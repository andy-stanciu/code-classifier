class Solution {

    public int mySqrt(int x) {
        //inbuilt
        //return (int)Math.sqrt(x);
        //core binary search way...we know square root will belong in between 0...x
        if (x < 2)
            return x;
        int start = 0;
        int end = x;
        int res = 0;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (mid == x / mid)
                return (int) mid;
            else if (mid < x / mid) {
                res = mid;
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return (int) res;
    }
}
