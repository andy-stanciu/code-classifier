class Solution {

    public int mySqrt(int x) {
        if (x < 2)
            return x;
        int left = 0, right = x / 2;
        int res = 0;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if ((long) mid * mid < x) {
                left = mid + 1;
                res = mid;
            } else if ((long) mid * mid > x) {
                right = mid - 1;
            } else {
                return mid;
            }
        }
        return res;
    }
}
