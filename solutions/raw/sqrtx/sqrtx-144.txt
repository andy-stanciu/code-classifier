class Solution {

    public int mySqrt(int x) {
        if (x < 2) {
            // Directly return x for 0 and 1
            return x;
        }
        long low = 1;
        // You could also set high to Math.max(1, x / 2)
        long high = x;
        while (low <= high) {
            long mid = (low + high) / 2;
            long midSquared = mid * mid;
            if (midSquared <= x && (mid + 1) * (mid + 1) > x) {
                // Found the integer square root
                return (int) mid;
            } else if (midSquared > x) {
                // mid is too large
                high = mid - 1;
            } else {
                // mid is too small
                low = mid + 1;
            }
        }
        // Fallback, though we should never reach here
        return (int) high;
    }
}
