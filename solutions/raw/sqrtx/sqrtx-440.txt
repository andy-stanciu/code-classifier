class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int i = 1;
        int j = x;
        int mid = 0;
        while (i <= j) {
            mid = i + (j - i) / 2;
            if (mid * mid == x) {
                return mid;
            } else if ((long) mid * mid > (long) x) {
                j = mid - 1;
            } else {
                i = mid + 1;
            }
        }
        return j;
    }
}
