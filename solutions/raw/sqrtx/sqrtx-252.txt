class Solution {

    public int mySqrt(int x) {
        return (int) Math.sqrt(x);
    }
}

public class answer {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("x = ");
        int x = sc.nextInt();
        Solution sol = new Solution();
        System.out.println(sol.mySqrt(x));
    }
}
