class Solution {

    public int mySqrt(int x) {
        if (x == 0) {
            return 0;
        } else {
            long s = 1;
            long e = x;
            long ans = 1;
            while (s <= e) {
                long mid = s + (e - s) / 2;
                if (mid * mid <= x) {
                    ans = mid;
                    s = mid + 1;
                } else {
                    e = mid - 1;
                }
            }
            return (int) e;
        }
    }
}
