class Solution {

    public int mySqrt(int x) {
        // check for base case.
        if (x == 0 || x == 1) {
            return x;
        }
        int low = 1, high = x, mid, ans = 1;
        while (low <= high) {
            mid = (high + low) / 2;
            // check if the square of mid is less than or equal to n
            // If yes, assign ans=mid and search for a larger value in the range mid+1 to high
            if (mid <= x / mid) {
                ans = mid;
                low = mid + 1;
                // otherwise initialize high=mid-1 and search for a smaller value in the range low to mid-1.
            } else {
                high = mid - 1;
            }
        }
        return ans;
    }
}
