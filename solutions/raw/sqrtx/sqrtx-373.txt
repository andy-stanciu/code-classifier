class Solution {

    public int mySqrt(int x) {
        if (x == 0) {
            return 0;
        }
        int low = 0;
        int high = x;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            long midSquared = (long) mid * mid;
            if (midSquared <= x) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return high;
    }
}
