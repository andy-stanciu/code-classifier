class Solution {

    public int mySqrt(int x) {
        if (x == 1) {
            return 1;
        }
        long xLong = (long) x;
        int low = 0;
        int high = x;
        long product = 1;
        while (low <= high) {
            long mid = low + (high - low) / 2;
            long prevProd = product;
            product = mid * mid;
            if (product == xLong) {
                return (int) mid;
            } else if (product < xLong) {
                low = (int) mid;
            } else {
                high = (int) mid;
            }
            if (prevProd == product) {
                return (int) mid;
            }
        }
        return low;
    }
}
