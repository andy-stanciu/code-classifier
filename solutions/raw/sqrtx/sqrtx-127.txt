class Solution {

    public int mySqrt(int x) {
        long start = 1;
        long end = x;
        while (start <= end) {
            long mid = start + (end - start) / 2;
            if (mid * mid > x) {
                end = (int) mid - 1;
            } else if (mid * mid < x) {
                start = (int) mid + 1;
            } else {
                return (int) mid;
            }
        }
        return (int) end;
    }
}
