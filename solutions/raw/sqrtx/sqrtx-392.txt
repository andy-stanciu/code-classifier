class Solution {

    public int mySqrt(int x) {
        long i = 0;
        while (i <= x / 2) {
            if (i * i == x) {
                return (int) i;
            } else if (i * i < x) {
                i++;
                continue;
            } else {
                return (int) i - 1;
            }
        }
        return i == 1 ? 1 : (int) i - 1;
    }
}
