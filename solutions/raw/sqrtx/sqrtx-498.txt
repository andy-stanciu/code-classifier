class Solution {

    public int mySqrt(long x) {
        long a = (long) x;
        long low = 1;
        long high = x / 2;
        long y = 1;
        if (x == 0)
            return 0;
        while (low <= high) {
            long mid = low + (high - low) / 2;
            if (a >= mid * mid) {
                y = mid;
                low = mid + 1;
            } else if (a < (mid * mid)) {
                high = mid - 1;
            }
        }
        return (int) y;
    }
}
