class Solution {

    public int mySqrt(int num) {
        if (num == 0) {
            return 0;
        } else if (num == 1) {
            return 1;
        } else {
            double t, sqrt;
            sqrt = num / 2;
            do {
                t = sqrt;
                sqrt = (t + (num / t)) / 2;
            } while ((t - sqrt) != 0);
            //System.out.println(sqrt);
            int num1 = (int) (sqrt);
            return num1;
        }
    }
}
