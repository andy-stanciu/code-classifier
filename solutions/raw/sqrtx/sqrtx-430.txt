class Solution {

    public int mySqrt(int x) {
        int i = 0, j = x;
        while (i <= j) {
            long m = (i + j) / 2;
            if (m * m == x)
                return (int) m;
            else if (m * m > x)
                j = (int) m - 1;
            else
                i = (int) m + 1;
        }
        return j;
    }
}
