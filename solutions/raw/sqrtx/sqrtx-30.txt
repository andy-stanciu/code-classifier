class Solution {

    public int mySqrt(int x) {
        // Edge case for zero
        if (x == 0)
            return 0;
        // Use long to prevent overflow
        long left = 1, right = x, result = 0;
        while (left <= right) {
            long mid = left + (right - left) / 2;
            long midSquared = mid * mid;
            if (midSquared == x) {
                return (int) mid;
            } else if (midSquared < x) {
                // Record the last valid mid
                result = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        // Return the integer part of sqrt(x)
        return (int) result;
    }
}
