class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        // the root will lie between these two values
        int low = 1;
        int high = x;
        int root = 0;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (mid <= (x / mid)) {
                // This could be the answer, but look for larger
                root = mid;
                low = mid + 1;
            } else {
                high = mid - 1;
                // mid is too large, look on the left
            }
        }
        return root;
    }
}
