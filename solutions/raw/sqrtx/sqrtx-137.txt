class Solution {

    public int mySqrt(int x) {
        if (x == 0) {
            return 0;
        }
        int start = 1;
        int last = x;
        int ans = 0;
        while (start <= last) {
            int mid = start + (last - start) / 2;
            if (mid <= x / mid) {
                ans = mid;
                start = mid + 1;
            } else {
                last = mid - 1;
            }
        }
        return ans;
    }
}
