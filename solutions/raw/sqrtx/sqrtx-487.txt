class Solution {

    public int mySqrt(int x) {
        int left = 0;
        int right = x;
        long mid;
        while (left <= right) {
            mid = (left + right) / 2;
            if (mid * mid == x) {
                return (int) mid;
            } else if (mid * mid > x) {
                right = (int) mid - 1;
            } else if (mid * mid < x) {
                left = (int) mid + 1;
            }
        }
        return right;
    }
}
