class Solution {

    public int mySqrt(int x) {
        int lo = 0, hi = x;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if ((long) mid * mid > (long) x)
                hi = mid - 1;
            else if ((long) mid * mid < (long) x)
                lo = mid + 1;
            else
                return mid;
        }
        return lo - 1;
    }
}
