class Solution {

    public int mySqrt(int x) {
        if (x < 2) {
            return x;
        }
        int start = 0, end = x / 2, sqrt = 0;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (1L * mid * mid <= x) {
                sqrt = mid;
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return sqrt;
    }
}
