class Solution {

    public int mySqrt(int num) {
        if (num == 0 || num == 1) {
            return num;
        }
        int start = 0;
        int end = num;
        int mid = 0;
        while (start <= end) {
            mid = start + (end - start) / 2;
            if (mid <= num / mid && (mid + 1) > num / (mid + 1)) {
                return mid;
            } else if (mid > num / mid) {
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return end;
    }
}
