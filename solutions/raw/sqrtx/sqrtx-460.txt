class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        double initial_guess = x / 2.0;
        double curr_x = initial_guess;
        double next_x = 0.5 * (curr_x + x / curr_x);
        while (Math.abs(curr_x - next_x) > 0.00001) {
            // Use a tolerance value to check for convergence
            curr_x = next_x;
            next_x = 0.5 * (curr_x + x / curr_x);
        }
        return (int) curr_x;
    }
}
