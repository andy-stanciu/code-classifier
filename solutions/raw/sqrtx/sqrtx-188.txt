class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int s = 1;
        int e = x;
        int m = -1;
        while (e >= s) {
            m = s + (e - s) / 2;
            if ((long) m * m == (long) x) {
                return m;
            }
            if ((long) m * m > (long) x) {
                e = m - 1;
            } else {
                s = m + 1;
            }
        }
        return Math.round(e);
    }
}
