class Solution {

    public int mySqrt(int x) {
        if (x <= 0) {
            return 0;
        }
        if (x == 1) {
            return 1;
        }
        double resultado = x;
        double anterior;
        do {
            anterior = resultado;
            resultado = (anterior + (x / anterior)) / 2;
        } while (Math.abs(anterior - resultado) > 0.1);
        return (int) resultado;
    }
}
