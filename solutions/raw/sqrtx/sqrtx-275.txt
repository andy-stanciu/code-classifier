class Solution {

    public int mySqrt(int n) {
        double guess = n;
        // Set the precision for accuracy
        double precision = 0.000001;
        // Continue iterating until the difference is small enough
        while ((guess - (n / guess)) > precision) {
            // Update the guess using Newton\'s formula
            guess = (guess + (n / guess)) / 2;
        }
        return (int) guess;
    }
}
