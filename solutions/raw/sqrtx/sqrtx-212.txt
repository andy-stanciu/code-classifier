class Solution {

    public int mySqrt(int n) {
        if (n < 2)
            return n;
        int start = 0;
        int end = n;
        int potentialAnswer = 0;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (mid == n / mid)
                return mid;
            else if (mid < n / mid) {
                potentialAnswer = mid;
                start = mid + 1;
            } else
                end = mid - 1;
        }
        return potentialAnswer;
    }
}
