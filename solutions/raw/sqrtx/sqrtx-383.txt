class Solution {

    public int mySqrt(int x) {
        if (x <= 0)
            return 0;
        for (long i = 0; i <= x; i++) {
            if (x == i * i) {
                return (int) i;
            } else if (x < i * i)
                return (int) (i - 1);
        }
        return -1;
    }
}
