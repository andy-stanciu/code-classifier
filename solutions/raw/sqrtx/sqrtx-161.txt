class Solution {

    public int mySqrt(int x) {
        long first = 0;
        long last = x;
        long ans = 0;
        while (first <= last) {
            long mid = first + (last - first) / 2;
            if (mid * mid == x) {
                return (int) mid;
            }
            if (mid * mid < x) {
                ans = mid;
                first = mid + 1;
            } else {
                last = mid - 1;
            }
        }
        return (int) ans;
    }
}
