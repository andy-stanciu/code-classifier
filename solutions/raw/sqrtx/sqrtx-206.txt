class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        int l = x;
        while (l > x / l) {
            l /= 2;
        }
        int u = x / l;
        while (Math.abs(l - u) > 1) {
            l = (l + u) / 2;
            u = x / l;
        }
        return l < u ? l : u;
    }
}
