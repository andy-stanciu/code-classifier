class Solution {

    public int mySqrt(int x) {
        int i = 0;
        int j = x;
        while (i <= j) {
            int mid = (i + j) / 2;
            long sqr = (long) mid * mid;
            if (sqr == x) {
                return mid;
            } else if (sqr < x) {
                i = mid + 1;
            } else {
                j = mid - 1;
            }
        }
        return j;
    }
}
