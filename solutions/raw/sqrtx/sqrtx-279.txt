class Solution {

    public int mySqrt(int x) {
        //1 2 3 4 5 6 7 8
        if (x == 0)
            return 0;
        if (x == 1)
            return 1;
        int low = 1, high = x;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            // Use long to prevent overflow
            long sq = (long) mid * mid;
            if (sq == x) {
                return mid;
            } else if (sq > x) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        // high will be the integer square root
        return high;
    }
}
