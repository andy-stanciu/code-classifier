class Solution {

    public int mySqrt(int x) {
        if (x <= 1)
            return x;
        // long i = 1;
        // long sq = i * i;
        // while (x >= sq) {
        //     i++;
        //     sq = i * i;
        // }
        long start = 1, end = Integer.MAX_VALUE, ans = 0;
        while (start <= end) {
            long mid = start + (end - start) / 2;
            if (mid * mid <= x) {
                ans = mid;
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return (int) ans;
    }
}
