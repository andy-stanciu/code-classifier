class Solution {

    public ListNode swapPairs(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode ans = new ListNode(0);
        ans.next = head;
        ListNode temp = head;
        ListNode temp1 = ans;
        while (temp != null && temp.next != null) {
            ListNode forw = temp.next;
            temp.next = forw.next;
            forw.next = temp;
            temp1.next = forw;
            temp1 = temp;
            temp = temp.next;
        }
        return ans.next;
    }
}
