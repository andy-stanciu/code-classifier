class Solution {

    public ListNode swapPairs(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode pre = head;
        ListNode curr = head.next;
        ListNode pre2 = head;
        head = head.next;
        while (curr != null) {
            pre.next = curr.next;
            curr.next = pre;
            pre2 = pre;
            pre = pre.next;
            if (pre == null || pre.next == null)
                return head;
            curr = pre.next;
            pre2.next = curr;
        }
        return head;
    }
}
