class Solution {

    public ListNode swapPairs(ListNode head) {
        ListNode temp = new ListNode();
        ListNode prev = new ListNode();
        ListNode cur = new ListNode();
        ListNode nxt = new ListNode();
        ListNode node = head;
        int start = 0;
        while (node != null && node.next != null) {
            temp = node.next.next;
            cur = node;
            nxt = node.next;
            nxt.next = cur;
            cur.next = temp;
            node = temp;
            if (prev != null)
                prev.next = nxt;
            if (start == 0) {
                head = nxt;
                start++;
            }
            prev = cur;
        }
        return head;
    }
}
