class Solution {

    public ListNode swapPairs(ListNode head) {
        ListNode f = new ListNode(-1);
        ListNode s = new ListNode(-1);
        f.next = s;
        s.next = head;
        while (s != null) {
            int temp = f.val;
            f.val = s.val;
            s.val = temp;
            if (// even case
            s.next == null) {
                return head;
            } else if (// odd value case
            s.next.next == null) {
                return head;
            }
            f = f.next.next;
            s = s.next.next;
        }
        return head;
    }
}
