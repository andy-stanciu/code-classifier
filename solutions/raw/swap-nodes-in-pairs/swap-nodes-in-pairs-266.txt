/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode swapPairs(ListNode head) {
        ListNode dummNode = new ListNode(0);
        dummNode.next = head;
        ListNode lastSwappedNode = dummNode;
        ListNode currNode = head;
        while (currNode != null && currNode.next != null) {
            // store the next pair:
            ListNode nextPair = currNode.next.next;
            // swapping:
            lastSwappedNode.next = currNode.next;
            currNode.next.next = currNode;
            currNode.next = nextPair;
            lastSwappedNode = currNode;
            currNode = nextPair;
        }
        return dummNode.next;
    }
}
