class Solution {

    public ListNode swapPairs(ListNode head) {
        Integer len = Solution.getLen(head);
        ListNode curr = head;
        ListNode toRet = Solution.helper(curr, len);
        return toRet;
    }

    public static ListNode helper(ListNode current, Integer len) {
        if (len < 2) {
            return current;
        }
        ListNode rest = current.next.next;
        ListNode Temp = current;
        ListNode Next = current.next;
        Temp.next = null;
        Next.next = null;
        current = Next;
        current.next = Temp;
        current.next.next = helper(rest, len - 2);
        return current;
    }

    public static int getLen(ListNode start) {
        int idx = 0;
        while (start != null) {
            idx += 1;
            start = start.next;
        }
        return idx;
    }
}
