class Solution {

    public ListNode swapPairs(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode res = head.next;
        ListNode midHead = null;
        while (head != null && head.next != null) {
            ListNode tail = head.next.next;
            ListNode temp = head;
            head = head.next;
            head.next = temp;
            temp.next = tail;
            if (midHead != null) {
                midHead.next = head;
            }
            midHead = head.next;
            head = tail;
        }
        return res;
    }
}
