class Solution {

    public ListNode swapPairs(ListNode head) {
        //base case
        if (head == null)
            return head;
        if (head.next == null)
            return head;
        ListNode first = head;
        ListNode second = head.next;
        ListNode third = second.next;
        //swap first and second node
        second.next = first;
        first.next = swapPairs(third);
        return second;
    }
}
