class Solution {

    public ListNode swapPairs(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode dummy = new ListNode(0, null);
        ListNode ans = dummy;
        dummy.next = head;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            ListNode temp = curr;
            curr = curr.next;
            dummy.next = curr;
            temp.next = curr.next;
            curr.next = temp;
            curr = temp.next;
            dummy = temp;
        }
        return ans.next;
    }
}
