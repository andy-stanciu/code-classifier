class Solution {

    public ListNode swapPairs(ListNode head) {
        ListNode temp = head;
        while (temp != null) {
            int x = temp.val;
            if (temp.next != null) {
                temp.val = temp.next.val;
                temp.next.val = x;
                temp = temp.next;
            }
            temp = temp.next;
        }
        return head;
    }
}
