class Solution {

    public ListNode swapPairs(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head.next;
        ListNode prev = head;
        ListNode next = null;
        ListNode newHead = curr;
        while (curr != null) {
            next = curr.next;
            curr.next = prev;
            if (next == null || next.next == null) {
                prev.next = next;
                curr = null;
            } else {
                curr = next.next;
                prev.next = curr;
                prev = next;
            }
        }
        return newHead;
    }
}
