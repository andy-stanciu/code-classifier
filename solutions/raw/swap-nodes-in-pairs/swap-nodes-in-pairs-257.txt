/**
 * Definition for singly-linked list.
 * public class ListNode {
 * int val;
 * ListNode next;
 * ListNode() {}
 * ListNode(int val) { this.val = val; }
 * ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode swapPairs(ListNode head) {
        ListNode trav = head, prev = null, res = null;
        while (trav != null && trav.next != null) {
            ListNode second = trav.next;
            if (prev == null) {
                trav.next = second.next;
                second.next = null;
            } else {
                prev.next = second;
                trav.next = second.next;
            }
            second.next = trav;
            prev = trav;
            trav = second;
            if (res == null) {
                res = trav;
            }
            trav = trav.next.next;
        }
        return (res == null) ? head : res;
    }
}
