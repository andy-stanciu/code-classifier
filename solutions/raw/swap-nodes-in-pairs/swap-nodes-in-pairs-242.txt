/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode swapPairs(ListNode head) {
        List<Integer> ls = new ArrayList<>();
        ListNode ph = head;
        while (ph != null) {
            ls.add(ph.val);
            ph = ph.next;
        }
        for (int i = 0; i + 1 < ls.size(); i = i + 2) {
            int temp = ls.get(i);
            ls.set(i, ls.get(i + 1));
            ls.set(i + 1, temp);
        }
        ListNode res = new ListNode(0), pr = res;
        for (int i : ls) {
            pr.next = new ListNode(i);
            pr = pr.next;
        }
        return res.next;
    }
}
