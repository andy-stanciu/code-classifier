class Solution {

    static int count;

    public static void solve(int[] nums, int target, int index, int sum) {
        if (index == nums.length) {
            if (sum == target) {
                count++;
            }
            return;
        }
        solve(nums, target, index + 1, sum + nums[index]);
        solve(nums, target, index + 1, sum - nums[index]);
        return;
    }

    public int findTargetSumWays(int[] nums, int target) {
        count = 0;
        solve(nums, target, 0, 0);
        return count;
    }
}
