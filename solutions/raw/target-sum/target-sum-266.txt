class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return helper(nums, nums.length - 1, target);
    }

    public int helper(int[] nums, int index, int target) {
        if (index < 0) {
            if (target == 0) {
                return 1;
            } else {
                return 0;
            }
        }
        int plus = helper(nums, index - 1, target - nums[index]);
        int minus = helper(nums, index - 1, nums[index] + target);
        return plus + minus;
    }
}
