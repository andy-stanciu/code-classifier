class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return f(nums, nums.length, 0, 0, target);
    }

    public int f(int[] nums, int n, int i, int sum, int k) {
        if (i == n) {
            if (sum == k)
                return 1;
            return 0;
        }
        return f(nums, n, i + 1, sum + nums[i], k) + f(nums, n, i + 1, sum + (-1 * nums[i]), k);
    }
}
