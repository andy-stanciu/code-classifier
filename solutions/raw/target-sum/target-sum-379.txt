class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        // by using memoisation ;
        HashMap<Integer, HashMap<Integer, Integer>> map = new HashMap<>();
        int count = answer(nums, 0, 0, target, map);
        return count;
    }

    public int answer(int[] nums, int index, int sum, int target, HashMap<Integer, HashMap<Integer, Integer>> map) {
        if (index == nums.length) {
            if (sum == target)
                return 1;
            else
                return 0;
        }
        if (map.containsKey(index)) {
            HashMap<Integer, Integer> temp = map.get(index);
            if (temp.containsKey(sum))
                return temp.get(sum);
        }
        int count = answer(nums, index + 1, sum + nums[index], target, map);
        count += answer(nums, index + 1, sum - nums[index], target, map);
        if (map.containsKey(index)) {
            HashMap<Integer, Integer> temp = map.get(index);
            temp.put(sum, count);
        } else {
            HashMap<Integer, Integer> temp = new HashMap<>();
            temp.put(sum, count);
            map.put(index, temp);
        }
        return count;
    }
}
