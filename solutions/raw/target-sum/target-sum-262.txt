class Solution {

    HashMap<Pair<Integer, Integer>, Integer> dp;

    public int findTargetSumWays(int[] nums, int target) {
        dp = new HashMap();
        return recursion(0, target, nums, 0);
    }

    public int recursion(int i, int target, int[] nums, int res) {
        Pair key = new Pair<>(i, target);
        if (dp.containsKey(key))
            return dp.get(key);
        if (i == nums.length && target == 0)
            return 1;
        if (i == nums.length)
            return 0;
        res += recursion(i + 1, target - nums[i], nums, res) + recursion(i + 1, target + nums[i], nums, res);
        dp.put(key, res);
        return res;
    }
}
