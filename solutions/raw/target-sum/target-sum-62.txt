class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWaysInner(nums, target, 0);
    }

    private int findTargetSumWaysInner(int[] nums, int target, int idx) {
        if (idx == nums.length) {
            if (target == 0)
                return 1;
            return 0;
        }
        return findTargetSumWaysInner(nums, target - nums[idx], idx + 1) + findTargetSumWaysInner(nums, target + nums[idx], idx + 1);
    }
}
