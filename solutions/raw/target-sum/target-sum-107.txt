class Solution {

    public int totalNumberOfWays(int[] nums, int target, int idx) {
        //base case
        if (idx == 0) {
            int ways = 0;
            if ((target - nums[idx]) == 0) {
                ways++;
            }
            if ((target + nums[idx]) == 0) {
                ways++;
            }
            return ways;
        }
        int add = totalNumberOfWays(nums, target + nums[idx], idx - 1);
        int sub = totalNumberOfWays(nums, target - nums[idx], idx - 1);
        return add + sub;
    }

    public int findTargetSumWays(int[] nums, int target) {
        return totalNumberOfWays(nums, target, nums.length - 1);
    }
}
