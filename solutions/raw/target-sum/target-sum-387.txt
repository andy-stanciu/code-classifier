class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int[] count = new int[1];
        backtrack(nums, 0, 0, target, count);
        return count[0];
    }

    public void backtrack(int[] nums, int index, int sum, int target, int[] count) {
        if (index == nums.length) {
            if (sum == target) {
                count[0] += 1;
            }
            return;
        }
        backtrack(nums, index + 1, sum + nums[index], target, count);
        backtrack(nums, index + 1, sum - nums[index], target, count);
    }
}
