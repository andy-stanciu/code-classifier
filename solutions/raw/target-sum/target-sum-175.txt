class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findWays(nums, 0, 0, target);
    }

    public int findWays(int[] nums, int index, int sum, int target) {
        int count = 0;
        if (index == nums.length) {
            if (sum == target)
                count++;
            return count;
        }
        count += findWays(nums, index + 1, sum - nums[index], target);
        count += findWays(nums, index + 1, sum + nums[index], target);
        return count;
    }
}
