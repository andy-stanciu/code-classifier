class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return countWays(target, nums, nums.length - 1);
    }

    int countWays(int t, int[] nums, int i) {
        // BASE CASE
        // returns 1 if target acheived or returns 0
        if (i < 0)
            return t == 0 ? 1 : 0;
        // Picking +ve num and
        // Update reqd target by subtracting +nums[i]
        int P = countWays(t - nums[i], nums, i - 1);
        // Picking -ve num
        // Update reqd target by subtracting -nums[i]
        int N = countWays(t + nums[i], nums, i - 1);
        return N + P;
    }
}
