class Solution {

    private int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        help(nums, target, 0, 0);
        return count;
    }

    private void help(int[] nums, int target, int index, int total) {
        if (index >= nums.length) {
            if (target == total)
                count++;
            return;
        }
        help(nums, target, index + 1, total - nums[index]);
        help(nums, target, index + 1, total + nums[index]);
    }
}
