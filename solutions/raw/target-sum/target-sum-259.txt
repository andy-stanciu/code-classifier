class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return helper(nums, target, nums.length);
    }

    public static int helper(int[] nums, int target, int i) {
        if (i == 0)
            return target == 0 ? 1 : 0;
        int p = helper(nums, target - nums[i - 1], i - 1);
        int m = helper(nums, target + nums[i - 1], i - 1);
        return p + m;
    }
}
