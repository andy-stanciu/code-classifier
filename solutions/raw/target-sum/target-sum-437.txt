class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int s = 0;
        for (int x : nums) s += x;
        if (s < target || (s - target) % 2 != 0)
            return 0;
        int n = (s - target) / 2;
        int[] c = new int[n + 1];
        c[0] = 1;
        for (int x : nums) for (int i = n; i >= x; i--) c[i] += c[i - x];
        return c[n];
    }
}
