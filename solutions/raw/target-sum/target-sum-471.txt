class Solution {

    int s = 0;

    public int findWays(int op, int n, int sumSoFar, int[] nums, int target) {
        if (n == nums.length)
            return 0;
        sumSoFar += nums[n] * op;
        if (n == nums.length - 1 && sumSoFar == target) {
            s += 1;
            return 0;
        }
        findWays(1, n + 1, sumSoFar, nums, target);
        findWays(-1, n + 1, sumSoFar, nums, target);
        return 0;
    }

    public int findTargetSumWays(int[] nums, int target) {
        findWays(1, 0, 0, nums, target);
        findWays(-1, 0, 0, nums, target);
        return s;
    }
}
