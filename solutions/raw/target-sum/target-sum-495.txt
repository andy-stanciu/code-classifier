class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return compute(0, 0, nums, target);
    }

    public int compute(int i, int sum, int[] nums, int target) {
        if (i == nums.length) {
            if (sum == target)
                return 1;
            return 0;
        }
        int minus = compute(i + 1, sum - nums[i], nums, target);
        int plus = compute(i + 1, sum + nums[i], nums, target);
        return minus + plus;
    }
}
