class Solution {

    int answer = 0;

    public int findTargetSumWays(int[] nums, int target) {
        recur(nums, target, 0);
        return answer;
    }

    public void recur(int[] nums, int target, int i) {
        if (i >= nums.length && 0 == target) {
            answer++;
            return;
        } else if (i >= nums.length) {
            return;
        }
        recur(nums, target + (nums[i] * (-1)), i + 1);
        recur(nums, target + (nums[i]), i + 1);
    }
}
