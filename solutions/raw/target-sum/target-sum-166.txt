class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return fun(nums, target, 0, 0);
    }

    int fun(int[] nums, int target, int start, int sum) {
        if (start == nums.length - 1) {
            int ans = 0;
            if (sum + nums[start] == target)
                ans++;
            if (sum - nums[start] == target)
                ans++;
            return ans;
        }
        int temp1 = fun(nums, target, start + 1, sum + nums[start]);
        int temp2 = fun(nums, target, start + 1, sum - nums[start]);
        return temp1 + temp2;
    }
}
