class Solution {

    public int findTargetSumWays(int[] arr, int target) {
        int n = arr.length;
        int sum = 0;
        for (int i : arr) sum += i;
        if (target > sum || target < -sum || (target + sum) % 2 != 0)
            return 0;
        int s = (target + sum) / 2;
        int[][] t = new int[n + 1][s + 1];
        for (int i = 0; i < t.length; i++) t[i][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= s; j++) {
                if (j - arr[i - 1] >= 0)
                    t[i][j] = t[i - 1][j - arr[i - 1]] + t[i - 1][j];
                else
                    t[i][j] = t[i - 1][j];
            }
        }
        return t[n][s];
    }
}
