class Solution {

    static int solve(int idx, int[] nums, int sum, int target) {
        if (idx == nums.length) {
            if (sum == target) {
                return 1;
            } else {
                return 0;
            }
        }
        int len = solve(idx + 1, nums, sum - nums[idx], target) + solve(idx + 1, nums, sum + nums[idx], target);
        return len;
    }

    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        return solve(0, nums, sum, target);
    }
}
