class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int[] arr = new int[1];
        findExpressions(0, 0, nums, target, arr);
        return arr[0];
    }

    private void findExpressions(int idx, int sum, int[] nums, int target, int[] arr) {
        if (idx == nums.length) {
            if (sum == target)
                arr[0]++;
            return;
        }
        sum -= nums[idx];
        findExpressions(idx + 1, sum, nums, target, arr);
        sum += nums[idx] * 2;
        findExpressions(idx + 1, sum, nums, target, arr);
    }
}
