class Solution {

    public int solve(int[] nums, int i, int sum, int k, int[][][] dp) {
        if (i >= nums.length) {
            if (sum == k) {
                return 1;
            } else {
                return 0;
            }
        }
        if (sum >= 0) {
            if (dp[i][0][sum] == Integer.MIN_VALUE) {
                int x = solve(nums, i + 1, sum + nums[i], k, dp);
                int y = solve(nums, i + 1, sum - nums[i], k, dp);
                dp[i][0][sum] = x + y;
            }
            return dp[i][0][sum];
        } else {
            if (dp[i][1][-1 * sum] == Integer.MIN_VALUE) {
                int x = solve(nums, i + 1, sum + nums[i], k, dp);
                int y = solve(nums, i + 1, sum - nums[i], k, dp);
                dp[i][1][-1 * sum] = x + y;
            }
            return dp[i][1][-1 * sum];
        }
    }

    public int findTargetSumWays(int[] nums, int target) {
        int[][][] dp = new int[nums.length][2][1001];
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < 2; j++) {
                for (int k = 0; k < 1001; k++) {
                    dp[i][j][k] = Integer.MIN_VALUE;
                }
            }
        }
        return solve(nums, 0, 0, target, dp);
    }
}
