class Solution {

    private int f(int i, int tempTarget, int target, int[] nums) {
        if (i < 0) {
            if (tempTarget == target) {
                return 1;
            } else {
                return 0;
            }
        }
        int plus = f(i - 1, tempTarget + nums[i], target, nums);
        int minus = f(i - 1, tempTarget - nums[i], target, nums);
        return plus + minus;
    }

    public int findTargetSumWays(int[] nums, int target) {
        return f(nums.length - 1, 0, target, nums);
    }
}
