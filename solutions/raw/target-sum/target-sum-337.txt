class Solution {

    static int targetSum(int[] nums, int target, int i, int sum) {
        if (i == nums.length) {
            if (sum == target) {
                return 1;
            } else {
                return 0;
            }
        }
        int count = 0;
        count += targetSum(nums, target, i + 1, sum + nums[i]);
        count += targetSum(nums, target, i + 1, sum - nums[i]);
        return count;
    }

    public int findTargetSumWays(int[] nums, int target) {
        return targetSum(nums, target, 0, 0);
    }
}
