class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return solve(nums, nums.length - 1, 0, target);
    }

    private int solve(int[] nums, int i, int sum, int target) {
        if (i < 0 && sum == target)
            return 1;
        else if (i < 0)
            return 0;
        //total number of ways if you added nums[i]
        int positive = solve(nums, i - 1, sum + nums[i], target);
        //total number of ways if you subtracted nums[i]
        int negative = solve(nums, i - 1, sum - nums[i], target);
        return positive + negative;
    }
}
