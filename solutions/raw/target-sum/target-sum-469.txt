class Solution {

    public static int target(int[] nums, int tar, int idx, int sum) {
        //base case
        if (sum == tar && idx == nums.length) {
            return 1;
        }
        if (idx >= nums.length) {
            return 0;
        }
        int a = target(nums, tar, idx + 1, sum + nums[idx]);
        int b = target(nums, tar, idx + 1, sum - nums[idx]);
        return a + b;
    }

    public int findTargetSumWays(int[] nums, int target) {
        return target(nums, target, 0, 0);
    }
}
