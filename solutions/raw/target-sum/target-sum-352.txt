class Solution {

    int result;

    public int findTargetSumWays(int[] nums, int target) {
        result = 0;
        rec(nums, target, 0, 0);
        return result;
    }

    private void rec(int[] nums, int target, int curSum, int index) {
        if (index == nums.length) {
            if (curSum == target) {
                result++;
            }
            return;
        }
        rec(nums, target, curSum + nums[index], index + 1);
        rec(nums, target, curSum - nums[index], index + 1);
    }
}
