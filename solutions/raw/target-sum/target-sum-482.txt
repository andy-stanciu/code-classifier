class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return sum(nums, target, 0, 0);
    }

    private int sum(int[] arr, int target, int summ, int n) {
        if (n == arr.length) {
            if (summ == target)
                return 1;
            else
                return 0;
        }
        int count = 0;
        count += sum(arr, target, summ + arr[n], n + 1);
        arr[n] *= -1;
        count += sum(arr, target, summ + arr[n], n + 1);
        arr[n] *= -1;
        return count;
    }
}
