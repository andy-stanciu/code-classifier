class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        Map<String, Integer> memo = new HashMap<>();
        return helper(nums, target, 0, memo);
    }

    private int helper(int[] nums, int target, int index, Map<String, Integer> memo) {
        if (index == nums.length) {
            return target == 0 ? 1 : 0;
        }
        String key = index + "," + target;
        if (memo.containsKey(key)) {
            return memo.get(key);
        }
        int ways = helper(nums, target - nums[index], index + 1, memo) + helper(nums, target + nums[index], index + 1, memo);
        memo.put(key, ways);
        return ways;
    }
}
