class Solution {

    public int findTargetSumWaysHelper(int[] nums, int target, int i, int counter) {
        if (i == nums.length) {
            if (counter == target) {
                return 1;
            } else {
                return 0;
            }
        }
        return findTargetSumWaysHelper(nums, target, i + 1, counter + nums[i]) + findTargetSumWaysHelper(nums, target, i + 1, counter - nums[i]);
    }

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWaysHelper(nums, target, 0, 0);
    }
}
