class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        findSum(nums, 0, 0, target);
        return count;
    }

    public void findSum(int[] array, int index, int tempT, int target) {
        if (index == array.length) {
            if (tempT == target) {
                count++;
            }
            return;
        }
        findSum(array, index + 1, tempT + array[index], target);
        findSum(array, index + 1, tempT - array[index], target);
    }
}
