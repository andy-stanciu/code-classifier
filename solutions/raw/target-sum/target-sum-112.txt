class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return solve(nums, nums.length - 1, target, 0);
    }

    public int solve(int[] nums, int i, int target, int totalNums) {
        if (i < 0 && target == totalNums) {
            return 1;
        }
        if (i < 0 && target != totalNums) {
            return 0;
        }
        return solve(nums, i - 1, target, totalNums - nums[i]) + solve(nums, i - 1, target, totalNums + nums[i]);
    }
}
