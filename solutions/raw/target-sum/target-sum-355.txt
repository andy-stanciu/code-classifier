class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return fun(0, 0, target, nums);
    }

    public int fun(int index, int sum, int target, int[] nums) {
        if (index == nums.length) {
            if (sum == target)
                return 1;
            return 0;
        }
        return fun(index + 1, sum - nums[index], target, nums) + fun(index + 1, sum + nums[index], target, nums);
    }
}
