class Solution {

    int ways = 0;

    public int findTargetSumWays(int[] nums, int target) {
        way(nums, target, 0);
        return ways;
    }

    void way(int[] nums, int target, int idx) {
        if (idx == nums.length) {
            if (target == 0)
                ways += 1;
            return;
        }
        way(nums, target - nums[idx], idx + 1);
        way(nums, target + nums[idx], idx + 1);
    }
}
