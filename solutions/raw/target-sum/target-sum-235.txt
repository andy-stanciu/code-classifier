class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int s = findsum(nums, target, 0, 0);
        return s;
    }

    public int check = 0;

    int findsum(int[] nums, int target, int i, int total) {
        if (i == nums.length) {
            if (total == target)
                check++;
            return 0;
        }
        total += nums[i];
        findsum(nums, target, ++i, total);
        --i;
        total -= nums[i];
        total -= nums[i];
        findsum(nums, target, ++i, total);
        return check;
    }
}
