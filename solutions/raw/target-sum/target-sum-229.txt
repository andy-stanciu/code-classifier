class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        int n = nums.length;
        //find sum
        for (int i : nums) {
            sum += i;
        }
        //s1 - s2 = target
        //s1 + s2 = sum
        //2s1 = target+sum
        //s1  = (target+sum)/2;
        if (((sum - target) % 2 == 1) || (target > sum))
            return 0;
        int sum1 = (sum - target) / 2;
        //solution 1
        /*
        int[][] dp = new int[n+1][sum1+1];
       
        dp[0][0] = 1;

        for(int i=1; i<=nums.length;i++){
            for(int j=0; j<=sum1; j++){
                if(nums[i-1] <=j)
                    dp[i][j] = dp[i-1][j] + dp[i-1][j-nums[i-1]];
                else
                    dp[i][j] = dp[i-1][j];
            }

        }
        return dp[nums.length][sum1];
        */
        //Collapse the 2D array into
        int[] dp = new int[sum1 + 1];
        dp[0] = 1;
        for (int num : nums) {
            for (int j = sum1; j >= 0; j--) {
                if (num <= j)
                    dp[j] = dp[j] + dp[j - num];
            }
        }
        return dp[sum1];
    }
}
