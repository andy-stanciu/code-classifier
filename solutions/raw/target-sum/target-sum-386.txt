class Solution {

    int res = 0;

    int t;

    public int findTargetSumWays(int[] nums, int target) {
        t = target;
        dfs(nums, 0, 0);
        return res;
    }

    public void dfs(int[] nums, int tempSum, int index) {
        if (index == nums.length) {
            if (tempSum == t) {
                res++;
            }
            return;
        }
        dfs(nums, tempSum + nums[index], index + 1);
        dfs(nums, tempSum - nums[index], index + 1);
    }
}
