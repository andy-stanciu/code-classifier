class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        // Arrays.sort(nums);
        int a = findtargetSum(nums, target, 0, 0);
        return a;
    }

    public int findtargetSum(int[] nums, int target, int index, int sum) {
        if (sum == target && index == nums.length) {
            return 1;
        }
        if (index >= nums.length && sum != target) {
            return 0;
        }
        int a = findtargetSum(nums, target, index + 1, sum + nums[index]);
        int b = findtargetSum(nums, target, index + 1, sum - nums[index]);
        return a + b;
    }
}
