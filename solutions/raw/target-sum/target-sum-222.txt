class Solution {

    public int solve(int ind, int[] nums, int target, int sum) {
        if (ind < 0) {
            if (sum == target) {
                return 1;
            }
            return 0;
        }
        int pos = solve(ind - 1, nums, target, sum + nums[ind]);
        int neg = solve(ind - 1, nums, target, sum - nums[ind]);
        return pos + neg;
    }

    public int findTargetSumWays(int[] nums, int target) {
        return solve(nums.length - 1, nums, target, 0);
    }
}
