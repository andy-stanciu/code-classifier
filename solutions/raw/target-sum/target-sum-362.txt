class Solution {

    public int findways(int[] nums, int target, int sum, int n, int i) {
        if (sum == target && i >= n) {
            return 1;
        }
        if (i >= n) {
            return 0;
        }
        int positive = findways(nums, target, sum + nums[i], n, i + 1);
        int negative = findways(nums, target, sum + (-nums[i]), n, i + 1);
        return positive + negative;
    }

    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        int n = nums.length;
        int i = 0;
        return findways(nums, target, sum, n, 0);
    }
}
