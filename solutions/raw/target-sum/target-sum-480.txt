class Solution {

    int ans = 0;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, 0, 0, target, nums.length);
        return ans;
    }

    public void helper(int[] arr, int index, int sum, int target, int n) {
        if (index == n) {
            if (sum == target)
                ans++;
            return;
        }
        helper(arr, index + 1, sum - arr[index], target, n);
        helper(arr, index + 1, sum + arr[index], target, n);
    }
}
