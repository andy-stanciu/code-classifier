class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int[] res = new int[1];
        dfs(nums, 0, target, 0, res);
        return res[0];
    }

    private void dfs(int[] nums, int index, int target, int curSum, int[] res) {
        if (index == nums.length) {
            if (curSum == target) {
                res[0]++;
            }
            return;
        }
        int tmp = nums[index];
        dfs(nums, index + 1, target, curSum + tmp, res);
        dfs(nums, index + 1, target, curSum - tmp, res);
    }
}
