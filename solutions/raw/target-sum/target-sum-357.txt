class Solution {

    public int findTargetSumWays(int[] arr, int target) {
        return recurse(arr, 0, 0, target);
    }

    private int recurse(int[] arr, int i, int sumsofar, int target) {
        int n = arr.length;
        if (i == n)
            return sumsofar == target ? 1 : 0;
        return recurse(arr, i + 1, sumsofar + arr[i], target) + recurse(arr, i + 1, sumsofar - arr[i], target);
    }
}
