class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        // Memory Table to store subproblem result...
        Map<String, Integer> memory = new HashMap<String, Integer>();
        // Function Call...
        return Memonise(nums, 0, 0, target, 0, memory);
    }

    public int Memonise(int[] nums, int index, int currentSum, int target, int totalWays, Map<String, Integer> memo) {
        if (index == nums.length) {
            // If the array is traversed...
            // If the sum equals target...
            return currentSum == target ? 1 : 0;
        }
        // generating a unique key (the sum at every instance will be unique)...
        String key = index + "|" + currentSum;
        if (// Checking memory...
        memo.containsKey(key))
            return memo.get(key);
        // If current number is signed positive...
        int positive = Memonise(nums, index + 1, currentSum + nums[index], target, totalWays, memo);
        // If current number is signed negative...
        int negative = Memonise(nums, index + 1, currentSum - nums[index], target, totalWays, memo);
        totalWays = positive + negative;
        // Updating the memory...
        memo.put(key, totalWays);
        // Returning the final output of the problem...
        return totalWays;
    }
}
