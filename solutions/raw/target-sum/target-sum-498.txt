class Solution {

    Integer n = null;

    public int findTargetSumWays(int[] nums, int target) {
        n = nums.length;
        return findWays(nums, target, 0, 0, 0);
    }

    private int findWays(int[] nums, int target, int curSum, int index, int count) {
        if (n == index && curSum == target) {
            count++;
            return count;
        }
        if (n == index) {
            return 0;
        }
        return findWays(nums, target, curSum + nums[index], index + 1, count) + findWays(nums, target, curSum - nums[index], index + 1, count);
    }
}
