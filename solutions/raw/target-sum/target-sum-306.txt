class Solution {

    int count = 0;

    private void countWays(int[] nums, int sum, int n, int target) {
        if (n == nums.length) {
            if (sum == target) {
                count++;
            }
        } else {
            countWays(nums, sum + nums[n], n + 1, target);
            countWays(nums, sum - nums[n], n + 1, target);
        }
    }

    public int findTargetSumWays(int[] nums, int target) {
        countWays(nums, 0, 0, target);
        return count;
    }
}
