class Solution {

    // O(n^2) time || O(n) space - where n is the length of the nums array
    int n;

    int ans;

    public int findTargetSumWays(int[] nums, int target) {
        n = nums.length;
        ans = 0;
        dp(nums, target, 0, 0);
        return ans;
    }

    public void dp(int[] nums, int target, int idx, int currSum) {
        if (idx == n) {
            if (currSum == target) {
                ans++;
            }
            return;
        }
        dp(nums, target, idx + 1, currSum - nums[idx]);
        dp(nums, target, idx + 1, currSum + nums[idx]);
    }
}
