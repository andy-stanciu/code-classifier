class Solution {

    // YOU HAVE NOT SOLVED IT USING DP, WHEN YOU KNOW DP , SOLVE IT AGAIN;
    public int findTargetSumWays(int[] nums, int target) {
        int[] ans = { 0 };
        helper(nums, target, 0, ans);
        return ans[0];
    }

    public void helper(int[] nums, int target, int i, int[] ans) {
        if (i == nums.length) {
            if (target == 0) {
                ans[0]++;
            }
            return;
        }
        // +nums[i];
        target = target - nums[i];
        helper(nums, target, i + 1, ans);
        // restoring the target value back to original;
        target = target + nums[i];
        // -nums[i];
        target = target + nums[i];
        helper(nums, target, i + 1, ans);
        target = target - nums[i];
    }
}
