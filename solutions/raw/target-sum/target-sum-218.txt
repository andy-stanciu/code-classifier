class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return fun(nums, 0, target, 0);
    }

    public int fun(int[] nums, int index, int target, int sum) {
        if (index == nums.length) {
            if (sum == target) {
                return 1;
            }
            return 0;
        }
        return fun(nums, index + 1, target, sum + nums[index]) + fun(nums, index + 1, target, sum - nums[index]);
    }
}
