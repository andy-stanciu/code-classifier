class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return helper(nums, target, 0, 0);
    }

    public int helper(int[] nums, int target, int sum, int ind) {
        if (ind == nums.length) {
            if (sum == target) {
                return 1;
            }
            return 0;
        }
        int add = helper(nums, target, sum + nums[ind], ind + 1);
        int subtract = helper(nums, target, sum - nums[ind], ind + 1);
        return add + subtract;
    }
}
