class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return count(nums.length - 1, target, nums);
    }

    public int count(int ind, int sum, int[] nums) {
        if (ind == 0) {
            if (sum == 0 && nums[0] == 0) {
                return 2;
            }
            if (sum == nums[0] || nums[0] == -sum) {
                return 1;
            }
            return 0;
        }
        int plus = count(ind - 1, sum + nums[ind], nums);
        int minus = count(ind - 1, sum - nums[ind], nums);
        return plus + minus;
    }
}
