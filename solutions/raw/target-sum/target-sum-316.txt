class Solution {

    /**
     *    1. backtracking to check all possible combination and
     *        check after using out all elements, the sum will equals target all not
     */
    int res = 0;

    public int findTargetSumWays(int[] nums, int target) {
        backtracking(nums, 0, target, 0);
        return res;
    }

    private void backtracking(int[] nums, int start, int target, int sum) {
        if (start == nums.length) {
            if (sum == target) {
                res++;
            }
            return;
        }
        backtracking(nums, start + 1, target, sum + nums[start]);
        backtracking(nums, start + 1, target, sum - nums[start]);
    }
}
