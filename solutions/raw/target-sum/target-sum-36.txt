class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return Sum(nums, target, 0, 0, 0);
    }

    public static int Sum(int[] nums, int target, int sum, int index, int count) {
        if (index == nums.length) {
            if (sum == target) {
                count++;
            }
            return count;
        }
        // adding the positive operator
        int answerPositive = Sum(nums, target, sum + nums[index], index + 1, count);
        // adding the negative operator
        int answerNegative = Sum(nums, target, sum - nums[index], index + 1, count);
        return answerPositive + answerNegative;
    }
}
