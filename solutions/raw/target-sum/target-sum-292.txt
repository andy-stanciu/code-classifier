class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        dfs(0, 0, nums, target);
        return count;
    }

    public void dfs(int total, int index, int[] nums, int target) {
        if (index == nums.length) {
            if (total == target) {
                count++;
            }
            return;
        }
        // decision tree part, we use 2 recursion calls to do so.
        // adding the num
        dfs(total + nums[index], index + 1, nums, target);
        // subtracting the num
        dfs(total - nums[index], index + 1, nums, target);
    }
}
