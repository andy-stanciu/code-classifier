class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return target(nums, 0, 0, target);
    }

    public static int target(int[] nums, int pos, int sum, int target) {
        if (pos == nums.length) {
            return sum == target ? 1 : 0;
        }
        int ans = target(nums, pos + 1, sum + nums[pos], target) + target(nums, pos + 1, sum + (-nums[pos]), target);
        return ans;
    }
}
