class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        if (nums.length == 1) {
            if (nums[0] == target || nums[0] + target == 0)
                return 1;
            else
                return 0;
        }
        String s = "";
        for (int i : nums) s += Integer.toString(i);
        int[] c = { 0 };
        fun(nums, target, 0, c);
        return c[0];
    }

    static void fun(int[] nums, int target, int i, int[] c) {
        if ((target == 0) && i == nums.length) {
            c[0]++;
            return;
        }
        if (i >= nums.length) {
            return;
        }
        // for(int i=ind;i<s.length();i++){
        fun(nums, target - (nums[i]), i + 1, c);
        fun(nums, target + (nums[i]), i + 1, c);
        // }
    }
}
