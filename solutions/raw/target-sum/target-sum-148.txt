class Solution {

    static int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        count = 0;
        dfs(nums, -1, target, 0);
        return count;
    }

    public static void dfs(int[] nums, int index, int target, int sum) {
        if (index >= nums.length - 1) {
            if (sum == target) {
                count++;
            }
            return;
        }
        dfs(nums, index + 1, target, sum + nums[index + 1]);
        dfs(nums, index + 1, target, sum - nums[index + 1]);
    }
}
