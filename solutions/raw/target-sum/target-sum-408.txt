class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWaysHelper(nums, target, 0);
    }

    private int findTargetSumWaysHelper(int[] nums, int target, int i) {
        if (nums.length == i)
            return target == 0 ? 1 : 0;
        return findTargetSumWaysHelper(nums, target + nums[i], i + 1) + findTargetSumWaysHelper(nums, target - nums[i], i + 1);
    }
}
