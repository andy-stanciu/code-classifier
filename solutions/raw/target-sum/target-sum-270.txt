class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return help(nums, target, 0, 0);
    }

    private int help(int[] nums, int target, int idx, int sum) {
        if (idx == nums.length) {
            if (target == sum) {
                return 1;
            }
            return 0;
        }
        int help1 = help(nums, target, idx + 1, sum + nums[idx]);
        int help2 = help(nums, target, idx + 1, sum - nums[idx]);
        return help1 + help2;
    }
}
