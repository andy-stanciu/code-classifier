class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return rec(nums, 0, 0, target);
    }

    public int rec(int[] nums, int idx, int sum, int target) {
        if (idx == nums.length) {
            if (sum == target)
                return 1;
            return 0;
        }
        return rec(nums, idx + 1, sum + nums[idx], target) + rec(nums, idx + 1, sum - nums[idx], target);
    }
}
