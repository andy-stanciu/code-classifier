class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        solve(nums, 0, 0, target);
        return count;
    }

    private void solve(int[] nums, int i, int sum, int target) {
        if (i == nums.length) {
            if (sum == target)
                count += 1;
            return;
        } else {
            solve(nums, i + 1, sum + nums[i], target);
            solve(nums, i + 1, sum - nums[i], target);
        }
    }
}
