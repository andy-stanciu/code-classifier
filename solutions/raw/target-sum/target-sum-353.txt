class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        if (nums.length == 0)
            return 0;
        return helper(nums, target, nums[0], 1) + helper(nums, target, -nums[0], 1);
    }

    public int helper(int[] nums, int target, int current, int index) {
        if (index == nums.length) {
            if (target == current)
                return 1;
            else
                return 0;
        }
        return helper(nums, target, current + nums[index], index + 1) + helper(nums, target, current - nums[index], index + 1);
    }
}
