class Solution {

    public static int findTargetSumWays(int[] nums, int target) {
        //taking array of size 1 for pass by reference
        int[] res = new int[1];
        findPaths(0, nums, target, 0, res);
        return res[0];
    }

    public static void findPaths(int ind, int[] arr, int target, int sum, int[] res) {
        //base case - end of array
        if (ind == arr.length) {
            if (sum == target)
                res[0]++;
            return;
        }
        //take- +ve path
        findPaths(ind + 1, arr, target, sum + arr[ind], res);
        //not take- -ve path
        findPaths(ind + 1, arr, target, sum - arr[ind], res);
        return;
    }
}
