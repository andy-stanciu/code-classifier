class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return solution(nums, target, 0, 0);
    }

    public int solution(int[] nums, int target, int i, int sum) {
        if (i == nums.length) {
            return (sum == target ? 1 : 0);
        }
        int add = solution(nums, target, i + 1, sum + nums[i]);
        int sub = solution(nums, target, i + 1, sum - nums[i]);
        return add + sub;
    }
}
