class Solution {

    public int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0, 0);
        return count;
    }

    public void helper(int[] nums, int target, int index, int value) {
        if (index == nums.length) {
            if (value == target) {
                count++;
            }
            return;
        }
        helper(nums, target, index + 1, value + nums[index]);
        helper(nums, target, index + 1, value - nums[index]);
    }
}
