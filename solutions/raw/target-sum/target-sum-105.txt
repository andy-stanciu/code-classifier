class Solution {

    public int f(int[] nums, int target, int index) {
        if (index == 0) {
            int x = 0;
            if (nums[index] - target == 0) {
                x++;
            }
            if (nums[index] + target == 0) {
                x++;
            }
            return x;
        }
        int add = f(nums, target + nums[index], index - 1);
        int sub = f(nums, target - nums[index], index - 1);
        return add + sub;
    }

    public int findTargetSumWays(int[] nums, int target) {
        return f(nums, target, nums.length - 1);
    }
}
