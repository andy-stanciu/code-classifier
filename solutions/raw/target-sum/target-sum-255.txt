class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0, 0);
        return count;
    }

    private void helper(int[] nums, int target, int currSum, int index) {
        if (index == nums.length) {
            if (currSum == target) {
                count++;
            }
            return;
        }
        helper(nums, target, currSum + nums[index], index + 1);
        helper(nums, target, currSum - nums[index], index + 1);
    }
}
