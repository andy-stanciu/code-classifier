class Solution {

    public int getWays(int[] nums, int target, int idx) {
        if (idx >= nums.length) {
            return target == 0 ? 1 : 0;
        }
        return getWays(nums, target - nums[idx], idx + 1) + getWays(nums, target + nums[idx], idx + 1);
    }

    public int findTargetSumWays(int[] nums, int target) {
        return getWays(nums, target, 0);
    }
}
