class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int totalSum = 0;
        for (int num : nums) {
            totalSum += num;
        }
        if ((totalSum - target) < 0 || (totalSum - target) % 2 == 1) {
            return 0;
        }
        int k = ((totalSum - target) / 2);
        int[] prev = new int[k + 1];
        if (nums[0] == 0) {
            prev[0] = 2;
        } else {
            prev[0] = 1;
        }
        if (nums[0] != 0 && nums[0] <= k) {
            prev[nums[0]] = 1;
        }
        for (int ind = 1; ind < nums.length; ind++) {
            int[] curr = new int[k + 1];
            for (int targetSum = 0; targetSum <= k; targetSum++) {
                int pick = 0;
                if (nums[ind] <= targetSum) {
                    pick = prev[targetSum - nums[ind]];
                }
                int notPick = prev[targetSum];
                curr[targetSum] = pick + notPick;
            }
            prev = curr;
        }
        return prev[k];
    }
}
