class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWaysHelper(nums, target, target, 0);
    }

    public int findTargetSumWaysHelper(int[] nums, int target, int rem, int index) {
        int count = 0;
        if (index == nums.length) {
            if (rem == 0) {
                count++;
            }
            return count;
        }
        int c1 = findTargetSumWaysHelper(nums, target, rem - nums[index], index + 1);
        int c2 = findTargetSumWaysHelper(nums, target, rem + nums[index], index + 1);
        count = c1 + c2;
        return count;
    }
}
