class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int n = nums.length, k = 0;
        int ans = targetSum(nums, target, n - 1, k);
        return ans;
    }

    public int targetSum(int[] nums, int target, int n, int k) {
        if (n < 0 && target == k) {
            return 1;
        }
        if (n < 0 && target != k)
            return 0;
        return targetSum(nums, target, n - 1, k + nums[n]) + targetSum(nums, target, n - 1, k - nums[n]);
    }
}
