class Solution {

    int nWays = 0;

    public int findTargetSumWays(int[] nums, int target) {
        dfs(0, 0, target, nums);
        return nWays;
    }

    void dfs(int index, int sumSoFar, int target, int[] nums) {
        if (index == nums.length) {
            if (sumSoFar == target) {
                nWays++;
            }
            return;
        }
        dfs(index + 1, sumSoFar - nums[index], target, nums);
        dfs(index + 1, sumSoFar + nums[index], target, nums);
    }
}
