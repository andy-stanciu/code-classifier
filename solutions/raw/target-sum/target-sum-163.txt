class Solution {

    public int solve(int[] a, int s) {
        int n = a.length;
        int[][] dp = new int[n + 1][s + 1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= s; j++) {
                if (i == 0 && j == 0) {
                    dp[i][j] = 1;
                } else if (i == 0) {
                    dp[i][j] = 0;
                } else {
                    if (a[i - 1] <= j) {
                        dp[i][j] = dp[i - 1][j - a[i - 1]] + dp[i - 1][j];
                    } else {
                        dp[i][j] = dp[i - 1][j];
                    }
                }
            }
        }
        return dp[n][s];
    }

    public int findTargetSumWays(int[] nums, int target) {
        int tsum = 0;
        for (int i : nums) {
            tsum += i;
        }
        int s1 = (target + tsum) / 2;
        //this is because the sum after division is a decimal number which cannot be divided into subsets of 2
        if ((tsum + target) % 2 != 0 || s1 < 0) {
            return 0;
        }
        return solve(nums, s1);
    }
}
