class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        dfs(nums, 0, 0, target);
        return count;
    }

    public void dfs(int[] nums, int index, int sum, int target) {
        if (index == nums.length) {
            if (sum == target)
                count++;
            return;
        }
        int pos = nums[index];
        int neg = -1 * pos;
        //System.out.println(sum+pos+" "+index);
        dfs(nums, index + 1, sum + pos, target);
        dfs(nums, index + 1, sum + neg, target);
    }
}
