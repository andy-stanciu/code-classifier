class Solution {

    static int count(int[] nums, int t, int i, int s) {
        if (i == nums.length) {
            if (s == t) {
                return 1;
            } else {
                return 0;
            }
        }
        // counting if we use + and if we use - operation
        return count(nums, t, i + 1, s + nums[i]) + count(nums, t, i + 1, s - nums[i]);
    }

    public int findTargetSumWays(int[] nums, int target) {
        return count(nums, target, 0, 0);
    }
}
