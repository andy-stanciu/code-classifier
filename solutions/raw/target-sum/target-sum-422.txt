class Solution {

    int ans = 0;

    public int findTargetSumWays(int[] nums, int target) {
        getWays(nums, target, 0, 0);
        return ans;
    }

    public void getWays(int[] nums, int target, int sum, int index) {
        if (index == nums.length) {
            if (sum == target)
                ans++;
            return;
        }
        //choose + number
        getWays(nums, target, sum + nums[index], index + 1);
        //choose negative
        getWays(nums, target, sum - nums[index], index + 1);
    }
}
