class Solution {

    int ans = 0;

    public int findTargetSumWays(int[] nums, int target) {
        solver(nums, 0, 0, target);
        return ans;
    }

    public void solver(int[] nums, int index, int currSum, int target) {
        if (index == nums.length) {
            if (currSum == target)
                ans += 1;
            return;
        }
        solver(nums, index + 1, currSum + nums[index], target);
        solver(nums, index + 1, currSum - nums[index], target);
    }
}
