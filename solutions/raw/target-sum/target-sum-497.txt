class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        backtrack(nums, target, 0);
        return count;
    }

    public void backtrack(int[] nums, int target, int curr) {
        if (curr == nums.length && target == 0) {
            count++;
            return;
        }
        if (curr >= nums.length)
            return;
        int pos = nums[curr] * 1;
        int neg = nums[curr] * (-1);
        backtrack(nums, target - pos, curr + 1);
        backtrack(nums, target - neg, curr + 1);
    }
}
