class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int ans = helper(nums, target, 0);
        return ans;
    }

    private int helper(int[] nums, int target, int i) {
        int ans = 0;
        if (i >= nums.length) {
            return target == 0 ? 1 : 0;
        }
        ans = ans + helper(nums, target - nums[i] * (-1), i + 1);
        ans = ans + helper(nums, target - nums[i], i + 1);
        return ans;
    }
}
