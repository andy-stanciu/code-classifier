class Solution {

    int count;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0, 0);
        return count;
    }

    private void helper(int[] nums, int target, int index, int currentValue) {
        if (index == nums.length) {
            if (currentValue == target) {
                count++;
            }
            return;
        }
        helper(nums, target, index + 1, currentValue + nums[index]);
        helper(nums, target, index + 1, currentValue - nums[index]);
    }
}
