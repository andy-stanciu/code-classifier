class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return rec(0, nums, target);
    }

    public int rec(int idx, int[] nums, int target) {
        if (idx == nums.length) {
            if (target == 0) {
                return 1;
            } else {
                return 0;
            }
        }
        int put = rec(idx + 1, nums, target - (0 - nums[idx]));
        int notPut = rec(idx + 1, nums, target - nums[idx]);
        return put + notPut;
    }
}
