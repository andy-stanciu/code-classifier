class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return xn(nums, target, 0, 0);
    }

    public int xn(int[] v, int x, int a, int i) {
        if (i == v.length) {
            if (a == x)
                return 1;
            else
                return 0;
        }
        int z = 0;
        z += xn(v, x, a - v[i], i + 1);
        z += xn(v, x, a + v[i], i + 1);
        return z;
    }
}
