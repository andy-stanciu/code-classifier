class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetCount(0, nums, target);
    }

    private int findTargetCount(int curr, int[] nums, int target) {
        if (nums.length == 0) {
            if (curr == target) {
                return 1;
            }
            return 0;
        }
        int[] newNums = Arrays.copyOfRange(nums, 1, nums.length);
        return findTarget(curr + nums[0], newNums, target) + findTarget(curr - nums[0], newNums, target);
    }
}
