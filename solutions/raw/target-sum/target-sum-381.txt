class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        findTarget(nums, 0, target, 0);
        return count;
    }

    public void findTarget(int[] nums, int i, int target, int sum) {
        if (i == nums.length) {
            if (sum == target)
                count++;
            return;
        }
        for (int j = i; j <= i; j++) {
            findTarget(nums, i + 1, target, sum - nums[j]);
            findTarget(nums, i + 1, target, sum + nums[j]);
        }
    }
}
