class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return getExpressions(nums, nums.length, 0, target);
    }

    public static int getExpressions(int[] nums, int n, int sum, int target) {
        if (n == 0) {
            if (sum == target) {
                return 1;
            }
            return 0;
        }
        int t1 = getExpressions(nums, n - 1, sum + nums[n - 1], target) + getExpressions(nums, n - 1, sum - nums[n - 1], target);
        return t1;
    }
}
