class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        for (int n : nums) sum += n;
        sum += target;
        if (sum < 0 || sum % 2 == 1)
            return 0;
        sum /= 2;
        int[] dp = new int[sum + 1];
        dp[0] = 1;
        for (int n : nums) {
            for (int j = sum; j >= n; j--) {
                dp[j] += dp[j - n];
            }
        }
        return dp[sum];
    }
}
