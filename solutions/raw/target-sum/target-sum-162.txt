class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWays(0, nums, target);
    }

    private int findTargetSumWays(int i, int[] nums, int target) {
        if (i == nums.length) {
            if (target == 0)
                return 1;
            else
                return 0;
        }
        return findTargetSumWays(i + 1, nums, target - nums[i]) + findTargetSumWays(i + 1, nums, target + nums[i]);
    }
}
