class Solution {

    int res = 0;

    public int findTargetSumWays(int[] nums, int target) {
        backTracking(nums, target, 0, 0);
        return res;
    }

    public void backTracking(int[] nums, int target, int idx, int sum) {
        if (idx == nums.length) {
            if (sum == target)
                res++;
            return;
        }
        backTracking(nums, target, idx + 1, sum + nums[idx]);
        backTracking(nums, target, idx + 1, sum - nums[idx]);
    }
}
