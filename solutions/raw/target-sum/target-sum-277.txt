class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWays(nums, 0, 0, target);
    }

    public static int findTargetSumWays(int[] nums, int index, int ans, int target) {
        if (index == nums.length) {
            if (ans == target)
                return 1;
            else
                return 0;
        }
        return findTargetSumWays(nums, index + 1, ans + nums[index], target) + findTargetSumWays(nums, index + 1, ans - nums[index], target);
    }
}
