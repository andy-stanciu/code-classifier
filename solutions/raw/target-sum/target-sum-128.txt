class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        // dp[nums.length +1][2*sum of all elements+1]
        //  max will be some + 1000 and min qill be somme -1000 according to constraints we are storing idx and the sum at that idx
        int[][] dp = new int[21][2002];
        for (int i = 0; i < 21; i++) {
            Arrays.fill(dp[i], -1);
        }
        return rec(nums, target, 0, 0, dp);
    }

    public int rec(int[] nums, int target, int sum, int idx, int[][] dp) {
        if (idx == nums.length) {
            if (target == sum) {
                return 1;
            } else {
                return 0;
            }
        }
        if (dp[idx][sum + 1000] != -1) {
            return dp[idx][sum + 1000];
        }
        int left = rec(nums, target, sum + nums[idx], idx + 1, dp);
        int right = rec(nums, target, sum - nums[idx], idx + 1, dp);
        return dp[idx][sum + 1000] = left + right;
    }
}
