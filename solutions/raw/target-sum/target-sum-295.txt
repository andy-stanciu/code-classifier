class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return backtrack(nums, 0, target, 0);
    }

    public static int backtrack(int[] nums, int sum, int target, int i) {
        if (i == nums.length) {
            if (sum == target)
                return 1;
            return 0;
        }
        return backtrack(nums, sum - nums[i], target, i + 1) + backtrack(nums, sum + nums[i], target, i + 1);
    }
}
