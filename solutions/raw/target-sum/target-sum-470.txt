class Solution {

    int ans = 0;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0, 0);
        return ans;
    }

    public void helper(int[] nums, int target, int sum, int idx) {
        if (idx == nums.length) {
            if (sum == target)
                ans += 1;
            return;
        }
        int a = nums[idx];
        helper(nums, target, sum + a, idx + 1);
        helper(nums, target, sum - a, idx + 1);
    }
}
