class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return findTargetSumWaysHelper(nums, target, 0, 0);
    }

    private int findTargetSumWaysHelper(int[] nums, int target, int k, int sum) {
        if (k == nums.length) {
            return sum == target ? 1 : 0;
        }
        int countPositive = findTargetSumWaysHelper(nums, target, k + 1, sum + nums[k]);
        int countNegative = findTargetSumWaysHelper(nums, target, k + 1, sum - nums[k]);
        return countPositive + countNegative;
    }
}
