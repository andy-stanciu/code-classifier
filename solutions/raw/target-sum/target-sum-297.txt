class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return helper(nums, 0, 0, target);
    }

    int helper(int[] nums, int index, int sum, int target) {
        if (index == nums.length) {
            if (sum == target) {
                return 1;
            } else {
                return 0;
            }
        }
        int left = helper(nums, index + 1, sum + nums[index], target);
        int right = helper(nums, index + 1, sum - nums[index], target);
        return left + right;
    }
}
