class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return solve(0, nums, target, 0);
    }

    private int solve(int curr, int[] nums, int target, int sum) {
        if (curr > nums.length - 1) {
            if (sum == target) {
                return 1;
            }
            return 0;
        }
        return solve(curr + 1, nums, target, sum + nums[curr]) + solve(curr + 1, nums, target, sum - nums[curr]);
    }
}
