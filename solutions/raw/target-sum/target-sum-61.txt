class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        Map<Pair<Integer, Integer>, Integer> dp = new HashMap<>();
        return findTargetSumWaysInner(nums, target, 0, dp);
    }

    private int findTargetSumWaysInner(int[] nums, int target, int idx, Map<Pair<Integer, Integer>, Integer> dp) {
        if (idx == nums.length) {
            if (target == 0)
                return 1;
            return 0;
        }
        Pair<Integer, Integer> key = new Pair(idx, target);
        Integer dpValue = dp.get(key);
        if (dpValue != null)
            return dpValue;
        int newValue = findTargetSumWaysInner(nums, target - nums[idx], idx + 1, dp) + findTargetSumWaysInner(nums, target + nums[idx], idx + 1, dp);
        dp.put(key, newValue);
        return newValue;
    }
}
