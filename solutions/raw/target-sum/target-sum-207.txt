class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return sumWayHelper(nums, target, 0, 0);
    }

    private int sumWayHelper(int[] arr, int target, int counter, int index) {
        if (index == arr.length) {
            if (counter == target) {
                return 1;
            }
            return 0;
        }
        return sumWayHelper(arr, target, counter + arr[index], index + 1) + sumWayHelper(arr, target, counter - arr[index], index + 1);
    }
}
