class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        //passing an array because it will not be
        int[] count = { 0 };
        //passed as call by value
        sum(nums, target, 0, count, 0);
        return count[0];
    }

    public void sum(int[] nums, int target, int ptr, int[] count, int value) {
        //base condition
        if (ptr == nums.length) {
            if (value == target) {
                count[0]++;
            }
            return;
        }
        //checking both the possibilities for all
        sum(nums, target, ptr + 1, count, value - nums[ptr]);
        sum(nums, target, ptr + 1, count, value + nums[ptr]);
    }
}
