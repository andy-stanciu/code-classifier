class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return getTotalTargetSumWays(nums, target, 0, 0);
    }

    public int getTotalTargetSumWays(int[] nums, int target, int sum, int start) {
        if (start == nums.length) {
            return sum == target ? 1 : -1;
        }
        int positiveNum = getTotalTargetSumWays(nums, target, sum + nums[start], start + 1);
        int negativeNum = getTotalTargetSumWays(nums, target, sum - nums[start], start + 1);
        return (positiveNum < 0 ? 0 : positiveNum) + (negativeNum < 0 ? 0 : negativeNum);
    }
}
