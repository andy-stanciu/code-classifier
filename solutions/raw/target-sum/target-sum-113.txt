class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        // since target is negative also so use hashMap
        int n = nums.length;
        return solve(nums, n - 1, target);
    }

    public static int solve(int[] nums, int idx, int target) {
        if (idx == -1)
            return (target == 0) ? 1 : 0;
        int minus = solve(nums, idx - 1, target + nums[idx]);
        int plus = 0;
        if (Math.abs(target) >= 0)
            plus = solve(nums, idx - 1, target - nums[idx]);
        return minus + plus;
    }
}
