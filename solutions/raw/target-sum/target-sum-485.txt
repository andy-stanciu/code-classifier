class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return f(0, 0, target, nums);
    }

    public int f(int idx, int s, int target, int[] nums) {
        if (idx == nums.length) {
            if (s == target)
                return 1;
            return 0;
        }
        s += nums[idx];
        int l = f(idx + 1, s, target, nums);
        s = s - 2 * nums[idx];
        int r = f(idx + 1, s, target, nums);
        return l + r;
    }
}
