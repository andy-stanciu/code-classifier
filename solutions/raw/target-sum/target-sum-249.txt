class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return getTargetSumWays(nums, target, 0);
    }

    public int getTargetSumWays(int[] nums, int target, int i) {
        if (i == nums.length) {
            if (target == 0)
                return 1;
            else
                return 0;
        }
        return getTargetSumWays(nums, target - nums[i], i + 1) + getTargetSumWays(nums, target + nums[i], i + 1);
    }
}
