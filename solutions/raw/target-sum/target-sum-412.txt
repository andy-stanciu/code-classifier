class Solution {

    int count = 0;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0, 0);
        return count;
    }

    public void helper(int[] nums, int target, int currSum, int i) {
        if (i == nums.length && currSum == target) {
            count++;
            return;
        }
        if (i >= nums.length)
            return;
        int positive = currSum + nums[i];
        helper(nums, target, positive, i + 1);
        int negative = currSum - nums[i];
        helper(nums, target, negative, i + 1);
    }
}
