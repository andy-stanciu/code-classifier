class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return calculate(nums, 0, 0, target);
    }

    public int calculate(int[] nums, int i, int sum, int S) {
        if (i == nums.length) {
            if (sum == S) {
                return 1;
            }
        } else {
            int left = calculate(nums, i + 1, sum + nums[i], S);
            int right = calculate(nums, i + 1, sum - nums[i], S);
            return left + right;
        }
        return 0;
    }
}
