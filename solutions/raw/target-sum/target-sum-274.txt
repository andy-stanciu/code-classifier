class Solution {

    int[] dp;

    public int findTargetSumWays(int[] nums, int target) {
        int tsum = 0;
        for (int i : nums) {
            tsum += i;
        }
        if ((tsum - target) % 2 != 0)
            return 0;
        int sum = (tsum - target) / 2;
        if (sum < 0)
            return 0;
        dp = new int[sum + 1];
        dp[0] = 1;
        for (int i = 1; i <= nums.length; i++) {
            for (int s = sum; s >= 0; s--) {
                int take = 0;
                if (nums[i - 1] <= s) {
                    take = dp[s - nums[i - 1]];
                }
                int notTake = dp[s];
                dp[s] = take + notTake;
            }
        }
        return dp[sum];
    }
}
