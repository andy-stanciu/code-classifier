class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return rec(0, nums, target, 0);
    }

    int rec(int i, int[] n, int target, int t) {
        if (i >= n.length) {
            if (target == t) {
                return 1;
            } else {
                return 0;
            }
        }
        int su = rec(i + 1, n, target, t + n[i]);
        int sb = rec(i + 1, n, target, t - n[i]);
        return su + sb;
    }
}
