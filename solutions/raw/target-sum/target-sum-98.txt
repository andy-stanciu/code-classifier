class Solution {

    public int count;

    public int findTargetSumWays(int[] nums, int target) {
        count = 0;
        backtrack(nums, target, 0, 0);
        return count;
    }

    public void backtrack(int[] nums, int target, int sum, int index) {
        if (index == nums.length) {
            if (sum == target)
                ++count;
        } else {
            backtrack(nums, target, sum + nums[index], index + 1);
            backtrack(nums, target, sum - nums[index], index + 1);
        }
    }
}
