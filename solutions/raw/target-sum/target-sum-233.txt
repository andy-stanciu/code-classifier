class Solution {

    private int final_ans = 0;

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0, 0);
        return final_ans;
    }

    private void helper(int[] nums, int target, int local, int index) {
        if (index == nums.length && local == target)
            final_ans++;
        if (index == nums.length)
            return;
        helper(nums, target, local + nums[index], index + 1);
        helper(nums, target, local - nums[index], index + 1);
    }
}
