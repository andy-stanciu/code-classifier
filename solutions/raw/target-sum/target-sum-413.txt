public class Solution {

    public int findTargetSumWays(int[] nums, int target) {
        return find(nums, target, 0, 0, new HashMap<String, Integer>());
    }

    int find(int[] nums, int target, int i, int currSum, Map<String, Integer> memo) {
        if (i >= nums.length && currSum == target)
            return 1;
        if (i >= nums.length)
            return 0;
        String state = i + ":" + currSum;
        if (memo.get(state) != null)
            return memo.get(state);
        int pos = find(nums, target, i + 1, currSum + nums[i], memo);
        int neg = find(nums, target, i + 1, currSum - nums[i], memo);
        memo.put(state, pos + neg);
        return pos + neg;
    }
}
