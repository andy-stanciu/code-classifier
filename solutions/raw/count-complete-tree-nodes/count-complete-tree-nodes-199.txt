class Solution {

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        int left = height(root, true);
        int right = height(root, false);
        if (left == right)
            return (1 << left) - 1;
        else
            return 1 + countNodes(root.left) + countNodes(root.right);
    }

    int height(TreeNode root, boolean isLeft) {
        int count = 0;
        while (root != null) {
            count++;
            root = isLeft ? root.left : root.right;
        }
        return count;
    }
}
