/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int countVal = 0;

    public int countNodes(TreeNode root) {
        count(root);
        return countVal;
    }

    public void count(TreeNode root) {
        if (root == null)
            return;
        count(root.left);
        count(root.right);
        countVal++;
    }
}
