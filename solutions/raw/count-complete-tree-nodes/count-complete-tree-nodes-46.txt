class Solution {

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        int lh = leftHeight(root);
        int rh = rightHeight(root);
        if (lh == rh)
            return (1 << lh) - 1;
        return 1 + countNodes(root.left) + countNodes(root.right);
    }

    int leftHeight(TreeNode root) {
        int ht = 0;
        while (root != null) {
            ht++;
            root = root.left;
        }
        return ht;
    }

    int rightHeight(TreeNode root) {
        int ht = 0;
        while (root != null) {
            ht++;
            root = root.right;
        }
        return ht;
    }
}
