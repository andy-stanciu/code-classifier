class Solution {

    public int countNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lDepth = depthLeft(root.left);
        int rDepth = depthRight(root.right);
        if (lDepth == rDepth) {
            return 1 + 2 * ((1 << lDepth) - 1);
        } else {
            return 1 + countNodes(root.left) + countNodes(root.right);
        }
    }

    private static int depthLeft(TreeNode root) {
        int res = 0;
        for (TreeNode node = root; node != null; node = node.left) {
            res++;
        }
        return res;
    }

    private static int depthRight(TreeNode root) {
        int res = 0;
        for (TreeNode node = root; node != null; node = node.right) {
            res++;
        }
        return res;
    }
}
