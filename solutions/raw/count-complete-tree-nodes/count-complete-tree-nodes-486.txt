/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int max = 0;

    public int countNodes(TreeNode r) {
        max = 0;
        if (r != null) {
            cmax(r);
        }
        return max;
    }

    public void cmax(TreeNode r) {
        if (r == null)
            return;
        max++;
        cmax(r.left);
        cmax(r.right);
    }
}
