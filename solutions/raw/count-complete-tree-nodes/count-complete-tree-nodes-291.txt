/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int sum = 0;

    public int countNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        // Reset sum before starting the count
        sum = 0;
        helper(root);
        return sum;
    }

    private void helper(TreeNode root) {
        if (root == null) {
            return;
        }
        sum++;
        helper(root.left);
        helper(root.right);
    }
}
