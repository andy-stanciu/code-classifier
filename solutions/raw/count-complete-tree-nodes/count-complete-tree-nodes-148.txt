/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int c = 0;

    int count(TreeNode root) {
        if (root != null) {
            count(root.left);
            c++;
            count(root.right);
        }
        return c;
    }

    public int countNodes(TreeNode root) {
        c = 0;
        return count(root);
    }
}
