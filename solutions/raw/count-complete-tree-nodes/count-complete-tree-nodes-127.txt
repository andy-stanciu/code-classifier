/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countLeftHeigth(TreeNode node) {
        int count = 1;
        while (node.left != null) {
            count++;
            node = node.left;
        }
        return count;
    }

    public int countRightHeigth(TreeNode node) {
        int count = 1;
        while (node.right != null) {
            count++;
            node = node.right;
        }
        return count;
    }

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        int leftHeight = countLeftHeigth(root);
        int rightHeight = countRightHeigth(root);
        if (rightHeight == leftHeight) {
            return ((int) Math.pow(2, leftHeight) - 1);
        }
        return (1 + countNodes(root.left) + countNodes(root.right));
    }
}
