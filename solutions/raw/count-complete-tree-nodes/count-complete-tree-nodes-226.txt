class Solution {

    public int countNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int counter = 1;
        if (root.left != null) {
            counter += countNodes(root.left);
        }
        if (root.right != null) {
            counter += countNodes(root.right);
        }
        return counter;
    }
}
