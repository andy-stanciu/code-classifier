/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int findl(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.left;
        }
        System.out.println(height);
        return height;
    }

    public int findR(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.right;
        }
        return height;
    }

    public int countNodes(TreeNode root) {
        //if null then return 0
        if (root == null) {
            return 0;
        }
        int lh = findl(root);
        int rh = findR(root);
        if (lh == rh) {
            //this is a complete binary tree
            return (1 << lh) - 1;
            //2^lh -1 ie 2^3 -1
        }
        return 1 + countNodes(root.left) + countNodes(root.right);
    }
}
