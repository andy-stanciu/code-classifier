/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode root) {
        int[] cntNodes = new int[1];
        PreOrder(root, cntNodes);
        return cntNodes[0];
    }

    public static void PreOrder(TreeNode root, int[] cntNodes) {
        if (root == null) {
            return;
        }
        cntNodes[0] += 1;
        PreOrder(root.left, cntNodes);
        PreOrder(root.right, cntNodes);
    }
}
