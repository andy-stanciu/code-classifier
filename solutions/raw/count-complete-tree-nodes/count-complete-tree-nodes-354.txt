/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode root) {
        return countR(root);
    }

    private int countR(TreeNode root) {
        int count = 0;
        if (root == null)
            return 0;
        else
            count++;
        return count + countR(root.left) + countR(root.right);
    }
}
