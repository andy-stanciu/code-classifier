/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void count(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        count(root.left, list);
        count(root.right, list);
        list.add(root.val);
    }

    public int countNodes(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        if (root == null) {
            return list.size();
        }
        count(root, list);
        return list.size();
    }
}
