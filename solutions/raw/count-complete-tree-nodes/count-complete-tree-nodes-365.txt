class Solution {

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        int[] count = new int[1];
        counter(root, count);
        return count[0];
    }

    private void counter(TreeNode root, int[] count) {
        if (root == null)
            return;
        counter(root.left, count);
        count[0] = count[0] + 1;
        counter(root.right, count);
    }
}
