class Solution {

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        int[] count = new int[1];
        helper(root, count);
        return count[0];
    }

    private static void helper(TreeNode root, int[] count) {
        if (root == null)
            return;
        count[0] += 1;
        helper(root.left, count);
        helper(root.right, count);
    }
}
