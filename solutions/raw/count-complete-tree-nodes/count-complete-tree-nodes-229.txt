/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode root) {
        int count = 0;
        return rootCount(root, count);
    }

    public int rootCount(TreeNode root, int count) {
        if (root == null)
            return count;
        return 1 + rootCount(root.left, count) + rootCount(root.right, count);
    }
}
