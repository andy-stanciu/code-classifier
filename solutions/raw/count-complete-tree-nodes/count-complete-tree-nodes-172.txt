/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = getLhCount(root);
        int rh = getRhCount(root);
        if (lh == rh) {
            return (int) (Math.pow(2, lh) - 1);
        }
        return countNodes(root.left) + countNodes(root.right) + 1;
    }

    public int getLhCount(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return getLhCount(root.left) + 1;
    }

    public int getRhCount(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return getRhCount(root.right) + 1;
    }
}
