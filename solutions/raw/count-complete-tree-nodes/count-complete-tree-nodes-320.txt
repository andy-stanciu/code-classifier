/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    public int countNodes(TreeNode root) {
        if (root == null) {
            return count;
        }
        tree(root);
        return count;
    }

    public void tree(TreeNode root) {
        if (root == null) {
            return;
        }
        count++;
        tree(root.left);
        tree(root.right);
        return;
    }
}
