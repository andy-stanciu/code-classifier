/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    public int countNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return Number(root);
    }

    public int Number(TreeNode p) {
        if (p == null) {
            return 0;
        }
        count++;
        Number(p.left);
        Number(p.right);
        return count;
    }
}
