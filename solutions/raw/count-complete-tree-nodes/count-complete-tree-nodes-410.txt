/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int counter = 0;

    public int countNodes(TreeNode root) {
        counterFun(root);
        return counter;
    }

    public void counterFun(TreeNode root) {
        if (root == null) {
            return;
        }
        counter++;
        counterFun(root.left);
        counterFun(root.right);
    }
}
