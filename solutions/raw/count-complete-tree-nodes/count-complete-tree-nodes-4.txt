class Solution {

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        TreeNode left = root;
        TreeNode right = root;
        int lHeight = 0;
        int rHeight = 0;
        while (left != null) {
            left = left.left;
            lHeight++;
        }
        while (right != null) {
            right = right.right;
            rHeight++;
        }
        if (lHeight == rHeight) {
            return (1 << lHeight) - 1;
        }
        return 1 + countNodes(root.left) + countNodes(root.right);
    }
}
