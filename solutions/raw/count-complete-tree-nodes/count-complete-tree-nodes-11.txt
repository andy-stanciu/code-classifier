/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode root) {
        ArrayList<Integer> count = new ArrayList<Integer>();
        cnt(root, count);
        return count.size();
    }

    public static void cnt(TreeNode root, ArrayList<Integer> count) {
        if (root == null)
            return;
        count.add(1);
        cnt(root.left, count);
        cnt(root.right, count);
    }
}
