class Solution {

    public int countNodes(TreeNode root) {
        if (root == null)
            return 0;
        int leftHeight = getLeftHeight(root);
        int rightHeight = getRightHeight(root);
        if (leftHeight == rightHeight) {
            // If the height of the leftmost path is the same as the height of the rightmost path,
            // it means the tree is a perfect binary tree.
            return (1 << leftHeight) - 1;
        } else {
            // Otherwise, recursively count the nodes in the left and right subtrees and add 1 for the root.
            return 1 + countNodes(root.left) + countNodes(root.right);
        }
    }

    private int getLeftHeight(TreeNode node) {
        int height = 0;
        while (node != null) {
            height++;
            node = node.left;
        }
        return height;
    }

    private int getRightHeight(TreeNode node) {
        int height = 0;
        while (node != null) {
            height++;
            node = node.right;
        }
        return height;
    }
}
