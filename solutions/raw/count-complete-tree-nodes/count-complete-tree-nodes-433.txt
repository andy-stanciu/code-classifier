/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode curNode, int count) {
        if (curNode == null) {
            return count;
        }
        count++;
        count = countNodes(curNode.left, count);
        count = countNodes(curNode.right, count);
        return count;
    }

    public int countNodes(TreeNode root) {
        return countNodes(root, 0);
    }
}
