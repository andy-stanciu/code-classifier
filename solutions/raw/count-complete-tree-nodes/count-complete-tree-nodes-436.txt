/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int countNodes(TreeNode root) {
        //base case if root is null so return 0
        if (root == null)
            return 0;
        //call left tree and count nodes
        int leftCount = countNodes(root.left);
        //call right tree and count nodes
        int rightCount = countNodes(root.right);
        //final left and right nodes count and include root node by 1
        int countNode = leftCount + rightCount + 1;
        //return final result or totals nodes
        return countNode;
    }
}
