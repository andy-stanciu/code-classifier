class Solution {

    public void moveZeroes(int[] nums) {
        int i = 0;
        int j = 1;
        int temp = 0;
        while (j != nums.length) {
            if (nums[i] == 0) {
                if (nums[j] == 0) {
                    j++;
                    continue;
                }
                nums[i] = nums[j];
                nums[j] = 0;
                i++;
                j++;
                continue;
            }
            i++;
            j++;
        }
    }
}
