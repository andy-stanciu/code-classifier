class Solution {

    public void moveZeroes(int[] nums) {
        int readPointer = 0;
        int writePointer = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[readPointer] != 0) {
                nums[writePointer] = nums[readPointer];
                writePointer++;
            }
            readPointer++;
        }
        for (int i = writePointer; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
