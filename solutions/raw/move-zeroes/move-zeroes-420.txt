class Solution {

    public void moveZeroes(int[] nums) {
        int curr_ind = 0;
        int isZero = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                isZero++;
            } else {
                nums[curr_ind] = nums[i];
                curr_ind++;
            }
        }
        int lenval = nums.length - 1;
        while (isZero != 0) {
            nums[lenval] = 0;
            isZero--;
            lenval--;
        }
    }
}
