class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return;
        }
        int lastNonZeroAt = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[lastNonZeroAt] = nums[i];
                lastNonZeroAt++;
            }
        }
        for (int i = lastNonZeroAt; i < n; i++) {
            nums[i] = 0;
        }
    }
}
