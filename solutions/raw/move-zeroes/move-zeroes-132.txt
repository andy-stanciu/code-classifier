class Solution {

    public void moveZeroes(int[] nums) {
        boolean flag = true;
        int indexChecker = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0 && flag) {
                indexChecker = i;
                flag = false;
            }
            if (nums[i] != 0 && indexChecker != -1) {
                nums[indexChecker] = nums[i];
                nums[i] = 0;
                i = indexChecker;
                indexChecker = -1;
                flag = true;
            }
        }
    }
}
