class Solution {

    public void moveZeroes(int[] nums) {
        int i = -1;
        for (int k = 0; k < nums.length; k++) {
            if (nums[k] == 0) {
                i = k;
                break;
            }
        }
        if (i == -1) {
            return;
        }
        for (int j = i + 1; j < nums.length; j++) {
            if (nums[i] == 0 && nums[j] != 0) {
                nums[i] = nums[j];
                nums[j] = 0;
                i++;
            }
        }
    }
}
