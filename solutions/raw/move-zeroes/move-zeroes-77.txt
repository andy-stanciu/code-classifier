class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int cout = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[cout] = nums[i];
                cout++;
            }
        }
        if (cout < n) {
            while (cout != n) {
                nums[cout] = 0;
                cout++;
            }
        }
        for (int i = 0; i < n; i++) {
            System.out.print(nums[i]);
        }
    }
}
