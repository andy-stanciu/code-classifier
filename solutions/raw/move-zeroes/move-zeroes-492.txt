class Solution {

    public void moveZeroes(int[] nums) {
        // This will point to the first zero found
        int j = -1;
        // Find the first zero in the array
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                // Pointing to the first zero
                j = i;
                break;
            }
        }
        // If no zero is found, return immediately
        if (j == -1) {
            return;
        }
        // Move non-zero elements to the front and place zeros at the end
        for (int i = j + 1; i < nums.length; i++) {
            if (nums[i] != 0) {
                // Swap
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                // Move the zero pointer forward
                j++;
            }
        }
    }
}
