class Solution {

    public void moveZeroes(int[] nums) {
        //i is index of non zero element
        int i = 0;
        //move non zeros to front
        for (int num : nums) {
            if (num != 0) {
                nums[i] = num;
                i++;
            }
        }
        //place all 0s till the end
        while (i < nums.length) {
            nums[i] = 0;
            i++;
        }
    }
}
