class Solution {

    public void moveZeroes(int[] nums) {
        int last = nums[nums.length - 1];
        int left = 0;
        int curr = left + 1;
        int right = nums.length - 1;
        while (nums[right] == 0 && right > 0) {
            right--;
        }
        while (curr <= right) {
            if (nums[left] != 0) {
                left++;
                curr++;
            } else if (nums[left] == 0 && nums[curr] != 0) {
                int temp = nums[curr];
                nums[curr] = nums[left];
                nums[left] = temp;
                left++;
                curr++;
            } else {
                curr++;
            }
        }
    }
}
