class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int p = 0;
        for (int i = 0; i < nums.length; i++) {
            int x = nums[i];
            if (x != 0) {
                nums[p] = x;
                p++;
            }
        }
        for (int i = p; i < nums.length; i++) {
            nums[p] = 0;
            p++;
        }
    }
}
