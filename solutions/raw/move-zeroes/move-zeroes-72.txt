class Solution {

    public void moveZeroes(int[] nums) {
        if (nums == null || nums.length == 0)
            return;
        int position = 0;
        for (int n : nums) {
            if (n != 0) {
                nums[position] = n;
                position++;
            }
        }
        while (position < nums.length) {
            nums[position++] = 0;
        }
    }
}
