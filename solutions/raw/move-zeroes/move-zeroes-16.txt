class Solution {

    public void moveZeroes(int[] nums) {
        int k = 0;
        int countZero = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                countZero++;
            } else {
                nums[k++] = nums[i];
            }
        }
        for (int i = 0; i < countZero; i++) {
            nums[k++] = 0;
        }
    }
}
