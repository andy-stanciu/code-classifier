class Solution {

    public void moveZeroes(int[] nums) {
        int pos = 0;
        // if the current element is not  \'0\' then swap the element with position
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                int temp = nums[i];
                nums[i] = nums[pos];
                nums[pos] = temp;
                // now we incremant
                pos++;
            }
        }
    }
}
