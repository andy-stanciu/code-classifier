class Solution {

    public void moveZeroes(int[] nums) {
        int i = 0, j = 0;
        int n = nums.length;
        for (int ele = 0; ele < n; ele++) {
            if (nums[ele] != 0) {
                nums[i] = nums[ele];
                System.out.println(nums[i]);
                ++i;
            }
        }
        while (i < n) {
            nums[i++] = 0;
        }
    }
}
