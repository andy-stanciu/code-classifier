class Solution {

    public void moveZeroes(int[] nums) {
        int index = 0, p = 0;
        int n = nums.length;
        while (p < n) {
            if (nums[p] != 0)
                nums[index++] = nums[p];
            p++;
        }
        while (index < n) {
            nums[index++] = 0;
        }
    }
}
