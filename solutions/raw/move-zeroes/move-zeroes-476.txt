class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        // Pointer for the position of the next non-zero element
        int index = 0;
        // First pass: Move non-zero elements to the front
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                // Swap only if index and i are different
                if (index != i) {
                    int temp = nums[index];
                    nums[index] = nums[i];
                    nums[i] = temp;
                }
                index++;
            }
        }
    }
}
