class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int pt = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[pt] = nums[i];
                pt++;
            }
        }
        for (int i = pt; i < n; i++) {
            nums[i] = 0;
        }
    }
}
