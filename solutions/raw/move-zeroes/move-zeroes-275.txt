class Solution {

    public void moveZeroes(int[] arr) {
        int x = 0;
        boolean found = false;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != 0) {
                //until we do not find a zero it goes on
                //transferring allnon zero elements to top left
                arr[x] = arr[i];
                x++;
            } else {
                found = true;
            }
        }
        if (found == false) {
            return;
        }
        for (int i = x; i < arr.length; i++) {
            //now x starts from the index where all non zero ends
            //trasferring all zero to left put right most indices
            arr[i] = 0;
        }
    }
}
