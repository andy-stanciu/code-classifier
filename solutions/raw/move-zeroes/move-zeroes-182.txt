class Solution {

    public void moveZeroes(int[] nums) {
        int index = 0;
        // First pass: move all non-zero elements to the front
        for (int num : nums) {
            if (num != 0) {
                nums[index++] = num;
            }
        }
        // Second pass: fill the rest of the array with zeros
        while (index < nums.length) {
            nums[index++] = 0;
        }
    }
}
