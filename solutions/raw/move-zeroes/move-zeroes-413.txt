class Solution {

    public void moveZeroes(int[] arr) {
        // 2nd pointer
        int j = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != 0) {
                arr[j++] = arr[i];
            }
        }
        for (; j < arr.length; j++) {
            arr[j] = 0;
        }
    }
}
