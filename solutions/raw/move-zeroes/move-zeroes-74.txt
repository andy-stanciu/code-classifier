class Solution {

    public void moveZeroes(int[] nums) {
        int size = nums.length;
        int i = 0;
        int j = 0;
        while (j < size) {
            if (nums[j] != 0) {
                nums[i++] = nums[j];
            }
            j++;
        }
        while (i < size) {
            nums[i++] = 0;
        }
    }
}
