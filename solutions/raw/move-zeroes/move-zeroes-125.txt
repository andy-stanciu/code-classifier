class Solution {

    public void moveZeroes(int[] nums) {
        if (nums.length < 2) {
            return;
        }
        int length = nums.length;
        int pivot = 0;
        for (int i = 0; i < length; i++) {
            if (nums[i] != 0) {
                swap(nums, pivot, i);
                pivot++;
            }
        }
    }

    public void swap(int[] nums, int pivot, int i) {
        int temp = nums[i];
        nums[i] = nums[pivot];
        nums[pivot] = temp;
    }
}
