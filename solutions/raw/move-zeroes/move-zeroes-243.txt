class Solution {

    public void moveZeroes(int[] nums) {
        int i = 0, j = 0;
        int n = nums.length;
        // Swap only when nums[j] is non-zero
        while (i < n && j < n) {
            if (nums[j] != 0) {
                //Swapping elements using temporary variable
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                // Move \'i\' to the next position for a non-zero element to get placed
                i++;
            }
            // Move \'j\' to the next element to check
            j++;
        }
    }
}
