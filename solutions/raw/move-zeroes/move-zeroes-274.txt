class Solution {

    public void moveZeroes(int[] nums) {
        ArrayList<Integer> arr = new ArrayList<>();
        int n1 = nums.length;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                arr.add(nums[i]);
            }
        }
        int n2 = arr.size();
        for (int i = 0; i < n2; i++) {
            nums[i] = arr.get(i);
        }
        for (int i = n2; i < n1; i++) {
            nums[i] = 0;
        }
    }
}
