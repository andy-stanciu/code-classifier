class Solution {

    public void moveZeroes(int[] nums) {
        int[] arrs = new int[nums.length];
        int i = 0;
        int j = 0;
        for (int vars : nums) {
            if (vars != 0) {
                arrs[i] = vars;
                i++;
            }
        }
        for (int varks : arrs) {
            nums[j] = varks;
            j++;
        }
    }
}
