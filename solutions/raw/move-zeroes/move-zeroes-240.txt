class Solution {

    public void moveZeroes(int[] nums) {
        int j = -1;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                j = i;
                // find first 0\'s position and change j to it
                break;
            }
        }
        if (j == -1)
            return;
        for (int i = j + 1; i < n; i++) {
            if (nums[i] != 0) {
                int temp = nums[i];
                // swap 0 to right in all iterations
                nums[i] = nums[j];
                nums[j] = temp;
                j++;
            }
        }
    }
}
