class Solution {

    public void moveZeroes(int[] nums) {
        int j = 0, t;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                t = nums[i];
                nums[i] = nums[j];
                nums[j] = t;
                j++;
            }
        }
    }
}
