class Solution {

    public void moveZeroes(int[] nums) {
        int count = nums.length - 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                int j = i;
                while (nums[j] == 0 && j + 1 < nums.length) {
                    j++;
                }
                nums[i] = nums[j];
                nums[j] = 0;
            }
        }
    }
}
