class Solution {

    public void moveZeroes(int[] nums) {
        // TWO POINTER APPROACH
        //1- we find zero in the array
        int n = nums.length;
        int j = -1;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                j = i;
                break;
            }
        }
        //2- if we don\'t find any zero in the array
        if (j == -1)
            return;
        //3- if we find first zero in the array
        for (int i = j + 1; i < n; i++) {
            if (nums[i] != 0) {
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                j++;
            }
        }
        return;
    }
}
