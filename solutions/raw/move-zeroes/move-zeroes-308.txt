class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        int i = 0, j = 0;
        while (i < n) {
            if (nums[i] != 0) {
                arr[j] = nums[i];
                j++;
            }
            i++;
        }
        while (j < n) {
            arr[j] = 0;
            j++;
        }
        for (int k = 0; k < n; k++) {
            nums[k] = arr[k];
        }
    }
}
