class Solution {

    public void moveZeroes(int[] nums) {
        int non_zero = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[non_zero++] = nums[i];
            }
        }
        while (non_zero < n) nums[non_zero++] = 0;
    }
}
