class Solution {

    public void moveZeroes(int[] nums) {
        int legal_index = 0;
        int i = 0;
        while (i < nums.length) {
            if (nums[i] != 0) {
                swap(nums, i, legal_index);
                legal_index++;
            }
            i++;
        }
    }

    public void swap(int[] arr, int i, int j) {
        int k = arr[i];
        arr[i] = arr[j];
        arr[j] = k;
    }
}
