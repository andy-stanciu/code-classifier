class Solution {

    public void moveZeroes(int[] nums) {
        int[] arr = nums;
        int c = 0;
        for (int i : arr) {
            if (i == 0)
                c++;
        }
        int j = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != 0) {
                nums[j] = arr[i];
                j++;
            }
        }
        while (j < nums.length) {
            nums[j] = 0;
            j++;
        }
    }
}
