class Solution {

    public void moveZeroes(int[] nums) {
        int[] nonZero = new int[nums.length];
        int i = 0;
        for (int num : nums) {
            if (num != 0) {
                nonZero[i++] = num;
            }
        }
        for (int j = 0; j < nonZero.length; j++) {
            nums[j] = nonZero[j];
        }
    }
}
