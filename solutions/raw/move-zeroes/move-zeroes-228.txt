class Solution {

    public void moveZeroes(int[] nums) {
        int j = 0;
        if (nums.length == 1) {
            if (nums[0] == 0)
                return;
        } else {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] != 0) {
                    int val = nums[i];
                    nums[i] = 0;
                    nums[j] = val;
                    j++;
                }
            }
        }
    }
}
