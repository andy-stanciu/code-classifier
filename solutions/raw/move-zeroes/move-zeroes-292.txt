class Solution {

    public void moveZeroes(int[] arr) {
        // take 2 pointers left and right
        // understand it this way :)
        // left wants to be at 0
        // right wants to be at non zero
        // whenever the above two condition satisfies , swap them
        // once you swap both of them, conditions change and need to incerement both
        int l = 0;
        int r = 0;
        while (r < arr.length) {
            if (arr[r] == 0) {
                r++;
            } else {
                // if not at same index
                if (l != r) {
                    int temp = arr[l];
                    arr[l] = arr[r];
                    arr[r] = temp;
                }
                l++;
                r++;
            }
        }
    }
}
