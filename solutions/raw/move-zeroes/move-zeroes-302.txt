class Solution {

    public void moveZeroes(int[] nums) {
        int index = fillNonZeroes(nums);
        fillZeroes(nums, index);
    }

    private int fillNonZeroes(int[] nums) {
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[index++] = nums[i];
            }
        }
        return index;
    }

    private void fillZeroes(int[] nums, int index) {
        for (int i = index; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
