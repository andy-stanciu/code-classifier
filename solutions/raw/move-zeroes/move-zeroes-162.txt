class Solution {

    public void moveZeroes(int[] nums) {
        int len = nums.length;
        int j = 0;
        int count = 0;
        for (int i = 0; i < len; i++) {
            if (nums[i] == 0) {
                count++;
                continue;
            } else {
                nums[j++] = nums[i];
            }
        }
        for (int i = j; i < len; i++) {
            nums[i] = 0;
        }
    }
}
