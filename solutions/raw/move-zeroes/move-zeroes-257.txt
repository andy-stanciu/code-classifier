class Solution {

    public void moveZeroes(int[] nums) {
        int i = 0;
        int count = 0;
        for (int num : nums) {
            if (num == 0)
                count++;
            else
                nums[i++] = num;
            if (i + count >= nums.length)
                break;
        }
        while (i < nums.length) {
            nums[i++] = 0;
        }
    }
}
