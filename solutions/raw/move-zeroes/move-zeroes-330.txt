class Solution {

    public void moveZeroes(int[] nums) {
        int l = nums.length;
        int counter = 0;
        for (int i = 0; i < l; i++) {
            if (nums[i] != 0) {
                nums[counter] = nums[i];
                counter++;
            }
        }
        int x = l - counter;
        for (int i = counter; i < l; i++) {
            nums[i] = 0;
        }
    }
}
