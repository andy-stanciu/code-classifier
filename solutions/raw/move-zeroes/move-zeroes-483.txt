class Solution {

    public void moveZeroes(int[] nums) {
        int len = nums.length;
        int lastCount = 0;
        for (int i = 0; i < len; i++) {
            if (nums[i] != 0) {
                nums[lastCount] = nums[i];
                lastCount++;
            }
        }
        for (int i = lastCount; i < len; i++) {
            nums[lastCount] = 0;
            lastCount++;
        }
    }
}
