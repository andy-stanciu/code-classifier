class Solution {

    public void moveZeroes(int[] nums) {
        int i, j = 0, n = nums.length;
        int count = 0, k;
        for (i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[j] = nums[i];
                j++;
            }
        }
        for (k = j; k < n; k++) {
            nums[k] = 0;
        }
    }
}
