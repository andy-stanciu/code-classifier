class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        // Pointer for the position to place non-zero elements
        int j = 0;
        // Move all non-zero elements to the front
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[j] = nums[i];
                j++;
            }
        }
        // Fill the rest of the array with zeros
        for (int i = j; i < n; i++) {
            nums[i] = 0;
        }
    }
}
