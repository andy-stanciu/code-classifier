class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        if (n < 2) {
            System.out.print(0);
            return;
        }
        int i = 0;
        int j = -1;
        while (i != n) {
            if (nums[i] != 0) {
                j++;
                int temp = nums[j];
                nums[j] = nums[i];
                nums[i] = temp;
            }
            i++;
        }
    }
}
