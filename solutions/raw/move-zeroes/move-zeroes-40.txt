class Solution {

    public void moveZeroes(int[] nums) {
        int dst = -1;
        // Find first zero
        while (++dst < nums.length && nums[dst] != 0) ;
        int src = dst;
        while (++src < nums.length) if (// Skip
        nums[src] != 0)
            // or copy content
            nums[dst++] = nums[src];
        while (// Fill with zeroes to the end!
        dst < nums.length) nums[dst++] = 0;
    }
}
