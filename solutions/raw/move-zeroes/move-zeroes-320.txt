class Solution {

    public void moveZeroes(int[] nums) {
        int j = 0;
        int count = 0;
        for (int i = 0; i <= nums.length - 1; i++) {
            if (nums[i] != 0) {
                nums[j] = nums[i];
                count++;
                j++;
            }
        }
        for (int i = count; i <= nums.length - 1; i++) {
            nums[i] = 0;
        }
    }
}
