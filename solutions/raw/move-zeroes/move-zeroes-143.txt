class Solution {

    public void moveZeroes(int[] nums) {
        Queue<Integer> idxZero = new LinkedList<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                idxZero.add(i);
            } else {
                if (!idxZero.isEmpty()) {
                    nums[idxZero.poll()] = nums[i];
                    nums[i] = 0;
                    idxZero.add(i);
                }
            }
        }
    }
}
