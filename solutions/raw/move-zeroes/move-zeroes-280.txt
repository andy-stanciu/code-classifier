class Solution {

    public void moveZeroes(int[] nums) {
        int lastNonZero = 0;
        for (int curr = 0; curr < nums.length; curr++) {
            if (nums[curr] != 0) {
                nums[lastNonZero] = nums[curr];
                lastNonZero++;
            }
        }
        for (int i = lastNonZero; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
