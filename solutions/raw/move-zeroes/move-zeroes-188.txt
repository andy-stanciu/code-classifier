class Solution {

    public void moveZeroes(int[] nums) {
        if (nums.length > 1) {
            int i = 0;
            int lastIndx = nums.length;
            while (i < lastIndx) {
                if (nums[i] == 0) {
                    for (int j = i + 1; j < nums.length; j++) {
                        nums[j - 1] = nums[j];
                    }
                    nums[nums.length - 1] = 0;
                    i = 0;
                    lastIndx--;
                } else {
                    i++;
                }
            }
        }
    }
}
