class Solution {

    public void moveZeroes(int[] nums) {
        int idxForNonZero = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                int temp = nums[idxForNonZero];
                nums[idxForNonZero] = nums[i];
                nums[i] = temp;
                idxForNonZero++;
            }
        }
    }
}
