class Solution {

    public void moveZeroes(int[] nums) {
        // Initialize a count variable to keep track of non-zero elements
        int count = 0;
        // Create a new array with the same length as the input array nums
        int[] arr = new int[nums.length];
        // Initialize a variable j to keep track of the index in the new array arr
        int j = 0;
        // Loop through the elements of the input array nums
        for (int i = 0; i < nums.length; i++) {
            // If the current element of nums is non-zero
            if (nums[i] != 0) {
                // Copy the non-zero element to the next available position in arr
                arr[j] = nums[i];
                // Increment the index j for the next position in arr
                j++;
            }
        }
        // Copy elements from arr back to nums
        for (int i = 0; i < arr.length; i++) {
            nums[i] = arr[i];
        }
    }
}
