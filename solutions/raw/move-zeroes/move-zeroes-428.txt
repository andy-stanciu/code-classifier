class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int ind = -1;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                ind = i;
                break;
            }
        }
        if (ind == -1)
            return;
        for (int i = ind + 1; i < n; i++) {
            if (nums[i] != 0) {
                int t = nums[ind];
                nums[ind] = nums[i];
                nums[i] = t;
                ind++;
            }
        }
    }
}
