class Solution {

    public void moveZeroes(int[] nums) {
        int counter = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                counter++;
            } else {
                int temp = nums[i];
                nums[i] = nums[i - counter];
                nums[i - counter] = temp;
            }
        }
    }
}
