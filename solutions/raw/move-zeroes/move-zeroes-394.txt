class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        int k = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0)
                arr[k++] = nums[i];
        }
        for (int i = k; i < n; i++) {
            arr[k++] = 0;
        }
        for (int i = 0; i < n; i++) {
            nums[i] = arr[i];
        }
    }
}
