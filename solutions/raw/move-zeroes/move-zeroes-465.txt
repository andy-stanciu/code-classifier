class Solution {

    public void moveZeroes(int[] nums) {
        int ele = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                ele = i;
                break;
            }
        }
        if (ele == -1) {
            return;
        }
        for (int i = ele + 1; i < nums.length; i++) {
            if (nums[i] != 0) {
                int temp = nums[ele];
                nums[ele] = nums[i];
                nums[i] = temp;
                ele++;
            }
        }
    }
}
