class Solution {

    public void moveZeroes(int[] nums) {
        int zero = 0, j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                zero++;
            else {
                nums[j++] = nums[i];
            }
        }
        while (j < nums.length && zero > 0) {
            nums[j++] = 0;
            zero--;
        }
    }
}
