class Solution {

    public void moveZeroes(int[] nums) {
        if (nums.length == 1)
            return;
        int i = 0;
        int c = 0;
        while (i < nums.length) {
            if (nums[i] == 0)
                c++;
            else if (c != 0) {
                nums[i - c] = nums[i];
                nums[i] = 0;
            }
            i++;
        }
    }
}
