class Solution {

    public void moveZeroes(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        int[] result = new int[nums.length];
        int resultIndex = 0;
        while (start <= end) {
            if (nums[start] != 0) {
                result[resultIndex] = nums[start];
                resultIndex++;
            }
            start++;
        }
        for (int i = 0; i < nums.length; i++) {
            nums[i] = result[i];
        }
    }
}
