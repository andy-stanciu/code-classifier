class Solution {

    public void moveZeroes(int[] nums) {
        int l = nums.length;
        int k = 0;
        int[] arr = new int[l];
        for (int i = 0; i < l; i++) {
            if (nums[i] != 0) {
                arr[k++] = nums[i];
            }
        }
        for (int i = 0; i < l; i++) {
            if (nums[i] == 0) {
                arr[k++] = nums[i];
            }
        }
        for (int i = 0; i < l; i++) {
            nums[i] = arr[i];
        }
    }
}
