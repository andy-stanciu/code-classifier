// Time complexity = O(n)
// Space complexity = O(1)
class Solution {

    public void moveZeroes(int[] nums) {
        int swapIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                int temp = nums[swapIndex];
                nums[swapIndex++] = nums[i];
                nums[i] = temp;
            }
        }
    }
}
