class Solution {

    public void moveZeroes(int[] nums) {
        int insertionPtr = 0;
        int n = nums.length;
        for (int traversalPtr = 0; traversalPtr < n; traversalPtr++) {
            if (nums[traversalPtr] != 0) {
                nums[insertionPtr] = nums[traversalPtr];
                insertionPtr++;
            }
        }
        for (int i = insertionPtr; i < n; i++) {
            if (nums[i] != 0) {
                nums[i] = 0;
            }
        }
    }
}
