class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int p1 = 0;
        int p2 = 1;
        while (p2 < n) {
            //swap zeros
            if (nums[p1] == 0 && nums[p2] != 0) {
                int temp = nums[p1];
                nums[p1] = nums[p2];
                nums[p2] = temp;
            }
            if (nums[p1] != 0) {
                p1++;
            }
            p2++;
        }
    }
}
