class Solution {

    public void moveZeroes(int[] nums) {
        int zero;
        for (int i = nums.length - 1; i > -1; i--) {
            if (nums[i] == 0) {
                for (int j = i; j < nums.length - 1; j++) {
                    if (nums[j + 1] == 0) {
                        break;
                    }
                    nums[j] = nums[j + 1];
                    nums[j + 1] = 0;
                }
            }
        }
    }
}
