class Solution {

    public void moveZeroes(int[] nums) {
        int left = 0;
        int right = 1;
        while (right < nums.length) {
            /**
             *            Currently both pointers are 0 0 so just move right pointer
             */
            if (nums[left] == 0 && nums[right] == 0) {
                right++;
                continue;
            } else //Left is 0 and right is not zero - SWAP
            if (nums[left] == 0 && nums[right] != 0) {
                int temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
            }
            //Move both pointers after a successful swap
            left++;
            right++;
        }
    }
}
