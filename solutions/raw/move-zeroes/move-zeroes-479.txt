class Solution {

    int[] moveZeroesToRight(int[] nums, int len) {
        int index = 0;
        int i = 0;
        while (i < len) {
            if (nums[i] != 0) {
                int temp = nums[i];
                nums[i++] = nums[index];
                nums[index++] = temp;
            } else {
                ++i;
            }
        }
        return nums;
    }

    public int[] moveZeroesToLeft(int[] nums, int len) {
        int index = 0;
        int i = 0;
        int temp;
        while (i < len) {
            if (nums[i] == 0) {
                temp = nums[i];
                nums[i++] = nums[index];
                nums[index++] = temp;
            } else {
                i++;
            }
        }
        return nums;
    }

    public void moveZeroes(int[] nums) {
        moveZeroesToRight(nums, nums.length);
        // moveZeroesToLeft(nums, nums.length);
    }
}
