class Solution {

    public void moveZeroes(int[] nums) {
        // Pointer to track the position where the next non-zero element should go
        int nonZeroIndex = 0;
        // Traverse through the array and place non-zero elements at the nonZeroIndex
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                // Swap or move the non-zero element to the front
                nums[nonZeroIndex] = nums[i];
                nonZeroIndex++;
            }
        }
        // After all non-zero elements are placed, fill the rest with 0s
        for (int i = nonZeroIndex; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
