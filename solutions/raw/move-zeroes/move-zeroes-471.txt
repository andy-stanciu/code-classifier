class Solution {

    public void moveZeroes(int[] nums) {
        int ptr1 = 0, ptr2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[ptr1] != 0) {
                int temp = nums[ptr1];
                nums[ptr1] = nums[ptr2];
                nums[ptr2] = temp;
                ptr2++;
            }
            ptr1++;
        }
    }
}
