class Solution {

    public void moveZeroes(int[] nums) {
        var n = nums.length;
        var lastNonNullIndex = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[lastNonNullIndex] = nums[i];
                lastNonNullIndex++;
            }
        }
        for (int i = lastNonNullIndex; i < n; i++) {
            nums[i] = 0;
        }
    }
}
