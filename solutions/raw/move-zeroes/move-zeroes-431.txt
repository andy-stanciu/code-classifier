class Solution {

    public void moveZeroes(int[] nums) {
        int p1 = 0;
        int temp;
        int p2 = 1;
        while (p2 < nums.length) {
            if (nums[p2] != 0) {
                if (nums[p1] == 0) {
                    // temp = nums[p1];         // you can also use
                    // nums[p1] = nums[p2];     // this for swapping
                    // nums[p2] = temp;         // algorithm
                    nums[p1] = nums[p2];
                    nums[p2] = 0;
                    p1++;
                }
            }
            if (nums[p1] != 0) {
                p1++;
            }
            p2++;
        }
    }
}
