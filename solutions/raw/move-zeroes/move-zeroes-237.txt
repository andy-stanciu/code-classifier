class Solution {

    public void moveZeroes(int[] nums) {
        int left = 0;
        int target = 0;
        while (left < nums.length) {
            if (nums[left] != 0) {
                nums[target++] = nums[left];
            }
            left++;
        }
        while (target < nums.length) {
            nums[target++] = 0;
        }
    }
}
