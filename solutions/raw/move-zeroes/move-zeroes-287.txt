class Solution {

    public void moveZeroes(int[] nums) {
        int l = nums.length;
        int i = 0;
        for (int j = 0; j < l; j++) {
            if (nums[j] != 0) {
                swap(nums, i, j);
                i++;
            }
        }
        System.out.println(Arrays.toString(nums));
    }

    public void swap(int[] nums, int a, int b) {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
}
