class Solution {

    public void moveZeroes(int[] nums) {
        int nonZeroPosition = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[nonZeroPosition] = nums[i];
                nonZeroPosition++;
            }
        }
        for (int i = nonZeroPosition; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
