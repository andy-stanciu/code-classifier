class Solution {

    public void moveZeroes(int[] nums) {
        int zero = 0, nonZero = 0;
        int i = 0;
        while (i < nums.length) {
            if (nums[i] != 0) {
                if (i != nonZero) {
                    int temp = nums[i];
                    nums[i] = nums[nonZero];
                    nums[nonZero] = temp;
                }
                nonZero++;
            }
            i++;
        }
    }
}
