class Solution {

    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        int j = 1, k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                arr[n - j] = nums[i];
            } else {
                arr[k++] = nums[i];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            nums[i] = arr[i];
        }
    }
}
