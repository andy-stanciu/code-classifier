class Solution {

    public void moveZeroes(int[] nums) {
        int countOfZeros = 0;
        for (int i = 0; i < nums.length; ++i) {
            if (nums[i] == 0) {
                countOfZeros++;
            } else if (countOfZeros > 0) {
                nums[i - countOfZeros] = nums[i];
                nums[i] = 0;
            }
        }
    }
}
