class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> answer = new ArrayList();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            if (//avoid dups
            i > 0 && nums[i] == nums[i - 1])
                continue;
            int sum = 0;
            while (j < k) {
                if (//avoid dups
                j > i + 1 && nums[j] == nums[j - 1]) {
                    j++;
                    continue;
                } else if (//avoid dups
                k < nums.length - 1 && nums[k] == nums[k + 1]) {
                    k--;
                    continue;
                }
                // check sum
                sum = nums[i] + nums[j] + nums[k];
                if (sum == 0) {
                    answer.add(new ArrayList(List.of(nums[i], nums[j], nums[k])));
                    j++;
                    k--;
                } else if (sum < 0) {
                    j++;
                } else
                    k--;
            }
        }
        return answer;
    }
}
