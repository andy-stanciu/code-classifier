class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> output = new ArrayList<>();
        // In order to avoid hashmap lookup, reduce time complexity by 2 pointer approach,
        // we sort the given array
        Arrays.sort(nums);
        int n = nums.length;
        // 2 for loops.
        // keep outer for loop value as k.  k will go until nums[n-3]
        for (int k = 0; k < n - 2; k++) {
            // to get distinct output, this check is made
            if (k == 0 || nums[k] != nums[k - 1]) {
                // two pointers where i starts k+1 and j starts n-1.
                // incrementor
                int i = k + 1;
                // decrementor
                int j = n - 1;
                int sum = 0;
                while (i < j) {
                    sum = nums[k] + nums[i] + nums[j];
                    if (sum > 0)
                        --j;
                    else if (sum < 0)
                        ++i;
                    else {
                        // if sum is equal to 0 then add to output
                        output.add(new ArrayList(Arrays.asList(nums[k], nums[i], nums[j])));
                        --j;
                        // to get distinct output, this check is made
                        while (i < j && nums[j] == nums[j + 1]) --j;
                    }
                }
            }
        }
        return output;
    }
}
