class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> a = new ArrayList<>();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            if (i == 0 || i > 0 && nums[i] != nums[i - 1]) {
                int lb = i + 1;
                int ub = nums.length - 1;
                int sum = 0 - nums[i];
                while (lb < ub) {
                    if (nums[lb] + nums[ub] == sum) {
                        a.add(Arrays.asList(nums[i], nums[lb], nums[ub]));
                        // to avoid duplicates within the pointers window
                        while (lb < ub && nums[lb] == nums[lb + 1]) lb++;
                        while (lb < ub && nums[ub] == nums[ub - 1]) ub--;
                        lb++;
                        ub--;
                    } else if (nums[lb] + nums[ub] > sum) {
                        ub--;
                    } else {
                        lb++;
                    }
                }
            }
        }
        return a;
    }
}
