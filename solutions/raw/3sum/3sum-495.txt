class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        //Sort the array to reduce complexity of the problem
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            // if duplicate element are there skip the element
            if (nums[i] > 0 || i > 0 && nums[i] == nums[i - 1])
                continue;
            // start the
            int start = i + 1;
            int end = nums.length - 1;
            while (start < end) {
                int sum = nums[i] + nums[start] + nums[end];
                // if sum is zero add the all three number in list
                if (sum == 0) {
                    result.add(Arrays.asList(nums[i], nums[start], nums[end]));
                    // Skip the duplicate element from starting
                    while (start < end && nums[start] == nums[start + 1]) start++;
                    // Skip the duplicate element from ending
                    while (start < end && nums[end] == nums[end - 1]) end--;
                    // move the pointer
                    start++;
                    end--;
                } else // Sum is less than zero, increment start pointer to increase the sum
                if (sum < 0)
                    start++;
                else
                    // Sum is greater than zero, increment end pointer to decrease the sum
                    end--;
            }
            // while
        }
        // for
        return result;
    }
}
