class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        //Size is less than 3 return empty List;
        if (nums == null || nums.length < 3)
            return Collections.emptyList();
        Set<List<Integer>> resultSet = new HashSet<>();
        //sorted the array to in ascending order
        Arrays.sort(nums);
        //First loop to keep track of first number in the possible solution of array.
        for (int low = 0; low < nums.length - 2; low++) {
            int mid = low + 1;
            int high = nums.length - 1;
            //This Loop would provided better insight of possible array having sum as zero .
            while (mid < high) {
                int sum = nums[low] + nums[mid] + nums[high];
                if (sum == 0) {
                    resultSet.add(new ArrayList<>(Arrays.asList(nums[low], nums[mid], nums[high])));
                    mid++;
                    high--;
                }
                if (sum < 0) {
                    mid++;
                }
                if (sum > 0) {
                    high--;
                }
            }
        }
        return new ArrayList<>(resultSet);
    }
}
