class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet<>();
        int[] finalArray = new int[] {};
        //Storing the element of nums1 in HashMap
        Map<Integer, Boolean> map = new HashMap<>();
        for (var getTheElements : nums1) {
            map.put(getTheElements, true);
        }
        //Here checking the common elements and storing in the Set to avoid duplicate
        for (var findInterSection : nums2) {
            if (map.containsKey(findInterSection)) {
                set.add(findInterSection);
            }
        }
        Object[] array = set.toArray();
        finalArray = new int[set.size()];
        for (int i = 0; i < set.size(); i++) {
            //At last type casting to int[] from Object[]
            finalArray[i] = (Integer) array[i];
        }
        return finalArray;
    }
}
