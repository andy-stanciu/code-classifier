class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> m = new HashSet<>();
        for (int i = 0; i < nums1.length; i++) {
            m.add(nums1[i]);
        }
        Set<Integer> n = new HashSet<>();
        for (int i = 0; i < nums2.length; i++) {
            n.add(nums2[i]);
        }
        n.retainAll(m);
        int[] ans = n.stream().mapToInt(i -> i).toArray();
        return ans;
    }
}
