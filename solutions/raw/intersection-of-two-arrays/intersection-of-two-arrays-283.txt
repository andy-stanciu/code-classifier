class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> containsOne = new HashSet<Integer>();
        Set<Integer> containsBoth = new HashSet<Integer>();
        for (int i = 0; i < nums1.length; i++) {
            containsOne.add(nums1[i]);
        }
        for (int j = 0; j < nums2.length; j++) {
            if (containsOne.contains(nums2[j]))
                containsBoth.add(nums2[j]);
        }
        int index = 0;
        int[] result = new int[containsBoth.size()];
        for (int i : containsBoth) {
            result[index] = i;
            index++;
        }
        return result;
    }
}
