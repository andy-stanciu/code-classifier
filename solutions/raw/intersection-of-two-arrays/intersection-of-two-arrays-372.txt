class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        //sorting the arrays
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        //create two hashsets for both arrays
        HashSet<Integer> set1 = new HashSet<Integer>();
        HashSet<Integer> set2 = new HashSet<Integer>();
        //adding element in set1 from nums1
        for (int i : nums1) set1.add(i);
        //adding element in set2 from nums2 if present in set1
        for (int j : nums2) {
            if (set1.contains(j))
                set2.add(j);
        }
        //resulting array of size set2
        int[] arr = new int[set2.size()];
        int index = 0;
        //adding elements in array from hashset
        for (int num : set2) {
            arr[index] = num;
            index++;
        }
        //returning resultig array
        return arr;
    }
}
