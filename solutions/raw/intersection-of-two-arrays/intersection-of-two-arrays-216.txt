class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> hs = new HashSet<>();
        for (int n : nums1) {
            hs.add(n);
        }
        HashSet<Integer> hr = new HashSet<>();
        for (int n : nums2) {
            if (hs.contains(n)) {
                hr.add(n);
            }
        }
        int[] res = new int[hr.size()];
        int i = 0;
        for (int n : hr) {
            res[i++] = n;
        }
        return res;
    }
}
