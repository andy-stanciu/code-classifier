class Solution {

    // TC: O(n)
    // SC: O(n)
    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> commonSet = new HashSet<>();
        Set<Integer> s1 = new HashSet<>();
        for (int n : nums1) {
            s1.add(n);
        }
        for (int n : nums2) {
            if (!s1.contains(n)) {
                continue;
            }
            commonSet.add(n);
        }
        int[] commonArr = new int[commonSet.size()];
        int i = 0;
        for (int n : commonSet) {
            commonArr[i] = n;
            i++;
        }
        return commonArr;
    }
}
