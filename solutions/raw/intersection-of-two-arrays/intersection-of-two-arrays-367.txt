class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = 0, j = 0, k = 0;
        int[] tempResult = new int[Math.min(nums1.length, nums2.length)];
        // Use two pointers to find the intersection
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] == nums2[j]) {
                if (k == 0 || tempResult[k - 1] != nums1[i]) {
                    tempResult[k++] = nums1[i];
                }
                i++;
                j++;
            } else if (nums1[i] < nums2[j]) {
                i++;
            } else {
                j++;
            }
        }
        // Copy the result to the final array
        int[] result = Arrays.copyOf(tempResult, k);
        return result;
    }
}
