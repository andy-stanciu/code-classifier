class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> intersection = new HashSet<>();
        for (int i : nums1) {
            set.add(i);
        }
        for (int i : nums2) {
            if (set.contains(i)) {
                intersection.add(i);
            }
        }
        int[] result = new int[intersection.size()];
        int count = 0;
        for (Integer value : intersection) {
            result[count] = value;
            count++;
        }
        return result;
    }
}
