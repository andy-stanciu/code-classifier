class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        int n = nums1.length;
        int m = nums2.length;
        //    int a[]=new int [n];
        Set<Integer> s = new HashSet<>();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (nums1[i] == nums2[j]) {
                    s.add(nums1[i]);
                }
            }
        }
        int[] j = new int[s.size()];
        int cout = 0;
        for (int res : s) {
            j[cout++] = res;
        }
        return j;
    }
}
