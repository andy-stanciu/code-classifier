class Solution {

    public int[] intersection(int[] arr1, int[] arr2) {
        Set<Integer> intersection = new HashSet<>();
        List list = Arrays.stream(arr1).boxed().collect(Collectors.toList());
        for (int i = 0; i < arr2.length; i++) {
            if (list.contains(arr2[i])) {
                intersection.add(arr2[i]);
            }
        }
        Integer[] integerArray = intersection.toArray(new Integer[intersection.size()]);
        int[] intArray = Arrays.stream(integerArray).mapToInt(Integer::intValue).toArray();
        return intArray;
    }
}
