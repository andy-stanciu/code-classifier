class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        int[] frequency = new int[1000];
        int count = 0;
        int[] temp = new int[Math.min(nums1.length, nums2.length)];
        //Find Frequency
        for (int i : nums1) {
            frequency[i]++;
        }
        //Find intersection.
        for (int i : nums2) {
            if (frequency[i] > 0) {
                temp[count] = i;
                // Setting 0 so that next time if same number comes it wont be counted as we need to ignore duplicates
                frequency[i] = 0;
                count++;
            }
        }
        int[] result = new int[count];
        for (int i = 0; i < count; i++) {
            result[i] = temp[i];
        }
        return result;
    }
}
