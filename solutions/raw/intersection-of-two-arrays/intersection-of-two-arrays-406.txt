class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> sets1 = new HashSet<>();
        ArrayList<Integer> intersectionList = new ArrayList<>();
        for (int i = 0; i < nums1.length; i++) {
            sets1.add(nums1[i]);
        }
        for (int j = 0; j < nums2.length; j++) {
            if (sets1.contains(nums2[j])) {
                sets1.remove(nums2[j]);
                intersectionList.add(nums2[j]);
            }
        }
        // Convert ArrayList to array
        int[] intersectionArray = new int[intersectionList.size()];
        for (int k = 0; k < intersectionList.size(); k++) {
            intersectionArray[k] = intersectionList.get(k);
        }
        return intersectionArray;
    }
}
