class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        boolean[] freq1 = new boolean[1001];
        for (int el : nums1) {
            freq1[el] = true;
        }
        boolean[] freq2 = new boolean[1001];
        for (int el : nums2) {
            freq2[el] = true;
        }
        List<Integer> intersection = new ArrayList<>();
        for (int i = 0; i < freq1.length; ++i) {
            if (freq1[i] && freq2[i]) {
                intersection.add(i);
            }
        }
        int ind = 0;
        int[] res = new int[intersection.size()];
        for (int el : intersection) {
            res[ind++] = el;
        }
        return res;
    }
}
