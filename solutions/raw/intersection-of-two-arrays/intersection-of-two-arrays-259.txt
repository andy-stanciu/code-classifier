class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        int i = 0;
        Set<Integer> set = new HashSet<>();
        Set<Integer> union = new HashSet<>();
        for (i = 0; i < nums1.length; i++) {
            set.add(nums1[i]);
        }
        i = 0;
        while (i < nums2.length) {
            if (set.contains(nums2[i])) {
                union.add(nums2[i]);
            }
            i++;
        }
        int[] unionArr = new int[union.size()];
        i = 0;
        for (int num : union) {
            unionArr[i++] = num;
        }
        return unionArr;
    }
}
