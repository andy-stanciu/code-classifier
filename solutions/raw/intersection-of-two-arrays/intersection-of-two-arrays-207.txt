class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        boolean[] count = new boolean[1001];
        int[] li = new int[nums1.length > nums2.length ? nums1.length : nums2.length];
        int j = 0;
        for (int num : nums1) count[num] = true;
        for (int num : nums2) {
            if (count[num]) {
                count[num] = false;
                li[j++] = num;
            }
        }
        return Arrays.copyOfRange(li, 0, j);
    }
}
