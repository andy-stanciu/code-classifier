class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        boolean[] histogram = new boolean[1001];
        // initialize histogram with nums1 values
        for (int n : nums1) {
            histogram[n] = true;
        }
        //the intersection array will always be smaller or equal to
        //min(nums1.length, nums2.length)
        int[] intersection = new int[nums1.length];
        int j = 0;
        for (int n : nums2) {
            //if a value exists, we add it to the intersection and consume it to avoid duplicates
            if (histogram[n])
                intersection[j++] = n;
            histogram[n] = false;
        }
        //truncate trailing zeroes of array
        return Arrays.copyOf(intersection, j);
    }
}
