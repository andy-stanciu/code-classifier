class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        // declare set to holde values from nums1
        Set<Integer> set = new HashSet<>();
        // declare set to hold all intersections of arrays
        Set<Integer> intersect = new HashSet<>();
        // iterate over nums1 array
        for (int i = 0; i < nums1.length; i++) {
            // add each value to set
            set.add(nums1[i]);
        }
        // iterate over nums2 array
        for (int i = 0; i < nums2.length; i++) {
            // check if value intersects with nums1 array, if yes add value to intersect set
            if (set.contains(nums2[i]))
                intersect.add(nums2[i]);
        }
        // convert set to int[] array with stream and return it
        return intersect.stream().mapToInt(i -> i).toArray();
    }
}
