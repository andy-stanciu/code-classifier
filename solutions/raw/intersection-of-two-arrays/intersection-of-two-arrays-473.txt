class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> set = new HashSet<>();
        HashSet<Integer> intersetion = new HashSet<>();
        for (Integer i : nums1) {
            set.add(i);
        }
        for (Integer i : nums2) {
            if (set.contains(i)) {
                intersetion.add(i);
            }
        }
        int[] answer = new int[intersetion.size()];
        int j = 0;
        for (Integer i : intersetion) {
            answer[j++] = i;
        }
        return answer;
    }
}
