class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        int[] f = new int[1001];
        for (int i : nums1) {
            f[i]++;
        }
        ArrayList<Integer> al = new ArrayList<>();
        for (int j : nums2) {
            if (f[j] > 0) {
                al.add(j);
                f[j] = 0;
            }
        }
        int[] ans = (new int[al.size()]);
        int i = 0;
        for (int a : al) {
            ans[i] = a;
            i++;
        }
        return ans;
    }
}
