class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> arr1 = new HashSet<>();
        HashSet<Integer> arr2 = new HashSet<>();
        for (int i : nums1) {
            arr1.add(i);
        }
        for (int j : nums2) {
            if (arr1.contains(j)) {
                arr2.add(j);
            }
        }
        int[] arr = new int[arr2.size()];
        int index = 0;
        for (int k : arr2) {
            arr[index] = k;
            index++;
        }
        return arr;
    }
}
