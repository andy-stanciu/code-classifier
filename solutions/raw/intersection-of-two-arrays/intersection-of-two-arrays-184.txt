class Solution {

    public int[] intersection(int[] arr1, int[] arr2) {
        HashSet<Integer> set1 = new HashSet<>();
        HashSet<Integer> set2 = new HashSet<>();
        for (int num : arr1) {
            set1.add(num);
        }
        for (int num : arr2) {
            if (set1.contains(num)) {
                set2.add(num);
            }
        }
        int[] array = set2.stream().mapToInt(Integer::intValue).toArray();
        return array;
    }
}
