class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        return countArray(nums1, nums2);
    }

    int[] countArray(int[] nums1, int[] nums2) {
        int n = nums1.length;
        ArrayList<Integer> a = new ArrayList<Integer>();
        int[] count = new int[1000];
        for (int i = 0; i < n; i++) {
            count[nums1[i]]++;
        }
        for (int i = 0; i < nums2.length; i++) {
            if (count[nums2[i]] > 0) {
                a.add(nums2[i]);
                count[nums2[i]] = 0;
            }
        }
        int[] arr1 = new int[a.size()];
        for (int i = 0; i < a.size(); i++) {
            arr1[i] = a.get(i);
        }
        return arr1;
    }
}
