class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        boolean[] bol = new boolean[1001];
        for (int n : nums1) {
            bol[n] = true;
        }
        int c = 0;
        for (int n : nums2) {
            if (bol[n] == true) {
                nums1[c++] = n;
                bol[n] = false;
            }
        }
        int[] res = new int[c];
        for (int i = 0; i < c; i++) {
            res[i] = nums1[i];
        }
        return res;
    }
}
