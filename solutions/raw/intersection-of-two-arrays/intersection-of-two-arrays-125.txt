class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> sp = new HashSet<>();
        for (int num : nums1) {
            sp.add(num);
        }
        ArrayList<Integer> inter = new ArrayList<>();
        for (int num : nums2) {
            if (sp.contains(num)) {
                inter.add(num);
                sp.remove(num);
            }
        }
        int[] result = new int[inter.size()];
        for (int i = 0; i < inter.size(); i++) {
            result[i] = inter.get(i);
        }
        return result;
    }
}
