class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        //to store the unique values of array 1
        HashSet<Integer> set = new HashSet<>();
        // to get the intersection values
        HashSet<Integer> ret = new HashSet<>();
        for (int i : nums1) {
            set.add(i);
        }
        for (int j : nums2) {
            if (set.contains(j)) {
                ret.add(j);
            }
        }
        int[] res = new int[ret.size()];
        int l = 0;
        for (int k : ret) {
            //storing in the returning array
            res[l] = k;
            l++;
        }
        return res;
    }
}
