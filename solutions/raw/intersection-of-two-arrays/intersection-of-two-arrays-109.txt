class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet();
        for (int i : nums1) {
            set.add(i);
        }
        List<Integer> temp = new ArrayList();
        for (int i : nums2) {
            if (set.contains(i)) {
                temp.add(i);
                set.remove(i);
            }
        }
        int[] ret = new int[temp.size()];
        for (int i = 0; i < ret.length; i++) {
            ret[i] = temp.get(i);
        }
        return ret;
    }
}
