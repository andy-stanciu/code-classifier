class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> set1 = new HashSet<>();
        HashSet<Integer> set2 = new HashSet<>();
        for (int ele : nums1) set1.add(ele);
        for (int ele : nums2) {
            if (set1.contains(ele))
                set2.add(ele);
        }
        int[] result = new int[set2.size()];
        int i = 0;
        for (int ele : set2) {
            result[i++] = ele;
        }
        return result;
    }
}
