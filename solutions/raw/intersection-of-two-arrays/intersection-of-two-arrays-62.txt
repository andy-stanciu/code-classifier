class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        //creating an arrayList for answers store
        ArrayList<Integer> ans = new ArrayList<>();
        // Creating a Hashset for adding all element of first nums1 array
        HashSet<Integer> set = new HashSet<>();
        for (int i = 0; i < n1; i++) {
            set.add(nums1[i]);
        }
        //Searching elements of nums2 from HashSet, if contains put into ans
        for (int num : nums2) {
            if (set.contains(num))
                ans.add(num);
        }
        //convert the ArrayList into the simple array using stream() api
        return ans.stream().distinct().mapToInt(Integer::intValue).toArray();
    }
}
