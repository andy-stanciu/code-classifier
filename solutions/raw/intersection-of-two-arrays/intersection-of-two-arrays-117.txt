class Solution {

    public int[] intersection(int[] nums1, int[] nums2) {
        List<Integer> li = new ArrayList<>();
        for (int n1 : nums1) {
            for (int n2 : nums2) {
                if (n1 == n2) {
                    if (!li.contains(n1))
                        li.add(n1);
                    break;
                }
            }
        }
        int[] result = new int[li.size()];
        int index = 0;
        for (int i : li) {
            result[index++] = i;
        }
        return result;
    }
}
