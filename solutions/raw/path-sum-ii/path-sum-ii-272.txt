/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static void checkPathSum(TreeNode root, int targetSum, List<List<Integer>> result, List<Integer> temp) {
        if (root == null)
            return;
        temp.add(root.val);
        if (root.left == null && root.right == null) {
            targetSum -= root.val;
            if (targetSum == 0) {
                result.add(new ArrayList<>(temp));
            }
        }
        checkPathSum(root.left, targetSum - root.val, result, temp);
        checkPathSum(root.right, targetSum - root.val, result, temp);
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        if (root == null)
            return new ArrayList<>();
        List<List<Integer>> result = new ArrayList<>();
        checkPathSum(root, targetSum, result, new ArrayList<>());
        return result;
    }
}
