/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> list1 = new ArrayList<>();

    public void path(TreeNode root, int targetSum, int sum, List<Integer> list2) {
        if (root == null) {
            return;
        }
        sum += root.val;
        list2.add(root.val);
        if (root.left == null && root.right == null) {
            if (sum == targetSum) {
                list1.add(new ArrayList<>(list2));
            }
        }
        path(root.left, targetSum, sum, list2);
        path(root.right, targetSum, sum, list2);
        list2.remove(list2.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<Integer> list2 = new ArrayList<>();
        int sum = 0;
        path(root, targetSum, sum, list2);
        return list1;
    }
}
