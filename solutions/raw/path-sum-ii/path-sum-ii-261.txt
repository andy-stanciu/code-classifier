/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> currentPath = new ArrayList<>();
        pathSumHelper(root, targetSum, result, currentPath, 0);
        return result;
    }

    private void pathSumHelper(TreeNode root, int targetSum, List<List<Integer>> result, List<Integer> currentPath, int currentSum) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && currentSum + root.val == targetSum) {
            result.add(currentPath);
        }
        currentPath.add(root.val);
        pathSumHelper(root.left, targetSum, result, new ArrayList<>(currentPath), currentSum + root.val);
        pathSumHelper(root.right, targetSum, result, new ArrayList<>(currentPath), currentSum + root.val);
    }
}
