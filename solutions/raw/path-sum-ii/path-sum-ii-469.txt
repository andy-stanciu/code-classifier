class Solution {

    List<List<Integer>> result = new ArrayList();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        result.clear();
        traverse(root, new ArrayList(), 0, targetSum);
        return result;
    }

    private void traverse(TreeNode root, List<Integer> list, int sum, int targetSum) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && (sum + root.val) == targetSum) {
            result.add(new ArrayList(list));
            list.remove(list.size() - 1);
            return;
        }
        traverse(root.left, list, sum + root.val, targetSum);
        traverse(root.right, list, sum + root.val, targetSum);
        // takes O(1) time
        list.remove(list.size() - 1);
    }
}
