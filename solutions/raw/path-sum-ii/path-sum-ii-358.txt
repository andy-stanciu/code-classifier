/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static void checkSum(TreeNode root, int path, List<Integer> arr, List<List<Integer>> ans, int t) {
        if (root == null) {
            return;
        }
        path = path + root.val;
        arr.add(root.val);
        if (root.left == null && root.right == null) {
            if (path == t) {
                ans.add(new ArrayList<>(arr));
            }
        }
        checkSum(root.left, path, arr, ans, t);
        checkSum(root.right, path, arr, ans, t);
        arr.remove(arr.size() - 1);
        path = path - root.val;
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> ans = new ArrayList<>();
        checkSum(root, 0, new ArrayList<>(), ans, targetSum);
        return ans;
    }
}
