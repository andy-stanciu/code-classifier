/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> local = new ArrayList<>();
        helper(root, targetSum, result, local);
        return result;
    }

    public static void helper(TreeNode root, int targetSum, List<List<Integer>> result, List<Integer> local) {
        if (root == null) {
            return;
        }
        local.add(root.val);
        if ((root.left == null && root.right == null) && targetSum == root.val) {
            result.add(new ArrayList<>(local));
        }
        helper(root.left, targetSum - root.val, result, local);
        helper(root.right, targetSum - root.val, result, local);
        local.remove(local.size() - 1);
    }
}
