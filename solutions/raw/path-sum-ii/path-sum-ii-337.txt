class Solution {

    public void solver(TreeNode root, int targetSum, List<List<Integer>> ans, List<Integer> dummy) {
        if (root == null)
            return;
        dummy.add(root.val);
        if (root.left == null && root.right == null && targetSum - root.val == 0) {
            ans.add(new ArrayList<>(dummy));
        }
        if (root.left != null) {
            solver(root.left, targetSum - root.val, ans, dummy);
            dummy.remove(dummy.size() - 1);
        }
        if (root.right != null) {
            solver(root.right, targetSum - root.val, ans, dummy);
            dummy.remove(dummy.size() - 1);
        }
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> dummy = new ArrayList<>();
        solver(root, targetSum, ans, dummy);
        return ans;
    }
}
