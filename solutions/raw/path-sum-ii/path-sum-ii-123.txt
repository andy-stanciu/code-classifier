/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        List<Integer> list = new ArrayList<>();
        helper(result, list, root, targetSum, 0);
        return result;
    }

    public void helper(List<List<Integer>> result, List<Integer> list, TreeNode root, int target, int sum) {
        if (root == null) {
            return;
        }
        sum += root.val;
        list.add(root.val);
        if (root.left == null && root.right == null) {
            if (sum == target) {
                // Add a copy of the current path
                result.add(new ArrayList<>(list));
            }
        } else {
            helper(result, list, root.left, target, sum);
            helper(result, list, root.right, target, sum);
        }
        // Remove the last element to backtrack
        list.remove(list.size() - 1);
    }
}
