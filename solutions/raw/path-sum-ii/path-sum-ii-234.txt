/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> mainlist = new ArrayList<>();

    public void helper(int sum, int targetsum, TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        sum += root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && sum == targetsum) {
            // Create a new ArrayList to store a copy of the current path
            mainlist.add(new ArrayList<>(list));
        }
        helper(sum, targetsum, root.left, list);
        helper(sum, targetsum, root.right, list);
        list.remove(list.size() - 1);
        return;
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return mainlist;
        }
        ArrayList<Integer> list = new ArrayList<>();
        helper(0, targetSum, root, list);
        return mainlist;
    }
}
