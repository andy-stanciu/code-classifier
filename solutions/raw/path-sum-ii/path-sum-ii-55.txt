class Solution {

    public void getPath(TreeNode node, int targetSum, List<Integer> list, List<List<Integer>> paths) {
        if (node == null) {
            return;
        }
        list.add(node.val);
        targetSum -= node.val;
        if (targetSum == 0 && node.left == null && node.right == null) {
            paths.add(list);
            return;
        }
        getPath(node.left, targetSum, new ArrayList<>(list), paths);
        getPath(node.right, targetSum, new ArrayList<>(list), paths);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> paths = new ArrayList<>();
        getPath(root, targetSum, new ArrayList<>(), paths);
        return paths;
    }
}
