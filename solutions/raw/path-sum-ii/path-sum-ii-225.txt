/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void solve(TreeNode root, int targetSum, int sum, List<Integer> tempAns, List<List<Integer>> ans) {
        if (root == null) {
            return;
        }
        tempAns.add(root.val);
        solve(root.left, targetSum, sum + root.val, tempAns, ans);
        solve(root.right, targetSum, sum + root.val, tempAns, ans);
        if (root.left == null && root.right == null && targetSum == sum + root.val) {
            ans.add(new ArrayList<>(tempAns));
        }
        tempAns.remove(tempAns.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<Integer> tempAns = new ArrayList<>();
        List<List<Integer>> ans = new ArrayList<>();
        solve(root, targetSum, 0, tempAns, ans);
        return ans;
    }
}
