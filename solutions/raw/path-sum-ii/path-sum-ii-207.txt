/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public void find(ArrayList<Integer> list, int cs, TreeNode root, int sum) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && cs + root.val == sum) {
            ans.add(new ArrayList<>(list));
        } else {
            find(list, cs + root.val, root.left, sum);
            find(list, cs + root.val, root.right, sum);
        }
        list.remove(list.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        ArrayList<Integer> list = new ArrayList<>();
        find(list, 0, root, sum);
        return ans;
    }
}
