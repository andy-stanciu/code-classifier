class Solution {

    public List<List<Integer>> res;

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        this.res = new LinkedList();
        LinkedList<Integer> input = new LinkedList();
        backtrack(root, targetSum, input);
        return res;
    }

    public void backtrack(TreeNode root, int targetSum, LinkedList<Integer> input) {
        if (root == null) {
            return;
        }
        input.add(root.val);
        if (root.val == targetSum && root.left == null && root.right == null) {
            res.add(new LinkedList(input));
        }
        if (root.left != null) {
            backtrack(root.left, targetSum - root.val, input);
            input.removeLast();
        }
        if (root.right != null) {
            backtrack(root.right, targetSum - root.val, input);
            input.removeLast();
        }
    }
}
