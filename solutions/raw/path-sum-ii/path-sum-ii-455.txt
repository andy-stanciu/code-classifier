/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<List<Integer>>();

    public void helper(TreeNode root, int sum, ArrayList<Integer> tempList, int targetSum) {
        //Base case
        if (root.left == null && root.right == null) {
            tempList.add(root.val);
            if ((sum + root.val) == targetSum) {
                result.add(new ArrayList<Integer>(tempList));
            }
            tempList.remove(tempList.size() - 1);
            return;
        }
        tempList.add(root.val);
        if (root.left != null) {
            helper(root.left, sum + root.val, tempList, targetSum);
        }
        if (root.right != null) {
            helper(root.right, sum + root.val, tempList, targetSum);
        }
        if (tempList.size() > 0) {
            tempList.remove(tempList.size() - 1);
        }
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        ArrayList<Integer> temp = new ArrayList<Integer>();
        if (root == null) {
            return result;
        }
        helper(root, 0, temp, targetSum);
        return result;
    }
}
