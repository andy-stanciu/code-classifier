/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        helper(root, new ArrayList<Integer>(), targetSum);
        return list;
    }

    private void helper(TreeNode root, List<Integer> lis, int targetSum) {
        if (root == null)
            return;
        lis.add(root.val);
        targetSum -= root.val;
        if (root.left == null && root.right == null && targetSum == 0) {
            list.add(new ArrayList(lis));
        }
        helper(root.left, new ArrayList(lis), targetSum);
        helper(root.right, new ArrayList(lis), targetSum);
    }
}
