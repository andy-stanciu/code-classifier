/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int sum = 0;
        return helper(root, targetSum, sum);
    }

    public boolean helper(TreeNode root, int targetSum, int sum) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null && targetSum == sum + root.val)
            return true;
        sum += root.val;
        boolean left = helper(root.left, targetSum, sum);
        boolean right = helper(root.right, targetSum, sum);
        return (left || right);
    }
}
