/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        dfs(root, result, targetSum, new ArrayList<>());
        return result;
    }

    public void dfs(TreeNode root, List<List<Integer>> result, int targetSum, List<Integer> temp) {
        if (root == null)
            return;
        targetSum = targetSum - root.val;
        temp.add(root.val);
        if (targetSum == 0 && root.left == null && root.right == null) {
            result.add(new ArrayList<>(temp));
            return;
        }
        if (root.left != null) {
            dfs(root.left, result, targetSum, temp);
            temp.remove(temp.size() - 1);
        }
        if (root.right != null) {
            dfs(root.right, result, targetSum, temp);
            temp.remove(temp.size() - 1);
        }
    }
}
