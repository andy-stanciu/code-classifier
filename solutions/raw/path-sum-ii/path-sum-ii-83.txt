/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> currPath = new ArrayList<>();
        dfs(root, currPath, targetSum, res);
        return res;
    }

    public void dfs(TreeNode root, List<Integer> currPath, int targetSum, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        currPath.add(root.val);
        if (root.left == null && root.right == null && root.val == targetSum) {
            res.add(new ArrayList<>(currPath));
        } else {
            dfs(root.left, currPath, targetSum - root.val, res);
            dfs(root.right, currPath, targetSum - root.val, res);
        }
        // backtrack karre hain
        currPath.remove(currPath.size() - 1);
    }
}
