class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return ans;
        }
        path(root, targetSum, new ArrayList<>());
        return ans;
    }

    public void path(TreeNode node, int targetRemaining, List<Integer> list) {
        if (node == null) {
            return;
        }
        // add the node\'s val in the list
        list.add(node.val);
        // remaining target should be fulfilled by the `leaf` node\'s value hence checking (target == node.val)
        if (node.left == null && node.right == null && targetRemaining == node.val) {
            ans.add(new ArrayList<>(list));
        } else {
            // check left and right
            path(node.left, targetRemaining - node.val, list);
            path(node.right, targetRemaining - node.val, list);
        }
        // backtracking
        list.remove(list.size() - 1);
    }
}
