/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        helper(root, targetSum, result, new ArrayList());
        return result;
    }

    private void helper(TreeNode root, int targetSum, List<List<Integer>> result, List<Integer> current) {
        if (root == null) {
            return;
        }
        current.add(root.val);
        targetSum -= root.val;
        if (root.right == null && root.left == null && targetSum == 0) {
            result.add(new ArrayList(current));
        } else {
            helper(root.left, targetSum, result, current);
            helper(root.right, targetSum, result, current);
        }
        current.remove(current.size() - 1);
    }
}
