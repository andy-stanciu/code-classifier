class Solution {

    private List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        path(root, targetSum, new ArrayList<>());
        return ans;
    }

    private void path(TreeNode root, int targetSum, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        targetSum -= root.val;
        if (root.left == null && root.right == null) {
            if (targetSum == 0)
                ans.add(list);
            return;
        }
        path(root.left, targetSum, new ArrayList<>(list));
        path(root.right, targetSum, new ArrayList<>(list));
    }
}
