class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList();
        dfs(root, res, new ArrayList<Integer>(), targetSum);
        return res;
    }

    private void dfs(TreeNode root, List<List<Integer>> res, List<Integer> temp, int targetSum) {
        if (root == null)
            return;
        temp.add(root.val);
        if (root.left == null && root.right == null && targetSum == root.val)
            res.add(new ArrayList<Integer>(temp));
        dfs(root.left, res, temp, targetSum - root.val);
        dfs(root.right, res, temp, targetSum - root.val);
        temp.remove(temp.size() - 1);
    }
}
