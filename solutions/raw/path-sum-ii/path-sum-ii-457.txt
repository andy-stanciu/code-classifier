/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int i = 0;

    private void solve(TreeNode root, int sum, int targetSum, List<Integer> currList, List<List<Integer>> list) {
        if (root == null)
            return;
        currList.add(root.val);
        sum += root.val;
        //at at last level(leaf node level) we check whether the sum of node values in the path equals to targetsum or not
        if (root.left == null && root.right == null && sum == targetSum) {
            list.add(new ArrayList<Integer>(currList));
            // return;
        } else {
            solve(root.left, sum, targetSum, currList, list);
            solve(root.right, sum, targetSum, currList, list);
        }
        currList.remove(currList.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> currList = new ArrayList<>();
        int sum = 0;
        solve(root, sum, targetSum, currList, list);
        return list;
    }
}
