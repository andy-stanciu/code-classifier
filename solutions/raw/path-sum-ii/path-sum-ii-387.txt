class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> l = new ArrayList<>();
        ;
        totals(root, targetSum, 0, new ArrayList<>(), l);
        return l;
    }

    public void totals(TreeNode root, int t, int total, List<Integer> l, List<List<Integer>> ans) {
        if (root == null)
            return;
        total += root.val;
        l.add(root.val);
        if (root.left == null && root.right == null) {
            if (t == total)
                ans.add(new ArrayList<>(l));
            l.remove(l.size() - 1);
            return;
        }
        totals(root.left, t, total, l, ans);
        totals(root.right, t, total, l, ans);
        total -= root.val;
        l.remove(l.size() - 1);
    }
}
