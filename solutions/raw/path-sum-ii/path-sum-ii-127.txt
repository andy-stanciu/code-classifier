/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<Integer> temp = new ArrayList<>();
        fill(root, targetSum, 0, temp);
        return result;
    }

    public void fill(TreeNode root, int target, int sum, List<Integer> temp) {
        if (root == null)
            return;
        sum = sum + root.val;
        temp.add(root.val);
        //if leaf node h to tabhi check karenge sum target k equal h ki nahi
        if (root.left == null && root.right == null) {
            if (sum == target) {
                result.add(new ArrayList<>(temp));
            }
        } else {
            fill(root.left, target, sum, temp);
            fill(root.right, target, sum, temp);
        }
        //backtracking
        temp.remove(temp.size() - 1);
    }
}
