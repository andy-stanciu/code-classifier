/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        helper(root, 0, targetSum, result, new ArrayList<>());
        return result;
    }

    void helper(TreeNode root, int ans, int targetSum, List<List<Integer>> result, List<Integer> list) {
        if (root == null)
            return;
        int sum = ans + root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && sum == targetSum) {
            result.add(new ArrayList<>(list));
        }
        helper(root.left, sum, targetSum, result, list);
        helper(root.right, sum, targetSum, result, list);
        list.remove(list.size() - 1);
    }
}
