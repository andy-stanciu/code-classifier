/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        pathSum(root, targetSum, list, res, 0);
        return res;
    }

    public void pathSum(TreeNode root, int target, List<Integer> list, List<List<Integer>> res, int sum) {
        if (root == null) {
            return;
        }
        sum += root.val;
        list.add(root.val);
        if (sum == target && root.left == null && root.right == null) {
            res.add(new ArrayList<>(list));
        }
        pathSum(root.left, target, list, res, sum);
        pathSum(root.right, target, list, res, sum);
        list.remove(list.size() - 1);
    }
}
