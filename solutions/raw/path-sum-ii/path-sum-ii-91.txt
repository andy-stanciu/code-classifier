/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> arr = new ArrayList<>();
        rootToLeaf(root, arr, res, targetSum);
        return res;
    }

    public void rootToLeaf(TreeNode root, List<Integer> lst, List<List<Integer>> res, int target) {
        int sum = 0;
        if (root == null) {
            return;
        }
        lst.add(root.val);
        if (root.left == null && root.right == null) {
            for (int i = 0; i < lst.size(); i++) {
                sum += lst.get(i);
            }
            if (sum == target) {
                res.add(new ArrayList<>(lst));
            }
            sum = 0;
        } else {
            if (root.left != null) {
                rootToLeaf(root.left, lst, res, target);
            }
            if (root.right != null) {
                rootToLeaf(root.right, lst, res, target);
            }
        }
        lst.remove(lst.size() - 1);
    }
}
