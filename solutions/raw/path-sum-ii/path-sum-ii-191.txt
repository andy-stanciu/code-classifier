/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> al = new ArrayList<>();

    public void solve(TreeNode root, String s, int sum, int target) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            sum = sum + root.val;
            s = s + "," + root.val;
            if (sum == target) {
                String[] str = s.split(",");
                List<String> a = new ArrayList<>(Arrays.asList(str));
                List<Integer> ans = new ArrayList<>();
                for (int i = 1; i < a.size(); i++) {
                    ans.add(Integer.valueOf(a.get(i)));
                }
                al.add(ans);
            }
            return;
        }
        solve(root.left, s + "," + root.val, sum + root.val, target);
        solve(root.right, s + "," + root.val, sum + root.val, target);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        solve(root, "", 0, targetSum);
        return al;
    }
}
