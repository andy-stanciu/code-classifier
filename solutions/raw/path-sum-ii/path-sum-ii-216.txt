/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<Integer> ls = new ArrayList<>();
        List<List<Integer>> res = new ArrayList<>();
        pathSumSol(root, targetSum, res, ls);
        return res;
    }

    public void pathSumSol(TreeNode root, int targetSum, List<List<Integer>> res, List<Integer> ls) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (targetSum - root.val == 0) {
                ls.add(root.val);
                res.add(new ArrayList<>(ls));
                ls.remove(ls.size() - 1);
            }
            return;
        }
        ls.add(root.val);
        pathSumSol(root.left, targetSum - root.val, res, ls);
        ls.remove(ls.size() - 1);
        ls.add(root.val);
        pathSumSol(root.right, targetSum - root.val, res, ls);
        ls.remove(ls.size() - 1);
    }
}
