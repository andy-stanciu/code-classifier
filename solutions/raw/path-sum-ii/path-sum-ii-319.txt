/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> a = new ArrayList<List<Integer>>();
        List<Integer> b = new ArrayList<>();
        PathSum(root, targetSum, 0, a, b);
        return a;
    }

    public void PathSum(TreeNode root, int targetSum, int currSum, List<List<Integer>> a, List<Integer> b) {
        if (root == null) {
            return;
        }
        currSum += root.val;
        b.add(root.val);
        if (root.left == null && root.right == null) {
            if (currSum == targetSum) {
                a.add(new ArrayList<>(b));
            }
        }
        PathSum(root.left, targetSum, currSum, a, b);
        PathSum(root.right, targetSum, currSum, a, b);
        b.remove(b.size() - 1);
        return;
    }
}
