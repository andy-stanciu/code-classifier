class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public void pathSum(TreeNode root, int targetSum, List<Integer> list1) {
        list1.add(root.val);
        if (root.left == null && root.right == null) {
            if (targetSum == root.val)
                list.add(new LinkedList<>(list1));
            list1.remove(list1.size() - 1);
            return;
        }
        if (root.left != null) {
            pathSum(root.left, targetSum - root.val, list1);
        }
        if (root.right != null) {
            pathSum(root.right, targetSum - root.val, list1);
        }
        list1.remove(list1.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        if (root == null)
            return list;
        pathSum(root, targetSum, new ArrayList<>());
        return list;
    }
}
