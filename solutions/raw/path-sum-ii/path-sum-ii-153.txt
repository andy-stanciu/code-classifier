/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> path = new ArrayList<>();
        if (root != null)
            traverse(root, path, list, targetSum);
        return list;
    }

    public void traverse(TreeNode root, List<Integer> path, List<List<Integer>> list, int targetsum) {
        path.add(root.val);
        if (root.left == null && root.right == null) {
            int sum = 0;
            for (int i = 0; i < path.size(); i++) {
                sum += path.get(i);
            }
            if (sum == targetsum) {
                list.add(new ArrayList<>(path));
                // path.clear();
            }
        }
        if (root.left != null)
            traverse(root.left, path, list, targetsum);
        if (root.right != null)
            traverse(root.right, path, list, targetsum);
        path.remove(path.size() - 1);
    }
}
