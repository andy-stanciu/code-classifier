// Runtime: 1 ms, faster than 99.88% of Java online submissions for Path Sum II.
// Memory Usage: 39.7 MB, less than 79.46% of Java online submissions for Path Sum II.
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        List<List<Integer>> result = new ArrayList<>();
        return dfs(root, sum, result, new ArrayList<Integer>());
    }

    private List<List<Integer>> dfs(TreeNode node, int sum, List<List<Integer>> result, List<Integer> slate) {
        if (node == null)
            return result;
        if (node.left == null && node.right == null) {
            if (sum == node.val) {
                slate.add(node.val);
                result.add(new ArrayList<Integer>(slate));
                slate.remove(slate.size() - 1);
            }
            return result;
        }
        slate.add(node.val);
        dfs(node.left, sum - node.val, result, slate);
        dfs(node.right, sum - node.val, result, slate);
        slate.remove(slate.size() - 1);
        return result;
    }
}
