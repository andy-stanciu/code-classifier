/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        if (root == null)
            return result;
        findSum(root, 0, targetSum, new ArrayList<>());
        return result;
    }

    private void findSum(TreeNode root, int sum, int targetSum, List<Integer> path) {
        if (root == null)
            return;
        path.add(root.val);
        sum += root.val;
        if (root.left == null && root.right == null) {
            if (sum == targetSum) {
                result.add(new ArrayList<>(path));
            }
        } else {
            findSum(root.left, sum, targetSum, path);
            findSum(root.right, sum, targetSum, path);
        }
        path.remove(path.size() - 1);
    }
}
