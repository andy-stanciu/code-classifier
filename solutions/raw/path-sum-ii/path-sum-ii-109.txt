/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<Integer> lis = new ArrayList<>();
        helper(root, targetSum, lis, 0);
        return res;
    }

    public void helper(TreeNode root, int target, List<Integer> arr, int sum) {
        if (root == null) {
            return;
        }
        arr.add(root.val);
        sum += root.val;
        if (root.left == null && root.right == null && sum == target) {
            res.add(arr);
        }
        helper(root.left, target, new ArrayList(arr), sum);
        helper(root.right, target, new ArrayList(arr), sum);
    }
}
