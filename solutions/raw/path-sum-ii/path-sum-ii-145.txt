/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> list = new ArrayList<>();
        ArrayList<Integer> arr = new ArrayList<>();
        function(root, targetSum, list, arr, 0);
        return list;
    }

    public void function(TreeNode root, int target, List<List<Integer>> list, ArrayList<Integer> arr, int sum) {
        if (root == null) {
            return;
        }
        sum += root.val;
        arr.add(root.val);
        if (root.left == null && root.right == null) {
            if (sum == target) {
                list.add(new ArrayList<>(arr));
            }
        }
        function(root.left, target, list, arr, sum);
        function(root.right, target, list, arr, sum);
        arr.remove(arr.size() - 1);
    }
}
