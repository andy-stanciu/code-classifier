/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ll = new ArrayList<>();
        check(root, targetSum, ans, ll);
        return ans;
    }

    public void check(TreeNode root, int sum, List<List<Integer>> ans, List<Integer> ll) {
        if (root == null) {
            return;
        }
        ll.add(root.val);
        if (root.left == null && root.right == null) {
            int path = 0;
            for (int val : ll) {
                path += val;
            }
            if (path == sum) {
                ans.add(new ArrayList<>(ll));
            }
        }
        check(root.left, sum, ans, ll);
        check(root.right, sum, ans, ll);
        ll.remove(ll.size() - 1);
    }
}
