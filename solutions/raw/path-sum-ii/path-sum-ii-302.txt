class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        dfs(root, 0, targetSum, new ArrayList<Integer>());
        return res;
    }

    public void dfs(TreeNode node, int sum, int tar, List<Integer> l) {
        if (node != null && node.left == null && node.right == null && sum + node.val == tar) {
            List<Integer> e = new ArrayList<>();
            e.addAll(l);
            e.add(node.val);
            if (!e.isEmpty()) {
                res.add(e);
            }
        } else if (node != null) {
            l.add(node.val);
            dfs(node.left, sum + node.val, tar, l);
            dfs(node.right, sum + node.val, tar, l);
            l.remove(l.size() - 1);
        }
    }
}
