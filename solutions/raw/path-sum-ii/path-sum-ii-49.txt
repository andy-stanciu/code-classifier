/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void slove(TreeNode root, int targetSum, List<List<Integer>> ans, List<Integer> temp, int sum) {
        if (root == null) {
            return;
        }
        if (root.right == null && root.left == null) {
            // include current node
            temp.add(root.val);
            sum += root.val;
            // check for target sum
            if (sum == targetSum) {
                //
                ans.add(new ArrayList<>(temp));
            }
            // exclude
            sum -= root.val;
            temp.remove(temp.size() - 1);
        }
        // include current node
        temp.add(root.val);
        sum += root.val;
        slove(root.left, targetSum, ans, temp, sum);
        slove(root.right, targetSum, ans, temp, sum);
        // backtrack
        temp.remove(temp.size() - 1);
        sum -= root.val;
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        int sum = 0;
        slove(root, targetSum, ans, temp, sum);
        return ans;
    }
}
