/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        solve(new ArrayList<>(), root, targetSum, 0, res);
        return res;
    }

    private void solve(List<Integer> temp, TreeNode root, int target, int currSum, List<List<Integer>> res) {
        if (root == null)
            return;
        currSum += root.val;
        temp.add(root.val);
        if (root.left == null && root.right == null && currSum == target) {
            res.add(new ArrayList<>(temp));
            //return;
        }
        solve(temp, root.left, target, currSum, res);
        solve(temp, root.right, target, currSum, res);
        temp.remove(temp.size() - 1);
    }
}
