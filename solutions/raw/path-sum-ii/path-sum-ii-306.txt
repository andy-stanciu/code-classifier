/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> ll = new ArrayList<>();
        List<Integer> lld = new ArrayList<>();
        hasPathSum(root, targetSum, lld, ll);
        return ll;
    }

    public void hasPathSum(TreeNode root, int targetSum, List<Integer> lld, List<List<Integer>> ll) {
        if (root == null)
            return;
        lld.add(root.val);
        if (root.left == null && root.right == null && targetSum == root.val) {
            ll.add(new ArrayList<Integer>(lld));
        } else {
            hasPathSum(root.left, targetSum - root.val, lld, ll);
            hasPathSum(root.right, targetSum - root.val, lld, ll);
        }
        lld.remove(lld.size() - 1);
    }
}
