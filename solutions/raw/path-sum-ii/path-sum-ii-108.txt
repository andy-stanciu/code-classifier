/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static void backtrack(TreeNode root, int targetSum, List<Integer> list, List<List<Integer>> ans) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        if (targetSum == root.val && root.left == null && root.right == null) {
            ans.add(new ArrayList<>(list));
        } else {
            backtrack(root.left, targetSum - root.val, list, ans);
            backtrack(root.right, targetSum - root.val, list, ans);
        }
        list.remove(list.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<Integer> list = new ArrayList<>();
        List<List<Integer>> ans = new ArrayList<>();
        backtrack(root, targetSum, list, ans);
        return ans;
    }
}
