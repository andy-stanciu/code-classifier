/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        ArrayList<Integer> temp = new ArrayList<>();
        int tempSum = 0;
        myFun(root, targetSum, res, tempSum, temp);
        return res;
    }

    public static void myFun(TreeNode root, int targetSum, List<List<Integer>> res, int tempSum, ArrayList<Integer> temp) {
        if (root == null) {
            return;
        }
        temp.add(root.val);
        tempSum += root.val;
        if (root.left == null && root.right == null) {
            if (tempSum == targetSum) {
                res.add(new ArrayList<>(temp));
            }
            tempSum -= root.val;
            temp.remove(temp.size() - 1);
            return;
        }
        myFun(root.left, targetSum, res, tempSum, temp);
        myFun(root.right, targetSum, res, tempSum, temp);
        temp.remove(temp.size() - 1);
        tempSum -= root.val;
    }
}
