/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void backtrack(TreeNode root, int targetSum, List<Integer> cur, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        // add to curr.
        cur.add(root.val);
        if (root.left == null && root.right == null && root.val == targetSum) {
            res.add(new ArrayList<>(cur));
        } else {
            backtrack(root.left, targetSum - root.val, cur, res);
            backtrack(root.right, targetSum - root.val, cur, res);
        }
        // bactrack.
        cur.remove(cur.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        backtrack(root, targetSum, cur, res);
        return res;
    }
}
