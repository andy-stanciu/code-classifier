class Solution {

    public List<List<Integer>> res;

    public void solve(TreeNode root, int sum, List<Integer> vals) {
        if (root == null)
            return;
        vals.add(root.val);
        if (root.left == null && root.right == null && root.val == sum) {
            res.add(vals);
            return;
        }
        solve(root.left, sum - root.val, new ArrayList<Integer>(vals));
        solve(root.right, sum - root.val, vals);
    }

    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        res = new ArrayList<>();
        solve(root, sum, new ArrayList<>());
        return res;
    }
}
