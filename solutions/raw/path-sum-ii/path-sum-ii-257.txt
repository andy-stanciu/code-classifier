/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> list = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        if (root == null)
            return list;
        int sum = 0;
        TreeNode node = root;
        TreeNode prev = null;
        while (node != null || !stack.isEmpty()) {
            while (node != null) {
                sum += node.val;
                stack.push(node);
                if (sum == targetSum && node.left == null && node.right == null) {
                    List<Integer> arySum = new ArrayList<>();
                    for (TreeNode a : stack) {
                        arySum.add(a.val);
                    }
                    list.add(arySum);
                }
                node = node.left;
            }
            node = stack.pop();
            sum -= node.val;
            if (node.right == null || node.right == prev) {
                prev = node;
                node = null;
            } else {
                stack.push(node);
                sum += node.val;
                node = node.right;
            }
        }
        return list;
    }
}
