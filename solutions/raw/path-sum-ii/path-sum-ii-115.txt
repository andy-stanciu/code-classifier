/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> finallist = new ArrayList<>();

    public void pathSumm(TreeNode root, int targetSum, List<Integer> list1) {
        if (root == null) {
            return;
        }
        list1.add(root.val);
        if (root.left == null && root.right == null && targetSum == root.val) {
            finallist.add(new ArrayList<>(list1));
        } else {
            pathSumm(root.left, targetSum - root.val, list1);
            pathSumm(root.right, targetSum - root.val, list1);
        }
        list1.remove(list1.size() - 1);
        return;
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        pathSumm(root, targetSum, new ArrayList<>());
        return finallist;
    }
}
