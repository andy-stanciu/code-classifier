class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null)
            return ans;
        calculate(root, targetSum, ans, new ArrayList<>());
        return ans;
    }

    public void calculate(TreeNode root, int sum, List<List<Integer>> ans, List<Integer> ds) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (sum == root.val) {
                ds.add(root.val);
                ans.add(new ArrayList<>(ds));
                ds.remove(ds.size() - 1);
            }
            return;
        }
        ds.add(root.val);
        calculate(root.left, sum - root.val, ans, ds);
        calculate(root.right, sum - root.val, ans, ds);
        ds.remove(ds.size() - 1);
    }
}
