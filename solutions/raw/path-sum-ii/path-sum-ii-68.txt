/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //helper function
    private void dfs(TreeNode node, int target, List<Integer> currentPath, List<List<Integer>> result) {
        if (node == null)
            return;
        //add that node\'s value to currentPath
        currentPath.add(node.val);
        //if we reach a leaf node, check if the sum is equal to target
        if (node.left == null && node.right == null && target == node.val)
            result.add(new ArrayList<>(currentPath));
        else {
            dfs(node.left, target - node.val, currentPath, result);
            dfs(node.right, target - node.val, currentPath, result);
        }
        //remove that current path or backtrack
        currentPath.remove(currentPath.size() - 1);
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        //stores current paths
        List<Integer> currentPath = new ArrayList<>();
        //stores all the paths that have sum equal to target
        List<List<Integer>> result = new ArrayList<>();
        dfs(root, targetSum, currentPath, result);
        return result;
    }
}
