/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        solve(root, targetSum, 0, new ArrayList<Integer>());
        return ans;
    }

    public void solve(TreeNode root, int target, int sum, List<Integer> values) {
        if (root == null)
            return;
        sum += root.val;
        values.add(root.val);
        if (sum == target && root.left == null && root.right == null) {
            ans.add(new ArrayList<Integer>(values));
        }
        solve(root.left, target, sum, values);
        solve(root.right, target, sum, values);
        values.remove(values.size() - 1);
    }
}
