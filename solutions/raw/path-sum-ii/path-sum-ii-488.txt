class Solution {

    List<List<Integer>> lists = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        pathSumList(root, sum, 0, new ArrayList<>());
        return lists;
    }

    public void pathSumList(TreeNode root, int sum, int initial, List<Integer> list) {
        if (root == null) {
            return;
        }
        initial = initial + root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && initial == sum)
            lists.add(new ArrayList<>(list));
        pathSumList(root.left, sum, initial, list);
        pathSumList(root.right, sum, initial, list);
        list.remove(list.size() - 1);
    }
}
