/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        sum(result, new ArrayList<Integer>(), root, targetSum);
        return result;
    }

    private void sum(List<List<Integer>> result, List<Integer> curr_path, TreeNode node, int targetSum) {
        if (node == null)
            return;
        curr_path.add(node.val);
        if (node.left == null && node.right == null && targetSum - node.val == 0) {
            result.add(new ArrayList<Integer>(curr_path));
        }
        sum(result, curr_path, node.left, targetSum - node.val);
        sum(result, curr_path, node.right, targetSum - node.val);
        curr_path.remove(curr_path.size() - 1);
    }
}
