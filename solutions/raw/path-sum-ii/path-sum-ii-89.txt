/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    List<List<Integer>> fAns;

    public void helper(TreeNode root, int targetSum, List<Integer> ans, int sum) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            sum += root.val;
            if (sum == targetSum) {
                ans.add(root.val);
                List<Integer> tAns = new ArrayList<>(ans);
                fAns.add(tAns);
                ans.remove(ans.size() - 1);
            }
            return;
        }
        sum += root.val;
        ans.add(root.val);
        if (root.left != null) {
            helper(root.left, targetSum, ans, sum);
        }
        if (root.right != null) {
            helper(root.right, targetSum, ans, sum);
        }
        if (ans.size() >= 1)
            ans.remove(ans.size() - 1);
        return;
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        fAns = new ArrayList<>();
        helper(root, targetSum, new ArrayList<Integer>(), 0);
        return fAns;
    }
}
