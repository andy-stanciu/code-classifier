/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        backtrack(root, 0, targetSum, result, new ArrayList<Integer>());
        return result;
    }

    private void backtrack(TreeNode node, int sum, int targetSum, List<List<Integer>> result, List<Integer> current) {
        sum += node.val;
        current.add(node.val);
        if (node.left == null && node.right == null) {
            if (sum == targetSum)
                result.add(new ArrayList<Integer>(current));
            sum -= node.val;
            current.remove(current.size() - 1);
            return;
        }
        if (node.left != null)
            backtrack(node.left, sum, targetSum, result, current);
        if (node.right != null)
            backtrack(node.right, sum, targetSum, result, current);
        sum -= node.val;
        current.remove(current.size() - 1);
    }
}
