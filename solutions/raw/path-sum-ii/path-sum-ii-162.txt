/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        hasPathSum(root, targetSum, new ArrayList<Integer>());
        return result;
    }

    public void hasPathSum(TreeNode root, int targetSum, List<Integer> path) {
        if (root == null)
            return;
        path.add(root.val);
        if (root.val == targetSum && root.left == null && root.right == null) {
            result.add(new ArrayList<>(path));
        } else {
            hasPathSum(root.left, targetSum - root.val, path);
            hasPathSum(root.right, targetSum - root.val, path);
        }
        path.remove(path.size() - 1);
    }
}
