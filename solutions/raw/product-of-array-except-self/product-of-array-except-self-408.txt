class Solution {

    public int[] productExceptSelf(int[] nums) {
        // Initialize result array
        int[] result = new int[nums.length];
        int left = 1;
        int right = 1;
        // First calculate left and store at each place except itself
        for (int i = 0; i < nums.length; i++) {
            result[i] = left;
            left *= nums[i];
        }
        // Then calculate right and store at each place except itself
        for (int i = nums.length - 1; i >= 0; i--) {
            result[i] *= right;
            right *= nums[i];
        }
        return result;
    }
}
