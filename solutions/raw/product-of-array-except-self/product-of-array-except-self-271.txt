class Solution {

    public int[] productExceptSelf(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        int leftProduct = 1;
        for (int i = 0; i < len; i++) {
            // Store product of elements to the left
            result[i] = leftProduct;
            // Calculate product of elements to the left
            leftProduct *= nums[i];
        }
        int rightProduct = 1;
        for (int i = len - 1; i >= 0; i--) {
            // Multiply by product of elements to the right
            result[i] *= rightProduct;
            // Calculate product of elements to the right
            rightProduct *= nums[i];
        }
        return result;
    }
}
