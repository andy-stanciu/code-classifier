class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] output = new int[n];
        output[0] = 1;
        //To left product
        for (int i = 1; i < n; i++) {
            output[i] = nums[i - 1] * output[i - 1];
        }
        int R = 1;
        for (int i = n - 1; i >= 0; i--) {
            output[i] = output[i] * R;
            R = R * nums[i];
        }
        return output;
    }
}
