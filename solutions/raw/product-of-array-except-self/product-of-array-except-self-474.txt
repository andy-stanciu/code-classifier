class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] answer = new int[nums.length];
        answer[0] = nums[0];
        for (int i = 1; i < nums.length; i++) answer[i] = answer[i - 1] * nums[i];
        int product = nums[nums.length - 1];
        answer[nums.length - 1] = answer[nums.length - 2];
        for (int i = nums.length - 2; i > 0; i--) {
            answer[i] = answer[i - 1] * product;
            product *= nums[i];
        }
        answer[0] = product;
        return answer;
    }
}
