class Solution {

    public int[] productExceptSelf(int[] nums) {
        int left = 1;
        int right = 1;
        int zeros = 0;
        int total = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zeros++;
            } else {
                total *= nums[i];
            }
            right *= nums[i];
        }
        int[] ans = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                right = right / nums[i];
                ans[i] = left * right;
                left *= nums[i];
            } else {
                if (zeros > 1) {
                    ans[i] = 0;
                } else {
                    ans[i] = total;
                }
            }
        }
        return ans;
    }
}
