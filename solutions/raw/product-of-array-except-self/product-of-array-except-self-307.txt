class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] arr1 = new int[nums.length];
        int pro = 1;
        boolean b = true;
        int zeroCount = 0;
        for (int i : nums) {
            if (i == 0) {
                zeroCount++;
                if (zeroCount > 1) {
                    return arr1;
                }
                b = false;
            } else {
                pro = pro * i;
            }
        }
        int k = 0;
        for (int i : nums) {
            if (b) {
                arr1[k] = pro / i;
            } else {
                if (i == 0) {
                    arr1[k] = pro;
                } else {
                    arr1[k] = 0;
                }
            }
            k++;
        }
        return arr1;
    }
}
