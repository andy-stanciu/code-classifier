class Solution {

    public int[] productExceptSelf(int[] A) {
        int n = A.length;
        var output = new int[n];
        fill(output, 1);
        for (int i = 0, prefix = 1; i < n; i++) {
            output[i] = prefix;
            prefix *= A[i];
        }
        for (int i = n - 1, suffix = 1; i >= 0; i--) {
            output[i] *= suffix;
            suffix *= A[i];
        }
        return output;
    }
}
