class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] pre = new int[nums.length];
        int[] suf = new int[nums.length];
        /*intializing first element as there is no element before*/
        pre[0] = 1;
        /*intializing last elemnt as there is no element after*/
        suf[nums.length - 1] = 1;
        /*product of all elements towards left*/
        for (int i = 1; i < nums.length; i++) {
            pre[i] = nums[i - 1] * pre[i - 1];
        }
        /*product of all elements towards right*/
        for (int i = nums.length - 2; i >= 0; i--) {
            suf[i] = suf[i + 1] * nums[i + 1];
        }
        /*multipliying both to get the product of array except self*/
        for (int i = 0; i < nums.length; i++) {
            nums[i] = pre[i] * suf[i];
        }
        return nums;
    }
}
