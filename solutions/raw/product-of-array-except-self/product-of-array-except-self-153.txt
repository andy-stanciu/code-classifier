class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] number = new int[nums.length];
        int suffix = 1;
        for (int i = 0; i < nums.length; i++) {
            number[i] = suffix;
            suffix *= nums[i];
        }
        int prefix = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            number[i] *= prefix;
            prefix *= nums[i];
        }
        return number;
    }
}
