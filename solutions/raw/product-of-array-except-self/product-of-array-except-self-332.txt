class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] result = new int[nums.length];
        Arrays.fill(result, 1);
        for (int i = 0, p = 1; i < nums.length; i++) {
            result[i] *= p;
            p *= nums[i];
        }
        for (int i = nums.length - 1, p = 1; i > -1; i--) {
            result[i] *= p;
            p *= nums[i];
        }
        return result;
    }
}
