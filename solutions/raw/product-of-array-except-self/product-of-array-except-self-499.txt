class Solution {

    public int[] productExceptSelf(int[] nums) {
        // 1, 2, 6, 24
        int[] result = new int[nums.length];
        int temp = 1;
        for (int i = 0; i < nums.length; i++) {
            result[i] = temp *= nums[i];
        }
        int temp2 = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (i == nums.length - 1) {
                result[i] = result[i - 1] * 1;
                temp2 = temp2 *= nums[i];
                continue;
            }
            if (i == 0) {
                result[i] = 1 * temp2;
                break;
            }
            result[i] = result[i - 1] * temp2;
            temp2 = temp2 *= nums[i];
        }
        return result;
    }
}
