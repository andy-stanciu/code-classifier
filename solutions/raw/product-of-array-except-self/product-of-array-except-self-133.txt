class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        int product = 1;
        for (int i = 0; i < nums.length; i++) {
            // left product of each nums[i]
            result[i] = product;
            product *= nums[i];
        }
        product = 1;
        for (int i = n - 1; i >= 0; i--) {
            // here product is the right product of nums[i] element which is getting multiplied by result[i] which contains left sum
            result[i] *= product;
            product *= nums[i];
        }
        return result;
    }
}
