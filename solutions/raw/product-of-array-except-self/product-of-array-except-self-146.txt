class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        arr[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            arr[i] = arr[i + 1] * nums[i];
        }
        int prevProduct = 1;
        for (int i = 0; i < n - 1; i++) {
            arr[i] = prevProduct * arr[i + 1];
            prevProduct = prevProduct * nums[i];
        }
        arr[n - 1] = prevProduct;
        return arr;
    }
}
