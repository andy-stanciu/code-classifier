class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] ans = new int[nums.length];
        int left = nums[0];
        int right = nums[nums.length - 1];
        Arrays.fill(ans, 1);
        for (int i = 1; i < nums.length; i++) {
            ans[i] = ans[i] * left;
            left = left * nums[i];
        }
        for (int i = nums.length - 2; i >= 0; i--) {
            ans[i] = ans[i] * right;
            right = right * nums[i];
        }
        return ans;
    }
}
