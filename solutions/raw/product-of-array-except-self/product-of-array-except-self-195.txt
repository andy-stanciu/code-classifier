class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        int[] lanswer = new int[n];
        int lproduct = 1;
        int[] ranswer = new int[n];
        int rproduct = 1;
        for (int i = 0; i < nums.length; i++) {
            lanswer[i] = lproduct;
            lproduct = lproduct * nums[i];
        }
        for (int i = n - 1; i >= 0; i--) {
            ranswer[i] = rproduct;
            rproduct = rproduct * nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            answer[i] = lanswer[i] * ranswer[i];
        }
        return answer;
    }
}
