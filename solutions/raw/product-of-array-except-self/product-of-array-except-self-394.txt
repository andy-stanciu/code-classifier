class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        // Initialize the answer array
        ans[0] = 1;
        for (int i = 1; i < n; i++) {
            ans[i] = ans[i - 1] * nums[i - 1];
        }
        // Now, calculate the product of elements to the right
        int rightProduct = 1;
        for (int i = n - 1; i >= 0; i--) {
            // Update the answer array
            ans[i] *= rightProduct;
            // Update the right product
            rightProduct *= nums[i];
        }
        return ans;
    }
}
