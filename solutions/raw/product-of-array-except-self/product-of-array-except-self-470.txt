class Solution {

    public int[] productExceptSelf(int[] nums) {
        int r = 1;
        int zero = 0;
        int ex = 0;
        int[] result = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                ex = 0;
                zero++;
                continue;
            }
            r = r * nums[i];
        }
        if (zero > 1) {
            r = 0;
        }
        for (int i = 0; i < nums.length; i++) {
            if (zero == 0) {
                result[i] = r / nums[i];
            } else {
                if (nums[i] == 0) {
                    result[i] = r;
                } else
                    result[i] = 0;
            }
        }
        return result;
    }
}
