class Solution {

    public int[] productExceptSelf(int[] nums) {
        int product = 1;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0)
                product *= nums[i];
            else
                count++;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0 && count <= 1)
                nums[i] = product;
            else if (nums[i] != 0 && count == 0)
                nums[i] = product / nums[i];
            else
                nums[i] = 0;
        }
        return nums;
    }
}
