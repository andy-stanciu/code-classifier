class Solution {

    public int[] productExceptSelf(int[] nums) {
        int x = nums.length;
        int[] answer = new int[x];
        int productLeft = 1;
        for (int i = 0; i < x; i++) {
            answer[i] = productLeft;
            productLeft *= nums[i];
        }
        int productRight = 1;
        for (int i = x - 1; i >= 0; i--) {
            answer[i] *= productRight;
            productRight *= nums[i];
        }
        return answer;
    }
}
