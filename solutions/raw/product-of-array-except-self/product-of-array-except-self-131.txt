class Solution {

    public int[] productExceptSelf(int[] array) {
        int N = array.length;
        int[] prefix_product_array = new int[N];
        prefix_product_array[0] = 1;
        for (int i = 1; (i < N); i++) {
            prefix_product_array[i] = (prefix_product_array[(i - 1)] * array[(i - 1)]);
        }
        int suffix_product_value = 1;
        for (int i = (N - 1); (i >= 0); i--) {
            prefix_product_array[i] *= suffix_product_value;
            suffix_product_value *= array[i];
        }
        return prefix_product_array;
    }
}
