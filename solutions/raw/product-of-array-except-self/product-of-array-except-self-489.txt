class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        int leftprod = 1;
        for (int i = 0; i < n; i++) {
            answer[i] = leftprod;
            leftprod = leftprod * nums[i];
        }
        int rightprod = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] *= rightprod;
            rightprod *= nums[i];
        }
        return answer;
    }
}
