class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length, prod = 1;
        int[] result = new int[n];
        result[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            result[i] = nums[i] * result[i + 1];
        }
        for (int i = 0; i < n - 1; i++) {
            result[i] = result[i + 1] * prod;
            prod *= nums[i];
        }
        result[n - 1] = prod;
        return result;
    }
}
