class Solution {

    public int[] productExceptSelf(int[] nums) {
        int len = nums.length;
        int[] r = new int[len];
        int l = 1;
        r[0] = l;
        int i = 1;
        for (; i < len; i++) {
            l = l * nums[i - 1];
            r[i] = l;
        }
        int rg = 1;
        int j = len - 2;
        for (; j >= 0; j--) {
            rg = rg * nums[j + 1];
            r[j] = r[j] * rg;
        }
        return r;
    }
}
