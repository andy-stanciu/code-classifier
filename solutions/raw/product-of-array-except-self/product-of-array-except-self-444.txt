class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] ans = new int[nums.length];
        int mul = 1;
        int zeroCount = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                mul *= nums[i];
            } else {
                zeroCount++;
            }
        }
        for (int j = 0; j < nums.length; j++) {
            if (zeroCount > 1) {
                ans[j] = 0;
            } else if (zeroCount == 1) {
                ans[j] = nums[j] == 0 ? mul : 0;
            } else {
                ans[j] = mul / nums[j];
            }
        }
        return ans;
    }
}
