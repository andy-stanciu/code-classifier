class Solution {

    public int[] productExceptSelf(int[] nums) {
        int product = 1;
        int zeroCount = 0;
        for (int num : nums) {
            if (num != 0)
                product *= num;
            else
                zeroCount += 1;
        }
        for (int i = 0; i < nums.length; ++i) {
            if (zeroCount > 1) {
                nums[i] = 0;
            } else if (zeroCount == 1) {
                if (nums[i] != 0)
                    nums[i] = 0;
                else
                    nums[i] = product;
            } else {
                nums[i] = product / nums[i];
            }
        }
        return nums;
    }
}
