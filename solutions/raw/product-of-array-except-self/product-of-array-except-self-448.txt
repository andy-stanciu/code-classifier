class Solution {

    public int[] productExceptSelf(int[] nums) {
        int length = nums.length;
        int[] prefixProduct = new int[length];
        int[] suffixProduct = new int[length];
        // compute the prefix product
        for (int i = 0; i < length; i++) {
            if (i == 0) {
                prefixProduct[i] = 1;
            } else {
                prefixProduct[i] = nums[i - 1] * prefixProduct[i - 1];
            }
        }
        // compute the suffix product
        for (int i = length - 1; i >= 0; i--) {
            if (i == length - 1) {
                suffixProduct[i] = 1;
            } else {
                suffixProduct[i] = nums[i + 1] * suffixProduct[i + 1];
            }
        }
        // compute the product except self
        for (int i = 0; i < length; i++) {
            nums[i] = prefixProduct[i] * suffixProduct[i];
        }
        return nums;
    }
}
