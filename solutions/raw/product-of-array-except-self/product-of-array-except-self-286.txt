class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] p = new int[nums.length];
        Arrays.fill(p, 1);
        int lp = 1, rp = 1;
        for (int i = 0, j = nums.length - 1; i < nums.length; i++, j--) {
            p[i] = p[i] * lp;
            lp = lp * nums[i];
            p[j] = p[j] * rp;
            rp = rp * nums[j];
        }
        return p;
    }
}
