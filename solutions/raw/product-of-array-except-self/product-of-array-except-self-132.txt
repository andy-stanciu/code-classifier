class Solution {

    public int[] productExceptSelf(int[] nums) {
        int zeroes = 0;
        int product = 1;
        for (int num : nums) {
            if (num == 0) {
                zeroes++;
            } else {
                product *= num;
            }
        }
        // If there are multiple zeroes, we could return a new empty array here.
        // That would skip the second pass, but it would use O(n) space instead of O(1) space.
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0 && zeroes == 1) {
                nums[i] = product;
            } else if (zeroes >= 1) {
                nums[i] = 0;
            } else {
                nums[i] = product / nums[i];
            }
        }
        return nums;
    }
}
