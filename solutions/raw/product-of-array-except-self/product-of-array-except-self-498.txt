class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        // Paso 1: Multiplicar todos los elementos a la izquierda de cada \xEDndice
        // No hay elementos a la izquierda del primer elemento
        result[0] = 1;
        for (int i = 1; i < n; i++) {
            result[i] = result[i - 1] * nums[i - 1];
        }
        // Paso 2: Multiplicar todos los elementos a la derecha de cada \xEDndice
        // Inicializamos con 1 porque no hay elementos a la derecha del \xFAltimo elemento
        int rightProduct = 1;
        for (int i = n - 1; i >= 0; i--) {
            result[i] = result[i] * rightProduct;
            rightProduct *= nums[i];
        }
        return result;
    }
}
