class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        // Step 1: Calculate prefix products
        int[] prefixProducts = new int[n];
        // There\'s no element before the first element
        prefixProducts[0] = 1;
        for (int i = 1; i < n; i++) {
            prefixProducts[i] = prefixProducts[i - 1] * nums[i - 1];
        }
        // Step 2: Calculate suffix products and the final answer
        // Start with 1 since there\'s no element after the last element
        int suffixProduct = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] = prefixProducts[i] * suffixProduct;
            // Update suffix product for next iteration
            suffixProduct *= nums[i];
        }
        return answer;
    }
}
