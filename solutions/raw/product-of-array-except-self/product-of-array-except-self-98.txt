class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return new int[0];
        int[] answer = new int[n];
        int leftProduct = 1;
        for (int i = 0; i < n; i++) {
            answer[i] = leftProduct;
            leftProduct *= nums[i];
        }
        int rightProduct = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] *= rightProduct;
            rightProduct *= nums[i];
        }
        return answer;
    }
}
