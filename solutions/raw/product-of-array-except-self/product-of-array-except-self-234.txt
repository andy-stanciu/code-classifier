class Solution {

    public int[] productExceptSelf(int[] nums) {
        //keep count of product
        int prod = 1;
        // count of zeros
        int count_0 = 0;
        //array to return ans
        int[] ans = new int[nums.length];
        for (int num : nums) {
            //iterating each element in nums
            if (num == 0) {
                //if num is 0 then increment count_0
                count_0++;
            } else
                //else multiply prod
                prod *= num;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0 && count_0 == 1) {
                //if count_0 is 1 that means  that all elements will be zero except 0
                ans[i] = prod;
            } else if (count_0 > 0) {
                //if 0 exists then ans will be zero
                ans[i] = 0;
            } else
                //if no 0s are there, then ans of that element will just be product divided by the number itself
                ans[i] = prod / nums[i];
        }
        return ans;
    }
}
