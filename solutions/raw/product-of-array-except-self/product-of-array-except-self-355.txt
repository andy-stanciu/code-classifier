class Solution {

    public static int[] productExceptSelf(int[] nums) {
        int[] ANSWER = new int[nums.length];
        int prefix = 1;
        for (int i = 0; i < nums.length; i++) {
            ANSWER[i] = prefix;
            prefix = prefix * nums[i];
        }
        int postfix = 1;
        for (int j = nums.length - 1; j >= 0; j--) {
            ANSWER[j] = ANSWER[j] * postfix;
            postfix = postfix * nums[j];
        }
        return ANSWER;
    }
}
