/**
 * Idea: Multiply all nonZero numbers and store it in a variable(multi).
 * If zerocount = 0, then product at that index = multi/nums[i];
 * If zerocount = 1, then only that particular index would have value, rest would be 0
 * If zerocount = 2, then all the elemnts would be 0
 */
class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] fin = new int[nums.length];
        int zeroCount = 0;
        int zeroInd = -1;
        long multi = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zeroCount++;
                zeroInd = i;
            } else {
                multi *= nums[i];
            }
        }
        if (zeroCount == 0) {
            for (int i = 0; i < nums.length; i++) {
                fin[i] = (int) multi / nums[i];
            }
        } else if (zeroCount == 1) {
            fin[zeroInd] = (int) multi;
        }
        return fin;
    }
}
