class Solution {

    public int[] productExceptSelf(int[] nums) {
        int totalProduct = 1;
        //keeping track of the number of 0s
        int zeroCount = 0;
        //keeping track of zero index
        int zeroIndex = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zeroCount++;
                zeroIndex = i;
            } else {
                totalProduct *= nums[i];
            }
        }
        //initializing an array of 0s
        int[] results = new int[nums.length];
        //deal with situations that contain 0s
        if (zeroCount > 1) {
            return new int[nums.length];
        } else if (zeroCount == 1) {
            results[zeroIndex] = totalProduct;
            return results;
        }
        //when we get here, we are promised that there is no 0 in the array
        for (int i = 0; i < nums.length; i++) {
            results[i] = totalProduct / nums[i];
        }
        return results;
    }
}
