class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] R = nums.clone();
        R[nums.length - 1] = 1;
        int[] L = nums.clone();
        L[0] = 1;
        int[] Result = nums;
        int Rcal = 1;
        int Lcal = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            Lcal = Lcal * nums[i];
            L[i + 1] = Lcal;
            Rcal = Rcal * nums[nums.length - (i + 1)];
            R[nums.length - (i + 2)] = Rcal;
        }
        for (int j = 0; j < nums.length; j++) {
            Result[j] = L[j] * R[j];
        }
        return Result;
    }
}
