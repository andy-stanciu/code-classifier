class Solution {

    public int[] productExceptSelf(int[] arr) {
        int n = arr.length;
        int[] product = new int[n];
        // Calculate left products
        int left = 1;
        for (int i = 0; i < n; i++) {
            product[i] = left;
            left *= arr[i];
        }
        // Calculate right products and simultaneously multiply with left products
        int right = 1;
        for (int i = n - 1; i >= 0; i--) {
            product[i] *= right;
            right *= arr[i];
        }
        return product;
    }
}
