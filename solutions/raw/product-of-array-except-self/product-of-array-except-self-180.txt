class Solution {

    public int[] productExceptSelf(int[] nums) {
        // TIME & SPACE complexity: O(N)
        int N = nums.length;
        int product = 1;
        int[] ans = new int[N];
        // Calculate product of all elements to the left of each element
        for (int i = 0; i < N; i++) {
            ans[i] = product;
            // Update product for next element up to the previous element
            product *= nums[i];
        }
        product = 1;
        // Calculate product of all elements to the right of each element
        for (int i = N - 1; i >= 0; i--) {
            ans[i] *= product;
            // Update product for next element up to the previous element
            product *= nums[i];
        }
        return ans;
    }
}
