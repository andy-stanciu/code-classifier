class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] pre = new int[n];
        pre[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            pre[i] = nums[i - 1] * pre[i - 1];
        }
        int nxt = 1;
        int[] ans = new int[n];
        ans[n - 1] = pre[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            nxt *= nums[i + 1];
            ans[i] = nxt * pre[i];
        }
        return ans;
    }
}
