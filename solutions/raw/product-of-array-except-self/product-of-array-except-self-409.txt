class Solution {

    public int[] productExceptSelf(int[] nums) {
        int len = nums.length;
        int[] productOfAllLeftElement = new int[len];
        productOfAllLeftElement[0] = 1;
        for (int i = 1; i < len; i++) {
            productOfAllLeftElement[i] = productOfAllLeftElement[i - 1] * nums[i - 1];
        }
        System.out.println("productOfAllLeftElement: ");
        for (int i : productOfAllLeftElement) {
            System.out.print(i + " ");
        }
        int[] productOfAllRightElement = new int[len];
        productOfAllRightElement[len - 1] = 1;
        for (int i = len - 2; i >= 0; i--) {
            productOfAllRightElement[i] = productOfAllRightElement[i + 1] * nums[i + 1];
        }
        System.out.println("productOfAllRightElement: ");
        for (int i : productOfAllRightElement) {
            System.out.print(i + " ");
        }
        int[] result = new int[len];
        for (int i = 0; i < len; i++) {
            result[i] = productOfAllLeftElement[i] * productOfAllRightElement[i];
        }
        return result;
    }
}
