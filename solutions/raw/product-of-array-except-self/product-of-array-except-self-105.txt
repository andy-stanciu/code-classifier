class Solution {

    public int[] productExceptSelf(int[] nums) {
        int prod = 1;
        int zc = 0;
        for (int k : nums) {
            if (k == 0) {
                zc++;
                continue;
            }
            prod *= k;
        }
        if (zc > 1) {
            Arrays.fill(nums, 0);
            return nums;
        }
        for (int i = 0; i < nums.length; i++) {
            if (zc > 0) {
                if (nums[i] == 0) {
                    nums[i] = prod;
                } else {
                    nums[i] = 0;
                }
                continue;
            }
            nums[i] = (prod / nums[i]);
        }
        return nums;
    }
}
