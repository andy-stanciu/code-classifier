class Solution {

    public int[] productExceptSelf(int[] nums) {
        int totalPs = 1;
        int zero = 0;
        for (int num : nums) {
            if (num == 0) {
                zero++;
                continue;
            } else {
                totalPs *= num;
            }
        }
        if (zero == 1) {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] == 0) {
                    nums[i] = totalPs;
                } else {
                    nums[i] = 0;
                }
            }
        } else if (zero > 1) {
            for (int i = 0; i < nums.length; i++) {
                nums[i] = 0;
            }
        } else {
            for (int i = 0; i < nums.length; i++) {
                nums[i] = totalPs / nums[i];
            }
        }
        return nums;
    }
}
