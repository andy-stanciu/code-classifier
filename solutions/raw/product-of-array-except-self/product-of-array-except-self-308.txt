class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] arr = new int[nums.length];
        int j = 1;
        for (int i = 0; i < nums.length; i++) {
            arr[i] = j;
            j *= nums[i];
        }
        int r = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            arr[i] *= r;
            r *= nums[i];
        }
        return arr;
    }
}
