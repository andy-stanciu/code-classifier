class Solution {

    public int[] productExceptSelf(int[] nums) {
        int product = 1;
        int zeroOccurances = 0;
        int index = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zeroOccurances++;
                index = i;
            } else
                product *= nums[i];
        }
        int[] res = new int[nums.length];
        if (zeroOccurances > 1)
            return res;
        else if (zeroOccurances == 1) {
            res[index] = product;
            return res;
        } else {
            for (int i = 0; i < nums.length; i++) {
                res[i] = product / nums[i];
            }
        }
        return res;
    }
}
