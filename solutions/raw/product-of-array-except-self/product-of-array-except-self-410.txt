class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] output = new int[nums.length];
        int product = 1;
        int count = 0;
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                product *= nums[i];
            } else {
                count++;
                index = i;
            }
        }
        if (count == 0) {
            for (int i = 0; i < nums.length; i++) {
                output[i] = product / nums[i];
            }
            return output;
        } else if (count == 1) {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] == 0) {
                    output[i] = product;
                }
            }
        }
        return output;
    }
}
