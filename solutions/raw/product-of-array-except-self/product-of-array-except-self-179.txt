class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] res = new int[n];
        int leftPro = 1;
        for (int i = 0; i < n; i++) {
            res[i] = leftPro;
            leftPro *= nums[i];
        }
        int rightPro = 1;
        for (int i = n - 1; i >= 0; i--) {
            res[i] *= rightPro;
            rightPro *= nums[i];
        }
        return res;
    }
}
