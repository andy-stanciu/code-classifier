class Solution {

    public int[] productExceptSelf(int[] nums) {
        int[] res = new int[nums.length];
        Arrays.fill(res, 1);
        for (int i = 0, pre = 1, suf = 1, n = nums.length; i < n; res[i] *= pre, pre *= nums[i], res[n - i - 1] *= suf, suf *= nums[n - i - 1], i++) ;
        return res;
    }
}
