class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        int tmp = 1;
        for (int i = 0; i < n - 1; i++) {
            tmp *= nums[i];
            answer[i + 1] = tmp;
        }
        tmp = 1;
        for (int i = n - 1; i > 1; i--) {
            tmp *= nums[i];
            answer[i - 1] *= tmp;
        }
        tmp *= nums[1];
        answer[0] = tmp;
        return answer;
    }
}
