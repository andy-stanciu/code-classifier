class Solution {

    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        int product = 1;
        int zeroCount = 0;
        // Calculate the product of all non-zero elements and count the zeroes
        for (int num : nums) {
            if (num != 0) {
                product *= num;
            } else {
                zeroCount++;
            }
        }
        // Fill the result array based on the number of zeroes
        for (int i = 0; i < n; i++) {
            if (zeroCount > 1) {
                // More than one zero, all elements are zero
                result[i] = 0;
            } else if (zeroCount == 1) {
                // One zero, set product only at the zero\'s position
                result[i] = nums[i] == 0 ? product : 0;
            } else {
                // No zero, regular division
                result[i] = product / nums[i];
            }
        }
        return result;
    }
}
