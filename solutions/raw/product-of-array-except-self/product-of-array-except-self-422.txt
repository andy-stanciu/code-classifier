class Solution {

    public int[] productExceptSelf(int[] nums) {
        int prod = 1;
        boolean hasZero = false;
        for (int n : nums) {
            if (n != 0) {
                prod *= n;
            } else if (hasZero) {
                return new int[nums.length];
            } else {
                hasZero = true;
            }
        }
        int[] answer = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0 && hasZero) {
                answer[i] = 0;
            } else if (nums[i] == 0) {
                answer[i] = prod;
            } else {
                answer[i] = prod / nums[i];
            }
        }
        return answer;
    }
}
