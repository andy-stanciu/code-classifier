class Solution {

    public int[] productExceptSelf(int[] nums) {
        int product_without_zero = 1;
        boolean is_zero_product = false;
        int[] result = new int[nums.length];
        int zero_count = 0;
        for (int index = 0; index < nums.length; ++index) {
            if (nums[index] != 0) {
                product_without_zero = product_without_zero * nums[index];
            } else {
                is_zero_product = true;
                ++zero_count;
            }
        }
        int product = product_without_zero * (is_zero_product ? 0 : 1);
        for (int index = 0; index < nums.length; ++index) {
            if (zero_count > 1) {
                result[index] = 0;
            } else {
                result[index] = nums[index] == 0 ? product_without_zero : (product / nums[index]);
            }
        }
        return result;
    }
}
