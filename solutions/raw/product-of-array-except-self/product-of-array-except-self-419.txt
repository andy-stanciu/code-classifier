class Solution {

    public int[] productExceptSelf(int[] nums) {
        int length = nums.length;
        if (length == 1)
            return nums;
        int[] product = new int[length];
        int prefix = 1;
        for (int i = 0; i < length; i++) {
            product[i] = prefix;
            prefix = nums[i] * prefix;
        }
        int suffix = 1;
        for (int j = length - 1; j >= 0; j--) {
            product[j] = product[j] * suffix;
            suffix = nums[j] * suffix;
        }
        return product;
    }
}
