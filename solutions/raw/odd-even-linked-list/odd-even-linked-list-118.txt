class Solution {

    public ListNode oddEvenList(ListNode head) {
        //no need for comment to understand this code, its self explanatory
        ListNode evenH = new ListNode(-1);
        ListNode oddH = new ListNode(-1);
        ListNode evenT = evenH;
        ListNode oddT = oddH;
        ListNode curr = head;
        int idx = 1;
        while (curr != null) {
            if (idx % 2 == 0) {
                evenT.next = curr;
                evenT = evenT.next;
            } else {
                oddT.next = curr;
                oddT = oddT.next;
            }
            curr = curr.next;
            idx++;
        }
        evenT.next = null;
        oddT.next = evenH.next;
        return oddH.next;
    }
}
