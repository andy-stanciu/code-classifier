/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null)
            return null;
        ListNode odd = new ListNode(-1);
        ListNode oHead = odd;
        ListNode even = new ListNode(-1);
        ListNode eHead = even;
        ListNode curr = head;
        int index = 1;
        while (curr != null) {
            if (index % 2 != 0) {
                oHead.next = curr;
                oHead = oHead.next;
            } else {
                eHead.next = curr;
                eHead = eHead.next;
            }
            curr = curr.next;
            index++;
        }
        eHead.next = null;
        oHead.next = even.next;
        return odd.next;
    }
}
