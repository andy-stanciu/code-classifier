class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode odd = head;
        ListNode evenTail = head.next;
        ListNode evenHead = head.next;
        while (evenTail != null && evenTail.next != null) {
            odd.next = evenTail.next;
            odd = odd.next;
            evenTail.next = odd.next;
            evenTail = evenTail.next;
            odd.next = evenHead;
        }
        return head;
    }
}
