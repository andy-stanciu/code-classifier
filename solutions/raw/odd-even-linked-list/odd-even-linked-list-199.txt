class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null)
            return head;
        ListNode tail = head;
        int numNodes = 1;
        //save the last element in list and count number of nodes
        while (tail.next != null) {
            tail = tail.next;
            numNodes++;
        }
        if (numNodes == 1 || numNodes == 2)
            return head;
        int n = numNodes / 2;
        ListNode dummy = head;
        for (int i = 0; i < n; i++) {
            // save the next odd node
            ListNode save = head.next.next;
            // append even node to the end
            tail.next = head.next;
            // set its next to null to avoid cycle
            tail.next.next = null;
            // append the saved odd number
            head.next = save;
            // move forward
            head = head.next;
            // move forward
            tail = tail.next;
        }
        return dummy;
    }
}
