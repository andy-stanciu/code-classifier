class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null)
            return head;
        // CREATING ODD AND EVEN POINTERS
        ListNode odd = head;
        ListNode even = head.next;
        // CREATING EVENHEAD TO CONNECT WITH ODD LIST LATER
        ListNode evenHead = even;
        while (even != null && even.next != null) {
            //point odd\'s next to even\'s next
            odd.next = even.next;
            //point even\'s next to even\'s next\'s next
            even.next = even.next.next;
            //move odd and even pointers
            odd = odd.next;
            even = even.next;
        }
        // CONNECT ODD WITH EVEN
        odd.next = evenHead;
        // return Head.
        return head;
    }
    // don\'t forget to upvote solution.
}
