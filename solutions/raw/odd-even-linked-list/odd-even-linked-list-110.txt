class Solution {

    public ListNode oddEvenList(ListNode head) {
        ListNode p = head, q = null;
        if (head != null && head.next != null) {
            q = head.next;
        } else {
            return head;
        }
        ListNode odd = p, even = q;
        while (even != null && even.next != null) {
            odd.next = even.next;
            odd = odd.next;
            even.next = odd.next;
            even = even.next;
        }
        odd.next = q;
        return p;
    }
}
