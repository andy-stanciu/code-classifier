/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null || head.next == null) {
            // If there\'s 0 or 1 node, no reordering is needed.
            return head;
        }
        // Start of the odd-indexed list
        ListNode odd = head;
        // Start of the even-indexed list
        ListNode even = head.next;
        // Save the head of the even list to link later
        ListNode evenHead = even;
        while (even != null && even.next != null) {
            // Link the next odd node
            odd.next = even.next;
            // Move to the next odd node
            odd = odd.next;
            // Link the next even node
            even.next = odd.next;
            // Move to the next even node
            even = even.next;
        }
        // Connect the end of the odd list to the start of the even list
        odd.next = evenHead;
        return head;
    }
}
