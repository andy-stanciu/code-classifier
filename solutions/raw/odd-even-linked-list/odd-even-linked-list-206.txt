class Solution {

    public ListNode oddEvenList(ListNode head) {
        ListNode temp = head;
        int count = 0;
        ListNode even = new ListNode();
        ListNode curr = even;
        ListNode odd = new ListNode();
        ListNode oddCurr = odd;
        while (temp != null) {
            count++;
            if (count % 2 == 0) {
                curr.next = new ListNode(temp.val);
                curr = curr.next;
            } else {
                oddCurr.next = new ListNode(temp.val);
                oddCurr = oddCurr.next;
            }
            temp = temp.next;
        }
        temp = odd.next;
        while (temp != null) {
            if (temp.next == null) {
                temp.next = even.next;
                break;
            }
            temp = temp.next;
        }
        return odd.next;
    }
}
