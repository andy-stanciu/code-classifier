class Solution2 {

    public ListNode oddEvenList(ListNode head) {
        ListNode res = head;
        ListNode ptr = head;
        ListNode resPtr = res;
        while (ptr != null && ptr.next != null) {
            resPtr.next = ptr.next.next;
            ptr = ptr.next.next;
            resPtr = resPtr.next;
        }
        ptr = head.next;
        return res;
    }
}
