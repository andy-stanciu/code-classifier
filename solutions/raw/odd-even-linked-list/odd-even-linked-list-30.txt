class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null)
            return null;
        ListNode start = head;
        ListNode sec = head.next;
        ListNode join = head.next;
        while (sec != null && start != null) {
            if (sec == null || sec.next == null)
                break;
            start.next = start.next.next;
            start = start.next;
            sec.next = sec.next.next;
            sec = sec.next;
        }
        start.next = join;
        return head;
    }
}
