class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode even = head, odd = head.next, oddHead = head.next;
        while (odd != null && odd.next != null) {
            even.next = even.next.next;
            odd.next = odd.next.next;
            even = even.next;
            odd = odd.next;
        }
        even.next = oddHead;
        return head;
    }
}
