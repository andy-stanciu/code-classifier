/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode nextOdd = head;
        ListNode nextEven = head;
        ListNode prevOdd = new ListNode(0);
        nextEven = nextEven.next;
        nextOdd = nextOdd.next.next;
        prevOdd = head;
        while (nextOdd != null) {
            ListNode temp = prevOdd.next;
            prevOdd.next = nextOdd;
            nextEven.next = nextOdd.next;
            nextOdd.next = temp;
            prevOdd = nextOdd;
            nextEven = nextEven.next;
            if (nextEven == null || nextEven.next == null)
                break;
            nextOdd = nextEven.next;
        }
        return head;
    }
}
