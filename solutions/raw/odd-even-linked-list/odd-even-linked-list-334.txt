class Solution {

    public ListNode oddEvenList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        // ListNode current=head;
        // int n=0;
        // while(current!=null){
        //     n++;
        //     current=current.next;
        // }
        // for(int i=1;i<n;i++){
        //     if(i%2==0){
        //         current.next=current.next.next;
        //     }
        //     if(i%2!==0){
        //         current.append();
        //     }
        // }
        // return head;
        ListNode odd = head;
        ListNode even = head.next;
        ListNode evenHead = even;
        while (even != null && even.next != null) {
            odd.next = even.next;
            odd = odd.next;
            even.next = odd.next;
            even = even.next;
        }
        odd.next = evenHead;
        return head;
    }
}
