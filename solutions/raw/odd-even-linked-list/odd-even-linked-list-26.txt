class Solution {

    public ListNode oddEvenList(ListNode head) {
        ListNode odd = new ListNode();
        ListNode p1 = odd;
        ListNode even = new ListNode();
        ListNode p2 = even;
        while (head != null) {
            if (head.next == null) {
                p1.next = head;
                p1 = p1.next;
                break;
            }
            p1.next = head;
            p2.next = head.next;
            p1 = p1.next;
            p2 = p2.next;
            head = head.next.next;
        }
        p1.next = even.next;
        p2.next = null;
        return odd.next;
    }
}
