class Solution {

    public ListNode oddEvenList(ListNode head) {
        ListNode even = null, odd = null, evenHead = null, oddHead = null;
        if (head == null)
            return head;
        int index = 1;
        while (head != null) {
            if ((index % 2) == 1) {
                if (oddHead == null)
                    oddHead = head;
                else
                    odd.next = head;
                odd = head;
            } else {
                if (evenHead == null)
                    evenHead = head;
                else
                    even.next = head;
                even = head;
            }
            head = head.next;
            index++;
        }
        if (even != null)
            even.next = null;
        if (oddHead != null)
            odd.next = evenHead;
        return oddHead;
    }
}
