class Solution {

    public int compress(char[] chars) {
        int resultLength = 0;
        int j = 0;
        while (j < chars.length) {
            char currentChar = chars[j];
            int count = 0;
            while (j < chars.length && chars[j] == currentChar) {
                count++;
                j++;
            }
            chars[resultLength] = currentChar;
            resultLength++;
            String countString = "" + count;
            int i = 0;
            while (count > 1 && i < countString.length()) {
                chars[resultLength] = countString.charAt(i);
                i++;
                resultLength++;
            }
        }
        return resultLength;
    }
}
