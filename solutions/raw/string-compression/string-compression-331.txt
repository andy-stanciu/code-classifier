class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1)
            return 1;
        int updateIndex = 0;
        char lastSeen = chars[0];
        int lastSeenCount = 1;
        for (int i = 1; i < chars.length; i++) {
            char curChar = chars[i];
            if (curChar == lastSeen) {
                lastSeenCount++;
            } else {
                chars[updateIndex++] = lastSeen;
                if (lastSeenCount > 1) {
                    String count = String.valueOf(lastSeenCount);
                    for (int j = 0; j < count.length(); j++) chars[updateIndex++] = count.charAt(j);
                }
                lastSeen = curChar;
                lastSeenCount = 1;
            }
        }
        chars[updateIndex++] = lastSeen;
        if (lastSeenCount > 1) {
            String count = String.valueOf(lastSeenCount);
            for (int j = 0; j < count.length(); j++) chars[updateIndex++] = count.charAt(j);
        }
        return updateIndex;
    }
}
