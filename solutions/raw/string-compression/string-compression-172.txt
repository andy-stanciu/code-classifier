class Solution {

    public static int compress(char[] chars) {
        // Index to write to in the chars array
        int writeindex = 0;
        // Index to read from the chars array
        int i = 0;
        while (i < chars.length) {
            char curr = chars[i];
            int point = i;
            // Count the number of consecutive characters
            while (i < chars.length && chars[i] == curr) {
                i++;
            }
            // Write the character to the array
            chars[writeindex++] = curr;
            // Write the count if greater than 1
            if (i - point > 1) {
                for (char c : Integer.toString(i - point).toCharArray()) {
                    chars[writeindex++] = c;
                }
            }
        }
        return writeindex;
    }
}
