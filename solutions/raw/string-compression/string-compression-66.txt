class Solution {

    public int compress(char[] chars) {
        if (chars.length < 1) {
            return -1;
        }
        if (chars.length == 1) {
            return 1;
        }
        int count = 1;
        // always points to the index where it needs to be modified
        int startIndex = 1;
        for (int i = 1; i < chars.length; i++) {
            if (chars[i] != chars[i - 1]) {
                if (count > 1) {
                    String num = "" + count;
                    // handle num more than signle digits
                    for (int j = 0; j < num.length(); j++) {
                        chars[startIndex] = num.charAt(j);
                        startIndex++;
                    }
                    count = 1;
                }
                // need to set this as curr character which is not equal and increment the index by 1
                // so we can point to new index which needs to be updated
                chars[startIndex] = chars[i];
                startIndex++;
            } else {
                count++;
            }
        }
        if (count > 1) {
            String num = "" + count;
            for (int j = 0; j < num.length(); j++) {
                chars[startIndex] = num.charAt(j);
                startIndex++;
            }
            count = 1;
        }
        return startIndex;
    }
}
