class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        int count = 1;
        for (int i = 0; i < chars.length - 1; i++) {
            if (chars[i] == chars[i + 1]) {
                count++;
            } else {
                if (count == 1) {
                    sb.append(chars[i]);
                } else {
                    sb.append(chars[i - 1]);
                    sb.append(count);
                }
                count = 1;
            }
        }
        if (count == 1) {
            sb.append(chars[chars.length - 1]);
        } else {
            sb.append(chars[chars.length - 1]);
            sb.append(count);
        }
        for (int i = 0; i < chars.length; i++) {
            if (i > sb.length() - 1) {
                break;
            }
            chars[i] = sb.charAt(i);
        }
        return sb.length();
    }
}
