class Solution {

    public int compress(char[] chars) {
        int stringLength = 0;
        int left = 0;
        int index = 0;
        while (left < chars.length) {
            int right = left;
            char c = chars[left];
            int length = 0;
            while (right < chars.length && chars[left] == chars[right]) {
                right++;
                length++;
            }
            chars[index] = c;
            index++;
            if (length > 1) {
                char[] digits = String.valueOf(length).toCharArray();
                for (int i = 0; i < digits.length; i++) {
                    chars[index] = digits[i];
                    index++;
                    stringLength++;
                }
            }
            left = right;
            stringLength++;
        }
        return stringLength;
    }
}
