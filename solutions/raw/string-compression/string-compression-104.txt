class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int c = 0;
        int i = 0;
        while (i < n) {
            char currentChar = chars[i];
            int count = 0;
            while (i < n && chars[i] == currentChar) {
                i++;
                count++;
            }
            chars[c++] = currentChar;
            if (count > 1) {
                String countStr = Integer.toString(count);
                for (char digit : countStr.toCharArray()) {
                    chars[c++] = digit;
                }
            }
        }
        return c;
    }
}
