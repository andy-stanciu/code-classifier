class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1)
            return 1;
        int flag = 1;
        String s = "";
        for (int i = 0; i < chars.length - 1; i++) {
            if (chars[i] != chars[i + 1]) {
                if (flag == 1) {
                    s = s + chars[i];
                } else {
                    s = s + chars[i] + flag;
                }
                flag = 1;
            } else {
                flag++;
            }
        }
        if (flag > 1) {
            s = s + chars[chars.length - 1] + flag;
        } else {
            s = s + chars[chars.length - 1];
        }
        for (int i = 0; i < s.length(); i++) {
            chars[i] = s.charAt(i);
        }
        chars = Arrays.copyOfRange(chars, 0, s.length());
        return chars.length;
    }
}
