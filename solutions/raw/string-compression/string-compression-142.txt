class Solution {

    public int compress(char[] chars) {
        // To place compressed characters in chars array
        int index = 0;
        // Pointer to traverse the chars array
        int i = 0;
        while (i < chars.length) {
            char currentChar = chars[i];
            int count = 0;
            // Count consecutive occurrences of currentChar
            while (i < chars.length && chars[i] == currentChar) {
                i++;
                count++;
            }
            // Place the character
            chars[index++] = currentChar;
            // If count > 1, append the digits of the count
            if (count > 1) {
                for (char c : String.valueOf(count).toCharArray()) {
                    chars[index++] = c;
                }
            }
        }
        // New length of the compressed array
        return index;
    }
}
