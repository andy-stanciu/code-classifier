class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1)
            return 1;
        int ans = 0;
        for (int i = 0; i < chars.length; ) {
            int c = 1;
            for (int j = i + 1; j < chars.length; j++) {
                if (chars[i] == chars[j])
                    c++;
                else
                    break;
            }
            chars[ans] = chars[i];
            ans++;
            if (c > 1) {
                String n = Integer.toString(c);
                for (char cn : n.toCharArray()) {
                    chars[ans] = cn;
                    ans++;
                }
            }
            i = i + c;
        }
        return ans;
    }
}
