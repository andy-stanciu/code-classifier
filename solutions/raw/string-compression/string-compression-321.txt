class Solution {

    public int compress(char[] chars) {
        if (chars.length > 1) {
            int ind = 0;
            int count = 1;
            for (int i = 0; i < chars.length; i++) {
                chars[ind++] = chars[i];
                while (i + 1 < chars.length && chars[i + 1] == chars[i]) {
                    count++;
                    i++;
                }
                if (count > 1) {
                    for (Character c : Integer.toString(count).toCharArray()) {
                        chars[ind++] = c;
                    }
                    count = 1;
                }
            }
            return ind;
        }
        return chars.length;
    }
}
