// TC: O(n)
// SC: O(1)
class Solution {

    public int compress(char[] chars) {
        // Index for writing
        int index = 0;
        // Index for reading
        int i = 0;
        while (i < chars.length) {
            char currChar = chars[i];
            int count = 0;
            // Count occurrences of currentChar
            while (i < chars.length && chars[i] == currChar) {
                count++;
                i++;
            }
            // Write the current character to chars
            chars[index++] = currChar;
            // Write the count if it\'s greater than 1
            if (count > 1) {
                // Convert count into a char array and write each digit
                String s = Integer.toString(count);
                for (char ch : s.toCharArray()) {
                    chars[index++] = ch;
                }
            }
        }
        return index;
    }
}
