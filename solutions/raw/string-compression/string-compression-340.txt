class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int read = 0;
        int write = 0;
        int anchor = 0;
        for (read = 0; read < n; read++) {
            char current = chars[read];
            if (read == n - 1 || current != chars[read + 1]) {
                chars[write++] = current;
                int count = read - anchor + 1;
                if (count > 1) {
                    char[] countDigits = Integer.toString(count).toCharArray();
                    for (char digit : countDigits) {
                        chars[write++] = digit;
                    }
                }
                anchor = read + 1;
            }
        }
        return write;
    }
}
