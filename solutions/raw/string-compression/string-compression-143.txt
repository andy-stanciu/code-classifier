class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int c = 1;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n - 1; i++) {
            if (chars[i] == chars[i + 1]) {
                c++;
            } else {
                sb.append(chars[i]);
                if (c > 1) {
                    sb.append(c);
                }
                c = 1;
            }
        }
        sb.append(chars[n - 1]);
        if (c > 1)
            sb.append(c);
        String compressed = sb.toString();
        for (int i = 0; i < compressed.length(); i++) {
            chars[i] = compressed.charAt(i);
        }
        return compressed.length();
    }
}
