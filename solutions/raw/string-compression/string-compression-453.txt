class Solution {

    public int compress(char[] chars) {
        int ptr = -1;
        for (int i = 0; i < chars.length; i++) {
            char c = chars[i];
            int counter = 1;
            while (i + 1 < chars.length && chars[i] == chars[i + 1]) {
                counter++;
                i++;
            }
            if (counter > 1) {
                ptr++;
                chars[ptr] = chars[i];
                String counterAsString = Integer.toString(counter);
                for (int x = 0; x < counterAsString.length(); x++) {
                    ptr++;
                    chars[ptr] = counterAsString.charAt(x);
                }
            } else {
                ptr++;
                chars[ptr] = chars[i];
            }
        }
        return ptr + 1;
    }
}
