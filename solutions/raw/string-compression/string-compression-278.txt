class Solution {

    public int compress(char[] chars) {
        var resultStringBuilder = new StringBuilder();
        char currentProcessingChar = chars[0];
        var count = 0;
        for (int i = 0; i < chars.length; i++) {
            if (chars[i] == currentProcessingChar) {
                count++;
            }
            if (chars[i] != currentProcessingChar || i == chars.length - 1) {
                resultStringBuilder.append(currentProcessingChar);
                if (count > 1) {
                    resultStringBuilder.append(count);
                }
                if (chars[i] != currentProcessingChar && i == chars.length - 1) {
                    resultStringBuilder.append(chars[i]);
                } else {
                    currentProcessingChar = chars[i];
                    count = 1;
                }
            }
        }
        var resultString = resultStringBuilder.toString();
        for (int i = 0; i < resultString.length(); i++) {
            chars[i] = resultString.charAt(i);
        }
        return resultString.length();
    }
}
