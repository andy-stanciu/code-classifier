class Solution {

    public int compress(char[] chars) {
        // Index to track where to place the next character in the compressed array
        int index = 0;
        // Loop through the characters array
        for (int i = 0; i < chars.length; i++) {
            char currentChar = chars[i];
            // Initialize count for the current character
            int count = 1;
            // Count consecutive occurrences of the current character
            while (i + 1 < chars.length && chars[i + 1] == currentChar) {
                count++;
                i++;
            }
            // Update the character array with the compressed version
            chars[index++] = currentChar;
            if (count > 1) {
                char[] countArray = String.valueOf(count).toCharArray();
                for (char c : countArray) {
                    chars[index++] = c;
                }
            }
        }
        // Return the length of the compressed array
        return index;
    }
}
