class Solution {

    public int compress(char[] chars) {
        int i = 0;
        int j = 0;
        while (i < chars.length) {
            char currentChar = chars[i];
            int counter = 0;
            while (i < chars.length && currentChar == chars[i]) {
                i++;
                counter++;
            }
            chars[j++] = currentChar;
            if (counter > 1) {
                for (char c : Integer.toString(counter).toCharArray()) chars[j++] = c;
            }
        }
        return j;
    }
}
