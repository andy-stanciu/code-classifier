class Solution {

    public int compress(char[] chars) {
        String s = "";
        int i = 0;
        while (i < chars.length) {
            s += chars[i];
            int length = 1;
            while ((i < chars.length - 1) && chars[i] == chars[i + 1]) {
                i++;
                length++;
            }
            if (length > 1) {
                s += String.valueOf(length);
            }
            i++;
        }
        int j;
        for (j = 0; j < s.length(); j++) {
            chars[j] = s.charAt(j);
        }
        for (int k = j; k < chars.length; k++) {
            chars[k] = 0;
        }
        return s.length();
    }
}
