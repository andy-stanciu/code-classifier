class Solution {

    public int compress(char[] chars) {
        char currentChar = chars[0];
        int index = 1;
        StringBuilder result = new StringBuilder();
        while (index <= chars.length) {
            int repetition\u0421ounter = 1;
            while (index <= chars.length - 1 && currentChar == chars[index]) {
                index++;
                repetition\u0421ounter++;
            }
            result.append(currentChar);
            if (repetition\u0421ounter > 1) {
                result.append(repetition\u0421ounter);
            }
            if (index <= chars.length - 1) {
                currentChar = chars[index];
            }
            index++;
        }
        char[] comp = result.toString().toCharArray();
        System.arraycopy(comp, 0, chars, 0, comp.length);
        return comp.length;
    }
}
