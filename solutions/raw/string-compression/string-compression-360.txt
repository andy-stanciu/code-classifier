class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        int count = 1;
        for (int i = 0; i < chars.length; i++) {
            if (i + 1 < chars.length && chars[i] == chars[i + 1]) {
                count++;
            } else {
                sb.append(chars[i]);
                if (count > 1) {
                    sb.append(count);
                }
                // Reset count for the next character
                count = 1;
            }
        }
        // Copy compressed characters back to the input array
        char[] compressedArray = sb.toString().toCharArray();
        for (int i = 0; i < compressedArray.length; i++) {
            chars[i] = compressedArray[i];
        }
        return sb.length();
    }
}
