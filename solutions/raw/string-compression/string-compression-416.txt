class Solution {

    public int compress(char[] chars) {
        int count = 1;
        char current = chars[0];
        StringBuilder sb = new StringBuilder();
        // if array has only char
        if (chars.length == 1) {
            return 1;
        }
        for (int i = 1; i < chars.length; i++) {
            // if current == next one also then increment the counter
            if (current == chars[i]) {
                count++;
            } else {
                //  in case next is new char then append the current char to sb along with count
                append(sb, current, count);
                // assign the next char as current and reset the count to 1
                current = chars[i];
                count = 1;
            }
        }
        append(sb, current, count);
        for (int i = 0; i < sb.length(); i++) {
            chars[i] = sb.charAt(i);
        }
        return sb.length();
    }

    public void append(StringBuilder sb, char current, int count) {
        // sb = sb+current;
        sb.append(current);
        // append the count only when count is greater than one
        if (count > 1) {
            // Convert the number to a string
            String numberAsString = Integer.toString(count);
            // Create a char array to store the digits
            // char[] charArray = new char[numberAsString.length()];
            // Iterate through each digit in the number
            for (int j = 0; j < numberAsString.length(); j++) {
                // Extract the current digit as a char
                char digit = numberAsString.charAt(j);
                // Append the digit to the char array
                sb.append(digit);
            }
        }
    }
}
