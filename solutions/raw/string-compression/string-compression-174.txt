class Solution {

    public int compress(char[] chars) {
        String s = Character.toString(chars[0]);
        int count = 1;
        for (int i = 0; i < chars.length - 1; i++) {
            if (chars[i] == chars[i + 1]) {
                count++;
            } else {
                if (count != 1) {
                    s += Integer.toString(count);
                }
                count = 1;
                s += Character.toString(chars[i + 1]);
            }
        }
        if (count != 1) {
            s += Integer.toString(count);
        }
        for (int i = 0; i < s.length(); i++) {
            chars[i] = s.charAt(i);
        }
        return s.length();
    }
}
