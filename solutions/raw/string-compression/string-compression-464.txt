class Solution {

    public int compress(char[] chars) {
        if (chars.length == 0) {
            return 0;
        }
        if (chars.length == 1) {
            return 1;
        }
        int cnt = 0;
        int pos = 0;
        for (int i = 0; i < chars.length; i++) {
            char cur = chars[i];
            char next = i == chars.length - 1 ? 0 : chars[i + 1];
            cnt++;
            if (cur != next) {
                chars[pos++] = cur;
                if (cnt > 1) {
                    for (char digit : Integer.toString(cnt).toCharArray()) {
                        chars[pos++] = digit;
                    }
                }
                cnt = 0;
            }
        }
        return pos;
    }
}
