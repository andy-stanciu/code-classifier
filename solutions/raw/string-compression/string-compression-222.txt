class Solution {

    public int compress(char[] chars) {
        // Index to place the compressed characters
        int index = 0;
        // Iterator over the chars array
        int i = 0;
        while (i < chars.length) {
            // Current character to compress
            char currentChar = chars[i];
            // Count occurrences of the current character
            int count = 0;
            // Count the number of occurrences of the current character
            while (i < chars.length && chars[i] == currentChar) {
                i++;
                count++;
            }
            // Place the character in the array
            chars[index++] = currentChar;
            // If count is more than 1, convert it to string and place the digits in the array
            if (count > 1) {
                for (char c : Integer.toString(count).toCharArray()) {
                    chars[index++] = c;
                }
            }
        }
        // Return the new length of the compressed array
        return index;
    }
}
