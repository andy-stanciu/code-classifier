class Solution {

    public int compress(char[] chars) {
        // To keep track of where to write in the array
        int index = 0;
        // Pointer to traverse through the array
        int i = 0;
        while (i < chars.length) {
            char currentChar = chars[i];
            int count = 0;
            // Count the occurrences of the current character
            while (i < chars.length && chars[i] == currentChar) {
                i++;
                count++;
            }
            // Write the character to the current position
            chars[index++] = currentChar;
            // If the count is greater than 1, write the count as well
            if (count > 1) {
                // Convert the count to string and write each digit separately
                for (char c : String.valueOf(count).toCharArray()) {
                    chars[index++] = c;
                }
            }
        }
        // Return the new length of the compressed array
        return index;
    }
}
