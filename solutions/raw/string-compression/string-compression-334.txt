class Solution {

    public int compress(char[] chars) {
        /*
         * The Time Complexity of this approach is O(N).
         * 
         * Don\'t get confused by the time Complexity as O(N*M) because of for loop 
         * for N and while loop M for occurrences thats  completely wrong. when you counting the occurences you are
         * incrementing the value of i that makes time complexity O(N).
         * 
         * Space Complexity : O(1)
         */
        if (chars.length <= 1)
            return chars.length;
        else {
            String result = "";
            for (int i = 0; i < chars.length; i++) {
                int characterCount = 1;
                while (i < chars.length - 1 && chars[i] == chars[i + 1]) {
                    i++;
                    characterCount++;
                }
                if (characterCount == 1)
                    result += chars[i];
                else
                    result += chars[i] + String.valueOf(characterCount);
            }
            for (int i = 0; i < result.length(); i++) chars[i] = result.charAt(i);
            return result.length();
        }
    }
}
