class Solution {

    public int[] twoSum(int[] a, int sum) {
        Map<Integer, Integer> map = new HashMap();
        int count = 0;
        for (int e : a) {
            int comp = sum - e;
            if (map.containsKey(comp)) {
                int[] arr = { count, map.get(comp) };
                return arr;
            }
            map.put(e, count);
            count++;
        }
        return null;
    }
}
