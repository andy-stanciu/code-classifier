class Solution {

    public int[] twoSum(int[] nums, int target) {
        Random random = new Random();
        int[] out = { 0, 0 };
        int r1, r2;
        while (1 == 1) {
            r1 = random.nextInt(nums.length);
            r2 = random.nextInt(nums.length);
            while (r1 == r2) {
                r2 = random.nextInt(nums.length);
            }
            if (nums[r1] + nums[r2] == target) {
                out[0] = r1;
                out[1] = r2;
                break;
            }
        }
        return out;
    }
}
