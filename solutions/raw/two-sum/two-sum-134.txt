class Solution {

    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> numsMap = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            int complement = target - nums[i];
            if (numsMap.containsKey(complement)) {
                return new int[] { i, numsMap.get(complement) };
            }
            numsMap.put(nums[i], i);
        }
        return new int[] {};
    }
}
