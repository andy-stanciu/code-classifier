class Solution {

    public int[] twoSum(int[] nums, int target) {
        int n = nums.length;
        //map
        HashMap<Integer, Integer> mp = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int tofind = target - nums[i];
            if (mp.containsKey(tofind)) {
                //return index of both terms
                return new int[] { mp.get(tofind), i };
            } else {
                mp.put(nums[i], i);
            }
        }
        //empty arr
        return new int[] {};
    }
}
