class Solution {

    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> mp = new HashMap<>();
        int[] empty = {};
        for (int i = 0; i < nums.length; i++) {
            if (mp.containsKey(target - nums[i])) {
                int[] out = { mp.get(target - nums[i]), i };
                return out;
            }
            mp.put(nums[i], i);
        }
        return empty;
    }
}
