class Solution {

    public int[] twoSum(int[] nums, int target) {
        HashMap<Long, Integer> table = new HashMap<Long, Integer>();
        for (int i = 0; i != nums.length; i++) {
            Integer id = table.get((long) target - nums[i]);
            if (id != null)
                return new int[] { id, i };
            table.put((long) nums[i], i);
        }
        return new int[] { -1, -1 };
    }
}
