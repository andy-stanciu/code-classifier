class Solution {

    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> visitedNums = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            //difference = given target - number given at ith index
            int difference = target - nums[i];
            if (// check if found difference is present in the MAP list
            visitedNums.containsKey(difference)) {
                //if difference in map matches with the ith index element in array
                return new int[] { i, visitedNums.get(difference) };
            }
            //addition of previous array element in map to match with future element if forms a pair
            visitedNums.put(nums[i], i);
        }
        //if no matches are found
        return new int[] { 0, 0 };
    }
}
