class Solution {

    public int[] twoSum(int[] nums, int target) {
        //new array for answer
        int[] ans = new int[2];
        for (int i = 0; i < nums.length; i++) {
            //first loop
            for (int j = i + 1; j < nums.length; j++) {
                // 2nd loop for intrting throgh other values of nums[]
                if (nums[i] + nums[j] == target) {
                    // if condition to check if sum equals to the target
                    // apemding i into answer array
                    ans[0] = i;
                    // appending j into the answer array
                    ans[1] = j;
                }
            }
        }
        // return answer
        return ans;
    }
}
