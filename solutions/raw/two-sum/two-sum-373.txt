class Solution {

    public int[] twoSum(int[] nums, int target) {
        HashMap hashMap = new HashMap();
        for (int i = 0; i < nums.length; i++) {
            if (hashMap.containsKey(nums[i]))
                return new int[] { (int) hashMap.get(nums[i]), i };
            else
                hashMap.put(target - nums[i], i);
        }
        return new int[] {};
    }
}
