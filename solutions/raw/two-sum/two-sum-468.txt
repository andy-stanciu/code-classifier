class Solution {

    public int[] twoSum(int[] nums, int target) {
        int[] arr = new int[2];
        int m = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (i != j) {
                    if (nums[i] + nums[j] == target) {
                        arr[m] = i;
                        arr[m + 1] = j;
                    }
                }
            }
        }
        return arr;
    }
}
