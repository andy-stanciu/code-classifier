class Solution {

    public int[] twoSum(int[] a, int t) {
        int n = a.length - 1;
        int[] ans = new int[2];
        HashMap<Integer, Integer> h = new HashMap<>();
        for (int i = 0; i <= n; i++) {
            int k = t - a[i];
            if (h.containsKey(k)) {
                ans[0] = i;
                ans[1] = h.get(k);
                break;
            }
            h.put(a[i], i);
        }
        return ans;
    }
}
