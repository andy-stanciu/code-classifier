class Solution {

    public int[] twoSum(int[] nums, int target) {
        var numToIdxMap = new HashMap<Integer, Integer>();
        for (int currentNumberIdx = 0; currentNumberIdx < nums.length; currentNumberIdx++) {
            var currentNumber = nums[currentNumberIdx];
            var possibleSecondNumberIdx = numToIdxMap.get(target - currentNumber);
            if (possibleSecondNumberIdx != null) {
                return new int[] { currentNumberIdx, possibleSecondNumberIdx };
            }
            numToIdxMap.put(currentNumber, currentNumberIdx);
        }
        return new int[0];
    }
}
