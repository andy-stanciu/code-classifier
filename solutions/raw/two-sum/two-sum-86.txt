class Solution {

    int ch = 0;

    boolean treag = false;

    int[] mas = new int[2];

    public int[] twoSum(int[] nums, int target) {
        while (treag != true) {
            for (int g = 1; g < nums.length; g++) {
                if ((nums[ch] + nums[g] == target) && (ch != g)) {
                    mas[0] = ch;
                    mas[1] = g;
                    treag = true;
                }
            }
            ch++;
        }
        return mas;
    }
}
