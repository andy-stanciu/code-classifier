class Solution {

    public int[] twoSum(int[] nums, int target) {
        int i = 0, j = 0, len = nums.length;
        int[] arr = new int[2];
        for (i = 0; i < len - 1; i++) {
            for (j = i + 1; j < len; j++) {
                if (nums[i] + nums[j] == target) {
                    arr[0] = i;
                    arr[1] = j;
                    return arr;
                }
            }
        }
        return arr;
    }
}
