class Solution {

    public int[] twoSum(int[] nums, int target) {
        for (int row = 0; row < nums.length; row++) {
            int value = nums[row];
            int sum = 0;
            for (int col = row + 1; col < nums.length; col++) {
                sum = value + nums[col];
                if (target == sum) {
                    return new int[] { row, col };
                }
            }
        }
        return new int[] { 0 };
    }
}
