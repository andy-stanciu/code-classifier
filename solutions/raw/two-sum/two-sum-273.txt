class Solution {

    public int[] twoSum(int[] nums, int target) {
        for (int i = 0; i < nums.length; i++) {
            //it takes the first starting numbers
            int num1 = nums[i];
            for (int j = i + 1; j < nums.length; j++) {
                // it takes the second number
                int num2 = nums[j];
                // add first number and second number
                int sum = num1 + num2;
                if (//check if the target meets the sum
                target == sum) {
                    //if yes then printing the index number
                    return new int[] { i, j };
                }
            }
        }
        return null;
    }
}
