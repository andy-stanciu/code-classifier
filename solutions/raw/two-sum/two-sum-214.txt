class Solution {

    public int[] twoSum(int[] nums, int target) {
        int[] arr = nums;
        int n = arr.length;
        int[] res = new int[2];
        outerloop: for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (j == i)
                    continue;
                if (arr[i] + arr[j] == target) {
                    res[0] = i;
                    res[1] = j;
                    break outerloop;
                }
            }
        }
        return res;
    }
}
