class Solution {

    public int[] twoSum(int[] nums, int target) {
        for (int i = 0; i < nums.length; i++) {
            for (int x = i + 1; x < nums.length; x++) {
                if (nums[i] + nums[x] == target) {
                    int[] donzos = { i, x };
                    return donzos;
                }
            }
        }
        int[] baddy = { 0, 0 };
        return baddy;
    }
}
