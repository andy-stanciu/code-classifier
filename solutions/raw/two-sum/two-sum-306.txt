class Solution {

    public int[] twoSum(int[] arr, int target) {
        ArrayList<Integer> arr2 = new ArrayList<>();
        for (int i = 0; i < arr.length - 1; i++) {
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[i] + arr[j] == target) {
                    arr2.add(i);
                    arr2.add(j);
                    break;
                }
            }
        }
        return arr2.stream().mapToInt(i -> i).toArray();
    }
}
