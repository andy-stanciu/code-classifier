class Solution {

    public int[] twoSum(int[] nums, int target) {
        for (int i = 1; i < nums.length; i++) {
            for (int l = i; l < nums.length; l++) {
                if (nums[l] + nums[l - i] == target) {
                    return new int[] { l, l - i };
                }
            }
        }
        return null;
    }
}
