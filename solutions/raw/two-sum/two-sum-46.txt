class Solution {

    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int number = nums[i];
            int remaining = target - nums[i];
            if (hm.containsKey(remaining)) {
                return new int[] { i, hm.get(remaining) };
            }
            hm.put(number, i);
        }
        return new int[] {};
    }
}
