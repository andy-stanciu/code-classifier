class Solution {

    public int[] twoSum(int[] nums, int target) {
        HashMap<Integer, Integer> numsAlreadyShowedUp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int indexOfCompliment = numsAlreadyShowedUp.getOrDefault(target - nums[i], -1);
            if (indexOfCompliment >= 0) {
                return new int[] { indexOfCompliment, i };
            }
            numsAlreadyShowedUp.put(nums[i], i);
        }
        return new int[] { 0, 0 };
    }
}
