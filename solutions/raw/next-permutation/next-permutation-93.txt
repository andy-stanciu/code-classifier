class Solution {

    public void nextPermutation(int[] nums) {
        int n = nums.length;
        int i = n - 2, temp = 0;
        while (i >= 0) {
            if (nums[i] < nums[i + 1])
                break;
            i--;
        }
        if (i < 0) {
            int l = 0, r = n - 1;
            while (l < r) Collections.swap(nums, l, r);
        } else {
            int j = n - 1;
            while (nums[i] >= nums[j]) j--;
            Collections.swap(nums, i, j);
            int l = i + 1, r = n - 1;
            while (l < r) {
                Collections.swap(nums, l, r);
                l++;
                r--;
            }
        }
    }
}
