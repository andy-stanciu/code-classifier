class Solution {

    public void nextPermutation(int[] nums) {
        int b = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                b = i;
                break;
            }
        }
        if (b == -1) {
            reverseArray(nums, 0, nums.length - 1);
            return;
        }
        for (int i = nums.length - 1; i > b; i--) {
            if (nums[i] > nums[b]) {
                int temp = nums[i];
                nums[i] = nums[b];
                nums[b] = temp;
                reverseArray(nums, b + 1, nums.length - 1);
                return;
            }
        }
    }

    public static void reverseArray(int[] arr, int start, int end) {
        while (start < end) {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }
}
