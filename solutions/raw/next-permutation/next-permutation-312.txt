class Solution {

    public void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public void printArr(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + " ");
        }
    }

    public void reverse(int[] nums, int start) {
        int i = start;
        int j = nums.length - 1;
        while (i < j) {
            swap(nums, i, j);
            i++;
            j--;
        }
    }

    public void nextPermutation(int[] nums) {
        int j = nums.length - 1, i = j - 1;
        int swapc = 0;
        int mini = Integer.MAX_VALUE, minindex = -1;
        while (j > 0) {
            if (nums[i] < nums[j]) {
                swapc++;
                break;
            }
            j--;
            i--;
        }
        if (swapc == 0) {
            reverse(nums, 0);
            return;
        }
        j = nums.length - 1;
        while (j >= 0) {
            if (nums[j] > nums[i]) {
                swap(nums, i, j);
                break;
            }
            j--;
        }
        reverse(nums, i + 1);
    }
}
