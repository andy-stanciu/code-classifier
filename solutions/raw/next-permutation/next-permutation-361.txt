class Solution {

    public void nextPermutation(int[] nums) {
        int len = nums.length;
        int temp;
        int sorted_point = -1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] < nums[i + 1])
                sorted_point = i;
        }
        if (//sort
        sorted_point == -1) {
            Arrays.sort(nums);
        } else //just find the minimum largest element than the sorted_point
        {
            int swapper = Integer.MAX_VALUE;
            int idx = -1;
            for (int i = sorted_point; i < len; i++) {
                if (nums[i] > nums[sorted_point]) {
                    if (nums[i] < swapper) {
                        swapper = nums[i];
                        idx = i;
                    }
                }
            }
            //replace sorted point with the min largest element after it
            temp = nums[idx];
            nums[idx] = nums[sorted_point];
            nums[sorted_point] = temp;
            // sort all elements after sorted_point
            Arrays.sort(nums, sorted_point + 1, len);
        }
    }
}
