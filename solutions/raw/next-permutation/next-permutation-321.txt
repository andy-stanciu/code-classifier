class Solution {

    public void findminreplace(int[] a, int i) {
        int min = Integer.MAX_VALUE;
        int t = a[i];
        int ans = 0;
        for (int j = i + 1; j < a.length; j++) {
            if (t < a[j]) {
                if (a[j] < min) {
                    ans = j;
                }
            }
        }
        swap(a, i, ans);
    }

    public void swap(int[] a, int i, int j) {
        int t = 0;
        t = a[i];
        a[i] = a[j];
        a[j] = t;
    }

    public void sort(int[] a, int k) {
        int i = k;
        int j = a.length - 1;
        int min = 0;
        int t = 0;
        while (i < j) {
            swap(a, i, j);
            i++;
            j--;
        }
    }

    public void nextPermutation(int[] a) {
        int min = Integer.MIN_VALUE;
        int k = 0;
        for (int i = a.length - 2; i >= -1; i--) {
            if (i >= 0 && a[i] < a[i + 1]) {
                findminreplace(a, i);
                k = i + 1;
                sort(a, k);
                break;
            }
            if (i == -1) {
                sort(a, 0);
            }
        }
    }
}
