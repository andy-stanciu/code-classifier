class Solution {

    public void nextPermutation(int[] a) {
        int less = -1;
        for (int i = a.length - 1; i >= 1; i--) {
            if (a[i - 1] < a[i]) {
                less = i - 1;
                break;
            }
        }
        int greater = less;
        if (less != -1) {
            for (int i = a.length - 1; i >= less + 1; i--) {
                if (a[less] < a[i]) {
                    greater = i;
                    break;
                }
            }
            int temp = a[less];
            a[less] = a[greater];
            a[greater] = temp;
        }
        int l = less + 1, r = a.length - 1;
        while (l < r) {
            int temp = a[l];
            a[l] = a[r];
            a[r] = temp;
            l++;
            r--;
        }
    }
}
