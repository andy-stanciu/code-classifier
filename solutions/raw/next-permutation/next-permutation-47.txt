class Solution {

    void reverse(int[] nums, int i, int j) {
        while (i < j) {
            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
            i++;
            j--;
        }
    }

    public void nextPermutation(int[] nums) {
        int n = nums.length;
        int i1 = -1, i2 = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                i1 = i;
                break;
            }
        }
        if (i1 == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        for (int i = n - 1; i > i1; i--) {
            if (nums[i] > nums[i1]) {
                i2 = i;
                break;
            }
        }
        int temp = nums[i1];
        nums[i1] = nums[i2];
        nums[i2] = temp;
        reverse(nums, i1 + 1, n - 1);
    }
}
