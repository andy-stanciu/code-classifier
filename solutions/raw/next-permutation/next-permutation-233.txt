class Solution {

    public void nextPermutation(int[] nums) {
        int ind = -1;
        // Find the break point
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        // If the break point is not there reverse the entire array and return
        if (ind == -1) {
            reverseArr(nums, 0, nums.length - 1);
            return;
        }
        // Find the samllest ele larger than arr[ind] ans swap with nums[i]
        for (int i = nums.length - 1; i > ind; i--) {
            if (nums[i] > nums[ind]) {
                swapEle(nums, i, ind);
                break;
            }
        }
        // Reverse the subarray after ind to get the next permutation
        reverseArr(nums, ind + 1, nums.length - 1);
    }

    static void reverseArr(int[] arr, int start, int end) {
        while (start < end) {
            swapEle(arr, start, end);
            start++;
            end--;
        }
    }

    static void swapEle(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
