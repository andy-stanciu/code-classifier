class Solution {

    public void nextPermutation(int[] nums) {
        int ind = -1;
        // Step 1: Find the first decreasing element from the right
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        // Step 2: Handle the case where no decreasing element is found
        if (ind == -1) {
            Arrays.sort(nums);
        } else {
            // Step 3: Find the smallest element greater than the decreasing element
            for (int i = nums.length - 1; i > ind; i--) {
                if (nums[i] > nums[ind]) {
                    int temp = nums[i];
                    nums[i] = nums[ind];
                    nums[ind] = temp;
                    break;
                }
            }
            // Step 4: Reverse the remaining part of the array
            int l = ind + 1;
            int r = nums.length - 1;
            while (l < r) {
                int temp = nums[l];
                nums[l] = nums[r];
                nums[r] = temp;
                l++;
                r--;
            }
        }
    }
}
