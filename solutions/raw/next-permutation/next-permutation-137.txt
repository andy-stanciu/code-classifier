class Solution {

    public void nextPermutation(int[] arr) {
        int index = -1;
        int n = arr.length;
        //find the pivot point where arr[i]<arr[i+1],(disrupts the increasing order)
        for (int i = n - 2; i >= 0; i--) {
            if (arr[i] < arr[i + 1]) {
                index = i;
                break;
            }
        }
        // if no pivot is found, then the entire array must be reversed
        if (index == -1) {
            reverse(arr, 0, n - 1);
            return;
        }
        //if pivot is found , swap it with the immediate greater element by traversing from the last
        for (int i = n - 1; i > index; i--) {
            if (arr[index] < arr[i]) {
                int temp = arr[index];
                arr[index] = arr[i];
                arr[i] = temp;
                break;
            }
        }
        //reverse from pivot+1 to the end
        reverse(arr, index + 1, n - 1);
    }

    public void reverse(int[] arr, int start, int end) {
        while (start < end) {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }
}
