class Solution {

    public void nextPermutation(int[] nums) {
        int breakPt = -1;
        int nxtLarge = -1;
        for (// step 1 --> Find breaking point
        // step 1 --> Find breaking point
        int i = nums.length - 2; // step 1 --> Find breaking point
        i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                //Found break Point, hence break it here.
                breakPt = i;
                break;
            }
        }
        if (// If there is no breaking  point
        breakPt == -1) {
            reverse(nums, 0);
        } else //If there is breaking point
        {
            // step 2 --> Find next greater element & swap with nxtLarge
            for (int i = nums.length - 1; i >= 0; i--) {
                if (nums[i] > nums[breakPt]) {
                    nxtLarge = i;
                    break;
                }
            }
            swap(nums, breakPt, nxtLarge);
            // step 3 reverse the rest right half
            reverse(nums, breakPt + 1);
        }
    }

    void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    void reverse(int[] nums, int start) {
        int i = start;
        int j = nums.length - 1;
        while (i < j) {
            swap(nums, i, j);
            i++;
            j--;
        }
    }
}
