class Solution {

    public void nextPermutation(int[] nums) {
        int ind = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            int x = 0;
            int y = nums.length - 1;
            while (x < y) {
                int tem = nums[y];
                nums[y] = nums[x];
                nums[x] = tem;
                x++;
                y--;
            }
        } else {
            for (int i = nums.length - 1; i >= ind; i--) {
                if (nums[i] > nums[ind]) {
                    int temp = nums[ind];
                    nums[ind] = nums[i];
                    nums[i] = temp;
                    break;
                }
            }
            int x = ind + 1;
            int y = nums.length - 1;
            while (x < y) {
                int tem = nums[y];
                nums[y] = nums[x];
                nums[x] = tem;
                x++;
                y--;
            }
        }
    }
}
