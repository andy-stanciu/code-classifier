class Solution {

    public void nextPermutation(int[] nums) {
        int pivot = nums.length - 2;
        while (pivot >= 0 && nums[pivot] >= nums[pivot + 1]) {
            pivot--;
        }
        if (pivot >= 0) {
            int successor = nums.length - 1;
            while (successor >= 0 && nums[pivot] >= nums[successor]) {
                successor--;
            }
            //swap pivot and successor values
            int temp = nums[successor];
            nums[successor] = nums[pivot];
            nums[pivot] = temp;
        }
        //reverse array after pivot
        int i = pivot + 1;
        int j = nums.length - 1;
        while (i < j) {
            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
            i++;
            j--;
        }
    }
}
