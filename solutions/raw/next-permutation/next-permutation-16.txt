class Solution {

    public void nextPermutation(int[] nums) {
        int index = -1;
        int n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                index = i;
                break;
            }
        }
        if (index == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        // next index which is greater than index
        for (int i = n - 1; i >= index; i--) {
            if (nums[index] < nums[i]) {
                swap(nums, index, i);
                break;
            }
        }
        reverse(nums, index + 1, n - 1);
    }

    public static void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public static void reverse(int[] arr, int left, int right) {
        while (left < right) {
            // Continue until the left index is less than the right index
            // Swap elements at left and right
            swap(arr, left, right);
            // Move left index to the right
            left++;
            // Move right index to the left
            right--;
        }
    }
}
