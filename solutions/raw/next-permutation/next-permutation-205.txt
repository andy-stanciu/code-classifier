class Solution {

    public void nextPermutation(int[] nums) {
        int n = nums.length, curr = n - 2;
        if (n == 1)
            return;
        if (nums[n - 1] > nums[n - 2]) {
            swap(nums, n - 1, n - 2);
            return;
        }
        while (curr >= 0 && nums[curr] >= nums[curr + 1]) {
            curr--;
        }
        if (curr == -1) {
            Arrays.sort(nums);
        } else {
            reverse(nums, curr + 1);
            int nextBig = curr + 1;
            while (nextBig < n && nums[nextBig] <= nums[curr]) nextBig++;
            if (nextBig < n)
                swap(nums, curr, nextBig);
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[j];
        nums[j] = nums[i];
        nums[i] = temp;
    }

    private void reverse(int[] nums, int i) {
        int n = nums.length - 1, last = n;
        while (i < last) {
            swap(nums, i, last);
            last--;
            i++;
        }
    }
}
