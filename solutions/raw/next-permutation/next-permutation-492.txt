class Solution {

    public static void reverse(int[] arr, int i, int j) {
        if (arr == null || i < 0 || j >= arr.length || i >= j) {
            return;
        }
        // Reverse the array segment between indices i and j
        while (i < j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
    }

    public void nextPermutation(int[] arr) {
        int n = arr.length;
        int index = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (arr[i] < arr[i + 1]) {
                index = i;
                break;
            }
        }
        if (index == -1) {
            reverse(arr, 0, n - 1);
        } else {
            for (int i = n - 1; i >= index; i--) {
                if (arr[i] > arr[index]) {
                    int temp = arr[i];
                    arr[i] = arr[index];
                    arr[index] = temp;
                    reverse(arr, index + 1, n - 1);
                    break;
                }
            }
        }
    }
}
