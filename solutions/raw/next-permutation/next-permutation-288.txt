class Solution {

    public void nextPermutation(int[] nums) {
        int n = nums.length;
        int max = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                max = i;
                break;
            }
        }
        if (max > -1) {
            for (int i = n - 1; i >= max; i--) {
                if (nums[max] < nums[i]) {
                    int temp = nums[i];
                    nums[i] = nums[max];
                    nums[max] = temp;
                    break;
                }
            }
            int p1 = max + 1;
            int p2 = n - 1;
            while (p1 < p2) {
                int temp = nums[p1];
                nums[p1] = nums[p2];
                nums[p2] = temp;
                p1++;
                p2--;
            }
        } else {
            int p1 = 0;
            int p2 = n - 1;
            while (p1 < p2) {
                int temp = nums[p1];
                nums[p1] = nums[p2];
                nums[p2] = temp;
                p1++;
                p2--;
            }
        }
    }
}
