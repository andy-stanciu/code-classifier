class Solution {

    public void nextPermutation(int[] nums) {
        int lastIndex = nums.length - 1;
        // if no number less than its right then there is no next perm
        //for example: [9,8,7,6,5,4,3,2,1] is the largest permutation
        //start on the element before the last
        int i = lastIndex - 1;
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            i--;
        }
        //we found a number that is less than its right so there is next permutation
        int j = lastIndex;
        if (i >= 0) {
            while (j >= 0 && nums[i] >= nums[j]) {
                j--;
            }
            swap(nums, i, j);
        }
        reverse(nums, i + 1, lastIndex);
    }

    public void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }
}
