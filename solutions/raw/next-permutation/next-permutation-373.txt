class Solution {

    public void nextPermutation(int[] nums) {
        int tIndex = -1;
        int temp = -1;
        for (int i = nums.length - 1; i > -1; i--) {
            if (nums[i] < temp) {
                tIndex = i;
                break;
            }
            temp = Math.max(temp, nums[i]);
        }
        if (tIndex == -1) {
            Arrays.sort(nums);
            return;
        }
        temp = 101;
        int minIndex = -1;
        for (int j = tIndex + 1; j < nums.length; j++) {
            if (nums[tIndex] < nums[j] && temp > nums[j]) {
                temp = nums[j];
                minIndex = j;
            }
        }
        temp = nums[minIndex];
        nums[minIndex] = nums[tIndex];
        nums[tIndex] = temp;
        Arrays.sort(nums, tIndex + 1, nums.length);
    }
}
