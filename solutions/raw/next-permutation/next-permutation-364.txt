class Solution {

    public void nextPermutation(int[] nums) {
        int v = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                v = i;
                break;
            }
        }
        if (v == -1) {
            reverse(nums, 0, nums.length - 1);
        } else {
            for (int i = nums.length - 1; i >= v; i--) {
                if (nums[v] < nums[i]) {
                    int temp = nums[v];
                    nums[v] = nums[i];
                    nums[i] = temp;
                    break;
                }
            }
            reverse(nums, v + 1, nums.length - 1);
        }
    }

    public void reverse(int[] nums, int l, int r) {
        while (l <= r) {
            int temp = nums[l];
            nums[l] = nums[r];
            nums[r] = temp;
            l++;
            r--;
        }
    }
}
