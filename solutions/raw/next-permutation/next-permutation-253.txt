class Solution {

    public void nextPermutation(int[] nums) {
        int idx = -1;
        int n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                idx = i;
                break;
            }
        }
        if (idx == -1) {
            rev(nums, 0, n - 1);
            return;
        }
        for (int i = n - 1; i >= 0; i--) {
            if (nums[i] > nums[idx]) {
                int temp = nums[i];
                nums[i] = nums[idx];
                nums[idx] = temp;
                break;
            }
        }
        rev(nums, idx + 1, n - 1);
    }

    public static int[] rev(int[] nums, int r, int l) {
        while (l > r) {
            int temp = nums[l];
            nums[l] = nums[r];
            nums[r] = temp;
            l--;
            r++;
        }
        return nums;
    }
}
