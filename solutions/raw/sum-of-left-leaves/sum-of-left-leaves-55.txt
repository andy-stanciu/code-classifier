class Solution {

    public int sum(TreeNode root, int f) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            if (f == 1)
                return root.val;
            else
                return 0;
        }
        int a = sum(root.left, 1);
        int b = sum(root.right, 0);
        return a + b;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        return sum(root, 0);
    }
}
