class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return helper(root, "");
    }

    private int helper(TreeNode node, String direction) {
        if (node.left == null && node.right == null && direction == "left")
            return node.val;
        int sum = 0;
        if (node.left != null)
            sum += helper(node.left, "left");
        if (node.right != null)
            sum += helper(node.right, "right");
        return sum;
    }
}
