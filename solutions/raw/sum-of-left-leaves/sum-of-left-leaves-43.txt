class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        return ifLeafGetValueOtherwiseZero(root.left) + sumOfLeftLeaves(root.left) + sumOfLeftLeaves(root.right);
    }

    private int ifLeafGetValueOtherwiseZero(TreeNode root) {
        return root != null && root.left == null && root.right == null ? root.val : 0;
    }
}
