/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int rec(TreeNode root, int n) {
        if (root.left == null && root.right == null && n == 0)
            return root.val;
        int left = 0, right = 0;
        if (root.left != null)
            left = rec(root.left, 0);
        if (root.right != null)
            right = rec(root.right, 1);
        return left + right;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        return rec(root, -1);
    }
}
