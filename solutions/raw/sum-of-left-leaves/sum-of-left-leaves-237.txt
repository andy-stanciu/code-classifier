class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return dfs(root, false);
    }

    public int dfs(TreeNode root, boolean isLeft) {
        if (root == null) {
            return 0;
        }
        int sum = 0;
        if (root.left == null && root.right == null && isLeft) {
            sum += root.val;
        }
        return sum + dfs(root.left, true) + dfs(root.right, false);
    }
}
