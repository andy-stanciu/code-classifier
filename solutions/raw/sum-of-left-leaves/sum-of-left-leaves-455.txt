class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        sumtree(root);
        return sum;
    }

    public void sumtree(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            sum += root.left.val;
        else
            sumtree(root.left);
        sumtree(root.right);
    }
}
