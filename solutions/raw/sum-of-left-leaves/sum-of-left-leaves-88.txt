/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    // reason to class level variable - In Java, when you pass an int variable (like
    // sum) to a method, the method gets a copy of that variable\u2019s value
    private int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        dfs(root, false);
        return sum;
    }

    public void dfs(TreeNode node, boolean isLeft) {
        // leave node
        if (node.left == null && node.right == null && isLeft) {
            sum += node.val;
        }
        // left//
        if (node.left != null)
            dfs(node.left, true);
        // right
        if (node.right != null)
            dfs(node.right, false);
    }
}
