/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int sum;

    public static void leftLeftSum(TreeNode root, int flag) {
        if (root == null)
            return;
        ;
        if (flag == 1 && root.left == null && root.right == null) {
            sum += root.val;
        }
        leftLeftSum(root.left, 1);
        leftLeftSum(root.right, 0);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        sum = 0;
        leftLeftSum(root, 0);
        return sum;
    }
}
