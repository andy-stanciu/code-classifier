/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //-1 left
    // 1 right
    int sum = 0;

    public void solve(TreeNode root, int flag) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && flag == -1) {
            sum += root.val;
        }
        solve(root.left, -1);
        solve(root.right, 1);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        solve(root, 0);
        return sum;
    }
}
