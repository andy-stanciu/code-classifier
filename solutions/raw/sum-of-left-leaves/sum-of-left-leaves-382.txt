class Solution {

    int res = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        dfs(root, "");
        return res;
    }

    void dfs(TreeNode node, String dir) {
        if (node == null) {
            return;
        }
        if (node.left == null && node.right == null) {
            if (dir == "left")
                res += node.val;
            return;
        }
        dfs(node.left, "left");
        dfs(node.right, "right");
    }
}
