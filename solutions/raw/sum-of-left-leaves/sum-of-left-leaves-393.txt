/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int res = 0;
        if (root.left != null)
            res += sum(root.left, true);
        if (root.right != null)
            res += sum(root.right, false);
        return res;
    }

    public int sum(TreeNode rt, boolean left) {
        if (rt.left == null && rt.right == null && left) {
            return rt.val;
        } else if (rt.left == null && rt.right == null && !left)
            return 0;
        int res = 0;
        if (rt.left != null)
            res += sum(rt.left, true);
        if (rt.right != null)
            res += sum(rt.right, false);
        return res;
    }
}
