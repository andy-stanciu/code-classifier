/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sumLeft;

    public int sumOfLeftLeaves(TreeNode root) {
        sumLeft = 0;
        if (root == null) {
            return 0;
        }
        sum(root.left, root.right);
        return sumLeft;
    }

    public void sum(TreeNode left, TreeNode right) {
        if (left != null && left.left == null && left.right == null) {
            sumLeft += left.val;
        }
        if (left != null)
            sum(left.left, left.right);
        if (right != null)
            sum(right.left, right.right);
    }
}
