/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    int sum = 0;

    public void addLeftLeaves(TreeNode root, boolean isLeftNode) {
        if (root.left == null && root.right == null && isLeftNode)
            sum += root.val;
        if (root.left != null)
            addLeftLeaves(root.left, true);
        if (root.right != null)
            addLeftLeaves(root.right, false);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        addLeftLeaves(root, false);
        return sum;
    }
}
