/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int getSum(TreeNode node, String type) {
        if (node == null)
            return 0;
        int sum = 0;
        if ((node.left == null && node.right == null) && type == "left")
            return node.val;
        return getSum(node.left, "left") + getSum(node.right, "right");
    }

    public int sumOfLeftLeaves(TreeNode root) {
        return getSum(root, "root");
    }
}
