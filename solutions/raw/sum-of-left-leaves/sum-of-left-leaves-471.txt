/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        calculate(root, false);
        return sum;
    }

    public void calculate(TreeNode t, boolean flag) {
        if (t == null) {
            return;
        }
        if (flag && (t.right == null && t.left == null)) {
            sum = sum + t.val;
        }
        calculate(t.left, true);
        calculate(t.right, false);
    }
}
