class Solution {

    private int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null) {
            return 0;
        }
        traverse(root, false);
        return this.sum;
    }

    void traverse(TreeNode root, boolean isLeft) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && isLeft) {
            this.sum += root.val;
            return;
        }
        traverse(root.left, true);
        traverse(root.right, false);
    }
}
