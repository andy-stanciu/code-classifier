class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return rec(false, root);
    }

    int rec(boolean left, TreeNode root) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && left) {
            return root.val;
        }
        return rec(true, root.left) + rec(false, root.right);
    }
}
