class Solution {

    int sum;

    public void summ(TreeNode root, int a) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && a == 1)
            sum += root.val;
        summ(root.left, 1);
        summ(root.right, 0);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        sum = 0;
        summ(root, -1);
        return sum;
    }
}
