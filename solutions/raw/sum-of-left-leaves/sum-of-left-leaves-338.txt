class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return sumOfLeftLeaves(root, false);
    }

    //same method, but we check if we are on a left child using a boolean
    public int sumOfLeftLeaves(TreeNode root, boolean left) {
        if (root == null)
            return 0;
        //if its a leaf and a left child, we add root val
        if (left == true && root.left == null && root.right == null)
            return root.val;
        //do the same to all left and right nodes
        return sumOfLeftLeaves(root.left, true) + sumOfLeftLeaves(root.right, false);
    }
}
