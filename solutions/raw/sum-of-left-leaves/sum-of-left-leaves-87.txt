/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int suml(TreeNode root, TreeNode pre) {
        if (root == null || pre == null)
            return 0;
        if (pre.left == root && (root.left == null && root.right == null))
            return root.val;
        int left = suml(root.left, root);
        int right = suml(root.right, root);
        return left + right;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        int sum = suml(root, root);
        return sum;
    }
}
