/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sum;

    public int sumOfLeftLeaves(TreeNode root) {
        sum = 0;
        return sumOfLeaves(root, false);
    }

    private int sumOfLeaves(TreeNode root, boolean left) {
        if (root == null) {
            return sum;
        }
        if (left && root.left == null && root.right == null) {
            sum += root.val;
        }
        sumOfLeaves(root.left, true);
        sumOfLeaves(root.right, false);
        return sum;
    }
}
