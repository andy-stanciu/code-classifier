/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void Magic(int[] res, TreeNode root) {
        if (root == null)
            return;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                res[0] += root.left.val;
        }
        Magic(res, root.left);
        Magic(res, root.right);
        return;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        int[] res = { 0 };
        Magic(res, root);
        return res[0];
    }
}
