class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return sumLeftLeaves(root, false);
    }

    public int sumLeftLeaves(TreeNode root, Boolean isleft) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && isleft == true)
            return root.val;
        if (root.left == null && root.right == null && isleft == false)
            return 0;
        return sumLeftLeaves(root.left, true) + sumLeftLeaves(root.right, false);
    }
}
