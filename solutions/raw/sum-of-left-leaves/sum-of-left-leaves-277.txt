class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        int res = 0;
        if (root.left != null) {
            res += root.left.right == null && root.left.left == null ? root.left.val : 0;
            res += sumOfLeftLeaves(root.left);
        }
        if (root.right != null)
            res += sumOfLeftLeaves(root.right);
        return res;
    }
}
