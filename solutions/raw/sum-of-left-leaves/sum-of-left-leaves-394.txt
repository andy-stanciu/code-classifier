/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int dfs(TreeNode node, Boolean left) {
        int sum = 0;
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null && left) {
            return node.val;
        }
        if (node.left != null) {
            sum += dfs(node.left, true);
        }
        if (node.right != null) {
            sum += dfs(node.right, false);
        }
        return sum;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        return dfs(root, false);
    }
}
