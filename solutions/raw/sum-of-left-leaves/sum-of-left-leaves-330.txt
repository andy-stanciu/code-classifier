/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        findleaf(root, list, false);
        int sum = 0;
        for (int num : list) {
            sum += num;
        }
        return sum;
    }

    public TreeNode findleaf(TreeNode root, List<Integer> list, boolean val) {
        if (root == null) {
            return null;
        }
        TreeNode left = findleaf(root.left, list, true);
        TreeNode right = findleaf(root.right, list, false);
        if (left == null && right == null && val == true) {
            list.add(root.val);
        }
        return root;
    }
}
