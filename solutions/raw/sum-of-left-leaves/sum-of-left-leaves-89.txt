/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] sum = new int[1];
        sumOfLeftLeaves(root.left, 0, sum);
        sumOfLeftLeaves(root.right, 1, sum);
        return sum[0];
    }

    private void sumOfLeftLeaves(TreeNode root, int index, int[] sum) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && index == 0) {
            sum[0] += root.val;
        }
        sumOfLeftLeaves(root.left, 0, sum);
        sumOfLeftLeaves(root.right, 1, sum);
    }
}
