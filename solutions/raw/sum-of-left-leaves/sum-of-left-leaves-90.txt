/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int[] leafSum = new int[1];
        preOrderBinaryTree(root, leafSum);
        return leafSum[0];
    }

    public static void preOrderBinaryTree(TreeNode root, int[] leafSum) {
        if (root == null) {
            return;
        }
        if (root.left != null && (root.left.left == null && root.left.right == null)) {
            leafSum[0] += root.left.val;
        }
        preOrderBinaryTree(root.left, leafSum);
        preOrderBinaryTree(root.right, leafSum);
    }
}
