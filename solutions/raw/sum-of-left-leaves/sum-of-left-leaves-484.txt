/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        sum = 0;
        summer(root, -1);
        return sum;
    }

    public static void summer(TreeNode root, int v) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && v == 0) {
            sum = sum + root.val;
            return;
        }
        summer(root.left, 0);
        summer(root.right, 1);
    }
}
