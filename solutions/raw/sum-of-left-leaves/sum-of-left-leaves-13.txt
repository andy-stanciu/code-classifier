class Solution {

    public static int sum;

    public static int mdepth;

    static void findleft(TreeNode root, int depth) {
        if (root == null)
            return;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                sum += root.left.val;
            }
        }
        findleft(root.left, depth + 1);
        findleft(root.right, depth + 1);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        sum = 0;
        mdepth = 0;
        findleft(root, 0);
        return sum;
    }
}
