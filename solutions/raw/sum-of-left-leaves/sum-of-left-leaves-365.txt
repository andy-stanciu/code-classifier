class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return getLeftWeight(root, false);
    }

    public static int getLeftWeight(TreeNode node, boolean left) {
        if (node.right == null && node.left == null) {
            if (left)
                return node.val;
            else
                return 0;
        } else {
            int leftWeight = (node.left != null) ? getLeftWeight(node.left, true) : 0;
            int rightWeight = (node.right != null) ? getLeftWeight(node.right, false) : 0;
            return leftWeight + rightWeight;
        }
    }
}
