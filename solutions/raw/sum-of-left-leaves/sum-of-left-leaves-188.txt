class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int[] arr = { 0 };
        computeSum(root, arr);
        return arr[0];
    }

    private void computeSum(TreeNode root, int[] arr) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            arr[0] += root.left.val;
        computeSum(root.left, arr);
        computeSum(root.right, arr);
    }
}
