class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return findsum(root);
    }

    public int findsum(TreeNode root) {
        int sum = 0;
        if (root == null) {
            return sum;
        }
        if (root.left != null && (root.left.right == null && root.left.left == null)) {
            sum += root.left.val;
        }
        sum += findsum(root.right);
        sum += findsum(root.left);
        return sum;
    }
}
