class Solution {

    public int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        if (root.left == null && root.right == null) {
            return 0;
        }
        sumLeaves(root);
        return sum;
    }

    private void sumLeaves(TreeNode node) {
        if (node == null) {
            return;
        } else if (node.left == null && node.right == null) {
            return;
        }
        if (node.left != null) {
            if (node.left.left == null && node.left.right == null) {
                sum += node.left.val;
            }
        }
        sumLeaves(node.left);
        sumLeaves(node.right);
    }
}
