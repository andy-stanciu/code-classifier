class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        else
            return sum(null, root);
    }

    public int sum(TreeNode prev, TreeNode trav) {
        if (trav == null)
            return 0;
        if (trav.left == null && trav.right == null && prev != null && prev.left == trav)
            return trav.val;
        if (trav.left != null && trav.right != null)
            return sum(trav, trav.left) + sum(trav, trav.right);
        else if (trav.left != null)
            return sum(trav, trav.left);
        else
            //(trav.right!=null)
            return sum(trav, trav.right);
    }
}
