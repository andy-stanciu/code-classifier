/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int summ(TreeNode root, boolean left) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            if (left) {
                return root.val;
            }
            // System.out.println("r "+ root.val + " ans " + ans);
            return 0;
        }
        int a = summ(root.left, true);
        int b = summ(root.right, false);
        return a + b;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        int a = summ(root, false);
        return a;
    }
}
