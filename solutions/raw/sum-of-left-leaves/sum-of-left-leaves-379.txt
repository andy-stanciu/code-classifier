/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        help(root, 0);
        return sum;
    }

    void help(TreeNode root, int f) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && f == 1) {
            sum += root.val;
            return;
        }
        help(root.left, 1);
        help(root.right, 0);
    }
}
