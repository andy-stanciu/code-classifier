/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        return helper(root, false);
    }

    // THIS method will help to find left leaves and do addition for only those. based on boolean isleftside assign latest node value
    public int helper(TreeNode root, boolean isleftside) {
        if (root.left == null && root.right == null) {
            return isleftside ? root.val : 0;
        }
        int totalleaves = 0;
        if (root.left != null) {
            totalleaves = totalleaves + helper(root.left, true);
        }
        if (root.right != null) {
            totalleaves = totalleaves + helper(root.right, false);
        }
        return totalleaves;
    }
}
