/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int[] sum = new int[1];
        goLeft(root, sum, false);
        return sum[0];
    }

    public void goLeft(TreeNode root, int[] sum, boolean isLeft) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && isLeft) {
            sum[0] += root.val;
        } else {
            goLeft(root.left, sum, true);
        }
        goLeft(root.right, sum, false);
    }
}
