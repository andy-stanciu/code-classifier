/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int sum = 0;
        if (root == null) {
            return sum;
        }
        TreeNode leftLeaf = root;
        boolean flag = false;
        while (leftLeaf.left != null) {
            flag = true;
            leftLeaf = leftLeaf.left;
            if (leftLeaf.left == null) {
                while (leftLeaf.right != null) {
                    leftLeaf = leftLeaf.right;
                    if (leftLeaf.left != null) {
                        flag = true;
                        break;
                    }
                    flag = false;
                }
            }
        }
        if (flag) {
            sum += leftLeaf.val;
            leftLeaf.val = 0;
        }
        sum += sumOfLeftLeaves(root.left);
        sum += sumOfLeftLeaves(root.right);
        return sum;
    }
}
