/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sum = 0;

    public void helper(TreeNode node, boolean leftok) {
        if (node == null) {
            return;
        }
        if (leftok == true && node.left == null && node.right == null) {
            sum = sum + node.val;
        }
        helper(node.left, true);
        helper(node.right, false);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null) {
            return 0;
        }
        helper(root, false);
        return sum;
    }
}
