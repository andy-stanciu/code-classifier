class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int rightRtn = 0;
        int leftRtn = 0;
        int newLeftRtn = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                newLeftRtn = root.left.val;
            } else {
                leftRtn = sumOfLeftLeaves(root.left);
            }
        }
        if (root.right != null) {
            rightRtn = sumOfLeftLeaves(root.right);
        }
        return leftRtn + rightRtn + newLeftRtn;
    }
}
