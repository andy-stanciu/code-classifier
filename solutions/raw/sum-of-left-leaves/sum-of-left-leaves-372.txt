/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return soln(root, false);
    }

    public int soln(TreeNode node, boolean flag) {
        if (node == null) {
            return 0;
        }
        if (flag && node.left == null && node.right == null) {
            return node.val;
        }
        int left = 0, right = 0;
        left += soln(node.left, true);
        right += soln(node.right, false);
        return (left + right);
    }
}
