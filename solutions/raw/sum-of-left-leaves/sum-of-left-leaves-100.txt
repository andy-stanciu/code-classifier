/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        int sum = 0;
        Queue<TreeNode> nodes = new LinkedList<>();
        Queue<TreeNode> leftNodes = new LinkedList<>();
        nodes.add(root);
        while (!nodes.isEmpty()) {
            TreeNode currentNode = nodes.poll();
            if (currentNode.left != null) {
                nodes.add(currentNode.left);
                leftNodes.add(currentNode.left);
            }
            if (currentNode.right != null) {
                nodes.add(currentNode.right);
            }
        }
        while (!leftNodes.isEmpty()) {
            TreeNode currentLeftNode = leftNodes.poll();
            if (currentLeftNode.left == null && currentLeftNode.right == null) {
                sum += currentLeftNode.val;
            }
        }
        return sum;
    }
}
