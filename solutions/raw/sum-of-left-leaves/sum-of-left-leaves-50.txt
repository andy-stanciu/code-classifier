/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        cal(root.left);
        cal1(root.right);
        return sum;
    }

    public void cal(TreeNode root) {
        if (root == null)
            return;
        if (root.left == null && root.right == null)
            sum = sum + root.val;
        cal(root.left);
        cal1(root.right);
    }

    public void cal1(TreeNode root) {
        if (root == null)
            return;
        cal(root.left);
        cal1(root.right);
    }
}
