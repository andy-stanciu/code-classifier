/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        helper(root, -1);
        return sum;
    }

    public int helper(TreeNode root, int idx) {
        if (root == null) {
            return 0;
        }
        if (idx == 1 && root.left == null && root.right == null) {
            sum += root.val;
        }
        int left = helper(root.left, 1);
        int right = helper(root.right, 0);
        return left + right;
    }
}
