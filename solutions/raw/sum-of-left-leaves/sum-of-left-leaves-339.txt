class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        return sumOfLeft(root.left) + sumOfRight(root.right);
    }

    private int sumOfLeft(TreeNode root) {
        if (root == null)
            return 0;
        if (root.right == null && root.left == null)
            return root.val;
        //return value beacause its left leaf
        return sumOfLeft(root.left) + sumOfRight(root.right);
    }

    private int sumOfRight(TreeNode root) {
        if (root == null)
            return 0;
        //Go ahead with the dfs instead of returning node val
        return sumOfLeft(root.left) + sumOfRight(root.right);
    }
}
