/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        findSumOfLeftLeaves(root, false);
        return count;
    }

    public void findSumOfLeftLeaves(TreeNode root, boolean isLeftNode) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && isLeftNode) {
            count += root.val;
        }
        findSumOfLeftLeaves(root.left, true);
        findSumOfLeftLeaves(root.right, false);
    }
}
