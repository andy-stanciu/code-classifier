class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        sum(root, arr);
        int addition = 0;
        for (int i = 0; i < arr.size(); i++) addition += arr.get(i);
        return addition;
    }

    public void sum(TreeNode root, ArrayList<Integer> arr) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            return;
        } else {
            if (root.left != null)
                if (root.left.left == null && root.left.right == null) {
                    arr.add(root.left.val);
                }
            sum(root.left, arr);
            sum(root.right, arr);
        }
    }
}
