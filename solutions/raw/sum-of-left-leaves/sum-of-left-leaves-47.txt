/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        traverse(root.left, 1);
        traverse(root.right, 2);
        return sum;
    }

    private void traverse(TreeNode root, int leave) {
        // base condition
        if (root == null)
            return;
        traverse(root.left, 1);
        if (root.left == null && root.right == null && leave == 1)
            sum += root.val;
        traverse(root.right, 2);
    }
}
