class Solution {

    int sum = 0;

    private void find(TreeNode root, boolean isLeft) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && isLeft)
            sum += root.val;
        find(root.left, true);
        find(root.right, false);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        find(root, false);
        return sum;
    }
}
