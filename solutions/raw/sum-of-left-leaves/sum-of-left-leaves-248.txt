class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        // Recursion Approach
        // TC: O(n), SC: O(h)
        int[] ans = { 0 };
        helper(root, ans);
        return ans[0];
    }

    private void helper(TreeNode root, int[] ans) {
        if (root == null) {
            return;
        }
        // Left leaf is found
        if (root.left != null && root.left.left == null && root.left.right == null) {
            ans[0] += root.left.val;
        }
        helper(root.left, ans);
        helper(root.right, ans);
    }
}
