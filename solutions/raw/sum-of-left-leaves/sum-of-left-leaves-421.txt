/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        return helper(root, 0);
    }

    private int helper(TreeNode root, int l) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null && l == 1) {
            return root.val;
        }
        int x = helper(root.left, 1);
        int y = helper(root.right, 0);
        return x + y;
    }
}
