/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null) {
            // If the root is null, there are no left leaves, so return 0.
            return 0;
        }
        int sum = 0;
        // Check if the left child of the current node is a leaf node.
        if (root.left != null && root.left.left == null && root.left.right == null) {
            // If it\'s a leaf node, add its value to the sum.
            sum += root.left.val;
        }
        // Recursively traverse both left and right subtrees and add their left leaves.
        // Recursively traverse the left subtree.
        sum += sumOfLeftLeaves(root.left);
        // Recursively traverse the right subtree.
        sum += sumOfLeftLeaves(root.right);
        // Return the total sum of left leaves.
        return sum;
    }
}
