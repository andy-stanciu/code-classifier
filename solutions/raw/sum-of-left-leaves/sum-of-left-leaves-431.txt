class Solution {

    int s = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        help(root, false);
        return s;
    }

    public void help(TreeNode root, boolean l) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (l)
                s += root.val;
        }
        help(root.left, true);
        help(root.right, false);
    }
}
