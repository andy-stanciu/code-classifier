class Solution {

    public void solve(TreeNode node, int[] sum, boolean flag) {
        if (Objects.isNull(node))
            return;
        if (!Objects.nonNull(node.left) && !Objects.nonNull(node.right) && flag)
            sum[0] += node.val;
        solve(node.left, sum, true);
        solve(node.right, sum, false);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        int[] sum = { 0 };
        solve(root, sum, false);
        return sum[0];
    }
}
