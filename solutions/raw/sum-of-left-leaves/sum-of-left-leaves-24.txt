class Solution {

    int s = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        sumLeft(root, false);
        return s;
    }

    private void sumLeft(TreeNode root, boolean isLeft) {
        if (root == null)
            return;
        //left leaf
        if (isLeft && root.left == null && root.right == null)
            s += root.val;
        sumLeft(root.left, true);
        sumLeft(root.right, false);
    }
}
