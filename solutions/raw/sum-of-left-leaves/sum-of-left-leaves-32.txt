class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int[] sum = new int[1];
        dfs(root, sum);
        return sum[0];
    }

    public void dfs(TreeNode root, int[] sum) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            sum[0] += root.left.val;
        if (root.left != null)
            dfs(root.left, sum);
        if (root.right != null)
            dfs(root.right, sum);
    }
}
