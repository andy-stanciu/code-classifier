class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        sumHelper(root);
        if (root.left == null && root.right == null)
            return 0;
        return sum;
    }

    public void sumHelper(TreeNode root) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            sum += root.val;
            return;
        }
        sumHelper(root.left);
        if (root.right == null)
            return;
        if (root.right.left != null || root.right.right != null)
            sumHelper(root.right);
    }
}
