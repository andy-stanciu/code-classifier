/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        display(root, false);
        return sum;
    }

    public void display(TreeNode node, boolean isleft) {
        if (node == null) {
            return;
        }
        display(node.left, true);
        if (node.left == null && node.right == null && isleft) {
            sum = sum + node.val;
        }
        display(node.right, false);
    }
}
