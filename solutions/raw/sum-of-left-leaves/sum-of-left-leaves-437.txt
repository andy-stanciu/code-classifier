class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        if (root == null)
            return 0;
        int x = sumOfLeftLeaves(root.left) + sumOfLeftLeaves(root.right);
        return root.left != null && root.left.left == null && root.left.right == null ? x + root.left.val : x;
    }
}
