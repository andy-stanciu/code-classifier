class Solution {

    int sum = 0;

    void check(TreeNode root, boolean isleft) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && isleft) {
            sum += root.val;
        }
        check(root.left, true);
        check(root.right, false);
    }

    public int sumOfLeftLeaves(TreeNode root) {
        check(root, false);
        return sum;
    }
}
