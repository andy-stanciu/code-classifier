/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        findSumOfLeaves(root, root);
        return sum;
    }

    public void findSumOfLeaves(TreeNode root, TreeNode prev) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && prev.left == root) {
            sum += root.val;
            return;
        }
        findSumOfLeaves(root.left, root);
        findSumOfLeaves(root.right, root);
    }
}
