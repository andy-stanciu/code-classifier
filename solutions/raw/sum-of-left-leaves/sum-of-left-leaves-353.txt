class Solution {

    private int sum;

    public int sumOfLeftLeaves(TreeNode root) {
        sum = 0;
        dfs(root, false);
        return sum;
    }

    public void dfs(TreeNode node, boolean isLeft) {
        if (node == null) {
            return;
        }
        if (isLeft && node.left == null && node.right == null) {
            sum += node.val;
        }
        dfs(node.left, true);
        dfs(node.right, false);
    }
}
