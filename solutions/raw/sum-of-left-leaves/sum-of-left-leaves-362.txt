/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int getAns(TreeNode root, boolean flag) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null && flag) {
            return root.val;
        }
        int sum = getAns(root.left, true);
        int sum2 = getAns(root.right, false);
        return sum + sum2;
    }

    public int sumOfLeftLeaves(TreeNode root) {
        return getAns(root, false);
    }
}
