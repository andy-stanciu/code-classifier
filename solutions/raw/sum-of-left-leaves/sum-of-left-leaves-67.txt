class Solution {

    public int sumOfLeftLeaves(TreeNode root) {
        int leaves = 0;
        if (root.left == null)
            ;
        else if (// root.left is not leaf
        root.left.left != null || root.left.right != null)
            // root.left is not leaf
            leaves += sumOfLeftLeaves(root.left);
        else
            // root.left is leaf
            leaves += root.left.val;
        if (root.right == null)
            ;
        else // root.right is not leaf
        if (root.right.left != null || root.right.right != null)
            leaves += sumOfLeftLeaves(root.right);
        // else leaves += root.right.val; // uncommenting this line would add values for right leaves
        return leaves;
    }
}
