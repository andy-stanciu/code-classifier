class Solution {

    int sum = 0;

    public int sumOfLeftLeaves(TreeNode root) {
        helper(root, 0);
        return sum;
    }

    void helper(TreeNode root, int mark) {
        if (//Checks a leaf node
        root.left == null && root.right == null) {
            //mark is 1 if the leaf node is Left leaf node
            if (mark == 1)
                sum += root.val;
            return;
        }
        if (root.left != null)
            helper(root.left, 1);
        if (root.right != null)
            helper(root.right, 0);
    }
}
