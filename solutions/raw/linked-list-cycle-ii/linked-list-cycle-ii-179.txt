public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head == null)
            return null;
        ListNode fast = head;
        ListNode slow = head;
        ListNode meet = null;
        while (true) {
            if (fast.next != null)
                fast = fast.next;
            else
                return null;
            if (fast.next != null)
                fast = fast.next;
            else
                return null;
            slow = slow.next;
            if (fast == slow) {
                meet = slow;
                break;
            }
        }
        fast = head;
        while (fast != slow) {
            fast = fast.next;
            slow = slow.next;
        }
        return slow;
    }
}
