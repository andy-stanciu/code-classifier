/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        int length = 0;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (fast == slow) {
                slow = slow.next;
                length++;
                while (slow != fast) {
                    slow = slow.next;
                    length++;
                }
                if (slow == fast) {
                    break;
                }
            }
        }
        if (length == 0) {
            return null;
        }
        fast = slow = head;
        for (int i = 0; i < length; i++) {
            slow = slow.next;
        }
        while (slow != fast) {
            fast = fast.next;
            slow = slow.next;
        }
        return slow;
    }
}
