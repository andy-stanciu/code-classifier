public class Solution {

    public ListNode detectCycle(ListNode head) {
        HashMap<ListNode, Integer> map = new HashMap<>();
        while (head != null) {
            if (map.containsKey(head)) {
                // Return the node where the cycle starts
                return head;
            }
            // Store the node in the map
            map.put(head, 1);
            head = head.next;
        }
        // No cycle detected
        return null;
    }
}
