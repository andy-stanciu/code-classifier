/**
 * Definition for singly-linked list.
 * class ListNode {
 * int val;
 * ListNode next;
 * ListNode(int x) {
 * val = x;
 * next = null;
 * }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        Set<ListNode> map = new HashSet<>();
        int value = -1;
        ListNode ptr = head;
        ListNode head2 = head;
        while (head != null) {
            if (map.contains(head)) {
                return head;
            }
            map.add(head);
            head = head.next;
        }
        return null;
    }
}
