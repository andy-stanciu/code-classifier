public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode slow = head;
        HashSet<ListNode> hs = new HashSet<>();
        while (slow.next != null) {
            if (hs.contains(slow)) {
                return slow;
            }
            hs.add(slow);
            slow = slow.next;
        }
        return slow.next;
    }
}
