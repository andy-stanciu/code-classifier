/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head != null) {
            if (head.val == Integer.MIN_VALUE) {
                return head;
            } else {
                head.val = Integer.MIN_VALUE;
                return detectCycle(head.next);
            }
        } else
            return null;
    }
}
