public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode s = head;
        ListNode f = head;
        while (s != null && f.next != null) {
            s = s.next;
            f = f.next.next;
            if (f == null)
                return null;
            if (s == f) {
                while (true) {
                    if (s == head)
                        return head;
                    head = head.next;
                    s = s.next;
                }
            }
        }
        return null;
    }
}
