/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head == null)
            return head;
        boolean isCycle = false;
        ListNode s1 = head, f = head;
        while (f != null && f.next != null) {
            s1 = s1.next;
            f = f.next.next;
            if (s1 == f) {
                isCycle = true;
                break;
            }
        }
        // System.out.println(isCycle);
        if (isCycle == false) {
            return null;
        }
        s1 = head;
        while (s1 != f) {
            s1 = s1.next;
            f = f.next;
        }
        return s1;
    }
}
