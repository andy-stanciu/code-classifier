/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        int size = findLength(head);
        System.out.println("Size received from length method is " + size);
        if (size > 0) {
            ListNode first = head;
            ListNode second = head;
            for (int i = 0; i < size; i++) {
                // shifting second pointer by size of cycle
                second = second.next;
            }
            System.out.println("Current pos of second pointer is " + second.val);
            while (true) {
                if (first == second) {
                    return first;
                }
                first = first.next;
                second = second.next;
            }
        }
        return null;
    }

    private int findLength(ListNode head) {
        int length = 0;
        // intializing slow and fast pointer
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            // checking if fast and slow pointing to same node
            if (slow == fast) {
                System.out.println("Cycle found");
                // calculating the length of cycle
                do {
                    slow = slow.next;
                    length += 1;
                } while (slow != fast);
                System.out.println("Size - " + length);
                return length;
            }
        }
        return 0;
    }
}
