/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
//HARE AND TORTOISE ALGORITHM
public class Solution {

    public ListNode detectCycle(ListNode head) {
        // ListNode current = head;
        ListNode slow = head;
        ListNode fast = head;
        while (//Through this loop we find if there is cycle or not
        fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (slow == fast)
                break;
        }
        slow = head;
        if (fast == null || fast.next == null) {
            //It means no cycle was found
            return null;
        } else {
            // slow = head;
            while (slow != fast) {
                slow = slow.next;
                fast = fast.next;
            }
        }
        return slow;
    }
}
