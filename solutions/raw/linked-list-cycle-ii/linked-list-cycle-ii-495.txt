public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode turtle = head;
        ListNode hare = head;
        while (hare != null && hare.next != null) {
            turtle = turtle.next;
            hare = hare.next.next;
            if (turtle == hare) {
                return find(head, hare);
            }
        }
        return null;
    }

    private ListNode find(ListNode turtle, ListNode hare) {
        while (hare != turtle) {
            hare = hare.next;
            turtle = turtle.next;
        }
        return turtle;
    }
}
