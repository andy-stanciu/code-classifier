public class Solution {

    public ListNode detectCycle(ListNode head) {
        HashSet<ListNode> linkset = new HashSet<>();
        ListNode temp = head;
        while (temp != null) {
            if (linkset.contains(temp)) {
                return temp;
            } else {
                linkset.add(temp);
                temp = temp.next;
            }
        }
        return null;
    }
}
