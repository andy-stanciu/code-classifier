/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode f = head;
        ListNode s = head;
        boolean n = false;
        while (f != null && f.next != null) {
            f = f.next.next;
            s = s.next;
            if (f == s) {
                n = true;
                break;
            }
        }
        if (n) {
            f = head;
            while (f != s) {
                f = f.next;
                s = s.next;
            }
            return f;
        }
        return null;
    }
}
