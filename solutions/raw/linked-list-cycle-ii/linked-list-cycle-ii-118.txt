public class Solution {

    public ListNode detectCycle(ListNode head) {
        //O(n) space and time complexity solution
        HashSet<ListNode> hs = new HashSet<>();
        while (head != null) {
            if (hs.contains(head))
                break;
            hs.add(head);
            head = head.next;
        }
        if (head == null)
            return null;
        else
            return head;
    }
}
