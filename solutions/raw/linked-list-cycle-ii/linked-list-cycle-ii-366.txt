/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head == null || head.next == null)
            return null;
        int clength = cycleLength(head);
        ListNode node = cycleStart(clength, head);
        if (clength == 0)
            return null;
        return node;
    }

    public int cycleLength(ListNode head) {
        ListNode f = head;
        ListNode s = head;
        while (f != null && f.next != null) {
            f = f.next.next;
            s = s.next;
            if (f == s) {
                int length = 0;
                do {
                    s = s.next;
                    length++;
                } while (f != s);
                return length;
            }
        }
        return 0;
    }

    public ListNode cycleStart(int clength, ListNode head) {
        ListNode f = head;
        ListNode s = head;
        for (int i = 0; i < clength; i++) {
            s = s.next;
        }
        while (f != s) {
            s = s.next;
            f = f.next;
        }
        return s;
    }
}
