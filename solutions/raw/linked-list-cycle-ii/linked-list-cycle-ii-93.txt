public class Solution {

    public ListNode detectCycle(ListNode head) {
        // Create a hashset to store nodes
        // not values because values can be same but node addesses
        // will always be different even with same values/data
        Set<ListNode> set = new HashSet<>();
        ListNode n = head;
        while (n != null) {
            // if set contains the node already
            if (set.contains(n)) {
                // return it, its our answer
                return n;
            }
            // add it to set
            set.add(n);
            // move to next node
            n = n.next;
        }
        // we reach till this statement when
        // iteration on our list ends, means no cycle
        return null;
    }
}
