/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode a = head;
        while (a != null && a.val != -999999) {
            a.val = -999999;
            a = a.next;
        }
        return a;
    }
}
