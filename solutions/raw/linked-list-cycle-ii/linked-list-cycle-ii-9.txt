public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode curr = head;
        HashMap<ListNode, Integer> hm = new HashMap<>();
        int index = 0;
        while (curr.next != null) {
            if (hm.containsKey(curr)) {
                return curr;
            } else {
                hm.put(curr, index);
                index++;
                curr = curr.next;
            }
        }
        return null;
    }
}
