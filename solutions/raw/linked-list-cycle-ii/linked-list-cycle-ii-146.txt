public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode sPtr = head;
        ListNode fPtr = head;
        while (fPtr != null && fPtr.next != null) {
            sPtr = sPtr.next;
            fPtr = fPtr.next.next;
            if (sPtr == fPtr)
                break;
        }
        if (fPtr == null || fPtr.next == null) {
            return null;
        }
        fPtr = head;
        while (sPtr != fPtr) {
            sPtr = sPtr.next;
            fPtr = fPtr.next;
        }
        return sPtr;
    }
}
