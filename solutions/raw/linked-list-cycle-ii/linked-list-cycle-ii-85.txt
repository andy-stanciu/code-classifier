public class Solution {

    public ListNode detectCycle(ListNode head) {
        ListNode first = new ListNode();
        ListNode second = new ListNode();
        first = head;
        second = head;
        while (second != null && second.next != null) {
            first = first.next;
            second = second.next.next;
            if (first == second) {
                first = head;
                while (first != second) {
                    first = first.next;
                    second = second.next;
                }
                return second;
            }
        }
        return null;
    }
}
