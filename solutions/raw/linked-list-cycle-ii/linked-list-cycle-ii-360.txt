/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode detectCycle(ListNode head) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode slowptr = head;
        ListNode fptr = head;
        while (fptr != null && fptr.next != null) {
            fptr = fptr.next.next;
            slowptr = slowptr.next;
            if (fptr == slowptr) {
                return start(slowptr, head);
            }
        }
        return null;
    }

    public ListNode start(ListNode slowptr, ListNode head) {
        ListNode temp = head;
        while (slowptr != temp) {
            temp = temp.next;
            slowptr = slowptr.next;
        }
        return temp;
    }
}
