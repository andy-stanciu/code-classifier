class Solution {

    public int maxProfit(int[] prices) {
        int buyPrice = prices[0];
        int profit = 0;
        int n = prices.length;
        for (int i = 1; i < n; i++) {
            if (buyPrice > prices[i]) {
                buyPrice = prices[i];
            } else if (0 < (prices[i] - buyPrice)) {
                profit += (prices[i] - buyPrice);
                buyPrice = prices[i];
            }
        }
        return profit;
    }
}
