class Solution {

    public int maxProfit(int[] prices) {
        int minBuy = prices[0];
        int profit = 0;
        int n = prices.length;
        for (int i = 1; i < n; i++) {
            if (prices[i] < minBuy) {
                minBuy = prices[i];
            } else if (0 < (prices[i] - minBuy)) {
                profit += prices[i] - minBuy;
                minBuy = prices[i];
            }
        }
        return profit;
    }
}
