class Solution {

    public int maxProfit(int[] prices) {
        int ans = 0, n = prices.length;
        int[] profValue = new int[n];
        for (int i = 0; i < n - 1; i++) {
            profValue[i] = prices[i + 1] - prices[i];
        }
        for (int i = 0; i < n; i++) {
            if (profValue[i] > 0) {
                ans += profValue[i];
            }
        }
        return ans;
    }
}
