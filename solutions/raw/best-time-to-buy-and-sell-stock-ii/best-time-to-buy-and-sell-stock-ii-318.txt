class Solution {

    public int maxProfit(int[] prices) {
        // If the price array has only one price so we cannot make any profit
        if (prices.length == 1)
            return 0;
        int buyPrice = prices[0];
        int sellPrice = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            // Think like travelling up the mountain in search of global maxima
            if (prices[i] >= prices[i - 1]) {
                //UpHill
                sellPrice = prices[i];
                // Think like travelling down the mountain in search of global minima
            } else {
                //DownHill
                buyPrice = prices[i];
            }
            // Else part handles when we are at last element of array so as to avoid index out of bound
            if (i < (prices.length - 1)) {
                if ((prices[i] >= prices[i - 1]) && (prices[i] > prices[i + 1])) {
                    profit = profit + (sellPrice - buyPrice);
                }
            } else {
                if ((prices[i] >= prices[i - 1])) {
                    profit = profit + (sellPrice - buyPrice);
                }
            }
        }
        return profit;
    }
}
