class Solution {

    public int maxProfit(int[] prices) {
        int cost = 0;
        int buy = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] - prices[buy] > 0) {
                cost += prices[i] - prices[buy];
                buy = i;
            }
            if (prices[i] < prices[buy]) {
                buy = i;
            }
        }
        return cost;
    }
}
