class Solution {

    public int maxProfit(int[] arr) {
        int n = arr.length;
        int[] next = new int[2];
        int profit = 0;
        for (int ind = n - 1; ind >= 0; ind--) {
            int[] curr = new int[2];
            for (int buy = 0; buy <= 1; buy++) {
                if (buy == 1) {
                    // can buy
                    curr[buy] = Math.max(next[0] - arr[ind], next[1]);
                } else {
                    curr[buy] = Math.max(next[1] + arr[ind], next[0]);
                }
            }
            next = curr;
        }
        return next[1];
    }
}
