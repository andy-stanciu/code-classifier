class Solution {

    public int maxProfit(int[] prices) {
        int i = 0, j = 1, k = 0, res = 0;
        if (prices.length == 0) {
            return 0;
        } else if (prices.length == 1) {
            return 0;
        } else if (prices.length == 2) {
            if (prices[i] < prices[j]) {
                return prices[j] - prices[i];
            } else
                return 0;
        }
        //Above all if statement check for length below 2
        while (j != prices.length) {
            //Run from 0-length of Array
            if (prices[i] < prices[j]) {
                //check if pointer i had less value then pointer j
                k = prices[j] - prices[i];
                //subtract and add in previous profit
                res = k + res;
                i = j;
                // increment both values
                j++;
            } else if (prices[i] >= prices[j]) {
                i = j;
                // if i values less or equal to j values then also inrement i and j position by 1
                j++;
            }
        }
        return res;
    }
}
