class Solution {

    public int maxProfit(int[] prices) {
        int earn = 0;
        // Loop through the price array once
        for (int i = 1; i < prices.length; i++) {
            // If today\'s price is greater than yesterday\'s, add the difference to earn
            if (prices[i] > prices[i - 1]) {
                earn += prices[i] - prices[i - 1];
            }
        }
        return earn;
    }
}
