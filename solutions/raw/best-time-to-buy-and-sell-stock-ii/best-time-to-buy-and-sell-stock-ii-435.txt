class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int max = prices[0];
        int totalProfit = 0;
        for (int i : prices) {
            if (i > max) {
                max = i;
            }
            if (i < max) {
                totalProfit += max - min;
                min = i;
                max = i;
            }
        }
        totalProfit += max - min;
        return totalProfit;
    }
}
