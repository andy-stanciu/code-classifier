class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        for (int sellingDay = 1; sellingDay < prices.length; sellingDay++) {
            int previousDay = sellingDay - 1;
            if (prices[sellingDay] > prices[previousDay]) {
                maxProfit += prices[sellingDay] - prices[previousDay];
            }
        }
        return maxProfit;
    }
}
