class Solution {

    public int maxProfit(int[] prices) {
        int buy = 0;
        int sell = 1;
        int sum = 0;
        while (sell < prices.length) {
            if (prices[sell] > prices[buy])
                sum += prices[sell++] - prices[buy++];
            else
                buy = sell++;
        }
        return sum;
    }
}
