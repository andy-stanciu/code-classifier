class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int purchase = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (purchase > prices[i]) {
                purchase = prices[i];
            }
            if (prices[i] - purchase > 0) {
                profit = profit + prices[i] - purchase;
                purchase = prices[i];
            }
        }
        return profit;
    }
}
