class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            //Buying only when you know price is going up
            // Example 1,5.. here you know price will increase from 1 to 5 so buy 1.
            if (prices[i] < prices[i + 1]) {
                profit += prices[i + 1] - prices[i];
            }
        }
        return profit;
    }
}
