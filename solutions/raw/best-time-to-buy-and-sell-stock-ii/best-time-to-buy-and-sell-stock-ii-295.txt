class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            // check the condition if the price of the stock is less then the next dat stock
            if (prices[i] < prices[i + 1]) {
                profit = profit + (prices[i + 1] - prices[i]);
            }
        }
        return profit;
    }
}
