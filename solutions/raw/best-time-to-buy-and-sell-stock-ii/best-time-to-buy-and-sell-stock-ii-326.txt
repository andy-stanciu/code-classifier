class Solution {

    public int maxProfit(int[] prices) {
        int max_profit = 0, low = prices[0], high = prices[0];
        int n = prices.length;
        int i = 0;
        int buy = prices[0];
        for (i = 1; i < n; i++) {
            //To find if profit is made
            if (prices[i] > buy) {
                max_profit += (prices[i] - buy);
            }
            buy = prices[i];
        }
        return max_profit;
    }
}
