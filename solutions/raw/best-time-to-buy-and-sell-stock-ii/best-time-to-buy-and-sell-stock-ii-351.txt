class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        //asusual the loop for traversing the array of prices
        for (int i = 0; i < prices.length - 1; i++) {
            //if current price is lesser then future buying and storing only the profit
            if (prices[i] < prices[i + 1]) {
                profit += prices[i + 1] - prices[i];
            }
        }
        return profit;
    }
}
