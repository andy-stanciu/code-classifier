class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i - 1] < prices[i]) {
                maxProfit = maxProfit + prices[i] - prices[i - 1];
            }
        }
        return maxProfit;
    }
}
// prices[i-1] = 5 ,  Prices[i] = 10  // 5
// maxprofit = 4 2nd itr + 3 3rd itr  -> 4 + 3 = 7
