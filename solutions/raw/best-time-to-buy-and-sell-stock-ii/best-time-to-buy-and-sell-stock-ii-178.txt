class Solution {

    public int maxProfit(int[] prices) {
        int bp = prices[0];
        int total_profit = 0;
        int profit;
        for (int i = 1; i < prices.length; i++) {
            profit = prices[i] - bp;
            if (profit > 0) {
                total_profit += profit;
                bp = prices[i];
            } else {
                bp = Math.min(bp, prices[i]);
            }
        }
        return total_profit;
    }
}
