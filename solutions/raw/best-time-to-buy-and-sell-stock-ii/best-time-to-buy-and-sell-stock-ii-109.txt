// 1) If there is any price increase p[i] > p[i-1], we can purchase at
// day[i-1] and see day[i] and make a profit.
// 2) If there is any price decrease, we would sell in the previous day
// and avoid that lost.
// So overall, the max will be sum(all_gain_delta).
class Solution {

    public int maxProfit(int[] prices) {
        int result = 0;
        for (int i = 1; i < prices.length; i++) {
            int delta = prices[i] - prices[i - 1];
            result += delta > 0 ? delta : 0;
            // result += Math.max(delta, 0);  // Same as above.
        }
        return result;
    }
}
