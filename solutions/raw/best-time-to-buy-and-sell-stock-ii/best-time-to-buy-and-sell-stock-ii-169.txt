class Solution {

    public int maxProfit(int[] prices) {
        // greedy approach to sell whenever profit in array
        int maxProfit = 0;
        int l = prices.length;
        if (l < 2)
            return 0;
        int buy = prices[0];
        int profit = 0;
        for (int i = 0; i < l; i++) {
            int x = prices[i];
            if (x < buy) {
                buy = x;
            } else if (x - buy > profit) {
                profit = x - buy;
                maxProfit += profit;
                buy = Integer.MAX_VALUE;
                profit = 0;
                i--;
            }
        }
        return maxProfit;
    }
}
