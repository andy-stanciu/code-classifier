class Solution {

    public int maxProfit(int[] prices) {
        int buy = -prices[0];
        int sell = 0;
        for (int i = 1; i < prices.length; i++) {
            int newBuy = Math.max(buy, sell - prices[i]);
            int newsell = Math.max(sell, buy + prices[i]);
            buy = newBuy;
            sell = newsell;
        }
        return sell;
    }
}
