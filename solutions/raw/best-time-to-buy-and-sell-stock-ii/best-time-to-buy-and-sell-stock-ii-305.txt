class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        long[][] dp = new long[n][2];
        for (long[] it : dp) {
            Arrays.fill(it, -1);
        }
        return (int) fun(prices, 0, n, 1, dp);
    }

    static long fun(int[] prices, int idx, int n, int buy, long[][] dp) {
        if (idx == n)
            return 0;
        if (dp[idx][buy] != -1)
            return dp[idx][buy];
        long profit = 0;
        if (buy == 1)
            profit = Math.max(-prices[idx] + fun(prices, idx + 1, n, 0, dp), fun(prices, idx + 1, n, 1, dp));
        else
            profit = Math.max(prices[idx] + fun(prices, idx + 1, n, 1, dp), fun(prices, idx + 1, n, 0, dp));
        return dp[idx][buy] = profit;
    }
}
