class Solution {

    public int maxProfit(int[] a) {
        int maxProfit = 0;
        int buy = a[0];
        for (int i = 1; i < a.length; i++) {
            int running_profit = 0;
            if (a[i] > buy) {
                running_profit = a[i] - buy;
                maxProfit = Math.max(maxProfit, maxProfit + running_profit);
            }
            buy = a[i];
        }
        return maxProfit;
    }
}
