class Solution {

    public int maxProfit(int[] prices) {
        int i = 0;
        int j = 1;
        int profit = 0;
        int profitTemp = 0;
        while (j < prices.length) {
            int buy = prices[i];
            int sell = prices[j];
            if (sell - buy <= profitTemp) {
                i = j;
                j++;
                profit += profitTemp;
                profitTemp = 0;
            } else {
                profitTemp = sell - buy;
                if (j == prices.length - 1) {
                    profit += profitTemp;
                }
                j++;
            }
        }
        return profit;
    }
}
