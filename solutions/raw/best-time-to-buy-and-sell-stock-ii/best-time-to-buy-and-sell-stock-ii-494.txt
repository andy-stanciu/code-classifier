class Solution {

    public int maxProfit(int[] prices) {
        int buy = prices[0], sell = prices[0], profit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            if (prices[i] > prices[i + 1]) {
                buy = prices[i + 1];
            } else {
                buy = sell;
            }
            if (prices[i + 1] > prices[i]) {
                sell = prices[i + 1];
            } else {
                sell = buy;
            }
            profit = profit + sell - buy;
        }
        return profit;
    }
}
