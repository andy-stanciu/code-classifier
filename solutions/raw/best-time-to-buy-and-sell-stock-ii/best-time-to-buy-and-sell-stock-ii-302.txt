class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int i = 1;
        int buy = prices[0];
        while (i < prices.length) {
            while (i < prices.length && prices[i] <= prices[i - 1]) {
                buy = prices[i];
                i++;
            }
            int sell = 0;
            while (i < prices.length && prices[i] > prices[i - 1]) {
                sell = prices[i];
                i++;
            }
            if (sell > 0)
                profit += sell - buy;
        }
        return profit;
    }
}
