class Solution {

    public int maxProfit(int[] prices) {
        // maximum profit
        int maxp = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            // loop through the prices
            if (prices[i + 1] > prices[i]) {
                // check the condition
                maxp += (prices[i + 1] - prices[i]);
            }
        }
        return maxp;
    }
}
