class Solution {

    public int maxProfit(int[] prices) {
        int totalProfit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            int profitToday = prices[i + 1] - prices[i];
            if (profitToday > 0)
                totalProfit += profitToday;
        }
        return totalProfit;
    }
}
