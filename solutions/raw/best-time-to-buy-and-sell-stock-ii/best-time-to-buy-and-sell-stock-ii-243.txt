class Solution {

    public int maxProfit(int[] prices) {
        int N = prices.length;
        int[] ahead = new int[2];
        int[] cur = new int[2];
        for (int ind = N - 1; ind >= 0; ind--) {
            for (int buy = 0; buy <= 1; buy++) {
                int profit = 0;
                if (buy == 1) {
                    profit = Math.max(-prices[ind] + ahead[0], 0 + ahead[1]);
                } else {
                    profit = Math.max(prices[ind] + ahead[1], 0 + ahead[0]);
                }
                cur[buy] = profit;
            }
            ahead = cur;
        }
        return cur[1];
    }
}
