class Solution {

    public int maxProfit(int[] prices) {
        int lowBuy = prices[0];
        int profit = 0;
        for (int price : prices) {
            //if lowBuy is Bigger then current, then it\'s no more low buy, switch to current
            if (lowBuy >= price) {
                lowBuy = price;
            } else {
                // if see Profit - sell and add to Profit
                profit += (price - lowBuy);
                // also switch lowBuy to current, so we don\'t double math\'s
                lowBuy = price;
            }
            // yes, lowBuy can go walk out of brackets, but this i feel more readeable
        }
        return profit;
    }
}
