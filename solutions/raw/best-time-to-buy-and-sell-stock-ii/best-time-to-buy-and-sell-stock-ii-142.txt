class Solution {

    public int maxProfit(int[] prices) {
        int minPrice = prices[0], maxProfit = 0, sellPrice = 0, cost = 0;
        for (int i = 0; i < prices.length; i++) {
            minPrice = Math.min(minPrice, prices[i]);
            sellPrice = prices[i];
            cost = sellPrice - minPrice;
            if (cost > 0) {
                maxProfit += cost;
                minPrice = sellPrice;
            }
        }
        return maxProfit;
    }
}
