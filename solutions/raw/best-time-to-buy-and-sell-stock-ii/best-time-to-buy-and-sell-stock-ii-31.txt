class Solution {

    public int maxProfit(int[] prices) {
        // to find the total profit
        int res = 0;
        int n = prices.length;
        // checking if previous element is less than current,  just add the difference
        for (int i = 1; i < n; i++) {
            if (prices[i] > prices[i - 1]) {
                res += (prices[i] - prices[i - 1]);
            }
        }
        return res;
    }
}
