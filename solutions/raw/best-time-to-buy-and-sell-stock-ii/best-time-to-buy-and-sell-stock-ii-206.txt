class Solution {

    public int maxProfit(int[] prices) {
        // Accumulate each arise price everyday
        int profit = 0;
        for (int today = 1; today < prices.length; today++) {
            int profitOfToday = prices[today] - prices[today - 1];
            if (profitOfToday > 0) {
                profit += profitOfToday;
            }
        }
        return profit;
    }
}
