class Solution {

    public int maxProfit(int[] prices) {
        //iterate through the array to the right of the miximum element repeat the same process again
        if (prices == null || prices.length < 2) {
            return 0;
        }
        int maxProfit = 0;
        for (int i = 1; i < prices.length; i++) {
            // If the price at day i is higher than at day i-1, we can make a profit by buying at day i-1 and selling at day i
            if (prices[i] > prices[i - 1]) {
                maxProfit += prices[i] - prices[i - 1];
            }
        }
        return maxProfit;
    }
}
