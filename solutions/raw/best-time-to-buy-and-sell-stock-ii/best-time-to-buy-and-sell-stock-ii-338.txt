class Solution {

    public int maxProfit(int[] prices) {
        int buy = 0, sell = 0, profit = 0, Total_profit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            for (int j = i + 1; j <= i + 1; j++) {
                if (prices[i] < prices[j]) {
                    if (prices[i] < prices[j]) {
                        buy = prices[i];
                        sell = prices[j];
                        profit = sell - buy;
                        Total_profit = Total_profit + profit;
                        break;
                    }
                }
            }
        }
        return Total_profit;
    }
}
