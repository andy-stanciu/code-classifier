class Solution {

    public int maxProfit(int[] prices) {
        int totalProfit = 0;
        int minIdx = 0;
        int maxLocalProfit = 0;
        for (int i = 1; i < prices.length; i++) {
            int profit = prices[i] - prices[minIdx];
            if (profit < maxLocalProfit) {
                totalProfit += maxLocalProfit;
                minIdx = i;
                maxLocalProfit = 0;
            } else {
                maxLocalProfit = profit;
            }
        }
        return totalProfit + maxLocalProfit;
    }
}
