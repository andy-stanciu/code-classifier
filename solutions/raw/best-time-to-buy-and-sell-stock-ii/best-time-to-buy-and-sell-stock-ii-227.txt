class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 1) {
            return 0;
        }
        int buy = prices[0];
        int sell = prices[1];
        int profit = 0;
        boolean sold = true;
        int fp = 0;
        for (int i = 0; i < prices.length; i++) {
            if (buy > prices[i]) {
                buy = prices[i];
            } else if (prices[i] - buy > profit) {
                profit = prices[i] - buy;
                sell = prices[i];
                buy = prices[i];
                sold = true;
            }
            if (sold) {
                fp = profit + fp;
                profit = 0;
                sold = false;
            }
        }
        return fp;
    }
}
