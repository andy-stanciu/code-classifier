class Solution {

    public int maxProfit(int[] prices) {
        // Initialize valie for the current day price
        int currentDayPrice = prices[0];
        // Variable to store the maximum profit
        int maxProfit = 0;
        // If there\'s only one price, return 0 profit
        if (prices.length == 1)
            return maxProfit;
        // Loop through the prices array until the second to last element
        for (int futurePointer = 1; futurePointer < prices.length; futurePointer++) {
            // Check if the price at the current day is less than the price at the next day
            if (currentDayPrice < prices[futurePointer]) {
                // Calculate the profit and add it to maxProfit
                maxProfit += (prices[futurePointer] - currentDayPrice);
            }
            // Assign the future price to the current day
            currentDayPrice = prices[futurePointer];
        }
        // Return the maximum profit calculated
        return maxProfit;
    }
}
