class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int diff, sol = 0;
        for (int i = 0; i < n - 1; i++) {
            if (prices[i] < prices[i + 1]) {
                diff = Math.abs(prices[i] - prices[i + 1]);
                sol += diff;
            }
        }
        return sol;
    }
}
