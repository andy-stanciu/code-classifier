class Solution {

    public int maxProfit(int[] prices) {
        int buyPrice = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            int todayPrice = prices[i];
            if (todayPrice >= buyPrice) {
                maxProfit += todayPrice - buyPrice;
                buyPrice = todayPrice;
            } else {
                buyPrice = todayPrice;
            }
        }
        return maxProfit;
    }
}
