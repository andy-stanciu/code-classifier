class Solution {

    public int maxProfit(int[] prices) {
        int buy = Integer.MAX_VALUE;
        int sum = 0;
        for (int i = 0; i < prices.length; i++) {
            buy = Math.min(buy, prices[i]);
            if (prices[i] > buy) {
                sum = sum + prices[i] - buy;
                buy = prices[i];
            }
        }
        return sum;
    }
}
