class Solution {

    public int maxProfit(int[] arr) {
        int max = 0;
        int ele = arr[0];
        int j = 1;
        while (j < arr.length) {
            if (arr[j] > ele) {
                max += (arr[j] - ele);
            }
            ele = arr[j];
            j++;
        }
        return max;
    }
}
