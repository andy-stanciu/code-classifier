class Solution {

    public int maxProfit(int[] prices) {
        int ans = 0;
        int min_buy = prices[0];
        int maxProfit = Integer.MIN_VALUE;
        for (int i = 0; i < prices.length; i++) {
            min_buy = Math.min(min_buy, prices[i]);
            maxProfit = Math.max(maxProfit, prices[i] - min_buy);
            if (maxProfit > 0) {
                ans = ans + maxProfit;
                min_buy = prices[i];
                maxProfit = 0;
            }
        }
        return ans;
    }
}
