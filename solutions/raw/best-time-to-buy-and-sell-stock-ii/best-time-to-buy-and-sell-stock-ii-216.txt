class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        // base case for day 1
        // do nothing
        int noStock = 0;
        // buy
        int inHand = -prices[0];
        for (int i = 1; i < n; i++) {
            // do nothing or sell
            noStock = Math.max(noStock, inHand + prices[i]);
            // do nothing or buy
            inHand = Math.max(inHand, noStock - prices[i]);
        }
        return noStock;
    }
}
