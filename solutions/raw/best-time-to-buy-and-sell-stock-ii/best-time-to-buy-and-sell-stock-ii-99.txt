class Solution {

    public int maxProfit(int[] arr) {
        int max = 0;
        int i = 0;
        int j = 1;
        while (j < arr.length) {
            if (arr[i] < arr[j]) {
                int temp = arr[j] - arr[i];
                max = max + temp;
                i++;
                j++;
            } else {
                i++;
                j++;
            }
        }
        return max;
    }
}
