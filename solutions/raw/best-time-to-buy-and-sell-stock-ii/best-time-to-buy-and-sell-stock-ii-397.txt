class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int buy = 0;
        int sell = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            buy = prices[i];
            sell = prices[i + 1];
            if (buy <= sell) {
                maxProfit += (sell - buy);
            }
        }
        return maxProfit;
    }
}
