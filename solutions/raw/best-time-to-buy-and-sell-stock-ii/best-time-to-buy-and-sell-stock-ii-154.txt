class Solution {

    public int maxProfit(int[] prices) {
        int currPrice = prices[0];
        int totalProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < currPrice) {
                currPrice = prices[i];
            } else if (prices[i] > currPrice) {
                totalProfit += (prices[i] - currPrice);
                currPrice = prices[i];
            }
        }
        return totalProfit;
    }
}
