class Solution {

    public int maxProfit(int[] prices) {
        int buyPrice = (int) 1e9, profit = 0;
        for (int i : prices) {
            if (buyPrice >= i) {
                buyPrice = i;
            } else {
                profit += i - buyPrice;
                buyPrice = i;
            }
        }
        return profit;
    }
}
