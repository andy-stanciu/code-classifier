class Solution {

    public int maxProfit(int[] prices) {
        int l = prices.length, valley = 0, peak = 0, maxProfit = 0, i = 0;
        while (i < l - 1) {
            while (i < l - 1 && prices[i] >= prices[i + 1]) i++;
            valley = prices[i];
            while (i < l - 1 && prices[i] <= prices[i + 1]) i++;
            peak = prices[i];
            maxProfit += peak - valley;
        }
        return maxProfit;
    }
}
