class Solution {

    public int maxProfit(int[] prices) {
        int bestBuy = 0;
        for (int i = 1; i < prices.length; i++) {
            int curr = prices[i];
            if (prices[i - 1] < curr) {
                bestBuy += curr - prices[i - 1];
            }
        }
        return bestBuy;
    }
}
