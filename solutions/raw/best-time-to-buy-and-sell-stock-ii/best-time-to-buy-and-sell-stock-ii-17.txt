class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int buy = 0;
        int diff = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            buy = prices[i];
            if (prices[i] < prices[i + 1]) {
                diff = prices[i + 1] - buy;
                profit += diff;
            }
        }
        return profit;
    }
}
