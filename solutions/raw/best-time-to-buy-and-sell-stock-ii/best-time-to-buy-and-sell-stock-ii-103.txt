class Solution {

    public int maxProfit(int[] prices) {
        int buy_price = prices[0];
        int maxprofit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (buy_price > prices[i]) {
                buy_price = prices[i];
            } else {
                int profit = prices[i] - buy_price;
                maxprofit = Math.max(profit + maxprofit, maxprofit);
                buy_price = prices[i];
            }
        }
        return maxprofit;
    }
}
