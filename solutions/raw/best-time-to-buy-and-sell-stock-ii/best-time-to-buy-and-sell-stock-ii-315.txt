class Solution {

    public int maxProfit(int[] prices) {
        int result = 0;
        int current = 0;
        int after = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            current = prices[i];
            after = prices[i + 1];
            if (current < after) {
                result += after - current;
            }
        }
        return result;
    }
}
