class Solution {

    public int maxProfit(int[] price) {
        int day = price[0];
        int maxProfit = 0;
        int i = 1;
        while (i < price.length) {
            if (day > price[i]) {
                day = price[i++];
            } else {
                int profit = price[i] - day;
                day = price[i++];
                maxProfit = maxProfit + profit;
            }
        }
        return maxProfit;
    }
}
