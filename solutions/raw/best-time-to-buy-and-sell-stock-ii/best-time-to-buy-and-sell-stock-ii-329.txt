class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 0 || prices.length == 1)
            return 0;
        int buy = Integer.MAX_VALUE;
        int sum = 0;
        for (int i = 0; i < prices.length; i++) {
            buy = Math.min(buy, prices[i]);
            if (i < prices.length - 1 && prices[i] > prices[i + 1]) {
                sum += prices[i] - buy;
                buy = prices[i + 1];
            }
            if (i == prices.length - 1 && prices[i] >= prices[i - 1]) {
                sum += prices[i] - buy;
            }
        }
        return sum;
    }
}
