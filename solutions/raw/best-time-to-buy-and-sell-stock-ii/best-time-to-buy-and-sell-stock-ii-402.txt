class Solution {

    public int maxProfit(int[] prices) {
        int i = 0;
        int n = prices.length;
        int hold = prices[0];
        int ans = 0;
        for (i = 1; i < n; i++) {
            if (hold < prices[i]) {
                ans += prices[i] - hold;
            }
            hold = prices[i];
        }
        return ans;
    }
}
