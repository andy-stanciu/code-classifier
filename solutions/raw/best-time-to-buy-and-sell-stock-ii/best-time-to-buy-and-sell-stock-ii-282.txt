class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int[] accProfits = new int[prices.length];
        accProfits[prices.length - 1] = 0;
        for (int i = prices.length - 2; i >= 0; i--) {
            if (prices[i] >= prices[i + 1]) {
                accProfits[i] = accProfits[i + 1];
            } else {
                accProfits[i] = accProfits[i + 1] + prices[i + 1] - prices[i];
            }
            if (accProfits[i] > maxProfit) {
                maxProfit = accProfits[i];
            }
        }
        return maxProfit;
    }
}
