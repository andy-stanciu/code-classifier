class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int buy = prices[0];
        for (int i : prices) {
            if (i < buy) {
                buy = i;
            } else if (i > buy) {
                profit = profit + i - buy;
                buy = i;
            }
        }
        return profit;
    }
}
