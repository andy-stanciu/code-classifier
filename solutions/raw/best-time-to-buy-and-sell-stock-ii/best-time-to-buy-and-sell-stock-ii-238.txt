class Solution {

    public int maxProfit(int[] prices) {
        int i = 0;
        int min = prices[0];
        int max = prices[0];
        int profit = 0;
        while (i < prices.length - 1) {
            while (prices[i] > prices[i + 1]) {
                i++;
                if (i == prices.length - 1)
                    return profit;
            }
            min = prices[i];
            while (prices[i] < prices[i + 1]) {
                i++;
                if (i == prices.length - 1)
                    break;
            }
            max = prices[i];
            profit += max - min;
            i++;
        }
        return profit;
    }
}
