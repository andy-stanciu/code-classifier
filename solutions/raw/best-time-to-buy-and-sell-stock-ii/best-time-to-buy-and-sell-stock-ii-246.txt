class Solution {

    public int maxProfit(int[] prices) {
        // initial price
        int number = prices[0];
        int profit = 0;
        int x = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < number) {
                profit += x;
                // Update the number to the new lower price
                number = prices[i];
                // Reset x since we are considering a new buy point
                x = 0;
            } else if (prices[i] - number > x) {
                // If a larger profit is found, update x
                x = prices[i] - number;
            }
            // When prices[i] doesn\'t give a better profit, finalize the current profit
            if (prices[i] - number < x) {
                // Add profit and reset x
                profit += x;
                // Reset for the next transaction
                x = 0;
                // Consider new buy point
                number = prices[i];
            }
        }
        // Add any remaining profit after the loop ends
        profit += x;
        return profit;
    }
}
