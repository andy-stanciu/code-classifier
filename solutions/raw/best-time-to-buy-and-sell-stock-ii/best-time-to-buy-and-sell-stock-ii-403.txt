class Solution {

    public int maxProfit(int[] prices) {
        int buy = prices[0], profit = 0, total = 0;
        int n = prices.length;
        for (int i = 1; i < n; i++) {
            if (prices[i] < buy) {
                buy = prices[i];
            } else {
                profit = prices[i] - buy;
                buy = prices[i];
                total += profit;
            }
        }
        return total;
    }
}
