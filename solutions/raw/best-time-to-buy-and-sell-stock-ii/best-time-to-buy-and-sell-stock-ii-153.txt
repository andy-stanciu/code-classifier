class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        boolean stockInHand = false;
        int cp = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            if (!stockInHand && prices[i + 1] > prices[i]) {
                cp = prices[i];
                stockInHand = true;
            }
            if (stockInHand && prices[i] > prices[i + 1]) {
                profit += prices[i] - cp;
                stockInHand = false;
            }
        }
        if (stockInHand)
            profit += prices[prices.length - 1] - cp;
        return profit;
    }
}
