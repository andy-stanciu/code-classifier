class Solution {

    public int maxProfit(int[] nums) {
        int min = nums[0], totalProfit = 0;
        int pos = 1, max = -1;
        while (pos < nums.length) {
            // Track the Min of the list and start tracking it
            if (nums[pos] < min)
                min = nums[pos];
            // Track the valley until it reaches down
            while (pos < nums.length && max < nums[pos]) max = nums[pos++];
            totalProfit += max - min;
            max = -1;
            if (pos < nums.length)
                min = nums[pos];
            pos++;
        }
        return totalProfit;
    }
}
