class Solution {

    public int maxProfit(int[] prices) {
        int total_profit = 0;
        for (int i = 1; i < prices.length; i++) {
            // If todays profit is greater than yesterday we make profit
            if (prices[i] > prices[i - 1]) {
                total_profit += prices[i] - prices[i - 1];
            }
        }
        return total_profit;
    }
}
