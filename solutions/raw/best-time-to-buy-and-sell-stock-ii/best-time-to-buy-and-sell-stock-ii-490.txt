class Solution {

    public int maxProfit(int[] prices) {
        int max = 0;
        int buy = 0, sell = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < prices[buy]) {
                buy = i;
                sell = i;
            } else if (prices[i] > prices[buy]) {
                max = max + prices[i] - prices[buy];
                buy = i;
                sell = i;
            }
        }
        return max;
    }
}
