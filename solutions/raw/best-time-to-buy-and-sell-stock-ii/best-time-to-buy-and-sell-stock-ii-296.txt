class Solution {

    public int maxProfit(int[] prices) {
        int left = 0;
        int count = 0;
        int buy = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] > buy) {
                count += prices[i] - buy;
            }
            buy = prices[i];
        }
        return count;
    }
}
