class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int nextbuy = 0;
        int nextsell = 0;
        for (int i = n - 1; i >= 0; i--) {
            int prevbuy = 0;
            int prevsell = 0;
            prevbuy = Math.max(-prices[i] + nextsell, nextbuy);
            prevsell = Math.max(prices[i] + nextbuy, nextsell);
            nextbuy = prevbuy;
            nextsell = prevsell;
        }
        return nextbuy;
    }
}
