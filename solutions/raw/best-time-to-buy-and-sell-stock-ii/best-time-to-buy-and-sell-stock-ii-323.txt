class Solution {

    public int maxProfit(int[] prices) {
        int maxprofit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            int value = prices[i + 1] - prices[i];
            if (value > maxprofit || value > 0)
                maxprofit += value;
        }
        return maxprofit;
    }
}
