class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        ArrayList<Integer> list = new ArrayList<>();
        Stack<Integer> s = new Stack<>();
        for (int i = 0; i < asteroids.length; i++) {
            if (s.isEmpty()) {
                s.push(asteroids[i]);
                continue;
            }
            if ((!s.isEmpty()) && (s.peek() > 0 && asteroids[i] > 0) || (s.peek() < 0 && asteroids[i] < 0) || s.peek() < 0 && asteroids[i] > 0) {
                s.push(asteroids[i]);
                continue;
            } else {
                if (!s.isEmpty() && s.peek() > 0 && asteroids[i] < 0) {
                    if (Math.abs(s.peek()) > Math.abs(asteroids[i])) {
                        continue;
                    } else if (!s.isEmpty() && Math.abs(s.peek()) == Math.abs(asteroids[i])) {
                        s.pop();
                        continue;
                    } else {
                        s.pop();
                        i--;
                        continue;
                    }
                    // }else{
                    //     if(!s.isEmpty() && s.peek() < 0 && asteroids[i] > 0){
                    //         if(Math.abs(s.peek()) > Math.abs(asteroids[i])){
                    //             continue;
                    //         }
                    //         else if ( !s.isEmpty() && Math.abs(s.peek()) == Math.abs(asteroids[i])){
                    //             s.pop();
                    //             continue;
                    //         }else{
                    //             s.pop();
                    //             i--;
                    //             continue;
                    //         }
                    //     }
                }
            }
        }
        while (!s.isEmpty()) {
            list.add(s.pop());
        }
        int[] arr = new int[list.size()];
        for (int i = list.size() - 1; i >= 0; i--) {
            arr[i] = list.get(list.size() - i - 1);
        }
        return arr;
    }
}
