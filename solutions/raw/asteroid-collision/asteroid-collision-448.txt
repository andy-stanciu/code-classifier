class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Deque<Integer> stack = new ArrayDeque<>();
        for (int i = 0; i < asteroids.length; i++) {
            if (stack.isEmpty() || stack.peek() < 0) {
                stack.push(asteroids[i]);
                continue;
            }
            if (stack.size() > 0 && asteroids[i] < 0) {
                while (!stack.isEmpty() && stack.peek() > 0 && stack.peek() < Math.abs(asteroids[i])) {
                    stack.pop();
                }
                if (!stack.isEmpty() && stack.peek() < 0) {
                    stack.push(stack.peek());
                }
                if (!stack.isEmpty() && stack.peek() == Math.abs(asteroids[i])) {
                    stack.pop();
                    continue;
                } else if (!stack.isEmpty() && stack.peek() > Math.abs(asteroids[i])) {
                    continue;
                } else if (!stack.isEmpty() && stack.peek() < Math.abs(asteroids[i])) {
                    stack.pop();
                    stack.push(asteroids[i]);
                } else if (stack.isEmpty()) {
                    stack.push(asteroids[i]);
                }
            } else if (stack.size() > 0 && asteroids[i] > 0) {
                stack.push(asteroids[i]);
            }
        }
        int[] remaining = new int[stack.size()];
        for (int i = remaining.length - 1; i >= 0; i--) {
            remaining[i] = stack.pop();
        }
        return remaining;
    }
}
