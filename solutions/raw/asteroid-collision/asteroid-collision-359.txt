class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> st = new Stack<>();
        int n = asteroids.length;
        for (int i = 0; i < n; i++) {
            if (!st.isEmpty() && (st.peek() > 0 && asteroids[i] < 0)) {
                if (Math.abs(asteroids[i]) == st.peek()) {
                    st.pop();
                } else {
                    boolean isBothDestroyed = false;
                    while (!st.isEmpty() && Math.abs(asteroids[i]) >= st.peek() && st.peek() > 0) {
                        if (Math.abs(asteroids[i]) == st.peek()) {
                            st.pop();
                            isBothDestroyed = true;
                            break;
                        } else {
                            st.pop();
                        }
                    }
                    if (!isBothDestroyed) {
                        if (!st.isEmpty() && st.peek() > 0 && asteroids[i] < 0 && st.peek() >= Math.abs(asteroids[i])) {
                            // st.push(asteroids[i]);
                            continue;
                        } else if (!st.isEmpty() && st.peek() < 0 && asteroids[i] < 0) {
                            if (!isBothDestroyed) {
                                st.push(asteroids[i]);
                            }
                        } else {
                            st.push(asteroids[i]);
                        }
                    }
                }
            } else {
                st.push(asteroids[i]);
            }
        }
        // System.out.println(st);
        int[] arr = new int[st.size()];
        for (int i = arr.length - 1; i >= 0; i--) {
            arr[i] = st.pop();
        }
        return arr;
    }
}
