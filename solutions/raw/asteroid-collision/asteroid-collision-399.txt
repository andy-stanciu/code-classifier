class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> stack = new Stack<>();
        for (int asteroid : asteroids) {
            if (asteroid > 0) {
                stack.push(asteroid);
            } else {
                boolean isDestroyed = false;
                while (!stack.isEmpty()) {
                    int top = stack.peek();
                    if (top < 0) {
                        break;
                    } else if (top == Math.abs(asteroid)) {
                        stack.pop();
                        isDestroyed = true;
                        break;
                    } else if (top > Math.abs(asteroid)) {
                        isDestroyed = true;
                        break;
                    } else {
                        stack.pop();
                    }
                }
                if (!isDestroyed) {
                    stack.push(asteroid);
                }
            }
        }
        int[] result = new int[stack.size()];
        int i = 0;
        for (int elem : stack) {
            result[i] = elem;
            i += 1;
        }
        return result;
    }
}
