class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> st = new Stack();
        for (int i = 0; i < asteroids.length; i++) {
            int a = asteroids[i];
            while (!st.empty() && a < 0 && st.peek() > 0) {
                int sum = a + st.peek();
                if (sum < 0) {
                    st.pop();
                } else if (sum > 0) {
                    a = 0;
                } else {
                    st.pop();
                    a = 0;
                }
            }
            if (a != 0)
                st.push(a);
        }
        int[] a = new int[st.size()];
        for (int i = a.length - 1; i >= 0; i--) {
            a[i] = st.pop();
        }
        return a;
    }
}
