class Solution {

    public int[] asteroidCollision(int[] a) {
        int[] st = new int[a.length];
        int si = 0;
        for (int x : a) {
            if (x > 0)
                st[si++] = x;
            else {
                for (; si > 0 && st[si - 1] > 0 && st[si - 1] + x < 0; si--) ;
                if (si > 0 && st[si - 1] > 0) {
                    if (st[si - 1] + x == 0)
                        si--;
                    continue;
                }
                st[si++] = x;
            }
        }
        int[] r = new int[si];
        for (int i = 0; i < si; i++) r[i] = st[i];
        return r;
    }
}
