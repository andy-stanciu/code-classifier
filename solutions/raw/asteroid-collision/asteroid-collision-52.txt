class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> st = new Stack<>();
        for (int a : asteroids) {
            boolean flag = true;
            while (!st.isEmpty() && (st.peek() > 0 && a < 0)) {
                //if the `asteroid` on the top is smaller, then we\'ll pop it from the stack as it\'ll explode and continue for the next asteroid
                if (Math.abs(st.peek()) < Math.abs(a)) {
                    st.pop();
                    continue;
                } else //if equal then both will explode,hence we\'ll pop if from the stack
                if (Math.abs(st.peek()) == Math.abs(a)) {
                    st.pop();
                }
                //if the asteroid is smaller then the asteroid on the top then it\'ll explode so we\'ll not pop it and thus mark flag as false
                flag = false;
                break;
            }
            if (flag) {
                st.push(a);
            }
        }
        int[] ans = new int[st.size()];
        for (int i = ans.length - 1; i >= 0; i--) {
            ans[i] = st.peek();
            st.pop();
        }
        return ans;
    }
}
