class Solution {

    public int[] asteroidCollision(int[] ast) {
        Stack<Integer> stack = new Stack();
        for (int i = 0; i < ast.length; i++) {
            if (stack.isEmpty() || ast[i] > 0) {
                stack.push(ast[i]);
            } else {
                while (true) {
                    int peek = stack.peek();
                    if (peek < 0) {
                        stack.push(ast[i]);
                        break;
                    } else if (peek == -ast[i]) {
                        stack.pop();
                        break;
                    } else if (peek > -ast[i]) {
                        break;
                    } else {
                        stack.pop();
                        if (stack.isEmpty()) {
                            stack.push(ast[i]);
                            break;
                        }
                    }
                }
            }
        }
        int[] output = new int[stack.size()];
        for (int i = stack.size() - 1; i >= 0; i--) {
            output[i] = stack.pop();
        }
        return output;
    }
}
