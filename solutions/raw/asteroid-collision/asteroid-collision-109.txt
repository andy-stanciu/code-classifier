class Solution {

    public int[] asteroidCollision(int[] arr) {
        ArrayList<Integer> list = new ArrayList<>();
        if (arr.length == 0) {
            return new int[] {};
        }
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            if (arr[i] > 0) {
                list.add(arr[i]);
            } else {
                while (list.size() != 0 && arr[i] < 0 && list.get(list.size() - 1) > 0 && Math.abs(list.get(list.size() - 1)) < Math.abs(arr[i])) {
                    list.remove(list.size() - 1);
                }
                if (list.size() != 0 && list.get(list.size() - 1) == Math.abs(arr[i])) {
                    list.remove(list.size() - 1);
                } else if (list.size() == 0 || (list.size() != 0 && list.get(list.size() - 1) < 0)) {
                    list.add(arr[i]);
                }
            }
        }
        int[] ans = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            ans[i] = list.get(i);
        }
        return ans;
    }
}
