class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        int ptr = 0;
        int j = 1;
        while (j < asteroids.length) {
            if (ptr == -1 || !(asteroids[ptr] > 0 && asteroids[j] < 0)) {
                asteroids[++ptr] = asteroids[j];
            } else {
                int a = Math.abs(asteroids[ptr]);
                int b = Math.abs(asteroids[j]);
                if (a == b) {
                    ptr--;
                } else if (a < b) {
                    ptr--;
                    continue;
                }
            }
            j++;
        }
        return (ptr == -1) ? (new int[0]) : Arrays.copyOf(asteroids, ptr + 1);
    }
}
