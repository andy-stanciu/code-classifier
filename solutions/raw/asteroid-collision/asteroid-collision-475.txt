class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        List<Integer> stack = new ArrayList();
        stack.add(asteroids[0]);
        for (int i = 1; i < asteroids.length; i++) {
            if (asteroids[i] > 0) {
                stack.add(asteroids[i]);
            } else {
                boolean flag = true;
                int j = stack.size() - 1;
                while (j >= 0 && stack.get(j) > 0) {
                    if (Math.abs(asteroids[i]) > stack.get(j)) {
                        stack.remove(stack.size() - 1);
                        j--;
                    } else if (Math.abs(asteroids[i]) == stack.get(j)) {
                        stack.remove(stack.size() - 1);
                        j--;
                        flag = false;
                        break;
                    } else {
                        flag = false;
                        break;
                    }
                }
                if (flag) {
                    stack.add(asteroids[i]);
                }
            }
        }
        int[] ans = new int[stack.size()];
        for (int i = 0; i < stack.size(); i++) {
            ans[i] = stack.get(i);
        }
        return ans;
    }
}
