class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < asteroids.length; i++) {
            addAsteroid(stack, asteroids[i]);
        }
        int[] result = new int[stack.size()];
        int i = stack.size() - 1;
        while (!stack.isEmpty()) {
            result[i] = stack.pop();
            i--;
        }
        return result;
    }

    private void addAsteroid(Stack<Integer> stack, int asteroid) {
        if (stack.isEmpty() || asteroid > 0 || stack.peek() < 0) {
            stack.add(asteroid);
            return;
        }
        int top = stack.pop();
        if (top + asteroid > 0) {
            stack.push(top);
        } else if (top + asteroid < 0) {
            addAsteroid(stack, asteroid);
        }
    }
}
