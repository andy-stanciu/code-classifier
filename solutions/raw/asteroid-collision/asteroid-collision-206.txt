class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < asteroids.length; i++) {
            checkCollision(stack, asteroids[i]);
        }
        int[] result = new int[stack.size()];
        for (int i = result.length - 1; i >= 0; i--) {
            result[i] = stack.pop();
        }
        return result;
    }

    public void checkCollision(Stack<Integer> stack, int currentAsteroid) {
        if (stack.empty())
            stack.push(currentAsteroid);
        else {
            int prev = stack.peek();
            boolean willCollide = prev > 0 && currentAsteroid < 0;
            if (!willCollide)
                stack.push(currentAsteroid);
            else {
                int modVal = Math.abs(currentAsteroid);
                if (modVal < prev)
                    return;
                else if (modVal == prev) {
                    stack.pop();
                    return;
                } else {
                    stack.pop();
                    checkCollision(stack, currentAsteroid);
                }
            }
        }
    }
}
