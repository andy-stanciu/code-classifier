class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> st = new Stack<>();
        for (int c : asteroids) {
            if (c > 0) {
                st.push(c);
            } else {
                while (st.size() > 0 && st.peek() > 0 && st.peek() < -c) {
                    st.pop();
                }
                if (st.size() > 0 && st.peek() == -c) {
                    st.pop();
                } else if (st.size() > 0 && st.peek() > -c) {
                } else {
                    st.push(c);
                }
            }
        }
        // Create the answer array with the size of the remaining asteroids in the stack
        int[] ans = new int[st.size()];
        // Initialize i to the last index of the array
        int i = st.size() - 1;
        while (i >= 0) {
            // Pop the elements from the stack and assign them to the array
            ans[i--] = st.pop();
        }
        return ans;
    }
}
