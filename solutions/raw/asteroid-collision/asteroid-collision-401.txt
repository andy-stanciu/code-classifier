class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> st = new Stack<>();
        for (int arr : asteroids) {
            if (arr > 0) {
                st.push(arr);
            } else {
                while (!st.empty() && st.peek() > 0 && Math.abs(arr) > st.peek()) {
                    st.pop();
                }
                if (st.empty() || st.peek() < 0) {
                    st.push(arr);
                } else if (!st.empty() && Math.abs(arr) == st.peek()) {
                    st.pop();
                }
            }
        }
        int[] ansr = new int[st.size()];
        for (int i = ansr.length - 1; i >= 0; i--) {
            ansr[i] = st.pop();
        }
        return ansr;
    }
}
