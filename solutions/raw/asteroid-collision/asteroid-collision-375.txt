class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        int j = -1;
        for (int i = 0; i < asteroids.length; i++) {
            int ast = asteroids[i];
            while (j > -1 && ast < 0 && asteroids[j] > 0 && -ast > asteroids[j]) j--;
            if (j == -1 || ast > 0 || asteroids[j] < 0)
                asteroids[++j] = ast;
            else if (-ast == asteroids[j])
                j--;
        }
        return Arrays.copyOfRange(asteroids, 0, j + 1);
    }
}
