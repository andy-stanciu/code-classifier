class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        // collection case
        Stack<Integer> s = new Stack<>();
        for (int asteroid : asteroids) {
            while (!s.isEmpty() && s.peek() > 0 && asteroid < 0) {
                int prev = s.pop();
                if (prev == -asteroid) {
                    asteroid = 0;
                } else if (prev > -asteroid) {
                    s.push(prev);
                    asteroid = 0;
                }
            }
            // if not then put it into the stack
            if (asteroid != 0) {
                s.push(asteroid);
            }
        }
        int[] res = new int[s.size()];
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = s.pop();
        }
        return res;
    }
}
