class Solution {

    public boolean findOppositeDirection(int asteroid1, int asteroid2) {
        if (asteroid1 > 0 && asteroid2 < 0 || asteroid1 < 0 && asteroid2 > 0)
            return true;
        return false;
    }

    public int[] asteroidCollision(int[] asteroids) {
        ArrayDeque<Integer> stack = new ArrayDeque<>();
        for (int i = 0; i < asteroids.length; i++) {
            if (stack.isEmpty()) {
                stack.push(asteroids[i]);
                continue;
            } else {
                if (!findOppositeDirection(stack.peek(), asteroids[i]))
                    stack.push(asteroids[i]);
                else {
                    int survival = asteroids[i];
                    while (!stack.isEmpty()) {
                        if (!findOppositeDirection(stack.peek(), survival))
                            break;
                        if (stack.peek() < 0)
                            break;
                        int nearAsteroid = stack.pop();
                        if (Math.abs(survival) == Math.abs(nearAsteroid)) {
                            survival = 0;
                            break;
                        }
                        survival = Math.abs(survival) > Math.abs(nearAsteroid) ? survival : nearAsteroid;
                    }
                    if (survival != 0)
                        stack.push(survival);
                }
            }
        }
        int[] arr = new int[stack.size()];
        int n = stack.size();
        for (int i = n - 1; i >= 0; i--) {
            arr[i] = stack.pop();
        }
        return arr;
    }
}
