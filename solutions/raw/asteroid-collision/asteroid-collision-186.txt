class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        List<Integer> listWithAsteriods = new ArrayList<>(Arrays.stream(asteroids).boxed().toList());
        for (int i = 1; i < listWithAsteriods.size(); ) /**/
        {
            int firstAsteroid = listWithAsteriods.get(i - 1);
            int secondAsteroid = listWithAsteriods.get(i);
            if ((firstAsteroid > 0 && secondAsteroid > 0) || (firstAsteroid < 0 && secondAsteroid < 0) || firstAsteroid < 0 && secondAsteroid > 0) {
                i++;
                continue;
            }
            firstAsteroid = Math.abs(firstAsteroid);
            secondAsteroid = Math.abs(secondAsteroid);
            if (firstAsteroid == secondAsteroid) {
                listWithAsteriods.remove(i);
                listWithAsteriods.remove(i - 1);
                if (i != 1)
                    i--;
            } else if (firstAsteroid > secondAsteroid)
                listWithAsteriods.remove(i);
            else {
                listWithAsteriods.remove(i - 1);
                if (i != 1)
                    i--;
            }
        }
        return listWithAsteriods.stream().mapToInt(Integer::intValue).toArray();
    }
}
