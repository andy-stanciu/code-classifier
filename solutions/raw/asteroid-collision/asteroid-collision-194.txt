class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        // create a stack data structure using integer array
        class Stack {

            // type of stack object
            int[] stack;

            int top;

            // initilization
            public Stack(int size) {
                stack = new int[size];
                top = 0;
            }

            // push
            public void push(int num) {
                if (top == stack.length) {
                    System.out.println("Stack is full.");
                } else {
                    stack[top] = num;
                    top++;
                }
            }

            // pop
            public int pop() {
                if (isEmpty()) {
                    System.out.println("Stack is empty.");
                    return 0;
                } else {
                    top--;
                    int num = stack[top];
                    stack[top] = 0;
                    return num;
                }
            }

            // is empty
            public boolean isEmpty() {
                return top == 0;
            }

            // peek
            public int peek() {
                if (isEmpty()) {
                    System.out.println("Stack is empty.");
                    return 0;
                } else {
                    return stack[top - 1];
                }
            }

            // size
            public int size() {
                return top;
            }
        }
        // create instance of stack
        Stack stack = new Stack(asteroids.length);
        // loop through asteroids
        for (int a : asteroids) {
            // if the top asteroid is moving to the right add it to the stack
            if (a > 0) {
                stack.push(a);
            } else {
                // in case of the current asteroid moving left check for collision
                while (!stack.isEmpty() && stack.peek() > 0 && stack.peek() < Math.abs(a)) {
                    // if the current asteroid (left) abs is > top then pop (destroy) the top
                    stack.pop();
                }
                if (stack.isEmpty() || stack.peek() < 0) {
                    // if there is only a left moving asteroid or both top and current are left push to stack
                    stack.push(a);
                } else if (stack.peek() == Math.abs(a)) {
                    // if top asteroid is == current asteroid then both get destroyed pop the top
                    stack.pop();
                }
                // if top asteroid > current asteroid we do nothing since current is destroyed
            }
        }
        // result list
        int[] result = new int[stack.size()];
        int i = stack.size() - 1;
        while (!stack.isEmpty()) {
            result[i--] = stack.pop();
        }
        return result;
    }
}
