class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> s = new Stack<>();
        boolean alive = true;
        for (int asteroid : asteroids) {
            alive = true;
            while (!s.isEmpty() && (asteroid < 0 && s.peek() > 0)) {
                if (Math.abs(asteroid) > s.peek()) {
                    s.pop();
                    alive = true;
                } else if (Math.abs(asteroid) == s.peek()) {
                    s.pop();
                    alive = false;
                    break;
                } else {
                    alive = false;
                    break;
                }
            }
            if (alive == true)
                s.push(asteroid);
        }
        int[] ans = new int[s.size()];
        while (!s.isEmpty()) {
            ans[s.size() - 1] = s.pop();
        }
        return ans;
    }
}
