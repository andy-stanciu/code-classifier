class Solution {

    public int[] asteroidCollision(int[] a) {
        int n = a.length;
        // Initialize a stack to hold asteroids
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < n; i++) {
            int current = a[i];
            // If the stack is empty, just push the current asteroid onto the stack
            if (stack.isEmpty()) {
                stack.push(current);
                continue;
            }
            // Get the sign of the current and top asteroids
            int prev = Integer.signum(stack.peek());
            int curr = Integer.signum(current);
            // Check for collision conditions
            if ((curr == prev) || (prev == -1 && curr == 1)) {
                // No collision, push onto stack
                stack.push(current);
            } else {
                // Handle collision cases
                while (true) {
                    // If two asteroids have the same size or same direction, they pass each other
                    if (current == stack.peek() || (Integer.signum(current) == Integer.signum(stack.peek()))) {
                        stack.push(current);
                        break;
                    } else // If the current asteroid is smaller in size, it gets destroyed
                    if (!stack.isEmpty() && (Math.abs(current) < Math.abs(stack.peek()))) {
                        break;
                    } else // If the current asteroid is larger in size, it destroys the top asteroid(s)
                    if (!stack.isEmpty() && Math.abs(current) > Math.abs(stack.peek())) {
                        stack.pop();
                        if (stack.isEmpty()) {
                            stack.push(current);
                            break;
                        }
                    }
                }
            }
        }
        // Convert stack to array
        int[] result = new int[stack.size()];
        for (int i = result.length - 1; i >= 0; i--) {
            result[i] = stack.pop();
        }
        return result;
    }
}
