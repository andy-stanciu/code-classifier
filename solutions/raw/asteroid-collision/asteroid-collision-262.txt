class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        // [1]
        Stack<Integer> stack = new Stack<>();
        for (int asteroid : asteroids) {
            //
            while (!stack.isEmpty() && stack.peek() >= 0 && asteroid < 0 && stack.peek() - Math.abs(asteroid) < 0) {
                stack.pop();
            }
            if (stack.isEmpty() || stack.peek() < 0 || asteroid >= 0) {
                stack.add(asteroid);
            }
            if (stack.peek() >= 0 && asteroid < 0 && stack.peek() - Math.abs(asteroid) == 0) {
                stack.pop();
            }
        }
        int[] res = new int[stack.size()];
        while (!stack.isEmpty()) res[stack.size() - 1] = stack.pop();
        return res;
    }
}
