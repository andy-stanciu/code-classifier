class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < asteroids.length; i++) {
            boolean needToAddCurrentAsteroids = true;
            while (asteroids[i] < 0 && !stack.isEmpty() && stack.peek() > 0) {
                int prev = stack.peek();
                if (prev < -asteroids[i]) {
                    stack.pop();
                } else if (prev == -asteroids[i]) {
                    needToAddCurrentAsteroids = false;
                    stack.pop();
                    break;
                } else {
                    needToAddCurrentAsteroids = false;
                    break;
                }
            }
            if (needToAddCurrentAsteroids) {
                stack.push(asteroids[i]);
            }
        }
        int[] res = new int[stack.size()];
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = stack.pop();
        }
        return res;
    }
}
