class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        int n = asteroids.length;
        Stack<Integer> st = new Stack<>();
        for (int i : asteroids) {
            boolean alive = true;
            while (!st.isEmpty() && i < 0 && st.peek() > 0) {
                if (st.peek() < -i) {
                    st.pop();
                    continue;
                } else if (st.peek() == -i)
                    st.pop();
                alive = false;
                break;
            }
            if (alive)
                st.push(i);
        }
        int[] arr = new int[st.size()];
        for (int i = st.size() - 1; i >= 0; i--) {
            arr[i] = st.pop();
        }
        return arr;
    }
}
