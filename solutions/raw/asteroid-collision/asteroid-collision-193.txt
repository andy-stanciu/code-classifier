class Solution {

    public int[] asteroidCollision(int[] asteroids) {
        Stack<Integer> ms = new Stack<>();
        for (int i = 0; i < asteroids.length; i++) {
            if (ms.isEmpty() || asteroids[i] > 0) {
                ms.push(asteroids[i]);
            } else {
                while (!ms.isEmpty()) {
                    int top = ms.peek();
                    if (top < 0) {
                        ms.push(asteroids[i]);
                        break;
                    }
                    int modval = Math.abs(asteroids[i]);
                    if (modval == top) {
                        ms.pop();
                        break;
                    } else if (modval < top) {
                        break;
                    } else {
                        ms.pop();
                        if (ms.isEmpty()) {
                            ms.push(asteroids[i]);
                            break;
                        }
                    }
                }
            }
        }
        int len = ms.size();
        int[] ss = new int[len];
        for (int i = len - 1; i >= 0; i--) {
            ss[i] = ms.pop();
        }
        return ss;
    }
}
