class Solution {

    public int[] plusOne(int[] digits) {
        if (digits[digits.length - 1] != 9) {
            digits[digits.length - 1] += 1;
            return digits;
        } else {
            digits[digits.length - 1] = 10;
            for (int i = digits.length - 2; i >= 0; i--) {
                if (digits[i + 1] == 10) {
                    if (digits[i] != 9) {
                        digits[i] += 1;
                        digits[i + 1] = 0;
                    } else {
                        digits[i] = 10;
                        digits[i + 1] = 0;
                    }
                }
            }
            if (digits[0] == 10) {
                digits[0] = 0;
                int[] new_digits = new int[digits.length + 1];
                new_digits[0] = 1;
                for (int i = 1; i < new_digits.length; i++) {
                    new_digits[i] = digits[i - 1];
                }
                return new_digits;
            } else {
                return digits;
            }
        }
    }
}
