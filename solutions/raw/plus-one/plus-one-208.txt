class Solution {

    public int[] plusOne(int[] arr) {
        int n = arr.length;
        int adder = 1;
        int r1 = 0;
        int r2 = 0;
        for (int i = n - 1; i >= 0; i--) {
            arr[i] = arr[i] + adder;
            if (arr[i] < 10) {
                return arr;
            }
            arr[i] = 0;
        }
        int[] tmp = new int[n + 1];
        tmp[0] = 1;
        return tmp;
    }
}
