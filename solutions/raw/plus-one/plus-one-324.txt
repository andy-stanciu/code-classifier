public class Solution {

    public int[] plusOne(int[] digits) {
        // Start from the last digit and work backwards
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                // No carry needed, return the result
                return digits;
            }
            // Set current digit to 0 and continue
            digits[i] = 0;
        }
        // If all digits were 9, the array would be all 0s now
        // We need a new array with an extra digit at the beginning
        int[] newNumber = new int[digits.length + 1];
        // Set the first digit to 1
        newNumber[0] = 1;
        return newNumber;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        int[] digits = { 1, 2, 3 };
        int[] result = sol.plusOne(digits);
        // Output: [1, 2, 4]
        System.out.println(java.util.Arrays.toString(result));
    }
}
