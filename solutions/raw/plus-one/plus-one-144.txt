class Solution {

    public int[] plusOne(int[] digits) {
        //iterate from right to left
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i] += 1;
                return digits;
            } else {
                digits[i] = 0;
            }
        }
        //if we reach here it means carry exists and needed to be added at index 0
        int[] op = new int[digits.length + 1];
        op[0] = 1;
        return op;
    }
}
