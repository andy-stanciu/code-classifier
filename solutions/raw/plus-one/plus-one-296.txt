class Solution {

    public int[] plusOne(int[] digits) {
        boolean allNines = true;
        //search for case of all nines as edge case
        for (int i = 0; i < digits.length && allNines; i++) {
            if (digits[i] != 9) {
                allNines = false;
            }
        }
        //case of all nines, create new array with extra
        if (allNines) {
            int[] res = new int[digits.length + 1];
            res[0] = 1;
            return res;
        } else {
            int[] res = digits;
            int index = res.length - 1;
            //search of for first number not a nine from right to left
            while (res[index] == 9) {
                res[index] = 0;
                index--;
            }
            //increment this digit
            res[index]++;
            return res;
        }
    }
}
