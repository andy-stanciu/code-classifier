class Solution {

    public int[] plusOne(int[] digits) {
        // Traverse the digits array from the last element
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                // Simply add 1 if the current digit is less than 9
                digits[i]++;
                // Return the array as no carry is needed
                return digits;
            }
            // If the digit is 9, set it to 0 and continue to handle carry
            digits[i] = 0;
        }
        // If all digits are 9, the array becomes [0, 0, ..., 0], so we need an additional space
        int[] newDigits = new int[digits.length + 1];
        // Set the first element to 1, rest will remain 0
        newDigits[0] = 1;
        return newDigits;
    }
}
