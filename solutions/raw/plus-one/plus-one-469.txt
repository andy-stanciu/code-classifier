class Solution {

    public static int[] plusOne(int[] digits) {
        StringBuilder numberBuilder = new StringBuilder();
        for (int i : digits) {
            numberBuilder.append(i);
        }
        String number = numberBuilder.toString();
        BigInteger result = new BigInteger(number).add(BigInteger.ONE);
        String resultNumber = result.toString();
        int[] digitArray = new int[resultNumber.length()];
        for (int i = 0; i < resultNumber.length(); i++) {
            digitArray[i] = Character.getNumericValue(resultNumber.charAt(i));
        }
        return digitArray;
    }
}
