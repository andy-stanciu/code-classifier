class Solution {

    public int[] plusOne(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            //Basically what this will do is if array is [1,9,9] then it will change
            // it to [1,9,0] and then again in loop now i will point at 9(length -1 value)
            //and then it will make it 0 [1,0,0] and then again loop happen and now
            // 1 will be change in 2 bcuz of if condition
            digits[i] = 0;
        }
        //if array is[9,9] then new array will be created of previous array length + 1
        //and then in new array the default value is 0 u have to just replace first digit with 1
        digits = new int[digits.length + 1];
        digits[0] = 1;
        return digits;
    }
}
