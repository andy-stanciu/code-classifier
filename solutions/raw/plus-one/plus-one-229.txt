class Solution {

    public int[] plusOne(int[] arr) {
        if (arr[arr.length - 1] != 9) {
            arr[arr.length - 1] += 1;
            return arr;
        }
        int[] arr1 = new int[arr.length + 1];
        int x = arr1.length - 1;
        int carry = 0;
        for (int i = arr.length - 1; i >= 0; i--) {
            int sum = 0;
            if (i == arr.length - 1) {
                sum = arr[i] + 1 + carry;
            } else {
                sum = arr[i] + carry;
            }
            if (sum > 9) {
                int rem = sum % 10;
                sum = sum / 10;
                arr1[x--] = rem;
                carry = sum;
                continue;
            }
            carry = 0;
            arr1[x--] = sum;
        }
        arr1[x] = carry;
        if (arr1[0] == 0) {
            for (int i = 0; i < arr1.length - 1; i++) {
                arr[i] = arr1[i + 1];
            }
            return arr;
        }
        return arr1;
    }
}
