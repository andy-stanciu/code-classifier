class Solution {

    public int[] plusOne(int[] digits) {
        int carry = 1;
        int[] ans = new int[digits.length];
        for (int i = digits.length - 1; i >= 0; i--) {
            int digitSum = carry + digits[i];
            carry = digitSum / 10;
            ans[i] = digitSum % 10;
        }
        if (carry == 1) {
            ans = new int[digits.length + 1];
            ans[0] = 1;
        }
        return ans;
    }
}
