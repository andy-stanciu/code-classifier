class Solution {

    public int[] plusOne(int[] digits) {
        int rightPointer = digits.length - 1;
        while (rightPointer >= 0) {
            if (digits[rightPointer] < 9) {
                digits[rightPointer] += 1;
                return digits;
            }
            digits[rightPointer] = 0;
            rightPointer--;
        }
        int[] result = new int[digits.length + 1];
        result[0] = 1;
        return result;
    }
}
