class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length - 1;
        int[] ans = new int[n + 2];
        while (n >= 0) {
            if (digits[n] == 9) {
                digits[n] = 0;
                ans[n + 1] = 0;
                n -= 1;
            } else {
                digits[n] += 1;
                ans[n + 1] += 1;
                break;
            }
        }
        if (digits[0] == 0) {
            ans[0] = 1;
            return ans;
        } else {
            return digits;
        }
    }
}
