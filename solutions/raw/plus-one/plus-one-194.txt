class Solution {

    public int[] plusOne(int[] digits) {
        int x = digits.length - 1;
        int[] f = new int[digits.length + 1];
        if (digits[digits.length - 1] != 9) {
            digits[digits.length - 1]++;
            return digits;
        } else {
            while (digits[x] == 9 && x != 0) {
                x--;
            }
            if (digits[x] != 9) {
                digits[x] = digits[x] + 1;
                for (int i = x + 1; i < digits.length; i++) {
                    digits[i] = 0;
                }
                return digits;
            }
            if (x == 0) {
                f[0] = 1;
                for (int i = 1; i < digits.length; i++) {
                    f[i] = 0;
                }
                return f;
            }
            digits[x - 1] = digits[x - 1] + 1;
        }
        return digits;
    }
}
