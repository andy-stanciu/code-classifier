class Solution {

    public int[] plusOne(int[] a) {
        for (int i = a.length - 1; i >= 0; i--) {
            if (a[i] < 9) {
                // when we do not encounter 9 in the last index.
                a[i]++;
                return a;
            }
            // when we encounter 9 in the last.
            a[i] = 0;
        }
        // array with the length 1 greater than the input array when we encounter only 9\'s in our input array.
        int[] newarray = new int[a.length + 1];
        newarray[0] = 1;
        return newarray;
    }
}
