class Solution {

    public int[] plusOne(int[] arr) {
        int n = arr.length;
        for (int i = n - 1; i >= 0; i--) {
            if (arr[i] < 9) {
                arr[i]++;
                return arr;
            } else {
                arr[i] = 0;
            }
        }
        // if all elements in the array is 9 we have to increase the array size by 1
        int[] newArr = new int[n + 1];
        //set first element to 1
        newArr[0] = 1;
        return newArr;
    }
}
