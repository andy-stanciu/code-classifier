class Solution {

    public int[] plusOne(int[] digits) {
        int l = digits.length;
        digits[l - 1]++;
        for (int i = l - 1; i > 0; i--) {
            if (digits[i] > 9) {
                digits[i] = 0;
                digits[i - 1]++;
            }
        }
        if (digits[0] == 10) {
            l++;
            int[] ans = new int[l];
            ans[0] = 1;
            return ans;
        } else
            return digits;
    }
}
