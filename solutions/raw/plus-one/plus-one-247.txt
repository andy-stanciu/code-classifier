class Solution {

    public int[] plusOne(int[] d) {
        int n = d.length;
        for (int i = n - 1; i >= 0; i--) {
            if (d[i] == 9) {
                d[i] = 0;
            } else {
                d[i]++;
                return d;
            }
        }
        int[] a = new int[n + 1];
        a[0] = 1;
        return a;
    }
}
