class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        // if(digits[n-1]!=9){
        //     digits[n-1]=digits[n-1]+1;
        //     return digits;
        // } else{
        //     digits[n-1]=0;
        // }
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] != 9) {
                digits[i] = digits[i] + 1;
                return digits;
            } else {
                digits[i] = 0;
            }
        }
        int[] arr = new int[n + 1];
        arr[0] = 1;
        return arr;
    }
}
