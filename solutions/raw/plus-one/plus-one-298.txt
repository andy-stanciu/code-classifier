class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        int j = n - 1;
        while (j > 0 && digits[j] == 9) {
            j--;
        }
        if (digits[j] == 9) {
            int[] arr = new int[n + 1];
            arr[0] = 1;
            return arr;
        } else {
            digits[j]++;
            for (int i = j + 1; i < n; i++) {
                digits[i] = 0;
            }
            return digits;
        }
    }
}
