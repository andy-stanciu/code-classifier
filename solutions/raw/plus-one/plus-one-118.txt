class Solution {

    public int[] plusOne(int[] digits) {
        int len = digits.length - 1;
        for (int i = len; i >= 0; i--) {
            //start checking from end of the array
            if (digits[i] == 9) {
                // If the digits are 9 then they are sure to be 10 and digit 0 will take it\'s palce
                digits[i] = 0;
            } else {
                // The first time a digit is NOT 9, add 1 to it and return from there
                digits[i] = digits[i] + 1;
                return digits;
            }
        }
        if (digits[0] == 0) {
            // If all the digits were 9 then they all will get converted to 0 e.g. 999 to 000 as in 1000
            //initialize with a new array with 1 extra place to put a \'1\' in beginning (len was already digits\' length -1 so.. now it\'s len+2)
            digits = new int[len + 2];
            //make the 1st value \'1\'
            digits[0] = 1;
        }
        return digits;
    }
}
