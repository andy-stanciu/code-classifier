class Solution {

    public int[] plusOne(int[] digits) {
        return add1(digits, digits.length - 1);
    }

    private int[] add1(int[] digits, int index) {
        if (index < 0) {
            int[] newArray = new int[digits.length + 1];
            newArray[0] = 1;
            return newArray;
        }
        if (digits[index] < 9) {
            digits[index]++;
            return digits;
        }
        digits[index] = 0;
        return add1(digits, index - 1);
    }
}
