class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] == 9) {
                digits[i] = 0;
                if (i == 0) {
                    int[] newArr = new int[n + 1];
                    System.arraycopy(digits, 0, newArr, 1, n);
                    newArr[0] = 1;
                    return newArr;
                }
            } else {
                digits[i]++;
                return digits;
            }
        }
        return digits;
    }
}
