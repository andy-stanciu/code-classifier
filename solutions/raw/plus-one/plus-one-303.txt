class Solution {

    public int[] plusOne(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // Set current digit to 0 if it\'s 9
            digits[i] = 0;
        }
        // If all the digits are 9, we need a larger array
        int[] newNumber = new int[digits.length + 1];
        // Set the first digit to 1, others are 0 by default
        newNumber[0] = 1;
        return newNumber;
    }
}
