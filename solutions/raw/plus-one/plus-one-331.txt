class Solution {

    public int[] plusOne(int[] digits) {
        for (int j = (digits.length - 1); j >= 0; j--) {
            if (digits[j] < 9) {
                digits[j] = digits[j] + 1;
                break;
            } else if ((digits[j] == 9) && (j != 0)) {
                digits[j] = 0;
            } else if ((digits[j] == 9) && (j == 0)) {
                int[] arr = new int[digits.length + 1];
                arr[0] = 1;
                arr[1] = 0;
                return arr;
            }
        }
        return digits;
    }
}
