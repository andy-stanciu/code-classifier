class Solution {

    public int[] plusOne(int[] digits) {
        if (digits[digits.length - 1] != 9) {
            digits[digits.length - 1]++;
        } else {
            int index = digits.length - 1;
            while (index >= 0 && digits[index] == 9) {
                digits[index--] = 0;
            }
            if (index == -1) {
                int[] ans = new int[digits.length + 1];
                ans[0] = 1;
                return ans;
            } else {
                digits[index]++;
            }
        }
        return digits;
    }
}
