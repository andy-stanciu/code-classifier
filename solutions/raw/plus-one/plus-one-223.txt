class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        // Traverse from the last digit to the first one
        for (int i = n - 1; i >= 0; i--) {
            // If the digit is less than 9, just increment and return the result
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the digit is 9, set it to 0 (handle carry over)
            digits[i] = 0;
        }
        // If we are here, it means all digits were 9, e.g., 999 -> 1000
        // Create a new array with an extra leading 1
        int[] newDigits = new int[n + 1];
        newDigits[0] = 1;
        return newDigits;
    }
}
