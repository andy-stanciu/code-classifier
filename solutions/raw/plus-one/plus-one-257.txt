class Solution {

    public int[] plusOne(int[] digits) {
        final int base = 10;
        final int digit_size = digits.length;
        int carry = 1;
        for (int i = digit_size - 1; i >= 0; i--) {
            int current_digit_sum = digits[i] + carry;
            carry = current_digit_sum - base + 1;
            if (carry > 0) {
                // continue looping
                digits[i] = 0;
            } else {
                // havent reached end and no need to continue
                digits[i] = current_digit_sum;
                return digits;
            }
        }
        int[] new_digits = new int[digit_size + 1];
        System.arraycopy(digits, 0, new_digits, 1, digit_size);
        new_digits[0] = 1;
        return new_digits;
    }
}
