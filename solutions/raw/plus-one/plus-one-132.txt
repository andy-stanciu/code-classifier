class Solution {

    public int[] plusOne(int[] arr) {
        for (int c = arr.length - 1; c >= 0; c--) {
            if (arr[c] != 9) {
                arr[c]++;
                return arr;
            } else {
                arr[c] = 0;
            }
        }
        int[] result = new int[arr.length + 1];
        result[0] = 1;
        return result;
    }
}
