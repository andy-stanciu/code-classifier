class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        if (digits[n - 1] < 9) {
            digits[n - 1] += 1;
            return digits;
        }
        int i = n - 1;
        while (digits[i] == 9 && i > 0) {
            digits[i] = 0;
            i--;
        }
        if (digits[i] == 9) {
            int[] result = new int[n + 1];
            result[0] = 1;
            return result;
        } else
            digits[i] += 1;
        return digits;
    }
}
