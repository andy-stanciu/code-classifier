class Solution {

    public int[] plusOne(int[] digits) {
        // Traverse the array from the last digit
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                // If the current digit is less than 9, just increment it and return the array
                digits[i]++;
                return digits;
            }
            // If the current digit is 9, set it to 0 and continue to the next digit
            digits[i] = 0;
        }
        // If all the digits are 9, we need a new array with an extra digit
        int[] newDigits = new int[digits.length + 1];
        // Set the first digit to 1 (e.g., [9,9,9] -> [1,0,0,0])
        newDigits[0] = 1;
        return newDigits;
    }
}
