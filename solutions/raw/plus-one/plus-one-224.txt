class Solution {

    public int[] plusOne(int[] digits) {
        int size = digits.length;
        if (digits[size - 1] >= 0 && digits[size - 1] < 9)
            digits[size - 1]++;
        else if (digits[size - 1] == 9) {
            if (size == 1) {
                digits = new int[2];
                digits[0] = 1;
                digits[1] = 0;
            } else {
                digits[size - 1] = 0;
                for (int i = size - 2; i >= 0; i--) {
                    if (digits[i] < 9) {
                        digits[i]++;
                        return digits;
                    }
                    digits[i] = 0;
                }
                int[] newDigits = new int[size + 1];
                newDigits[0] = 1;
                return newDigits;
            }
        }
        return digits;
    }
}
