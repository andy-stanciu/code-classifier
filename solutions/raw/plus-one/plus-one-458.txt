class Solution {

    public int[] plusOne(int[] digits) {
        if (digits == null)
            return null;
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        int[] ne = new int[n + 1];
        ne[0] = 1;
        return ne;
    }
}
