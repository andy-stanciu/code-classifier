class Solution {

    public int[] plusOne(int[] digits) {
        int carry = 1;
        for (int i = digits.length - 1; i >= 0; --i) {
            int sum = digits[i] + carry;
            carry = sum / 10;
            if (carry > 0) {
                // remainder
                digits[i] = sum % 10;
            } else {
                digits[i] = sum;
                // we can cut the iteration short since the other digits will stay the same
                return digits;
            }
        }
        if (carry == 0) {
            return digits;
        }
        int[] ans = new int[digits.length + 1];
        for (int i = digits.length - 1; i >= 0; --i) {
            ans[i + 1] = digits[i];
        }
        ans[0] = carry;
        return ans;
    }
}
