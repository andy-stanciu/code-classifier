class Solution {

    public int[] plusOne(int[] digits) {
        for (//if all elements are not 9 if all elements  are == 9 than line no :- 14
        //if all elements are not 9 if all elements  are == 9 than line no :- 14
        //if all elements are not 9 if all elements  are == 9 than line no :- 14
        int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i] += 1;
                return digits;
            }
            digits[i] = 0;
        }
        //Other will default zero
        int[] newNumber = new int[digits.length + 1];
        newNumber[0] = 1;
        return newNumber;
    }
}
