class Solution {

    public int[] plusOne(int[] digits) {
        final int len = digits.length;
        for (int index = len - 1; index >= 0; --index) {
            if (digits[index] != 9) {
                // if element is not equal to 9, then increment it by 1 and return it
                digits[index] += 1;
                return digits;
            }
            // else if element is 9, then make that element as 0(zero)
            digits[index] = 0;
        }
        //  for cases where all elements are 9, then we have to create a new array of 1 larger length than the original array
        final int[] result = new int[len + 1];
        // making the 1st element as 1 and rest will be 0 as the default value of int is 0, so all elements will be 0 except for 1st element which will be 1
        result[0] = 1;
        return result;
    }
}
