class Solution {

    public int[] plusOne(int[] digits) {
        // lets focus on traversing backwards because we want to increment number from behind
        for (int i = digits.length - 1; i >= 0; i--) {
            if (//if last digit of no is less then 9 so simply increase it by 1
            digits[i] < 9) {
                digits[i]++;
                // and we are done here coz eg 123 we have made 3 as 4
                return digits;
            }
            // else make all digits 0 , eg like 999 ans should be 1000 so three 9\'s converted to 3 zeros
            digits[i] = 0;
        }
        // in cases of 999 we want to increase length of array
        digits = new int[digits.length + 1];
        // make the initial digit 1 so we already had 000 now we have addec 1 so , 1000
        digits[0] = 1;
        return digits;
    }
}
