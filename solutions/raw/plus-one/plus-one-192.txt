class Solution {

    public int[] plusOne(int[] digits) {
        // convert to String
        String s = "";
        for (int i = 0; i < digits.length; i++) {
            s += String.valueOf(digits[i]);
        }
        // do the maths while in big int
        BigInteger bigInt = new BigInteger(s);
        BigInteger num = bigInt.add(BigInteger.ONE);
        System.out.println(num);
        // back to String
        String valueString = String.valueOf(num);
        System.out.println(valueString);
        // back to integer array
        int[] digitArray = new int[valueString.length()];
        for (int i = 0; i < valueString.length(); i++) {
            digitArray[i] = Character.getNumericValue(valueString.charAt(i));
        }
        return digitArray;
    }
}
