class Solution {

    public int[] plusOne(int[] digits) {
        // Start from the last digit and work backwards
        for (int i = digits.length - 1; i >= 0; i--) {
            // If the current digit is less than 9, increment it and return the array
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the current digit is 9, set it to 0 (carry over to the next digit)
            digits[i] = 0;
        }
        // If all digits were 9, we need an additional digit at the beginning
        int[] ans = new int[digits.length + 1];
        // Set the first digit to 1, all others default to 0
        ans[0] = 1;
        return ans;
    }
}
