class Solution {

    public int[] plusOne(int[] digits) {
        int[] Array = new int[digits.length + 1];
        for (int i = 1; i < Array.length; ++i) {
            Array[i] = digits[i - 1];
        }
        for (int i = digits.length - 1; i >= 0; --i) {
            Array[i + 1] = ++digits[i];
            if (digits[i] == 10) {
                if ((i - 1) >= 0) {
                    digits[i] = 0;
                    Array[i + 1] = 0;
                } else {
                    Array[0] = 1;
                    Array[1] = 0;
                    return Array;
                }
            } else
                return digits;
        }
        return digits;
    }
}
