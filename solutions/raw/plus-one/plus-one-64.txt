class Solution {

    public int[] plusOne(int[] num) {
        int n = num.length;
        // Check if there is no carry after incrementing the last digit
        if (num[n - 1] + 1 != 10) {
            num[n - 1]++;
            return num;
        }
        // If there is a carry, propagate it to the higher digits
        int carry = 1;
        for (int i = n - 1; i >= 0; i--) {
            num[i] += carry;
            carry = num[i] / 10;
            num[i] %= 10;
            // If no more carry, break the loop
            if (carry == 0) {
                break;
            }
        }
        // If there is still a carry after processing all digits, create a new array
        if (carry == 1) {
            int[] result = new int[n + 1];
            result[0] = 1;
            System.arraycopy(num, 0, result, 1, n);
            return result;
        }
        return num;
    }
}
