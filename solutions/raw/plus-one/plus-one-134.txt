class Solution {

    public int[] plusOne(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            } else
                // i.e digits[0]==0;
                digits[i] = 0;
        }
        int[] plus = new int[digits.length + 1];
        plus[0] = 1;
        return plus;
    }
}
// traversing from right to left , if there is any element less than 9 increment it by 1 and return array
// else  assign 0 at that position ( i.e if digits[i] == 9)
// when they all becomes 0 , for example [9,9,9] will becomes [0,0,0] but after adding 1 it should have become //[1,0,0,0] , so we creating new array with +1 length to add 1 at first position of newly created array
