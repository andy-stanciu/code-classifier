class Solution {

    public int[] plusOne(int[] digits) {
        String num = "";
        for (int i = 0; i < digits.length; i++) {
            num = num + digits[i];
        }
        BigInteger bi = new BigInteger(num);
        char[] res = String.valueOf(bi.add(BigInteger.ONE)).toCharArray();
        int[] result = new int[res.length];
        for (int i = 0; i < res.length; i++) {
            result[i] = Character.getNumericValue(res[i]);
        }
        return result;
    }
}
