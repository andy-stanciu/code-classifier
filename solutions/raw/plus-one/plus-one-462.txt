class Solution {

    public int[] plusOne(int[] digits) {
        int carry = 0;
        int n = digits.length;
        digits[n - 1] += 1;
        for (int i = n - 1; i >= 0; i -= 1) {
            int sum = digits[i] + carry;
            if (sum >= 10) {
                carry = 1;
                digits[i] = sum % 10;
            } else {
                digits[i] = sum;
                carry = 0;
            }
        }
        if (carry == 0) {
            return digits;
        } else {
            int[] res = new int[n + 1];
            res[0] = 1;
            int index = 1;
            for (int i = 0; i < n; i += 1) {
                res[index] = digits[i];
                index += 1;
            }
            return res;
        }
    }
}
