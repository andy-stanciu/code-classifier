class Solution {

    public int[] plusOne(int[] digits) {
        int originalArraySize = digits.length;
        int[] result = digits.clone();
        int[] resultWithLeadingOne;
        int i = originalArraySize - 1;
        while (true) {
            boolean carryOne = false;
            if (digits[i] == 9) {
                carryOne = true;
                result[i] = 0;
            } else
                result[i] += 1;
            i--;
            if (carryOne && i < 0) {
                resultWithLeadingOne = new int[originalArraySize + 1];
                resultWithLeadingOne[0] = 1;
                for (int j = 0; j < originalArraySize; j++) {
                    resultWithLeadingOne[j + 1] = result[j];
                }
                return resultWithLeadingOne;
            }
            if (!carryOne)
                break;
        }
        return result;
    }
}
