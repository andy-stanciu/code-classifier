class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        // Start from the least significant digit and move leftwards
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                // If the current digit is less than 9, we can just add 1 and return the array
                digits[i]++;
                return digits;
            }
            // If the digit is 9, set it to 0 and continue the loop to handle the carry
            digits[i] = 0;
        }
        // If all digits were 9 (like 999 + 1 = 1000), we need to create a new array
        int[] newDigits = new int[n + 1];
        // Set the most significant digit to 1, rest will be 0
        newDigits[0] = 1;
        return newDigits;
    }
}
