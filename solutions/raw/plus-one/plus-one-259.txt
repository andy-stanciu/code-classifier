class Solution {

    public int[] plusOne(int[] digits) {
        int n = digits.length;
        // Start from the least significant digit and work backwards
        for (int i = n - 1; i >= 0; i--) {
            // Increment the current digit
            digits[i]++;
            // If the increment doesn\'t result in a carry, we are done
            if (digits[i] < 10) {
                return digits;
            }
            // Otherwise, set the current digit to 0 (carry over to the next digit)
            digits[i] = 0;
        }
        // If all digits were 9, we need to add an extra digit at the front (carry case)
        int[] result = new int[n + 1];
        // All other elements are already 0
        result[0] = 1;
        return result;
    }
}
