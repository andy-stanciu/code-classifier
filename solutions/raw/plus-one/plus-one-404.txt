class Solution {

    public int[] plusOne(int[] digits) {
        digits = carryDecider(digits, digits.length - 1);
        return digits;
    }

    private int[] carryDecider(int[] digits, int index) {
        if (digits[index] != 9) {
            digits[index] = digits[index] + 1;
            return digits;
        } else {
            digits[index] = 0;
            if (index > 0) {
                return carryDecider(digits, index - 1);
            } else {
                int[] newDigits = new int[digits.length + 1];
                newDigits[0] = 1;
                for (int i = 1; i < newDigits.length; i++) {
                    newDigits[i] = digits[i - 1];
                }
                return newDigits;
            }
        }
    }
}
