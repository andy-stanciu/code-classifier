class Solution {

    public int[] plusOne(int[] digits) {
        if (digits[digits.length - 1] < 9) {
            digits[digits.length - 1]++;
            return digits;
        } else {
            for (int i = digits.length - 1; i >= 0; i--) {
                digits[i]++;
                if (digits[i] > 9)
                    digits[i] = digits[i] % 10;
                else
                    break;
            }
            if (digits[0] == 0) {
                int[] x = new int[digits.length + 1];
                x[0] = 1;
                return x;
            }
            return digits;
        }
    }
}
