class Solution {

    public int[] plusOne(int[] digits) {
        // adding the intial +1
        int remainder = 1;
        for (int i = digits.length - 1; i >= 0; i--) {
            int newNumber = digits[i] + remainder;
            int newDigit = newNumber % 10;
            digits[i] = newDigit;
            remainder = newNumber >= 10 ? 1 : 0;
            // if remainder is 0 no need to continue the loop
            if (remainder == 0) {
                return digits;
            }
        }
        // Create a new array with one extra space
        int[] newArray = new int[digits.length + 1];
        // Insert the new element at the 0th position
        newArray[0] = 1;
        // Copy the original array elements to the new array
        System.arraycopy(digits, 0, newArray, 1, digits.length);
        return newArray;
    }
}
