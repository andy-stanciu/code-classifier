class Solution {

    public int[] plusOne(int[] digits) {
        // create a pointer and a variable carry
        int j = digits.length - 1;
        while (j >= 0) {
            // if digit is not 9 increment and return array else make it 0
            if (digits[j] != 9) {
                digits[j] = digits[j] + 1;
                return digits;
            } else {
                digits[j] = 0;
                j--;
            }
        }
        // create new array of size+1 and assign first digit 1 else 0
        int[] expandedDigits = new int[digits.length + 1];
        expandedDigits[0] = 1;
        // return expanded array
        return expandedDigits;
    }
}
