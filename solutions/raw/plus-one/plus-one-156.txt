class Solution {

    public int[] plusOne(int[] digits) {
        int[] a;
        if (digits[digits.length - 1] != 9) {
            digits[digits.length - 1]++;
            return digits;
        }
        int i = digits.length - 1;
        int carry = 1;
        while (i > -1) {
            int t = digits[i] + carry;
            digits[i] = t % 10;
            i--;
            carry = t / 10;
        }
        if (carry > 0) {
            a = new int[digits.length + 1];
            for (int j = 0; j < digits.length; j++) {
                a[j + 1] = digits[j];
            }
            a[0] = carry;
        } else {
            a = new int[digits.length];
            for (int j = 0; j < digits.length; j++) {
                a[j] = digits[j];
            }
        }
        return a;
    }
}
