class Solution {

    public int[] plusOne(int[] digits) {
        int carryforward = 0;
        int value = digits[digits.length - 1] + 1;
        if (value > 9) {
            carryforward = value / 10;
            value = value % 10;
            digits[digits.length - 1] = value;
        } else {
            digits[digits.length - 1] = value;
        }
        if (carryforward > 0) {
            for (int i = digits.length - 2; i >= 0; i--) {
                int val = digits[i] + carryforward;
                if (val > 9) {
                    carryforward = val / 10;
                    val = val % 10;
                    digits[i] = val;
                } else {
                    carryforward = 0;
                    digits[i] = val;
                }
            }
        }
        if (carryforward > 0) {
            int[] result = new int[digits.length + 1];
            result[0] = 1;
            for (int i = 0; i < digits.length; i++) {
                result[i + 1] = digits[i];
            }
            return result;
        }
        return digits;
    }
}
