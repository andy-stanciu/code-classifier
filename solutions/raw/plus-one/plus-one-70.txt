class Solution {

    public int[] plusOne(int[] digits) {
        int carry = 1;
        for (int i = digits.length - 1; i >= 0; i--) {
            digits[i] += carry;
            if (// early return
            digits[i] != 10)
                return digits;
            digits[i] = 0;
        }
        //if upper return statement didn\'t run, it means [9],[9,9] type of test case is present.
        int[] ans = new int[digits.length + 1];
        //As other elements will be 0, therefore only updating the 0th element
        ans[0] = 1;
        return ans;
    }
}
