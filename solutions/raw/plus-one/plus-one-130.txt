class Solution {

    public int[] plusOne(int[] digits) {
        // Check if the first digit is 9, indicating a special case where the number is all 9s.
        if (digits[0] == 9) {
            // Check if all other digits are also 9s.
            boolean allNines = true;
            for (int i = 1; i < digits.length; i++) {
                if (digits[i] != 9) {
                    allNines = false;
                    break;
                }
            }
            // If all digits are 9s, create a new array with one extra digit and return it with the first digit set to 1.
            if (allNines) {
                int[] num = new int[digits.length + 1];
                Arrays.fill(num, 0);
                num[0] = 1;
                return num;
            }
        }
        // Start from the rightmost digit and add 1 to it.
        int i = digits.length - 1;
        digits[i] = digits[i] + 1;
        // Handle carry propagation if the current digit becomes 10.
        while (digits[i] == 10) {
            digits[i] = 0;
            digits[i - 1] = digits[i - 1] + 1;
            i--;
        }
        // Return the updated digits array.
        return digits;
    }
}
