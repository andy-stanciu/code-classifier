class Solution {

    public int[] plusOne(int[] digits) {
        int i = 0;
        int[] novaMatriz = new int[digits.length + 1];
        int numeroAtual = 0;
        for (i = digits.length - 1; i >= 0; i--) {
            numeroAtual = digits[i];
            if (numeroAtual + 1 != 10) {
                digits[i] = numeroAtual + 1;
                return digits;
            }
            digits[i] = 0;
        }
        novaMatriz[0] = 1;
        return novaMatriz;
    }
}
