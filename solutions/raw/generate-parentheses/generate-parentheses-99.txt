class Solution {

    private List<String> permutations = new ArrayList<>();

    public List<String> generateParenthesis(int n) {
        generateAcceptablePermutations("", n, 0);
        return permutations;
    }

    private void generateAcceptablePermutations(String oldString, int n, int openBrackets) {
        if (n > 0) {
            generateAcceptablePermutations(oldString + "(", n - 1, openBrackets + 1);
        }
        if (openBrackets > 0) {
            generateAcceptablePermutations(oldString + ")", n, openBrackets - 1);
        }
        if (n == 0 && openBrackets == 0) {
            permutations.add(oldString);
        }
    }
}
