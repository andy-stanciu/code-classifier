class Solution {

    ArrayList<String> ans = new ArrayList<>();

    int n;

    void generate(String s, int o, int c) {
        if (o > n || c > n)
            return;
        if ((o + c) == n * 2) {
            ans.add(s);
            return;
        }
        generate(s + "(", o + 1, c);
        if (c < o)
            generate(s + ")", o, c + 1);
    }

    public List<String> generateParenthesis(int n) {
        this.n = n;
        generate("", 0, 0);
        return ans;
    }
}
