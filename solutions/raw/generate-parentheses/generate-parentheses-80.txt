class Solution {

    List<String> res = new ArrayList<>();

    public List<String> generateParenthesis(int n) {
        paranths(n, n, new StringBuilder());
        return res;
    }

    public void paranths(int open, int closed, StringBuilder curr) {
        if (open == 0 && closed == 0) {
            res.add(curr.toString());
            return;
        }
        if (open > 0) {
            curr.append("(");
            paranths(open - 1, closed, curr);
            curr.deleteCharAt(curr.length() - 1);
        }
        if (closed > open) {
            curr.append(")");
            paranths(open, closed - 1, curr);
            curr.deleteCharAt(curr.length() - 1);
        }
    }
}
