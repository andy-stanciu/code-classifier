class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> list = new ArrayList<String>();
        return br(list, "", n, n);
    }

    public static List<String> br(List<String> list, String s, int l, int r) {
        if (l > r)
            return list;
        if (l < 0)
            return list;
        if (l == 0 && r == 0) {
            list.add(s);
            return list;
        }
        list = br(list, s + "(", l - 1, r);
        list = br(list, s + ")", l, r - 1);
        return list;
    }
}
