class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        helper(n, 0, 0, "", result);
        return result;
    }

    private void helper(int n, int open, int close, String current, List<String> result) {
        // stop condition
        if (current.length() == n * 2) {
            result.add(current);
            return;
        }
        // check if we can add open parentheses
        if (open < n) {
            helper(n, open + 1, close, current + "(", result);
        }
        // check if we can add close parentheses
        if (close < open) {
            helper(n, open, close + 1, current + ")", result);
        }
    }
}
