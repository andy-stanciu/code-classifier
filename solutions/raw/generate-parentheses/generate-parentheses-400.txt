class Solution {

    int numberOfBrackets;

    List<String> ans;

    public List<String> generateParenthesis(int n) {
        numberOfBrackets = n * 2;
        ans = new ArrayList<>();
        generateParenthesis(0, 0, 0, "");
        return ans;
    }

    public void generateParenthesis(int i, int open, int close, String str) {
        if (i == numberOfBrackets) {
            ans.add(str);
            return;
        }
        if (!(open == numberOfBrackets / 2)) {
            generateParenthesis(i + 1, open + 1, close, str + "(");
        }
        if (!(open == close)) {
            generateParenthesis(i + 1, open, close + 1, str + ")");
        }
        return;
    }
}
