class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        generate("", result, 0, 0, n);
        return result;
    }

    private static void generate(String current, List<String> result, int open, int close, int n) {
        if (current.length() == n * 2) {
            if (!result.contains(current))
                result.add(current);
            return;
        }
        if (open < n)
            generate(current + "(", result, open + 1, close, n);
        if (close < open)
            generate(current + ")", result, open, close + 1, n);
    }
}
