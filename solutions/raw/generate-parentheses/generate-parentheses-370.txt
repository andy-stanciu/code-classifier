class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        generate("", 0, 0, n, result);
        return result;
    }

    private void generate(String value, int open, int closed, int n, List<String> result) {
        if (open >= n && closed >= n) {
            //if(!result.contains(value))
            result.add(value);
            return;
        }
        if (open < n) {
            generate(value + "(", open + 1, closed, n, result);
        }
        if (closed < n && closed < open) {
            generate(value + ")", open, closed + 1, n, result);
        }
    }
}
