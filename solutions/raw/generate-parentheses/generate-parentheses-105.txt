class Solution {

    public List<String> generateParenthesis(int n) {
        return helper("(", n - 1, n);
    }

    private ArrayList<String> helper(String s, int opn, int cls) {
        ArrayList<String> lst = new ArrayList();
        if (opn == 0) {
            lst.add(s + ")".repeat(cls));
            return lst;
        }
        if (cls > opn) {
            for (String i : helper(s + ")", opn, cls - 1)) {
                lst.add(i);
            }
        }
        for (String i : helper(s + "(", opn - 1, cls)) {
            lst.add(i);
        }
        return lst;
    }
}
