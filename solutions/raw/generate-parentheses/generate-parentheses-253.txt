public class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        generateParenthesis(n, 0, 0, "", result);
        return result;
    }

    private void generateParenthesis(int n, int open, int close, String ans, List<String> result) {
        if (open == n && close == n) {
            result.add(ans);
            return;
        }
        if (open > n || close > open)
            return;
        generateParenthesis(n, open + 1, close, ans + "(", result);
        generateParenthesis(n, open, close + 1, ans + ")", result);
    }
}
