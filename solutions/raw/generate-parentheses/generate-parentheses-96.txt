class Solution {

    public List<String> generateParenthesis(int open, int close, int n) {
        List<String> newOutput = new ArrayList<>();
        if (open == close) {
            if (close == n) {
                return List.of("");
            }
            List<String> output = generateParenthesis(open + 1, close, n);
            for (String out : output) {
                newOutput.add("(" + out);
            }
            return newOutput;
        }
        if (open < n) {
            List<String> outputOpen = generateParenthesis(open + 1, close, n);
            for (String out : outputOpen) {
                newOutput.add("(" + out);
            }
        }
        List<String> outputClose = generateParenthesis(open, close + 1, n);
        for (String out : outputClose) {
            newOutput.add(")" + out);
        }
        return newOutput;
    }

    public List<String> generateParenthesis(int n) {
        return generateParenthesis(0, 0, n);
    }
}
