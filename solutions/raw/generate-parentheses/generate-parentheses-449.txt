class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        backTrack(n, 0, 0, "", result);
        return result;
    }

    private void backTrack(int n, int openN, int closedN, String cur, List<String> result) {
        //if open parentheses count  = closed parentheses count in current string is already n
        if (openN == n && closedN == n) {
            result.add(cur);
            return;
        }
        //if open count is less than n -  we can add one open parentheses
        if (openN < n) {
            String s = cur.concat("(");
            backTrack(n, openN + 1, closedN, s, result);
        }
        //if closed count is less than open - we can add one close parentheses
        if (closedN < openN) {
            String s = cur.concat(")");
            backTrack(n, openN, closedN + 1, s, result);
        }
    }
}
