class Solution {

    public void generator(int n, int openCount, int closeCount, int index, String str, List<String> answer) {
        if (index == (2 * n)) {
            answer.add(str);
            return;
        }
        if (openCount < n)
            generator(n, openCount + 1, closeCount, index + 1, str + "(", answer);
        if (closeCount < openCount && closeCount <= n)
            generator(n, openCount, closeCount + 1, index + 1, str + ")", answer);
        return;
    }

    public List<String> generateParenthesis(int n) {
        List<String> list = new ArrayList<>();
        generator(n, 0, 0, 0, "", list);
        return list;
    }
}
