class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> list = new ArrayList<String>();
        setParanthesis(n, 0, 0, "", list);
        return list;
    }

    public void setParanthesis(int n, int left, int right, String s, List<String> al) {
        if (2 * n == s.length()) {
            al.add(s);
            return;
        }
        if (left < n)
            setParanthesis(n, left + 1, right, s + "(", al);
        if (right < left)
            setParanthesis(n, left, right + 1, s + ")", al);
    }
}
