class Solution {

    List<String> output;

    public List<String> generateParenthesis(int n) {
        output = new ArrayList<>();
        Stack<String> stk = new Stack<>();
        backTrack(n, n, "");
        return output;
    }

    public void backTrack(int open, int close, String current) {
        if (open == 0 & close == 0) {
            output.add(current);
            return;
        }
        if (open < 0 || close < 0) {
            return;
        }
        if (close < open)
            return;
        backTrack(open - 1, close, current + "(");
        backTrack(open, close - 1, current + ")");
    }
}
