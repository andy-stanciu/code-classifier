class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> res = new LinkedList<>();
        generate(res, "(", n - 1, n);
        return res;
    }

    private void generate(List<String> result, String current, int open, int closed) {
        if (open <= 0 && closed <= 0) {
            result.add(current);
            return;
        }
        if (open > 0)
            generate(result, current.concat("("), open - 1, closed);
        if (open < closed) {
            generate(result, current.concat(")"), open, closed - 1);
        }
        return;
    }
}
