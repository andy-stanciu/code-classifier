class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> ans = new ArrayList<>();
        solve(ans, n, n, "");
        return ans;
    }

    public void solve(List<String> ans, int open, int close, String s) {
        if (open == 0) {
            for (int i = 0; i < close; i++) {
                s += ")";
            }
            ans.add(s);
            return;
        }
        if (open == close) {
            solve(ans, open - 1, close, s + "(");
        }
        if (open < close) {
            solve(ans, open - 1, close, s + "(");
            solve(ans, open, close - 1, s + ")");
        }
    }
}
