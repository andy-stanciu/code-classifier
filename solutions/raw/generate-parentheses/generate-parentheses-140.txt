class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> ans = new ArrayList<>();
        addParentheses(0, 0, "", n, ans);
        return ans;
    }

    public void addParentheses(int opening, int colosing, String s, int n, List<String> ans) {
        if (opening == colosing && opening + colosing == n * 2) {
            ans.add(s);
            return;
        }
        if (opening < n) {
            addParentheses(opening + 1, colosing, s + "(", n, ans);
        }
        if (colosing < opening) {
            addParentheses(opening, colosing + 1, s + ")", n, ans);
        }
    }
}
