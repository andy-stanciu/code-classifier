class Solution {

    public void solve(int open, int close, String output, List<String> arr) {
        if (open == 0 && close == 0) {
            arr.add(output);
        }
        if (open != 0) {
            String op1 = output + "(";
            solve(open - 1, close, op1, arr);
        }
        if (close > open) {
            String op2 = output + ")";
            solve(open, close - 1, op2, arr);
        }
    }

    public List<String> generateParenthesis(int n) {
        ArrayList<String> arr = new ArrayList<>();
        int open = n;
        int close = n;
        String output = "";
        solve(open, close, output, arr);
        return arr;
    }
}
