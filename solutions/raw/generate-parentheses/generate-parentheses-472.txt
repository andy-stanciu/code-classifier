class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        generateParentheses(n, 0, 0, result, new StringBuilder());
        return result;
    }

    public void generateParentheses(int n, int open, int close, List<String> ll, StringBuilder current) {
        if (open == n && close == n) {
            ll.add(current.toString());
            return;
        }
        if (open < n) {
            current.append("(");
            generateParentheses(n, open + 1, close, ll, current);
            current.deleteCharAt(current.length() - 1);
        }
        if (close < open) {
            current.append(")");
            generateParentheses(n, open, close + 1, ll, current);
            current.deleteCharAt(current.length() - 1);
        }
    }
}
