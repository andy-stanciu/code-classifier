class Solution {

    List<String> res = new ArrayList<>();

    public void gen(int n, int l, int r, String s) {
        if (l == n && r == n) {
            res.add(s);
            return;
        }
        if (l < n) {
            gen(n, l + 1, r, s + "(");
        }
        if (r < n && l > r) {
            gen(n, l, r + 1, s + ")");
        }
        return;
    }

    public List<String> generateParenthesis(int n) {
        gen(n, 0, 0, "");
        return res;
    }
}
