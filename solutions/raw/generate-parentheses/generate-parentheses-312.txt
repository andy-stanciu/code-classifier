class Solution {

    public void generate(List<String> result, String patt, int n, int noOfLeft, int open) {
        if (patt.length() == 2 * n) {
            result.add(patt);
            return;
        }
        if (noOfLeft == n) {
            generate(result, patt + ")", n, noOfLeft, open - 1);
        } else {
            if (open != 0)
                generate(result, patt + ")", n, noOfLeft, open - 1);
            generate(result, patt + "(", n, noOfLeft + 1, open + 1);
        }
    }

    public List<String> generateParenthesis(int n) {
        if (n == 0)
            return new ArrayList<String>();
        List<String> res = new ArrayList<String>();
        generate(res, "", n, 0, 0);
        return res;
    }
}
