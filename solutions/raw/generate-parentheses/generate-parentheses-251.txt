class Solution {

    private void helpfindParenthesis(int open, int closed, String curr, List<String> ans) {
        if (open == 0 && closed == 0) {
            ans.add(curr);
            return;
        }
        if (open > 0) {
            helpfindParenthesis(open - 1, closed, curr + "(", ans);
        }
        if (closed > open) {
            helpfindParenthesis(open, closed - 1, curr + ")", ans);
        }
    }

    public List<String> generateParenthesis(int n) {
        List<String> ans = new ArrayList<>();
        helpfindParenthesis(n, n, "", ans);
        return ans;
    }
}
