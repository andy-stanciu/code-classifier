class Solution {

    public List<String> generateParenthesis(int n) {
        if (n == 0) {
            return List.of("");
        } else if (n == 1) {
            return List.of("()");
        }
        List<String> result = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            List<String> solutionsInParenthesis = generateParenthesis(i);
            List<String> solutionsOutOfParenthesis = generateParenthesis(n - 1 - i);
            for (String in : solutionsInParenthesis) {
                for (String out : solutionsOutOfParenthesis) {
                    result.add("(" + in + ")" + out);
                }
            }
        }
        return result;
    }
}
