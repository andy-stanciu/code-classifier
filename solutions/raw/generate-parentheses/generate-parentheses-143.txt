class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> list = new ArrayList<>();
        helper(n, 0, 0, 0, "", list);
        return list;
    }

    private void helper(int n, int idx, int startCount, int endCount, String str, List<String> list) {
        if (startCount < endCount)
            return;
        if (idx >= n * 2) {
            list.add(str);
            return;
        }
        if (startCount < n) {
            helper(n, idx + 1, startCount + 1, endCount, str + "(", list);
        }
        helper(n, idx + 1, startCount, endCount + 1, str + ")", list);
    }
}
