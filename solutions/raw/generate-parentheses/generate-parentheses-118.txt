class Solution {

    public void fun(int n, int left, int right, List<String> li, String str) {
        if (str.length() == n * 2) {
            li.add(str);
        }
        if (left < n) {
            fun(n, left + 1, right, li, str + "(");
        }
        if (right < left) {
            fun(n, left, right + 1, li, str + ")");
        }
    }

    public List<String> generateParenthesis(int n) {
        List<String> li = new ArrayList<>();
        fun(n, 0, 0, li, "");
        return li;
    }
}
