class Solution {

    static void generateparanthissis(List<String> list, String p, int open, int close) {
        if (open == 0 && close == 0) {
            list.add(p);
            return;
        }
        if (open > 0) {
            generateparanthissis(list, p + "(", open - 1, close);
        }
        if (close > 0) {
            if (open < close) {
                generateparanthissis(list, p + ")", open, close - 1);
            }
        }
    }

    public List<String> generateParenthesis(int n) {
        ArrayList<String> list = new ArrayList<>();
        generateparanthissis(list, "", n, n);
        return list;
    }
}
