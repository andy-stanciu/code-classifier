class Solution {

    private void backtrack(int n, int open, int close, String str, List<String> solns) {
        if (open == close && open == n)
            solns.add(str);
        if (open > n || close > open)
            return;
        backtrack(n, open + 1, close, str + "(", solns);
        if (open > close)
            backtrack(n, open, close + 1, str + ")", solns);
    }

    public List<String> generateParenthesis(int n) {
        List<String> solns = new ArrayList<>();
        backtrack(n, 0, 0, "", solns);
        return solns;
    }
}
