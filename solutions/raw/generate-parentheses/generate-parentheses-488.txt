class Solution {

    List<String> list;

    public List<String> generateParenthesis(int n) {
        list = new ArrayList<>();
        String s = "(";
        generate(s, n - 1, n);
        return list;
    }

    public void generate(String s, int opencount, int closecount) {
        if (opencount == 0 && closecount == 0) {
            if (!list.contains(s))
                list.add(s);
            return;
        }
        if (opencount > 0) {
            generate(s + "(", opencount - 1, closecount);
        }
        if (closecount > 0 && opencount < closecount) {
            generate(s + ")", opencount, closecount - 1);
        }
        return;
    }
}
