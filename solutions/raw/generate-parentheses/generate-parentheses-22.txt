class Solution {

    List<String> result = new ArrayList();

    void generate(int open, int close, String ans) {
        if (close == 0) {
            result.add(ans);
            return;
        }
        if (open >= close) {
            generate(open - 1, close, ans + "(");
        } else {
            if (//to check if we still have open parenthesis
            open > 0)
                generate(open - 1, close, ans + "(");
            generate(open, close - 1, ans + ")");
        }
    }

    public List<String> generateParenthesis(int n) {
        generate(n, n, "");
        return result;
    }
}
