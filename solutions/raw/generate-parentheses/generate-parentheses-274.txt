class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> outputList = new ArrayList<String>();
        generatedParenthesis(0, 0, "", n, outputList);
        return outputList;
    }

    public void generatedParenthesis(int i, int j, String s, int n, List<String> output) {
        if (j == n) {
            output.add(s);
            return;
        }
        if (i < n) {
            generatedParenthesis(i + 1, j, s + "(", n, output);
        }
        if (i > j) {
            generatedParenthesis(i, j + 1, s + ")", n, output);
        }
    }
}
