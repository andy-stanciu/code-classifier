class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> allCombinations = new ArrayList<>();
        generateCombinations("", 0, 0, n, allCombinations);
        return allCombinations;
    }

    private static void generateCombinations(String current, int open, int close, int max, List<String> result) {
        if (current.length() == max * 2) {
            result.add(current);
            return;
        }
        if (open < max) {
            generateCombinations(current + "(", open + 1, close, max, result);
        }
        if (close < max && close < open) {
            generateCombinations(current + ")", open, close + 1, max, result);
        }
    }
}
