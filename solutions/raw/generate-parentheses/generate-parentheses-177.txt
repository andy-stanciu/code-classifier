class Solution {

    public List<String> generateParenthesis(int n) {
        ArrayList<String> result = new ArrayList<>();
        fill(0, 0, "", n, result);
        return result;
    }

    public static void fill(int open, int close, String s, int n, List<String> ans) {
        if (open == close && open + close == n * 2) {
            ans.add(s);
            return;
        }
        if (open < n) {
            fill(open + 1, close, s + "(", n, ans);
        }
        if (close < open) {
            fill(open, close + 1, s + ")", n, ans);
        }
    }
}
