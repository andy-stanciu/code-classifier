class Solution {

    static List<String> ans;

    public void print(int open, int closed, int n, String s) {
        if (s.length() == 2 * n) {
            ans.add(s);
            return;
        }
        if (open < n)
            print(open + 1, closed, n, s + "(");
        if (open > closed)
            print(open, closed + 1, n, s + ")");
    }

    public List<String> generateParenthesis(int n) {
        ans = new ArrayList<>();
        print(0, 0, n, "");
        return ans;
    }
}
