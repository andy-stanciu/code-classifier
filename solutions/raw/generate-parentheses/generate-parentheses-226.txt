class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> res = new ArrayList<>();
        generateParanthesis(n, n, "", res);
        return res;
    }

    private void generateParanthesis(int open, int close, String curr, List<String> res) {
        if (open == 0 && close == 0) {
            res.add(curr);
            return;
        }
        if (open > 0) {
            generateParanthesis(open - 1, close, curr + "(", res);
        }
        if (close > open) {
            generateParanthesis(open, close - 1, curr + ")", res);
        }
    }
}
