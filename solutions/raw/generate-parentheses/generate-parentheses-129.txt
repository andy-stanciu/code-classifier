class Solution {

    List<String> helper(int open, int close, String s, List<String> res) {
        if (open == 0 && close == 0) {
            res.add(s);
            return res;
        }
        if (open > 0)
            helper(open - 1, close, s + "(", res);
        if (open < close)
            helper(open, close - 1, s + ")", res);
        return res;
    }

    public List<String> generateParenthesis(int n) {
        return helper(n, n, "", new ArrayList<>());
    }
}
