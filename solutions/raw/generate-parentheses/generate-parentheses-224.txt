class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        int openP = 0, closeP = 0;
        generateParenthesis(n, openP, closeP, "", result);
        return result;
    }

    public void generateParenthesis(int n, int openP, int closeP, String s, List<String> result) {
        if (openP < n) {
            generateParenthesis(n, openP + 1, closeP, s + "(", result);
        }
        if (closeP < openP) {
            generateParenthesis(n, openP, closeP + 1, s + ")", result);
        }
        if (closeP == openP && openP == n) {
            result.add(s);
        }
    }
}
