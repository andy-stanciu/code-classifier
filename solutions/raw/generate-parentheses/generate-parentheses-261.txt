class Solution {

    void gen(List list, String str, int n, int open, int close) {
        if (str.length() == (2 * n)) {
            list.add(str);
            return;
        }
        if (open < n) {
            gen(list, str + "(", n, open + 1, close);
        }
        if (close < open) {
            gen(list, str + ")", n, open, close + 1);
        }
    }

    public List<String> generateParenthesis(int n) {
        List list = new ArrayList();
        gen(list, "", n, 0, 0);
        return list;
    }
}
