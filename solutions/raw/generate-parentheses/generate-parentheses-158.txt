class Solution {

    private List<String> ls = new ArrayList<>();

    public List<String> generateParenthesis(int n) {
        helper(2 * n, 0, 0, "");
        return ls;
    }

    private void helper(int N, int lP, int rP, String soFar) {
        if (soFar.length() == N) {
            ls.add(soFar);
            return;
        }
        if (rP < lP) {
            helper(N, lP, rP + 1, soFar + ")");
        }
        if (lP < N / 2) {
            helper(N, lP + 1, rP, soFar + "(");
        }
    }
}
