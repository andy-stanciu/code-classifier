class Solution {

    private List<String> allParanthesis = new ArrayList<>();

    public List<String> generateParenthesis(int n) {
        generateParenthesis(n, n, new StringBuilder());
        return allParanthesis;
    }

    public void generateParenthesis(int l, int r, StringBuilder combo) {
        if (l == 0 && r == 0) {
            allParanthesis.add(combo.toString());
            return;
        }
        if (l != 0) {
            combo.append("(");
            generateParenthesis(l - 1, r, combo);
            combo.delete(combo.length() - 1, combo.length());
        }
        if (r > l) {
            combo.append(")");
            generateParenthesis(l, r - 1, combo);
            combo.delete(combo.length() - 1, combo.length());
        }
    }
}
