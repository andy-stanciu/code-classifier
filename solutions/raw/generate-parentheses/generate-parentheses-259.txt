class Solution {

    List<String> result = new ArrayList<>();

    String leftBracket = "(";

    String rightBracket = ")";

    public List<String> generateParenthesis(int n) {
        generateParenthesisHelper(n, 0, 0, 0, "");
        return result;
    }

    private void generateParenthesisHelper(int n, int openBracketCount, int closeBracketCount, int currentOpenCount, String brackets) {
        // Base case
        if ((openBracketCount > n) || (closeBracketCount > n))
            return;
        if ((openBracketCount == n) && closeBracketCount == n)
            result.add(brackets);
        generateParenthesisHelper(n, openBracketCount + 1, closeBracketCount, currentOpenCount + 1, brackets + leftBracket);
        if (currentOpenCount > 0)
            generateParenthesisHelper(n, openBracketCount, closeBracketCount + 1, currentOpenCount - 1, brackets + rightBracket);
    }
}
