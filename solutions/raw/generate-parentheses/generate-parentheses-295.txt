class Solution {

    public void generateParenthesisUtil(int open, int close, String temp, List<String> res) {
        if (open == 0 && close == 0) {
            res.add(temp);
            return;
        }
        if (open == close) {
            temp = temp + "(";
            generateParenthesisUtil(open - 1, close, temp, res);
        } else if (open == 0) {
            temp = temp + ")";
            generateParenthesisUtil(open, close - 1, temp, res);
        } else {
            generateParenthesisUtil(open - 1, close, temp + "(", res);
            generateParenthesisUtil(open, close - 1, temp + ")", res);
        }
    }

    public List<String> generateParenthesis(int n) {
        int open = n;
        int close = n;
        List<String> res = new ArrayList<String>();
        generateParenthesisUtil(open, close, "", res);
        return res;
    }
}
