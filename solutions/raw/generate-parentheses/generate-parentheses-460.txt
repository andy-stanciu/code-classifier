class Solution {

    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        findAllValidParenthesis("(", 1, 0, result, n);
        return result;
    }

    void findAllValidParenthesis(String current, int openCount, int closeCount, List<String> result, int n) {
        if (current.length() == 2 * n) {
            result.add(current);
            return;
        }
        if (openCount < n)
            findAllValidParenthesis(current + "(", openCount + 1, closeCount, result, n);
        if (closeCount < openCount)
            findAllValidParenthesis(current + ")", openCount, closeCount + 1, result, n);
    }
}
