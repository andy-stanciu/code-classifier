class Solution {

    List<String> res = new ArrayList<>();

    public List<String> generateParenthesis(int n) {
        StringBuilder sb = new StringBuilder();
        generate(sb, n, n);
        return res;
    }

    private void generate(StringBuilder sb, int open, int close) {
        if (open == 0 && close == 0) {
            res.add(sb.toString());
        }
        if (open > 0) {
            sb.append("(");
            generate(sb, open - 1, close);
            // Backtrack
            sb.deleteCharAt(sb.length() - 1);
        }
        if (close > open) {
            sb.append(")");
            generate(sb, open, close - 1);
            // Backtrack
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
