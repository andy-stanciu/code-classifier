class Solution {

    public void parenthesisGenerator(int open, int close, List<String> list, String s, int n) {
        if (s.length() == 2 * n)
            list.add(s);
        if (open < n)
            parenthesisGenerator(open + 1, close, list, s + "(", n);
        if (open > close)
            parenthesisGenerator(open, close + 1, list, s + ")", n);
    }

    public List<String> generateParenthesis(int n) {
        List<String> list = new ArrayList<>();
        String s = "";
        parenthesisGenerator(0, 0, list, s, n);
        return list;
    }
}
