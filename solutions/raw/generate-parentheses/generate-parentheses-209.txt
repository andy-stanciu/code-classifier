class Solution {

    List<String> ans;

    public void generate(int open, int close, int n, String s) {
        if (s.length() == 2 * n) {
            // we need 2*n parenthesis in the answer
            ans.add(s);
            return;
        }
        // add "(" open cant be greater than n
        if (open < n)
            generate(open + 1, close, n, s + "(");
        // add ")" close cant be greater then open at any stage
        if (close < open)
            generate(open, close + 1, n, s + ")");
    }

    public List<String> generateParenthesis(int n) {
        // reinitialise for running all the test cases
        ans = new ArrayList<>();
        generate(0, 0, n, "");
        return ans;
    }
}
