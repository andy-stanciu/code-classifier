class Solution {

    public List<String> generateParenthesis(int n) {
        ArrayList<String> list = new ArrayList<>();
        generate(list, "", 0, 0, n);
        return list;
    }

    public void generate(List<String> list, String str, int op, int cl, int n) {
        if (str.length() == n * 2) {
            list.add(str);
            return;
        }
        if (op < n) {
            generate(list, str + "(", op + 1, cl, n);
        }
        if (cl < op) {
            generate(list, str + ")", op, cl + 1, n);
        }
    }
}
