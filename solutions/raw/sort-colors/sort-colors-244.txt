class Solution {

    public void sortColors(int[] a) {
        //counting each number of color
        int zeroes = 0, ones = 0, twos = 0;
        for (int i = 0; i < a.length; i++) {
            if (a[i] == 0)
                //counting and setting 0(Red) to save loops down the road
                a[zeroes++] = 0;
            else if (a[i] == 1)
                //counting ones (green)
                ones++;
            else
                //counting twos (blue)
                twos++;
        }
        for (//setting 1s after all the zeroes
        //setting 1s after all the zeroes
        //setting 1s after all the zeroes
        int i = zeroes; i < zeroes + ones; i++) a[i] = 1;
        for (//setting 2s after all the ones
        //setting 2s after all the ones
        //setting 2s after all the ones
        int i = zeroes + ones; i < a.length; i++) a[i] = 2;
    }
}
