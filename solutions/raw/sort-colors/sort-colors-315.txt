class Solution {

    public void sortColors(int[] nums) {
        int[] count = new int[3];
        for (int i = 0; i < nums.length; i++) {
            count[nums[i]]++;
        }
        int k = 0, i = 0;
        while (i < nums.length) {
            for (int j = 0; j < count[k]; j++) {
                nums[i] = k;
                i++;
            }
            k++;
        }
    }
}
