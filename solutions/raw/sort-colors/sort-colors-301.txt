class Solution {

    public void sortColors(int[] nums) {
        int i = 0;
        int j = 0;
        int k = nums.length - 1;
        while (j <= k) {
            if (nums[j] == 1) {
                j++;
            } else if (nums[j] == 2) {
                int temp = nums[k];
                nums[k] = nums[j];
                nums[j] = temp;
                k--;
            } else {
                int temp1 = nums[j];
                nums[j] = nums[i];
                nums[i] = temp1;
                i++;
                j++;
            }
        }
    }
}
