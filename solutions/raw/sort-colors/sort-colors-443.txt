class Solution {

    public void sortColors(int[] nums) {
        int[] temp = { 0, 0, 0 };
        for (int i = 0; i < nums.length; i++) {
            temp[nums[i]]++;
        }
        int cur = 0;
        for (int i = 0; i < temp[0]; i++) {
            nums[cur++] = 0;
        }
        for (int i = 0; i < temp[1]; i++) {
            nums[cur++] = 1;
        }
        for (int i = 0; i < temp[2]; i++) {
            nums[cur++] = 2;
        }
    }
}
