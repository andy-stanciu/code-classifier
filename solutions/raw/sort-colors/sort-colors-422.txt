class Solution {

    public void sortColors(int[] nums) {
        int countOf0 = 0, countOf1 = 0, countOf2 = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                countOf0++;
            } else if (nums[i] == 1) {
                countOf1++;
            } else {
                countOf2++;
            }
        }
        int index = 0;
        // set zeros
        for (int i = 0; i < countOf0; i++) {
            nums[index] = 0;
            index++;
        }
        // set ones
        for (int i = 0; i < countOf1; i++) {
            nums[index] = 1;
            index++;
        }
        // set twos
        for (int i = 0; i < countOf2; i++) {
            nums[index] = 2;
            index++;
        }
    }
}
