class Solution {

    public void sortColors(int[] nums) {
        // Arrays.sort(nums);
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            //add
            pq.offer(nums[i]);
        }
        for (int i = 0; i < nums.length; i++) {
            //poll
            nums[i] = pq.remove();
        }
    }
}
