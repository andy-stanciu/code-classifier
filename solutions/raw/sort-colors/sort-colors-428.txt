class Solution {

    public void sortColors(int[] nums) {
        int a0 = 0;
        int a1 = 0;
        int a2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                a0++;
            }
            if (nums[i] == 1) {
                a1++;
            }
            if (nums[i] == 2) {
                a2++;
            }
        }
        for (int i = 0; i < a0; i++) {
            nums[i] = 0;
        }
        for (int i = a0; i < a0 + a1; i++) {
            nums[i] = 1;
        }
        for (int i = a0 + a1; i < nums.length; i++) {
            nums[i] = 2;
        }
    }
}
