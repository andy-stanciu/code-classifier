class Solution {

    public void sortColors(int[] nums) {
        int r = 0, w = 0, b = 0;
        for (int i : nums) {
            if (i == 0)
                r++;
            if (i == 1)
                w++;
            if (i == 2)
                b++;
        }
        int idx = 0;
        for (int i = 0; i < r; i++) {
            nums[idx++] = 0;
        }
        for (int i = 0; i < w; i++) {
            nums[idx++] = 1;
        }
        for (int i = 0; i < b; i++) {
            nums[idx++] = 2;
        }
    }
}
