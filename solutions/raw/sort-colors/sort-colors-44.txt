class Solution {

    public void sortColors(int[] nums) {
        int[] count = new int[3];
        for (int i = 0; i < nums.length; i++) {
            count[nums[i]]++;
        }
        int idx = 0;
        for (int color = 0; color < 3; color++) {
            for (int j = 0; j < count[color]; j++) {
                nums[idx] = color;
                idx++;
            }
        }
    }
}
