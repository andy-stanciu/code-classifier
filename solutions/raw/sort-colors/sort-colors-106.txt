class Solution {

    public void sortColors(int[] nums) {
        int[] color = new int[3];
        for (int ele : nums) color[ele]++;
        int j = 0;
        for (int x = 0; x < 3; x++) {
            for (int i = 0; i < color[x]; i++) {
                nums[j] = x;
                j++;
            }
        }
    }
}
