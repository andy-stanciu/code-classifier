class Solution {

    public void sortColors(int[] nums) {
        int swap = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i; j < nums.length; j++) {
                if (nums[i] > nums[j]) {
                    swap = nums[j];
                    nums[j] = nums[i];
                    nums[i] = swap;
                }
            }
        }
    }
}
