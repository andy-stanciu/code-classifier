class Solution {

    public void sortColors(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            int temp = 0;
            for (int j = 1; j < nums.length - i; j++) {
                if (nums[j] < nums[j - 1]) {
                    temp = nums[j];
                    nums[j] = nums[j - 1];
                    nums[j - 1] = temp;
                }
            }
        }
    }
}
