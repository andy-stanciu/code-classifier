class Solution {

    public void sortColors(int[] nums) {
        int left = 0;
        int right = 1;
        while (left < nums.length - 1) {
            if (nums[left] > nums[right]) {
                int temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
                left = 0;
                right = 1;
            } else {
                left += 1;
                right += 1;
            }
        }
    }
}
