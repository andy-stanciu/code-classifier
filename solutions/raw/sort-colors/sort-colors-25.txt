class Solution {

    public void sortColors(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {
            //n-1
            for (int j = 0; j < arr.length - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {
                    //swap
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
    }
}
