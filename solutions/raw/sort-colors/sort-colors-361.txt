class Solution {

    public void sortColors(int[] nums) {
        int n = nums.length;
        int zeros = 0, ones = 0, twos = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                zeros++;
            } else if (nums[i] == 1) {
                ones++;
            } else {
                twos++;
            }
        }
        for (int i = 0; i < zeros; i++) {
            nums[i] = 0;
        }
        for (int j = zeros; j < zeros + ones; j++) {
            nums[j] = 1;
        }
        for (int k = zeros + ones; k < n; k++) {
            nums[k] = 2;
        }
    }
}
