class Solution {

    public void sortColors(int[] nums) {
        int leftPt = 0;
        int rightPt = nums.length - 1;
        int iterator = 0;
        while (iterator <= rightPt) {
            if (nums[iterator] == 0) {
                int temp = nums[iterator];
                nums[iterator] = nums[leftPt];
                nums[leftPt] = temp;
                leftPt++;
            }
            if (nums[iterator] == 2) {
                int temp = nums[iterator];
                nums[iterator] = nums[rightPt];
                nums[rightPt] = temp;
                rightPt--;
            } else
                iterator++;
        }
    }
}
