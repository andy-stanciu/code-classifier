class Solution {

    public void sortColors(int[] nums) {
        int s = 0;
        int e = nums.length - 1;
        int m = 0;
        while (m <= e) {
            switch(nums[m]) {
                case 0:
                    {
                        int temp = nums[s];
                        nums[s] = nums[m];
                        nums[m] = temp;
                        m++;
                        s++;
                        break;
                    }
                case 1:
                    {
                        m++;
                        break;
                    }
                case 2:
                    {
                        int temp = nums[e];
                        nums[e] = nums[m];
                        nums[m] = temp;
                        e--;
                        break;
                    }
            }
        }
    }
}
