class Solution {

    public void sortColors(int[] nums) {
        int[] temp = new int[3];
        for (int i = 0; i < nums.length; i++) {
            temp[nums[i]]++;
        }
        int j = 0;
        for (int i = 0; i < temp.length; i++) {
            int len = temp[i];
            while (len > 0) {
                nums[j++] = i;
                len--;
            }
        }
    }
}
