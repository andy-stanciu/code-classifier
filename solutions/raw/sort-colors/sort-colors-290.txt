class Solution {

    public void sortColors(int[] nums) {
        int low = 0;
        ;
        int high = nums.length - 1;
        int mid = 0;
        //step 1 traverse the array mid point start at low index and if the equal to 0 swap it and increment the mid and low and if it equal to 1 do nothing and just increment the mid and else means it equal to 2 then swap it and reduce the high and and don\'t increment the mid;
        while (mid <= high) {
            if (nums[mid] == 0) {
                swap(mid, low, nums);
                mid++;
                low++;
            } else if (nums[mid] == 1) {
                mid++;
            } else {
                swap(mid, high, nums);
                high--;
            }
        }
    }

    void swap(int i, int j, int[] nums) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
