public class Solution {

    public void sortColors(int[] nums) {
        //paso#1
        int[] numeros = new int[3];
        //paso#2
        for (int num : nums) numeros[num]++;
        //paso#3
        int i = 0;
        for (int j = 0; j < 3; j++) {
            while (numeros[j] > 0) {
                nums[i] = j;
                numeros[j]--;
                i++;
            }
        }
    }
}
