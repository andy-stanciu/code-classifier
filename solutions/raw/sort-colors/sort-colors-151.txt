class Solution {

    public void sortColors(int[] nums) {
        int[] color = new int[3];
        int idx = 0;
        for (int i = 0; i < nums.length; i++) {
            color[nums[i]]++;
        }
        for (int i = 0; i < color.length; i++) {
            while (color[i] > 0) {
                nums[idx++] = i;
                color[i]--;
            }
        }
    }
}
