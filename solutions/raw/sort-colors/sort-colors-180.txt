class Solution {

    public void sortColors(int[] nums) {
        //initializaing the variables
        int low = 0;
        int mid = 0;
        int high = nums.length - 1;
        //traverse through all the elements
        for (int i = 0; i < nums.length; i++) {
            //nums[mid]==0, then swap it with the nums[low] and increment both low and high
            if (nums[mid] == 0) {
                int temp = nums[low];
                nums[low] = nums[mid];
                nums[mid] = temp;
                low++;
                mid++;
            } else //check if nums[mid==1] , increment the mid because in the middle part , we need to have 1 only
            if (nums[mid] == 1) {
                mid++;
            } else //the else case contains when nums[mid==2], this will swap the values which needs to have 2 only .
            {
                int temp2 = nums[high];
                nums[high] = nums[mid];
                nums[mid] = temp2;
                //decrement the high
                high--;
            }
        }
    }
}
