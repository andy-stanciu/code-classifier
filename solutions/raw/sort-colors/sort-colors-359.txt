class Solution {

    public void sortColors(int[] nums) {
        int red = 0, white = 0, blue = 0, index = 0;
        for (int x : nums) {
            if (x == 0) {
                red++;
            }
            if (x == 1) {
                white++;
            }
            if (x == 2) {
                blue++;
            }
        }
        for (int i = 0; i < red; i++) {
            nums[index++] = 0;
        }
        for (int i = 0; i < white; i++) {
            nums[index++] = 1;
        }
        for (int i = 0; i < blue; i++) {
            nums[index++] = 2;
        }
    }
}
