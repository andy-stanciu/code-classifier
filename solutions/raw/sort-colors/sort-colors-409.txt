class Solution {

    public void sortColors(int[] nums) {
        HashMap<Integer, Integer> numberCount = new HashMap<>();
        for (int num : nums) {
            if (numberCount.containsKey(num)) {
                numberCount.put(num, numberCount.get(num) + 1);
            } else {
                numberCount.put(num, 1);
            }
        }
        int hashKey = 0;
        for (int i = 0; i < nums.length; i++) {
            if (numberCount.containsKey(0) && numberCount.get(0) != 0) {
                nums[i] = 0;
                numberCount.put(0, numberCount.get(0) - 1);
            } else if (numberCount.containsKey(1) && numberCount.get(1) != 0) {
                nums[i] = 1;
                numberCount.put(1, numberCount.get(1) - 1);
            } else if (numberCount.containsKey(2) && numberCount.get(2) != 0) {
                nums[i] = 2;
                numberCount.put(2, numberCount.get(2) - 1);
            }
        }
    }
}
