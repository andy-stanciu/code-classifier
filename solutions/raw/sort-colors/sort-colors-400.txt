class Solution {

    public void swap(int[] arr, int a, int b) {
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }

    // Main method called by LeetCode
    public void sortColors(int[] nums) {
        // Call the helper method
        quickSort(nums, 0, nums.length - 1);
    }

    // Helper method for recursive sorting
    public void quickSort(int[] nums, int lb, int ub) {
        if (lb < ub) {
            // Perform partition
            int loc = partition(nums, lb, ub);
            // Sort left part
            quickSort(nums, lb, loc - 1);
            // Sort right part
            quickSort(nums, loc + 1, ub);
        }
    }

    // Partition logic
    public int partition(int[] arr, int lb, int ub) {
        int pivot = arr[lb];
        int start = lb;
        int end = ub;
        while (start < end) {
            while (arr[start] <= pivot && start < ub) {
                start++;
            }
            while (arr[end] > pivot) {
                end--;
            }
            if (start < end) {
                swap(arr, start, end);
            }
        }
        swap(arr, lb, end);
        return end;
    }
}
