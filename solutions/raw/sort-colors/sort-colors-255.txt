class Solution {

    public void sortColors(int[] nums) {
        int zero = 0;
        int one = 0;
        int two = 0;
        for (int val : nums) {
            if (val == 0) {
                zero++;
            } else if (val == 1) {
                one++;
            } else {
                //val == 2
                two++;
            }
        }
        int i = 0;
        while (zero-- > 0) {
            nums[i++] = 0;
        }
        while (one-- > 0) {
            nums[i++] = 1;
        }
        while (two-- > 0) {
            nums[i++] = 2;
        }
    }
}
