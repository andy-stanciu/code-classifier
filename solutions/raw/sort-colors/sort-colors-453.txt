class Solution {

    public void sortColors(int[] nums) {
        int zeroCount = 0;
        int oneCount = 0;
        int twoCount = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                zeroCount++;
            else if (nums[i] == 1)
                oneCount++;
            else
                twoCount++;
        }
        int start = 0;
        while (start < zeroCount) {
            nums[start] = 0;
            start++;
        }
        while (start < zeroCount + oneCount) {
            nums[start] = 1;
            start++;
        }
        while (start < zeroCount + oneCount + twoCount) {
            nums[start] = 2;
            start++;
        }
    }
}
