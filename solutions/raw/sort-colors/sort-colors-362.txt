class Solution {

    public void sortColors(int[] nums) {
        int[] val = new int[3];
        for (int i = 0; i < nums.length; i++) {
            switch(nums[i]) {
                case 0:
                    val[0] += 1;
                case 1:
                    val[1] += 1;
                case 2:
                    val[2] += 1;
            }
        }
        for (int i = 0; i < val[0]; i++) {
            nums[i] = 0;
        }
        for (int i = val[0]; i < val[1]; i++) {
            nums[i] = 1;
        }
        for (int i = val[1]; i < val[2]; i++) {
            nums[i] = 2;
        }
    }
}
