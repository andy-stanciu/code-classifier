class Solution {

    public void sortColors(int[] nums) {
        int i = 0, z = 0, o = 0;
        for (i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                z++;
            if (nums[i] == 1)
                o++;
        }
        for (i = 0; i < z; i++) nums[i] = 0;
        for (i = z; i < (o + z); i++) nums[i] = 1;
        for (i = (o + z); i < nums.length; i++) nums[i] = 2;
    }
}
