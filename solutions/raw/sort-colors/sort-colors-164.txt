class Solution {

    public void sortColors(int[] nums) {
        int[] counter = new int[3];
        for (int i = 0; i < nums.length; i++) {
            counter[nums[i]]++;
        }
        for (int i = 0; i < counter[0]; i++) {
            nums[i] = 0;
        }
        for (int i = counter[0]; i < counter[0] + counter[1]; i++) {
            nums[i] = 1;
        }
        for (int i = counter[0] + counter[1]; i < nums.length; i++) {
            nums[i] = 2;
        }
    }
}
