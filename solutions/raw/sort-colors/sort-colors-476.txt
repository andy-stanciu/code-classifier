class Solution {

    public void sortColors(int[] nums) {
        int n = nums.length;
        int zero = 0, one = 0, two = 0, c = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0)
                zero++;
            if (nums[i] == 1)
                one++;
            if (nums[i] == 2)
                two++;
        }
        for (int i = 0; i < zero; i++) {
            nums[c] = 0;
            c++;
        }
        for (int i = 0; i < one; i++) {
            nums[c] = 1;
            c++;
        }
        for (int i = 0; i < two; i++) {
            nums[c] = 2;
            c++;
        }
    }
}
