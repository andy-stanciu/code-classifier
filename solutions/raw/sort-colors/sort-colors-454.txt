class Solution {

    public void sortColors(int[] nums) {
        int[] colors = new int[3];
        for (int i = 0; i < nums.length; i++) {
            int x = nums[i];
            colors[x] += 1;
        }
        int j = 0;
        for (int i = 0; i < colors.length; i++) {
            int n = colors[i];
            while (n > 0) {
                nums[j] = i;
                n--;
                j++;
            }
        }
    }
}
