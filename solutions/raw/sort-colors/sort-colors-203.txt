class Solution {

    public void sortColors(int[] nums) {
        {
            int zero, one, two;
            zero = one = two = -1;
            for (int i = 0; i < nums.length; i++) {
                int num = nums[i];
                switch(num) {
                    case 2:
                        two++;
                        break;
                    case 1:
                        one++;
                        two++;
                        swap(nums, one, two);
                        break;
                    case 0:
                        one++;
                        two++;
                        swap(nums, one, two);
                        zero++;
                        swap(nums, zero, one);
                        break;
                    default:
                        break;
                }
            }
        }
    }

    private static void swap(int[] nums, int a, int b) {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
}
