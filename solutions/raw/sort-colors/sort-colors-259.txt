class Solution {

    public void sortColors(int[] nums) {
        int[] count = new int[3];
        for (int num : nums) count[num]++;
        for (int i = 1; i < count.length; i++) count[i] += count[i - 1];
        for (int i = 0; i < count.length; i++) {
            int start = i > 0 ? count[i - 1] : 0;
            int end = count[i];
            while (end > start) nums[--end] = i;
        }
    }
}
