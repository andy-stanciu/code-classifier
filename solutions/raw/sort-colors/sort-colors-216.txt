class Solution {

    public void sortColors(int[] nums) {
        int[] ans = new int[3];
        for (int num : nums) {
            ans[num]++;
        }
        int ind = 0;
        for (int i = 0; i < ans.length; i++) {
            int times = ans[i];
            for (int j = 0; j < times; j++) {
                nums[ind] = i;
                ind++;
            }
        }
    }
}
