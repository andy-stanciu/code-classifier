class Solution {

    public void sortColors(int[] nums) {
        int s = 0;
        int[] Arr = new int[nums.length];
        int target = 0;
        while (target < 3) {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] == target) {
                    Arr[s] = target;
                    s++;
                }
            }
            target++;
        }
        for (int k = 0; k < Arr.length; k++) {
            nums[k] = Arr[k];
        }
    }
}
