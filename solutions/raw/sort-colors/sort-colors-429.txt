class Solution {

    public void sortColors(int[] nums) {
        int[] colors = new int[3];
        for (int num : nums) {
            colors[num]++;
        }
        int i = 0;
        for (int color = 0; color < colors.length; color++) {
            while (colors[color] > 0) {
                nums[i] = color;
                i++;
                colors[color]--;
            }
        }
    }
}
