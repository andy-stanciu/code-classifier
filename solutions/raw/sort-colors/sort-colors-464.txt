class Solution {

    public void sortColors(int[] n) {
        Arrays.sort(n);
        System.out.print("[");
        for (int i = 0; i < n.length; i++) {
            System.out.print(n[i]);
            if (i == n.length - 1)
                continue;
            else
                System.out.print(",");
        }
        System.out.print("]");
    }
}
