class Solution {

    public void sortColors(int[] arr) {
        //using dutch national flag algorithm we will solve this problem
        int low = 0;
        int high = arr.length - 1;
        //the pointer with which we will iterate the array
        int mid = 0;
        // We have keep iterating till all the elements are sorted
        while (mid <= high) {
            switch(arr[mid]) {
                // Here mid is 0.
                case 0:
                    swap(low, mid, arr);
                    //behind low everyone is 0
                    low++;
                    //behind low everyone is 0
                    mid++;
                    break;
                // Here mid is 1.
                case 1:
                    //from low to high everyone should be 1
                    mid++;
                    break;
                // Here mid is 2.
                case 2:
                    swap(mid, high, arr);
                    //after high everyone is 2
                    high--;
                    break;
            }
        }
    }

    private void swap(int i, int j, int[] arr) {
        //function to swap two elements
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
