class Solution {

    public static String intToRoman(int num) {
        int[] ms = { 1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000 };
        String[] ms1 = { "I", "IV", "V", "IX", "X", "XL", "L", "XC", "C", "CD", "D", "CM", "M" };
        String str = "";
        for (int i = ms.length - 1; i > -1; i--) {
            if (ms[i] <= num) {
                num -= ms[i];
                str += ms1[i];
                i++;
            }
        }
        return str;
    }
}
