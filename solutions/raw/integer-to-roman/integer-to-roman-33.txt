class Solution {

    public String intToRoman(int num) {
        String str = "";
        String[] strArray = { "M", "D", "C", "L", "X", "V", "I" };
        int[] arr = { 1000, 500, 100, 50, 10, 5, 1 };
        for (int i = 0; i < arr.length; i++) {
            int ans = num / arr[i];
            num = num % arr[i];
            if (ans != 0) {
                for (int j = 0; j < ans; j++) {
                    str = str + strArray[i];
                }
            }
            if (num >= 900 && num < 1000) {
                num = num % 900;
                str = str + "CM";
            } else if (num >= 400 && num < 500) {
                num = num % 400;
                str = str + "CD";
            } else if (num >= 90 && num < 100) {
                num = num % 90;
                str = str + "XC";
            } else if (num >= 40 && num < 50) {
                num = num % 40;
                str = str + "XL";
            } else if (num >= 9 && num < 10) {
                num = num % 9;
                str = str + "IX";
            } else if (num >= 4 && num < 5) {
                num = num % 4;
                str = str + "IV";
            }
        }
        return str;
    }
}
