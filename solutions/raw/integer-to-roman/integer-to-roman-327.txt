class Solution {

    public String intToRoman(int num) {
        // Initialize arrays to store the integer values and corresponding Roman numerals
        // and a StringBuilder object to store the resulting Roman numeral string
        int[] symbol = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] value = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        StringBuilder sb = new StringBuilder();
        // Iterate over the symbol array using a while loop until num becomes zero
        // and build the resulting Roman numeral string by appending the corresponding
        // Roman numeral values to sb
        int i = 0;
        while (num > 0) {
            if (num >= symbol[i]) {
                sb.append(value[i]);
                num = num - symbol[i];
            } else {
                i++;
            }
        }
        // Return the resulting Roman numeral string
        return sb.toString();
    }
}
