class Solution {

    public String intToRoman(int num) {
        String he = "";
        int num1 = num / 1000;
        if (num1 == 1) {
            he += "M";
        }
        if (num1 == 2) {
            he += "MM";
        }
        if (num1 == 3) {
            he += "MMM";
        }
        int num2 = (num % 1000) / 100;
        if (num2 == 9) {
            he += "CM";
        } else if (num2 >= 5) {
            he += "D";
            for (int i = 0; i < num2 - 5; i++) {
                he += "C";
            }
        } else if (num2 == 4) {
            he += "CD";
        } else {
            for (int i = 0; i < num2; i++) {
                he += "C";
            }
        }
        int num3 = (num % 100) / 10;
        if (num3 == 9) {
            he += "XC";
        } else if (num3 >= 5) {
            he += "L";
            for (int i = 0; i < num3 - 5; i++) {
                he += "X";
            }
        } else if (num3 == 4) {
            he += "XL";
        } else {
            for (int i = 0; i < num3; i++) {
                he += "X";
            }
        }
        int num4 = num % 10;
        if (num4 == 9) {
            he += "IX";
        } else if (num4 >= 5) {
            he += "V";
            for (int i = 0; i < num4 - 5; i++) {
                he += "I";
            }
        } else if (num4 == 4) {
            he += "IV";
        } else {
            for (int i = 0; i < num4; i++) {
                he += "I";
            }
        }
        return he;
    }
}
