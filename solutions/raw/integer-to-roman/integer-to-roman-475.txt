class Solution {

    public String intToRoman(int num) {
        StringBuilder sb = new StringBuilder();
        int[] ans = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] n = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        for (int i = 0; i < ans.length; i++) {
            while (num >= ans[i]) {
                num -= ans[i];
                sb.append(n[i]);
            }
        }
        return sb.toString();
    }
}
