class Solution {

    public String intToRoman(int num) {
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] symbols = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        String output = "";
        for (int i = 0; i < 13; i++) {
            while (values[i] <= num) {
                String currS = symbols[i];
                int currV = values[i];
                output = output + currS;
                num = num - currV;
            }
        }
        return output;
    }
}
