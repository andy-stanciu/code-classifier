class Solution {

    public String intToRoman(int num) {
        // Array of Roman numeral symbols and their corresponding values
        String[] romanNumerals = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        // To build the resulting Roman numeral string
        StringBuilder result = new StringBuilder();
        // Iterate through the values and corresponding Roman symbols
        for (int i = 0; i < values.length; i++) {
            // Determine how many times the current value can fit into num
            while (num >= values[i]) {
                // Append the Roman numeral
                result.append(romanNumerals[i]);
                // Subtract the value from num
                num -= values[i];
            }
        }
        // Return the final Roman numeral string
        return result.toString();
    }
}
