class Solution {

    public String intToRoman(int num) {
        String roman = "";
        int i = 0;
        int value = 0;
        String[] str = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] arr = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        while (value < num) {
            if (value + arr[i] <= num) {
                value = value + arr[i];
                roman = roman + str[i];
            } else {
                i++;
            }
        }
        return roman;
    }
}
