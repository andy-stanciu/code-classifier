class Solution {

    private int[] value = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };

    private String[] code = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };

    public String intToRoman(int num) {
        String result;
        result = "";
        while (num > 0) {
            for (int index = 0; index < value.length; index++) {
                if (num >= value[index]) {
                    result = result + code[index];
                    num = num - value[index];
                    break;
                }
            }
        }
        return result;
    }
}
