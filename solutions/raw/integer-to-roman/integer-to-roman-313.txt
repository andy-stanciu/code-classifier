class Solution {

    public String intToRoman(int num) {
        String roman = "";
        int[] nums = { 1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000 };
        String[] sym = { "I", "IV", "V", "IX", "X", "XL", "L", "XC", "C", "CD", "D", "CM", "M" };
        for (int i = nums.length - 1; i >= 0; i--) {
            while (num >= nums[i]) {
                roman = roman + sym[i];
                num = num - nums[i];
            }
        }
        return roman;
    }
}
