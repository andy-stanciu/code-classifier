class Solution {

    public String intToRoman(int num) {
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] romans = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        StringBuilder str = new StringBuilder();
        for (int i = 0; i < values.length; i++) {
            if (num / values[i] > 0) {
                int q = num / values[i];
                num = num % values[i];
                while (q != 0) {
                    str.append(romans[i]);
                    q--;
                }
            }
            if (num == 0) {
                break;
            }
        }
        return str.toString();
    }
}
