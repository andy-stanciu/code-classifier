class Solution {

    public String intToRoman(int num) {
        // Arrays to store the integer values and their corresponding Roman numeral strings
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] symbols = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        // StringBuilder is more efficient than using a String for concatenation
        StringBuilder roman = new StringBuilder();
        // Iterate through the values and symbols arrays
        for (int i = 0; i < values.length; i++) {
            // Append the symbol to the result while reducing the number
            while (num >= values[i]) {
                roman.append(symbols[i]);
                num -= values[i];
            }
        }
        // Convert StringBuilder to String and return
        return roman.toString();
    }
}
