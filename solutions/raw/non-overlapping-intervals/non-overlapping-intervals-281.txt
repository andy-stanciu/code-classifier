class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        //Sorting on the 2nd argument of the array
        Arrays.sort(intervals, Comparator.comparing(a -> a[1]));
        int end = intervals[0][1];
        int count = 1;
        int n = intervals.length;
        //checking the overlapping intervals
        for (int i = 1; i < n; i++) {
            if (intervals[i][0] >= end) {
                end = intervals[i][1];
                count++;
            }
        }
        //return total - overlapping = non-overlapping intervals
        return (n - count);
    }
}
