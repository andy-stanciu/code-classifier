class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (o1, o2) -> o1[1] - o2[1]);
        int r = intervals.length;
        int cnt = 0, i = 1;
        int[] start = intervals[0];
        while (i < r) {
            if (intervals[i][0] < start[1]) {
                cnt++;
            } else {
                start = intervals[i];
            }
            i++;
        }
        return cnt;
    }
}
