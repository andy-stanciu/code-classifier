class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int count = 0;
        int lastind = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (lastind > intervals[i][0]) {
                count++;
            } else {
                lastind = intervals[i][1];
            }
        }
        return count;
    }
}
