class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        // sort intervals
        mergeSort(intervals, 0, intervals.length - 1);
        int inCompatibles = 0;
        int lastFinishedIntervalEnd = Integer.MIN_VALUE;
        for (int[] interval : intervals) {
            int start = interval[0];
            int end = interval[1];
            if (start >= lastFinishedIntervalEnd) {
                lastFinishedIntervalEnd = end;
            } else {
                inCompatibles++;
            }
        }
        return inCompatibles;
    }

    public void mergeSort(int[][] intervals, int left, int right) {
        if (left < right) {
            int mid = (left + right) / 2;
            mergeSort(intervals, left, mid);
            mergeSort(intervals, mid + 1, right);
            merge(intervals, left, mid, right);
        }
    }

    public void merge(int[][] intervals, int low, int mid, int high) {
        int[][] left = new int[mid - low + 1][2];
        int[][] right = new int[high - (mid + 1) + 1][2];
        // copy left array
        for (int i = 0; i < (mid - low + 1); i++) {
            left[i][0] = intervals[low + i][0];
            left[i][1] = intervals[low + i][1];
        }
        // copy right array
        for (int j = 0; j < (high - mid); j++) {
            right[j][0] = intervals[mid + 1 + j][0];
            right[j][1] = intervals[mid + 1 + j][1];
        }
        // merge
        int k = low;
        int i = 0;
        int j = 0;
        while (i < left.length || j < right.length) {
            if (i < left.length && j < right.length) {
                int toBeComparedLeft = left[i][1];
                int toBeComparedRight = right[j][1];
                if (toBeComparedLeft < toBeComparedRight) {
                    intervals[k][0] = left[i][0];
                    intervals[k++][1] = left[i++][1];
                } else {
                    intervals[k][0] = right[j][0];
                    intervals[k++][1] = right[j++][1];
                }
            } else if (i < left.length) {
                intervals[k][0] = left[i][0];
                intervals[k++][1] = left[i++][1];
            } else {
                intervals[k][0] = right[j][0];
                intervals[k++][1] = right[j++][1];
            }
        }
    }
}
