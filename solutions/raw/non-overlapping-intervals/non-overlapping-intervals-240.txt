class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        if (n == 0)
            return 0;
        Arrays.sort(intervals, (a, b) -> (a[1] - b[1]));
        int res = 0;
        int index = 0;
        for (int i = 1; i < n; i++) {
            if (intervals[i][0] >= intervals[index][1]) {
                index = i;
            } else
                res++;
        }
        return res;
    }
}
