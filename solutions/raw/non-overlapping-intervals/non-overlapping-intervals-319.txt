class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> (a[0] - b[0]));
        int count = 0;
        int start = intervals[0][0];
        for (int[] nums : intervals) {
            if (nums[0] < start) {
                count++;
                start = Math.min(start, nums[1]);
            } else {
                start = nums[1];
            }
        }
        return count;
    }
}
