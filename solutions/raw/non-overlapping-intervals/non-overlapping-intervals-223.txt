class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int prev = 0;
        int ans = 0;
        for (int curr = 1; curr < intervals.length; curr++) {
            if (intervals[curr][0] < intervals[prev][1]) {
                ans++;
                if (intervals[prev][1] >= intervals[curr][1]) {
                    prev = curr;
                }
            } else {
                prev = curr;
            }
        }
        return ans;
    }
}
