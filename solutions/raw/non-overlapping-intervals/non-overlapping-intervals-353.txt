class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int l = intervals.length;
        if (l == 0) {
            return 0;
        }
        Arrays.sort(intervals, (a, b) -> {
            return a[1] - b[1];
        });
        int ans = 1;
        int ind = 0;
        for (int i = 1; i < l; i++) {
            if (intervals[i][0] >= intervals[ind][1]) {
                ans++;
                ind = i;
            }
        }
        return l - ans;
    }
}
