class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[0], b[0]));
        int prev = Integer.MIN_VALUE, c = 0;
        for (int[] arr : intervals) {
            if (arr[0] >= prev)
                //no overlap
                prev = arr[1];
            else {
                c++;
                prev = Math.min(prev, arr[1]);
            }
        }
        return c;
    }
}
