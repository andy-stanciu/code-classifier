class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int ans = 0;
        int pend = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] < pend) {
                ans++;
                pend = Math.min(pend, intervals[i][1]);
            } else {
                pend = intervals[i][1];
            }
        }
        return ans;
    }
}
