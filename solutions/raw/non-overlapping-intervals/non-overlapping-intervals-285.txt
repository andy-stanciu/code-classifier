class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> {
            if (a[0] == b[0]) {
                return Integer.compare(a[1], b[1]);
            }
            return Integer.compare(a[0], b[0]);
        });
        int count = 0;
        // intervals can be -ve and positive so storing the min value
        int lastInt = Integer.MIN_VALUE;
        for (int[] data : intervals) {
            if (data[0] >= lastInt) {
                // non overlaping case we don\'t need to remove it update last int to latest end time
                lastInt = data[1];
            } else {
                // overlaping interval remove the one with highest ending time
                count++;
                lastInt = Math.min(lastInt, data[1]);
            }
        }
        return count;
    }
}
