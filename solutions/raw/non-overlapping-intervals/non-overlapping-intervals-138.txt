class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        // Sort intervals by their end times
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[1], b[1]));
        int n = intervals.length;
        // This will count the number of intervals to remove
        int count = 0;
        // Initial end time set to the end of the first interval
        int endtime = intervals[0][1];
        // Start from the second interval
        for (int i = 1; i < n; i++) {
            int start = intervals[i][0];
            if (start < endtime) {
                // Overlapping interval found, increment count
                count++;
            } else {
                // No overlap, update endtime to current interval\'s end time
                endtime = intervals[i][1];
            }
        }
        return count;
    }
}
