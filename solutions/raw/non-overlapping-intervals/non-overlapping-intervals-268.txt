class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int count = 1;
        Arrays.sort(intervals, Comparator.comparing(o -> o[1]));
        int prev = intervals[0][1];
        for (int i = 0; i < intervals.length; i++) {
            if (intervals[i][0] >= prev) {
                prev = intervals[i][1];
                count++;
            }
        }
        return intervals.length - count;
    }
}
