class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int end = Integer.MIN_VALUE, count = 0;
        Arrays.sort(intervals, (x, y) -> x[1] - y[1]);
        for (int i = 0; i < intervals.length; i++) {
            if (intervals[i][0] >= end)
                end = intervals[i][1];
            else
                count++;
        }
        return count;
    }
}
