class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        int ans = 0;
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int prevLast = intervals[0][1];
        for (int i = 1; i < n; i++) {
            int first = intervals[i][0];
            int last = intervals[i][1];
            if (first < prevLast) {
                if (last > prevLast) {
                    ans++;
                } else {
                    prevLast = last;
                    ans++;
                }
            } else {
                prevLast = last;
            }
        }
        return ans;
    }
}
