class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (int[] x, int[] y) -> x[1] - y[1]);
        int total = intervals.length;
        int lastTime = Integer.MIN_VALUE;
        int count = 0;
        for (int[] interval : intervals) {
            if (interval[0] >= lastTime) {
                count++;
                lastTime = interval[1];
            }
        }
        return total - count;
    }
}
