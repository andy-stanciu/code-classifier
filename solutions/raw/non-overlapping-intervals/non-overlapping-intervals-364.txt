class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        // Sort by ending time
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int end = Integer.MIN_VALUE;
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (end <= intervals[i][0]) {
                end = intervals[i][1];
                cnt++;
            }
        }
        // Minimum deletion=n-Maximum number of non-overlapping intervals.
        return n - cnt;
    }
}
