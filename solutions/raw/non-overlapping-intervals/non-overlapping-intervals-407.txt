class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        //we can also find by finding the non overlapping intervals and subract from the total
        //or find the overlapping intervals
        Arrays.sort(intervals, new Comparator<int[]>() {

            public int compare(int[] o1, int[] o2) {
                return o1[1] - o2[1];
            }
        });
        int count = 0;
        int end = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (end > intervals[i][0]) {
                count++;
            } else {
                end = intervals[i][1];
            }
        }
        return count;
    }
}
