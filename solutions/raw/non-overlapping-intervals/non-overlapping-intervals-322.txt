class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        intervals = sorting(intervals, 1);
        int end = intervals[0][1];
        int ans = 0;
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] < end) {
                ans++;
            } else
                end = intervals[i][1];
        }
        return ans;
    }

    public int[][] sorting(int[][] arr, int col) {
        Arrays.sort(arr, new Comparator<int[]>() {

            public int compare(int[] first, int[] second) {
                if ((first[col] > second[col]) || (first[col] == second[col] && first[col - 1] > second[col - 1])) {
                    return 1;
                } else
                    return -1;
            }
        });
        return arr;
    }
}
