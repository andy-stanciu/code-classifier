class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (x, y) -> x[1] - y[1]);
        int n = intervals.length;
        if (n <= 1)
            return 0;
        int end = intervals[0][1];
        int ans = 0;
        for (int i = 1; i < n; i++) {
            if (end > intervals[i][0]) {
                ans++;
            } else {
                end = intervals[i][1];
            }
        }
        return ans;
    }
}
