class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, new thisComparator());
        int end = intervals[0][1];
        int count = 1;
        int len = intervals.length;
        for (int i = 1; i < len; i++) {
            if (intervals[i][0] >= end) {
                end = intervals[i][1];
                count++;
            }
        }
        return len - count;
    }

    class thisComparator implements Comparator<int[]> {

        public int compare(int[] a, int[] b) {
            return a[1] - b[1];
        }
    }
}
