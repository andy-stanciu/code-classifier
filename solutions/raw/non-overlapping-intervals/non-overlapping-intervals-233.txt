class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (i1, i2) -> Integer.compare(i1[0], i2[0]));
        int prev = intervals[0][1];
        int count = 0;
        for (int i = 1; i < intervals.length; i++) {
            int start = intervals[i][0];
            int end = intervals[i][1];
            if (prev > start) {
                count++;
                if (prev > end) {
                    prev = end;
                }
                continue;
            }
            prev = end;
        }
        return count;
    }
}
