class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int lastEnd = Integer.MIN_VALUE, ans = 0;
        for (int i = 0; i < intervals.length; i++) {
            int curStart = intervals[i][0];
            int curEnd = intervals[i][1];
            if (curStart >= lastEnd) {
                lastEnd = curEnd;
            } else {
                ans++;
            }
        }
        return ans;
    }
}
