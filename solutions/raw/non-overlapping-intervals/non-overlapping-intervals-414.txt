class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, Comparator.comparingDouble(o -> o[0]));
        int common = intervals[0][1];
        int overlap = 0;
        for (int i = 1; i < intervals.length; ++i) {
            if (common > intervals[i][0]) {
                ++overlap;
                common = Math.min(common, intervals[i][1]);
            } else {
                common = intervals[i][1];
            }
        }
        return overlap;
    }
}
