class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, Comparator.comparingInt(a -> a[1]));
        int p = 0;
        int c = 0;
        for (int i = 0; i < intervals.length; i++) {
            if (intervals[p][1] > intervals[i][0]) {
                c++;
            } else {
                p = i;
            }
        }
        return c - 1;
    }
}
