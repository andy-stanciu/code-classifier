class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> (a[1] == b[1]) ? (a[0] - b[0]) : (a[1] - b[1]));
        int count = 0;
        Stack<int[]> st = new Stack<int[]>();
        for (int i = 0; i < intervals.length; i++) {
            if (st.isEmpty()) {
                st.push(intervals[i]);
            } else {
                if (intervals[i][0] < st.peek()[1]) {
                    count++;
                } else {
                    st.push(intervals[i]);
                }
            }
        }
        return count;
    }
}
