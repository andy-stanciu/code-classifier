class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[0], b[0]));
        int n = intervals.length;
        int ans = 0;
        int j = 1;
        int i = 0;
        while (j < n) {
            if (intervals[i][1] > intervals[j][0]) {
                if (intervals[i][1] >= intervals[j][1]) {
                    i = j;
                }
                j++;
                ans++;
            } else {
                i = j;
                j++;
            }
        }
        return ans;
    }
}
