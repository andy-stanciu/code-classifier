class Solution {

    public int eraseOverlapIntervals(int[][] act) {
        int n = act.length;
        Arrays.sort(act, (a, b) -> Integer.compare(a[1], b[1]));
        int maxAct = 1;
        int removed = 0;
        int lastEnd = act[0][1];
        for (int i = 1; i < n; i++) {
            if (act[i][0] >= lastEnd) {
                maxAct++;
                lastEnd = act[i][1];
            } else
                removed++;
        }
        return removed;
    }
}
