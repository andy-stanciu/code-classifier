class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        if (intervals.length < 2)
            return 0;
        // sort the array based on 0th index of intervals
        // if the 0th index are equal, then the interval with smaller 1st index should have higher priority
        Arrays.sort(intervals, (a, b) -> a[0] != b[0] ? a[0] - b[0] : a[1] - b[1]);
        int ans = 0;
        int minR = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] < minR) {
                ans++;
                minR = Math.min(intervals[i][1], minR);
            } else {
                // update right boundary otherwise
                minR = intervals[i][1];
            }
        }
        return ans;
    }
}
