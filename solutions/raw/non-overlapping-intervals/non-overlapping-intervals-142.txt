class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        // sort by interval end, quicker a task end more room for other non overllaped ones
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int count = 1, prev = 0;
        for (int i = 1; i < n; i++) {
            // find out if the intervals are non overlapping
            if (intervals[i][0] >= intervals[prev][1]) {
                // count it
                count++;
                // set the current one as prev to compare with upcoming
                // interval
                prev = i;
            }
        }
        return n - count;
    }
}
