class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (x, y) -> x[1] - y[1]);
        int res = 0, start = intervals[0][0], end = Integer.MIN_VALUE;
        for (int[] arr : intervals) {
            int s = arr[0], e = arr[1];
            if (s >= end) {
                end = e;
            } else
                res++;
        }
        return res;
    }
}
