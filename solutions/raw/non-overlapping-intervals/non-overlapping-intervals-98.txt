class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        Arrays.sort(intervals, (x, y) -> x[1] - y[1]);
        int count = 1;
        int l = 0;
        int r = 1;
        while (r < n) {
            if (intervals[l][1] <= intervals[r][0]) {
                count++;
                l++;
                r++;
            } else {
                while (r < n && intervals[l][1] > intervals[r][0]) r++;
                if (r >= n)
                    continue;
                count++;
                l = r;
                r++;
            }
        }
        return n - count;
    }
}
