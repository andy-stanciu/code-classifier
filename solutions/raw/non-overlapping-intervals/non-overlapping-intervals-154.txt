class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> (a[1] - b[1]));
        Stack<Integer> end_time_stack = new Stack<>();
        end_time_stack.push(intervals[0][1]);
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] >= end_time_stack.peek())
                end_time_stack.push(intervals[i][1]);
        }
        return intervals.length - end_time_stack.size();
    }
}
