class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (e1, e2) -> e1[1] - e2[1]);
        int prev = -(int) 1e6;
        int res = 0;
        for (int idx = 0; idx < intervals.length; idx++) {
            if (intervals[idx][0] >= prev) {
                prev = intervals[idx][1];
            } else {
                res++;
            }
        }
        return res;
    }
}
