class Solution {

    public int eraseOverlapIntervals(int[][] arr) {
        if (arr.length == 0)
            return 0;
        Arrays.sort(arr, (a, b) -> Integer.compare(a[1], b[1]));
        int cnt = 0;
        int lastEndTime = arr[0][1];
        for (int i = 1; i < arr.length; ++i) {
            if (arr[i][0] >= lastEndTime) {
                lastEndTime = arr[i][1];
            } else
                cnt++;
        }
        return cnt;
    }
}
