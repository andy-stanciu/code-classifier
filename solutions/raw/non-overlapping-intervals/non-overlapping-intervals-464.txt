class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int c = 0;
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int s = intervals[0][1];
        int e = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] < e) {
                e = Math.min(e, intervals[i][1]);
                c += 1;
            } else {
                e = intervals[i][1];
            }
        }
        return c;
    }
}
