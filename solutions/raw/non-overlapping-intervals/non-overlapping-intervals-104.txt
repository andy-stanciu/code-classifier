class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, java.util.Comparator.comparingDouble(a -> a[1]));
        int r = intervals.length;
        int i = 1, n = 0, f = intervals[0][1];
        while (i < r) {
            if (intervals[i][0] < f)
                n++;
            else {
                f = intervals[i][1];
            }
            i++;
        }
        return n;
    }
}
