class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[0], b[0]));
        int prevIndex = 0;
        int ans = 0;
        for (int i = 1; i < intervals.length; i++) {
            int prevStart = intervals[prevIndex][0], prevEnd = intervals[prevIndex][1];
            int currStart = intervals[i][0], currEnd = intervals[i][1];
            if (prevEnd > currStart) {
                // there is overlap, so we will need to remove 1 of these intervals
                ans++;
                // need to decide which to remove
                if (currEnd <= prevEnd) {
                    // we are better off with the shorter interval
                    prevIndex = i;
                }
            } else {
                // prevEnd <= currStart
                // move the prevIndex forward since there is no overlap
                prevIndex = i;
            }
        }
        return ans;
    }
}
