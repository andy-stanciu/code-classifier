class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        // for counting the overlapping intervals
        int count = 0;
        // for previous selcting interval index
        int prev = 0;
        // sorting the given array on column 1 to select as many intervals
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] >= intervals[prev][1]) {
                // updating previous selecting interval
                prev = i;
            } else {
                count++;
            }
        }
        return count;
    }
}
