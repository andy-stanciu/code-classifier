class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[1], b[1]));
        int i = 0, count = 1;
        for (int j = 1; j < intervals.length; j++) {
            if (intervals[j][0] >= intervals[i][1]) {
                i = j;
                count++;
            }
        }
        return intervals.length - count;
    }
}
