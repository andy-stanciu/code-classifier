class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, Comparator.comparingInt(a -> a[1]));
        int count = 0;
        int m = intervals.length;
        int n = intervals[0].length;
        int prev = intervals[0][1];
        for (int i = 1; i < m; i++) {
            if (intervals[i][0] < prev)
                count++;
            else
                prev = intervals[i][1];
        }
        return count;
    }
}
