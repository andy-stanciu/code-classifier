class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int prevEnd = Integer.MIN_VALUE;
        int res = 0;
        for (int[] interval : intervals) {
            if (interval[0] < prevEnd) {
                if (interval[1] <= prevEnd) {
                    prevEnd = interval[1];
                }
                res++;
            } else {
                prevEnd = interval[1];
            }
        }
        return res;
    }
}
