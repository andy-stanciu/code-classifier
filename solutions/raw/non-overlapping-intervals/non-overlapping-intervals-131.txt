class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (arr1, arr2) -> Integer.compare(arr1[1], arr2[1]));
        int prevIntervalEnd = intervals[0][1];
        int nonOverlappingIntervals = 0;
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] < prevIntervalEnd) {
                nonOverlappingIntervals += 1;
            } else {
                prevIntervalEnd = intervals[i][1];
            }
        }
        return nonOverlappingIntervals;
    }
}
