class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, Comparator.comparingDouble(o -> o[1]));
        int count = 0;
        int x = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (x > intervals[i][0]) {
                count++;
            } else {
                x = intervals[i][1];
            }
        }
        return count;
    }
}
