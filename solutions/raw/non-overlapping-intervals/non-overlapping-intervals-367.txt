class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        List<List<Integer>> list = Arrays.stream(intervals).map(row -> Arrays.stream(row).boxed().collect(Collectors.toList())).collect(Collectors.toList());
        int c = 0;
        int i = 0;
        while (!list.isEmpty() && i < list.size() - 1) {
            if ((list.get(i + 1).get(0) < list.get(i).get(1)) && ((list.get(i + 1).get(1) <= list.get(i).get(1)))) {
                list.remove(i);
                c++;
            } else if ((list.get(i + 1).get(0) < list.get(i).get(1)) && ((list.get(i + 1).get(1) > list.get(i).get(1)))) {
                list.remove(i + 1);
                c++;
            } else {
                i++;
            }
        }
        System.out.println(c);
        return c;
    }
}
