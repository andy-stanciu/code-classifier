class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        int cnt = 0;
        Arrays.sort(intervals, (int[] a, int[] b) -> a[1] - b[1]);
        int i = 0;
        while (i < n) {
            int b = intervals[i][1];
            while (i + 1 < n && b > intervals[i + 1][0]) {
                i++;
                cnt++;
            }
            i++;
        }
        return cnt;
    }
}
