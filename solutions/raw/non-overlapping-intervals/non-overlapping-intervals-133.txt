class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int totalIntervals = intervals.length;
        //Sort by intervals ending first
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        // By default include intervals[0]
        int countOfNonOverLappinIntervals = 1;
        int previousNonOverLappingIntervalIndex = 0;
        for (int index = 1; index < totalIntervals; index++) {
            if (intervals[index][0] >= intervals[previousNonOverLappingIntervalIndex][1]) {
                countOfNonOverLappinIntervals++;
                previousNonOverLappingIntervalIndex = index;
            }
        }
        return totalIntervals - countOfNonOverLappinIntervals;
    }
}
