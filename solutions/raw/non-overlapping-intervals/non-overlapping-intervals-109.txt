class Solution {

    public int eraseOverlapIntervals(int[][] arr) {
        Arrays.sort(arr, (a, b) -> Integer.compare(a[0], b[0]));
        int count = 0;
        int[] temp = arr[0];
        for (int j = 1; j < arr.length; j++) {
            int[] i = arr[j];
            if (i[0] < temp[1]) {
                count++;
                if (i[1] < temp[1]) {
                    temp = i;
                }
            } else {
                temp = i;
            }
        }
        return count;
    }
}
