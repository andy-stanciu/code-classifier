class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        Arrays.sort(intervals, (a, b) -> {
            if (a[1] < b[1])
                return -1;
            else if (a[1] > b[1])
                return 1;
            else
                return 0;
        });
        int ans = 0;
        int prevE = intervals[0][1];
        for (int i = 1; i < n; i++) {
            if (intervals[i][0] < prevE)
                ans++;
            else
                prevE = intervals[i][1];
        }
        return ans;
    }
}
