class Solution {

    public int eraseOverlapIntervals(int[][] arr) {
        int n = arr.length;
        Arrays.sort(arr, (a, b) -> Integer.compare(a[1], b[1]));
        int ans = 1;
        int prev = 0;
        for (int i = 1; i < n; i++) {
            if (arr[i][0] >= arr[prev][1]) {
                ans++;
                prev = i;
            }
        }
        return n - ans;
    }
}
