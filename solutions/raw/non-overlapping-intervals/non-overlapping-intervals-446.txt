class Solution {

    public int eraseOverlapIntervals(int[][] A) {
        Arrays.sort(A, (o1, o2) -> o1[1] - o2[1]);
        System.out.println(Arrays.deepToString(A));
        int res = A[0][1], prev = 1;
        for (int i = 1; i < A.length; i++) {
            if (A[i][0] >= res) {
                prev++;
                res = A[i][1];
            }
        }
        return Math.abs(A.length - prev);
    }
}
