class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int n = intervals.length;
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int last = intervals[0][1], c = 1;
        for (int i = 1; i < n; i++) {
            if (last <= intervals[i][0]) {
                c++;
                last = intervals[i][1];
            }
        }
        return n - c;
    }
}
