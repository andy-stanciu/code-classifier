class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int[] pre = intervals[0];
        int res = 0;
        for (int i = 1; i < intervals.length; i++) {
            int[] cur = intervals[i];
            if (cur[0] < pre[1]) {
                //overlapping
                res++;
                if (pre[1] <= cur[1])
                    continue;
            }
            pre = cur;
        }
        return res;
    }
}
