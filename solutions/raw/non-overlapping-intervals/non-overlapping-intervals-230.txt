class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        // Sort on endTime
        Arrays.sort(intervals, (a1, a2) -> (a1[1] - a2[1]));
        int previousEndTime = Integer.MIN_VALUE;
        int ignoredIntervals = 0;
        for (int i = 0; i < intervals.length; i++) {
            int[] interval = intervals[i];
            int startTime = interval[0];
            int endTime = interval[1];
            if (startTime >= previousEndTime) {
                // Take this interval because not overlapping..
                previousEndTime = endTime;
            } else {
                // Sadly this is overlapping. So, ignore this.
                ignoredIntervals++;
            }
        }
        return ignoredIntervals;
    }
}
