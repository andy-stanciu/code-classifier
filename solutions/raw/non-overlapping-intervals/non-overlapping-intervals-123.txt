class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int count = 0;
        int last = intervals[0][1];
        for (int i = 1; i < intervals.length; ++i) {
            int[] interval = intervals[i];
            if (interval[0] < last) {
                count++;
                last = Math.min(last, interval[1]);
            } else
                last = interval[1];
        }
        return count;
    }
}
