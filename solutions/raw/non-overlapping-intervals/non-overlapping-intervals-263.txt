class Solution {

    public int eraseOverlapIntervals(int[][] ins) {
        Arrays.sort(ins, (a, b) -> {
            return a[1] - b[1];
        });
        int c = 0;
        for (int i = 1; i < ins.length; i++) {
            if (ins[i][0] < ins[i - 1][1]) {
                c++;
                ins[i][0] = ins[i - 1][0];
                ins[i][1] = ins[i - 1][1];
            }
        }
        return c;
    }
}
