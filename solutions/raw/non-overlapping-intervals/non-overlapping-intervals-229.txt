class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, Comparator.comparingInt(row -> row[1]));
        int ans = -1;
        int[] prev = intervals[0];
        for (int[] interval : intervals) {
            if (interval[0] < prev[1])
                ans++;
            else
                prev = interval;
        }
        return ans;
    }
}
