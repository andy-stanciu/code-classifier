class Solution {

    private boolean checkOverlap(int[] intervalOne, int[] intervalTwo) {
        return (intervalTwo[0] < intervalOne[1]);
    }

    public int eraseOverlapIntervals(int[][] intervals) {
        int minimumIntervalsRemoval = 0;
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int lastIntervalIndex = 0;
        int currentIntervalIndex = 1;
        while (currentIntervalIndex < intervals.length) {
            if (checkOverlap(intervals[lastIntervalIndex], intervals[currentIntervalIndex])) {
                minimumIntervalsRemoval++;
            } else {
                lastIntervalIndex = currentIntervalIndex;
            }
            currentIntervalIndex++;
        }
        return minimumIntervalsRemoval;
    }
}
