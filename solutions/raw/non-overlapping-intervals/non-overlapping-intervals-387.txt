class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        int[] dp = new int[intervals.length + 1];
        Arrays.fill(dp, -1);
        int max = Recu(intervals, 0, dp);
        return intervals.length - max;
    }

    public static int Recu(int[][] intervals, int i, int[] dp) {
        if (i == intervals.length)
            return 0;
        if (dp[i] != -1)
            return dp[i];
        int nextIndex = Binary(intervals, i, intervals[i][1]);
        return dp[i] = Math.max(Recu(intervals, i + 1, dp), 1 + Recu(intervals, nextIndex, dp));
    }

    public static int Binary(int[][] intervals, int curr, int target) {
        int left = curr, right = intervals.length;
        while (left < right) {
            int mid = (left + right) / 2;
            if (intervals[mid][0] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }
}
