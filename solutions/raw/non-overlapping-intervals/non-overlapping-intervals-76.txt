class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        // sort based on the right side of the interval
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int count = 0;
        // keep track of the largest value encountered so far
        int prevIntervalMax = Integer.MIN_VALUE;
        for (int[] interval : intervals) {
            //  if the value of the left side of the interval is less
            // than our max then this would overlap with other intervals
            // so we increase the counter
            if (interval[0] < prevIntervalMax) {
                count++;
                continue;
            }
            // else we update the value of our max
            prevIntervalMax = interval[1];
        }
        return count;
    }
}
