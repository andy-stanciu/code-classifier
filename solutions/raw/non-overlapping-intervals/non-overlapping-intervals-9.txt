class Solution {

    public int eraseOverlapIntervals(int[][] arr) {
        if (arr.length <= 1)
            return 0;
        int[] a = new int[100001];
        Arrays.fill(a, -1);
        int x = 50000;
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            int start = arr[i][0] + x;
            int end = arr[i][1] + x;
            if (a[start] != -1) {
                count++;
                a[start] = Math.min(a[start], end);
            } else {
                a[start] = end;
            }
        }
        int j = -1, k = -1;
        for (int i = 0; i < 100001; i++) {
            if (a[i] == -1)
                continue;
            if (j == -1) {
                j = i;
                k = a[i];
            } else {
                if (i < k) {
                    count++;
                    if (k > a[i]) {
                        j = i;
                        k = a[i];
                    }
                } else {
                    j = i;
                    k = a[i];
                }
            }
        }
        return count;
    }
}
