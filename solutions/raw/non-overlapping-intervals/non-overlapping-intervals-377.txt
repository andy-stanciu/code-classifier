class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1, curr = 1, prev = 0, n = intervals.length;
        while (curr < n) {
            if (intervals[prev][1] <= intervals[curr][0]) {
                count++;
                prev = curr;
            }
            curr++;
        }
        return n - count;
    }
}
