// intutive: sort the array
// if there is overlap, remove the longest one
// in other words leave the ends ealier one counter ++
//
class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int counter = 0;
        // sort base on start point, low -> high
        Arrays.sort(intervals, (a, b) -> {
            if (a[0] == b[0]) {
                return a[1] - b[1];
            }
            return a[0] - b[0];
        });
        int[] prev = intervals[0];
        for (int i = 1; i < intervals.length; i++) {
            int[] curr = intervals[i];
            if (prev[1] > curr[0]) {
                // there is overlap
                if (curr[1] < prev[1]) {
                    prev = curr;
                }
                counter++;
            } else {
                prev = curr;
            }
        }
        return counter;
    }
}
