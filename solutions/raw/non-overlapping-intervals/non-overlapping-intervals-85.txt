class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        //sort the array by end time
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        int last = Integer.MIN_VALUE;
        int cnt = 0;
        for (int i = 0; i < intervals.length; i++) {
            if (intervals[i][0] >= last) {
                last = intervals[i][1];
            } else {
                cnt++;
            }
        }
        return cnt;
    }
}
