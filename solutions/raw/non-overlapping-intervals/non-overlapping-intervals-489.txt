class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        int minCount = 0;
        // Empty case
        if (intervals.length == 0)
            return minCount;
        // Sort intervals first - O(nlogn)
        Arrays.sort(intervals, (a, b) -> {
            return a[0] - b[0];
        });
        int[] compareTo = intervals[0];
        // O(n)
        for (int i = 1; i < intervals.length; i++) {
            //Check for overlap
            if (compareTo[1] > intervals[i][0]) {
                minCount++;
                //Overlap found, keep the interval with the smallest end value
                if (compareTo[1] > intervals[i][1]) {
                    compareTo = intervals[i];
                }
            } else {
                //Set compareTo as the next value, we have found all the overlap for this one
                compareTo = intervals[i];
            }
        }
        return minCount;
    }
}
