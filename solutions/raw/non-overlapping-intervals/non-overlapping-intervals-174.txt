class Solution {

    public int eraseOverlapIntervals(int[][] A) {
        int ans = 0;
        if (A.length <= 1)
            return ans;
        Arrays.sort(A, (a, b) -> Integer.compare(a[1], b[1]));
        int[] arr = A[0];
        int n = A.length;
        for (int i = 1; i < n; i++) {
            if (A[i][0] < arr[1])
                ans++;
            else {
                arr = A[i];
            }
        }
        return ans;
    }
}
