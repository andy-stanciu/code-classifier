class Solution {

    public int eraseOverlapIntervals(int[][] intervals) {
        if (intervals == null || intervals.length == 0) {
            return 0;
        }
        // Sort intervals based on end times
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);
        // Number of non-overlapping intervals
        int count = 1;
        int prevEnd = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] >= prevEnd) {
                // Non-overlapping interval found
                count++;
                prevEnd = intervals[i][1];
            }
            // If the interval overlaps with the previous interval, skip it
        }
        // Number of intervals to remove = total intervals - non-overlapping intervals
        return intervals.length - count;
    }
}
