class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        int sum = len1 + len2;
        // Create a new array to hold the merged arrays
        int[] arr = new int[sum];
        System.arraycopy(nums1, 0, arr, 0, len1);
        System.arraycopy(nums2, 0, arr, len1, len2);
        // Sort the merged array
        Arrays.sort(arr);
        // Find the median
        if (sum % 2 == 1) {
            // If the total length is odd, the median is the middle element
            return arr[sum / 2];
        } else {
            // If the total length is even, the median is the average of the two middle elements
            return (arr[sum / 2 - 1] + arr[sum / 2]) / 2.0;
        }
    }
}
