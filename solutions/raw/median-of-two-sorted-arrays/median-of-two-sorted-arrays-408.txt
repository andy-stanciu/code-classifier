class Solution {

    public double findMedianSortedArrays(int[] array1, int[] array2) {
        int pointer1 = 0, pointer2 = 0, mergeIndex = 0, totalLength = array1.length + array2.length;
        int[] mergedArray = new int[totalLength];
        while (mergeIndex < totalLength) {
            if (pointer1 < array1.length && pointer2 < array2.length) {
                if (array1[pointer1] < array2[pointer2])
                    mergedArray[mergeIndex++] = array1[pointer1++];
                else
                    mergedArray[mergeIndex++] = array2[pointer2++];
            } else {
                if (pointer1 < array1.length) {
                    while (pointer1 < array1.length) mergedArray[mergeIndex++] = array1[pointer1++];
                } else {
                    while (pointer2 < array2.length) mergedArray[mergeIndex++] = array2[pointer2++];
                }
            }
        }
        if (mergeIndex % 2 != 0) {
            return mergedArray[mergeIndex / 2];
        } else {
            return (mergedArray[mergeIndex / 2 - 1] + mergedArray[mergeIndex / 2]) / 2.0;
        }
    }
}
