class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int i = 0;
        int j = 0;
        ArrayList<Integer> arr = new ArrayList<>();
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] < nums2[j]) {
                arr.add(nums1[i]);
                i++;
            } else {
                arr.add(nums2[j]);
                j++;
            }
        }
        while (i < nums1.length) {
            arr.add(nums1[i]);
            i++;
        }
        while (j < nums2.length) {
            arr.add(nums2[j]);
            j++;
        }
        int k = arr.size() / 2;
        if (arr.size() % 2 != 0) {
            return (double) arr.get(k);
        }
        return ((double) arr.get(k - 1) + (double) arr.get(k)) / 2.0;
    }
}
