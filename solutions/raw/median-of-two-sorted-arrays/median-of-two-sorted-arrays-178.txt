class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        //length of first array
        int l1 = nums1.length;
        //length of second array
        int l2 = nums2.length;
        int l3 = l1 + l2;
        int[] mergeArr = new int[l3];
        //copying first array in new array
        System.arraycopy(nums1, 0, mergeArr, 0, l1);
        //copying second array in new array
        System.arraycopy(nums2, 0, mergeArr, l1, l2);
        //using inbuilt sorting method
        Arrays.sort(mergeArr);
        if (l3 % 2 == 0) {
            //for even length array
            int index1 = l3 / 2;
            int index2 = (l3 / 2) + 1;
            double median = (double) ((mergeArr[index1 - 1]) + (mergeArr[index2 - 1])) / 2;
            return median;
        }
        int index = (l3 + 1) / 2;
        double median = mergeArr[index - 1];
        return median;
    }
}
