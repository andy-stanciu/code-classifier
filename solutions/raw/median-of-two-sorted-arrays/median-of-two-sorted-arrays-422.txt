class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        // Create a new array with a size equal to the sum of both arrays\' lengths
        int[] combinedArray = new int[m + n];
        // Copy elements from the first array
        System.arraycopy(nums1, 0, combinedArray, 0, m);
        // Copy elements from the second array
        System.arraycopy(nums2, 0, combinedArray, m, n);
        //sort the combined array
        Arrays.sort(combinedArray);
        if ((m + n) % 2 == 0) {
            int val = (m + n) / 2;
            return (double) (combinedArray[val - 1] + combinedArray[val]) / 2;
        } else {
            int val = (m + n) / 2;
            return combinedArray[val];
        }
    }
}
