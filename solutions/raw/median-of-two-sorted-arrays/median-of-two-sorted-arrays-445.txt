class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        int length = m + n;
        int[] arr = new int[length];
        int i = 0, j = 0, k = 0;
        while (i < m && j < n) {
            if (nums1[i] <= nums2[j]) {
                arr[k++] = nums1[i++];
            } else {
                arr[k++] = nums2[j++];
            }
        }
        while (i < m) {
            arr[k++] = nums1[i++];
        }
        while (j < n) {
            arr[k++] = nums2[j++];
        }
        if (length % 2 == 1) {
            return arr[length / 2];
        } else {
            int mid1 = (length / 2) - 1;
            int mid2 = length / 2;
            return (arr[mid1] + arr[mid2]) / 2.0;
        }
    }
}
