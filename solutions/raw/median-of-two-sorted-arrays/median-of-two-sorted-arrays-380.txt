class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        int[] newArr = new int[n1 + n2];
        // Merge the two arrays
        System.arraycopy(nums1, 0, newArr, 0, n1);
        System.arraycopy(nums2, 0, newArr, n1, n2);
        // Sort the merged array
        Arrays.sort(newArr);
        // Calculate the median
        int totalLength = newArr.length;
        if (totalLength % 2 == 0) {
            // If even, return the average of the two middle numbers
            return (newArr[totalLength / 2 - 1] + newArr[totalLength / 2]) / 2.0;
        } else {
            // If odd, return the middle number
            return newArr[totalLength / 2];
        }
    }
}
