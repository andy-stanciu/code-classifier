class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int l = nums1.length + nums2.length;
        int[] total = new int[l / 2 + 1];
        int c1 = 0;
        int c2 = 0;
        for (int i = 0; i <= l / 2; i++) {
            if (c1 < nums1.length && c2 < nums2.length) {
                if (nums1[c1] < nums2[c2]) {
                    total[i] = nums1[c1];
                    c1++;
                } else {
                    total[i] = nums2[c2];
                    c2++;
                }
            } else if (c1 < nums1.length) {
                total[i] = nums1[c1];
                c1++;
            } else if (c2 < nums2.length) {
                total[i] = nums2[c2];
                c2++;
            }
        }
        if (l % 2 == 0) {
            return ((double) (total[l / 2 - 1] + total[l / 2])) / 2;
        } else {
            return (double) total[l / 2];
        }
    }
}
