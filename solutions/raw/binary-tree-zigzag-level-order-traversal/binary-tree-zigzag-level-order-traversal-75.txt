class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> res = new LinkedList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int flag = 0;
        while (!q.isEmpty()) {
            int len = q.size();
            LinkedList<Integer> l = new LinkedList<>();
            for (int i = 0; i < len; i++) {
                TreeNode node = q.poll();
                if (node != null) {
                    q.add(node.left);
                    q.add(node.right);
                    if (flag == 1) {
                        l.addFirst(node.val);
                    } else {
                        l.add(node.val);
                    }
                }
            }
            if (flag == 0)
                flag = 1;
            else
                flag = 0;
            if (!l.isEmpty())
                res.add(l);
        }
        return res;
    }
}
