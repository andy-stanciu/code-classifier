/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> ans = new ArrayList();
        if (root == null) {
            return ans;
        }
        boolean leftToRight = true;
        Queue<TreeNode> qu = new LinkedList();
        qu.offer(root);
        while (!qu.isEmpty()) {
            LinkedList<Integer> level = new LinkedList<>();
            // number of ele in that level
            int size = qu.size();
            for (int i = 0; i < size; i++) {
                TreeNode node = qu.poll();
                //to decide direction
                if (leftToRight) {
                    // Adding node value at the end if left to right
                    level.addLast(node.val);
                } else {
                    // Adding node value at the beginning if right to left
                    level.addFirst(node.val);
                }
                if (node.left != null) {
                    // if left child exist
                    qu.offer(node.left);
                }
                if (node.right != null) {
                    // if right child exist
                    qu.offer(node.right);
                }
            }
            //change direction
            leftToRight = !leftToRight;
            //adding level
            ans.add(level);
        }
        return ans;
    }
}
