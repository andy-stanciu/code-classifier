class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> zigzagList = new ArrayList<>();
        if (root == null)
            return zigzagList;
        Deque<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        boolean popFromBack = true;
        while (!queue.isEmpty()) {
            int nodesAtPrevLevel = queue.size();
            List<Integer> list = new ArrayList<>();
            while (nodesAtPrevLevel > 0) {
                if (popFromBack) {
                    TreeNode node = queue.pollLast();
                    list.add(node.val);
                    if (node.left != null)
                        queue.addFirst(node.left);
                    if (node.right != null)
                        queue.addFirst(node.right);
                } else {
                    TreeNode node = queue.pollFirst();
                    list.add(node.val);
                    if (node.right != null)
                        queue.addLast(node.right);
                    if (node.left != null)
                        queue.addLast(node.left);
                }
                nodesAtPrevLevel--;
            }
            zigzagList.add(list);
            popFromBack = !popFromBack;
        }
        return zigzagList;
    }
}
