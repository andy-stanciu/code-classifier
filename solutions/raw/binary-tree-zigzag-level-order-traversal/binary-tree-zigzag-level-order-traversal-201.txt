/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        zigzag(result, root, 0, true);
        return result;
    }

    private void zigzag(List<List<Integer>> result, TreeNode root, int level, boolean ltr) {
        if (root == null) {
            return;
        }
        List<Integer> list;
        if (result.size() == level) {
            list = new ArrayList<>();
            result.add(list);
        } else {
            list = result.get(level);
        }
        if (ltr) {
            list.add(root.val);
        } else {
            list.add(0, root.val);
        }
        zigzag(result, root.left, level + 1, !ltr);
        zigzag(result, root.right, level + 1, !ltr);
    }
}
