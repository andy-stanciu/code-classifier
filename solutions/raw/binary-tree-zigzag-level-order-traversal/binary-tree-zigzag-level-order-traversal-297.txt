class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> arr = new ArrayList<>();
        int level = 0;
        list(root, arr, 0);
        for (int i = 0; i < arr.size(); i++) {
            if (i % 2 != 0) {
                Collections.reverse(arr.get(i));
            }
        }
        return arr;
    }

    public void list(TreeNode node, List<List<Integer>> arr, int level) {
        if (node == null)
            return;
        List<Integer> temp = new ArrayList<>();
        if (arr.size() <= level)
            arr.add(level, temp);
        arr.get(level).add(node.val);
        level++;
        list(node.left, arr, level);
        list(node.right, arr, level);
    }
}
