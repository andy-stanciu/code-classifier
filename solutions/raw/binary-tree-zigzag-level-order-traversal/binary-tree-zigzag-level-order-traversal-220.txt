/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> r = new ArrayList<>();
        if (root == null)
            return r;
        Queue<TreeNode> q = new LinkedList<>();
        boolean f = true;
        q.add(root);
        while (!q.isEmpty()) {
            int s = q.size();
            LinkedList<Integer> t = new LinkedList<>();
            for (int i = 0; i < s; i++) {
                TreeNode temp = q.poll();
                if (temp.left != null)
                    q.add(temp.left);
                if (temp.right != null)
                    q.add(temp.right);
                if (f)
                    t.add(temp.val);
                else
                    t.addFirst(temp.val);
            }
            f = !f;
            r.add(t);
        }
        return r;
    }
}
