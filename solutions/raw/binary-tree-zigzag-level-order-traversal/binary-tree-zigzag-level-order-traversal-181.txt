class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null)
            return ans;
        Queue<TreeNode> Q = new LinkedList<>();
        Q.add(root);
        TreeNode head;
        int c = 1;
        boolean flag = false;
        int temp;
        while (!Q.isEmpty()) {
            temp = 0;
            var Level = new ArrayList<Integer>(c);
            for (int i = 0; i < c; i++) {
                head = Q.poll();
                Level.add(head.val);
                if (head.left != null) {
                    temp++;
                    Q.add(head.left);
                }
                if (head.right != null) {
                    temp++;
                    Q.add(head.right);
                }
            }
            if (flag == true) {
                flag = false;
                Collections.reverse(Level);
            } else {
                flag = true;
            }
            ans.add(Level);
            c = temp;
        }
        return ans;
    }
}
