/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> arrLst;

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        arrLst = new ArrayList<>();
        helper(root, 0);
        return arrLst;
    }

    private void helper(TreeNode node, int depth) {
        if (node == null)
            return;
        if (arrLst.size() == depth) {
            arrLst.add(new LinkedList<Integer>());
        }
        if (depth % 2 == 0) {
            arrLst.get(depth).add(node.val);
        } else {
            arrLst.get(depth).addFirst(node.val);
        }
        helper(node.left, depth + 1);
        helper(node.right, depth + 1);
    }
}
