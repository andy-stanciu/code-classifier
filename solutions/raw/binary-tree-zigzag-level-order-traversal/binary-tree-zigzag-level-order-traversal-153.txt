/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> lnode = new ArrayList<>();
        if (root == null) {
            return lnode;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        boolean leftToright = true;
        while (!queue.isEmpty()) {
            int quSize = queue.size();
            List<Integer> curr = new ArrayList<>();
            for (int i = 0; i < quSize; i++) {
                TreeNode treeVal = queue.poll();
                if (leftToright) {
                    curr.add(treeVal.val);
                } else {
                    curr.add(0, treeVal.val);
                }
                if (treeVal.left != null) {
                    queue.add(treeVal.left);
                }
                if (treeVal.right != null) {
                    queue.add(treeVal.right);
                }
            }
            lnode.add(curr);
            leftToright = !leftToright;
        }
        // lnode.add(que);
        return lnode;
    }
}
