class Solution {

    public void traverse(TreeNode root, List<List<Integer>> zigzagOrderList, int level) {
        if (root == null)
            return;
        if (zigzagOrderList.size() == level) {
            List<Integer> currLevelNodes = new ArrayList<>();
            currLevelNodes.add(root.val);
            zigzagOrderList.add(currLevelNodes);
        } else {
            if (level % 2 == 1) {
                zigzagOrderList.get(level).add(0, root.val);
            } else {
                zigzagOrderList.get(level).add(root.val);
            }
        }
        traverse(root.left, zigzagOrderList, level + 1);
        traverse(root.right, zigzagOrderList, level + 1);
    }

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> zigzagOrderList = new ArrayList<>();
        traverse(root, zigzagOrderList, 0);
        return zigzagOrderList;
    }
}
