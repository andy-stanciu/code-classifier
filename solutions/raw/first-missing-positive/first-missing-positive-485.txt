class Solution {

    public int firstMissingPositive(int[] nums) {
        int n = nums.length;
        Arrays.sort(nums);
        int res = 1;
        for (int i = 0; i < n; i++) {
            if (nums[i] == res) {
                res++;
            } else if (nums[i] > res) {
                break;
            }
        }
        return res;
    }
}
