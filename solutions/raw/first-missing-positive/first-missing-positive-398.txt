class Solution {

    public int firstMissingPositive(int[] nums) {
        if (nums.length == 1) {
            return nums[0] == 1 ? 2 : 1;
        }
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            map.put(num, num);
        }
        for (int i = 1; i < nums.length + 2; i++) {
            if (!map.containsKey(i))
                return i;
        }
        return -1;
    }
}
