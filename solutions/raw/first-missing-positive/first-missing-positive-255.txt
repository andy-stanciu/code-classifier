class Solution {

    public int firstMissingPositive(int[] nums) {
        boolean[] arr = new boolean[nums.length + 1];
        for (int num : nums) {
            if (num >= 0 && num < arr.length) {
                arr[num] = true;
            }
        }
        int i = 1;
        while (i < arr.length) {
            if (!arr[i])
                return i;
            i++;
        }
        return i;
    }
}
