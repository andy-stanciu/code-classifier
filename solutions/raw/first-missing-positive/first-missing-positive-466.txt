class Solution {

    public int firstMissingPositive(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] >= 0)
                set.add(nums[i]);
        }
        int small = 1;
        while (set.contains(small)) {
            small++;
        }
        return small;
    }
}
