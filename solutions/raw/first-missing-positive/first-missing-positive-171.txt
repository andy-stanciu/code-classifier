// first made all numbers in array positive
//if any number is found in range [1,N] then attach -ve sign to the corresponding index. So for 1, 0th //element becomes -ve, for 2 it is 1st considering 0 based index
class Solution {

    public int firstMissingPositive(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] <= 0 || nums[i] > nums.length) {
                nums[i] = nums.length + 1;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            int num = Math.abs(nums[i]);
            if (num > nums.length) {
                continue;
            }
            // -1 for zero index based array
            num--;
            if (nums[num] > 0) {
                // prevents double negative operations or duplicate numbers
                nums[num] = -1 * nums[num];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] >= 0) {
                //first non negative number is absent
                return i + 1;
            }
        }
        //if all numbers between 1 to n is found then n+1
        return nums.length + 1;
    }
}
