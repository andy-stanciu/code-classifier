class Solution {

    public int firstMissingPositive(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            int ci = nums[i] - 1;
            if (nums[i] > 0 && nums[i] <= nums.length && nums[ci] != nums[i]) {
                swap(nums, i, ci);
            } else {
                i++;
            }
        }
        for (int index = 0; index < nums.length; index++) if (nums[index] != index + 1) {
            return index + 1;
        }
        return nums.length + 1;
    }

    static void swap(int[] nums, int faake, int riyal) {
        int temp = nums[faake];
        nums[faake] = nums[riyal];
        nums[riyal] = temp;
    }
}
