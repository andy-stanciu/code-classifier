class Solution {

    public int firstMissingPositive(int[] nums) {
        // System.out.print(Arrays.toString(cyclicsort(nums)));
        cyclicsort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        // 0 1 2 3
        // case 2
        // if array doesnot have missing numbers then next element is consider as missing here (4)
        return nums.length + 1;
    }

    public void cyclicsort(int[] arr) {
        int i = 0;
        while (i < arr.length) {
            // starting from 0 thatswhy
            int correctIndex = arr[i] - 1;
            // checking number is positive, and number is not exceed from arr.length
            if (arr[i] > 0 && arr[i] <= arr.length && arr[i] != arr[correctIndex]) {
                // swapping
                int temp = arr[i];
                arr[i] = arr[correctIndex];
                arr[correctIndex] = temp;
            } else {
                i++;
            }
        }
    }
}
