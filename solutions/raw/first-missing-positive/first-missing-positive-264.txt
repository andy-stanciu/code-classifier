class Solution {

    public int firstMissingPositive(int[] nums) {
        if (nums.length == 1) {
            if (nums[0] == 1) {
                return 2;
            } else {
                return 1;
            }
        }
        Arrays.sort(nums);
        int j = 0;
        while (nums[j] <= 0) {
            if (j < nums.length - 1)
                j++;
            else
                return 1;
        }
        if (nums[0] > 1) {
            return 1;
        }
        if (nums[j] > 1)
            return 1;
        for (int i = ++j; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] > 1) {
                return nums[i - 1] + 1;
            }
        }
        return nums[nums.length - 1] + 1;
    }
}
