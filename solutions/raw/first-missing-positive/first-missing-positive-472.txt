class Solution {

    public int firstMissingPositive(int[] nums) {
        //so here this question is hard because we have not been given that it is cycle sort.
        //other sorting algorithm does not give O(n) they need extra time complexity.
        //and we have to solve it in O(n) hence by using cyle sort we can sort it in O(n).
        int i = 0;
        int n = nums.length;
        while (i < n) {
            int correctidx = nums[i] - 1;
            if (nums[i] <= n && nums[i] > 0 && nums[i] != nums[correctidx]) {
                swap(nums, i, correctidx);
            } else {
                i++;
            }
        }
        for (int k = 0; k < n; k++) {
            if (nums[k] != k + 1) {
                return k + 1;
            }
        }
        return n + 1;
    }

    public static void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
