class Solution {

    public int firstMissingPositive(int[] nums) {
        int k = 1;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < k) {
                continue;
            } else if (nums[i] == k) {
                k++;
            } else {
                break;
            }
        }
        return k;
    }
}
