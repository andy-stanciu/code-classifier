class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        if (nums[nums.length - 1] <= 0 || nums[0] > 1)
            return 1;
        int n = nums[0];
        boolean vis = false;
        for (int i = 1; i < nums.length; i++) {
            if (n == nums[i])
                continue;
            if (n == 1)
                vis = true;
            if (n <= 0) {
                n = nums[i];
            } else if (nums[i] == n + 1) {
                n = n + 1;
            } else if (vis) {
                return n + 1;
            }
        }
        if (n == 1)
            vis = true;
        if (vis)
            return n + 1;
        return 1;
    }
}
