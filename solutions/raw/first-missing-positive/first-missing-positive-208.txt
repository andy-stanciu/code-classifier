class Solution {

    public int firstMissingPositive(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] - 1 != i) {
                if (arr[i] >= arr.length || arr[i] <= 0 || arr[i] == arr[arr[i] - 1]) {
                    continue;
                }
                int temp = arr[i];
                arr[i] = arr[temp - 1];
                arr[temp - 1] = temp;
                i--;
            }
        }
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != i + 1) {
                return i + 1;
            }
        }
        return arr.length + 1;
    }
}
