class Solution {

    public int firstMissingPositive(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            int val = nums[i];
            if (val <= 0) {
                nums[i] = 0;
            }
        }
        for (int i = 0; i < n; i++) {
            int val = nums[i];
            if (val > 0 && val <= n) {
                if (nums[val - 1] > 0) {
                    nums[val - 1] = -nums[val - 1];
                } else if (nums[val - 1] == 0) {
                    nums[val - 1] = -val;
                }
            } else if (val < 0) {
                if (nums[-val - 1] > 0) {
                    nums[-val - 1] = -nums[-val - 1];
                } else if (nums[-val - 1] == 0) {
                    nums[-val - 1] = val;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] >= 0)
                return i + 1;
        }
        return n + 1;
    }
}
