class Solution {

    public int firstMissingPositive(int[] nums) {
        HashSet<Integer> h = new HashSet<Integer>();
        int max = 0;
        for (int i : nums) {
            h.add(i);
            if (i > max)
                max = i;
        }
        for (int i = 1; i < max; i++) {
            if (!h.contains(i))
                return i;
        }
        return max + 1;
    }
}
