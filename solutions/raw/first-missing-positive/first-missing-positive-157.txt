class Solution {

    public int firstMissingPositive(int[] nums) {
        int i = 0, j = 0;
        while (j < nums.length) {
            if (nums[j] <= 0) {
                j++;
            } else {
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                i++;
                j++;
            }
        }
        for (j = 0; j < i; j++) {
            int temp = Math.abs(nums[j]);
            if (temp - 1 < i && nums[temp - 1] > 0)
                nums[temp - 1] = -nums[temp - 1];
        }
        for (j = 0; j < i; j++) {
            if (nums[j] > 0)
                return j + 1;
        }
        return i + 1;
    }
}
