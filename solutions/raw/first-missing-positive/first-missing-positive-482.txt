class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        int minmiss = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0) {
                if (nums[i] == minmiss) {
                    minmiss++;
                } else if (nums[i] > minmiss) {
                    return minmiss;
                }
            }
        }
        return minmiss;
    }
}
