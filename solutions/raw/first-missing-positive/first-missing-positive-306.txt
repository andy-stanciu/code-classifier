class Solution {

    public int firstMissingPositive(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        Boolean oneFound = false;
        int biggestNumber = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 1)
                oneFound = true;
            set.add(nums[i]);
            if (nums[i] > biggestNumber)
                biggestNumber = nums[i];
        }
        if (oneFound == false)
            return 1;
        for (int j = 1; j < biggestNumber; j++) {
            if (!set.contains(j))
                return j;
        }
        return biggestNumber + 1;
    }
}
