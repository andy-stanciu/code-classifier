class Solution {

    public int firstMissingPositive(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        int missingPositive = 1;
        while (set.contains(missingPositive)) {
            missingPositive++;
        }
        return missingPositive;
    }
}
