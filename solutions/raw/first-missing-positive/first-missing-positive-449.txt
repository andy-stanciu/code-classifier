class Solution {

    public int firstMissingPositive(int[] nums) {
        HashSet<Integer> check = new HashSet<>();
        for (int i : nums) {
            check.add(i);
        }
        int i = 1;
        while (true) {
            if (!check.contains(i)) {
                break;
            }
            i++;
        }
        return i;
    }
}
