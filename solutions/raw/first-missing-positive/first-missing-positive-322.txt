class Solution {

    boolean binarySrch(int[] a, int len, int key) {
        int start = 0, end = len - 1;
        while (start <= end) {
            int mid = (start + end) / 2;
            if (a[mid] == key)
                return true;
            if (a[mid] < key)
                start = mid + 1;
            else
                end = mid - 1;
        }
        return false;
    }

    public int firstMissingPositive(int[] nums) {
        int len = nums.length;
        Arrays.sort(nums);
        int val = 1;
        while (binarySrch(nums, len, val)) {
            val++;
        }
        return val;
    }
}
