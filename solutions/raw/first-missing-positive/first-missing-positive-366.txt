class Solution {

    public int firstMissingPositive(int[] nums) {
        int[] s = new int[nums.length];
        for (int n : nums) {
            if (n > 0 && n <= nums.length) {
                s[n - 1] = n;
            }
        }
        for (int i = 0; i < s.length; i++) {
            if (s[i] == 0) {
                return i + 1;
            }
        }
        return nums.length + 1;
    }
}
