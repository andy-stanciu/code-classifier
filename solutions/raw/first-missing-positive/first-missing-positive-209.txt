class Solution {

    public int firstMissingPositive(int[] nums) {
        HashSet<Integer> b = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            b.add(nums[i]);
        }
        for (int i = 1; i > 0; i++) {
            if (!b.contains(i)) {
                return i;
            }
        }
        return 0;
    }
}
