class Solution {

    public int firstMissingPositive(int[] nums) {
        boolean[] b = new boolean[nums.length + 1];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0 && nums[i] <= nums.length) {
                b[nums[i]] = true;
            }
        }
        for (int i = 1; i < b.length; i++) {
            if (!b[i])
                return i;
        }
        return b.length;
    }
}
