class Solution {

    public int firstMissingPositive(int[] nums) {
        HashSet<Integer> m = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0) {
                m.add(nums[i]);
            }
        }
        int n = 1;
        int out = 0;
        while (n > 0) {
            if (!m.contains(n)) {
                out = n;
                break;
            }
            n++;
        }
        return out;
    }
}
