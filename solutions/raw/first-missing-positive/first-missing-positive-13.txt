class Solution {

    public int firstMissingPositive(int[] nums) {
        //create a new HashSet and store all greater than 0
        HashSet<Integer> hst = new HashSet<>();
        //create a new variable store max value
        int maxVal = Integer.MIN_VALUE;
        //Iteration loop zero index to nums.length
        for (int num : nums) {
            if (num > 0) {
                //check greater than 0
                //here store num in HashSet
                hst.add(num);
                //find max value
                maxVal = Math.max(maxVal, num);
            }
            if (num == Integer.MAX_VALUE) {
                // Handling the large value
                maxVal = nums.length;
            }
        }
        //here check if maxVal is not found so return 1 That means all arrays number lesser than 1
        if (maxVal == Integer.MIN_VALUE)
            return 1;
        for (int i = 1; i <= maxVal + 1; i++) {
            //here check number not contains in HashSet so return i
            if (!hst.contains(i)) {
                return i;
            }
        }
        //finally return max Value + 1
        return maxVal + 1;
    }
}
