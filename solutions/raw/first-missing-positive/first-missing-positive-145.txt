class Solution {

    public void swap(int[] nums, int start, int end) {
        int temp = nums[start];
        nums[start] = nums[end];
        nums[end] = temp;
    }

    public int firstMissingPositive(int[] nums) {
        int left = 0;
        while (left < nums.length) {
            int number = nums[left] - 1;
            // System.out.print(left+" ");
            // System.out.print(nums[left]+" ");
            if (number < nums.length && number >= 0 && number != left) {
                if (nums[left] == nums[number]) {
                    left++;
                } else {
                    swap(nums, left, number);
                }
            } else {
                left++;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        return nums.length + 1;
    }
}
