class Solution {

    public int firstMissingPositive(int[] nums) {
        int n = nums.length, lowest = 1;
        Arrays.sort(nums);
        if (nums[n - 1] <= 0)
            return 1;
        for (int i = 0; i < n; i++) {
            if (nums[i] > lowest) {
                return lowest;
            }
            if (nums[i] == lowest) {
                lowest++;
            }
        }
        return lowest;
    }
}
