class Solution {

    public int firstMissingPositive(int[] nums) {
        boolean[] contains = new boolean[nums.length];
        for (int i : nums) {
            if (i <= nums.length && i > 0) {
                contains[i - 1] = true;
            }
        }
        for (int i = 0; i < contains.length; i++) {
            if (!contains[i]) {
                return i + 1;
            }
        }
        return nums.length + 1;
    }
}
