class Solution {

    public int firstMissingPositive(int[] nums) {
        // Sort the array
        Arrays.sort(nums);
        // The smallest positive integer we are looking for
        int expected = 1;
        for (int num : nums) {
            // Skip non-positive numbers and duplicates
            if (num == expected) {
                expected++;
            }
        }
        // The first missing positive integer
        return expected;
    }
}
