class Solution {

    public static int firstMissingPositive(int[] nums) {
        HashSet<Integer> st = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            st.add(nums[i]);
        }
        for (int i = 1; i <= 2147483647; i++) {
            if (!st.contains(i)) {
                return i;
            }
        }
        return 2147483647;
    }
}
