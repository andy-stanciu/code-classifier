class Solution {

    public int firstMissingPositive(int[] nums) {
        int len = nums.length;
        String[] str = new String[len];
        for (int a = 0; a < len; a++) {
            if ((nums[a] <= len) && (nums[a] > 0)) {
                str[nums[a] - 1] = "";
            }
        }
        for (int a = 0; a < len; a++) {
            if (str[a] == null) {
                return a + 1;
            }
        }
        return len + 1;
    }
}
