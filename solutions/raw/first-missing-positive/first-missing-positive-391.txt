class Solution {

    public int firstMissingPositive(int[] nums) {
        int[] arr = new int[nums.length + 2];
        for (var temp : nums) if (temp <= nums.length && temp > 0)
            arr[temp] = 1;
        for (int i = 1; i < arr.length; i++) if (arr[i] == 0)
            return i;
        return -1;
    }
}
