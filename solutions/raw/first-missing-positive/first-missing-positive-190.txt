class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        int num = -1;
        if (nums[nums.length - 1] < 1 || nums[0] > 1) {
            num = 1;
            return num;
        }
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] + 1 != nums[i + 1] && nums[i] != nums[i + 1]) {
                num = nums[i] + 1;
                if (num < 1 && nums[i + 1] > 1) {
                    num = 1;
                    break;
                } else if (num > 0) {
                    break;
                }
            }
        }
        if (num < 0) {
            num = nums[nums.length - 1] + 1;
        }
        return num;
    }
}
