class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        HashSet<Integer> hs = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            hs.add(nums[i]);
        }
        int i = 1;
        while (i <= nums.length) {
            if (!hs.contains(i))
                return i;
            i++;
        }
        return i;
    }
}
