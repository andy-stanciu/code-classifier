class Solution {

    public int firstMissingPositive(int[] nums) {
        HashSet<Integer> h = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            h.add(nums[i]);
        }
        int max = Collections.max(h);
        //for the corner cases where max value is 0 or less than 0 or negative elements. For example [-5, -2], [0].
        if (max <= 0) {
            return 1;
        }
        for (int i = 1; i <= max; i++) {
            // below if is added for corner cases like [1, 2]
            if (i == max) {
                return max + 1;
            }
            if (!h.contains(i)) {
                return i;
            }
        }
        return 0;
    }
}
