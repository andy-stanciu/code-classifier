class Solution {

    public int firstMissingPositive(int[] nums) {
        HashMap<Integer, Boolean> m = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            m.put(nums[i], true);
        }
        for (int i = 1; i <= nums.length + 1; i++) {
            if (!m.containsKey(i)) {
                return i;
            }
        }
        return 0;
    }
}
