class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        int s = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] <= 0 || s - 1 == nums[i])
                continue;
            if (s != nums[i])
                return s;
            s++;
        }
        //System.out.println(s);
        return s == 1 ? s : nums[nums.length - 1] + 1;
    }
}
