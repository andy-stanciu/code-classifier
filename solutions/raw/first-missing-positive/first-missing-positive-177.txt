class Solution {

    public int firstMissingPositive(int[] nums) {
        HashMap<Integer, Boolean> map = new HashMap<>();
        for (int num : nums) {
            if (num > 0) {
                map.put(num, true);
            }
        }
        int smallestMissing = 1;
        while (map.containsKey(smallestMissing)) {
            smallestMissing++;
        }
        return smallestMissing;
    }
}
