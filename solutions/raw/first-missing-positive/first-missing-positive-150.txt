class Solution {

    public int firstMissingPositive(int[] nums) {
        int[] arr = new int[nums.length + 1];
        if (nums.length == 1 && nums[0] == 1)
            return 2;
        for (int num : nums) if (num > 0 && num <= nums.length)
            arr[num] = 1;
        int i = 1;
        while (i <= nums.length) {
            if (arr[i] == 0)
                return i;
            i++;
        }
        return i;
    }
}
