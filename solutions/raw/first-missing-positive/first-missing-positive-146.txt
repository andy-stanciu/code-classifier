class Solution {

    public int firstMissingPositive(int[] nums) {
        int size = nums.length;
        boolean[] counts = new boolean[size + 1];
        for (int i = 0; i < size; i++) {
            if (nums[i] > 0 && nums[i] <= size) {
                counts[nums[i]] = true;
            }
        }
        for (int i = 1; i <= size; i++) {
            if (!counts[i]) {
                return i;
            }
        }
        return size + 1;
    }
}
