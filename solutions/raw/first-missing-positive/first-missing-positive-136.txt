class Solution {

    public int firstMissingPositive(int[] nums) {
        int missingPositive = 1;
        Arrays.sort(nums);
        for (int num : nums) {
            if (num == missingPositive) {
                missingPositive++;
            }
        }
        return missingPositive;
    }
}
//  TC : O(N)
//  SC : O(1)
