class Solution {

    public int firstMissingPositive(int[] nums) {
        int min = 0;
        int n = nums.length;
        int pos = 0;
        Arrays.sort(nums);
        if (nums[n - 1] <= 0) {
            return 1;
        }
        //Find first position where positive no. starts
        if (nums[0] >= 0) {
            pos = 0;
        } else {
            pos = firstPositive(nums, 0, n - 2);
        }
        for (int i = pos; i < n; i++) {
            if (nums[i] == min || nums[i] == min + 1) {
                min = nums[i];
            } else {
                return min + 1;
            }
        }
        return min + 1;
    }

    //Binary search
    public int firstPositive(int[] A, int left, int right) {
        if (left <= right) {
            int mid = left + (right - left) / 2;
            if (A[mid] <= 0 && A[mid + 1] >= 0) {
                return mid + 1;
            } else if (A[mid] <= 0 && A[mid + 1] <= 0) {
                return firstPositive(A, mid + 1, right);
            } else {
                return firstPositive(A, left, mid - 1);
            }
        } else {
            return 0;
        }
    }
}
