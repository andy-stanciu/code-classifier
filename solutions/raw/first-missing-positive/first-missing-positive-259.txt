class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        int n = 1, j = 0;
        while (j < nums.length) {
            if (nums[j] >= 1) {
                if (j != 0 && nums[j] != nums[j - 1]) {
                    if (nums[j] != n) {
                        return n;
                    }
                    j++;
                    n++;
                } else if (j == 0) {
                    if (nums[j] != n) {
                        return n;
                    }
                    j++;
                    n++;
                } else {
                    j++;
                }
            } else {
                j++;
            }
        }
        return n;
    }
}
