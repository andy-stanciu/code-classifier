class Solution {

    public int firstMissingPositive(int[] nums) {
        Arrays.sort(nums);
        Set<Integer> hs = new HashSet<>();
        for (int i : nums) {
            hs.add(i);
        }
        for (int i = 1; i <= nums.length; i++) {
            if (!(hs.contains(i))) {
                return i;
            }
        }
        return nums[nums.length - 1] + 1;
    }
}
