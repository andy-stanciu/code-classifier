class Solution {

    public int firstMissingPositive(int[] nums) {
        boolean found = false;
        int j = 1;
        while (true) {
            found = false;
            for (int i : nums) {
                if (j == i) {
                    found = true;
                    break;
                }
            }
            if (!found) {
                return j;
            }
            j++;
        }
    }
}
