class Solution {

    public int firstMissingPositive(int[] arr) {
        int n = arr.length;
        boolean one = false;
        for (int i = 0; i < n; i++) {
            if (arr[i] == 1)
                one = true;
            if (arr[i] < 1 || arr[i] > n)
                arr[i] = 1;
        }
        if (one == false)
            return 1;
        for (int i = 0; i < n; i++) {
            int val = Math.abs(arr[i]);
            int idx = val - 1;
            arr[idx] = -Math.abs(arr[idx]);
        }
        for (int i = 0; i < n; i++) {
            if (arr[i] > 0)
                return i + 1;
        }
        return n + 1;
    }
}
