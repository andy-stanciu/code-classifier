class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totaldifference = 0;
        int ans = 0;
        int difference = 0;
        int fuel = 0;
        for (int i = 0; i < cost.length; i++) {
            difference = gas[i] - cost[i];
            totaldifference = difference + totaldifference;
            fuel = fuel + difference;
            if (fuel < 0) {
                ans = i + 1;
                fuel = 0;
            }
        }
        return (totaldifference < 0) ? -1 : ans;
    }
}
