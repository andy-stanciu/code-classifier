class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int gasSum = 0;
        for (int i : gas) gasSum += i;
        int costSum = 0;
        for (int i : cost) costSum += i;
        if (gasSum < costSum)
            return -1;
        int sum = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            sum += gas[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                start = i + 1;
            }
        }
        return start;
    }
}
