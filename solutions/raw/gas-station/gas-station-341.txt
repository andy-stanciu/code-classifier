class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int currTank = 0, maxTank = 0, resInd = 0;
        for (int i = gas.length - 1; i >= 0; i--) {
            currTank += gas[i] - cost[i];
            if (currTank >= maxTank) {
                maxTank = currTank;
                resInd = i;
            }
        }
        return currTank < 0 ? -1 : resInd;
    }
}
