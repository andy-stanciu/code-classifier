class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total_gas = 0;
        int temp_gas = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            total_gas += gas[i] - cost[i];
            temp_gas += gas[i] - cost[i];
            if (temp_gas < 0) {
                temp_gas = 0;
                start = i + 1;
            }
        }
        return total_gas >= 0 ? start : -1;
    }
}
