class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalcost = 0;
        int sum = 0;
        int idx = 0;
        for (int i = 0; i < gas.length; i++) {
            totalcost += gas[i] - cost[i];
            sum += gas[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                idx = i + 1;
            }
        }
        if (totalcost < 0)
            return -1;
        else {
            return idx;
        }
    }
}
