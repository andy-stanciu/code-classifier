class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int[] diff = new int[n];
        for (int i = 0; i < n; i++) diff[i] = gas[i] - cost[i];
        for (int start = 0; start < n; start++) {
            if (diff[start] < 0)
                continue;
            int gasInCar = diff[start];
            for (int curr = start; curr < start + n; curr++) {
                gasInCar += diff[(curr + 1) % n];
                if (gasInCar < 0)
                    break;
            }
            if (gasInCar >= 0)
                return start;
        }
        return -1;
    }
}
