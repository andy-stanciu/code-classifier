class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int gasRemainingOrFillTo = 0;
        int answer = 0;
        for (int i = 0; i < gas.length; i++) {
            gasRemainingOrFillTo += gas[i] - cost[i];
        }
        if (gasRemainingOrFillTo < 0)
            return -1;
        gasRemainingOrFillTo = 0;
        for (int i = 0; i < gas.length; i++) {
            gasRemainingOrFillTo += gas[i] - cost[i];
            if (gasRemainingOrFillTo < 0) {
                gasRemainingOrFillTo = 0;
                answer = i + 1;
            }
        }
        return answer;
    }
}
