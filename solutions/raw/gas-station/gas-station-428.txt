class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int index = 0, token = 0, total = 0;
        for (int i = 0; i < gas.length; i++) {
            gas[i] = gas[i] - cost[i];
            total += gas[i];
        }
        if (total < 0) {
            return -1;
        }
        for (int i = 0; i < gas.length; i++) {
            token += gas[i];
            if (gas[i] > token) {
                index = i;
                token = gas[i];
            }
        }
        return index;
    }
}
