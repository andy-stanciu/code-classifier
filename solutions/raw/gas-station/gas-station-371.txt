class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int g = 0;
        int c = 0;
        for (int i = 0; i < n; i++) {
            g += gas[i];
            c += cost[i];
        }
        if (c > g) {
            return -1;
        }
        int res = 0;
        int total = 0;
        for (int i = 0; i < n; i++) {
            total += gas[i] - cost[i];
            if (total < 0) {
                total = 0;
                res = i + 1;
            }
        }
        return res;
    }
}
