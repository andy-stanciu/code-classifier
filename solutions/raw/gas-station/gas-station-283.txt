class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalNetGas = 0;
        for (int i = 0; i < gas.length; i++) {
            totalNetGas += gas[i] - cost[i];
        }
        // If total net gas is less than zero, return -1 (impossible to complete the circuit)
        if (totalNetGas < 0) {
            return -1;
        }
        // Initialize starting index and current net gas
        int startingIndex = 0;
        int currentNetGas = 0;
        // Iterate through each station
        for (int i = 0; i < gas.length; i++) {
            // Calculate net gas at the current station
            currentNetGas += gas[i] - cost[i];
            // If current net gas becomes negative, reset starting index and current net gas
            if (currentNetGas < 0) {
                startingIndex = i + 1;
                currentNetGas = 0;
            }
        }
        // Return the starting index
        return startingIndex;
    }
}
