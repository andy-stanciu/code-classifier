class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int start = 0;
        int totgas = 0;
        int fuelreq = 0;
        int remaining = 0;
        for (int i = 0; i < gas.length; i++) {
            totgas = totgas + gas[i];
            fuelreq = fuelreq + cost[i];
        }
        if (totgas < fuelreq) {
            return -1;
        }
        for (int i = 0; i < gas.length; i++) {
            remaining = remaining + gas[i] - cost[i];
            if (remaining < 0) {
                start = i + 1;
                remaining = 0;
            }
        }
        return start;
    }
}
