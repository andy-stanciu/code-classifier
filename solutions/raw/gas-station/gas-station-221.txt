class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int index = gas.length;
        int count = 0;
        int startingindex = 0;
        int totgas = 0, totcost = 0;
        for (int i = 0; i < gas.length; i++) {
            // add the total gas
            totgas += gas[i];
            //add the total cost of gas required for move to next Station
            totcost += cost[i];
            count += gas[i] - cost[i];
            if (count < 0) {
                startingindex = i + 1;
                count = 0;
            }
        }
        if (totgas < totcost) {
            return -1;
        } else {
            return startingindex;
        }
    }
}
