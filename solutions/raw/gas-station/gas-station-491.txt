class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        //total amount of gas in all gas station;
        int totalgas = 0;
        //total amount of dist in travelling from one station to all
        int totalcost = 0;
        //station in circural ;
        for (int i = 0; i < gas.length; i++) {
            totalgas += gas[i];
            totalcost += cost[i];
        }
        if (totalgas < totalcost) {
            //if the amount of gas is less then total dist return -1;
            return -1;
        }
        //checking at all station that the gas is sufficent to travel
        int remmaningGas = 0;
        //to next station if not then move to next one and assign remmingGas to zero;
        int st = 0;
        for (int i = 0; i < gas.length; i++) {
            remmaningGas += gas[i] - cost[i];
            if (remmaningGas < 0) {
                st = i + 1;
                remmaningGas = 0;
            }
        }
        return st;
    }
}
