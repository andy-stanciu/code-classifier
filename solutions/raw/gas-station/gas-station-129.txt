class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int start = 0, tank = 0, diff = 0;
        for (int i = 0; i < gas.length; i++) {
            diff += gas[i] - cost[i];
            tank += gas[i] - cost[i];
            if (tank < 0) {
                tank = 0;
                start = i + 1;
            }
        }
        // if(diff<0){
        //     return -1;
        // }
        return diff < 0 ? -1 : start;
    }
}
