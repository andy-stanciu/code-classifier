class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        for (int i = 0; i < n; i++) {
            int tank = 0;
            boolean roundTrip = true;
            for (int j = i; j < n + i + 1; j++) {
                tank = tank + gas[j % n] - cost[j % n];
                // if at any point, our gas in tank becomes Negative, we can not reach the next station...
                if (tank < 0) {
                    roundTrip = false;
                    break;
                }
            }
            if (roundTrip)
                return i;
        }
        return -1;
    }
}
