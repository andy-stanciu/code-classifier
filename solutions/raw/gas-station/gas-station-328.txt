class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int extra = 0;
        int startIndex = 0;
        int totalExtra = 0;
        for (int i = 0; i < gas.length; i++) {
            int diff = gas[i] - cost[i];
            totalExtra += diff;
            extra += diff;
            if (extra < 0) {
                //not choosing this index because insuffient fuel
                extra = 0;
                startIndex = i + 1;
            }
        }
        return (totalExtra < 0) ? -1 : startIndex;
    }
}
