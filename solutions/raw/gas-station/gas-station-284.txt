class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int startStation = 0;
        int gasSpent = 0;
        int currentTank = gas[0] - cost[0];
        for (int station = 1; station < gas.length; station++) {
            if (currentTank < 0) {
                startStation = station;
                gasSpent += (-currentTank);
                currentTank = 0;
            }
            currentTank += gas[station] - cost[station];
        }
        if (currentTank >= gasSpent) {
            return startStation;
        } else {
            return -1;
        }
    }
}
