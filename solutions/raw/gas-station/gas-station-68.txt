class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int tmpSum = 0;
        int index = 0;
        int sum = gas[0] - cost[0];
        for (int i = 1; i < gas.length; i++) {
            if (sum < 0) {
                index = i;
                //transfering to temp sum to later check in the output whether the started index has sufficient petrol for prev!
                tmpSum += sum;
                sum = 0;
            }
            sum += gas[i] - cost[i];
        }
        sum += tmpSum;
        return sum >= 0 ? index : -1;
    }
}
