class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total = 0, gasSum = 0, start = 0;
        for (int i = 0; i < gas.length; i++) {
            int cur = gas[i] - cost[i];
            total += cur;
            gasSum += cur;
            // we cal from the start/index_gasSum==0
            // if gas sum < 0, then we cannot reach to the next from start/index_gasSum==0
            // if gas sum < 0. then we start from the next
            if (gasSum < 0) {
                gasSum = 0;
                // if to the end, then also means cannot goal
                start = i + 1;
            }
        }
        if (total < 0)
            return -1;
        return start;
    }
}
