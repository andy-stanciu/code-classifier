class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int g = 0, c = 0;
        int total = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            g += gas[i];
            c += cost[i];
            total += gas[i] - cost[i];
            if (total < 0) {
                start = i + 1;
                total = 0;
            }
        }
        if (g < c) {
            return -1;
        }
        return start;
    }
}
