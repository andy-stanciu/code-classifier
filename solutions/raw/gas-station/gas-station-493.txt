class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int defecit = 0;
        int balance = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            balance += gas[i] - cost[i];
            if (balance < 0) {
                defecit += balance;
                start = i + 1;
                balance = 0;
            }
        }
        return (defecit + balance >= 0) ? start : -1;
    }
}
