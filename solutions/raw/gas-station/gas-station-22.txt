class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int total_gas = 0;
        int gasVal = 0;
        int start = 0;
        // traverse both array with a single loop
        for (int i = 0; i < n; i++) {
            total_gas += gas[i] - cost[i];
            gasVal += gas[i] - cost[i];
            if (gasVal < 0) {
                gasVal = 0;
                start = i + 1;
            }
        }
        // if total gas is remaining then return this otherwise return -1
        return (total_gas < 0) ? -1 : start;
    }
}
