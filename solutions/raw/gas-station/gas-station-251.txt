class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0, totalCost = 0;
        int ind = 0, cap = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
            cap += gas[i] - cost[i];
            if (cap < 0) {
                cap = 0;
                ind = i + 1;
            }
        }
        return totalGas >= totalCost ? ind : -1;
    }
}
