class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int totalGas = 0;
        int totalCost = 0;
        for (int a : gas) {
            totalGas += a;
        }
        for (int b : cost) {
            totalCost += b;
        }
        //IF THE GAS WE HAVE IS LESS THAN WE NEED TO SPEND THEN WE CAN NOT TRAEL
        if (totalGas < totalCost)
            return -1;
        int index = 0;
        int remainingFuel = 0;
        for (int i = 0; i < n; i++) {
            remainingFuel += gas[i] - cost[i];
            //IF remainig fuel gets less than 0 then we will make remainig fuel as 0 and increase index by i + 1
            if (remainingFuel < 0) {
                index = i + 1;
                remainingFuel = 0;
            }
        }
        return index;
    }
}
