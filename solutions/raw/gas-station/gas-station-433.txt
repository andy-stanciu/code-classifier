class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // Total gas in the tank
        int totalGas = 0;
        // Gas in the tank at the current station
        int currentGas = 0;
        // Starting station index
        int startStation = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i] - cost[i];
            currentGas += gas[i] - cost[i];
            // If the gas at the current station is negative, we can\'t start from here
            if (currentGas < 0) {
                // Move to the next station
                startStation = i + 1;
                // Reset gas in the tank
                currentGas = 0;
            }
        }
        // If the total gas is non-negative, it means we can complete the circuit
        return totalGas >= 0 ? startStation : -1;
    }
}
