class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int alreadyVisited = 0;
        for (int i = 0; i < gas.length; i++) {
            int tank = gas[i] - cost[i];
            int j = i + 1;
            for (; tank >= 0 && j < gas.length; j++) {
                tank += gas[j] - cost[j];
            }
            alreadyVisited += tank;
            if (Math.min(tank, alreadyVisited) >= 0)
                return i;
            else
                i = --j;
        }
        return -1;
    }
}
