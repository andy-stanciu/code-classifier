class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0;
        int totalCost = 0;
        int currentGas = 0;
        int startStation = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
            currentGas += gas[i] - cost[i];
            // If currentGas is negative, we can\'t start from startStation to i
            if (currentGas < 0) {
                startStation = i + 1;
                currentGas = 0;
            }
        }
        // If totalGas is less than totalCost, we can\'t complete the circuit
        if (totalGas < totalCost) {
            return -1;
        } else {
            return startStation;
        }
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        int[] gas1 = { 1, 2, 3, 4, 5 };
        int[] cost1 = { 3, 4, 5, 1, 2 };
        // Output: 3
        System.out.println(sol.canCompleteCircuit(gas1, cost1));
        int[] gas2 = { 2, 3, 4 };
        int[] cost2 = { 3, 4, 3 };
        // Output: -1
        System.out.println(sol.canCompleteCircuit(gas2, cost2));
    }
}
