class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        for (int i = 0; i < gas.length; ) {
            int distance = getDistance(i, gas, cost);
            if (distance >= gas.length) {
                return i;
            }
            i = i + distance + 1;
        }
        return -1;
    }

    int getDistance(int start, int[] gas, int[] cost) {
        int distance = 0;
        int curGas = gas[start];
        while (curGas >= cost[start] && distance < gas.length) {
            distance++;
            curGas -= cost[start];
            start = (start + 1) % gas.length;
            curGas += gas[start];
        }
        return distance;
    }
}
