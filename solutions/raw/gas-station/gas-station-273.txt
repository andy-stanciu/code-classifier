class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int startIndex = 0;
        int totalGas = 0;
        //whenever it becomes 0 then reset the value when iteration
        int sum = 0;
        for (int i = 0; i < n; i++) {
            totalGas += gas[i] - cost[i];
            sum += gas[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                startIndex = i + 1;
            }
        }
        return totalGas < 0 ? -1 : startIndex;
    }
}
