class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int curr = gas[gas.length - 1] - cost[gas.length - 1];
        int min_gas = curr;
        int min_index = 0;
        for (int i = 0; i < gas.length - 1; i++) {
            curr += gas[i] - cost[i];
            if (curr < min_gas) {
                min_gas = curr;
                min_index = i + 1;
            }
        }
        return curr >= 0 ? min_index : -1;
    }
}
