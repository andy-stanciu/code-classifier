class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int g = gas.length;
        int totalcost = 0;
        int totalgas = 0;
        int currentgas = 0;
        int startingstation = 0;
        for (int i = 0; i < g; i++) {
            totalgas = totalgas + gas[i];
            totalcost = totalcost + cost[i];
            currentgas = currentgas + gas[i] - cost[i];
            if (currentgas < 0) {
                startingstation = i + 1;
                currentgas = 0;
            }
        }
        if (totalgas >= totalcost) {
            return startingstation;
        } else {
            return -1;
        }
    }
}
