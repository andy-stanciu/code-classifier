class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int tg = 0, tank = 0, gs_st = 0;
        for (int i = 0; i < gas.length; i++) {
            tg = tg + (gas[i] - cost[i]);
            tank = tank + (gas[i] - cost[i]);
            if (tank < 0) {
                tank = 0;
                gs_st = i + 1;
            }
        }
        return tg < 0 ? -1 : gs_st;
    }
}
