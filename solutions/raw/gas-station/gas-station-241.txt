class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        if (gas.length != cost.length || gas.length == 0 || cost.length == 0) {
            return -1;
        }
        int gasSum = 0;
        int costSum = 0;
        int diff = 0;
        int index = -1;
        for (int i = 0; i < gas.length; i++) {
            gasSum += gas[i];
            costSum += cost[i];
            diff += gas[i] - cost[i];
            if (diff <= 0) {
                diff = 0;
                index = i;
            }
        }
        if (gasSum >= costSum) {
            return (index + 1) % gas.length;
        } else {
            return -1;
        }
    }
}
