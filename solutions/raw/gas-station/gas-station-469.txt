class Solution {

    public int canCompleteCircuit(int[] gasAvailable, int[] travelCost) {
        int totalGas = 0, totalCost = 0, startIndex = 0, fuelInTank = 0;
        for (int i = 0; i < gasAvailable.length; i++) {
            totalGas += gasAvailable[i];
            totalCost += travelCost[i];
        }
        if (totalGas < totalCost)
            return -1;
        for (int i = 0; i < gasAvailable.length; i++) {
            fuelInTank += gasAvailable[i] - travelCost[i];
            if (fuelInTank < 0) {
                fuelInTank = 0;
                startIndex = i + 1;
            }
        }
        return startIndex;
    }
}
