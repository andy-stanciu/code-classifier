class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int sum = 0;
        int ans = 0;
        int t = 0;
        for (int i = 0; i < gas.length; i++) {
            sum = sum + gas[i] - cost[i];
            if (sum < 0) {
                t = t + sum;
                sum = 0;
                ans = i + 1;
            }
        }
        t = t + sum;
        if (t >= 0) {
            return ans;
        } else {
            return -1;
        }
    }
}
