class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int sumBeforePoint = 0;
        int sumFromPoint = 0;
        int point = -1;
        for (int i = 0; i < gas.length; i++) {
            int sumAtPoint = gas[i] - cost[i];
            if (sumAtPoint < 0 && (sumFromPoint + sumAtPoint < 0)) {
                point = -1;
                sumBeforePoint += (sumAtPoint + sumFromPoint);
                sumFromPoint = 0;
            } else {
                if (point == -1) {
                    point = i;
                }
                sumFromPoint += sumAtPoint;
            }
        }
        return sumFromPoint + sumBeforePoint >= 0 ? point : -1;
    }
}
