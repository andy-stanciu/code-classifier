public class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int currRemaining = 0;
        int totalRemaining = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            currRemaining += gas[i] - cost[i];
            totalRemaining += gas[i] - cost[i];
            if (currRemaining < 0) {
                currRemaining = 0;
                start = i + 1;
            }
        }
        return totalRemaining < 0 ? -1 : start;
    }
}
