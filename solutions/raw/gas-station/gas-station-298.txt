class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int currentFuel = 0;
        int debitFuel = 0;
        int startingStation = 0;
        for (int i = 0; i < gas.length; i++) {
            currentFuel += gas[i];
            currentFuel -= cost[i];
            if (currentFuel < 0) {
                debitFuel += currentFuel;
                startingStation = i + 1;
                currentFuel = 0;
            }
        }
        return debitFuel + currentFuel >= 0 ? startingStation : -1;
    }
}
