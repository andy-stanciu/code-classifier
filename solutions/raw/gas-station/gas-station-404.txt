class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total = 0, sumOfGas = 0, sumOfCost = 0;
        int ans = 0;
        for (int i = 0; i < gas.length; i++) {
            total += gas[i] - cost[i];
            if (total < 0) {
                total = 0;
                ans = i + 1;
            }
            sumOfGas += gas[i];
            sumOfCost += cost[i];
        }
        total = sumOfGas - sumOfCost;
        if (total < 0) {
            return -1;
        }
        return ans;
    }
}
