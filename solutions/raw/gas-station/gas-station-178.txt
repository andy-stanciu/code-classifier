class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // verify that solution exists..
        // if sum( gas ) >= sum( cost ) :: then solution exist
        int totalGas = 0, totalCost = 0;
        int[] diff = new int[gas.length];
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
            diff[i] = gas[i] - cost[i];
        }
        if (totalGas < totalCost) {
            return -1;
        }
        // here :: definately an answer exist
        int cap = 0;
        int result = -1;
        for (int i = 0; i < diff.length; i++) {
            cap += diff[i];
            if (cap < 0) {
                result = -1;
                cap = 0;
                continue;
            } else if (result == -1) {
                result = i;
            }
        }
        return result;
    }
}
