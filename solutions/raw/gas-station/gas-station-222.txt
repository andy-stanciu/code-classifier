class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int tot = 0;
        int curr = 0;
        int start = 0;
        for (int i = 0; i < n; i++) {
            curr += gas[i] - cost[i];
            tot += gas[i] - cost[i];
            if (curr < 0) {
                curr = 0;
                start = i + 1;
            }
        }
        return tot < 0 ? -1 : start;
    }
}
