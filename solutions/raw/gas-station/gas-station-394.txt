class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int start = 0, total = 0, currentSum = 0;
        for (int i = 0; i < gas.length; i++) {
            total += gas[i] - cost[i];
            currentSum += gas[i] - cost[i];
            if (currentSum < 0) {
                start = i + 1;
                currentSum = 0;
            }
        }
        return total >= 0 ? start : -1;
    }
}
