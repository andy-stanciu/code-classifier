class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int t = 0;
        for (int i = 0; i < n; ++i) t += gas[i] - cost[i];
        if (t < 0)
            return -1;
        t = 0;
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            t += gas[i] - cost[i];
            if (t < 0) {
                ans = i + 1;
                t = 0;
            }
        }
        return ans;
    }
}
