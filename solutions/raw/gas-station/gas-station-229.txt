class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int k = 0;
        int gassum = 0;
        int costsum = 0;
        int gascostdiff = 0;
        for (int i = 0; i < gas.length; i++) {
            gascostdiff += gas[i] - cost[i];
            gassum += gas[i];
            costsum += cost[i];
            if (gascostdiff < 0) {
                k = i + 1;
                gascostdiff = 0;
            }
        }
        return costsum > gassum ? -1 : k;
    }
}
