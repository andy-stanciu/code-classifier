class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int totalGas = 0;
        int currentGas = 0;
        int startStation = 0;
        for (int i = 0; i < n; i++) {
            int netGas = gas[i] - cost[i];
            totalGas += netGas;
            currentGas += netGas;
            // If the current net gas is negative, reset the starting station
            // to the next station and reset the current net gas to zero.
            if (currentGas < 0) {
                startStation = i + 1;
                currentGas = 0;
            }
        }
        // If the total net gas is non-negative, return the starting station.
        // Otherwise, return -1.
        return totalGas >= 0 ? startStation : -1;
    }
}
