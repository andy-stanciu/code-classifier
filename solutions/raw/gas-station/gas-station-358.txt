class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        if (sum(gas) < sum(cost))
            return -1;
        int start = 0;
        int totalGas = 0;
        for (int i = 0; i < n; i++) {
            totalGas += gas[i] - cost[i];
            if (totalGas < 0) {
                start = i + 1;
                totalGas = 0;
            }
        }
        return start;
    }

    public int sum(int[] arr) {
        int sum = 0;
        for (int n : arr) sum += n;
        return sum;
    }
}
