class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0;
        int totalCost = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
        }
        if (totalGas < totalCost)
            return -1;
        int gasInTank = 0, start = 0;
        for (int i = 0; i < gas.length; i++) {
            gasInTank += gas[i];
            gasInTank -= cost[i];
            if (gasInTank < 0) {
                start = i + 1;
                gasInTank = 0;
            }
        }
        return start;
    }
}
