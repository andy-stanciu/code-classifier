class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total_cost = 0;
        int total_gas = 0;
        int start = 0;
        int curr = 0;
        for (int i = 0; i < gas.length; i++) {
            total_gas += gas[i];
        }
        for (int i = 0; i < cost.length; i++) {
            total_cost += cost[i];
        }
        if (total_gas < total_cost) {
            return -1;
        }
        for (int i = 0; i < gas.length; i++) {
            curr += (gas[i] - cost[i]);
            if (curr < 0) {
                start = i + 1;
                curr = 0;
            }
        }
        return start;
    }
}
