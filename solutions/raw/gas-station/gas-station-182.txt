class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        if (gas.length != cost.length)
            return -1;
        int start = 0, tank = 0, totalCost = 0, totalGas = 0;
        for (int i = 0; i < gas.length; i++) {
            totalCost += cost[i];
            totalGas += gas[i];
            tank = tank + gas[i] - cost[i];
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        if (totalCost > totalGas)
            return -1;
        return start;
    }
}
