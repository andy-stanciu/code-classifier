class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int sum = 0;
        int totalg = 0;
        int totalc = 0;
        int pointer = 0;
        int i = 0;
        // ----- boundary condition ------
        if (gas.length != cost.length)
            return -1;
        // sum = gas[i] - cost[i];
        // ----------- main logic ---------
        for (i = 0; i < gas.length; i++) {
            totalg = totalg + gas[i];
        }
        for (i = 0; i < cost.length; i++) {
            totalc = totalc + cost[i];
        }
        if (totalg < totalc)
            return -1;
        for (i = 0; i < gas.length; i++) {
            sum = sum + gas[i] - cost[i];
            if (sum < 0) {
                pointer = i + 1;
                sum = 0;
            }
        }
        return pointer;
    }
}
