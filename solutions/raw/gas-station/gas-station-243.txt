class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // find potential position
        int potentialPosition = -1;
        int largestStartPossible = -1;
        int sumDiffBtwGasAndCost = 0;
        int len = gas.length;
        for (int i = len - 1; i >= 0; i--) {
            int diff = gas[i] - cost[i];
            sumDiffBtwGasAndCost += diff;
            if (sumDiffBtwGasAndCost > largestStartPossible) {
                potentialPosition = i;
                largestStartPossible = sumDiffBtwGasAndCost;
            }
        }
        if (potentialPosition == -1 || sumDiffBtwGasAndCost < 0)
            return -1;
        return potentialPosition;
    }
}
