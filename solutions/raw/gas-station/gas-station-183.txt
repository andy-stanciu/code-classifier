class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int diff = 0;
        int max = 0;
        int ans = cost.length - 1;
        for (int i = gas.length - 1; i >= 0; i--) {
            diff += gas[i] - cost[i];
            if (diff >= max) {
                max = diff;
                ans = i;
            }
        }
        if (diff < 0) {
            return -1;
        }
        return ans;
    }
}
