class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int gasTank = 0, i = 0, j = gas.length - 1;
        while (i <= j) {
            if (gasTank >= 0)
                gasTank += gas[i] - cost[i++];
            else
                gasTank += gas[j] - cost[j--];
        }
        return gasTank < 0 ? -1 : (j + 1) % gas.length;
    }
}
