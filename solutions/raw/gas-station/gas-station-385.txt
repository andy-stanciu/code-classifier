class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int sum = 0;
        int minPos = 0;
        int minSum = Integer.MAX_VALUE;
        for (int i = 0; i < cost.length; i++) {
            sum -= cost[i];
            if (sum <= minSum) {
                minSum = sum;
                minPos = i;
            }
            sum += gas[(i + 1) % gas.length];
        }
        return sum >= 0 ? (minPos + 1) % gas.length : -1;
    }
}
