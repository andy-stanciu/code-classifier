class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        if ((sumOFArray(gas) - sumOFArray(cost)) < 0) {
            return -1;
        }
        int start = 0;
        int totalSum = 0;
        for (int i = 0; i < gas.length; i++) {
            totalSum += gas[i] - cost[i];
            if (totalSum < 0) {
                totalSum = 0;
                start = i + 1;
            }
        }
        if (start > gas.length - 1) {
            return -1;
        } else {
            return start;
        }
    }

    int sumOFArray(int[] arr) {
        int sum = 0;
        for (int i = 0; i < arr.length; i++) {
            sum += arr[i];
        }
        return sum;
    }
}
