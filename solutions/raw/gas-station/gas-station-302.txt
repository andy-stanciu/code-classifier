class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalExtraFuel = 0;
        int extraFuel = 0;
        int index = 0;
        int l = gas.length;
        for (int i = 0; i < l; i++) {
            int diff = gas[i] - cost[i];
            totalExtraFuel += diff;
            extraFuel += diff;
            if (extraFuel < 0) {
                //  if extrafuel is <0 that means currently fuel in tank at the station is less than the cost, in that case we cant move further so move to next index to restart (yes restart that\'s why we are doing extrafuel =0 \')
                index = i + 1;
                extraFuel = 0;
            }
        }
        // total fuel needed has to be less than or equal to fuel we have in total then only we can travel
        return (totalExtraFuel < 0) ? -1 : index;
    }
}
