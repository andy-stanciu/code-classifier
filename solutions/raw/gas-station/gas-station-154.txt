class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // Total gas available
        int totalGas = 0;
        // Total gas required
        int totalCost = 0;
        // Gas remaining during the trip
        int currentGas = 0;
        // Starting gas station
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
            currentGas += gas[i] - cost[i];
            // If at any point, currentGas is negative, we can\'t start from \'start\'
            if (currentGas < 0) {
                // Start from the next station
                start = i + 1;
                // Reset currentGas for the new start
                currentGas = 0;
            }
        }
        // If total gas is greater than or equal to total cost, we can complete the
        // circuit
        return (totalGas >= totalCost) ? start : -1;
    }
}
