class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0;
        int totalCost = 0;
        int startStation = 0;
        int tank = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
            tank += gas[i] - cost[i];
            // If the current tank becomes negative, reset the startStation and tank
            if (tank < 0) {
                startStation = i + 1;
                tank = 0;
            }
        }
        // If total gas is less than total cost, it\'s impossible to complete the circuit
        if (totalGas < totalCost) {
            return -1;
        } else {
            return startStation;
        }
    }
}
