class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int sum_gas = 0;
        int sum_cost = 0;
        for (int i = 0; i < gas.length; i++) {
            sum_gas += gas[i];
        }
        for (int i = 0; i < cost.length; i++) {
            sum_cost += cost[i];
        }
        if (sum_gas < sum_cost)
            return -1;
        int start = 0;
        int curr = 0;
        for (int i = 0; i < gas.length; i++) {
            curr += (gas[i] - cost[i]);
            if (curr < 0) {
                start = i + 1;
                curr = 0;
            }
        }
        return start;
    }
}
