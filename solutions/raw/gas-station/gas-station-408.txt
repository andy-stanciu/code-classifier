class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0, totalCost = 0, i = 0;
        for (i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
        }
        if (totalGas < totalCost)
            return -1;
        int result = 0, tank = 0;
        for (i = 0; i < gas.length; i++) {
            tank = tank + (gas[i] - cost[i]);
            if (tank < 0) {
                tank = 0;
                result = i + 1;
            }
        }
        return result;
    }
}
