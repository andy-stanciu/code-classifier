class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGasLeft = 0, start = 0, remainGas = 0;
        for (int ind = 0; ind < gas.length; ind++) {
            var gasUsed = gas[ind] - cost[ind];
            remainGas += gasUsed;
            totalGasLeft += gasUsed;
            //if remainGas negative, start from next Index
            if (remainGas < 0) {
                start = ind + 1;
                remainGas = 0;
            }
        }
        //if totalGasLeft is negative, can\'t complete circle
        if (totalGasLeft < 0) {
            return -1;
        }
        return (start == gas.length) ? 0 : start;
    }
}
