class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int gasNeededOnLeft = 0;
        int gasLeft = 0;
        int startPoint = -1;
        for (int i = 0; i < gas.length; i++) {
            gasLeft = gasLeft + gas[i] - cost[i];
            if (gasLeft < 0) {
                gasNeededOnLeft += gasLeft;
                gasLeft = 0;
                startPoint = -1;
            } else if (startPoint == -1) {
                startPoint = i;
            }
        }
        return ((gasLeft + gasNeededOnLeft) >= 0) ? startPoint : -1;
    }
}
