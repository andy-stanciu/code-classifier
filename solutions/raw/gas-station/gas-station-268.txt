class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int total = 0;
        for (int i = 0; i < n; i++) {
            total += gas[i] - cost[i];
        }
        if (total < 0)
            return -1;
        int remaining = 0;
        int startIndex = 0;
        for (int i = 0; i < n; i++) {
            remaining += gas[i] - cost[i];
            if (remaining < 0) {
                remaining = 0;
                startIndex = i + 1;
            }
        }
        return startIndex;
    }
}
