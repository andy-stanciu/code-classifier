class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int tank = 0;
        int lowestIndex = 0;
        int lowestGas = gas[0];
        for (int i = 0; i < gas.length; i++) {
            tank += gas[i] - cost[i];
            if (tank <= lowestGas) {
                lowestGas = tank;
                lowestIndex = i;
            }
        }
        if (tank < 0)
            return -1;
        return (lowestIndex + 1) % gas.length;
    }
}
