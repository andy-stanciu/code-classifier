class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        //We will track the net cost at every station
        int totalCost = 0;
        //Check if it\'s possible for a circuit to exist
        int circuitCheck = 0;
        //Set the index to 0 as default
        int index = 0;
        //To test every possible circuit, visit every station
        for (int i = 0; i < gas.length; i++) {
            /*
            If the net cost < 0 up to station i, the 
            circuit can\'t start at the last i - 1 stations
            
            So bump the index up to station i and reset the net cost
            */
            if (totalCost < 0) {
                index = i;
                totalCost = 0;
            }
            //Add up the cost at every station
            totalCost += (gas[i] - cost[i]);
            circuitCheck += (gas[i] - cost[i]);
        }
        //Return index if the circuit value is positive (meaning a circuit exists)
        return (circuitCheck >= 0) ? index : -1;
    }
}
