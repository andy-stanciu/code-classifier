class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        // Starting index candidate
        int startIdx = 0;
        // Cumulative sum of gas[i] - cost[i]
        int total = 0;
        // Current tank capacity
        int tank = 0;
        for (int i = 0; i < n; i++) {
            int gain = gas[i] - cost[i];
            tank += gain;
            total += gain;
            if (tank < 0) {
                // If tank becomes negative, reset the starting index candidate
                startIdx = i + 1;
                tank = 0;
            }
        }
        if (total < 0) {
            // If the total sum is negative, it is impossible to complete a circuit
            return -1;
        }
        return startIdx;
    }
}
