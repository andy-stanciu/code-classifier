class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        // Total gas over all stations
        int totalGas = 0;
        // Total cost over all stations
        int totalCost = 0;
        // Track gas left in the tank as we check each station
        int currentTank = 0;
        // Index of the starting gas station
        int startStation = 0;
        for (int i = 0; i < n; i++) {
            // Total gas in all stations
            totalGas += gas[i];
            // Total cost for all stations
            totalCost += cost[i];
            // Track the gas remaining in the tank
            currentTank += gas[i] - cost[i];
            // If we cannot reach the next station from the current start, change start
            if (currentTank < 0) {
                // Move the start station to the next station
                startStation = i + 1;
                // Reset the current tank since we\'re starting fresh
                currentTank = 0;
            }
        }
        // If the total gas is less than the total cost, return -1 (no solution)
        if (totalGas < totalCost) {
            return -1;
        }
        // Return the start station index where we can complete the circuit
        return startStation;
    }
}
