class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int startingGasIndex = -1, currentLeftGas = 0, totalGasFromStartingIndex = 0, totalLeftGas = 0;
        for (int i = 0; i < gas.length; i++) {
            currentLeftGas = gas[i] - cost[i];
            totalLeftGas += currentLeftGas;
            if (startingGasIndex != -1) {
                totalGasFromStartingIndex += currentLeftGas;
                if (totalGasFromStartingIndex < 0) {
                    startingGasIndex = -1;
                    totalGasFromStartingIndex = 0;
                }
            } else if (currentLeftGas >= 0) {
                startingGasIndex = i;
                totalGasFromStartingIndex += currentLeftGas;
            }
        }
        if (totalLeftGas >= 0) {
            return startingGasIndex;
        }
        return -1;
    }
}
