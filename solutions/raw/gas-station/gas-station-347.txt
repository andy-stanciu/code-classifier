class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        final int len = gas.length;
        int startStation = 0;
        int maxDeficit = 0;
        int currentFuel = 0;
        for (int i = 0; i < len; ++i) {
            currentFuel += gas[i] - cost[i];
            if (currentFuel < maxDeficit) {
                maxDeficit = currentFuel;
                startStation = i + 1;
            }
        }
        if (currentFuel < 0)
            return -1;
        return startStation;
    }
}
