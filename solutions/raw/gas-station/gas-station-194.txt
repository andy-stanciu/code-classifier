class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int len = gas.length;
        int fuel = 0, exp = 0, tot = 0;
        int idx = 0;
        for (int i = 0; i < len; i++) {
            exp = gas[i] - cost[i];
            tot = tot + exp;
            fuel = fuel + exp;
            if (fuel < 0) {
                fuel = 0;
                idx = i + 1;
            }
        }
        if (tot >= 0)
            return idx;
        return -1;
    }
}
