class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int size = gas.length;
        // int [] res = new int[size];
        int total_gas = 0;
        int total_cost = 0;
        //filling the res array
        for (int i = 0; i < size; i++) {
            // res[i] = gas[i] - cost[i];
            total_gas += gas[i];
            total_cost += cost[i];
        }
        if (total_gas < total_cost) {
            return -1;
        }
        int total = 0;
        int index = 0;
        for (int i = 0; i < size; i++) {
            total += gas[i] - cost[i];
            if (total < 0) {
                total = 0;
                index = i + 1;
            }
        }
        return index;
    }
}
