class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int ans = 0;
        int totalDif = 0;
        int currDif = 0;
        int dif = 0;
        for (int i = 0; i < gas.length; i++) {
            // calculating if the dif between total gas and total cost
            dif = gas[i] - cost[i];
            // if the totalDif is negative means we have less gas compared to cost
            totalDif += dif;
            // calculating if the curDif in gas and cost is < 0 then it is not possible solution.
            currDif += dif;
            if (currDif < 0) {
                ans = i + 1;
                currDif = 0;
            }
        }
        if (totalDif < 0) {
            return -1;
        } else {
            return ans;
        }
    }
}
