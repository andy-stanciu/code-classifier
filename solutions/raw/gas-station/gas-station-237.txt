class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int netFuel = 0;
        int start = 0;
        int tank = 0;
        for (int i = 0; i < n; i++) {
            netFuel += gas[i] - cost[i];
            tank += gas[i] - cost[i];
            if (tank < 0) {
                tank = 0;
                start = i + 1;
            }
        }
        return (netFuel < 0) ? (-1) : (start);
    }
}
