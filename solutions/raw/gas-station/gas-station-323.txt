class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int g = Arrays.stream(gas).sum();
        int c = Arrays.stream(cost).sum();
        int start = 0;
        int currFuel = 0;
        if (c > g) {
            return -1;
        } else {
            for (int i = 0; i < gas.length; i++) {
                currFuel += (gas[i] - cost[i]);
                if (currFuel < 0) {
                    currFuel = 0;
                    start = i + 1;
                }
            }
        }
        return start;
    }
}
