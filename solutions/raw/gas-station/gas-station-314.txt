class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int ans = 0;
        int currDiff = 0;
        int totalDiff = 0;
        for (int i = 0; i < gas.length; i++) {
            currDiff += gas[i] - cost[i];
            totalDiff += gas[i] - cost[i];
            if (currDiff < 0) {
                ans = i + 1;
                currDiff = 0;
            }
        }
        if (totalDiff < 0)
            return -1;
        return ans;
    }
}
