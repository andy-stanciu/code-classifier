class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totgas = 0;
        int totcost = 0;
        int total = 0;
        int res = 0;
        for (int i = 0; i < gas.length; i++) {
            totgas += gas[i];
            totcost += cost[i];
        }
        if (totgas < totcost) {
            return -1;
        }
        for (int i = 0; i < gas.length; i++) {
            total += gas[i] - cost[i];
            if (total < 0) {
                total = 0;
                res = i + 1;
            }
        }
        return res;
    }
}
