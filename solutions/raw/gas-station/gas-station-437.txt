class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalSum = 0;
        int maxSum = 0;
        int startIndex = 0;
        for (int i = 0; i < gas.length; i++) {
            int difference = gas[i] - cost[i];
            totalSum += difference;
            if (difference > maxSum + difference) {
                maxSum = difference;
                startIndex = i;
            } else {
                maxSum += difference;
            }
        }
        return totalSum < 0 ? -1 : startIndex;
    }
}
