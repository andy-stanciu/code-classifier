class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int gassum = 0;
        int costsum = 0;
        for (int i = 0; i < n; i++) {
            gassum += gas[i];
            costsum += cost[i];
        }
        if (gassum < costsum)
            return -1;
        else {
            int temp = 0;
            int result = 0;
            for (int i = 0; i < n; i++) {
                temp += gas[i] - cost[i];
                if (temp < 0) {
                    temp = 0;
                    result = i + 1;
                }
            }
            return result;
        }
    }
}
