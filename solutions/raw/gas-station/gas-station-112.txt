class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int start = 0, total = 0, current = 0;
        for (int i = 0; i < gas.length; i++) {
            total += gas[i] - cost[i];
        }
        if (total < 0) {
            return -1;
        }
        for (int i = 0; i < gas.length; i++) {
            int costs = gas[i] - cost[i];
            current += costs;
            if (current < 0) {
                start = i + 1;
                current = 0;
            }
        }
        return start;
    }
}
