class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int fuel = 0, sum = 0, start = 0;
        for (int i = 0; i < gas.length; i++) {
            sum += (gas[i] - cost[i]);
            fuel += (gas[i] - cost[i]);
            if (fuel < 0) {
                fuel = 0;
                start = i + 1;
            }
        }
        if (sum >= 0) {
            return start % (gas.length);
        }
        return -1;
    }
}
