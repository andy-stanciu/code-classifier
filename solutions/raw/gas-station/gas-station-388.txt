class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int deficit = 0, extragas = 0, sp = 0, idx = 0;
        while (idx < gas.length) {
            extragas += gas[idx] - cost[idx];
            if (extragas < 0) {
                deficit += extragas;
                extragas = 0;
                sp = idx + 1;
            }
            idx++;
        }
        return (sp == gas.length || deficit + extragas < 0) ? -1 : sp;
    }
}
