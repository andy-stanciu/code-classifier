class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totaldiff = 0, n = gas.length, fuel = 0, index = 0;
        for (int i = 0; i < n; i++) {
            int diff = gas[i] - cost[i];
            totaldiff = totaldiff + diff;
            fuel = fuel + diff;
            if (fuel < 0) {
                index = i + 1;
                fuel = 0;
            }
        }
        return (totaldiff < 0) ? -1 : index;
    }
}
