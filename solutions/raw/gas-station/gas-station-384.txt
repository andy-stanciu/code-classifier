class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int totalgas = 0, current = 0, totalcost = 0, start = 0;
        for (int i = 0; i < n; i++) {
            current += gas[i];
            totalgas += gas[i];
            totalcost += cost[i];
            if (current < cost[i]) {
                current = 0;
                start = i + 1;
                continue;
            }
            current -= cost[i];
        }
        if (totalgas >= totalcost)
            return start;
        return -1;
    }
}
