public class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalTank = 0;
        int currTank = 0;
        int startingStation = 0;
        for (int i = 0; i < gas.length; i++) {
            int difference = gas[i] - cost[i];
            totalTank += difference;
            currTank += difference;
            if (currTank < 0) {
                currTank = 0;
                startingStation = i + 1;
            }
        }
        return totalTank >= 0 ? startingStation : -1;
    }
}
