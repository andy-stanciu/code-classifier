class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int start = 0;
        int currentFuel = 0;
        int totalfuel = 0;
        int totalcost = 0;
        for (int i = 0; i < gas.length; i++) {
            totalfuel += gas[i];
        }
        for (int i = 0; i < cost.length; i++) {
            totalcost += cost[i];
        }
        if (totalfuel < totalcost)
            return -1;
        for (int i = 0; i < gas.length; i++) {
            currentFuel = currentFuel + gas[i] - cost[i];
            if (currentFuel < 0) {
                //we have chosen the wrong startting index
                //with that starting indez we won\'t be able to complete the cycle\'
                //hence we need to reset
                start = i + 1;
                currentFuel = 0;
            }
        }
        return start;
    }
}
