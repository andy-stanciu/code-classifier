class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int diff = 0, current_gas = 0, pos = 0;
        int n = gas.length;
        for (int i = 0; i < n; i++) {
            diff += gas[i] - cost[i];
            current_gas += gas[i] - cost[i];
            if (current_gas < 0) {
                pos = i + 1;
                current_gas = 0;
            }
        }
        if (diff < 0) {
            pos = -1;
        }
        return pos;
    }
}
