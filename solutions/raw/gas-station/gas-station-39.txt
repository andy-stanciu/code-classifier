class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int amt = 0;
        int t = 0;
        int idx = 0;
        for (int i = 0; i < gas.length; i++) {
            amt += gas[i] - cost[i];
            t += gas[i] - cost[i];
            if (amt < 0) {
                amt = 0;
                idx = i + 1;
            }
        }
        if (t < 0) {
            //incomplete path
            return -1;
        }
        return idx;
    }
}
