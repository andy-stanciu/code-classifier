class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int totalGas = 0, totaCost = 0;
        for (int i = 0; i < n; i++) {
            totalGas += gas[i];
            totaCost += cost[i];
        }
        if (totalGas < totaCost)
            return -1;
        int gasInCar = 0, start = 0;
        for (int i = 0; i < n; i++) {
            gasInCar += gas[i] - cost[i];
            if (gasInCar < 0) {
                gasInCar = 0;
                start = i + 1;
            }
        }
        return start;
    }
}
