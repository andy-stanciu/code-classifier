class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int GasAvailable = 0;
        int GasUsed = 0;
        int AmountPaid = 0;
        int StartPoint = 0;
        for (int i = 0; i < n; i++) {
            GasUsed += gas[i];
            AmountPaid += cost[i];
            GasAvailable += gas[i] - cost[i];
            if (GasAvailable < 0) {
                GasAvailable = 0;
                StartPoint = i + 1;
            }
        }
        return AmountPaid > GasUsed ? -1 : StartPoint;
    }
}
