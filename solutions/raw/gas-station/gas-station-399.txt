class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalFuelVsCost = 0;
        int diffFuelVsCost = 0;
        int ans = 0;
        for (int i = 0; i < cost.length; i++) {
            if (diffFuelVsCost < 0) {
                diffFuelVsCost = 0;
                ans = i;
            }
            diffFuelVsCost = diffFuelVsCost + gas[i] - cost[i];
            totalFuelVsCost = totalFuelVsCost + gas[i] - cost[i];
        }
        if (totalFuelVsCost >= 0) {
            return ans;
        }
        return -1;
    }
}
