class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int def = 0;
        int bal = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            bal = bal + gas[i] - cost[i];
            if (bal < 0) {
                def += Math.abs(bal);
                start = i + 1;
                bal = 0;
            }
        }
        if (bal - def >= 0)
            return start;
        else
            return -1;
    }
}
