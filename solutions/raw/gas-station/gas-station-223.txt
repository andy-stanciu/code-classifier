class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        // Starting point index
        int start = 0;
        // Net gas count from current start
        int gasCount = 0;
        for (int i = 0; i < n; i++) {
            // Calculate net gas at each station
            gasCount += gas[i] - cost[i];
            if (gasCount < 0) {
                // Reset start to next station
                start = i + 1;
                // Reset gasCount
                gasCount = 0;
            }
        }
        // Check if the total gas count starting from \'start\' is sufficient
        int totalGas = 0;
        int totalCost = 0;
        for (int i = 0; i < n; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
        }
        if (totalGas < totalCost) {
            // If total gas is less than total cost, return -1
            return -1;
        }
        return start;
    }
}
