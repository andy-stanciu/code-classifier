class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int accu = 0;
        int start = -1;
        int min = 0;
        for (int i = 0; i < gas.length; i++) {
            //add the net
            accu += gas[i] - cost[i];
            if (accu < min) {
                min = accu;
                start = i;
            }
        }
        if (accu < 0) {
            return -1;
        } else {
            return start + 1;
        }
    }
}
