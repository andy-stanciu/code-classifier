class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int avialable = 0;
        int ind = 0;
        int total = 0;
        for (int i = 0; i < gas.length; i++) {
            total += gas[i] - cost[i];
            avialable += gas[i] - cost[i];
            if (avialable < 0) {
                avialable = 0;
                ind = i + 1;
            }
        }
        if (total < 0)
            return -1;
        return ind;
    }
}
