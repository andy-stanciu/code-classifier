class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int leftOver = 0;
        int start = 0;
        int cur = 0;
        for (int i = 0; i < n; i++) {
            leftOver += gas[i] - cost[i];
            cur += gas[i] - cost[i];
            if (cur < 0) {
                start = i + 1;
                cur = 0;
            }
        }
        return leftOver >= 0 ? start : -1;
    }
}
