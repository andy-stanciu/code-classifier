class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int startIndex = 0;
        int gasLeft = 0;
        int prevGas = 0;
        for (int i = 0; i < gas.length; i++) {
            gasLeft += gas[i] - cost[i];
            if (gasLeft < prevGas) {
                prevGas = gasLeft;
                startIndex = (i + 1) % gas.length;
            }
        }
        if (gasLeft >= 0) {
            return startIndex;
        }
        return -1;
    }
}
