class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // Initialize variables to track total difference in gas and cost,
        // current index where starting point is being tested,
        // current difference in gas and cost, and current fuel in the tank.
        int totalDiff = 0;
        int index = 0;
        int diff = 0;
        int fuel = 0;
        // Iterate through the gas stations.
        for (int i = 0; i < gas.length; i++) {
            // Calculate the difference in gas and cost for the current station.
            diff = gas[i] - cost[i];
            // Accumulate the total difference in gas and cost.
            totalDiff += diff;
            // Add the difference to the fuel tank.
            fuel += diff;
            // If the fuel becomes negative, we cannot start from the current station,
            // so update the starting index to the next station and reset the fuel tank.
            if (fuel < 0) {
                index = i + 1;
                fuel = 0;
            }
        }
        // If the total difference between gas and cost is negative,
        // it means there is no possible solution, return -1.
        if (totalDiff < 0)
            return -1;
        // Otherwise, return the index of the starting gas station.
        return index;
    }
}
