class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int tsum = 0;
        int csum = 0;
        int si = 0;
        for (int i = 0; i < cost.length; i++) {
            int delta = gas[i] - cost[i];
            tsum += delta;
            csum += delta;
            if (csum < 0) {
                csum = 0;
                si = i + 1;
            }
        }
        if (tsum >= 0)
            return si;
        return -1;
    }
}
