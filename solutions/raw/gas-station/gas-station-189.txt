class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int sumG = 0;
        int sumC = 0;
        // Calculate total gas and total cost
        for (int i = 0; i < gas.length; i++) {
            sumG += gas[i];
            sumC += cost[i];
        }
        // If total gas is less than total cost, return -1
        if (sumG < sumC)
            return -1;
        // Initialize variables to track fuel left and starting index
        int fuelLeft = 0;
        int startIdx = 0;
        // Iterate through each station
        for (int i = 0; i < gas.length; i++) {
            fuelLeft += gas[i] - cost[i];
            if (fuelLeft < 0) {
                // If fuel left is negative, reset starting index and fuel left
                fuelLeft = 0;
                startIdx = i + 1;
            }
        }
        return startIdx;
    }
}
