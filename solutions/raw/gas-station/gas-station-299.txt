class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total_gas = 0;
        int total_cost = 0;
        for (int i = 0; i < gas.length; i++) {
            //lenght of gas and cost is same so gas.length==cost.length
            total_gas += gas[i];
            total_cost += cost[i];
        }
        if (total_gas < total_cost) {
            return -1;
        }
        int gas_at_station = 0;
        int station = 0;
        for (int i = 0; i < gas.length; i++) {
            gas_at_station += gas[i] - cost[i];
            if (gas_at_station < 0) {
                gas_at_station = 0;
                station = i + 1;
            }
        }
        return station;
    }
}
