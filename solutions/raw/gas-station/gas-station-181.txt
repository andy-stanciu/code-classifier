class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int total = 0;
        int current = 0;
        int start_pos = 0;
        for (int i = 0; i < n; i++) {
            total += gas[i] - cost[i];
            current += gas[i] - cost[i];
            if (current < 0) {
                start_pos = i + 1;
                current = 0;
            }
        }
        return total >= 0 ? start_pos : -1;
    }
}
