class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        var bor = 0;
        var curGas = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            curGas += gas[i] - cost[i];
            if (curGas < 0) {
                bor += -curGas;
                curGas = 0;
                start = i + 1;
            }
        }
        return (curGas < bor) ? -1 : start;
    }
}
