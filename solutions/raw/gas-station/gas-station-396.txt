class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        int tf = 0;
        int f = 0;
        int index = 0;
        for (int i = 0; i < n; i++) {
            tf += gas[i] - cost[i];
            f += gas[i] - cost[i];
            if (f < 0) {
                f = 0;
                index = i + 1;
            }
        }
        return (tf < 0) ? -1 : index;
    }
}
