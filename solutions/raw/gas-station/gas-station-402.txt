class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int start = 0;
        int current = 0;
        int fuelLeft = 0;
        int totalSteps = 0;
        while (totalSteps <= gas.length * 2) {
            totalSteps++;
            int currentCost = cost[current];
            int availableFuel = gas[current] + fuelLeft;
            int nextStation = (current + 1) % gas.length;
            if (availableFuel < currentCost) {
                start = nextStation;
                current = nextStation;
                fuelLeft = 0;
                continue;
            }
            if (nextStation == start) {
                return start;
            }
            current = nextStation;
            fuelLeft = availableFuel - currentCost;
        }
        return -1;
    }
}
