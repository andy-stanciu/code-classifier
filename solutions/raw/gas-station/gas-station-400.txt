class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int cumulatedGas = 0;
        int gasFromIndexZero = 0;
        int startingIndex = 0;
        for (int i = 0; i < gas.length; i++) {
            cumulatedGas += gas[i] - cost[i];
            if (cumulatedGas < 0) {
                gasFromIndexZero += cumulatedGas;
                cumulatedGas = 0;
                startingIndex = i + 1;
            }
        }
        return cumulatedGas + gasFromIndexZero >= 0 ? startingIndex : -1;
    }
}
