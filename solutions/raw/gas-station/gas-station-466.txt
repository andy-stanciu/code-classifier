class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int diff = 0, total = 0;
        int idx = 0;
        for (int i = 0; i < gas.length; i++) {
            diff += gas[i] - cost[i];
            total += gas[i] - cost[i];
            if (diff < 0) {
                diff = 0;
                idx = i + 1;
            }
        }
        return total < 0 ? -1 : idx;
    }
}
