class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int need = 0, balance = 0, s = 0;
        int n = gas.length;
        for (int i = 0; i < n; i++) {
            int tank = gas[i] - cost[i] + balance;
            // Check go to next station with available gas. If possible then add remain gas to balance
            if (tank >= 0) {
                balance = tank;
            } else // If not possible then calculate how many gas needed to reach next station and set next station is a possible stating point
            {
                need += Math.abs(tank);
                balance = 0;
                s = i + 1;
            }
        }
        if (balance >= need)
            return s;
        return -1;
    }
}
