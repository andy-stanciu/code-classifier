class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // inclusive
        int start = gas.length - 1;
        // exclusive
        int end = 0;
        int gasInTank = gas[start] - cost[start];
        while (start != end) {
            if (gasInTank > 0) {
                gasInTank += gas[end] - cost[end];
                end++;
            } else {
                start--;
                gasInTank += gas[start] - cost[start];
            }
        }
        return gasInTank >= 0 ? start : -1;
    }
}
