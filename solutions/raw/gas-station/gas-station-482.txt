class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int remaining_fuel = 0;
        int ans = 0;
        for (int i : gas) remaining_fuel += i;
        for (int j : cost) remaining_fuel -= j;
        if (remaining_fuel < 0)
            return -1;
        remaining_fuel = 0;
        for (int i = 0; i < gas.length; i++) {
            remaining_fuel = gas[i] - cost[i] + remaining_fuel;
            if (remaining_fuel < 0) {
                ans = i + 1;
                remaining_fuel = 0;
            }
        }
        return ans;
    }
}
