class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int current = 0, totalfuel = 0, totalcost = 0, start = 0;
        for (int i = 0; i < gas.length; i++) {
            totalfuel += gas[i];
        }
        for (int i = 0; i < cost.length; i++) {
            totalcost += cost[i];
        }
        if (totalcost > totalfuel) {
            return -1;
        }
        for (int i = 0; i < gas.length; i++) {
            current += (gas[i] - cost[i]);
            if (current < 0) {
                start = i + 1;
                current = 0;
            }
        }
        return start;
    }
}
