class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int max = 0;
        int maxPos = 0;
        int min = 0;
        int tmp = 0;
        for (int i = gas.length - 1; i >= 0; i--) {
            gas[i] = gas[i] - cost[i];
            tmp += gas[i];
            if (tmp > max) {
                maxPos = i;
                max = tmp;
            }
        }
        if (tmp >= 0) {
            return maxPos;
        } else {
            return -1;
        }
    }
}
