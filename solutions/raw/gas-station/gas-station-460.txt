class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total = 0, sum = 0, i = 0, start = 0;
        while (start < gas.length) {
            int value = i % gas.length;
            sum = sum + gas[value] - cost[value];
            if (total >= gas.length) {
                return start;
            }
            if (sum < 0) {
                start = ++i;
                total = 0;
                sum = 0;
                continue;
            } else {
                i++;
                total++;
            }
        }
        return -1;
    }
}
