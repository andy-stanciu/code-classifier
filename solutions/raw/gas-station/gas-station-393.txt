class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int gSum = 0;
        for (int i : gas) {
            gSum += i;
        }
        int cSum = 0;
        for (int i : cost) {
            cSum += i;
        }
        if (gSum < cSum) {
            return -1;
        }
        int tSum = 0;
        int s = 0;
        for (int i = 0; i < gas.length; i++) {
            int d = tSum + gas[i] - cost[i];
            if (d >= 0) {
                tSum = d;
            } else {
                tSum = 0;
                s = i + 1;
            }
        }
        return s;
    }
}
