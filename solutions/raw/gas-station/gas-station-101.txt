class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total = 0, remaining = 0, index = 0;
        for (int i = 0; i < gas.length; i++) {
            total += (gas[i] - cost[i]);
            remaining += (gas[i] - cost[i]);
            if (remaining < 0) {
                remaining = 0;
                index = i + 1;
            }
        }
        if (total < 0)
            return -1;
        return index;
    }
}
