class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int fuelSum = 0, n = gas.length;
        int gasInTank = 0, start = 0;
        for (int i = 0; i < n; i++) {
            gasInTank += gas[i] - cost[i];
            fuelSum += gas[i] - cost[i];
            //important part
            // if we are not able to reach next station from i,
            if (gasInTank < 0) {
                start = i + 1;
                gasInTank = 0;
            }
        }
        return fuelSum >= 0 ? start : -1;
    }
}
//we need to return either one value from 2(start(index)/-1)
