        // if(total >= 0){

        //-----------------------------------------------------------------------------
        int ans = 0, sum = 0, total = 0;
        for (int i = 0; i < gas.length; i++) {
            total += gas[i] - cost[i];
            sum += gas[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                ans = i + 1;
            }
        }
        if (total >= 0) {
            return ans;
        } else {
            return -1;
        }
    }
