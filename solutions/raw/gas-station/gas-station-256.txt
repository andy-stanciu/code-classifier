class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0, totalCost = 0;
        for (int i = 0; i < gas.length; i++) {
            totalGas += gas[i];
            totalCost += cost[i];
        }
        // If the total gas is less than total cost, return -1
        if (totalGas < totalCost) {
            return -1;
        }
        int start = 0;
        int tank = 0;
        // Find the starting station
        for (int i = 0; i < gas.length; i++) {
            tank += gas[i] - cost[i];
            // If tank is negative, reset starting point to i + 1
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        return start;
    }
}
