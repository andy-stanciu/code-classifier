class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total = 0;
        int cursum = 0;
        int start = 0;
        for (int i = 0; i < gas.length; i++) {
            int diff = gas[i] - cost[i];
            total += diff;
            cursum += diff;
            if (cursum < 0) {
                cursum = 0;
                start = i + 1;
            }
        }
        return total >= 0 ? start : -1;
    }
}
