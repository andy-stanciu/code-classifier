class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        // Current fuel at the current gas station
        int curr = 0;
        int totalFuel = 0, totalCost = 0;
        int start = 0;
        // Calculate total fuel and total cost
        for (int i : gas) {
            totalFuel += i;
        }
        for (int i : cost) {
            totalCost += i;
        }
        // If total fuel is less than total cost, it\'s impossible to complete the circuit
        if (totalFuel < totalCost)
            return -1;
        // Traverse through the gas stations to find the starting index
        for (int i = 0; i < gas.length; i++) {
            // Update the current fuel
            curr += (gas[i] - cost[i]);
            if (curr < 0) {
                start = i + 1;
                // Reset the current fuel to 0
                curr = 0;
            }
        }
        return start;
    }
}
