class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int totalSum = 0;
        int startIndex = 0;
        int currentCost = 0;
        for (int i = 0; i < gas.length; i++) {
            totalSum += gas[i] - cost[i];
            currentCost += gas[i] - cost[i];
            if (currentCost < 0) {
                currentCost = 0;
                startIndex = i + 1;
            }
        }
        return totalSum >= 0 ? startIndex : -1;
    }
}
