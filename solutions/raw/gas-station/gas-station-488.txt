class Solution {

    public int canCompleteCircuit(int[] gas, int[] cost) {
        int total_gas = 0;
        int start = 0;
        int curr_gas = 0;
        //checking if solution even exists or not
        //gas[i]-cost[i] gives us the gas left after we are done moving from one to another station
        for (int i = 0; i < gas.length; i++) total_gas += gas[i] - cost[i];
        // check if no gas left
        if (total_gas < 0)
            return -1;
        for (int i = 0; i < gas.length - 1; i++) {
            curr_gas += gas[i] - cost[i];
            if (curr_gas < 0) {
                start = i + 1;
                curr_gas = 0;
            }
        }
        return start;
    }
}
