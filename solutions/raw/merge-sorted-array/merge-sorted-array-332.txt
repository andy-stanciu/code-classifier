class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int k = nums1.length - 1, a = m - 1, b = n - 1;
        while (b >= 0) {
            if (a >= 0 && nums1[a] > nums2[b]) {
                nums1[k] = nums1[a];
                k--;
                a--;
            } else {
                nums1[k] = nums2[b];
                k--;
                b--;
            }
        }
    }
}
