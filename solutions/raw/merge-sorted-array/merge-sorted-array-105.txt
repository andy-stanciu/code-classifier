class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int insertIndex = 0;
        int nums1Index = n;
        int nums2Index = 0;
        int totalSize = m + n;
        System.arraycopy(nums1, 0, nums1, n, m);
        while (insertIndex < totalSize && nums1Index < totalSize && nums2Index < n) {
            if (nums1[nums1Index] < nums2[nums2Index]) {
                nums1[insertIndex] = nums1[nums1Index];
                nums1Index++;
            } else {
                nums1[insertIndex] = nums2[nums2Index];
                nums2Index++;
            }
            insertIndex++;
        }
        System.arraycopy(nums2, nums2Index, nums1, insertIndex, n - nums2Index);
    }
}
