class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        for (int i = 0; i < n; i++) {
            int current = nums2[i];
            int j = m + i;
            while (j > 0 && j - 1 >= 0 && nums1[j - 1] > current) {
                nums1[j] = nums1[j - 1];
                j--;
            }
            nums1[j] = current;
        }
    }
}
