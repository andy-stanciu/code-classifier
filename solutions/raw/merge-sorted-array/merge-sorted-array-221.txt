class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int index1 = m - 1, index2 = n - 1, indexAll = m + n - 1;
        // start inserting from the back and do the numeral comparison
        for (int i = indexAll; i >= 0 && index2 >= 0; i--) {
            if (index1 >= 0 && nums1[index1] > nums2[index2]) {
                nums1[i] = nums1[index1--];
            } else {
                nums1[i] = nums2[index2--];
            }
        }
    }
}
