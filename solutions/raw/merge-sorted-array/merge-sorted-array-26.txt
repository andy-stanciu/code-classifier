class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        if ((m + n) == 0)
            return;
        if (n == 0)
            return;
        int i1 = m - 1;
        int i2 = n - 1;
        int i = n + m - 1;
        while (i > -1) {
            if (i1 == -1) {
                nums1[i--] = nums2[i2--];
                continue;
            }
            if (i2 == -1) {
                nums1[i--] = nums1[i1--];
                continue;
            }
            if (nums1[i1] > nums2[i2])
                nums1[i--] = nums1[i1--];
            else
                nums1[i--] = nums2[i2--];
        }
    }
}
