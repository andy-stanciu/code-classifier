class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        // Last element in the valid portion of nums1
        int r1 = m - 1;
        // Last element in nums2
        int r2 = n - 1;
        // Last index of nums1 (combined size)
        int w = m + n - 1;
        while (r2 >= 0) {
            // Process all elements of nums2
            if (r1 >= 0 && nums1[r1] > nums2[r2]) {
                // Place nums1[r1] at the end
                nums1[w--] = nums1[r1--];
            } else {
                // Place nums2[r2] at the end
                nums1[w--] = nums2[r2--];
            }
        }
    }
}
