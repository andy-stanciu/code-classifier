class Solution {

    public void merge(int[] nums1, int len1, int[] nums2, int len2) {
        int i = len1 - 1;
        int j = len2 - 1;
        int k = nums1.length - 1;
        while (i >= 0 && j >= 0 && k >= 0) {
            if (nums1[i] <= nums2[j]) {
                nums1[k] = nums2[j];
                k--;
                j--;
            } else if (nums1[i] > nums2[j]) {
                nums1[k] = nums1[i];
                k--;
                i--;
            }
        }
        while (j >= 0 && k >= 0) {
            nums1[k] = nums2[j];
            j--;
            k--;
        }
    }
}
