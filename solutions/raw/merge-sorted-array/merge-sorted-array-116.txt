class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int[] temp = new int[m + n];
        for (int i = 0; i < m; i++) {
            temp[i] = nums1[i];
        }
        for (int j = 0; j < n; j++) {
            temp[j + m] = nums2[j];
        }
        for (int k = 0; k < temp.length; k++) {
            nums1[k] = temp[k];
        }
        Arrays.sort(nums1);
    }
}
