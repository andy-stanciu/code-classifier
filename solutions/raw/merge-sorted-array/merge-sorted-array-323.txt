class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int rp = m + n - 1;
        int m1 = m - 1;
        int m2 = n - 1;
        if (n == 0) {
            return;
        }
        while (m2 >= 0) {
            if (m1 < 0 || nums2[m2] >= nums1[m1]) {
                nums1[rp] = nums2[m2];
                m2--;
            } else {
                nums1[rp] = nums1[m1];
                m1--;
            }
            rp--;
        }
    }
}
