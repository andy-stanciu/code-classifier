class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int i = m;
        int j = 0;
        while (i < nums1.length) {
            nums1[i] = nums2[j];
            j += 1;
            i += 1;
        }
        int temp;
        for (int i1 = 0; i1 < nums1.length - 1; i1++) {
            for (int j1 = 0; j1 < nums1.length - 1 - i1; j1++) {
                if (nums1[j1] > nums1[j1 + 1]) {
                    temp = nums1[j1];
                    nums1[j1] = nums1[j1 + 1];
                    nums1[j1 + 1] = temp;
                }
            }
        }
    }
}
