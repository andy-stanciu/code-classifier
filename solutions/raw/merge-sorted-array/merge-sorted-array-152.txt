class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        // inialize j;
        int j = 0;
        // traversal start from length of nums1
        for (int i = m; i < m + n; i++) {
            nums1[i] = nums2[j++];
        }
        // after sorting this
        Arrays.sort(nums1);
    }
}
