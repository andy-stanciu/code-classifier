class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        //     int j=0;
        //     for(int i=m;i<m+n;i++){
        //        nums1[i]=nums2[j];
        //         j++;
        //     }
        //     Arrays.sort(nums1);
        // }
        int i = m - 1, j = n - 1;
        for (int k = m + n - 1; k >= 0; k--) {
            if (i >= 0 && j >= 0) {
                if (nums1[i] >= nums2[j]) {
                    nums1[k] = nums1[i];
                    i--;
                } else {
                    nums1[k] = nums2[j];
                    j--;
                }
            }
        }
        if (j >= 0) {
            for (int c = 0; c <= j; c++) {
                nums1[c] = nums2[c];
            }
        }
    }
}
