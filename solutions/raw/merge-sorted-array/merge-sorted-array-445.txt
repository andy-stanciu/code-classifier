class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int left = 0, right = 0;
        // Temporary array to store the merged result
        int[] arr3 = new int[n + m];
        int index = 0;
        // Merge the two arrays
        while (left < m && right < n) {
            if (nums1[left] < nums2[right]) {
                arr3[index] = nums1[left];
                left++;
            } else {
                // Correct array being copied
                arr3[index] = nums2[right];
                right++;
            }
            index++;
        }
        // Copy remaining elements from nums1, if any
        while (left < m) {
            arr3[index] = nums1[left];
            left++;
            index++;
        }
        // Copy remaining elements from nums2, if any
        while (right < n) {
            arr3[index] = nums2[right];
            right++;
            index++;
        }
        // Copy merged array back into nums1 (no need to touch nums2)
        for (int i = 0; i < n + m; i++) {
            // Merged result is placed into nums1 only
            nums1[i] = arr3[i];
        }
    }
}
