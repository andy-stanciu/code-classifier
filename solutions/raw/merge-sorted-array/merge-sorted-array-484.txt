class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int l1 = nums1.length;
        int j = 0;
        for (int i = 0; i < l1; i++) {
            if (nums1[i] == 0 && j < nums2.length) {
                nums1[i] = nums2[j];
                j++;
            }
        }
        Arrays.sort(nums1);
        System.out.println(nums1);
    }
}
