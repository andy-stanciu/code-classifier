class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        for (int i = nums1.length - 1, length1 = m - 1, length2 = n - 1; i >= 0; i--) {
            if (length2 < 0 || (length1 >= 0 && nums1[length1] > nums2[length2])) {
                nums1[i] = nums1[length1--];
                continue;
            }
            nums1[i] = nums2[length2--];
        }
    }
}
