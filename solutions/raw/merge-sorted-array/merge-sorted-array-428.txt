public class Solution {

    public static void merge(int[] nums1, int m, int[] nums2, int n) {
        // Pointer for the end of the initialized elements in nums1
        int i = m - 1;
        // Pointer for the end of nums2
        int j = n - 1;
        // Pointer for the end of nums1 (merged array)
        int k = m + n - 1;
        // Merge in reverse order
        while (i >= 0 && j >= 0) {
            if (nums1[i] > nums2[j]) {
                nums1[k--] = nums1[i--];
            } else {
                nums1[k--] = nums2[j--];
            }
        }
        // If there are remaining elements in nums2, add them
        while (j >= 0) {
            nums1[k--] = nums2[j--];
        }
        // No need to copy the remaining elements of nums1, they are already in place
    }
}
