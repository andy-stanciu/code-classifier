class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        //Brute-force solution
        //take a counter to traverse through nums2 so that no extra loop is needed
        int counter = 0;
        //insert nums2 into nums1 by traversing through nums1 after m(size of nums1)
        for (int i = m; i < nums1.length; i++) {
            //insert nums2 in empty place inside nums1
            nums1[i] = nums2[counter];
            //increase counter by  1 to insert next element of nums2 in next iteration
            counter++;
        }
        //Apply Sort
        Arrays.sort(nums1);
    }
}
