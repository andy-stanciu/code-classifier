class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        Integer[] merged = Stream.concat(Arrays.stream(nums1, 0, m).boxed(), Arrays.stream(nums2).boxed()).sorted().toArray(Integer[]::new);
        for (int i = 0; i < merged.length; i++) {
            nums1[i] = merged[i];
        }
    }
}
