public class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        //pointer for nums1 - merge
        int p1 = m - 1;
        //pointer for nums2 - merge
        int p2 = n - 1;
        //pointer for new nums1
        int p = m + n - 1;
        while (p1 >= 0 && p2 >= 0) {
            if (nums2[p2] >= nums1[p1]) {
                nums1[p] = nums2[p2];
                p2--;
            } else {
                nums1[p] = nums1[p1];
                p1--;
            }
            p--;
        }
        while (p2 >= 0) {
            nums1[p] = nums2[p2];
            p2--;
            p--;
        }
    }
}
