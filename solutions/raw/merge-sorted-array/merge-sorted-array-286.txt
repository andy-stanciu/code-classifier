class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int cont1 = 0, cont2 = 0, total = 0;
        int[] res = new int[n + m];
        if (m == 0) {
            for (int a = 0; a < n; a++) {
                nums1[a] = nums2[a];
            }
        } else if (n != 0) {
            do {
                if (nums1[cont1] < nums2[cont2]) {
                    res[total] = nums1[cont1];
                    cont1++;
                } else {
                    res[total] = nums2[cont2];
                    cont2++;
                }
                total++;
            } while (cont1 < m && cont2 < n);
            while (cont1 != m) {
                res[total] = nums1[cont1];
                cont1++;
                total++;
            }
            while (cont2 != n) {
                res[total] = nums2[cont2];
                cont2++;
                total++;
            }
            for (int a = 0; a < total; a++) {
                nums1[a] = res[a];
            }
        }
    }
}
