class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        //nums1 = array m = nums1 length
        //nums2 = array n = nums2 length
        int finalLength = n + m;
        int[] aux = new int[finalLength];
        int num1Index = 0;
        int num2Index = 0;
        int auxIndex = 0;
        while (auxIndex < finalLength) {
            if (n <= 0) {
                aux = nums1;
                break;
            }
            if (m <= 0) {
                aux = nums2;
                break;
            }
            if (num1Index < m && num2Index >= n || (num1Index < m && nums1[num1Index] < nums2[num2Index])) {
                aux[auxIndex++] = nums1[num1Index++];
            } else if (num2Index < n) {
                aux[auxIndex++] = nums2[num2Index++];
            }
        }
        for (int i = 0; i < aux.length; i++) nums1[i] = aux[i];
    }
}
