//2nd approach
class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        // 2nd approach
        int i = n - 1;
        int j = m + n - 1;
        while (i >= 0) {
            nums1[j] = nums2[i];
            i--;
            j--;
        }
        Arrays.sort(nums1);
    }
}
