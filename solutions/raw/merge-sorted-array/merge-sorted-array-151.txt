class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int x = m + n;
        int[] arr = new int[x];
        int i = 0, j = 0, k = 0;
        while (j < m && k < n) {
            if (nums1[j] < nums2[k]) {
                arr[i] = nums1[j];
                j++;
            } else {
                arr[i] = nums2[k];
                k++;
            }
            i++;
        }
        while (j < m) {
            arr[i++] = nums1[j++];
        }
        while (k < n) {
            arr[i++] = nums2[k++];
        }
        for (int l = 0; l < x; l++) {
            nums1[l] = arr[l];
        }
    }
}
