class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int i = 0, j = 0;
        int[] sortedArray = new int[m + n];
        int pos = 0;
        while (i < m && j < n) {
            if (nums1[i] <= nums2[j]) {
                sortedArray[pos++] = nums1[i++];
                continue;
            }
            if (nums2[j] < nums1[i]) {
                sortedArray[pos++] = nums2[j++];
            }
        }
        while (i < m) {
            sortedArray[pos++] = nums1[i++];
        }
        while (j < n) {
            sortedArray[pos++] = nums2[j++];
        }
        for (int q = 0; q < m + n; q++) {
            nums1[q] = sortedArray[q];
        }
    }
}
