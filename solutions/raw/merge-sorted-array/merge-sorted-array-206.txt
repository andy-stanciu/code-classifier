class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        //Time Complexity: O(m + n)
        //Fill in nums1 array from end, comparing biggest element from each array
        int firstIndex = m - 1;
        int secondIndex = n - 1;
        for (int i = m + n - 1; i > -1; i--) {
            if (secondIndex >= 0 && (firstIndex < 0 || nums2[secondIndex] >= nums1[firstIndex])) {
                nums1[i] = nums2[secondIndex];
                secondIndex--;
            } else {
                nums1[i] = nums1[firstIndex];
                firstIndex--;
            }
        }
    }
}
