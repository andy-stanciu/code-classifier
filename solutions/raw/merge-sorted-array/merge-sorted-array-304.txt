class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int[] a = new int[m + n];
        int pos = 0;
        for (int i = 0; i < m; i++) {
            a[pos] = nums1[i];
            pos++;
        }
        for (int j = 0; j < n; j++) {
            a[pos] = nums2[j];
            pos++;
        }
        Arrays.sort(a);
        for (int i = 0; i < a.length; i++) nums1[i] = a[i];
    }
}
