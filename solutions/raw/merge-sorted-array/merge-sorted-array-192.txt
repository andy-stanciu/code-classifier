class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int j = 0;
        for (int i = m; i < nums1.length; i++) {
            nums1[i] = nums2[j];
            int k = i;
            while ((k > 0) && (nums1[k] < nums1[k - 1])) {
                int tmp = nums1[k];
                nums1[k] = nums1[k - 1];
                nums1[k - 1] = tmp;
                k--;
            }
            j++;
        }
    }
}
