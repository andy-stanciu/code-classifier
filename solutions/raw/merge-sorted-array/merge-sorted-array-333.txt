class Solution {

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int k = m - 1, l = n - 1, o = nums1.length - 1;
        while (l >= 0) {
            if (k >= 0 && nums1[k] > nums2[l]) {
                nums1[o] = nums1[k];
                k--;
                o--;
            } else {
                nums1[o] = nums2[l];
                o--;
                l--;
            }
        }
        System.out.println(Arrays.toString(nums1));
    }
}
