class Solution {

    public boolean wordPattern(String pattern, String s) {
        List<String> s1 = Arrays.asList(s.split(" "));
        if (pattern.length() != s1.size()) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            if (pattern.indexOf(pattern.charAt(i)) != s1.indexOf(s1.get(i))) {
                return false;
            }
        }
        return true;
    }
}
