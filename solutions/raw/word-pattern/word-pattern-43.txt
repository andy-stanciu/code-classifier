class Solution {

    public boolean wordPattern(String pattren, String s) {
        HashSet<String> hs = new HashSet<>();
        HashMap<Character, String> hm = new HashMap<>();
        String[] arr = s.trim().split(" ");
        if (arr.length != pattren.length())
            return false;
        for (int i = 0; i < pattren.length(); i++) {
            char temp = pattren.charAt(i);
            if (hm.containsKey(temp)) {
                if (hm.get(temp).equals(arr[i])) {
                } else {
                    return false;
                }
            } else {
                if (hs.contains(arr[i]))
                    return false;
                hm.put(temp, arr[i]);
                hs.add(arr[i]);
            }
        }
        return true;
    }
}
