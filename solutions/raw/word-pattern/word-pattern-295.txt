class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        char[] chars = pattern.toCharArray();
        if (words.length != chars.length)
            return false;
        Map<Character, String> relations = new HashMap<>();
        for (int i = 0; i < chars.length; i++) {
            if (relations.containsKey(chars[i])) {
                if (!words[i].equals(relations.get(chars[i])))
                    return false;
            } else if (relations.containsValue(words[i]))
                return false;
            relations.put(chars[i], words[i]);
        }
        return true;
    }
}
