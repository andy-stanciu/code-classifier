class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> dict = new HashMap<>();
        //Creates a String array from s. Ex: ["dog", "cat", "dog"] from "dog cat dog"
        String[] words = s.split(" ");
        if (pattern.length() != words.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            if (dict.containsKey(pattern.charAt(i))) {
                if (!dict.get(pattern.charAt(i)).equals(words[i])) {
                    return false;
                }
            } else if (dict.containsValue(words[i])) {
                return false;
            } else {
                dict.put(pattern.charAt(i), words[i]);
            }
        }
        return true;
    }
}
