class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] aa = s.split(" ");
        HashMap<Character, String> map = new HashMap();
        HashSet<String> set = new HashSet();
        if (pattern.length() != aa.length) {
            return false;
        }
        for (int i = 0; i < aa.length; i++) {
            String word = aa[i];
            char val2 = pattern.charAt(i);
            if (map.containsKey(val2) && !map.get(val2).equals(word)) {
                return false;
            } else if (!map.containsKey(val2) && set.contains(word)) {
                return false;
            }
            map.put(val2, word);
            set.add(word);
        }
        return true;
    }
}
