class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        if (pattern.length() != words.length) {
            // Different lengths means different patterns
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < pattern.length(); i++) {
            char ch = pattern.charAt(i);
            String word = words[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(word)) {
                    // If the pattern does not match the word
                    return false;
                }
            } else {
                if (map.containsValue(word)) {
                    // If the same word corresponds to different patterns
                    return false;
                }
                // Add the pattern and word to the map
                map.put(ch, word);
            }
        }
        // If the pattern and word mappings hold true
        return true;
    }
}
