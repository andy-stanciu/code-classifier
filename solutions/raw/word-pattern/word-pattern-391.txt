class Solution {

    public boolean wordPattern(String pattern, String s) {
        var words = s.split(" ");
        var chars = pattern.toCharArray();
        if (words.length != chars.length) {
            return false;
        }
        var wordMap = new HashMap<Character, String>();
        for (int i = 0; i < chars.length; i++) {
            var mapWord = wordMap.get(chars[i]);
            if (mapWord == null) {
                if (wordMap.containsValue(words[i])) {
                    return false;
                }
                wordMap.put(chars[i], words[i]);
                continue;
            }
            if (!Objects.equals(mapWord, words[i])) {
                return false;
            }
        }
        return true;
    }
}
