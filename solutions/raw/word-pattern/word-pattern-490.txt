class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> m = new HashMap<>();
        Map<String, Character> cm = new HashMap<>();
        String[] a = s.split(" ");
        if (pattern.length() != a.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            char c = pattern.charAt(i);
            if (m.containsKey(c)) {
                if (!a[i].equals(m.get(c))) {
                    return false;
                }
                // if(cm.get(a[i])!=c){
                //     return true;
                // }
            } else {
                m.put(c, a[i]);
                if (cm.containsKey(a[i])) {
                    return false;
                }
                cm.put(a[i], c);
            }
        }
        return true;
    }
}
