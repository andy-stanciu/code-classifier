class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] stringList = s.split("\\\\s+");
        Map<Character, String> charMap = new HashMap<>();
        if (pattern.length() != stringList.length)
            return false;
        for (int i = 0; i < pattern.length(); i++) {
            Character currentChar = pattern.charAt(i);
            String currentString = stringList[i];
            if (charMap.containsKey(currentChar)) {
                if (!charMap.get(currentChar).equals(currentString)) {
                    return false;
                }
            } else {
                if (charMap.containsValue(currentString))
                    return false;
                charMap.put(currentChar, currentString);
            }
        }
        return true;
    }
}
