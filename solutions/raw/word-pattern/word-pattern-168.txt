class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] tocken = s.split(" ");
        HashMap<String, Character> map = new HashMap<>();
        Set<Character> used = new HashSet<>();
        if (pattern.length() != tocken.length) {
            return false;
        }
        for (int i = 0; i < tocken.length; i++) {
            char ch = pattern.charAt(i);
            String word = tocken[i];
            if (map.containsKey(word) && ch == map.get(word)) {
                continue;
            } else if (map.containsKey(word) && ch != map.get(word)) {
                return false;
            } else if (!map.containsKey(word) && used.contains(ch)) {
                return false;
            } else {
                map.put(word, ch);
                used.add(ch);
            }
        }
        return true;
    }
}
