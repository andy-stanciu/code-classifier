class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> a = new HashMap<>();
        String[] splitArray = s.split("\\\\s+");
        if (pattern.length() != splitArray.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            if (a.containsKey(pattern.charAt(i))) {
                if (!a.get(pattern.charAt(i)).equals(splitArray[i])) {
                    return false;
                }
            } else {
                if (a.containsValue(splitArray[i])) {
                    return false;
                }
                a.put(pattern.charAt(i), splitArray[i]);
            }
        }
        return true;
    }
}
