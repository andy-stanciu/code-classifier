class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] arr = s.split(" ");
        if (arr.length != pattern.length())
            return false;
        for (int i = 0; i < pattern.length(); i++) {
            char ch = pattern.charAt(i);
            if (//if key character exist
            map.containsKey(ch)) {
                if (//has to match with respective string
                !map.get(ch).equals(arr[i]))
                    return false;
            } else {
                if (//this string is already mapped to another key
                map.containsValue(arr[i]))
                    //this is not in fit in rule
                    return false;
                map.put(ch, arr[i]);
            }
        }
        return true;
    }
}
