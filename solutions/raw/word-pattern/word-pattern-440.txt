class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] words = s.split("\\\\s");
        if (pattern.length() != words.length)
            return false;
        for (int i = 0; i < pattern.length(); i++) {
            char patternCh = pattern.charAt(i);
            String replacedWord = words[i];
            if (!map.containsKey(patternCh)) {
                if (!map.containsValue(replacedWord)) {
                    map.put(patternCh, replacedWord);
                } else {
                    return false;
                }
            } else {
                String mappedWord = map.get(patternCh);
                if (!mappedWord.equals(replacedWord)) {
                    return false;
                }
            }
        }
        return true;
    }
}
