class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<String, String> patWord = new HashMap<>();
        String[] words = s.split(" ");
        char[] chars = pattern.toCharArray();
        if (chars.length != words.length)
            return false;
        for (int i = 0; i < words.length; i++) {
            String patternLetter = String.valueOf(chars[i]);
            //String
            if (patWord.containsKey(words[i]) && !patternLetter.equals(patWord.get(words[i])))
                return false;
            else if (patWord.containsValue(patternLetter) && patWord.get(words[i]) == null)
                return false;
            else
                patWord.put(words[i], String.valueOf(chars[i]));
        }
        return true;
    }
}
