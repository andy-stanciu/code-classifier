class Solution {

    public boolean wordPattern(String word, String values) {
        Map<Character, String> map = new HashMap<>();
        String[] words = values.split(" ");
        if (words.length != word.length()) {
            return false;
        }
        for (int i = 0; i < word.length(); i++) {
            if (map.containsKey(word.charAt(i))) {
                if (!map.get(word.charAt(i)).equals(words[i])) {
                    return false;
                }
            } else {
                if (map.containsValue(words[i])) {
                    return false;
                }
                map.put(word.charAt(i), words[i]);
            }
        }
        return true;
    }
}
