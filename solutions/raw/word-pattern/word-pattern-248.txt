class Solution {

    public boolean wordPattern(String pattern, String st) {
        String[] arr = st.split(" ");
        Map<Character, String> map = new HashMap<>();
        Map<String, Character> revmap = new HashMap<>();
        if (pattern.length() != arr.length) {
            return false;
        }
        for (int i = 0; i < arr.length; i++) {
            char c = pattern.charAt(i);
            String s = arr[i];
            if (map.containsKey(c)) {
                String val = map.get(c);
                if (!val.equals(s)) {
                    return false;
                }
            } else {
                map.put(c, s);
            }
            if (revmap.containsKey(s)) {
                if (!revmap.get(s).equals(c)) {
                    return false;
                }
            } else {
                revmap.put(s, c);
            }
        }
        return true;
    }
}
