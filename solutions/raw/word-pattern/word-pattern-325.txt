class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> woord = new HashMap<>();
        String[] arr = s.split(" ");
        HashSet<String> set = new HashSet<>();
        if (pattern.length() != arr.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            char a = pattern.charAt(i);
            String b = arr[i];
            set.add(arr[i]);
            if (woord.containsKey(a)) {
                if (!woord.get(a).equals(b)) {
                    return false;
                }
            }
            // if(i>=1&&arr[i]==arr[i-1]){
            //     return false;
            // }
            woord.put(a, b);
        }
        if (set.size() != woord.size()) {
            return false;
        }
        return true;
    }
}
