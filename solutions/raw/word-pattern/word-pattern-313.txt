class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] arr = s.split(" ");
        int n = arr.length;
        if (arr.length != pattern.length())
            return false;
        for (int i = 0; i < arr.length; i++) {
            char currpat = pattern.charAt(i);
            String currs = arr[i];
            for (int j = i + 1; j < n; j++) {
                if (pattern.charAt(j) == currpat) {
                    if (!currs.equals(arr[j]))
                        return false;
                }
                if (arr[j].equals(currs)) {
                    if (pattern.charAt(j) != currpat)
                        return false;
                }
            }
        }
        return true;
    }
}
