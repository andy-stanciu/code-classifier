class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> h = new HashMap<>();
        String[] arr = s.split(" ");
        if (pattern.length() != arr.length)
            return false;
        for (int i = 0; i < pattern.length(); i++) {
            char ch = pattern.charAt(i);
            String th = arr[i];
            if (h.containsKey(ch)) {
                if (!h.get(ch).equals(th))
                    return false;
            } else if (h.containsValue(th))
                return false;
            else
                h.put(ch, th);
        }
        return true;
    }
}
