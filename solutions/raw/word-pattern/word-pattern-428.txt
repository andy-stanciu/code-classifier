class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] arr1 = pattern.split("");
        String[] arr2 = s.split(" ");
        if (arr1.length != arr2.length)
            return false;
        Map<String, String> map = new HashMap<>();
        for (int i = 0; i < arr1.length; i++) {
            String pdemo = arr1[i];
            String sdemo = arr2[i];
            if (map.containsKey(pdemo)) {
                if (!map.get(pdemo).equals(sdemo))
                    return false;
            } else {
                if (map.containsValue(sdemo))
                    return false;
                map.put(pdemo, sdemo);
            }
        }
        return true;
    }
}
