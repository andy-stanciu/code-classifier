class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] pattern_array = pattern.split("");
        String[] s_array = s.split(" ");
        if (pattern_array.length != s_array.length) {
            return false;
        }
        HashMap<String, String> map = new HashMap<>();
        for (int lett = 0; lett < s_array.length; lett++) {
            if (!map.containsKey(pattern_array[lett]) && !map.containsValue(s_array[lett])) {
                map.put(pattern_array[lett], s_array[lett]);
            } else {
                String value = map.get(pattern_array[lett]);
                if (value == null) {
                    return false;
                }
                if (!value.equals(s_array[lett])) {
                    return false;
                }
            }
        }
        return true;
    }
}
