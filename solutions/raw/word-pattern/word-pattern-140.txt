class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> mp = new HashMap<>();
        Map<String, Character> ms = new HashMap<>();
        String[] words = s.split(" ");
        if (pattern.length() != words.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            // verify letter-to-word
            char key = pattern.charAt(i);
            if (mp.containsKey(key)) {
                String val = mp.get(key);
                if (!val.equals(words[i])) {
                    return false;
                }
            } else {
                mp.put(key, words[i]);
            }
            String keys = words[i];
            if (ms.containsKey(keys)) {
                char vals = ms.get(keys);
                if (vals != pattern.charAt(i)) {
                    return false;
                }
            } else {
                ms.put(keys, pattern.charAt(i));
            }
        }
        return true;
    }
}
