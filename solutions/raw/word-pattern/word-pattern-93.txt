class Solution {

    public boolean wordPattern(String pattern, String s) {
        char[] arrOfLetter = pattern.toCharArray();
        String[] arrOfWord = s.split(" ");
        // for ( String h: arrOfWord) System.out.println(h);
        // for ( char h: arrOfLetter) System.out.print(h);
        boolean res = true;
        if (arrOfWord.length != arrOfLetter.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        HashSet<String> usedWords = new HashSet<>();
        for (int i = 0; i < arrOfLetter.length; i++) {
            if (map.get(arrOfLetter[i]) != null && !map.get(arrOfLetter[i]).equals(arrOfWord[i])) {
                // System.out.println("Key: " + arrOfLetter[i] + ", Expected: " + map.get(arrOfLetter[i]) + ", we have: " + arrOfWord[i]);
                return false;
            } else if (map.get(arrOfLetter[i]) == null && usedWords.contains(arrOfWord[i])) {
                return false;
            }
            map.put(arrOfLetter[i], arrOfWord[i]);
            usedWords.add(arrOfWord[i]);
            // for (char ch: map.keySet()) System.out.print(ch + ": " + map.get(ch) + ", ");System.out.println("");
        }
        return res;
    }
}
