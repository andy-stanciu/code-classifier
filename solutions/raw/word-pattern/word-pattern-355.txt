class Solution {

    public boolean wordPattern(String pattern, String s) {
        // initialize the map
        Map<Character, String> map = new HashMap<>();
        // convert the words in s into an array
        String[] s_arr = s.split(" ");
        // check if arr.length == pattern.length -> if not return false
        if (s_arr.length != pattern.length()) {
            return false;
        }
        // map the chars of pattern to the words in s array
        // if mapping not present, put
        // else get and compare the value and arr element
        for (int i = 0; i < pattern.length(); i++) {
            String s_str = s_arr[i];
            Character c_pattern = pattern.charAt(i);
            if (map.containsKey(c_pattern)) {
                if (map.get(c_pattern).equals(s_str)) {
                    continue;
                } else {
                    return false;
                }
            } else {
                if (map.containsValue(s_str)) {
                    return false;
                } else {
                    map.put(c_pattern, s_str);
                }
            }
        }
        return true;
    }
}
