class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> pkey = new HashMap<>();
        Map<String, Character> skey = new HashMap<>();
        String[] str = s.split(" ");
        if (pattern.length() != str.length)
            return false;
        for (int i = 0; i < pattern.length(); i++) {
            char pchar = pattern.charAt(i);
            String schar = str[i];
            if (pkey.containsKey(pchar) && !pkey.get(pchar).equals(schar))
                return false;
            else
                pkey.put(pchar, schar);
            if (skey.containsKey(schar) && skey.get(schar) != pchar)
                return false;
            else
                skey.put(schar, pchar);
        }
        return true;
    }
}
