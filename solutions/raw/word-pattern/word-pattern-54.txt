class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> dict = new HashMap<>();
        String[] match = s.split(" ");
        if (match.length != pattern.length())
            return false;
        List<String> lt = new ArrayList<>();
        int i = 0;
        for (char ind : pattern.toCharArray()) {
            if (dict.containsKey(ind)) {
                if (!(dict.get(ind).equals(match[i]))) {
                    return false;
                }
            } else {
                if (lt.contains(match[i]))
                    return false;
                dict.put(ind, match[i]);
                lt.add(match[i]);
            }
            i++;
        }
        return true;
    }
}
