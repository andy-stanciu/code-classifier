class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.trim().split("\\\\s+");
        if (words.length != pattern.length()) {
            return false;
        }
        Map<Character, Integer> appear = new HashMap<>();
        Set<String> wordsSoFar = new HashSet<>();
        for (int i = 0; i < pattern.length(); i++) {
            char curr = pattern.charAt(i);
            if (appear.containsKey(curr)) {
                if (!words[i].equals(words[appear.get(curr)])) {
                    return false;
                }
            } else {
                if (!wordsSoFar.add(words[i])) {
                    return false;
                }
            }
            appear.put(curr, i);
        }
        return true;
    }
}
