class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] split_string = s.split(" ");
        // Check if the number of words matches the length of the pattern
        if (pattern.length() != split_string.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            char currentChar = pattern.charAt(i);
            String currentWord = split_string[i];
            if (map.containsKey(currentChar)) {
                // If the pattern character already has a mapping, it must match the current word
                if (!map.get(currentChar).equals(currentWord)) {
                    return false;
                }
            } else {
                // If the pattern character doesn\'t have a mapping yet, but the word is already mapped
                if (map.containsValue(currentWord)) {
                    return false;
                }
                // Add the mapping
                map.put(currentChar, currentWord);
            }
        }
        return true;
    }
}
