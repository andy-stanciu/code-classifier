class Solution {

    public boolean wordPattern(String pattern, String s) {
        String a = getString(pattern);
        int n;
        int incrementor = 1;
        String result = "";
        String[] subs = s.split(" ");
        HashMap<String, Integer> hmap = new HashMap<>();
        for (String ch : subs) {
            n = 0;
            if (hmap.containsKey(ch)) {
                n = hmap.get(ch);
                result = result + n;
            } else {
                hmap.put(ch, incrementor);
                result = result + incrementor;
                incrementor++;
            }
        }
        return a.equals(result);
    }

    public String getString(String s) {
        int n;
        int incrementor = 1;
        String result = "";
        HashMap<Character, Integer> hmap = new HashMap<>();
        for (char ch : s.toCharArray()) {
            n = 0;
            if (hmap.containsKey(ch)) {
                n = hmap.get(ch);
                result = result + n;
            } else {
                hmap.put(ch, incrementor);
                result = result + incrementor;
                incrementor++;
            }
        }
        return result;
    }
}
