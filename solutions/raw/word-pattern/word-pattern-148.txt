class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] arr = s.split(" ");
        if (pattern.length() != arr.length) {
            return false;
        }
        HashMap<String, String> map = new HashMap<String, String>();
        for (int arrIndex = 0, patternIndex = 0; arrIndex < arr.length; arrIndex++, patternIndex++) {
            String strPattern = String.valueOf(pattern.charAt(patternIndex));
            if (map.containsKey(strPattern)) {
                if (!map.get(strPattern).equals(arr[arrIndex])) {
                    return false;
                }
            } else {
                if (map.containsValue(arr[arrIndex])) {
                    return false;
                }
                map.put(strPattern, arr[arrIndex]);
            }
        }
        return true;
    }
}
