class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] str = s.split(" ");
        if (str.length != pattern.length()) {
            return false;
        }
        Map<Character, String> patMap = new HashMap<>();
        Map<String, Character> sMap = new HashMap<>();
        int length = str.length;
        for (int i = 0; i < length; i++) {
            Character pChar = pattern.charAt(i);
            String word = str[i];
            if (patMap.containsKey(pChar)) {
                if (!patMap.get(pChar).equals(word)) {
                    return false;
                }
            } else {
                patMap.put(pChar, word);
            }
            if (sMap.containsKey(word)) {
                if (!Objects.equals(sMap.get(word), pChar)) {
                    return false;
                }
            } else {
                sMap.put(word, pChar);
            }
        }
        return true;
    }
}
