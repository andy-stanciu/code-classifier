class Solution {

    public boolean wordPattern(String pattern, String s) {
        char[] char_array = pattern.toCharArray();
        String[] str_array = s.split(" ");
        if (str_array.length != char_array.length) {
            return false;
        }
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < str_array.length; i++) {
            if (map.containsKey(char_array[i])) {
                if (!map.get(char_array[i]).equals(str_array[i])) {
                    return false;
                }
            } else if (!map.containsKey(char_array[i]) && map.containsValue(str_array[i])) {
                return false;
            } else {
                map.put(char_array[i], str_array[i]);
            }
        }
        return true;
    }
}
