class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        HashMap<Character, String> mp = new HashMap<Character, String>();
        int n = pattern.length();
        if (n != words.length)
            return false;
        for (int i = 0; i < n; i++) {
            if (mp.get(pattern.charAt(i)) == null) {
                if (!mp.values().contains(words[i])) {
                    mp.put(pattern.charAt(i), words[i]);
                } else
                    return false;
            } else {
                if (!mp.get(pattern.charAt(i)).equals(words[i]))
                    return false;
            }
        }
        return true;
    }
}
