class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] arr = s.split(" ");
        if (pattern.length() != arr.length)
            return false;
        Hashtable<Character, String> h = new Hashtable<>();
        for (int i = 0; i < pattern.length(); i++) {
            char c1 = pattern.charAt(i);
            String word = arr[i];
            if (h.containsKey(c1)) {
                if (!h.get(c1).equals(word))
                    return false;
            } else {
                if (h.containsValue(word))
                    return false;
                h.put(c1, word);
            }
        }
        return true;
    }
}
