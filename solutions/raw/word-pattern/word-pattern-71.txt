class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] str = s.split(" ");
        if (str.length != pattern.length())
            return false;
        HashMap<Character, String> hm = new HashMap<>();
        // Iterating through each element.
        for (int i = 0; i < pattern.length(); i++) {
            char a = pattern.charAt(i);
            if (!hm.containsKey(a)) {
                if (// current word in str[i] is not already mapped.
                hm.containsValue(str[i]))
                    return false;
                hm.put(a, str[i]);
            } else {
                // Using Objects class to compare actual values.
                if (!Objects.equals(hm.get(a), str[i]))
                    return false;
            }
        }
        return true;
    }
}
