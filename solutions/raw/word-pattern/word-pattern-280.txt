class Solution {

    public boolean wordPattern(String pattern, String s) {
        char[] p = pattern.toCharArray();
        String[] str = s.split(" ");
        if (p.length != str.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < p.length; i++) {
            char word = p[i];
            String tring = str[i];
            if (map.containsKey(word)) {
                if (!map.get(word).equals(tring)) {
                    return false;
                }
            } else {
                if (map.containsValue(tring)) {
                    return false;
                }
                map.put(word, tring);
            }
        }
        return true;
    }
}
