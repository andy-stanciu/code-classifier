class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        if (pattern.length() != words.length)
            return false;
        HashMap<Character, String> mapPat = new HashMap<>();
        HashMap<String, Character> mapWord = new HashMap<>();
        for (int i = 0; i < pattern.length(); i++) {
            char pat = pattern.charAt(i);
            String word = words[i];
            if (mapPat.containsKey(pat)) {
                if (!mapPat.get(pat).equals(word)) {
                    return false;
                }
            } else {
                mapPat.put(pat, word);
            }
            if (mapWord.containsKey(word)) {
                if (mapWord.get(word) != pat) {
                    return false;
                }
            } else {
                mapWord.put(word, pat);
            }
        }
        return true;
    }
}
