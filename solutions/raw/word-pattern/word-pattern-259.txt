class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        return stringToDiff(List.of(words)).equals(stringToDiff(pattern.chars().mapToObj(c -> (char) c).collect(Collectors.toList())));
    }

    public <T> String stringToDiff(List<T> elements) {
        HashMap<T, Integer> wordMap = new HashMap<>();
        StringBuilder sb = new StringBuilder();
        int count = 0;
        for (T word : elements) {
            if (!wordMap.containsKey(word)) {
                wordMap.put(word, count);
                count++;
            }
            sb.append(wordMap.get(word));
        }
        return sb.toString();
    }
}
