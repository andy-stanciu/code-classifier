class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] str = s.split(" ");
        if (str.length != pattern.length()) {
            return false;
        }
        int j = 0;
        HashMap<Character, String> mpp = new HashMap<>();
        for (int i = 0; i < str.length; i++) {
            if (mpp.containsKey(pattern.charAt(i)) && !mpp.get(pattern.charAt(i)).equals(str[i])) {
                return false;
            }
            if (!mpp.containsKey(pattern.charAt(i))) {
                if (mpp.containsValue(str[i])) {
                    return false;
                }
                mpp.put(pattern.charAt(i), str[i]);
            }
        }
        return true;
    }
}
