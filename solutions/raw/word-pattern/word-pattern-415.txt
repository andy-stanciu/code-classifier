class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] word = s.split(" ");
        // separate words in string s
        if (word.length != pattern.length())
            return false;
        HashMap<Character, String> map = new HashMap();
        for (int i = 0; i < pattern.length(); i++) {
            char c = pattern.charAt(i);
            if (map.containsKey(c)) {
                // if map contains character(key) but is not matching to its value
                if (!map.get(c).equals(word[i])) {
                    return false;
                }
            } else {
                // if map contains that value but corresponding to some other key
                if (map.containsValue(word[i])) {
                    return false;
                }
                // add to map
                map.put(c, word[i]);
            }
        }
        return true;
    }
}
