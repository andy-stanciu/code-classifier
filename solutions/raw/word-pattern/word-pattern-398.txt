class Solution {

    public boolean wordPattern(String pattern, String str) {
        String[] splited_Str = str.split(" ");
        if (splited_Str.length != pattern.length())
            return false;
        Map string_Map = new HashMap();
        for (Integer i = 0; i < splited_Str.length; ++i) {
            if (string_Map.put(pattern.charAt(i), i) != string_Map.put(splited_Str[i], i))
                return false;
        }
        return true;
    }
}
