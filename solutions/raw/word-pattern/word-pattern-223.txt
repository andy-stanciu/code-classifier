class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] w = s.split(" ");
        if (w.length != pattern.length())
            return false;
        Map i = new HashMap();
        for (Integer j = 0; j < w.length; j++) if (i.put(pattern.charAt(j), j) != i.put(w[j], j))
            return false;
        return true;
    }
}
