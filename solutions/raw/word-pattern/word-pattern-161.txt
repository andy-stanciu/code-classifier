class Solution {

    public boolean wordPattern(String pattern, String s) {
        // All declarations
        HashMap<Character, String> h1 = new HashMap<>();
        HashMap<String, Character> h2 = new HashMap<>();
        int l = pattern.length();
        String[] array = s.split(" ");
        // Check if the length of pattern and number of words match
        if (l != array.length) {
            return false;
        }
        // Main logic for code
        for (int i = 0; i < l; i++) {
            // Establishing relation and also checking
            char c = pattern.charAt(i);
            String word = array[i];
            if (h1.containsKey(c)) {
                if (!h1.get(c).equals(word)) {
                    return false;
                }
            } else {
                h1.put(c, word);
            }
            if (h2.containsKey(word)) {
                if (h2.get(word) != c) {
                    return false;
                }
            } else {
                h2.put(word, c);
            }
        }
        return true;
    }
}
