class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] allStrings = s.split(" ");
        if (pattern.length() != allStrings.length)
            return false;
        int n = pattern.length();
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (map.containsKey(pattern.charAt(i))) {
                if (map.get(pattern.charAt(i)).equals(allStrings[i]))
                    continue;
                else
                    return false;
            } else {
                if (map.containsValue(allStrings[i]))
                    return false;
                map.put(pattern.charAt(i), allStrings[i]);
            }
        }
        return true;
    }
}
