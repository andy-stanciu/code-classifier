class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> char_word = new HashMap<>();
        Map<String, Character> word_char = new HashMap<>();
        String[] words = s.split(" ");
        if (words.length != pattern.length()) {
            return false;
        }
        for (int i = 0; i < words.length; i++) {
            char a = pattern.charAt(i);
            String word = words[i];
            if (!char_word.containsKey(a)) {
                char_word.put(a, word);
            }
            if (!word_char.containsKey(word)) {
                word_char.put(word, a);
            }
            if (!char_word.get(a).equals(word) || !word_char.get(word).equals(a)) {
                return false;
            }
        }
        return true;
    }
}
