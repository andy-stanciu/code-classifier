class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> ctw = new HashMap<>();
        Map<String, Character> wtc = new HashMap<>();
        String[] str = s.split(" ");
        int n = pattern.length();
        if (n != str.length) {
            return false;
        }
        for (int i = 0; i < n; i++) {
            char c = pattern.charAt(i);
            String word = str[i];
            if (ctw.containsKey(c)) {
                if (!ctw.get(c).equals(word)) {
                    return false;
                }
            } else {
                ctw.put(c, word);
            }
            if (wtc.containsKey(word)) {
                if (!wtc.get(word).equals(c)) {
                    return false;
                }
            } else {
                wtc.put(word, c);
            }
        }
        return true;
    }
}
