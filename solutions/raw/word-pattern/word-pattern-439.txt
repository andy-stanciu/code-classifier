class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> m = new HashMap<>();
        String[] words = s.split("\\\\s");
        if (pattern.length() != words.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            char original = pattern.charAt(i);
            String replace = words[i];
            if (!m.containsKey(original)) {
                if (!m.containsValue(replace)) {
                    m.put(original, replace);
                } else {
                    return false;
                }
            } else {
                String mappedWord = m.get(original);
                if (!mappedWord.equals(replace)) {
                    return false;
                }
            }
        }
        return true;
    }
}
