class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        char[] p = pattern.toCharArray();
        if (words.length != p.length || p == null || words == null) {
            return false;
        }
        HashMap<Character, String> H1 = new HashMap<>();
        HashMap<String, Character> H2 = new HashMap<>();
        for (int i = 0; i < p.length; i++) {
            if (H1.containsKey(p[i])) {
                String value = H1.get(p[i]);
                if (value.equals(words[i])) {
                    continue;
                } else {
                    return false;
                }
            } else {
                H1.put(p[i], words[i]);
            }
            if (H2.containsKey(words[i])) {
                char value1 = H2.get(words[i]);
                if (value1 == p[i]) {
                    continue;
                } else {
                    return false;
                }
            } else {
                H2.put(words[i], p[i]);
            }
        }
        return true;
    }
}
