class Solution {

    public boolean wordPattern(String s1, String s) {
        String[] s2 = s.split(" ");
        if (s1.length() != s2.length) {
            return false;
        }
        HashMap<Character, String> hs = new HashMap<>();
        for (int i = 0; i < s1.length(); i++) {
            if (!hs.containsKey(s1.charAt(i))) {
                if (hs.containsValue(s2[i])) {
                    return false;
                }
                hs.put(s1.charAt(i), s2[i]);
            } else if (hs.containsKey(s1.charAt(i))) {
                String a = hs.get(s1.charAt(i));
                String b = s2[i];
                if (!a.equals(b)) {
                    System.out.println(a + b);
                    return false;
                }
            }
        }
        return true;
    }
}
