class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] ts = s.split(" ");
        if (pattern.length() != ts.length)
            return false;
        HashMap<Character, String> hm = new HashMap<Character, String>();
        for (int i = 0; i < pattern.length(); i++) {
            char cs = pattern.charAt(i);
            if (hm.containsKey(cs)) {
                if (!hm.get(cs).equals(ts[i])) {
                    return false;
                }
            } else if (hm.containsValue(ts[i])) {
                return false;
            } else {
                hm.put(cs, ts[i]);
            }
        }
        return true;
    }
}
