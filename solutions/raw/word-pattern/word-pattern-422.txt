class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] wordsArray = s.split(" ");
        if (wordsArray.length != pattern.length()) {
            return false;
        }
        HashMap<Character, String> mapPattern = new HashMap<Character, String>();
        for (int i = 0; i < pattern.length(); i++) {
            if (mapPattern.containsKey(pattern.charAt(i)) && !mapPattern.get(pattern.charAt(i)).equals(wordsArray[i])) {
                return false;
            }
            if (!mapPattern.containsKey(pattern.charAt(i))) {
                if (!mapPattern.containsValue(wordsArray[i]))
                    mapPattern.put(pattern.charAt(i), wordsArray[i]);
                else
                    return false;
            }
        }
        return true;
    }
}
