class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        char[] characters = pattern.toCharArray();
        //check the size of both string if not same return false
        if (words.length != characters.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < characters.length; i++) {
            if (map.containsKey(characters[i])) {
                //contain pr match nhi kar raha phle kuchh tha ab kuchh hai
                if (!map.get(characters[i]).equals(words[i]))
                    return false;
            } else {
                //contains nhi kra the pr iska dost phle se h to duplicate ho raha to return false;
                if (map.containsValue(words[i]))
                    return false;
                map.put(characters[i], words[i]);
            }
        }
        return true;
    }
}
