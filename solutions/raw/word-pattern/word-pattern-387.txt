class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] a = s.split(" ");
        int l = pattern.length();
        if (l != a.length)
            return false;
        HashMap<Character, String> h = new HashMap<>();
        HashMap<String, Character> g = new HashMap<>();
        String t;
        Character u;
        for (int i = 0; i < l; i++) {
            t = h.get(pattern.charAt(i));
            u = g.get(a[i]);
            if (t == null && u == null) {
                h.put(pattern.charAt(i), a[i]);
                g.put(a[i], pattern.charAt(i));
            }
            if (t != null && u != null) {
                if (!t.equals(a[i]) || u != pattern.charAt(i))
                    return false;
            } else if ((t == null && u != null) || (t != null && u == null))
                return false;
        }
        return true;
    }
}
