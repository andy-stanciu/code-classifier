class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<String, String> patternMap = new HashMap<>();
        Map<String, String> sMap = new HashMap<>();
        String[] str = s.split(" ");
        if (pattern.length() != str.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            String st = String.valueOf(pattern.charAt(i));
            String strn = str[i];
            if (patternMap.containsKey(st)) {
                if (!patternMap.get(st).equals(strn)) {
                    return false;
                }
            } else {
                patternMap.put(st, strn);
            }
            if (sMap.containsKey(strn)) {
                if (!sMap.get(strn).equals(st)) {
                    return false;
                }
            } else {
                sMap.put(strn, st);
            }
        }
        return true;
    }
}
