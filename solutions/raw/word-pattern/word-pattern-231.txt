class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> hashmap = new HashMap<>();
        String[] splittedText = s.split(" ");
        if (pattern.length() != splittedText.length) {
            return false;
        }
        for (int i = 0; i < splittedText.length; i++) {
            if (hashmap.get(pattern.charAt(i)) == null && hashmap.containsValue(splittedText[i]) || (hashmap.get(pattern.charAt(i)) != null && !hashmap.get(pattern.charAt(i)).equals(splittedText[i]))) {
                return false;
            } else {
                hashmap.put(pattern.charAt(i), splittedText[i]);
            }
        }
        return true;
    }
}
