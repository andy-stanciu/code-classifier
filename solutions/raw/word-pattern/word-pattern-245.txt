class Solution {

    public boolean wordPattern(String pattern, String s) {
        // to map a string to its character
        Map<Character, String> m = new HashMap();
        int leftp = 0;
        String[] arr = s.split(" ");
        if (pattern.length() != arr.length)
            return false;
        while (leftp < pattern.length()) {
            if (!m.containsKey(pattern.charAt(leftp))) {
                if (m.containsValue(arr[leftp])) {
                    return false;
                }
                m.put(pattern.charAt(leftp), arr[leftp]);
            } else {
                if (!m.get(pattern.charAt(leftp)).equals(arr[leftp]))
                    return false;
            }
            leftp++;
        }
        return true;
    }
}
