class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> hm = new HashMap<>();
        // make one String array by splitting s by blank space
        String[] stA = s.split(" ");
        if (stA.length != pattern.length())
            return false;
        else {
            // Map pattern character and stA array string
            // if key repeats in map and correspoding String is different then return false
            // if key is new but but correspoding String repeats then return false
            for (int i = 0; i < stA.length; i++) {
                if (hm.containsKey(pattern.charAt(i))) {
                    if (!stA[i].trim().equals(hm.get(pattern.charAt(i)).trim()))
                        return false;
                } else if (hm.containsValue(stA[i].trim()))
                    return false;
                hm.put(pattern.charAt(i), stA[i].trim());
            }
            return true;
        }
    }
}
