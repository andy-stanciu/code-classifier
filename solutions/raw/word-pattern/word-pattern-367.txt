class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] arr = s.split(" ");
        if (pattern.length() != arr.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < pattern.length(); i++) {
            char ch = pattern.charAt(i);
            boolean conKey = map.containsKey(ch);
            if (!conKey && map.containsValue(arr[i]))
                return false;
            if (conKey && !map.get(ch).equals(arr[i]))
                return false;
            else {
                map.put(ch, arr[i]);
            }
        }
        return true;
    }
}
