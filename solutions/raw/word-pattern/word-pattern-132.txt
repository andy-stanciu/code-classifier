class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] arr = s.split(" ");
        if (pattern.length() != arr.length) {
            return false;
        }
        HashMap<Character, String> h = new HashMap<>();
        for (int i = 0; i < pattern.length(); i++) {
            // Fix typo here
            char c = pattern.charAt(i);
            // If the key is already mapped but to a different word
            if (h.containsKey(c)) {
                if (!h.get(c).equals(arr[i])) {
                    return false;
                }
            } else {
                // If the value is already mapped to a different key
                if (h.containsValue(arr[i])) {
                    return false;
                }
                h.put(c, arr[i]);
            }
        }
        return true;
    }
}
