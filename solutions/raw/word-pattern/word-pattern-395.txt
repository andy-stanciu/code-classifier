class Solution {

    public boolean wordPattern(String pattern, String s) {
        Map<Character, String> mp = new HashMap<>();
        int i = 0;
        String[] ss = s.split(" ");
        if (ss.length != pattern.length())
            return false;
        for (String st : ss) {
            if (!mp.containsKey(pattern.charAt(i))) {
                if (mp.containsValue(st))
                    return false;
                mp.put(pattern.charAt(i++), st);
            } else {
                if (!mp.get(pattern.charAt(i++)).equals(st))
                    return false;
            }
        }
        return true;
    }
}
