class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        if (pattern.length() != words.length) {
            return false;
        }
        Map<Character, String> patternMap = new HashMap<>();
        Map<String, Character> wordMap = new HashMap<>();
        for (int i = 0; i < pattern.length(); i++) {
            char currChar = pattern.charAt(i);
            String currWord = words[i];
            if (patternMap.containsKey(currChar)) {
                if (!patternMap.get(currChar).equals(currWord)) {
                    return false;
                }
            } else {
                patternMap.put(currChar, currWord);
            }
            if (wordMap.containsKey(currWord)) {
                if (wordMap.get(currWord) != currChar) {
                    return false;
                }
            } else {
                wordMap.put(currWord, currChar);
            }
        }
        return true;
    }
}
