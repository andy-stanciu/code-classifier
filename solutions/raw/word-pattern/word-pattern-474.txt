class Solution {

    public boolean wordPattern(String pattern, String s) {
        // convert both strings to a arrays
        var sArray = s.split(" ");
        var patternArray = pattern.split("");
        if (sArray.length != patternArray.length) {
            return false;
        }
        // map string to number
        HashMap<String, Integer> sToNum = new HashMap<String, Integer>();
        HashMap<String, Integer> patternToNum = new HashMap<String, Integer>();
        // set order of words in s
        for (String str : sArray) {
            if (!sToNum.containsKey(str)) {
                sToNum.put(str, sToNum.size());
            }
        }
        // set order of words in pattern
        for (String str : patternArray) {
            if (!patternToNum.containsKey(str)) {
                patternToNum.put(str, patternToNum.size());
            }
        }
        // convert order to int array
        int[] patternSequence = new int[patternArray.length];
        for (int i = 0; i < patternArray.length; i++) {
            patternSequence[i] = patternToNum.get(patternArray[i]);
        }
        // convert order to int array
        int[] sSequence = new int[sArray.length];
        for (int i = 0; i < sArray.length; i++) {
            sSequence[i] = sToNum.get(sArray[i]);
        }
        return Arrays.equals(sSequence, patternSequence);
    }
}
