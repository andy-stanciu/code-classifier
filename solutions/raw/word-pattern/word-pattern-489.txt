class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] ans = s.split(" ");
        int n = ans.length;
        HashMap<Character, String> map = new HashMap<>();
        if (n != pattern.length()) {
            return false;
        }
        for (int i = 0; i < n; i++) {
            char ch = pattern.charAt(i);
            String word = ans[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(word)) {
                    return false;
                }
            } else {
                if (map.containsValue(word)) {
                    return false;
                }
                map.put(ch, word);
            }
        }
        return true;
    }
}
