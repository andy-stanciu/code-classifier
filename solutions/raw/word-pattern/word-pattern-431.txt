class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] words = s.split(" ");
        if (pattern.length() != words.length) {
            return false;
        }
        Map<String, Character> wordToCharMap = new HashMap<>();
        Set<Character> usedChars = new HashSet<>();
        for (int i = 0; i < words.length; i++) {
            Character ch = wordToCharMap.get(words[i]);
            if (ch == null) {
                ch = pattern.charAt(i);
                if (usedChars.contains(ch)) {
                    return false;
                }
                usedChars.add(ch);
                wordToCharMap.put(words[i], ch);
            } else if (ch != pattern.charAt(i)) {
                return false;
            }
        }
        return true;
    }
}
