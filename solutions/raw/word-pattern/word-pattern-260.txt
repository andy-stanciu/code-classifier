class Solution {

    // same as IsomorphicStrings
    public boolean wordPattern(String pattern, String s) {
        Map<Character, Integer> patternIndex = new HashMap<>();
        Map<String, Integer> sIndex = new HashMap<>();
        String[] words = s.split(" ");
        if (pattern.length() != words.length) {
            return false;
        }
        for (int i = 0; i < pattern.length(); i++) {
            int index1 = patternIndex.getOrDefault(pattern.charAt(i), 0);
            int index2 = sIndex.getOrDefault(words[i], 0);
            if (index1 != index2) {
                return false;
            }
            patternIndex.put(pattern.charAt(i), i + 1);
            sIndex.put(words[i], i + 1);
        }
        return true;
    }
}
