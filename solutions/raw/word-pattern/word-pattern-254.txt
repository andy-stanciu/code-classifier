class Solution {

    public boolean wordPattern(String pattern, String s) {
        HashMap<Character, String> pMap = new HashMap<Character, String>();
        HashMap<String, Character> sMap = new HashMap<String, Character>();
        String[] words = s.split(" ");
        char[] chars = pattern.toCharArray();
        if (chars.length != words.length)
            return false;
        for (int i = 0; i < words.length; i++) {
            char ch = chars[i];
            String word = words[i];
            if (pMap.containsKey(ch) && !pMap.get(ch).equals(word))
                return false;
            else if (sMap.containsKey(word) && !sMap.get(word).equals(ch))
                return false;
            else {
                pMap.put(chars[i], words[i]);
                sMap.put(words[i], chars[i]);
            }
        }
        return true;
    }
}
