class Solution {

    public boolean wordPattern(String pattern, String s) {
        String[] sArr = s.split(" ");
        if (pattern.length() != sArr.length)
            return false;
        Map<Character, String> hashPat = new HashMap<>();
        for (int i = 0; i < pattern.length(); i++) {
            if (hashPat.containsKey(pattern.charAt(i))) {
                if (!hashPat.get(pattern.charAt(i)).equals(sArr[i]))
                    return false;
            } else {
                if (hashPat.containsValue(sArr[i]))
                    return false;
            }
            hashPat.put(pattern.charAt(i), sArr[i]);
        }
        return true;
    }
}
