class Solution {

    public int majorityElement(int[] nums) {
        HashMap<Integer, Integer> numberCountMap = new HashMap<>();
        if (nums.length == 1) {
            return nums[0];
        }
        int count = nums.length / 2;
        for (int num : nums) {
            if (numberCountMap.containsKey(num)) {
                int numCount = numberCountMap.get(num);
                if (numCount >= count) {
                    return num;
                }
                numberCountMap.put(num, numCount + 1);
            } else {
                numberCountMap.put(num, 1);
            }
        }
        return -1;
    }
}
