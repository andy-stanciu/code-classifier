class Solution {

    public int majorityElement(int[] nums) {
        int count = 1;
        int n = nums.length;
        int ans = nums[0];
        int i = 1;
        while (i < n) {
            if (nums[i] != ans) {
                count--;
            } else {
                count++;
            }
            if (count == 0) {
                count = 1;
                ans = nums[i];
            }
            i++;
        }
        return ans;
    }
}
