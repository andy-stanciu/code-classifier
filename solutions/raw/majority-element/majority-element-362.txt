class Solution {

    public int majorityElement(int[] arr) {
        //optimal solution using the moors voting
        int n = arr.length;
        //using tep pointer approch
        int count = 0;
        int ele = 0;
        for (int i = 0; i < n; i++) {
            if (count == 0) {
                count++;
                ele = arr[i];
            } else if (ele == arr[i]) {
                count++;
            } else
                count--;
        }
        return ele;
    }
}
