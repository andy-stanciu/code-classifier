class Solution {

    public int majorityElement(int[] nums) {
        int n = nums.length;
        int dupTimes = n / 2;
        int majorityElement = nums[0];
        int count = 0;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == majorityElement) {
                count++;
                if (count > dupTimes) {
                    return majorityElement;
                }
            } else {
                count = 0;
                majorityElement = nums[i];
                count++;
            }
        }
        return majorityElement;
    }
}
