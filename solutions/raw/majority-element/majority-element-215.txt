class Solution {

    public int majorityElement(int[] nums) {
        //using map to store element and its frequency
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int ele : nums) {
            if (map.containsKey(ele)) {
                //if the map already contains the element
                //incrementing the value by 1
                map.put(ele, map.get(ele) + 1);
            } else //(map.get(key) ==> finds the value of the key)
            {
                //by default the element occurs 1 time
                map.put(ele, 1);
            }
        }
        int element = 0;
        for (int ele : nums) {
            if (map.get(ele) > (nums.length / 2)) {
                //Checking if the frequency of the element is greater than n/2
                element = ele;
                break;
            }
        }
        return element;
    }
}
