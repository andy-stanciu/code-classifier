class Solution {

    public int majorityElement(int[] nums) {
        int element = 0;
        int count = 0;
        for (int x : nums) {
            if (count == 0) {
                element = x;
            }
            if (element == x) {
                count++;
            } else {
                count--;
            }
        }
        return element;
    }
}
