class Solution {

    public int majorityElement(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (map.containsKey(nums[i])) {
                // if the map  contains the key, then increase the value by 1
                map.put(nums[i], map.get(nums[i]) + 1);
            } else {
                map.put(nums[i], 1);
            }
            if (map.get(nums[i]) > n / 2) {
                // when the value of any specific key becomes greater than m/2 , then return that key
                return nums[i];
            }
        }
        return -1;
    }
}
