class Solution {

    public int majorityElement(int[] nums) {
        // Step 1: Find the candidate for majority element
        int candidate = nums[0];
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                // Set new candidate
                candidate = num;
            }
            count += (num == candidate) ? 1 : -1;
        }
        // Step 2: Return the candidate (since the problem guarantees a majority element exists)
        return candidate;
    }
}
