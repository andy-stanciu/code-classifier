class Solution {

    public int majorityElement(int[] nums) {
        int count = 0;
        int cand = 0;
        for (int a = 0; a < nums.length; a++) {
            if (count == 0) {
                cand = nums[a];
            }
            if (cand == nums[a]) {
                count++;
            } else {
                count--;
            }
        }
        return (cand);
    }
}
