class Solution {

    public int majorityElement(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int majorityCount = n / 2;
        for (int i = 0; i < n - majorityCount; i++) {
            if (nums[i] == nums[i + majorityCount]) {
                return nums[i];
            }
        }
        return -1;
    }
}
