class Solution {

    public int majorityElement(int[] nums) {
        int major = nums[0];
        int counter = 1;
        for (int i = 1; i < nums.length; i++) {
            if (counter > 0 && major != nums[i]) {
                counter--;
            } else {
                major = nums[i];
                counter++;
            }
        }
        return major;
    }
}
