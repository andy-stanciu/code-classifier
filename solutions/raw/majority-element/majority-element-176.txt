class Solution {

    public int majorityElement(int[] nums) {
        int count = 0;
        int highest = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                highest = nums[i];
                count = 1;
            } else if (nums[i] == highest) {
                count++;
            } else {
                count--;
            }
        }
        return highest;
    }
}
