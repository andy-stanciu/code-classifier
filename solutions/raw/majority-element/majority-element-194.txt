class Solution {

    public int majorityElement(int[] nums) {
        int count = 0;
        int majority_element = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == majority_element) {
                count += 1;
            } else {
                count -= 1;
            }
            if (count == 0) {
                majority_element = nums[i];
                count = 1;
            }
        }
        return majority_element;
    }
}
