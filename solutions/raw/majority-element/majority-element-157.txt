class Solution {

    public int majorityElement(int[] nums) {
        int count = 0;
        int candidate = nums[0];
        for (int x : nums) {
            if (x != candidate) {
                count--;
                if (count == 0) {
                    candidate = x;
                    count = 1;
                }
            } else
                count++;
        }
        return candidate;
    }
}
