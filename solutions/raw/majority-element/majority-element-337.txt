class Solution {

    public int majorityElement(int[] nums) {
        int candidate = -1, count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                candidate = nums[i];
                count = 1;
            } else if (candidate == nums[i]) {
                count++;
            } else {
                count--;
            }
        }
        if (count == nums.length)
            return candidate;
        count = 0;
        for (int i : nums) {
            if (candidate == i) {
                count++;
            }
        }
        if (count > nums.length / 2)
            return candidate;
        else
            return -1;
    }
}
