class Solution {

    public int majorityElement(int[] nums) {
        int potentialCandidate = nums[0];
        int count = 0;
        for (int num : nums) {
            if (count == 0)
                potentialCandidate = num;
            if (num == potentialCandidate)
                count++;
            else
                count--;
        }
        return potentialCandidate;
    }
}
