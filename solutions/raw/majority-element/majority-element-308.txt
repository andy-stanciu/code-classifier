class Solution {

    public int majorityElement(int[] nums) {
        int ans = 0, freq = 0;
        for (int i = 0; i < nums.length; i++) {
            // If frequency counter is zero, update the candidate answer
            // Set current element as potential majority
            if (freq == 0)
                ans = nums[i];
            // If the current element is the same as the candidate, increment frequency
            if (nums[i] == ans)
                freq++;
            else
                // Otherwise, decrement frequency
                freq--;
        }
        return ans;
    }
}
