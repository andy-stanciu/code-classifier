class Solution {

    public int majorityElement(int[] nums) {
        int count = 0, candidate = 0;
        for (int i : nums) {
            if (//selecting a candidate from among the array elements
            count == 0)
                candidate = i;
            if (//checking if the current element is equal to the candidate
            candidate == i)
                //incrementing count
                count++;
            else
                //when canidate &current element is different
                count--;
        }
        return candidate;
    }
}
