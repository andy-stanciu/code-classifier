class Solution {

    public int majorityElement(int[] nums) {
        //Better approach would be using Moore\'s Voting algorithm where space complexity is O(1)
        Map<Integer, Integer> hmap = new HashMap<>();
        for (int p : nums) {
            if (!hmap.containsKey(p)) {
                hmap.put(p, 1);
            } else {
                hmap.put(p, hmap.get(p) + 1);
            }
        }
        int n = (int) Math.floor(nums.length / 2);
        int res = Integer.MIN_VALUE;
        for (int keyFinder : hmap.keySet()) {
            if (hmap.get(keyFinder) > n) {
                res = keyFinder;
            }
        }
        return res;
    }
}
