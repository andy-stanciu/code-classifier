class Solution {

    public int majorityElement(int[] nums) {
        int m = nums[0];
        int count = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (m == nums[i]) {
                count++;
            } else {
                count--;
            }
            if (count == 0) {
                count = 1;
                m = nums[i];
            }
        }
        return m;
    }
}
