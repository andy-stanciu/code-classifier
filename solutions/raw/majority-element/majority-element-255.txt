class Solution {

    //Based on Boyer-Moore Voting Algorithm
    //The idea is using the fact the majority element occurs more than n/2 times
    public int majorityElement(int[] nums) {
        int count = 0;
        int check = 0;
        for (int num : nums) {
            if (count == 0) {
                check = num;
            }
            count += (num == check) ? 1 : -1;
        }
        return check;
    }
}
