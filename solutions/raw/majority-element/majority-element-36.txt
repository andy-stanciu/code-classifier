class Solution {

    public int majorityElement(int[] nums) {
        HashMap<Integer, Integer> countForNumber = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int number = nums[i];
            Integer count = countForNumber.get(number);
            if (count == null) {
                count = Integer.valueOf(1);
            } else {
                count = Integer.valueOf(count.intValue() + 1);
            }
            countForNumber.put(number, count);
        }
        int majority = 0;
        int half = nums.length / 2;
        for (Integer candidate : countForNumber.keySet()) {
            Integer count = countForNumber.get(candidate);
            if (count.intValue() > half) {
                majority = candidate;
                break;
            }
        }
        return majority;
    }
}
