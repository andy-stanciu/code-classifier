class Solution {

    public int majorityElement(int[] nums) {
        int n = nums.length;
        // count
        int cnt = 0;
        // Element
        int el = 0;
        // applying the algorithm:
        for (int i = 0; i < n; i++) {
            if (cnt == 0) {
                cnt = 1;
                // change element
                el = nums[i];
            } else if (el == nums[i]) {
                cnt++;
            } else {
                cnt--;
            }
        }
        // Make el effectively final
        final int finalEl = el;
        // checking if the stored element
        // is the majority element:
        long cnt1 = Arrays.stream(nums).filter(num -> num == finalEl).count();
        if (cnt1 > (n / 2)) {
            return finalEl;
        }
        return -1;
    }
}
