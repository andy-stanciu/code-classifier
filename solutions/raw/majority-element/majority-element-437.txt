class Solution {

    public int majorityElement(int[] nums) {
        int n = nums.length / 2;
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int b : nums) {
            hm.put(b, hm.getOrDefault(b, 0) + 1);
        }
        int major = -1;
        for (int i : hm.keySet()) {
            if (hm.get(i) > n) {
                major = i;
            }
        }
        return major;
    }
}
