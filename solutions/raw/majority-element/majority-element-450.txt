class Solution {

    public int majorityElement(int[] nums) {
        Arrays.sort(nums);
        int n = nums[0], count = 1, len = nums.length;
        for (int i = 1; i < len; i++) {
            if (nums[i] != n) {
                if (count > len / 2)
                    return n;
                else {
                    n = nums[i];
                    count++;
                }
            } else
                count++;
        }
        return n;
    }
}
