class Solution {

    public int majorityElement(int[] nums) {
        Map<Integer, Integer> mp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            mp.put(nums[i], mp.getOrDefault(nums[i], 0) + 1);
        }
        int max = 0;
        int res = 0;
        for (Map.Entry<Integer, Integer> e : mp.entrySet()) {
            if (max < e.getValue()) {
                max = e.getValue();
                res = e.getKey();
            }
        }
        return res;
    }
}
