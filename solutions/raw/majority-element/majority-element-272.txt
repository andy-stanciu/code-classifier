class Solution {

    public int majorityElement(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        Arrays.stream(nums).forEach(v -> map.merge(v, 1, Integer::sum));
        int max = Integer.MIN_VALUE;
        int maxElement = -1;
        for (Integer key : map.keySet()) {
            if (map.get(key) > max) {
                max = map.get(key);
                maxElement = key;
            }
        }
        return maxElement;
    }
}
