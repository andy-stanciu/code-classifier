class Solution {

    public int majorityElement(int[] nums) {
        HashMap<Integer, Integer> mapping = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int value = mapping.getOrDefault(nums[i], 0);
            mapping.put(nums[i], value + 1);
        }
        for (Map.Entry<Integer, Integer> it : mapping.entrySet()) {
            if (it.getValue() > (nums.length / 2)) {
                return it.getKey();
            }
        }
        return -1;
    }
}
