// Moores voting Algorithm part 2 not needed as majority present
class Solution {

    public int majorityElement(int[] nums) {
        int estimated_majority = nums[0];
        int count = 1;
        for (int index = 1; index < nums.length; index++) {
            if (nums[index] != estimated_majority) {
                if (count == 0) {
                    estimated_majority = nums[index];
                    count = 1;
                } else {
                    count--;
                }
            } else {
                count++;
            }
        }
        return estimated_majority;
    }
}
