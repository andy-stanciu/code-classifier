class Solution {

    public int majorityElement(int[] n) {
        int c = 0, r = 0;
        for (int i : n) {
            if (c == 0) {
                r = i;
                c = 1;
            } else if (i == r) {
                c++;
            } else {
                c--;
            }
        }
        return r;
    }
}
