class Solution {

    public int majorityElement(int[] nums) {
        int count = 0, mj = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                count = 1;
                mj = nums[i];
            } else {
                if (mj == nums[i])
                    count++;
                else
                    count--;
            }
        }
        return mj;
    }
}
