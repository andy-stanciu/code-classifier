class Solution {

    public int majorityElement(int[] nums) {
        int n = nums.length;
        int temp = 0;
        for (int i = 0; i < n; i++) {
            int c = 0;
            for (int j = 0; j < n; j++) {
                if (nums[i] == nums[j]) {
                    c++;
                    temp = nums[j];
                }
            }
            if (c > (n / 2)) {
                break;
            }
        }
        return temp;
    }
}
