class Solution {

    public int majorityElement(int[] nums) {
        int e = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == e) {
                count++;
            } else {
                count--;
                if (count == 0)
                    if (i + 1 < nums.length)
                        e = nums[i + 1];
            }
        }
        return e;
    }
}
