class Solution {

    public int majorityElement(int[] nums) {
        TreeMap<Integer, Integer> a = new TreeMap<>();
        for (int i : nums) {
            if (a.containsKey(i)) {
                a.replace(i, a.get(i) + 1);
            } else {
                a.put(i, 1);
            }
        }
        int max = 0;
        int o = 0;
        for (int i : a.keySet()) {
            if (a.get(i) > max) {
                max = a.get(i);
                o = i;
            }
        }
        return o;
    }
}
