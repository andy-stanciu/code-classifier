class Solution {

    public int majorityElement(int[] nums) {
        int len = nums.length;
        Arrays.sort(nums);
        int count = 1;
        int t = 0;
        for (int i = 0; i < len - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                count++;
                if (count > t) {
                    t = i;
                }
            } else {
                count = 0;
            }
        }
        return nums[t];
    }
}
