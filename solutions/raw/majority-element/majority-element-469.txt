class Solution {

    public int majorityElement(int[] nums) {
        System.gc();
        int count = 1;
        int number = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == number) {
                count++;
            } else {
                count--;
                if (count == 0) {
                    count++;
                    number = nums[i];
                }
            }
        }
        return number;
    }
}
