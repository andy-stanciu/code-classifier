class Solution {

    public int majorityElement(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<Integer, Integer>();
        for (int k : nums) {
            map.put(k, map.getOrDefault(k, 0) + 1);
        }
        int ak = nums.length / 2;
        int j = 0;
        int fg = 0;
        for (Map.Entry<Integer, Integer> sa : map.entrySet()) {
            fg = sa.getValue();
            if (fg > ak) {
                j = sa.getKey();
            }
        }
        return j;
    }
}
