class Solution {

    public int majorityElement(int[] nums) {
        Map<Integer, Integer> m = new HashMap<>();
        for (int num : nums) {
            m.put(num, m.getOrDefault(num, 0) + 1);
        }
        int ans = 0;
        int global_freq = 0;
        for (Map.Entry<Integer, Integer> entry : m.entrySet()) {
            int num = entry.getKey();
            int freq = entry.getValue();
            if (freq > global_freq) {
                global_freq = freq;
                ans = num;
            }
        }
        return ans;
    }
}
