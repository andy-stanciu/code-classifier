class Solution {

    public int majorityElement(int[] nums) {
        Map<Integer, Integer> a = new HashMap<>();
        int b = nums.length / 2;
        for (int i : nums) {
            a.put(i, a.getOrDefault(i, 0) + 1);
            if (a.get(i) > b)
                return i;
        }
        return 0;
    }
}
