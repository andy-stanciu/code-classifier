class Solution {

    public int majorityElement(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        int max = Integer.MIN_VALUE;
        for (int n : nums) {
            map.put(n, map.getOrDefault(n, 0) + 1);
        }
        for (Integer i : map.keySet()) {
            if (map.get(i) > nums.length / 2) {
                max = i;
            }
        }
        return max;
    }
}
// TC: O(nlogn), SC: O(n)
