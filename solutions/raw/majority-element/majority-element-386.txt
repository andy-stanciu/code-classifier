class Solution {

    public int majorityElement(int[] nums) {
        int candidate = 0, count = 0;
        // Phase 1: Find a candidate for majority
        for (int num : nums) {
            if (count == 0) {
                candidate = num;
            }
            count += (num == candidate) ? 1 : -1;
        }
        // Phase 2 (optional for strict problems): Verify the candidate is actually the majority
        count = 0;
        for (int num : nums) {
            if (num == candidate) {
                count++;
            }
        }
        if (count > nums.length / 2) {
            return candidate;
        }
        // This line is typically not needed since a majority element is always assumed to exist
        return -1;
    }
}
