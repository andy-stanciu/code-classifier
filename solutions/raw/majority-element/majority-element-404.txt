class Solution {

    public int majorityElement(int[] nums) {
        int n = nums.length;
        Map<Integer, Long> occurences = Arrays.stream(nums).boxed().collect(Collectors.groupingBy(o -> o, Collectors.counting()));
        return occurences.entrySet().stream().filter(entry -> entry.getValue() > n / 2).map(Map.Entry::getKey).findFirst().orElse(-1);
    }
}
