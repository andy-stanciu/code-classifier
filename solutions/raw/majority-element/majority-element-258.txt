class Solution {

    public int majorityElement(int[] nums) {
        // declare hash map to store frequency of nums
        HashMap<Integer, Integer> map = new HashMap<>();
        // calculate majority value
        int maj = nums.length / 2;
        // iterate over nums array
        for (int num : nums) {
            // store frequency of num in hashmap
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        // declare result
        int res = 0;
        // iterate over frequency map
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            // get key
            int key = entry.getKey();
            // get value
            int val = entry.getValue();
            // check if frequency is greater than majority
            if (val > maj) {
                // set current key as result
                res = key;
            }
        }
        // return result
        return res;
    }
}
