class Solution {

    public int majorityElement(int[] nums) {
        int ele = -1;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count != 0) {
                if (nums[i] == ele)
                    count++;
                else
                    count--;
            }
            if (count == 0) {
                ele = nums[i];
                count++;
            }
        }
        return ele;
    }
}
