class Solution {

    public int majorityElement(int[] nums) {
        int x = 0;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            int count = 1;
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[i] == nums[j]) {
                    count++;
                }
            }
            if (count > nums.length / 2) {
                x = nums[i];
            }
        }
        return x;
    }
}
