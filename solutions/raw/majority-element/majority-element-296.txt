class Solution {

    public int majorityElement(int[] nums) {
        int major = nums[0];
        int cnt = 1;
        int n = nums.length;
        for (int i = 1; i < n; i++) {
            if (nums[i] == major) {
                cnt++;
            } else {
                if (cnt == 0) {
                    major = nums[i];
                    cnt = 1;
                } else {
                    cnt--;
                }
            }
        }
        return major;
    }
}
