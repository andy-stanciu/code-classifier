class Solution {

    public int majorityElement(int[] nums) {
        int ans = 0;
        int element = 0;
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                element = num;
                count++;
            } else if (element == num) {
                count++;
            } else {
                count--;
            }
        }
        return element;
    }
}
