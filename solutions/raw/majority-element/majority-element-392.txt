class Solution {

    public int majorityElement(int[] nums) {
        int majority = nums[0];
        int voter = 1;
        for (int i = 1; i < nums.length; i++) {
            if (voter == 0) {
                majority = nums[i];
                voter++;
            } else if (nums[i] == majority) {
                voter++;
            } else {
                voter--;
            }
        }
        return majority;
    }
}
