class Solution {

    public int majorityElement(int[] nums) {
        int cnt = 0;
        Integer element = null;
        for (int num : nums) {
            if (cnt == 0) {
                element = num;
            }
            if (num == element) {
                cnt++;
            } else {
                cnt--;
            }
        }
        return element;
    }
}
