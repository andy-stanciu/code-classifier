class Solution {

    public int majorityElement(int[] a) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < a.length; i++) {
            if (map.containsKey(a[i])) {
                int value = map.get(a[i]) + 1;
                map.put(a[i], value);
            } else {
                map.put(a[i], 1);
            }
            if (map.get(a[i]) > a.length / 2)
                return a[i];
        }
        return -1;
    }
}
