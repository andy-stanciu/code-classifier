class Solution {

    public int majorityElement(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int val : nums) {
            map.put(val, map.getOrDefault(val, 0) + 1);
        }
        for (int val : map.keySet()) {
            int freq = map.get(val);
            if (freq > n / 2) {
                return val;
            }
        }
        return -1;
    }
}
