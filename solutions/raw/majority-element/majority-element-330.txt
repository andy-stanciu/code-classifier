class Solution {

    public int majorityElement(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            int count = map.containsKey(num) ? map.get(num) : 0;
            map.put(num, count + 1);
        }
        for (var item : map.keySet()) {
            if (map.get(item) > (nums.length / 2))
                return item;
        }
        return 0;
    }
}
