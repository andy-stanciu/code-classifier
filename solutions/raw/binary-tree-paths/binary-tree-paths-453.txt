class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> res = new ArrayList<>();
        binaryPath(root, new ArrayList<>(), res);
        return res;
    }

    public void binaryPath(TreeNode root, List<String> str, List<String> res) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            str.add(root.val + "");
            res.add(String.join("->", str));
            str.remove(str.size() - 1);
            return;
        }
        str.add(root.val + "");
        binaryPath(root.left, str, res);
        binaryPath(root.right, str, res);
        str.remove(str.size() - 1);
    }
}
