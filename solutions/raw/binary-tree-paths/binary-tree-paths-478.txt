class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ls = new ArrayList<>();
        treepaths(root, "", ls);
        return ls;
    }

    public void treepaths(TreeNode root, String s, List<String> ls) {
        if (root == null) {
            return;
        }
        s += root.val;
        if (root.right == null && root.left == null) {
            ls.add(s);
        }
        treepaths(root.left, s + "->", ls);
        treepaths(root.right, s + "->", ls);
    }
}
