/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> res = new ArrayList<>();
        String s = "";
        findPaths(root, res, s);
        return res;
    }

    public void findPaths(TreeNode node, List<String> paths, String s) {
        if (node == null) {
            return;
        }
        if (node.left == null && node.right == null) {
            paths.add(s + String.valueOf(node.val));
            s = "";
        }
        s = s + String.valueOf(node.val) + "->";
        findPaths(node.left, paths, s);
        findPaths(node.right, paths, s);
    }
}
