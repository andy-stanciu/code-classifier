/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> res = new ArrayList<>();
        String ls = "";
        binaryTreePathsSol(root, res, ls);
        return res;
    }

    public void binaryTreePathsSol(TreeNode root, List<String> res, String ls) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            ls = ls + Integer.toString(root.val);
            res.add(ls);
            // if (ls.length() > 2)
            ls = ls.substring(0, ls.length());
        }
        // ls = ls + Integer.toString(root.val);
        binaryTreePathsSol(root.left, res, ls + root.val + "->");
        // if (ls.length() > 2)
        // ls = ls.substring(0, ls.length()-1);
        // ls = ls + Integer.toString(root.val);
        binaryTreePathsSol(root.right, res, ls + root.val + "->");
        // if (ls.length() > 2)
        // ls = ls.substring(0, ls.length()-1);
    }
}
