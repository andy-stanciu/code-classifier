class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ans = new ArrayList<>();
        dfs(root, "", ans);
        return ans;
    }

    public void dfs(TreeNode root, String pathTillNow, List<String> ans) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            pathTillNow += root.val;
            ans.add(pathTillNow);
            return;
        }
        pathTillNow += root.val + "->";
        dfs(root.left, pathTillNow, ans);
        dfs(root.right, pathTillNow, ans);
    }
}
