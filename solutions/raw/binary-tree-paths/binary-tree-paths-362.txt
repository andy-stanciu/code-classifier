/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> arr = new ArrayList<String>();
        tree(root, arr, "");
        return arr;
    }

    public void tree(TreeNode root, List<String> arr, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            arr.add(s + root.val);
        }
        tree(root.left, arr, s + root.val + "->");
        tree(root.right, arr, s + root.val + "->");
    }
}
