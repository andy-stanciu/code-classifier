/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    ArrayList<String> li = new ArrayList<>();

    public List<String> binaryTreePaths(TreeNode r) {
        String q = "";
        if (r == null)
            return li;
        else {
            path(r, q);
        }
        return li;
    }

    TreeNode path(TreeNode r, String q) {
        if (r != null) {
            if (r.left == null && r.right == null) {
                q += r.val;
                li.add(q);
            } else {
                q += r.val + "->";
                path(r.left, q);
                path(r.right, q);
            }
        }
        return null;
    }
}
