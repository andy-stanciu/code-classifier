class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> path = new ArrayList<String>();
        // Base case: empty tree
        if (root == null) {
            return path;
        }
        // Base case: leaf node
        if (root.left == null && root.right == null) {
            return Arrays.asList(Integer.toString(root.val));
        }
        // General cases:
        for (String leftPath : binaryTreePaths(root.left)) {
            path.add(Integer.toString(root.val) + "->" + leftPath);
        }
        for (String rightPath : binaryTreePaths(root.right)) {
            path.add(Integer.toString(root.val) + "->" + rightPath);
        }
        return path;
    }
}
