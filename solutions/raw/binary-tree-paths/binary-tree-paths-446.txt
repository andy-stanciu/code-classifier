/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList();
        if (root == null)
            return paths;
        traversePaths(root, "", paths);
        return paths;
    }

    public void traversePaths(TreeNode root, String currentPath, List<String> paths) {
        currentPath += root.val;
        if (root.left == null && root.right == null) {
            paths.add(currentPath);
            return;
        }
        if (root.left != null) {
            traversePaths(root.left, currentPath + "->", paths);
        }
        if (root.right != null) {
            traversePaths(root.right, currentPath + "->", paths);
        }
    }
}
