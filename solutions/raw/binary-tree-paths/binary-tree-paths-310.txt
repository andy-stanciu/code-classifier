class TreeNode {

    int val;

    TreeNode left;

    TreeNode right;

    TreeNode(int x) {
        val = x;
    }
}

public class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        constructPaths(root, "", res);
        return res;
    }

    private void constructPaths(TreeNode node, String pathSoFar, List<String> res) {
        if (node == null) {
            return;
        }
        if (pathSoFar.isEmpty()) {
            pathSoFar += node.val;
        } else {
            pathSoFar += "->" + node.val;
        }
        if (node.left == null && node.right == null) {
            res.add(pathSoFar);
        } else {
            constructPaths(node.left, pathSoFar, res);
            constructPaths(node.right, pathSoFar, res);
        }
    }
}
