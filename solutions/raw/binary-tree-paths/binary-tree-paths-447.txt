/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        dfs(result, "", root);
        return result;
    }

    public void dfs(List<String> result, String temp, TreeNode node) {
        if (node == null)
            return;
        if (node.left == null && node.right == null) {
            result.add(temp + node.val);
            return;
        }
        temp = temp + node.val + "->";
        dfs(result, temp, node.left);
        dfs(result, temp, node.right);
    }
}
