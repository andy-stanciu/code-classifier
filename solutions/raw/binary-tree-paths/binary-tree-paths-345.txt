/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> list = new ArrayList<>();
        collectBinaryTreePath(root, list, new StringBuilder());
        return list;
    }

    public void collectBinaryTreePath(TreeNode node, List<String> list, StringBuilder builder) {
        if (node == null)
            return;
        builder.append(node.val);
        if (node.left == null && node.right == null) {
            String path = builder.toString();
            list.add(path);
            return;
        }
        builder.append("->");
        collectBinaryTreePath(node.left, list, new StringBuilder(builder));
        collectBinaryTreePath(node.right, list, new StringBuilder(builder));
    }
}
