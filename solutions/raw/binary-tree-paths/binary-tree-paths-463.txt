class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> treePaths = new ArrayList<>();
        if (root == null) {
            return treePaths;
        }
        binaryTreePathUtil(root, "", treePaths);
        return treePaths;
    }

    public void binaryTreePathUtil(TreeNode root, String currPath, List<String> treePaths) {
        if (root == null) {
            return;
        }
        currPath = currPath + root.val;
        if (root.left == null && root.right == null) {
            treePaths.add(currPath);
            return;
        }
        currPath = currPath + "->";
        binaryTreePathUtil(root.left, currPath, treePaths);
        binaryTreePathUtil(root.right, currPath, treePaths);
    }
}
