/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void solve(TreeNode root, List<String> list, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            String s = Integer.toString(root.val);
            str += s;
            list.add(str);
            return;
        }
        String s = Integer.toString(root.val);
        str += s;
        str += "->";
        solve(root.left, list, str);
        solve(root.right, list, str);
    }

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> list = new ArrayList<>();
        solve(root, list, "");
        return list;
    }
}
