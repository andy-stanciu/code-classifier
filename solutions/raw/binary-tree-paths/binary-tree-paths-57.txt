/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        ArrayList<String> res = new ArrayList<String>();
        if (root != null) {
            solvedfs(root, "", res);
        }
        return res;
    }

    public void solvedfs(TreeNode root, String way, List<String> res) {
        if (root.left == null && root.right == null)
            res.add(way + root.val);
        if (root.left != null)
            solvedfs(root.left, way + root.val + "->", res);
        if (root.right != null)
            solvedfs(root.right, way + root.val + "->", res);
    }
}
