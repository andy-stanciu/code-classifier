class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        if (root == null)
            return result;
        pathToLeaf(root, "", result);
        return result;
    }

    void pathToLeaf(TreeNode node, String str, List<String> result) {
        if (node == null)
            return;
        str = str + node.val;
        if (node.left == null && node.right == null) {
            result.add(str);
            return;
        }
        str = str + "->";
        pathToLeaf(node.left, str, result);
        pathToLeaf(node.right, str, result);
    }
}
