class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        path(root, "", ans);
        return ans;
    }

    private void path(TreeNode node, String s, List<String> ans) {
        if (node == null) {
            return;
        }
        // If the node is a leaf node, add the path to the result list
        if (node.left == null && node.right == null) {
            ans.add(s + node.val);
            return;
        }
        // If the node is not a leaf, continue traversing
        path(node.left, s + node.val + "->", ans);
        path(node.right, s + node.val + "->", ans);
    }
}
