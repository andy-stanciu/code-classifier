class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> list = new ArrayList<>();
        recursion(root, new StringBuilder(), list);
        return list;
    }

    private void recursion(TreeNode root, StringBuilder str, List<String> list) {
        str.append(root.val);
        if (root.left == null && root.right == null) {
            list.add(str.toString());
            return;
        }
        str.append("->");
        if (root.left != null)
            recursion(root.left, new StringBuilder().append(str), list);
        if (root.right != null)
            recursion(root.right, new StringBuilder().append(str), list);
    }
}
