/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> path = new ArrayList<>();
        showPath(root, path, "");
        return path;
    }

    public void showPath(TreeNode root, List<String> path, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null)
            path.add(str + root.val);
        showPath(root.left, path, str + root.val + "->");
        showPath(root.right, path, str + root.val + "->");
    }
}
