class Solution {

    List<String> paths = new ArrayList<>();

    public List<String> binaryTreePaths(TreeNode root) {
        if (root == null)
            return paths;
        allPaths(root, new StringBuilder());
        return paths;
    }

    private void allPaths(TreeNode node, StringBuilder path) {
        if (node == null)
            return;
        path.append(node.val);
        if (node.left == null && node.right == null)
            paths.add(path.toString());
        else {
            path.append("->");
            String currNodePath = path.toString();
            allPaths(node.left, path);
            allPaths(node.right, new StringBuilder(currNodePath));
        }
    }
}
