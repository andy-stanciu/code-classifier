class Solution {

    public void solve(List<String> str, String x, TreeNode root) {
        if (root.right == null && root.left == null) {
            x += Integer.toString(root.val);
            str.add(x);
            return;
        }
        x += Integer.toString(root.val);
        x += "->";
        if (root.left != null) {
            solve(str, x, root.left);
        }
        if (root.right != null) {
            solve(str, x, root.right);
        }
        return;
    }

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> str = new ArrayList<>();
        String x = "";
        if (root == null)
            return str;
        solve(str, x, root);
        return str;
    }
}
