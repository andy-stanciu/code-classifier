class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> binaryTreePaths = new ArrayList<>();
        if (root != null) {
            traverse(binaryTreePaths, root, "");
        }
        return binaryTreePaths;
    }

    private void traverse(List<String> binaryTreePaths, TreeNode root, String str) {
        if (root.left == null && root.right == null) {
            binaryTreePaths.add(str + root.val);
        }
        if (root.left != null) {
            traverse(binaryTreePaths, root.left, str + root.val + "->");
        }
        if (root.right != null) {
            traverse(binaryTreePaths, root.right, str + root.val + "->");
        }
    }
}
