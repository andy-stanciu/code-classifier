class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        ArrayList<String> al = new ArrayList<String>();
        comb(root, al, "");
        return al;
    }

    private void comb(TreeNode root, ArrayList<String> al, String str) {
        if (root.left == null && root.right == null)
            al.add(str + root.val);
        if (root.left != null)
            comb(root.left, al, str + root.val + "->");
        if (root.right != null)
            comb(root.right, al, str + root.val + "->");
    }
}
