/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        binaryTreePathHelper(root, result, "");
        return result;
    }

    public void binaryTreePathHelper(TreeNode root, List<String> result, String stack) {
        if (root == null) {
            return;
        }
        stack += root.val;
        if (root.left == null && root.right == null) {
            result.add(new String(stack));
            return;
        }
        stack += "->";
        binaryTreePathHelper(root.left, result, stack);
        binaryTreePathHelper(root.right, result, stack);
    }
}
