class Solution {

    ArrayList<String> path = new ArrayList<String>();

    public List<String> binaryTreePaths(TreeNode root) {
        dfs(root, "");
        return path;
    }

    void dfs(TreeNode root, String p) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            p += "->" + Integer.toString(root.val);
            path.add(p.substring(2));
        } else {
            dfs(root.right, p + "->" + Integer.toString(root.val));
            dfs(root.left, p + "->" + Integer.toString(root.val));
        }
    }
}
