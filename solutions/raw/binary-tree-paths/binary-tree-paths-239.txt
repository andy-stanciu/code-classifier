/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        String s = "";
        List<String> result = new ArrayList<>();
        return getPaths(root, result, s);
    }

    public static List<String> getPaths(TreeNode root, List<String> result, String s) {
        if (root == null)
            return result;
        if (root.left == null && root.right == null) {
            result.add(s + root.val);
            return result;
        }
        s = s + root.val + "->";
        getPaths(root.left, result, s);
        getPaths(root.right, result, s);
        return result;
    }
}
