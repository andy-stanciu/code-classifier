/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        String path = "";
        treeTraversal(result, path, root);
        return result;
    }

    public void treeTraversal(List<String> result, String path, TreeNode root) {
        if (root.left == null && root.right == null) {
            path += root.val;
            result.add(path);
        }
        if (root.left != null) {
            treeTraversal(result, path + root.val + "->", root.left);
        }
        if (root.right != null) {
            treeTraversal(result, path + root.val + "->", root.right);
        }
    }
}
