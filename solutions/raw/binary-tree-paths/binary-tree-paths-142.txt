/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> l = new ArrayList<>();
        sol(root, "", l);
        return l;
    }

    public void sol(TreeNode root, String s, List<String> l) {
        if (root == null)
            return;
        if (s.length() == 0)
            s = "" + root.val;
        else
            s = s + "->" + root.val;
        if (root.left == null && root.right == null) {
            l.add(s);
            return;
        }
        sol(root.left, s, l);
        sol(root.right, s, l);
    }
}
