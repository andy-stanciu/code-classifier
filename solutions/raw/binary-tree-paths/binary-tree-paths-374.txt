/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> list = new ArrayList<>();
        backtracking(list, root, new StringBuilder());
        return list;
    }

    public void backtracking(List<String> list, TreeNode root, StringBuilder sb) {
        if (root == null)
            return;
        int length = sb.length();
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            list.add(sb.toString());
        }
        sb.append("->");
        if (root.left != null)
            backtracking(list, root.left, sb);
        if (root.right != null)
            backtracking(list, root.right, sb);
        sb.setLength(length);
    }
}
