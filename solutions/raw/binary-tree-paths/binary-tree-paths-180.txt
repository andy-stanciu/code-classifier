/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void pathofBinaryTree(TreeNode root, String str, List<String> list) {
        if (root == null) {
            return;
        }
        str += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            list.add(str);
            return;
        }
        str += "->";
        pathofBinaryTree(root.left, str, list);
        pathofBinaryTree(root.right, str, list);
    }

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ans = new ArrayList<>();
        if (root.left == null && root.right == null) {
            ans.add(Integer.toString(root.val));
            return ans;
        }
        pathofBinaryTree(root, "", ans);
        return ans;
    }
}
