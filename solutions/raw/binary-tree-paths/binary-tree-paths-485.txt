class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> li = new ArrayList<>();
        String s = "";
        me(root, s, li);
        return li;
    }

    public void me(TreeNode root, String s, List<String> li) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            s = s + root.val;
            li.add(s);
            return;
        }
        String output = s + root.val + "->";
        me(root.left, output, li);
        me(root.right, output, li);
    }
}
