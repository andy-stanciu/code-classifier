class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        String str = root.val + "";
        List<String> list = new ArrayList<>();
        if (root.left == null && root.right == null) {
            list.add(str);
            return list;
        }
        collectAll(root.left, str, list);
        collectAll(root.right, str, list);
        return list;
    }

    public static void collectAll(TreeNode root, String str, List<String> list) {
        if (root == null)
            return;
        str += "->" + root.val;
        if (root.left == null && root.right == null) {
            list.add(str);
            return;
        }
        collectAll(root.left, str, list);
        collectAll(root.right, str, list);
    }
}
