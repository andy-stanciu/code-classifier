/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePathsHelper(TreeNode root, String s, List<String> lst) {
        if (root.left == null && root.right == null) {
            s = s + "" + root.val;
            lst.add(s);
        }
        if (root.left != null) {
            binaryTreePathsHelper(root.left, s + "" + root.val + "->", lst);
        }
        if (root.right != null) {
            binaryTreePathsHelper(root.right, s + "" + root.val + "->", lst);
        }
        return lst;
    }

    public List<String> binaryTreePaths(TreeNode root) {
        return binaryTreePathsHelper(root, "", new LinkedList<>());
    }
}
