class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList<>();
        if (root != null) {
            printPaths(root, "", paths);
        }
        return paths;
    }

    public void printPaths(TreeNode root, String ans, List<String> paths) {
        // Append the current node\'s value to the current path
        ans += root.val;
        if (root.left == null && root.right == null) {
            // If leaf node, add the current path to paths
            paths.add(ans);
            return;
        }
        if (root.left != null) {
            // Recursively call for left child
            printPaths(root.left, ans + "->", paths);
        }
        if (root.right != null) {
            // Recursively call for right child
            printPaths(root.right, ans + "->", paths);
        }
    }
}
