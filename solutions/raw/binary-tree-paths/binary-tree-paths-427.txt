/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        List<String> pathString = new ArrayList<>();
        listPaths(root, result, pathString);
        return result;
    }

    private void listPaths(TreeNode root, List<String> result, List<String> pathString) {
        if (root == null) {
            return;
        }
        // add current nodes value to the pathlist string
        pathString.add(Integer.toString(root.val));
        if (root.left == null && root.right == null) {
            result.add(String.join("->", pathString));
        } else {
            listPaths(root.left, result, pathString);
            listPaths(root.right, result, pathString);
        }
        // backtracking, remove each val from string
        pathString.remove(pathString.size() - 1);
    }
}
