/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        btpUtil(root, result, "");
        return result;
    }

    public void btpUtil(TreeNode root, List<String> result, String path) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            path = path == "" ? root.val + "" : path + "->" + root.val;
            result.add(path);
        }
        path = path == "" ? root.val + "" : path + "->" + root.val;
        btpUtil(root.left, result, path);
        btpUtil(root.right, result, path);
    }
}
