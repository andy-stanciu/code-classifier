/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        binaryTreePathsH(root, "", list);
        return list;
    }

    public void binaryTreePathsH(TreeNode n, String currNVal, List<String> list) {
        currNVal += n.val;
        if (n.left == null && n.right == null) {
            // base case
            list.add(currNVal);
            return;
        }
        if (n.left != null) {
            binaryTreePathsH(n.left, currNVal + "->", list);
        }
        if (n.right != null) {
            binaryTreePathsH(n.right, currNVal + "->", list);
        }
    }
}
