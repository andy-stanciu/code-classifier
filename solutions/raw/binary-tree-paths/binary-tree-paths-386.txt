/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        String s = "";
        List<String> l = new ArrayList<>();
        return returnPaths(root, l, s);
    }

    private List<String> returnPaths(TreeNode root, List<String> l, String s) {
        if (root == null)
            return l;
        if (root.left == null && root.right == null) {
            l.add(s + root.val);
        }
        s = s + root.val + "->";
        returnPaths(root.left, l, s);
        returnPaths(root.right, l, s);
        return l;
    }
}
