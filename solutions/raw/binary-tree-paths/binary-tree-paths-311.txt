class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList<>();
        if (root != null) {
            dfs(paths, "", root);
        }
        return paths;
    }

    public void dfs(List<String> paths, String path, TreeNode node) {
        if (node.right == null && node.left == null) {
            paths.add(path + node.val);
        }
        if (node.left != null) {
            dfs(paths, path + node.val + "->", node.left);
        }
        if (node.right != null) {
            dfs(paths, path + node.val + "->", node.right);
        }
    }
}
