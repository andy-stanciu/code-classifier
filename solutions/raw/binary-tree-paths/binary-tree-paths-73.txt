/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        if (root == null) {
            return null;
        }
        ArrayList<String> a = new ArrayList<String>();
        bin(root, "", a);
        return a;
    }

    public void bin(TreeNode root, String curr, List<String> paths) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            paths.add(curr + root.val);
            return;
        }
        bin(root.left, curr + root.val + "->", paths);
        bin(root.right, curr + root.val + "->", paths);
    }
}
