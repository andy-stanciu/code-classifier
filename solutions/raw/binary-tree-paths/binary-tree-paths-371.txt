class Solution {

    List<String> res = new ArrayList();

    public List<String> binaryTreePaths(TreeNode root) {
        paths(root, "");
        return res;
    }

    private void paths(TreeNode root, String str) {
        if (root == null)
            return;
        if (isLeaf(root))
            res.add(str.concat(String.valueOf(root.val)));
        paths(root.left, str.concat(String.valueOf(root.val) + "->"));
        paths(root.right, str.concat(String.valueOf(root.val) + "->"));
    }

    private boolean isLeaf(TreeNode root) {
        return root.left == null && root.right == null;
    }
}
