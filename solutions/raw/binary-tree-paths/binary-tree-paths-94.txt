class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList<>();
        String path = "";
        calpath(root, path, paths);
        return paths;
    }

    public static void calpath(TreeNode root, String path, List<String> paths) {
        if (root == null) {
            return;
        }
        path += root.val;
        if (root.left == null && root.right == null) {
            paths.add(path);
        }
        path += "->";
        calpath(root.left, path, paths);
        calpath(root.right, path, paths);
    }
}
