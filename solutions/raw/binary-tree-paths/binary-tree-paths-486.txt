class Solution {

    private static final String arrow = "->";

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList<>();
        StringBuilder sb = new StringBuilder();
        return topDown(root, paths, sb);
    }

    private List<String> topDown(TreeNode root, List<String> paths, StringBuilder sb) {
        // recursive helper function
        // error checking for empty trees
        if (root == null)
            return paths;
        // regardless of the type of node, append the value
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            // leaf node, so add to List and return
            paths.add(sb.toString());
            return paths;
        }
        // new StringBuilder is used to prevent unwanted mutations
        // not a leaf node, so append arrow
        if (root.left != null)
            topDown(root.left, paths, new StringBuilder(sb).append(this.arrow));
        // not a leaf node, so append arrow
        if (root.right != null)
            topDown(root.right, paths, new StringBuilder(sb).append(this.arrow));
        return paths;
    }
}
