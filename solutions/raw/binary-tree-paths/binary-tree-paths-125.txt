class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> out = new ArrayList<>();
        helper(root, out, new String(""));
        return out;
    }

    public void helper(TreeNode root, List<String> out, String curr) {
        if (null == root)
            return;
        if (curr.length() == 0)
            curr = "" + root.val;
        else
            curr = curr + "->" + root.val;
        if (null == root.left && null == root.right) {
            out.add(curr);
            return;
        }
        helper(root.left, out, curr);
        helper(root.right, out, curr);
    }
}
