/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        ArrayList<String> al = new ArrayList<>();
        paths(root, al, "");
        return al;
    }

    public void paths(TreeNode root, List<String> al, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            str += root.val;
            al.add(str);
            return;
        }
        paths(root.left, al, str + root.val + "->");
        paths(root.right, al, str + root.val + "->");
        return;
    }
}
