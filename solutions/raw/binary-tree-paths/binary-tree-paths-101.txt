/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> binaryTreePaths(TreeNode root) {
        if (root == null)
            return ans;
        find(root, "");
        return ans;
    }

    void find(TreeNode root, String path) {
        if (root == null)
            return;
        // Append current node to the path
        if (path.length() == 0) {
            path += root.val;
        } else {
            path += "->" + root.val;
        }
        // If it\'s a leaf node, add the path to the answer list
        if (root.left == null && root.right == null) {
            ans.add(path);
            return;
        }
        // Continue the path with the left and right children
        find(root.left, path);
        find(root.right, path);
    }
}
