/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<String> list = new ArrayList<>();

    public List<String> binaryTreePaths(TreeNode root) {
        if (root == null) {
            return list;
        }
        sec(root, "", list);
        return list;
    }

    public void sec(TreeNode root, String s, List<String> list) {
        if (root.left == null && root.right == null) {
            list.add(s + root.val);
        }
        if (root.left != null) {
            sec(root.left, s + root.val + "->", list);
        }
        if (root.right != null) {
            sec(root.right, s + root.val + "->", list);
        }
    }
}
