/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ans = new ArrayList<>();
        paths(root, ans, "");
        return ans;
    }

    public void paths(TreeNode root, List<String> ans, String paths) {
        if (root == null) {
            return;
        }
        if (paths.isEmpty()) {
            paths = root.val + "";
        } else {
            paths = paths + "->" + root.val;
        }
        if (root.left == null && root.right == null) {
            ans.add(paths);
            return;
        }
        paths(root.left, ans, paths);
        paths(root.right, ans, paths);
    }
}
