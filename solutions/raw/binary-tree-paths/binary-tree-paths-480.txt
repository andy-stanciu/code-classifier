class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        String s = "";
        List l = new ArrayList<>();
        return solve(root, l, s);
    }

    public List<String> solve(TreeNode root, List l, String s) {
        if (root == null)
            return l;
        if (root.left == null && root.right == null) {
            l.add(s + root.val);
            return l;
        }
        s = s + root.val + "->";
        solve(root.left, l, s);
        solve(root.right, l, s);
        return l;
    }
}
