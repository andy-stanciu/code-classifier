/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> result = new ArrayList<>();
        if (root == null)
            return result;
        // to plug in the node values that we traversed
        List<Integer> paths = new ArrayList<>();
        traversal(root, result, paths);
        return result;
    }

    void traversal(TreeNode root, List<String> result, List<Integer> paths) {
        //middle
        paths.add(root.val);
        if (root.left == null && root.right == null) {
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < paths.size() - 1; i++) {
                sb.append(paths.get(i)).append("->");
            }
            sb.append(paths.get(paths.size() - 1));
            result.add(sb.toString());
            return;
        }
        if (root.left != null) {
            traversal(root.left, result, paths);
            paths.remove(paths.size() - 1);
        }
        if (root.right != null) {
            traversal(root.right, result, paths);
            paths.remove(paths.size() - 1);
        }
    }
}
