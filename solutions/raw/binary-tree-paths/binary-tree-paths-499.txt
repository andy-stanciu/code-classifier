class Solution {

    public List<String> binaryTreePaths(TreeNode root) {
        List<String> treePaths = new ArrayList<>();
        if (root == null)
            return treePaths;
        getBinaryTreePaths(root, "", treePaths);
        return treePaths;
    }

    public void getBinaryTreePaths(TreeNode root, String currPath, List<String> treePaths) {
        if (root == null)
            return;
        currPath = currPath + root.val;
        if (root.left == null && root.right == null) {
            treePaths.add(currPath);
        }
        currPath = currPath + "->";
        getBinaryTreePaths(root.left, currPath, treePaths);
        getBinaryTreePaths(root.right, currPath, treePaths);
    }
}
