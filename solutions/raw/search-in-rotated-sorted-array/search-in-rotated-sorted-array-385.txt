class Solution {

    public int search(int[] nums, int target) {
        // Handle edge cases
        if (nums.length == 1) {
            return nums[0] == target ? 0 : -1;
        }
        // Find the pivot point
        int pivot_point = pivot_find(nums);
        if (pivot_point == 0) {
            // Array is not rotated
            return binary_search(nums, 0, nums.length - 1, target);
        }
        if (nums[0] <= target && target <= nums[pivot_point - 1]) {
            // Target is in the left half
            return binary_search(nums, 0, pivot_point - 1, target);
        } else {
            // Target is in the right half
            return binary_search(nums, pivot_point, nums.length - 1, target);
        }
    }

    // Finding the pivot
    public int pivot_find(int[] nums) {
        int ans = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                return i + 1;
            }
        }
        // Array is not rotated
        return 0;
    }

    // Binary search
    public int binary_search(int[] array, int start, int end, int target) {
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (array[mid] == target) {
                return mid;
            } else if (array[mid] > target) {
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return -1;
    }
}
