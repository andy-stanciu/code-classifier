class Solution {

    public int search(int[] nums, int target) {
        int count = 0;
        int ans = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == target) {
                ans = target;
                break;
            }
            count++;
        }
        if (ans == target) {
            return count;
        }
        return -1;
    }
}
