class Solution {

    public int search(int[] arr, int target) {
        int start = 0;
        int end = arr.length - 1;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (arr[mid] == target)
                return mid;
            // 1st case: first half of array is sorted
            if (arr[mid] >= arr[start]) {
                // target lies in the range, search in first half
                if (target >= arr[start] && target <= arr[mid]) {
                    end = mid - 1;
                } else // search in other half
                {
                    start = mid + 1;
                }
                // 2nd case: second half of array is sorted
            } else {
                // target lies in the range, search in second half
                if (target >= arr[mid] && target <= arr[end]) {
                    start = mid + 1;
                    // search in first half
                } else {
                    end = mid - 1;
                }
            }
        }
        return -1;
    }
}
