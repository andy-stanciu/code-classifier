class Solution {

    public int search(int[] nums, int target) {
        return bin(nums, target, 0, nums.length - 1);
    }

    public int bin(int[] nums, int target, int si, int ei) {
        if (si > ei) {
            return -1;
        }
        int mid = si + (ei - si) / 2;
        if (nums[mid] == target) {
            return mid;
        }
        //target on L1
        if (nums[si] <= nums[mid]) {
            //case a, target on left
            if (nums[si] <= target && target < nums[mid]) {
                return bin(nums, target, si, mid - 1);
            } else {
                return bin(nums, target, mid + 1, ei);
            }
        } else //target on L2
        {
            //case a, target on right
            if (nums[mid] <= target && target <= nums[ei]) {
                return bin(nums, target, mid + 1, ei);
            } else {
                return bin(nums, target, si, mid - 1);
            }
        }
    }
}
