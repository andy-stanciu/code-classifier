class Solution {

    public int search(int[] nums, int target) {
        //Binary search kind of
        int low = 0;
        int high = nums.length - 1;
        while (low <= high) {
            int mid = (low + high + 1) / 2;
            if (//found
            target == nums[mid])
                return mid;
            else if (//left side is sorted
            nums[mid] > nums[low]) {
                if (//target in the left side
                target >= nums[low] && target < nums[mid])
                    high = mid - 1;
                else
                    low = mid + 1;
            } else //right side is sorted
            {
                if (//target in the right side
                target > nums[mid] && target <= nums[high])
                    low = mid + 1;
                else
                    high = mid - 1;
            }
        }
        return -1;
    }
}
