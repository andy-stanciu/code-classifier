class Solution {

    public int search(int[] arr, int data) {
        int n = arr.length, si = 0, ei = n - 1;
        while (si <= ei) {
            int mid = (si + ei) / 2;
            if (arr[mid] == data)
                return mid;
            else if (arr[si] <= arr[mid]) {
                if (arr[si] <= data && data < arr[mid])
                    ei = mid - 1;
                else
                    si = mid + 1;
            } else {
                if (arr[mid] < data && data <= arr[ei])
                    si = mid + 1;
                else
                    ei = mid - 1;
            }
        }
        return -1;
    }
}
