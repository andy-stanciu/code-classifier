class Solution {

    public int search(int[] nums, int target) {
        int left = 0;
        int right = nums.length - 1;
        int pivot = 0;
        // Find the pivot, which is the smallest element in the rotated array
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] > nums[right]) {
                // Pivot is in the right half
                left = mid + 1;
            } else {
                // Pivot could be mid
                right = mid;
            }
        }
        pivot = left;
        // Perform binary search in the two sorted halves
        int result = doBinarySearch(0, pivot - 1, nums, target);
        if (result != -1) {
            return result;
        }
        return doBinarySearch(pivot, nums.length - 1, nums, target);
    }

    int doBinarySearch(int left, int right, int[] nums, int target) {
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                return mid;
            }
            if (target > nums[mid]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return -1;
    }
}
