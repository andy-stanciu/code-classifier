class Solution {

    public int search(int[] nums, int target) {
        int n = nums.length;
        if (n == 1) {
            if (nums[0] == target)
                return 0;
            else
                return -1;
        }
        if (nums[0] < nums[n - 1])
            return search(nums, 0, n - 1, target);
        int min = findmin(nums, 0, n - 1);
        //System.out.println("min" + min);
        int a = search(nums, 0, min - 1, target);
        int b = search(nums, min, n - 1, target);
        return a == -1 ? b : a;
    }

    int findmin(int[] nums, int l, int r) {
        int n = nums.length;
        while (l <= r) {
            int m = l + (r - l) / 2;
            int prev = nums[(m - 1 + n) % n];
            int next = nums[(m + 1) % n];
            if (prev > nums[m] && next > nums[m])
                return m;
            else if (nums[m] >= nums[n - 1])
                l = m + 1;
            else if (nums[m] <= nums[n - 1])
                r = m - 1;
        }
        return -1;
    }

    int search(int[] nums, int l, int r, int t) {
        while (l <= r) {
            int m = l + (r - l) / 2;
            if (nums[m] == t)
                return m;
            else if (nums[m] > t)
                r = m - 1;
            else if (nums[m] < t)
                l = m + 1;
        }
        return -1;
    }
}
