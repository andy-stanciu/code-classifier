class Solution {

    public int search(int[] nums, int target) {
        return findindex(nums, target, 0, nums.length - 1);
    }

    public static int findindex(int[] arr, int target, int low, int high) {
        if (low > high)
            return -1;
        int mid = (low + high) / 2;
        if (arr[mid] == target)
            return mid;
        int leftHalfResult = findindex(arr, target, low, mid - 1);
        int rightHalfResult = findindex(arr, target, mid + 1, high);
        return leftHalfResult + rightHalfResult + 1;
    }
}
