class Solution {

    public int search(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        int ans = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                ans = mid;
                break;
            }
            if (nums[mid] >= nums[left]) {
                //Check if the left side is sorted
                if (nums[mid] > target && nums[left] <= target) {
                    //Check if the sorted left side contains target
                    right = mid - 1;
                } else {
                    //If it doesn\'t contains target move to right side
                    left = mid + 1;
                }
            } else {
                //If left side is not sorted definitely right side is sorted
                if (nums[mid] < target && nums[right] >= target) {
                    //Check if the sorted right side contains target
                    left = mid + 1;
                } else {
                    //If it doesn\'t contains target move to left side
                    right = mid - 1;
                }
            }
        }
        return ans;
    }
}
