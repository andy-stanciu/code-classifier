class Solution {

    public int search(int[] arr, int t) {
        int k = -1;
        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i] > arr[i + 1]) {
                k = i;
                break;
            }
        }
        int l = 0;
        int r = k;
        for (int i = 0; i < 2; i++) {
            while (l <= r) {
                int mid = (l + r) / 2;
                if (arr[mid] == t) {
                    return mid;
                } else if (arr[mid] < t) {
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
            l = k + 1;
            r = arr.length - 1;
        }
        return -1;
    }
}
