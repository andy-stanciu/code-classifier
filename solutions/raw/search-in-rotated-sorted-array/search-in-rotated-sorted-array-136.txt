class Solution {

    public int search(int[] nums, int target) {
        //Given rotated sorted array , target -> to find the target element
        int n = nums.length;
        int low = 0, high = n - 1;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] > nums[high])
                low = mid + 1;
            else if (nums[mid] < nums[high])
                high = mid;
        }
        // found the smallest element in the array . .
        int small = low;
        // Now applying the simple binary Search
        low = 0;
        high = n - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            int realMid = (small + mid) % n;
            if (nums[realMid] == target)
                return realMid;
            else if (nums[realMid] < target)
                low = mid + 1;
            else
                high = mid - 1;
        }
        return -1;
    }
}
