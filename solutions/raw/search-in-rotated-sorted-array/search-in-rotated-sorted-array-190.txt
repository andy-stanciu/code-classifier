class Solution {

    public int search(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        while (left <= right) {
            final int mid = left + (right - left) / 2;
            if (nums[left] == target)
                return left;
            else if (nums[right] == target)
                return right;
            else if (nums[mid] == target)
                return mid;
            if (nums[mid] < nums[right] && nums[mid] > target)
                right = mid - 1;
            else
                left += 1;
        }
        return -1;
    }
}
