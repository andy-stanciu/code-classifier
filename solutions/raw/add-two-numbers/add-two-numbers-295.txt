class Solution {

    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        int kalan = 0;
        int bolen = 0;
        ListNode sonuc = new ListNode(0);
        ListNode current = sonuc;
        while (l1 != null || l2 != null || bolen != 0) {
            int toplam = bolen;
            if (l1 != null) {
                toplam += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                toplam += l2.val;
                l2 = l2.next;
            }
            kalan = toplam % 10;
            bolen = toplam / 10;
            current.next = new ListNode(kalan);
            current = current.next;
        }
        return sonuc.next;
    }
}
