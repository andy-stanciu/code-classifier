class Solution {

    public boolean containsNearbyDuplicate(int[] n, int k) {
        if (n == null || n.length < 2 || k == 0)
            return false;
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < n.length; i++) {
            if (hm.containsKey(n[i]) && i - hm.get(n[i]) <= k) {
                return true;
            }
            hm.put(n[i], i);
        }
        return false;
    }
}
