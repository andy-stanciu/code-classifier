class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Set<Integer> set = new HashSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            if (//abs value
            i > k)
                //i cross range remove prev
                set.remove(nums[i - k - 1]);
            if (//set.add returns false if duplicate
            !set.add(nums[i]))
                return true;
        }
        return false;
    }
}
