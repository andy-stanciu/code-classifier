/*
 * Idea: Add  all the k elements in a set and see if we get dublicates while adding. Use HashSet instead of HashMap
*/
class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashSet<Integer> hm = new HashSet<Integer>();
        int l = 0, r = 0;
        int cnt = 0;
        while (l <= r && r < nums.length) {
            if ((r - l) > k) {
                hm.remove(nums[l]);
                l++;
            }
            if (!hm.add(nums[r])) {
                return true;
            }
            r++;
        }
        return false;
    }
}
