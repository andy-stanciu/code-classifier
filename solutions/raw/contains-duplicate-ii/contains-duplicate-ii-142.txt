class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        //int count = 0;
        // Arrays.sort(nums);
        HashMap<Integer, Integer> h1 = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (h1.containsKey(nums[i])) {
                int value = h1.get(nums[i]);
                if ((i - value) <= k) {
                    return true;
                } else {
                    h1.put(nums[i], i);
                }
            } else {
                h1.put(nums[i], i);
            }
        }
        return false;
    }
}
