class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        int start = 0, end = k;
        boolean flag = false;
        LinkedHashSet<Integer> hashSet = new LinkedHashSet<Integer>(k + 1);
        int bufferCounter = 0;
        for (int i = 0; i < nums.length; i++) {
            boolean isInserted = hashSet.add(nums[i]);
            //System.out.println("after adding hashSet "+hashSet);
            if (!isInserted) {
                return true;
            }
            bufferCounter++;
            if (bufferCounter == k + 1 || flag) {
                hashSet.removeFirst();
                //System.out.println(hashSet);
                flag = true;
            }
        }
        return false;
    }
}
