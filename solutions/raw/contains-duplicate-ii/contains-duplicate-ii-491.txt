class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> map = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; i++) {
            int key = nums[i];
            if (map.containsKey(key) == false)
                map.put(key, i);
            else if (i - map.get(key) <= k)
                return true;
            else
                map.put(key, i);
        }
        return false;
    }
}
