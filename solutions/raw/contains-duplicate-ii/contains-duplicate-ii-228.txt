class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        if (k == 0)
            return false;
        int n = nums.length;
        if (k > n)
            return containsNearbyDuplicate(nums, n);
        HashSet<Integer> hs = new HashSet<>(k);
        int i = 0;
        while (i < k) {
            if (hs.contains(nums[i]))
                return true;
            hs.add(nums[i++]);
        }
        while (i < n) {
            if (hs.contains(nums[i]))
                return true;
            hs.remove(nums[i - k]);
            hs.add(nums[i++]);
        }
        return false;
    }
}
