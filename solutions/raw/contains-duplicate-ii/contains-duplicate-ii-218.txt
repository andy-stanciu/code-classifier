class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> ma = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (ma.containsKey(nums[i])) {
                int lastindex = ma.get(nums[i]);
                if (Math.abs(lastindex - i) <= k) {
                    return true;
                }
            }
            ma.put(nums[i], i);
        }
        return false;
    }
}
