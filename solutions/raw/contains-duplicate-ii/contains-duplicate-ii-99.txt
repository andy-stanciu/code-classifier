class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> hash = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            Integer lastIndex = hash.put(nums[i], i);
            if (lastIndex != null && i - lastIndex <= k) {
                return true;
            }
        }
        return false;
    }
}
