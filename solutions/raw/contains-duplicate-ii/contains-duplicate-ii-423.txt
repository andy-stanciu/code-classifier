class Solution {

    /**
     * For which values we know that | i - j | <= k ?
     * As j < i, then i - j <= k
     * That means, i <= j + k
     * Which is the same than, i - k <= j < i
     */
    public boolean containsNearbyDuplicate(int[] nums, int k) {
        for (int i = 0; i < nums.length; i++) for (int j = Math.max(i - k, 0); j < i; j++) if (nums[i] == nums[j])
            return true;
        return false;
    }
}
