class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        boolean result = false;
        Map<Integer, Integer> occr = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (occr.containsKey(nums[i])) {
                if (Math.abs(i - occr.get(nums[i])) <= k) {
                    result = true;
                    break;
                } else {
                    occr.put(nums[i], i);
                }
            } else {
                occr.put(nums[i], i);
            }
        }
        return result;
    }
}
