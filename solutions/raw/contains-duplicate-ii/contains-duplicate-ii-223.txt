class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> account = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (account.containsKey(nums[i])) {
                if ((i - account.get(nums[i])) <= k)
                    return true;
            }
            account.put(nums[i], i);
        }
        return false;
    }
}
