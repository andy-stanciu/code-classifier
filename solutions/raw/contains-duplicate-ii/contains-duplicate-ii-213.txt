class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> hm = new HashMap<Integer, Integer>();
        hm.put(nums[0], 0);
        for (int i = 1; i < nums.length; i++) {
            if (hm.containsKey(nums[i]) && Math.abs(hm.get(nums[i]) - i) <= k) {
                return true;
            } else {
                hm.put(nums[i], i);
            }
        }
        return false;
    }
}
