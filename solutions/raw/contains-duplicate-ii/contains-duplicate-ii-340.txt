class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashSet<Integer> mySet = new HashSet<>();
        mySet.add(nums[0]);
        Integer i = 0;
        Integer j = i + 1;
        while (j < nums.length) {
            if (Math.abs(i - j) <= k) {
                if (mySet.contains(nums[j])) {
                    return true;
                } else {
                    mySet.add(nums[j]);
                    j = j + 1;
                    ;
                }
            } else {
                mySet.remove(nums[i]);
                i = i + 1;
            }
        }
        return false;
    }
}
