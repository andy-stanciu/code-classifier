class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> dict = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (dict.containsKey(nums[i])) {
                int prevIndex = dict.get(nums[i]);
                if (i - prevIndex <= k) {
                    System.out.println(dict);
                    return true;
                } else {
                    dict.put(nums[i], i);
                }
            } else {
                dict.put(nums[i], i);
            }
        }
        System.out.println(dict);
        return false;
    }
}
