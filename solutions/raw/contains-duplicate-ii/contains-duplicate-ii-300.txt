class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (set.contains(nums[i])) {
                //return true,if we found a duplicate within k distance
                return true;
            }
            // Add the current element nums[i] to the set
            set.add(nums[i]);
            // Maintain the size of the set to be at most k
            if (set.size() > k) {
                // Remove the oldest element if set size exceeds k
                set.remove(nums[i - k]);
            }
        }
        // If no duplicate within k distance is found, return false
        return false;
    }
}
