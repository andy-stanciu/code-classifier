class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i])) {
                int j = map.get(nums[i]);
                int diff = Math.abs(j - i);
                if (diff <= k) {
                    return true;
                }
            }
            map.put(nums[i], i);
        }
        return false;
    }
}
