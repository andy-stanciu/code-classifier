class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        // Initial variables
        Map<Integer, Integer> map = new HashMap<>();
        // Iterate the array
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i]) && Math.abs(i - map.get(nums[i])) <= k) {
                return true;
            }
            map.put(nums[i], i);
        }
        // Return the result
        return false;
    }
}
