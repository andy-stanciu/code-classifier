class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashSet<Integer> g = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (i > k) {
                g.remove(nums[i - k - 1]);
            }
            if (!g.add(nums[i])) {
                return true;
            }
        }
        return false;
    }
}
