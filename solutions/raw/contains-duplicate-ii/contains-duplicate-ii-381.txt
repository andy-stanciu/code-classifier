class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> lastOccurrenceIndices = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int num = nums[i];
            if (lastOccurrenceIndices.containsKey(num) && i - lastOccurrenceIndices.get(num) <= k)
                return true;
            lastOccurrenceIndices.put(num, i);
        }
        return false;
    }
}
