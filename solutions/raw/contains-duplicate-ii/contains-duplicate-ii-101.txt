class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> hm = new HashMap<Integer, Integer>();
        int i = 0;
        while (i < nums.length) {
            int key = nums[i];
            if (hm.containsKey(key)) {
                int prevIndex = hm.get(key);
                if ((i - prevIndex) <= k) {
                    return true;
                }
            }
            hm.put(key, i);
            i++;
        }
        return false;
    }
}
