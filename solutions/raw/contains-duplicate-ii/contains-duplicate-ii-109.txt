class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        // 1 2 3 1
        // <1: 0, 2: 1, 3: 2, >
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i])) {
                int diff = i - map.get(nums[i]);
                if (diff <= k) {
                    return true;
                }
            }
            map.put(nums[i], i);
        }
        return false;
    }
}
