class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int n = nums[i];
            Integer value = map.get(n);
            if (value != null && Math.abs(value - i) <= k)
                return true;
            map.put(n, i);
        }
        return false;
    }
}
