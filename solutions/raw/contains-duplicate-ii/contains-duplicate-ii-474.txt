class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        int index = 0;
        for (Integer n : nums) {
            if (map.containsKey(n)) {
                if (Math.abs(map.get(n) - index) <= k) {
                    return true;
                }
                map.put(n, index);
            }
            map.put(n, index);
            index++;
        }
        return false;
    }
}
