class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        int e = 0, s = 0;
        HashMap<Integer, Integer> hm = new HashMap<>();
        while (e < nums.length) {
            if (hm.containsKey(nums[e])) {
                int ans = Math.abs(hm.get(nums[e]) - e);
                if (ans <= k)
                    return true;
                hm.remove(nums[s]);
                hm.put(nums[e], e);
                s++;
            } else {
                hm.put(nums[e], e);
            }
            e++;
        }
        return false;
    }
}
