class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> dupli = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (dupli.containsKey(nums[i])) {
                int last_seen = dupli.get(nums[i]);
                if (Math.abs(i - last_seen) <= k) {
                    return true;
                }
            }
            dupli.put(nums[i], i);
        }
        return false;
    }
}
