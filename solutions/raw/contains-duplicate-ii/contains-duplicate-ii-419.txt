class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        boolean result = false;
        int end;
        for (int start = 0; start < nums.length - 1; start++) {
            end = start + k;
            if (end > nums.length - 1) {
                end = nums.length - 1;
            }
            for (int i = start + 1; i <= end; i++) {
                if (nums[start] == nums[i]) {
                    result = true;
                    return result;
                }
            }
        }
        return result;
    }
}
