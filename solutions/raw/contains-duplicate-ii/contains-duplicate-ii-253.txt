class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        int n = nums.length;
        int l = 0, r = 0, len = 0;
        HashMap<Integer, Integer> hmap = new HashMap<>();
        while (r < n) {
            if (!hmap.containsKey(nums[r])) {
                hmap.put(nums[r], r);
            } else {
                int ind = hmap.get(nums[r]);
                len = r - ind;
                System.out.println(len);
                if (len <= k)
                    return true;
                l = ind + 1;
                hmap.put(nums[r], r);
            }
            r++;
        }
        return false;
    }
}
