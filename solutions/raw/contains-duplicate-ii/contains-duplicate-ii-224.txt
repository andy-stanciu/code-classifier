class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, String> hashMap = new HashMap<Integer, String>();
        int chk = 0;
        String str = "";
        for (int i = 0; i < nums.length; i++) {
            if (hashMap.containsKey(nums[i])) {
                String temp = hashMap.get(nums[i]);
                str = str + temp + Integer.toString(i);
                //hashMap.replace(nums[i], str);
                int value = Character.getNumericValue(str.charAt(0)) - Character.getNumericValue(str.charAt(1));
                int absoluteValue = Math.abs(value);
                if (k >= absoluteValue) {
                    chk = 1;
                    break;
                } else {
                    chk = 0;
                    str = "";
                    str = str + Integer.toString(i);
                    hashMap.replace(nums[i], str);
                    str = "";
                }
            } else {
                str = str + Integer.toString(i);
                hashMap.put(nums[i], str);
                str = "";
            }
        }
        if (chk == 1) {
            return true;
        } else {
            return false;
        }
    }
}
