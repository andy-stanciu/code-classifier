class //using HashMap
Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i <= nums.length - 1; i++) {
            if (map.containsKey(nums[i]) && (Math.max(map.get(nums[i]), i) - Math.min(map.get(nums[i]), i)) <= k) {
                return true;
            } else
                map.put(nums[i], i);
        }
        return false;
    }
}
