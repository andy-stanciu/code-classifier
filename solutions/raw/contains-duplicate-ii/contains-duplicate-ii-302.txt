class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        // Sliding window
        Set<Integer> set = new HashSet<>();
        for (int i = 0, l = 0; i < nums.length; i++) {
            if (i - l > k)
                set.remove(nums[l++]);
            if (set.contains(nums[i]))
                return true;
            else
                set.add(nums[i]);
        }
        return false;
    }
}
