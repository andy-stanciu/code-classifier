class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> h = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (h.get(nums[i]) == null)
                h.put(nums[i], i);
            else {
                int m = i - h.get(nums[i]);
                h.put(nums[i], i);
                if (m <= k)
                    return true;
            }
        }
        return false;
    }
}
