class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> boom = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int doom = nums[i];
            if (boom.containsKey(doom)) {
                int j = boom.get(doom);
                if (Math.abs(i - j) <= k) {
                    return true;
                }
            }
            boom.put(doom, i);
        }
        return false;
    }
}
