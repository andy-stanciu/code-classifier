class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        int n = nums.length;
        int left = 0;
        HashSet<Integer> set = new HashSet<>();
        for (int r = 0; r < n; r++) {
            if (r - left > k) {
                set.remove(nums[left]);
                left++;
            }
            if (set.contains(nums[r]))
                return true;
            set.add(nums[r]);
        }
        return false;
    }
}
