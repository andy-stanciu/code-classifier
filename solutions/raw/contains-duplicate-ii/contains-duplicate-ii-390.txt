class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            Integer ret = map.put(nums[i], i);
            if (ret != null) {
                if (Math.abs(ret - i) <= k)
                    return true;
            }
        }
        return false;
    }
}
