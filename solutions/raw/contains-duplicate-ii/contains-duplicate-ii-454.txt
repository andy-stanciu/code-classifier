class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        int n = nums.length;
        for (int r = 0; r < n; r++) {
            int curr = nums[r];
            if (map.containsKey(curr)) {
                if (Math.abs(r - map.get(curr)) <= k)
                    return true;
            }
            map.put(curr, r);
        }
        return false;
    }
}
