class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Map<Integer, Integer> mp = new HashMap<>();
        int index = 0;
        for (int num : nums) {
            if (mp.containsKey(num) && index - mp.get(num) <= k)
                return true;
            mp.put(num, index++);
        }
        return false;
    }
}
