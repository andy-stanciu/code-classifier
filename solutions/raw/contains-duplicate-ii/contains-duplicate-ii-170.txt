class Solution {

    public boolean containsNearbyDuplicate(int[] nums, int k) {
        Set<Integer> jtemp = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (jtemp.contains(nums[i])) {
                return true;
            }
            jtemp.add(nums[i]);
            if (jtemp.size() > k) {
                jtemp.remove(nums[i - k]);
            }
        }
        return false;
    }
}
