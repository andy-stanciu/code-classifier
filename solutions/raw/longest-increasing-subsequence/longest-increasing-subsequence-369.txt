class Solution {

    public int lengthOfLIS(int[] nums) {
        Integer[] memo = new Integer[nums.length];
        return lengthOfLISR(nums, 0, memo, nums[0]);
    }

    int lengthOfLISR(int[] nums, int start, Integer[] memo, int no) {
        if (start >= nums.length)
            return 0;
        if (memo[start] != null)
            return memo[start];
        int maxCount = 0;
        for (int i = start; i < nums.length; i++) {
            int count = 0;
            if (nums[i] > no || start == 0) {
                count = 1 + lengthOfLISR(nums, i + 1, memo, nums[i]);
            }
            maxCount = Math.max(count, maxCount);
        }
        memo[start] = maxCount;
        return memo[start];
    }
}
