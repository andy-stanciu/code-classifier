class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        int max = 1;
        Arrays.fill(dp, 1);
        dp[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (dp[i] < dp[j] + 1 && nums[i] > nums[j])
                    dp[i] = dp[j] + 1;
            }
            if (dp[i] > max)
                max = dp[i];
        }
        return max;
    }
}
