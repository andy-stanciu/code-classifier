class Solution {

    public int lengthOfLIS(int[] nums) {
        int ans = 0, dp[] = new int[nums.length];
        for (int i = nums.length - 1; i >= 0; i--) {
            for (int j = i + 1, k = 0; j < nums.length; j++) {
                if (nums[j] > nums[i]) {
                    dp[i] = Math.max(dp[j], dp[i]);
                }
            }
            dp[i]++;
            ans = Math.max(ans, dp[i]);
        }
        return ans;
    }
}
