class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        if (n <= 1)
            return n;
        int longest = 0;
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    dp[i] = Math.max(dp[i], 1 + dp[j]);
                }
            }
            longest = Math.max(longest, dp[i]);
        }
        return longest;
    }
}
