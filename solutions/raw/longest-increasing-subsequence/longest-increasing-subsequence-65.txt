class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] l = new int[n];
        for (int i = 0; i < n; i++) {
            l[i] = 1;
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i])
                    l[i] = Math.max(l[i], l[j] + 1);
            }
        }
        Arrays.sort(l);
        return l[n - 1];
    }
}
