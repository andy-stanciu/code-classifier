class Solution {

    public int lengthOfLIS(int[] a) {
        int[] dp = new int[a.length];
        Arrays.fill(dp, 1);
        for (int i = 1; i < a.length; i++) {
            int max = 0;
            for (int j = 0; j < i; j++) {
                if (a[i] > a[j]) {
                    int y = dp[j];
                    max = Math.max(max, dp[j]);
                }
            }
            dp[i] = 1 + max;
        }
        return Arrays.stream(dp).sorted().max().getAsInt();
    }
}
