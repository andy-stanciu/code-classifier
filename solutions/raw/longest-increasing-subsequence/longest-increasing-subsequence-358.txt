class Solution {

    public int lengthOfLIS(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) {
            if (set.isEmpty() || num > set.last()) {
                set.add(num);
            } else {
                int greater = set.ceiling(num);
                set.remove(greater);
                set.add(num);
            }
        }
        return set.size();
    }
}
