class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] sort = new int[nums.length];
        int len = 0;
        for (int n : nums) {
            int index = Arrays.binarySearch(sort, 0, len, n);
            if (index < 0)
                index = -index - 1;
            sort[index] = n;
            if (index >= len) {
                len++;
            }
        }
        return len;
    }
}
