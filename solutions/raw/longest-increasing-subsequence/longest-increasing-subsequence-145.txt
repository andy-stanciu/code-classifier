class Solution {

    public int lengthOfLIS(int[] nums) {
        int ans = 1;
        int i = 1, j = 0;
        int[] memo = new int[nums.length];
        Arrays.fill(memo, 1);
        while (i < nums.length) {
            while (j < i) {
                if (nums[j] < nums[i]) {
                    memo[i] = Math.max(memo[i], memo[j] + 1);
                    ans = Math.max(ans, memo[i]);
                }
                j++;
            }
            j = 0;
            i++;
        }
        return ans;
    }
}
