class Solution {

    public int nextGreater(ArrayList<Integer> temp, int val) {
        int s = 0;
        int e = temp.size() - 1;
        while (s <= e) {
            int mid = s + (e - s) / 2;
            if (temp.get(mid) == val) {
                return mid;
            } else if (temp.get(mid) > val) {
                e = mid - 1;
            } else {
                s = mid + 1;
            }
        }
        return s;
    }

    public int lengthOfLIS(int[] nums) {
        ArrayList<Integer> temp = new ArrayList<>();
        temp.add(nums[0]);
        int len = 1;
        int n = nums.length;
        for (int i = 1; i < n; i++) {
            if (nums[i] > temp.get(temp.size() - 1)) {
                temp.add(nums[i]);
                len++;
            } else {
                int ind = nextGreater(temp, nums[i]);
                temp.set(ind, nums[i]);
            }
        }
        return len;
    }
}
