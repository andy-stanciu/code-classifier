class Solution {

    public int lengthOfLIS(int[] arr) {
        String amax = Integer.toString(arr[0]);
        String[] dp = new String[arr.length];
        for (int i = 0; i < arr.length; i++) {
            String val = Integer.toString(arr[i]);
            String max = val;
            for (int j = 0; j < i; j++) if (arr[j] < arr[i] && dp[j].length() + 1 > max.length())
                max = dp[j] + ", " + val;
            dp[i] = max;
            if (dp[i].length() > amax.length())
                amax = dp[i];
        }
        System.out.println(amax);
        return amax.split(" ").length;
    }
}
