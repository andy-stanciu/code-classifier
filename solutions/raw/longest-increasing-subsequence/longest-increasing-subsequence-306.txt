class Solution {

    public int lengthOfLIS(int[] nums) {
        //Dp is gonna be of length nums and have the longest increasing ss
        //That the number at i is a part of at each
        int[] dp = new int[nums.length];
        dp[0] = 1;
        //Track the max sequence of it all
        int maxPat = 1;
        for (int i = 1; i < dp.length; i++) {
            //Find maximum sized sequence that this number at i is part of
            int max = Integer.MIN_VALUE;
            int maxIndex = -1;
            for (int j = i - 1; j >= 0; j--) {
                if (nums[j] < nums[i]) {
                    if (dp[j] > max) {
                        max = dp[j];
                        maxIndex = j;
                    }
                }
            }
            //This was part of no sequence
            if (maxIndex == -1) {
                dp[i] = 1;
            } else //Add 1 to the maximum sequence
            {
                dp[i] = dp[maxIndex] + 1;
            }
            //Update max sequence
            if (dp[i] > maxPat) {
                maxPat = dp[i];
            }
        }
        return maxPat;
    }
}
