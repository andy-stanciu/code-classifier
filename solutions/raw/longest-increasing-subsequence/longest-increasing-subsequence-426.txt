class Solution {

    public int lengthOfLIS(int[] nums) {
        // Solve the problem for nums[:i]
        int[] dp = new int[nums.length];
        // Every index is at least a one number LIS
        Arrays.fill(dp, 1);
        int max = 1;
        for (int i = 1; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] > nums[j]) {
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                    max = Math.max(max, dp[i]);
                }
            }
        }
        return max;
    }
}
