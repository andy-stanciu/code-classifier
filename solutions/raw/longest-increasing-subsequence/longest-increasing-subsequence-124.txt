class Solution {

    public int lengthOfLIS(int[] a) {
        if (a.length == 0)
            return 0;
        int[] dp = new int[a.length];
        Arrays.fill(dp, 1);
        for (int i = 1; i < a.length; ++i) for (int j = 0; j < i; ++j) if (a[j] < a[i])
            dp[i] = Math.max(dp[i], dp[j] + 1);
        int max = 0;
        for (int i : dp) max = Math.max(i, max);
        return max;
    }
}
