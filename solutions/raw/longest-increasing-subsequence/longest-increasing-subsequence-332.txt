class Solution {

    public int lengthOfLIS(int[] a) {
        int n = a.length;
        int[] dp = new int[n];
        for (int i = 0; i < n; i++) {
            dp[i] = 1;
            for (int j = i - 1; j >= 0; j--) {
                if (a[j] >= a[i])
                    continue;
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) ans = Math.max(ans, dp[i]);
        return ans;
    }
}
