class Solution {

    public int lengthOfLIS(int[] nums) {
        ArrayList<Integer> dp = new ArrayList<>();
        for (int num : nums) {
            int pos = Collections.binarySearch(dp, num);
            if (pos < 0) {
                pos = -(pos + 1);
            }
            if (pos < dp.size()) {
                dp.set(pos, num);
            } else {
                dp.add(num);
            }
        }
        return dp.size();
    }
}
