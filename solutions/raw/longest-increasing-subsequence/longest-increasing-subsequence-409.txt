class Solution {

    public int lengthOfLIS(int[] nums) {
        // Brute Force - we will be thinking whether to take or not take each element of an array, hence the time complexity will be 2^n.
        // DP bottom up approach
        int[] dp = new int[nums.length];
        int max = 0;
        Arrays.fill(dp, 1);
        for (int i = nums.length - 1; i >= 0; i--) {
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[i] < nums[j])
                    dp[i] = Math.max(dp[i], 1 + dp[j]);
            }
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
