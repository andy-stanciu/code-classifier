class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] a = new int[nums.length];
        int maxLength = 1;
        for (int i = 0; i < nums.length; i++) {
            a[i] = 1;
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    a[i] = Math.max(a[i], a[j] + 1);
                }
                maxLength = Math.max(maxLength, a[i]);
            }
        }
        return maxLength;
    }
}
