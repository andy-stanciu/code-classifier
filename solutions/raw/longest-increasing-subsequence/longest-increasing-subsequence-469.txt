class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] helper = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            int count = -1;
            for (int j = i; j >= 0; j--) {
                if (nums[i] > nums[j]) {
                    count = Math.max(count, helper[j]);
                }
            }
            helper[i] = count + 1;
        }
        int ans = -1;
        for (int elem : helper) {
            ans = Math.max(ans, elem);
        }
        return ans + 1;
    }
}
