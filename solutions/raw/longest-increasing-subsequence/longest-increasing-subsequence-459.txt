class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] longest = new int[nums.length];
        int ans = 1;
        longest[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            int max = 0;
            for (int j = 0; j < i; j++) {
                if (longest[j] > max && nums[j] < nums[i])
                    max = longest[j];
            }
            longest[i] = max + 1;
            ans = Math.max(longest[i], ans);
        }
        return ans;
    }
}
