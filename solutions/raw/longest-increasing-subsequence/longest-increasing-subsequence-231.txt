class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[][] dp = new int[n][n + 1];
        for (int[] d : dp) {
            Arrays.fill(d, -1);
        }
        return solve(n - 1, -1, nums, dp);
    }

    int solve(int i, int prev, int[] arr, int[][] dp) {
        if (i < 0)
            return 0;
        if (dp[i][prev + 1] != -1)
            return dp[i][prev + 1];
        int len = 0;
        if (prev == -1 || arr[i] < arr[prev])
            len = Math.max(len, 1 + solve(i - 1, i, arr, dp));
        len = Math.max(len, 0 + solve(i - 1, prev, arr, dp));
        return dp[i][prev + 1] = len;
    }
}
