class Solution {

    public int lengthOfLIS(int[] nums) {
        int N = nums.length;
        return findLongestLIS(0, -1, nums);
    }

    int findLongestLIS(int index, int prev, int[] nums) {
        if (index == nums.length) {
            return 0;
        }
        //base case
        int notTaken = 0 + findLongestLIS(index + 1, prev, nums);
        int taken = (int) -1e6;
        if (prev == -1 || nums[index] > nums[prev]) {
            taken = 1 + findLongestLIS(index + 1, index, nums);
        }
        return Math.max(notTaken, taken);
    }
}
