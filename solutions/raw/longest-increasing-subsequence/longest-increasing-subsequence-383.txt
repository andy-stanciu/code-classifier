class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] lengths = new int[nums.length + 1];
        lengths[0] = Integer.MIN_VALUE;
        int longest = 0;
        for (int i = 0; i < nums.length; i++) {
            int predecessorPosition = Arrays.binarySearch(lengths, 0, longest + 1, nums[i]);
            if (predecessorPosition > 0)
                continue;
            predecessorPosition = -1 * predecessorPosition - 1;
            lengths[predecessorPosition] = nums[i];
            if (predecessorPosition > longest) {
                longest = predecessorPosition;
            }
        }
        return longest;
    }
}
