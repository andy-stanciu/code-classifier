class Solution {

    public int lengthOfLIS(int[] nums) {
        int N = nums.length;
        int[] B = new int[N];
        for (int i = 0; i < N; i++) {
            B[i] = 1;
        }
        for (int i = 1; i < N; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] > nums[j] && B[i] < B[j] + 1) {
                    B[i] = B[j] + 1;
                }
            }
        }
        int count = 0;
        for (int i = 0; i < N; i++) {
            if (B[i] > count) {
                count = B[i];
            }
        }
        return count;
    }
}
