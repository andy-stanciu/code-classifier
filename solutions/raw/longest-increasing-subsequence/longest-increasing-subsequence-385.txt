class Solution {

    public int lengthOfLIS(int[] nums) {
        List<Integer> list = new ArrayList<>();
        list.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (list.get(list.size() - 1) < nums[i])
                list.add(nums[i]);
            else {
                int start = 0;
                int end = list.size() - 1;
                while (start < end) {
                    int mid = (start + end) / 2;
                    if (list.get(mid) < nums[i])
                        start = mid + 1;
                    else
                        end = mid;
                }
                list.set(start, nums[i]);
            }
        }
        return list.size();
    }
}
