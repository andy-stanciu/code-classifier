class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        Integer[][] dp = new Integer[n][n + 1];
        return lis(0, nums, -1, dp);
    }

    public int lis(int ind, int[] nums, int prev, Integer[][] dp) {
        if (ind == nums.length)
            return 0;
        int prevInd = prev + 1;
        if (dp[ind][prevInd] != null)
            return dp[ind][prevInd];
        int len = lis(ind + 1, nums, prev, dp);
        int l1 = 0;
        if (prev == -1 || nums[ind] > nums[prev]) {
            l1 = 1 + lis(ind + 1, nums, ind, dp);
        }
        return dp[ind][prevInd] = Math.max(len, l1);
    }
}
