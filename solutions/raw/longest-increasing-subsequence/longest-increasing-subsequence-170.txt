class Solution {

    public int lengthOfLIS(int[] nums) {
        int result = 1;
        int n = nums.length;
        int[] ls = new int[n];
        Arrays.fill(ls, 1);
        for (int i = n - 2; i >= 0; i--) {
            for (int j = i + 1; j < n; j++) {
                if (nums[j] > nums[i]) {
                    ls[i] = Math.max(ls[i], 1 + ls[j]);
                }
            }
            result = Math.max(result, ls[i]);
        }
        return result;
    }
}
