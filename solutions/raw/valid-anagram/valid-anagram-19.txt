class Solution {

    public int transform(String s) {
        int sum = 0;
        for (int i = 0; i < s.length(); i++) {
            int ch = s.codePointAt(i);
            sum += ch * ch * ch * ch;
        }
        return sum;
    }

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length())
            return false;
        return transform(s) == transform(t);
    }
}
