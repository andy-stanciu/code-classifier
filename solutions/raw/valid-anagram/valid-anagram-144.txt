class Solution {

    public boolean isAnagram(String s, String t) {
        char[] s1 = s.toCharArray();
        char[] s2 = t.toCharArray();
        Arrays.sort(s1);
        String st1 = Arrays.toString(s1);
        Arrays.sort(s2);
        String st2 = Arrays.toString(s2);
        if (st1.equals(st2))
            return true;
        else
            return false;
    }
}
