class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length())
            return false;
        char[] str = s.toCharArray();
        char[] x = t.toCharArray();
        Arrays.sort(str);
        Arrays.sort(x);
        for (int i = 0; i < str.length; i++) if (str[i] != x[i])
            return false;
        return true;
    }
}
