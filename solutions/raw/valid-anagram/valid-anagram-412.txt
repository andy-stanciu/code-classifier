class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length())
            return false;
        var sDict = new HashMap<Character, Integer>();
        for (int i = 0; i < s.length(); i++) {
            var current = sDict.getOrDefault(s.charAt(i), 0);
            sDict.put(s.charAt(i), current + 1);
        }
        for (int i = 0; i < t.length(); i++) {
            var current = sDict.getOrDefault(t.charAt(i), 0);
            if (current == 0)
                return false;
            sDict.put(t.charAt(i), current - 1);
        }
        return true;
    }
}
