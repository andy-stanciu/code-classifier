class Solution {

    public int hash(String s) {
        int prod = 1;
        int sum = 0;
        for (int i = 0; i < s.length(); i++) {
            prod *= (int) s.charAt(i);
            sum += (int) s.charAt(i);
        }
        prod += sum;
        prod *= s.length();
        return prod;
    }

    public boolean isAnagram(String s, String t) {
        int h1 = hash(s);
        int h2 = hash(t);
        return h1 == h2;
    }
}
