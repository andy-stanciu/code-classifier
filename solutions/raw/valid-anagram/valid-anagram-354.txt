class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        // change string to character arrray
        char[] Z = s.toCharArray();
        char[] Y = t.toCharArray();
        //sort the array for check the arrays are equal or not
        Arrays.sort(Z);
        Arrays.sort(Y);
        //return the boolean value true or false if they equal or not
        return Arrays.equals(Z, Y);
    }
}
