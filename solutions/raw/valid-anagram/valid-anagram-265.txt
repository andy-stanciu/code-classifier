class Solution {

    public boolean isAnagram(String s, String t) {
        boolean ans = true;
        if (s.length() != t.length()) {
            ans = false;
            return ans;
        } else {
            char[] arr1 = s.toCharArray();
            Arrays.sort(arr1);
            char[] arr2 = t.toCharArray();
            Arrays.sort(arr2);
            for (int i = 0; i < s.length(); i++) {
                if (arr1[i] != arr2[i]) {
                    ans = false;
                    return ans;
                } else {
                }
            }
        }
        return ans;
    }
}
