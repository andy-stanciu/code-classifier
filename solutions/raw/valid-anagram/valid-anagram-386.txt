class Solution {

    public boolean isAnagram(String s, String t) {
        char[] stringToChar1 = s.toCharArray();
        Arrays.sort(stringToChar1);
        char[] stringToChar2 = t.toCharArray();
        Arrays.sort(stringToChar2);
        if (Arrays.equals(stringToChar1, stringToChar2) == true) {
            return true;
        } else {
            return false;
        }
    }
}
