class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        Map<Character, Integer> characterAndCount = new HashMap<>();
        char[] firstString = s.toCharArray();
        char[] secondString = t.toCharArray();
        for (int i = 0; i < s.length(); i++) {
            characterAndCount.put(firstString[i], characterAndCount.getOrDefault(firstString[i], 0) + 1);
            characterAndCount.put(secondString[i], characterAndCount.getOrDefault(secondString[i], 0) - 1);
        }
        for (int value : characterAndCount.values()) {
            if (value != 0) {
                return false;
            }
        }
        return true;
    }
}
