class Solution {

    public boolean isAnagram(String s, String t) {
        var dict = new HashMap<Character, Integer>();
        for (char c : s.toCharArray()) {
            if (!dict.containsKey(c))
                dict.put(c, 1);
            else
                dict.put(c, dict.get(c) + 1);
        }
        for (char c : t.toCharArray()) {
            if (!dict.containsKey(c))
                dict.put(c, 1);
            else
                dict.put(c, dict.get(c) - 1);
        }
        for (char c : dict.keySet()) {
            if (dict.get(c) != 0)
                return false;
        }
        return true;
    }
}
