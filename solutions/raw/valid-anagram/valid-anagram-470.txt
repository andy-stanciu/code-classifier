class Solution {

    public boolean isAnagram(String s, String t) {
        boolean a = true;
        char[] arrS = s.toCharArray();
        Arrays.sort(arrS);
        char[] arrT = t.toCharArray();
        Arrays.sort(arrT);
        if (arrS.length != arrT.length) {
            a = false;
        } else {
            for (int i = 0; i < arrS.length; i++) {
                if (arrS[i] != arrT[i]) {
                    a = false;
                }
            }
        }
        return a;
    }
}
