class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        } else {
            char[] news = s.toCharArray();
            char[] newt = t.toCharArray();
            Arrays.sort(news);
            Arrays.sort(newt);
            int count = 0;
            for (int i = 0; i < s.length(); i++) {
                if (news[i] == newt[i]) {
                    count++;
                }
            }
            if (count == s.length()) {
                return true;
            } else {
                return false;
            }
        }
    }
}
