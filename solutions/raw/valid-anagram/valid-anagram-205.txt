class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] freq = new int[26];
        int len = s.length();
        int indexA = 0;
        int indexB = 0;
        while (indexA < len && indexB < len) {
            char chA = s.charAt(indexA);
            int frqA = chA - 97;
            freq[frqA]++;
            char chB = t.charAt(indexA);
            int frqB = chB - 97;
            freq[frqB]--;
            indexA++;
            indexB++;
        }
        for (int i = 0; i < 26; i++) {
            if (freq[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
