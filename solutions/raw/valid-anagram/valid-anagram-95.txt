class Solution {

    public boolean isAnagram(String s, String t) {
        Map<Character, Integer> dic = new HashMap<>();
        for (char character : s.toCharArray()) {
            dic.put(character, dic.getOrDefault(character, 0) + 1);
        }
        for (char character : t.toCharArray()) {
            if (dic.containsKey(character)) {
                dic.put(character, dic.get(character) - 1);
                if (dic.get(character) < 0) {
                    return false;
                }
            } else {
                return false;
            }
        }
        for (int count : dic.values()) {
            if (count != 0) {
                return false;
            }
        }
        return true;
    }
}
