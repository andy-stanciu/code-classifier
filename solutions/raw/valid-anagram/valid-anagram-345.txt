class Solution {

    public boolean isAnagram(String s, String t) {
        String a = sorted(s);
        String b = sorted(t);
        int x = a.compareTo(b);
        if (x == 0) {
            return true;
        }
        return false;
    }

    public static String sorted(String s) {
        char[] ch = s.toCharArray();
        Arrays.sort(ch);
        String s1 = new String(ch);
        return s1;
    }
}
