class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() == t.length()) {
            StringBuilder copieS = new StringBuilder(s);
            for (char y : t.toCharArray()) {
                int indexY = copieS.indexOf(String.valueOf(y));
                if (indexY == -1) {
                    return false;
                } else {
                    copieS.deleteCharAt(indexY);
                }
                ;
            }
            return true;
        } else {
            return false;
        }
    }
}
