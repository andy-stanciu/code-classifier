class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length())
            return false;
        char[] arrayOfS = s.toCharArray();
        char[] arrayOfT = t.toCharArray();
        Arrays.sort(arrayOfS);
        Arrays.sort(arrayOfT);
        if (Arrays.equals(arrayOfS, arrayOfT))
            return true;
        return false;
    }
}
