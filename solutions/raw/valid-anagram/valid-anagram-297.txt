class Solution {

    public boolean isAnagram(String s, String t) {
        int m = s.length();
        int n = t.length();
        HashMap<Character, Integer> stable = new HashMap<>();
        char[] sarray = s.toCharArray();
        char[] tarray = t.toCharArray();
        if (s.length() != t.length())
            return false;
        for (int i = 0; i < sarray.length; i++) {
            if (stable.containsKey(sarray[i])) {
                stable.put(sarray[i], stable.get(sarray[i]) + 1);
            } else {
                stable.put(sarray[i], 1);
            }
        }
        for (int i = 0; i < sarray.length; i++) {
            char tocheck = tarray[i];
            if (stable.containsKey(tocheck) && stable.get(tocheck) > 0) {
                stable.put(tocheck, stable.get(tocheck) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
