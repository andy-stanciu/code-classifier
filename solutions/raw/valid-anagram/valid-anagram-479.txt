class Solution {

    public boolean isAnagram(String s, String t) {
        //If lengths are different, they can\'t be anagrams
        if (s.length() != t.length())
            return false;
        char[] word1 = s.toCharArray();
        char[] word2 = t.toCharArray();
        Arrays.sort(word1);
        Arrays.sort(word2);
        for (int i = 0; i < word1.length; i++) {
            if (word1[i] != word2[i])
                return false;
        }
        return true;
    }
}
