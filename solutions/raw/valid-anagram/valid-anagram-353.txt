class Solution {

    public boolean isAnagram(String s, String t) {
        char[] ch1 = s.toCharArray();
        char[] ch2 = t.toCharArray();
        Arrays.sort(ch1);
        Arrays.sort(ch2);
        boolean val = false;
        if (ch1.length != ch2.length) {
            val = false;
        }
        if (Arrays.equals(ch1, ch2)) {
            val = true;
        } else {
            val = false;
        }
        return val;
    }
}
