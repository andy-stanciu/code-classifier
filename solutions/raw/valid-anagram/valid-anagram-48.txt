class Solution {

    public boolean isAnagram(String s, String t) {
        //check whether the lengths of both the strings are same or not . If they are not same, then return false .
        if (s.length() != t.length()) {
            return false;
        }
        //Converting the given strings into character arrays
        char[] arr1 = s.toCharArray();
        char[] arr2 = t.toCharArray();
        //Sort the arrays so that they can become the same strings to be compared
        Arrays.sort(arr1);
        Arrays.sort(arr2);
        // check whether the two arrays are true or false and return a boolean value
        return Arrays.equals(arr1, arr2);
    }
}
