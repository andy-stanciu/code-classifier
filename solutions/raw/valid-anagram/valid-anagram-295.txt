class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int ans = -1;
        char[] a = s.toCharArray();
        Arrays.sort(a);
        s = new String(a);
        char[] b = t.toCharArray();
        Arrays.sort(b);
        t = new String(b);
        ans = s.indexOf(t);
        if (ans > -1) {
            return true;
        } else {
            return false;
        }
    }
}
