class Solution {

    public boolean isAnagram(String s, String t) {
        char[] temp = s.toCharArray();
        char[] temp_1 = t.toCharArray();
        Arrays.sort(temp);
        Arrays.sort(temp_1);
        String str = new String(temp);
        String str_1 = new String(temp_1);
        return str.equals(str_1);
    }
}
