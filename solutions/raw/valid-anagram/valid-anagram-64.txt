class Solution {

    public boolean isAnagram(String s, String t) {
        char[] c = s.toCharArray();
        char[] c1 = t.toCharArray();
        Arrays.sort(c);
        Arrays.sort(c1);
        if (s.length() != t.length())
            return false;
        for (int i = 0; i < c.length; i++) {
            if (c[i] != c1[i])
                return false;
        }
        return true;
    }
}
