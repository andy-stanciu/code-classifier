class Solution {

    public boolean isAnagram(String s, String t) {
        ArrayList<String> arr = new ArrayList<String>(Arrays.asList(t.split("")));
        for (int i = 0; i < s.length(); i++) {
            if (arr.contains(s.charAt(i) + "")) {
                //if a letter contains it will remove from th  list
                arr.remove(arr.indexOf(s.charAt(i) + ""));
                continue;
            }
            return false;
        }
        if (// if arr is not empty then it is not a anagram
        arr.size() == 0) {
            return true;
        }
        return false;
    }
}
