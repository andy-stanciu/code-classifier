public class Solution {

    public boolean isAnagram(String s, String t) {
        // Step 1: Check if lengths are equal
        if (s.length() != t.length())
            return false;
        // Step 2: Initialize a character count map
        HashMap<Character, Integer> charCount = new HashMap<>();
        // Count characters in string s
        for (char c : s.toCharArray()) {
            // Count characters in s
            charCount.put(c, charCount.getOrDefault(c, 0) + 1);
        }
        // Step 3: Compare with characters in string t
        for (char c : t.toCharArray()) {
            // If char is not found
            if (!charCount.containsKey(c) || charCount.get(c) == 0)
                return false;
            // If char is there, reduce the count
            charCount.put(c, charCount.get(c) - 1);
        }
        // Step 4: If we complete the loop without returning false, return true
        return true;
    }
}
