class Solution {

    public boolean isAnagram(String s, String t) {
        int len = s.length();
        if (len != t.length())
            return false;
        int[] arr = new int[26];
        for (int i = 0; i < len; i++) {
            arr[s.charAt(i) - 97]++;
            arr[t.charAt(i) - 97]--;
        }
        for (int i = 0; i < 26; i++) if (arr[i] != 0)
            return false;
        return true;
    }
}
