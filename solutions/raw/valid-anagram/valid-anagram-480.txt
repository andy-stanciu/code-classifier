class Solution {

    public boolean isAnagram(String s, String t) {
        char[] sArray = s.toCharArray();
        char[] tArray = t.toCharArray();
        Arrays.sort(sArray);
        Arrays.sort(tArray);
        if (String.valueOf(sArray).equals(String.valueOf(tArray)))
            return true;
        else
            return false;
    }
}
