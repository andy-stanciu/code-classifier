class Solution {

    public boolean isAnagram(String s, String t) {
        if (t.length() < s.length())
            return false;
        HashMap<Character, Integer> parentString = new HashMap<Character, Integer>();
        for (int i = 0; i < s.length(); i++) {
            if (parentString.containsKey(s.charAt(i))) {
                parentString.put(s.charAt(i), parentString.get(s.charAt(i)) + 1);
            } else {
                parentString.put(s.charAt(i), 1);
            }
        }
        for (int i = 0; i < t.length(); i++) {
            if (!parentString.containsKey(t.charAt(i))) {
                return false;
            } else if (parentString.get(t.charAt(i)) == 1) {
                parentString.remove(t.charAt(i));
            } else {
                parentString.put(t.charAt(i), parentString.get(t.charAt(i)) - 1);
            }
        }
        return true;
    }
}
