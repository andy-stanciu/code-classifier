class Solution {

    public boolean isAnagram(String s, String t) {
        char[] s1 = s.toCharArray();
        char[] t1 = t.toCharArray();
        Arrays.sort(s1);
        Arrays.sort(t1);
        String s2 = String.valueOf(s1);
        String t2 = String.valueOf(t1);
        return s2.equals(t2);
    }
}
