class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() == t.length()) {
            HashMap<Character, Integer> record = new HashMap<>();
            for (char c : s.toCharArray()) {
                if (record.containsKey(c)) {
                    int v = record.get(c);
                    record.put(c, v + 1);
                } else {
                    record.put(c, 1);
                }
            }
            for (char c : t.toCharArray()) {
                if (record.containsKey(c)) {
                    int v = record.get(c);
                    record.put(c, v - 1);
                    if ((v - 1) < 0) {
                        return false;
                    }
                } else {
                    return false;
                }
            }
            Set<Character> keys = record.keySet();
            for (char c : keys) {
                int v = record.get(c);
                if (v != 0) {
                    return false;
                }
            }
            return true;
        }
        return false;
    }
}
