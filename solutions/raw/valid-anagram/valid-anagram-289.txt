class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length() || s.length() == 0)
            return false;
        if (s.length() == 1 && s.charAt(0) != t.charAt(0))
            return false;
        if (s.length() == 1 && s.charAt(0) == t.charAt(0))
            return true;
        char[] stemp = s.toCharArray();
        char[] ttemp = t.toCharArray();
        Arrays.sort(stemp);
        Arrays.sort(ttemp);
        s = String.valueOf(stemp);
        t = String.valueOf(ttemp);
        int endIndex = s.length() - 1;
        for (int i = 0; i < s.length() / 2; i++) {
            if (s.charAt(i) != t.charAt(i))
                return false;
            if (s.charAt(endIndex) != t.charAt(endIndex))
                return false;
            endIndex--;
        }
        return true;
    }
}
