class Solution {

    public boolean isAnagram(String s, String t) {
        HashMap<Character, Integer> hm = new HashMap<>();
        int n = s.length();
        int m = t.length();
        if (n != m)
            return false;
        for (int i = 0; i < n; i++) {
            char ss = s.charAt(i);
            char tt = t.charAt(i);
            hm.put(ss, hm.getOrDefault(ss, 0) + 1);
            hm.put(tt, hm.getOrDefault(tt, 0) - 1);
        }
        for (char c : s.toCharArray()) {
            if (hm.get(c) != 0)
                return false;
        }
        return true;
    }
}
