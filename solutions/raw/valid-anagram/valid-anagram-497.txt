class Solution {

    public boolean isAnagram(String s, String t) {
        char[] ch = s.toCharArray();
        char[] ch1 = t.toCharArray();
        Arrays.sort(ch);
        Arrays.sort(ch1);
        String str = Arrays.toString(ch);
        String str1 = Arrays.toString(ch1);
        if (str.equals(str1)) {
            return true;
        } else {
            return false;
        }
    }
}
