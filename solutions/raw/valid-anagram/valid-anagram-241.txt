class Solution {

    public boolean isAnagram(String s, String t) {
        //first of all check if the length of two strings is equal if no, return false;
        if (s.length() != t.length())
            return false;
        //Now create a map to store the character and its occurance in a string
        Map<Character, Integer> map = new HashMap<>();
        //Now create a loop and add all the characters of a string in it
        for (char c : s.toCharArray()) {
            //now get the char at present index in first string
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        //Now create one more loop to check if the characters of it are present in map and check
        //their frequecies
        for (char c : t.toCharArray()) {
            if (!map.containsKey(c)) {
                return false;
            } else {
                map.put(c, map.get(c) - 1);
                if (map.get(c) == 0) {
                    map.remove(c);
                }
            }
        }
        //if map is empty means strings are anagram else not anagram
        //fingers crossed!
        return map.isEmpty();
    }
}
