class Solution {

    public boolean isAnagram(String s, String t) {
        char[] chrs = s.toCharArray();
        char[] chrs1 = t.toCharArray();
        Arrays.sort(chrs);
        Arrays.sort(chrs1);
        for (int i = 0; i < chrs.length; i++) {
            if (chrs.length != chrs1.length) {
                return false;
            }
            if (chrs[i] != chrs1[i]) {
                return false;
            }
        }
        return true;
    }
}
