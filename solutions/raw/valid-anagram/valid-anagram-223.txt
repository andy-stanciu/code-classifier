class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        String sorted1 = sortinglogic(s);
        String sorted2 = sortinglogic(t);
        if (!sorted1.equals(sorted2)) {
            return false;
        }
        return true;
    }

    public String sortinglogic(String str) {
        char[] ch = str.toCharArray();
        Arrays.sort(ch);
        return new String(ch);
    }
}
