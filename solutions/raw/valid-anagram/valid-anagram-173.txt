class Solution {

    public boolean isAnagram(String s, String t) {
        int[] arrS = new int[26];
        int[] arrT = new int[26];
        for (int i = 0; i < s.length(); i++) {
            int index = ((int) s.charAt(i)) - 97;
            arrS[index] += 1;
        }
        for (int i = 0; i < t.length(); i++) {
            int index = ((int) t.charAt(i)) - 97;
            arrT[index] += 1;
        }
        boolean result = true;
        for (int i = 0; i < 26; i++) if (arrS[i] != arrT[i])
            result = false;
        return result;
    }
}
