class Solution {

    public boolean isAnagram(String s, String t) {
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        int n1 = a.length, n2 = b.length;
        if (n1 != n2)
            return false;
        Arrays.sort(a);
        Arrays.sort(b);
        int i = 0, j = 0;
        boolean flag = true;
        while (i < n1 && j < n2) {
            if (a[i] != b[j]) {
                flag = false;
                break;
            }
            i++;
            j++;
        }
        return flag;
    }
}
