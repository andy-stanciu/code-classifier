class Solution {

    public boolean isAnagram(String s, String t) {
        boolean status = false;
        if (s.length() != t.length()) {
            status = false;
        } else {
            char[] arr1 = s.toLowerCase().toCharArray();
            char[] arr2 = t.toLowerCase().toCharArray();
            Arrays.sort(arr1);
            Arrays.sort(arr2);
            status = Arrays.equals(arr1, arr2);
        }
        return status;
    }
}
