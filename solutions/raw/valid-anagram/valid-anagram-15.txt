class Solution {

    public boolean isAnagram(String s, String t) {
        HashMap<Character, Integer> characterMap = new HashMap<>();
        // this loop stores frequencies of characters of string s
        for (char ch : s.toCharArray()) {
            // System.out.println("Created hashmap is" + characterMap);
            if (characterMap.containsKey(ch)) {
                characterMap.put(ch, characterMap.get(ch) + 1);
            } else {
                characterMap.put(ch, 1);
            }
        }
        // System.out.println("Created hashmap is" + characterMap);
        // this loop matches the character with hashmap and updates the count
        for (char ch : t.toCharArray()) {
            if (characterMap.containsKey(ch)) {
                characterMap.put(ch, characterMap.get(ch) - 1);
            } else {
                return false;
            }
        }
        // this loop checks if there\'s any character that\'s still not 0 in the map
        for (Map.Entry<Character, Integer> me : characterMap.entrySet()) {
            // char key = me.getKey();
            if (me.getValue() != 0) {
                return false;
            }
        }
        return true;
    }
}
