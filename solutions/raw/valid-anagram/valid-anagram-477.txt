class Solution {

    public boolean isAnagram(String s, String t) {
        HashMap<Character, Integer> counter = new HashMap<>();
        if (s.length() != t.length())
            return false;
        int n = s.length();
        for (int i = 0; i < n; i++) {
            char ch = s.charAt(i);
            if (counter.containsKey(ch)) {
                counter.put(ch, counter.get(ch) + 1);
            } else {
                counter.put(ch, 1);
            }
        }
        for (int i = 0; i < n; i++) {
            char ch = t.charAt(i);
            if (counter.containsKey(ch)) {
                counter.put(ch, counter.get(ch) - 1);
            } else {
                return false;
            }
        }
        for (var item : counter.values()) {
            if (item != 0) {
                return false;
            }
        }
        return true;
    }
}
