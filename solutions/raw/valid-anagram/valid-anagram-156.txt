class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length())
            return false;
        int[] ascArr = new int[123];
        for (int i = 0; i < s.length(); i++) {
            ascArr[s.charAt(i)] += 1;
        }
        for (int i = 0; i < t.length(); i++) {
            if (ascArr[t.charAt(i)] <= 0) {
                return false;
            } else {
                ascArr[t.charAt(i)] -= 1;
            }
        }
        return true;
    }
}
