class Solution {

    public boolean isAnagram(String s, String t) {
        char[] n = s.toCharArray();
        Arrays.sort(n);
        int k = n.length;
        char[] m = t.toCharArray();
        Arrays.sort(m);
        int l = m.length;
        if (k != l) {
            return false;
        }
        for (int i = 0; i < k; i++) {
            if (n[i] != m[i]) {
                return false;
            }
        }
        return true;
    }
}
