class Solution {

    public boolean isAnagram(String s, String t) {
        // For anagram to be valid, both strings have to be the same length to ensure all
        // characters are used in s and t.
        if (s.length() != t.length()) {
            return false;
        }
        // Convert to char array
        char[] sCharacters = s.toCharArray();
        char[] tCharacters = t.toCharArray();
        // Sorting here is necessary for elementwise comparison in the created char array.
        // Preventing the need to check for the frequency of characters in both strings.
        Arrays.sort(sCharacters);
        Arrays.sort(tCharacters);
        // Use equals to compare element by element and check if equal
        return Arrays.equals(sCharacters, tCharacters);
    }
}
