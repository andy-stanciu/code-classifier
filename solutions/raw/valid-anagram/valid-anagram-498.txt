class Solution {

    public boolean isAnagram(String s, String t) {
        HashMap<Character, Integer> c = new HashMap<>();
        for (char str : s.toCharArray()) {
            c.put(str, c.getOrDefault(str, 0) + 1);
        }
        for (char str : t.toCharArray()) {
            c.put(str, c.getOrDefault(str, 0) - 1);
        }
        for (Integer val : c.values()) {
            if (val != 0) {
                return false;
            }
        }
        return true;
    }
}
