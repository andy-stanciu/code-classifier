class Solution {

    public boolean isAnagram(String s, String t) {
        int[] a = new int[26];
        if (s.length() != t.length()) {
            return false;
        }
        for (int i = 0; i < s.length(); i++) {
            //int x=(int) s.charAt(i)-97;
            a[(int) s.charAt(i) - 97]++;
        }
        for (int i = 0; i < t.length(); i++) {
            //int x=(int) t.charAt(i)-97;
            if (a[(int) t.charAt(i) - 97] == 0) {
                return false;
            }
            a[(int) t.charAt(i) - 97]--;
        }
        return true;
    }
}
