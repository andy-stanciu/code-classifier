class Solution {

    public boolean isAnagram(String s, String t) {
        int ls = s.length();
        int lt = t.length();
        if (ls == lt) {
            char[] lsa = new char[ls];
            char[] lta = new char[lt];
            int f = 0;
            //convert into string
            for (int i = 0; i < ls; i++) {
                lsa[i] = s.charAt(i);
                lta[i] = t.charAt(i);
            }
            Arrays.sort(lsa);
            Arrays.sort(lta);
            //comprasion
            for (int j = 0; j < ls; j++) {
                if (lsa[j] == lta[j]) {
                    f = 0;
                } else {
                    return false;
                }
            }
            if (f == 0) {
                return true;
            }
        } else {
            return false;
        }
        return false;
    }
}
