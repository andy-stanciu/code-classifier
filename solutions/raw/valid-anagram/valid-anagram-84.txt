class Solution {

    public boolean isAnagram(String s, String t) {
        // Step 1: Create a HashMap for character counting
        HashMap<Character, Integer> hm = new HashMap<>();
        // Iterate through characters of the first string
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            hm.put(ch, hm.getOrDefault(ch, 0) + 1);
        }
        // Step 2: Decrement counts in the second string
        for (int i = 0; i < t.length(); i++) {
            char ch = t.charAt(i);
            if (hm.containsKey(ch)) {
                if (hm.get(ch) == 1) {
                    hm.remove(ch);
                } else {
                    hm.put(ch, hm.get(ch) - 1);
                }
            } else {
                // Return false if the character is not present in the HashMap
                return false;
            }
        }
        // Step 3: Check if the HashMap is empty
        return hm.isEmpty();
    }
}
