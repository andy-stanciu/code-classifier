class Solution {

    public boolean isAnagram(String s, String t) {
        int sumS = 0, sumT = 0;
        int productS = 1, productT = 1;
        if (t.length() != s.length())
            return false;
        for (int i = 0; i < s.length(); i++) {
            sumS += s.charAt(i) - 96;
            productS *= s.charAt(i) - 96;
            sumT += t.charAt(i) - 96;
            productT *= t.charAt(i) - 96;
        }
        if (sumS != sumT || productS != productT)
            return false;
        return true;
    }
}
