class Solution {

    public boolean isAnagram(String s, String t) {
        char[] arr = s.toCharArray();
        Arrays.sort(arr);
        String ans = new String(arr);
        char[] arr1 = t.toCharArray();
        Arrays.sort(arr1);
        String ans1 = new String(arr1);
        if (s.length() != t.length()) {
            return false;
        }
        if (ans.equals(ans1)) {
            return true;
        }
        return false;
    }
}
