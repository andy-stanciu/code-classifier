class Solution {

    public boolean isAnagram(String s, String t) {
        boolean isAnagram = false;
        int[] array = new int[128];
        if (s.length() != t.length())
            return isAnagram;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == t.charAt(i))
                continue;
            array[s.charAt(i)]++;
            array[t.charAt(i)]--;
        }
        return Arrays.stream(array).allMatch(c -> c == 0);
    }
}
