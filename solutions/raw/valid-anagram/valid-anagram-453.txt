class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length())
            return false;
        int n = s.length();
        int[] map = new int[128];
        for (char c : s.toCharArray()) {
            map[c]++;
        }
        for (char c : t.toCharArray()) {
            map[c]--;
        }
        for (int count : map) {
            if (count != 0)
                return false;
        }
        return true;
    }
}
