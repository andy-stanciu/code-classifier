class Solution {

    public boolean isAnagram(String s, String t) {
        // Check if lengths are not equal
        if (s.length() != t.length()) {
            return false;
        } else {
            // Convert strings to lowercase character arrays
            char[] s1 = s.toLowerCase().toCharArray();
            char[] t1 = t.toLowerCase().toCharArray();
            // Sort character arrays
            Arrays.sort(s1);
            Arrays.sort(t1);
            // Compare sorted arrays for equality
            if (Arrays.equals(s1, t1)) {
                // Return true if arrays are equal (anagrams)
                return true;
            } else {
                // Return false if arrays are not equal (not anagrams)
                return false;
            }
        }
    }
}
