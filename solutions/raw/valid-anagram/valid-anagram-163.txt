class Solution {

    public boolean isAnagram(String s, String t) {
        char[] sChars = s.toCharArray();
        Arrays.sort(sChars);
        char[] tChars = t.toCharArray();
        Arrays.sort(tChars);
        int res = Arrays.compare(sChars, tChars);
        if (res == 0)
            return true;
        return false;
    }
}
