class Solution {

    public boolean isAnagram(String s, String t) {
        Scanner sc = new Scanner(System.in);
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        if (Arrays.equals(a, b)) {
            return true;
        } else {
            return false;
        }
    }
}
