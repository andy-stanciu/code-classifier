class Solution {

    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        var len = s.length();
        Map<Character, Integer> m = new HashMap<>();
        char sChar, tChar;
        for (int i = 0; i < len; i++) {
            sChar = s.charAt(i);
            tChar = t.charAt(i);
            m.put(sChar, m.getOrDefault(sChar, 0) + 1);
            m.put(tChar, m.getOrDefault(tChar, 0) - 1);
        }
        for (var entry : m.entrySet()) {
            if (entry.getValue() != 0) {
                return false;
            }
        }
        return true;
    }
}
