class Solution {

    public String longestPalindrome(String s) {
        int n = s.length();
        int max = 0;
        int left = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                if (check(s, i, j)) {
                    if (j - i + 1 > max) {
                        max = j - i + 1;
                        left = i;
                    }
                }
            }
        }
        return s.substring(left, left + max);
    }

    public boolean check(String s, int i, int j) {
        while (i < j) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            } else {
                i++;
                j--;
            }
        }
        return true;
    }
}
