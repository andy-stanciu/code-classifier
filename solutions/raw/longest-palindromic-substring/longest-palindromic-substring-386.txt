class Solution {

    public String longestPalindrome(String s) {
        String p = "";
        int n = s.length();
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            //odd
            int l = i;
            int r = i;
            while (l >= 0 && r < n && s.charAt(l) == s.charAt(r)) {
                if (r - l + 1 > max) {
                    max = r - l + 1;
                    p = s.substring(l, r + 1);
                }
                l--;
                r++;
            }
            //even
            l = i;
            r = i + 1;
            while (l >= 0 && r < n && s.charAt(l) == s.charAt(r)) {
                if (r - l + 1 > max) {
                    max = r - l + 1;
                    p = s.substring(l, r + 1);
                }
                l--;
                r++;
            }
        }
        return p;
    }
}
