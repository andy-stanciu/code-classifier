class Solution {

    int maxLen = 0;

    int lo = 0;

    public String longestPalindrome(String s) {
        char[] arr = s.toCharArray();
        if (arr.length < 2) {
            return s;
        }
        for (int i = 0; i < arr.length; i++) {
            expandP(arr, i, i);
            expandP(arr, i, i + 1);
        }
        return s.substring(lo, lo + maxLen);
    }

    public void expandP(char[] arr, int l, int h) {
        while (l >= 0 && h < arr.length && arr[l] == arr[h]) {
            l--;
            h++;
        }
        if (maxLen < h - l - 1) {
            maxLen = h - l - 1;
            lo = l + 1;
        }
    }
}
