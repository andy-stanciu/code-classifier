class Solution {

    public String longestPalindrome(String s) {
        String longestStr = "";
        for (int i = 0; i < s.length(); i++) {
            String oddPalin = checkPalindrome(s, i, i);
            if (oddPalin.length() > longestStr.length()) {
                longestStr = oddPalin;
            }
            String evenPalin = checkPalindrome(s, i, i + 1);
            if (evenPalin.length() > longestStr.length()) {
                longestStr = evenPalin;
            }
        }
        return longestStr;
    }

    public String checkPalindrome(String s, int left, int right) {
        while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
            left--;
            right++;
        }
        return s.substring(left + 1, right);
    }
}
