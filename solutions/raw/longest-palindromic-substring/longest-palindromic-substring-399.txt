class Solution {

    public String longestPalindrome(String s) {
        int n = s.length();
        if (n < 2) {
            return s;
        }
        String str = s.substring(0, 1);
        int len = 1;
        for (int i = 0; i < n; i++) {
            int f = i, l = i;
            while (f >= 0 && l < n && s.charAt(f) == s.charAt(l)) {
                if (l - f + 1 > len) {
                    str = s.substring(f, l + 1);
                    len = l - f + 1;
                }
                f--;
                l++;
            }
            f = i;
            l = i + 1;
            while (f >= 0 && l < n && s.charAt(f) == s.charAt(l)) {
                if (l - f + 1 > len) {
                    str = s.substring(f, l + 1);
                    len = l - f + 1;
                }
                f--;
                l++;
            }
        }
        return str;
    }
}
