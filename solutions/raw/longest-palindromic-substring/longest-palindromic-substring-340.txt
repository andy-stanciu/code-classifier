class Solution {

    public String longestPalindrome(String s) {
        int start = 0, end = 1;
        int l = 0, r = 0;
        for (int i = 0; i < s.length(); i++) {
            //even
            l = i;
            r = i + 1;
            while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
                if (r - l + 1 > end) {
                    start = l;
                    end = r - l + 1;
                }
                l--;
                r++;
            }
            //odd
            l = i - 1;
            r = i + 1;
            while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
                if (r - l + 1 > end) {
                    start = l;
                    end = r - l + 1;
                }
                l--;
                r++;
            }
        }
        StringBuilder sb = new StringBuilder("");
        for (int i = start; i < start + end; i++) {
            sb.append(s.charAt(i));
        }
        return sb.toString();
    }
}
