class Solution {

    public String longestPalindrome(String s) {
        for (int i = s.length() - 1; true; i--) for (int j = 0; j < s.length() - i; j++) if (isPalindrome(s.substring(j, i + j + 1)))
            return s.substring(j, i + j + 1);
    }

    public boolean isPalindrome(String str) {
        for (int i = 0; i < str.length() / 2; i++) if (str.charAt(i) != str.charAt(str.length() - 1 - i))
            return false;
        return true;
    }
}
