class Solution {

    public String longestPalindrome(String s) {
        int n = s.length();
        int start = 0;
        int end = 1;
        for (int i = 0; i < n; i++) {
            int low = i - 1;
            int high = i;
            while (low >= 0 && high < n && s.charAt(low) == s.charAt(high)) {
                if (high - low + 1 > end) {
                    start = low;
                    end = high - low + 1;
                }
                low--;
                high++;
            }
            low = i - 1;
            high = i + 1;
            while (low >= 0 && high < n && s.charAt(low) == s.charAt(high)) {
                if (high - low + 1 > end) {
                    start = low;
                    end = high - low + 1;
                }
                low--;
                high++;
            }
        }
        return s.substring(start, end + start);
    }
}
