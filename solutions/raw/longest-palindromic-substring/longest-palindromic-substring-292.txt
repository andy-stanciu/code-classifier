class Solution {

    public static void main(String[] args) {
        String na = longestPalindrome("babad");
        System.out.println(na);
    }

    public static String longestPalindrome(String s) {
        int l = s.length();
        String na = "";
        String p = "";
        int max = 0;
        for (int i = 0; i < l; i++) {
            for (int j = i + 1; j <= l; j++) {
                na = s.substring(i, j);
                if (isPalindromic(na) && na.length() > max) {
                    max = na.length();
                    p = na;
                }
            }
        }
        return p;
    }

    static boolean isPalindromic(String s) {
        int i = 0;
        int j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i) != s.charAt(j))
                return false;
            i++;
            j--;
        }
        return true;
    }
}
