class Solution {

    public String longestPalindrome(String s) {
        String retVal = "";
        int left = 0;
        int right = s.length() - 1;
        boolean stop = false;
        for (int i = left, j = right; j < s.length(); j--) {
            if (stop)
                break;
            for (int a = i, b = j; b < s.length(); a++, b++) {
                if (stop)
                    break;
                int innerLeft = a;
                int innerRight = b;
                while (true) {
                    if (stop)
                        break;
                    if (innerLeft >= innerRight) {
                        stop = Boolean.TRUE;
                        retVal = s.substring(a, b + 1);
                        break;
                    }
                    if (s.charAt(innerLeft) == s.charAt(innerRight)) {
                        innerLeft++;
                        innerRight--;
                    } else {
                        break;
                    }
                }
            }
        }
        return retVal;
    }
}
