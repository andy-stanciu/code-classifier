class Solution {

    public String longestPalindrome(String s) {
        String longestSymetry = "";
        for (int border = 0; border < s.length(); border++) {
            // 1
            String symetry = findLargestSymetry(border, border, s);
            if (symetry.length() >= longestSymetry.length()) {
                longestSymetry = symetry;
            }
            // 2
            symetry = findLargestSymetry(border, border + 1, s);
            if (symetry.length() >= longestSymetry.length()) {
                longestSymetry = symetry;
            }
        }
        return longestSymetry;
    }

    public String findLargestSymetry(int left, int right, String s) {
        String result = "";
        while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
            if (right - left + 1 >= result.length()) {
                // +1 because indexes are starting by 0
                result = s.substring(left, right + 1);
            }
            left--;
            right++;
        }
        return result;
    }
}
