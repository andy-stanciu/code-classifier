class Solution {

    public String longestPalindrome(String s) {
        int n = s.length();
        String LPS = "";
        int maxlength = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                String S = s.substring(i, j + 1);
                if (isPalindrome(S) && S.length() > maxlength) {
                    LPS = S;
                    maxlength = S.length();
                }
            }
        }
        return LPS;
    }

    public boolean isPalindrome(String s) {
        int n = s.length() - 1;
        int low = 0;
        int high = n;
        while (low < high) {
            if (s.charAt(low) != s.charAt(high)) {
                return false;
            }
            low++;
            high--;
        }
        return true;
    }
}
