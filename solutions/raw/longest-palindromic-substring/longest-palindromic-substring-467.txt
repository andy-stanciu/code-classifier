class Solution {

    public String longestPalindrome(String s) {
        int[][] mat = new int[s.length()][s.length()];
        int start = 0, end = 0;
        for (int i = 1; i <= mat.length; i++) {
            for (int j = 0; j + i - 1 < mat.length; j++) {
                if (i > 2) {
                    if (s.charAt(j) == s.charAt(j + i - 1)) {
                        if (mat[j + 1][j + i - 2] != 0) {
                            mat[j][j + i - 1] = 1;
                            start = j;
                            end = j + i - 1;
                        }
                    }
                } else {
                    if (s.charAt(j) == s.charAt(j + i - 1)) {
                        mat[j][j + i - 1] = 1;
                        start = j;
                        end = j + i - 1;
                    }
                }
            }
        }
        return s.substring(start, end + 1);
    }
}
