class Solution {

    public String longestPalindrome(String s) {
        if (s == null || s.length() < 1)
            return "";
        int sLen = s.length();
        int startIdx = 0, len = 0;
        for (int idx = 0; idx < sLen; idx++) {
            // odd length palindromes
            int len1 = expandAroundCenter(s, idx, idx);
            // even length palindromes
            int len2 = expandAroundCenter(s, idx, idx + 1);
            int maxLen = Math.max(len1, len2);
            if (maxLen > len) {
                startIdx = idx - (maxLen - 1) / 2;
                len = maxLen;
            }
        }
        return s.substring(startIdx, startIdx + len);
    }

    private int expandAroundCenter(String s, int left, int right) {
        int sLen = s.length();
        while (left >= 0 && right < sLen && s.charAt(left) == s.charAt(right)) {
            left--;
            right++;
        }
        // length of the palindrome
        return right - left - 1;
    }
}
