class Solution {

    public String longestPalindrome(String s) {
        String largestPalindrome = s.substring(0, 1);
        for (int i = 0; i < s.length(); i++) {
            char currentChar = s.charAt(i);
            int foundMatchAtIndex = i;
            while (foundMatchAtIndex < s.length()) {
                foundMatchAtIndex++;
                if (foundMatchAtIndex < s.length() && currentChar == s.charAt(foundMatchAtIndex)) {
                    String currentSubstring = s.substring(i, foundMatchAtIndex + 1);
                    if (isPalindrome(currentSubstring)) {
                        if (currentSubstring.length() > largestPalindrome.length()) {
                            largestPalindrome = currentSubstring;
                        }
                    }
                }
            }
        }
        return largestPalindrome;
    }

    private boolean isPalindrome(String substring) {
        return substring.equals(new StringBuilder(substring).reverse().toString());
    }
}
