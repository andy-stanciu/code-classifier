class Solution {

    public String longestPalindrome(String s) {
        if (s.length() == 1)
            return s;
        else {
            for (int finale = s.length(); finale > 0; finale--) {
                for (int principio = 0; principio <= s.length() - finale; principio++) {
                    if (palindromeChecker(principio, finale + principio, s)) {
                        return s.substring(principio, finale + principio);
                    }
                }
            }
        }
        return "";
    }

    private boolean palindromeChecker(int i, int j, String s) {
        int principio = i;
        int finale = j - 1;
        while (principio < finale) {
            if (s.charAt(principio) != s.charAt(finale)) {
                return false;
            }
            principio++;
            finale--;
        }
        return true;
    }
}
