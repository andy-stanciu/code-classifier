class Solution {

    public int strStr(String haystack, String needle) {
        char firstChar = needle.charAt(0);
        int lastIndex = haystack.length() - needle.length();
        for (int i = 0; i <= lastIndex; i++) {
            if (haystack.charAt(i) == firstChar) {
                if (subString(haystack, i, needle)) {
                    return i;
                }
            }
        }
        return -1;
    }

    private boolean subString(String haystack, int index, String needle) {
        int i = 0;
        for (; i < needle.length() && index + i < haystack.length(); i++) {
            if (haystack.charAt(index + i) != needle.charAt(i)) {
                return false;
            }
        }
        return i == needle.length();
    }
}
