class Solution {

    public int strStr(String haystack, String needle) {
        int n = haystack.length();
        int m = needle.length();
        if (m == 0) {
            return -1;
        }
        for (int i = 0; i < n; i++) {
            if (m == 0 && haystack.charAt(i) == needle.charAt(0)) {
                return i;
            }
            if (haystack.charAt(i) == needle.charAt(0) && i + (m - 1) < n) {
                if (haystack.substring(i, i + m).equals(needle)) {
                    return i;
                }
            }
        }
        return -1;
    }
}
