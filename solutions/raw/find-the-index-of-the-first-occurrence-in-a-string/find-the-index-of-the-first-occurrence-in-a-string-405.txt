class Solution {

    public int strStr(String haystack, String needle) {
        int m = haystack.length();
        int n = needle.length();
        for (int i = 0; i <= m - n; i++) {
            // haystack loop
            for (int j = 0; j < n; j++) {
                // needle loop
                if (haystack.charAt(i + j) != needle.charAt(j)) {
                    break;
                }
                if (j == n - 1) {
                    // Return the starting index of the match
                    return i;
                }
            }
        }
        // Return -1 if no match is found
        return -1;
    }
}
