class Solution {

    public int strStr(String haystack, String needle) {
        int i, j = 0;
        for (i = 0; i < haystack.length(); i++) {
            if (j == needle.length()) {
                return i - j;
            }
            if (haystack.charAt(i) == needle.charAt(j)) {
                j++;
            } else {
                i = i - j;
                j = 0;
            }
        }
        if (j == needle.length()) {
            return i - j;
        }
        return -1;
    }
}
