class Solution {

    public int strStr(String haystack, String needle) {
        int ans = -1;
        if (haystack.equals(needle))
            return 0;
        int n = haystack.length() - needle.length();
        if (n < 0)
            return ans;
        for (int i = 0; i < n + 1; i++) {
            String str = haystack.substring(i, i + needle.length());
            if (str.equals(needle)) {
                ans = i;
                break;
            }
        }
        return ans;
    }
}
