class Solution {

    public int strStr(String haystack, String needle) {
        int n = needle.length();
        int h = haystack.length();
        if (h < n) {
            return -1;
        }
        for (int i = 0; i <= h - n; i++) {
            if (haystack.substring(i, i + n).equals(needle)) {
                return i;
            }
        }
        return -1;
    }
}
