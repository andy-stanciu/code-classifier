class Solution {

    public int strStr(String haystack, String needle) {
        if (needle == "")
            return 0;
        for (int i = 0; i + (needle.length() - 1) < haystack.length(); i++) {
            if (needle.equals(haystack.substring(i, i + (needle.length())))) {
                return i;
            }
        }
        return -1;
    }
}
