class Solution {

    public int strStr(String haystack, String needle) {
        int m = haystack.length();
        int n = needle.length();
        for (int i = 0; i <= m - n; i++) {
            int first = i, second = 0;
            while (second < n) {
                if (haystack.charAt(first) != needle.charAt(second)) {
                    break;
                } else {
                    first++;
                    second++;
                }
                if (second == needle.length())
                    return first - second;
            }
        }
        return -1;
    }
}
