class Solution {

    public int strStr(String haystack, String needle) {
        int needleIndex = 0;
        int needleLength = needle.length();
        int index = 0;
        int haystackLength = haystack.length();
        while (index < haystackLength) {
            //haystack.charAt(index) == needle.charAt(needleIndex)
            if (haystack.charAt(index) == needle.charAt(needleIndex)) {
                needleIndex++;
            } else {
                index = index - needleIndex;
                needleIndex = 0;
            }
            if (needleIndex >= needleLength) {
                return index - (needleLength - 1);
            }
            index++;
        }
        return -1;
    }
}
