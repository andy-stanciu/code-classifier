class Solution {

    public int strStr(String haystack, String needle) {
        int nSize = needle.length();
        int nFoundAt = -1;
        for (int i = 0; i <= (haystack.length() - nSize); i++) {
            if (haystack.substring(i, i + nSize).equals(needle)) {
                nFoundAt = i;
                break;
            }
        }
        return nFoundAt;
    }
}
