class Solution {

    public int strStr(String haystack, String needle) {
        if (needle.length() > haystack.length()) {
            return -1;
        }
        int aPointer = 0;
        int bPointer = 0;
        int result = 0;
        while (aPointer < haystack.length() && bPointer < needle.length()) {
            if (haystack.charAt(aPointer) == needle.charAt(bPointer)) {
                aPointer++;
                bPointer++;
            } else {
                bPointer = 0;
                aPointer = result + 1;
                result = aPointer;
            }
        }
        return bPointer < needle.length() ? -1 : result;
    }
}
