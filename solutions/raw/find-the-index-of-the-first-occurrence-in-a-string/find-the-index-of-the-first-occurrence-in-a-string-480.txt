class Solution {

    public int strStr(String longStr, String shortStr) {
        if (longStr.length() < shortStr.length())
            return -1;
        char[] longChar = longStr.toCharArray();
        char[] shortChar = shortStr.toCharArray();
        int length = 0;
        int retIndexOfStartWord = 0;
        for (int longIndex = 0; longIndex < longChar.length; longIndex++) {
            if (longChar[longIndex] == shortChar[0]) {
                retIndexOfStartWord = longIndex;
                for (int i = 0; i < shortChar.length && longIndex < longChar.length; i++) {
                    if (longChar[longIndex] != shortChar[i])
                        break;
                    length++;
                    longIndex++;
                }
                if (length == shortChar.length)
                    return retIndexOfStartWord;
                longIndex = retIndexOfStartWord;
                length = 0;
            }
        }
        return -1;
    }
}
