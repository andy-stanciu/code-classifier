class Solution {

    public int strStr(String haystack, String needle) {
        //intialize result to -1
        int result = -1;
        if (haystack.contains(needle)) {
            //check haystack contains needle or not
            // get the index of haystack
            int index = haystack.indexOf(needle);
            // if "YES" , assign the index of needle to result
            result = index;
        }
        //if "NO", return -1
        return result;
    }
}
