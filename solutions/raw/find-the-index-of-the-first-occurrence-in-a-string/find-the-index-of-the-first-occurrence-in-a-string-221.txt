class Solution {

    public int strStr(String haystack, String needle) {
        int n = needle.length();
        for (int i = 0; i <= haystack.length() - n; i++) {
            String substring = haystack.substring(i, i + n);
            if (substring.equals(needle)) {
                return i;
            }
        }
        return -1;
    }
}
