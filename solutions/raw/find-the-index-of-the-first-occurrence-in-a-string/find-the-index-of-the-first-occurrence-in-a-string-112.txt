class Solution {

    public int strStr(String s, String g) {
        if (s == null || g == null)
            return -1;
        if (s == g)
            return 0;
        int n = g.length();
        int h = s.length();
        for (int i = 0; i < (h - n + 1); i++) {
            if (s.substring(i, i + n).equals(g)) {
                return i;
            }
        }
        return -1;
    }
}
