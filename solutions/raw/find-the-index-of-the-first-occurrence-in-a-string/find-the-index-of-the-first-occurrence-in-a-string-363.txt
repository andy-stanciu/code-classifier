class Solution {

    public int strStr(String haystack, String needle) {
        int j = needle.length() - 1;
        int hLen = haystack.length() - j;
        for (int k = 0; k < hLen; k++) {
            String sub = haystack.substring(k, j + 1);
            if (sub.equals(needle)) {
                return k;
            }
            j++;
        }
        return -1;
    }
}
