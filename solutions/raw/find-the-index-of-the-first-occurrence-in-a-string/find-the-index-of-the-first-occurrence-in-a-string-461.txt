class Solution {

    public int strStr(String haystack, String needle) {
        if (haystack.contains(needle)) {
            int j = 0;
            boolean matchNotFound = true;
            int iIndex = haystack.indexOf(needle.charAt(0));
            int index = haystack.indexOf(needle.charAt(0));
            while (matchNotFound) {
                boolean isIndexCorrect = true;
                for (int i = 0; i < needle.length(); i++) {
                    if (haystack.charAt(iIndex) != needle.charAt(j)) {
                        isIndexCorrect = false;
                        index = haystack.indexOf(needle.charAt(0), index + 1);
                        iIndex = index;
                        j = 0;
                        break;
                    }
                    iIndex++;
                    j++;
                }
                if (isIndexCorrect) {
                    matchNotFound = false;
                }
            }
            return index;
        } else {
            return -1;
        }
    }
}
