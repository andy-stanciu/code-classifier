class Solution {

    public int strStr(String haystack, String needle) {
        int pointerOne = 0, pointerTwo = 0;
        int lastMet = -1;
        while (pointerOne < needle.length() && pointerTwo < haystack.length()) {
            if (needle.charAt(pointerOne) == haystack.charAt(pointerTwo)) {
                pointerOne++;
                pointerTwo++;
            } else if (pointerOne == 0)
                pointerTwo++;
            else {
                pointerTwo = pointerTwo - pointerOne + 1;
                pointerOne = 0;
            }
        }
        if (pointerOne == needle.length())
            lastMet = pointerTwo - pointerOne;
        return lastMet;
    }
}
