class Solution {

    public int strStr(String haystack, String needle) {
        char[] ch1 = haystack.toCharArray();
        char[] ch2 = needle.toCharArray();
        int i, j;
        for (i = 0, j = 0; i < ch1.length; i++) {
            if (ch1[i] == ch2[j]) {
                j++;
            } else {
                i = i - j;
                j = 0;
            }
            if (j == ch2.length) {
                return i - j + 1;
            }
        }
        return -1;
    }
}
