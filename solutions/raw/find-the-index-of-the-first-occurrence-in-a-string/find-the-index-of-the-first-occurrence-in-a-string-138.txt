class Solution {

    public int strStr(String haystack, String needle) {
        int l1 = haystack.length();
        int l2 = needle.length();
        for (int i = 0; i < l1; i++) {
            try {
                if (haystack.substring(i, l2 + i).equals(needle)) {
                    return i;
                }
            } catch (Exception ignored) {
            }
        }
        return -1;
    }
}
