class Solution {

    public int strStr(String haystack, String needle) {
        int result = -1;
        int pointer = 0;
        int right = needle.length();
        while (right <= haystack.length()) {
            if (haystack.substring(pointer, right).equals(needle)) {
                result = pointer;
                break;
            } else {
                pointer++;
                right++;
            }
        }
        return result;
    }
}
