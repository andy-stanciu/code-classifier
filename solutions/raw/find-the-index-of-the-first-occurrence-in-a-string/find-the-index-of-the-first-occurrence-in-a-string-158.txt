class Solution {

    public int strStr(String haystack, String needle) {
        if (haystack.equals(needle)) {
            return 0;
        }
        int startIndex = 0;
        int endIndex = needle.length();
        while (endIndex <= haystack.length()) {
            if (haystack.substring(startIndex, endIndex).equals(needle)) {
                //found
                return startIndex;
            }
            startIndex++;
            endIndex++;
        }
        return -1;
    }
}
