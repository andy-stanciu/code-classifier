class Solution {

    public int strStr(String haystack, String needle) {
        if (!haystack.contains(needle))
            return -1;
        else {
            int m = haystack.length();
            int n = needle.length();
            for (int i = 0; i < m; i++) if (haystack.substring(i, i + n).equals(needle))
                return i;
        }
        return -1;
    }
}
