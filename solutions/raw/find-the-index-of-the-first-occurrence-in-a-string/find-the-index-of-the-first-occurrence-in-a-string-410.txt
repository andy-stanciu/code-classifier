class Solution {

    public int strStr(String haystack, String needle) {
        int N = haystack.length();
        int M = needle.length();
        for (int i = 0; i < N - M + 1; i++) {
            String str = "";
            for (int j = i; j < i + M; j++) {
                str = str + haystack.charAt(j);
            }
            if (str.equals(needle)) {
                return i;
            }
        }
        return -1;
    }
}
