class Solution {

    public int strStr(String haystack, String needle) {
        int n = needle.length();
        int m = haystack.length();
        int ans = -1;
        for (int i = 0; i <= m - n; i++) {
            String s = haystack.substring(i, i + n);
            if (s.equals(needle)) {
                ans = i;
                break;
            }
        }
        return ans;
    }
}
