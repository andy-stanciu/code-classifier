class Solution {

    public int strStr(String haystack, String needle) {
        int n = haystack.length(), m = needle.length(), lo = 0;
        for (int hi = m - 1; hi < n; hi++) {
            if (haystack.substring(lo, hi + 1).equals(needle)) {
                return lo;
            }
            lo++;
        }
        return -1;
    }
}
