class Solution {

    public String convert(String s, int numRows) {
        //base cases
        if (numRows >= s.length() || numRows == 1) {
            return s;
        }
        //StringBuilder to return
        StringBuilder sb = new StringBuilder();
        //index in string
        int index = 0;
        //keep track of current row
        int row = 0;
        //first amount to increment string index by
        int fstInc = (numRows - 1) * 2;
        //second amount to increment string index by
        int secInc = 0;
        //bool switch between fstInc and secInc
        boolean doFst = true;
        int length = s.length();
        while (row != numRows) {
            //iterates through each row
            if (index >= length) {
                //reset point if index is out of bounds
                //move to next row and set index to that row num
                row++;
                index = row;
                doFst = true;
                //increment the increments
                fstInc -= 2;
                secInc += 2;
            } else {
                if (doFst) {
                    //switch points to fstInc
                    if (index < length && fstInc != 0) {
                        //add to stringBuilder
                        sb.append(s.charAt(index));
                        //increment index by fstInc
                        index += fstInc;
                    }
                    //flip the switch
                    doFst = false;
                }
                if (!doFst) {
                    //switch points to secInc
                    if (index < length && secInc != 0) {
                        //add to stringBuilder
                        sb.append(s.charAt(index));
                        //increment index by secInc
                        index += secInc;
                    }
                    //flip the switch
                    doFst = true;
                }
            }
        }
        //return stringBuilder as string
        return sb.toString();
    }
}
