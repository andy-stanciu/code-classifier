class Solution {

    public String convert(String s, int numRows) {
        int n = s.length();
        if (// if numsRows < 2 means we have to return same string as it is.
        numRows < 2) {
            return s;
        }
        // total number of character has been passed after reaching at same row
        int col = numRows + numRows - 2;
        // t will give the flexiblity to help which char will come in each level
        int t = col;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            for (int j = i; j < n; j = j + col) {
                char ch = s.charAt(j);
                if (i == 0) {
                    sb.append(ch);
                } else {
                    sb.append(ch);
                    if ((j + t < n) && (j + t != j)) {
                        char ch1 = s.charAt(j + t);
                        sb.append(ch1);
                    }
                }
            }
            // decreasing t by 2 each time
            t -= 2;
        }
        return sb.toString();
    }
}
