class Solution {

    public String convert(String s, int numRows) {
        int n = s.length();
        char[][] arr = new char[n][2];
        if (numRows == 1)
            return s;
        int count = 1;
        int step = 1;
        for (int i = 0; i < n; i++) {
            arr[i][0] = s.charAt(i);
            arr[i][1] = (char) count;
            count += step;
            if (count == numRows || count == 1)
                step *= -1;
        }
        int val = 1;
        char[] res = new char[n];
        int k = 0;
        while (val <= numRows) {
            for (int i = 0; i < n; i++) {
                if (arr[i][1] == val) {
                    res[k++] = arr[i][0];
                }
            }
            val++;
        }
        return new String(res, 0, k);
    }
}
