class Solution {

    public String convert(String s, int numRows) {
        StringBuilder[] ans = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            ans[i] = new StringBuilder();
        }
        // special case : if only one character
        if (numRows == 1) {
            return s;
        }
        // to determine the direction upward or downward
        boolean flag = false;
        // from first row
        int i = 0;
        for (char ch : s.toCharArray()) {
            ans[i].append(ch);
            // at first row or last row: reverse the direction
            if (i == 0 || i == numRows - 1) {
                flag = !flag;
            }
            // true so move downwards
            if (flag) {
                i++;
            } else // move upwards
            {
                i--;
            }
        }
        // combine all string:
        StringBuilder res = new StringBuilder();
        for (StringBuilder row : ans) {
            res.append(row);
        }
        return res.toString();
    }
}
