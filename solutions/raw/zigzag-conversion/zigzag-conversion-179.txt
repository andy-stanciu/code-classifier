class Solution {

    public String convert(String s, int numRows) {
        //return the string becuase in this case no Zigzag is formed
        if (numRows <= 1 || numRows >= s.length()) {
            return s;
        }
        int idx = 0, d = -1;
        //creating new array of string
        String[] rows = new String[numRows];
        //filling the array with empty string so to concatinate
        for (int i = 0; i < numRows; i++) {
            rows[i] = "";
        }
        for (int i = 0; i < s.length(); i++) {
            rows[idx] += s.charAt(i);
            // adjusting d so that idx increases from 0 position and decreases from last position
            if (idx == 0 || idx == (numRows - 1)) {
                d *= -1;
            }
            idx += d;
        }
        return String.join("", rows);
    }
}
