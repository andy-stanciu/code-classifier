class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1 || s.length() <= numRows) {
            return s;
        }
        StringBuilder[] x = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) x[i] = new StringBuilder();
        int k = 0, j = 0;
        boolean godown = true;
        while (k != s.length()) {
            x[j].append(s.charAt(k++));
            if (j == 0)
                godown = true;
            else if (j == numRows - 1)
                godown = false;
            j += godown ? 1 : -1;
        }
        StringBuilder y = new StringBuilder();
        for (int i = 0; i < numRows; i++) y.append(x[i]);
        return y.toString();
    }
}
