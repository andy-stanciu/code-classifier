class Solution {

    public String convert(String s, int numRows) {
        // special case
        if (numRows == 1 || numRows >= s.length())
            return s;
        // will store different lines in the different elements in this array
        StringBuilder[] zigzag = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            zigzag[i] = new StringBuilder();
        }
        // variable to insert in the position of the array
        int j = 0;
        // variable to check j will increase or decrease
        boolean x = false;
        for (int i = 0; i < s.length(); i++) {
            if (j == 0)
                x = true;
            if (j == numRows - 1)
                x = false;
            zigzag[j].append(s.charAt(i));
            if (x == true)
                j++;
            else
                j--;
        }
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            ans.append(zigzag[i]);
        }
        return ans.toString();
    }
}
