class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        List<List<Character>> arr = new ArrayList<>();
        int n = s.length();
        for (int i = 0; i < numRows; i++) arr.add(new ArrayList<>());
        boolean flag = false;
        int i = 0;
        while (i < n) {
            if (!flag) {
                for (int j = 0; j < numRows; j++) {
                    if (i < n)
                        arr.get(j).add(s.charAt(i++));
                }
            } else {
                for (int j = numRows - 2; j >= 1; j--) {
                    if (i < n)
                        arr.get(j).add(s.charAt(i++));
                }
            }
            flag = !flag;
        }
        String ans = "";
        for (int j = 0; j < numRows; j++) {
            for (char e : arr.get(j)) ans += e;
        }
        return ans;
    }
}
