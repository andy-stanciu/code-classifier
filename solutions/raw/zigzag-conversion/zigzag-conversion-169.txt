class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        int n = s.length(), row = 0;
        boolean down = true;
        StringBuilder[] strings = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) strings[i] = new StringBuilder();
        for (int i = 0; i < n; i++) {
            // System.out.println(Arrays.toString(strings));
            char c = s.charAt(i);
            strings[row].append(c);
            if (down) {
                if (row == numRows - 1) {
                    down = false;
                    row--;
                } else
                    row++;
            } else {
                if (row == 0) {
                    down = true;
                    row++;
                } else
                    row--;
            }
        }
        StringBuilder out = new StringBuilder();
        for (StringBuilder sb : strings) out.append(sb);
        return out.toString();
    }
}
