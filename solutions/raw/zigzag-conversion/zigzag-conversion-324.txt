class Solution {

    public String convert(String s, int numRows) {
        if (numRows <= 1) {
            return s;
        }
        Character[][] matrix = new Character[numRows][s.length()];
        int i = 0;
        int j = 0;
        boolean traverseDown = false;
        for (Character c : s.toCharArray()) {
            matrix[i][j] = c;
            if (i == 0) {
                traverseDown = true;
            } else if (i == numRows - 1) {
                traverseDown = false;
            }
            if (traverseDown) {
                i++;
            } else {
                i--;
                j++;
            }
        }
        String solution = "";
        for (i = 0; i < numRows; i++) {
            for (j = 0; j < s.length(); j++) {
                if (matrix[i][j] != null && matrix[i][j] != 0) {
                    solution += matrix[i][j];
                }
            }
        }
        return solution;
    }
}
