class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        String[] ans = new String[numRows];
        Arrays.fill(ans, "");
        int cycleLength = 2 * numRows - 2;
        int reminder;
        for (int i = 0; i < s.length(); i++) {
            reminder = i % cycleLength;
            if (reminder < numRows) {
                ans[reminder] += s.charAt(i);
            } else {
                ans[cycleLength - reminder] += s.charAt(i);
            }
        }
        s = "";
        for (int i = 0; i < ans.length; i++) {
            s += ans[i];
        }
        return s;
    }
}
