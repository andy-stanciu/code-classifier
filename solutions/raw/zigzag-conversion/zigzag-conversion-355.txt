class Solution {

    public String convert(String s, int numRows) {
        String response = "";
        Map<Integer, String> tmp = new HashMap<>();
        int tmpCounter = 1;
        boolean toAdd = true;
        if (numRows != 1) {
            for (int counter = 0; counter < s.length(); counter++) {
                if (tmp.containsKey(tmpCounter)) {
                    tmp.put(tmpCounter, (tmp.get(tmpCounter) + String.valueOf(s.charAt(counter))));
                } else {
                    tmp.put(tmpCounter, String.valueOf(s.charAt(counter)));
                }
                if (toAdd) {
                    tmpCounter += 1;
                    toAdd = (tmpCounter == numRows) ? false : true;
                } else {
                    tmpCounter -= 1;
                    toAdd = (tmpCounter == 1) ? true : false;
                }
            }
            for (int counter = 1; counter <= numRows; counter++) {
                response = (tmp.get(counter) == null) ? response + "" : response + tmp.get(counter);
            }
        } else {
            response = s;
        }
        return response;
    }
}
