class Solution {

    public String convert(String s, int numRows) {
        //if no of row is less than equal to length of string then we will simply return string
        if (numRows == 1 || s.length() <= numRows)
            return s;
        //string builder to store ans
        StringBuilder ans = new StringBuilder();
        //for simplicity take variable r and store no of rows
        int r = numRows;
        //similarly for length
        int l = s.length();
        //find the diffrence between the character for the first row.
        int diff = 2 * (r - 1);
        //initialize index as 0;
        int i = 0;
        //check if i< no of rows
        while (i < r) {
            //so if i==0 or i==r-1 then the character difference will be difference only
            if (i == 0 || i == r - 1) {
                //initialize another variable j =i;
                int j = i;
                //check if j< length of string
                while (j < l) {
                    //append character to ans
                    ans.append(s.charAt(j));
                    //and we will increment j variable by diff(i.e = 2(r-1))
                    j += diff;
                }
                //now we have stored ans for a row
                //so increment the value of i
                i++;
            } else //if i is nor first row neither the last row then
            {
                //similarly we will take varibale j=i
                int j = i;
                // this c will track the column of the row
                int c = 0;
                //check if j< length of string
                while (j < l) {
                    //if the column will be 1st, 3rd , 5th,..... then  we will append character to ans
                    // and we will increment j by diff-(2*i);
                    if (c % 2 == 0) {
                        ans.append(s.charAt(j));
                        j += diff - (2 * i);
                        c++;
                    } else //otherwise  we will append character to ans
                    // and we will increment j by (2*i) only;
                    {
                        ans.append(s.charAt(j));
                        j += (2 * i);
                        c++;
                    }
                }
                //now we have stored ans for the a row
                //so increment the value of i
                i++;
            }
        }
        return ans.toString();
    }
}
