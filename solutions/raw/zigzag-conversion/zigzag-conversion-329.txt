class Solution {

    public String convert(String s, int numRows) {
        StringBuffer[] arr = new StringBuffer[numRows];
        for (int i = 0; i < numRows; i++) arr[i] = new StringBuffer();
        boolean isZig = false;
        int zigVal = numRows - 2;
        for (int i = 0; i < s.length(); i++) {
            int tempRowCount = 0;
            if (isZig && zigVal > 0) {
                tempRowCount = zigVal;
                while (i < s.length() && tempRowCount > 0) {
                    arr[tempRowCount].append(s.charAt(i));
                    i++;
                    tempRowCount--;
                }
            } else {
                while (i < s.length() && tempRowCount < numRows) {
                    arr[tempRowCount].append(s.charAt(i));
                    i++;
                    tempRowCount++;
                }
            }
            i--;
            isZig = !isZig;
        }
        String response = "";
        for (int i = 0; i < arr.length; i++) {
            response = response + arr[i];
        }
        return response;
    }
}
