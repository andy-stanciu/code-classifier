class Solution {

    public String convert(String s, int numRows) {
        // corner case 1 : if string in empty
        if (s == null || s.length() == 0) {
            return "";
        }
        // corner case 1 : numb of rows is 1 , nothing to do
        if (numRows == 1) {
            return s;
        }
        // build line wise string
        StringBuilder[] sbs = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            sbs[i] = new StringBuilder();
        }
        int lineIndex = 0;
        int direction = 1;
        int counter = 1;
        for (int i = 0; i < s.length(); i++) {
            // append char to current line
            sbs[lineIndex].append(s.charAt(i));
            // change direction and counter
            if (counter % numRows == 0) {
                counter = 1;
                direction *= -1;
            }
            counter++;
            lineIndex += direction;
        }
        // concatenate all line string.
        return String.join("", sbs);
    }
}
