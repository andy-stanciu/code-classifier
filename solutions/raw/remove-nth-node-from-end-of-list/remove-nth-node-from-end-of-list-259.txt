class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode d1 = dummy;
        ListNode d2 = dummy;
        for (int i = 1; i <= n + 1; i++) {
            d2 = d2.next;
        }
        while (d2 != null) {
            d1 = d1.next;
            d2 = d2.next;
        }
        d1.next = d1.next.next;
        return dummy.next;
    }
}
