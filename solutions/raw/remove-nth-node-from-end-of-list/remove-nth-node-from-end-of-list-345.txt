/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    ListNode removed;

    int removeWhenN(ListNode head, int n) {
        if (head == null)
            return 0;
        int nextDepth = removeWhenN(head.next, n);
        if (nextDepth == n) {
            removed = head.next;
            head.next = head.next.next;
        }
        return nextDepth + 1;
    }

    public ListNode removeNthFromEnd(ListNode head, int n) {
        if (head == null)
            return null;
        int removedAt = removeWhenN(head, n);
        if (removedAt == n) {
            return head.next;
        }
        return head;
    }
}
