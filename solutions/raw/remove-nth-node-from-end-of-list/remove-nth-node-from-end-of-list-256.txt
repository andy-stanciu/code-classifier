class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode curr = head, req = head, prev = head;
        for (int i = 0; i < n; i++) {
            curr = curr.next;
        }
        while (curr != null) {
            curr = curr.next;
            prev = req;
            req = req.next;
        }
        if (req == head)
            head = head.next;
        prev.next = req.next;
        req.next = null;
        return head;
    }
}
