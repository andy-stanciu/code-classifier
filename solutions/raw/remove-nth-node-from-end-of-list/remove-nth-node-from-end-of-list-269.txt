//couting index from end instead of starting
//bottom - up
class Solution {

    int index = 0;

    public ListNode removeNthFromEnd(ListNode head, int n) {
        removeNode(head.next, head, n);
        index++;
        if (index == n) {
            return head.next;
        }
        return head;
    }

    private void removeNode(ListNode current, ListNode prev, int n) {
        if (current == null) {
            return;
        } else {
            removeNode(current.next, current, n);
            index++;
            if (index == n) {
                prev.next = current.next;
            }
        }
    }
}
