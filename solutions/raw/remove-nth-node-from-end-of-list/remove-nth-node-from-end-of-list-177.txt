class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        // O(L) time | O(1) space
        // two pointer technique
        ListNode dummyHead = new ListNode(-1);
        dummyHead.next = head;
        ListNode left = dummyHead;
        ListNode right = head;
        while (n > 0) {
            n -= 1;
            right = right.next;
        }
        while (right != null) {
            left = left.next;
            right = right.next;
        }
        left.next = left.next.next;
        return dummyHead.next;
    }
}
