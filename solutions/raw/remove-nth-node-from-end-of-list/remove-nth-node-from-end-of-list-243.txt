class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode temp = head;
        int l = 0;
        while (temp != null) {
            temp = temp.next;
            l++;
        }
        l = l - n;
        System.out.println(l);
        temp = head;
        if (l == 0) {
            head = head.next;
        } else {
            for (int i = 1; i < l; i++) {
                temp = temp.next;
            }
            temp.next = temp.next.next;
        }
        return head;
    }
}
