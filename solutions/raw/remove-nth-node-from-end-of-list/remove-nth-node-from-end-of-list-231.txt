class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode p = head;
        head = dummy;
        ListNode temp = head;
        int len = size(head);
        int i = 0;
        if (len <= 1) {
            return null;
        }
        while (i < len - n - 1) {
            temp = temp.next;
            i += 1;
        }
        temp.next = temp.next.next;
        return head.next;
    }

    public int size(ListNode head) {
        int size = 0;
        while (head != null) {
            head = head.next;
            size++;
        }
        return size;
    }
}
