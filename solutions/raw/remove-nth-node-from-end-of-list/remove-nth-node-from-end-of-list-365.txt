class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode nodeNext = head;
        ListNode nodePrevious = head;
        for (int i = 0; i < n; i++) {
            nodeNext = nodeNext.next;
        }
        if (nodeNext == null) {
            head = head.next;
            return head;
        }
        while (nodeNext.next != null) {
            nodePrevious = nodePrevious.next;
            nodeNext = nodeNext.next;
        }
        nodePrevious.next = nodePrevious.next.next;
        return head;
    }
}
