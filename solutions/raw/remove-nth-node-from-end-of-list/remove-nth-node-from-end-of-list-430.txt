class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode r = reverse(head);
        if (r == null)
            return r;
        else {
            ListNode t = r;
            ListNode p = null;
            int i = 1;
            while (i < n && t != null) {
                p = t;
                i++;
                t = t.next;
            }
            if (t != null) {
                if (p == null) {
                    r = t.next;
                } else {
                    p.next = t.next;
                }
            }
            return reverse(r);
        }
    }

    static ListNode reverse(ListNode head) {
        ListNode c = head;
        ListNode t = null;
        ListNode temp = null;
        while (c != null) {
            temp = c.next;
            c.next = t;
            t = c;
            c = temp;
        }
        return t;
    }
}
