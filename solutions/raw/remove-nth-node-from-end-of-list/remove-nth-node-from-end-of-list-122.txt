class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode curr = head;
        int len = 0;
        while (curr != null) {
            len++;
            curr = curr.next;
        }
        // n from start
        int nFromStart = len - n + 1;
        if (nFromStart == 1)
            return head.next;
        curr = head;
        while (--nFromStart > 1) {
            curr = curr.next;
        }
        curr.next = curr.next.next;
        return head;
    }
}
