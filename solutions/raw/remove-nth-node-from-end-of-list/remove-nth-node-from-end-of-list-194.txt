class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode node = head;
        if (head.next == null) {
            head = null;
            return head;
        }
        int length = 0;
        while (node != null) {
            length++;
            node = node.next;
        }
        if (n == length) {
            head = head.next;
            return head;
        }
        int num = length - n + 1;
        length = 0;
        node = head;
        while (length + 2 != num) {
            length++;
            node = node.next;
        }
        node.next = node.next.next;
        return head;
    }
}
