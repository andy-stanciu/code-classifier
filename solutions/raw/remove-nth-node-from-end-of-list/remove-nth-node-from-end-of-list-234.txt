class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode p1 = head;
        ListNode p2 = head;
        ListNode prev = head;
        // Move p1 k nodes into the list
        for (int i = 0; i < n; i++) {
            if (p1 == null) {
                return null;
            }
            p1 = p1.next;
        }
        // Move them at same pace, When p1 hits the end, p2 will
        // be at the right element
        while (p1 != null) {
            p1 = p1.next;
            prev = p2;
            p2 = p2.next;
        }
        if (p2 == head) {
            head = head.next;
            return head;
        }
        prev.next = p2.next;
        return head;
    }
}
