class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode a = null;
        for (var b = head; b.next != null; b = b.next) if (--n <= 0)
            a = a == null ? head : a.next;
        if (a == null)
            head = head.next;
        else
            a.next = a.next.next;
        return head;
    }
}
