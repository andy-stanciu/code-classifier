class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        //5
        ListNode current = head;
        for (int i = 0; i < n; i++) {
            current = current.next;
        }
        //1
        ListNode slow = head;
        //5
        ListNode fast = current;
        //1
        ListNode previous = null;
        while (fast != null) {
            previous = slow;
            slow = slow.next;
            fast = fast.next;
        }
        if (previous == null) {
            return head.next;
        }
        previous.next = slow.next;
        return head;
    }
}
