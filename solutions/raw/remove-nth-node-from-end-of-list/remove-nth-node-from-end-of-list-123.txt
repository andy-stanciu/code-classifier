class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        Stack<ListNode> stack = new Stack<>();
        ListNode curr = head;
        while (curr != null) {
            stack.push(curr);
            curr = curr.next;
        }
        while (n-- > 0) stack.pop();
        if (stack.isEmpty())
            return head.next;
        ListNode pop = stack.pop();
        pop.next = pop.next.next;
        return head;
    }
}
