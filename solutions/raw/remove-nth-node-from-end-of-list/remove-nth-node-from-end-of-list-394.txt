class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        if (head.next == null && n == 1)
            return null;
        ListNode i = head, j = head;
        while (n-- > 0) j = j.next;
        if (j == null)
            return head.next;
        while (j.next != null) {
            j = j.next;
            i = i.next;
        }
        i.next = i.next.next;
        return head;
    }
}
