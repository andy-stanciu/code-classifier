class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        // edge case
        if (head == null)
            return head;
        // move n steps forward
        int n1 = n;
        // calculates length of the list
        int len = 1;
        ListNode end = head;
        while (n1-- > 0 && end.next != null) {
            end = end.next;
            len++;
        }
        // we can\'t remove the node which is not in the list
        if (n1 > 0)
            return null;
        // reach one node previous to the deleting node
        ListNode lastN = head;
        while (end.next != null) {
            end = end.next;
            lastN = lastN.next;
            len++;
        }
        // removing the head node
        if (len == n)
            return head.next;
        // removing node which is not the head.
        lastN.next = lastN.next.next;
        return head;
    }
}
