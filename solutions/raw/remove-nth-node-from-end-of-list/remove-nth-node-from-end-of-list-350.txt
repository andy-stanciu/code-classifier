class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        if (head == null || (head.next == null && n == 1))
            return null;
        int nodes = 1;
        ListNode curr = head;
        while (curr.next != null) {
            curr = curr.next;
            nodes++;
        }
        nodes -= n;
        curr = head;
        ListNode prev = null;
        while (nodes != 0) {
            prev = curr;
            curr = curr.next;
            nodes--;
        }
        if (prev != null) {
            prev.next = curr.next;
        } else {
            return head.next;
        }
        return head;
    }
}
