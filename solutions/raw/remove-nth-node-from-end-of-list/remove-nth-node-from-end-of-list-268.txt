class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode f = head;
        ListNode s = head;
        int c = 1;
        while (c <= n) {
            f = f.next;
            c++;
        }
        if (f == null) {
            return s.next;
        }
        while (f.next != null) {
            f = f.next;
            s = s.next;
        }
        s.next = s.next.next;
        return head;
    }
}
