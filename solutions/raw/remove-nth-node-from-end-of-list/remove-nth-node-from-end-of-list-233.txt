class Solution {

    int counter = 0;

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dummy = new ListNode(-1, head);
        helper(dummy, n);
        return dummy.next;
    }

    public void helper(ListNode head, int n) {
        if (head == null)
            return;
        helper(head.next, n);
        if (counter == n) {
            head.next = head.next.next;
        }
        counter++;
    }
}
