class Solution {

    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode first = head, second = head;
        while (n != 0) {
            second = second.next;
            n--;
        }
        // If we have to remove the head of list then second will be null.
        if (second == null) {
            return head.next;
        }
        while (second.next != null) {
            second = second.next;
            first = first.next;
        }
        first.next = first.next.next;
        return head;
    }
}
