class Solution {

    public List<List<Integer>> generate(int numRows) {
        if (numRows == 0)
            return new ArrayList<>();
        List<List<Integer>> res = new ArrayList<>();
        int k = 1;
        while (k <= numRows) {
            //To add the first element in the triangle
            if (k == 1) {
                List<Integer> temp = new ArrayList<>();
                temp.add(1);
                res.add(temp);
            } else // to add the second element
            if (k == 2) {
                List<Integer> temp = new ArrayList<>();
                temp.add(1);
                temp.add(1);
                res.add(temp);
            } else //to add the rest of the elements into the list, which starts from numRows =3.
            {
                List<Integer> temp = new ArrayList<>();
                temp.add(1);
                // add 1 as first element in the current list
                //gets the previous list
                List<Integer> prev = res.get(k - 2);
                int j = 1;
                for (int i = 0; i < prev.size() - 1 && j < prev.size(); i++) {
                    //iterate over the previous list
                    int a = prev.get(i);
                    int b = prev.get(j++);
                    // add sum of the elements from the previous list into the curr one
                    temp.add(a + b);
                }
                temp.add(1);
                //add the last element into the list
                res.add(temp);
            }
            k++;
        }
        return res;
    }
}
