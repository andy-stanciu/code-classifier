class Solution {

    public List<List<Integer>> generate(int numRows) {
        // Final answer List
        List<List<Integer>> ans = new ArrayList<>();
        //iterating the list
        for (int i = 0; i < numRows; i++) {
            //creating an empty list for row
            List<Integer> ls = new ArrayList<>();
            //first row is list of number 1
            if (i == 0) {
                ls.add(1);
                ans.add(ls);
                continue;
            }
            //every row starts with 1
            ls.add(1);
            //calculating the inner values
            for (int j = 1; j < i; j++) {
                ls.add(ans.get(i - 1).get(j) + ans.get(i - 1).get(j - 1));
            }
            //every row ends with 1
            ls.add(1);
            //adding the row to final answer
            ans.add(ls);
        }
        return ans;
    }
}
