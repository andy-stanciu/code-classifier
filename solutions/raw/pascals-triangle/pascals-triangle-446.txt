class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> list = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            list.add(getRow(i));
        }
        return list;
    }

    private List<Integer> getRow(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = i; j > 0; j--) {
                arr[j] += arr[j - 1];
            }
        }
        List<Integer> l = new ArrayList<>();
        for (int number : arr) {
            l.add(number);
        }
        return l;
    }
}
