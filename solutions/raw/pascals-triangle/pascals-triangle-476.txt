class Solution {

    public List<List<Integer>> generate(int n) {
        if (n == 1) {
            return new ArrayList<>() {

                {
                    add(new ArrayList<>() {

                        {
                            add(1);
                        }
                    });
                }
            };
        }
        List<List<Integer>> ans = new ArrayList<>() {

            {
                add(new ArrayList<>() {

                    {
                        add(1);
                    }
                });
                add(new ArrayList<>() {

                    {
                        add(1);
                        add(1);
                    }
                });
            }
        };
        for (int i = 2; i < n; i++) {
            List<Integer> l1 = ans.get(i - 1);
            List<Integer> l2 = new ArrayList<>();
            l2.add(1);
            for (int j = 0; j < l1.size() - 1; j++) {
                int x = l1.get(j) + l1.get(j + 1);
                l2.add(x);
            }
            l2.add(1);
            ans.add(l2);
        }
        return ans;
    }
}
