class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> triangle = new ArrayList<>();
        //1 based to 0 based indexing
        numRows -= 1;
        for (int row = 0; row <= numRows; row++) {
            triangle.add(new ArrayList<Integer>());
            // 0th column of row
            triangle.get(row).add(1);
            for (int col = 1; col < row; col++) {
                // 1st column to 2nd last column
                int topLeft = triangle.get(row - 1).get(col - 1);
                int top = triangle.get(row - 1).get(col);
                triangle.get(row).add(topLeft + top);
            }
            if (row != 0)
                // (row)th column of row
                triangle.get(row).add(1);
        }
        return triangle;
    }
}
