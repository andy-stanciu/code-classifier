class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> firstElement = List.of(1);
        result.add(firstElement);
        int readyRows = 1;
        while (readyRows < numRows) {
            generate(result.get(readyRows - 1), result);
            readyRows++;
        }
        return result;
    }

    private void generate(List<Integer> previous, List<List<Integer>> finalArray) {
        List<Integer> newOne = new ArrayList<>(previous.size() + 1);
        newOne.add(1);
        for (int i = 0; i < previous.size() - 1; i++) {
            newOne.add(previous.get(i) + previous.get(i + 1));
        }
        newOne.add(1);
        finalArray.add(newOne);
    }
}
