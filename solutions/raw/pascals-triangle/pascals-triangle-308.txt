class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> res = new ArrayList<>();
        //"res" list k andr ek list add kro
        res.add(new ArrayList<>());
        // Sbse phele 1st row m "1" daalo !
        res.get(0).add(1);
        for (int i = 1; i < numRows; i++) {
            // "i" is the row number
            List<Integer> curr = new ArrayList<>();
            //Firstly add "1" to the list at 1st position
            curr.add(1);
            for (int j = 1; j < i; j++) {
                // "j" is the place
                //vo upaar waale row k dono ko add krke, neeche waale row m daalna hai
                curr.add(res.get(i - 1).get(j - 1) + res.get(i - 1).get(j));
            }
            //after adding add "1" more into the list
            curr.add(1);
            // add the row in the "res" list
            res.add(curr);
        }
        return res;
    }
}
