class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> arrl = new ArrayList<>();
        List<Integer> arr = new ArrayList<>();
        arr.add(1);
        arrl.add(arr);
        for (int i = 2; i < numRows + 1; i++) {
            List<Integer> arr1 = new ArrayList<>();
            arr1.add(1);
            for (int j = 1; j < i; j++) {
                int a = i - j;
                int b = (a * arr1.getLast()) / j;
                arr1.add(b);
            }
            arrl.add(arr1);
        }
        return arrl;
    }
}
