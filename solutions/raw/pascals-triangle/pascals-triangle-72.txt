class Solution {

    // Calculate n choose r (nCr) using a loop
    public int calculateCombination(int n, int r) {
        int result = 1;
        for (int i = 0; i < r; i++) {
            result = result * (n - i) / (i + 1);
        }
        return result;
    }

    // Generate Pascal\'s Triangle with \'numRows\' rows
    public List<List<Integer>> generate(int numRows) {
        // Initialize a list of lists to represent Pascal\'s Triangle
        List<List<Integer>> pascalTriangle = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            pascalTriangle.add(new ArrayList<>());
            List<Integer> currentRow = pascalTriangle.get(i);
            for (int j = 0; j <= i; j++) {
                // Calculate and insert the binomial coefficient (nCr) into the current row
                currentRow.add(calculateCombination(i, j));
            }
        }
        return pascalTriangle;
    }
}
