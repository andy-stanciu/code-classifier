class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> lst = new ArrayList<>();
        for (int l = 1; l <= numRows; l++) {
            List<Integer> subLst = new ArrayList<>();
            int C = 1;
            for (int i = 1; i <= l; i++) {
                subLst.add(C);
                C = C * (l - i) / i;
            }
            lst.add(subLst);
        }
        return lst;
    }
}
