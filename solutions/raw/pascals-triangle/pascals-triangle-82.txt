class Solution {

    public List<List<Integer>> generate(int n) {
        List<List<Integer>> res = new ArrayList<>(n);
        for (int i = 0; i < n; i++) {
            // pointer flag
            int left = 0;
            List<Integer> l = new ArrayList<>(i + 1);
            for (int j = 0; j < i + 1; j++) {
                // add 1 to \'first\' and \'last\'
                if (j == 0 || j == i) {
                    l.add(1);
                } else {
                    l.add(res.get(i - 1).get(left) + res.get(i - 1).get(left + 1));
                    left++;
                }
            }
            res.add(l);
        }
        return res;
    }
}
