class Solution {

    public List<List<Integer>> generate(int numRows) {
        // Initialize the main list
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            // List to store the current row values
            List<Integer> temp = new ArrayList<>();
            for (int j = 0; j <= i; j++) {
                // Traverse up to and including i
                if (j == 0 || j == i) {
                    // First and last elements are always 1
                    temp.add(1);
                } else {
                    // Get the previous row
                    List<Integer> pre = ans.get(i - 1);
                    // Sum of the two values above
                    temp.add(pre.get(j - 1) + pre.get(j));
                }
            }
            // Add the current row to the main list
            ans.add(temp);
        }
        return ans;
    }
}
