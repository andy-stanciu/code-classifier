class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> retList = new ArrayList<List<Integer>>();
        retList.add(Arrays.asList(1));
        if (numRows == 1)
            return retList;
        List<Integer> prevList = retList.get(0);
        int count = 1;
        while (count < numRows) {
            List<Integer> currList = new ArrayList<Integer>();
            for (int i = 0; i < prevList.size() + 1; i++) {
                if (i == 0 || i == prevList.size())
                    currList.add(prevList.get(0));
                else
                    currList.add((prevList.get(i - 1) + prevList.get(i)));
            }
            retList.add(currList);
            prevList = currList;
            count++;
        }
        return retList;
    }
}
