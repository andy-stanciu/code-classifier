class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        int i = 0;
        while (i < numRows) {
            int j = 0;
            int k = 0;
            while (j <= i) {
                if (j == 0 || j == i) {
                    list.add(1);
                } else {
                    list.add(ans.get(i - 1).get(k) + ans.get(i - 1).get(k + 1));
                    k++;
                }
                j++;
            }
            ans.add(new ArrayList(list));
            list.clear();
            i++;
        }
        return ans;
    }
}
