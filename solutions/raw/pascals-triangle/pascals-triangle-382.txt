class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ansList = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            List<Integer> tmpArr = compute(i);
            ansList.add(tmpArr);
        }
        return ansList;
    }

    public List<Integer> compute(int row) {
        List<Integer> temp = new ArrayList<>();
        int ans = 1;
        temp.add(ans);
        for (int col = 1; col < row; col++) {
            ans = ans * (row - col);
            ans = ans / col;
            temp.add(ans);
        }
        return temp;
    }
}
