class Solution {

    public int findMinArrowShots(int[][] a) {
        Arrays.sort(a, (x, y) -> Integer.compare(x[0], y[0]));
        int len = a.length;
        int arrow_count = 1;
        int current_pos = a[len - 1][0];
        for (int i = len - 2; i >= 0; i--) {
            if (current_pos > a[i][1]) {
                arrow_count++;
                current_pos = a[i][0];
            }
        }
        return arrow_count;
    }
}
