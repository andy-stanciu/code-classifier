class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(a -> a[0]));
        int[] newPoint = points[0];
        int count = 1;
        for (int[] point : points) {
            if (point[0] <= newPoint[1]) {
                //overlapped
                newPoint[1] = Math.min(point[1], newPoint[1]);
            } else {
                count++;
                newPoint = point;
            }
        }
        return count;
    }
}
