class Solution {

    public int findMinArrowShots(int[][] points) {
        int size = points.length;
        if (size == 0) {
            return 0;
        }
        // sort the intervals by the end point
        Comparator<int[]> comparator = (i1, i2) -> Integer.compare(i1[1], i2[1]);
        Arrays.sort(points, comparator);
        int result = 1;
        int previous = 0;
        for (int current = 1; current < size; current++) {
            if (points[current][0] > points[previous][1]) {
                result++;
                previous = current;
            }
        }
        return result;
    }
}
