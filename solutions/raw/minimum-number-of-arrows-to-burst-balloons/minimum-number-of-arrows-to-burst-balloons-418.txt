class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(a -> a[1]));
        int x = 0;
        for (int i = 0; i < points.length; i++) {
            int max = points[i][1];
            while (i + 1 < points.length && points[i + 1][0] <= max) i++;
            x++;
        }
        return x;
    }
}
