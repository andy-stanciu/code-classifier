class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> {
            if (a[0] == b[0]) {
                return Integer.compare(a[1], b[1]);
            }
            return Integer.compare(a[0], b[0]);
        });
        int count = 1;
        int prevStartPoint = points[0][0];
        int prevEndPoint = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int currStartPoint = points[i][0];
            int currEndPoint = points[i][1];
            if (currStartPoint > prevEndPoint) {
                // no overlap
                count++;
                prevStartPoint = currStartPoint;
                prevEndPoint = currEndPoint;
            } else {
                // overlap
                prevStartPoint = Math.max(prevStartPoint, currStartPoint);
                prevEndPoint = Math.min(prevEndPoint, currEndPoint);
            }
        }
        return count;
    }
}
