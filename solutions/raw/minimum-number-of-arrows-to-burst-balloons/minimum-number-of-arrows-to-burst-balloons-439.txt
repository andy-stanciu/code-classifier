class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int eraseCount = 0;
        int len = points.length;
        int end = points[0][1];
        for (int i = 1; i < len; i++) {
            if (points[i][0] <= end) {
                eraseCount++;
                end = Math.min(end, points[i][1]);
            } else {
                end = points[i][1];
            }
        }
        return len - eraseCount;
    }
}
