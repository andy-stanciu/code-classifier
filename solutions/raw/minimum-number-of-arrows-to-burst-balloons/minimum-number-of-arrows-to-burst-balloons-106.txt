class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (p, q) -> {
            return Integer.compare(p[1], q[1]);
        });
        int arrow = 1;
        int prev = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int start = points[i][0];
            int end = points[i][1];
            //previous arrow got utilised to burst this one as well.
            if (start <= prev) {
                continue;
            }
            prev = end;
            arrow++;
        }
        return arrow;
    }
}
