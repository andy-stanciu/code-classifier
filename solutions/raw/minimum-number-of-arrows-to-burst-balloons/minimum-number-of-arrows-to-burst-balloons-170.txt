class Solution {

    public int findMinArrowShots(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1;
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] <= intervals[i - 1][1]) {
                intervals[i][1] = intervals[i - 1][1];
            } else {
                count++;
            }
        }
        return count;
    }
}
