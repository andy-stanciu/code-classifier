class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int loc = 0;
        int i = 0;
        while (i < points.length) {
            loc = points[i][1];
            while (i < points.length && loc <= points[i][1] && loc >= points[i][0]) {
                i++;
            }
            count++;
        }
        return count;
    }
}
