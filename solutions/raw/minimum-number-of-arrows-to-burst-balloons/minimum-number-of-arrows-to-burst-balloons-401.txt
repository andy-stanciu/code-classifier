class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int currPos = points[0][1];
        int count = 1;
        for (int i = 1; i < points.length; i++) {
            if (currPos >= points[i][0] && currPos <= points[i][1]) {
                continue;
            } else {
                count++;
                currPos = points[i][1];
            }
        }
        return count;
    }
}
