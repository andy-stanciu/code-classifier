class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int chainLen = 1;
        int chainEnd = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > chainEnd) {
                chainLen++;
                chainEnd = points[i][1];
            }
        }
        return chainLen;
    }
}
