class Solution {

    public int findMinArrowShots(int[][] points) {
        //Sorting acc to end points
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0, arrow = 0;
        for (int i = 0; i < points.length; i++) {
            if (count == 0 || points[i][0] > arrow) {
                //Comparing with starting point
                count++;
                //End point, the whole segment will become arrow
                arrow = points[i][1];
            }
        }
        return count;
    }
}
