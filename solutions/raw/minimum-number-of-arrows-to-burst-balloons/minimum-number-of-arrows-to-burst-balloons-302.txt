class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> a[1] - b[1]);
        int arrow = 1;
        int i = -1;
        int shoot = points[0][1];
        for (int[] point : points) {
            i++;
            if (i == 0)
                continue;
            if (shoot > point[1] || shoot < point[0]) {
                arrow++;
                shoot = point[1];
            }
        }
        return arrow;
    }
}
