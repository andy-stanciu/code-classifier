class Solution {

    public int findMinArrowShots(int[][] points) {
        int n = points.length;
        int count = 1;
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int last = points[0][1];
        for (int i = 1; i < n; i++) {
            if (last < points[i][0]) {
                count++;
                last = points[i][1];
            } else {
                continue;
            }
        }
        return count;
    }
}
