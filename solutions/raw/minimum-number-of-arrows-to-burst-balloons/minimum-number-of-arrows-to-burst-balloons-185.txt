class Solution {

    public static int findMinArrowShots(int[][] points) {
        int arrows = 0;
        //sorting on basis of ending point
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        if (points.length <= 0) {
            return arrows;
        }
        arrows = 1;
        int endintCoordinate = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int start = points[i][0];
            int end = points[i][1];
            // end>=endintCoordinate (sorting on basis of ending point)
            if (start <= endintCoordinate) {
                //ballons share a common point
                // it will burst without need of new arrows
            } else {
                arrows++;
                endintCoordinate = end;
            }
        }
        return arrows;
    }
}
