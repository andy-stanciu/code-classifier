class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> {
            if (a[0] <= b[0])
                return -1;
            return 1;
        });
        int low = points[0][0], high = points[0][1], c = 0;
        for (int i = 1; i < points.length; i++) {
            int currStart = points[i][0], currEnd = points[i][1];
            if (currStart <= high) {
                high = Math.min(high, currEnd);
            } else {
                c++;
                low = currStart;
                high = currEnd;
            }
        }
        c++;
        return c;
    }
}
