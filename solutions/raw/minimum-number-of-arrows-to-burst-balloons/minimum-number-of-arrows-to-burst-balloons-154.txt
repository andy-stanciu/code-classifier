class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int lastArrow = points[0][1];
        int count = 1;
        // Always shoot arror at last point or end point
        for (int i = 1; i < points.length; i++) {
            // Check if last arrow shot covers the current point or not
            if (lastArrow >= points[i][0])
                continue;
            else {
                // Update lastArrow shot and inc count
                lastArrow = points[i][1];
                count++;
            }
        }
        return count;
    }
}
