class Solution {

    public int findMinArrowShots(int[][] points) {
        if (points.length == 1)
            return 1;
        Arrays.sort(points, (p1, p2) -> Integer.compare(p1[1], p2[1]));
        int count = 1;
        int curr = points[0][1];
        for (int[] i : points) {
            if (i[1] > curr && i[0] > curr) {
                curr = i[1];
                count++;
            }
        }
        return count;
    }
}
