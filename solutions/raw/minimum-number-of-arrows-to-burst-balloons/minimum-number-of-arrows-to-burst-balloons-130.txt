class Solution {

    public int findMinArrowShots(int[][] points) {
        // Sort points by their start time
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        // Initialize merged list with the first point after sorting
        List<int[]> merged = new ArrayList<>();
        merged.add(points[0]);
        // Iterate through the points using a for-each loop
        for (int[] point : points) {
            int[] last = merged.get(merged.size() - 1);
            if (last[1] >= point[0]) {
                // Check if intervals overlap
                if (last[1] >= point[1]) {
                    // Update end to the minimum end
                    last[1] = point[1];
                }
                // Update start to the current point\'s start
                last[0] = point[0];
            } else {
                merged.add(point);
            }
        }
        return merged.size();
    }
}
