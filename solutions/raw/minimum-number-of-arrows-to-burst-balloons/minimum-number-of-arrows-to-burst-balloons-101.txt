class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int range = points[0][1], count = 1;
        for (int i = 0; i < points.length; i++) {
            if (range < points[i][0]) {
                count++;
                range = points[i][1];
            }
        }
        return count;
    }
}
