class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int[] prev = points[0];
        int n = points.length, count = 1;
        for (int i = 1; i < n; i++) {
            int[] cur = points[i];
            if (prev[1] >= cur[0]) {
                prev[0] = Math.max(prev[0], cur[0]);
                prev[1] = Math.min(prev[1], cur[1]);
            } else {
                count++;
                prev = cur;
            }
        }
        return count;
    }
}
