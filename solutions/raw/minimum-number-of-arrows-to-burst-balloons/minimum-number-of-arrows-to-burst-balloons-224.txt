class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int temp = points[points.length - 1][0];
        int count = 1;
        for (int i = points.length - 1; i >= 0; i--) {
            if (points[i][1] < temp) {
                count++;
                temp = points[i][0];
            }
        }
        return count;
    }
}
