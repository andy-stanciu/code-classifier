class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int count = 1, start = points[0][1];
        for (int[] point : points) {
            if (point[0] <= start) {
                start = Math.min(start, point[1]);
                continue;
            }
            count++;
            start = point[1];
        }
        return count;
    }
}
