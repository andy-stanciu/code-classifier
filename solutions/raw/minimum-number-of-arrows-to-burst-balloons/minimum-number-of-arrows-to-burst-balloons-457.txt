class Solution {

    public int findMinArrowShots(int[][] points) {
        if (points.length == 0) {
            return 0;
        }
        Arrays.sort(points, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return Integer.compare(a[0], b[0]);
            }
        });
        int minArrowRequired = 1;
        int arrowBurstPoint = points[0][1];
        for (int iCounter = 1; iCounter < points.length; ) {
            if (arrowBurstPoint >= points[iCounter][0]) {
                arrowBurstPoint = Math.min(arrowBurstPoint, points[iCounter][1]);
                iCounter++;
            } else {
                minArrowRequired++;
                arrowBurstPoint = points[iCounter][1];
            }
        }
        return minArrowRequired;
    }
}
