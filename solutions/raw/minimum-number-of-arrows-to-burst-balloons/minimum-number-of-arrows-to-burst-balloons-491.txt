class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[0]));
        int arrow = 1;
        int common = points[0][1];
        for (int i = 1; i < points.length; ++i) {
            if (common >= points[i][0]) {
                common = Math.min(common, points[i][1]);
            } else {
                ++arrow;
                common = points[i][1];
            }
        }
        return arrow;
    }
}
