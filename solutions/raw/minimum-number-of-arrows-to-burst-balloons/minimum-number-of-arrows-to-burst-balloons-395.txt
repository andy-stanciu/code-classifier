class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(r -> r[0]));
        int arrows = 1;
        // window right side index
        int wr = 1;
        // 0 - start, 1 - end
        int[] intersection = points[0];
        while (true) {
            if (wr == points.length) {
                break;
            }
            if (points[wr][0] <= intersection[1]) {
                intersection[0] = Math.max(intersection[0], points[wr][0]);
                intersection[1] = Math.min(intersection[1], points[wr][1]);
            } else {
                arrows++;
                intersection = points[wr];
            }
            wr++;
        }
        return arrows;
    }
}
