class Solution {

    public int findMinArrowShots(int[][] points) {
        int i = 0;
        int arrow = 0;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        while (i < points.length) {
            int[] start = points[i];
            while (i < points.length && start[1] >= points[i][0]) i++;
            arrow += 1;
        }
        return arrow;
    }
}
