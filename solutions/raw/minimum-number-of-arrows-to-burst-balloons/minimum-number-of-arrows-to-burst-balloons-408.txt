class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> a[1] - b[1]);
        int noOfArrows = 1, idx = 1;
        int sp = points[0][1];
        while (idx < points.length) {
            if (points[idx][0] > sp || points[idx][1] < sp) {
                noOfArrows++;
                sp = points[idx][1];
            }
            idx++;
        }
        return noOfArrows;
    }
}
