class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int c = 1;
        int k = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (!(k <= points[i][1] && k >= points[i][0])) {
                c++;
                k = points[i][1];
            }
        }
        return c;
    }
}
