class Solution {

    public int findMinArrowShots(int[][] points) {
        // Arrays.sort(points , (a,b)->{
        // //  if(a[0] == b[0]) return a[1]- b[1];
        //  return a[1] - b[1];
        // });
        // one test case wont help for the above test case
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int arrowLocation = points[0][1];
        int arrows = 1;
        for (int i = 0; i < points.length; i++) {
            int st = points[i][0];
            int end = points[i][1];
            if (st <= arrowLocation) {
                // if the starting location was lessthan the arrowlocation
                //set the arrowlocation at the edge of the smaller ballon
                arrowLocation = Math.min(arrowLocation, end);
            } else {
                // if the ballons are not arrowlocation arrows increase
                arrows++;
                // arrange the arrow here
                arrowLocation = end;
            }
        }
        return arrows;
    }
}
