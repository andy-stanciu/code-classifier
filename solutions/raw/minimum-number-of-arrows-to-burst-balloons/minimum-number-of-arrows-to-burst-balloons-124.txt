class Solution {

    public int findMinArrowShots(int[][] p) {
        Arrays.sort(p, (arr1, arr2) -> Integer.compare(arr1[1], arr2[1]));
        int i = 0;
        int arrows = 0;
        while (i < p.length) {
            int next = p[i][1];
            while (i + 1 < p.length && p[i + 1][0] <= next) i++;
            i++;
            arrows++;
        }
        return arrows;
    }
}
