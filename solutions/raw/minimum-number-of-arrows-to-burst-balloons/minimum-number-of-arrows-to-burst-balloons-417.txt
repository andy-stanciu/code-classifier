class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparing(a -> a[1]));
        int len = points.length;
        int count = 0;
        int i;
        int start = points[0][1];
        for (i = 0; i < len; i++) {
            if (start < points[i][0]) {
                count++;
                start = points[i][1];
            }
        }
        return count + 1;
    }
}
