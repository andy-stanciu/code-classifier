class point {

    int start;

    int end;

    point(int start, int end) {
        this.start = start;
        this.end = end;
    }
}

class pointsComparator implements Comparator<point> {

    @Override
    public int compare(point p1, point p2) {
        if (p1.end < p2.end)
            return -1;
        else if (p1.end > p2.end)
            return 1;
        else if (p1.start < p2.start)
            return -1;
        else
            return 1;
    }
}

class Solution {

    public int findMinArrowShots(int[][] points) {
        ArrayList<point> list = new ArrayList<>();
        for (int i = 0; i < points.length; i++) {
            list.add(new point(points[i][0], points[i][1]));
        }
        pointsComparator pc = new pointsComparator();
        // Sort the points array according to the end point
        Collections.sort(list, pc);
        int res = 0;
        // initially the first end point
        int endPoint = list.get(0).end;
        for (int i = 1; i < points.length; i++) {
            // if the points less than endPoint then continue
            if (list.get(i).end <= endPoint || list.get(i).start <= endPoint) {
                continue;
            } else {
                res++;
                endPoint = list.get(i).end;
            }
        }
        return res + 1;
    }
}
