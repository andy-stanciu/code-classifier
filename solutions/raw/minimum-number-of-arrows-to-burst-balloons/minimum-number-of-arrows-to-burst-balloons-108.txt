class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (p1, p2) -> Long.compare(p1[1], p2[1]));
        int res = 1;
        long endOfInterval = points[0][1];
        for (int index = 1; index < points.length; index++) {
            if (points[index][0] > endOfInterval) {
                res++;
                endOfInterval = points[index][1];
            }
        }
        return res;
    }
}
