class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (o1, o2) -> Integer.compare(o1[1], o2[1]));
        int arrows = 1;
        int reach = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (reach < points[i][0]) {
                // check for the condition whether the next balloon is in the range of previous balloon
                // if it is not in the range of first one then we need extra arrow
                arrows++;
                reach = points[i][1];
            } else {
                // since the balloon is in the range of previous balloon we will not do anything.
                // we will skip and check for the next balloon.
            }
        }
        return arrows;
    }
}
