class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int n = points.length, arrows = 0;
        for (int i = 0; i < n; i++) {
            int end = points[i][1];
            while (i + 1 < n && end >= points[i + 1][0]) i++;
            arrows++;
        }
        return arrows;
    }
}
