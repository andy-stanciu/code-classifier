class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1;
        int[] prevPoint = points[0];
        for (int[] currPoint : points) {
            if (prevPoint[1] < currPoint[0]) {
                count++;
                prevPoint = currPoint;
            }
        }
        return count;
    }
}
