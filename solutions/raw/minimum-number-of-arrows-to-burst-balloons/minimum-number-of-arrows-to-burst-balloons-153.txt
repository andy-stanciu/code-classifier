class Solution {

    public int findMinArrowShots(int[][] points) {
        int numOfBallons = points.length;
        if (numOfBallons == 1)
            return 1;
        Arrays.sort(points, Comparator.comparingInt(e -> e[0]));
        int[] targetBallon = points[0];
        int arrowCount = 1;
        for (int i = 1; i < numOfBallons; i++) {
            if (points[i][0] <= targetBallon[1]) {
                targetBallon[0] = Math.max(targetBallon[0], points[i][0]);
                targetBallon[1] = Math.min(targetBallon[1], points[i][1]);
            } else {
                arrowCount++;
                targetBallon = points[i];
            }
        }
        return arrowCount;
    }
}
