class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (b, a) -> Double.compare(b[0], a[0]));
        int ans = 1;
        int end = points[0][1];
        int start = points[0][0];
        for (int i = 1; i < points.length; i++) {
            if (points[i][1] >= start && points[i][1] <= end) {
                end = Math.min(points[i][1], end);
                continue;
            } else if (end >= points[i][0] && end <= points[i][1])
                continue;
            else {
                ans++;
                start = points[i][0];
                end = points[i][1];
            }
        }
        return ans;
    }
}
