class Solution {

    public int findMinArrowShots(int[][] brr) {
        Arrays.sort(brr, Comparator.comparing(o -> o[0]));
        int ans = 1;
        int range = brr[0][1];
        for (int i = 1; i < brr.length; i++) {
            if (brr[i][0] <= range) {
                range = Math.min(range, brr[i][1]);
            } else {
                ans++;
                range = brr[i][1];
            }
        }
        return ans;
    }
}
