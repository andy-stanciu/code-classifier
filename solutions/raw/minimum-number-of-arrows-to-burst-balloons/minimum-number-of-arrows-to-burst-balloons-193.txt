class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> a[1] - b[1]);
        int numArrows = 1;
        int arrPos = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (arrPos < points[i][0] || arrPos > points[i][1]) {
                numArrows++;
                arrPos = points[i][1];
            }
        }
        return numArrows;
    }
}
