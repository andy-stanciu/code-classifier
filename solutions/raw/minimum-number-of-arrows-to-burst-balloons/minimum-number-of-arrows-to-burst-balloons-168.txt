class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(point -> point[0]));
        int arrows = 1;
        int end = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (end < points[i][0]) {
                arrows++;
                end = points[i][1];
            } else {
                end = Math.min(end, points[i][1]);
            }
        }
        return arrows;
    }
}
