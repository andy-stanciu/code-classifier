class Solution {

    public int findMinArrowShots(int[][] points) {
        int count = 1;
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int min = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] <= min) {
                min = Math.min(min, points[i][1]);
            } else {
                count++;
                min = points[i][1];
            }
        }
        return count;
    }
}
