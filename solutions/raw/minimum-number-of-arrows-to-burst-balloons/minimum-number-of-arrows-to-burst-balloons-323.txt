class Solution {

    public int findMinArrowShots(int[][] points) {
        int n = points.length;
        Arrays.sort(points, Comparator.comparingInt(o -> o[1]));
        int cnt = 1;
        int prev = points[0][1];
        for (int i = 1; i < n; i++) {
            if (prev < points[i][0]) {
                cnt++;
                prev = points[i][1];
            }
        }
        return cnt;
    }
}
