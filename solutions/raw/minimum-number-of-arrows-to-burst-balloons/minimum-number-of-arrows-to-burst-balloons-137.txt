class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, Comparator.comparing(point -> point[0]));
        var lastBalloon = points[0];
        var minimumNumberOfArrows = 1;
        for (var point : points) {
            if (Math.max(point[0], lastBalloon[0]) <= Math.min(point[1], lastBalloon[1])) {
                lastBalloon[1] = Math.min(point[1], lastBalloon[1]);
                continue;
            }
            lastBalloon = point;
            minimumNumberOfArrows++;
        }
        return minimumNumberOfArrows;
    }
}
