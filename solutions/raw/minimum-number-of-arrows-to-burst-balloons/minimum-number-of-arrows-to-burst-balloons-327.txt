class Solution {

    public int findMinArrowShots(int[][] points) {
        if (points.length == 0)
            return 0;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int pointer = points[0][1], res = 1;
        for (int i = 0; i < points.length; i++) {
            if (points[i][0] > pointer) {
                res++;
                pointer = points[i][1];
            }
        }
        return res;
    }
}
