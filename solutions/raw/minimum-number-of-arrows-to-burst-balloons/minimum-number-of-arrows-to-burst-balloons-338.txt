class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int i = 0, count = 0;
        while (i < points.length) {
            int j = i + 1;
            while (j < points.length && (points[j][0] <= points[i][1] && points[j][1] >= points[i][1])) {
                j++;
            }
            count++;
            i = j;
        }
        return count;
    }
}
