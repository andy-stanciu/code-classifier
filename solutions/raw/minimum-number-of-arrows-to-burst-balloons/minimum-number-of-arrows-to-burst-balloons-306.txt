class Solution {

    public int findMinArrowShots(int[][] points) {
        Arrays.sort(points, new Comparator<int[]>() {

            @Override
            public int compare(int[] o1, int[] o2) {
                if (o1[0] == o2[0])
                    return Integer.compare(o1[1], o2[1]);
                return Integer.compare(o1[0], o2[0]);
            }
        });
        int n = points.length;
        int[] prev = points[0];
        int count = 1, mini = prev[1];
        for (int i = 1; i < n; i++) {
            int startp = prev[0];
            int endp = prev[1];
            int startc = points[i][0];
            int endc = points[i][1];
            if (endp >= startc && mini >= startc) {
                prev[1] = Math.max(endp, endc);
                mini = Math.min(mini, endc);
            } else {
                prev = points[i];
                mini = endc;
                count++;
            }
        }
        return count;
    }
}
