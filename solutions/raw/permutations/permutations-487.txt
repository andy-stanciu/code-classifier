class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        getway_permu(nums, 0, nums.length);
        return ans;
    }

    public void swap(int[] n, int i, int j) {
        int t = n[i];
        n[i] = n[j];
        n[j] = t;
    }

    public void getway_permu(int[] n, int i, int l) {
        if (i == l) {
            List<Integer> temp = new ArrayList<>();
            for (int x : n) {
                temp.add(x);
            }
            ans.add(temp);
            return;
        }
        for (int j = i; j < l; j++) {
            swap(n, i, j);
            getway_permu(n, i + 1, l);
            swap(n, i, j);
        }
    }
}
