class Solution {

    static void solve(int i, int[] nums, List<List<Integer>> res) {
        if (i == nums.length) {
            List<Integer> arr = new ArrayList<>();
            for (int x : nums) arr.add(x);
            res.add(new ArrayList<>(arr));
            return;
        }
        for (int a = i; a < nums.length; a++) {
            swap(nums, i, a);
            solve(i + 1, nums, res);
            swap(nums, i, a);
        }
    }

    static int[] swap(int[] nums, int i, int a) {
        int temp = nums[i];
        nums[i] = nums[a];
        nums[a] = temp;
        return nums;
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        solve(0, nums, res);
        return res;
    }
}
