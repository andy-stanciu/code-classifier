class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        solve(visited, nums, ans, new ArrayList<>());
        return ans;
    }

    static void solve(boolean[] visited, int[] nums, List<List<Integer>> ans, List<Integer> curr) {
        if (curr.size() == nums.length) {
            ans.add(new ArrayList<>(curr));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (visited[i] == false) {
                visited[i] = true;
                curr.add(nums[i]);
                solve(visited, nums, ans, curr);
                curr.remove(curr.size() - 1);
                visited[i] = false;
            }
        }
    }
}
