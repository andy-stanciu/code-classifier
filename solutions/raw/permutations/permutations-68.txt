class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        generate(res, nums, 0);
        return res;
    }

    private void generate(List<List<Integer>> res, int[] nums, int ind) {
        if (ind == nums.length) {
            List<Integer> arr = new ArrayList<>();
            for (int num : nums) {
                arr.add(num);
            }
            res.add(arr);
            return;
        }
        for (int i = ind; i < nums.length; i++) {
            swap(nums, i, ind);
            generate(res, nums, ind + 1);
            swap(nums, i, ind);
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
