class Solution {

    private List<List<Integer>> result;

    private int finalMask;

    public List<List<Integer>> permute(int[] nums) {
        result = new ArrayList<>();
        // for arr of 3 elements we have bit mask like that 111(2) => 7(10)
        // 2 ^ 3 = 8 - 1
        // we use this mask for determine that all numbers were used
        finalMask = (int) Math.pow(2, nums.length) - 1;
        // instead of creating tmp list and change in each step we can keep postion of number in static array
        int[] positions = new int[nums.length];
        backtrack(nums, positions, 0, 0);
        return result;
    }

    private void backtrack(int[] nums, int[] positions, int mask, int index) {
        if (mask == finalMask) {
            addPermutation(nums, positions);
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            // by bit mask we can determine did we use or not this number
            if ((mask & 1 << i) != 0)
                continue;
            // save position of number
            positions[index] = i;
            backtrack(nums, positions, mask | 1 << i, index + 1);
        }
    }

    private void addPermutation(int[] nums, int[] positions) {
        List<Integer> tmp = new ArrayList<>();
        for (int idx : positions) tmp.add(nums[idx]);
        result.add(tmp);
    }
}
