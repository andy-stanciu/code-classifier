class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        boolean[] map = new boolean[nums.length];
        findPermutations(nums, ds, ans, map);
        return ans;
    }

    public void findPermutations(int[] nums, List<Integer> ds, List<List<Integer>> ans, boolean[] map) {
        //base condition
        if (ds.size() == nums.length) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!map[i]) {
                // if entry is true and not picked already
                // picked now and marked it true
                map[i] = true;
                // add the number to the data structure
                ds.add(nums[i]);
                //recursive call
                findPermutations(nums, ds, ans, map);
                // remove the number
                ds.remove(ds.size() - 1);
                // mark the entry false to make it available
                map[i] = false;
            }
        }
    }
}
