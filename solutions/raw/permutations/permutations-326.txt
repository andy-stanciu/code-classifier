class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> firstPermutation = new ArrayList<>(Arrays.asList(nums[0]));
        result.add(firstPermutation);
        for (int i = 1; i < nums.length; i++) {
            int num = nums[i];
            List<List<Integer>> newPermutations = new ArrayList<>();
            for (List<Integer> perm : result) {
                int permLength = perm.size();
                for (int j = 0; j <= permLength; j++) {
                    List<Integer> newPerm = new ArrayList<>(perm);
                    newPerm.add(j, num);
                    newPermutations.add(newPerm);
                }
            }
            result = newPermutations;
        }
        return result;
    }
}
