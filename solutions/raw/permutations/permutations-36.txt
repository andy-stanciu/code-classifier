class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> answer = new ArrayList<>();
        helper(0, nums, answer);
        return answer;
    }

    public void helper(int index, int[] nums, List<List<Integer>> answer) {
        if (index == nums.length) {
            List<Integer> temp = new ArrayList<>();
            for (int i = 0; i < nums.length; i++) {
                temp.add(nums[i]);
            }
            answer.add(temp);
            return;
        }
        for (int i = index; i < nums.length; i++) {
            int temp = nums[i];
            nums[i] = nums[index];
            nums[index] = temp;
            // ek ek baar us position par usse aage ke saare number ek ek karke laayenge ;
            //  aur wapas aate wakt dubara phle jaise aarray kar denge ;
            helper(index + 1, nums, answer);
            // taaki same phle jaia array ho sake ;
            temp = nums[i];
            nums[i] = nums[index];
            nums[index] = temp;
        }
    }
}
