class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> list = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        backtrack(list, nums, new ArrayList(), visited);
        return list;
    }

    void backtrack(List<List<Integer>> list, int[] nums, List<Integer> curr, boolean[] visited) {
        if (curr.size() == nums.length) {
            list.add(new ArrayList(curr));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (visited[i] == true)
                continue;
            curr.add(nums[i]);
            visited[i] = true;
            backtrack(list, nums, curr, visited);
            curr.remove(curr.size() - 1);
            visited[i] = false;
        }
    }
}
