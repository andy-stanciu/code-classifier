class Solution {

    public void findOtherPermutation(List<List<Integer>> result, List<Integer> singlePermute, int[] nums) {
        // If the current permutation is complete add it to the result
        if (singlePermute.size() == nums.length) {
            result.add(new ArrayList<>(singlePermute));
            return;
        }
        // Iterate through the nums array
        for (int num : nums) {
            if (singlePermute.contains(num)) {
                continue;
            }
            singlePermute.add(num);
            findOtherPermutation(result, singlePermute, nums);
            singlePermute.remove(singlePermute.size() - 1);
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        findOtherPermutation(result, new ArrayList<>(), nums);
        return result;
    }
}
