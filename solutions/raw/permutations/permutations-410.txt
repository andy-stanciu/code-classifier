class Solution {

    static void generatePermutations(int[] nums, List<Integer> inner, List<List<Integer>> outer) {
        // Base Case
        if (inner.size() == nums.length) {
            outer.add(new ArrayList<Integer>(inner));
            return;
        }
        // Adding element to inner (generating permutations)
        for (int i = 0; i < nums.length; i++) {
            if (!inner.contains(nums[i])) {
                inner.add(nums[i]);
                generatePermutations(nums, inner, outer);
                inner.remove(inner.size() - 1);
            }
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> outer = new ArrayList();
        List<Integer> inner = new ArrayList();
        generatePermutations(nums, inner, outer);
        return outer;
    }
}
