class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        solve(ans, new ArrayList<Integer>(), nums);
        return ans;
    }

    public void solve(List<List<Integer>> ans, ArrayList<Integer> arrayList, int[] nums) {
        if (arrayList.size() == nums.length) {
            ans.add(new ArrayList<Integer>(arrayList));
            return;
        }
        for (int num : nums) {
            if (arrayList.contains(num))
                continue;
            arrayList.add(num);
            solve(ans, arrayList, nums);
            arrayList.remove(arrayList.size() - 1);
        }
    }
}
