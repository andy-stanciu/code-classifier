class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> permutations = new ArrayList<>();
        LinkedList<Integer> start = new LinkedList<>();
        start.add(nums[0]);
        permutations.add(start);
        for (int i = 1; i < nums.length; i++) {
            List<List<Integer>> newPermutations = new ArrayList<>();
            for (List permutation : permutations) {
                for (int j = 0; j <= permutation.size(); j++) {
                    LinkedList<Integer> newPermutation = new LinkedList<>(permutation);
                    newPermutation.add(j, nums[i]);
                    newPermutations.add(newPermutation);
                }
            }
            permutations = newPermutations;
        }
        return permutations;
    }
}
