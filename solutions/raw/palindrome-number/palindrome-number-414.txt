class Solution {

    public boolean isPalindrome(int n) {
        int reverse = 0;
        int x = n;
        if (n < 0) {
            return false;
        }
        while (n > 0) {
            int lastdigit = n % 10;
            reverse = reverse * 10 + lastdigit;
            n = n / 10;
        }
        return reverse == x;
    }
}
