class Solution {

    public boolean isPalindrome(int x) {
        int duplicate = x;
        int ld = -1;
        int revnum = 0;
        while (x > 0) {
            ld = x % 10;
            x = x / 10;
            revnum = (revnum * 10) + ld;
        }
        if (revnum == duplicate) {
            return true;
        }
        return false;
    }
}
