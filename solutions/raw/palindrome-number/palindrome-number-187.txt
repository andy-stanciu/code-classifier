class Solution {

    public boolean isPalindrome(int x) {
        char[] numberAsCharArray = String.valueOf(x).toCharArray();
        for (int i = 0; i < numberAsCharArray.length / 2; i++) {
            if (numberAsCharArray[i] != numberAsCharArray[numberAsCharArray.length - 1 - i]) {
                return false;
            }
        }
        return true;
    }
}
