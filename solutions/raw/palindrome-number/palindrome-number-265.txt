class Solution {

    public boolean isPalindrome(int x) {
        return (x < 0) ? false : reverse(x, 0) == x;
    }

    //if x negative return false; otherwise reverse and check
    public int reverse(int n, int reversed) {
        return (n == 0) ? reversed : reverse(n / 10, reversed * 10 + n % 10);
    }
    //if n == 0 return our built number reversed. otherwise keep building
}
