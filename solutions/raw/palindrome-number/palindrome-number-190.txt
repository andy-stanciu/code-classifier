class Solution {

    public boolean isPalindrome(int x) {
        // creating variables for reversed integer and a copy for x
        long rev = 0, xCopy = x;
        // I did xCopy > 0 instead of xCopy != 0 so that it excludes negative integers
        while (xCopy > 0) {
            rev = rev * 10 + xCopy % 10;
            xCopy /= 10;
        }
        // Returns true only when rev is equal to x AND xCopy isn\'t negative
        return (rev == x && xCopy > -1);
    }
}
