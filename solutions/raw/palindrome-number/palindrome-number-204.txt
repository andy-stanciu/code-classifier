class Solution {

    public boolean isPalindrome(int x) {
        int digit = 0;
        int rev = 0;
        int m = x;
        while (x > 0) {
            digit = x % 10;
            rev = rev * 10 + digit;
            x = x / 10;
        }
        if (rev == m) {
            return true;
        } else {
            return false;
        }
    }
}
