class Solution {

    public boolean isPalindrome(int x) {
        int y = Math.abs(x);
        int s = 0;
        while (y != 0) {
            int b = y % 10;
            s = s * 10 + b;
            y = y / 10;
        }
        if ((s == x) && x >= 0)
            return true;
        else
            return false;
    }
}
