class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int num = x;
        int palindrome = 0;
        while (num > 0) {
            palindrome = palindrome * 10 + num % 10;
            num /= 10;
        }
        return x == palindrome;
    }
}
