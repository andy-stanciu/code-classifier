class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0)
            return false;
        int reverseNum = 0;
        int original = x;
        while (x != 0) {
            int LastDigit = x % 10;
            reverseNum = reverseNum * 10 + LastDigit;
            x = x / 10;
        }
        return reverseNum == original;
    }
}
