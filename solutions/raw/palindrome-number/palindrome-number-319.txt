class Solution {

    public boolean isPalindrome(int x) {
        // Negative numbers or numbers ending in 0 (but not 0 itself) are not palindromes.
        if (x < 0 || (x != 0 && x % 10 == 0))
            return false;
        int rev = 0;
        // Reverse half of the digits of the number.
        while (x > rev) {
            // Append the last digit of x to rev.
            rev = rev * 10 + x % 10;
            // Remove the last digit of x.
            x /= 10;
        }
        // Check if the number is a palindrome.
        return (x == rev || x == rev / 10);
    }
}
