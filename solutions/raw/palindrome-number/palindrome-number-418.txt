class Solution {

    public int reverseIt(int org) {
        int rev = 0;
        // single digit value is always palindrome
        if (org < 10 && org > -1)
            return org;
        // reversing the number
        while (org > 0) {
            int rem = org % 10;
            org /= 10;
            // Checking the reverse is exeeding the limit or not
            if (rev / 10 > Integer.MAX_VALUE)
                return 0;
            rev = rev * 10 + rem;
        }
        return rev;
    }

    public boolean isPalindrome(int x) {
        int org = x;
        // zero return will doesn\'t match with the org and proint false
        if (org < 0) {
            return false;
        }
        int rev = reverseIt(x);
        ;
        // checking both are equal or not
        if (rev == org)
            return true;
        else
            return false;
    }
}
