class Solution {

    public boolean isPalindrome(int x) {
        // Break the int down into a char array
        char[] chars = String.valueOf(x).toCharArray();
        // Establish our bounds
        int left = 0;
        int right = chars.length - 1;
        // Loop through and check each element at beginning and end, moving inward with each loop
        while (left < right) {
            if (chars[left] != chars[right]) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
