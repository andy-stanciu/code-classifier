class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        } else if (x == Peversed(x)) {
            return true;
        }
        return false;
    }

    static int Peversed(int x) {
        int reversed = 0;
        while (x != 0) {
            reversed = reversed * 10 + x % 10;
            x = x / 10;
        }
        return reversed;
    }
}
