class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0 || (x != 0 && x % 10 == 0)) {
            return false;
        }
        int c = x;
        int sum = 0;
        while (x > 0) {
            int tem = x % 10;
            sum = sum * 10 + tem;
            x /= 10;
        }
        return (sum == c);
    }
}
