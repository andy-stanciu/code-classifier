class Solution {

    public boolean isPalindrome(int x) {
        String s = String.valueOf(x);
        int len = s.length();
        int left = 0;
        int right = len - 1;
        while (left <= right) {
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
