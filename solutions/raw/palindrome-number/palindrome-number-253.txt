class Solution {

    public boolean isPalindrome(int x) {
        int temp = x;
        int rev = 0;
        while (temp > 0) {
            int rem = temp % 10;
            rev = rem + rev * 10;
            temp = temp / 10;
        }
        if (x == rev) {
            return true;
        }
        return false;
    }
}
