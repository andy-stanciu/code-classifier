class Solution {

    public static boolean isPalindrome(int x) {
        int r, sum = 0;
        int tem = x;
        while (x > 0) {
            r = x % 10;
            sum = (sum * 10) + r;
            x = x / 10;
        }
        if (sum == tem) {
            return true;
        }
        return false;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Number :- ");
        int x = sc.nextInt();
        boolean ans = isPalindrome(x);
        System.out.print(ans);
    }
}
