class Solution {

    public boolean isPalindrome(int x) {
        int copyX = x;
        int newX = 0;
        while (x > 0) {
            int n = x % 10;
            newX = (newX * 10) + n;
            x = x / 10;
        }
        if (newX == copyX) {
            return true;
        }
        return false;
    }
}
