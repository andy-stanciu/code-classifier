class Solution {

    public boolean isPalindrome(int x) {
        String number = String.valueOf(x);
        for (int i = 0; i < number.length(); i++) {
            char c = number.charAt(i);
            char d = number.charAt(number.length() - 1 - i);
            if (c != d) {
                return false;
            }
        }
        return true;
    }
}
