class Solution {

    /**
     *    Time Complexity: O(d), d = number of digits.
     */
    public boolean isPalindrome(int x) {
        // first we need to define one temporary variable to store the original value of x
        int temp = x, rem = 0, res = 0;
        if (x < 0) {
            // If the number is negative then it will never be palindrome due to the -Ve sign.
            return false;
        }
        while (x != 0) {
            rem = x % 10;
            res = res * 10 + rem;
            x /= 10;
        }
        if (res == temp) {
            return true;
        }
        return false;
    }
}
