class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int y = x;
        int r = 0;
        long rev = 0;
        while (x != 0) {
            r = x % 10;
            rev = (rev * 10) + r;
            x /= 10;
        }
        if (y == rev) {
            return true;
        } else {
            return false;
        }
    }
}
