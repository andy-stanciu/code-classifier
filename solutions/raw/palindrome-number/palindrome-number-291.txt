class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int original = x;
        int y = 0;
        while (x != 0) {
            int rem = x % 10;
            y = y * 10 + rem;
            x /= 10;
        }
        return original == y;
    }
}
