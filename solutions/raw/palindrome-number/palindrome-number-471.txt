class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int palindrome = 0, originalNumber = x, currentDigit;
        while (x > 0) {
            currentDigit = x % 10;
            palindrome = palindrome * 10 + currentDigit;
            x /= 10;
        }
        return palindrome == originalNumber;
    }
}
