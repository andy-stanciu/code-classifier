class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        String num = Integer.toString(x);
        int i = 0;
        int j = num.length() - 1;
        while (i < j) {
            char ci = num.charAt(i);
            char cj = num.charAt(j);
            if (ci != cj) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
