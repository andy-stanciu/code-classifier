class Solution {

    public boolean isPalindrome(int x) {
        int p = x, remainder, pallin = 0;
        while (x > 0) {
            remainder = x % 10;
            pallin = (pallin * 10) + remainder;
            x /= 10;
        }
        if (p == pallin)
            return true;
        else
            return false;
    }
}
