class Solution {

    public boolean isPalindrome(int x) {
        int d, sum = 0, temp = x;
        if (x < 0) {
            return false;
        }
        while (x != 0) {
            d = x % 10;
            sum = sum * 10 + d;
            x = x / 10;
        }
        return temp == sum;
    }
}
