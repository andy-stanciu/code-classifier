class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int b = x;
        int n = 0;
        while (x > 0) {
            int a = x % 10;
            n = (n * 10) + a;
            x = x / 10;
        }
        if (b == n) {
            return true;
        }
        return false;
    }
}
