class Solution {

    public boolean isPalindrome(int x) {
        if (x == 0) {
            return true;
        } else if (x > 0) {
            int temp = x;
            int palindrome = 0;
            //MAIN LOGIC
            while (x > 0) {
                int remainder = x % 10;
                palindrome = palindrome * 10 + remainder;
                x = x / 10;
            }
            if (palindrome == temp) {
                return true;
            } else {
                return false;
            }
        } else {
            return false;
        }
    }
}
