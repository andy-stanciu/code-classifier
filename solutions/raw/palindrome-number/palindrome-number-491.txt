class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        } else {
            int k = x;
            int count = 0;
            while (k > 0) {
                count++;
                k = k / 10;
            }
            int[] num = new int[count];
            k = x;
            for (int i = num.length - 1; i >= 0; i--) {
                num[i] = k % 10;
                k = k / 10;
            }
            int i = 0;
            int j = num.length - 1;
            while (i < num.length) {
                if (num[i] != num[j]) {
                    return false;
                }
                i++;
                j--;
            }
            return true;
        }
    }
}
