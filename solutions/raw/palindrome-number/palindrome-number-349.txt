class Solution {

    public boolean isPalindrome(int x) {
        int sign = (x >= 0) ? 1 : -1;
        if (sign == 1) {
            int num = 0;
            int original = x;
            while (x > 0) {
                int val = x % 10;
                num = num * 10 + val;
                x /= 10;
            }
            return original == num;
        }
        return false;
    }
}
