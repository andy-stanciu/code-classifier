class Solution {

    public boolean isPalindrome(int x) {
        //We can reverse a number and compare.
        if (x < 0) {
            return false;
        }
        int num = reverse(x, 0);
        return num == x;
    }

    public int reverse(int x, int res) {
        if (x == 0) {
            return res;
        }
        //x/10 moves the places of digit and x%10 get the last digit
        return reverse(x / 10, (x % 10) + res * 10);
    }
}
