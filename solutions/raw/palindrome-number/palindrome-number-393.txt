class Solution {

    public boolean isPalindrome(int x) {
        //for checking if number is negative
        if (x < 0) {
            return false;
        }
        int rev = 0;
        int xcopy = x;
        while (x > 0) {
            //append x to rev multiplying by 10
            //extract last digit
            rev = (rev * 10) + (x % 10);
            //removes last digit
            x /= 10;
        }
        return rev == xcopy;
    }
}
