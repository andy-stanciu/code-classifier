class Solution {

    public boolean isPalindrome(int x) {
        int temp = x;
        int ans = 0;
        for (int i = x; i > 0; i = i / 10) {
            int digit = i % 10;
            ans = (ans * 10) + digit;
        }
        if (ans == temp) {
            return true;
        } else {
            return false;
        }
    }
}
