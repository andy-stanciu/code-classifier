class Solution {

    public boolean isPalindrome(int x) {
        int temp = x;
        int dig;
        int rev = 0;
        if (x < 0)
            return false;
        while (x != 0) {
            dig = x % 10;
            rev = rev * 10 + dig;
            x /= 10;
        }
        if (rev == temp)
            return true;
        else
            return false;
    }
}
