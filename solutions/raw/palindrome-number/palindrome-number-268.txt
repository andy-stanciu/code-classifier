class Solution {

    public boolean isPalindrome(int x) {
        int reversedX = 0;
        int copiedX = x;
        if (x < 0) {
            return false;
        }
        while (copiedX != 0) {
            //get the last digit of x
            int digit = copiedX % 10;
            //create space and append digit to the next available position
            reversedX = reversedX * 10 + digit;
            //get rid of the last digit of x
            copiedX = copiedX / 10;
        }
        if (x == reversedX) {
            return true;
        } else {
            return false;
        }
    }
}
