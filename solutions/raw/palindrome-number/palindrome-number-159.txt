class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0)
            return false;
        long n = 0;
        int temp = x, d;
        while (temp > 0) {
            d = temp % 10;
            n = n * 10 + d;
            temp /= 10;
        }
        if (n == (long) x)
            return true;
        else
            return false;
    }
}
