class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0 || (x != 0 && x % 10 == 0))
            return false;
        int inverted = 0;
        while (x > inverted) {
            inverted = inverted * 10 + x % 10;
            x /= 10;
        }
        return x == inverted || x == inverted / 10;
    }
}
