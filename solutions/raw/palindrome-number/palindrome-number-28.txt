class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0)
            return false;
        int i = x, rev = 0;
        while (x > 0) {
            rev = (rev * 10) + (x % 10);
            x = (int) (x / 10);
        }
        if (rev == i)
            return true;
        else
            return false;
    }
}
