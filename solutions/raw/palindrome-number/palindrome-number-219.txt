class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        String str = Integer.toString(x);
        int length = str.length();
        for (int i = 0, y = length - 1; i <= length / 2; i++, y--) {
            if (str.charAt(i) != str.charAt(y)) {
                return false;
            }
        }
        return true;
    }
}
