class Solution {

    public boolean isPalindrome(int x) {
        int j = 0;
        char[] th = String.valueOf(x).toCharArray();
        int size = th.length;
        char[] rev = new char[size];
        for (int i = th.length - 1; i >= 0; i--) {
            rev[j] = th[i];
            j++;
        }
        int result = Arrays.compare(rev, th);
        if (result == 0)
            return true;
        else
            return false;
    }
}
