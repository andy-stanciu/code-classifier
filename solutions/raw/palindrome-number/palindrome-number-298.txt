class Solution {

    public boolean isPalindrome(int x) {
        String str = Integer.toString(x);
        int len = str.length();
        if (len == 0 || len == 1) {
            return true;
        }
        for (int i = 0; i < len / 2; i++) {
            char curr = str.charAt(i);
            char opp = str.charAt(len - i - 1);
            if (curr != opp) {
                return false;
            }
        }
        return true;
    }
}
