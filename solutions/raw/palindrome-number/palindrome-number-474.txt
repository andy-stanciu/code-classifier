class Solution {

    public boolean isPalindrome(int x) {
        int dup = x;
        int reverse = 0;
        if (x < 0) {
            return false;
        }
        while (x != 0) {
            int num = x % 10;
            x = x / 10;
            reverse = (reverse * 10) + num;
        }
        return dup == reverse;
    }
}
