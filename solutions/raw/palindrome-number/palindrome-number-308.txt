class Solution {

    public boolean isPalindrome(int x) {
        long rev = 0, num;
        if (x < 0) {
            num = x;
            num = -num;
        } else {
            num = x;
        }
        while (x > 0) {
            rev = rev * 10 + x % 10;
            x /= 10;
        }
        return rev == num;
    }
}
