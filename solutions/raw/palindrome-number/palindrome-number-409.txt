class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        } else if (x == 0) {
            return true;
        }
        char[] numbers = String.valueOf(x).toCharArray();
        boolean result = false;
        for (int i = 0; i < numbers.length; i++) {
            if (numbers[i] == numbers[numbers.length - i - 1]) {
                result = true;
                continue;
            }
            result = false;
            break;
        }
        return result;
    }
}
