class Solution {

    public boolean isPalindrome(int x) {
        int nn = x, r = 0;
        if (nn < 0) {
            return false;
        }
        while (nn > 0) {
            int u = nn % 10;
            r = r * 10 + u;
            nn = nn / 10;
        }
        if (x == r) {
            return true;
        } else {
            return false;
        }
    }
}
