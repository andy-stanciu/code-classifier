class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        List<Integer> num = new ArrayList<>();
        while (x > 0) {
            int dig = x % 10;
            x = x / 10;
            num.add(dig);
        }
        int left = 0, right = num.size() - 1;
        while (left <= right) {
            if (num.get(left) != num.get(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
