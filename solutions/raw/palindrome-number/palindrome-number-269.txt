class Solution {

    public boolean isPalindrome(int x) {
        int reminder = 0;
        int num = x;
        int reverseNumber = 0;
        if (num < 0) {
            return false;
        } else {
            while (num != 0) {
                reminder = num % 10;
                reverseNumber = (reverseNumber * 10) + reminder;
                num /= 10;
            }
        }
        if (x == reverseNumber) {
            return true;
        } else {
            return false;
        }
    }
}
