class Solution {

    public boolean isPalindrome(int x) {
        int a = x;
        boolean o = false;
        int s = 0;
        while (a > 0) {
            int ld = a % 10;
            s = (s * 10) + ld;
            a = a / 10;
        }
        if (s == x) {
            o = true;
        }
        return o;
    }
}
