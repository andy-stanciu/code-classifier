class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int temp = x, rem;
        int rev = 0;
        while (temp != 0) {
            rem = temp % 10;
            rev = rev * 10 + rem;
            temp = temp / 10;
        }
        boolean isPallindrome = match(rev, x);
        return isPallindrome;
    }

    boolean match(int reverse, int original) {
        return original == reverse ? true : false;
    }
}
