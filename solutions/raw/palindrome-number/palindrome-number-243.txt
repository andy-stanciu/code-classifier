class Solution {

    public boolean isPalindrome(int x) {
        int c = 0;
        int t = x;
        int s = 0;
        while (x != 0) {
            int r = x % 10;
            s = s * 10 + r;
            x = x / 10;
        }
        if (t == s && t >= 0) {
            return true;
        } else {
            return false;
        }
    }
}
