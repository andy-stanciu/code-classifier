class Solution {

    public boolean isPalindrome(int x) {
        int num = 0, temp = x, rem;
        if (x < 0)
            return false;
        else {
            while (x > 0) {
                rem = x % 10;
                num = num * 10 + rem;
                x /= 10;
            }
        }
        return num == temp;
    }
}
