class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0)
            return false;
        int count = 0;
        int reverse = 0;
        int original = x;
        while (x > 0) {
            count = x % 10;
            x /= 10;
            reverse = reverse * 10 + count;
        }
        return (reverse == original);
    }
}
