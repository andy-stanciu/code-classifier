class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        } else if (x == 0) {
            return true;
        }
        int temp = x;
        int result = 0;
        if (x > 0) {
            while (x > 0) {
                result = result * 10 + x % 10;
                x = x / 10;
            }
            return result == temp;
        }
        return false;
    }
}
