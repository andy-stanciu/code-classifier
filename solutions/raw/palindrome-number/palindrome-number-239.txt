class Solution {

    public boolean isPalindrome(int x) {
        int remainder = 0, newNumber = 0, oldnumber = x;
        if (x < 0) {
            return false;
        } else if (x < 10) {
            return true;
        } else {
            while (x > 0) {
                remainder = x % 10;
                newNumber = newNumber * 10 + remainder;
                x = x / 10;
            }
            if (newNumber == oldnumber) {
                return true;
            } else {
                return false;
            }
        }
    }
}
