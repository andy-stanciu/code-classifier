class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        List<Integer> numbers = new ArrayList();
        while (x > 0) {
            numbers.add(x % 10);
            x = x / 10;
        }
        for (int i = 0; i < numbers.size(); i++) {
            if (numbers.get(i) != numbers.get(numbers.size() - (i + 1))) {
                return false;
            }
        }
        return true;
    }
}
