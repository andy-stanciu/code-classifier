class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0)
            return false;
        String s = String.valueOf(x);
        StringBuilder a = new StringBuilder(s.substring(s.length() / 2));
        StringBuilder b = new StringBuilder(s.substring(0, s.length() / 2));
        if (a.length() > b.length())
            a.delete(0, 1);
        return b.reverse().toString().contentEquals(a);
    }
}
