class Solution {

    public boolean isPalindrome(int x) {
        StringBuilder stringBuilder = new StringBuilder(String.valueOf(x));
        String firstHalf = stringBuilder.substring(0, stringBuilder.length() / 2);
        String secondHalf = stringBuilder.reverse().substring(0, stringBuilder.length() / 2);
        if (firstHalf.equals(secondHalf))
            return true;
        return false;
    }
}
