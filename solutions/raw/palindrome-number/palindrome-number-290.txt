class Solution {

    public boolean isPalindrome(int x) {
        int number = x;
        if (x == 0)
            return true;
        if (x < 0 || x % 10 == 0)
            return false;
        int rev = 0;
        while (x > 0) {
            int digit = x % 10;
            rev = rev * 10 + digit;
            x = x / 10;
        }
        if (number == rev) {
            return true;
        } else {
            return false;
        }
    }
}
