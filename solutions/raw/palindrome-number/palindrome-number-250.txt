class Solution {

    public boolean isPalindrome(int x) {
        int reverseNumber = 0;
        int dup = x;
        while (x > 0) {
            int ld = x % 10;
            reverseNumber = reverseNumber * 10 + ld;
            x = x / 10;
        }
        if (reverseNumber == dup) {
            return true;
        } else {
            return false;
        }
    }
}
