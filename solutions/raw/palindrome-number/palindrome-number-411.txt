class Solution {

    public boolean isPalindrome(int x) {
        boolean isPalindrome;
        int originalNumber = x;
        int reversed = 0;
        if (x < 0) {
            isPalindrome = false;
        } else {
            while (x != 0) {
                reversed = reversed * 10 + x % 10;
                x /= 10;
            }
            isPalindrome = reversed == originalNumber;
        }
        return isPalindrome;
    }
}
