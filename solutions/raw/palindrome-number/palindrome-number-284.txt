class Solution {

    public boolean isPalindrome(int x) {
        if (x >= 0 && x < 10) {
            return true;
        }
        if (x < 0) {
            return false;
        }
        var s = String.valueOf(x);
        var left = 0;
        var right = s.length() - 1;
        while (left < right) {
            if (!Objects.equals(s.charAt(left), s.charAt(right))) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
