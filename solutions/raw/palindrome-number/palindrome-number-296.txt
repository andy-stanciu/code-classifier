class Solution {

    public static boolean isPalindrome(int x) {
        int original = x, reversed = 0;
        while (x > 0) {
            int num = x % 10;
            reversed = reversed * 10 + num;
            x = x / 10;
        }
        return original == reversed;
    }

    public static void main(String[] args) {
        System.out.println(isPalindrome(121));
    }
}
