class Solution {

    public boolean isPalindrome(int x) {
        // when number is (-Ve).
        if (x < 0)
            return false;
        // conver integer to string
        String s = Integer.toString(x);
        int i = 0, j = s.length() - 1;
        while (i != s.length() / 2) {
            //check the palindrome or note.
            if (s.charAt(i) != s.charAt(j))
                return false;
            i++;
            j--;
        }
        return true;
    }
}
