class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        int dhanush = x;
        int dhanavel = 0;
        while (x > 0) {
            int temp = x % 10;
            dhanavel = (dhanavel * 10) + temp;
            x /= 10;
        }
        return dhanavel == dhanush;
    }
}
