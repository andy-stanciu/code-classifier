class Solution {

    public boolean isPalindrome(int x) {
        int no = 0;
        if (x < 0) {
            return false;
        }
        int c = x;
        while (x != 0) {
            int rem = x % 10;
            no = no * 10 + rem;
            x /= 10;
        }
        return no == c;
    }
}
