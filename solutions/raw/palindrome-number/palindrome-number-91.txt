class Solution {

    public boolean isPalindrome(int num) {
        if (num < 0)
            return false;
        int reverseInt = reverseNumber(num);
        if (num == reverseInt) {
            return true;
        } else {
            return false;
        }
    }

    public int reverseNumber(int num) {
        int reverseInt = 0;
        while (num > 0) {
            int digit = num % 10;
            reverseInt = (reverseInt * 10) + digit;
            num /= 10;
        }
        return reverseInt;
    }
}
