class Solution {

    public boolean isPalindrome(int x) {
        // Negative numbers and numbers ending with 0 (except 0 itself) are not palindromes
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int original = x;
        int rev = 0;
        while (x != 0) {
            int i = x % 10;
            rev = rev * 10 + i;
            x = x / 10;
        }
        // Compare the reversed number with the original number
        return rev == original;
    }
}
