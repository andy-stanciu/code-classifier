class Solution {

    public boolean isPalindrome(int x) {
        // Is it an integer?
        if (x < Integer.MIN_VALUE || x > Integer.MAX_VALUE)
            return false;
        //Is it positive? Negtive numbers can\'t be a palindrome
        if (x < 0)
            return false;
        // Any positive number under 10 is palindromic
        if (x < 10)
            return true;
        String numberAsString = String.valueOf(x);
        char[] charArrayOfNumber = numberAsString.toCharArray();
        int lastIndexPointer = charArrayOfNumber.length - 1;
        for (int i = 0; (i < charArrayOfNumber.length / 2); i++) {
            if (charArrayOfNumber[i] != charArrayOfNumber[lastIndexPointer])
                return false;
            lastIndexPointer--;
        }
        return true;
    }
}
