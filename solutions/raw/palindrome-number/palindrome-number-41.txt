class Solution {

    public boolean isPalindrome(int x) {
        //x will be safe
        int n = x;
        int rem, rev = 0;
        while (n > 0) {
            //last digit of n is stored in rem
            rem = n % 10;
            //we are now adding the remainders one by one to our rev.
            rev = rev * 10 + rem;
            //removing the last digit from n, as it had been already added to rev
            n = n / 10;
        }
        if (rev == x) {
            return true;
        }
        return false;
    }
}
