class Solution {

    public boolean isPalindrome(int x) {
        int sum = 0;
        int temp = x;
        if (x == 0) {
            return true;
        }
        while (x > 0) {
            int r = x % 10;
            sum = (sum * 10) + r;
            x /= 10;
            if (sum == temp) {
                return true;
            }
        }
        return false;
    }
}
