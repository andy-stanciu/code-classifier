class Solution {

    public static boolean isPalindrome(int x) {
        int a = x;
        int pali = 0;
        while (x != 0) {
            int rem = x % 10;
            pali = pali * 10 + rem;
            x /= 10;
        }
        if (a >= 0 && a == pali) {
            return true;
        } else {
            return false;
        }
    }
}
