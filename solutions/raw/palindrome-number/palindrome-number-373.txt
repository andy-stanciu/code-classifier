class Solution {

    public boolean isPalindrome(int x) {
        if (//-> 1st we had declare that if x is nagtive return false;
        x < 0) {
            return false;
        }
        // main logic start
        // here we are take O data to store x; (O is now x)
        int O = x;
        // remainder is now 0;
        int r = 0;
        // take d; for % divide O(mean x)
        int d;
        while (O != 0) {
            // these are maths so,  do it on notebook
            d = O % 10;
            r = r * 10 + d;
            O = O / 10;
        }
        //here  if r == x(mean O ) return true it\'s palindrome
        return r == x;
    }
}
