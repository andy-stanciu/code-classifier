class Solution {

    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;
        }
        String s = Integer.toString(x);
        int size = s.length();
        if (size == 1) {
            return true;
        }
        int i = size - 1;
        int j = 0;
        while (i > j) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            j++;
            i--;
        }
        return true;
    }
}
