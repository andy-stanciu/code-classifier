class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int n1 = 0, n2 = 0;
        int i = 0;
        while (n1 < nums1.length && n2 < nums2.length) {
            if (nums1[n1] == nums2[n2]) {
                nums2[i++] = nums1[n1];
                n1++;
                n2++;
            } else if (nums1[n1] < nums2[n2])
                n1++;
            else
                n2++;
        }
        int[] arr = new int[i];
        for (int q = 0; q < i; q++) {
            arr[q] = nums2[q];
        }
        return arr;
    }
}
