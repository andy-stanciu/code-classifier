class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] count = new int[1001];
        List<Integer> list = new ArrayList<>();
        for (int i : nums1) count[i]++;
        for (int i : nums2) {
            if (count[i] != 0) {
                list.add(i);
                count[i]--;
            }
        }
        int[] a = new int[list.size()];
        int i = 0;
        for (int e : list) a[i++] = e;
        return a;
    }
}
