class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int c = 0;
        ArrayList<Integer> a = new ArrayList<Integer>();
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 0; j < nums2.length; j++) {
                if (nums1[i] == nums2[j]) {
                    c++;
                    a.add(nums2[j]);
                    nums2[j] = -1;
                    break;
                } else {
                    continue;
                }
            }
        }
        int[] b = new int[c];
        for (int i = 0; i < a.size(); b[i] = a.get(i), i++) ;
        return b;
    }
}
