class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        HashMap<Integer, Integer> countNums1 = new HashMap<>();
        for (int num : nums1) {
            countNums1.put(num, countNums1.getOrDefault(num, 0) + 1);
        }
        // Step 2: Create a list to store the intersection elements
        List<Integer> intersection = new ArrayList<>();
        for (int num : nums2) {
            if (countNums1.getOrDefault(num, 0) > 0) {
                intersection.add(num);
                countNums1.put(num, countNums1.get(num) - 1);
            }
        }
        // Step 3: Convert the list to an array and return it
        int[] result = new int[intersection.size()];
        for (int i = 0; i < intersection.size(); i++) {
            result[i] = intersection.get(i);
        }
        return result;
    }
}
