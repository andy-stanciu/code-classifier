class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        ArrayList<Integer> ans1 = new ArrayList<>();
        Arrays.sort(nums2);
        Arrays.sort(nums1);
        int i = 0;
        int j = 0;
        while (i != nums1.length && j != nums2.length) {
            if (nums1[i] == nums2[j]) {
                ans1.add(nums1[i++]);
                j++;
            } else {
                if (nums1[i] > nums2[j]) {
                    j++;
                } else {
                    i++;
                }
            }
        }
        int[] result = new int[ans1.size()];
        for (int k = 0; k < ans1.size(); k++) {
            result[k] = ans1.get(k);
        }
        return result;
    }
}
