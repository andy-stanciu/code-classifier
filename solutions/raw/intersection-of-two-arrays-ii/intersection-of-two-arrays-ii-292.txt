class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int ans = 0;
        for (int i = 0, j = 0; i < nums1.length; i++) {
            while (j < nums2.length && nums2[j] < nums1[i]) j++;
            if (j == nums2.length)
                break;
            if (nums1[i] == nums2[j]) {
                ans++;
                j++;
                nums1[i] = (nums1[i] + 1) * -1;
            }
        }
        int[] a = new int[ans];
        int i = 0;
        for (int x : nums1) if (x < 0)
            a[i++] = (x * -1) - 1;
        return a;
    }
}
