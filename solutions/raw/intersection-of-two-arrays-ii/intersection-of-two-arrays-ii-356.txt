class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int m = Math.min(nums1.length, nums2.length);
        int[] a = new int[m];
        int i = 0, j = 0, x = 0;
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int k = 0;
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] == nums2[j]) {
                a[x] = nums1[i];
                i++;
                j++;
                x++;
            } else if (nums1[i] < nums2[j])
                i++;
            else
                j++;
        }
        int[] result = Arrays.copyOfRange(a, 0, x);
        return result;
    }
}
