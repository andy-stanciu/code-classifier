class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] count = new int[1001];
        int[] ans = new int[1000];
        for (int i = 0; i < nums1.length; i++) {
            count[nums1[i]]++;
        }
        int j = 0;
        for (int i = 0; i < nums2.length; i++) {
            if (count[nums2[i]]-- > 0) {
                ans[j++] = nums2[i];
            }
        }
        return Arrays.copyOf(ans, j);
    }
}
