class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        ArrayList<Integer> intermediaryList = new ArrayList<>();
        ArrayList<Integer> nums1List = Arrays.stream(nums1).boxed().collect(Collectors.toCollection(ArrayList::new));
        ArrayList<Integer> nums2List = Arrays.stream(nums2).boxed().collect(Collectors.toCollection(ArrayList::new));
        for (int i = 0; i < nums2.length; i++) {
            if (nums1List.contains(nums2List.get(i))) {
                intermediaryList.add(nums2List.get(i));
                nums1List.remove(nums2List.get(i));
            }
        }
        return intermediaryList.stream().mapToInt(i -> i).toArray();
    }
}
