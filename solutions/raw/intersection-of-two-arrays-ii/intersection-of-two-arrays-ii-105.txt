class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] ans = new int[1001];
        int[] finalAns = new int[1001];
        for (int i : nums1) {
            ans[i]++;
        }
        int count = 0;
        for (int i : nums2) {
            if (ans[i] > 0) {
                finalAns[count++] = i;
                ans[i]--;
            }
        }
        return Arrays.copyOfRange(finalAns, 0, count);
    }
}
