class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        List<Integer> list = new ArrayList<>();
        int ind = 0;
        int index2 = 0;
        while (ind < nums1.length && index2 < nums2.length) {
            // Corrected \'num2.length\' to \'nums2.length\'
            if (nums1[ind] == nums2[index2]) {
                list.add(nums1[ind]);
                ind++;
                index2++;
            } else if (nums1[ind] < nums2[index2]) {
                // Corrected \'nums[ind]\' to \'nums1[ind]\'
                ind++;
            } else {
                index2++;
            }
        }
        // Corrected \'list.length\' to \'list.size()\'
        int[] numbers = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            // Corrected \'number[i]\' to \'numbers[i]\'
            numbers[i] = list.get(i);
        }
        return numbers;
    }
}
