class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int n = nums1.length;
        int m = nums2.length;
        int i = 0, j = 0;
        ArrayList<Integer> arr = new ArrayList<>();
        while (i < n && j < m) {
            if (nums1[i] == nums2[j]) {
                arr.add(nums1[i]);
                i++;
                j++;
            } else if (nums1[i] < nums2[j]) {
                i++;
            } else {
                j++;
            }
        }
        int k = arr.size();
        int[] ans = new int[k];
        for (int l = 0; l < k; l++) {
            ans[l] = arr.get(l);
        }
        return ans;
    }
}
