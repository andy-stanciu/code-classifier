class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        //T.C. = O(n) + O(nlogn).
        int n = nums1.length;
        //S.C. = O(1).
        int m = nums2.length;
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        //point with nums1
        int i = 0;
        //point with nums2
        int j = 0;
        ArrayList<Integer> list = new ArrayList<>();
        while (i < n && j < m) {
            if (nums1[i] == nums2[j]) {
                //if the both vslue equal we increase the i and j by 1.
                list.add(nums1[i]);
                i++;
                j++;
            } else if (nums1[i] < nums2[j]) {
                //the value of the nums1[i] is smaller than the value of the nums2[j] so we increase the i by 1.
                i++;
            } else {
                //the value of the nums2[j] is smaller than the value of the nums1[i] so we increase the j by 1.
                j++;
            }
        }
        //make the array of the size of the list.
        int[] result = new int[list.size()];
        int id = 0;
        for (int temp : list) {
            //add all elements in the result array.
            result[id++] = temp;
        }
        return result;
    }
}
