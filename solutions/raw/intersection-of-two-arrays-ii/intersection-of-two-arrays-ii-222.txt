class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        // To count frequencies of elements in nums1
        int[] arr = new int[1001];
        // To store the result
        int[] result = new int[1001];
        // Count the frequencies of each number in nums1
        for (int num : nums1) {
            arr[num]++;
        }
        // Counter for the result array
        int ctr = 0;
        // Traverse nums2 to find common elements
        for (int num : nums2) {
            if (arr[num] > 0) {
                result[ctr++] = num;
                arr[num]--;
            }
        }
        // Return only the filled portion of the result array
        return Arrays.copyOfRange(result, 0, ctr);
    }
}
