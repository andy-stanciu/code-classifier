class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] dataCountNum = new int[1001];
        int[] tempIntersection = new int[Math.min(nums1.length, nums2.length)];
        int index = 0;
        for (int num : nums1) {
            dataCountNum[num]++;
        }
        for (int num : nums2) {
            if (dataCountNum[num] > 0) {
                tempIntersection[index++] = num;
                dataCountNum[num]--;
            }
        }
        return Arrays.copyOfRange(tempIntersection, 0, index);
    }
}
