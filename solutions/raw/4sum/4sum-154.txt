class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        int n = nums.length;
        // Return empty list if the array has less than 4 elements
        if (nums == null || nums.length < 4) {
            return new ArrayList<>();
        }
        // Sort the array to use the two-pointer technique
        Arrays.sort(nums);
        List<List<Integer>> ans = new ArrayList<>();
        // Set to track duplicates
        HashSet<String> val = new HashSet<>();
        for (int i = 0; i < n - 3; i++) {
            for (int j = i + 1; j < n - 2; j++) {
                int left = j + 1;
                int right = n - 1;
                while (left < right) {
                    // Calculate the sum
                    long sum = (long) nums[i] + nums[j] + nums[left] + nums[right];
                    if (sum > target) {
                        right--;
                    } else if (sum < target) {
                        left++;
                    } else {
                        // Create a StringBuilder for unique tracking
                        StringBuilder sb = new StringBuilder();
                        sb.append(nums[i]);
                        sb.append(nums[j]);
                        sb.append(nums[left]);
                        sb.append(nums[right]);
                        String value = sb.toString();
                        // If the combination hasn\'t been added before, add it
                        if (!val.contains(value)) {
                            val.add(value);
                            ans.add(Arrays.asList(nums[i], nums[j], nums[left], nums[right]));
                        }
                        left++;
                        right--;
                    }
                }
            }
        }
        return ans;
    }
}
