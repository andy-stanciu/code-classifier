class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        int n = nums.length;
        Arrays.sort(nums);
        List<List<Integer>> ans = new ArrayList<>();
        for (int a = 0; a < n - 3; a++) {
            if (a > 0 && nums[a] == nums[a - 1])
                continue;
            for (int b = a + 1; b < n - 2; b++) {
                if ((b > (a + 1)) && nums[b] == nums[b - 1])
                    continue;
                int c = b + 1, d = n - 1;
                while (c < d) {
                    // this step is very important for storing the value greater than int range
                    long sum = nums[a] + nums[b];
                    sum += nums[c] + nums[d];
                    if (sum == target) {
                        List<Integer> temp = new ArrayList<>();
                        temp.add(nums[a]);
                        temp.add(nums[b]);
                        temp.add(nums[c]);
                        temp.add(nums[d]);
                        ans.add(temp);
                        c++;
                        d--;
                        while (c < d && nums[c] == nums[c - 1]) c++;
                        while (c < d && nums[d] == nums[d + 1]) d--;
                    } else if (sum < target)
                        c++;
                    else
                        d--;
                }
            }
        }
        return ans;
    }
}
