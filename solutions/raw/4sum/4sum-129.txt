class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length, k, l;
        long sum, need;
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            // skip duplicate, do 1 and then skip
            if (i != 0 && nums[i - 1] == nums[i])
                continue;
            for (int j = i + 1; j < n; j++) {
                // we need k,l so that  kth + lth == target - (ith + jth)
                // skip duplicate, do 1 and then skip
                if (j - 1 != i && nums[j - 1] == nums[j])
                    continue;
                k = j + 1;
                l = n - 1;
                need = target - nums[i];
                need -= nums[j];
                while (k < l) {
                    // i, j is fixed then 2 pointer k l move and apply 2 sum
                    sum = nums[k];
                    sum += nums[l];
                    if (sum < need)
                        k++;
                    else if (sum > need)
                        l--;
                    else {
                        List<Integer> list = new ArrayList<>();
                        list.add(nums[i]);
                        list.add(nums[j]);
                        list.add(nums[k]);
                        list.add(nums[l]);
                        ans.add(list);
                        k++;
                        l--;
                        // skip duplicate, do 1 and then skip
                        while (k < l && nums[k] == nums[k - 1]) k++;
                        // skip duplicate, do 1 and then skip
                        while (k < l && nums[l] == nums[l + 1]) l--;
                    }
                }
            }
        }
        return ans;
    }
}
