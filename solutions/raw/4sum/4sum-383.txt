class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        List<List<Integer>> outPut_arr = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            for (int j = i + 1; j < nums.length; j++) {
                if (j > i + 1 && nums[j] == nums[j - 1]) {
                    continue;
                }
                int low = j + 1;
                int end = nums.length - 1;
                while (low < end) {
                    long sum = (long) nums[i] + nums[j] + nums[low] + nums[end];
                    if (sum == target) {
                        outPut_arr.add(Arrays.asList(nums[i], nums[j], nums[low], nums[end]));
                        low++;
                        end--;
                        while (low < end && nums[low] == nums[low - 1]) {
                            low++;
                        }
                        while (low < end && nums[end] == nums[end + 1]) {
                            end--;
                        }
                    } else if (sum > target) {
                        end--;
                    } else {
                        low++;
                    }
                }
            }
        }
        return outPut_arr;
    }
}
