public class Solution {

    public static double myPow(double x, int n) {
        return Math.pow(x, n);
    }

    public static void main(String[] arg) {
        Scanner sc = new Scanner(System.in);
        double x = sc.nextDouble();
        int n = sc.nextInt();
        sc.close();
        double result = myPow(x, n);
        System.out.println(result);
    }
}
