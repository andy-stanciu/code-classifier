class Solution {

    public double myPow(double x, int n) {
        if (n >= 0) {
            return ansPow(x, n);
        }
        return 1 / ansPow(x, n);
    }

    public double ansPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (x == 0) {
            return 0;
        }
        double res = ansPow(x, n / 2);
        res *= res;
        if (n % 2 == 0) {
            return res;
        } else {
            return res * x;
        }
    }
}
