class Solution {

    public double myPow(double x, int n) {
        double p = x;
        if (n == 0)
            return (double) 1;
        if (n > 0) {
            p = positive(x, n);
        } else {
            p = 1 / positive(x, n);
        }
        return p;
    }

    static double positive(double x, int n) {
        if (n == 0)
            return 1;
        double temp = positive(x, n / 2);
        if (n % 2 == 0)
            return temp * temp;
        else
            return temp * temp * x;
    }
}
