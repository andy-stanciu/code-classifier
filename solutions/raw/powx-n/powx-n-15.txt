class Solution {

    public double myPow(double x, int n) {
        if (n < 0) {
            x = 1 / x;
            n = -n;
        }
        return pow(x, n);
    }

    double pow(double x, int n) {
        if (n == 0)
            return 1;
        double t = pow(x, n / 2);
        t = t * t;
        if (n % 2 == 0) {
            return t;
        } else {
            return t * x;
        }
    }
}
