class Solution {

    public double myPow(double x, int n) {
        return calPow(x, (long) n);
    }

    private double calPow(double x, long n) {
        if (n == 0)
            return 1;
        if (n < 0)
            return 1 / calPow(x, -n);
        if (n % 2 == 0) {
            return calPow(x * x, n / 2);
        } else
            return x * calPow(x * x, (n - 1) / 2);
    }
}
