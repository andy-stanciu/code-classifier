class Solution {

    public double myPow(double x, int n) {
        long N = n;
        if (N < 0) {
            x = 1 / x;
            N = -N;
        }
        double result = 1.0;
        while (N > 0) {
            if ((N % 2) == 1) {
                // If N is odd, multiply result by x
                result *= x;
            }
            // Square the base
            x *= x;
            // Divide the exponent by 2
            N /= 2;
        }
        return result;
    }
}
