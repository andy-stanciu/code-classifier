class Solution {

    public double myPow(double x, int n) {
        if (x == 1.0)
            return 1;
        if (x == -1.0 && n % 2 == 0)
            return 1.0;
        if (x == -1.0 && n % 2 != 0)
            return -1.0;
        if (x == 0.0)
            return 0.0;
        if (n == Integer.MIN_VALUE)
            return myPow(x * x, n / 2);
        if (n == 0)
            return 1;
        else if (n > 0) {
            if (n % 2 == 0)
                return myPow(x * x, n / 2);
            else
                return x * myPow(x, n - 1);
        } else
            return myPow(1.0 / x, -1 * n);
    }
}
