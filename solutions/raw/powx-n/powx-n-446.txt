class Solution {

    public double myPow(double x, int n) {
        return pow(x, (long) n);
    }

    public double pow(double x, long n) {
        double ans = 1;
        if (n < 0) {
            return 1 / pow(x, -n);
        }
        while (n > 0) {
            if (n % 2 == 1) {
                ans = ans * x;
                n = n - 1;
            } else {
                x = x * x;
                n = n / 2;
            }
        }
        return ans;
    }
}
