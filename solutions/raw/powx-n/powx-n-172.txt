class Solution {

    public double myPow(double x, long n) {
        double ans = 1;
        if (n < 0) {
            // very important edge case
            n = -n;
            x = 1 / x;
        }
        while (n > 0) {
            if (n % 2 == 1) {
                ans = ans * x;
                n--;
            } else {
                n = n / 2;
                x = x * x;
            }
        }
        return ans;
    }
}
