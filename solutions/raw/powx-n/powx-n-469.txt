class Solution {

    public double myPow(double x, int n) {
        if (n == 0 || x == 1) {
            return 1;
        }
        if (x == -1) {
            if (n % 2 == 0)
                return 1;
            else
                return -1;
        }
        if (n > 0)
            return findPow(x, n);
        else {
            return 1 / (findPow(x, (n + 1) * (-1)) * x);
        }
    }

    public static double findPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        double half = findPow(x, n / 2);
        if (n % 2 == 0)
            return half * half;
        else
            return half * half * x;
    }
}
