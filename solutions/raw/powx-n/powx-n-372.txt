class Solution {

    public double pow(double x, int n) {
        if (n == 1)
            return x;
        if (n == 0)
            return 1;
        double p = pow(x, n / 2);
        if (n % 2 == 0) {
            return p * p;
        } else {
            return p * p * x;
        }
    }

    public double myPow(double x, int n) {
        if (n < 0) {
            return pow(1 / x, n);
        }
        return pow(x, n);
    }
}
