class Solution {

    public double pow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (n % 2 == 0) {
            // (x^2)^(n/2)
            return pow(x * x, n / 2);
        }
        // x*(x^2)^(n-1)/2
        return x * pow(x * x, (n - 1) / 2);
    }

    public double myPow(double x, int n) {
        double ans = 0.0;
        if (n < 0) {
            if (n == Integer.MIN_VALUE) {
                return 1.0 / pow(x, Integer.MAX_VALUE) * x;
            }
            return 1.0 / pow(x, -1 * n);
        }
        return pow(x, n);
    }
}
