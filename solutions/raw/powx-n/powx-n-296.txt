class Solution {

    public double myPow(double x, int n) {
        double ans = 1;
        long m = Math.abs((long) n);
        while (m > 0) {
            if (m % 2 == 1) {
                ans = ans * x;
                m = m - 1;
            } else {
                m = m / 2;
                x = x * x;
            }
        }
        return n < 0 ? 1 / ans : ans;
    }
}
