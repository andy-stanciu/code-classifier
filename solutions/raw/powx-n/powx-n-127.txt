class Solution {

    public double myPow(double x, int n) {
        double temp;
        if (n == 0) {
            return 1;
        }
        temp = myPow(x, n / 2);
        // For even powers
        if (n % 2 == 0)
            return temp * temp;
        else // For odd powers
        {
            // For odd positive powers
            if (n > 0)
                return x * temp * temp;
            else
                // For odd negative powers
                return (temp * temp) / x;
        }
    }
}
