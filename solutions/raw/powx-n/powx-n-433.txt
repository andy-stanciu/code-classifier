class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (n == Integer.MIN_VALUE) {
            return myPow(x * x, n / 2);
        }
        if (n < 0) {
            n = -n;
            x = 1 / x;
        }
        double power = myPow(x, n / 2);
        double halfPowSq = power * power;
        if (n % 2 != 0) {
            halfPowSq = x * power * power;
        }
        return halfPowSq;
    }
}
