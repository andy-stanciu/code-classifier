class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (x == 1) {
            return 1;
        }
        if (x == 0) {
            return 0;
        }
        if (x == -1 && n % 2 == 0) {
            return 1.0;
        }
        if (x == -1 && n % 2 != 0) {
            return -1.0;
        }
        long bF = n;
        if (n < 0) {
            x = 1 / x;
            bF = -bF;
        }
        double ans = 1;
        while (bF > 0) {
            if (bF % 2 == 1) {
                ans *= x;
            }
            x *= x;
            bF /= 2;
        }
        return ans;
    }
}
