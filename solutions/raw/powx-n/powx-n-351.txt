class Solution {

    public double myPow(double x, int n) {
        if (n == 0 || x == 1)
            return 1.0;
        boolean isNegative = (n < 0) ? true : false;
        long itr = isNegative ? -(long) n : n;
        double ans = 1;
        while (itr > 0) {
            if (itr % 2 == 0) {
                x *= x;
                itr /= 2;
            }
            ans = ans * x;
            itr--;
        }
        System.out.println(ans);
        if (isNegative)
            return 1.00 / ans;
        return ans;
    }
}
