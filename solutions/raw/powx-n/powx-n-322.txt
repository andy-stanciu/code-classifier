class Solution {

    public double myPow(double x, int n) {
        if (n == 0 || x == 1) {
            return 1;
        } else if (n == Integer.MIN_VALUE) {
            return x == -1 ? 1 : 0;
        } else {
            x = n > 0 ? x : 1.0 / x;
            n = Math.abs(n);
            return pow(x, n);
        }
    }

    public double pow(double x, int n) {
        if (n == 1) {
            return x;
        } else if (n % 2 == 0) {
            return pow(x * x, n / 2);
        } else {
            return pow(x, n - 1) * x;
        }
    }
}
