public class Solution {

    public double solve(double x, long n) {
        if (n == 0) {
            // power of 0 is 1
            return 1;
        }
        double temp = solve(x, n / 2);
        temp = temp * temp;
        if (n % 2 == 0) {
            // if even, return just without doing anything
            return temp;
        } else {
            // if odd, return by multiplying once more with the given number
            return temp * x;
        }
    }

    public double myPow(double x, int n) {
        // Convert n to a long to handle Integer.MIN_VALUE
        long longN = n;
        double ans = solve(x, Math.abs(longN));
        if (longN < 0) {
            return 1 / ans;
        }
        return ans;
    }
}
