class Solution {

    public double myPow(double x, int n) {
        double ans = 1;
        long m = Math.abs((long) n);
        while (m > 0) {
            if ((m & 1) != 0) {
                ans = ans * x;
            }
            x = x * x;
            m = m >> 1;
        }
        return n < 0 ? 1 / ans : ans;
    }
}
