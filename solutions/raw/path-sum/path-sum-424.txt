class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return sol(0, root, targetSum);
    }

    public boolean sol(int sum, TreeNode root, int target) {
        if (root == null)
            return false;
        sum = sum + root.val;
        if (root.left == null && root.right == null) {
            if (sum == target)
                return true;
        }
        return sol(sum, root.left, target) || sol(sum, root.right, target);
    }
}
