class Solution {

    int sum;

    public boolean hasPathSum(TreeNode root, int sum) {
        this.sum = sum;
        return getSum(root, 0);
    }

    boolean getSum(TreeNode node, int pSum) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            return pSum + node.val == sum;
        return getSum(node.left, pSum + node.val) || getSum(node.right, pSum + node.val);
    }
}
