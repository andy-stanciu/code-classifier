class Solution {

    private boolean found = false;

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        dfs(root, targetSum, root.val);
        return found;
    }

    private void dfs(TreeNode node, int targetSum, int cur) {
        if (node.left == null && node.right == null) {
            if (cur == targetSum) {
                found = true;
            }
        }
        if (node.left != null) {
            dfs(node.left, targetSum, cur + node.left.val);
        }
        if (node.right != null) {
            dfs(node.right, targetSum, cur + node.right.val);
        }
    }
}
