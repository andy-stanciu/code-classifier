/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return hasPathSumRec(root, targetSum);
    }

    public boolean hasPathSumRec(TreeNode root, int currentBal) {
        if (root == null) {
            return false;
        }
        currentBal -= root.val;
        if (root.left == null && root.right == null && currentBal == 0) {
            return true;
        }
        return hasPathSumRec(root.left, currentBal) || hasPathSumRec(root.right, currentBal);
    }
}
