/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int currentSum = 0;
        return (findSum(root, currentSum, targetSum));
    }

    public boolean findSum(TreeNode root, int currentSum, int targetSum) {
        if (root == null) {
            return false;
        } else if (root.left == null && root.right == null) {
            currentSum = currentSum + root.val;
            if (currentSum == targetSum) {
                return true;
            }
        } else {
            currentSum = currentSum + root.val;
        }
        return (findSum(root.left, currentSum, targetSum) || findSum(root.right, currentSum, targetSum));
    }
}
