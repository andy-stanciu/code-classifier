class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return recPathSum(root, 0, targetSum);
    }

    private boolean recPathSum(TreeNode root, int curSum, int targetSum) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            return curSum + root.val == targetSum;
        }
        return recPathSum(root.left, curSum + root.val, targetSum) || recPathSum(root.right, curSum + root.val, targetSum);
    }
}
