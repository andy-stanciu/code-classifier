/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return hasPathSum(root, targetSum, 0);
    }

    private boolean hasPathSum(TreeNode root, int targetSum, int acc) {
        if ((targetSum == acc + root.val) && root.left == null && root.right == null) {
            return true;
        }
        if (root.left != null && hasPathSum(root.left, targetSum, acc + root.val)) {
            return true;
        }
        if (root.right != null && hasPathSum(root.right, targetSum, acc + root.val)) {
            return true;
        }
        return false;
    }
}
