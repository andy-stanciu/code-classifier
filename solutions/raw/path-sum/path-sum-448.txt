/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private boolean hasPathSum;

    public boolean hasPathSum(TreeNode root, int targetSum) {
        hasPathSum = false;
        if (root != null) {
            traverse(root, targetSum, 0);
        }
        return hasPathSum;
    }

    private void traverse(TreeNode node, int targetSum, int currentSum) {
        if (node.left == null && node.right == null) {
            if (((currentSum + node.val) == targetSum)) {
                hasPathSum = true;
            }
        } else {
            if (node.left != null) {
                traverse(node.left, targetSum, currentSum + node.val);
            }
            if (node.right != null) {
                traverse(node.right, targetSum, currentSum + node.val);
            }
        }
    }
}
