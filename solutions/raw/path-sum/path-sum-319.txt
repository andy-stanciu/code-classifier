class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return helper(root, targetSum, 0);
    }

    public boolean helper(TreeNode node, int target, int sum) {
        if (node == null && target == sum)
            return true;
        if (node == null)
            return false;
        // there is possibility node\'s left is empty
        if (node.left == null)
            return helper(node.right, target, sum + node.val);
        // node\'s right is empty
        if (node.right == null)
            return helper(node.left, target, sum + node.val);
        return helper(node.left, target, sum + node.val) || helper(node.right, target, sum + node.val);
    }
}
