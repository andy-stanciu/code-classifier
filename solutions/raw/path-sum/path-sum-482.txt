/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return sumNodes(root, targetSum, 0);
    }

    public boolean sumNodes(TreeNode node, int targetSum, int s) {
        if (node == null)
            return targetSum == s;
        if (node.left == null && node.right == null)
            return targetSum == (s + node.val);
        if (node.left == null)
            return sumNodes(node.right, targetSum, s + node.val);
        if (node.right == null)
            return sumNodes(node.left, targetSum, s + node.val);
        if (sumNodes(node.left, targetSum, s + node.val) || sumNodes(node.right, targetSum, s + node.val))
            return true;
        return false;
    }
}
