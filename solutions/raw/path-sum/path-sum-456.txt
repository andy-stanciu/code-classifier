/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        boolean flag = false;
        int target = targetSum;
        if (root == null)
            return false;
        /*if(root.val == targetSum){
            if(root.right != null)
                return false;
            if(root.left != null)
                return false;
         }*/
        flag = answer(root, 0, target, false);
        if (flag == true)
            return true;
        return false;
    }

    public boolean answer(TreeNode root, int sum, int target, boolean flag) {
        if (root == null) {
            return false;
        }
        if (root.right == null && root.left == null) {
            sum = sum + root.val;
            if (sum == target) {
                System.out.println("true");
                return true;
            }
            return false;
        } else {
            sum = sum + root.val;
            return answer(root.left, sum, target, flag) | answer(root.right, sum, target, flag);
        }
        //return false;
    }
}
