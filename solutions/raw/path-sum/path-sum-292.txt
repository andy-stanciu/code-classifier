/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasTargetSum(int sum, TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && sum + root.val == targetSum) {
            return true;
        }
        return hasTargetSum(sum + root.val, root.left, targetSum) || hasTargetSum(sum + root.val, root.right, targetSum);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return hasTargetSum(0, root, targetSum);
    }
}
