/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int t) {
        //using inorder traversal
        boolean[] check = new boolean[1];
        check[0] = false;
        int sum = 0;
        //Stack<Integer>s2 = new Stack<>();
        PreOrder(root, t, check, sum);
        return check[0];
    }

    public void PreOrder(TreeNode root, int t, boolean[] check, int sum) {
        if (root == null) {
            return;
        }
        sum += root.val;
        if (root.left == null && root.right == null && sum == t) {
            check[0] = true;
        }
        PreOrder(root.left, t, check, sum);
        PreOrder(root.right, t, check, sum);
        sum -= root.val;
    }
}
