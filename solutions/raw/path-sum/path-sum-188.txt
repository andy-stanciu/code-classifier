class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return hasPathSum(root, 0, targetSum);
    }

    public boolean hasPathSum(TreeNode root, int sum, int targetSum) {
        TreeNode t = root;
        if (t == null) {
            return false;
        }
        if (t.left == null && t.right == null) {
            sum = sum + t.val;
            if (sum == targetSum) {
                return true;
            } else {
                return false;
            }
        } else {
            boolean leftBranch = hasPathSum(t.left, sum + t.val, targetSum);
            boolean rightBranch = hasPathSum(t.right, sum + t.val, targetSum);
            return leftBranch || rightBranch;
        }
        // AKHRI MAI RETURN TAB HI MANGTA HAI AGAR HUMNE UPAR EK BHI ESE CONDITION CHOD DI JISKA
        // RESULT USE NICHE YAHA TAK LAI AAYE. AGAR SARA KUCH UPAR SAI E RETURN HO RAHA TO YAHA
        // KABHI NAHI RETURN NAHI MANGE GA;
    }
}
