/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return pathSum(root, targetSum, 0);
    }

    private boolean pathSum(TreeNode root, int targetSum, int prevSum) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return (root.val + prevSum) == targetSum;
        prevSum += root.val;
        if (pathSum(root.left, targetSum, prevSum))
            return true;
        return pathSum(root.right, targetSum, prevSum);
    }
}
