/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        return PreOrder(root, 0, targetSum);
    }

    public boolean PreOrder(TreeNode root, int cal, int targetSum) {
        if (root == null) {
            return false;
        }
        cal += root.val;
        if (root.left == null && root.right == null && cal == targetSum) {
            return true;
        }
        return PreOrder(root.left, cal, targetSum) || PreOrder(root.right, cal, targetSum);
    }
}
