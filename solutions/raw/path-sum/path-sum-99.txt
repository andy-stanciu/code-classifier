/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasSum(TreeNode root, int targetSum, int currentSum) {
        if (root.left == null && root.right == null) {
            if (currentSum + root.val == targetSum)
                return true;
            return false;
        }
        if (root.left != null && hasSum(root.left, targetSum, currentSum + root.val))
            return true;
        if (root.right != null && hasSum(root.right, targetSum, currentSum + root.val))
            return true;
        return false;
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return hasSum(root, targetSum, 0);
    }
}
