/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int tar) {
        if (root == null) {
            return false;
        }
        return sum(root, tar, 0);
    }

    public boolean sum(TreeNode root, int tar, int sum) {
        if (root.left == null && root.right == null) {
            sum += root.val;
            if (sum == tar) {
                return true;
            }
            return false;
        }
        sum += root.val;
        boolean left = false, right = false;
        if (root.left != null) {
            left = sum(root.left, tar, sum);
        }
        if (root.right != null) {
            right = sum(root.right, tar, sum);
        }
        return left || right;
    }
}
