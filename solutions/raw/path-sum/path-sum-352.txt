/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(final TreeNode root, final int targetSum) {
        if (root == null) {
            return false;
        }
        final int localSum = targetSum - root.val;
        final boolean isLeaf = root.left == null && root.right == null;
        if (isLeaf && localSum == 0) {
            return true;
        }
        final boolean left = hasPathSum(root.left, localSum);
        final boolean right = hasPathSum(root.right, localSum);
        return left || right;
        /* If one of these values goes to zero at the leaf then it has a path sum! */
    }
}
