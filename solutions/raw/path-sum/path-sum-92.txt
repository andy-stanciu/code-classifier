/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return path(root, targetSum, 0);
    }

    public boolean path(TreeNode root, int target, int curr) {
        if (root == null) {
            return false;
        }
        curr += root.val;
        if (root.left == null && root.right == null && curr == target) {
            return true;
        }
        return path(root.left, target, curr) || path(root.right, target, curr);
    }
}
