/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    /**
     * 20240324
     * try 2
     *
     * @param root
     * @param targetSum
     * @return
     */
    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            // first node, null
            // OR
            // non-leaf node\'s null child
            return false;
        }
        // edge case, leaf node
        if (root.left == null && root.right == null) {
            if (root.val == targetSum) {
                return true;
            } else {
                return false;
            }
        }
        // not leaf node
        int restTargetSum = targetSum - root.val;
        return hasPathSum(root.left, restTargetSum) || hasPathSum(root.right, restTargetSum);
    }
}
