class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return hasTargetSum(root, 0, targetSum);
    }

    private static boolean hasTargetSum(TreeNode node, int prevSum, int targetSum) {
        if (node.right == null && node.left == null) {
            prevSum = prevSum + node.val;
            return prevSum == targetSum;
        } else {
            boolean rightHasTargetSum = (node.right != null) ? hasTargetSum(node.right, prevSum + node.val, targetSum) : false;
            boolean leftHasTargetSum = (node.left != null) ? hasTargetSum(node.left, prevSum + node.val, targetSum) : false;
            return rightHasTargetSum || leftHasTargetSum;
        }
    }
}
