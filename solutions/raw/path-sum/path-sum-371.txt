/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean ps(TreeNode root, int targetSum, int sum) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            sum = sum + root.val;
            if (sum == targetSum)
                return true;
        }
        return ps(root.left, targetSum, sum + root.val) || ps(root.right, targetSum, sum + root.val);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int sum = 0;
        return ps(root, targetSum, sum);
    }
}
