/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public boolean dfs(TreeNode root, int t, int csum) {
        if (root == null) {
            return false;
        }
        csum += root.val;
        if (root.left == null && root.right == null) {
            return t == csum;
        }
        return dfs(root.left, t, csum) || dfs(root.right, t, csum);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return dfs(root, targetSum, 0);
    }
}
