class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        if (root.right == null && root.left == null)
            return targetSum == root.val;
        targetSum -= root.val;
        return hasPathSum(root.right, targetSum) || hasPathSum(root.left, targetSum);
    }
}
