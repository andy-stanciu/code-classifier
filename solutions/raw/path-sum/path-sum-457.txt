class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return hasPathSum(root, targetSum, 0);
    }

    private boolean hasPathSum(TreeNode node, int targetSum, int currSum) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            if (targetSum == (currSum + node.val))
                return true;
        return hasPathSum(node.left, targetSum, currSum + node.val) || hasPathSum(node.right, targetSum, currSum + node.val);
    }
}
