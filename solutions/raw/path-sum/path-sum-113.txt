/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        // Check all paths with DFS
        // No bookkeeping is needed, just reduce the target by the current node as we go.
        // Leaf
        if (root.left == null && root.right == null) {
            return root.val == targetSum;
        }
        // Non-leaf
        boolean leftPathFound = hasPathSum(root.left, targetSum - root.val);
        boolean rightPathFound = hasPathSum(root.right, targetSum - root.val);
        return leftPathFound || rightPathFound;
    }
}
