class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int currsum = 0;
        return path(root, currsum, targetSum);
    }

    public boolean path(TreeNode root, int currsum, int targetSum) {
        if (root == null) {
            return false;
        }
        currsum += root.val;
        if (root.left == null && root.right == null) {
            if (currsum == targetSum) {
                return true;
            } else {
                return false;
            }
        }
        boolean left = path(root.left, currsum, targetSum);
        boolean right = path(root.right, currsum, targetSum);
        return left || right;
    }
}
