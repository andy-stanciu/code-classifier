/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int sum = 0;
        return check(root, targetSum, sum);
    }

    public boolean check(TreeNode root, int targetSum, int sum) {
        if (root == null)
            return false;
        if (targetSum == sum + root.val && root.left == null && root.right == null)
            return true;
        return check(root.left, targetSum, sum + root.val) || check(root.right, targetSum, sum + root.val);
    }
}
