/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return path(root, 0, targetSum);
    }

    boolean path(TreeNode root, int sum, int target) {
        if (root == null) {
            return false;
        }
        sum += root.val;
        if (root.right == null && root.left == null) {
            return sum == target;
        }
        boolean left = path(root.left, sum, target);
        boolean right = path(root.right, sum, target);
        return left || right;
    }
}
