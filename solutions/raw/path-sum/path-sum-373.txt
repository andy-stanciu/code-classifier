/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private boolean deep(TreeNode node, int sum) {
        if (node == null) {
            if (sum == 0) {
                return true;
            }
            return false;
        }
        if (node.left == null && node.right != null) {
            return deep(node.right, sum - node.val);
        } else if (node.right == null && node.left != null) {
            return deep(node.left, sum - node.val);
        }
        return deep(node.left, sum - node.val) || deep(node.right, sum - node.val);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        return deep(root, targetSum);
    }
}
