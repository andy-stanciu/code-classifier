class Solution {

    boolean result = false;

    public boolean hasPathSum(TreeNode root, int targetSum) {
        dfs(root, targetSum);
        return result;
    }

    private void dfs(TreeNode root, int targetSum) {
        if (root == null) {
            return;
        }
        targetSum -= root.val;
        //leaf Node
        if (root.left == null && root.right == null) {
            if (targetSum == 0) {
                result = true;
            }
            return;
        }
        //path not found yet
        if (result == false) {
            dfs(root.left, targetSum);
            dfs(root.right, targetSum);
        }
    }
}
