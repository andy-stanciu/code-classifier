/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return findSum(root, targetSum, 0);
    }

    boolean findSum(TreeNode root, int targetSum, int currentSum) {
        if (root == null)
            return targetSum == currentSum;
        currentSum += root.val;
        if (root.left != null && root.right != null)
            return findSum(root.left, targetSum, currentSum) || findSum(root.right, targetSum, currentSum);
        if (root.left != null && root.right == null)
            return findSum(root.left, targetSum, currentSum);
        else
            return findSum(root.right, targetSum, currentSum);
    }
}
