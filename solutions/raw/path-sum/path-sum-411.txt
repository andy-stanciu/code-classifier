/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return dfs(root, targetSum, 0, false);
    }

    static boolean dfs(TreeNode root, int targetSum, int sum, boolean flag) {
        if (root == null)
            return flag;
        if (root.left == null && root.right == null) {
            return sum + root.val == targetSum;
        }
        return dfs(root.left, targetSum, sum + root.val, flag) || dfs(root.right, targetSum, sum + root.val, flag);
    }
}
