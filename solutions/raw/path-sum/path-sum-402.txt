/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean isTargetSum = false;

    public void targetSum(TreeNode root, int sum) {
        if (root == null)
            return;
        if (sum - root.val == 0 && root.left == null && root.right == null)
            isTargetSum = true;
        targetSum(root.left, sum - root.val);
        targetSum(root.right, sum - root.val);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        targetSum(root, targetSum);
        return isTargetSum;
    }
}
