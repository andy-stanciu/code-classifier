/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return find(root, 0, targetSum);
    }

    boolean find(TreeNode root, int sum, int t) {
        if (root == null)
            return false;
        sum += root.val;
        if (root.left == null && root.right == null) {
            return sum == t;
        }
        return (find(root.left, sum, t) || find(root.right, sum, t));
    }
}
