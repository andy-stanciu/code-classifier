/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return function(root, targetSum, 0);
    }

    static boolean function(TreeNode root, int targetSum, int sum) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            sum = sum + root.val;
            if (sum == targetSum)
                return true;
            return false;
        }
        sum = sum + root.val;
        boolean f1 = function(root.left, targetSum, sum);
        boolean f2 = function(root.right, targetSum, sum);
        return f1 || f2;
    }
}
