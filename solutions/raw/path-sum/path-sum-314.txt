/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        return find(root, targetSum);
    }

    boolean find(TreeNode node, int f) {
        if (node == null) {
            return false;
        }
        if (node.val == f && node.left == null && node.right == null) {
            return true;
        }
        return find(node.right, f - node.val) || find(node.left, f - node.val);
    }
}
