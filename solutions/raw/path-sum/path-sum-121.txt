/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        boolean[] isThere = new boolean[1];
        preOrderBinaryTree(root, 0, isThere, targetSum);
        return isThere[0];
    }

    public static void preOrderBinaryTree(TreeNode root, int sum, boolean[] isThere, int targetSum) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if (sum + root.val == targetSum) {
                isThere[0] = true;
            }
        }
        preOrderBinaryTree(root.left, sum + root.val, isThere, targetSum);
        preOrderBinaryTree(root.right, sum + root.val, isThere, targetSum);
    }
}
