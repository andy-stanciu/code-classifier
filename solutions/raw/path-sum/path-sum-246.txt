/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return sum(root, targetSum, 0);
    }

    private boolean sum(TreeNode root, int targetSum, int sumSoFar) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (sumSoFar + root.val == targetSum) {
                return true;
            }
            return false;
        }
        return sum(root.left, targetSum, sumSoFar + root.val) || sum(root.right, targetSum, sumSoFar + root.val);
    }
}
