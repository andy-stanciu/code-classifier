class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null && targetSum == 0)
            return false;
        else
            return hasPath(root, targetSum);
    }

    public boolean hasPath(TreeNode trav, int targetSum) {
        if (trav == null && targetSum == 0)
            return true;
        else if (trav == null)
            return false;
        if (trav.left != null && trav.right != null)
            return hasPath(trav.left, targetSum - trav.val) || hasPath(trav.right, targetSum - trav.val);
        if (trav.left != null)
            return hasPath(trav.left, targetSum - trav.val);
        else
            return hasPath(trav.right, targetSum - trav.val);
    }
}
