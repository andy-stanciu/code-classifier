/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int targetSum = 0;

    public boolean rec(TreeNode root, int sum) {
        if (root == null) {
            return false;
        }
        sum = sum + root.val;
        if (root.left == null && root.right == null) {
            if (sum == targetSum) {
                return true;
            }
        }
        return rec(root.right, sum) || rec(root.left, sum);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        this.targetSum = targetSum;
        return rec(root, 0);
    }
}
