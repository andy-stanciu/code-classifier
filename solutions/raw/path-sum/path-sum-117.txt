class Solution {

    public boolean sumfind(TreeNode root, int sum, int targetsum) {
        if (root == null) {
            return false;
        }
        sum += root.val;
        if (root.left == null && root.right == null) {
            if (sum == targetsum) {
                return true;
            }
        }
        boolean a = sumfind(root.left, sum, targetsum);
        boolean b = sumfind(root.right, sum, targetsum);
        return a || b;
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return sumfind(root, 0, targetSum);
    }
}
