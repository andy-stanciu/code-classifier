class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return test(root, 0, targetSum);
    }

    boolean test(TreeNode node, int sum, int target) {
        if (null == node)
            return false;
        var current = sum + node.val;
        if (target == current && node.left == null && node.right == null) {
            return true;
        }
        return test(node.left, current, target) || test(node.right, current, target);
    }
}
