/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean help(TreeNode node, int target, int sum) {
        if (node == null)
            return false;
        if (target == sum + node.val && node.left == null && node.right == null) {
            return true;
        }
        return help(node.left, target, sum + node.val) || help(node.right, target, sum + node.val);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return help(root, targetSum, 0);
    }
}
