/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return findPath(root, targetSum, 0);
    }

    public boolean findPath(TreeNode root, int targetSum, int currentSum) {
        if (root == null)
            return false;
        currentSum += root.val;
        if (root.left == null & root.right == null) {
            if (currentSum == targetSum) {
                return true;
            }
        }
        return findPath(root.left, targetSum, currentSum) || findPath(root.right, targetSum, currentSum);
    }
}
