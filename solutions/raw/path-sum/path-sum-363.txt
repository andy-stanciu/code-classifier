/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        targetSum = targetSum - root.val;
        if (targetSum == 0 && root.right == null && root.left == null) {
            return true;
        }
        if (root.right == null && root.left != null) {
            return hasPathSum(root.left, targetSum);
        }
        if (root.left == null && root.right != null) {
            return hasPathSum(root.right, targetSum);
        }
        if (root.right != null & root.left != null) {
            if (targetSum - root.right.val == 0 || targetSum - root.left.val == 0) {
                if (root.right.right == null && root.right.left == null && root.left.left == null && root.right.right == null)
                    return true;
            }
            return hasPathSum(root.right, targetSum) || hasPathSum(root.left, targetSum);
        }
        return false;
    }
}
