/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // private int count = 0;
    public boolean checksum(TreeNode root, int targetSum, int count) {
        if (root == null) {
            return false;
        }
        count += root.val;
        if (root.left == null && root.right == null) {
            return count == targetSum;
        }
        return checksum(root.left, targetSum, count) || checksum(root.right, targetSum, count);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null && targetSum == 0) {
            return false;
        }
        // int count = root.val;
        int count = 0;
        return checksum(root, targetSum, count);
    }
}
