/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int[] currsum = new int[1];
        return has(root, targetSum, currsum);
    }

    public boolean has(TreeNode root, int target, int[] currsum) {
        if (root == null) {
            return false;
        }
        currsum[0] += root.val;
        if (root.left == null && root.right == null && currsum[0] == target) {
            return true;
        }
        boolean left = has(root.left, target, currsum);
        boolean right = has(root.right, target, currsum);
        currsum[0] -= root.val;
        return left || right;
    }
}
