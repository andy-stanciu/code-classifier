/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean flag = false;

    public void dfsSum(TreeNode root, int sum, int target) {
        sum += root.val;
        if (root.left == null && root.right == null) {
            if (sum == target) {
                flag = true;
            }
            return;
        }
        if (root.left != null)
            dfsSum(root.left, sum, target);
        if (root.right != null)
            dfsSum(root.right, sum, target);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root != null)
            dfsSum(root, 0, targetSum);
        return flag;
    }
}
