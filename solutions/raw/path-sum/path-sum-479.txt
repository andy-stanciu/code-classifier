/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return findPath(root, targetSum, 0);
    }

    public static boolean findPath(TreeNode root, int ref, int sum) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (sum + root.val == ref)
                return true;
        }
        if (findPath(root.left, ref, sum + root.val))
            return true;
        if (findPath(root.right, ref, sum + root.val))
            return true;
        return false;
    }
}
