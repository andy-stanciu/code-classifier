/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return hasPathSumHelper(root, targetSum, 0);
    }

    private boolean hasPathSumHelper(TreeNode node, int targetSum, int currentSum) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null) {
            return targetSum == currentSum + node.val;
        }
        return hasPathSumHelper(node.left, targetSum, currentSum + node.val) || hasPathSumHelper(node.right, targetSum, currentSum + node.val);
    }
}
