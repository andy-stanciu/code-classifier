class Solution {

    boolean pathSumExists = false;

    int targetSum;

    public boolean hasPathSum(TreeNode root, int targetSum) {
        this.targetSum = targetSum;
        isLeafWithSum(root, 0);
        return pathSumExists;
    }

    private void isLeafWithSum(TreeNode node, int currentSum) {
        if (this.pathSumExists || node == null) {
            // we already found target path sum
            // OR node is null
            return;
        }
        int sum = node.val + currentSum;
        if (node.left == null && node.right == null) {
            if (sum == this.targetSum) {
                pathSumExists = true;
            }
        } else {
            isLeafWithSum(node.left, sum);
            isLeafWithSum(node.right, sum);
        }
    }
}
