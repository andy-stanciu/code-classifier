/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    private boolean hasFound = false;

    public boolean hasPathSum(TreeNode root, int targetSum) {
        readTree(root, targetSum, 0);
        return hasFound;
    }

    private void readTree(TreeNode node, int targetSum, int preSum) {
        if (node == null || hasFound)
            return;
        if (node.left == null && node.right == null && (node.val + preSum == targetSum))
            hasFound = true;
        readTree(node.left, targetSum, preSum + node.val);
        readTree(node.right, targetSum, preSum + node.val);
    }
}
