/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        return hasPath(root, 0, targetSum);
    }

    private boolean hasPath(TreeNode node, int current, int target) {
        current += node.val;
        if (node.left == null && node.right == null && current == target) {
            return true;
        }
        if (node.left != null && hasPath(node.left, current, target)) {
            return true;
        }
        if (node.right != null && hasPath(node.right, current, target)) {
            return true;
        }
        return false;
    }
}
