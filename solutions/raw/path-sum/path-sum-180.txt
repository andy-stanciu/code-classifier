/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private boolean res(TreeNode root, int targetSum, int sum) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && targetSum == sum + root.val) {
            return true;
        }
        return res(root.left, targetSum, sum + root.val) || res(root.right, targetSum, sum + root.val);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return res(root, targetSum, 0);
    }
}
