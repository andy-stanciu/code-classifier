/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        return sum(root, targetSum, 0, false);
    }

    public static boolean sum(TreeNode root, int targetSum, int currentSum, boolean leaf) {
        if (root == null) {
            if (leaf) {
                return currentSum == targetSum;
            }
            return false;
        }
        return sum(root.left, targetSum, currentSum + root.val, root.right == null && root.left == null) || sum(root.right, targetSum, currentSum + root.val, root.right == null && root.left == null);
    }
}
