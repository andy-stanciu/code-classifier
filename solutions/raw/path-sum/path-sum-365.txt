/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean check(TreeNode root, int k, int curr) {
        if (root == null)
            return false;
        int path = root.val + curr;
        if (root.left == null && root.right == null) {
            if (path == k)
                return true;
            else
                return false;
        }
        return check(root.left, k, path) || check(root.right, k, path);
    }

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null)
            return false;
        return check(root, targetSum, 0);
    }
}
