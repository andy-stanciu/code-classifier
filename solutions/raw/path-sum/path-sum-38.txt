class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        int[] sum = { 0 };
        return findSumUtil(root, sum, targetSum);
    }

    public boolean findSumUtil(TreeNode root, int[] sum, int target) {
        if (root == null)
            return false;
        sum[0] += root.val;
        if (root.left == null && root.right == null) {
            if (sum[0] == target)
                return true;
        }
        if (findSumUtil(root.left, sum, target) || findSumUtil(root.right, sum, target))
            return true;
        else {
            sum[0] -= root.val;
            return false;
        }
    }
}
