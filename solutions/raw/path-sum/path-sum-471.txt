/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        return checkPathSum(root, 0, targetSum);
    }

    public boolean checkPathSum(TreeNode node, int currentSum, int targetSum) {
        if (node.left == null && node.right == null) {
            if (currentSum + node.val == targetSum) {
                return true;
            }
        }
        if (node.left != null) {
            boolean possible = checkPathSum(node.left, currentSum + node.val, targetSum);
            if (possible) {
                return true;
            }
        }
        if (node.right != null) {
            boolean possible = checkPathSum(node.right, currentSum + node.val, targetSum);
            if (possible) {
                return true;
            }
        }
        return false;
    }
}
