class Solution {

    int count = 0;

    public boolean canFinish(int numCourses, int[][] prerequisites) {
        // creation of graph (prereq course -> course)
        List<List<Integer>> graph = new ArrayList<>();
        for (int i = 0; i < numCourses; i++) graph.add(new ArrayList<>());
        for (int[] edge : prerequisites) {
            graph.get(edge[1]).add(edge[0]);
        }
        for (int i = 0; i < numCourses; i++) {
            if (hasCycle(graph, new int[numCourses], i))
                return false;
        }
        return true;
    }

    private boolean hasCycle(List<List<Integer>> graph, int[] state, int idx) {
        // processed
        if (state[idx] == 1)
            return false;
        // processing -> we have a cycle
        if (state[idx] == -1)
            return true;
        state[idx] = -1;
        for (int neighbor : graph.get(idx)) {
            if (hasCycle(graph, state, neighbor))
                return true;
        }
        // mark the node processed
        state[idx] = 1;
        return false;
    }
}
