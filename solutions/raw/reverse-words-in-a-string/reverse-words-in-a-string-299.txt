class Solution {

    public String reverseWords(String s) {
        StringBuilder str = new StringBuilder();
        s = s.trim();
        String[] st = s.split("\\\\s+");
        for (int i = st.length - 1; i >= 0; i--) {
            str.append(st[i].trim() + " ");
        }
        return str.toString().trim();
    }
}
