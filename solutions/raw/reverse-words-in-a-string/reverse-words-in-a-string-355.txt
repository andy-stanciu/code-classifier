class Solution {

    public String reverseWords(String s) {
        String trimmed = s.trim();
        String[] words = trimmed.split("\\\\s+");
        StringBuilder st = new StringBuilder();
        int l = words.length;
        for (int i = l - 1; i > 0; i--) {
            st.append(words[i]);
            st.append(" ");
        }
        st.append(words[0]);
        return st.toString();
    }
}
