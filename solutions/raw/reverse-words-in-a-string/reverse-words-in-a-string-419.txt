class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] words = s.split("\\\\s+");
        StringBuilder newString = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            if (i == 0)
                newString.append(words[i]);
            else
                newString.append(words[i] + " ");
        }
        return newString.toString();
    }
}
