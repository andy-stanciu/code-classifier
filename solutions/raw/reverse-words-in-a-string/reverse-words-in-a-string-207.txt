class Solution {

    public String reverseWords(String s) {
        StringBuilder result = new StringBuilder();
        int end = s.length() - 1;
        while (end >= 0) {
            // Skip trailing spaces
            while (end >= 0 && Character.isWhitespace(s.charAt(end))) {
                end--;
            }
            // Break if no more words
            if (end < 0)
                break;
            int start = end;
            // Find the start of the word
            while (start >= 0 && !Character.isWhitespace(s.charAt(start))) {
                start--;
            }
            // Append the word and a space
            result.append(s.substring(start + 1, end + 1)).append(" ");
            // Move the end pointer to the beginning of the next word
            end = start;
        }
        // Trim to remove the trailing space and return the result
        return result.toString().trim();
    }
}
