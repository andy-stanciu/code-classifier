class Solution {

    public String reverseWords(String s) {
        String ss = s.trim();
        String[] str = ss.split("\\\\s+");
        StringBuilder sd = new StringBuilder();
        int n = str.length - 1;
        for (int i = n; i >= 0; i--) {
            sd.append(str[i] + " ");
        }
        return sd.toString().trim();
    }
}
