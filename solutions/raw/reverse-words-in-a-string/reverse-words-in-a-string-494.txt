class Solution {

    public String reverseWords(String s) {
        String s1 = s.trim();
        String[] str = s1.split("\\\\s+");
        String ans = "";
        for (int i = str.length - 1; i >= 0; i--) {
            ans = ans + str[i];
            if (i != 0) {
                ans = ans + " ";
            }
        }
        return ans;
    }
}
