class Solution {

    public String reverseWords(String s) {
        int n = s.length();
        s = s.trim();
        String[] words = s.split("\\\\s+");
        //now reversing the elements of array words
        int left = 0;
        int right = words.length - 1;
        while (left < right) {
            String temp = words[left];
            words[left] = words[right];
            words[right] = temp;
            left++;
            right--;
        }
        //to covert words array back to string
        String result = String.join(" ", words);
        return result;
    }
}
