class Solution {

    public String reverseWords(String s) {
        StringBuilder sb = new StringBuilder();
        //conver single string into array of words and remove spaces
        String[] words = s.trim().split("\\\\s+");
        //iterate backwards over array and use StringBuilder to construct the reverse string
        for (int i = words.length - 1; i >= 0; i--) {
            sb.append(words[i]);
            //check if is not the first sting tecnically last value to append
            if (i != 0) {
                //append space in between words expect first and last one
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
