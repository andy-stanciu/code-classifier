class Solution {

    public String reverseWords(String s) {
        String reverse = "";
        if (s != null) {
            s = s.replaceAll("\\\\s+", " ");
            String[] initialString = s.split(" ");
            for (int i = initialString.length - 1; i >= 0; i--) reverse += initialString[i] + " ";
            reverse = reverse.trim();
        }
        return reverse;
    }
}
