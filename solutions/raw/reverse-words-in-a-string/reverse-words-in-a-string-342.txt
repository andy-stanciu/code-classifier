class Solution {

    public String reverseWords(String s) {
        StringBuilder newString = new StringBuilder();
        s = s.trim();
        String[] arr = s.split("\\\\s+");
        for (int i = arr.length - 1; i >= 0; i--) {
            newString.append(arr[i]);
            if (i != 0) {
                newString.append(" ");
            }
        }
        return newString.toString();
    }
}
