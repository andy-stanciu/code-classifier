class Solution {

    public String reverseWords(String s) {
        StringBuilder result = new StringBuilder();
        //removes any trailing or ending spaces
        s = s.strip();
        //splits strings by spaces allowing for each word to appear separately
        String[] word_arr = s.split(" ");
        for (//for loop to iterate through all words in the string in reverse order (because string has to be reversed)
        //for loop to iterate through all words in the string in reverse order (because string has to be reversed)
        //for loop to iterate through all words in the string in reverse order (because string has to be reversed)
        int i = word_arr.length - 1; i >= 0; i--) {
            //removes any trailing or ending spaces
            word_arr[i] = word_arr[i].strip();
            if (word_arr[i].length() != 0) {
                if (i == 0)
                    result.append(word_arr[i]);
                else {
                    result.append(word_arr[i]);
                    result.append(" ");
                }
            }
        }
        return result.toString();
    }
}
