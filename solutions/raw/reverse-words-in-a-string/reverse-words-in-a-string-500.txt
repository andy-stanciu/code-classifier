class Solution {

    public String reverseWords(String s) {
        if (s.length() <= 1 || s == null)
            return s;
        String[] words = s.split("\\\\s+");
        StringBuilder ans = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            ans.append(words[i]);
            ans.append(" ");
        }
        return ans.toString().trim();
    }
}
