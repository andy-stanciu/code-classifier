class Solution {

    public String reverseWords(String s) {
        // Split by spaces after trimming the string
        String[] words = s.trim().split("\\\\s+");
        // Create an array to store reversed words
        String[] rs = new String[words.length];
        // Index for reversed array
        int j = 0;
        // Reverse the words
        for (int i = words.length - 1; i >= 0; i--) {
            rs[j] = words[i];
            j++;
        }
        // Join reversed words with a single space
        String ns = String.join(" ", rs);
        return ns;
    }
}
