class Solution {

    public String reverseWords(String s) {
        s = s.replaceAll("\\\\s+", " ").trim();
        String[] string1 = s.split(" ");
        int j = string1.length - 1, i = 0;
        while (i < j) {
            String temp = string1[i];
            string1[i] = string1[j];
            string1[j] = temp;
            j--;
            i++;
        }
        s = String.join(" ", string1);
        return s;
    }
}
