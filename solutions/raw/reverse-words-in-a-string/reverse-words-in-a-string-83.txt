class Solution {

    public String reverseWords(String s) {
        return Arrays.stream(s.split(" ")).map(String::trim).filter(word -> !word.isEmpty()).reduce((word1, word2) -> word2 + " " + word1).orElse("");
    }
}

class Solution {

    public String reverseWords(String s) {
        String[] words = s.trim().split(" +");
        Collections.reverse(Arrays.asList(words));
        return String.join(" ", words);
    }
}
