class Solution {

    public String reverseWords(String s) {
        String temp = "";
        Stack<String> stck = new Stack<>();
        StringTokenizer st = new StringTokenizer(s);
        while (st.hasMoreTokens()) {
            stck.push(st.nextToken());
        }
        while (!stck.isEmpty()) {
            temp += stck.pop() + " ";
        }
        temp = temp.trim();
        return temp;
    }
}
