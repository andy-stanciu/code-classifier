class Solution {

    public String reverseWords(String s) {
        String[] parts = s.split("\\\\s+");
        String reversed = parts[parts.length - 1];
        for (int i = parts.length - 2; i > -1; i--) {
            if (parts[i] == "") {
                continue;
            }
            reversed = reversed + " " + parts[i];
        }
        return reversed;
    }
}
