class Solution {

    public String reverseWords(String s) {
        String[] splitedWords = s.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = splitedWords.length - 1; i >= 0; i--) {
            sb.append(splitedWords[i]);
            if (i > 0) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
