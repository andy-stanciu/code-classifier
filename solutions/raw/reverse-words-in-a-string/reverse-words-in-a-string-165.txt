class Solution {

    public String reverseWords(String s) {
        String[] words = s.trim().split(" ");
        StringJoiner joiner = new StringJoiner(" ");
        for (int i = words.length - 1; i >= 0; i--) {
            if (words[i].length() > 0)
                joiner.add(words[i]);
        }
        return joiner.toString();
    }
}
