class Solution {

    public String reverseWords(String s) {
        var words = s.split(" ");
        var result = new ArrayList<String>();
        for (var i = words.length - 1; i >= 0; i--) {
            if (words[i].isEmpty())
                continue;
            result.add(words[i]);
        }
        return String.join(" ", result);
    }
}
