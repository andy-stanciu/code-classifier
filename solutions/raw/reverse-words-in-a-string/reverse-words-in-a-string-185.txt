class Solution {

    public String reverseWords(String s) {
        String reversedS = reverseWord(s.trim());
        String[] words = reversedS.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < words.length; i++) {
            String reversedWord = reverseWord(words[i].trim());
            sb.append(reversedWord);
            sb.append(" ");
        }
        return sb.toString().trim();
    }

    public String reverseWord(String w) {
        if (w.length() <= 1) {
            return w;
        }
        StringBuilder sb = new StringBuilder(w);
        int left = 0;
        int right = w.length() - 1;
        while (left < right) {
            sb.setCharAt(left, w.charAt(right));
            sb.setCharAt(right, w.charAt(left));
            left++;
            right--;
        }
        return sb.toString();
    }
}
