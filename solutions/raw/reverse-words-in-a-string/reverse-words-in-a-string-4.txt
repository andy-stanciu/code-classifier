class Solution {

    public String reverseWords(String s) {
        String[] arr = s.split("\\\\s+");
        List<String> list = new ArrayList<>();
        Collections.addAll(list, arr);
        Collections.reverse(list);
        StringBuilder sb = new StringBuilder();
        for (String a : list) {
            sb.append(a).append(" ");
        }
        return sb.toString().trim();
    }
}
