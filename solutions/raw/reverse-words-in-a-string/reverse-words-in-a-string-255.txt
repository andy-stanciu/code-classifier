class Solution {

    public String reverseWords(String s) {
        StringBuilder sb = new StringBuilder();
        String[] ary = s.split(" ");
        for (int i = ary.length - 1; i >= 0; i--) {
            if (ary[i].isBlank())
                continue;
            sb.append(ary[i]);
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
