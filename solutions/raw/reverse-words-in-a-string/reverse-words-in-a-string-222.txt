class Solution {

    public String reverseWords(String s) {
        StringBuffer sb = new StringBuffer();
        String[] st = s.trim().split("\\\\s+");
        for (int i = st.length - 1; i >= 0; i--) {
            String inp = st[i];
            sb = sb.append(inp).append(" ");
        }
        return sb.toString().trim();
    }
}
