class Solution {

    public String reverseWords(String s) {
        String[] words = s.split(" ");
        StringBuilder res = new StringBuilder();
        int i;
        for (i = words.length - 1; i >= 0; i--) {
            if (!words[i].isEmpty()) {
                if (res.length() > 0) {
                    res.append(" ");
                }
                res.append(words[i]);
            }
        }
        String result = res.toString();
        return result;
    }
}
