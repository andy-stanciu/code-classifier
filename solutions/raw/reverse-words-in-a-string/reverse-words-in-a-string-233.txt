class Solution {

    public String reverseWords(String s) {
        s = s.trim().replaceAll("\\\\s+", " ");
        String[] words = s.split(" ");
        StringBuilder strRev = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            if (i < words.length - 1) {
                strRev.append(" ");
            }
            strRev.append(words[i]);
        }
        return strRev.toString();
    }
}
