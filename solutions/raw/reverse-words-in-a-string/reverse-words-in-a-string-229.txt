class Solution {

    public String reverseWords(String s) {
        //remove multiple space
        String[] arr = s.split("\\\\s+");
        StringBuilder sb = new StringBuilder("");
        for (int i = arr.length - 1; i >= 0; i--) {
            sb.append(arr[i] + " ");
        }
        String ans = sb.toString();
        //remove space from trailing and leading
        ans = ans.trim();
        return ans;
    }
}
