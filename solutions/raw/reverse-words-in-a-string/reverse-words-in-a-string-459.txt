class Solution {

    public String reverseWords(String word) {
        List<String> words = new ArrayList<>();
        String[] strArray = word.split("\\\\s+");
        for (int i = strArray.length - 1; i >= 0; i--) {
            words.add(strArray[i]);
        }
        String reverse = words.stream().collect(Collectors.joining(" "));
        return reverse.trim();
    }
}
