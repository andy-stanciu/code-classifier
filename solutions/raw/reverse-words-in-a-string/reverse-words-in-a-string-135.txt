class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] strArr = s.split("\\\\s+");
        int n = strArr.length;
        int start = 0;
        int end = n - 1;
        while (start <= end) {
            String temp = strArr[start];
            strArr[start] = strArr[end];
            strArr[end] = temp;
            start++;
            end--;
        }
        String ans = String.join(" ", strArr);
        return ans;
    }
}
