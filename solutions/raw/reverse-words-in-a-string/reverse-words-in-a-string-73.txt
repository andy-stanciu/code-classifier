class Solution {

    public String reverseWords(String s) {
        //to remove leading and trailing spaces
        s = s.trim();
        //splits string into an array of words whenever a single space is detected
        String[] arr = s.split(" ");
        String ans = "";
        for (int i = arr.length - 1; i >= 0; i--) {
            if (// checks if the string has blank space if yes then do no bother to go through the rest continue
            arr[i].isEmpty())
                continue;
            {
                if (i != 0)
                    //addition of spaces if we have not reached the last word
                    ans = ans + arr[i] + " ";
                else
                    ans = ans + arr[i];
            }
        }
        return ans;
    }
}
