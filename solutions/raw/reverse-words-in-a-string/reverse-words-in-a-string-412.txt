class Solution {

    public String reverseWords(String s) {
        s = s.replaceAll("\\\\s+", " ").trim();
        String[] str = s.split(" ");
        String ptr = "";
        for (int i = str.length - 1; i >= 0; i--) {
            ptr = ptr + str[i] + " ";
        }
        return ptr.substring(0, ptr.length() - 1);
    }
}
