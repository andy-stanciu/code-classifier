class Solution {

    public String reverseWords(String s) {
        StringBuilder str1 = new StringBuilder();
        String[] words = s.trim().split("\\\\s+");
        for (int i = words.length - 1; i >= 0; i--) {
            str1.append(words[i]);
            if (i != 0) {
                str1.append(" ");
            }
        }
        return str1.toString();
    }
}
