class Solution {

    public String reverseWords(String s) {
        String[] sArr = s.split("\\\\s+");
        StringBuilder result = new StringBuilder();
        for (int i = sArr.length - 1; i >= 0; i--) {
            result.append(sArr[i]);
            if (i != 0) {
                result.append(" ");
            }
        }
        return result.toString().trim();
    }
}
