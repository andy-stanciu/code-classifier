class Solution {

    public String reverseWords(String s) {
        String[] words = s.trim().split("\\\\s+");
        StringBuilder sen = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            sen.append(words[i]);
            if (i != 0) {
                sen.append(" ");
            }
        }
        return sen.toString();
    }
}
