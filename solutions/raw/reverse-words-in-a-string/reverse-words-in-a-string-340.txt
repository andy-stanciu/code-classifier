class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        StringBuilder st = new StringBuilder();
        String[] str = s.split("\\\\s+");
        st.append(str[str.length - 1]);
        for (int i = str.length - 2; i >= 0; i--) {
            st.append(" ");
            st.append(str[i]);
        }
        return st.toString();
    }
}
