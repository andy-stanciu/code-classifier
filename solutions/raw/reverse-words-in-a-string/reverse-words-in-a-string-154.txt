class Solution {

    public String reverseWords(String s) {
        String[] words = s.trim().split(" ");
        StringBuilder response = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            if (words[i].length() == 0)
                continue;
            response.append(words[i]);
            if (i != 0)
                response.append(" ");
        }
        return response.toString();
    }
}
