class Solution {

    public String reverseWords(String s) {
        String[] word = s.trim().split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = word.length - 1; i >= 0; i--) {
            if (word[i].length() > 0) {
                sb.append(word[i]);
                sb.append(" ");
            }
        }
        return sb.substring(0, sb.length() - 1).toString();
    }
}
