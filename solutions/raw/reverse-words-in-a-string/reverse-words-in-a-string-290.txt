class Solution {

    public String reverseWords(String s) {
        //trim() => remove leading and trailing white spaces
        //split(" ") split the on the basis of space and return array
        String[] array = s.trim().split(" ");
        StringBuilder ans = new StringBuilder();
        int length = array.length - 1;
        //iterate the array from last index to index 1
        while (length > 0) {
            if (array[length].matches("[a-zA-Z0-9]+")) {
                //add string and space to string builder
                ans.append(array[length]);
                ans.append(" ");
            }
            length--;
        }
        //add index 0 string in answer string
        ans.append(array[0]);
        return ans.toString();
    }
}
