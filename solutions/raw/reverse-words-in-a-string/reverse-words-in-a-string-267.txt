class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] sArray = s.split("\\\\s+");
        StringBuilder res = new StringBuilder();
        for (int i = sArray.length - 1; i >= 0; i--) {
            res.append(sArray[i]);
            if (i != 0) {
                res.append(" ");
            }
        }
        return res.toString();
    }
}
