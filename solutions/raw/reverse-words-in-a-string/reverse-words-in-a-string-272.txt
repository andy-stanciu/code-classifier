class Solution {

    public String reverseWords(String s) {
        String reversedString = processString(s);
        // Print the resulting string
        // System.out.println("Reversed String:");
        // System.out.println(reversedString);
        return reversedString;
    }

    // Method to process the string: trim, split, reverse, and join
    public static String processString(String s) {
        // Trim leading and trailing spaces
        String trimmedInput = s.trim();
        // Split the trimmed string into words
        String[] substrings = trimmedInput.split("\\\\s+");
        // Convert the array to an ArrayList
        ArrayList<String> stringList = new ArrayList<>(Arrays.asList(substrings));
        // Convert the reversed ArrayList to a single string
        String reversedString = convertReversedListToString(stringList);
        return reversedString;
    }

    // Method to convert reversed ArrayList to a single string
    public static String convertReversedListToString(ArrayList<String> list) {
        StringBuilder sb = new StringBuilder();
        for (int i = list.size() - 1; i >= 0; i--) {
            sb.append(list.get(i));
            if (i > 0) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
