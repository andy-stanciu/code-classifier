class Solution {

    public String reverseWords(String s) {
        // usede to remove start and end space
        s = s.trim();
        // spliting
        String[] wrds = s.split("\\\\s+");
        int start = 0;
        int end = wrds.length - 1;
        while (start < end) {
            String x = wrds[start];
            wrds[start] = wrds[end];
            wrds[end] = x;
            start++;
            end--;
        }
        String res = String.join(" ", wrds);
        return res;
    }
}
