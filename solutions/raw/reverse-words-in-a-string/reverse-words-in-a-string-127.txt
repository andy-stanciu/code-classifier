class Solution {

    public String reverseWords(String s) {
        String[] words = s.split("\\\\s+");
        String result = "";
        for (int i = words.length - 1; i >= 0; i--) {
            if (i == words.length - 1) {
                result = result + words[i];
            } else {
                if (words[i] != "") {
                    result = result + " " + words[i];
                }
            }
        }
        return result;
    }
}
