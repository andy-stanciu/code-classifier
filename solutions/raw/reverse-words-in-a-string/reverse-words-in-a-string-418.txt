class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] se = s.split(" ");
        StringBuilder st = new StringBuilder();
        for (int i = se.length - 1; i >= 0; i--) {
            if (se[i] != "") {
                st.append(i != se.length - 1 ? " " + se[i] : se[i]);
            }
        }
        return st.toString();
    }
}
