class Solution {

    public String reverseWords(String s) {
        String[] str = s.replaceAll("\\\\s+", " ").trim().split(" ");
        String temp = "";
        int length = str.length;
        for (int i = 0; i < length / 2; i++) {
            temp = str[i];
            str[i] = str[length - i - 1];
            str[length - i - 1] = temp;
        }
        String ans = str[0];
        for (int i = 1; i < length; i++) {
            ans += " " + str[i];
        }
        return ans;
    }
}
