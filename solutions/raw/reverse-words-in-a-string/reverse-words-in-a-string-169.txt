class Solution {

    public String reverseWords(String s) {
        String[] arr = s.split("\\\\s+");
        String ans = new String();
        int n = arr.length;
        for (int i = n - 1; i >= 0; i--) {
            ans += arr[i];
            if (i != 0)
                ans += " ";
        }
        return ans.trim();
    }
}
