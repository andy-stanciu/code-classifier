class Solution {

    public String reverseWords(String s) {
        String[] set = s.split(" ");
        StringBuilder temp = new StringBuilder("");
        for (int i = set.length - 1; i >= 0; i--) {
            if (!set[i].isBlank()) {
                temp.append(set[i]);
                temp.append(" ");
            }
        }
        temp.deleteCharAt(temp.length() - 1);
        return temp.toString();
    }
}
