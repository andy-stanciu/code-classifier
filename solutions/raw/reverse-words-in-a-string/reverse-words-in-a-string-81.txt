class Solution {

    public String reverseWords(String s) {
        // splitting the word whenever a space appears
        String[] str = s.split("\\\\s+");
        // Initializing an empty string buffer
        StringBuffer sbf = new StringBuffer();
        for (int i = str.length - 1; i >= 0; i--) {
            //appending the string array value to string buffer
            sbf.append(str[i]);
            if (i != 0)
                //adding a space at the end
                sbf.append(" ");
        }
        //returning the stringbuffer after converting it to a string and trimming it for spaces
        return sbf.toString().trim();
    }
}
