class Solution {

    public String reverseWords(String s) {
        StringTokenizer sz = new StringTokenizer(s);
        StringBuilder sb = new StringBuilder();
        Stack<String> st = new Stack<>();
        String space = " ";
        while (sz.hasMoreTokens()) {
            st.push(sz.nextToken());
            st.push(space);
        }
        if (!st.isEmpty()) {
            st.pop();
        }
        while (!st.isEmpty()) {
            sb.append(st.pop());
        }
        return sb.toString();
    }
}
