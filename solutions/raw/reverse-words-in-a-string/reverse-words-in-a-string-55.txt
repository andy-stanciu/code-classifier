class Solution {

    public String reverseWords(String s) {
        // O(n) time | O(n) space
        String[] strs = s.trim().split("\\\\s+");
        StringBuilder res = new StringBuilder();
        for (int i = strs.length - 1; i >= 0; i--) res.append(strs[i]).append(" ");
        return res.toString().trim();
    }
}
