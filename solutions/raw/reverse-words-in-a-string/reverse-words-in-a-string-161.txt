class Solution {

    public String reverseWords(String s) {
        // use split function and regex
        String[] arr = s.split(" +");
        StringBuilder res = new StringBuilder();
        for (int index = arr.length - 1; index >= 0; index--) {
            res.append(arr[index]);
            res.append(" ");
        }
        return res.toString().trim();
    }
}
