class Solution {

    public String reverseWords(String s) {
        String[] split = s.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = split.length - 1; i >= 0; i--) {
            //dont insert space for the last element
            if (i < split.length - 1) {
                sb.append(" ");
            }
            sb.append(split[i]);
        }
        return sb.toString();
    }
}
