class Solution {

    public String reverseWords(String s) {
        String[] strs = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = strs.length - 1; i >= 0; i--) {
            if (strs[i].length() != 0) {
                sb.append(strs[i].trim());
                sb.append(" ");
            }
        }
        return sb.toString().trim();
    }
}
