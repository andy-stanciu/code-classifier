class Solution {

    public String reverseWords(String s) {
        String trimmed = s.trim().replaceAll("\\\\s{2,}", " ");
        String[] words = trimmed.split(" ");
        StringBuilder reversedStatement = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            reversedStatement.append(words[i]);
            if (i > 0) {
                reversedStatement.append(" ");
            }
        }
        return reversedStatement.toString();
    }
}
