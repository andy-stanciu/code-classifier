class Solution {

    public String reverseWords(String s) {
        Stack<String> st = new Stack<>();
        StringTokenizer stt = new StringTokenizer(s, " ", false);
        String ans = "";
        while (stt.hasMoreTokens()) {
            st.push(stt.nextToken());
        }
        while (!st.isEmpty()) {
            ans = ans.concat(st.peek());
            st.pop();
            if (!st.isEmpty()) {
                ans = ans.concat(" ");
            }
        }
        return ans;
    }
}
