class Solution {

    public String reverseWords(String s) {
        //using two pointers
        String[] words = s.trim().split("\\\\s+");
        int start = 0;
        int end = words.length - 1;
        while (start < end) {
            String tempString = words[start];
            words[start] = words[end];
            words[end] = tempString;
            start++;
            end--;
        }
        return String.join(" ", words);
    }
}
