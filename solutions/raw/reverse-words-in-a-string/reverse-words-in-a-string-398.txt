class Solution {

    public String reverseWords(String s) {
        String[] strArr = s.split("\\s+");
        int first = 0;
        int last = strArr.length - 1;
        while (first < last) {
            String temp = strArr[first];
            strArr[first] = strArr[last];
            strArr[last] = temp;
            first++;
            last--;
        }
        return String.join(" ", strArr).trim();
    }
}
