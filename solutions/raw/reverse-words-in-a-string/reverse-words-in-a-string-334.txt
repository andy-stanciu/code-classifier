class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String retarr = "";
        StringTokenizer st = new StringTokenizer(s);
        int l = st.countTokens();
        String[] arr = new String[l];
        for (int x = 0; x < l; x++) {
            arr[x] = st.nextToken();
        }
        for (int x = (l - 1); x >= 0; x--) {
            retarr = retarr + arr[x] + " ";
        }
        retarr = retarr.trim();
        return retarr;
    }
}
