class Solution {

    public String reverseWords(String s) {
        String result = "";
        String[] words = s.split("\\\\W+");
        int left = 0;
        int right = words.length - 1;
        while (left < right) {
            String j = words[left];
            words[left] = words[right];
            words[right] = j;
            left++;
            right--;
        }
        result = String.join(" ", words).trim();
        return result;
    }
}
