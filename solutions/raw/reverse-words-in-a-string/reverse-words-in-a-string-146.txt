class Solution {

    public String reverseWords(String s) {
        String[] words = s.trim().split("\\\\s+");
        int st = 0;
        int e = words.length - 1;
        while (st < e) {
            String temp = words[st];
            words[st] = words[e];
            words[e] = temp;
            st++;
            e--;
        }
        String str = String.join(" ", words);
        return str;
    }
}
