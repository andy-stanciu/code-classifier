class Solution {

    public String reverseWords(String s) {
        String[] words = s.replaceAll("\\\\s+", " ").trim().split(" ");
        int start = 0;
        int end = words.length - 1;
        while (start < end) {
            String temp = words[start];
            words[start] = words[end];
            words[end] = temp;
            start++;
            end--;
        }
        String answer = String.join(" ", words);
        return answer;
    }
}
