class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] arr = s.split(" ");
        String ret = "";
        for (int i = arr.length - 1; i >= 0; i--) {
            arr[i] = arr[i].trim();
            if (arr[i] != "") {
                ret = ret + arr[i].trim() + " ";
            }
        }
        return ret.trim();
    }
}
