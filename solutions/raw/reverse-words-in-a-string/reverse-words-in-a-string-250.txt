class Solution {

    public String reverseWords(String s) {
        String[] arr = s.trim().split("\\\\s+");
        int n = arr.length;
        if (n == 1 || n == 0)
            return s.trim();
        int i = 0;
        int j = n - 1;
        while (i <= j) {
            String temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
        return String.join(" ", arr);
    }
}
