class Solution {

    public String reverseWords(String s) {
        s.trim();
        String[] splited = s.split(" ");
        String res = "";
        for (int i = splited.length - 1; i >= 0; i--) {
            if (splited[i] != "") {
                res += splited[i] + " ";
            }
        }
        return res.trim();
    }
}
