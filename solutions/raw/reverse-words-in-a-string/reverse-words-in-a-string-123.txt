class Solution {

    public String reverseWords(String s) {
        String str = s.trim().replaceAll("\\\\s+", " ");
        String[] temp = str.split(" ");
        StringBuilder res = new StringBuilder();
        for (int i = temp.length - 1; i >= 0; i--) {
            res.append(temp[i]);
            if (i != 0) {
                res.append(" ");
            }
        }
        String after = res.toString();
        return after;
    }
}
