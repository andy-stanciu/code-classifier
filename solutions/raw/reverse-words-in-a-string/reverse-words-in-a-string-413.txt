class Solution {

    public String reverseWords(String s) {
        String[] words = s.trim().split("\\\\s+");
        int n = words.length;
        String res = "";
        for (int i = n - 1; i >= 0; i--) {
            if (i < n - 1 && i >= 0)
                res += " ";
            res += words[i];
        }
        return res;
    }
}
