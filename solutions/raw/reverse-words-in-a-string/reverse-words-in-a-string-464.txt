class Solution {

    public String reverseWords(String s) {
        String[] wrd = s.split("\\\\s+");
        StringBuffer s2 = new StringBuffer();
        for (int i = (wrd.length - 1); i >= 0; i--) {
            s2.append(wrd[i]);
            s2.append(" ");
        }
        return s2.toString().trim();
    }
}
