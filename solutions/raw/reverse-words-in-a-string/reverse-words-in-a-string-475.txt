class Solution {

    public String reverseWords(String s) {
        var words = s.split(" ");
        var sb = new StringBuilder();
        for (var i = words.length - 1; i >= 0; i--) {
            if (!words[i].isEmpty()) {
                sb.append(words[i] + " ");
            }
        }
        sb.deleteCharAt(sb.length() - 1);
        return sb.toString();
    }
}
