class Solution {

    public String reverseWords(String s) {
        String[] sr = s.trim().split("\\\\s+");
        StringBuilder res = new StringBuilder();
        int len = sr.length - 1;
        for (int i = len; i >= 0; i--) {
            res.append(sr[i]);
            if (i > 0) {
                res.append(" ");
            }
        }
        return res.toString();
    }
}
