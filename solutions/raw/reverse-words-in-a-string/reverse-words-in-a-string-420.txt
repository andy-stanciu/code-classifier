class Solution {

    public static String reverseWords(String word2Reverse) {
        // Initialize an empty string to store the reversed words
        String reversed = "";
        // Split the string into words & remove trailing spaces
        word2Reverse = word2Reverse.trim();
        String[] StringArray = word2Reverse.split(" ");
        // Iterate through the words array from end to beginning
        for (int i = StringArray.length - 1; i >= 0; i--) {
            //add word from array at i to reversed string
            reversed += StringArray[i];
            //add space between reversed words
            if (i != 0 && !reversed.endsWith(" "))
                reversed += " ";
        }
        //end for loop
        return reversed;
    }
    //end reverseWords()
}
//end class
