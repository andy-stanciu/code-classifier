class Solution {

    public String reverseWords(String s) {
        String[] tr = s.trim().split("\\\\s+");
        String rev = "";
        for (int i = tr.length - 1; i >= 0; i--) {
            rev += tr[i];
            if (i != 0) {
                rev += " ";
            }
        }
        return rev;
    }
}
