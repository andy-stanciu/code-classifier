class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] words = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            String word = words[i];
            if (word.trim().length() != 0) {
                sb.append(word).append(" ");
            }
        }
        return sb.toString().trim();
    }
}
