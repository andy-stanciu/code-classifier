class Solution {

    public String reverseWords(String s) {
        s.trim();
        String[] c = s.split("\\\\s+");
        String[] d = new String[c.length];
        int p = 0;
        for (int i = c.length - 1; i >= 0; i--) {
            d[p] = c[i];
            p++;
        }
        String o = String.join(" ", d);
        return o.trim();
    }
}
