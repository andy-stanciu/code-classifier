class Solution {

    public String reverseWords(String s) {
        String[] arr = s.trim().split(" ");
        String res = "";
        for (int i = arr.length - 1; i >= 0; i--) {
            if (arr[i] == "") {
                continue;
            }
            res += arr[i] + " ";
        }
        return res.trim();
    }
}
