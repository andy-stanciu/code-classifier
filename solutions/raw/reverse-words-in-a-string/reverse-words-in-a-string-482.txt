class Solution {

    public String reverseWords(String s) {
        String[] words = s.split(" ");
        String rev = "";
        for (int i = words.length - 1; i >= 0; i--) {
            if (!words[i].equals("")) {
                rev += words[i];
                if (i != 0 && !words[i - 1].equals("")) {
                    rev += " ";
                }
            }
        }
        return rev;
    }
}
