class Solution {

    public String reverseWords(String s) {
        String[] str = s.split(" ");
        String s1 = "";
        int i = 0;
        for (String sb : str) {
            if (sb.length() != 0) {
                if (i == 0) {
                    s1 = sb + s1;
                    i++;
                } else {
                    s1 = sb + " " + s1;
                }
            }
        }
        return s1;
    }
}
