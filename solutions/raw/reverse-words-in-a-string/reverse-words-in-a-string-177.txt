class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        StringBuilder ans = new StringBuilder();
        String[] words = s.split(" ");
        int r = words.length - 1;
        while (r >= 0) {
            if (words[r] != "") {
                ans.append(words[r]);
                if (r != 0) {
                    ans.append(" ");
                }
            }
            r--;
        }
        return ans.toString();
    }
}
