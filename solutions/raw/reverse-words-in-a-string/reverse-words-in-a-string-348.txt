class Solution {

    public String reverseWords(String s) {
        String[] m = s.trim().split("\\\\s+");
        StringBuilder reverse = new StringBuilder();
        for (int i = m.length - 1; i >= 0; i--) {
            reverse.append(m[i]);
            if (i != 0) {
                reverse.append(" ");
            }
        }
        return reverse.toString();
    }
}
