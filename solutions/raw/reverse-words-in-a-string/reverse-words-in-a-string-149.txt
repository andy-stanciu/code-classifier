class Solution {

    public String reverseWords(String s) {
        String[] str = s.trim().split("\\\\s+");
        String[] reverse = new String[str.length];
        int pos = 0;
        for (int i = str.length - 1; i >= 0; i--) {
            reverse[pos] = str[i];
            pos++;
        }
        return String.join(" ", reverse);
    }
}
