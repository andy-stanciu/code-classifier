class Solution {

    public String reverseWords(String s) {
        String[] list = s.trim().split("\\\\s+");
        for (int i = 0; i < list.length / 2; i++) {
            String temp = list[i];
            list[i] = list[list.length - 1 - i];
            list[list.length - i - 1] = temp;
        }
        String sk = String.join(" ", list);
        return sk;
    }
}
