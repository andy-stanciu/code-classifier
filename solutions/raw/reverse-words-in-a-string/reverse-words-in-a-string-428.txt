class Solution {

    public String reverseWords(String s) {
        StringBuilder result = new StringBuilder();
        String[] words = s.split(" ");
        for (int i = words.length - 1; i >= 0; i--) {
            if (!words[i].isEmpty()) {
                result.append(" ");
                result.append(words[i]);
            }
        }
        result.deleteCharAt(0);
        return result.toString();
    }
}
