class Solution {

    public String reverseWords(String s) {
        String[] subs = s.trim().split("\\\\s+");
        int len = subs.length;
        String ans = "";
        for (int i = len - 1; i > 0; i--) {
            ans += subs[i] + " ";
            System.out.println(ans + ":");
        }
        return ans + subs[0];
    }
}
