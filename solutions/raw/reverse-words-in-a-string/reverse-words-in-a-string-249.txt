class Solution {

    public String reverseWords(String s) {
        //trim the string
        s = s.trim();
        //Split the string
        String[] words = s.split("\\\\s+");
        //reversing a string using stringbuilder
        StringBuilder Rev = new StringBuilder();
        for (int i = words.length - 1; i >= 0; i--) {
            Rev.append(words[i]);
            if (i != 0)
                Rev.append(" ");
        }
        return Rev.toString();
    }
}
