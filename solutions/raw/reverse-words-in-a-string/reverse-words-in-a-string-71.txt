class Solution {

    public String reverseWords(String s) {
        s = s.trim();
        String[] str = s.split("\\\\s+");
        String rev = "";
        for (int i = str.length - 1; i >= 0; i--) {
            if (i == 0)
                rev += str[i];
            else
                rev += (str[i] + " ");
        }
        return rev;
    }
}
