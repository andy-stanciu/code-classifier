class Solution {

    static int uniquePath(int m, int n, int[][] dp, int[][] a) {
        if (m >= 0 && n >= 0 && a[m][n] == 1)
            return 0;
        if (m == 0 && n == 0)
            return 1;
        if (m < 0 || n < 0)
            return 0;
        if (dp[m][n] != -1)
            return dp[m][n];
        int up = uniquePath(m - 1, n, dp, a);
        int down = uniquePath(m, n - 1, dp, a);
        return dp[m][n] = up + down;
    }

    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length, n = obstacleGrid[0].length;
        // if(obstacleGrid[0][0]==1) return 0;
        int[][] dp = new int[m][n];
        for (int[] row : dp) Arrays.fill(row, -1);
        return uniquePath(m - 1, n - 1, dp, obstacleGrid);
    }
}
