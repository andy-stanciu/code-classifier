class Solution {

    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        // do recursive dfs and when there is more than one \'adjacent\' square
        // add it to the stack and add it to a shared list if you are able to reach the end
        // x, 0, 0          start at x, add [1,0] and [0,1] to the stack
        // 0  1, 0          we are at two potential paths
        // 0, 0, 0          for each other square there is only one other \'adjacent\' square (we can only move down and right)
        //  so we stay at two paths and both can reach the bottom right so both are true.
        if (obstacleGrid[0][0] == 1) {
            return 0;
        }
        List<Integer> list = new ArrayList<Integer>();
        dfs(obstacleGrid, list, 0, 0);
        return list.size();
    }

    public void dfs(int[][] matrix, List<Integer> paths, int row, int col) {
        int botRow = matrix.length - 1;
        int botCol = matrix[0].length - 1;
        if (row == botRow && col == botCol) {
            paths.add(1);
            return;
        }
        if (row + 1 < matrix.length && matrix[row + 1][col] != 1) {
            dfs(matrix, paths, row + 1, col);
        }
        if (col + 1 < matrix[0].length && matrix[row][col + 1] != 1) {
            dfs(matrix, paths, row, col + 1);
        }
    }
}
