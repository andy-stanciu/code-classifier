class Solution {

    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int row = obstacleGrid.length, col = obstacleGrid[0].length;
        int[][] dp = new int[row][col];
        //Ask the recursion to do the work
        return findPath(dp, 0, 0, obstacleGrid);
    }

    public int findPath(int[][] dp, int row, int col, int[][] arr) {
        // when we get to the last out of index
        if (row == dp.length || col == dp[0].length)
            return 0;
        //if we get a obstacle
        if (arr[row][col] == 1)
            return 0;
        //checking the dp array
        if (dp[row][col] != 0)
            return dp[row][col];
        //Achieved Goal
        if (row == dp.length - 1 && col == dp[0].length - 1)
            return 1;
        //Recurse to right-side
        int right = findPath(dp, row, col + 1, arr);
        //Recurse to down-side
        int left = findPath(dp, row + 1, col, arr);
        //store it in a dynamic array
        dp[row][col] = left + right;
        //return the dp val
        return dp[row][col];
    }
}
