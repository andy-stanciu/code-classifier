class Solution {

    public int uniquePathsWithObstacles(int[][] og) {
        int m = og.length;
        int n = og[0].length;
        if (og[0][0] == 1)
            return 0;
        og[0][0] = 1;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (!(i == 0 && j == 0) && og[i][j] == 1)
                    og[i][j] = 0;
                else {
                    if (i > 0) {
                        og[i][j] += og[i - 1][j];
                    }
                    if (j > 0) {
                        og[i][j] += og[i][j - 1];
                    }
                }
            }
        }
        return og[m - 1][n - 1];
    }
}
