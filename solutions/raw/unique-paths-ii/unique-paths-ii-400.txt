class Solution {

    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        int[][] dp = new int[m][n];
        dp[0][0] = obstacleGrid[0][0] == 1 ? 0 : 1;
        for (int i = 1; i < n; i++) {
            if (obstacleGrid[0][i] == 1) {
                // If there is an obstacle, no path can reach this cell
                dp[0][i] = 0;
            } else {
                // Otherwise, it depends on the left cell
                dp[0][i] = dp[0][i - 1];
            }
        }
        // Fill the first column
        for (int i = 1; i < m; i++) {
            if (obstacleGrid[i][0] == 1) {
                // If there is an obstacle, no path can reach this cell
                dp[i][0] = 0;
            } else {
                // Otherwise, it depends on the upper cell
                dp[i][0] = dp[i - 1][0];
            }
        }
        // Calculate the number of unique paths for each cell
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (obstacleGrid[i][j] == 1) {
                    // If there is an obstacle, no path can reach this cell
                    dp[i][j] = 0;
                } else {
                    // Sum of paths from above and left
                    dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
                }
            }
        }
        return dp[m - 1][n - 1];
    }
}
