class Solution {

    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int ans = searchPaths(obstacleGrid, 0, 0);
        return ans;
    }

    public int searchPaths(int[][] obstacleGrid, int row, int col) {
        if (row == obstacleGrid.length - 1 && col == obstacleGrid[0].length - 1)
            return 1;
        if (obstacleGrid[row][col] == 1)
            return 0;
        int down = 0, right = 0;
        if (col + 1 <= obstacleGrid[0].length - 1)
            right = searchPaths(obstacleGrid, row, col + 1);
        if (row + 1 <= obstacleGrid.length - 1)
            down = searchPaths(obstacleGrid, row + 1, col);
        return down + right;
    }
}
