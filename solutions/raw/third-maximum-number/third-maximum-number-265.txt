class Solution {

    public int thirdMax(int[] nums) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int num : nums) {
            map.put(num, 1);
        }
        int result = 0, count = 0;
        if (map.size() > 2) {
            for (Map.Entry<Integer, Integer> temp : map.entrySet()) {
                result = temp.getKey();
                count++;
                if (map.size() - count <= 2) {
                    break;
                }
            }
        } else {
            for (Map.Entry<Integer, Integer> temp : map.entrySet()) {
                result = temp.getKey();
            }
        }
        return result;
    }
}
