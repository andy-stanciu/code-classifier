class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> zbior = new TreeSet<>();
        for (int num : nums) {
            zbior.add(num);
        }
        if (zbior.size() > 2) {
            zbior.remove(zbior.last());
            zbior.remove(zbior.last());
        }
        return zbior.last();
    }
}
