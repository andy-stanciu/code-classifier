class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        List<Integer> array = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (!array.contains(nums[i])) {
                array.add(nums[i]);
            }
        }
        int arraySize = array.size();
        if (arraySize >= 3) {
            return array.get(arraySize - 3);
        }
        return array.get(arraySize - 1);
    }
}
