class Solution {

    public int thirdMax(int[] nums) {
        int max1 = -2147483648;
        int max2 = -2147483648;
        int max3 = -2147483648;
        int c = 0;
        for (int i = 0; i < nums.length; i++) {
            max1 = Math.max(max1, nums[i]);
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < max1)
                max2 = Math.max(max2, nums[i]);
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < max2) {
                max3 = Math.max(max3, nums[i]);
                c++;
            }
        }
        if (c > 0)
            return max3;
        return max1;
    }
}
