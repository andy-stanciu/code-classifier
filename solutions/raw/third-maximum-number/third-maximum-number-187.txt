class Solution {

    public int thirdMax(int[] a) {
        HashSet<Integer> hs = new LinkedHashSet<>();
        for (int v : a) {
            hs.add(v);
        }
        ArrayList<Integer> al = new ArrayList<>(hs);
        Collections.sort(al, Collections.reverseOrder());
        if (al.size() >= 3)
            return al.get(2);
        return al.get(0);
    }
}
