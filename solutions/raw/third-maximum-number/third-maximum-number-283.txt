class Solution {

    public int thirdMax(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }
        // Remove duplicates
        int[] uniquesNums = Arrays.stream(nums).distinct().toArray();
        // Sort
        Arrays.sort(uniquesNums);
        // Validation to return correct value
        int length = uniquesNums.length;
        return (length < 3) ? uniquesNums[length - 1] : uniquesNums[length - 3];
    }
}
