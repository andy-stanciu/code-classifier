class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> flag = new TreeSet<>();
        for (int i = 0; i < nums.length; i++) {
            flag.add(nums[i]);
        }
        int i = 0;
        int big = 0;
        for (int val : flag) {
            nums[i] = val;
            big = val;
            i++;
        }
        if (flag.size() >= 3) {
            return nums[flag.size() - 3];
        }
        return big;
    }
}
