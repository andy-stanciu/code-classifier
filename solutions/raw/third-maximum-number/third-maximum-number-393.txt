class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        System.out.println(Arrays.toString(nums));
        Set<Integer> set = new HashSet<>();
        int ans = 0, count = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            ans = nums[i];
            if (set.contains(ans)) {
            } else {
                set.add(ans);
                count++;
                if (count == 3) {
                    break;
                }
            }
        }
        return count < 3 ? nums[nums.length - 1] : ans;
    }
}
