class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        ArrayList<Integer> arr = new ArrayList<Integer>();
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                arr.add(nums[i - 1]);
            }
        }
        // Add the last element of nums
        arr.add(nums[nums.length - 1]);
        if (arr.size() < 3) {
            // If there are less than three distinct elements, return the maximum
            return arr.get(arr.size() - 1);
        } else {
            // Return the third maximum distinct element
            return arr.get(arr.size() - 3);
        }
    }
}
