class Solution {

    public int thirdMax(int[] nums) {
        HashSet<Integer> hash = new HashSet<>();
        for (int i : nums) {
            hash.add(i);
        }
        int[] arr = new int[hash.size()];
        int k = 0;
        for (int j : hash) {
            arr[k] = j;
            k++;
        }
        Arrays.sort(arr);
        if (arr.length < 3)
            return arr[arr.length - 1];
        else
            return arr[arr.length - 3];
    }
}
