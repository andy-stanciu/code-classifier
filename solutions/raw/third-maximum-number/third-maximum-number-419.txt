class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                sum++;
            }
            if (sum == 3 - 1) {
                return nums[i - 1];
            }
        }
        return nums[nums.length - 1];
    }
}
