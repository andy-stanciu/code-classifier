class Solution {

    public int thirdMax(int[] nums) {
        int min = nums[0];
        for (int i : nums) {
            if (i < min) {
                min = i;
            }
        }
        int first = min;
        int second = min;
        int third = min;
        for (int i : nums) {
            if (i > first) {
                third = second;
                second = first;
                first = i;
            } else if (i > second && i < first) {
                third = second;
                second = i;
            } else if (i > third && i < second) {
                third = i;
            }
        }
        if (third == second) {
            return first;
        }
        return third;
    }
}
