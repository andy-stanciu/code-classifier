class Solution {

    public int thirdMax(int[] nums) {
        //to maintan the order.
        HashSet<Integer> container = new LinkedHashSet<>();
        for (int i : nums) {
            container.add(i);
        }
        if (container.size() < 3) {
            Object obj = Collections.max(container);
            return (int) obj;
        }
        List<Integer> arr = new ArrayList<>(container);
        Collections.sort(arr, Collections.reverseOrder());
        return arr.get(2);
    }
}
