class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> set = new HashSet();
        int maxNumber = Integer.MIN_VALUE;
        Arrays.sort(nums);
        int distincts = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            int val = nums[i];
            if (!set.contains(val)) {
                distincts++;
            }
            set.add(val);
            if (distincts == 3) {
                return val;
            }
            maxNumber = Math.max(maxNumber, val);
        }
        return maxNumber;
    }
}
