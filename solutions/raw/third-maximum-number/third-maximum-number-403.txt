class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int max = nums[nums.length - 1];
        if (nums.length >= 3) {
            int count = 0;
            for (int i = nums.length - 1; i > 0; i--) {
                if (nums[i] > nums[i - 1]) {
                    count = count + 1;
                }
                if (count == 2) {
                    return nums[i - 1];
                }
            }
        }
        return max;
    }
}
