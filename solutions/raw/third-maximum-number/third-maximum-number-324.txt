class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        Set<Integer> set = new TreeSet<>();
        for (int i = nums.length - 1; i >= 0; i--) {
            set.add(nums[i]);
            if (set.size() == 3) {
                return nums[i];
            }
        }
        return nums[nums.length - 1];
    }
}
