class Solution {

    public int thirdMax(int[] nums) {
        //First Sort the Array to arrange the element of given array
        Arrays.sort(nums);
        //if array has single element that mean this num is maximum in array
        int num = nums[nums.length - 1];
        //if we found the any max number tht mean we found first largest number
        int isThird = 1;
        int i = nums.length - 1;
        while (isThird < 3 && i > 0) {
            //if two number are equal we skip them that means
            //we count the consecuent distinct number
            if (nums[i] != nums[i - 1]) {
                isThird++;
                // if we get the third number we return it
                if (isThird == 3)
                    return nums[i - 1];
            }
            i--;
        }
        //if we not get third larget number so we return max number which we store in num variable
        return num;
    }
}
