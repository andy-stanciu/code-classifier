class Solution {

    public int thirdMax(int[] nums) {
        int n = nums.length;
        Set<Integer> set = new HashSet<Integer>();
        for (int i : nums) {
            set.add(i);
        }
        if (set.size() < 3) {
            int max = Integer.MIN_VALUE;
            for (int num : set) {
                if (num > max) {
                    max = num;
                }
            }
            return max;
        } else {
            Integer[] arr = set.toArray(new Integer[0]);
            Arrays.sort(arr, Collections.reverseOrder());
            return arr[2];
        }
    }
}
