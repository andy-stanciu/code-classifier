class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int max = nums[n - 1];
        int count = 1;
        int ans = max;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                count++;
                ans = nums[i];
                if (count == 3) {
                    return ans;
                }
            }
        }
        return max;
    }
}
