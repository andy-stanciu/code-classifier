class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int distinctCount = 1, element = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] != element) {
                element = nums[i];
                distinctCount++;
            }
            if (distinctCount == 3)
                return element;
        }
        return nums[nums.length - 1];
    }
}
