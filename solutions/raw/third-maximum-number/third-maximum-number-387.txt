class Solution {

    public int thirdMax(int[] nums) {
        int n = nums.length;
        if (n < 3) {
            // If there are less than 3 nums amounts, return the maximum.
            Arrays.sort(nums);
            return nums[n - 1];
        } else {
            Arrays.sort(nums);
            int distinctCount = 1;
            int thirdHighest = nums[n - 1];
            for (int i = n - 2; i >= 0; i--) {
                if (nums[i] < nums[i + 1]) {
                    distinctCount++;
                }
                if (distinctCount == 3) {
                    thirdHighest = nums[i];
                    break;
                }
            }
            return thirdHighest;
        }
    }
}
