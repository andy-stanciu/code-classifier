class Solution {

    public int thirdMax(int[] nums) {
        int n = nums.length - 1;
        Arrays.sort(nums);
        int count = 0;
        int com = (int) 1e9;
        while (n >= 0) {
            if (com != nums[n]) {
                count++;
                com = nums[n];
            }
            if (count == 3) {
                return nums[n];
            }
            n--;
        }
        return nums[nums.length - 1];
    }
}
