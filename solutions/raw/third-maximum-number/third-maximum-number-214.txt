class Solution {

    public int thirdMax(int[] nums) {
        long fLargest = Long.MIN_VALUE;
        long sLargest = Long.MIN_VALUE;
        long tLargest = Long.MIN_VALUE;
        boolean found = false;
        for (int num : nums) {
            if (num == fLargest || num == sLargest || num == tLargest)
                continue;
            if (num > fLargest) {
                tLargest = sLargest;
                sLargest = fLargest;
                fLargest = num;
            } else if (num > sLargest) {
                tLargest = sLargest;
                sLargest = num;
            } else if (num > tLargest) {
                tLargest = num;
            }
        }
        return tLargest == Long.MIN_VALUE ? (int) fLargest : (int) tLargest;
    }
}
