class Solution {

    public int thirdMax(int[] nums) {
        Integer first, second, third;
        if (nums.length == 0) {
            return Integer.MIN_VALUE;
        }
        first = nums[0];
        second = null;
        third = null;
        boolean isThirstSet = false;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == first || (second != null && nums[i] == second))
                continue;
            if (nums[i] > first) {
                third = second;
                second = first;
                first = nums[i];
            } else if (second == null || nums[i] > second) {
                third = second;
                second = nums[i];
            } else if (third == null || nums[i] >= third) {
                //isThirstSet = true;
                third = nums[i];
            }
        }
        return third == null ? first : third;
    }
}
