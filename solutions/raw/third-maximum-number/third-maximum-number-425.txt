class Solution {

    public int thirdMax(int[] nums) {
        int count = 2;
        int i = nums.length - 1;
        Arrays.sort(nums);
        int max = nums[nums.length - 1];
        while (count > 0 && i >= 0) {
            if (nums[i] < max) {
                max = nums[i];
                count--;
            }
            i--;
        }
        if (count != 0) {
            max = nums[nums.length - 1];
        }
        return max;
    }
}
