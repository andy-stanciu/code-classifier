class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> disnums = new TreeSet<>();
        for (int num : nums) {
            disnums.add(num);
        }
        if (disnums.size() < 3) {
            return disnums.last();
        }
        disnums.pollLast();
        disnums.pollLast();
        return disnums.last();
    }
}
