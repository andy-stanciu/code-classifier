class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int maximum = nums[nums.length - 1];
        int secondMaximum = Integer.MIN_VALUE;
        int thirdMaximum = Integer.MIN_VALUE;
        boolean secondMaximumFound = false;
        boolean thirdMaximumFound = false;
        if (nums.length > 2)
            for (int index = nums.length - 2; index >= 0; index--) if (!secondMaximumFound) {
                if (nums[index] < maximum) {
                    secondMaximum = nums[index];
                    secondMaximumFound = true;
                }
            } else if (nums[index] < secondMaximum) {
                thirdMaximum = nums[index];
                thirdMaximumFound = true;
                break;
            }
        if (thirdMaximumFound)
            return thirdMaximum;
        else
            return maximum;
    }
}
