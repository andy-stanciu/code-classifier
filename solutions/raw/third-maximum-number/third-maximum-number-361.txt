class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 0;
        int max = -10;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (max == nums[i])
                continue;
            else {
                max = nums[i];
                count++;
            }
            if (count == 3) {
                return max;
            }
        }
        return nums[nums.length - 1];
    }
}
