class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int ele : nums) {
            set.add(ele);
        }
        int[] a = new int[set.size()];
        int i = 0;
        for (int x : set) a[i++] = x;
        Arrays.sort(a);
        if (a.length <= 2) {
            return a[a.length - 1];
        }
        return a[a.length - 3];
    }
}
