class Solution {

    public int thirdMax(int[] arr) {
        int len = arr.length;
        Arrays.sort(arr);
        HashSet<Integer> hp = new HashSet<>();
        for (int i = len - 1; i >= 0; i--) {
            int a = arr[i];
            hp.add(a);
            if (hp.size() == 3)
                return a;
        }
        return arr[len - 1];
    }
}
