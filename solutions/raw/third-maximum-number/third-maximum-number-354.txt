class Solution {

    public int thirdMax(int[] a) {
        long large = Long.MIN_VALUE;
        long secondLarge = Long.MIN_VALUE;
        long thirdLarge = Long.MIN_VALUE;
        for (int i = 0; i < a.length; i++) {
            if (a[i] > thirdLarge && a[i] != secondLarge && a[i] != large) {
                if (a[i] >= secondLarge) {
                    if (a[i] >= large) {
                        thirdLarge = secondLarge;
                        secondLarge = large;
                        large = a[i];
                        continue;
                    }
                    thirdLarge = secondLarge;
                    secondLarge = a[i];
                    continue;
                }
                if (a[i] > thirdLarge)
                    thirdLarge = a[i];
            }
        }
        return thirdLarge == Long.MIN_VALUE ? (int) large : (int) thirdLarge;
    }
}
