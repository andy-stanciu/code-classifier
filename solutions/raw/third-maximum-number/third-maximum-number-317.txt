class Solution {

    public int thirdMax(int[] nums) {
        long max = Long.MIN_VALUE;
        long maxMinus1 = Long.MIN_VALUE;
        long maxMinus2 = Long.MIN_VALUE;
        for (int i : nums) {
            if (i > max) {
                maxMinus2 = maxMinus1;
                maxMinus1 = max;
                max = i;
            } else if (i > maxMinus1 && i < max) {
                maxMinus2 = maxMinus1;
                maxMinus1 = i;
            } else if (i > maxMinus2 && i < maxMinus1 && i < max)
                maxMinus2 = i;
        }
        if (maxMinus2 > Long.MIN_VALUE)
            return (int) maxMinus2;
        return (int) max;
    }
}
