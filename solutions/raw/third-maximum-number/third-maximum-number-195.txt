class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        HashSet<Integer> hs = new HashSet<>();
        for (int n : nums) {
            hs.add(n);
        }
        int[] arr = new int[hs.size()];
        int i = 0;
        for (int n : hs) {
            arr[i++] = n;
        }
        Arrays.sort(arr);
        if (arr.length < 3)
            return arr[arr.length - 1];
        return arr[arr.length - 3];
    }
}
