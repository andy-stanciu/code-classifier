class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int[] arr = new int[set.size()];
        int index = 0;
        for (int num : set) {
            arr[index++] = num;
        }
        Arrays.sort(arr);
        if (arr.length < 3) {
            return arr[arr.length - 1];
        }
        return arr[arr.length - 3];
    }
}
