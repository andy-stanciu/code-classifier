class Solution {

    public int thirdMax(int[] nums) {
        ArrayList<Integer> arr = new ArrayList<Integer>();
        TreeSet<Integer> set = new TreeSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        for (int num : set) {
            arr.add(num);
        }
        if (arr.size() <= 2) {
            return arr.get(arr.size() - 1);
        } else {
            return arr.get(arr.size() - 3);
        }
    }
}
