class Solution {

    public int thirdMax(int[] nums) {
        long thirdmax = Long.MIN_VALUE;
        long secondmax = Long.MIN_VALUE;
        long firstmax = Long.MIN_VALUE;
        for (int n : nums) {
            if (n > firstmax) {
                thirdmax = secondmax;
                secondmax = firstmax;
                firstmax = n;
            } else if (n < firstmax && n > secondmax) {
                thirdmax = secondmax;
                secondmax = n;
            } else if (n < secondmax && n > thirdmax) {
                thirdmax = n;
            }
        }
        return (thirdmax != Long.MIN_VALUE) ? (int) thirdmax : (int) firstmax;
    }
}
