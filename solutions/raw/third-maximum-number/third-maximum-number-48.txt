class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int n : nums) {
            set.add(n);
        }
        if (set.size() < 3)
            return set.stream().reduce((a, b) -> b).get();
        PriorityQueue<Integer> minheap = new PriorityQueue<>();
        for (int n : set) {
            minheap.offer(n);
            if (minheap.size() > 3) {
                minheap.poll();
            }
        }
        return minheap.peek();
    }
}
