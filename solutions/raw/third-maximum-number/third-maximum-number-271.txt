class Solution {

    public int thirdMax(int[] nums) {
        //create hashset to it exclude duplicate values
        HashSet<Integer> hs = new HashSet<>();
        for (int i : nums) {
            hs.add(i);
        }
        //transfer values of hashset into array
        int[] arr = new int[hs.size()];
        int index = 0;
        for (int i : hs) {
            arr[index] = i;
            index++;
        }
        // sort the array
        Arrays.sort(arr);
        if (arr.length < 3) {
            return arr[arr.length - 1];
        } else {
            return arr[arr.length - 3];
        }
    }
}
