class Solution {

    public int thirdMax(int[] nums) {
        HashSet<Integer> numSet = new HashSet<>();
        Integer max = null;
        Integer second = null;
        Integer third = null;
        for (int n : nums) {
            boolean inSet = numSet.contains(n);
            if (inSet)
                continue;
            numSet.add(n);
            if (max == null || n > max) {
                third = second;
                second = max;
                max = n;
                continue;
            }
            if (second == null || n > second) {
                third = second;
                second = n;
                continue;
            }
            if (third == null || n > third) {
                third = n;
            }
        }
        return third == null ? max : third;
    }
}
