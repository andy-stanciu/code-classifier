class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int max = nums[nums.length - 1];
        int secondMax = max;
        int thirdMax = max;
        int len = nums.length;
        int distinctCount = 0;
        for (int i = len - 2; i >= 0; i--) {
            if (nums[i] < max) {
                thirdMax = secondMax;
                secondMax = max;
                max = nums[i];
                distinctCount++;
                if (distinctCount == 2) {
                    break;
                }
            }
        }
        if (distinctCount < 2) {
            return nums[nums.length - 1];
        } else {
            return max;
        }
    }
}
