class Solution {

    public int thirdMax(int[] nums) {
        long highest = Long.MIN_VALUE, secondHighest = Long.MIN_VALUE, thirdHighest = Long.MIN_VALUE;
        boolean thirdMax = false;
        for (int i = 0; i < nums.length; i++) {
            long num = (long) nums[i];
            if (num > highest) {
                thirdHighest = secondHighest;
                secondHighest = highest;
                highest = num;
            } else if (num != highest && num > secondHighest) {
                thirdHighest = secondHighest;
                secondHighest = num;
            } else if (num != highest && num != secondHighest && num >= thirdHighest) {
                thirdHighest = num;
            }
            thirdMax = true;
        }
        return thirdMax ? (thirdHighest != Long.MIN_VALUE ? (int) thirdHighest : (int) highest) : -1;
    }
}
