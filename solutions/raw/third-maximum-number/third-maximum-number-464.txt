class Solution {

    public int thirdMax(int[] nums) {
        int flag = 0;
        int i, max1 = Integer.MIN_VALUE, max2 = Integer.MIN_VALUE, max3 = Integer.MIN_VALUE;
        for (i = 0; i < nums.length; i++) {
            if (nums[i] > max1)
                max1 = nums[i];
        }
        for (i = 0; i < nums.length; i++) {
            if (nums[i] < max1 && nums[i] >= max2) {
                max2 = nums[i];
                flag = 1;
            }
        }
        if (max2 == Integer.MIN_VALUE && flag == 0)
            max2 = max1;
        flag = 0;
        for (i = 0; i < nums.length; i++) {
            if (nums[i] < max2 && nums[i] >= max3) {
                max3 = nums[i];
                flag = 1;
            }
        }
        if (max3 == Integer.MIN_VALUE && flag == 0)
            max3 = max1;
        if (nums.length >= 3)
            return max3;
        return max1;
    }
}
