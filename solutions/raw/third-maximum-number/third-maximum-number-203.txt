class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> al = new TreeSet<>();
        for (int i : nums) {
            al.add(i);
            if (al.size() > 3) {
                al.pollFirst();
            }
        }
        if (al.size() < 3)
            return al.pollLast();
        else
            //stored in asc order in treeset
            return al.pollFirst();
    }
}
