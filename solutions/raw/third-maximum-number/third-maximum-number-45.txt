class Solution {

    public int thirdMax(int[] a) {
        long firstMax, secondMax, thirdMax;
        firstMax = secondMax = thirdMax = (long) Integer.MIN_VALUE - 1;
        for (int i = 0; i < a.length; i++) {
            if (firstMax < a[i]) {
                thirdMax = secondMax;
                secondMax = firstMax;
                firstMax = a[i];
            } else if (secondMax < a[i] && firstMax != a[i]) {
                thirdMax = secondMax;
                secondMax = a[i];
            } else if (thirdMax < a[i] && secondMax != a[i] && firstMax != a[i]) {
                thirdMax = a[i];
            }
        }
        if (thirdMax != (long) Integer.MIN_VALUE - 1) {
            return (int) thirdMax;
        } else {
            return (int) firstMax;
        }
    }
}
