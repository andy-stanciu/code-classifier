class Solution {

    public int thirdMax(int[] nums) {
        ArrayList<Integer> ar = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (!ar.contains(nums[i])) {
                ar.add(nums[i]);
            }
        }
        if (ar.size() < 3) {
            Collections.sort(ar, Collections.reverseOrder());
            return ar.get(0);
        } else {
            Collections.sort(ar, Collections.reverseOrder());
            return ar.get(2);
        }
    }
}
