class Solution {

    public int thirdMax(int[] nums) {
        long max1 = Long.MIN_VALUE;
        long max2 = Long.MIN_VALUE;
        long max3 = Long.MIN_VALUE;
        for (int cur : nums) {
            if (cur > max1) {
                max3 = max2;
                max2 = max1;
                max1 = cur;
            } else if (cur > max2 && cur < max1) {
                max3 = max2;
                max2 = cur;
            } else if (cur > max3 && cur < max2) {
                max3 = cur;
            }
        }
        return (int) (max3 >= Integer.MIN_VALUE ? max3 : max1);
    }
}
