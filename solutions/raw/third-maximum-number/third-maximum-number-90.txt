class Solution {

    public int thirdMax(int[] nums) {
        int n = nums.length, count = 1;
        Arrays.sort(nums);
        for (int i = n - 1; i >= 1; i--) {
            if (nums[i] != nums[i - 1])
                count++;
            if (count == 3)
                return nums[i - 1];
        }
        return nums[n - 1];
    }
}
