class Solution {

    public int thirdMax(int[] nums) {
        Queue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            if (!pq.contains(nums[i])) {
                pq.add(nums[i]);
            }
            if (pq.size() > 3) {
                pq.poll();
            }
        }
        if (pq.size() == 2) {
            pq.poll();
        }
        return pq.peek();
    }
}
