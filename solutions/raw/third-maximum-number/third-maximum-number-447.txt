class Solution {

    public int thirdMax(int[] nums) {
        ArrayList<Integer> lst = new ArrayList<>();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (!lst.contains(nums[i]))
                lst.add(nums[i]);
        }
        if (lst.size() < 3)
            return lst.get(lst.size() - 1);
        return lst.get(lst.size() - 3);
    }
}
