class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int n : nums) {
            set.add(n);
            //remove the smallest element
            if (set.size() > 3)
                set.pollFirst();
        }
        //return largest element in array
        if (set.size() < 3)
            return set.last();
        //return third largest
        return set.first();
    }
}
