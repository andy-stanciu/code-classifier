class Solution {

    public int thirdMax(int[] nums) {
        // so now we have to do this solution in the o(n) here so now how we can do this here
        // so now here take the 3 integer 1 and 2 and 3 and now we have to update the max according to it
        Integer max1 = null;
        Integer max2 = null;
        Integer max3 = null;
        // so now we have to run the for loop here
        for (Integer num : nums) {
            // but here as what if the number is already present here then what
            if (num.equals(max1) || num.equals(max2) || num.equals(max3)) {
                continue;
            }
            // so now here
            if (max1 == null || num > max1) {
                // so now if the max1 is null or when the num is greater than the max1 so we can just do this
                // here
                max3 = max2;
                max2 = max1;
                max1 = num;
                // so we have just updatin the value here
            } else if (max2 == null || num > max2) {
                // so now we need to update the value
                max3 = max2;
                max2 = num;
            } else if (max3 == null || num > max3) {
                // so we have to update the value here
                max3 = num;
            }
        }
        // so now here we have to get the value here
        // so agar max3 ka value null hai so hame to max 1 return karna chaiye
        return max3 == null ? max1 : max3;
    }
}
