class Solution {

    public int thirdMax(int[] nums) {
        int first, second, third;
        first = second = third = Integer.MIN_VALUE;
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        Iterator it = set.iterator();
        while (it.hasNext()) {
            int item = (int) it.next();
            if (item > first) {
                if (second != Integer.MIN_VALUE) {
                    third = second;
                    second = first;
                    first = item;
                } else {
                    second = first;
                    first = item;
                }
            } else if (item > second) {
                third = second;
                second = item;
            } else if (item > third) {
                third = item;
            }
        }
        if (set.size() < 3)
            return first;
        return third;
    }
}
