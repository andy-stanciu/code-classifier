class Solution {

    public int thirdMax(int[] nums) {
        int idx = 0;
        Arrays.sort(nums);
        for (int i = 1; i < nums.length; ++i) {
            if (nums[i] != nums[idx]) {
                nums[++idx] = nums[i];
            }
        }
        if (idx < 2) {
            return nums[nums.length - 1];
        }
        return nums[idx - 2];
    }
}
