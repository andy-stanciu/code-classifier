class Solution {

    public int thirdMax(int[] nums) {
        int max1 = Integer.MIN_VALUE, max2 = Integer.MIN_VALUE, max3 = Integer.MIN_VALUE, flag = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > max1) {
                max1 = nums[i];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < max1 && nums[i] > max2) {
                max2 = nums[i];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < max2 && nums[i] >= max3) {
                flag = 1;
                max3 = nums[i];
            }
        }
        if (flag == 0) {
            return max1;
        }
        return max3;
    }
}
