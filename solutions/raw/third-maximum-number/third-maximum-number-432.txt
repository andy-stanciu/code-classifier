class Solution {

    public int thirdMax(int[] nums) {
        double secondMax = 0.5678;
        double thirdMax = 0.5678;
        double firstMax = 0.5678;
        for (int i = 0; i < nums.length; i++) {
            int num = nums[i];
            if (num == firstMax || num == secondMax || num == thirdMax) {
                // Skip duplicates
                continue;
            }
            if (firstMax == 0.5678 || num > firstMax) {
                thirdMax = secondMax;
                secondMax = firstMax;
                firstMax = num;
            } else if (secondMax == 0.5678 || num > secondMax) {
                thirdMax = secondMax;
                secondMax = num;
            } else if (thirdMax == 0.5678 || num > thirdMax) {
                thirdMax = num;
            }
        }
        if (thirdMax != 0.5678) {
            return (int) thirdMax;
        } else {
            return (int) firstMax;
        }
    }
}
