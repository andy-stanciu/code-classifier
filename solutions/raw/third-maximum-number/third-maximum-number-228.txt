class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        Vector<Integer> s = new Vector<>();
        for (int i = 0; i < nums.length; i++) {
            if (!(s.contains(nums[i]))) {
                s.add(nums[i]);
            }
        }
        if (s.size() == 1) {
            return (s.get(0));
        }
        if (s.size() == 2) {
            return (s.get(1));
        }
        if (s.size() == 3) {
            return (s.get(0));
        }
        return s.get(s.size() - 3);
    }
}
