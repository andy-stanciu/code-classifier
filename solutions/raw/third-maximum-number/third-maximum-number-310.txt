class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 2;
        if (nums.length < 3) {
            return (nums[nums.length - 1]);
        }
        int temp = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < temp) {
                if (count == 0) {
                    break;
                } else {
                    temp = nums[i];
                    count--;
                }
            }
        }
        if (count == 0) {
            return temp;
        } else {
            return nums[nums.length - 1];
        }
    }
}
