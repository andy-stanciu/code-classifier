class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int x = nums.length;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                x--;
            }
        }
        int[] arr = new int[x];
        for (int i = 0, j = 0; i < nums.length - 1; i++) {
            if (nums[i] < nums[i + 1]) {
                arr[j] = nums[i];
                j++;
            }
        }
        arr[x - 1] = nums[nums.length - 1];
        if (x < 3) {
            return arr[arr.length - 1];
        } else {
            int y = arr[x - 3];
            return y;
        }
    }
}
