class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> obj = new TreeSet<>();
        for (int i = 0; i < nums.length; i++) {
            obj.add(nums[i]);
        }
        Object[] arr = obj.toArray();
        if (arr.length >= 3) {
            return (int) arr[arr.length - 3];
        } else {
            return (int) arr[arr.length - 1];
        }
    }
}
