class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        // Iterate over the array and add elements to the set
        for (int num : nums) {
            set.add(num);
            // Ensure the set contains at most 3 elements
            if (set.size() > 3) {
                // Remove the smallest element if size exceeds 3
                set.pollFirst();
            }
        }
        // If the set contains 3 elements, the third maximum is the smallest in the set
        // Otherwise, the maximum is the largest element in the set
        return set.size() == 3 ? set.first() : set.last();
    }
}
