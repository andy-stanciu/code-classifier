class Solution {

    public int thirdMax(int[] nums) {
        // Use a HashSet to store unique elements
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        // Use a PriorityQueue (min-heap) to track the top 3 maximum numbers
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int num : set) {
            pq.add(num);
            if (pq.size() > 3) {
                // Remove the smallest element if size exceeds 3
                pq.poll();
            }
        }
        // If there are less than 3 unique elements, return the maximum
        if (pq.size() < 3) {
            while (pq.size() > 1) {
                pq.poll();
            }
        }
        return pq.peek();
    }
}
