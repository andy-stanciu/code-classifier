class Solution {

    public int thirdMax(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int var : nums) {
            set.add(var);
        }
        int n = set.size();
        int[] arr = new int[n];
        int idx = 0;
        for (int var : set) {
            arr[idx] = var;
            idx++;
        }
        Arrays.sort(arr);
        if (n < 3) {
            return arr[n - 1];
        }
        return arr[n - 3];
    }
}
