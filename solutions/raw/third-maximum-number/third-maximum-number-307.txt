class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> hs = new HashSet<>();
        for (Integer x : nums) hs.add(x);
        Integer[] arr = hs.toArray(new Integer[0]);
        Arrays.sort(arr);
        if (arr.length >= 3)
            return arr[arr.length - 3];
        return arr[arr.length - 1];
    }
}
