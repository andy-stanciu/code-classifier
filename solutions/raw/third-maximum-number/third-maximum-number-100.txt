class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 1;
        int max = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            //if distinct value the count will increase and max is set to the current distinct value
            if (nums[i] != max) {
                max = nums[i];
                count++;
            }
            //if we get third distinct value we directly return it
            if (count == 3)
                return nums[i];
        }
        //if array did not contains three distinct values return max value
        return nums[nums.length - 1];
    }
}
