class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        //first we sort the array, then initialize
        int fMax = nums[nums.length - 1];
        //fMax to the greatest number and sMax and tMax
        int sMax = 100000000;
        //to very large integers
        int tMax = 100000000;
        int i = nums.length - 1;
        while (i > 0 && sMax == 100000000) {
            //for finding out second max
            if (nums[i - 1] == nums[i])
                i--;
            else {
                sMax = nums[i - 1];
                i--;
            }
        }
        while (i > 0 && tMax == 100000000) {
            //for finding out third max
            if (nums[i - 1] == nums[i])
                i--;
            else {
                tMax = nums[i - 1];
                i--;
            }
        }
        if (fMax > sMax) {
            //if there is a possible third max then
            //return it
            if (sMax > tMax)
                return tMax;
        }
        //otherwise return fMax
        return fMax;
    }
}
