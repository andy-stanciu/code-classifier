class Solution {

    public int thirdMax(int[] nums) {
        int firstMax = Integer.MIN_VALUE;
        int secondMax = Integer.MIN_VALUE;
        int thirdMax = Integer.MIN_VALUE;
        int numChanges = 0;
        boolean isMinIntPresent = false;
        for (int num : nums) {
            if (num > firstMax) {
                thirdMax = secondMax;
                secondMax = firstMax;
                firstMax = num;
                numChanges++;
            } else if (num > secondMax && num < firstMax) {
                thirdMax = secondMax;
                secondMax = num;
                numChanges++;
            } else if (num > thirdMax && num < secondMax) {
                thirdMax = num;
                numChanges++;
            } else if (!isMinIntPresent && num == Integer.MIN_VALUE && thirdMax == Integer.MIN_VALUE) {
                isMinIntPresent = true;
                numChanges++;
            }
        }
        int returnValue = thirdMax;
        if (numChanges < 3) {
            returnValue = firstMax;
        }
        return returnValue;
    }
}
