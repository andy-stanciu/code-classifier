class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> s = new HashSet<>();
        Arrays.sort(nums);
        for (int i = nums.length - 1; i >= 0; i--) {
            s.add(nums[i]);
            if (s.size() == 3)
                return nums[i];
        }
        return nums[nums.length - 1];
    }
}
