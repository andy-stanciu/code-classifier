class Solution {

    public int thirdMax(int[] nums) {
        // create a arrayslist
        ArrayList<Integer> list = new ArrayList<>();
        for (// Iterate the nums of the element
        // Iterate the nums of the element
        // Iterate the nums of the element
        int i = 0; i < nums.length; i++) {
            if (// cheack the element is not alredy present in the list
            !list.contains(nums[i]))
                // add those the element
                list.add(nums[i]);
        }
        // sort the given the array
        Collections.sort(list);
        // and then print
        System.out.println(list);
        if (// the size is less the 3 return the last element of the arrays
        list.size() < 3)
            return list.get(list.size() - 1);
        else
            // If the size is 3 or more, return the third-to-last element in the list
            return list.get(list.size() - 3);
    }
