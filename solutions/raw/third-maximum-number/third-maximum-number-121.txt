class Solution {

    public int thirdMax(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int ele : nums) {
            set.add(ele);
        }
        int[] ans = new int[set.size()];
        int i = 0;
        for (int ele : set) {
            ans[i++] = ele;
        }
        Arrays.sort(ans);
        int n = ans.length;
        if (n >= 3)
            return ans[n - 3];
        return ans[n - 1];
    }
}
