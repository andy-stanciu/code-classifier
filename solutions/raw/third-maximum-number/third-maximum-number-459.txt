class Solution {

    public int thirdMax(int[] nums) {
        long a = Long.MIN_VALUE, b = a, c = a;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > a) {
                c = b;
                b = a;
                a = nums[i];
            } else if (nums[i] != a && nums[i] > b) {
                c = b;
                b = nums[i];
            } else if (nums[i] != a && nums[i] != b && nums[i] > c)
                c = nums[i];
        }
        int ans = (c == Long.MIN_VALUE) ? (int) a : (int) c;
        return ans;
    }
}
