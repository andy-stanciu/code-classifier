class Solution {

    public int thirdMax(int[] nums) {
        // Sort the array in ascending order
        Arrays.sort(nums);
        // Initialize variables
        int min = Integer.MAX_VALUE + 1;
        int i = nums.length - 1;
        int count = 0;
        // Iterate over the sorted array from the end
        while (i >= 0) {
            // Check if the current element is different from the previous one
            if (min != nums[i]) {
                // Increment the count of unique elements
                count++;
                // Check if we have found the third maximum
                if (count == 3) {
                    // Return the third maximum
                    return nums[i];
                }
                // Update the minimum value encountered so far
                min = nums[i];
            }
            // Move to the previous element in the sorted array
            i--;
        }
        // If there are less than three unique elements, return the maximum value
        return nums[nums.length - 1];
    }
}
