class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 1, max = nums[nums.length - 1], thirdmax = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (count < 3 && nums[i] != nums[i - 1]) {
                thirdmax = nums[i - 1];
                count++;
            }
        }
        return (count == 3) ? thirdmax : max;
    }
}
