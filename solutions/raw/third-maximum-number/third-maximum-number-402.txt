class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        int result = 0;
        int k = (set.size() >= 3) ? 3 : 1;
        for (int i = 0; i < k; ++i) {
            result = Integer.MIN_VALUE;
            for (int val : set) {
                result = (val > result) ? val : result;
            }
            set.remove(result);
        }
        return result;
    }
}
