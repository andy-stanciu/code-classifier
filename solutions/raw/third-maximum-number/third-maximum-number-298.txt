class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 1;
        int max = nums[nums.length - 1];
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] == max) {
                continue;
            } else if (count == 3) {
                break;
            } else {
                max = nums[i];
                count++;
            }
        }
        return (count == 3) ? max : nums[nums.length - 1];
    }
}
