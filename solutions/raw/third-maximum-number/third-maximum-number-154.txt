class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int c = 0;
        int n = nums.length;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                //iff dono equals nhi hue to count ko increse kr denge
                c++;
            }
            //jab count 2 huaa to uske next wale number ko print kra denge
            if (c == 2) {
                return nums[i - 1];
            }
        }
        return nums[n - 1];
    }
}
