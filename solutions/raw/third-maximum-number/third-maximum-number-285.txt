class Solution {

    public int thirdMax(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        int index = 0;
        int[] a = new int[set.size()];
        for (int i : set) {
            a[index++] = i;
        }
        Arrays.sort(a);
        if (a.length < 3) {
            return a[a.length - 1];
        }
        return a[a.length - 3];
    }
}
