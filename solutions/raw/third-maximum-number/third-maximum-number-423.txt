class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        HashSet<Integer> hs = new HashSet<>();
        if (nums.length < 3)
            return nums[nums.length - 1];
        for (int i = nums.length - 1; i >= 0; i--) {
            hs.add(nums[i]);
            if (hs.size() == 3)
                return nums[i];
        }
        return nums[nums.length - 1];
    }
}
