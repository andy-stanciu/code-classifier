class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i : nums) set.add(i);
        int a[] = new int[set.size()], c = 0;
        for (int i : set) a[c++] = i;
        Arrays.sort(a);
        int n = a.length;
        if (n == 1)
            return a[0];
        if (n == 2)
            return a[1];
        if (n == 3)
            return a[0];
        else
            return a[n - 3];
    }
}
