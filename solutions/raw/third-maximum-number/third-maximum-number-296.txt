class Solution {

    public int thirdMax(int[] nums) {
        List<Integer> li = new ArrayList<>();
        for (int x : nums) {
            if (!li.contains(x))
                li.add(x);
        }
        Collections.sort(li);
        return li.size() < 3 ? (li.size() == 2 ? Math.max(li.get(0), li.get(1)) : li.get(0)) : li.get(li.size() - 3);
    }
}
