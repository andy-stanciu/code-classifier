class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> se = new HashSet<>();
        for (int num : nums) {
            se.add(num);
        }
        List<Integer> list = new LinkedList<>(se);
        Collections.sort(list, Collections.reverseOrder());
        if (list.size() >= 3) {
            return list.get(2);
        } else {
            return list.get(0);
        }
    }
}
