class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        for (int i = 0; i < nums.length / 2; i++) {
            int a = nums[i];
            nums[i] = nums[nums.length - 1 - i];
            nums[nums.length - 1 - i] = a;
        }
        int[] arr = new int[3];
        int y = 0;
        boolean b = false;
        for (int i = 0; i < nums.length; i++) {
            arr[y] = nums[i];
            y++;
            while (i + 1 < nums.length && nums[i] == nums[i + 1]) {
                i++;
            }
            if (y == 3) {
                b = true;
                break;
            }
        }
        if (b)
            return arr[2];
        else
            return arr[0];
    }
}
