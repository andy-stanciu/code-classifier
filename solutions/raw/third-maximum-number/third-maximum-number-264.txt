class Solution {

    public int thirdMax(int[] nums) {
        long a = Long.MIN_VALUE;
        long b = Long.MIN_VALUE;
        long c = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int val = nums[i];
            if (val > a) {
                c = b;
                b = a;
                a = val;
            } else if (val > b && val < a) {
                c = b;
                b = val;
            } else if (val > c && val < b) {
                c = val;
            }
        }
        return (c == Long.MIN_VALUE) ? (int) a : (int) c;
    }
}
