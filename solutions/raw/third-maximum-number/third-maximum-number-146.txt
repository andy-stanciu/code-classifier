class Solution {

    public int thirdMax(int[] nums) {
        HashSet<Integer> hm = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            hm.add(nums[i]);
        }
        int[] arr = new int[hm.size()];
        int x = 0;
        for (int i : hm) {
            arr[x++] = i;
        }
        Arrays.sort(arr);
        if (arr.length < 3)
            return arr[arr.length - 1];
        return arr[arr.length - 3];
    }
}
