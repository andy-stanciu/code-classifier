class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> s = new HashSet<>();
        List<Integer> list = new ArrayList<>();
        for (Integer i : nums) {
            if (s.contains(i)) {
                continue;
            }
            list.add(i);
            s.add(i);
        }
        Collections.sort(list);
        if (list.size() < 3) {
            return list.get(list.size() - 1);
        }
        return list.get(list.size() - 3);
    }
}
