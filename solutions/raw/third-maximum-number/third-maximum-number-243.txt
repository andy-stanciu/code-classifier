class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> sortedNums = new TreeSet<Integer>();
        for (int i : nums) sortedNums.add(i);
        if (sortedNums.size() >= 3)
            for (int i = 0; i < 2; i++) sortedNums.pollLast();
        return sortedNums.last();
    }
}
