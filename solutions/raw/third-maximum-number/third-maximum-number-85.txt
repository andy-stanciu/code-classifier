class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 1;
        int n = nums.length - 1;
        int pre = nums[n];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] != pre)
                count += 1;
            pre = nums[i];
            if (count == 3) {
                return pre;
            }
        }
        return nums[nums.length - 1];
    }
}
