class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int i, max = nums[0], c = 0;
        for (i = nums.length - 1; i >= 0; i--) {
            if (nums[i] != max) {
                c++;
                max = nums[i];
            }
            if (c == 3) {
                return max;
            }
        }
        return nums[nums.length - 1];
    }
}
