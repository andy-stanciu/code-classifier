class Solution {

    public int thirdMax(int[] nums) {
        Set<Integer> numbers = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            numbers.add(nums[i]);
        }
        List<Integer> distinctNumbers = new ArrayList<Integer>(numbers);
        Collections.sort(distinctNumbers, Collections.reverseOrder());
        int lengthOfList = distinctNumbers.size();
        if (lengthOfList <= 2) {
            Integer max = distinctNumbers.get(0);
            int maxEle = Integer.valueOf(max);
            return maxEle;
        }
        Integer thirdMax = distinctNumbers.get(2);
        int thirdMaxRet = Integer.valueOf(thirdMax);
        return thirdMaxRet;
    }
}
