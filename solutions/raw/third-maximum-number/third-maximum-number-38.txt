class Solution {

    public int thirdMax(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>(Comparator.reverseOrder());
        for (int n : nums) {
            set.add(n);
        }
        if (set.size() < 3) {
            return set.first();
        }
        set.remove(set.first());
        set.remove(set.first());
        return set.first();
    }
}
