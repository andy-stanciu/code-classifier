class Solution {

    public int thirdMax(int[] nums) {
        int a = -2147483648;
        int b = -2147483648;
        int c = -2147483648;
        boolean x = true;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] >= a) {
                a = nums[i];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < a && nums[i] >= b) {
                b = nums[i];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < a && nums[i] < b && nums[i] >= c) {
                c = nums[i];
                x = false;
            }
        }
        if (c == -2147483648 && x) {
            Arrays.sort(nums);
            return nums[nums.length - 1];
        }
        return c;
    }
}
