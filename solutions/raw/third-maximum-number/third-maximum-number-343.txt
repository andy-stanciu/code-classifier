class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int count = 1;
        int min = nums[nums.length - 1];
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] < min) {
                min = nums[i];
                count++;
            }
            if (count == 3) {
                return min;
            }
        }
        return nums[n - 1];
    }
}
