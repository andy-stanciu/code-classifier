class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int n = nums[nums.length - 1];
        int count = 1;
        int max = n;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] != n) {
                count++;
                n = nums[i];
                max = Math.max(n, max);
                if (count >= 3) {
                    break;
                }
            }
        }
        return (count < 3) ? max : n;
    }
}
