class Solution {

    public int thirdMax(int[] nums) {
        int[] arr = Arrays.stream(nums).distinct().toArray();
        Arrays.sort(arr);
        int n = arr.length;
        if (n == 3) {
            return arr[0];
        } else if (n < 3) {
            return arr[n - 1];
        }
        return arr[n - 3];
    }
}
