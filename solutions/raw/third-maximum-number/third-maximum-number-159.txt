class Solution {

    public int thirdMax(int[] nums) {
        Arrays.sort(nums);
        int count = 2;
        int largest = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (count == 1 && nums[i] < nums[i + 1]) {
                return nums[i];
            }
            if (nums[i + 1] > nums[i]) {
                count = count - 1;
                continue;
            }
        }
        return nums[nums.length - 1];
    }
}
