class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int closeSum = nums[0] + nums[1] + nums[nums.length - 1];
        int sum = 0;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                sum = nums[i] + nums[j] + nums[k];
                if (Math.abs(sum - target) < Math.abs(closeSum - target)) {
                    closeSum = sum;
                }
                if (sum == target)
                    return sum;
                else if (sum < target)
                    j++;
                else
                    k--;
            }
        }
        return closeSum;
    }
}
