class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        if (nums.length == 1)
            return nums[0];
        int maxDiff = Integer.MAX_VALUE, sum, ans = 0;
        for (int i = 0; i < nums.length; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                sum = nums[i] + nums[j] + nums[k];
                int diff = Math.abs(target - sum);
                if (diff < maxDiff) {
                    maxDiff = diff;
                    ans = sum;
                }
                if (sum < target)
                    j++;
                else
                    k--;
            }
        }
        return ans;
    }
}
