class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int numsLength = nums.length;
        Integer closest = null;
        for (int i = 0; i < numsLength; i++) {
            int currentNum = nums[i];
            int startCursor = i + 1;
            int endCursor = numsLength - 1;
            while (startCursor < endCursor) {
                int sum = currentNum + nums[startCursor] + nums[endCursor];
                if (closest == null) {
                    closest = sum;
                } else {
                    int distanceWithSum = Math.abs(target - sum);
                    int distanceWithClosest = Math.abs(target - closest);
                    if (distanceWithSum < distanceWithClosest) {
                        closest = sum;
                    }
                }
                if (sum > target) {
                    endCursor--;
                } else if (sum < target) {
                    startCursor++;
                } else {
                    startCursor++;
                    endCursor--;
                }
            }
        }
        return closest;
    }
}
