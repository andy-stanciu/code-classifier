class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int diff = Integer.MAX_VALUE;
        int ans = 0;
        for (int i = 0; i < nums.length; i++) {
            int f = nums[i];
            int s = i + 1;
            int e = nums.length - 1;
            while (s < e) {
                int sum = f + nums[s] + nums[e];
                if (sum == target)
                    return target;
                else if (Math.abs(sum - target) < diff) {
                    diff = Math.abs(sum - target);
                    ans = sum;
                }
                if (sum > target)
                    e--;
                else
                    s++;
            }
        }
        return ans;
    }
}
