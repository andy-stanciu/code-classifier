class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length, ans = Integer.MAX_VALUE, op = 0;
        for (int i = 0; i < n - 2; ++i) {
            int j = i + 1, k = n - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (Math.abs(sum - target) < ans) {
                    ans = Math.abs(sum - target);
                    op = sum;
                    if (sum == target)
                        return op;
                }
                if (sum > target)
                    k--;
                else
                    j++;
            }
        }
        return op;
    }
}
