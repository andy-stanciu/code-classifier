class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int n = nums.length;
        int res = 0;
        int gap = Integer.MAX_VALUE;
        Arrays.sort(nums);
        for (int i = 0; i < n - 2; i++) {
            int left = i + 1;
            int right = n - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum < target) {
                    int ngap = target - sum;
                    if (ngap < gap) {
                        gap = ngap;
                        res = sum;
                    }
                    left++;
                } else if (sum > target) {
                    int ngap = sum - target;
                    if (ngap < gap) {
                        gap = ngap;
                        res = sum;
                    }
                    right--;
                } else {
                    return sum;
                }
            }
        }
        return res;
    }
}
