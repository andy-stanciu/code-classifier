class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int closest = Integer.MAX_VALUE;
        //use three pointers & find the closest no to the target
        for (int i = 0; i < n - 2; i++) {
            int start = i + 1;
            int end = n - 1;
            while (start < end) {
                int sum = nums[i] + nums[start] + nums[end];
                if (sum == target)
                    return sum;
                if (Math.abs(sum - target) < Math.abs(closest - target))
                    closest = sum;
                if (sum < target) {
                    start++;
                } else {
                    end--;
                }
            }
        }
        return closest;
    }
}
