class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int closest_sum = Integer.MAX_VALUE;
        int result = 0;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i += 1) {
            int low = i + 1, high = nums.length - 1;
            while (low < high) {
                int sum = nums[i] + nums[low] + nums[high];
                if (Math.abs(sum - target) <= closest_sum) {
                    closest_sum = Math.abs(sum - target);
                    result = sum;
                }
                if (sum < target) {
                    low += 1;
                } else {
                    high -= 1;
                }
            }
        }
        return result;
    }
}
