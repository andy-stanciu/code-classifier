class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int sum = 0;
        int closestSum = nums[0] + nums[1] + nums[nums.length - 1];
        // Setting intial value of our smallest difference
        int smallestdiff = Math.abs(closestSum - target);
        for (int i = 0; i < nums.length - 2; i++) {
            int startindex = i + 1;
            int endindex = nums.length - 1;
            while (startindex < endindex) {
                sum = nums[i] + nums[startindex] + nums[endindex];
                int diff = Math.abs(target - sum);
                if (//checking whether the current difference is the smallest
                diff <= smallestdiff) {
                    closestSum = sum;
                    smallestdiff = diff;
                }
                if (sum < target) {
                    startindex++;
                } else
                    endindex--;
            }
            //end of while loop
        }
        // end of for loop
        return closestSum;
    }
}
