class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int Sum = 0;
        int result = 0;
        int CloseSum = 0;
        int diff = 0;
        int ReferenceSum = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                Sum = nums[i] + nums[j] + nums[k];
                diff = target - Sum;
                CloseSum = target - ReferenceSum;
                if (diff < 0) {
                    diff = -(diff);
                }
                if (CloseSum < 0) {
                    CloseSum = -(CloseSum);
                }
                if (diff < CloseSum) {
                    ReferenceSum = Sum;
                }
                if (Sum < target) {
                    j++;
                } else {
                    k--;
                }
            }
        }
        return ReferenceSum;
    }
}
