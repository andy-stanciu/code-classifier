class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int res = 0;
        int diff = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int start = i + 1, end = nums.length - 1;
            int req = target - nums[i];
            while (start < end) {
                int currSum = nums[start] + nums[end];
                if (Math.abs(currSum + nums[i] - target) < diff) {
                    diff = Math.abs(currSum + nums[i] - target);
                    res = currSum + nums[i];
                }
                if (req == currSum) {
                    return res;
                } else if (req < currSum) {
                    end--;
                } else {
                    start++;
                }
            }
        }
        return res;
    }
}
