class //Algo -- One element fix and rest two element using BS + hurastics
Solution {

    public int threeSumClosest(int[] nums, int target) {
        //To reduce the problem
        Arrays.sort(nums);
        //Diff Ensure problem is reducing(converging) //Hurastic For BS
        int diff = Integer.MAX_VALUE;
        int ansSoFar = 0;
        for (int i = 0; i < nums.length; i++) {
            //Fixing First No
            int first = nums[i];
            int start = i + 1;
            int end = nums.length - 1;
            while (start < end) {
                //BS
                if ((first + nums[start] + nums[end]) == target)
                    return target;
                else if (Math.abs((first + nums[start] + nums[end]) - target) < diff) {
                    diff = Math.abs((first + nums[start] + nums[end]) - target);
                    ansSoFar = first + nums[start] + nums[end];
                }
                //Trying to pick the right  choice
                if (first + nums[start] + nums[end] > target)
                    end--;
                else
                    start++;
            }
            //Not a valid position selection
            continue;
        }
        return ansSoFar;
    }
}
