class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int N = nums.length;
        int historicalClose = Integer.MAX_VALUE, closestSum = Integer.MAX_VALUE;
        for (int i = 0; i < N - 2; i++) {
            for (int j = i + 1; j < N - 1; j++) {
                for (int k = j + 1; k < N; k++) {
                    int v1 = nums[i], v2 = nums[j], v3 = nums[k];
                    int currentSum = v1 + v2 + v3;
                    int currentDistance = Math.abs(target - currentSum);
                    if (currentDistance < historicalClose) {
                        historicalClose = currentDistance;
                        closestSum = currentSum;
                    }
                }
            }
        }
        return closestSum;
    }
}
