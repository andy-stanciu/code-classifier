class Solution {

    static int threeSumClosest(int[] array, int target) {
        int n = array.length;
        int result = 0;
        int minDiff = Integer.MAX_VALUE;
        int sum, diff;
        Arrays.sort(array);
        for (int i = 0; i < n; i++) {
            int left = i + 1, right = n - 1;
            while (left < right) {
                sum = array[i] + array[left] + array[right];
                diff = Math.abs(target - sum);
                if (minDiff > diff) {
                    minDiff = diff;
                    result = sum;
                }
                if (sum < target) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return result;
    }
}
