class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int lastClosestDifference = Integer.MAX_VALUE;
        int lastClosestSum = 0;
        Arrays.sort(nums);
        int length = nums.length;
        for (int i = 0; i < length - 2; i++) {
            int left = i + 1;
            int right = length - 1;
            while (left < length && right < length && left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target) {
                    return sum;
                } else if (sum < target) {
                    int differnce = target - sum;
                    if (differnce < lastClosestDifference) {
                        lastClosestSum = sum;
                        lastClosestDifference = differnce;
                    }
                    left++;
                } else {
                    int differnce = sum - target;
                    if (differnce < lastClosestDifference) {
                        lastClosestSum = sum;
                        lastClosestDifference = differnce;
                    }
                    right--;
                }
            }
        }
        return lastClosestSum;
    }
}
