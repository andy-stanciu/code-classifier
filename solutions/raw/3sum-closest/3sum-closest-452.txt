class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int result = 0;
        int difference = Integer.MAX_VALUE;
        for (int middle = 1; middle < nums.length; middle++) {
            int start = 0;
            int end = nums.length - 1;
            int preDifference = Integer.MAX_VALUE;
            while (start < middle && middle < end) {
                int total = nums[start] + nums[middle] + nums[end];
                int currDifference = Math.abs(target - total);
                if (currDifference > preDifference) {
                    break;
                }
                if (total > target) {
                    end--;
                } else {
                    start++;
                }
                if (difference > currDifference) {
                    difference = currDifference;
                    result = total;
                }
                preDifference = currDifference;
            }
        }
        return result;
    }
}
