class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int bestDelta = nums[0] + nums[1] + nums[2] - target;
        int bestDeltaAbs = Math.abs(bestDelta);
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            int remainingDelta = nums[i] - target;
            while (j < k) {
                int delta = remainingDelta + nums[j] + nums[k];
                if (delta < 0) {
                    // Increase delta
                    j++;
                    if (delta > -bestDeltaAbs) {
                        bestDelta = delta;
                        bestDeltaAbs = Math.abs(delta);
                    }
                } else {
                    // Decrease delta
                    k--;
                    if (delta < bestDeltaAbs) {
                        bestDelta = delta;
                        bestDeltaAbs = Math.abs(delta);
                    }
                }
            }
        }
        return bestDelta + target;
    }
}
