class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int size = nums.length;
        int closest = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < size; i++) {
            int left = i + 1;
            int right = size - 1;
            int sum = 0;
            while (left < right) {
                sum = nums[i] + nums[left] + nums[right];
                if (Math.abs(target - sum) < Math.abs(target - closest)) {
                    closest = sum;
                } else if (sum > target)
                    right--;
                else
                    left++;
            }
        }
        return closest;
    }
}
