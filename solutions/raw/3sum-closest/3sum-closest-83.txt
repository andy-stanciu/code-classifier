class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int minDist = Integer.MAX_VALUE;
        int minSum = 0;
        for (int i = 0; i < nums.length; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                int dist = Math.abs(target - sum);
                if (dist <= minDist) {
                    minSum = sum;
                    minDist = dist;
                }
                if (sum > target) {
                    right--;
                } else if (sum < target) {
                    left++;
                } else {
                    break;
                }
            }
        }
        return minSum;
    }
}
