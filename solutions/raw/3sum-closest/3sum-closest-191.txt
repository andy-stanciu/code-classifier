class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int minDiff = Integer.MAX_VALUE;
        int result = 0;
        int length = nums.length;
        for (int i = 0, iLength = length - 2; i < iLength; i++) {
            for (int j = i + 1, jLength = length - 1; j < jLength; j++) {
                for (int k = j + 1, kLength = length; k < kLength; k++) {
                    long sum = (long) nums[i] + (long) nums[j] + (long) nums[k];
                    if (sum > Integer.MAX_VALUE | sum < Integer.MIN_VALUE) {
                        continue;
                    }
                    int diff = Math.abs(target - (int) sum);
                    if (diff < minDiff) {
                        minDiff = diff;
                        result = (int) sum;
                    }
                }
            }
        }
        return result;
    }
}
