class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int rs = 0;
        int dif = Integer.MAX_VALUE;
        for (int i = 0; i < n - 2; i++) {
            int left = i + 1;
            int right = n - 1;
            while (left < right) {
                int curSum = nums[i] + nums[left] + nums[right];
                int curDif;
                if (curSum == target) {
                    return target;
                } else if (curSum > target) {
                    curDif = Math.abs(curSum - target);
                    if (curDif < dif) {
                        rs = curSum;
                        dif = curDif;
                    }
                    right--;
                } else {
                    curDif = Math.abs(curSum - target);
                    if (curDif < dif) {
                        rs = curSum;
                        dif = curDif;
                    }
                    left++;
                }
            }
        }
        return rs;
    }
}
