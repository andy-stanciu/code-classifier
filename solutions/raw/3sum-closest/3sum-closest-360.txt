class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int delta = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 2; i++) {
            //choose the first ele
            int sum = nums[i] + twoMin(nums, i + 1, target - nums[i]);
            if (Math.abs(delta) > Math.abs(target - sum)) {
                delta = target - sum;
            }
        }
        return target - delta;
    }

    public int twoMin(int[] nums, int index, int target) {
        int left = index, right = nums.length - 1;
        int delta = Integer.MAX_VALUE;
        while (left < right) {
            int sum = nums[left] + nums[right];
            if (Math.abs(delta) > Math.abs(target - sum)) {
                delta = target - sum;
            }
            if (sum < target) {
                left++;
            } else {
                right--;
            }
        }
        return target - delta;
    }
}
