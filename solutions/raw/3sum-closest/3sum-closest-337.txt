class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int closestSum = Integer.MAX_VALUE;
        int diff = Math.abs(target - closestSum);
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                int newDiff = target - sum;
                if (newDiff == 0)
                    return sum;
                else if (newDiff > 0)
                    j++;
                else {
                    k--;
                    newDiff *= -1;
                }
                if (newDiff < diff) {
                    closestSum = sum;
                    diff = newDiff;
                }
            }
        }
        return closestSum;
    }
}
