class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int n = nums.length;
        int closestsum = 10000000;
        Arrays.sort(nums);
        for (int k = 0; k <= n; k++) {
            int i = k + 1;
            int j = n - 1;
            while (i < j) {
                int sum = nums[k] + nums[i] + nums[j];
                if (Math.abs(target - sum) < Math.abs(target - closestsum))
                    closestsum = sum;
                if (sum > target) {
                    j--;
                } else {
                    i++;
                }
            }
        }
        return closestsum;
    }
}
