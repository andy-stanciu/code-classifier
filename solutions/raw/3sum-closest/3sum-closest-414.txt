class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int left, sum, right, min = Integer.MAX_VALUE, temp, ans = 0;
        for (int i = 0; i <= nums.length - 3; i++) {
            left = i + 1;
            right = nums.length - 1;
            while (left < right) {
                sum = nums[i] + nums[left] + nums[right];
                temp = min;
                min = Math.min(Math.abs(target - sum), min);
                if (temp != min) {
                    temp = min;
                    ans = sum;
                }
                if (sum == target) {
                    return sum;
                } else if (sum > target)
                    right--;
                else
                    left++;
            }
        }
        return ans;
    }
}
