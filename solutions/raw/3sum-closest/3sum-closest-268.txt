class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int closestGap = Integer.MAX_VALUE;
        int ans = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (sum < target) {
                    int gap = target - sum;
                    if (gap < closestGap) {
                        closestGap = gap;
                        ans = sum;
                    }
                    j++;
                } else if (sum > target) {
                    int gap = sum - target;
                    if (gap < closestGap) {
                        closestGap = gap;
                        ans = sum;
                    }
                    k--;
                } else {
                    return sum;
                }
            }
        }
        return ans;
    }
}
