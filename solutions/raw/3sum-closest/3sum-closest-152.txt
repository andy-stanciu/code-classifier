class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        // Initialize to a large number
        int closestSum = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                // Absolute difference
                int difference = Math.abs(target - sum);
                // Update the closest sum if the current difference is smaller
                if (difference < Math.abs(target - closestSum)) {
                    closestSum = sum;
                }
                // Move pointers
                if (sum < target) {
                    left++;
                } else if (sum > target) {
                    right--;
                } else {
                    // If sum equals target, we\'ve found the closest possible sum
                    return sum;
                }
            }
        }
        return closestSum;
    }
}
