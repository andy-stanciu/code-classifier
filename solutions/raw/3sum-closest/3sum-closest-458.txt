class Solution {

    public int threeSumClosest(int[] nums, int target) {
        int close = nums[0] + nums[1] + nums[2], i, j, k, len = nums.length, sum = 0;
        Arrays.sort(nums);
        for (i = 0; i < len - 2; i++) {
            j = i + 1;
            k = len - 1;
            while (j < k) {
                sum = nums[i] + nums[j] + nums[k];
                if (Math.abs(sum - target) < Math.abs(close - target))
                    close = sum;
                else if (sum > target)
                    k--;
                else
                    j++;
            }
        }
        return close;
    }
}
