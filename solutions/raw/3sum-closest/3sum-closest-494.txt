class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int ans = 0;
        int min_diff = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                int diff = Math.abs(sum - target);
                if (diff <= min_diff) {
                    min_diff = diff;
                    ans = sum;
                }
                if (sum < target) {
                    j++;
                } else if (sum == target)
                    return target;
                else {
                    k--;
                }
            }
        }
        return ans;
    }
}
