class Solution {

    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int near = 100000;
        int ans = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            // low
            int l = i + 1;
            // high
            int h = nums.length - 1;
            while (l < h) {
                if (near > Math.abs(nums[i] + nums[l] + nums[h] - target)) {
                    near = Math.abs(nums[i] + nums[l] + nums[h] - target);
                    ans = nums[i] + nums[l] + nums[h];
                }
                if (nums[i] + nums[l] + nums[h] > target) {
                    h--;
                } else {
                    l++;
                }
            }
        }
        return ans;
    }
}
