class Solution {

    public int threeSumClosest(int[] nums, int target) {
        //sort  number increceing order  me kerte hai
        Arrays.sort(nums);
        int n = nums.length;
        // initialize closest sum
        int closest_sum = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < n - 2; i++) {
            int left = i + 1, right = n - 1;
            while (left < right) {
                // two-pointer approach lagate hai
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target) {
                    // sum equals target, return immediately hoga
                    return sum;
                } else if (sum < target) {
                    //sum less then target value jab hoga
                    left++;
                } else {
                    // sum greater then value jab hoga
                    right--;
                }
                if (Math.abs(sum - target) < Math.abs(closest_sum - target)) {
                    // update closest sum
                    closest_sum = sum;
                }
            }
        }
        return closest_sum;
    }
}
