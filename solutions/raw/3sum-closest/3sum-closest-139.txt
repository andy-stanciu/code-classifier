class Solution {

    public int threeSumClosest(int[] nums, int t) {
        Arrays.sort(nums);
        int n = nums.length;
        // we have to print the maximum sum in so
        int min = Integer.MAX_VALUE;
        int res = 0;
        // No need Of Skipping  The value no problem of skipping anyways to faltu me write the logic
        // closset use element ka voh hoga na jo agar tino ka sum se bada aye
        for (int i = 0; i < nums.length; i++) {
            // Differce Calcualte Karna Padega
            int target = t - nums[i];
            int st = i + 1;
            int end = n - 1;
            while (st < end) {
                int sum = nums[st] + nums[end] + nums[i];
                // mereko kisi bhi trah Apna Differnce minimum kana ahi
                if (Math.abs(t - sum) < min) {
                    min = Math.abs(t - sum);
                    res = sum;
                }
                if (nums[st] + nums[end] == target) {
                }
                if (nums[st] + nums[end] > target) {
                    end--;
                } else {
                    st++;
                }
            }
        }
        return res;
    }
}
