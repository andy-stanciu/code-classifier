class Solution {

    public boolean isHappy(int n) {
        // Using Floyd\'s cycle detection algorithm
        if (n == 1)
            return true;
        int p, q, np = n, nq = n;
        while (np != 1 || nq != 1) {
            // take p to 2 steps ahead
            p = sumSqD(sumSqD(np));
            // q to 1 step ahead
            q = sumSqD(nq);
            // if any one of them is 1 then return true
            if (p == 1 || q == 1)
                return true;
            // check if they meet each other at some point then break loop
            if (p == q)
                break;
            np = p;
            nq = q;
        }
        return false;
    }

    private int sumSqD(int n) {
        int sum = 0;
        while (n != 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }
}
