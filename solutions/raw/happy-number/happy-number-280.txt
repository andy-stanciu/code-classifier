class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> hs = new HashSet<>();
        hs.add(n);
        while (n != 1) {
            if (n == 1)
                return true;
            int sum = 0;
            while (n > 0) {
                int mod = n % 10;
                n /= 10;
                sum += mod * mod;
            }
            if (hs.contains(sum))
                return false;
            else
                hs.add(sum);
            n = sum;
        }
        return true;
    }
}
