class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> set = new HashSet<>();
        while (true) {
            List<Integer> arr_num = get_numarr(n);
            int sum = getsum(arr_num);
            if (sum == 1)
                return true;
            if (set.contains(sum))
                return false;
            set.add(sum);
            n = sum;
        }
    }

    public List<Integer> get_numarr(int n) {
        List<Integer> arrlist = new ArrayList<>();
        while (n >= 1) {
            arrlist.add(n % 10);
            n = n / 10;
        }
        return arrlist;
    }

    public int getsum(List<Integer> arr_num) {
        int sum = 0;
        for (int num : arr_num) {
            sum += num * num;
        }
        return sum;
    }
}
