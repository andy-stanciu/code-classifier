class Solution {

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = findHappy(slow);
            fast = findHappy(findHappy(fast));
        } while (slow != fast);
        return slow == 1;
    }

    static int findHappy(int n) {
        int sum = 0;
        while (n != 0) {
            int x = n % 10;
            sum += x * x;
            n /= 10;
        }
        return sum;
    }
}
