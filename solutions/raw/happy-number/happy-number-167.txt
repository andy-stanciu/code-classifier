class Solution {

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = findSqr(slow);
            fast = findSqr(findSqr(fast));
            if (slow == 1) {
                return true;
            }
        } while (slow != fast);
        return false;
    }

    private int findSqr(int n) {
        int sum = 0;
        while (n > 0) {
            int m = n % 10;
            m *= m;
            sum = sum + m;
            n = n / 10;
        }
        return sum;
    }
}
