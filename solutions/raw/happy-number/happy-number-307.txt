class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> h = new HashSet<>();
        while (n != 1 && !h.contains(n)) {
            h.add(n);
            int s = 0;
            while (n > 0) {
                int r = n % 10;
                s += r * r;
                n /= 10;
            }
            n = s;
        }
        return n == 1;
    }
}
