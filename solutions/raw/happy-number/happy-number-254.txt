class Solution {

    public boolean isHappy(int n) {
        while (n > 9) {
            n = sum(n);
        }
        if (n == 1 || n == 7) {
            return true;
        } else {
            return false;
        }
    }

    public static int sum(int num) {
        int sum = 0;
        while (num != 0) {
            int rem = num % 10;
            sum = sum + rem * rem;
            num = num / 10;
        }
        return sum;
    }
}
