class Solution {

    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<>();
        while (n != 1 && !set.contains(n)) {
            set.add(n);
            n = helper(n);
        }
        return n == 1;
    }

    private int helper(int n) {
        int sum = 0;
        while (n > 0) {
            int number = n % 10;
            sum += number * number;
            n /= 10;
        }
        return sum;
    }
}
