class Solution {

    public boolean isHappy(int n) {
        int sum = 0;
        while (n > 9) {
            do {
                int d = n % 10;
                sum = sum + d * d;
                n = n / 10;
            } while (n != 0);
            n = sum;
            sum = 0;
        }
        return n == 1 || n == 7;
    }
}
