class Solution {

    public boolean isHappy(int n) {
        int temp = n;
        while (true) {
            int sum = sum(n);
            if (n == 1)
                return true;
            if (n == 4)
                return false;
            n = sum;
        }
    }

    private static int sum(int num) {
        int sum = 0;
        while (num > 0) {
            int digit = num % 10;
            sum += digit * digit;
            num /= 10;
        }
        return sum;
    }
}
