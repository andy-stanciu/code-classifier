class Solution {

    public static int happynum(int x) {
        int a = 0;
        while (x > 0) {
            a += (int) Math.pow(x % 10, 2);
            x = x / 10;
        }
        return a;
    }

    public boolean isHappy(int n) {
        int m = n;
        while (true) {
            m = happynum(m);
            if (m == 1)
                return true;
            if (m < 1 || m == 89)
                return false;
        }
    }
}
