class Solution {

    public boolean isHappy(int n) {
        int sum = 0;
        do {
            while (n > 0) {
                // Square the last digit and add to sum
                sum += Math.pow(n % 10, 2);
                // Remove the last digit
                n = n / 10;
            }
            // Assign the sum of squares to n
            n = sum;
            // Reset sum for the next iteration
            sum = 0;
        } while (// Keep doing this until n becomes a single-digit number
        n >= 10);
        if (// Special case: if n reaches 7, return true
        n == 7)
            return true;
        // Return true if n is 1 (happy number), otherwise false
        return (n % 10 == 1);
    }
}
