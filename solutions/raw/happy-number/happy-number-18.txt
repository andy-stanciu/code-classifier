class Solution {

    public boolean isHappy(int n) {
        int x;
        while (n / 10 > 0) {
            x = 0;
            while (n != 0) {
                x = x + (int) Math.pow((n % 10), 2);
                n = n / 10;
            }
            n = x;
        }
        if (n == 1 || n == 7)
            return true;
        return false;
    }
}
