class Solution {

    public boolean isHappy(int n) {
        int t = n;
        while (t != 1 && t != 4) {
            int sum = 0;
            while (t > 0) {
                int k = t % 10;
                t = t / 10;
                sum += k * k;
            }
            t = sum;
        }
        if (t == 1) {
            return true;
        }
        return false;
    }
}
