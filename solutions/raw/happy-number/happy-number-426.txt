class Solution {

    public boolean isHappy(int n) {
        return checkHappy(n);
    }

    public boolean checkHappy(int r) {
        if (r == 1)
            return true;
        if (r > 1 && r <= 4)
            return false;
        int sum = 0;
        while (r != 0) {
            sum += (r % 10) * (r % 10);
            r /= 10;
        }
        return checkHappy(sum);
    }
}
