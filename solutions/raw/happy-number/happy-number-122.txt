class Solution {

    public boolean isHappy(int n) {
        int res = n;
        while (res != 1) {
            if (res == 7)
                return true;
            else if (res <= 9)
                return false;
            res = compress(res);
        }
        return true;
    }

    public int compress(int num) {
        int res = 0;
        while (num > 0) {
            res += Math.pow(num % 10, 2);
            num /= 10;
        }
        return res;
    }
}
