class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> set = new HashSet<Integer>();
        while (set.contains(n) == false) {
            set.add(n);
            int result = 0;
            while (n != 0) {
                result += (n % 10) * (n % 10);
                n = n / 10;
            }
            n = result;
            if (result == 1) {
                return true;
            }
        }
        return false;
    }
}
