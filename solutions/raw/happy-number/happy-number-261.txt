class Solution {

    public static int suml(int n) {
        int sum = 0;
        while (n != 0) {
            sum += Math.pow(n % 10, 2);
            n = n / 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        if (n == 1) {
            return true;
        }
        HashSet<Integer> usedSum = new HashSet<>();
        boolean res = false;
        int new_sum = n;
        while (new_sum != 1) {
            new_sum = suml(new_sum);
            if (new_sum == 1) {
                res = true;
                break;
            } else if (!usedSum.contains(new_sum)) {
                usedSum.add(new_sum);
            } else {
                res = false;
                break;
            }
        }
        return res;
    }
}
