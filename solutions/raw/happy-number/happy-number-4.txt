class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> hs = new HashSet<>();
        while (true) {
            int sum = 0;
            while (n > 0) {
                int r = n % 10;
                sum = sum + r * r;
                n = n / 10;
            }
            if (sum == 1) {
                return true;
            } else if (hs.contains(sum)) {
                return false;
            }
            hs.add(sum);
            n = sum;
        }
    }
}
