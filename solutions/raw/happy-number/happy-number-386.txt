class Solution {

    public boolean isHappy(int n) {
        if (n == 1)
            return true;
        if (n / 10 == 0 && n % 2 == 0)
            return false;
        int currentNum = 0;
        while (n != 0) {
            currentNum += (int) Math.pow(n % 10, 2);
            n = n / 10;
        }
        return isHappy(currentNum);
    }
}
