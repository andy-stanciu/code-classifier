class Solution {

    public boolean isHappy(int n) {
        List<Integer> numList = String.valueOf(n).chars().map(Character::getNumericValue).boxed().collect(Collectors.toList());
        int sum = 0;
        for (Integer number : numList) {
            sum += number * number;
        }
        if (sum == 1) {
            return true;
        } else if ((sum > 1 && sum <= 9) && (sum % 2 == 0)) {
            return false;
        } else {
            return isHappy(sum);
        }
    }
}
