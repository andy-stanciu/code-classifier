class Solution {

    public int SquareDigit(int n) {
        int digit = -1, sum = 0;
        while (n > 0) {
            digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        int sum = -1;
        if (n == 1 || n == 7) {
            return true;
        }
        sum = SquareDigit(n);
        while (sum != 1 && sum != 4) sum = SquareDigit(sum);
        return (sum == 1);
    }
}
