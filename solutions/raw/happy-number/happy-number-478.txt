class Solution {

    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<>();
        int temp = n;
        while (temp != 1) {
            if (set.contains(temp)) {
                return false;
            }
            set.add(temp);
            int sum = 0;
            while (temp > 0) {
                int val = temp % 10;
                temp = temp / 10;
                sum += val * val;
            }
            temp = sum;
        }
        return true;
    }
}
