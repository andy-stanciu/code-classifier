class Solution {

    public boolean isHappy(int n) {
        int m = 0;
        while (n != 1 && n != 4) {
            int ans = 0;
            while (n > 0) {
                int r = n % 10;
                ans += r * r;
                n /= 10;
            }
            n = ans;
        }
        if (n == 4)
            return false;
        return true;
    }
}
