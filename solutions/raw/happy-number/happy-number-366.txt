class Solution {

    public boolean isHappy(int n) {
        int fast = n;
        int slow = n;
        do {
            slow = getSquareOfDigits(slow);
            fast = getSquareOfDigits(getSquareOfDigits(fast));
        } while (slow != fast);
        if (slow == 1) {
            return true;
        }
        return false;
    }

    static int getSquareOfDigits(int num) {
        int ans = 0;
        if (num == 0) {
            return ans;
        }
        int rem = num % 10;
        ans += rem * rem;
        return ans + getSquareOfDigits(num / 10);
    }
}
