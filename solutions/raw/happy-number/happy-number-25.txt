class Solution {

    public boolean isHappy(int n) {
        int temp = n;
        while (true) {
            int sum = sum(n);
            if (n == 1)
                return true;
            if (n == 4)
                return false;
            n = sum;
        }
    }

    static int sum(int n) {
        int sum = 0;
        while (n > 0) {
            int r = n % 10;
            sum += (int) Math.pow(r, 2);
            n /= 10;
        }
        return sum;
    }
}
