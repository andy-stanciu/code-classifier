class Solution {

    public boolean isHappy(int n) {
        Map<Integer, Integer> hp = new HashMap<>();
        int result = 0;
        do {
            result = sumvalue(n);
            if (result == 1) {
                return true;
            }
            hp.put(n, result);
            n = result;
        } while (!hp.containsKey(result));
        return false;
    }

    public int sumvalue(int n) {
        int sumdata = 0;
        while (n != 0) {
            int ab = n % 10;
            n = n / 10;
            sumdata = sumdata + (ab * ab);
        }
        return sumdata;
    }
}
