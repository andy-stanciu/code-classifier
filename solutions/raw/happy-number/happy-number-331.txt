class Solution {

    public boolean isHappy(int n) {
        if (n == 1 || n == 7) {
            return true;
        }
        while (n > 9) {
            int num = n;
            int sum = 0;
            while (num > 0) {
                sum = sum + (int) Math.pow(num % 10, 2);
                num = num / 10;
            }
            n = sum;
            if (n == 1 || n == 7) {
                return true;
            }
        }
        return false;
    }
}
