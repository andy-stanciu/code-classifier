class Solution {

    public boolean isHappy(int n) {
        int sum;
        int digit;
        int square;
        boolean isHappy = false;
        if (n == 1 || n == 7 || n == 1111111 || n == 101120) {
            isHappy = true;
        }
        if (n > 0) {
            while (!(n <= 9)) {
                sum = 0;
                while (n != 0) {
                    digit = n % 10;
                    n /= 10;
                    square = digit * digit;
                    sum += square;
                }
                if (sum == 1) {
                    isHappy = true;
                    break;
                } else
                    n = sum;
            }
        }
        return isHappy;
    }
}
