class Solution {

    public int Loop(int n) {
        int digi = 0;
        while (n != 0) {
            digi += (int) Math.pow((n % 10), 2);
            n = n / 10;
        }
        return digi;
    }

    public boolean isHappy(int n) {
        int sl = n;
        int fs = n;
        do {
            sl = Loop(sl);
            fs = Loop(Loop(fs));
        } while (sl != fs);
        if (sl == 1)
            return true;
        return false;
    }
}
