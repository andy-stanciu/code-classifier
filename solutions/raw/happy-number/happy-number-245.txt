class Solution {

    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<>();
        while (set.contains(n) == false) {
            set.add(n);
            n = nextdigit(n);
            if (n == 1) {
                return true;
            }
        }
        return false;
    }

    private int nextdigit(int n) {
        int sum = 0;
        while (n > 0) {
            // retrieving last digit
            int digit = n % 10;
            // summing up
            sum += digit * digit;
            // changing the number;
            n = n / 10;
        }
        return sum;
    }
}
