class Solution {

    public boolean isHappy(int n) {
        int slow = n, fast = n;
        do {
            slow = findSquareNum(slow);
            fast = findSquareNum(findSquareNum(fast));
        } while (slow != fast);
        return slow == 1;
    }

    int findSquareNum(int num) {
        int sum = 0, digit;
        while (num > 0) {
            digit = num % 10;
            sum += digit * digit;
            num /= 10;
        }
        return sum;
    }
}
