class Solution {

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            //cycle detection do while loop
            //slow pointer one time
            slow = Square(slow);
            //fast pointer two times
            fast = Square(Square(fast));
        } while (slow != fast);
        if (slow == 1)
            return true;
        return false;
    }

    public int Square(int num) {
        //fuction to find the squares
        int sq = 0;
        while (num > 0) {
            int r = num % 10;
            sq = sq + r * r;
            num = num / 10;
        }
        return sq;
    }
}
