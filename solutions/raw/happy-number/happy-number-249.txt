class Solution {

    public int sumOfSquares(int a) {
        int sum = 0;
        while (a > 0) {
            int d = a % 10;
            sum = sum + (d * d);
            a = a / 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        while (n > 9) {
            n = sumOfSquares(n);
        }
        if (n == 1 || n == 7)
            return true;
        else
            return false;
    }
}
