class Solution {

    public int checkHappy(int n) {
        int sum = 0;
        while (n != 0) {
            int lastDigit = n % 10;
            sum += lastDigit * lastDigit;
            n /= 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = checkHappy(slow);
            fast = checkHappy(checkHappy(fast));
        } while (slow != fast);
        if (slow == 1)
            return true;
        return false;
    }
}
