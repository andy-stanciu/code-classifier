class Solution {

    public boolean isHappy(int n) {
        if (n == 1 || n == 7)
            return true;
        if (Integer.toString(n).length() == 1)
            return false;
        int temp = n, sum = 0, rem = 0;
        while (temp != 0) {
            rem = temp % 10;
            sum += Math.pow(rem, 2);
            temp /= 10;
        }
        return isHappy(sum);
    }
}
