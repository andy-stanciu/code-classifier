class Solution {

    public boolean isHappy(int n) {
        int square = findSquare(n);
        int slow = square;
        int fast = square;
        while (slow != 1) {
            slow = findSquare(slow);
            fast = findSquare(findSquare(fast));
            if (slow == fast && slow != 1) {
                return false;
            }
        }
        return true;
    }

    private int findSquare(int number) {
        int ans = 0;
        while (number > 0) {
            int rem = number % 10;
            ans = ans + (rem * rem);
            number = number / 10;
        }
        return ans;
    }
}
