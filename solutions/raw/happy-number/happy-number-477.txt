class Solution {

    public boolean isHappy(int n) {
        if (n == 1)
            return true;
        if (n > 1 && n < 7)
            return false;
        int res = 0;
        while (n > 0) {
            int digit = n % 10;
            res += digit * digit;
            n /= 10;
        }
        return isHappy(res);
    }
}
