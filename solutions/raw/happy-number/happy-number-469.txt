class Solution {

    public boolean isHappy(int n) {
        int f = n;
        int s = n;
        do {
            s = square(s);
            f = square(square(f));
        } while (s != f);
        if (s == 1) {
            return true;
        }
        return false;
    }

    public int square(int number) {
        int sum = 0;
        while (number > 0) {
            int r = number % 10;
            sum += r * r;
            number = number / 10;
        }
        return sum;
    }
}
