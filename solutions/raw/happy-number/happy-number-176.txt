class Solution {

    public boolean isHappy(int n) {
        if (n < 0) {
            return false;
        }
        HashMap<Integer, Integer> map = new HashMap<>();
        while (n != 1) {
            int temp = n;
            int sum = 0;
            while (temp > 0) {
                int rem = temp % 10;
                sum = sum + (rem * rem);
                temp = temp / 10;
            }
            if (map.containsKey(sum)) {
                return false;
            }
            map.put(sum, map.getOrDefault(sum, 0) + 1);
            n = sum;
        }
        return true;
    }
}
