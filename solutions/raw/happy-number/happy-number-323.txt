class Solution {

    public boolean isHappy(int n) {
        Set<Integer> seen = new HashSet<>();
        while (n != 1 && !seen.contains(n)) {
            seen.add(n);
            n = square(n);
        }
        return n == 1;
    }

    public int square(int n) {
        int ans = 0;
        int remainder = 0;
        while (n > 0) {
            remainder = n % 10;
            ans = ans + remainder * remainder;
            n = n / 10;
        }
        return ans;
    }
}
