class Solution {

    public boolean isHappy(int n) {
        Set<Integer> seen = new HashSet<>();
        while (n != 1) {
            int sumOfSquares = 0;
            while (n > 0) {
                int digit = n % 10;
                sumOfSquares += digit * digit;
                n /= 10;
            }
            if (seen.contains(sumOfSquares)) {
                return false;
            }
            seen.add(sumOfSquares);
            n = sumOfSquares;
        }
        return true;
    }
}
