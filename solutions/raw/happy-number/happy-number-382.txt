class Solution {

    int findSquare(int n) {
        int n1 = 0;
        while (n != 0) {
            int rem = n % 10;
            n = n / 10;
            n1 += rem * rem;
        }
        return n1;
    }

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = findSquare(slow);
            fast = findSquare(findSquare(fast));
        } while (slow != fast);
        if (slow == 1) {
            return true;
        }
        return false;
    }
}
