class Solution {

    public boolean isHappy(int n) {
        int sum = 0;
        while (n != 1 && n != 4) {
            while (n > 0) {
                int rem = n % 10;
                sum = sum + (rem * rem);
                n = n / 10;
            }
            n = sum;
            sum = 0;
        }
        return n == 1;
    }
}
