class Solution {

    public long getSumOfSquare(long n) {
        long sum = 0;
        while (n > 0) {
            long digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        long num = n;
        while (num > 9) {
            num = getSumOfSquare(num);
        }
        return num == 1 || num == 7;
    }
}
