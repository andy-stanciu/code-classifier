class Solution {

    public boolean isHappy(int n) {
        Set<Integer> hashSet = new HashSet<>();
        while (n != 1 && !hashSet.contains(n)) {
            hashSet.add(n);
            n = getNxt(n);
        }
        if (n == 1) {
            return true;
        }
        return false;
    }

    private static int getNxt(int num) {
        int digit = 0;
        int totalSum = 0;
        while (num > 0) {
            digit = num % 10;
            totalSum += digit * digit;
            num = num / 10;
        }
        return totalSum;
    }
}
