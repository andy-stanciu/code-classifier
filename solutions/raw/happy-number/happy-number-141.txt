class Solution {

    public boolean isHappy(int n) {
        int c = 0;
        while (c < 200) {
            n = sumOfSquareOfDigits(n);
            if (n == 1) {
                return true;
            }
            c++;
        }
        return false;
    }

    public int sumOfSquareOfDigits(int n) {
        int num;
        int sum = 0;
        while (n != 0) {
            num = n % 10;
            n /= 10;
            sum += (int) Math.pow(num, 2);
        }
        return sum;
    }
}
