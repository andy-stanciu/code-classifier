class Solution {

    public boolean isHappy(int N) {
        while (DigitSquareSum(N) != 1 && DigitSquareSum(N) != 4) {
            N = DigitSquareSum(N);
        }
        N = DigitSquareSum(N);
        if (N == 1) {
            return true;
        }
        return false;
    }

    public static int DigitSquareSum(int number) {
        int temp = number;
        int sum = 0;
        while (temp > 0) {
            int digit = temp % 10;
            sum += (digit * digit);
            temp = temp / 10;
        }
        return sum;
    }
}
