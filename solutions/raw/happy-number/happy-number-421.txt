class Solution {

    public boolean isHappy(int n) {
        int sum, rem;
        Set<Integer> list = new HashSet<>();
        while (n != 1 && !list.contains(n)) {
            list.add(n);
            sum = 0;
            while (n > 0) {
                rem = n % 10;
                sum = sum + rem * rem;
                n = n / 10;
            }
            n = sum;
        }
        return n == 1;
    }
}
