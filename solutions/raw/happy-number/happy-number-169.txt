class Solution {

    public boolean isHappy(int n) {
        int ans = 0;
        HashSet<Integer> set = new HashSet<>();
        set.add(n);
        while (ans != 1) {
            ans = 0;
            if (n == 0)
                return false;
            while (n != 0) {
                ans += (n % 10) * (n % 10);
                n = n / 10;
            }
            n = ans;
            if (n == 1)
                return true;
            else if (set.contains(ans))
                return false;
            else
                set.add(ans);
        }
        return false;
    }
}
