class Solution {

    public boolean isHappy(int n) {
        int sums;
        for (int i = 0; i < n; i++) {
            sums = 0;
            while (n != 0) {
                int s = n % 10;
                sums = sums + s * s;
                n = n / 10;
            }
            n = sums;
            if (n == 1) {
                return true;
            }
        }
        return false;
    }
}
