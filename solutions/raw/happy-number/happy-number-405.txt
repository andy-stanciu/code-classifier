class Solution {

    public boolean isHappy(int n) {
        if (n == 1 || n == 7)
            return true;
        if (n < 10)
            return false;
        int sum = 0;
        while (n > 0) {
            int remainder = n % 10;
            n = (n - remainder) / 10;
            sum += remainder * remainder;
        }
        if (sum == n)
            return false;
        return isHappy(sum);
    }
}
