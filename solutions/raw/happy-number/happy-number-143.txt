class Solution {

    public boolean isHappy(int n) {
        int slow = n;
        int fast = happy(n);
        while (fast != 1 && slow != fast) {
            slow = happy(slow);
            fast = happy(happy(fast));
        }
        return fast == 1;
    }

    public int happy(int num) {
        int sum = 0;
        while (num != 0) {
            int d = num % 10;
            num = num / 10;
            sum += d * d;
        }
        return sum;
    }
}
