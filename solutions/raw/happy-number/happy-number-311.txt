class Solution {

    public boolean isHappy(int n) {
        int fast = n;
        int slow = n;
        do {
            fast = sq(sq(fast));
            slow = sq(slow);
            if (slow == 1 || fast == 1) {
                return true;
            }
        } while (slow != fast);
        return false;
    }

    int sq(int n) {
        int val = 0;
        while (n > 0) {
            int a = n % 10;
            val += a * a;
            n /= 10;
        }
        return val;
    }
}
