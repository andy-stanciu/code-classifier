class Solution {

    public static int getSquareSum(int n) {
        int x = 0;
        while (n > 0) {
            int y = n % 10;
            x += y * y;
            n /= 10;
        }
        return x;
    }

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = getSquareSum(slow);
            fast = getSquareSum(getSquareSum(fast));
        } while (fast != 1 && fast != slow);
        return fast == 1;
    }
}
