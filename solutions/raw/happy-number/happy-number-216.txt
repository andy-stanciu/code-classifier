class Solution {

    public boolean isHappy(int n) {
        int result = n;
        while (result != 1 && result != 4) {
            result = happy(result);
        }
        if (result == 1)
            return true;
        else
            return false;
    }

    private int happy(int n) {
        int sum = 0;
        while (n > 0) {
            int dig = n % 10;
            sum += dig * dig;
            n /= 10;
        }
        return sum;
    }
}
