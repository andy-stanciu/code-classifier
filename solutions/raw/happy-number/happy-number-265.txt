class Solution {

    public static int sumOfSquares(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        for (int i = 0; i < 50; i++) {
            n = sumOfSquares(n);
            if (n == 1) {
                return true;
            }
        }
        return false;
    }
}
