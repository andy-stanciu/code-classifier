class Solution {

    private boolean isReallyHappy(int n, Set<Integer> seen) {
        int nextNumber = 0;
        while (n > 0) {
            nextNumber = nextNumber + ((n % 10) * (n % 10));
            // System.out.println(">>>>>>"+n);
            n = n / 10;
        }
        // System.out.println("******"+nextNumber);
        if (nextNumber == 1) {
            return true;
        }
        if (seen.contains(nextNumber)) {
            return false;
        }
        seen.add(nextNumber);
        return isReallyHappy(nextNumber, seen);
    }

    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<Integer>();
        return isReallyHappy(n, set);
    }
}
