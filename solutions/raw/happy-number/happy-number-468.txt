class Solution {

    public boolean isHappy(int n) {
        if (n == 1)
            return true;
        if (n < 7)
            return false;
        return isHappy(sum(n));
    }

    public int sum(int num) {
        int sum = 0;
        while (num != 0) {
            int dig = num % 10;
            sum += dig * dig;
            num /= 10;
        }
        return sum;
    }
}
