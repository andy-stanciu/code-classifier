class Solution {

    public boolean isHappy(int n) {
        if (n == 1 || n == 7)
            // 1 and 7 are already happy number
            return true;
        int sum = squareDigits(n);
        while (// if sum==1 or sum==4 exit the loop
        sum != 1 && sum != 4) sum = squareDigits(sum);
        if (sum == 1) {
            // if sum==1 retuen true because 1 is happy number
            return true;
        } else {
            // if sum==4 return false because 4 is unhappy number
            return false;
        }
    }

    private int squareDigits(int n) {
        int sum = 0;
        while (n > 0) {
            int rem = n % 10;
            sum += rem * rem;
            n /= 10;
        }
        return sum;
    }
}
