class Solution {

    public boolean isHappy(int n) {
        if (n == 1 || n == 7) {
            return true;
        } else if (n < 9) {
            return false;
        } else {
            int sum = calculateSum(n);
            return isHappy(sum);
        }
    }

    private static int calculateSum(int i) {
        if (i <= 9) {
            return i * i;
        } else {
            return (((i % 10) * (i % 10)) + calculateSum(i / 10));
        }
    }
}
