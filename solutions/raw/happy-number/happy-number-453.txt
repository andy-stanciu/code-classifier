class Solution {

    public int getNext(int n) {
        int sum = 0;
        while (n > 0) {
            int d = n % 10;
            n = n / 10;
            sum += d * d;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<>();
        while (n > 1 && !set.contains(getNext(n))) {
            n = getNext(n);
            set.add(n);
        }
        return n == 1;
    }
}
