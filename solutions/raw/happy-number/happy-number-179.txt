class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> set = new HashSet<>();
        while (n != 1 && !set.contains(n)) {
            set.add(n);
            n = convert(n);
        }
        return n == 1;
    }

    public static int convert(int n) {
        int ld = 0;
        int sum = 0;
        while (n > 0) {
            ld = n % 10;
            sum = sum + ld * ld;
            n = n / 10;
        }
        return sum;
    }
}
