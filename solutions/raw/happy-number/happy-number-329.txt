class Solution {

    public boolean isHappy(int n) {
        while (n != 1 && n != 4) {
            n = resultHappy(n);
        }
        return n == 1;
    }

    public int resultHappy(int n) {
        int temp = 0;
        while (n > 0) {
            temp += Math.pow(n % 10, 2);
            n /= 10;
        }
        return temp;
    }
}
