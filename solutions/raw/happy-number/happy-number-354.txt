class Solution {

    public boolean isHappy(int n) {
        int s = n;
        int f = n;
        int x = 0, y = 0;
        while (true) {
            for (int i = 1; i <= 1; i++) {
                s = Sum_digits(s);
            }
            for (int i = 1; i <= 2; i++) {
                f = Sum_digits(f);
            }
            if (f == 1) {
                x = 1;
                break;
            } else if (f == s) {
                y = 1;
                break;
            }
        }
        if (x == 1)
            return true;
        else if (y == 1)
            return false;
        return false;
    }

    public int Sum_digits(int n) {
        int temp = n;
        int sum = 0;
        while (temp != 0) {
            int r = temp % 10;
            int k = (int) Math.pow(r, 2);
            sum = sum + k;
            temp = temp / 10;
        }
        return sum;
    }
}
