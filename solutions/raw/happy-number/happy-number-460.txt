class Solution {

    public boolean isHappy(int n) {
        int slow = n, fast = n;
        while (fast != 1 && happySquareOf(fast) != 1) {
            slow = happySquareOf(slow);
            fast = happySquareOf(happySquareOf(fast));
            if (fast == slow)
                return false;
        }
        return true;
    }

    public int happySquareOf(int num) {
        int temp = num;
        int sum = 0;
        while (temp != 0) {
            sum += Math.pow((temp % 10), 2);
            temp /= 10;
        }
        return sum;
    }
}
