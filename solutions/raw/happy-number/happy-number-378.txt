class Solution {

    public boolean isHappy(int n) {
        Set<Integer> set = new HashSet<>();
        while (n != 1) {
            if (set.contains(n)) {
                return false;
            }
            set.add(n);
            int copy = n;
            n = 0;
            while (copy > 0) {
                int r = copy % 10;
                copy = copy / 10;
                n += r * r;
            }
        }
        return true;
    }
}
