class Solution {

    public boolean isHappy(int n) {
        List<Integer> list = new ArrayList<>();
        int num = n;
        while (num != 1) {
            int sum = 0;
            while (num > 0) {
                int lastDigit = num % 10;
                num /= 10;
                sum += (lastDigit * lastDigit);
            }
            num = sum;
            if (list.contains(num))
                return false;
            list.add(num);
        }
        return true;
    }
}
