class Solution {

    public boolean isHappy(int n) {
        int num = n;
        int sum = 0;
        int length = String.valueOf(num).length();
        ;
        int digit = 0;
        while (num != 1 && num != 4) {
            for (int i = 0; i < length; i++) {
                digit = (int) (num / Math.pow(10, i) % 10);
                sum += digit * digit;
            }
            length = String.valueOf(sum).length();
            num = sum;
            sum = 0;
        }
        return num == 1;
    }
}
