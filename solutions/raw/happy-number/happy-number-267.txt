class Solution {

    int fun(int num) {
        int sum = 0;
        while (num != 0) {
            sum = sum + ((num % 10) * (num % 10));
            num = num / 10;
        }
        return sum;
    }

    public boolean isHappy(int n) {
        if (n == 1)
            return true;
        while (true) {
            n = fun(n);
            if (n == 4) {
                return false;
            }
            if (n == 1) {
                return true;
            }
        }
    }
}
