class Solution {

    public boolean isHappy(int n) {
        int first = n;
        int second = n;
        do {
            second = square(second);
            first = square(square(first));
        } while (first != second);
        if (second == 1) {
            return true;
        }
        return false;
    }

    public int square(int n) {
        int ans = 0;
        while (n > 0) {
            int rem = n % 10;
            ans = ans + (rem * rem);
            n = n / 10;
        }
        return ans;
    }
}
