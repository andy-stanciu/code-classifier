class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> hs = new HashSet<>();
        while (n != 1) {
            int sum = 0;
            if (hs.contains(n)) {
                return false;
            } else {
                hs.add(n);
            }
            while (n > 0) {
                //234
                int r = n % 10;
                sum = sum + r * r;
                n = n / 10;
            }
            n = sum;
        }
        return true;
    }
}
