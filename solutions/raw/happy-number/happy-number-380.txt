class Solution {

    public int sumSquares(int n) {
        int counter = 0;
        int result = 0;
        while (n > 0) {
            counter = n % 10;
            result = result + counter * counter;
            n /= 10;
        }
        return result;
    }

    public boolean isHappy(int n) {
        while (1 > 0) {
            n = sumSquares(n);
            if (n == 1 || n == 7)
                return true;
            else if (n < 10)
                return false;
        }
    }
}
