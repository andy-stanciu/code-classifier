class Solution {

    public boolean isHappy(int n) {
        if (n == 1)
            return true;
        if (n < 5) {
            return false;
        }
        int sum = 0;
        while (n > 0) {
            int r = n % 10;
            sum += r * r;
            n /= 10;
        }
        if (sum == 1)
            return true;
        return isHappy(sum);
    }
}
