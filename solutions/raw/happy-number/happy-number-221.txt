class Solution {

    public boolean isHappy(int n) {
        Map<Integer, Integer> visited = new HashMap<>();
        boolean ans = true;
        while (n != 1) {
            if (visited.containsKey(n)) {
                ans = false;
                break;
            }
            visited.put(n, 1);
            n = sumOfDigitsSquare(n);
        }
        return ans;
    }

    public int sumOfDigitsSquare(int n) {
        int sum = 0;
        while (n > 0) {
            int d = n % 10;
            sum += d * d;
            n = n / 10;
        }
        return sum;
    }
}
