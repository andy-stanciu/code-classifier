class Solution {

    public int squareNum(int n) {
        int squareSum = 0;
        while (n != 0) {
            squareSum += (n % 10) * (n % 10);
            n = n / 10;
        }
        return squareSum;
    }

    public boolean isHappy(int n) {
        // here we apply a approach which find a loop in a linklist type
        //intilise both by n
        int slow = n;
        int fast = n;
        do {
            // move slow number by one iteration
            slow = squareNum(slow);
            // move fast number by two iteration
            fast = squareNum(squareNum(fast));
        } while (slow != fast);
        return (slow == 1);
    }
}
