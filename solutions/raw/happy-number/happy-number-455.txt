class Solution {

    public boolean isHappy(int n) {
        for (int i = 0; i < 7; i++) {
            n = calc(n);
            if (n == 1) {
                return true;
            }
        }
        return false;
    }

    // 82 %10 = 2
    // 2 +
    // 82/10=8
    // 2+8
    public int calc(int n) {
        int sum = 0;
        int zaid = 0;
        while (n > 0) {
            zaid = n % 10;
            sum += zaid * zaid;
            n = n / 10;
        }
        return sum;
    }
}
