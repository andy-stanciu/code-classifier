class Solution {

    private static HashSet<Integer> set = new HashSet<>();

    public boolean isHappy(int n) {
        set.clear();
        if (ishappy(n)) {
            return true;
        }
        return false;
    }

    public boolean ishappy(int n) {
        int temp = n;
        int sum = 0;
        while (temp > 0) {
            int rem = temp % 10;
            sum = sum + (rem * rem);
            temp = temp / 10;
        }
        if (sum == 1) {
            return true;
        } else {
            if (set.contains(sum)) {
                return false;
            } else {
                set.add(sum);
                return ishappy(sum);
            }
        }
    }
}
