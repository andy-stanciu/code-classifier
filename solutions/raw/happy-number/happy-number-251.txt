class Solution {

    public boolean isHappy(int n) {
        int temp = n;
        int sum = sumOfSquares(temp);
        while (sum != 1) {
            sum = sumOfSquares(sum);
            if (sum != 1 && sum / 10 == 0)
                return false;
        }
        return true;
    }

    public int sumOfSquares(int n) {
        int sum = 0;
        while (n > 0) {
            int dig = n % 10;
            sum = sum + dig * dig;
            n = n / 10;
        }
        return sum;
    }
}
