class Solution {

    public boolean isHappy(int n) {
        if (n < 0)
            return false;
        int slow = n, fast = n;
        do {
            slow = square(slow);
            fast = (square(square(fast)));
        } while (slow != fast);
        return slow == 1;
    }

    private int square(int x) {
        int sq = 0;
        while (x > 0) {
            int d = x % 10;
            sq += d * d;
            x /= 10;
        }
        return sq;
    }
}
