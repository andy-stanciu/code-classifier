class Solution {

    public boolean isHappy(int n) {
        HashSet<Integer> ishappy = new HashSet<Integer>();
        int c = 0, m = 0;
        if (n == 1 || n == 7) {
            return true;
        }
        if (ishappy.contains(n) || n < 10) {
            return false;
        }
        ishappy.add(n);
        while (n != 0) {
            c = n % 10;
            m += c * c;
            n = n / 10;
        }
        n = m;
        return isHappy(n);
    }
}
