class Solution {

    private HashSet<Integer> seenSums = new HashSet<>();

    public boolean isHappy(int n) {
        if (n == 1) {
            return true;
        }
        int sum = 0;
        while (n != 0) {
            int r = n % 10;
            n /= 10;
            sum += r * r;
        }
        if (seenSums.contains(sum)) {
            return false;
        } else {
            seenSums.add(sum);
            return isHappy(sum);
        }
    }
}
