class Solution {

    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = getSumOfSquare(slow);
            fast = getSumOfSquare(getSumOfSquare(fast));
        } while (slow != fast);
        return slow == 1;
    }

    private static int getSumOfSquare(int n) {
        int sum = 0;
        while (n > 0) {
            int remainder = n % 10;
            // square the remainder And sum of them
            sum = sum + remainder * remainder;
            // decrease from the last number
            n = n / 10;
        }
        return sum;
    }
}
