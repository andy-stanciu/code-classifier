class Solution {

    private int findsqr(int n) {
        int ans = 0;
        while (n > 0) {
            int rem = n % 10;
            ans += rem * rem;
            n = n / 10;
        }
        return ans;
    }

    public boolean isHappy(int n) {
        int s = n;
        int f = n;
        do {
            s = findsqr(s);
            f = findsqr(findsqr(f));
        } while (f != s);
        if (s == 1) {
            return true;
        }
        return false;
    }
}
