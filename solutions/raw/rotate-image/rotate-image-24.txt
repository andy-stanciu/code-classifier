class Solution {

    public void rotate(int[][] matrix) {
        int n = matrix.length;
        // matrissani ko\'chirish
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int temp = matrix[i][j];
                // Swap elements
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        // harbir qator teskari aylantiradi
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n / 2; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[i][n - j - 1];
                matrix[i][n - j - 1] = temp;
            }
        }
    }
}
