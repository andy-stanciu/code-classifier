class Solution {

    public void rotate(int[][] matrix) {
        int n = matrix.length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        for (int[] rows : matrix) {
            int left = 0;
            int right = rows.length - 1;
            while (left < right) {
                int temp = rows[left];
                rows[left] = rows[right];
                rows[right] = temp;
                left++;
                right--;
            }
        }
    }
}
