class Solution {

    public void rotate(int[][] matrix) {
        int i;
        int j;
        int row = 0;
        int col = 0;
        while (row < matrix.length - 1 && col < matrix.length - 1) {
            j = row + 1;
            i = row + 1;
            while (i < matrix.length && j < matrix.length) {
                int temp = matrix[i][row];
                matrix[i][row] = matrix[col][j];
                matrix[col][j] = temp;
                i++;
                j++;
            }
            row++;
            col++;
        }
        for (i = 0; i < matrix.length; i++) {
            int k = matrix[i].length - 1;
            for (j = 0; j < k; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[i][k];
                matrix[i][k] = temp;
                k--;
            }
        }
    }
}
