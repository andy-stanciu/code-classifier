class Solution {

    public void rotate(int[][] matrix) {
        int row = 0;
        int startCol = 0;
        int endCol = matrix.length - 1;
        while (startCol < endCol) {
            for (int c = startCol; c < endCol; c++) {
                rotate(matrix, 4, row, c);
            }
            row += 1;
            startCol += 1;
            endCol -= 1;
        }
    }

    private void rotate(int[][] matrix, int count, int row, int col) {
        if (count == 0) {
            return;
        }
        int value = matrix[row][col];
        int targetRow = col;
        int targetCol = matrix.length - row - 1;
        rotate(matrix, count - 1, targetRow, targetCol);
        matrix[targetRow][targetCol] = value;
    }
}
