class Solution {

    public void rotate(int[][] a) {
        int x = 0, temp, y;
        int n = a.length - 1;
        while (x < n) {
            y = n;
            for (int i = x; i < n; i++) {
                temp = a[x][i];
                a[x][i] = a[y][x];
                a[y][x] = a[n][y];
                a[n][y--] = a[i][n];
                a[i][n] = temp;
            }
            n--;
            x++;
        }
    }
}
