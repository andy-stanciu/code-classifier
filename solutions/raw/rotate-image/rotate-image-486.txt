class Solution {

    public void rotate(int[][] matrix) {
        int layer = 0;
        int center = matrix.length / 2;
        int buf = 0;
        while (layer < center) {
            int bound = matrix.length - layer - 1;
            for (int i = layer; i < bound; i++) {
                buf = matrix[layer][i];
                matrix[layer][i] = matrix[bound + layer - i][layer];
                matrix[bound + layer - i][layer] = matrix[bound][bound + layer - i];
                matrix[bound][bound + layer - i] = matrix[i][bound];
                matrix[i][bound] = buf;
            }
            layer++;
        }
    }
}
