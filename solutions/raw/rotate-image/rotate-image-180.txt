class Solution {

    public void rotate(int[][] matrix) {
        int n = matrix.length;
        //transpose of the matrix
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) swap(matrix, i, j);
        }
        //swapping first row and last column
        for (int i = 0; i < matrix.length; i++) {
            int low = 0;
            int high = matrix.length - 1;
            while (low < high) {
                int temp = matrix[i][low];
                matrix[i][low] = matrix[i][high];
                matrix[i][high] = temp;
                low++;
                high--;
            }
        }
    }

    public void swap(int[][] arr, int i, int j) {
        int temp = arr[i][j];
        arr[i][j] = arr[j][i];
        arr[j][i] = temp;
    }
}
