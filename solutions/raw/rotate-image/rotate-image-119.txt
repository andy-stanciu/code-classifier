class Solution {

    public void rotate(int[][] arr) {
        int n = arr.length;
        int[][] arr2 = new int[n][n];
        int k = n - 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                arr2[i][j] = arr[i][j];
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                arr[j][k] = arr2[i][j];
            }
            k--;
        }
        return;
    }
}
