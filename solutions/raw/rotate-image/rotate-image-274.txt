class Solution {

    public void rotate(int[][] matrix) {
        int r = matrix.length;
        int c = matrix[0].length;
        int[][] Transpose = new int[r][c];
        for (int i = 0; i < r; i++) {
            for (int j = c - 1, k = 0; j >= 0; j--, k++) {
                Transpose[i][k] = matrix[j][i];
            }
        }
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                matrix[i][j] = Transpose[i][j];
            }
        }
    }
}
