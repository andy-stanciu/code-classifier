class Solution {

    public void rotate(int[][] matrix) {
        int rows = matrix.length;
        int[][] newMat = new int[rows][rows];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < rows; j++) {
                newMat[i][j] = matrix[rows - j - 1][i];
            }
        }
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < rows; j++) {
                matrix[i][j] = newMat[i][j];
            }
        }
    }
}
