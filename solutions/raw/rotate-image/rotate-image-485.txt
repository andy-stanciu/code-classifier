class Solution {

    public void rotate(int[][] matrix) {
        int n = matrix.length;
        // dummy matrix
        int[][] ans = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                // Rotating elements
                ans[j][n - i - 1] = matrix[i][j];
            }
        }
        // we can directly return the dummy matrix as  (return ans)
        // we can copy all the elements of dummy matrix back to original matrix to get out ans
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                matrix[i][j] = ans[i][j];
            }
        }
    }
}
