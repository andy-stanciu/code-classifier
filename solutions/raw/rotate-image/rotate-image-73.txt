class Solution {

    public void rotate(int[][] matrix) {
        int[][] ans = new int[matrix.length][matrix.length];
        int k = 0;
        int n = matrix.length - 1;
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix.length; j++) {
                ans[i][j] = matrix[i][j];
            }
        }
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix.length; j++) {
                matrix[i][j] = ans[n - j][i];
            }
        }
    }
}
