class Solution {

    class Edge {

        int src;

        int dest;

        Edge(int src, int dest) {
            this.src = src;
            this.dest = dest;
        }
    }

    public void createGraph(int[][] prerequisites, ArrayList<Edge>[] graph) {
        for (int i = 0; i < graph.length; i++) {
            graph[i] = new ArrayList<>();
        }
        for (int i = 0; i < prerequisites.length; i++) {
            graph[prerequisites[i][1]].add(new Edge(prerequisites[i][1], prerequisites[i][0]));
        }
    }

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        ArrayList<Edge>[] graph = new ArrayList[numCourses];
        createGraph(prerequisites, graph);
        boolean[] vis = new boolean[numCourses];
        // Used for cycle detection
        boolean[] recStack = new boolean[numCourses];
        Stack<Integer> s = new Stack<>();
        int[] ans = new int[numCourses];
        int y = 0;
        for (int i = 0; i < numCourses; i++) {
            if (!vis[i]) {
                if (!topSortUtil(graph, i, vis, recStack, s)) {
                    // Return empty array if a cycle is detected
                    return new int[0];
                }
            }
        }
        // Populate the result array from the stack
        while (!s.isEmpty()) {
            ans[y] = s.pop();
            y++;
        }
        return ans;
    }

    public boolean topSortUtil(ArrayList<Edge>[] graph, int curr, boolean[] vis, boolean[] recStack, Stack<Integer> s) {
        vis[curr] = true;
        // Mark the current node in the recursion stack
        recStack[curr] = true;
        for (int i = 0; i < graph[curr].size(); i++) {
            Edge e = graph[curr].get(i);
            if (recStack[e.dest]) {
                // Cycle detected
                return false;
            }
            if (!vis[e.dest]) {
                if (!topSortUtil(graph, e.dest, vis, recStack, s)) {
                    // Propagate the failure if a cycle is detected
                    return false;
                }
            }
        }
        // Remove the current node from the recursion stack
        recStack[curr] = false;
        s.push(curr);
        return true;
    }
}
