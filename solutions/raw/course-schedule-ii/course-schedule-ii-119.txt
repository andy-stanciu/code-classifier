class Solution {

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        int n = prerequisites.length;
        List<List<Integer>> adjList = new ArrayList<>();
        for (int i = 0; i < numCourses; i++) {
            adjList.add(new ArrayList<>());
        }
        for (int i = 0; i < n; i++) {
            int to = prerequisites[i][0];
            int from = prerequisites[i][1];
            adjList.get(from).add(to);
        }
        boolean[] visited = new boolean[numCourses];
        boolean[] recStack = new boolean[numCourses];
        List<Integer> inorder = new ArrayList<>();
        int[] numOrder = new int[numCourses];
        for (int i = 0; i < numCourses; i++) if (dfs(i, adjList, visited, recStack, inorder))
            return new int[0];
        for (int i = 0; i < numCourses; i++) {
            numOrder[numCourses - i - 1] = inorder.get(i);
        }
        return numOrder;
    }

    public boolean dfs(int vertex, List<List<Integer>> adjLst, boolean[] visited, boolean[] recStack, List<Integer> inorder) {
        if (recStack[vertex]) {
            return true;
        }
        if (visited[vertex]) {
            return false;
        }
        visited[vertex] = true;
        recStack[vertex] = true;
        for (int i : adjLst.get(vertex)) {
            if (dfs(i, adjLst, visited, recStack, inorder)) {
                return true;
            }
        }
        inorder.add(vertex);
        recStack[vertex] = false;
        return false;
    }
}
