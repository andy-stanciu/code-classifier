class Solution {

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        Map<Integer, List<Integer>> adjList = new HashMap<>();
        int[] inDegree = new int[numCourses];
        int[] ans = new int[numCourses];
        Queue<Integer> q = new ArrayDeque<>();
        int idx = 0;
        for (int[] edge : prerequisites) {
            //edge[1] is the prerequisite course for edge[0]
            adjList.computeIfAbsent(edge[1], v -> new ArrayList<>()).add(edge[0]);
            //mark the indegree here
            inDegree[edge[0]]++;
        }
        //add all courses without prerequisites to the queue
        for (int i = 0; i < inDegree.length; i++) {
            if (inDegree[i] == 0)
                q.offer(i);
        }
        while (!q.isEmpty()) {
            int currCourse = q.poll();
            ans[idx++] = currCourse;
            //remove this prerequisite course from its neighbors in adjList
            if (adjList.containsKey(currCourse)) {
                for (int coursesInWait : adjList.get(currCourse)) {
                    inDegree[coursesInWait]--;
                    if (inDegree[coursesInWait] == 0)
                        q.offer(coursesInWait);
                }
            }
        }
        return idx == numCourses ? ans : new int[0];
    }
}
