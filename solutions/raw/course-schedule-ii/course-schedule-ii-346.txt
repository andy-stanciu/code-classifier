class Solution {

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        int[] order = new int[numCourses];
        int[] indegree = new int[numCourses];
        HashMap<Integer, List<Integer>> nodes = new HashMap<>();
        for (var p : prerequisites) {
            int from = p[1];
            int to = p[0];
            List<Integer> inNodes = nodes.getOrDefault(from, new ArrayList<Integer>());
            inNodes.add(to);
            nodes.put(from, inNodes);
            indegree[to]++;
        }
        Stack<Integer> availableCourses = new Stack<Integer>();
        for (int i = 0; i < numCourses; i++) {
            if (indegree[i] == 0)
                availableCourses.push(i);
        }
        int index = 0;
        while (!availableCourses.isEmpty()) {
            int course = availableCourses.pop();
            order[index++] = course;
            if (nodes.get(course) == null)
                continue;
            for (int c : nodes.get(course)) {
                if (--indegree[c] == 0) {
                    availableCourses.push(c);
                }
            }
        }
        return index == numCourses ? order : new int[0];
    }
}
