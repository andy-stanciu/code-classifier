class Solution {

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        // Create an array to store the in-degree of each course
        int[] inDegree = new int[numCourses];
        // Create a map to represent the directed graph of prerequisites
        Map<Integer, List<Integer>> graph = new HashMap<>();
        // Iterate through the prerequisites array and update the in-degree and graph
        for (int[] prereq : prerequisites) {
            int course = prereq[0];
            int prereqCourse = prereq[1];
            inDegree[course]++;
            graph.computeIfAbsent(prereqCourse, k -> new ArrayList<>()).add(course);
        }
        // Create a queue to store courses with zero in-degree
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < numCourses; i++) {
            if (inDegree[i] == 0) {
                queue.offer(i);
            }
        }
        // Create an array to store the order of courses
        int[] order = new int[numCourses];
        int index = 0;
        // Perform a BFS on the graph
        while (!queue.isEmpty()) {
            int course = queue.poll();
            order[index++] = course;
            // Update the in-degree of each course and add any courses with zero in-degree to the queue
            if (graph.containsKey(course)) {
                for (int nextCourse : graph.get(course)) {
                    inDegree[nextCourse]--;
                    if (inDegree[nextCourse] == 0) {
                        queue.offer(nextCourse);
                    }
                }
            }
        }
        // If all courses can be taken, return the order of courses, otherwise return an empty array
        return index == numCourses ? order : new int[0];
    }
}
