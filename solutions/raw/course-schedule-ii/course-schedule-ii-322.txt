class Solution {

    public int[] findOrder(int numCourses, int[][] graph) {
        int[] indegree = new int[numCourses];
        List<List<Integer>> li = new ArrayList<>();
        for (int i = 0; i < numCourses; i++) {
            li.add(new ArrayList<>());
        }
        for (int i = 0; i < graph.length; i++) {
            li.get(graph[i][1]).add(graph[i][0]);
            indegree[graph[i][0]]++;
        }
        List<Integer> courses = new ArrayList<>();
        Queue<Integer> q = new ArrayDeque<>();
        for (int i = 0; i < numCourses; i++) if (indegree[i] == 0)
            q.add(i);
        while (!q.isEmpty()) {
            int curr = q.poll();
            courses.add(curr);
            for (int node : li.get(curr)) {
                indegree[node]--;
                if (indegree[node] == 0) {
                    q.add(node);
                }
            }
        }
        if (courses.size() != numCourses)
            return new int[0];
        else
            return courses.stream().mapToInt(i -> i).toArray();
    }
}
