class Solution {

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        List<List<Integer>> adjList = new ArrayList<>();
        int n = numCourses;
        while (n-- != 0) {
            //prep placeholder for adj
            adjList.add(new ArrayList<>());
        }
        //eligible vertex for ordering
        int[] freq = new int[numCourses];
        for (int[] course : prerequisites) {
            //add edges and maintain outbound fre
            int index = course[1];
            int neigh = course[0];
            adjList.get(index).add(neigh);
            freq[neigh] += 1;
        }
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < freq.length; i++) {
            if (freq[i] == 0)
                queue.offer(i);
        }
        int i = 0;
        int[] result = new int[numCourses];
        while (!queue.isEmpty()) {
            int ans = queue.poll();
            result[i++] = ans;
            List<Integer> neighList = adjList.get(ans);
            for (int neigh : neighList) {
                freq[neigh] -= 1;
                if (freq[neigh] == 0)
                    queue.offer(neigh);
            }
        }
        return i != numCourses ? new int[0] : result;
    }
}
