class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int len = nums.length;
        for (int i = 0; i < len; i++) {
            pq.add(nums[i]);
        }
        for (int i = 0; i < len - k; i++) {
            pq.remove();
        }
        return pq.peek();
    }
}
