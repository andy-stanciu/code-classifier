class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>((a, b) -> b - a);
        for (int num : nums) {
            q.offer(num);
        }
        int ans = 0;
        while (!q.isEmpty() && k-- > 0) {
            ans = q.poll();
        }
        return ans;
    }
}
