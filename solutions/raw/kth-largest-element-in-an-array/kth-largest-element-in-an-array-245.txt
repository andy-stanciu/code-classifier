class Solution {

    public int findKthLargest(int[] nums, int k) {
        int kthlargest = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(Comparator.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            pq.add(nums[i]);
        }
        int j = 0;
        while (!pq.isEmpty() && j < k) {
            kthlargest = pq.remove();
            j++;
        }
        return kthlargest;
    }
}
