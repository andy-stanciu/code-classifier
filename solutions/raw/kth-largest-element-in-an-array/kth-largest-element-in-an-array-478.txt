class Solution {

    public int findKthLargest(int[] nums, int k) {
        Map<Integer, Integer> m = new HashMap();
        PriorityQueue<Integer> p = new PriorityQueue<>();
        for (int val : nums) {
            p.add(val);
            if (p.size() > k) {
                p.remove();
            }
        }
        return p.peek();
    }
}
