class Solution {

    public int findKthLargest(int[] nums, int k) {
        int[] arr = new int[20001];
        for (int n : nums) arr[n + 10000]++;
        for (int i = 20000; i >= 0; i--) {
            int count = arr[i];
            if (count >= k)
                return i - 10000;
            k -= count;
        }
        return -1;
    }
}
