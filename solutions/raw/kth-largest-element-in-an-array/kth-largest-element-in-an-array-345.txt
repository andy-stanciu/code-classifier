class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        SortedSet<Integer> set = new TreeSet<>((a, b) -> b - a);
        for (int num : nums) pq.add(num);
        int i = 0;
        while (i++ < k) {
            set.add(pq.remove());
        }
        return set.last();
    }
}
