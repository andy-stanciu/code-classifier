class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> minheap = new PriorityQueue<>();
        int counter = 0;
        for (int x : nums) {
            if (counter >= k) {
                if (x > minheap.peek()) {
                    minheap.poll();
                    minheap.add(x);
                }
            } else
                minheap.add(x);
            counter++;
        }
        return minheap.peek();
    }
}
