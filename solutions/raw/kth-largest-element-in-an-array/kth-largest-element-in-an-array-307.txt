class Solution {

    public int findKthLargest(int[] nums, int k) {
        int[] hash = new int[20001];
        for (int i : nums) {
            hash[i + 10000]++;
        }
        int c = 0;
        for (int i = 0; i < hash.length; i++) {
            c += hash[i];
            if (c >= nums.length - k + 1)
                return i - 10000;
        }
        return -1;
    }
}
