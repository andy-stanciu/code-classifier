class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            minHeap.add(nums[i]);
        }
        int index = k;
        while (index < nums.length) {
            if (minHeap.peek() < nums[index]) {
                minHeap.poll();
                minHeap.add(nums[index]);
            }
            index++;
        }
        return minHeap.peek();
    }
}
