class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> pQueue = new PriorityQueue<>(Collections.reverseOrder());
        for (int num : nums) {
            pQueue.add(num);
        }
        int top = 0;
        for (int i = 0; i < k; i++) {
            top = pQueue.poll();
        }
        return top;
    }
}
