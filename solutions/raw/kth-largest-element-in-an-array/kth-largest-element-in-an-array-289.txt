class Solution {

    PriorityQueue<Integer> pq;

    public void add(int val, int k) {
        if (pq.size() < k) {
            pq.add(val);
        } else if (val > pq.peek()) {
            pq.poll();
            pq.add(val);
        }
    }

    public int findKthLargest(int[] nums, int k) {
        pq = new PriorityQueue<>();
        for (int i : nums) {
            add(i, k);
        }
        return pq.peek();
    }
}
