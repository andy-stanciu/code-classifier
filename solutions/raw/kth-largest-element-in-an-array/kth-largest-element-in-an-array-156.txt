class Solution {

    public int findKthLargest(int[] nums, int k) {
        PrioorityQueue<Integer> pq = new PrioorityQueue<>();
        for (int el : nums) {
            pq.add(el);
            while (pq.size() > k) {
                pq.poll();
            }
        }
        return pq.poll();
    }
}
