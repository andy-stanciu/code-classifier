class Solution {

    static final int Max_Num = 10000;

    public int findKthLargest(int[] nums, int k) {
        int[] counter = new int[Max_Num * 2 + 1];
        for (int i = 0; i < nums.length; i++) {
            counter[nums[i] + Max_Num]++;
        }
        int j = Max_Num * 2;
        while (k > counter[j]) {
            k -= counter[j--];
        }
        return j - Max_Num;
    }
}
