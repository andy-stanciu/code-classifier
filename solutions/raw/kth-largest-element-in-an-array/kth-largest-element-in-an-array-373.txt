class Solution {

    public int findKthLargest(int[] nums, int k) {
        //w/o sorting
        //Prority Queue can be used to implement min heap
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            //adding all the elements
            pq.add(nums[i]);
            if (pq.size() > k)
                //as the size of pq  will become more than k, we\'ll remove the the smallest value because it\'s min heap :)
                pq.remove();
        }
        //returning the last element
        return pq.peek();
    }
}
