class Solution {

    public int findKthLargest(int[] nums, int k) {
        Arrays.sort(nums);
        int max = 0, n = nums.length;
        // calculating the index of kth element
        int r = n - k;
        for (int i = 0; i <= r; i++) {
            max = nums[i];
        }
        return max;
    }
}
