class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            queue.add(nums[i]);
        }
        int ans = nums.length - k;
        while (ans > 0) {
            queue.poll();
            ans--;
        }
        return queue.peek();
    }
}
