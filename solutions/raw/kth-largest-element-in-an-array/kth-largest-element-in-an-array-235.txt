class Solution {

    public int findKthLargest(int[] nums, int k) {
        int res = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>(Collections.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            pq.add(nums[i]);
        }
        int loop = 0;
        while (loop < k) {
            loop++;
            res = pq.poll();
        }
        return res;
    }
}
