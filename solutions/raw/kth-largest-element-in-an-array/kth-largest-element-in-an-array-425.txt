class Solution {

    public int findKthLargest(int[] nums, int k) {
        if (nums.length == 1) {
            return nums[0];
        }
        Arrays.sort(nums);
        int largest = Integer.MIN_VALUE;
        int count = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (i == 0) {
                break;
            }
            if (k == 1) {
                return nums[nums.length - 1];
            }
            if (nums[i] >= nums[i - 1]) {
                largest = nums[i - 1];
                count++;
                if (count == k) {
                    return largest;
                }
            }
        }
        return -1;
    }
}
