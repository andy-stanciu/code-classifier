class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> heap = new PriorityQueue<>();
        for (int n : nums) {
            if (heap.size() < k)
                heap.add(n);
            else if (n > heap.peek()) {
                heap.poll();
                heap.add(n);
            }
        }
        return heap.peek();
    }
}
