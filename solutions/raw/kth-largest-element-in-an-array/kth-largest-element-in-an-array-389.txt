class Solution {

    public int findKthLargest(int[] nums, int k) {
        /**
         * Using Min heap - This solution is uses min heap of size k, so Space Complexity O(K))
         */
        //By default Priority queue is min Heap
        Queue<Integer> minHeap = new PriorityQueue<>();
        int answer = 0;
        for (Integer e : nums) {
            minHeap.add(e);
            if (minHeap.size() > k) {
                //If size of min heap is more than K, then poll one, maintain the size of min heap at k
                minHeap.poll();
            }
        }
        return minHeap.poll();
    }
}
