class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>();
        for (int e : nums) {
            q.offer(e);
        }
        for (int i = 0; i < nums.length - k; i++) {
            q.poll();
        }
        return q.poll();
    }
}
