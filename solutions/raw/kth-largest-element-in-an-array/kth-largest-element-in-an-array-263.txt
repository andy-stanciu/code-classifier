class Solution {

    public int findKthLargest(int[] nums, int k) {
        int n = nums.length;
        PriorityQueue<Integer> minHeap = new PriorityQueue<>((a, b) -> (b - a));
        for (int i = 0; i < n; i++) {
            minHeap.add(nums[i]);
        }
        while (k - 1 > 0) {
            minHeap.poll();
            k--;
        }
        return minHeap.peek();
    }
}
