class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) pq.add(nums[i]);
        // 6 - 2 = 4
        int find = nums.length - k;
        // 1 2 3 4 5 6
        while (find > -1) {
            if (find == 0) {
                return pq.peek();
            }
            pq.poll();
            find--;
        }
        return 0;
    }
}
