public class Solution {

    public int findKthLargest(int[] nums, int k) {
        // Convert k-th largest to the index in a sorted array (0-based)
        return quickSelect(nums, 0, nums.length - 1, nums.length - k);
    }

    private int quickSelect(int[] nums, int left, int right, int k) {
        // Base case: if the search range has one element
        if (left == right)
            return nums[left];
        // Randomly choose a pivot index within the current range
        int pIndex = new Random().nextInt(right - left + 1) + left;
        // Partition around the chosen pivot and get its sorted position
        pIndex = partition(nums, left, right, pIndex);
        // Check if pivot is at the k-th position
        if (pIndex == k)
            return nums[k];
        else // If pivot is too small, search in the right half
        if (pIndex < k)
            return quickSelect(nums, pIndex + 1, right, k);
        else
            // If pivot is too large, search in the left half
            return quickSelect(nums, left, pIndex - 1, k);
    }

    private int partition(int[] nums, int left, int right, int pIndex) {
        int pivot = nums[pIndex];
        // Move pivot to the end of the range
        swap(nums, pIndex, right);
        pIndex = left;
        // Move elements less than pivot to the left of the partition
        for (int i = left; i < right; i++) {
            if (nums[i] < pivot) {
                swap(nums, i, pIndex);
                pIndex++;
            }
        }
        // Move pivot to its final position
        swap(nums, pIndex, right);
        return pIndex;
    }

    private void swap(int[] nums, int x, int y) {
        int temp = nums[x];
        nums[x] = nums[y];
        nums[y] = temp;
    }
}
