class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> minH = new PriorityQueue<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            minH.offer(nums[i]);
            if (minH.size() > k) {
                minH.poll();
            }
        }
        return minH.peek();
    }
}
