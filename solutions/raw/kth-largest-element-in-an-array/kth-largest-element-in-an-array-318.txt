class Solution {

    public int findKthLargest(int[] nums, int k) {
        Arrays.sort(nums);
        int n = nums.length - k;
        int max = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (n == i) {
                max = nums[i];
            }
        }
        return max;
    }
}
