class Solution {

    public int findKthLargest(int[] nums, int k) {
        // Create a min heap and if size increases than k then remove and return the top element ,the top element is kth largest element
        PriorityQueue<Integer> minheap = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            minheap.add(nums[i]);
            if (minheap.size() > k)
                minheap.remove();
        }
        /* while(k!=1){
        maxheap.remove();
        k--;
      }*/
        return minheap.peek();
    }
}
