Solution {

    public int findKthLargest(int[] nums, int k) {
        // here k th largest element so we have to intoduce min. heap
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            pq.add(nums[i]);
            if (pq.size() > k) {
                pq.remove();
            }
        }
        return pq.peek();
    }
}
