class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> Pqueue = new PriorityQueue<Integer>();
        for (int i = 0; i < nums.length; i++) {
            Pqueue.add(nums[i]);
            if (Pqueue.size() > k) {
                Pqueue.poll();
            }
        }
        return Pqueue.peek();
    }
}
