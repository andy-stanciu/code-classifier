class Solution {

    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>();
        int n = nums.length;
        if (n == 1)
            return nums[0];
        for (int i = 0; i < k; i++) pq.add(nums[i]);
        for (int i = k; i < n; i++) {
            if (nums[i] > pq.peek()) {
                pq.poll();
                pq.add(nums[i]);
            }
        }
        return pq.peek();
    }
}
