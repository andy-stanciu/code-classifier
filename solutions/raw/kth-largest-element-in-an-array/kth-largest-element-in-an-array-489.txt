class Solution {

    public int findKthLargest(int[] nums, int k) {
        final PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Collections.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            maxHeap.add(nums[i]);
        }
        int kthNumber = 0;
        for (int i = 0; i < k; i++) {
            int number = maxHeap.poll();
            if (i + 1 == k) {
                kthNumber = number;
            }
        }
        return kthNumber;
    }
}
