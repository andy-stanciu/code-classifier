class Solution {

    public int findKthLargest(int[] nums, int k) {
        Queue<Integer> q = new PriorityQueue<>(Comparator.reverseOrder());
        q.addAll(Arrays.stream(nums).boxed().toList());
        for (int i = 1; i < k; i++) {
            q.poll();
        }
        return q.peek();
    }
}
