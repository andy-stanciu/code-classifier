class Solution {

    public static boolean check_symmetric(TreeNode L_root, TreeNode R_root) {
        if (R_root == null && L_root == null)
            return true;
        if (R_root == null || L_root == null)
            return false;
        return L_root.val == R_root.val && check_symmetric(L_root.left, R_root.right) && check_symmetric(L_root.right, R_root.left);
    }

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return check_symmetric(root.left, root.right);
    }
}
