/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return isSymmetric(root.left, root.right);
    }

    public boolean isSymmetric(TreeNode first, TreeNode second) {
        if (first == null && second == null)
            return true;
        if ((first != null && second == null) || (first == null && second != null) || (first.val != second.val))
            return false;
        if (isSymmetric(first.left, second.right) && isSymmetric(first.right, second.left))
            return true;
        return false;
    }
}
