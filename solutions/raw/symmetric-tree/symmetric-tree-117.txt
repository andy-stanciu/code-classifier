/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean checksymmetric(TreeNode n1, TreeNode n2) {
        if (n1 == null || n2 == null) {
            return n1 == n2;
        }
        boolean c1 = checksymmetric(n1.left, n2.right);
        boolean c2 = checksymmetric(n1.right, n2.left);
        return (n1.val == n2.val) && c1 && c2;
    }

    public boolean isSymmetric(TreeNode root) {
        if (root == null) {
            return false;
        }
        boolean check = checksymmetric(root.left, root.right);
        return check;
    }
}
