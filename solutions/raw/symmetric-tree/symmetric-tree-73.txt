class Solution {

    private boolean depthFirstTraversal(TreeNode node1, TreeNode node2) {
        if (node1 == null && node2 == null) {
            return true;
        }
        if (node1 == null || node2 == null) {
            return false;
        }
        return (node1.val == node2.val) && (depthFirstTraversal(node1.left, node2.right)) && (depthFirstTraversal(node1.right, node2.left));
    }

    public boolean isSymmetric(TreeNode root) {
        return depthFirstTraversal(root, root);
    }
}
