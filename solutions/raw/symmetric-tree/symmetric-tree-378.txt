class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return isMirroring(root.left, root.right);
    }

    public static boolean isMirroring(TreeNode LTree, TreeNode RTree) {
        if (LTree == null || RTree == null)
            return LTree == RTree;
        return (LTree.val == RTree.val) && isMirroring(LTree.left, RTree.right) && isMirroring(LTree.right, RTree.left);
    }
}
