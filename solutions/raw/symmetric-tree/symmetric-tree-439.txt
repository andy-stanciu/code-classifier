/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return mirror(root.left, root.right);
    }

    public boolean mirror(TreeNode lroot, TreeNode rroot) {
        if (lroot == null || rroot == null)
            return lroot == rroot;
        if (lroot.val != rroot.val)
            return false;
        return mirror(lroot.left, rroot.right) && mirror(lroot.right, rroot.left);
    }
}
