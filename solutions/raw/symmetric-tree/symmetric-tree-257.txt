class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null) {
            return true;
        }
        return isSymmetricNodes(root.left, root.right);
    }

    private boolean isSymmetricNodes(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (p == null || q == null) {
            return false;
        }
        return p.val == q.val && isSymmetricNodes(p.right, q.left) && isSymmetricNodes(p.left, q.right);
    }
}
