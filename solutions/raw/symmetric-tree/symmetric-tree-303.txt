/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return symmetric(root.left, root.right);
    }

    public boolean symmetric(TreeNode lef, TreeNode righ) {
        if (lef == null && righ == null)
            return true;
        if (lef == null || righ == null)
            return false;
        if (lef.val != righ.val)
            return false;
        return symmetric(lef.left, righ.right) && symmetric(lef.right, righ.left);
    }
}
