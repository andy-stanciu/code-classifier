/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean traversal(TreeNode p, TreeNode q) {
        if (p == null || q == null) {
            if (p == null && q == null)
                return true;
            return false;
        }
        boolean l = traversal(p.left, q.right);
        boolean r = traversal(p.right, q.left);
        return l && r && (p.val == q.val);
    }

    public boolean isSymmetric(TreeNode root) {
        return traversal(root, root);
    }
}
