class Solution {

    static boolean symmetricity(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            if (left == null && right == null)
                return true;
            else
                return false;
        }
        if (left.val != right.val)
            return false;
        boolean a = symmetricity(left.left, right.right);
        boolean b = symmetricity(left.right, right.left);
        return a && b;
    }

    public boolean isSymmetric(TreeNode root) {
        return symmetricity(root.left, root.right);
    }
}
