class Solution {

    public boolean symmetricityCheck(TreeNode a, TreeNode b) {
        //base case of recursion
        if (a == null && b == null)
            return true;
        if (a == null && b != null || a != null && b == null || a.val != b.val)
            return false;
        //general case of recursion
        if (a.val == b.val)
            return symmetricityCheck(a.left, b.right) && symmetricityCheck(a.right, b.left);
        //agar value equal hi nai hai toh age jaane ki zarurat nai hai we return false..
        return false;
    }

    public boolean isSymmetric(TreeNode root) {
        //we will be needing 2 references from which one reference will be on either side of the mirror
        //initially both. references point to the same root node
        TreeNode a = root;
        TreeNode b = root;
        return symmetricityCheck(a, b);
    }
}
