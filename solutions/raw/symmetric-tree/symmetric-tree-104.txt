class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return false;
        return traverse(root.left, root.right);
    }

    public boolean traverse(TreeNode l, TreeNode r) {
        if (l == null || r == null)
            return l == r;
        if (l.val != r.val)
            return false;
        return traverse(l.left, r.right) && traverse(l.right, r.left);
    }
}
