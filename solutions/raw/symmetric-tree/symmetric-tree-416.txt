class Solution {

    public boolean isSymmetric(TreeNode root) {
        return similarTree(root.left, root.right);
    }

    public boolean similarTree(TreeNode p1, TreeNode p2) {
        if (p1 == null && p2 == null)
            return true;
        if (p1 != null && p2 != null && p1.val == p2.val) {
            return similarTree(p1.left, p2.right) && similarTree(p1.right, p2.left);
        }
        return false;
    }
}
