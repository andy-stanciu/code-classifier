class Solution {

    public boolean result(TreeNode leftNode, TreeNode rightNode) {
        if (leftNode == null || rightNode == null)
            return (leftNode == rightNode);
        return (leftNode.val == rightNode.val) && result(leftNode.left, rightNode.right) && result(leftNode.right, rightNode.left);
    }

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return false;
        return result(root.left, root.right);
    }
}
