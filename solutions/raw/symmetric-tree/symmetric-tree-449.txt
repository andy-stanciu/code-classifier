class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return isMirror(root.left, root.right);
    }

    public boolean isMirror(TreeNode lN, TreeNode rN) {
        if (lN == null && rN == null)
            return true;
        if (lN == null || rN == null || lN.val != rN.val)
            return false;
        return (lN.val == rN.val) && isMirror(lN.left, rN.right) && isMirror(lN.right, rN.left);
    }
}
