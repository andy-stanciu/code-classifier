class Solution {

    public boolean isSymmetric(TreeNode root) {
        return root == null || is(root.left, root.right);
    }

    static boolean is(TreeNode l, TreeNode r) {
        // so if(l==null) then r==null to make them symmetrical and vice versa
        if (l == null || r == null)
            return l == r;
        // check if the left node and right node val are equal if not then return false
        if (l.val != r.val)
            return false;
        // then check left of the left node and right of right node are equal or not and    //similarly for right of left node and left of right node are equal . if both yes the return yes
        return is(l.left, r.right) && is(l.right, r.left);
    }
}
