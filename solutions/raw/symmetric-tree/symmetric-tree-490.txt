class Solution {

    boolean x = true;

    public boolean isSymmetric(TreeNode root) {
        return isSymmetric(root.left, root.right);
    }

    private boolean isSymmetric(TreeNode n, TreeNode m) {
        if (m == null || n == null)
            return m == n;
        if (n.val != m.val)
            return false;
        return isSymmetric(n.left, m.right) && isSymmetric(m.left, n.right);
    }
}
