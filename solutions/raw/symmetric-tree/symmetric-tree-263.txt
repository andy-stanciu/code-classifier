/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean isSymmUtil(TreeNode leftRoot, TreeNode rightRoot) {
        if (leftRoot == null || rightRoot == null) {
            return leftRoot == rightRoot;
        }
        if (leftRoot.val != rightRoot.val) {
            return false;
        }
        boolean l = isSymmUtil(leftRoot.left, rightRoot.right);
        boolean r = isSymmUtil(leftRoot.right, rightRoot.left);
        return (l && r);
    }

    public boolean isSymmetric(TreeNode root) {
        if (root == null) {
            return true;
        }
        return isSymmUtil(root.left, root.right);
    }
}
