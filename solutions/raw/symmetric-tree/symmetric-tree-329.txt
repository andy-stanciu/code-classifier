// Definition for a binary tree node.
class TreeNode {

    int val;

    TreeNode left;

    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null) {
            return true;
        }
        return isMirror(root.left, root.right);
    }

    private boolean isMirror(TreeNode lef, TreeNode rht) {
        if (lef == null && rht == null) {
            return true;
        }
        if (lef == null || rht == null) {
            return false;
        }
        return (lef.val == rht.val) && isMirror(lef.left, rht.right) && isMirror(lef.right, rht.left);
    }
}
