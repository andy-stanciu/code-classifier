/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        else
            return ifMirror(root.left, root.right);
    }

    public boolean ifMirror(TreeNode leftNode, TreeNode rightNode) {
        if (leftNode == null && rightNode == null)
            return true;
        else if (leftNode == null && rightNode != null)
            return false;
        else if (leftNode != null && rightNode == null)
            return false;
        else {
            return (leftNode.val == rightNode.val) && ifMirror(leftNode.left, rightNode.right) && ifMirror(leftNode.right, rightNode.left);
        }
    }
}
