/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static boolean fn(TreeNode r, TreeNode l) {
        if (r == null && l == null)
            return true;
        if (r == null || l == null)
            return false;
        if (r.val != l.val)
            return false;
        return (fn(r.left, l.right) && fn(r.right, l.left));
    }

    public boolean isSymmetric(TreeNode r) {
        return fn(r.left, r.right);
    }
}
