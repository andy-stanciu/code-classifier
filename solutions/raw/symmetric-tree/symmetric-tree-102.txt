class Solution {

    public boolean isSymmetric(TreeNode root) {
        if (root == null)
            return true;
        return checkLeftAndRightNodes(root.left, root.right);
    }

    public boolean checkLeftAndRightNodes(TreeNode a, TreeNode b) {
        if (a == null && b == null)
            return true;
        if (a == null || b == null)
            return false;
        if (a.val != b.val)
            return false;
        return checkLeftAndRightNodes(a.left, b.right) && checkLeftAndRightNodes(a.right, b.left);
    }
}
