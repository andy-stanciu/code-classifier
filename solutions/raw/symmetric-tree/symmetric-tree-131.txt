class Solution {

    public boolean helper(TreeNode rootLeft, TreeNode rootRight) {
        if (rootLeft == null && rootRight == null)
            return true;
        if (rootLeft == null || rootRight == null)
            return false;
        boolean ans = (rootLeft.val == rootRight.val) ? true : false;
        return ans && helper(rootLeft.left, rootRight.right) && helper(rootLeft.right, rootRight.left);
    }

    public boolean isSymmetric(TreeNode root) {
        return helper(root.left, root.right);
    }
}
