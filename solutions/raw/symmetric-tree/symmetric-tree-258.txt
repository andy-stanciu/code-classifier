/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean mirror(TreeNode lTree, TreeNode rTree) {
        if (lTree == null && rTree == null) {
            return true;
        }
        if (lTree == null || rTree == null) {
            return false;
        }
        return (lTree.val == rTree.val) && mirror(lTree.left, rTree.right) && mirror(lTree.right, rTree.left);
    }

    public boolean isSymmetric(TreeNode root) {
        if (root == null) {
            return true;
        }
        return mirror(root.left, root.right);
    }
}
