class Solution {

    public void gameOfLife(int[][] board) {
        boolean[][] matrix = new boolean[board.length][board[0].length];
        gameOfLife(board, 0, 0, board.length, board[0].length, matrix);
    }

    public void gameOfLife(int[][] board, int row, int col, int m, int n, boolean[][] visited) {
        if (row < 0 || row > m - 1 || col < 0 || col > n - 1 || visited[row][col] == true)
            return;
        visited[row][col] = true;
        //Pattern
        int count = 0;
        if (row != 0 && board[row - 1][col] == 1)
            count++;
        if (row != m - 1 && board[row + 1][col] == 1)
            count++;
        if (col != 0 && board[row][col - 1] == 1)
            count++;
        if (col != n - 1 && board[row][col + 1] == 1)
            count++;
        if (row != 0 && col != 0 && board[row - 1][col - 1] == 1)
            count++;
        if (row != m - 1 && col != n - 1 && board[row + 1][col + 1] == 1)
            count++;
        if (row != 0 && col != n - 1 && board[row - 1][col + 1] == 1)
            count++;
        if (row != m - 1 && col != 0 && board[row + 1][col - 1] == 1)
            count++;
        gameOfLife(board, row - 1, col, m, n, visited);
        gameOfLife(board, row + 1, col, m, n, visited);
        gameOfLife(board, row, col - 1, m, n, visited);
        gameOfLife(board, row, col + 1, m, n, visited);
        gameOfLife(board, row - 1, col - 1, m, n, visited);
        gameOfLife(board, row + 1, col + 1, m, n, visited);
        gameOfLife(board, row - 1, col + 1, m, n, visited);
        gameOfLife(board, row + 1, col - 1, m, n, visited);
        if (board[row][col] == 1 && count < 2) {
            board[row][col] = 0;
        } else if (board[row][col] == 1 && (count == 2 || count == 3)) {
            board[row][col] = 1;
        } else if (board[row][col] == 1 && count > 3) {
            board[row][col] = 0;
        } else if (board[row][col] == 0 && count == 3) {
            board[row][col] = 1;
        }
    }
}
