class Solution {

    public void gameOfLife(int[][] nums) {
        int n = nums.length;
        int m = nums[0].length;
        int[][] arr = new int[n][m];
        int[] x = { -1, -1, -1, 1, 1, 1, 0, 0 };
        int[] y = { -1, 0, 1, -1, 0, 1, -1, 1 };
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int count = 0;
                for (int p = 0; p < 8; p++) {
                    int row = i + x[p];
                    int col = j + y[p];
                    if (row < n && row >= 0 && col < m && col >= 0 && nums[row][col] == 1)
                        count++;
                }
                arr[i][j] = count;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (arr[i][j] < 2 && nums[i][j] == 1)
                    nums[i][j] = 0;
                else if ((arr[i][j] == 2 || arr[i][j] == 3) && nums[i][j] == 1)
                    nums[i][j] = 1;
                else if (arr[i][j] == 3 && nums[i][j] == 0)
                    nums[i][j] = 1;
                else if (arr[i][j] > 3 && nums[i][j] == 1)
                    nums[i][j] = 0;
            }
        }
    }
}
