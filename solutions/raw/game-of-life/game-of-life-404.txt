class Solution {

    public void gameOfLife(int[][] board) {
        int[][] check = new int[board.length + 2][board[0].length + 2];
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j] == 1) {
                    check[i][j] += 1;
                    check[i][j + 1] += 1;
                    check[i][j + 2] += 1;
                    check[i + 1][j] += 1;
                    check[i + 1][j + 2] += 1;
                    check[i + 2][j] += 1;
                    check[i + 2][j + 1] += 1;
                    check[i + 2][j + 2] += 1;
                }
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j] == 0 && check[i + 1][j + 1] == 3) {
                    board[i][j] = 1;
                }
                if (board[i][j] == 1) {
                    if (check[i + 1][j + 1] > 3 || check[i + 1][j + 1] < 2) {
                        board[i][j] = 0;
                    }
                }
            }
        }
        return;
    }
}
