class Solution {

    public void gameOfLife(int[][] board) {
        int[][] res = new int[board.length + 2][board[0].length + 2];
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                res[i + 1][j + 1] = board[i][j];
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                board[i][j] = helper(res, i + 1, j + 1);
            }
        }
    }

    static int helper(int[][] board, int n, int m) {
        int[] dirs1 = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dirs2 = { -1, 0, 1, -1, 1, -1, 0, 1 };
        int sum = 0;
        int i = 0, j = dirs1.length;
        for (i = 0; i < j; i++) {
            sum += board[n + dirs1[i]][m + dirs2[i]];
        }
        if (sum < 2 && board[n][m] == 1)
            return 0;
        else if ((sum == 3 || sum == 2) && board[n][m] == 1)
            return 1;
        else if (sum > 3 && board[n][m] == 1)
            return 0;
        else if (sum == 3 && board[n][m] == 0)
            return 1;
        return board[n][m];
    }
}
