class Solution {

    public void gameOfLife(int[][] board) {
        int n = board.length;
        int m = board[0].length;
        int[][] ans = new int[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int c = l(board, i, j) + r(board, i, j) + u(board, i, j) + d(board, i, j) + ul(board, i, j) + ur(board, i, j) + dl(board, i, j) + dr(board, i, j);
                if (board[i][j] == 1 && (c < 2 || c > 3)) {
                    ans[i][j] = 0;
                } else if (board[i][j] == 1 && (c == 2 || c == 3)) {
                    ans[i][j] = 1;
                } else if (board[i][j] == 0 && c == 3) {
                    ans[i][j] = 1;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                board[i][j] = ans[i][j];
            }
        }
    }

    public int l(int[][] board, int i, int j) {
        if (j > 0 && board[i][j - 1] == 1) {
            return 1;
        }
        return 0;
    }

    public int r(int[][] board, int i, int j) {
        if (j < board[0].length - 1 && board[i][j + 1] == 1) {
            return 1;
        }
        return 0;
    }

    public int u(int[][] board, int i, int j) {
        if (i > 0 && board[i - 1][j] == 1) {
            return 1;
        }
        return 0;
    }

    public int d(int[][] board, int i, int j) {
        if (i < board.length - 1 && board[i + 1][j] == 1) {
            return 1;
        }
        return 0;
    }

    public int ul(int[][] board, int i, int j) {
        if (i > 0 && j > 0 && board[i - 1][j - 1] == 1) {
            return 1;
        }
        return 0;
    }

    public int ur(int[][] board, int i, int j) {
        if (i > 0 && j < board[0].length - 1 && board[i - 1][j + 1] == 1) {
            return 1;
        }
        return 0;
    }

    public int dl(int[][] board, int i, int j) {
        if (i < board.length - 1 && j > 0 && board[i + 1][j - 1] == 1) {
            return 1;
        }
        return 0;
    }

    public int dr(int[][] board, int i, int j) {
        if (i < board.length - 1 && j < board[0].length - 1 && board[i + 1][j + 1] == 1) {
            return 1;
        }
        return 0;
    }
}
