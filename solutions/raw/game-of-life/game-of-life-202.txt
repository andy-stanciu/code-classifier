class Solution {

    public static int getSum(int[][] board, int i, int j) {
        int n = board.length, m = board[0].length;
        int[] dir = { -1, 0, 1 };
        int res = 0;
        for (int x : dir) {
            for (int y : dir) {
                if (x != 0 || y != 0) {
                    int indx = i + x;
                    int indy = j + y;
                    if (indx >= 0 && indx < n && indy >= 0 && indy < m) {
                        res += board[indx][indy];
                    }
                }
            }
        }
        return res;
    }

    public void gameOfLife(int[][] board) {
        int n = board.length, m = board[0].length;
        int[][] mat = new int[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int res = this.getSum(board, i, j);
                if (board[i][j] == 1) {
                    if (res < 2 || res > 3) {
                        mat[i][j] = 0;
                    } else {
                        mat[i][j] = 1;
                    }
                } else {
                    if (res == 3) {
                        mat[i][j] = 1;
                    }
                }
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                board[i][j] = mat[i][j];
            }
        }
    }
}
