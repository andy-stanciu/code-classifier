class Solution {

    int[][] board;

    public void gameOfLife(int[][] board) {
        this.board = board;
        int rows = board.length, cols = board[0].length;
        int[][] temp = new int[rows][cols];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                int liveneighbours = neighbour(i - 1, j - 1) + neighbour(i, j - 1) + neighbour(i - 1, j) + neighbour(i + 1, j + 1) + neighbour(i + 1, j) + neighbour(i, j + 1) + neighbour(i - 1, j + 1) + neighbour(i + 1, j - 1);
                if (board[i][j] == 1) {
                    temp[i][j] = (liveneighbours < 2 || liveneighbours > 3) ? 0 : 1;
                } else {
                    temp[i][j] = (liveneighbours == 3) ? 1 : 0;
                }
            }
        }
        for (int i = 0; i < rows; i++) {
            board[i] = temp[i].clone();
        }
    }

    int neighbour(int i, int j) {
        if (i < 0 || i >= board.length || j < 0 || j >= board[0].length || board[i][j] == 0) {
            return 0;
        }
        return 1;
    }
}
