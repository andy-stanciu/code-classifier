class Solution {

    public void gameOfLife(int[][] arr) {
        int m = arr.length;
        int n = arr[0].length;
        int[][] sol = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int dies = 0;
                int live = 0;
                if (i - 1 >= 0) {
                    if (arr[i - 1][j] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (j - 1 >= 0) {
                    if (arr[i][j - 1] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (i + 1 < m) {
                    if (arr[i + 1][j] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (j + 1 < n) {
                    if (arr[i][j + 1] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (i - 1 >= 0 && j - 1 >= 0) {
                    if (arr[i - 1][j - 1] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (i + 1 < m && j + 1 < n) {
                    if (arr[i + 1][j + 1] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (i - 1 >= 0 && j + 1 < n) {
                    if (arr[i - 1][j + 1] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (i + 1 < m && j - 1 >= 0) {
                    if (arr[i + 1][j - 1] == 0) {
                        dies++;
                    } else {
                        live++;
                    }
                }
                if (arr[i][j] == 1 && live < 2) {
                    sol[i][j] = 0;
                } else if (arr[i][j] == 0 && live == 3) {
                    sol[i][j] = 1;
                } else if (arr[i][j] == 1 && (live == 2 || live == 3)) {
                    sol[i][j] = 1;
                } else if (arr[i][j] == 1 && live > 3) {
                    sol[i][j] = 0;
                }
            }
        }
        for (int i = 0; i < sol.length; i++) {
            for (int j = 0; j < sol[0].length; j++) {
                arr[i][j] = sol[i][j];
            }
        }
    }
}
