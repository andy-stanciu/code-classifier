class Solution {

    public static void gameOfLife(int[][] board) {
        int row = board.length;
        int col = board[0].length;
        int[][] newBoard = new int[row][col];
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                int zeroCount = 0;
                int oneCount = 0;
                int current = board[i][j];
                for (int itrIndexX = i - 1; itrIndexX <= i + 1; itrIndexX++) {
                    for (int itrIndexY = j - 1; itrIndexY <= j + 1; itrIndexY++) {
                        if ((itrIndexX >= 0 && itrIndexX < row) && (itrIndexY >= 0 && itrIndexY < col)) {
                            if (board[itrIndexX][itrIndexY] == 0) {
                                zeroCount++;
                            } else {
                                oneCount++;
                            }
                        }
                    }
                }
                if (current == 1)
                    oneCount -= 1;
                if (current == 0)
                    zeroCount -= 1;
                if (current == 0 && oneCount == 3) {
                    newBoard[i][j] = 1;
                }
                if (current == 1) {
                    if (oneCount < 2) {
                        newBoard[i][j] = 0;
                    } else if (oneCount == 2 || oneCount == 3) {
                        newBoard[i][j] = 1;
                    } else {
                        newBoard[i][j] = 0;
                    }
                }
            }
        }
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                board[i][j] = newBoard[i][j];
            }
        }
    }
}
