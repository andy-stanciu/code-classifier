class Solution {

    public void gameOfLife(int[][] board) {
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                boolean live = board[i][j] % 10 == 1;
                int neighbors = countLiveNeighbors(board, i, j);
                if (live)
                    board[i][j] = (neighbors == 2 || neighbors == 3) ? 11 : 1;
                else
                    board[i][j] = (neighbors == 3) ? 10 : 0;
            }
        }
        for (int i = 0; i < board.length; i++) for (int j = 0; j < board[i].length; j++) board[i][j] /= 10;
    }

    public int countLiveNeighbors(int[][] board, int row, int col) {
        int result = 0;
        for (int i = Math.max(0, row - 1); i < Math.min(board.length, row + 2); i++) for (int j = Math.max(0, col - 1); j < Math.min(board[row].length, col + 2); j++) if (i != row || j != col)
            result += board[i][j] % 10;
        return result;
    }
}
