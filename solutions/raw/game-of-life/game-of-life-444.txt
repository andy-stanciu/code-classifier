class Solution {

    public void gameOfLife(int[][] board) {
        int m = board.length;
        int n = board[0].length;
        int[][] res = new int[m][n];
        int ones = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                ones = 0;
                if (i - 1 >= 0 && j - 1 >= 0 && board[i - 1][j - 1] == 1)
                    ones++;
                if (i - 1 >= 0 && board[i - 1][j] == 1)
                    ones++;
                if (i - 1 >= 0 && j + 1 < n && board[i - 1][j + 1] == 1)
                    ones++;
                if (j - 1 >= 0 && board[i][j - 1] == 1)
                    ones++;
                if (j + 1 < n && board[i][j + 1] == 1)
                    ones++;
                if (i + 1 < m && j - 1 >= 0 && board[i + 1][j - 1] == 1)
                    ones++;
                if (i + 1 < m && board[i + 1][j] == 1)
                    ones++;
                if (i + 1 < m && j + 1 < n && board[i + 1][j + 1] == 1)
                    ones++;
                if (board[i][j] == 1 && ones < 2)
                    res[i][j] = 0;
                else if (board[i][j] == 1 && ones < 4)
                    res[i][j] = 1;
                else if (board[i][j] == 1 && ones > 3)
                    res[i][j] = 0;
                else if (board[i][j] == 0 && ones == 3)
                    res[i][j] = 1;
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                board[i][j] = res[i][j];
            }
        }
    }
}
