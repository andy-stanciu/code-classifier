class Solution {

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> list = new ArrayList<>();
        // sort the array
        Arrays.sort(candidates);
        // recrusion
        findTheCombination(0, candidates, target, list, new ArrayList<>());
        // return list
        return list;
    }

    // recrusion function
    private void findTheCombination(int ind, int[] candidates, int target, List<List<Integer>> list, List<Integer> ds) {
        if (target == 0) {
            list.add(new ArrayList<>(ds));
            // return back
            return;
        }
        // iterate over the array
        for (int idx = ind; idx < candidates.length; idx++) {
            if (idx > ind && candidates[idx] == candidates[idx - 1])
                continue;
            if (candidates[idx] > target)
                break;
            ds.add(candidates[idx]);
            findTheCombination(idx + 1, candidates, target - candidates[idx], list, ds);
            ds.remove(ds.size() - 1);
        }
    }
}
