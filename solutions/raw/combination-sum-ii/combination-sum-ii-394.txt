class Solution {

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        Arrays.sort(candidates);
        getCombinations(0, target, candidates, ans, ds);
        return ans;
    }

    public void getCombinations(int index, int target, int[] arr, List<List<Integer>> ans, List<Integer> ds) {
        if (target == 0) {
            List<Integer> ads = new ArrayList<>(ds);
            ans.add(ads);
            return;
        }
        for (int i = index; i < arr.length; i++) {
            if (i > index && arr[i] == arr[i - 1]) {
                continue;
            }
            if (arr[i] > target) {
                break;
            }
            ds.add(arr[i]);
            getCombinations(i + 1, target - arr[i], arr, ans, ds);
            ds.remove(ds.size() - 1);
        }
    }
}
