class Solution {

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        findSolution(0, candidates, target, ans, new ArrayList<>());
        return ans;
    }

    public static void findSolution(int idx, int[] cand, int target, List<List<Integer>> ans, List<Integer> res) {
        if (target == 0) {
            ans.add(new ArrayList<>(res));
            return;
        }
        for (int i = idx; i < cand.length; i++) {
            if (i > idx && cand[i] == cand[i - 1])
                continue;
            if (cand[i] > target)
                break;
            res.add(cand[i]);
            findSolution(i + 1, cand, target - cand[i], ans, res);
            res.remove(res.size() - 1);
        }
    }
}
