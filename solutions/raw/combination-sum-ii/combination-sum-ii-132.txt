class Solution {

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        findCombinations(0, target, candidates, ans, new ArrayList<>());
        return ans;
    }

    void findCombinations(int index, int target, int[] candidates, List<List<Integer>> ans, List<Integer> ds) {
        //base case
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = index; i < candidates.length; i++) {
            //edge cases
            if (i > index && candidates[i] == candidates[i - 1])
                continue;
            //candidate exceeds target value, then break
            if (candidates[i] > target)
                break;
            //pick
            ds.add(candidates[i]);
            findCombinations(i + 1, target - candidates[i], candidates, ans, ds);
            ds.remove(ds.size() - 1);
        }
    }
}
