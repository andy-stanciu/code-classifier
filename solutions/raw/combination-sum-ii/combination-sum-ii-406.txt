class Solution {

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        findCombo(candidates, target, ans, new ArrayList(), 0);
        return ans;
    }

    public void findCombo(int[] candidates, int target, List<List<Integer>> ans, List<Integer> ds, int ind) {
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = ind; i < candidates.length; i++) {
            if (i > ind && candidates[i] == candidates[i - 1])
                continue;
            if (candidates[i] > target)
                break;
            ds.add(candidates[i]);
            findCombo(candidates, target - candidates[i], ans, ds, i + 1);
            ds.remove(ds.size() - 1);
        }
    }
}
