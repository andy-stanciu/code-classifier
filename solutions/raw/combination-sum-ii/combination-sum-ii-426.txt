class Solution {

    public void backtracking(int i, int[] candidates, List<List<Integer>> answer, List<Integer> buffer, int sum) {
        if (sum == 0) {
            answer.add(new ArrayList<>(buffer));
            return;
        } else if (sum > 0) {
            for (int j = i; j < candidates.length; j++) {
                if (candidates[j] > sum)
                    break;
                if (j > i && candidates[j] == candidates[j - 1])
                    continue;
                buffer.add(candidates[j]);
                backtracking(j + 1, candidates, answer, buffer, sum - candidates[j]);
                buffer.remove(buffer.size() - 1);
            }
        }
    }

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<Integer> buffer = new ArrayList<>();
        List<List<Integer>> answer = new ArrayList<>();
        Arrays.sort(candidates);
        backtracking(0, candidates, answer, buffer, target);
        return answer;
    }
}
