class Solution {

    List<List<Integer>> listFinal = new ArrayList<List<Integer>>();

    int n;

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        n = candidates.length;
        Arrays.sort(candidates);
        solve(candidates, target, 0, new ArrayList(), 0);
        return listFinal;
    }

    void solve(int[] candidates, int target, int idx, List<Integer> list, int sum) {
        if (idx > n || sum > target) {
            return;
        }
        if (sum == target) {
            listFinal.add(new ArrayList(list));
            return;
        }
        for (int i = idx; i < n && (sum + candidates[i]) <= target; i++) {
            if (i > idx && candidates[i] == candidates[i - 1]) {
                // Skip duplicates
                continue;
            }
            list.add(candidates[i]);
            solve(candidates, target, i + 1, list, sum + candidates[i]);
            // Backtrack
            list.remove(list.size() - 1);
        }
    }
}
