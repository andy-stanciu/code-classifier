class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        }
        int sum = Integer.MIN_VALUE;
        int currMax = 0;
        for (int num : nums) {
            int currSum = num + currMax;
            currMax = Math.max(num, currSum);
            sum = Math.max(sum, currMax);
        }
        return sum;
    }
}
