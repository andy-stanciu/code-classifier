class Solution {

    public int maxSubArray(int[] nums) {
        int maxSum = nums[0];
        int total = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (total < 0) {
                total = 0;
            }
            total += nums[i];
            maxSum = Math.max(maxSum, total);
        }
        return maxSum;
    }
}
