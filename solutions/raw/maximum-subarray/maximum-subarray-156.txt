//KADANE\'S ALGORITHM
class Solution {

    public int maxSubArray(int[] nums) {
        int s = 0;
        int x = nums[0];
        for (int i = 0; i < nums.length; i++) {
            s += nums[i];
            if (x < s) {
                x = s;
            }
            if (s < 0) {
                s = 0;
                //System.out.println(i);
            }
        }
        return x;
    }
}
