class Solution {

    public int maxSubArray(int[] nums) {
        int currentsum = 0;
        int ms = Integer.MIN_VALUE;
        for (int val : nums) {
            currentsum += val;
            ms = Math.max(currentsum, ms);
            if (currentsum < 0) {
                currentsum = 0;
            }
        }
        return ms;
    }
}
