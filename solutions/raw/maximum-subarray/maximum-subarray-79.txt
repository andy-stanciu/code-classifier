class Solution {

    public int maxSubArray(int[] nums) {
        int sum = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int localSum = 0;
            for (int j = i; j < nums.length; j++) {
                localSum += nums[j];
                sum = Math.max(sum, localSum);
            }
        }
        return sum;
    }
}
