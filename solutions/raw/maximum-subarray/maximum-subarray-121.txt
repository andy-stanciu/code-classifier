class Solution {

    public int maxSubArray(int[] nums) {
        int min = 0;
        int runningSum = 0;
        int max = nums[0];
        for (int num : nums) {
            runningSum += num;
            max = Math.max(max, runningSum - min);
            min = Math.min(runningSum, min);
        }
        return max;
    }
}
