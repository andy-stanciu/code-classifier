class Solution {

    public int maxSubArray(int[] nums) {
        // Initialize to the smallest possible integer value
        int max = Integer.MIN_VALUE;
        // Variable to keep track of the current subarray sum
        int subArr = 0;
        for (int num : nums) {
            subArr += num;
            if (subArr > max) {
                max = subArr;
            }
            if (subArr < 0) {
                subArr = 0;
            }
        }
        return max;
    }
}
