class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 0)
            return 0;
        int max = Integer.MIN_VALUE, sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = Math.max(nums[i], sum + nums[i]);
            max = Math.max(sum, max);
        }
        return max;
    }
}
