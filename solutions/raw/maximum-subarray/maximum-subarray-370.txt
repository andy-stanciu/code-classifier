class Solution {

    public int maxSubArray(int[] a) {
        int sum = 0, maxSum = Integer.MIN_VALUE;
        int i = 0;
        while (i < a.length) {
            sum += a[i];
            maxSum = Math.max(sum, maxSum);
            i++;
            if (sum <= 0) {
                sum = 0;
            }
        }
        return maxSum;
    }
}
