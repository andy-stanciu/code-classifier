class Solution {

    public int maxSubArray(int[] nums) {
        int maxSum = nums[0];
        int currentSum = 0;
        for (int i = 0; i < nums.length; i++) {
            //Add the current element to currentSum to expand the subarray
            currentSum += nums[i];
            // If the current subarray sum is greater than maxSum, update maxSum
            if (currentSum > maxSum)
                maxSum = currentSum;
            // If currentSum is negative, reset it to 0
            // Starting fresh with the next element may lead to a higher sum
            if (currentSum < 0)
                currentSum = 0;
        }
        return maxSum;
    }
}
