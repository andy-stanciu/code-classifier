class Solution {

    public int maxSubArray(int[] nums) {
        int currentMax = 0;
        int totalMax = Integer.MIN_VALUE;
        if (nums.length == 1) {
            return nums[0];
        }
        for (int i = 0; i < nums.length; i++) {
            currentMax = currentMax + nums[i];
            if (currentMax > totalMax) {
                totalMax = currentMax;
            }
            if (currentMax < 0) {
                currentMax = 0;
            }
        }
        return totalMax;
    }
}
