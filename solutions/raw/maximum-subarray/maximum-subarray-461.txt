class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 0)
            return 0;
        int maxCurrSum = 0;
        int maxSum = Integer.MIN_VALUE;
        for (int num : nums) {
            maxCurrSum = Math.max(num, maxCurrSum + num);
            maxSum = Math.max(maxCurrSum, maxSum);
        }
        return maxSum;
    }
}
