class Solution {

    public int maxSubArray(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        int i;
        dp[0] = nums[0];
        int max = nums[0];
        for (i = 1; i < n; ++i) {
            dp[i] = Math.max(nums[i], dp[i - 1] + nums[i]);
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
