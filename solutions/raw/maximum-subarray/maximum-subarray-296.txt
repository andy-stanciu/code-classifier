class Solution {

    public int maxSubArray(int[] arrnum) {
        int heighestsum = arrnum[0];
        int totalsum = 0;
        for (int nums : arrnum) {
            if (totalsum <= -1) {
                totalsum = 0;
            }
            totalsum += nums;
            heighestsum = totalsum > heighestsum ? totalsum : heighestsum;
        }
        return heighestsum;
    }
}
