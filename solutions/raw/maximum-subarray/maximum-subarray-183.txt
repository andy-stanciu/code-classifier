class Solution {

    public int maxSubArray(int[] nums) {
        int current_sum = 0;
        int maximum = nums[0];
        for (int i = 0; i < nums.length; i++) {
            current_sum += nums[i];
            maximum = Math.max(maximum, current_sum);
            if (current_sum <= 0) {
                current_sum = 0;
            }
        }
        return maximum;
    }
}
