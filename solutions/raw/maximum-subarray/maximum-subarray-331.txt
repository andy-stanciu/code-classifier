class Solution {

    public int maxSubArray(int[] nums) {
        int j = 0;
        int sum = 0;
        int max = Integer.MIN_VALUE;
        while (j < nums.length) {
            sum = Math.max(sum + nums[j], nums[j]);
            max = Math.max(max, sum);
            j++;
        }
        return max;
    }
}
