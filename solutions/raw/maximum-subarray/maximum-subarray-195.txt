class Solution {

    public int maxSubArray(int[] nums) {
        int sum = 0;
        // First declare SUM as 0 and max_sum as INT.MIN.
        int max_sum = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
            max_sum = Math.max(max_sum, sum);
            if (sum < 0) {
                // If sum is less than 0 compare it to max_sum and set it to 0.
                sum = 0;
            }
        }
        // and return max_sum.
        return max_sum;
    }
}
