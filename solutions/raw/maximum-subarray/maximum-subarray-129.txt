class Solution {

    public int maxSubArray(int[] nums) {
        int maxSum = nums[0];
        int currSum = nums[0];
        if (nums.length == 1) {
            return nums[0];
        }
        for (int i = 1; i < nums.length; i++) {
            if (currSum < 0) {
                currSum = 0;
            }
            currSum = nums[i] + currSum;
            maxSum = Math.max(maxSum, currSum);
        }
        return maxSum;
    }
}
