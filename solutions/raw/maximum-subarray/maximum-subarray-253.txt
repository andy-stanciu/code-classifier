class Solution {

    public int maxSubArray(int[] nums) {
        int curr = 0;
        int max = Integer.MIN_VALUE;
        if (allnegative(nums)) {
            Arrays.sort(nums);
            return nums[nums.length - 1];
        }
        for (int i : nums) {
            curr += i;
            if (curr < 0)
                curr = 0;
            max = Math.max(curr, max);
        }
        return max;
    }

    public boolean allnegative(int[] nums) {
        for (int i : nums) {
            if (i >= 0)
                return false;
        }
        return true;
    }
}
