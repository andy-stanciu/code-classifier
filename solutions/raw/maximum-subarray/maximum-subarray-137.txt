class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        }
        int card = 0;
        int maxsub = nums[0];
        for (int n : nums) {
            if (card < 0) {
                card = 0;
            }
            card = card + n;
            maxsub = Math.max(maxsub, card);
        }
        return maxsub;
    }
}
