class Solution {

    public int maxSubArray(int[] nums) {
        int[] maxSum = new int[nums.length];
        maxSum[0] = nums[0];
        int maxx = nums[0];
        for (int i = 1; i < nums.length; i++) {
            maxSum[i] = Math.max(maxSum[i - 1] + nums[i], nums[i]);
            maxx = Math.max(maxx, maxSum[i]);
        }
        return maxx;
    }
}
