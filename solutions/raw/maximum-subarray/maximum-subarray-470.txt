class Solution {

    public int maxSubArray(int[] nums) {
        int a = Integer.MIN_VALUE;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
            a = Math.max(a, sum);
            sum = Math.max(sum, 0);
        }
        return a;
    }
}
