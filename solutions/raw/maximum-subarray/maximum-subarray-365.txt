class Solution {

    public int maxSubArray(int[] numbers) {
        if (numbers.length == 0)
            return 0;
        if (numbers.length == 1)
            return numbers[0];
        int max = numbers[0];
        int sum = Math.max(0, max);
        for (int i = 1; i < numbers.length; i++) {
            sum += numbers[i];
            max = Math.max(sum, max);
            if (sum < 0) {
                sum = 0;
            }
        }
        return max;
    }
}
