class Solution {

    public int maxSubArray(int[] nums) {
        // Initialize sum with the first element
        int sum = nums[0];
        // Initialize result with the first element
        int result = nums[0];
        for (int i = 1; i < nums.length; i++) {
            // Update sum
            sum = Math.max(nums[i], sum + nums[i]);
            // Update result
            result = Math.max(result, sum);
        }
        return result;
    }
}
