class Solution {

    public int maxSubArray(int[] nums) {
        int cur = nums[0];
        int total = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (cur < 0) {
                cur = nums[i];
            } else {
                cur += nums[i];
            }
            if (cur > total) {
                total = cur;
            }
        }
        return total;
    }
}
