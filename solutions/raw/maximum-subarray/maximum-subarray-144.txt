class Solution {

    public int maxSubArray(int[] nums) {
        int currsum = 0;
        int oversum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            currsum += nums[i];
            if (currsum < 0) {
                currsum = 0;
            }
            if (currsum > oversum) {
                oversum = currsum;
            }
        }
        return oversum;
    }
}
