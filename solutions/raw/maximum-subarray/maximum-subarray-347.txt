class Solution {

    public int maxSubArray(int[] nums) {
        // Corrected to Integer.MIN_VALUE
        int max_s = Integer.MIN_VALUE;
        // Current sum
        int cs = 0;
        for (int i = 0; i < nums.length; i++) {
            // Add current number to current sum
            cs += nums[i];
            if (cs > max_s) {
                max_s = cs;
            }
            if (cs < 0) {
            }
        }
        // Return the maximum subarray sum
        return max_s;
    }
}
