class Solution {

    public int maxSubArray(int[] nums) {
        int subArraySum = -10000;
        int maxSubSum = nums[0];
        for (int num : nums) {
            subArraySum = Math.max(num + subArraySum, num);
            maxSubSum = Math.max(subArraySum, maxSubSum);
        }
        return maxSubSum;
    }
}
