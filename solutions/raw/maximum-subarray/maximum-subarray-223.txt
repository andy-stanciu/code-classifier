class Solution {

    public int maxSubArray(int[] nums) {
        int ans = nums[0], sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (sum < 0) {
                sum = 0;
                ans = Math.max(ans, nums[i]);
            } else
                ans = Math.max(sum, ans);
        }
        return ans;
    }
}
