class Solution {

    public int maxSubArray(int[] nums) {
        int prefixSum = 0;
        int max = Integer.MIN_VALUE;
        for (int i = 0; i <= nums.length - 1; i++) {
            if (prefixSum + nums[i] > nums[i]) {
                prefixSum = prefixSum + nums[i];
            } else {
                prefixSum = nums[i];
            }
            if (prefixSum > max) {
                max = prefixSum;
            }
        }
        return max;
    }
}
