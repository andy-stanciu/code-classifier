class Solution {

    public int maxSubArray(int[] nums) {
        // omax: overall maximum
        int omax = nums[0];
        //  cmax: current maximum
        int cmax = nums[0];
        for (int i = 1; i < nums.length; i++) {
            cmax = Math.max(nums[i], nums[i] + cmax);
            omax = Math.max(cmax, omax);
        }
        return omax;
    }
}
