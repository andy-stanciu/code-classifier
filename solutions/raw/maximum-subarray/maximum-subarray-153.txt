class Solution {

    public int maxSubArray(int[] nums) {
        // Store 1st element in for maxSum and currSum
        int currSum = nums[0];
        int maxSum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            // If ongoing subarray sum is less than 0, discard it and start taking sum for new sub array
            if (currSum < 0) {
                currSum = 0;
            }
            // add current element from nums arr to our ongoing subarray sum
            currSum += nums[i];
            // check if current sum of going subarray is greater than max sum till now,
            // then set max sum to current sum
            if (currSum > maxSum) {
                maxSum = currSum;
            }
        }
        return maxSum;
    }
}
