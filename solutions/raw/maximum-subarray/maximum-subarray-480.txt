class Solution {

    public int maxSubArray(int[] nums) {
        int CurrSum = 0;
        int MaxSum = nums[0];
        for (int i = 0; i < nums.length; i++) {
            CurrSum = CurrSum + nums[i];
            MaxSum = Math.max(CurrSum, MaxSum);
            if (CurrSum < 0) {
                CurrSum = 0;
            }
        }
        return MaxSum;
    }
}
