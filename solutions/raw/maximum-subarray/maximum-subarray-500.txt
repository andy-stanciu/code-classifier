class Solution {

    //Solution by Tabulation.
    public int maxSubArray(int[] nums) {
        //ans variable for stor the max sum of subarray.
        int ans = nums[0];
        //create new array of nums length.
        int[] dp = new int[nums.length];
        //dp[0] store max subarray sum ending at 0 index.
        dp[0] = nums[0];
        for (int i = 1; i < dp.length; i++) {
            //dp[i] store max subarray sum ending at ith index.
            //Either the one behind will be max or this one will be add behind and this is max.
            dp[i] = Math.max(nums[i], nums[i] + dp[i - 1]);
            //check ans is greater or not. If not greater than dp[i] then change the ans value.
            if (ans < dp[i]) {
                ans = dp[i];
            }
        }
        return ans;
    }
}
