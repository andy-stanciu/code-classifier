class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        }
        int currsum = 0;
        int maxsum = Integer.MIN_VALUE;
        // to calculate if only negative no. are present in an array
        int maxnegsum = 0;
        int answer = 0;
        // CODE TO CALCULATE IF ONLY NEG NO. ARE PRESENT
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > maxsum) {
                maxsum = nums[i];
            }
            if (maxsum <= 0) {
                // this condition is used if one zero and other are negative no.s
                maxnegsum = maxsum;
            }
        }
        // THIS IS SIMPLE KEDANS ALGO FOR BOTH POSITIVE AND NEG NO
        for (int i = 0; i < nums.length; i++) {
            currsum = currsum + nums[i];
            if (currsum < 0) {
                currsum = 0;
            }
            if (currsum > maxsum) {
                maxsum = currsum;
            }
        }
        if (maxsum > 0) {
            answer = maxsum;
        } else {
            answer = maxnegsum;
        }
        return answer;
    }
}
