class Solution {

    public int maxSubArray(int[] nums) {
        int max = nums[0];
        int dp = nums[0];
        for (int i = 1; i < nums.length; i++) {
            //ex:[-4,2,1] if i=1 then -4+2<2, so we should store 2 as the max value as current rather than -2
            dp = Math.max(dp + nums[i], nums[i]);
            //max current value with ex-value, and store the bigger one
            max = Math.max(dp, max);
        }
        return max;
    }
}
