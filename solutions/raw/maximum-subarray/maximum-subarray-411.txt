class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int maxSum = nums[0];
        int sum = 0;
        for (int n : nums) {
            sum = Math.max(n, sum + n);
            maxSum = Math.max(maxSum, sum);
        }
        return maxSum;
    }
}
