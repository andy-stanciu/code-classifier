class Solution {

    public int maxSubArray(int[] nums) {
        int soFar = nums[0], tillNow = nums[0], i = 1;
        while (i < nums.length) {
            tillNow = Math.max(tillNow + nums[i], nums[i]);
            soFar = Math.max(soFar, tillNow);
            i++;
        }
        return soFar;
    }
}
