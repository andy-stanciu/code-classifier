class Solution {

    public int maxSubArray(int[] nums) {
        int max = Integer.MIN_VALUE;
        int meh = 0;
        for (int i = 0; i < nums.length; i++) {
            meh = meh + nums[i];
            if (meh > max) {
                max = meh;
            }
            if (meh < 0) {
                meh = 0;
            }
        }
        return max;
    }
}
