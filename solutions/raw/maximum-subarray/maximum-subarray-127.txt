class Solution {

    public int maxSubArray(int[] nums) {
        int sum = 0, ans = Integer.MIN_VALUE;
        for (int num : nums) {
            //Add each element to sum
            sum += num;
            //Update ans if sum > ans
            ans = Math.max(sum, ans);
            if (sum < 0) {
                //If sum become -ve, reset it to 0
                sum = 0;
            }
        }
        return ans;
    }
}
