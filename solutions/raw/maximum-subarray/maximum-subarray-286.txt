class Solution {

    public int maxSubArray(int[] arr) {
        int sum = 0;
        int n = arr.length;
        int maxs = arr[0];
        for (int i = 0; i < n; i++) {
            sum += arr[i];
            if (sum > maxs) {
                maxs = sum;
            }
            if (sum < 0)
                sum = 0;
        }
        return maxs;
    }
}
