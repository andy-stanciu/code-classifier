class Solution {

    public int maxSubArray(int[] a) {
        int i;
        int max = a[0];
        int result = a[0];
        for (i = 1; i < a.length; i++) {
            // computing max of current or prev+current
            max = Math.max(max + a[i], a[i]);
            result = Math.max(result, max);
        }
        return (result);
    }
}
