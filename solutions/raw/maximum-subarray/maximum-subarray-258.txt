class Solution {

    public int maxSubArray(int[] nums) {
        int max = nums[0];
        int currSubArraysum = 0;
        for (int i = 0; i < nums.length; i++) {
            currSubArraysum += nums[i];
            max = Math.max(max, currSubArraysum);
            if (currSubArraysum < 0) {
                currSubArraysum = 0;
            }
        }
        return max;
    }
}
