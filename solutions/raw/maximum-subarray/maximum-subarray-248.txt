class Solution {

    public int maxSubArray(int[] nums) {
        int prefixSum = 0;
        int minPrefixSum = 0;
        int result = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            prefixSum += nums[i];
            int maxHere = prefixSum - minPrefixSum;
            result = Math.max(result, maxHere);
            minPrefixSum = Math.min(minPrefixSum, prefixSum);
        }
        return result;
    }
}
