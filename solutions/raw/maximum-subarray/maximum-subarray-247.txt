class Solution {

    public int maxSubArray(int[] nums) {
        int maxcurrent = nums[0];
        int maxglobal = nums[0];
        for (int i = 1; i < nums.length; i++) {
            maxcurrent = Math.max(nums[i], maxcurrent + nums[i]);
            if (maxcurrent > maxglobal) {
                maxglobal = maxcurrent;
            }
        }
        return maxglobal;
    }
}
