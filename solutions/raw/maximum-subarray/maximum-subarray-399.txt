class Solution {

    public int maxSubArray(int[] nums) {
        int min = nums[0];
        int preSum = min;
        int ans = min;
        // 5,4,-1,7,8
        for (int i = 1; i < nums.length; i++) {
            int x = nums[i];
            preSum += x;
            ans = Math.max(preSum - min, ans);
            ans = Math.max(preSum, ans);
            min = Math.min(min, preSum);
        }
        return ans;
    }
}
