class Solution {

    public int maxSubArray(int[] nums) {
        int currSum = 0, maxSum = 0;
        int minVal = Integer.MIN_VALUE;
        for (int i : nums) {
            currSum += i;
            if (currSum > 0) {
                maxSum = Math.max(maxSum, currSum);
            } else {
                currSum = 0;
                minVal = Math.max(minVal, i);
            }
        }
        if (maxSum > 0) {
            return maxSum;
        }
        return minVal;
    }
}
