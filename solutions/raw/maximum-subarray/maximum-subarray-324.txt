class Solution {

    public int maxSubArray(int[] nums) {
        int max_Sum = Integer.MIN_VALUE;
        int curr_Sum = 0;
        for (int i = 0; i < nums.length; i++) {
            curr_Sum += nums[i];
            if (max_Sum < curr_Sum)
                max_Sum = curr_Sum;
            if (curr_Sum < 0)
                curr_Sum = 0;
        }
        return max_Sum;
    }
}
