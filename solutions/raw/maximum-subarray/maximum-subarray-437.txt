class Solution {

    public int maxSubArray(int[] nums) {
        int CurrSum = 0;
        int Maxsubarray = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            CurrSum += nums[i];
            Maxsubarray = Math.max(CurrSum, Maxsubarray);
            if (CurrSum < 0) {
                CurrSum = 0;
            }
        }
        return Maxsubarray;
    }
}
