class Solution {

    public int maxSubArray(int[] nums) {
        int i = 0, j = 0;
        int sum = 0, max = Integer.MIN_VALUE;
        while (j <= nums.length - 1) {
            sum += nums[j];
            max = Math.max(sum, max);
            if (sum <= 0) {
                i = j;
                sum = 0;
            }
            j++;
        }
        return max;
    }
}
