class Solution {

    public int maxSubArray(int[] nums) {
        int max = nums[0];
        int currentSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (currentSum < 0) {
                currentSum = 0;
            }
            currentSum += nums[i];
            if (currentSum > max) {
                max = currentSum;
            }
        }
        return max;
    }
}
