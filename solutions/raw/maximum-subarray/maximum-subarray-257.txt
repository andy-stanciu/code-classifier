class Solution {

    public int maxSubArray(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int res = Integer.MIN_VALUE, currSum = 0;
        // \u8D2A\u5FC3\u7B97\u6CD5
        for (int i : nums) {
            currSum += i;
            res = Math.max(res, currSum);
            if (currSum < 0)
                currSum = 0;
        }
        return res;
    }
}
