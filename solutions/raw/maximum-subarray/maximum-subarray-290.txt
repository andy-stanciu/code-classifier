class Solution {

    public int maxSubArray(int[] nums) {
        int n = nums.length;
        int currSum = nums[0];
        int longestSum = nums[0];
        if (n == 1)
            return nums[0];
        for (int start = 1; start < n; start++) {
            currSum = Math.max(nums[start], currSum + nums[start]);
            longestSum = Math.max(longestSum, currSum);
        }
        return longestSum;
    }
}
