class Solution {

    public int maxSubArray(int[] nums) {
        int max = nums[0];
        int suba = 0;
        for (int i = 0; i < nums.length; i++) {
            suba += nums[i];
            if (suba > max) {
                max = suba;
            }
            if (suba < 0) {
                suba = 0;
            }
        }
        return max;
    }
}
