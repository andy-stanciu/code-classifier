class Solution {

    public int maxSubArray(int[] nums) {
        int currentOptimal = nums[0];
        int max = currentOptimal;
        for (int i = 1; i < nums.length; i++) {
            currentOptimal = Math.max(currentOptimal + nums[i], nums[i]);
            max = Math.max(max, currentOptimal);
        }
        return max;
    }
}
