class Solution {

    public int maxSubArray(int[] nums) {
        int sum = 0;
        int max = 0;
        int n = nums.length;
        int nmax = nums[0];
        boolean flag = false;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            if (sum <= 0 && i != n) {
                sum = 0;
                if (nums[i] > nmax) {
                    nmax = nums[i];
                }
            } else if (sum > max) {
                flag = true;
                max = sum;
            }
        }
        if (flag) {
            return max;
        }
        return nmax;
    }
}
