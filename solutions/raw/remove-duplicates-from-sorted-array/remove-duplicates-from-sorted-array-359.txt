class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0)
            return 0;
        int f = 0;
        int n = 1;
        while (n < nums.length) {
            if (nums[n] != nums[f]) {
                f++;
                nums[f] = nums[n];
            }
            n++;
        }
        return f + 1;
    }
}
