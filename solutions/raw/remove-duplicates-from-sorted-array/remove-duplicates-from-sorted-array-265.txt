class Solution {

    public int removeDuplicates(int[] nums) {
        int k = 0;
        int prevUnique = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != prevUnique) {
                nums[k] = nums[i];
                prevUnique = nums[k];
                k++;
            }
        }
        return k;
    }
}
