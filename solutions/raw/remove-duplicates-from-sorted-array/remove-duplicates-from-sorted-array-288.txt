class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return n;
        int uniIdx = 0;
        for (int i = 0; i < n; i++) {
            if (nums[uniIdx] != nums[i]) {
                uniIdx += 1;
                nums[uniIdx] = nums[i];
            }
        }
        return uniIdx + 1;
    }
}
