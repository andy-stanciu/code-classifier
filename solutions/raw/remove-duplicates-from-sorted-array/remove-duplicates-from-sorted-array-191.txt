class Solution {

    public int removeDuplicates(int[] nums) {
        Stack<Integer> st = new Stack<>();
        for (int i = nums.length - 1; i >= 0; i--) {
            if (st.isEmpty() || nums[i] != st.peek()) {
                st.push(nums[i]);
            }
        }
        int count = 0;
        for (int j = 0; j < nums.length; j++) {
            if (!st.isEmpty()) {
                nums[j] = st.pop();
                count++;
            }
        }
        return count;
    }
}
