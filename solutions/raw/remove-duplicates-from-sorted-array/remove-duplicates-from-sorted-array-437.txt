class Solution {

    public int removeDuplicates(int[] nums) {
        int k = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != k) {
                nums[count] = nums[i];
                count++;
                k = nums[i];
            }
        }
        return count;
    }
}
