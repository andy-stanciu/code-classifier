class Solution {

    public int removeDuplicates(int[] nums) {
        int count = 0;
        int val = nums[0];
        int size = nums.length;
        for (int i = 1; i < size; i++) {
            if (nums[i] == val && nums[i - 1] == val) {
                count++;
            } else {
                val = nums[i];
                nums[i - count] = nums[i];
            }
        }
        return size - count;
    }
}
