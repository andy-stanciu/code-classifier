class Solution {

    public int removeDuplicates(int[] nums) {
        int N = nums.length;
        // initialize variable for array index
        int i = 0, j = 1;
        while (j < N) {
            // apply condition
            if (nums[i] != nums[j]) {
                // check condition
                // increase value of i (+1)
                i++;
                // assign nums[j]\'s value in nums[i]
                nums[i] = nums[j];
            }
            // increased value of j
            j++;
        }
        return i + 1;
    }
}
