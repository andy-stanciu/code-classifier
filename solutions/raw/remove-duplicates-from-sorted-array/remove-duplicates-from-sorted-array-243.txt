class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0, j = nums.length - 1, c = 0;
        while (i < j) {
            int val = nums[i];
            while (val == nums[i + 1]) {
                nums[i + 1] = 998909;
                c++;
                if (i + 1 < j) {
                    i++;
                }
            }
            i++;
        }
        Arrays.sort(nums);
        return j - c + 1;
    }
}
