class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length, j = 1, prev = nums[0];
        for (int i = 1; i < n; i++) {
            if (nums[i] != prev) {
                prev = nums[i];
                nums[j] = nums[i];
                j++;
            }
        }
        return j;
    }
}
