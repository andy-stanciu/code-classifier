class Solution {

    public int removeDuplicates(int[] nums) {
        List<Integer> listInt = new ArrayList(Arrays.stream(nums).boxed().collect(Collectors.toCollection(LinkedHashSet::new)));
        if (listInt.size() == 1)
            return 1;
        int pointer = 0;
        for (Integer el : listInt) {
            nums[pointer++] = el;
        }
        ;
        //        int numsLength = 0;
        //        while(numsLength < nums.length) {
        //
        //            if(numsLength < listInt.size()) nums[numsLength] = listInt.get(numsLength);
        //            numsLength++;
        //        }
        return listInt.size();
    }
}
