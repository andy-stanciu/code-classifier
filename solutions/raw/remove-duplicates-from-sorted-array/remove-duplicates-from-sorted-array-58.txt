class Solution {

    public int removeDuplicates(int[] nums) {
        int low = 0, fast = 1;
        int n = nums.length;
        if (n == 1)
            return 1;
        int k = 0;
        while (fast < nums.length) {
            if (nums[low] != nums[fast]) {
                low++;
                nums[low] = nums[fast];
                k++;
            }
            fast++;
        }
        return k + 1;
    }
}
