class Solution {

    public int removeDuplicates(int[] nums) {
        HashSet<Integer> sec = new HashSet<>();
        int newSizeArray = 0;
        for (int i = 0; i < nums.length; i++) {
            if (!sec.contains(nums[i])) {
                sec.add(nums[i]);
                nums[newSizeArray++] = nums[i];
            }
        }
        return newSizeArray;
    }
}
