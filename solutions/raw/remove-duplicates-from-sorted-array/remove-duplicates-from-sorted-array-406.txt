class Solution {

    public int removeDuplicates(int[] nums) {
        LinkedHashSet<Integer> set = Arrays.stream(nums).boxed().collect(Collectors.toCollection(LinkedHashSet::new));
        List<Integer> listInt = new ArrayList(set);
        if (set.size() == 1)
            return 1;
        int numsLength = 0;
        while (numsLength < nums.length) {
            if (numsLength < listInt.size())
                nums[numsLength] = listInt.get(numsLength);
            numsLength++;
        }
        return set.size();
    }
}
