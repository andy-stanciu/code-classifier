class Solution {

    public int removeDuplicates(int[] nums) {
        // the first element is always an unique element that\'s why we started the count from 1
        int count = 1;
        //start the pointer from the index 1 and check with the previous element i.e.,
        //check the 1st index with 0th index if those two elements are not equal increse the count
        //and swap the indexes of i and count
        //return count
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[count] = nums[i];
                count++;
            }
        }
        return count;
    }
}
