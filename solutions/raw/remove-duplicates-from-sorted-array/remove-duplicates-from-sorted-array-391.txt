class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        // t1 points to the position where the next unique element should be placed
        int t1 = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i - 1] != nums[i]) {
                nums[t1] = nums[i];
                t1++;
            }
        }
        return t1;
    }
}
