class Solution {

    public int removeDuplicates(int[] nums) {
        //[0,1,2,3,4,2,2,3,3,4]
        //       u           i
        int lastUniqueElement = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[lastUniqueElement] != nums[i]) {
                nums[lastUniqueElement + 1] = nums[i];
                lastUniqueElement++;
            }
        }
        return lastUniqueElement + 1;
    }
}
