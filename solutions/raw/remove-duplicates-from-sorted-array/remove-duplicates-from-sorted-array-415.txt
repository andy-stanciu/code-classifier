class Solution {

    public int removeDuplicates(int[] nums) {
        int insertionPointer = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                insertionPointer++;
                nums[insertionPointer] = nums[i];
            }
        }
        return insertionPointer + 1;
    }
}
