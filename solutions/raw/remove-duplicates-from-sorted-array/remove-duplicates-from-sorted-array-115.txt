class Solution {

    public int removeDuplicates(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) set.add(num);
        Arrays.fill(nums, 0);
        ArrayList<Integer> list = new ArrayList<>(set);
        Collections.sort(list);
        for (int i = 0; i < list.size(); i++) {
            nums[i] = list.get(i);
        }
        return set.size();
    }
}
