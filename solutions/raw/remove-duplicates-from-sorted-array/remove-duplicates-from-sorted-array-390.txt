class Solution {

    public int removeDuplicates(int[] nums) {
        // If the array is empty, return 0
        if (nums.length == 0) {
            return 0;
        }
        // Initialize result to track the position of unique elements
        // Since the first element is always unique
        int result = 1;
        // Start iterating from the second element (index 1)
        for (int i = 1; i < nums.length; i++) {
            // Check if current element is different from the previous one
            if (nums[i] != nums[i - 1]) {
                // Move the unique element to the result position
                nums[result] = nums[i];
                // Increment the result to mark the next unique position
                result++;
            }
        }
        // Return the number of unique elements
        return result;
    }
}
