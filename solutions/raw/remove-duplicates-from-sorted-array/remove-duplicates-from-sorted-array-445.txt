class Solution {

    public int removeDuplicates(int[] nums) {
        int temp = nums[0];
        int check = 0;
        for (int i = 1; i < nums.length; i++) {
            if (check != 0) {
                nums[i - check] = nums[i];
            }
            if (temp == nums[i]) {
                check++;
            } else {
                temp = nums[i];
            }
        }
        return nums.length - check;
    }
}
