class Solution {

    public int removeDuplicates(int[] nums) {
        int prev = 0;
        int curr = 1;
        nums[prev] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[prev]) {
                nums[curr] = nums[i];
                curr++;
                prev++;
            }
        }
        return curr;
    }
}
