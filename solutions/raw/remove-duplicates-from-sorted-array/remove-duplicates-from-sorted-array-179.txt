class Solution {

    public int removeDuplicates(int[] nums) {
        TreeSet<Integer> set = new TreeSet();
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        int num = set.size();
        for (int i = 0; i < num; i++) {
            nums[i] = set.pollFirst();
        }
        return num;
    }
}
