class Solution {

    public static int removeDuplicates(int[] nums) {
        int uniqueNums = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[uniqueNums - 1] != nums[i]) {
                nums[uniqueNums] = nums[i];
                uniqueNums++;
            }
        }
        return uniqueNums;
    }
}
