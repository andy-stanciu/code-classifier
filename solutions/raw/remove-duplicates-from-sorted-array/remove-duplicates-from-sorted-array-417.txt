class Solution {

    public int[] noDuplicatesNums;

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        noDuplicatesNums = new int[nums.length];
        int uniqueCount = 0;
        noDuplicatesNums[uniqueCount] = nums[0];
        uniqueCount++;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != noDuplicatesNums[uniqueCount - 1]) {
                noDuplicatesNums[uniqueCount] = nums[i];
                uniqueCount++;
            }
        }
        for (int i = 0; i < uniqueCount; i++) {
            nums[i] = noDuplicatesNums[i];
        }
        return uniqueCount;
    }
}
