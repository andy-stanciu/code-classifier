class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0)
            return 0;
        // Pointer for the last unique element
        int i = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i]) {
                // Move to the next position for unique element
                i++;
                // Update the array in place
                nums[i] = nums[j];
            }
        }
        // Length of unique elements
        return i + 1;
    }
}
