class Solution {

    public int removeDuplicates(int[] nums) {
        HashMap<Integer, Boolean> hashMap = new HashMap<>();
        int newIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (!hashMap.containsKey(nums[i])) {
                nums[newIndex] = nums[i];
                newIndex++;
                hashMap.put(nums[i], true);
            }
        }
        return newIndex;
    }
}
