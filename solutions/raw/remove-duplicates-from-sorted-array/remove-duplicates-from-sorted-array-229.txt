class Solution {

    public int removeDuplicates(int[] nums) {
        int previous = nums[0];
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != previous) {
                nums[k] = nums[i];
                previous = nums[i];
                k += 1;
            }
        }
        return k;
    }
}
