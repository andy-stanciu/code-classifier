class Solution {

    public int removeDuplicates(int[] nums) {
        Set<Integer> s = new HashSet<>();
        for (int n : nums) {
            s.add(n);
        }
        int idx = 0;
        for (int x : s) {
            nums[idx++] = x;
        }
        Arrays.sort(nums, 0, idx);
        for (int j = 0; j < idx; j++) {
            System.out.print(nums[j] + ", ");
        }
        return idx;
    }
}
