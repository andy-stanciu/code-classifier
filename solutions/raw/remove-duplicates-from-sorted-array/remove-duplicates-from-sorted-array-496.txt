class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 1;
        int j = 1;
        int count = 1;
        int n = nums.length;
        while (i < n) {
            if (nums[i] != nums[i - 1]) {
                nums[j] = nums[i];
                count = count + 1;
                j++;
                i++;
            } else {
                i++;
            }
        }
        return count;
    }
}
