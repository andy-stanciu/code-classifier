class Solution {

    public int removeDuplicates(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        PriorityQueue<Integer> q = new PriorityQueue<>(set);
        int index = 0;
        while (!q.isEmpty()) {
            nums[index] = q.poll();
            index++;
        }
        return set.size();
    }
}
