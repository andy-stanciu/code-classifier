class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        int c = 1;
        int i = 1;
        while (i < n) {
            if (nums[i - 1] != nums[i]) {
                nums[c++] = nums[i];
            }
            i++;
        }
        return c;
    }
}
