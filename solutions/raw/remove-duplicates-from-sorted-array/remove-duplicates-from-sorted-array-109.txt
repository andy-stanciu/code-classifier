class Solution {

    public int removeDuplicates(int[] nums) {
        int N = nums.length;
        // First it will check for the edge cases
        if (N == 0 || N == 1) {
            return N;
        }
        int i = 0;
        int j = 1;
        while (j < N) {
            /*
            Check the value at ith index with jth index.
            If both are same then j will increment by 1 to
            check the next index value and so on and so foth;
            */
            if (nums[i] != nums[j]) {
                nums[++i] = nums[j];
            }
            j++;
        }
        // To include the last element which don\'t have duplicate value (i + 1)
        return i + 1;
    }
}
