class Solution {

    public int removeDuplicates(int[] nums) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (mp.containsKey(nums[i])) {
                mp.put(nums[i], mp.get(nums[i]) + 1);
                //if  hashmap already contains the key, increment its frequency.
            } else {
                mp.put(nums[i], 1);
                //else, put the key value pair into the hashmap.
            }
            if (mp.get(nums[i]) == 1) {
                nums[k] = nums[i];
                k++;
                // Check whether the frequency is 1 and replace the values on the nums array
            }
        }
        return k;
    }
}
