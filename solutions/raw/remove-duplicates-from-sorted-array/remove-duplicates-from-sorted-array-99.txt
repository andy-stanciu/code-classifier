class Solution {

    public int removeDuplicates(int[] nums) {
        int temp = -101, index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == temp)
                temp = nums[i];
            else {
                nums[index] = nums[i];
                index++;
                temp = nums[i];
            }
        }
        return index;
    }
}
