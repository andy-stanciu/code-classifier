class Solution {

    public int removeDuplicates(int[] nums) {
        int count = 0;
        int firstPointer = 0;
        int secondPointer = 1;
        while (secondPointer < nums.length) {
            if (nums[firstPointer] == nums[secondPointer]) {
                secondPointer++;
                count++;
            } else {
                nums[secondPointer - count] = nums[secondPointer];
                firstPointer = secondPointer;
                secondPointer++;
            }
        }
        return nums.length - count;
    }
}
