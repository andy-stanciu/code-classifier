class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0)
            return 0;
        // Pointer for the unique elements position
        int i = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i]) {
                i++;
                nums[i] = nums[j];
            }
        }
        // Number of unique elements
        return i + 1;
    }
}
