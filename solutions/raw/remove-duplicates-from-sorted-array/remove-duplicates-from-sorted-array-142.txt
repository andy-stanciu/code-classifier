class Solution {

    public int removeDuplicates(int[] nums) {
        HashSet<Integer> ms = new HashSet<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (ms.contains(nums[i])) {
                continue;
            } else {
                ms.add(nums[i]);
                nums[k++] = nums[i];
            }
        }
        return k;
    }
}
