class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return nums[0];
        }
        int check = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[check] = nums[i];
                check++;
            }
        }
        return check;
    }
}
