class Solution {

    public int removeDuplicates(int[] nums) {
        int lastNumber = nums[0];
        int counter = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != lastNumber) {
                nums[counter] = nums[i];
                lastNumber = nums[i];
                counter++;
            }
        }
        return counter;
    }
}
