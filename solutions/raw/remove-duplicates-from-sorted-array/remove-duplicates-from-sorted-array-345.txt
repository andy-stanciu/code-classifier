class Solution {

    public int removeDuplicates(int[] nums) {
        Stack<Integer> stack = new Stack<>();
        for (int n : nums) {
            if (stack.isEmpty() || stack.peek() != n) {
                stack.push(n);
            }
        }
        int i = 0;
        while (!stack.isEmpty()) {
            nums[i++] = stack.remove(0);
        }
        return i;
    }
}
