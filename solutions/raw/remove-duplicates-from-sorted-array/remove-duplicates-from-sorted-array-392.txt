class Solution {

    public int removeDuplicates(int[] nums) {
        int current = -101;
        int numUnique = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != current) {
                current = nums[i];
                nums[numUnique] = current;
                numUnique++;
            }
        }
        return numUnique;
    }
}
