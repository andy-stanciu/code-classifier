class Solution {

    public int removeDuplicates(int[] nums) {
        Stack<Integer> stack = new Stack<>();
        int count = 1;
        stack.push(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            int peek = stack.peek();
            if (peek == nums[i]) {
                continue;
            } else {
                stack.push(nums[i]);
                count++;
            }
        }
        for (int i = count - 1; i >= 0; i--) {
            nums[i] = stack.pop();
        }
        return count;
    }
}
