class Solution {

    public static void main(String[] args) {
        int[] nums = { 0, 0, 1, 1, 1, 2, 2, 3, 3, 4 };
        // Calling Method inside SOP
        System.out.println(removeDuplicates(nums));
    }

    static int removeDuplicates(int[] nums) {
        // This Set is taken to track all the visited elements and will contain unique elements only
        Set<Integer> seen = new HashSet<>();
        // List to store unique elements
        List<Integer> nonDuplicates = new ArrayList<>();
        // Iterate over all the elements
        for (int i = 0; i < nums.length; i++) {
            // If the element is not in the set, add it to the list
            if (!seen.contains(nums[i])) {
                nonDuplicates.add(nums[i]);
            }
            // Add the element to the set to mark it as seen
            seen.add(nums[i]);
        }
        // Updating the original array with unique elements
        for (int i = 0; i < nonDuplicates.size(); i++) {
            nums[i] = nonDuplicates.get(i);
        }
        // Return the size of the non-duplicate list
        return nonDuplicates.size();
    }
}
