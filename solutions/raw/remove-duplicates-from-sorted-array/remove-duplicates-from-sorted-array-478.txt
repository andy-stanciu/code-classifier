class Solution {

    public int removeDuplicates(int[] nums) {
        int size = nums.length;
        // Edge case: if the array has less than two elements meanse no duplicate value ?////possible
        if (size < 1)
            return 0;
        int slow = 0;
        for (int fast = 0; fast < size; fast++) {
            //if the current element is different from the last element
            if (nums[fast] != nums[slow]) {
                //move the slow pointer to next position
                slow++;
                //place the new unique element
                nums[slow] = nums[fast];
            }
        }
        // return the number of unique element which is slow+1
        return slow + 1;
    }
}
