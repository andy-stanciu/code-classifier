class Solution {

    public int removeDuplicates(int[] nums) {
        int L = nums.length;
        if (L == 0)
            return 0;
        if (L == 1)
            return 1;
        int i = 0, j = 1;
        while (j < L) {
            while (j < L && (nums[i] == nums[j] || nums[i] > nums[j])) {
                j++;
            }
            if (j < L)
                nums[++i] = nums[j];
        }
        return i + 1;
    }
}
