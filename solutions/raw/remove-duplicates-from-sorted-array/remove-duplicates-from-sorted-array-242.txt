class Solution {

    public int removeDuplicates(int[] nums) {
        int len = nums.length;
        if (len <= 1) {
            return len;
        }
        int insertIndex = 1;
        int current = nums[0];
        for (int i = 1; i < len; i++) {
            if (nums[i] != current) {
                current = nums[i];
                nums[insertIndex++] = current;
            }
        }
        return insertIndex;
    }
}
