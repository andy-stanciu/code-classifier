class Solution {

    public int removeDuplicates(int[] nums) {
        int pointer = 0;
        int pointerIndex = 0;
        while (pointerIndex < nums.length - 1) {
            pointerIndex++;
            if (nums[pointerIndex] != nums[pointer]) {
                pointer++;
                nums[pointer] = nums[pointerIndex];
            }
        }
        ;
        return pointer + 1;
    }
}
