class Solution {

    public int removeDuplicates(int[] nums) {
        SortedSet<Integer> treeSet = new TreeSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            treeSet.add(nums[i]);
        }
        int k = 0;
        Iterator<Integer> itr = treeSet.iterator();
        while (itr.hasNext()) {
            nums[k] = itr.next();
            k++;
        }
        return treeSet.size();
    }
}
