class Solution {

    public int removeDuplicates(int[] arr) {
        int j = 0;
        int i = 0;
        while (j < arr.length) {
            if (arr[i] != arr[j]) {
                i++;
                arr[i] = arr[j];
            } else {
                j++;
            }
        }
        return i + 1;
    }
}
