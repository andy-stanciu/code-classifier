class Solution {

    public int removeDuplicates(int[] nums) {
        int[] // Remove duplicates
        // Convert stream back to array
        uniqueNums = Arrays.stream(nums).distinct().toArray();
        // Copy unique elements back to the original array
        for (int i = 0; i < uniqueNums.length; i++) {
            nums[i] = uniqueNums[i];
        }
        // Return the number of unique elements (size of uniqueNums)
        return uniqueNums.length;
    }
}
