public class Solution {

    public int removeDuplicatesRecursive(int[] nums, int writeIndex, int readIndex) {
        // Base case: If we\'ve reached the end of the array
        if (readIndex == nums.length) {
            return writeIndex;
        }
        // If the current element is not the same as the previous one, write it to the writeIndex
        if (readIndex == 0 || nums[readIndex] != nums[readIndex - 1]) {
            nums[writeIndex] = nums[readIndex];
            writeIndex++;
        }
        // Recursive call to process the next element
        return removeDuplicatesRecursive(nums, writeIndex, readIndex + 1);
    }

    public int removeDuplicates(int[] nums) {
        return removeDuplicatesRecursive(nums, 0, 0);
    }

    public static void main(String[] args) {
        int[] arr = { 1, 2, 3, 4, 4, 5, 6 };
        Solution solution = new Solution();
        int newLength = solution.removeDuplicates(arr);
        // Print the resulting array up to the new length
        for (int i = 0; i < newLength; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}
