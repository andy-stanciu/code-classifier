class Solution {

    public int removeDuplicates(int[] nums) {
        int temp = 1;
        int initial_idx = 0;
        if (nums.length == 1) {
            return 1;
        } else {
            for (int i = 0; i < nums.length; i++) {
                if (nums[initial_idx] != nums[i]) {
                    initial_idx++;
                    nums[initial_idx] = nums[i];
                    temp++;
                }
            }
        }
        return temp;
    }
}
