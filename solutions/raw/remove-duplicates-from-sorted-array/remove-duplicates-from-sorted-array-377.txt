class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0, j = i + 1, count = 1;
        while (j < nums.length) {
            if (nums[i] != nums[j]) {
                nums[count] = nums[j];
                i = j++;
                count++;
            } else {
                j++;
            }
        }
        return count;
    }
}
