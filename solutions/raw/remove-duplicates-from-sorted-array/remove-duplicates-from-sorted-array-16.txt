class Solution {

    public int removeDuplicates(int[] nums) {
        // using a 2 pointer approach
        // compare
        int r = 0;
        // place
        int l = 1;
        int n = nums.length;
        while (r < n - 1) {
            if (nums[r] == nums[r + 1]) {
                r++;
            } else {
                nums[l] = nums[r + 1];
                l++;
                r++;
            }
        }
        return l;
    }
}
