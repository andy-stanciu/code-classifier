class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 1)
            return 1;
        int i = 0;
        int j = 1;
        while (true) {
            if (j >= nums.length)
                return i + 1;
            if (nums[i] != nums[j]) {
                i++;
                nums[i] = nums[j];
            }
            j++;
        }
    }
}
