class Solution {

    public int removeDuplicates(int[] nums) {
        int[] expectedNums = new int[nums.length];
        int k = 0;
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > 0 && nums[i - 1] == nums[i])
                j++;
            else {
                nums[i - j] = nums[i];
                k++;
            }
        }
        return k;
    }
}
