public class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        // Initialize the count of unique elements to 1
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[k - 1]) {
                // Overwrite the next unique element
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
