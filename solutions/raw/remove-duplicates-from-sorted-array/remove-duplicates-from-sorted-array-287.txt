class Solution {

    public int removeDuplicates(int[] nums) {
        int last_ele = -101, j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != last_ele) {
                nums[j] = nums[i];
                last_ele = nums[i];
                j++;
            }
        }
        return j;
    }
}
