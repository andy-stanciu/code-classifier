class Solution {

    public int removeDuplicates(int[] nums) {
        HashSet<Integer> numsSet = new HashSet<>();
        for (int num : nums) {
            numsSet.add(num);
        }
        int[] tempArr = new int[numsSet.size()];
        int i = 0;
        for (int num : numsSet) {
            tempArr[i] = num;
            i++;
        }
        Arrays.sort(tempArr);
        i = 0;
        for (int tmp : tempArr) {
            nums[i] = tmp;
            i++;
        }
        return tempArr.length;
    }
}
