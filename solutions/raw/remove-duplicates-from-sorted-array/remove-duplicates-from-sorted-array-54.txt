class Solution {

    public int removeDuplicates(int[] nums) {
        int actualCounter = 1;
        int lastIdentifiedNum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            while (i < nums.length && nums[i] == lastIdentifiedNum) {
                i++;
            }
            if (i >= nums.length)
                break;
            lastIdentifiedNum = nums[i];
            swap(nums, actualCounter, i);
            actualCounter++;
        }
        return actualCounter;
    }

    public void swap(int[] nums, int actualCounter, int i) {
        int temp = nums[i];
        nums[i] = nums[actualCounter];
        nums[actualCounter] = temp;
    }
}
