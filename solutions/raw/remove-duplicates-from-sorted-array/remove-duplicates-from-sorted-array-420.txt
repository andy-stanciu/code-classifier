class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        // Initialize the slow pointer
        int i = 0;
        // Loop through the array with the fast pointer
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i]) {
                // Move the slow pointer
                i++;
                // Place the unique element at position i
                nums[i] = nums[j];
            }
        }
        // The number of unique elements is i + 1
        return i + 1;
    }
}
