class Solution {

    public int removeDuplicates(int[] nums) {
        //base condition
        if (nums.length == 0)
            return 0;
        // taking first element index as a=0
        //for loop -- compare two elements at a time,
        // if they are not equal increment a and store i in a
        // return new created a
        int a = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[a] != nums[i]) {
                a++;
                nums[a] = nums[i];
            }
        }
        return a + 1;
    }
}
