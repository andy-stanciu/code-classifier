class Solution {

    public int removeDuplicates(int[] nums) {
        // inplace is required
        int lastPos = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i - 1] != nums[i]) {
                nums[++lastPos] = nums[i];
            }
        }
        return lastPos + 1;
    }
}
