class Solution {

    //ALgo Used: Simple logic using temp index
    //TC: O N , SC: O 1
    public int removeDuplicates(int[] nums) {
        // len of nums
        int n = nums.length;
        // temp index to modify the array
        int j = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] != nums[i - 1]) {
                // if the ith element is not duplicate, then keep on adding on jth index and move on,
                nums[j] = nums[i];
                j++;
            }
            // if duplicates will be there then we will not add it in jth index so in that way we will get the no. of duplicates and also the array will be modified with unique elements
        }
        // return the duplicate count
        return j;
    }
}
