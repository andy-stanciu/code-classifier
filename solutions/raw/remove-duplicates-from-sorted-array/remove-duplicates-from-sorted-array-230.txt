class Solution {

    public int removeDuplicates(int[] nums) {
        int uniqueElementsOccurenece = 1;
        int existingElement = nums[0];
        int changedIndex = 1;
        for (int i = 1; i < nums.length; i++) {
            if (existingElement != nums[i]) {
                uniqueElementsOccurenece += 1;
                existingElement = nums[i];
                nums[changedIndex++] = nums[i];
            }
        }
        return uniqueElementsOccurenece;
    }
}
