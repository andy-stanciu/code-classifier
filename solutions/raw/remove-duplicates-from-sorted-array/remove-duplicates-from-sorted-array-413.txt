class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 1;
        int n = nums.length;
        int l = 0;
        int m = 1;
        while (i < n) {
            if (nums[m] == nums[l]) {
                m++;
            } else if (nums[m] != nums[l] && l < n - 1) {
                nums[l + 1] = nums[m];
                l++;
                m++;
            }
            i++;
        }
        return l + 1;
    }
}
