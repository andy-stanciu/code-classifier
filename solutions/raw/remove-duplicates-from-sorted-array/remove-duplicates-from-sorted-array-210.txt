class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0)
            return 0;
        // Pointer for the position of last unique element
        int i = 0;
        for (int j = 1; j < nums.length; j++) {
            // Check if the current element is different from the last unique element
            if (nums[j] != nums[i]) {
                // Move the unique pointer forward
                i++;
                // Update the unique position with the new unique element
                nums[i] = nums[j];
            }
        }
        // `i + 1` is the number of unique elements
        return i + 1;
    }
}
