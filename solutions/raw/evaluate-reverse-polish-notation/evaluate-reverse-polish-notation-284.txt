class Solution {

    public int evalRPN(String[] tokens) {
        int result = 0;
        Stack<Integer> stack = new Stack<>();
        Set<String> operators = new HashSet<>(Arrays.asList("+", "-", "*", "/"));
        for (String s : tokens) {
            if (operators.contains(s))
                polish(result, s, stack);
            else
                stack.push(Integer.parseInt(s));
        }
        return stack.pop();
    }

    private void polish(int result, String s, Stack<Integer> stack) {
        int a = stack.pop();
        int b = stack.pop();
        if (s.equals("+"))
            stack.push(a + b);
        else if (s.equals("*"))
            stack.push(b * a);
        else if (s.equals("-"))
            stack.push(b - a);
        else if (s.equals("/"))
            stack.push(b / a);
    }
}
