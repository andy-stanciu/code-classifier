class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> s = new Stack<>();
        for (String st : tokens) {
            if (st.equals("+")) {
                int a = s.pop();
                int b = s.pop();
                int res = a + b;
                s.push(res);
            } else if (st.equals("-")) {
                int a = s.pop();
                int b = s.pop();
                int res = b - a;
                s.push(res);
            } else if (st.equals("*")) {
                int a = s.pop();
                int b = s.pop();
                int res = a * b;
                s.push(res);
            } else if (st.equals("/")) {
                int a = s.pop();
                int b = s.pop();
                int res = b / a;
                s.push(res);
            } else {
                s.push(Integer.parseInt(st));
            }
        }
        return s.pop();
    }
}
