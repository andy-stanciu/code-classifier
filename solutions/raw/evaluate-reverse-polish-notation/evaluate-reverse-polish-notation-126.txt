class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String token : tokens) {
            // Enhanced for loop for readability
            switch(token) {
                case "+":
                    stack.push(stack.pop() + stack.pop());
                    break;
                case "-":
                    int num1 = stack.pop();
                    int num2 = stack.pop();
                    stack.push(num2 - num1);
                    break;
                case "*":
                    stack.push(stack.pop() * stack.pop());
                    break;
                case "/":
                    num1 = stack.pop();
                    num2 = stack.pop();
                    // Integer division, ensure correct sign handling
                    stack.push(num2 / num1);
                    break;
                default:
                    // Parse integers directly
                    stack.push(Integer.parseInt(token));
            }
        }
        // Return the final result
        return stack.pop();
    }
}
