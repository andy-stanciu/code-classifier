class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String str : tokens) {
            // we need two number of the mathematical operation, so we pop from the stack twice
            // and since it is given that the input is a valid RPN, so we dont have to check for invalid statck state by calling stack.isEmpty() within every if/else-if block
            if ("+".equals(str)) {
                int i2 = stack.pop();
                int i1 = stack.pop();
                stack.push(i1 + i2);
            } else if ("-".equals(str)) {
                int i2 = stack.pop();
                int i1 = stack.pop();
                stack.push(i1 - i2);
            } else if ("*".equals(str)) {
                int i2 = stack.pop();
                int i1 = stack.pop();
                stack.push(i1 * i2);
            } else if ("/".equals(str)) {
                int i2 = stack.pop();
                int i1 = stack.pop();
                stack.push(i1 / i2);
            } else {
                stack.push(Integer.parseInt(str));
            }
        }
        return stack.pop();
    }
}
