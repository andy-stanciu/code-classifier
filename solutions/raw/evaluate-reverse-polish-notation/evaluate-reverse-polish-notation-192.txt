class Solution {

    public int evalRPN(String[] tokens) {
        int n = tokens.length;
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < n; i++) {
            String c = tokens[i];
            if (c.equals("+")) {
                int temp = st.pop() + st.pop();
                st.push(temp);
            } else if (c.equals("-")) {
                int temp1 = st.pop();
                int temp2 = st.pop();
                st.push(temp2 - temp1);
            } else if (c.equals("*")) {
                int temp = st.pop() * st.pop();
                st.push(temp);
            } else if (c.equals("/")) {
                int temp1 = st.pop();
                int temp2 = st.pop();
                st.push(temp2 / temp1);
            } else
                st.push(Integer.parseInt(c));
        }
        return st.pop();
    }
}
