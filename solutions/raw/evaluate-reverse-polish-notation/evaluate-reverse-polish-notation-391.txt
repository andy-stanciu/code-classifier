class Solution {

    public int evalRPN(String[] tokens) {
        Stack<String> stack = new Stack<>();
        for (int i = 0; i < tokens.length; i++) {
            stack.push(tokens[i]);
            String lastElement = stack.peek();
            if (lastElement.equals("+") || lastElement.equals("-") || lastElement.equals("*") || lastElement.equals("/")) {
                stack.pop();
                int second = Integer.parseInt(stack.pop());
                int first = Integer.parseInt(stack.pop());
                int value = 0;
                if (lastElement.equals("+")) {
                    value = first + second;
                } else if (lastElement.equals("-")) {
                    value = first - second;
                } else if (lastElement.equals("*")) {
                    value = first * second;
                } else if (lastElement.equals("/")) {
                    value = first / second;
                }
                stack.push(Integer.toString(value));
            }
        }
        int ans = Integer.parseInt(stack.pop());
        return ans;
    }
}
