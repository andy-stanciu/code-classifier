class Solution {

    public int evalRPN(String[] arr) {
        int t, b;
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < arr.length; i++) {
            if (arr[i].equals("+")) {
                st.push(st.pop() + st.pop());
            } else if (arr[i].equals("-")) {
                st.push(-st.pop() + st.pop());
            } else if (arr[i].equals("*")) {
                st.push(st.pop() * st.pop());
            } else if (arr[i].equals("/")) {
                t = st.pop();
                b = st.pop();
                st.push(b / t);
            } else {
                st.push(Integer.parseInt(arr[i]));
            }
        }
        return st.pop();
    }
}
