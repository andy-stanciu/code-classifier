class Solution {

    public int evalRPN(String[] tokens) {
        int answer = 0;
        Stack<Integer> st = new Stack<>();
        String expressions = "+-*/";
        for (String token : tokens) {
            if (expressions.indexOf(token) != -1) {
                int d2 = st.pop();
                int d1 = st.pop();
                if (token.equals("+")) {
                    answer = d1 + d2;
                } else if (token.equals("-")) {
                    answer = d1 - d2;
                } else if (token.equals("*")) {
                    answer = d1 * d2;
                } else if (token.equals("/")) {
                    answer = d1 / d2;
                }
                st.push(answer);
            } else {
                st.push(Integer.parseInt(token));
            }
        }
        return st.peek();
    }
}
