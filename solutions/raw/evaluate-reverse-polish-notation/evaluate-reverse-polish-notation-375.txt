class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String tok : tokens) {
            if (tok.equals("/") || tok.equals("*") || tok.equals("+") || tok.equals("-")) {
                int op2 = stack.pop();
                int op1 = stack.pop();
                switch(tok) {
                    case "+":
                        stack.push(op2 + op1);
                        break;
                    case "-":
                        stack.push(op1 - op2);
                        break;
                    case "/":
                        stack.push(op1 / op2);
                        break;
                    case "*":
                        stack.push(op1 * op2);
                        break;
                }
            } else
                stack.push(Integer.parseInt(tok));
        }
        return stack.pop();
    }
}
