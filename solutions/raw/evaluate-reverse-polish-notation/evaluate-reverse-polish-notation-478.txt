class Solution {

    private int operate(int num2, int num1, String s) {
        if (s.equals("*")) {
            return num2 * num1;
        }
        if (s.equals("+")) {
            return num2 + num1;
        }
        if (s.equals("-")) {
            return num2 - num1;
        } else
            return num2 / num1;
    }

    private boolean operator(String s) {
        return s.equals("*") || s.equals("-") || s.equals("+") || s.equals("/");
    }

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        int ans = 0;
        for (String s : tokens) {
            if (operator(s)) {
                int num1 = stack.pop();
                int num2 = stack.pop();
                int x = operate(num2, num1, s);
                stack.push(x);
            } else {
                stack.push(Integer.parseInt(s));
            }
        }
        return stack.pop();
    }
}
