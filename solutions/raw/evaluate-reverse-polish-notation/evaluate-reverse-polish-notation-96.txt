class Solution {

    public int evalRPN(String[] tokens) {
        if (tokens.length == 1)
            return Integer.valueOf(tokens[0]);
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < tokens.length; i++) {
            String item = tokens[i];
            switch(item) {
                case "+":
                    int f1 = stack.pop();
                    int f2 = stack.pop();
                    stack.push(f1 + f2);
                    break;
                case "-":
                    int f3 = stack.pop();
                    int f4 = stack.pop();
                    stack.push(f4 - f3);
                    break;
                case "*":
                    int f5 = stack.pop();
                    int f6 = stack.pop();
                    stack.push(f5 * f6);
                    break;
                case "/":
                    int f7 = stack.pop();
                    int f8 = stack.pop();
                    stack.push(f8 / f7);
                    break;
                default:
                    stack.push(Integer.valueOf(item));
            }
        }
        return stack.pop();
    }
}
