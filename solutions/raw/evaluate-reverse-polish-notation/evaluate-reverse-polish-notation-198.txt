class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> n = new Stack<>();
        for (String s : tokens) {
            if (s.equals("+")) {
                int n2 = n.pop();
                int n1 = n.pop();
                n.push(n1 + n2);
            } else if (s.equals("*")) {
                int n2 = n.pop();
                int n1 = n.pop();
                n.push(n1 * n2);
            } else if (s.equals("/")) {
                int n2 = n.pop();
                int n1 = n.pop();
                n.push(n1 / n2);
            } else if (s.equals("-")) {
                int n2 = n.pop();
                int n1 = n.pop();
                n.push(n1 - n2);
            } else {
                n.push(Integer.valueOf(s));
            }
        }
        return n.pop();
    }
}
