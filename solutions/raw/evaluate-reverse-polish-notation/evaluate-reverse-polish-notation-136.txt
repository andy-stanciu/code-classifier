class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String c : tokens) {
            if (c.equals("+") || c.equals("-") || c.equals("*") || c.equals("/")) {
                int ele1 = stack.pop();
                int ele2 = stack.pop();
                switch(c) {
                    case "+":
                        stack.push(ele1 + ele2);
                        break;
                    case "-":
                        stack.push(ele2 - ele1);
                        break;
                    case "*":
                        stack.push(ele1 * ele2);
                        break;
                    case "/":
                        stack.push(ele2 / ele1);
                        break;
                }
            } else {
                stack.push(Integer.parseInt(c));
            }
        }
        return stack.peek();
    }
}
