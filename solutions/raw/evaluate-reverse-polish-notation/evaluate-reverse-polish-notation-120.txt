class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < tokens.length; i++) {
            if (tokens[i].equals("+")) {
                int val2 = st.pop();
                int val1 = st.pop();
                int t = val1 + val2;
                st.push(t);
            } else if (tokens[i].equals("*")) {
                int val2 = st.pop();
                int val1 = st.pop();
                int t = val1 * val2;
                st.push(t);
            } else if (tokens[i].equals("/")) {
                int val1 = st.pop();
                int val2 = st.pop();
                int t = val2 / val1;
                st.push(t);
            } else if (tokens[i].equals("-")) {
                int val1 = st.pop();
                int val2 = st.pop();
                int t = val2 - val1;
                st.push(t);
            } else {
                st.push(Integer.parseInt(tokens[i]));
            }
        }
        return st.peek();
    }
}
