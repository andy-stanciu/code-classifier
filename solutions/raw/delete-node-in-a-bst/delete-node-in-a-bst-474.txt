/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    TreeNode deleteNode(TreeNode root, int key) {
        if (root == null)
            return null;
        root.left = deleteNode(root.left, key);
        root.right = deleteNode(root.right, key);
        if (root.val == key) {
            TreeNode temp = root.right;
            if (temp == null)
                return root.left;
            if (root.left == null)
                return root.right;
            root = root.left;
            TreeNode gr = root;
            while (gr.right != null) {
                gr = gr.right;
            }
            gr.right = temp;
        }
        return root;
    }
}
