class Solution {

    // Helper function to find the rightmost node in a subtree
    public TreeNode findRight(TreeNode root) {
        while (root.right != null) {
            root = root.right;
        }
        return root;
    }

    public TreeNode deleteRootNode(TreeNode root) {
        if (root.left == null)
            return root.right;
        if (root.right == null)
            return root.left;
        TreeNode rightChild = root.right;
        TreeNode lastRight = findRight(root.left);
        lastRight.right = rightChild;
        return root.left;
    }

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            return deleteRootNode(root);
        }
        TreeNode dummy = root;
        while (root != null) {
            if (root.val > key) {
                if (root.left != null && root.left.val == key) {
                    root.left = deleteRootNode(root.left);
                    break;
                } else {
                    root = root.left;
                }
            } else {
                if (root.right != null && root.right.val == key) {
                    root.right = deleteRootNode(root.right);
                    break;
                } else {
                    root = root.right;
                }
            }
        }
        return dummy;
    }
}
