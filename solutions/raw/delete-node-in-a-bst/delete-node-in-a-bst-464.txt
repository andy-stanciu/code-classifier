/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static TreeNode findlastnode(TreeNode root) {
        while (root.right != null) {
            root = root.right;
        }
        return root;
    }

    public static TreeNode deletion(TreeNode root) {
        if (root == null) {
            return null;
        }
        if (root.left == null) {
            return root.right;
        }
        if (root.right == null) {
            return root.left;
        }
        TreeNode r = root.right;
        TreeNode lastrightnode = findlastnode(root.left);
        lastrightnode.right = r;
        return root.left;
    }

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val == key) {
            return deletion(root);
        }
        //searching the node
        TreeNode temp = root;
        while (root != null) {
            if (root.val > key) {
                if (root.left != null && root.left.val != key) {
                    root = root.left;
                } else {
                    root.left = deletion(root.left);
                    break;
                }
            } else {
                if (root.right != null && root.right.val != key) {
                    root = root.right;
                } else {
                    root.right = deletion(root.right);
                    break;
                }
            }
        }
        return temp;
    }
}
