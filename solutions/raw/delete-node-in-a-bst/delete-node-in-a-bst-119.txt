/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root != null && root.val == key)
            return deleteHelper(root);
        TreeNode cur = root;
        while (cur != null) {
            if (cur.val < key) {
                if (cur.right != null && cur.right.val == key) {
                    cur.right = deleteHelper(cur.right);
                } else {
                    cur = cur.right;
                }
            } else {
                if (cur.left != null && cur.left.val == key) {
                    cur.left = deleteHelper(cur.left);
                } else {
                    cur = cur.left;
                }
            }
        }
        return root;
    }

    public TreeNode deleteHelper(TreeNode root) {
        if (root == null)
            return root;
        if (root.left == null) {
            return root.right;
        }
        if (root.right == null) {
            return root.left;
        }
        TreeNode right = root.right;
        TreeNode left = root.left;
        TreeNode largestOfLeft = findExtremeRight(root.left);
        largestOfLeft.right = right;
        return left;
    }

    public TreeNode findExtremeRight(TreeNode root) {
        if (root == null)
            return root;
        while (root.right != null) {
            root = root.right;
        }
        return root;
    }
}
