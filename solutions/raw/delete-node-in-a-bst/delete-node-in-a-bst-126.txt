final class Solution {

    public TreeNode deleteNode(final TreeNode root, final int key) {
        // key not found
        if (root == null)
            return root;
        if (key < root.val)
            root.left = deleteNode(root.left, key);
        else if (key > root.val)
            root.right = deleteNode(root.right, key);
        else {
            // key found delete process begins
            if (// works when the deleted node is a leaf or have no left
            root.left == null)
                // works when the deleted node is a leaf or have no left
                return root.right;
            else // works only when the deleted node has no right
            if (root.right == null)
                return root.left;
            // deleted node has both child
            var curr = root.right;
            // find the inorder sucessor
            while (curr.left != null) curr = curr.left;
            root.val = curr.val;
            root.right = deleteNode(root.right, root.val);
        }
        return root;
    }
}
