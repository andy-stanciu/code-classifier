class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val) {
            root.left = deleteNode(root.left, key);
        } else if (key > root.val) {
            root.right = deleteNode(root.right, key);
        } else {
            //No left Node available
            if (root.left == null)
                return root.right;
            else //No right Node available
            if (root.right == null)
                return root.left;
            //pick the smaller value of the right Node
            root.val = small(root.right);
            //delete the picked smaller Node from right Node
            root.right = deleteNode(root.right, root.val);
        }
        return root;
    }

    private int small(TreeNode root) {
        TreeNode temp = root;
        while (temp.left != null) {
            temp = temp.left;
        }
        return temp.val;
    }
}
