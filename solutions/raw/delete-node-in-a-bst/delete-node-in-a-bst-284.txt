/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val > key) {
            root.left = deleteNode(root.left, key);
        } else if (root.val < key) {
            root.right = deleteNode(root.right, key);
        } else {
            //case when root.val == key //node to be deleted
            //case 1 : root is having no child
            if (root.left == null && root.right == null)
                return null;
            //case 2 : root is having single child
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                //case 3 : root is having 2 children
                TreeNode inOrderNext = getInOrderNext(root.right);
                root.val = inOrderNext.val;
                root.right = deleteNode(root.right, inOrderNext.val);
            }
        }
        return root;
    }

    public TreeNode getInOrderNext(TreeNode node) {
        while (node.left != null) {
            node = node.left;
        }
        return node;
    }
}
