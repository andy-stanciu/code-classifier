/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        // found the node to delete
        if (root.val == key) {
            // now there are three cases
            // degree=0,1,2
            if (root.left == null && root.right == null) {
                // make root null and return
                root = null;
                return null;
            } else // if root has one child
            if (root.left == null || root.right == null) {
                TreeNode child = root.left != null ? root.left : root.right;
                root = null;
                return child;
            } else // if root has both children
            {
                // get the minValue node in right side
                int minValueInRight = getMinValue(root.right);
                // update the value
                root.val = minValueInRight;
                // now delete the node with min value i.e leftmost node in right side
                root.right = deleteNode(root.right, minValueInRight);
                return root;
            }
        }
        // find in left and right
        root.left = deleteNode(root.left, key);
        root.right = deleteNode(root.right, key);
        return root;
    }

    private int getMinValue(TreeNode node) {
        if (node == null) {
            return Integer.MIN_VALUE;
        }
        if (node.left == null) {
            return node.val;
        }
        return getMinValue(node.left);
    }
}
