/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (root.val == key) {
            if (root.left == null && root.right == null) {
                return null;
            } else if (root.left == null && root.right != null) {
                return root.right;
            } else if (root.left != null && root.right == null) {
                return root.left;
            } else {
                TreeNode minInRight = search(root.right);
                if (minInRight == null) {
                    root.right.left = root.left;
                    return root.right;
                } else {
                    // System.out.println(minInRight.val);
                    TreeNode updatedroot = deleteNode(root, minInRight.val);
                    root.val = minInRight.val;
                    return root;
                }
            }
        } else if (key < root.val) {
            root.left = deleteNode(root.left, key);
            return root;
        } else {
            root.right = deleteNode(root.right, key);
            return root;
        }
    }

    public TreeNode search(TreeNode root) {
        if (root == null) {
            return root;
        }
        if (root.left == null)
            return root;
        return search(root.left);
    }
}
