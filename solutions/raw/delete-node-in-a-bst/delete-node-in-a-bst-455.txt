/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key < root.val) {
            TreeNode lres = deleteNode(root.left, key);
            root.left = lres;
        } else if (key > root.val) {
            TreeNode rres = deleteNode(root.right, key);
            root.right = rres;
        } else {
            if (root.left == null && root.right == null) {
                return null;
            } else if (root.left != null && root.right == null) {
                return root.left;
            } else if (root.left == null && root.right != null) {
                return root.right;
            } else {
                int justsmaller = max(root.left);
                root.val = justsmaller;
                TreeNode res = deleteNode(root.left, justsmaller);
                root.left = res;
            }
        }
        return root;
    }

    static int max(TreeNode node) {
        if (node == null) {
            return Integer.MAX_VALUE;
        }
        while (node.right != null) {
            node = node.right;
        }
        return node.val;
    }
}
