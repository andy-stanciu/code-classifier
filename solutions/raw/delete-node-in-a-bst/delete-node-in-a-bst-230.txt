/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            TreeNode left = root.left;
            TreeNode right = root.right;
            if (left == null)
                return root.right;
            if (right == null)
                return root.left;
            TreeNode newNode;
            newNode = deleteAndGetSmallest(right);
            newNode.left = left;
            return newNode;
        } else if (root.val > key) {
            root.left = deleteNode(root.left, key);
        } else {
            root.right = deleteNode(root.right, key);
        }
        return root;
    }

    private TreeNode deleteAndGetSmallest(TreeNode node) {
        if (node == null)
            return null;
        TreeNode head = node;
        TreeNode prev = node;
        while (node.left != null) {
            prev = node;
            node = node.left;
        }
        prev.left = node.right;
        if (head == node) {
            return head;
        } else {
            return new TreeNode(node.val, null, head);
        }
    }
}
