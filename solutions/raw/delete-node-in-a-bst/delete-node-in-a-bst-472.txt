/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode delete(TreeNode root) {
        if (root.left == null && root.right == null) {
            // zero child
            return null;
        } else if (root.left == null) {
            // having one child
            return root.right;
        } else if (root.right == null) {
            return root.left;
        } else {
            // having two children
            System.out.println(successor(root.right).val);
            // to find the last elements and replace
            root.val = successor(root.right).val;
            // to remove the last elements
            root.right = deleteSuccessor(root.right);
            return root;
        }
    }

    public TreeNode deleteSuccessor(TreeNode root) {
        if (root.left == null && root.right != null) {
            return root.right;
        } else if (root.left == null) {
            return null;
        }
        root.left = deleteSuccessor(root.left);
        return root;
    }

    public TreeNode successor(TreeNode root) {
        if (root.left == null) {
            return root;
        }
        return successor(root.left);
    }

    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (root.val == key) {
            root = delete(root);
        } else if (key > root.val) {
            root.right = deleteNode(root.right, key);
        } else {
            root.left = deleteNode(root.left, key);
        }
        return root;
    }
}
