/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        findtraversal(root, li);
        return li;
    }

    public void findtraversal(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        findtraversal(root.left, ans);
        findtraversal(root.right, ans);
        ans.add(root.val);
    }
}
