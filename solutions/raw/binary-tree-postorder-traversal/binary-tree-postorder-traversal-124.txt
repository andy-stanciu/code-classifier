class Solution {

    List<Integer> l = new ArrayList<Integer>();

    public void find(TreeNode root) {
        if (root == null)
            return;
        postorderTraversal(root.left);
        postorderTraversal(root.right);
        l.add(root.val);
    }

    public List<Integer> postorderTraversal(TreeNode root) {
        find(root);
        return l;
    }
}
