class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> post = new ArrayList<>();
        postorder(post, root);
        return post;
    }

    static void postorder(List<Integer> post, TreeNode node) {
        if (node == null)
            return;
        TreeNode temp = node;
        postorder(post, node.left);
        postorder(post, node.right);
        post.add(temp.val);
    }
}
