class Solution {

    public void postorder(TreeNode node, List<Integer> post) {
        if (node == null)
            return;
        postorder(node.left, post);
        postorder(node.right, post);
        post.add(node.val);
    }

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> post = new ArrayList<>();
        postorder(root, post);
        return post;
    }
}
