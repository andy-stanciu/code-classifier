class Solution {

    public List<Integer> l = new ArrayList<>();

    public void postorderT(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            l.add(root.val);
            return;
        }
        if (root.left != null) {
            postorderT(root.left);
        }
        if (root.right != null) {
            postorderT(root.right);
        }
        l.add(root.val);
        return;
    }

    public List<Integer> postorderTraversal(TreeNode root) {
        postorderT(root);
        return l;
    }
}
