class Solution {

    List<Integer> nodes;

    public List<Integer> postorderTraversal(TreeNode root) {
        nodes = new ArrayList<>();
        traverse(root);
        return nodes;
    }

    private void traverse(TreeNode node) {
        if (node == null) {
            return;
        }
        traverse(node.left);
        traverse(node.right);
        nodes.add(node.val);
    }
}
