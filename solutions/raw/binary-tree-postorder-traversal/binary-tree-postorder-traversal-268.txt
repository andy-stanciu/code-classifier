class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> postOrder = new ArrayList<>();
        if (root == null)
            return postOrder;
        List<Integer> leftTraversal = postorderTraversal(root.left);
        postOrder.addAll(leftTraversal);
        List<Integer> rightTraversal = postorderTraversal(root.right);
        postOrder.addAll(rightTraversal);
        postOrder.add(root.val);
        return postOrder;
    }
}
