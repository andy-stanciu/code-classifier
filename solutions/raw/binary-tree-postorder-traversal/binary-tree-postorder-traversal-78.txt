class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        Stack<TreeNode> st = new Stack<>();
        TreeNode node = root;
        while (node != null || !st.isEmpty()) {
            while (node != null) {
                st.push(node);
                st.push(node);
                node = node.left;
            }
            node = st.pop();
            if (!st.isEmpty() && st.peek() == node) {
                node = node.right;
            } else {
                res.add(node.val);
                node = null;
            }
        }
        return res;
    }
}
