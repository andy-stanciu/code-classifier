class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        Stack<TreeNode> stk = new Stack<>();
        TreeNode prev = null;
        for (; !stk.isEmpty() || root != null; ) {
            if (root != null) {
                stk.push(root);
                root = root.left;
            } else {
                root = stk.peek();
                if (root.right != null && root.right != prev) {
                    root = root.right;
                } else {
                    res.add(root.val);
                    prev = root;
                    stk.pop();
                    root = null;
                }
            }
        }
        return res;
    }
}
