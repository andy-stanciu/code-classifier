/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        postTraversal(root, ans);
        return ans;
    }

    public void postTraversal(TreeNode node, List<Integer> ans) {
        if (node == null) {
            return;
        }
        postTraversal(node.left, ans);
        postTraversal(node.right, ans);
        ans.add(node.val);
        return;
    }
}
