/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        ArrayList<Integer> ar = new ArrayList<>();
        PostOrder(root, ar);
        System.out.println(ar);
        return ar;
    }

    public void PostOrder(TreeNode head, ArrayList<Integer> ar) {
        if (head == null)
            return;
        PostOrder(head.left, ar);
        PostOrder(head.right, ar);
        ar.add(head.val);
    }
}
