class Solution {

    // This is the main method that starts the postorder traversal.
    public List<Integer> postorderTraversal(TreeNode root) {
        // Create a new empty list to store the traversal result.
        List<Integer> list = new ArrayList<>();
        // Call the helper method to perform the traversal and return the result list.
        return postorderTraversal(root, list);
    }

    // This helper method performs the postorder traversal recursively.
    public List<Integer> postorderTraversal(TreeNode root, List<Integer> list) {
        // If the current node is null, return the list as it is.
        if (root == null)
            return list;
        // Recursively perform postorder traversal on the left subtree.
        postorderTraversal(root.left, list);
        // Recursively perform postorder traversal on the right subtree.
        postorderTraversal(root.right, list);
        // Add the value of the current node to the list.
        list.add(root.val);
        // Return the updated list.
        return list;
    }
}
