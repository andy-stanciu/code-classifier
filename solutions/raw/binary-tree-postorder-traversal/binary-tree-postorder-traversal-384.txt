class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        postOrder(root);
        return ans;
    }

    List<Integer> ans = new ArrayList<>();

    public void postOrder(TreeNode root) {
        if (root == null) {
            return;
        }
        postOrder(root.left);
        postOrder(root.right);
        ans.add(root.val);
    }
}
