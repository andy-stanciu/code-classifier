class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        TreeNode curr = root;
        while (curr != null) {
            if (curr.right == null) {
                list.add(curr.val);
                curr = curr.left;
            } else {
                TreeNode next = curr.right;
                while (next.left != null && next.left != curr) {
                    next = next.left;
                }
                if (next.left == null) {
                    next.left = curr;
                    list.add(curr.val);
                    curr = curr.right;
                } else {
                    next.left = null;
                    curr = curr.left;
                }
            }
        }
        Collections.reverse(list);
        return list;
    }
}
