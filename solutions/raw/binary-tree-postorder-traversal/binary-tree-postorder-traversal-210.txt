class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        postOrder(root, ans);
        return ans;
    }

    public void postOrder(TreeNode node, List<Integer> ans) {
        if (node == null)
            return;
        postOrder(node.left, ans);
        postOrder(node.right, ans);
        ans.add(node.val);
    }
}
