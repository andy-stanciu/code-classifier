class Solution {

    List<Integer> res = new ArrayList<>();

    void postOrder(TreeNode root) {
        if (root != null) {
            postOrder(root.left);
            postOrder(root.right);
            res.add(root.val);
        }
    }

    public List<Integer> postorderTraversal(TreeNode root) {
        postOrder(root);
        return res;
    }
}
