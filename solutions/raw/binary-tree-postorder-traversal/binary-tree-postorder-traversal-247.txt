/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    Stack<Integer> stack = new Stack<>();

    public List<Integer> postorderTraversal(TreeNode root) {
        if (root == null) {
            return new ArrayList();
        }
        stack.push(root.val);
        preParePostOrder(root, stack);
        List<Integer> result = new ArrayList();
        while (!stack.isEmpty()) {
            result.add(stack.pop());
        }
        return result;
    }

    private void preParePostOrder(TreeNode root, Stack<Integer> stack) {
        if (root == null) {
            return;
        }
        if (root.right != null) {
            stack.push(root.right.val);
            preParePostOrder(root.right, stack);
        }
        if (root.left != null) {
            stack.push(root.left.val);
            preParePostOrder(root.left, stack);
        }
    }
}
