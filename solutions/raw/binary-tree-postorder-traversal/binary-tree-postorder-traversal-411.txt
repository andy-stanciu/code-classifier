class Solution {

    private void postOrder(List<Integer> numbers, TreeNode root) {
        if (root == null)
            return;
        postOrder(numbers, root.left);
        postOrder(numbers, root.right);
        numbers.add(root.val);
    }

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> numbers = new ArrayList<>();
        postOrder(numbers, root);
        return numbers;
    }
}
