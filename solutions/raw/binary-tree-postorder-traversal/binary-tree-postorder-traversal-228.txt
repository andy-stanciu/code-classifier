class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        pot(root, res);
        return res;
    }

    public void pot(TreeNode root, List<Integer> res) {
        if (root != null) {
            pot(root.left, res);
            pot(root.right, res);
            res.add(root.val);
        }
    }
}
