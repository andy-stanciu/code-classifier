class Solution {

    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        postorder(root, ans);
        return ans;
    }

    public void postorder(TreeNode r, List<Integer> ans) {
        if (r == null)
            return;
        postorder(r.left, ans);
        postorder(r.right, ans);
        ans.add(r.val);
    }
}
