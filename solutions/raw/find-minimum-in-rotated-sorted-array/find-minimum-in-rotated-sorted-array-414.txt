class Solution {

    public int findMin(int[] nums) {
        int si = 0;
        int ei = nums.length - 1;
        int min = Integer.MAX_VALUE;
        while (si <= ei) {
            int mid = si + (ei - si) / 2;
            if (nums[si] <= nums[mid]) {
                min = Math.min(nums[si], min);
                si++;
            } else {
                min = Math.min(nums[mid], min);
                ei--;
            }
        }
        return min;
    }
}
