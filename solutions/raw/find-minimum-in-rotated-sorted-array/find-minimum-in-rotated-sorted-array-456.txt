class Solution {

    public int findMin(int[] nums) {
        int num = 0;
        int min_num = nums[0];
        for (int i = 0; i < nums.length; i++) {
            num = nums[i];
            if (num < min_num) {
                min_num = num;
            }
        }
        return min_num;
    }
}
