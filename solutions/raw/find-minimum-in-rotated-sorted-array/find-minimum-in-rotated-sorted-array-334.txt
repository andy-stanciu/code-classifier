class Solution {

    public int findMin(int[] nums) {
        int n = nums.length;
        int minimum = Integer.MAX_VALUE;
        int low = 0;
        int high = n - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (nums[low] <= nums[mid]) {
                // if left half is sorted
                minimum = Math.min(minimum, nums[low]);
                low = mid + 1;
            } else {
                // if right half is  sorted
                minimum = Math.min(minimum, nums[mid]);
                high = mid - 1;
            }
        }
        return minimum;
    }
}
