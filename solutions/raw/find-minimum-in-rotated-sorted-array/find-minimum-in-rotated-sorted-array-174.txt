class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        while (low <= high) {
            int check = low + (high - low) / 2;
            if (nums[low] > nums[check]) {
                high = check;
            } else if (nums[high] < nums[check]) {
                low = check + 1;
            } else {
                return nums[low];
            }
        }
        return low;
    }
}
