class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] > nums[high]) {
                //small element is in the right side
                low = mid + 1;
            } else {
                // small element is in the left side
                high = mid;
            }
        }
        return nums[high];
    }
}
