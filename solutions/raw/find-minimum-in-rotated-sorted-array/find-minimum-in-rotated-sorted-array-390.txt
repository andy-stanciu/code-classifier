class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        while (// cause in this case they will possible over lap cause the mid is not returning;
        low < high) {
            int mid = low + (high - low) / 2;
            if (// then possibly the minimum numbers are on the right
            nums[high] < nums[mid]) {
                low = mid + 1;
            } else // then possibly the minimum number are on the left
            {
                high = mid;
            }
        }
        return nums[low];
    }
}
