class Solution {

    public int findMin(int[] nums) {
        int n = nums.length;
        int s = 0;
        int e = n - 1;
        while (s <= e) {
            int mid = s + (e - s) / 2;
            if (nums[s] <= nums[mid]) {
                if (mid < nums.length - 1 && nums[mid] > nums[mid + 1]) {
                    return nums[mid + 1];
                } else {
                    s = mid + 1;
                }
            } else {
                e = mid - 1;
            }
        }
        return nums[0];
    }
}
