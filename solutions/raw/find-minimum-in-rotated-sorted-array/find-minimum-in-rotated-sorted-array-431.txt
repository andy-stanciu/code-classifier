class Solution {

    public int findMin(int[] nums, int leftIndex, int rightIndex) {
        // base case: array between leftIndex and rightIndex is sorted ?
        if (nums[leftIndex] <= nums[rightIndex]) {
            return nums[leftIndex];
        }
        int key = leftIndex + (rightIndex - leftIndex) / 2;
        return nums[key] > nums[rightIndex] ? findMin(nums, key + 1, rightIndex) : findMin(nums, leftIndex, key);
    }

    public int findMin(int[] nums) {
        // binary search , find range where nums [i] > nums [j]
        return findMin(nums, 0, nums.length - 1);
    }
}
