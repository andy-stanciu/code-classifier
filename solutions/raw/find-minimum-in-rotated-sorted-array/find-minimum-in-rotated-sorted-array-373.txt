class Solution {

    public int findMin(int[] nums) {
        int right = nums.length - 1;
        int left = 0;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (nums[mid] > nums[nums.length - 1])
                left = mid + 1;
            else
                right = mid - 1;
        }
        return nums[left];
    }
}
