class Solution {

    public int findMin(int[] nums) {
        return binarySearch(nums, 0, nums.length - 1);
    }

    int binarySearch(int[] nums, int i, int j) {
        int p = (i + j) / 2;
        if (i == j) {
            return nums[i];
        }
        return Math.min(binarySearch(nums, i, p), binarySearch(nums, p + 1, j));
    }
}
