class Solution {

    public int findMin(int[] nums) {
        int len = nums.length - 1;
        int mid = len / 2;
        int start = 0;
        int end = len;
        if (nums.length == 1) {
            return nums[0];
        }
        if (nums.length == 2) {
            return Math.min(nums[0], nums[1]);
        }
        if (nums[mid] < nums[mid + 1]) {
            end = mid;
        } else {
            start = mid + 1;
        }
        if (nums[start] < nums[end]) {
            start = 0;
            end = len;
        }
        int min = nums[start];
        for (int i = start; i <= end; i++) {
            min = Math.min(min, nums[i]);
        }
        return min;
    }
}
