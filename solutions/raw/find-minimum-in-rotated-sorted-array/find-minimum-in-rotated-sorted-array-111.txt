public class Solution {

    public int FindMin(int[] nums) {
        //low index
        int li = 0;
        //high index
        int hi = nums.Length - 1;
        while (hi > li) {
            int midi = li + (hi - li) / 2;
            if (nums[midi] < nums[hi]) {
                hi = midi;
            } else {
                li = midi + 1;
            }
        }
        return nums[li];
    }
}
