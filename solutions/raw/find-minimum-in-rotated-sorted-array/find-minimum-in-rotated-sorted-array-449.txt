class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int min = 5001;
        while (low <= high) {
            int mid = (low + high) / 2;
            //left sorted
            if (nums[low] <= nums[mid]) {
                if (nums[low] < min)
                    min = nums[low];
                low = mid + 1;
            } else {
                if (nums[mid] < min)
                    min = nums[mid];
                high = mid - 1;
            }
        }
        return min;
    }
}
