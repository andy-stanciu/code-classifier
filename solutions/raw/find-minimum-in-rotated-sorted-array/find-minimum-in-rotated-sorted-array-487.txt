class Solution {

    private int get(int[] nums, int l, int r) {
        if (l < r) {
            int m = l + (r - l) / 2;
            if (nums[m] > nums[r]) {
                return get(nums, m + 1, r);
            } else
                return get(nums, l, m);
        }
        return nums[l];
    }

    public int findMin(int[] nums) {
        return get(nums, 0, nums.length - 1);
    }
}
