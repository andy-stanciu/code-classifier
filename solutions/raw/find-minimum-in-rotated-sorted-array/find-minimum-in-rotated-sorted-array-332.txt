class Solution {

    public int findMin(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int mid;
        while (left < right) {
            // Calculate mid using left + right / 2
            mid = (left + right) / 2;
            if (nums[mid] > nums[right]) {
                // The min is in the right half
                left = mid + 1;
            } else {
                // The min is in the left half, including mid
                right = mid;
            }
        }
        // The minimum element
        return nums[left];
    }
}
