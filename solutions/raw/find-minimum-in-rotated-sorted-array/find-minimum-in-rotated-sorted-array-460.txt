class Solution {

    public int findMin(int[] nums) {
        int st = 0, end = nums.length - 1;
        if (nums[st] < nums[end])
            return nums[st];
        while (st + 1 < end) {
            int mid = st + (end - st) / 2;
            if (nums[mid] > nums[end])
                st = mid;
            else
                end = mid;
        }
        return Math.min(nums[st], nums[end]);
    }
}
