class Solution {

    public int findMin(int[] nums) {
        int ans = Integer.MAX_VALUE;
        int p1 = 0, p2 = nums.length - 1;
        ans = Math.min(nums[p1], nums[p2]);
        while (p1 <= p2) {
            ans = Math.min(ans, nums[p1]);
            ans = Math.min(ans, nums[p2]);
            p1++;
            p2--;
        }
        return ans;
    }
}
