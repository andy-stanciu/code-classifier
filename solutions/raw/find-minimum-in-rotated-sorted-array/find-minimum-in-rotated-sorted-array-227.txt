class Solution {

    public int findMin(int[] nums) {
        int left = 0;
        int right = 0;
        int min = nums[0];
        int i = 0;
        int j = nums.length - 1;
        for (j = nums.length - 1; i < j; i++, j--) {
            min = Math.min(min, Math.min(nums[i], nums[j]));
        }
        if (nums[i] < min) {
            return nums[i];
        } else if (nums[j] < min) {
            return nums[j];
        }
        return min;
    }
}
