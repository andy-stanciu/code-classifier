class Solution {

    public int findMin(int[] nums) {
        int a = nums[0];
        int right = nums.length - 1;
        int left = 0;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (nums[mid] < a) {
                a = nums[mid];
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return a;
    }
}
