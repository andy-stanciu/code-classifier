class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        if (nums[high] >= nums[low])
            return nums[low];
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] >= nums[0])
                low = mid + 1;
            else
                high = mid - 1;
        }
        return nums[low];
    }
}
