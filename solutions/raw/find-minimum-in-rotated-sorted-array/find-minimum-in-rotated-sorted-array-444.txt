class Solution {

    public int findMin(int[] nums) {
        int s = 0, e = nums.length - 1;
        int min = Integer.MAX_VALUE;
        while (s <= e) {
            int mid = (s + e) / 2;
            if (nums[mid] < nums[e]) {
                min = Math.min(min, nums[mid]);
                e = mid - 1;
            } else {
                min = Math.min(nums[s], min);
                s = mid + 1;
            }
        }
        return min;
    }
}
