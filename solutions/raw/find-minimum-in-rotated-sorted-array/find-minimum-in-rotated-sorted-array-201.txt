class Solution {

    public int bs(int[] nums, int l, int u) {
        int mid = (l + u) / 2;
        if (l >= u)
            return nums[mid];
        if (nums[mid] < nums[l] || nums[mid] < nums[u]) {
            return bs(nums, l, mid);
        } else
            return bs(nums, mid + 1, u);
    }

    public int findMin(int[] nums) {
        return bs(nums, 0, nums.length - 1);
    }
}
