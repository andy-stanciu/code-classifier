class Solution {

    public int findMin(int[] nums) {
        int left = 0, right = nums.length - 1;
        int ans = -1;
        int min = Integer.MAX_VALUE;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] >= nums[left]) {
                //If left is sorted take the minimum value and move to right
                ans = nums[left];
                if (ans < min) {
                    min = ans;
                }
                left = mid + 1;
            } else {
                //Else if left is not sorted, definitely right is sorted, so take minimum value from right and move to left
                ans = nums[mid];
                if (ans < min) {
                    min = ans;
                }
                right = mid - 1;
            }
        }
        return min;
    }
}
