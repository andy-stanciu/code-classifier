class Solution {

    public int findMin(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int min = nums[right];
        if (nums.length == 1) {
            min = nums[0];
        }
        while (left < right) {
            //int mid=left+(right-left)/2;
            int mid = (left + right) / 2;
            if (nums[mid] < nums[right]) {
                right = mid;
                min = nums[right];
            } else {
                left = mid + 1;
            }
        }
        return min;
    }
}
