class Solution {

    public int findMin(int[] nums) {
        int low = 0, high = nums.length - 1;
        int mid = 0;
        int ans = Integer.MAX_VALUE;
        while (low <= high) {
            mid = low + (high - low) / 2;
            if (nums[mid] >= nums[low]) {
                ans = Math.min(ans, nums[low]);
                low = mid + 1;
            } else {
                ans = Math.min(nums[mid], ans);
                high = mid - 1;
            }
        }
        return ans;
    }
}
