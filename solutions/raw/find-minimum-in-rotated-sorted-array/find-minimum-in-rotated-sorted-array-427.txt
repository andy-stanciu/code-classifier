class Solution {

    public int findMin(int[] nums) {
        int l = 0, u = nums.length - 1, mid = 0;
        while (l < u) {
            mid = (l + u) / 2;
            if (nums[mid] < nums[nums.length - 1])
                u = mid;
            else
                l = mid + 1;
        }
        return nums[l];
    }
}
