class Solution {

    public int findMin(int[] nums) {
        int min = nums[0];
        // Loop through the array
        for (int i = 1; i < nums.length; i++) {
            // If a smaller value is found, update min
            if (nums[i] < min) {
                min = nums[i];
            }
        }
        return min;
    }
}
