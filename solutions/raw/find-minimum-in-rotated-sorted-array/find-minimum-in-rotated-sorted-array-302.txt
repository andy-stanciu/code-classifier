class Solution {

    public int findMin(int[] nums) {
        int min = nums[0];
        int val = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < min) {
                min = nums[i];
                min = Math.min(min, nums[i]);
                return min;
            }
        }
        return min;
    }
}
