class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int min = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = (high + low) / 2;
            if (//if 0th element is smaller than mid means left portion is sorted..
            nums[low] <= nums[mid]) {
                min = Math.min(min, nums[low]);
                low = mid + 1;
            } else if (//if mid element is smaller than n-1th element so right side is sorted and mid is smallest.
            nums[mid] <= nums[high]) {
                min = Math.min(min, nums[mid]);
                high = mid - 1;
            }
        }
        return min;
    }
}
