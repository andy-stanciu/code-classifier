class Solution {

    public int findMin(int[] nums) {
        int pivot = -1;
        int min = nums[0];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i + 1] < nums[i]) {
                pivot = i + 1;
                break;
            }
        }
        if (pivot == -1) {
            return nums[0];
        }
        if (nums[pivot] < min) {
            min = nums[pivot];
        }
        return min;
    }
}
