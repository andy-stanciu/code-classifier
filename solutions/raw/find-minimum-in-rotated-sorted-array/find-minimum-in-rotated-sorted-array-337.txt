class Solution {

    public int findMin(int[] nums) {
        int low = 0, high = nums.length - 1;
        while (high - low >= 2) {
            int mid = low + (high - low) / 2;
            if (nums[mid] <= nums[high]) {
                high = mid;
            } else {
                low = mid;
            }
        }
        return Math.min(nums[low], nums[high]);
    }
}
