class Solution {

    public int findMin(int[] nums) {
        int low = 0, high = nums.length - 1, min = nums[0];
        while (low <= high) {
            // to handle the case when array is already sorted
            if (nums[low] < nums[high]) {
                // not putting <= because the array contains unique elements
                min = Math.min(min, nums[low]);
                break;
            }
            int mid = low + (high - low) / 2;
            min = Math.min(min, nums[mid]);
            if (// doing >= because of the edge case when mid and low points at the same index
            nums[mid] >= nums[high])
                low = mid + 1;
            else
                high = mid - 1;
        }
        return min;
    }
}
