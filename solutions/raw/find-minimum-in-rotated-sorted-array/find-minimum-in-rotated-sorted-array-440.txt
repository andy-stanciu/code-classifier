class Solution {

    public int findMin(int[] arr) {
        int start = 0, end = arr.length - 1, mid = 0, min = 5001;
        while (start <= end) {
            mid = start + (end - start) / 2;
            if (arr[start] <= arr[mid]) {
                min = Math.min(min, arr[start]);
                start = mid + 1;
            } else {
                min = Math.min(min, arr[mid + 1]);
                end = mid;
            }
        }
        return min;
    }
}
