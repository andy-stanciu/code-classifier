class Solution {

    public int findMin(int[] nums) {
        int min1 = Integer.MAX_VALUE;
        if (nums.length == 1) {
            return nums[0];
        }
        if (nums[0] < nums[nums.length - 1]) {
            return nums[0];
        } else {
            for (int i = 0; i < nums.length - 1; i++) {
                if (nums[i] > nums[i + 1]) {
                    return nums[i + 1];
                }
            }
        }
        return 0;
    }
}
