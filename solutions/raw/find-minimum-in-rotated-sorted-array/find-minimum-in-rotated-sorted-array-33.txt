class Solution {

    public int findMin(int[] nums) {
        final int last = nums.length - 1;
        final int cut = nums[last];
        if (cut >= nums[0])
            return nums[0];
        int left = 0, right = last, mid;
        while (left <= right) {
            //Safe due to the constraint on nums.length;
            mid = (left + right) / 2;
            int midVal = nums[mid];
            if (midVal > cut) {
                if (midVal > nums[mid + 1])
                    return nums[mid + 1];
                left = mid + 1;
            } else {
                if (midVal < nums[mid - 1])
                    return midVal;
                right = mid - 1;
            }
        }
        return -1;
    }
}
