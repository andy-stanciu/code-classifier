class Solution {

    public int findMin(int[] nums) {
        return findMinElement(nums, 0, nums.length - 1, Integer.MAX_VALUE);
    }

    public static int findMinElement(int[] nums, int low, int high, int min) {
        if (low > high)
            return min;
        int mid = (low + high) / 2;
        if (nums[mid] < nums[high]) {
            min = Math.min(min, nums[mid]);
            high = mid - 1;
        } else if (nums[mid] >= nums[low]) {
            min = Math.min(min, nums[low]);
            low = mid + 1;
        }
        return findMinElement(nums, low, high, min);
    }
}
