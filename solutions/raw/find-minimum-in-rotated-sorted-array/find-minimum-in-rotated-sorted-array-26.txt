class Solution {

    public int findMin(int[] nums) {
        // l - left,  r - right
        int l = 0, r = nums.length - 1;
        //mid
        int m;
        while (l < r) {
            //default mid
            m = l + (r - l) / 2;
            if (nums[m] > nums[r]) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return nums[l];
    }
}
