class Solution {

    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int lowest = 0;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] < nums[high]) {
                high = mid;
            } else if (nums[mid] > nums[high]) {
                low = mid + 1;
            } else {
                mid = high;
            }
        }
        return nums[high];
    }
}
