class Solution {

    public int findMin(int[] nums) {
        int count = 0;
        int num = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                num = nums[i + 1];
                count++;
            }
        }
        if (count == 0) {
            return nums[0];
        }
        return (count == 1) ? num : -1;
    }
}
