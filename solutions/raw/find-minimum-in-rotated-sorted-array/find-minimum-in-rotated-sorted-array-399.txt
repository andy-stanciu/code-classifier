class Solution {

    static int findCountInrotatedSortedArray(int[] nums) {
        int s = 0;
        int e = nums.length - 1;
        if (nums[s] < nums[e]) {
            return 0;
        }
        while (s < e) {
            int m = s + (e - s) / 2;
            if (nums[m] <= nums[e]) {
                e = m;
            } else {
                s = m + 1;
            }
        }
        return s;
    }

    public int findMin(int[] nums) {
        int pivot = findCountInrotatedSortedArray(nums);
        return nums[pivot];
    }
}
