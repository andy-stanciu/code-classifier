class Solution {

    public int findMin(int[] nums) {
        int n = nums.length;
        int s = 0;
        int e = n - 1;
        if (nums[0] <= nums[n - 1])
            return nums[0];
        while (s <= e) {
            int mid = s + (e - s) / 2;
            if (nums[mid] >= nums[0]) {
                s = mid + 1;
            } else {
                if (nums[mid - 1] > nums[mid])
                    return nums[mid];
                else
                    e = mid - 1;
            }
        }
        return -1;
    }
}
