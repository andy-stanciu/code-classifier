class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.length() == 0)
            return true;
        //we will create 2 pointers sPointer and tPointer
        int sPointer = 0, tPointer = 0;
        while (tPointer < t.length()) {
            if (t.charAt(tPointer) == s.charAt(sPointer)) {
                sPointer++;
                if (sPointer == s.length())
                    return true;
            }
            tPointer++;
        }
        return false;
    }
}
