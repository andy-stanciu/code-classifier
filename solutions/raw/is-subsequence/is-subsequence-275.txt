class Solution {

    public boolean isSubsequence(String s, String t) {
        int j = 0;
        int i = 0;
        char[] schar = s.toCharArray();
        char[] tchar = t.toCharArray();
        while (j < tchar.length && i < schar.length) {
            if (tchar[j] == schar[i]) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        if (i >= schar.length) {
            return true;
        }
        return false;
    }
}
