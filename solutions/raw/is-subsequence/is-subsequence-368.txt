class Solution {

    public boolean isSubsequence(String s, String t) {
        int index = -1;
        for (int i = 0; i < s.length(); i++) {
            int newIndex = t.indexOf(s.charAt(i), index + 1);
            if (newIndex == -1) {
                return false;
            }
            index = newIndex;
        }
        return true;
    }
}
