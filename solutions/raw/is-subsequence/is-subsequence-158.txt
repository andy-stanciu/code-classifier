class Solution {

    public boolean isSubsequence(String s, String t) {
        int l1 = s.length();
        int l2 = t.length();
        int min = Math.min(l1, l2);
        int[][] a = new int[l1 + 1][l2 + 1];
        for (int i = 1; i <= l1; i++) {
            for (int j = 1; j <= l2; j++) {
                char c1 = s.charAt(i - 1);
                char c2 = t.charAt(j - 1);
                if (c1 == c2)
                    a[i][j] = a[i - 1][j - 1] + 1;
                else
                    a[i][j] = Math.max(a[i][j - 1], a[i - 1][j]);
            }
        }
        int m = a[l1][l2];
        if (m == l1 && l1 >= 0)
            return true;
        else
            return false;
    }
}
