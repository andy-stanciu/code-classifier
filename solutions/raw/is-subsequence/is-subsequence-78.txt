class Solution {

    public boolean isSubsequence(String s, String t) {
        int idx = 0;
        if (idx >= s.length()) {
            return true;
        }
        for (char c : t.toCharArray()) {
            if (c == s.charAt(idx)) {
                idx++;
            }
            if (idx >= s.length()) {
                return true;
            }
        }
        return false;
    }
}
