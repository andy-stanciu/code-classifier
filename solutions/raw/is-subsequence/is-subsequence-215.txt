class Solution {

    public boolean isSubsequence(String s, String t) {
        // cases:
        // empty string -> true
        // same string -> true
        // letters can be plucked out of t to create s but not rearranged -> true
        // there is no way to remove letters from t to create s without rearranging -> false
        // there is no way to remove/rearrange letters from t to create s-> false
        // thinking: walk through t and tick off each letter along the way, if not all letters ticked off by the end, false
        char[] s_chars = s.toCharArray();
        char[] t_chars = t.toCharArray();
        int s_index = 0;
        for (int t_index = 0; (s_index < s.length()) && (t_index < t.length()); t_index++) {
            if (t_chars[t_index] == s_chars[s_index]) {
                s_index++;
            }
        }
        if (s_index == s.length()) {
            return true;
        } else {
            return false;
        }
    }
}
