class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.isEmpty()) {
            return true;
        }
        StringBuilder sbS = new StringBuilder(s);
        StringBuilder sbT = new StringBuilder(t);
        int countS = 0;
        int countT = 0;
        while (countT < sbT.length()) {
            if (sbS.charAt(countS) == sbT.charAt(countT)) {
                countS++;
            }
            if (countS >= sbS.length()) {
                return true;
            }
            countT++;
        }
        return false;
    }
}
