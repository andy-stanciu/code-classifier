class Solution {

    public boolean isSubsequence(String s, String t) {
        int len = s.length();
        if (len == 0)
            return true;
        int index = 0;
        for (int i = 0; index < len && i < t.length(); i++) {
            char c = s.charAt(index);
            if (c == t.charAt(i)) {
                index++;
            }
        }
        return index == len;
    }
}
