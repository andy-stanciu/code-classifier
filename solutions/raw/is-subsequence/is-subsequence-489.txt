class Solution {

    public boolean isSubsequence(String s, String t) {
        int count = 0;
        int p = 0;
        for (int i = 0; i < s.length(); i++) {
            for (int j = p; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    count++;
                    p = j + 1;
                    break;
                }
            }
        }
        return s.length() == count;
    }
}
