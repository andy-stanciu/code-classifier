class Solution {

    public boolean isSubsequence(String s, String t) {
        int count = 0;
        int n = 0;
        for (int i = 0; i < s.length(); i++) {
            boolean found = false;
            for (int j = n; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    n = j + 1;
                    count++;
                    found = true;
                    break;
                }
            }
            if (!found) {
                return false;
            }
        }
        return count == s.length();
    }
}
