class Solution {

    public boolean isSubsequence(String s, String t) {
        int i = 0;
        int j = 0;
        while (i < s.length()) {
            if (j == t.length())
                return false;
            char a = s.charAt(i);
            char b = t.charAt(j);
            if (a == b) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        return true;
    }
}
