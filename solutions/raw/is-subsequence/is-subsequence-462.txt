class Solution {

    public boolean isSubsequence(String S, String T) {
        int s = 0, t = 0, sl = S.length(), tl = T.length();
        if (sl < 1)
            return true;
        if (tl < 1)
            return false;
        while (s < sl && t < tl) {
            if (S.charAt(s) == T.charAt(t)) {
                s++;
            }
            t++;
        }
        if (s != sl)
            return false;
        return T.charAt(Math.max(t - 1, 0)) == S.charAt(Math.max(s - 1, 0));
    }
}
