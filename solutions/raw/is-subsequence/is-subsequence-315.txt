class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.isEmpty()) {
            return true;
        }
        StringBuilder sb = new StringBuilder(s);
        for (int i = 0; i < t.length(); i++) {
            if (sb.length() == 0) {
                break;
            }
            if (t.charAt(i) == sb.charAt(0)) {
                sb.delete(0, 1);
            }
        }
        return sb.length() == 0 ? true : false;
    }
}
