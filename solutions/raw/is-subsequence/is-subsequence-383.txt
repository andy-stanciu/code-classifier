class Solution {

    public boolean isSubsequence(String s, String t) {
        int i, j, k = 0, l1, l2;
        StringBuilder sb = new StringBuilder(t);
        l1 = sb.length();
        StringBuilder sb2 = new StringBuilder(s);
        l2 = sb2.length();
        if (l2 == 0) {
            return true;
        }
        j = 0;
        for (i = 0; i < l1; i++) {
            if (sb2.charAt(j) == sb.charAt(i)) {
                j++;
            }
            if (j == l2) {
                return true;
            }
        }
        return false;
    }
}
