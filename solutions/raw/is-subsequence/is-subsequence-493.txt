class Solution {

    int[][] memo;

    public boolean isSubsequence(String s, String t) {
        int n = s.length();
        int m = t.length();
        this.memo = new int[n + 1][m + 1];
        for (int i = 0; i < n + 1; i++) {
            for (int j = 0; j < m + 1; j++) {
                if (i == 0 || j == 0) {
                    memo[i][j] = 0;
                } else if (s.charAt(i - 1) == t.charAt(j - 1)) {
                    memo[i][j] = 1 + memo[i - 1][j - 1];
                } else {
                    memo[i][j] = memo[i][j - 1];
                }
            }
        }
        return s.length() == memo[n][m];
    }
}
