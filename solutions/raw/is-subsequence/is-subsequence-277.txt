class Solution {

    public boolean isSubsequence(String s, String t) {
        int p = 0, q = 0;
        // If s is empty, it\'s always a subsequence of t
        if (s.length() == 0) {
            return true;
        }
        // Iterate through string \'t\'
        while (p < t.length()) {
            // If characters match, move the pointer in \'s\'
            if (s.charAt(q) == t.charAt(p)) {
                q++;
            }
            // If we\'ve matched all characters of \'s\', return true
            if (q == s.length()) {
                return true;
            }
            // Move the pointer in \'t\'
            p++;
        }
        // If not all characters of \'s\' are matched, return false
        return false;
    }
}
