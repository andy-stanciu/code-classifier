class Solution {

    public boolean isSubsequence(String s, String t) {
        char[] sc = s.toCharArray();
        char[] tc = t.toCharArray();
        int p1 = 0;
        for (int p2 = 0; p2 < tc.length; p2++) {
            if (sc.length > 0 && sc[p1] == tc[p2]) {
                ++p1;
                if (p1 >= sc.length) {
                    return true;
                }
            }
        }
        return p1 >= sc.length;
    }
}
