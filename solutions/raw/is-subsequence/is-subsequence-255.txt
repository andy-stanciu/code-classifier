class Solution {

    public boolean isSubsequence(String s, String t) {
        int c = 0, i = 0, j = 0;
        while (i < s.length() && j < t.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                c++;
                i++;
            }
            j++;
        }
        if (c == s.length())
            return true;
        else
            return false;
    }
}
