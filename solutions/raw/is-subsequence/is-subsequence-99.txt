class Solution {

    public boolean isSubsequence(String s, String t) {
        int lastIndx = 0;
        int i = 0;
        while (i < s.length()) {
            char toFind = s.charAt(i);
            lastIndx = t.indexOf(toFind, lastIndx);
            if (lastIndx == -1)
                return false;
            i++;
            lastIndx++;
        }
        return true;
    }
}
