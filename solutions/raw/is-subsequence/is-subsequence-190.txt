class Solution {

    public boolean isSubsequence(String s, String t) {
        int previousIndex = -1;
        for (int i = 0; i < s.length(); i++) {
            int currentIndex = t.indexOf(s.charAt(i), previousIndex + 1);
            if (currentIndex == -1 || currentIndex <= previousIndex) {
                return false;
            }
            previousIndex = currentIndex;
        }
        return true;
    }
}
