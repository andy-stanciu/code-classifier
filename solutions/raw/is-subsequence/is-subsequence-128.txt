class Solution {

    public boolean isSubsequence(String s, String t) {
        int counter1 = 0;
        int counter2 = 0;
        int sLength = s.length();
        int tLength = t.length();
        while (counter1 < sLength && counter2 < tLength) {
            if (s.charAt(counter1) == t.charAt(counter2)) {
                counter1++;
            }
            counter2++;
        }
        return counter1 == sLength;
    }
}
