class Solution {

    public boolean isSubsequence(String s, String t) {
        String[] sarray = s.split("");
        String[] tarray = t.split("");
        boolean isSub = false;
        for (int i = 0; i < sarray.length; i++) {
            isSub = false;
            for (int n = 0; n < tarray.length; n++) {
                if (!tarray[n].equals(sarray[i])) {
                    tarray[n] = "0";
                }
                if (tarray[n].equals(sarray[i])) {
                    isSub = true;
                    tarray[n] = "0";
                    break;
                }
            }
            if (isSub == false) {
                break;
            }
        }
        if (s.equals("")) {
            isSub = true;
        }
        return isSub;
    }
}
