class Solution {

    public boolean isSubsequence(String s, String t) {
        int n1 = s.length();
        int n2 = t.length();
        for (int i = 0; i < n1; i++) {
            char chi = s.charAt(i);
            int idx = t.indexOf(chi);
            if (idx == -1) {
                return false;
            }
            t = t.substring(idx + 1, n2);
            n2 = t.length();
        }
        return true;
    }
}
