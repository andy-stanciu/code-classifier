class Solution {

    public boolean isSubsequence(String s, String t) {
        int i = 0, j = 0;
        if (i == s.length())
            return true;
        while (j < t.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                j++;
                if (i == s.length())
                    return true;
            } else
                j++;
        }
        return false;
    }
}
