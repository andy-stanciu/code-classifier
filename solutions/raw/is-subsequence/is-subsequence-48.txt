class Solution {

    public boolean isSubsequence(String s, String t) {
        Queue<Character> q = new LinkedList<>();
        for (char c : s.toCharArray()) q.add(c);
        if (q.isEmpty())
            return true;
        for (char c : t.toCharArray()) {
            if (!q.isEmpty() && c == q.peek())
                q.poll();
        }
        return q.isEmpty();
    }
}
