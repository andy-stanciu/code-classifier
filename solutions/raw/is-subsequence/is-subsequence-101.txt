class Solution {

    public boolean isSubsequence(String s, String t) {
        String subsequence = "";
        int j = 0;
        for (int i = 0; i < t.length(); i++) {
            if (j < s.length() && (t.charAt(i) == s.charAt(j))) {
                subsequence += t.charAt(i);
                j++;
            }
        }
        return subsequence.equals(s);
    }
}
