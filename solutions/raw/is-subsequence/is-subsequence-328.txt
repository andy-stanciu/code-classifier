class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.length() < 1) {
            return true;
        }
        if (t.length() < s.length()) {
            return false;
        }
        int sidx = 0;
        int tidx = 0;
        while (sidx < s.length() && tidx < t.length()) {
            if (s.charAt(sidx) == t.charAt(tidx)) {
                sidx++;
            }
            tidx++;
        }
        return sidx == s.length();
    }
}
