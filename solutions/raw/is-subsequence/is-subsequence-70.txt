class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.length() > t.length()) {
            return false;
        }
        // for A string
        int j = 0;
        for (int i = 0; i < t.length(); i++) {
            // for B string
            if (j < s.length() && s.charAt(j) == t.charAt(i)) {
                j++;
            }
        }
        return (j == s.length());
    }
}
