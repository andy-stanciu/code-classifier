class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.length() == 0)
            return true;
        int i = 0;
        int j = 0;
        int count = 0;
        while (j < t.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                count++;
            }
            if (i == s.length())
                break;
            j++;
        }
        if (count == s.length()) {
            return true;
        }
        return false;
    }
}
