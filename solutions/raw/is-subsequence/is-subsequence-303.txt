class Solution {

    public boolean isSubsequence(String s, String t) {
        int m = s.length();
        int n = t.length();
        int index = 0;
        for (char c : t.toCharArray()) {
            if (index == m)
                break;
            if (c == s.charAt(index))
                index++;
        }
        return index == m;
    }
}
