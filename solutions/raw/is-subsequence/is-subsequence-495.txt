class Solution {

    public boolean isSubsequence(String s, String t) {
        int ptr = 0;
        if (s.equals(""))
            return true;
        for (int i = 0; i < t.length(); i++) {
            if (s.charAt(ptr) == t.charAt(i)) {
                ptr++;
            }
            if (ptr == s.length()) {
                return true;
            }
        }
        return false;
    }
}
