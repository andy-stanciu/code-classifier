class Solution {

    public boolean isSubsequence(String s, String t) {
        // Use an index to keep track of the position in `s`
        int sIndex = 0;
        // Iterate over each character in `t`
        for (int i = 0; i < t.length(); i++) {
            // Check if current character in `t` matches the current character in `s`
            if (sIndex < s.length() && s.charAt(sIndex) == t.charAt(i)) {
                // Move to the next character in `s`
                sIndex++;
            }
        }
        // If `sIndex` reached the end of `s`, all characters were matched in order
        return sIndex == s.length();
    }
}
