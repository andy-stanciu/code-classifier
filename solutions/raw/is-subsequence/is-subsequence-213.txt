class Solution {

    public boolean isSubsequence(String s, String t) {
        int idt = -1;
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            for (int j = idt + 1; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    idt = j;
                    count++;
                    break;
                }
            }
        }
        return count == s.length();
    }
}
