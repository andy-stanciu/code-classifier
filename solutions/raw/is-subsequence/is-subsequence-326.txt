class Solution {

    public boolean isSubsequence(String s, String t) {
        int sLen = s.length();
        int tLen = t.length();
        int i = 0;
        int j = 0;
        while (i < sLen && j < tLen) {
            char sChar = s.charAt(i);
            char tChar = t.charAt(j);
            if (sChar == tChar) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        System.out.println(i);
        return i == sLen;
    }
}
