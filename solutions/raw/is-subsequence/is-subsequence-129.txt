class Solution {

    public boolean isSubsequence(String s, String t) {
        int index = 0;
        for (char c : s.toCharArray()) {
            int ind = t.indexOf(c);
            if (ind < 0)
                return false;
            else {
                t = t.substring(ind + 1);
            }
        }
        return true;
    }
}
