class Solution {

    public boolean isSubsequence(String s, String t) {
        char[] shortArr = s.toCharArray();
        char[] longArr = t.toCharArray();
        int n = s.length();
        int m = t.length();
        int pointer = 0;
        int k = 0;
        while (pointer < n && k < m) {
            if (shortArr[pointer] == longArr[k]) {
                pointer++;
            }
            k++;
        }
        return pointer == s.length();
    }
}
