class Solution {

    public boolean isSubsequence(String s, String t) {
        int lastIndex = 0;
        for (char c : s.toCharArray()) {
            int i = getIndex(t, String.valueOf(c), lastIndex);
            if (i >= lastIndex) {
                lastIndex = i + 1;
            } else {
                return false;
            }
        }
        return true;
    }

    public int getIndex(String t, String letter, int start) {
        return t.indexOf(letter, start);
    }
}
