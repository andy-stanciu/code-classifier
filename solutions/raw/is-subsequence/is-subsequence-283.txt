class Solution {

    public boolean isSubsequence(String s, String t) {
        int i = 0;
        int j = 0;
        int sLen = s.length();
        int tLen = t.length();
        while (j < tLen && i < sLen) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
            }
            j++;
        }
        if (i == sLen)
            return true;
        return false;
    }
}
