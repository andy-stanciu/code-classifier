class Solution {

    public boolean isSubsequence(String s, String t) {
        char[] c1 = s.toCharArray();
        char[] c2 = t.toCharArray();
        int i = 0, j = 0, l1 = c1.length, l2 = c2.length;
        while (i < l1 && j < l2) {
            if (c1[i] == c2[j]) {
                i++;
            }
            j++;
        }
        return i >= l1;
    }
}
