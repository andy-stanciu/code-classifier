class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.equals("")) {
            return true;
        }
        int s_pointer = 0;
        for (int i = 0; i < t.length(); i++) {
            if (s.charAt(s_pointer) == t.charAt(i)) {
                s_pointer++;
                if (s_pointer == s.length()) {
                    return true;
                }
            }
        }
        return false;
    }
}
