class Solution {

    public boolean isSubsequence(String s, String t) {
        int tind = 0, sind = 0;
        while (sind < s.length() && tind < t.length()) {
            if (s.charAt(sind) == t.charAt(tind)) {
                sind++;
            }
            tind++;
        }
        return sind == s.length();
    }
}
