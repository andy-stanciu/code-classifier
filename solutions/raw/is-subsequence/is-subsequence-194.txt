class Solution {

    public boolean isSubsequence(String s, String t) {
        int index = 0;
        int count = -1;
        for (int i = 0; i < s.length(); i++) {
            boolean found = false;
            for (int j = count + 1; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    index++;
                    count = j;
                    found = true;
                    break;
                }
            }
            if (!found) {
                return false;
            }
        }
        return index == s.length();
    }
}
