class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        char[] sa = s.toCharArray();
        char[] ta = t.toCharArray();
        int counter = 0;
        for (int i = 0; i < ta.length; i++) {
            if (ta[i] == sa[counter]) {
                counter++;
            }
            if (counter == sa.length) {
                return true;
            }
        }
        return false;
    }
}
