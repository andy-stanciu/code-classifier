class Solution {

    public boolean isSubsequence(String s, String t) {
        int indS = 0, indT = 0;
        while (indS < s.length() && indT < t.length()) {
            while (indS < s.length() && indT < t.length() && t.charAt(indT) != s.charAt(indS)) ++indT;
            if (indS < s.length() && indT < t.length()) {
                ++indS;
                ++indT;
            }
        }
        if (indS >= s.length())
            return true;
        return false;
    }
}
