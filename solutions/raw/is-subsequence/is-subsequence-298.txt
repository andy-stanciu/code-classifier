class Solution {

    public boolean isSubsequence(String s, String t) {
        int index = 0;
        char[] arr = t.toCharArray();
        int counter = 0;
        for (char c : s.toCharArray()) {
            for (int i = index; i < arr.length; i++) {
                if (c == arr[i]) {
                    counter++;
                    index = i + 1;
                    break;
                }
            }
        }
        return counter == s.length();
    }
}
