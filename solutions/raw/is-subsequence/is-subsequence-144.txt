class Solution {

    public boolean isSubsequence(String s, String t) {
        Stack<Character> stack = new Stack<Character>();
        for (int i = s.length() - 1; i >= 0; i--) {
            stack.push(s.charAt(i));
        }
        for (int i = 0; i < t.length(); i++) {
            if (stack.size() != 0 && stack.peek() == t.charAt(i))
                stack.pop();
        }
        if (stack.size() == 0)
            return true;
        return false;
    }
}
