class Solution {

    public boolean isSubsequence(String subsequence, String targetSequence) {
        char[] subsequenceArr = subsequence.toCharArray();
        char[] targetSequenceArr = targetSequence.toCharArray();
        int j = 0;
        for (int i = 0; i < targetSequenceArr.length; i++) {
            if (j == subsequenceArr.length)
                break;
            if (targetSequenceArr[i] == subsequenceArr[j]) {
                j++;
            }
        }
        return j == subsequenceArr.length;
    }
}
