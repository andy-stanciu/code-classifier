class Solution {

    public boolean isSubsequence(String s, String t) {
        int cnt = 0;
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            int m = t.indexOf(ch, cnt);
            if (m == -1) {
                return false;
            } else if (m == t.length() - 1 && i != s.length() - 1) {
                return false;
            } else {
                cnt = m + 1;
            }
        }
        return true;
    }
}
