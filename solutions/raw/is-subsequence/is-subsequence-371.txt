class Solution {

    public boolean isSubsequence(String s, String t) {
        int j = 0;
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            for (j = j; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    count += 1;
                    j++;
                    break;
                }
            }
        }
        return (count == s.length());
    }
}
