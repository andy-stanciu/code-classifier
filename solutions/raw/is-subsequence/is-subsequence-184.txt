class Solution {

    public boolean isSubsequence(String s, String t) {
        //two pointers
        int ps = 0;
        int pt = 0;
        //lengths
        int n = s.length();
        int m = t.length();
        //the subsequence can\'t be longer than the main sequence
        if (m < n) {
            return false;
        }
        //till we reach the end of any of the sequences
        while (ps < n && pt < m) {
            char cs = s.charAt(ps);
            char ct = t.charAt(pt);
            //found character match
            if (cs == ct) {
                ps++;
            }
            pt++;
        }
        if (ps >= n) {
            //found all the subsequence
            return true;
        } else {
            return false;
        }
    }
}
