class Solution {

    public boolean isSubsequence(String s, String t) {
        int it = 0;
        int is = 0;
        while (is < s.length() && it < t.length()) {
            if (s.charAt(is) == t.charAt(it)) {
                is++;
            }
            it++;
        }
        return is == s.length();
    }
}
