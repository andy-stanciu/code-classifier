class Solution {

    public boolean isSubsequence(String s, String t) {
        return check(s, t);
    }

    public boolean check(String s, String t) {
        int curIndex = 0;
        for (int i = 0; i < s.length(); i++) {
            curIndex = t.indexOf(s.charAt(i), curIndex) + 1;
            if (curIndex == 0)
                return false;
        }
        return true;
    }
}
