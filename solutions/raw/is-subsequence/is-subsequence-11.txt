class Solution {

    public boolean isSubsequence(String s, String t) {
        int sLength = s.length() - 1, tLength = t.length() - 1;
        while (tLength > -1) {
            if (sLength > -1 && s.charAt(sLength) == t.charAt(tLength)) {
                sLength--;
            }
            tLength--;
        }
        if (sLength <= -1) {
            return true;
        }
        return false;
    }
}
