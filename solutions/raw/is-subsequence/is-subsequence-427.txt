class Solution {

    public boolean isSubsequence(String s, String t) {
        int ns = s.length();
        int nt = t.length();
        int ps = 0;
        int pt = 0;
        while ((ps < ns) && (pt < nt)) {
            if (s.charAt(ps) == t.charAt(pt)) {
                ps++;
            }
            pt++;
        }
        if (ps == ns) {
            return true;
        } else {
            return false;
        }
    }
}
