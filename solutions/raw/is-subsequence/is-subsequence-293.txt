class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.length() > t.length())
            return false;
        int start = 0;
        for (int i = 0; i < t.length() && start < s.length(); i++) {
            if (s.charAt(start) == t.charAt(i))
                start++;
        }
        return start == s.length();
    }
}
