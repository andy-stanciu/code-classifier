class Solution {

    public boolean isSubsequence(String s, String t) {
        int sLen = s.length();
        int tLen = t.length();
        int j = 0;
        if (sLen == 0)
            return true;
        for (int i = 0; i < tLen; i++) {
            if (t.charAt(i) == s.charAt(j)) {
                j++;
            }
            if (j == sLen) {
                return true;
            }
        }
        return false;
    }
}
