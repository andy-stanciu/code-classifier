class Solution {

    public boolean isSubsequence(String s, String t) {
        if (s.isEmpty()) {
            return true;
        }
        if (t.isEmpty()) {
            return false;
        }
        int shortStrIndex = 0;
        for (int i = 0; i < t.length(); i++) {
            if (t.charAt(i) == s.charAt(shortStrIndex)) {
                shortStrIndex++;
            }
            if (shortStrIndex == s.length()) {
                return true;
            }
        }
        return false;
    }
}
