class Solution {

    public boolean isSubsequence(String S, String T) {
        int sr = 0, tr = 0, sLen = S.length(), tLen = T.length();
        while (sr < sLen && tr < tLen) {
            if (S.charAt(sr) == T.charAt(tr)) {
                sr++;
            }
            tr++;
        }
        return sr == sLen;
    }
}
