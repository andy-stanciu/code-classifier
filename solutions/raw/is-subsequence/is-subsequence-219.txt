class Solution {

    public boolean isSubsequence(String s, String t) {
        String sub = t;
        int k = -1;
        if (s.length() > t.length())
            return false;
        for (int i = 0; i < s.length(); i++) {
            k = sub.indexOf(s.charAt(i));
            if (k == -1)
                return false;
            else {
                sub = sub.substring(k + 1);
            }
        }
        return true;
    }
}
