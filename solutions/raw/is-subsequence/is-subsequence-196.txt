class Solution {

    public boolean isSubsequence(String s, String t) {
        int i = 0;
        int j = 0;
        int isize = s.length(), jsize = t.length();
        while (i < isize && j < jsize) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        return i == isize;
    }
}
