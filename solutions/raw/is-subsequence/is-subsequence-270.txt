class Solution {

    public boolean isSubsequence(String s, String t) {
        int max = s.length();
        int i = 0;
        if (max == 0) {
            return true;
        }
        for (char ch : t.toCharArray()) {
            if (ch == s.charAt(i)) {
                i++;
            }
            if (i == max) {
                return true;
            }
        }
        return i == max;
    }
}
