class Solution {

    public boolean isSubsequence(String s, String t) {
        int i = 0;
        for (char c : t.toCharArray()) {
            if (i < s.length() && s.charAt(i) == c) {
                i++;
                continue;
            }
        }
        if (i == s.length())
            return true;
        return false;
    }
}
