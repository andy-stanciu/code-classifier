class Solution {

    public boolean isSubsequence(String s, String t) {
        int l = s.length();
        int start = 0, index = 0;
        while (start < l) {
            char ch = s.charAt(start);
            boolean found = false;
            for (int i = index; i < t.length(); i++) {
                if (ch == t.charAt(i)) {
                    index = i + 1;
                    found = true;
                    break;
                }
            }
            if (!found) {
                return false;
            }
            start++;
        }
        return true;
    }
}
