class Solution {

    public boolean isSubsequence(String s, String t) {
        int sp = 0;
        int tp = 0;
        if (t.length() < s.length()) {
            return false;
        }
        while (sp < s.length() && tp < t.length()) {
            if (s.charAt(sp) == t.charAt(tp)) {
                sp++;
            }
            tp++;
        }
        if (sp == s.length()) {
            return true;
        } else {
            return false;
        }
    }
}
