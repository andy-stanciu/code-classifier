class Solution {

    public boolean isSubsequence(String s, String t) {
        int lenS = s.length();
        int lenT = t.length();
        if (lenS > lenT)
            return false;
        int curindex = 0;
        int count = 0;
        for (int i = 0; i < lenS; i++) {
            char current = s.charAt(i);
            for (int j = curindex; j < lenT; j++) {
                if (current == t.charAt(j)) {
                    count++;
                    curindex = j + 1;
                    break;
                }
            }
        }
        if (count == lenS) {
            return true;
        }
        return false;
    }
}
