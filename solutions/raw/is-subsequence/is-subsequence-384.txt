class Solution {

    public boolean isSubsequence(String s, String t) {
        char[] c = s.toCharArray();
        char[] d = t.toCharArray();
        int i = 0, j = 0;
        while (i < c.length && j < d.length) {
            if (c[i] == d[j])
                i++;
            j++;
        }
        if (i == c.length)
            return true;
        return false;
    }
}
