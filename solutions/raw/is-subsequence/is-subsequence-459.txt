class Solution {

    public boolean isSubsequence(String s, String t) {
        int count = 0;
        if (null == s || s.trim().isEmpty())
            return true;
        for (char c : t.toCharArray()) {
            if (c == s.charAt(count)) {
                count++;
                if (count >= s.length())
                    return true;
            }
        }
        return false;
    }
}
