class Solution {

    public boolean isSubsequence(String s, String t) {
        return isSubsequence(s, t, 0, 0);
    }

    public boolean isSubsequence(String s, String t, int sCount, int tCount) {
        if (sCount == s.length())
            return true;
        for (int i = tCount; i < t.length(); i++) {
            if (t.charAt(i) == s.charAt(sCount)) {
                return isSubsequence(s, t, sCount + 1, i + 1);
            }
        }
        return false;
    }
}
