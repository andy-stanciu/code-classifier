class Solution {

    public boolean isSubsequence(String s, String t) {
        int x = 0;
        int j = 0;
        for (int i = 0; i < s.length(); i++) {
            for (j = x; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    x = j + 1;
                    break;
                }
            }
            if (j == t.length())
                return false;
        }
        return true;
    }
}
