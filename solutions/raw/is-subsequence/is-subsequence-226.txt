class Solution {

    public boolean isSubsequence(String s, String t) {
        char[] s1 = s.toCharArray();
        char[] t1 = t.toCharArray();
        int x = 0, y = 0;
        for (x = 0; x < t.length() && y < s.length(); x++) {
            if (t1[x] == s1[y])
                y++;
        }
        if (y == s.length())
            return true;
        else
            return false;
    }
}
