class Solution {

    public String frequencySort(String s) {
        HashMap<Character, Integer> map = new HashMap<>();
        //here comparator order the String
        PriorityQueue<String> pq = new PriorityQueue<>((a, b) -> b.length() - a.length());
        //add the character and occurrence in the map
        for (char i : s.toCharArray()) {
            int count = map.getOrDefault(i, 0);
            map.put(i, ++count);
        }
        //add the String(char occurrence time) in PriorityQueue
        for (Character i : map.keySet()) {
            StringBuilder br = new StringBuilder();
            int len = map.get(i);
            for (int j = 0; j < len; j++) {
                br.append(i + "");
            }
            pq.add(br.toString());
        }
        //finally add the String to StringBuilder and remove that in PriorityQueue
        StringBuilder answer = new StringBuilder();
        while (!pq.isEmpty()) {
            answer.append(pq.remove());
        }
        return answer.toString();
    }
}
