class Solution {

    public String frequencySort(String s) {
        Map<Character, Integer> map = new HashMap();
        for (char c : s.toCharArray()) map.put(c, map.getOrDefault(c, 0) + 1);
        return map.entrySet().stream().sorted((i1, i2) -> i2.getValue().compareTo(i1.getValue())).map(i -> String.valueOf(i.getKey()).repeat(i.getValue())).collect(Collectors.joining());
    }
}
