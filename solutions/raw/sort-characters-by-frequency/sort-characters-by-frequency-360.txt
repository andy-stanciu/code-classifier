class Solution {

    public String frequencySort(String s) {
        int[][] map = new int[127][2];
        for (char c : s.toCharArray()) {
            map[c][0] = c;
            map[c][1]++;
        }
        Arrays.sort(map, (a, b) -> b[1] - a[1]);
        StringBuilder sb = new StringBuilder();
        int i = 0;
        while (i < 127 && map[i][1] != 0) {
            sb.append(String.valueOf((char) (map[i][0])).repeat(map[i][1]));
            i++;
        }
        return sb.toString();
    }
}
