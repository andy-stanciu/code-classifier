class Solution {

    public String frequencySort(String s) {
        // store the frequency of each character in hashmap
        Map<Character, Integer> hm = new HashMap<>();
        for (char ch : s.toCharArray()) {
            hm.put(ch, hm.getOrDefault(ch, 0) + 1);
        }
        // against ascii value of each char, populate its frequency
        int[][] asciiCount = new int[hm.size()][2];
        int index = 0;
        for (char ch : hm.keySet()) {
            asciiCount[index][0] = (int) ch;
            asciiCount[index++][1] = hm.get(ch);
        }
        // Reverse Sort the pair array of ascii and count on the basis of count
        Arrays.sort(asciiCount, (a, b) -> b[1] - a[1]);
        // append each character count times to stringbuilder
        StringBuilder sb = new StringBuilder("");
        for (int i = 0; i < asciiCount.length; i++) {
            char asciiConverted = (char) asciiCount[i][0];
            for (int j = 0; j < asciiCount[i][1]; j++) {
                sb.append(asciiConverted);
            }
        }
        return sb.toString();
    }
}
