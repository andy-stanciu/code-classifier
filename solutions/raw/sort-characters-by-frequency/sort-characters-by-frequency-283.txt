class Solution {

    public String frequencySort(String s) {
        if (s == null || s.isEmpty())
            return s;
        HashMap<Character, Integer> freqMap = new HashMap();
        for (char c : s.toCharArray()) {
            freqMap.put(c, freqMap.getOrDefault(c, 0) + 1);
        }
        TreeMap<Integer, TreeSet<Character>> sortedMap = new TreeMap<>(Collections.reverseOrder());
        for (char key : freqMap.keySet()) {
            int freq = freqMap.get(key);
            sortedMap.putIfAbsent(freq, new TreeSet());
            sortedMap.get(freq).add(key);
        }
        StringBuilder result = new StringBuilder();
        for (int freq : sortedMap.keySet()) {
            for (char c : sortedMap.get(freq)) {
                for (int i = 0; i < freq; i++) {
                    result.append(c);
                }
            }
        }
        return result.toString();
    }
}
