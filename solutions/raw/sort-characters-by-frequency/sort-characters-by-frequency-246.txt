class Solution {

    public String frequencySort(String s) {
        int[] freq = new int[255];
        // here i am finding freq of characters
        for (int i = 0; i < s.length(); i++) {
            freq[s.charAt(i)]++;
        }
        StringBuilder sp = new StringBuilder();
        // TC : O(255 * 255 + k )   SC : O(N)
        for (int i = 0; i < 255; i++) {
            // taking two  pointers so that i can find max value at every iteration and arrange it in lexiographical order also
            int idx = -1;
            int val = -1;
            for (int j = 0; j < 255; j++) {
                if (freq[j] > val) {
                    // just updating the val and index index bcz i have to add alphabets
                    val = freq[j];
                    idx = j;
                }
            }
            // just adding elements till the number of occurences
            while (val > 0) {
                if (val != 0)
                    sp.append((char) idx);
                val = val - 1;
            }
            freq[idx] = 0;
        }
        return sp.toString();
    }
}
