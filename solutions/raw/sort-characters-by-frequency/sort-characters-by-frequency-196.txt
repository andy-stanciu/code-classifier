class Solution {

    public String frequencySort(String s) {
        Map<Character, Integer> hashmap = new HashMap<>();
        for (char ch : s.toCharArray()) {
            hashmap.put(ch, hashmap.getOrDefault(ch, 0) + 1);
        }
        PriorityQueue<Map.Entry<Character, Integer>> pqueue = new // comparing based on second element freq . to store in a decreasing order
        PriorityQueue<>((a, b) -> b.getValue() - a.getValue());
        pqueue.addAll(hashmap.entrySet());
        StringBuilder result = new StringBuilder();
        while (!pqueue.isEmpty()) {
            Map.Entry<Character, Integer> entry = pqueue.poll();
            result.append(String.valueOf(entry.getKey()).repeat(entry.getValue()));
        }
        return result.toString();
    }
}
