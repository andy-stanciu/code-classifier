class Solution {

    public String frequencySort(String s) {
        // Step 1: Count frequencies of each character using HashMap
        HashMap<Character, Integer> frequencyMap = new HashMap<>();
        for (char c : s.toCharArray()) {
            frequencyMap.put(c, frequencyMap.getOrDefault(c, 0) + 1);
        }
        // Step 2: Sort characters by their frequencies in descending order
        List<Map.Entry<Character, Integer>> entryList = new ArrayList<>(frequencyMap.entrySet());
        // Sort by frequency (descending)
        entryList.sort((a, b) -> b.getValue().compareTo(a.getValue()));
        // Step 3: Construct the result string based on sorted frequencies
        StringBuilder result = new StringBuilder();
        for (Map.Entry<Character, Integer> entry : entryList) {
            char character = entry.getKey();
            int frequency = entry.getValue();
            for (int i = 0; i < frequency; i++) {
                result.append(character);
            }
        }
        return result.toString();
    }
}
