class Solution {

    public String frequencySort(String s) {
        Map<Character, String> frequencyMap = new HashMap<>();
        for (char ch : s.toCharArray()) {
            String str = String.valueOf(ch);
            frequencyMap.put(ch, frequencyMap.getOrDefault(ch, "") + str);
        }
        return // here we just need to do reverse sorting
        frequencyMap.values().stream().sorted((e1, e2) -> Integer.compare(e2.length(), e1.length())).collect(Collectors.joining());
    }
}
