class Solution {

    public String frequencySort(String s) {
        TreeMap<Character, Integer> fmap = new TreeMap<>();
        for (int i = 0; i < s.length(); i++) {
            fmap.put(s.charAt(i), fmap.getOrDefault(s.charAt(i), 0) + 1);
        }
        StringBuilder st = new StringBuilder();
        List<Map.Entry<Character, Integer>> entrymap = new ArrayList<>(fmap.entrySet());
        entrymap.sort((a, b) -> b.getValue() - a.getValue());
        for (Map.Entry<Character, Integer> map : entrymap) {
            Character ch = map.getKey();
            int n = map.getValue();
            for (int i = 0; i < n; i++) {
                st.append(ch);
            }
        }
        return st.toString();
    }
}
