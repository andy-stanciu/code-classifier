class Solution {

    public String frequencySort(String s) {
        return s.chars().mapToObj(ch -> (char) ch).collect(Collectors.groupingBy(Function.identity(), HashMap::new, Collectors.counting())).entrySet().stream().sorted(Map.Entry.<Character, Long>comparingByValue().reversed()).map(entryset -> Character.toString(entryset.getKey()).repeat(entryset.getValue().intValue())).collect(Collectors.joining(""));
    }
}
