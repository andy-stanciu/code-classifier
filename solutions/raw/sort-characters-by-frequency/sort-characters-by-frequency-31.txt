class Solution {

    class Temp {

        public char letter;

        public int count;

        public Temp(char letter, int count) {
            this.letter = letter;
            this.count = count;
        }
    }

    public String frequencySort(String s) {
        HashMap<Character, Integer> map = new HashMap<>();
        StringBuilder sb = new StringBuilder();
        for (char c : s.toCharArray()) map.put(c, map.getOrDefault(c, 0) + 1);
        Temp[] letters = new Temp[map.size()];
        int i = 0;
        for (Map.Entry<Character, Integer> entry : map.entrySet()) letters[i++] = new Temp(entry.getKey(), entry.getValue());
        Arrays.sort(letters, (a, b) -> b.count - a.count);
        for (Temp t : letters) {
            for (int j = 0; j < t.count; j++) sb.append(t.letter);
        }
        return sb.toString();
    }
}
