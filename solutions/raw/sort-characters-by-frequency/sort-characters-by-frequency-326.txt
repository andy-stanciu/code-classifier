class Solution {

    public String frequencySort(String s) {
        // HAsh map for frequency count
        HashMap<Character, Integer> x = new HashMap<>();
        // string builder for adding sorted arraylist on it
        StringBuilder sb = new StringBuilder();
        //add to hash map
        for (char c : s.toCharArray()) {
            x.put(c, x.getOrDefault(c, 0) + 1);
        }
        // array list
        ArrayList<Character> l = new ArrayList<>(x.keySet());
        // sort this it\'s also new for me google it
        Collections.sort(l, (a, b) -> x.get(b) - x.get(a));
        // add list to sb
        for (Character ch : l) {
            for (int i = 0; i < x.get(ch); i++) {
                sb.append(ch);
            }
        }
        return sb.toString();
    }
}
