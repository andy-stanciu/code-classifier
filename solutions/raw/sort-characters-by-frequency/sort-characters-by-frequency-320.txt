class Solution {

    public String frequencySort(String s) {
        Map<Character, Integer> hmap = new HashMap<>();
        for (int i = 0; i < s.length(); i++) hmap.put(s.charAt(i), hmap.getOrDefault(s.charAt(i), 0) + 1);
        List<Map.Entry<Character, Integer>> entryList = new ArrayList<>(hmap.entrySet());
        entryList.sort((a, b) -> b.getValue().compareTo(a.getValue()));
        StringBuilder sb = new StringBuilder();
        for (Map.Entry<Character, Integer> entry : entryList) {
            for (int i = 0; i < entry.getValue(); i++) sb.append(entry.getKey());
        }
        return sb.toString();
    }
}
