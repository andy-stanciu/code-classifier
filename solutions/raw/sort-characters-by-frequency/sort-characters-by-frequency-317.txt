class Solution {

    public String frequencySort(String s) {
        Map<Character, Integer> frequency = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            frequency.put(s.charAt(i), frequency.getOrDefault(s.charAt(i), 0) + 1);
        }
        List<Character> chars = new ArrayList<>(frequency.keySet());
        chars.sort((a, b) -> frequency.get(b) - frequency.get(a));
        StringBuilder sb = new StringBuilder();
        for (Character c : chars) {
            sb.append(String.valueOf(c).repeat(Math.max(0, frequency.get(c))));
        }
        return sb.toString();
    }
}
