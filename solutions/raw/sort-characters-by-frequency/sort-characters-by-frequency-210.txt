class Solution {

    public String frequencySort(String s) {
        int[] freq = new int[150];
        for (int i = 0; i < s.length(); i++) {
            freq[s.charAt(i)]++;
        }
        StringBuilder str = new StringBuilder();
        int max = -1;
        int index = -1;
        while (true) {
            for (int i = 0; i < 150; i++) {
                if (max <= freq[i]) {
                    max = freq[i];
                    index = i;
                }
            }
            if (max == 0)
                break;
            while (max > 0) {
                str.append((char) index);
                max--;
            }
            max = -1;
            freq[index] = 0;
            index = -1;
        }
        return str.toString();
    }
}
