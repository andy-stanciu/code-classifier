class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        return jumpFrom(0, nums);
    }

    public static boolean jumpFrom(int index, int[] nums) {
        if (index >= nums.length - 1)
            return true;
        for (int i = 1; i <= nums[index]; i++) {
            if (jumpFrom(index + i, nums) == true)
                return true;
        }
        return false;
    }
}
