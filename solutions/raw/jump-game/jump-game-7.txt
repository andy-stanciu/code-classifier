class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 0 || (nums[0] == 0 && nums.length != 1))
            return false;
        int rem = nums[0];
        for (int i = 1; i < nums.length - 1; i++) {
            rem = Math.max(rem - 1, nums[i]);
            if (rem == 0)
                return false;
        }
        if (rem > 0)
            return true;
        return true;
    }
}
