class Solution {

    public int func(int i, int[] arr, int[] dp) {
        if (i >= arr.length - 1)
            return 1;
        if (arr[i] == 0)
            return 0;
        if (dp[i] != -1)
            return dp[i];
        int t = arr[i];
        for (int j = 1; j <= t; j++) {
            if (func(i + j, arr, dp) == 1)
                return 1;
        }
        return dp[i] = 0;
    }

    public boolean canJump(int[] nums) {
        int n = nums.length;
        if (n == 1)
            return true;
        int[] dp = new int[n];
        Arrays.fill(dp, -1);
        return func(0, nums, dp) == 1 ? true : false;
    }
}
