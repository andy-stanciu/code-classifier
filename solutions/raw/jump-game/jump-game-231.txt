class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        int needJump = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < needJump) {
                needJump++;
            } else {
                needJump = 1;
            }
        }
        return nums[0] >= needJump;
    }
}
