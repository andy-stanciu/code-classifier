class Solution {

    public boolean canJump(int[] nums) {
        int lastZero = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] == 0 && lastZero == -1) {
                lastZero = i;
            }
            if (nums[i] > lastZero - i) {
                lastZero = -1;
            }
        }
        return lastZero == -1;
    }
}
