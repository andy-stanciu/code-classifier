class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1)
            return true;
        int farthest = 0;
        int curr = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            farthest = Math.max(farthest, i + nums[i]);
            if (i == curr) {
                curr = farthest;
            }
            if (curr >= nums.length - 1)
                return true;
        }
        return false;
    }
}
