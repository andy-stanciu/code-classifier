class Solution {

    public boolean canJump(int[] nums) {
        int i = nums.length - 1;
        int goal = nums.length - 1;
        while (i >= 0) {
            if (i + nums[i] >= goal) {
                goal = i;
            }
            i -= 1;
        }
        if (goal == 0)
            return true;
        else
            return false;
    }
}
