class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1)
            return true;
        int sol = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= sol)
                sol = i;
        }
        if (sol == 0)
            return true;
        else
            return false;
    }
}
