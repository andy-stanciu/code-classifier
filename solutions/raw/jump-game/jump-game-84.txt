class Solution {

    public boolean canJump(int[] nums) {
        return DFS(nums, 0);
    }

    public boolean DFS(int[] nums, int index) {
        if (index >= nums.length - 1)
            return true;
        if (nums[index] == 0)
            return false;
        //this is the part where it would lead to a TLE.
        int jump = nums[index];
        boolean result = false;
        while (jump > 0) {
            result = result || DFS(nums, index + jump);
            jump--;
        }
        return result;
    }
}
