class Solution {

    public boolean canJump(int[] nums) {
        //int pos = [2,3,1,1,4]
        int goal = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= goal) {
                goal = i;
            }
        }
        return goal == 0 ? true : false;
    }
}
