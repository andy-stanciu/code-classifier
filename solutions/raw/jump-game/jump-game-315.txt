class Solution {

    public boolean canJump(int[] nums) {
        int[] n = new int[nums.length];
        int c = 0;
        for (int i = 0; i < nums.length; i++) {
            if (c < i) {
                return false;
            }
            c = Math.max(c, i + nums[i]);
        }
        return true;
    }
}
