class Solution {

    public boolean canJump(int[] nums) {
        /*
         * Here we will be tracking last reachable point
         * so first we consider last point (nums.length-1) as last reachable point -LRP
         * then will will go one by one from right to left and will be checking if we
         * can reach the LRP from there or not, 
         * to check that we will use this condition :
         * 
         * int nextPointItCanReach = currentIndex + maxReachFromHere
        // ie. nextPointItCanReach = i + nums[i]
         * if( nextPointItCanReach >= LRP)
         * 
         * If \'yes\' then we will mark that as LRP, otherwise we will check same for more points
         * using this if we get LRP as 0th index then we reached the pos
         * 
         */
        int lastPointToReachEnd = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            int nextPointItCanReach = i + nums[i];
            if (nextPointItCanReach >= lastPointToReachEnd) {
                lastPointToReachEnd = i;
            }
        }
        return lastPointToReachEnd == 0;
    }
}
