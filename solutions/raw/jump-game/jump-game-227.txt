class Solution {

    public boolean canJump(int[] nums) {
        int maxi = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i <= maxi)
                maxi = Math.max(maxi, i + nums[i]);
            else
                break;
        }
        if (maxi >= nums.length - 1)
            return true;
        return false;
    }
}
