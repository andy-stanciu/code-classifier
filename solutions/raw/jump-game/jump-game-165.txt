class Solution {

    public boolean canJump(int[] nums) {
        return canjump(nums, 0);
    }

    public boolean canjump(int[] num, int ind) {
        if (ind > num.length)
            return false;
        if (ind == num.length - 1)
            return true;
        for (int i = 1; i <= num[ind]; i++) {
            if (canjump(num, ind + i)) {
                return true;
            }
        }
        return false;
    }
}
