class Solution {

    public boolean canJump(int[] nums) {
        int requiredJumps = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] >= requiredJumps) {
                requiredJumps = 1;
            } else {
                requiredJumps++;
            }
        }
        return nums[0] >= requiredJumps;
    }
}
