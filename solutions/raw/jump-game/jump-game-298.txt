class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        int lastIdx = n - 1;
        for (int i = 1; i < n; i++) {
            int k = n - 1 - i;
            if ((k + nums[k]) >= lastIdx) {
                lastIdx = k;
            }
        }
        return lastIdx == 0;
    }
}
