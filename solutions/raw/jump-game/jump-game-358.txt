class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1)
            return true;
        int curr = 0;
        int stopAt = nums[0];
        int nextStop = nums[0];
        while (curr < nums.length - 1) {
            if (curr > nextStop)
                return false;
            nextStop = Math.max(nextStop, curr + nums[curr]);
            if (nextStop >= nums.length - 1) {
                return true;
            }
            curr++;
        }
        return false;
    }
}
