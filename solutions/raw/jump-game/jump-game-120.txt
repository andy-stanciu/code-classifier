class Solution {

    public boolean canJump(int[] nums) {
        return canSolve(nums, 0);
    }

    public boolean canSolve(int[] nums, int i) {
        if (i == nums.length - 1)
            return true;
        if (i > nums.length - 1)
            return false;
        for (int j = 1; j <= nums[i]; j++) {
            if (canSolve(nums, i + j))
                return true;
        }
        return false;
    }
}
