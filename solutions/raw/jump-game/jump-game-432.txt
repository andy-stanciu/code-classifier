class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        if (n <= 1) {
            return true;
        }
        if (n == 0)
            return true;
        int max = 0;
        int maxRange = 0;
        int j = 0;
        for (int i = 0; i < n; i++) {
            max = Math.max(max, i + nums[i]);
            if (maxRange == i) {
                maxRange = max;
                j++;
                if (maxRange >= n - 1) {
                    return true;
                }
            }
        }
        return false;
    }
}
