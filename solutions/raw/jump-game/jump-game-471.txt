class Solution {

    public boolean canJump(int[] nums) {
        // keep track of the target
        int target = nums.length - 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            //if the previous element can reach the target, update the target
            if (i + nums[i] >= target) {
                target = i;
            }
        }
        return target == 0;
    }
}
