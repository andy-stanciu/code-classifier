class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > j) {
                return false;
            }
            if (j >= nums.length - 1) {
                return true;
            }
            j = Math.max(j, i + nums[i]);
        }
        return false;
    }
}
