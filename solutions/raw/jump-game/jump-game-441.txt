class Solution {

    public boolean canJump(int[] nums) {
        int gas = 0;
        for (int i = 0; i < nums.length; i++) {
            if (gas < 0)
                return false;
            else if (nums[i] > gas) {
                gas = nums[i];
            }
            gas--;
        }
        return true;
    }
}
