class Solution {

    public boolean canJump(int[] nums) {
        // max index we can jump to
        int indexCanJumpMax = 0;
        // on each index, check if we can jump to next index. keep doing this until we can jump to last index
        for (int i = 0; i < nums.length - 1; i++) {
            // calculate index to jump = number of jumps + current index
            int indexCanJumpCurrent = nums[i] + i;
            // update max index we can jump
            indexCanJumpMax = Math.max(indexCanJumpMax, indexCanJumpCurrent);
            // return FALSE if we can\'t jump to next index
            if (indexCanJumpMax < i + 1) {
                return false;
            }
        }
        // we can jump to last index since we got out of the loop
        return true;
    }
}
