class Solution {

    public boolean canJump(int[] nums) {
        // [3,2,1,0,4]
        int lastGoodIndex = nums[nums.length - 1];
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] + i >= lastGoodIndex) {
                lastGoodIndex = i;
            }
        }
        return lastGoodIndex == 0;
    }
}
