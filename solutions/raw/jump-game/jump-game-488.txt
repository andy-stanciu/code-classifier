class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length <= 1) {
            return true;
        }
        int maxjump = 0;
        int i = 0;
        while (i <= nums.length - 1) {
            if (i > maxjump) {
                return false;
            }
            if (i + nums[i] > maxjump) {
                maxjump = i + nums[i];
            }
            i++;
        }
        return true;
    }
}
