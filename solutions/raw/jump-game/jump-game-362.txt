class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1 || nums[0] + 1 >= nums.length)
            return true;
        int maxidx = 0;
        int i = 0;
        while (i < nums.length) {
            if (i > maxidx)
                return false;
            maxidx = Math.max(maxidx, i + nums[i]);
            i++;
        }
        return true;
    }
}
