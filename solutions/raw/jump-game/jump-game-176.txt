class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        int maxi = nums[0];
        for (int i = 0; i < n; i++) {
            if (maxi < i) {
                return false;
            }
            if (maxi >= n - 1) {
                return true;
            }
            maxi = Math.max(maxi, nums[i] + i);
        }
        return true;
    }
}
