class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length, max, flag;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0 && i != n - 1) {
                flag = 0;
                for (int j = 0; j < i; j++) {
                    if (j + nums[j] > i) {
                        flag = 1;
                        continue;
                    }
                }
                if (flag == 0) {
                    return false;
                }
            }
        }
        return true;
    }
}
