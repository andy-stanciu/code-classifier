class Solution {

    public boolean canJump(int[] nums) {
        int targetIndex = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (targetIndex <= nums[i] + i) {
                targetIndex = i;
            }
        }
        if (targetIndex == 0) {
            return true;
        }
        return false;
    }
}
