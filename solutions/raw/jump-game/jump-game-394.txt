class Solution {

    public boolean canJump(int[] nums) {
        int maxjump = 0;
        if (nums.length == 1)
            return true;
        for (int i = 0; i < nums.length; i++) {
            int jump = nums[i] + i;
            if (i <= maxjump) {
                maxjump = Math.max(maxjump, jump);
                if (maxjump >= nums.length - 1)
                    return true;
            }
            // if(i>maxjump) return false;
        }
        return false;
    }
}
