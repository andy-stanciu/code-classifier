class Solution {

    public boolean canJump(int[] nums) {
        // We will check, can we reach final position from second last position if yes
        // then we will make second last as final if not then we will not update final
        // and this time we\'ll check from one more previous position and so on....
        int finalPosition = nums.length - 1;
        for (int index = nums.length - 2; index >= 0; index--) {
            if (index + nums[index] >= finalPosition)
                finalPosition = index;
        }
        if (finalPosition == 0)
            return true;
        return false;
    }
}
