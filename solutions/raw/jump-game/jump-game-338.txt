class Solution {

    int[] dp = new int[10004];

    boolean jump(int idx, int[] nums) {
        if (idx >= nums.length - 1)
            return true;
        if (dp[idx] != -1)
            return dp[idx] == 1;
        boolean res = false;
        for (int i = 1; i <= nums[idx]; i++) {
            res |= jump(idx + i, nums);
            if (res == true)
                return true;
        }
        dp[idx] = res ? 1 : 0;
        return res;
    }

    public boolean canJump(int[] nums) {
        Arrays.fill(dp, -1);
        return jump(0, nums);
    }
}
