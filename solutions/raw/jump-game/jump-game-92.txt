class Solution {

    public boolean canJump(int[] nums) {
        int reachedIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > reachedIndex)
                return false;
            reachedIndex = Math.max(reachedIndex, nums[i] + i);
        }
        return true;
    }
}
