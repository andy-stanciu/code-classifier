class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1)
            return true;
        boolean flag = false;
        int cnt = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == 0) {
                if (cnt > 0 && flag == true) {
                    flag = false;
                    for (int j = i - 1; j >= 0; j--) {
                        if (nums[j] + j > i)
                            flag = true;
                    }
                }
                if (cnt == 0) {
                    for (int j = i - 1; j >= 0; j--) {
                        if (nums[j] + j > i)
                            flag = true;
                    }
                }
                cnt++;
            }
        }
        if (cnt == 0)
            return true;
        else
            return flag;
    }
}
