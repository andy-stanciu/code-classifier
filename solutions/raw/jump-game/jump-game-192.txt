class Solution {

    public boolean canJump(int[] nums) {
        int reachIdx = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= reachIdx) {
                reachIdx = i;
            }
        }
        // if reachIdx is zero it returns true otherwise false
        return reachIdx == 0;
    }
}
