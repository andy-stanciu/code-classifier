class Solution {

    public boolean canJump(int[] nums) {
        int counter = 1;
        boolean jump = true;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] >= counter) {
                jump = true;
                counter = 1;
            } else {
                jump = false;
                counter++;
            }
        }
        return jump;
    }
}
