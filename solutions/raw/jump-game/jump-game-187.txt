class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        boolean[] vis = new boolean[n];
        vis[n - 1] = true;
        int latest = n - 1;
        for (int i = n - 2; i >= 0; i--) {
            if (i + nums[i] >= latest) {
                vis[i] = true;
                latest = i;
            }
        }
        return vis[0];
    }
}
