public class Solution {

    public boolean canJump(int[] nums) {
        int maxJumpSoFar = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > maxJumpSoFar) {
                // If we reach index which cannot be reached from previous indexes
                return false;
            }
            maxJumpSoFar = Math.max(maxJumpSoFar, i + nums[i]);
        }
        return maxJumpSoFar >= nums.length - 1;
    }
}
