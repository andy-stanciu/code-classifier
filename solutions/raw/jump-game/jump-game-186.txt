class Solution {

    public boolean canJump(int[] nums) {
        int jumpsNeeded = 1;
        int i = nums.length - 2;
        while (i >= 0) {
            if (nums[i] >= jumpsNeeded) {
                jumpsNeeded = 1;
            } else {
                jumpsNeeded++;
            }
            i--;
        }
        return jumpsNeeded == 1;
    }
}
