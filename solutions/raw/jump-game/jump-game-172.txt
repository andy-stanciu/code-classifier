class Solution {

    public boolean canJump(int[] nums) {
        // define reachable variable
        int reachable = 0;
        for (int i = 0; i < nums.length; i++) {
            // if reachable become less than our current index then we cannot make it to the end, hence return false!
            if (reachable < i)
                return false;
            // else update the reachable with the updated value
            reachable = Math.max(reachable, i + nums[i]);
        }
        // at last return true
        return true;
    }
}
