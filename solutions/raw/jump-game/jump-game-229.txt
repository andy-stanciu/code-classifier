class Solution {

    public boolean canJump(int[] nums) {
        // we start from last
        // loop from last
        // main condtion -
        // target index <= index + interget at that index
        int targetNumIndex = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (targetNumIndex <= i + nums[i]) {
                targetNumIndex = i;
            }
        }
        return targetNumIndex == 0;
    }
}
