class Solution {

    public boolean canJump(int[] nums) {
        int farthestGood = nums[0];
        for (int pos = 1; pos < nums.length && pos <= farthestGood; pos++) {
            farthestGood = Math.max(nums[pos] + pos, farthestGood);
        }
        return farthestGood >= nums.length - 1;
    }
}
