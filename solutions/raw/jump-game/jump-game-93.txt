class Solution {

    public boolean canJump(int[] nums) {
        int maxJump = 0, lenMinus1 = nums.length - 1;
        for (int i = 0; maxJump >= i && maxJump < lenMinus1; i++) {
            if (i + nums[i] > maxJump) {
                //if Ai=0, maxJump = i
                maxJump = i + nums[i];
            }
        }
        return maxJump >= lenMinus1;
    }
}
