class Solution {

    public boolean canJump(int[] nums) {
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            //checks whether the current index is reachable from the previous indices or not
            if (i > count)
                return false;
            count = Math.max(count, i + nums[i]);
        }
        return true;
    }
}
