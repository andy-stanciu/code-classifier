class Solution {

    public boolean canJump(int[] nums) {
        return helper(nums, new boolean[nums.length], 0);
    }

    private boolean helper(int[] nums, boolean[] badPaths, int idx) {
        if (idx == nums.length - 1) {
            return true;
        }
        if (badPaths[idx])
            return false;
        int distance = nums[idx];
        for (int i = 1; i <= distance; i++) {
            if (helper(nums, badPaths, idx + i)) {
                return true;
            }
        }
        badPaths[idx] = true;
        return false;
    }
}
