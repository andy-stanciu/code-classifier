class Solution {

    public boolean canJump(int[] nums) {
        int len = nums.length;
        if (len == 1) {
            return true;
        }
        int maxReachFromZero = 0;
        for (int i = 0; i < len; i++) {
            if (i <= maxReachFromZero) {
                maxReachFromZero = Math.max(maxReachFromZero, i + nums[i]);
            } else {
                return false;
            }
            if (maxReachFromZero >= len - 1) {
                break;
            }
        }
        return true;
    }
}
