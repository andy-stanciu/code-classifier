class Solution {

    public boolean canJump(int[] nums) {
        int finalIndex = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if ((i + nums[i]) >= finalIndex) {
                finalIndex = i;
            }
        }
        if (finalIndex == 0)
            return true;
        else
            return false;
    }
}
