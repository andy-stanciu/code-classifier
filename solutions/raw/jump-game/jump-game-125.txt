class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 0)
            return false;
        if (nums[0] == 0 && nums.length > 1)
            return false;
        int reachable = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (i > reachable) {
                return false;
            }
            if (i + nums[i] > reachable) {
                reachable = i + nums[i];
            }
        }
        return true;
    }
}
