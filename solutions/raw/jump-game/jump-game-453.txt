class Solution {

    private boolean canJumpHelper(int[] nums, boolean[] visited, int pos) {
        if (visited[pos] == true) {
            return false;
        }
        if (pos + nums[pos] >= nums.length - 1) {
            return true;
        }
        for (int i = nums[pos]; i >= 1; i--) {
            if (canJumpHelper(nums, visited, i + pos)) {
                return true;
            }
        }
        visited[pos] = true;
        return false;
    }

    public boolean canJump(int[] nums) {
        boolean[] visited = new boolean[nums.length];
        return canJumpHelper(nums, visited, 0);
    }
}
