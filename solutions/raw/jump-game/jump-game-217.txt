class Solution {

    public boolean canJump(int[] nums) {
        int jumpIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > jumpIndex) {
                return false;
            }
            jumpIndex = (jumpIndex >= i + nums[i]) ? jumpIndex : i + nums[i];
        }
        return true;
    }
}
