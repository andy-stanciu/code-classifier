class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        boolean[] dp = new boolean[n];
        dp[n - 1] = true;
        for (int ind = n - 2; ind >= 0; ind--) {
            boolean ans = false;
            for (int i = 1; i <= nums[ind]; i++) {
                if (dp[ind + i]) {
                    ans = true;
                    break;
                }
            }
            dp[ind] = ans;
        }
        return dp[0];
    }
}
