class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1)
            return true;
        int maxInd = 0;
        int i = 0;
        while (i <= maxInd && i < nums.length) {
            maxInd = Math.max(maxInd, nums[i] + i);
            i++;
            if (maxInd >= nums.length - 1)
                return true;
        }
        return maxInd >= nums.length - 1;
    }
}
