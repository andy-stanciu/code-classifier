class Solution {

    public boolean canJump(int[] nums) {
        int ind = 0;
        int n = nums.length;
        for (int i = 0; i < nums.length; i++) {
            if (i > ind) {
                return false;
            }
            ind = Math.max(ind, i + nums[i]);
            if (ind >= nums.length - 1) {
                // If we can reach or exceed the last index
                return true;
            }
            // if(ind==n-1){
            //     return true;
            // }
            // ind+=nums[ind];
        }
        return false;
    }
}
