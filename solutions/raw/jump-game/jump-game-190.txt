class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] >= nums.length - 1 - i) {
                return canArrive(nums, i);
            }
        }
        return false;
    }

    private boolean canArrive(int[] nums, int index) {
        if (index == 0) {
            return true;
        }
        for (int i = index - 1; i >= 0; i--) {
            if (nums[i] >= index - i) {
                return canArrive(nums, i);
            }
        }
        return false;
    }
}
