class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        for (int i = 0; i < n - 1; i++) {
            if (i == 0) {
                dp[i] = nums[i];
            } else {
                dp[i] = Math.max(nums[i], dp[i - 1] - 1);
            }
            if (dp[i] == 0)
                return false;
        }
        return true;
    }
}
