class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1)
            return true;
        int count = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] >= count) {
                count = 1;
            } else {
                count++;
            }
        }
        if (count == 1)
            return true;
        return false;
    }
}
