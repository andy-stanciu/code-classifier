class Solution {

    public boolean canJump(int[] nums) {
        int jump = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i == 0 && nums[i] < jump) {
                return false;
            }
            if (nums[i] < jump) {
                jump++;
            } else {
                jump = 1;
            }
        }
        return true;
    }
}
