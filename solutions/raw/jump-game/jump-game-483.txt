class Solution {

    public boolean canJump(int[] nums) {
        int max = 0;
        for (int i = 0; i < nums.length; i++) {
            if (max < i) {
                return false;
            }
            int currMax = i + nums[i];
            max = Math.max(max, currMax);
            System.out.println("Max = " + max + " CurrMax=" + currMax);
        }
        return true;
    }
}
