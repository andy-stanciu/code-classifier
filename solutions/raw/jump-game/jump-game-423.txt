class Solution {

    public boolean canJump(int[] nums) {
        int[] dp = new int[nums.length];
        dp[nums.length - 1] = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= dp[i + 1]) {
                dp[i] = i;
            } else {
                dp[i] = dp[i + 1];
            }
            if (dp[i] <= nums[0]) {
                return true;
            }
        }
        return dp[0] == 0;
    }
}
