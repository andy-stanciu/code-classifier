class Solution {

    public boolean canJump(int[] nums) {
        // starting from the end, keep track of
        // what is the min index you have to reach (reachAtLeast)
        // to ensure you will reach the end
        int reachAtLeast = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] + i >= reachAtLeast) {
                reachAtLeast = i;
            }
        }
        return reachAtLeast == 0;
    }
}
