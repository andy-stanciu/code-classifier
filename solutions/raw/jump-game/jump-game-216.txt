class Solution {

    public boolean canJump(int[] nums) {
        int newTarget = nums.length - 1;
        if (nums.length == 1)
            return true;
        if (nums[0] == 0)
            return false;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= newTarget) {
                newTarget = i;
            }
        }
        if (newTarget == 0) {
            return true;
        } else
            return false;
    }
}
