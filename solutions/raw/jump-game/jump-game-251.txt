class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length - 1;
        int step = 1;
        for (int i = n - 1; i >= 0; i--) {
            if (nums[i] < step) {
                step++;
            } else {
                step = 1;
            }
        }
        return step == 1;
    }
}
