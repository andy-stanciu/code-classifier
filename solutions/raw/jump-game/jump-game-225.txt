class Solution {

    public boolean canJump(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        int[] memo = new int[nums.length];
        return jump(nums, 0, memo);
    }

    private boolean jump(int[] nums, int index, int[] memo) {
        if (index >= nums.length) {
            return false;
        }
        if (memo[index] == 1) {
            return false;
        }
        memo[index] = 1;
        if (nums.length == index + 1) {
            return true;
        }
        int root = nums[index];
        while (root > 0) {
            if (jump(nums, index + root, memo))
                return true;
            root--;
        }
        return false;
    }
}
