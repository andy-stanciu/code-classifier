class Solution {

    public boolean canJump(int[] nums) {
        int n = nums.length;
        boolean canJump = true;
        //if only one element, directly return
        if (n == 1)
            return true;
        //the minimum to jump over/cleanup the blocker
        int cleanUpNum = 1;
        for (int i = n - 2; i >= 0; i--) {
            int num = nums[i];
            //if it is the first blocker, set off alarm
            //the minimum num which can clean up the alarm
            if (canJump && num == 0) {
                canJump = false;
                cleanUpNum = 2;
            } else if (!canJump) {
                //if it is under alarm, we check if the current num
                //is bigger enough to clean up alarm;
                //if it can not, then increment the minimum num to cleanup
                if (num >= cleanUpNum) {
                    canJump = true;
                    cleanUpNum = 0;
                } else {
                    cleanUpNum++;
                }
            }
        }
        return canJump;
    }
}
