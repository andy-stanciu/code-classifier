class Solution {

    public int maxProfit(int[] prices) {
        int buy = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            int sell = prices[i];
            int diff = sell - buy;
            if (diff > profit)
                profit = diff;
            if (sell < buy)
                buy = sell;
        }
        return profit;
    }
}
