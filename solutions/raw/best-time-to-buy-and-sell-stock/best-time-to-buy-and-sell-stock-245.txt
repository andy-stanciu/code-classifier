class Solution {

    public int maxProfit(int[] prices) {
        //int p=0;
        int n = prices.length;
        int max_value = 0;
        int low = prices[0];
        for (int i = 1; i < n; i++) {
            int difference = prices[i] - low;
            max_value = Math.max(max_value, difference);
            low = Math.min(low, prices[i]);
        }
        return max_value;
    }
}
