class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int max_profit = 0, profit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (//finding minimum in the array
            prices[i] < min) {
                min = prices[i];
            } else {
                profit = prices[i] - min;
                if (//finding maximum profit
                profit > max_profit)
                    max_profit = profit;
            }
        }
        return max_profit;
    }
}
