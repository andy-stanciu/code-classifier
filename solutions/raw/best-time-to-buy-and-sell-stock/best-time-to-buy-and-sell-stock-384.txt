class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int n = prices.length;
        int maxpro = 0;
        for (int i = 0; i < n; i++) {
            if (prices[i] < min) {
                min = prices[i];
            } else {
                int profit = prices[i] - min;
                maxpro = Math.max(maxpro, profit);
            }
        }
        return maxpro;
    }
}
