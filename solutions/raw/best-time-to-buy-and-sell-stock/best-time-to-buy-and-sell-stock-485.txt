class Solution {

    public int maxProfit(int[] prices) {
        // to be minimized
        int buyPrice = prices[0];
        // to be maximized
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < buyPrice) {
                // minimize buyPrice
                buyPrice = prices[i];
            } else if (prices[i] - buyPrice > profit) {
                // maximize profit
                profit = prices[i] - buyPrice;
            }
        }
        return profit;
    }
}
