class Solution {

    public int maxProfit(int[] prices) {
        int buy_Price = prices[0];
        int profit = Integer.MIN_VALUE;
        for (int i = 0; i < prices.length; i++) {
            if (buy_Price > prices[i]) {
                buy_Price = prices[i];
            }
            profit = Math.max(profit, prices[i] - buy_Price);
        }
        return profit;
    }
}
