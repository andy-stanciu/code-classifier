class Solution {

    public int maxProfit(int[] prices) {
        // If the array is empty or contains only one price, no transaction is possible
        if (prices == null || prices.length == 0) {
            return 0;
        }
        // Initialize variables to track the minimum price and maximum profit
        int minPrice = Integer.MAX_VALUE;
        int maxProfit = 0;
        // Iterate through the price array
        for (int i = 0; i < prices.length; i++) {
            // Update the minimum price if a lower price is found
            if (prices[i] < minPrice) {
                minPrice = prices[i];
            } else // Calculate profit if the stock were sold on day i
            {
                int profit = prices[i] - minPrice;
                // Update maximum profit if current profit is higher
                if (profit > maxProfit) {
                    maxProfit = profit;
                }
            }
        }
        return maxProfit;
    }
}
