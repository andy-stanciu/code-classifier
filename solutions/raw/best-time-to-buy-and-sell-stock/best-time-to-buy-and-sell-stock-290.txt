class Solution {

    public int maxProfit(int[] prices) {
        int pro = prices[0];
        int max = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < pro) {
                pro = prices[i];
            } else if (prices[i] - pro > max) {
                max = prices[i] - pro;
            }
        }
        return max;
    }
}
