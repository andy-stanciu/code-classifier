class Solution {

    public int maxProfit(int[] prices) {
        int maxProfitSoFar = 0;
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        for (int price : prices) {
            if (!minHeap.isEmpty())
                maxProfitSoFar = Math.max(maxProfitSoFar, price - minHeap.peek());
            minHeap.add(price);
        }
        return maxProfitSoFar;
    }
}
