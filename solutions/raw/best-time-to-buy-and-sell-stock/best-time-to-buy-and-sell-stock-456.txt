class Solution {

    public int maxProfit(int[] prices) {
        // Variable to store the maximum profit
        int maxprofit = 0;
        // Variable to store the minimum price to buy stock (initially the first day\'s price)
        int buy = prices[0];
        // Iterate over the prices array
        for (int i = 0; i < prices.length; i++) {
            // Update the \'buy\' variable to be the minimum price encountered so far
            buy = Math.min(buy, prices[i]);
            // Calculate the potential profit if selling on day \'i\'
            int profit = prices[i] - buy;
            // Update maxprofit if this profit is larger than the previous one
            maxprofit = Math.max(maxprofit, profit);
        }
        // Return the maximum profit found
        return maxprofit;
    }
}
