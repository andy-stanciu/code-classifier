class Solution {

    public int maxProfit(int[] prices) {
        //if we are selling on one day then the buying rate should be the minimum
        int min = Integer.MAX_VALUE;
        int n = prices.length;
        int ans = 0;
        for (int i = 1; i < n; i++) {
            min = Math.min(min, prices[i - 1]);
            int cpro = prices[i] - min;
            ans = Math.max(ans, cpro);
        }
        return ans;
    }
}
