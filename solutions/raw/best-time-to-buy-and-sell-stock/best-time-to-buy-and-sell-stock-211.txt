class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 1)
            return 0;
        int buy_index = 0;
        int max_profit = 0;
        for (int i = 1; i < prices.length; i++) {
            /*if getting stock at lower price than before then buy it*/
            if (prices[buy_index] > prices[i]) {
                buy_index = i;
            } else /*if current price is greater than buying prices, sell it and keep mark of max profit*/
            if (prices[i] > prices[buy_index]) {
                max_profit = Math.max(max_profit, prices[i] - prices[buy_index]);
            }
        }
        return max_profit;
    }
}
