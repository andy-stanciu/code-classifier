class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int profit = Integer.MIN_VALUE;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] - min > profit)
                profit = prices[i] - min;
            if (prices[i] < min)
                min = prices[i];
        }
        if (profit < 0)
            return 0;
        else
            return profit;
    }
}
