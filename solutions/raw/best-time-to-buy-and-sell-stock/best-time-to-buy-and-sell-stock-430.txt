class Solution {

    public int maxProfit(int[] prices) {
        int min = Integer.MAX_VALUE;
        int maxdiff = 0;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < min) {
                min = prices[i];
            } else {
                profit = prices[i] - min;
                if (maxdiff < profit) {
                    maxdiff = profit;
                }
            }
        }
        return maxdiff;
    }
}
