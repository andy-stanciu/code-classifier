class Solution {

    public int maxProfit(int[] prices) {
        int p = 0;
        int buy = prices[0];
        for (int i = 1; i < prices.length; i++) {
            //if today\'s price is less than buy price, buy today
            if (buy > prices[i]) {
                buy = prices[i];
            } else {
                //if today\'s price is making better profit than before, sell today
                p = Math.max(p, prices[i] - buy);
            }
        }
        return p;
    }
}
