class Solution {

    public int maxProfit(int[] prices) {
        // Edge case: If prices array is null or has less than 2 elements, return 0
        if (prices == null || prices.length < 2) {
            return 0;
        }
        // Variable to store the maximum profit
        int maxp = 0;
        int minPrice = Integer.MAX_VALUE;
        for (int price : prices) {
            // Update minPrice to the minimum value encountered so far
            if (price < minPrice) {
                minPrice = price;
            } else {
                // Calculate the current profit if selling at the current price
                int currentProfit = price - minPrice;
                // Update maxp if current profit is greater than the previously recorded maxp
                if (currentProfit > maxp) {
                    maxp = currentProfit;
                }
            }
        }
        // Return the maximum profit found
        return maxp;
    }
}
