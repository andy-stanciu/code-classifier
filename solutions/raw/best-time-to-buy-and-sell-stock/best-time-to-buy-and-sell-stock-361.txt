class Solution {

    public int maxProfit(int[] prices) {
        int netProfit = 0;
        int buyPrice = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (buyPrice < prices[i]) {
                int currProfit = prices[i] - buyPrice;
                netProfit = Integer.max(currProfit, netProfit);
            } else
                buyPrice = prices[i];
        }
        return netProfit;
    }
}
