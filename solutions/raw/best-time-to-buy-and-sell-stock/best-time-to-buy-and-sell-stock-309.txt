class Solution {

    public int maxProfit(int[] prices) {
        if ((prices.length) == 0)
            return 0;
        int buy_amt = prices[0];
        int diff = 0;
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            buy_amt = Math.min(buy_amt, prices[i]);
            diff = (prices[i] - buy_amt);
            profit = Math.max(profit, diff);
        }
        return profit;
    }
}
