class Solution {

    public int maxProfit(int[] prices) {
        int minPrice = Integer.MAX_VALUE;
        int maxProfit = 0;
        // Loop through each price in the array using a normal for loop
        for (int i = 0; i < prices.length; i++) {
            // Update minPrice if the current price is lower than the minPrice so far
            if (prices[i] < minPrice) {
                minPrice = prices[i];
            }
            // Calculate the profit and update maxProfit without using Math.max
            int profit = prices[i] - minPrice;
            if (profit > maxProfit) {
                maxProfit = profit;
            }
        }
        return maxProfit;
    }
}
