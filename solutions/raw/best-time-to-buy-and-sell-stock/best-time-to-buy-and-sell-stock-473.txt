class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int maxm = Integer.MIN_VALUE, sell = 0;
        for (int i = n - 1; i >= 0; i--) {
            maxm = Math.max(maxm, prices[i]);
            sell = Math.max(sell, maxm - prices[i]);
        }
        return sell;
    }
}
