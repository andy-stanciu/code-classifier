class Solution {

    public int maxProfit(int[] prices) {
        int Max = 0;
        int buy = prices[0];
        for (int i : prices) {
            if (buy < i) {
                int profit = i - buy;
                Max = Math.max(Max, profit);
            } else {
                buy = i;
            }
        }
        return Max;
    }
}
