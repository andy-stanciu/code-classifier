class Solution {

    public int maxProfit(int[] prices) {
        //It will store the minimum price till that price.
        //Let\'s assume that the price on first day is minimum.
        int minLeft = prices[0];
        //It will store the maximum profit when we buy on minimum price and sell on current price.
        //Assume we have no profit initially.
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            //minimum price will be the minimum of prev and current price.
            minLeft = Math.min(minLeft, prices[i]);
            //minLeft = minimum price.
            //prices[i] = current price.
            //current profit = current - minimum price =  prices[i] - minLeft.
            //maximum profit will be the maximum of current and previous profit.
            profit = Math.max(profit, (prices[i] - minLeft));
        }
        //Finally, return the profit.
        return profit;
    }
}
