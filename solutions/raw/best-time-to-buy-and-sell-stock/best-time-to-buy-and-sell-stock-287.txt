class Solution {

    public int maxProfit(int[] prices) {
        int minCost = prices[0], profit = 0;
        for (int i = 1; i < prices.length; i++) {
            profit = Math.max(profit, prices[i] - minCost);
            minCost = Math.min(minCost, prices[i]);
        }
        return profit;
    }
}
