class Solution {

    public int maxProfit(int[] prices) {
        //edge case in case stock bought and sold on same day
        if (prices == null || prices.length < 2) {
            return 0;
        }
        int maxProfit = 0;
        int leftBuy = 0;
        int rightSell = 1;
        int n = prices.length;
        while (rightSell < n) {
            //price starts from day 1 to achive profit
            int currentPrice = prices[rightSell];
            //price to buy
            int buyPrice = prices[leftBuy];
            if (buyPrice < currentPrice) {
                int currentProfit = currentPrice - buyPrice;
                maxProfit = Math.max(maxProfit, currentProfit);
            } else {
                //to set for the lesser price
                leftBuy = rightSell;
            }
            rightSell++;
        }
        return maxProfit;
    }
}
