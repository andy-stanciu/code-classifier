class Solution {

    public int maxProfit(int[] arr) {
        int maxPro = 0;
        int minprice = Integer.MAX_VALUE;
        for (int i = 0; i < arr.length; i++) {
            minprice = Math.min(minprice, arr[i]);
            maxPro = Math.max(maxPro, arr[i] - minprice);
        }
        return maxPro;
    }
}
