class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int buyPrice = Integer.MAX_VALUE;
        for (int price : prices) {
            if (price < buyPrice) {
                buyPrice = price;
            }
            maxProfit = Math.max(maxProfit, price - buyPrice);
        }
        return maxProfit;
    }
}
