class Solution {

    public int maxProfit(int[] prices) {
        //Using One Pass technique or Max sub-array
        //min buy price
        int min = prices[0];
        //max delta
        int maxD = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < min) {
                //get new lowest min
                min = prices[i];
                System.out.println("min " + min);
            } else {
                //get new highest delta
                int newMaxD = prices[i] - min;
                if (newMaxD > maxD) {
                    maxD = newMaxD;
                }
            }
        }
        return maxD;
    }
}
