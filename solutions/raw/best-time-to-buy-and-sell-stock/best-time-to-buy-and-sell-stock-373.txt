class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int buyIndex = 0;
        int sellIndex = 1;
        int len = prices.length;
        while (sellIndex < len) {
            if (prices[buyIndex] < prices[sellIndex]) {
                maxProfit = Math.max(maxProfit, prices[sellIndex] - prices[buyIndex]);
            } else {
                buyIndex = sellIndex;
            }
            sellIndex++;
        }
        return maxProfit;
    }
}
