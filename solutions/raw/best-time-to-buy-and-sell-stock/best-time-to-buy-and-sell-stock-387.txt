class Solution {

    public int maxProfit(int[] prices) {
        return topDown(0, 0, prices, Integer.MAX_VALUE);
    }

    public int topDown(int pos, int profit, int[] prices, int lowest) {
        if (pos >= prices.length) {
            return profit;
        }
        if (prices[pos] < lowest) {
            lowest = prices[pos];
        }
        profit = Math.max(profit, prices[pos] - lowest);
        return topDown(pos + 1, profit, prices, lowest);
    }
}
