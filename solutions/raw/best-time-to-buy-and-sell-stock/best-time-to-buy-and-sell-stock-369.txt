class Solution {

    public int maxProfit(int[] prices) {
        int maxprofit = 0;
        int profit = 0;
        int buy = Integer.MAX_VALUE;
        for (int i = 0; i < prices.length; i++) {
            if (buy < prices[i]) {
                profit = prices[i] - buy;
                maxprofit = Math.max(profit, maxprofit);
            } else {
                buy = prices[i];
            }
        }
        return maxprofit;
    }
}
