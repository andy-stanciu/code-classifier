class Solution {

    public int maxProfit(int[] prices) {
        //let us assmue the minimum value as the
        //first element in the given array
        int min = prices[0];
        int profit = 0;
        //starting iterating from the 1st index hence the 0 index
        //is considered as the min in the abvoe steps
        for (int i = 1; i < prices.length; i++) {
            int cost = prices[i] - min;
            profit = Math.max(profit, cost);
            min = Math.min(prices[i], min);
        }
        return profit;
    }
}
