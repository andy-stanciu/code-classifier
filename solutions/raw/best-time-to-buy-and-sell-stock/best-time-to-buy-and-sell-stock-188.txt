class Solution {

    public int maxProfit(int[] prices) {
        rec(prices, prices.length, Integer.MIN_VALUE);
        return ans;
    }

    int ans = Integer.MIN_VALUE;

    public void rec(int[] arr, int N, int max) {
        if (N == 0) {
            return;
        }
        if (arr[N - 1] > max) {
            max = arr[N - 1];
        }
        int possibleCandidate = max - arr[N - 1];
        if (ans < possibleCandidate) {
            ans = possibleCandidate;
        }
        rec(arr, N - 1, max);
    }
}
