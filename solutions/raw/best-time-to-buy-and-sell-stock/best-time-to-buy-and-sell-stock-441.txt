class Solution {

    public int maxProfit(int[] arr) {
        if (arr == null || arr.length == 0) {
            // No profit possible
            return 0;
        }
        // Initialize min price
        int minPrice = arr[0];
        // Initialize max profit
        int maxProfit = 0;
        for (int i = 1; i < arr.length; i++) {
            // Update min price if a lower price is found
            if (arr[i] < minPrice) {
                minPrice = arr[i];
            }
            // Calculate profit if stock is sold on day i and update max profit if necessary
            int potentialProfit = arr[i] - minPrice;
            if (potentialProfit > maxProfit) {
                maxProfit = potentialProfit;
            }
        }
        return maxProfit;
    }
}
