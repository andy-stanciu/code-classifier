class Solution {

    public int maxProfit(int[] prices) {
        int sell = 0;
        int smleft = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < smleft) {
                smleft = prices[i];
            }
            sell = Math.max(sell, prices[i] - smleft);
        }
        return sell;
    }
}
