class Solution {

    public int maxProfit(int[] prices) {
        int len = prices.length;
        int profit = 0;
        int min = prices[0];
        for (int i = 1; i < len; i++) {
            if (prices[i] <= min) {
                min = prices[i];
            } else {
                profit = Math.max(profit, prices[i] - min);
            }
        }
        return profit;
    }
}
