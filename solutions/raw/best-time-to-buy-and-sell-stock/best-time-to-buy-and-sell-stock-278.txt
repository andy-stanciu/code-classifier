class Solution {

    public int maxProfit(int[] prices) {
        int min = Integer.MAX_VALUE;
        int max = 0;
        for (Integer i : prices) {
            if (i < min) {
                min = i;
            }
            int profit = i - min;
            if (max < profit) {
                max = profit;
            }
        }
        return max;
    }
}
