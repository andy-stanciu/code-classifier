class Solution {

    public int maxProfit(int[] prices) {
        int max = Integer.MIN_VALUE;
        int ans = Integer.MIN_VALUE;
        for (// creating suffix max array to take max price and taking max profit from orignal price and max price
        // creating suffix max array to take max price and taking max profit from orignal price and max price
        // creating suffix max array to take max price and taking max profit from orignal price and max price
        int i = prices.length - 1; i >= 0; i--) {
            max = Math.max(prices[i], max);
            ans = Math.max(max - prices[i], ans);
        }
        return ans;
    }
}
