class Solution {

    public int maxProfit(int[] prices) {
        int bp = prices[0];
        int profit = 0;
        int curr = 0;
        for (int i = 1; i < prices.length; i++) {
            curr = prices[i] - bp;
            profit = Math.max(profit, curr);
            bp = Math.min(prices[i], bp);
        }
        return profit;
    }
}
