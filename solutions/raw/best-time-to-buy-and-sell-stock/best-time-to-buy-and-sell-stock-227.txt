class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int mini = prices[0];
        int maxProfit = 0;
        int cost;
        for (int i = 1; i < n; i++) {
            cost = prices[i] - mini;
            maxProfit = Math.max(maxProfit, cost);
            mini = Math.min(mini, prices[i]);
        }
        return maxProfit;
    }
}
