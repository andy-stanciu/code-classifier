class Solution {

    public int maxProfit(int[] prices) {
        int ans = 0;
        int min = Integer.MAX_VALUE;
        int max = 0;
        for (int i = 0; i < prices.length; i++) {
            if (min > prices[i])
                min = prices[i];
            max = prices[i] - min;
            if (max > ans)
                ans = max;
        }
        return ans;
    }
}
