class Solution {

    public int maxProfit(int[] prices) {
        int minPri = Integer.MAX_VALUE;
        int maxPro = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < minPri) {
                minPri = prices[i];
            } else if (prices[i] - minPri > maxPro) {
                maxPro = prices[i] - minPri;
            }
        }
        return maxPro;
    }
}
