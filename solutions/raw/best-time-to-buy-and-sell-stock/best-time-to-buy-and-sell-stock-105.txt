class Solution {

    public int maxProfit(int[] prices) {
        int length = prices.length;
        int minValue = prices[0];
        int maxProfit = 0;
        for (int i = 1; i < length; i++) {
            // Every time check the profit, i.e, current price - previous minimum price
            maxProfit = Math.max(prices[i] - minValue, maxProfit);
            // If current value is less than the existing minimum value, that means new minimum value will be current value
            minValue = Math.min(minValue, prices[i]);
        }
        return maxProfit;
    }
}
