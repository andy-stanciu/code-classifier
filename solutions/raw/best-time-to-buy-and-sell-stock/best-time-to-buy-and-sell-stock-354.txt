class Solution {

    public int maxProfit(int[] arr) {
        int profit = 0;
        int cost;
        int maxi;
        int mini = arr[0];
        for (int i = 0; i < arr.length; i++) {
            cost = arr[i] - mini;
            profit = Math.max(profit, cost);
            mini = Math.min(mini, arr[i]);
        }
        return profit;
    }
}
