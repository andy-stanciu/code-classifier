class Solution {

    public int maxProfit(int[] prices) {
        int max_profit = 0, min_profit = prices[0];
        for (int i = 1; i < prices.length; i++) {
            int price = prices[i] - min_profit;
            max_profit = price > max_profit ? price : max_profit;
            min_profit = prices[i] < min_profit ? prices[i] : min_profit;
        }
        return max_profit;
    }
}
