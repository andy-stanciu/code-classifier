class Solution {

    public int maxProfit(int[] prices) {
        int buy = 0;
        int profit = 0;
        for (int j = buy + 1; j < prices.length; j++) {
            if (prices[j] < prices[buy]) {
                // && j != prices.length-1(unnecessary)
                buy = j;
            }
            profit = Math.max(profit, prices[j] - prices[buy]);
        }
        return profit;
    }
}
