class Solution {

    public int maxProfit(int[] prices) {
        // profit = max(sp - min(cp));
        int cp = prices[0], profit = 0;
        for (int i = 1; i < prices.length; i++) {
            cp = Math.min(prices[i], cp);
            profit = Math.max(profit, prices[i] - cp);
        }
        return profit;
    }
}
