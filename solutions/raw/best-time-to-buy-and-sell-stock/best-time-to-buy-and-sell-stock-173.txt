class Solution {

    public int maxProfit(int[] prices) {
        int minStockValue = Integer.MAX_VALUE;
        int maxProfit = Integer.MIN_VALUE;
        for (int i : prices) {
            if (i < minStockValue)
                minStockValue = i;
            if (maxProfit < i - minStockValue)
                maxProfit = i - minStockValue;
        }
        return maxProfit;
    }
}
