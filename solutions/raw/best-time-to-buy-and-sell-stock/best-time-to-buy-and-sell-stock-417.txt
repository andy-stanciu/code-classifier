class Solution {

    public int maxProfit(int[] prices) {
        int max_profit = 0;
        int mini_price = prices[0];
        for (int i = 1; i < prices.length; i++) {
            int cost = prices[i] - mini_price;
            max_profit = Math.max(max_profit, cost);
            mini_price = Math.min(mini_price, prices[i]);
        }
        return max_profit;
    }
}
