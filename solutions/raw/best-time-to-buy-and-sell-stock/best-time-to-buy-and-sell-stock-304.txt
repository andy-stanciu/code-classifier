class Solution {

    public int maxProfit(int[] prices) {
        int sell = prices[0], buy = prices[0], ans = 0;
        for (int i = 0; i < prices.length; i++) {
            if (buy > prices[i]) {
                buy = prices[i];
                sell = prices[i];
            }
            sell = Math.max(sell, prices[i]);
            ans = Math.max(sell - buy, ans);
        }
        return ans;
    }
}
