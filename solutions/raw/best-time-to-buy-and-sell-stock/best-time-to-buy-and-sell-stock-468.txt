class Solution {

    public int maxProfit(int[] prices) {
        int buyprice = prices[0];
        int maxpro = 0;
        for (int i = 0; i < prices.length; i++) {
            if (buyprice < prices[i]) {
                int profit = prices[i] - buyprice;
                maxpro = Math.max(profit, maxpro);
            } else {
                buyprice = prices[i];
            }
        }
        return maxpro;
    }
}
