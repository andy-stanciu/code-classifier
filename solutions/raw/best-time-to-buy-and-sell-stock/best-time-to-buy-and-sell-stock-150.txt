class Solution {

    public int maxProfit(int[] prices) {
        //keep a minprice variable which denotes the min price uptill that particular day
        //profit on that day = cost of the stock on that day - minprice
        //if the price of stock on that day < minprice then update minprice
        //keep a maxprofit variable and keep updating it as you traverse along the array
        int minprice = prices[0], maxprofit = 0;
        for (int day = 1; day < prices.length; day++) {
            int present_day_profit = prices[day] - minprice;
            if (present_day_profit > maxprofit)
                maxprofit = present_day_profit;
            if (prices[day] < minprice) {
                minprice = prices[day];
            }
        }
        return maxprofit;
    }
}
