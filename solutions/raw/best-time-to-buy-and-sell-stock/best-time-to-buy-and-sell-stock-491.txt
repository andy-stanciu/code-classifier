class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 1)
            return 0;
        int output = 0, lowestPrice = Integer.MAX_VALUE;
        for (int price : prices) {
            lowestPrice = price < lowestPrice ? price : lowestPrice;
            output = (price - lowestPrice) > output ? (price - lowestPrice) : output;
        }
        return output;
    }
}
