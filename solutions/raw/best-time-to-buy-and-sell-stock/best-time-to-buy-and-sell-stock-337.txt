class Solution {

    public int maxProfit(int[] prices) {
        int min = Integer.MAX_VALUE;
        int max = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < min) {
                // Update min with the lowest price so far
                min = prices[i];
            } else {
                // Calculate max profit so far
                max = Math.max(max, prices[i] - min);
            }
        }
        return max;
    }
}
