class Solution {

    public int maxProfit(int[] prices) {
        int buy = Integer.MAX_VALUE;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            //Buy
            if (prices[i] < buy) {
                buy = prices[i];
            } else //Sell
            {
                if ((prices[i] - buy) > profit)
                    profit = (prices[i] - buy);
            }
        }
        return profit;
    }
}
