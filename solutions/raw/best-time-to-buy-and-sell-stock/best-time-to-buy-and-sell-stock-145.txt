class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int minPrice = 100000;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < minPrice)
                minPrice = prices[i];
            else {
                maxProfit = prices[i] - minPrice < 0 ? 0 : Math.max(maxProfit, prices[i] - minPrice);
            }
        }
        return maxProfit;
    }
}
