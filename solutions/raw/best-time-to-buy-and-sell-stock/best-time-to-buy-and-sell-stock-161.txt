class Solution {

    public int maxProfit(int[] prices) {
        // Initialize {profit} with 0, as if there is only one element in array we can\'t have a sell day
        int profit = 0;
        // Minimum buy price till current day
        int min = prices[0];
        // Iterate from the second day to last day
        for (int day = 1; day < prices.length; day++) {
            // Update {min}, if current day price is less than minimum price
            if (prices[day] < min) {
                min = prices[day];
            } else {
                // else, update profit if selling on current day {day} is more profitable than selling on any other day in past
                profit = Math.max(profit, prices[day] - min);
            }
        }
        return profit;
    }
}
