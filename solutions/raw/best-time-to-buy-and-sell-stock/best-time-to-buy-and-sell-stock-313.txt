class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int buy = prices[0];
        for (int i : prices) {
            if (i > buy) {
                if ((i - buy) > maxProfit)
                    maxProfit = i - buy;
            } else {
                buy = i;
            }
        }
        return maxProfit;
    }
}
