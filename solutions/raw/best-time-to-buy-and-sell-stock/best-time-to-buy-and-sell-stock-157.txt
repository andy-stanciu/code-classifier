class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int right = prices[n - 1];
        int max = 0;
        for (int i = n - 2; i >= 0; i--) {
            right = Math.max(prices[i], right);
            max = Math.max(max, right - prices[i]);
        }
        return max;
    }
}
