class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int ans = 0;
        int mini = prices[0];
        for (int i = 1; i < n; i++) {
            int temp = prices[i] - mini;
            ans = Math.max(temp, ans);
            mini = Math.min(mini, prices[i]);
        }
        return ans;
    }
}
