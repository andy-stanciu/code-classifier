class Solution {

    public int maxProfit(int[] prices) {
        int MaxProfit = 0;
        int bestbuy = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] > bestbuy) {
                MaxProfit = Math.max(MaxProfit, prices[i] - bestbuy);
            }
            bestbuy = Math.min(bestbuy, prices[i]);
        }
        return MaxProfit;
    }
}
