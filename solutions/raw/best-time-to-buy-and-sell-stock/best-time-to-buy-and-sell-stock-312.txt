class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int profit;
        int buyPrice = prices[0];
        for (int price : prices) {
            if (buyPrice < price) {
                profit = price - buyPrice;
                if (profit > maxProfit)
                    maxProfit = profit;
            } else {
                buyPrice = price;
            }
        }
        return maxProfit;
    }
}
