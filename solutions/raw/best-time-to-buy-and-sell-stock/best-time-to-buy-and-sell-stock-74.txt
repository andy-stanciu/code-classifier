class Solution {

    public int maxProfit(int[] prices) {
        int buy = Integer.MAX_VALUE;
        int maxP = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] > buy) {
                int profit = prices[i] - buy;
                maxP = Math.max(profit, maxP);
            } else
                buy = prices[i];
        }
        return maxP;
    }
}
