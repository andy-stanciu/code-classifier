class Solution {

    public int maxProfit(int[] prices) {
        int[] min = new int[prices.length];
        int max = 0;
        for (int i = 0; i < prices.length; i++) min[i] = i == 0 ? prices[i] : Math.min(min[i - 1], prices[i]);
        for (int i = 0; i < prices.length; i++) {
            int ans = prices[i] - min[i];
            max = Math.max(max, ans);
        }
        return max;
    }
}
