class Solution {

    public int maxProfit(int[] prices) {
        int least = Integer.MAX_VALUE;
        int totalProfit = 0;
        int ifSell = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < least) {
                least = prices[i];
            } else {
                ifSell = prices[i] - least;
                if (totalProfit < ifSell) {
                    totalProfit = ifSell;
                }
            }
        }
        return totalProfit;
    }
}
