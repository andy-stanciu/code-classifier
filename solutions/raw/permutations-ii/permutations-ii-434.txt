class Solution {

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> out = new ArrayList();
        List<Integer> in = new ArrayList();
        for (int i : nums) in.add(i);
        permute(in, 0, out);
        return out;
    }

    public void permute(List<Integer> in, int k, List<List<Integer>> res) {
        for (int i = k; i < in.size(); i++) {
            Collections.swap(in, i, k);
            permute(in, k + 1, res);
            Collections.swap(in, k, i);
        }
        if (k == in.size() - 1) {
            if (!res.contains(in))
                res.add(new ArrayList<>(in));
        }
    }
}
