class Solution {

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        solution(nums, 0, ans);
        return ans;
    }

    public void solution(int[] arr, int ind, List<List<Integer>> ans) {
        if (ind == arr.length) {
            List<Integer> ds = new ArrayList<>();
            for (int i = 0; i < arr.length; ++i) {
                ds.add(arr[i]);
            }
            if (!ans.contains(ds)) {
                ans.add(new ArrayList<>(ds));
            }
        }
        for (int i = ind; i < arr.length; ++i) {
            int t = arr[ind];
            arr[ind] = arr[i];
            arr[i] = t;
            solution(arr, ind + 1, ans);
            t = arr[ind];
            arr[ind] = arr[i];
            arr[i] = t;
        }
    }
}
