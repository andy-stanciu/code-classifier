class Solution {

    public void swap(int i, int j, int[] s) {
        int temp = s[i];
        s[i] = s[j];
        s[j] = temp;
    }

    public void findperm(int i, List<List<Integer>> li, int[] nums, HashSet<List<Integer>> s) {
        if (i == nums.length - 1) {
            //    ArrayList<Integer>list=new ArrayList<>();
            //    for(int ik=0;ik<nums.length;ik++)
            // //    {
            //     list.add(nums[i]);
            //    }
            List<Integer> lis = new ArrayList<>();
            for (int ip = 0; ip < nums.length; ip++) {
                lis.add(nums[ip]);
            }
            li.add(lis);
            s.add(lis);
            return;
        }
        for (int j = i; j < nums.length; j++) {
            swap(i, j, nums);
            findperm(i + 1, li, nums, s);
            swap(i, j, nums);
        }
    }

    public List<List<Integer>> permuteUnique(int[] nums) {
        int i = 0;
        List<List<Integer>> li = new ArrayList<>();
        HashSet<List<Integer>> s = new HashSet<>();
        findperm(i, li, nums, s);
        List<List<Integer>> l = new ArrayList<>(s);
        return l;
    }
}
