class Solution {

    public int[] countBits(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 0;
        int i = 1;
        while (i <= n) {
            int afterSetLastBitOff = i - (i & -i);
            dp[i] = 1 + dp[afterSetLastBitOff];
            i++;
        }
        return dp;
    }
}
