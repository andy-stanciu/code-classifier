class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            res[i] = getDigitCount(i);
        }
        return res;
    }

    public int getDigitCount(int dec) {
        int count = 0;
        while (dec > 0) {
            int x = dec % 2;
            if (x == 1) {
                count++;
            }
            dec = dec / 2;
        }
        return count;
    }
}
