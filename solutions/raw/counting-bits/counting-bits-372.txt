class Solution {

    public int[] countBits(int n) {
        int[] arrayBits = new int[n + 1];
        String stringBits;
        for (int i = 0; i < arrayBits.length; i++) {
            stringBits = Integer.toBinaryString(i);
            arrayBits[i] = stringBits.length() - stringBits.replace("1", "").length();
        }
        return arrayBits;
    }
}
