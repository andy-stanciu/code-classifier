class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int count = 0;
            if (i == 0)
                arr[i] = 0;
            else {
                int t = i;
                while (t > 0) {
                    if ((t % 2) == 1)
                        count++;
                    t = t / 2;
                }
                arr[i] = count;
            }
        }
        return arr;
    }
}
