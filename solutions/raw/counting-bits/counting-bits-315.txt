class Solution {

    public int[] countBits(int n) {
        if (n == 0) {
            return new int[] { 0 };
        }
        if (n == 1) {
            return new int[] { 0, 1 };
        }
        int[] result = new int[n + 1];
        result[0] = 0;
        result[1] = 1;
        for (int i = 2; i <= n; i++) {
            int num = i;
            result[i] = result[num % 2] + result[num / 2];
        }
        return result;
    }
}
