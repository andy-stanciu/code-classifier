class Solution {

    public int[] countBits(int n) {
        if (n == 0)
            return new int[] { 0 };
        if (n == 1)
            return new int[] { 0, 1 };
        int[] result = new int[n + 1];
        result[0] = 0;
        result[1] = 1;
        if (n < 2)
            return result;
        for (int i = 2; i <= n; i++) {
            int q = i / 2;
            int r = i % 2;
            result[i] = result[q] + r;
        }
        return result;
    }
}
