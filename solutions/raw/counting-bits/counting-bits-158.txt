class Solution {

    public int countOne(int n) {
        int ans = 0;
        while (n != 0) {
            int lBit = n & 1;
            if (lBit == 1)
                ans++;
            n = n >> 1;
        }
        return ans;
    }

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = n; i >= 0; i--) {
            ans[i] = countOne(i);
        }
        return ans;
    }
}
