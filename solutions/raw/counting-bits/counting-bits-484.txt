class Solution {

    public int[] countBits(int n) {
        int[] result = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int sum = 0;
            String num = Integer.toBinaryString(i);
            for (int j = 0; j < num.length(); j++) {
                sum += Integer.valueOf(String.valueOf(num.charAt(j)));
            }
            result[i] = sum;
        }
        return result;
    }
}
