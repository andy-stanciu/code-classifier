class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        for (int i = 0; i < n; i++) {
            int j = n - i;
            int count = 0;
            while (j != 0) {
                j = j & (j - 1);
                count++;
            }
            arr[n - i] = count;
        }
        return arr;
    }
}
