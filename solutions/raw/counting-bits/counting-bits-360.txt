class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i < arr.length; i++) {
            int count = 0;
            n = i;
            while (n != 0) {
                count += n & 1;
                n = n >>> 1;
            }
            arr[i] = count;
        }
        return arr;
    }
}
