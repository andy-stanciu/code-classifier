class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            int count = 0, curr = i;
            while (curr != 0) {
                curr = curr & (curr - 1);
                count++;
            }
            arr[i] = count;
        }
        return arr;
    }
}
