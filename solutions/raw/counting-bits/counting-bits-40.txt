class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = countSetBits(i);
        }
        return ans;
    }

    public int countSetBits(int n) {
        int count = 0;
        for (int i = 0; i < 4 * 8; i++) {
            if ((n & (1 << i)) != 0)
                count++;
        }
        return count;
    }
}
// TC: O(n), SC: O(n)
