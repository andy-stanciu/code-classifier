class Solution {

    public int[] countBits(int n) {
        int len = n + 1;
        int[] arr = new int[len];
        arr[0] = 0;
        for (int j = 1; j < len; j++) arr[j] = getNoOfSetBits(j);
        return arr;
    }

    private int getNoOfSetBits(int j) {
        int count = 0;
        for (int i = 0; i < 32; i++) if ((j & (1 << i)) > 0)
            count++;
        return count;
    }
}
