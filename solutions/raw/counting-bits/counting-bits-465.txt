class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        int j = 1;
        for (int i = 0; i <= n; i++) {
            arr[i] = countSetBit(i);
        }
        return arr;
    }

    private int countSetBit(int num) {
        int count = 0;
        while (num > 0) {
            count += num & 1;
            num = num >> 1;
        }
        return count;
    }
}
