class Solution {

    public int[] countBits(int n) {
        int[] setBits = new int[n + 1];
        setBits[0] = 0;
        for (int i = 1; i <= n; i++) {
            setBits[i] = countSetBits(i);
        }
        return setBits;
    }

    public int countSetBits(int n) {
        String s = Integer.toBinaryString(n).replaceAll("0", "");
        return s.length();
    }
}
// TC: O(n^2 * logn), SC: O(n)
