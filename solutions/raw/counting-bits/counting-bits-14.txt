//Approach -3
//Even Faster
class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int count = 0;
            int num = i;
            while (num != 0) {
                // this removes the least significant set bit
                num = (num & (num - 1));
                count++;
            }
            // Store the count of 1\'s in the ans array at index i.
            ans[i] = count;
        }
        // Return the array containing the count of 1\'s for each number from 0 to n.
        return ans;
    }
}
