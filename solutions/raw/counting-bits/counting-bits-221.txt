class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        while (n != 0) {
            int i = n;
            int counter = 0;
            while (i != 0) {
                int rmsb = i & -i;
                i -= rmsb;
                counter++;
            }
            res[n] = counter;
            n--;
        }
        return res;
    }
}
