class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i < ans.length; i++) {
            int a = i;
            int c = 0;
            while (a > 0) {
                if ((a & 1) == 1) {
                    c++;
                }
                a >>= 1;
            }
            ans[i] = c;
        }
        return ans;
    }
}
