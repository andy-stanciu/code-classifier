class Solution {

    public int[] countBits(int n) {
        int[] out = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            //count is resetted for each iteration
            int count = 0;
            int attender = i;
            while (attender > 0) {
                //covertion of binary digits
                int temp = attender % 2;
                attender = attender / 2;
                if (temp == 1) {
                    //if 1 occurred in binary digit count is increased
                    count++;
                }
            }
            out[i] = count;
        }
        return out;
    }
}
