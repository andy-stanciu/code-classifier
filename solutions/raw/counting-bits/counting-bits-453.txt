class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            res[i] = countOnes(i);
        }
        return res;
    }

    private int countOnes(int n) {
        int ones = 0;
        while (n > 0) {
            if ((n & 1) == 1) {
                ones++;
            }
            n >>= 1;
        }
        return ones;
    }
}
