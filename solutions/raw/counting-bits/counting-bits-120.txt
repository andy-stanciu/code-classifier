class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        //One Line solution using Integer.bitCount() method
        for (int i = 0; i <= n; i++) {
            res[i] = Integer.bitCount(i);
        }
        /*Without using any ready made method in Java 
         if(n == 0){
            return res;
         }

         res[0] = 0;
         if(n >=1){
            res[1] = 1;
         }

        for(int i=2;i<res.length;i++){
            
            if(i%2 == 0){
                res[i] = res[i/2];
            }
            else{
                res[i] = res[i/2]+1;
            }
        }
        */
        return res;
    }
}
