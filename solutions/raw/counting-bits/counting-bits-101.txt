class Solution {

    public int[] countBits(int n) {
        int[] result = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            int temp = i;
            int count = 0;
            while (temp != 0) {
                temp &= temp - 1;
                count++;
            }
            result[i] = count;
        }
        return result;
    }
}
