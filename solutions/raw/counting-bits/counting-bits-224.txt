class Solution {

    public int[] countBits(int n) {
        int[] dp = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            if ((i & (i + 1)) == i) {
                dp[i] = dp[i >> 1];
            } else {
                dp[i] = 1 + dp[i & (i - 1)];
            }
        }
        return dp;
    }
}
