class Solution {

    public int[] countBits(int n) {
        int[] p = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            p[i] = count(i);
        }
        return p;
    }

    public int count(int n) {
        int count1s = 0;
        while (n != 0) {
            count1s += n & 1;
            n >>= 1;
        }
        return count1s;
    }
}
