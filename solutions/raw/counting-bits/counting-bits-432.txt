class Solution {

    public int[] countBits(int n) {
        int[] list = new int[n + 1];
        int k = 1;
        for (int i = 0; i <= n; i++) {
            if (i >= 2) {
                if (i % 2 == 0)
                    list[i] = list[i / 2];
                else
                    list[i] = list[i - 1] + 1;
            } else
                list[i] = i;
        }
        return list;
    }
}
