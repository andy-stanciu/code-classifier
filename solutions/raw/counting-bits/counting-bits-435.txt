class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = countIBits(i);
        }
        return ans;
    }

    public int countIBits(int i) {
        int count = 0;
        while (i != 0) {
            int last = (i & 1);
            if (last == 1) {
                count++;
            }
            i = i >> 1;
        }
        return count;
    }
}
