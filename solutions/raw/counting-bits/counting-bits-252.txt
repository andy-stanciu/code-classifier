class Solution {

    public int[] countBits(int num) {
        int[] result = new int[num + 1];
        for (int i = 1; i <= num; i++) {
            int temp = i;
            int amount = 0;
            while (temp != 0) {
                if (temp % 2 == 1)
                    amount++;
                temp /= 2;
            }
            result[i] = amount;
        }
        return result;
    }
}
