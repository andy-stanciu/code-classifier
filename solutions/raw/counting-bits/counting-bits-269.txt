class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        ans[0] = 0;
        int curPow = 1;
        int nextPow = 2;
        for (int i = 1; i <= n; i++) {
            if (i == nextPow) {
                ans[i] = 1;
                curPow = i;
                nextPow *= 2;
                continue;
            }
            ans[i] = 1 + ans[i - curPow];
        }
        return ans;
    }
}
