class Solution {

    static int getNrOfOnes(int n) {
        int cont = 0;
        for (int i = 17; i >= 0; i--) {
            if ((n & (1 << i)) != 0)
                cont++;
        }
        return cont;
    }

    public int[] countBits(int n) {
        return IntStream.range(0, n + 1).mapToObj(Integer::valueOf).mapToInt(Solution::getNrOfOnes).toArray();
    }
}
