class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        int count = 0;
        int j = 0;
        for (int i = 0; i <= n; i++) {
            count = 0;
            count += set(i);
            ans[j] = count;
            j++;
        }
        return ans;
    }

    public int set(int n) {
        int count = 0;
        while (n != 0) {
            if ((n & 1) != 0) {
                count++;
            }
            n = n >> 1;
        }
        return count;
    }
}
