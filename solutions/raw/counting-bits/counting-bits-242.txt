class Solution {

    public int[] countBits(int n) {
        int[] dp = new int[n + 1];
        if (n == 0)
            return dp;
        dp[1] = 1;
        int nextPow = 2;
        int prevPow = 2;
        for (int i = 2; i < n + 1; ++i) {
            if (i == nextPow) {
                dp[i] = 1;
                prevPow = nextPow;
                nextPow *= 2;
            } else {
                dp[i] = 1 + dp[i - prevPow];
            }
        }
        return dp;
    }
}
