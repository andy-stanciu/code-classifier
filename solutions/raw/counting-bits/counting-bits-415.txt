class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        Arrays.fill(arr, 0);
        for (int i = 1; i <= n; i++) {
            arr[i] = countSetBits(i);
        }
        return arr;
    }

    int countSetBits(int x) {
        int c = 0;
        int temp = 1;
        while (x > 0) {
            temp = x & 1;
            if (temp != 0) {
                c++;
            }
            x = x >> 1;
        }
        return c;
    }
}
