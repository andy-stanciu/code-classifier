class Solution {

    public int[] countBits(int n) {
        if (n == 0) {
            int[] ans = { 0 };
            return ans;
        }
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        int val = 2;
        for (int i = 2; i <= n; i++) {
            if (((i) & (i - 1)) == 0) {
                dp[i] = 1;
                val = i;
            } else {
                dp[i] = dp[val] + dp[i - val];
            }
        }
        return dp;
    }
}
