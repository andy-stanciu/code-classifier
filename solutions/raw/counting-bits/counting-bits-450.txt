class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        int position = 0;
        for (int i = 1; i <= n; i++) {
            position = i / 2;
            // even case
            if (i % 2 == 0) {
                arr[i] = arr[position];
            } else {
                // Odd case
                arr[i] = arr[position] + 1;
            }
        }
        return arr;
    }
}
