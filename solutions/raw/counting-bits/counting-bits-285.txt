class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        int ind = 0;
        for (int i = 0; i <= n; i++) {
            countOnes(ans, i, ind++);
        }
        return ans;
    }

    public void countOnes(int[] ans, int x, int ind) {
        int count = 0;
        while (x > 0) {
            if ((x & 1) == 1) {
                count++;
            }
            x >>= 1;
        }
        ans[ind] = count;
    }
}
