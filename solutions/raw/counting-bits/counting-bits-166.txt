class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i < ans.length; i++) {
            String binaryVal = Integer.toBinaryString(i);
            int count = binaryVal.length() - binaryVal.replace("1", "").length();
            ans[i] = count;
        }
        return ans;
    }
}
