class Solution {

    public int[] countBits(int n) {
        int[] bits = new int[n + 1];
        int pow = 0;
        for (int i = 1; i < n + 1; i++) {
            bits[i] = bits[i - (1 << pow)] + 1;
            if ((i & (i + 1)) == 0) {
                pow++;
            }
        }
        return bits;
    }
}
