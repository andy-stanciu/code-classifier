class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = checkBits(i);
        }
        return ans;
    }

    private int checkBits(int n) {
        int cnt = 0;
        while (n > 0) {
            n = n - (n & -n);
            cnt++;
        }
        return cnt;
    }
}
