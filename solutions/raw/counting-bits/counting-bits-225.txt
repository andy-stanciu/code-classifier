class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int bits = 0;
            int num = i;
            while (num > 0) {
                bits += num % 2;
                num /= 2;
            }
            arr[i] = bits;
        }
        return arr;
    }
}
