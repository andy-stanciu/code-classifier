class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        res[0] = 0;
        for (int i = 1; i <= n; i++) {
            int num = i, cnt = 0;
            while (num > 0) {
                if (num % 2 == 1)
                    cnt++;
                num = num / 2;
            }
            res[i] = cnt;
        }
        return res;
    }
}
