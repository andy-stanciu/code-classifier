class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i < arr.length; i++) {
            int temp = i;
            int count = 0;
            while (i > 0) {
                if ((i & 1) == 1) {
                    count++;
                }
                i = i >> 1;
            }
            i = temp;
            arr[i] = count;
        }
        return arr;
    }
}
