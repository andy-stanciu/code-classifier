class Solution {

    public int[] countBits(int n) {
        int offset = 1;
        int[] ans = new int[n + 1];
        ans[0] = 0;
        for (int i = 1; i <= n; i++) {
            if (offset * 2 == i) {
                offset *= 2;
            }
            ans[i] = 1 + ans[i - offset];
        }
        return ans;
    }
}
