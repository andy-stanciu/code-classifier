class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        int i = 0;
        while (i <= n) {
            ans[i] = counti(i);
            i++;
        }
        return ans;
    }

    public int counti(int n) {
        int cnt = 0;
        while (n != 0) {
            int rmsb = n & -n;
            n -= rmsb;
            cnt++;
        }
        return cnt;
    }
}
