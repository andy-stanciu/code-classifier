class Solution {

    public int[] countBits(int n) {
        int[] bits = new int[n + 1];
        bits[0] = 0;
        if (n == 0)
            return bits;
        bits[1] = 1;
        for (int i = 2; i < n + 1; i++) {
            if (i % 2 == 1)
                bits[i] = Math.min(bits[i - 1], bits[i - 2]) + 1;
            else
                bits[i] = bits[i / 2];
        }
        return bits;
    }
}
