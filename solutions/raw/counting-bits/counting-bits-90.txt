class Solution {

    public int countbits(int n) {
        if (n > 1)
            return n % 2 + countbits(n / 2);
        return n % 2;
    }

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) res[i] = countbits(i);
        return res;
    }
}
