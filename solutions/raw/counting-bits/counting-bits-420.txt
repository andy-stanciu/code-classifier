class Solution {

    public int[] countBits(int n) {
        int[] ar = new int[n + 1];
        ar[0] = 0;
        if (n > 0) {
            for (int i = 1; i <= n; i++) {
                int c = 0, j = i;
                while (j > 0) {
                    c += j & 1;
                    j = j >> 1;
                }
                ar[i] = c;
            }
        }
        return ar;
    }
}
