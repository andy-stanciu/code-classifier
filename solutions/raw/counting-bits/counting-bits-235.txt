class Solution {

    public int[] countBits(int n) {
        int[] bitsCounter = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            bitsCounter[i] = counter(i);
        }
        return bitsCounter;
    }

    private int counter(int n) {
        if (n == 0) {
            return 0;
        }
        int c = counter(n / 2);
        return n % 2 == 1 ? c + 1 : c;
    }
}
