class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int count = 0;
            int d = i;
            while (d != 0) {
                count++;
                d = d & (d - 1);
            }
            res[i] = count;
        }
        return res;
    }
}
