class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        dfs(ans, 1, 1);
        return ans;
    }

    private static void dfs(int[] ans, int index, int setBits) {
        if (index >= ans.length)
            return;
        ans[index] = setBits;
        dfs(ans, index * 2, setBits);
        dfs(ans, index * 2 + 1, setBits + 1);
    }
}
