class Solution {

    public int[] countBits(int n) {
        int[] cbit = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            cbit[i] = bit(i);
        }
        return cbit;
    }

    public static int bit(int n) {
        if (n == 0) {
            return 0;
        }
        int count = 0;
        while (n != 0) {
            if (n % 2 == 1) {
                count++;
            }
            n /= 2;
        }
        return count;
    }
}
