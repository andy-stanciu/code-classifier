class Solution {

    public int[] countBits(int n) {
        int val = 1;
        int[] res = new int[n + 1];
        res[0] = 0;
        for (int i = 1; i <= n; i++) {
            if (val == i) {
                res[i] = 1;
                val = val << 1;
            } else {
                int val2 = val / 2;
                res[i] = 1 + (res[i - val2]);
            }
        }
        return res;
    }
}
