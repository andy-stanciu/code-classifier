class Solution {

    public int[] countBits(int n) {
        int[] answer = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            answer[i] = CountBinary(i);
        }
        return answer;
    }

    int CountBinary(int i) {
        int count = 0;
        while (i != 0) {
            count += i & 1;
            i = i >> 1;
        }
        return count;
    }
}
