class Solution {

    public int[] countBits(int n) {
        int[] nums = new int[n + 1];
        for (int i = 0; i < nums.length; i++) {
            int count = 0;
            for (int j = 0; j < 32; j++) {
                if (countSetBits(i, j) == true) {
                    count++;
                }
            }
            nums[i] = count;
        }
        return nums;
    }

    public static boolean countSetBits(int n, int k) {
        if ((n & (1 << k)) == 0) {
            return false;
        } else {
            return true;
        }
    }
}
