class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        // Base case: 0 has zero 1s in its binary representation
        ans[0] = 0;
        for (int i = 1; i <= n; i++) {
            // Use the relationship: ans[i] = ans[i >> 1] + (i & 1)
            ans[i] = ans[i >> 1] + (i & 1);
        }
        return ans;
    }
}
