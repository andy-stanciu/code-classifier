class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        int k = 0;
        int rem = 0;
        int x = 0;
        for (int i = 0; i <= n; i++) {
            x = i;
            k = 0;
            while (x != 0) {
                rem = x % 2;
                if (rem == 1) {
                    k++;
                }
                x = x / 2;
            }
            res[i] = k;
        }
        return res;
    }
}
