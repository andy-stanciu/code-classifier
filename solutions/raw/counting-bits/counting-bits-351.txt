class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = countbit(i);
        }
        return ans;
    }

    public int countbit(int n) {
        int bitSum = 0;
        while (n != 0) {
            bitSum += (n & 1);
            n = n >>> 1;
        }
        return bitSum;
    }
}
