class Solution {

    public int setBits(int n) {
        int ans = 0;
        while (n > 0) {
            if ((n & 1) == 1)
                ans++;
            n >>= 1;
        }
        return ans;
    }

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            arr[i] = setBits(i);
        }
        return arr;
    }
}
