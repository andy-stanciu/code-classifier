class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        for (int i = 1; i <= n; i++) {
            int sum = i;
            int count = 0;
            while (sum > 0) {
                if ((sum & 1) == 1) {
                    count++;
                }
                sum = sum >> 1;
            }
            arr[i] = count;
        }
        return arr;
    }
}
