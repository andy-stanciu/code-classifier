class Solution {

    public int[] countBits(int n) {
        /////////// -> ////////
        int[] ans = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            int k = i;
            int c = 0;
            while (k != 0) {
                //mask for the given but;
                k = k & (k - 1);
                c = c + 1;
            }
            ans[i] = c;
        }
        return ans;
    }
}
