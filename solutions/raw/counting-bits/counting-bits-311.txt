class Solution {

    public int[] countBits(int n) {
        int twoBitNumber = 1;
        int[] answer = new int[n + 1];
        int count = 0;
        answer[0] = 0;
        for (int i = 1; i < n + 1; i++) {
            int x = i & i - 1;
            answer[i] = answer[x] + 1;
        }
        return answer;
    }
}
