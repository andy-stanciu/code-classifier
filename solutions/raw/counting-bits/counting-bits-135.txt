class Solution {

    public int hammingWeight(int n) {
        int n_bits = (int) (Math.log(n) / Math.log(2)) + 1;
        int cnt = 0;
        for (int i = 0; i < n_bits; i++) {
            if ((n & (1 << i)) != 0)
                cnt++;
        }
        return cnt;
    }

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            res[i] = hammingWeight(i);
        }
        return res;
    }
}
