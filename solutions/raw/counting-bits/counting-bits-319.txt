class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int x = i;
            int c = 0;
            while (x != 0) {
                int rsbm = (x & -x);
                x -= rsbm;
                c++;
            }
            arr[i] = c;
        }
        return arr;
    }
}
