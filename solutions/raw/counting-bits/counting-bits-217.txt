class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i < res.length; i++) {
            res[i] = hammingWeight(i);
        }
        return res;
    }

    public int hammingWeight(int n) {
        if (n == 1 || n == 0)
            return n;
        return (n & 1) + hammingWeight(n >> 1);
    }
}
