class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int setBits = 0;
            int num = i;
            while (num != 0) {
                num = num & (num - 1);
                setBits++;
            }
            arr[i] = setBits;
        }
        return arr;
    }
}
