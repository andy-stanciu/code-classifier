class Solution {

    public int[] countBits(int n) {
        if (n == 0)
            return new int[] { 0 };
        if (n == 1)
            return new int[] { 0, 1 };
        if (n == 2)
            return new int[] { 0, 1, 1 };
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 1;
        for (int i = 3; i <= n; i++) {
            int div = i / 2;
            int mod = i % 2;
            dp[i] = dp[div] + mod;
        }
        return dp;
    }
}
