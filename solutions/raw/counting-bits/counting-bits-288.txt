class Solution {

    public int[] countBits(int n) {
        int[] a = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            a[i] = fun(i);
        }
        return a;
    }

    int fun(int x) {
        int count = 0;
        while (x > 0) {
            count += x & 1;
            x = x >> 1;
        }
        return count;
    }
}
