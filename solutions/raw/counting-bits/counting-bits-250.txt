class Solution {

    public int[] countBits(int num) {
        int[] ans = new int[num + 1];
        int count = 0;
        ans[0] = 0;
        for (int i = 1; i <= num; i++) {
            if ((i & (i - 1)) == 0) {
                ans[i] = 1;
                count++;
            } else {
                ans[i] = 1 + ans[i - (1 << (count - 1))];
            }
        }
        return ans;
    }
}
