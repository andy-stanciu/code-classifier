class Solution {

    public int[] countBits(int n) {
        int[] arr = new int[n + 1];
        int power = 1;
        for (int i = 1; i <= n; i++) {
            if (((i) & (i - 1)) == 0) {
                power = i;
            }
            arr[i] = arr[i - power] + 1;
        }
        return arr;
    }
}
