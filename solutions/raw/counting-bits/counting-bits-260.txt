class Solution {

    public int[] countBits(int n) {
        int[] results = new int[n + 1];
        for (int i = 0; i < results.length; i++) {
            if (i % 2 != 0)
                results[i] = results[i / 2] + 1;
            else
                results[i] = results[i / 2];
        }
        return results;
    }
}
