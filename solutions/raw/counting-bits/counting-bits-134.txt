class Solution {

    private static int count(int n) {
        int counting = 0;
        while (n > 0) {
            if ((n & 1) != 0) {
                counting++;
            }
            n = n >> 1;
        }
        return counting;
    }

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = count(i);
        }
        return ans;
    }
}
