class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (var i = 1; i < ans.length; i++) {
            int remainder = i;
            while (remainder != 0) {
                ans[i] += remainder & 1;
                remainder >>>= 1;
            }
        }
        return ans;
    }
}
