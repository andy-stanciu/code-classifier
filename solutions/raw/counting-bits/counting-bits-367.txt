class Solution {

    public int[] countBits(int n) {
        String w = "";
        int[] arr = new int[n + 1];
        int k = 0;
        for (int i = 0; i <= n; i++) {
            k = 0;
            for (int j = i; j > 0; j /= 2) {
                if (j % 2 != 0) {
                    k++;
                }
            }
            arr[i] = k;
        }
        return arr;
    }
}
