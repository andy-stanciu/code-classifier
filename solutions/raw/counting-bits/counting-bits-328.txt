class Solution {

    int[] bitCount = new int[100001];

    public int[] countBits(int n) {
        int[] count = new int[n + 1];
        count[0] = 0;
        for (int i = 1; i <= n; i++) {
            if (i % 2 == 0) {
                count[i] = count[i / 2];
            } else {
                count[i] = count[i / 2] + 1;
            }
        }
        return count;
    }
}
