class Solution {

    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int num = 0; num <= n; num++) {
            // int count =0;
            // for(int i=0 ;i<=16;i++){
            //     if(((num >> i) & 1) !=0){ //if ith bit is set
            //         count++;
            //     }
            // }
            res[num] = countBit(num);
        }
        return res;
    }

    private int countBit(int n) {
        int count = 0;
        while (n > 0) {
            n = n & (n - 1);
            count++;
        }
        return count;
    }
}
