class Solution {

    public int[] countBits(int n) {
        if (n == 0) {
            return new int[] { 0 };
        }
        int[] bitsDp = new int[n + 1];
        bitsDp[0] = 0;
        bitsDp[1] = 1;
        for (int i = 2; i < n + 1; i++) {
            bitsDp[i] = bitsDp[i / 2] + (i % 2 == 0 ? 0 : 1);
        }
        return bitsDp;
    }
}
