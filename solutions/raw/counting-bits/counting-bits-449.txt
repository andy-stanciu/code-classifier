class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            ans[i] = (bitcount(i));
        }
        return ans;
    }

    private int bitcount(int num) {
        int cnt = 0;
        while (num > 0) {
            cnt++;
            num &= (num - 1);
        }
        return cnt;
    }
}
