class Solution {

    public int[] countBits(int n) {
        List<Integer> temp = new ArrayList<>();
        for (int i = 0; i <= n; i++) {
            int current = i;
            int ones = 0;
            while (current > 0) {
                int remainder = current % 2;
                if (remainder == 1) {
                    ones++;
                }
                current = current / 2;
            }
            temp.add(ones);
        }
        int[] arr = new int[temp.size()];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = temp.get(i);
        }
        return arr;
    }
}
