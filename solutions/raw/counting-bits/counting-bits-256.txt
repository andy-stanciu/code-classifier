class Solution {

    public int[] countBits(int n) {
        int[] a = new int[n + 1];
        a[0] = 0;
        int j = 1;
        for (int i = 1; i < a.length; i++) {
            if (j == i) {
                a[i] = 1;
                j = j << 1;
            } else if ((i & 1) == 1)
                a[i] = a[1] + a[i - 1];
            else
                a[i] = a[i / 2];
        }
        return a;
    }
}
