class Solution {

    public int[] countBits(int n) {
        return IntStream.range(0, n + 1).map(Solution::count1InNum).toArray();
    }

    public static int count1InNum(int num) {
        int count = 0;
        while (num != 0) {
            num = num & (num - 1);
            count++;
        }
        return count;
    }
}
