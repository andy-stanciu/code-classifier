class Solution {

    public int[] countBits(int n) {
        int[] ans = new int[n + 1];
        for (int i = 1; i <= n; i++) ans[i] = 1 + ans[i & (i - 1)];
        // the above line is the most crucial is cracking the solution
        // the 3 & 2 are broken into binary representation.
        //  & 3 are also broken dowb and so on.
        return ans;
    }
}
