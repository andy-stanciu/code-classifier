class Solution {

    int n;

    List<List<String>> ans;

    public List<List<String>> partition(String s) {
        n = s.length();
        ans = new ArrayList<List<String>>();
        FindSubstrings(s, 0, new ArrayList<String>());
        return ans;
    }

    void FindSubstrings(String ss, int ind, ArrayList<String> list) {
        // end condition
        if (ind == n) {
            ans.add(new ArrayList<String>(list));
            return;
        }
        // optional list
        // in this current option,we can start form index ,and end of n
        for (int i = ind + 1; i <= n; i++) {
            if (!IsPalindrome(ss.substring(ind, i)))
                continue;
            // we choose [index.i] as a substrings
            list.add(ss.substring(ind, i));
            FindSubstrings(ss, i, list);
            // we dont choose [index,i] as a substrings
            // so we remove it
            list.remove(list.size() - 1);
        }
    }

    boolean IsPalindrome(String s) {
        int lower = 0;
        int higher = s.length() - 1;
        while (lower < higher) {
            if (s.charAt(lower) != s.charAt(higher))
                return false;
            lower++;
            higher--;
        }
        return true;
    }
}
