class Solution {

    List<List<String>> result = new ArrayList<List<String>>();

    // in worst case when string is like AAAAAA
    // Time: O(2n*2^n) -> O(n*2^n)
    public List<List<String>> partition(String s) {
        findAll(0, s, new ArrayList<String>());
        return result;
    }

    private void findAll(int start, String s, List<String> partition) {
        if (start == s.length()) {
            // O(m) - m - size of partition
            result.add(new ArrayList<String>(partition));
            return;
        }
        //O(2^N) calls for this method
        for (int end = start; end < s.length(); end++) {
            // O(n) to check if palindrome
            if (isPalindrome(start, end, s)) {
                // O(n) to make a substring
                // n - in worst case, when substring is entire string
                partition.add(s.substring(start, end + 1));
                findAll(end + 1, s, partition);
                partition.remove(partition.size() - 1);
            }
        }
    }

    // O(n) to check if substring is palindrome
    // n - in worst case, when substring is entire string
    private boolean isPalindrome(int start, int end, String s) {
        if (start >= end)
            return true;
        while (start < end) {
            if (s.charAt(start) != s.charAt(end))
                return false;
            start++;
            end--;
        }
        return true;
    }
}
