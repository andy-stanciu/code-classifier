class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> result = new ArrayList<>();
        Backtrack(s, 0, new ArrayList<>(), result);
        return result;
    }

    public void Backtrack(String s, int start, List<String> path, List<List<String>> result) {
        if (start == s.length()) {
            result.add(new ArrayList<>(path));
            return;
        }
        for (int end = start + 1; end <= s.length(); end++) {
            String currentsub = s.substring(start, end);
            if (isPlaindrome(currentsub)) {
                path.add(currentsub);
                Backtrack(s, end, path, result);
                path.remove(path.size() - 1);
            }
        }
    }

    public boolean isPlaindrome(String sub) {
        int left = 0, right = sub.length() - 1;
        while (left < right) {
            if (sub.charAt(left) != sub.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
