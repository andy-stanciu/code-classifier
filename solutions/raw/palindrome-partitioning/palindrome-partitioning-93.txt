class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> list = new ArrayList<>();
        List<String> ans = new ArrayList<>();
        palindromePart(0, s, list, ans);
        return list;
    }

    private void palindromePart(int idx, String s, List<List<String>> list, List<String> ans) {
        if (idx == s.length()) {
            // Add a new copy of the current partition to the result
            list.add(new ArrayList<>(ans));
            return;
        }
        for (int i = idx; i < s.length(); i++) {
            // If the substring from idx to i is a palindrome
            if (palindrome(idx, i, s)) {
                // Add the palindrome substring
                ans.add(s.substring(idx, i + 1));
                // Recurse to find further partitions
                palindromePart(i + 1, s, list, ans);
                // Backtrack
                ans.remove(ans.size() - 1);
            }
        }
    }

    private boolean palindrome(int start, int end, String s) {
        while (start < end) {
            if (s.charAt(start) != s.charAt(end)) {
                // Not a palindrome
                return false;
            }
            start++;
            end--;
        }
        // It\'s a palindrome
        return true;
    }
}
