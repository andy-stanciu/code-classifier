class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> partitionDataList = new ArrayList<>();
        List<String> path = new ArrayList<>();
        isPalindromeString(0, s, path, partitionDataList);
        return partitionDataList;
    }

    public void isPalindromeString(int index, String s, List<String> path, List<List<String>> partitiondataList) {
        if (index == s.length()) {
            partitiondataList.add(new ArrayList<>(path));
            return;
        }
        for (int i = index; i < s.length(); ++i) {
            if (isPalindromeStr(s, index, i)) {
                path.add(s.substring(index, i + 1));
                isPalindromeString(i + 1, s, path, partitiondataList);
                path.remove(path.size() - 1);
            }
        }
    }

    public boolean isPalindromeStr(String s, int startIndex, int endIndex) {
        while (startIndex <= endIndex) {
            if (s.charAt(startIndex++) != s.charAt(endIndex--)) {
                return false;
            }
        }
        return true;
    }
}
