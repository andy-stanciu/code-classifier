class Solution {

    public List<List<Integer>> subsets(int[] nums) {
        return helper(nums, 0, new ArrayList<Integer>());
    }

    private List<List<Integer>> helper(int[] up, int index, List<Integer> p) {
        List<List<Integer>> ans = new LinkedList<>();
        if (index == up.length) {
            ans.add(p);
            return ans;
        }
        var temp = new ArrayList<Integer>(p);
        temp.add(up[index]);
        ans.addAll(helper(up, index + 1, temp));
        ans.addAll(helper(up, index + 1, p));
        return ans;
    }
}
