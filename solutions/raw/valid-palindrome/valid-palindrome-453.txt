class Solution {

    public boolean isPalindrome(String s) {
        String s1 = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        int low = 0;
        int high = s1.length() - 1;
        while (low < high) {
            if (s1.charAt(low) == s1.charAt(high)) {
                low++;
                high--;
            } else {
                return false;
            }
        }
        return true;
    }
}
