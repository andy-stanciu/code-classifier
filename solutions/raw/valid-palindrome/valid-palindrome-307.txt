class Solution {

    public boolean isPalindrome(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "").trim();
        int len = s.length();
        for (int i = 0; i < len; i++) {
            if (s.substring(i, i + 1).equalsIgnoreCase(s.substring(len - i - 1, len - i))) {
                continue;
            }
            return false;
        }
        return true;
    }
}
