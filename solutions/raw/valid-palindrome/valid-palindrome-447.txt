class Solution {

    public boolean isPalindrome(String s) {
        // start
        int i = 0;
        //size of the string
        int j = s.length() - 1;
        while (i < j) {
            //loop start i to j
            //we use (.charAt(i)) to check the value/character of index i;
            while (i < j && !Character.isLetterOrDigit(s.charAt(i))) {
                i++;
            }
            //we use (.charAt(j)) to check the value/character of index j;
            while (i < j && !Character.isLetterOrDigit(s.charAt(j))) {
                j--;
            }
            //we add (.toLowerCase) functio ,to check (A=a)
            if (Character.toLowerCase(s.charAt(i)) != Character.toLowerCase(s.charAt(j))) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
