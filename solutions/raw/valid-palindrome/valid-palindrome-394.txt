class Solution {

    public boolean isPalindrome(String s) {
        if (s.length() == 0 || s.length() == 1) {
            return true;
        }
        String n = s.replaceAll("[^a-zA-Z0-9]", "");
        String m = n.toLowerCase();
        for (int i = 0; i < m.length(); i++) {
            char st = m.charAt(i);
            char end = m.charAt(m.length() - 1 - i);
            while (st != end) {
                return false;
            }
        }
        return true;
    }
}
