class Solution {

    public boolean isPalindrome(String s) {
        String input = s.replaceAll("\\\\s", "");
        String inter = input.replaceAll("_", "");
        String output = inter.replaceAll("\\\\W", "");
        char[] v = output.toLowerCase().toCharArray();
        int size = v.length;
        char[] reverse = new char[size];
        int j = 0;
        for (int i = v.length - 1; i >= 0; i--) {
            reverse[j] = v[i];
            j++;
        }
        String ss = new String(reverse);
        String sh = new String(v);
        if (sh.equals(ss))
            return true;
        else
            return false;
    }
}
