class Solution {

    public boolean isPalindrome(String s) {
        return isitpalindrome(s);
    }

    public static boolean isitpalindrome(String s) {
        int left = 0;
        int right = s.length() - 1;
        while (left < right) {
            while (left < right && !Character.isLetterOrDigit(s.charAt(left))) {
                // character.isLetterOrDigit is use to select only letter or digits
                // but we have to check all digit or alphabet so we used ! operature to
                // skip non alphanumeric character
                left++;
            }
            while (left < right && !Character.isLetterOrDigit(s.charAt(right))) {
                right--;
            }
            // compare left and right side
            if (Character.toLowerCase(s.charAt(left)) != Character.toLowerCase(s.charAt(right))) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
