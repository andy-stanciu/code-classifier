class Solution {

    public boolean isPalindrome(String s) {
        int org = 0;
        int reverse = s.length() - 1;
        while (org < reverse) {
            char temp = s.charAt(org);
            char temp1 = s.charAt(reverse);
            temp = Character.toLowerCase(temp);
            temp1 = Character.toLowerCase(temp1);
            if (!Character.isLetterOrDigit(temp)) {
                org++;
            } else if (!Character.isLetterOrDigit(temp1)) {
                reverse--;
            } else {
                if (temp != temp1) {
                    return false;
                }
                org++;
                reverse--;
            }
        }
        return true;
    }
}
