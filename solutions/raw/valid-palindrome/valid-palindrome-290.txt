class Solution {

    public boolean isPalindrome(String s) {
        String newParagraph = s.toLowerCase();
        newParagraph = newParagraph.replaceAll("[\\\\p{Punct}]", "");
        newParagraph = newParagraph.replaceAll("\\\\s", "");
        int length = newParagraph.length();
        for (int i = 0; i < length / 2; i++) {
            if (newParagraph.charAt(i) != newParagraph.charAt(length - 1 - i))
                return false;
        }
        return true;
    }
}
