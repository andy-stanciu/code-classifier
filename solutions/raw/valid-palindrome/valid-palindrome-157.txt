class Solution {

    public boolean isPalindrome(String s) {
        String cleanString = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int start = 0, end = cleanString.length() - 1;
        char[] charArr = cleanString.toCharArray();
        while (start < end) {
            if (charArr[start++] != charArr[end--])
                return false;
        }
        return true;
    }
}
