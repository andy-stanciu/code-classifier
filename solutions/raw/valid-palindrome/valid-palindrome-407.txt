class Solution {

    public boolean isPalindrome(String s) {
        return checkPalindrome(0, s.length() - 1, s);
    }

    public boolean checkPalindrome(int startPtr, int endPtr, String s) {
        if (startPtr >= endPtr)
            return true;
        boolean a = Character.isLetterOrDigit(s.charAt(startPtr));
        boolean b = Character.isLetterOrDigit(s.charAt(endPtr));
        if (a && b) {
            char c1 = Character.toLowerCase(s.charAt(startPtr));
            char c2 = Character.toLowerCase(s.charAt(endPtr));
            if (c1 != c2)
                return false;
            startPtr++;
            endPtr--;
        }
        if (!a)
            startPtr++;
        if (!b)
            endPtr--;
        return checkPalindrome(startPtr, endPtr, s);
    }
}
