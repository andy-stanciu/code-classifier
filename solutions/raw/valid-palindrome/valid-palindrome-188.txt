class Solution {

    public boolean isPalindrome(String s) {
        s = s.toUpperCase();
        String forwards = "";
        String backwards = "";
        for (int y = 0; y < s.length(); y++) {
            if (((int) (s.charAt(y)) > 64) && ((int) (s.charAt(y)) < 91) || ((int) (s.charAt(y)) > 47 && s.charAt(y) < 58)) {
                forwards += s.charAt(y);
                backwards = s.charAt(y) + backwards;
            }
        }
        System.out.println(forwards);
        System.out.println(backwards);
        return (forwards.equals(backwards));
    }
}
