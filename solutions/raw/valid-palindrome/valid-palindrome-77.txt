class Solution {

    public boolean isPalindrome(String s) {
        // converting given string to lowercase
        // then replacing all alpha numeric characters.
        // all non - alphanumeric characters will be replaced by "" , means nothing will be there.
        s = s.toLowerCase().replaceAll("[^A-Za-z0-9]", "");
        // starting a pointer i from 0 index and j from last .
        int i = 0;
        int j = s.length() - 1;
        //until they meet keep checking the values at each index
        // if not equal return false .
        // keep increasing i and decreasing j pointer till they meet.
        while (i <= j) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        // return true
        return true;
    }
}
