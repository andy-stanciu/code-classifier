class Solution {

    public boolean isPalindrome(String s) {
        s = s.toLowerCase();
        int j = s.length() - 1;
        int i = 0;
        return isPalindromes(s, i, j);
    }

    public boolean isPalindromes(String s, int i, int j) {
        if (i >= j) {
            return true;
        }
        if (i < j && !Character.isLetterOrDigit(s.charAt(i))) {
            return isPalindromes(s, i + 1, j);
        }
        if (i < j && !Character.isLetterOrDigit(s.charAt(j))) {
            return isPalindromes(s, i, j - 1);
        }
        if (s.charAt(i) != s.charAt(j)) {
            return false;
        } else {
            return isPalindromes(s, i + 1, j - 1);
        }
    }
}
