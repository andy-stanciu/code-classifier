class Solution {

    public boolean isPalindrome(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        s = s.toLowerCase();
        char[] ch = s.toCharArray();
        StringBuilder str = new StringBuilder();
        for (int i = (ch.length - 1); i >= 0; i--) {
            str.append(ch[i]);
        }
        String strr = str.toString();
        return s.equals(strr);
    }
}
