class Solution {

    public boolean isPalindrome(String s) {
        String a = s.toLowerCase();
        String b = a.replaceAll("[^a-zA-Z0-9]", "");
        System.out.println(b);
        String rev = "";
        for (int i = 0; i < b.length(); i++) {
            rev = b.charAt(i) + rev;
        }
        if (b.equals(rev)) {
            return true;
        }
        return false;
    }
}
