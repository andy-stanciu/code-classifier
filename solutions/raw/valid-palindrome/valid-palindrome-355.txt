class Solution {

    public boolean isPalindrome(String s) {
        String res = "";
        if (s == " ") {
            return true;
        }
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (Character.isLetterOrDigit(c)) {
                res += Character.toLowerCase(c);
            }
        }
        String reve = "";
        for (int i = res.length() - 1; i >= 0; i--) {
            reve += res.charAt(i);
        }
        return res.equals(reve);
    }
}
