class Solution {

    public boolean isPalindrome(String s) {
        String replace = s.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        String rev = "";
        if (replace == "")
            return true;
        for (int i = replace.length() - 1; i >= 0; i--) {
            rev = rev + replace.charAt(i);
        }
        if (replace.equals(rev)) {
            return true;
        }
        return false;
    }
}
