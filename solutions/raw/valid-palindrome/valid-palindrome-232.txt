class Solution {

    public boolean isPalindrome(String s) {
        StringBuilder allString = new StringBuilder();
        for (char c : s.toCharArray()) {
            if (Character.isLetterOrDigit(c)) {
                allString.append(Character.toLowerCase(c));
            }
        }
        StringBuilder reversedString = new StringBuilder(allString).reverse();
        // Check if the original filtered string is equal to the reversed string
        return allString.toString().equals(reversedString.toString());
    }
}
