class Solution {

    public boolean isPalindrome(String s) {
        // s = s.replaceAll(" ","");
        s = s.replaceAll("[^A-Za-z0-9]", "");
        s = s.toLowerCase();
        // System.out.println("->"+s+"<-");
        int len = s.length();
        for (int i = 0; i < len / 2; i++) {
            if (s.charAt(i) != s.charAt(len - 1 - i)) {
                return false;
            }
        }
        return true;
    }
}
