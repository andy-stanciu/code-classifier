class Solution {

    public boolean isPalindrome(String s) {
        StringBuilder ns = new StringBuilder();
        for (int i = 0; i < s.length(); i++) if (Character.isLetterOrDigit(s.charAt(i)))
            ns.append(s.substring(i, i + 1).toLowerCase());
        //return n.equals(new StringBuilder(n).reverse().toString());
        String n = ns.toString();
        for (int i = 0; i < n.length() / 2; i++) if (n.charAt(i) != n.charAt(n.length() - i - 1))
            return false;
        return true;
    }
}
