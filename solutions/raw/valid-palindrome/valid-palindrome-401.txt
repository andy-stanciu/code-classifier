class Solution {

    public boolean isPalindrome(String s) {
        s = s.replaceAll("[^A-Za-z0-9]", "");
        return rec(s.toLowerCase(), 0, s.length() - 1);
    }

    public boolean rec(String s, int l, int r) {
        if (l >= r) {
            return true;
        }
        if (r > 0 && s.charAt(l) == s.charAt(r)) {
            return rec(s, l + 1, r - 1);
        }
        return false;
    }
}
