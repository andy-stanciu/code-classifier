public class Solution {

    public boolean isPalindrome(String s) {
        // Convert the string to lower case
        s = s.toLowerCase();
        char[] spl = new char[s.length()];
        int k = 0;
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (Character.isLetterOrDigit(ch)) {
                // Check for alphanumeric characters
                spl[k++] = ch;
            }
        }
        char[] temp = new char[k];
        System.arraycopy(spl, 0, temp, 0, k);
        // Call the recursive function with the cleaned array
        return isP(temp, 0);
    }

    public boolean isP(char[] s, int k) {
        if (k >= s.length / 2) {
            // Base case: if we\'ve checked halfway, it\'s a palindrome
            return true;
        }
        if (s[k] != s[s.length - k - 1]) {
            // Compare characters for equality
            // If characters don\'t match, it\'s not a palindrome
            return false;
        }
        // Recursive call to check the next pair of characters
        return isP(s, k + 1);
    }
}
