class Solution {

    public boolean isPalindrome(String s) {
        int f = 0;
        int l = s.length() - 1;
        while (f <= l) {
            if (!(Character.isLetterOrDigit(s.charAt(f)))) {
                f++;
            } else if (!(Character.isLetterOrDigit(s.charAt(l)))) {
                l--;
            } else if (Character.toLowerCase(s.charAt(f)) != Character.toLowerCase(s.charAt(l))) {
                return false;
            } else {
                f++;
                l--;
            }
        }
        return true;
    }
}
