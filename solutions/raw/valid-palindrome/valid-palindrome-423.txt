class Solution {

    public boolean isPalindrome(String s) {
        int p1 = 0;
        int p2 = s.length() - 1;
        while (p1 <= p2) {
            char currP1 = s.charAt(p1);
            char currP2 = s.charAt(p2);
            if (!Character.isLetterOrDigit(currP1)) {
                p1++;
            } else if (!Character.isLetterOrDigit(currP2)) {
                p2--;
            } else {
                if (Character.toLowerCase(currP1) != Character.toLowerCase(currP2))
                    return false;
                p1++;
                p2--;
            }
        }
        return true;
    }
}
