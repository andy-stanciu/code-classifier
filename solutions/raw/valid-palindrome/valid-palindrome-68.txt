class Solution {

    public boolean isPalindrome(String s) {
        // normalisez and remove unwanted chars
        s = s.trim().toLowerCase().replaceAll("[^a-z0-9]", "");
        int i = 0;
        int j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i++) != s.charAt(j--)) {
                return false;
            }
        }
        return true;
    }
}
