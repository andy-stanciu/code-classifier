class Solution {

    public boolean isPalindrome(String s) {
        //remove spaces
        s = s.replace(" ", "");
        int n = s.length();
        //into lowercase
        s = s.toLowerCase();
        //initialising stringBuilder
        StringBuilder sb = new StringBuilder();
        //adding alphabets and numbers to stringBuilder => Alphanumeric
        for (int i = 0; i < n; i++) {
            if (Character.isAlphabetic(s.charAt(i)) || Character.isDigit(s.charAt(i))) {
                sb.append(s.charAt(i));
            }
        }
        //StringBuilder --> String
        s = sb.toString();
        //String has less than 1 length then surely be palindrome
        if (s.length() <= 1) {
            return true;
        }
        for (int i = 0; i < s.length() / 2; i++) {
            if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
