class Solution {

    public boolean isPalindrome(String s) {
        String str2 = "";
        String str = s.replaceAll("[^a-zA-Z0-9]", "");
        str = str.toLowerCase();
        for (int i = str.length() - 1; i >= 0; i--) {
            str2 = str2 + str.charAt(i);
        }
        if (str.equals(str2)) {
            return true;
        } else {
            return false;
        }
    }
}
