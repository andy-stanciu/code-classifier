class Solution {

    public boolean isPalindrome(String s) {
        if (s.length() == 0)
            return true;
        String original = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int start = 0;
        int end = original.length() - 1;
        while (start <= end) {
            if (original.charAt(start) != original.charAt(end))
                return false;
            start++;
            end--;
        }
        return true;
    }
}
