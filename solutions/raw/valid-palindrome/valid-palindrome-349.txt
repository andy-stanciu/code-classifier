class Solution {

    public boolean isPalindrome(String s) {
        String s2 = s.replaceAll("[^a-zA-Z0-9]", "");
        String s3 = "";
        for (int i = s2.length() - 1; i >= 0; i--) {
            s3 = s3 + s2.charAt(i);
        }
        if (s2.equalsIgnoreCase(s3)) {
            return true;
        } else {
            return false;
        }
    }
}
