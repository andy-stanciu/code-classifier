class Solution {

    public boolean isPalindrome(String s) {
        s = s.toLowerCase();
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        if (s.length() < 1)
            return true;
        System.out.print(s);
        return fun(s, 0);
    }

    public boolean fun(String s, int i) {
        if (i >= s.length() / 2)
            return true;
        if (s.charAt(i) != s.charAt(s.length() - i - 1))
            return false;
        return fun(s, i + 1);
    }
}
