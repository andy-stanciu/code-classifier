class Solution {

    public boolean isPalindrome(String s) {
        // replacing not of alphabets
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        //and number with "" nothing, empty
        s = s.toLowerCase();
        System.out.println(s);
        boolean result = true;
        if (s.length() == 1) {
            return true;
        }
        char[] c = s.toCharArray();
        int j = s.length();
        //Checking only half Part of the String so s.length()/2
        for (int i = 0; i < s.length() / 2; i++) {
            if (c[i] != c[j - 1]) {
                result = false;
            }
            j--;
        }
        return result;
    }
}
