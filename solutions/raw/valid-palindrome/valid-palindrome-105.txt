class Solution {

    public boolean isPalindrome(String s) {
        //Here we created two pointers which will help us to keep track
        int low = 0;
        int high = s.length() - 1;
        //We created two char variable to store head & tail char at a time for comparison.
        char p_head, p_tail;
        while (low < high) {
            //Here we assign the char at low & high index in p_head & p_tail resp.
            p_head = s.charAt(low);
            p_tail = s.charAt(high);
            //Fuction Character.isLetterOrDigit checks if the char passed is either letter or Digit & return boolean & we are using the same property to skip any special characters and spaces by increasing their index.
            if (!Character.isLetterOrDigit(p_head)) {
                low++;
            } else if (!Character.isLetterOrDigit(p_tail)) {
                high--;
            } else {
                //used function Character.toLowerCase to convert any UpperCase letter to lowercase letter as it is a requirement of Question.
                //we are also checking if the char at head and tail are same, if they are differnt will directly return false as it signifies that the string is not palindrome.
                if (Character.toLowerCase(p_head) != Character.toLowerCase(p_tail)) {
                    return false;
                }
                low++;
                high--;
            }
        }
        return true;
    }
}
