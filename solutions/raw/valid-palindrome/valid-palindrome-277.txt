class Solution {

    public boolean isPalindrome(String s) {
        String sc = s.toLowerCase();
        String replacestring = sc.replaceAll("[^a-z0-9]", "");
        StringBuilder sb = new StringBuilder(replacestring);
        sb.reverse();
        String str = sb.toString();
        if (replacestring.equals(str)) {
            return true;
        } else
            return false;
    }
}
