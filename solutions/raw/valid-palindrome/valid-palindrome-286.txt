class Solution {

    public boolean isPalindrome(String s) {
        int front = 0;
        int last = s.length() - 1;
        if (s.isEmpty() == true) {
            return true;
        } else {
            while (front <= last) {
                char atfirst = s.charAt(front);
                char atlast = s.charAt(last);
                if (!Character.isLetterOrDigit(atfirst)) {
                    front++;
                } else if (!Character.isLetterOrDigit(atlast)) {
                    last--;
                } else {
                    if (Character.toLowerCase(atfirst) == Character.toLowerCase(atlast)) {
                        front++;
                        last--;
                    } else {
                        return false;
                    }
                }
            }
            return true;
        }
    }
}
