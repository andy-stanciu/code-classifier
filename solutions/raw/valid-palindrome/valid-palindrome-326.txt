class Solution {

    public boolean isPalindrome(String s) {
        String result = "";
        int i = 0;
        for (i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (Character.isLetterOrDigit(ch)) {
                if (ch >= 65 && ch <= 90) {
                    result += Character.toLowerCase(ch);
                } else {
                    result += ch;
                }
            }
        }
        if (result.length() == 0 || result.length() == 1) {
            return true;
        }
        if (result.length() == 2) {
            if (result.charAt(0) != result.charAt(1)) {
                return false;
            }
            return true;
        }
        int j = 0;
        int k = result.length() - 1;
        while (true) {
            if (j > (result.length() / 2) || k < (result.length() / 2)) {
                break;
            }
            if (result.charAt(j) != result.charAt(k)) {
                return false;
            }
            j++;
            k--;
        }
        return true;
    }
}
