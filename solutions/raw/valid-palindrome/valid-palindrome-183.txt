class Solution {

    public boolean isPalindrome(String s) {
        int start = 0;
        char[] charArray = s.toCharArray();
        int end = s.length() - 1;
        while (start <= end) {
            char startCh = charArray[start];
            char endCh = charArray[end];
            if (!Character.isLetterOrDigit(startCh)) {
                start++;
                continue;
            }
            if (!Character.isLetterOrDigit(endCh)) {
                end--;
                continue;
            }
            startCh = Character.toLowerCase(startCh);
            endCh = Character.toLowerCase(endCh);
            if (startCh != endCh) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
