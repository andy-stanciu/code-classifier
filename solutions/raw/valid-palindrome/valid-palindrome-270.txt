class Solution {

    public boolean isPalindrome(String s) {
        boolean flag = true;
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        s = s.toLowerCase();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(s.length() - i - 1))
                flag = true;
            else {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
