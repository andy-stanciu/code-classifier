class Solution {

    public static boolean isPalindrome(String s) {
        if (s.isEmpty())
            return false;
        char[] chars = s.toLowerCase().toCharArray();
        for (int i = 0, j = chars.length - 1; i < j; ) {
            if (isAlphanumeric(chars[i]) && isAlphanumeric(chars[j])) {
                if (chars[i] != chars[j]) {
                    return false;
                }
                i++;
                j--;
            } else if (!isAlphanumeric(chars[i])) {
                i++;
            } else if (!isAlphanumeric(chars[j])) {
                j--;
            }
        }
        return true;
    }

    public static boolean isAlphanumeric(final int codePoint) {
        return (codePoint >= 97 && codePoint <= 122) || (codePoint >= 48 && codePoint <= 57);
    }
}
