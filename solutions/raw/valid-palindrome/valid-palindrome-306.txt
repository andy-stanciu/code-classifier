class Solution {

    public boolean isPalindrome(String s) {
        var lowerCase = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        StringBuilder str = new StringBuilder(lowerCase);
        StringBuilder reverse = str.reverse();
        if (reverse.toString().equals(lowerCase)) {
            return true;
        } else {
            return false;
        }
    }
}
