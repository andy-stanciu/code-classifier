class Solution {

    public boolean isPalindrome(String s) {
        String alphaNumChars = s.toLowerCase().strip().replaceAll("[^a-z0-9]", "");
        int point1 = 0;
        int point2 = alphaNumChars.length() - 1;
        while (point1 <= point2) {
            // comparing front to back pointers
            if (alphaNumChars.charAt(point1) != alphaNumChars.charAt(point2)) {
                return false;
            }
            point1++;
            point2--;
        }
        return true;
    }
}
