class Solution {

    public boolean isPalindrome(String s) {
        String s1 = s.replaceAll("[^a-zA-Z0-9]", "");
        boolean flag = true;
        int n = s1.length();
        for (int i = 0; i < n / 2; i++) {
            if (Character.toLowerCase(s1.charAt(i)) != Character.toLowerCase(s1.charAt(n - i - 1))) {
                flag = false;
            }
        }
        return flag;
    }
}
