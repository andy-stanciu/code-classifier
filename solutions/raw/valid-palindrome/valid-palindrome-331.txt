class Solution {

    public boolean isPalindrome(String s) {
        int p1 = 0;
        int p2 = s.length() - 1;
        while (p1 < p2) {
            char s1 = s.charAt(p1);
            char s2 = s.charAt(p2);
            while (!Character.isLetterOrDigit(s1) && p1 < p2) {
                p1++;
                s1 = s.charAt(p1);
            }
            while (!Character.isLetterOrDigit(s2) && p2 > p1) {
                p2--;
                s2 = s.charAt(p2);
            }
            if (Character.toLowerCase(s1) == Character.toLowerCase(s2)) {
                p1++;
                p2--;
            } else {
                return false;
            }
        }
        return true;
    }
}
