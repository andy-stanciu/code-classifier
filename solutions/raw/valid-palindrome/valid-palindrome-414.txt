class Solution {

    public boolean isPalindrome(String s) {
        StringBuilder originalStringCleaned = new StringBuilder();
        StringBuilder palindromeString = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (Character.isLetterOrDigit(s.charAt(i))) {
                originalStringCleaned.append(s.charAt(i));
            }
        }
        for (int i = originalStringCleaned.length() - 1; i >= 0; i--) {
            palindromeString.append(originalStringCleaned.charAt(i));
        }
        return originalStringCleaned.toString().equalsIgnoreCase(palindromeString.toString());
    }
}
