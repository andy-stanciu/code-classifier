class Solution {

    public boolean isPalindrome(String s) {
        String nonAlpha = s.replaceAll("[^a-zA-Z0-9]", "");
        nonAlpha = nonAlpha.toLowerCase();
        String res = "";
        for (int i = nonAlpha.length() - 1; i >= 0; i--) {
            res = res + nonAlpha.charAt(i);
        }
        if (nonAlpha.equals(res)) {
            return true;
        } else {
            return false;
        }
    }
}
