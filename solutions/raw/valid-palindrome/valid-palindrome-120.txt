class Solution {

    private static final String VALID_STRING = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";

    public boolean isPalindrome(String s) {
        String cleanString = Arrays.stream(s.split("")).filter(i -> VALID_STRING.contains(i)).map(String::toLowerCase).collect(Collectors.joining());
        return cleanString.equals(new StringBuilder(cleanString).reverse().toString());
    }
}
