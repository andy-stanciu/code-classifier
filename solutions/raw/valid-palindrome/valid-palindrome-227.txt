class Solution {

    public boolean isPalindrome(String s) {
        String result = s.trim().toLowerCase().replaceAll("[^a-zA-Z0-9]", "").replaceAll("\\\\s+", "");
        String reversedResult = new StringBuilder(result).reverse().toString();
        if (result.equals(reversedResult)) {
            return true;
        } else {
            return false;
        }
    }
}
