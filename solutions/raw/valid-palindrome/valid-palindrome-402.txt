class Solution {

    public boolean isPalindrome(String s) {
        // convert to lowercase & check alphanumeric by replacing non-alphanumeric with empty string
        String checkString = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        //initialize pointers to compare letters from both ends
        int left = 0;
        int right = checkString.length() - 1;
        // Prevent OutOfBound and check that new string characters are same on both ends
        while (left < right) {
            //once a difference is found, returns false
            if (checkString.charAt(left) != checkString.charAt(right)) {
                return false;
            }
            // increment points to go through all characters of the string
            left++;
            right--;
        }
        // return true once loop is done and no difference was found
        return true;
    }
}
