class Solution {

    public boolean isPalindrome(String s) {
        String nov = s.trim().toLowerCase().chars().mapToObj(c -> String.valueOf((char) c)).filter(c -> Character.isLetterOrDigit(c.charAt(0))).collect(Collectors.joining());
        if (new StringBuilder(nov).reverse().toString().equalsIgnoreCase(nov)) {
            return true;
        }
        return false;
    }
}
