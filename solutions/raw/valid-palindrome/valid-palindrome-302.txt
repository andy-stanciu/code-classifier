class Solution {

    public boolean isPalindrome(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        s = s.toLowerCase();
        String temp = new String();
        temp = s;
        String rev = new String();
        for (int i = 0; i < s.length(); i++) {
            rev = s.charAt(i) + rev;
        }
        if (rev.equals(temp)) {
            return true;
        } else
            return false;
    }
}
