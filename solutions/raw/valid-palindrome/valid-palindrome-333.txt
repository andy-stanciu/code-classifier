class Solution {

    public boolean isPalindrome(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        if (s.length() == 0 || s.length() == 1)
            return true;
        int x = 0;
        int len = s.length();
        boolean res = false;
        while (x < len) {
            if (s.charAt(x) == s.charAt(len - x - 1))
                res = true;
            else
                return false;
            x++;
        }
        return res;
    }
}
