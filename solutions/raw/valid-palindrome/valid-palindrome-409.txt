class Solution {

    public boolean isPalindrome(String s) {
        s = s.replaceAll("\\\\W|[_]", "").toLowerCase();
        char[] strAfter = s.toCharArray();
        int i = 0, j = strAfter.length - 1;
        while (i < j) {
            if (strAfter[i] != strAfter[j]) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
