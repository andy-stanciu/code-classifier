class Solution {

    public boolean isPalindrome(String s) {
        // Step 1: Convert the string to lowercase and remove non-alphanumeric characters
        StringBuilder cleanedString = new StringBuilder();
        for (char c : s.toCharArray()) {
            if (Character.isLetterOrDigit(c)) {
                cleanedString.append(Character.toLowerCase(c));
            }
        }
        // Step 2: Check if the cleaned string is a palindrome
        String filtered = cleanedString.toString();
        String reversed = cleanedString.reverse().toString();
        return filtered.equals(reversed);
    }
}
