class Solution {

    public boolean isPalindrome(String s) {
        s = s.toLowerCase();
        String final1 = "";
        String final2;
        boolean result;
        //removing everything except any letters or digit
        for (int i = 0; i < s.length(); i++) {
            if (Character.isLetterOrDigit(s.charAt(i))) {
                final1 += s.charAt(i);
            }
        }
        StringBuilder stringBuilder = new StringBuilder(final1);
        stringBuilder.reverse();
        final2 = stringBuilder.toString();
        result = final1.equals(final2);
        return result;
    }
}
