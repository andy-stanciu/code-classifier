class Solution {

    public boolean isPalindrome(String s) {
        int left = 0;
        int right = s.length() - 1;
        while (left < right) {
            char cLeft = Character.toLowerCase(s.charAt(left));
            char cRight = Character.toLowerCase(s.charAt(right));
            //check if cLeft is non-alphabetic, if yes, increase left
            if (!Character.isLetterOrDigit(cLeft)) {
                left++;
                continue;
            }
            //check if cRight is non-alphabetic, if yes, decrease right
            if (!Character.isLetterOrDigit(cRight)) {
                right--;
                continue;
            }
            //check if they are equal, if not return False
            if (cLeft != cRight) {
                return false;
            }
            //modify left and right variables
            left++;
            right--;
        }
        return true;
    }
}
