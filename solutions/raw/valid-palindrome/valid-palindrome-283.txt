class Solution {

    public boolean isPalindrome(String s) {
        if (s.trim().isEmpty())
            return true;
        String result = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        char[] nameArr = result.toCharArray();
        int start = 0;
        int end = nameArr.length - 1;
        while (start < end) {
            if (nameArr[start] != nameArr[end]) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
