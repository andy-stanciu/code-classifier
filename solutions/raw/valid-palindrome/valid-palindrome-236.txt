class Solution {

    public boolean isPalindrome(String s) {
        char[] chArr = s.replaceAll("[\\\\s\\\\p{Punct}]", "").toLowerCase().toCharArray();
        int left = 0;
        int right = chArr.length - 1;
        while (left < right) {
            if (chArr[left] != chArr[right]) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
