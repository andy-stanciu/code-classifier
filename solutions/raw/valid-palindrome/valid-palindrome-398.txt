class Solution {

    public boolean isPalindrome(String s) {
        // # Step 1: Convert to lowercase
        String txt1 = s.toLowerCase();
        // # Step 2: Remove non-alphanumeric characters
        String txt2 = txt1.replaceAll("[^a-zA-Z0-9]", "");
        // # Step 3: Initialize two pointers
        int left = 0;
        int right = txt2.length() - 1;
        // # Step 4: Check characters from both ends
        while (left < right) {
            if (txt2.charAt(left) == txt2.charAt(right)) {
                left = left + 1;
                right = right - 1;
            } else {
                return false;
            }
        }
        // # Step 5: If no mismatches, return true
        return true;
    }
}
