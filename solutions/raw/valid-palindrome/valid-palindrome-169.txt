class Solution {

    public boolean isPalindrome(String s) {
        int start = 0;
        int end = s.length() - 1;
        while (start < end) {
            if (!isChar(s.charAt(start)) && !isNumber(s.charAt(start))) {
                start++;
                continue;
            }
            if (!isChar(s.charAt(end)) && !isNumber(s.charAt(end))) {
                end--;
                continue;
            }
            if (!isCharEqual(s.charAt(start), s.charAt(end))) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }

    private boolean isCharEqual(char a, char b) {
        if (isNumber(a) || isNumber(b)) {
            return a == b;
        }
        int result = a - b;
        if (result == -32 || result == 32 || result == 0) {
            return true;
        }
        return false;
    }

    private boolean isChar(char a) {
        return (a >= 65 && a <= 90) || (a >= 97 && a <= 122);
    }

    private boolean isNumber(char a) {
        return (a >= 48 && a <= 57);
    }
}
