class Solution {

    public boolean isPalindrome(String s) {
        int i = 0;
        int j = s.length() - 1;
        if (s.length() == 1) {
            return true;
        }
        boolean isPallindrome = false;
        while (i < j) {
            char x = s.charAt(i);
            char y = s.charAt(j);
            if (!Character.isLetterOrDigit(x)) {
                i++;
            } else if (!Character.isLetterOrDigit(y)) {
                j--;
            } else {
                if (Character.toLowerCase(x) == (Character.toLowerCase(y))) {
                    i++;
                    j--;
                } else {
                    return false;
                }
            }
        }
        return true;
    }
}
