class Solution {

    public boolean isPalindrome(String s) {
        //replaceAll("[^a-z0-9]", ""): Removes all characters that are not lowercase letters (a-z) or digits (0-9).
        s = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        int n = s.length();
        // in this we are only going to the half part of the string
        for (int i = 0; i < n / 2; i++) {
            if (s.charAt(i) != s.charAt(n - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
