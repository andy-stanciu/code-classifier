class Solution {

    public boolean isPalindrome(String s) {
        String original = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        if (original.isEmpty()) {
            return true;
        } else {
            StringBuilder SB = new StringBuilder(original);
            String reversedText = SB.reverse().toString();
            if (original.equals(reversedText)) {
                return true;
            } else {
                return false;
            }
        }
    }
}
