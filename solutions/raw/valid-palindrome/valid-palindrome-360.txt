class Solution {

    public boolean isPalindrome(String s) {
        // first converting the s to lower cases and then by using replace all it means it will not not contain alpha-numeric data.
        s = s.toLowerCase().replaceAll("[^A-Za-z0-9]", "");
        int i = 0;
        int j = s.length() - 1;
        while (i <= j) {
            if (s.charAt(i) != s.charAt(j)) {
                // checking if not equals
                return false;
            } else {
                // increment of i pointer
                i++;
                // decrement of j pointer
                j--;
            }
        }
        return true;
    }
}
