class Solution {

    public boolean isPalindrome(String s) {
        // Clean the string by removing non-alphanumeric characters and converting to lowercase
        String str = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        // If the string is empty after cleaning, it\'s considered a palindrome
        if (str.length() == 0) {
            return true;
        }
        // Use StringBuilder to reverse the cleaned string
        StringBuilder sb = new StringBuilder(str);
        sb.reverse();
        // Compare the reversed string with the original cleaned string
        return sb.toString().equals(str);
    }
}
