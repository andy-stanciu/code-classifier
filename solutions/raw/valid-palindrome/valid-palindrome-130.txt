class Solution {

    public boolean isPalindrome(String s) {
        if (s.length() == 0 || s.length() == 1) {
            return true;
        }
        char[] chars = s.toCharArray();
        int i = 0, j = chars.length - 1;
        while (i <= j) {
            char c = chars[i];
            char cPrime = chars[j];
            //65-90 (large), 97-122 (lowers)
            if (c >= 65 && c <= 90) {
                // convert to lower
                c += 32;
            }
            if (cPrime >= 65 && cPrime <= 90) {
                cPrime += 32;
            }
            // now ignore invalids
            if (c < 48 || (c > 57 && c < 97) || c > 122) {
                i++;
                continue;
            }
            if (cPrime < 48 || (cPrime > 57 && cPrime < 97) || cPrime > 122) {
                j--;
                continue;
            }
            // actual compare
            if (c != cPrime) {
                return false;
            } else {
                i++;
                j--;
            }
        }
        return true;
    }
}
