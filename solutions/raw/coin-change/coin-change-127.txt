class Solution {

    public int coinChange(int[] coins, int amount) {
        if (amount == 0) {
            return 0;
        }
        int[] coins_table = new int[amount + 1];
        coins_table[0] = 0;
        for (int index = 1; index <= amount; ++index) {
            int current_coin = amount + 1;
            for (int coin_index = 0; coin_index < coins.length; ++coin_index) {
                if (coins[coin_index] <= index) {
                    current_coin = Math.min(current_coin, 1 + coins_table[index - coins[coin_index]]);
                }
            }
            coins_table[index] = current_coin;
        }
        return coins_table[amount] > amount ? -1 : coins_table[amount];
    }
}
