class Solution {

    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int[][] arr = new int[n + 1][amount + 1];
        for (int i = 0; i <= amount; i++) {
            arr[0][i] = -1;
        }
        for (int i = 0; i <= n; i++) {
            arr[i][0] = 0;
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= amount; j++) {
                int a1 = arr[i - 1][j];
                int a2 = -1;
                if (j - coins[i - 1] >= 0) {
                    a2 = (arr[i][j - coins[i - 1]] == -1) ? -1 : arr[i][j - coins[i - 1]] + 1;
                }
                if (a1 == -1 && a2 == -1) {
                    arr[i][j] = -1;
                } else if (a1 == -1) {
                    arr[i][j] = a2;
                } else if (a2 == -1) {
                    arr[i][j] = a1;
                } else
                    arr[i][j] = Math.min(a1, a2);
            }
        }
        return arr[n][amount];
    }
}
