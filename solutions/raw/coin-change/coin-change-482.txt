class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount + 1];
        // fill array with value (amount + 1)
        Arrays.fill(dp, 1, dp.length, amount + 1);
        for (int i = 1; i <= amount; i++) {
            // check from 1 to amount if possible arrangement of coin is possible
            for (int coin : coins) {
                if (coin <= i) {
                    dp[i] = Math.min(dp[i], dp[i - coin] + 1);
                }
            }
        }
        if (dp[amount] == amount + 1) {
            // if final value is still as that was initialised, it means no such arrangement is possible
            return -1;
        }
        return dp[amount];
    }
}
