class Solution {

    private static final int NO_CHANGE = Integer.MAX_VALUE;

    public int coinChange(int[] coins, int amount) {
        int[] change = new int[amount + 1];
        Arrays.fill(change, NO_CHANGE);
        change[0] = 0;
        for (int currAmt = 0; currAmt <= amount; currAmt++) {
            for (int coin : coins) {
                // checking if there is any solution for currAmt - coin.
                // If so then only take minimum.
                if (coin <= currAmt && change[currAmt - coin] != NO_CHANGE)
                    change[currAmt] = Math.min(change[currAmt], change[currAmt - coin] + 1);
            }
        }
        return (change[amount] == NO_CHANGE) ? -1 : change[amount];
    }
}
