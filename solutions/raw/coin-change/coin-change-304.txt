class Solution {

    public int coinChange(int[] coins, int value, int index) {
        if (index < 0 || value < 0)
            return Integer.MAX_VALUE;
        if (value == 0)
            return 0;
        int firstWay = coinChange(coins, value - coins[index], index);
        int secondWay = coinChange(coins, value, index - 1);
        if (firstWay < secondWay) {
            return 1 + firstWay;
        }
        return secondWay;
    }

    public int coinChange(int[] coins, int value) {
        int result = coinChange(coins, value, coins.length - 1);
        return result == Integer.MAX_VALUE ? -1 : result;
    }
}
