class Solution {

    public int coinChange(int[] coins, int amount) {
        int[][] t = new int[coins.length][amount + 1];
        for (int[] i : t) Arrays.fill(i, (int) Math.pow(10, 9));
        for (int i = 0; i < amount + 1; i++) if (i % coins[0] == 0)
            t[0][i] = i / coins[0];
        for (int i = 1; i < coins.length; i++) {
            for (int j = 0; j <= amount; j++) {
                int taken = (int) Math.pow(10, 9);
                if (coins[i] <= j)
                    taken = 1 + t[i][j - coins[i]];
                t[i][j] = Math.min(taken, t[i - 1][j]);
            }
        }
        if (t[coins.length - 1][amount] == (int) Math.pow(10, 9))
            return -1;
        return t[coins.length - 1][amount];
    }
}
