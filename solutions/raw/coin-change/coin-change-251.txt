class Solution {

    public int coinChange(int[] coins, int amount) {
        if (amount < 0)
            return 0;
        int[] minCoinsCount = new int[amount + 1];
        for (int i = 1; i <= amount; i++) {
            minCoinsCount[i] = Integer.MAX_VALUE;
            for (int coin : coins) {
                if (coin <= i && minCoinsCount[i - coin] != Integer.MAX_VALUE)
                    minCoinsCount[i] = Math.min(minCoinsCount[i], 1 + minCoinsCount[i - coin]);
            }
        }
        return minCoinsCount[amount] != Integer.MAX_VALUE ? minCoinsCount[amount] : -1;
    }
}
