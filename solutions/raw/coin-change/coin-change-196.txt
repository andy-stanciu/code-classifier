class Solution {

    public int coinChange(int[] coins, int amount) {
        Arrays.sort(coins);
        int n = coins.length;
        int[] coinsReq = new int[amount + 1];
        for (int i = 1; i <= amount; i++) {
            boolean canMake = false;
            int minCoinReq = Integer.MAX_VALUE;
            for (int j = 0; j < coins.length; j++) {
                if (coins[j] <= i) {
                    int coinReq = i - coins[j];
                    if (coinsReq[coinReq] != -1) {
                        minCoinReq = Math.min(minCoinReq, coinsReq[coinReq] + 1);
                    }
                }
            }
            coinsReq[i] = minCoinReq != Integer.MAX_VALUE ? minCoinReq : -1;
        }
        return coinsReq[amount];
    }
}
