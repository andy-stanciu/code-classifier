class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] coinChange = new int[amount + 1];
        for (int currentAmount = 1; currentAmount <= amount; currentAmount++) {
            int minCoinChangeNeeded = Integer.MAX_VALUE;
            for (int j = 0; j < coins.length; j++) {
                if (currentAmount >= coins[j] && coinChange[currentAmount - coins[j]] != -1) {
                    minCoinChangeNeeded = Math.min(minCoinChangeNeeded, coinChange[currentAmount - coins[j]]);
                }
            }
            coinChange[currentAmount] = minCoinChangeNeeded == Integer.MAX_VALUE ? -1 : minCoinChangeNeeded + 1;
        }
        return coinChange[amount];
    }
}
