class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] t = new int[amount + 1];
        Arrays.fill(t, (int) Math.pow(10, 9));
        for (int i = 0; i < amount + 1; i++) if (i % coins[0] == 0)
            t[i] = i / coins[0];
        for (int i = 1; i < coins.length; i++) {
            int[] temp = new int[amount + 1];
            for (int j = 0; j <= amount; j++) {
                int taken = (int) Math.pow(10, 9);
                if (coins[i] <= j)
                    taken = 1 + temp[j - coins[i]];
                temp[j] = Math.min(taken, t[j]);
            }
            t = temp;
        }
        if (t[amount] == (int) Math.pow(10, 9))
            return -1;
        return t[amount];
    }
}
