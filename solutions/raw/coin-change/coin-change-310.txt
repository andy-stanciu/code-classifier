class Solution {

    public int coinChange(int[] coins, int amount) {
        int l = coins.length;
        int[][] hash = new int[l + 1][amount + 1];
        for (int i = 0; i < l + 1; i++) {
            for (int j = 0; j < amount + 1; j++) {
                if (i == 0)
                    hash[i][j] = Integer.MAX_VALUE;
                if (j == 0)
                    hash[i][j] = 0;
            }
        }
        for (int i = 1; i < l + 1; i++) {
            for (int j = 1; j < amount + 1; j++) {
                if (coins[i - 1] <= j) {
                    if (hash[i][j - coins[i - 1]] == Integer.MAX_VALUE) {
                        hash[i][j] = hash[i - 1][j];
                    } else {
                        hash[i][j] = Math.min(hash[i - 1][j], 1 + hash[i][j - coins[i - 1]]);
                    }
                } else {
                    hash[i][j] = hash[i - 1][j];
                }
            }
        }
        if (hash[l][amount] == Integer.MAX_VALUE)
            return -1;
        return hash[l][amount];
    }
}
