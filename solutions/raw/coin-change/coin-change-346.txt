class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] coinsNeededForAmount = new int[amount + 1];
        Arrays.fill(coinsNeededForAmount, amount + 1);
        coinsNeededForAmount[0] = 0;
        for (int a = 1; a <= amount; a++) {
            for (int c : coins) {
                if (c > a) {
                    continue;
                }
                coinsNeededForAmount[a] = Math.min(coinsNeededForAmount[a], coinsNeededForAmount[a - c] + 1);
            }
        }
        return coinsNeededForAmount[amount] > amount ? -1 : coinsNeededForAmount[amount];
    }
}
