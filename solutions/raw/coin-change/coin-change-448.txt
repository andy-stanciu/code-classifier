class Solution {

    public int coinChange(int[] coins, int amount) {
        return getCoinChange(coins, amount, 0, 0);
    }

    public int getCoinChange(int[] coins, int amount, int i, int coinsCount) {
        if (amount == 0)
            return coinsCount;
        if (amount < 0 || i == coins.length)
            return -1;
        int notTake = getCoinChange(coins, amount, i + 1, coinsCount);
        int take = getCoinChange(coins, amount - coins[i], i, coinsCount + 1);
        if (notTake == -1)
            return take;
        else if (take == -1)
            return notTake;
        return Math.min(take, notTake);
    }
}
