class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] t = new int[amount + 1];
        Arrays.fill(t, amount + 1);
        t[0] = 0;
        for (int i = 1; i <= amount; i++) {
            for (int coin : coins) {
                if (i >= coin) {
                    t[i] = Math.min(t[i], 1 + t[i - coin]);
                }
            }
        }
        return t[amount] > amount ? -1 : t[amount];
    }
}
