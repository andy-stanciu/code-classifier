class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] t = new int[amount + 1];
        Arrays.fill(t, amount + 1);
        t[0] = 0;
        for (int i = 1; i <= amount; ++i) {
            for (int c : coins) {
                if (i - c >= 0) {
                    t[i] = Math.min(t[i], 1 + t[i - c]);
                }
            }
        }
        return (t[amount] == amount + 1) ? -1 : t[amount];
    }
}
