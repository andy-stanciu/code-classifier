class Solution {

    // Memory array defined...
    public int[] memory;

    public int minCoin = Integer.MAX_VALUE;

    public int coinChange(int[] coins, int amount) {
        // Initializing and filling the Memory array...
        memory = new int[amount + 1];
        Arrays.fill(memory, -1);
        // Function call...
        int result = Memonize(coins, amount);
        return result == Integer.MAX_VALUE ? -1 : result;
    }

    public void Recurse(int[] coins, int amount, int current, int coinsCount) {
        // Brute-Force Recursion...
        // If the amount exceeds, the required amount...
        if (current > amount)
            return;
        if (current == amount) {
            // If the amount equals...
            // Update the number of coins required...
            minCoin = Math.min(minCoin, coinsCount);
            return;
        }
        for (// For every unique coin...
        // For every unique coin...
        // Recursively branch the tree...
        int coin : coins) Recurse(coins, amount, current + coin, coinsCount + 1);
        return;
    }

    public int Memonize(int[] coins, int current) {
        // Memonization (much faster approach)...
        // If problem is feasible...
        if (current == 0)
            return 0;
        // If the problem is not feasible...
        if (current < 0)
            return Integer.MAX_VALUE;
        if (// When the subproblem is already computed...
        memory[current] != -1)
            return memory[current];
        // Tracking the minimum number of coins...
        int minCoins = Integer.MAX_VALUE;
        for (int coin : coins) {
            // For every unqiue coin...
            // Branch the Recursion Tree...
            int subProblem = Memonize(coins, current - coin);
            if (// If subproblem is feasible...
            subProblem != Integer.MAX_VALUE)
                // Update the coins count...
                minCoins = Math.min(minCoins, 1 + subProblem);
        }
        // Update the Memory and backtrack...
        return memory[current] = minCoins;
    }
}
