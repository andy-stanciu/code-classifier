class Solution {

    public int coinChange(int[] coins, int amount) {
        if (amount == 0)
            return 0;
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, -1);
        for (int i = 1; i <= amount; i++) {
            for (int coin : coins) {
                if (coin == i) {
                    dp[i] = 1;
                    break;
                } else if (coin < i && dp[i - coin] > 0) {
                    if (dp[i] > 0) {
                        dp[i] = Math.min(dp[i], dp[i - coin] + 1);
                    } else {
                        dp[i] = dp[i - coin] + 1;
                    }
                }
            }
        }
        return dp[amount];
    }
}
