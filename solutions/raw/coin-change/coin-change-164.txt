class Solution {

    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int[][] dp = new int[n + 1][amount + 1];
        //To avoid overflow issues, you should use Integer.MAX_VALUE -        1 or another large value that won\u2019t cause overflow      when         performing arithmetic operations.
        for (int i = 0; i <= amount; i++) {
            dp[0][i] = Integer.MAX_VALUE - 1;
        }
        dp[0][0] = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= amount; j++) {
                dp[i][j] = dp[i - 1][j];
                if (coins[i - 1] <= j) {
                    dp[i][j] = Math.min(dp[i][j], dp[i][j - coins[i - 1]] + 1);
                }
            }
        }
        return (dp[n][amount] == Integer.MAX_VALUE - 1) ? -1 : dp[n][amount];
    }
}
