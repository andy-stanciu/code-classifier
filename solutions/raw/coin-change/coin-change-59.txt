class Solution {

    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount + 1];
        coinChange(coins, amount, dp);
        return dp[amount] == dp.length ? -1 : dp[amount];
    }

    public int coinChange(int[] coins, int amount, int[] dp) {
        if (amount == 0) {
            return 0;
        }
        if (dp[amount] != 0) {
            return dp[amount];
        }
        dp[amount] = dp.length;
        for (int coin : coins) {
            if (amount >= coin) {
                dp[amount] = Math.min(dp[amount], coinChange(coins, amount - coin, dp) + 1);
            }
        }
        return dp[amount];
    }
}
