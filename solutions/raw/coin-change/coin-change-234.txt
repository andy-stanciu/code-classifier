class Solution {

    public int coinChange(int[] coins, int amount) {
        return dfs(coins, amount);
    }

    //brute force solution
    int dfs(int[] coins, int amount, int sum, int steps) {
        if (sum == amount) {
            return steps;
        }
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < coins.length; i++) {
            if (sum + coins[i] <= amount && sum + coins[i] >= 0) {
                min = Math.min(dfs(coins, amount, sum + coins[i], steps + 1), min);
            }
        }
        if (min == Integer.MAX_VALUE) {
            return -1;
        }
        return min;
    }
}
