class Solution {

    public int coinChange(int[] coins, int amount) {
        Arrays.sort(coins);
        int[] minCoins = new int[amount + 1];
        Arrays.fill(minCoins, amount + 1);
        minCoins[0] = 0;
        for (int subAmount = 1; subAmount <= amount; subAmount++) {
            {
                for (int coin : coins) {
                    if (coin > subAmount)
                        break;
                    minCoins[subAmount] = Math.min(minCoins[subAmount], 1 + minCoins[subAmount - coin]);
                }
            }
        }
        return minCoins[amount] > amount ? -1 : minCoins[amount];
    }
}
