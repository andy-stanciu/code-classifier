class Solution {

    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int[][] dp = new int[n][amount + 1];
        for (int[] i : dp) {
            Arrays.fill(i, -1);
        }
        int res = minimumChange(n - 1, amount, coins, dp);
        if (res >= (int) Math.pow(10, 9))
            return -1;
        return res;
    }

    static int minimumChange(int ind, int target, int[] arr, int[][] dp) {
        if (ind == 0) {
            if (target % arr[0] == 0)
                return target / arr[ind];
            else
                return (int) Math.pow(10, 9);
        }
        if (dp[ind][target] != -1)
            return dp[ind][target];
        int nTake = 0 + minimumChange(ind - 1, target, arr, dp);
        int take = (int) Math.pow(10, 9);
        if (arr[ind] <= target) {
            take = 1 + minimumChange(ind, target - arr[ind], arr, dp);
        }
        return dp[ind][target] = Math.min(take, nTake);
    }
}
