class Solution {

    public int coinChange(int[] coins, int amount) {
        if (amount < 1)
            return 0;
        int[] minimumCoinDP = new int[amount + 1];
        for (int i = 1; i <= amount; i++) {
            minimumCoinDP[i] = Integer.MAX_VALUE;
            for (int coin : coins) {
                if (coin <= i && minimumCoinDP[i - coin] != Integer.MAX_VALUE) {
                    minimumCoinDP[i] = Math.min(minimumCoinDP[i], 1 + minimumCoinDP[i - coin]);
                }
            }
        }
        if (minimumCoinDP[amount] == Integer.MAX_VALUE)
            return -1;
        return minimumCoinDP[amount];
    }
}
