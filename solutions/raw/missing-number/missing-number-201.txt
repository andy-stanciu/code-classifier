class Solution {

    public int missingNumber(int[] nums) {
        int range = nums.length;
        //make starting  sum= 0
        int numsSum = 0;
        //make staring length 0
        int rangeSum = 0;
        //loop to update the value of rangeSum
        for (int i = 0; i <= range; i++) {
            rangeSum += i;
        }
        //for loop for  each element in array and get sum
        for (int num : nums) {
            numsSum += num;
        }
        //implimenting formula
        int missingNo = rangeSum - numsSum;
        return missingNo;
    }
}
