class Solution {

    public int missingNumber(int[] nums) {
        //make a frequency array of length (10^4 +1) coz the constraints are gien 1<=n<=10^4
        int[] f = new int[10001];
        //for every element in the array, increment its count in the freq array
        for (int i = 0; i < nums.length; i++) {
            f[nums[i]]++;
        }
        for (int i = 0; i <= 10001; i++) {
            //loop through the freq array and check which no wasnt present in the array
            if (f[i] == 0)
                return i;
        }
        return -1;
    }
}
