class Solution {

    public int missingNumber(int[] nums) {
        Arrays.sort(nums);
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (i == nums[j]) {
                    count++;
                }
            }
            if (count != 0) {
                count = 0;
            } else {
                return i;
            }
        }
        return nums.length;
    }
}
