class Solution {

    public int missingNumber(int[] nums) {
        int maxNumber = nums.length;
        int minNumber = nums[0];
        boolean check = false;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] < minNumber) {
                minNumber = nums[i];
            }
        }
        int i = minNumber;
        int result = 0;
        outer: while (i < maxNumber) {
            i++;
            inner: for (int j = 0; j < nums.length; j++) {
                if (nums[j] == i) {
                    check = true;
                    break inner;
                }
            }
            if (check == false) {
                result = i;
                break;
            }
            check = false;
        }
        return result;
    }
}
