class Solution {

    public int missingNumber(int[] nums) {
        int total1 = 0;
        int total2 = 0;
        for (int i = 0; i < nums.length; i++) {
            total1 += nums[i];
        }
        for (int i = 0; i < nums.length + 1; i++) {
            total2 += i;
        }
        if (total1 == total2)
            return 0;
        return total2 - total1;
    }
}
