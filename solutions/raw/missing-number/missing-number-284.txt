class Solution {

    public int missingNumber(int[] nums) {
        // Get the length of the array
        int n = nums.length;
        // Calculate the expected sum from 0 to n
        int sum = n * (n + 1) / 2;
        // Initialize the actual sum
        int sum2 = 0;
        for (int i = 0; i < nums.length; i++) {
            // Calculate the actual sum
            sum2 = sum2 + nums[i];
        }
        // Return the missing number
        return sum - sum2;
    }
}
