class Solution {

    public int missingNumber(int[] nums) {
        int len = nums.length;
        // Expected sum from 0 to n
        int count = len * (len + 1) / 2;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            // Sum of elements in the array
            sum += nums[i];
        }
        // Missing number is the difference
        return count - sum;
    }
}
