class Solution {

    public int missingNumber(int[] nums) {
        int len = nums.length;
        int number = 0;
        for (int i = 0; i < len; i++) {
            number ^= nums[i];
        }
        for (int i = 0; i <= len; i++) {
            number ^= i;
        }
        return number;
    }
}
