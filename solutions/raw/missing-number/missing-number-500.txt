class Solution {

    public int missingNumber(int[] nums) {
        if (nums.length == 0)
            return 0;
        int n = nums.length;
        int sum = n * (n + 1) / 2;
        int counter = 0;
        for (int num : nums) {
            counter += num;
        }
        return sum - counter;
    }
}
