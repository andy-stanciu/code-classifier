class Solution {

    public int missingNumber(int[] nums) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            hm.put(nums[i], hm.getOrDefault(nums[i], 0) + 1);
        }
        for (int i = 0; i <= n; i++) {
            if (!hm.containsKey(i)) {
                return i;
            }
        }
        return 0;
    }
}
