class Solution {

    public int missingNumber(int[] nums) {
        int n = nums.length;
        int sum = 0;
        int tsum = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        for (int i = 0; i < n + 1; i++) {
            tsum += i;
        }
        return tsum - sum;
    }
}
