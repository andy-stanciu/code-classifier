class Solution {

    public int returnMissingNumber(int[] result, int index) {
        if (result[index] == -1) {
            return index;
        }
        return returnMissingNumber(result, index + 1);
    }

    public int missingNumberHelper(int[] nums, int[] result, int size, int index) {
        if (index >= size) {
            return returnMissingNumber(result, 0);
        }
        result[nums[index]] = nums[index];
        return missingNumberHelper(nums, result, size, index + 1);
    }

    public int missingNumber(int[] nums) {
        int size = nums.length;
        int[] result = new int[size + 1];
        Arrays.fill(result, -1);
        return missingNumberHelper(nums, result, size, 0);
    }
}
