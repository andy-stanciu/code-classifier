class Solution {

    public int missingNumber(int[] nums) {
        int sumValues = 0;
        int sumIndexes = 0;
        for (int i = 0; i < nums.length; i++) {
            sumValues += nums[i];
            sumIndexes += i;
        }
        sumIndexes += nums.length;
        return sumIndexes - sumValues;
    }
}
