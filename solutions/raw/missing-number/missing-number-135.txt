class Solution {

    public int missingNumber(int[] nums) {
        int n = nums.length;
        boolean[] present = new boolean[n + 1];
        for (int i = 0; i < n; i++) {
            present[nums[i]] = true;
        }
        int ans = 0;
        for (int i = 0; i < present.length; i++) {
            if (present[i] == false) {
                ans = i;
                break;
            }
        }
        return ans;
    }
}
