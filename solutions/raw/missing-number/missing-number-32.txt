class Solution {

    public int missingNumber(int[] nums) {
        if (nums.length == 1) {
            if (nums[0] == 1) {
                return 0;
            } else {
                return 1;
            }
        }
        //finding the max value and create a boolean array
        int max = 0;
        for (int i : nums) {
            max = (max < i) ? i : max;
        }
        boolean[] arr = new boolean[max + 1];
        //if it is present in the array then mark true for easy searching
        for (int i : nums) {
            arr[i] = true;
        }
        //Now check tha boolean array if u get false that means that element is absent in the given array so retrn that element.
        for (int i = 0; i <= max; i++) {
            if (arr[i] != true) {
                return i;
            }
        }
        return max + 1;
    }
}
