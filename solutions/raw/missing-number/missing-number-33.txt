class Solution {

    public int missingNumber(int[] nums) {
        // This is simple and easy solution but time and space complexity is not much gud.
        // Arrays.sort(nums);
        // for(int i = 0; i < nums.length; i++){
        //     if(nums[i] != i) return i;
        // }
        // return nums.length;
        // By using HashMap
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            map.put(nums[i], nums[i]);
        }
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(i)) {
                continue;
            } else {
                return i;
            }
        }
        return nums.length;
    }
}
