class Solution {

    public int missingNumber(int[] nums) {
        int sum = 0, N = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            //Taking all the sums of the array
            if (nums[i] != n) {
                sum += nums[i];
            } else //Checking if n is present in the array
            {
                N = 1;
            }
        }
        int indx_sum;
        int ans = 0;
        //taking all the sum of indexes from 0 to n - 1
        indx_sum = (((n - 1) * (n)) / 2);
        //checking if index sum - array\'s sum gives me the       number in the range [0, n]
        if ((indx_sum - sum) != 0) {
            ans = indx_sum - sum;
        } else //if it is 0 then there might be two conditions
        //which is n is missing or 0 is missing
        {
            if (N == 1) {
                ans = 0;
            } else {
                ans = n;
            }
        }
        return ans;
    }
}
