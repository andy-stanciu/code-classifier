class Solution {

    public int missingNumber(int[] nums) {
        HashSet<Integer> s = new HashSet<>();
        int cp = 0;
        for (int i = 0; i < nums.length; i++) {
            s.add(nums[i]);
        }
        for (int i = 0; i <= nums.length; i++) {
            if (s.contains(i))
                cp++;
            else
                return i;
        }
        return -1;
    }
}
