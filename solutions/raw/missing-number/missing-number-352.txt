class Solution {

    public int missingNumber(int[] nums) {
        int n = nums.length;
        int add = 0;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            add += nums[i];
        }
        for (int i = 0; i <= n; i++) {
            sum += i;
        }
        return sum - add;
    }
}
