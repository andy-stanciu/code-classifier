class Solution {

    public int missingNumber(int[] nums) {
        int n = nums.length;
        int n1 = 0;
        int n2 = 0;
        int ans = 0;
        n2 = (n * (n + 1)) / 2;
        for (int i = 0; i < n; i++) {
            n1 += nums[i];
            ans = n2 - n1;
        }
        return ans;
    }
}
