class Solution {

    public int missingNumber(int[] nums) {
        if (nums.length == 1)
            return nums[0] == 0 ? 1 : 0;
        Arrays.sort(nums);
        int start = 0, end = nums.length - 1;
        int number = 0;
        boolean isFound = true;
        while (start <= end) {
            int mid = (start + end) / 2;
            if (number == nums[mid]) {
                number++;
                start = 0;
                end = nums.length - 1;
            } else if (number > nums[mid]) {
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return number;
    }
}
