class Solution {

    public int missingNumber(int[] nums) {
        int N = nums.length;
        int sum_with_missingnum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum_with_missingnum += nums[i];
        }
        int sum_without_missingnum = (nums.length * (nums.length + 1)) / 2;
        int missingNumber = sum_without_missingnum - sum_with_missingnum;
        return missingNumber;
    }
}
