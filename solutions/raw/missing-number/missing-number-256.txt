class Solution {

    public int missingNumber(int[] nums) {
        int sum = 0;
        int max = 0;
        for (int num : nums) {
            if (max < num) {
                max = num;
            }
            sum += num;
        }
        if (max < nums.length) {
            return nums.length;
        }
        int value = (max * (max + 1) / 2);
        return value - sum;
    }
}
