class Solution {

    public int missingNumber(int[] nums) {
        int sum = 0;
        int n = nums.length;
        int sum1 = n * (n + 1) / 2;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        if (sum1 == sum) {
            return 0;
        } else {
            return sum1 - sum;
        }
    }
}
