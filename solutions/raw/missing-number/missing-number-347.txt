class Solution {

    public int missingNumber(int[] nums) {
        // using this as a n term
        int l = nums.length;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        // we will use the mathematical induction formula
        int nsum = (l * (l + 1)) / 2;
        int miss = nsum - sum;
        return miss;
    }
}
