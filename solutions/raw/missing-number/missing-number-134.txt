class Solution {

    public int missingNumber(int[] nums) {
        Arrays.sort(nums);
        int beg = 0, end = nums.length;
        while (beg < end) {
            int mid = (beg + end) / 2;
            if (nums[mid] > mid) {
                end = mid;
            } else {
                beg = mid + 1;
            }
        }
        return end;
    }
}
