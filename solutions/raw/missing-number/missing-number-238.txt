class Solution {

    public int missingNumber(int[] nums) {
        int max = nums[0];
        for (int i = 1; i < nums.length; i++) if (nums[i] > max)
            max = nums[i];
        int[] map = new int[max + 1];
        for (int i = 0; i < nums.length; i++) {
            map[nums[i]]++;
        }
        int i = 0;
        for (i = 0; i < max; i++) {
            if (map[i] == 0)
                return i;
        }
        return (i + 1);
    }
}
