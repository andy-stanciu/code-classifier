class Solution {

    public int missingNumber(int[] nums) {
        int totalsum = 0;
        int sum = 0;
        for (int i = 0; i <= nums.length; i++) {
            totalsum += i;
        }
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        return totalsum - sum;
    }
}
