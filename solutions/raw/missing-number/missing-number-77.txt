class Solution {

    public int missingNumber(int[] nums) {
        //Since the missing number is inside the range of nums length.
        //So the only way to get that number is to xor with length as well.
        int sum = nums.length;
        int i = 0;
        for (i = 0; i < nums.length; i++) {
            sum ^= i ^ nums[i];
        }
        return sum;
    }
}
