class Solution {

    public int missingNumber(int[] nums) {
        int miss = 0;
        int max = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > max) {
                max = nums[i];
            }
        }
        if ((nums.length - 1) == (max)) {
            miss = max + 1;
        } else {
            int sum = 0;
            for (int i = 0; i < nums.length; i++) {
                sum = sum + nums[i];
            }
            int actualSum = max * ((nums.length + 1) / 2);
            if (actualSum == 2) {
                actualSum++;
            }
            miss = actualSum - sum;
        }
        return miss;
    }
}
