class Solution {

    public int missingNumber(int[] nums) {
        int n = nums.length;
        int i = 0;
        // Cyclic sort
        while (i < n) {
            if (nums[i] < n && nums[i] != nums[nums[i]]) {
                // Swap nums[i] with nums[nums[i]]
                int temp = nums[i];
                nums[i] = nums[nums[i]];
                nums[temp] = temp;
            } else {
                i++;
            }
        }
        // Find the missing number
        for (int j = 0; j < n; j++) {
            if (nums[j] != j) {
                return j;
            }
        }
        return n;
    }
}
