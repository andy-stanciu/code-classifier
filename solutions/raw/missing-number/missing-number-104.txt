class Solution {

    public int missingNumber(int[] nums) {
        int[] numbers = new int[nums.length];
        for (int a : nums) {
            if (a <= nums.length && a > 0)
                numbers[a - 1]++;
        }
        for (int b = 0; b < numbers.length; b++) {
            if (numbers[b] == 0)
                return b + 1;
        }
        return 0;
    }
}
