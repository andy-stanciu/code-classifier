class Solution {

    public int missingNumber(int[] nums) {
        int sum = 0;
        int N = nums.length;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int tot = (N * (N + 1)) / 2;
        int res = tot - sum;
        return res;
    }
}
