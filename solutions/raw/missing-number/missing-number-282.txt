class Solution {

    public int missingNumber(int[] nums) {
        int missingNo = 0;
        int n = nums.length;
        int sum2 = 0;
        int sum1 = (n * (n + 1)) / 2;
        for (int i = 0; i <= n - 1; i++) {
            sum2 = sum2 + nums[i];
        }
        return missingNo = sum1 - sum2;
    }
}
