class Solution {

    public int missingNumber(int[] nums) {
        return result(nums);
    }

    int result(int[] arr) {
        int i = 0;
        int sum = 0;
        int sumN = 0;
        for (i = 0; i < arr.length; i++) {
            sum += arr[i];
        }
        for (int j = 1; j <= arr.length; j++) {
            sumN += j;
        }
        return sumN - sum;
    }
}
