class Solution {

    public int missingNumber(int[] nums) {
        int n = nums.length;
        int result = n;
        for (int i = 0; i < n; i++) result ^= (i ^ nums[i]);
        return result;
    }
}
// XOR of all the number and all the indexes will be the xor of (missing no and total length)
// xor that by total length, will be the missing number
// [3,0,1]
// indexes = [0,1,2]
// numbers = [3,0,1]
// xor indexes^numbers = 2^3, which xor with 3 = 2 => ans
