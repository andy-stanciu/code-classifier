class Solution {

    public int missingNumber(int[] nums) {
        int sum = 0;
        // sum the numbers from 1 to n
        for (int i = 1; i <= nums.length; i++) {
            sum += i;
        }
        // subtract the values in nums
        for (int i = 0; i < nums.length; i++) {
            sum = sum - nums[i];
        }
        // the remainder is the missing number
        return sum;
    }
}
