class Solution {

    public int missingNumber(int[] nums) {
        // Ensure the array is sorted
        Arrays.sort(nums);
        int n = nums.length;
        int low = 0;
        int high = n;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (nums[mid] == mid) {
                // Move to the right side
                low = mid + 1;
            } else {
                // Move to the left side
                high = mid;
            }
        }
        // The missing number will be at the \'low\' index
        return low;
    }
}
