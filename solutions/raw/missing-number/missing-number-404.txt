class Solution {

    public int missingNumber(int[] nums) {
        int big_sum = 0, small_sum = 0;
        for (int i = 0; i <= nums.length; i++) {
            big_sum += i;
        }
        for (int i = 0; i < nums.length; i++) {
            small_sum += nums[i];
        }
        return (big_sum - small_sum);
    }
}
