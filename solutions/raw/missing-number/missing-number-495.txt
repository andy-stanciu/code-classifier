class Solution {

    public int missingNumber(int[] nums) {
        int sum1 = 0;
        int sum2 = 0;
        int n = nums.length;
        for (int e : nums) {
            sum1 += e;
        }
        for (int i = 0; i <= n; i++) {
            sum2 += i;
        }
        return sum2 - sum1;
    }
}
