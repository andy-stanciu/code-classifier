class Solution {

    public int missingNumber(int[] nums) {
        Arrays.sort(nums);
        int item = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (i + item != nums[i]) {
                return i + item;
            }
        }
        return Arrays.stream(nums).anyMatch(i -> i == 0) ? nums[nums.length - 1] + 1 : 0;
    }
}
