class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        ListNode out = new ListNode(-1);
        ListNode temp = out;
        ListNode l1 = list1;
        ListNode l2 = list2;
        while (l1 != null && l2 != null) {
            if (l1.val < l2.val) {
                out.next = new ListNode(l1.val);
                l1 = l1.next;
                out = out.next;
            } else if (l1.val > l2.val) {
                out.next = new ListNode(l2.val);
                l2 = l2.next;
                out = out.next;
            } else if (l1.val == l2.val) {
                out.next = new ListNode(l1.val);
                out = out.next;
                out.next = new ListNode(l2.val);
                out = out.next;
                l1 = l1.next;
                l2 = l2.next;
            }
        }
        if (l1 != null)
            out.next = l1;
        if (l2 != null)
            out.next = l2;
        return temp.next;
    }
}
