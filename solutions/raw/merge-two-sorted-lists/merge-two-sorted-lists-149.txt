/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        ListNode temp1 = new ListNode();
        ListNode temp = temp1;
        if (list1 == null) {
            return list2;
        }
        if (list2 == null) {
            return list1;
        }
        if (list1.val < list2.val) {
            temp1.val = list1.val;
            list1 = list1.next;
        } else {
            temp1.val = list2.val;
            list2 = list2.next;
        }
        while (list1 != null && list2 != null) {
            if (list1.val > list2.val) {
                temp1.next = new ListNode();
                temp1 = temp1.next;
                temp1.val = list2.val;
                list2 = list2.next;
            } else {
                temp1.next = new ListNode();
                temp1 = temp1.next;
                temp1.val = list1.val;
                list1 = list1.next;
            }
        }
        if (list1 != null) {
            while (list1 != null) {
                temp1.next = new ListNode();
                temp1 = temp1.next;
                temp1.val = list1.val;
                list1 = list1.next;
            }
        }
        if (list2 != null) {
            while (list2 != null) {
                temp1.next = new ListNode();
                temp1 = temp1.next;
                temp1.val = list2.val;
                list2 = list2.next;
            }
        }
        return temp;
    }
}
