/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
/**
 * Idea: To do a inline merge instead of new list, choose a linkedlist with greater 1st element and keep it as l1 and other as l2
 */
class Solution {

    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if (l1 == null) {
            return l2;
        } else if (l2 == null) {
            return l1;
        }
        ListNode nh;
        if (l1.val <= l2.val) {
            nh = l1;
        } else {
            nh = l1;
            l1 = l2;
            l2 = nh;
            nh = l1;
        }
        ListNode prev = nh;
        while (l1 != null && l2 != null) {
            if (l1.val <= l2.val) {
                prev = l1;
                l1 = l1.next;
            } else {
                ListNode l2tmp = l2.next;
                prev.next = l2;
                l2.next = l1;
                l2 = l2tmp;
                prev = prev.next;
            }
        }
        if (l2 != null) {
            prev.next = l2;
        }
        return nh;
    }
}
