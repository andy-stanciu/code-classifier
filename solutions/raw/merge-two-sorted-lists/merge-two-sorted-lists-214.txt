class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        } else if (list2 != null && list1.val > list2.val) {
            return mergeTwoLists(list2, list1);
        }
        return new ListNode(list1.val, mergeTwoLists(list1.next, list2));
    }
}
