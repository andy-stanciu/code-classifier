class Solution {

    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        ListNode ans;
        if (l1 == null) {
            return l2;
        } else if (l2 == null) {
            return l1;
        }
        if (l1.val > l2.val) {
            ans = l2;
            ans.next = mergeTwoLists(l1, l2.next);
        } else {
            ans = l1;
            ans.next = mergeTwoLists(l1.next, l2);
        }
        return ans;
    }
}
