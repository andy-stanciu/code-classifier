class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        ListNode res = helper(list1, list2);
        return res;
    }

    public ListNode helper(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        } else if (list2 == null) {
            return list1;
        }
        if (list1.val < list2.val) {
            ListNode nextNode = helper(list1.next, list2);
            list1.next = nextNode;
            return list1;
        } else {
            ListNode nextNode = helper(list1, list2.next);
            list2.next = nextNode;
            return list2;
        }
    }
}
