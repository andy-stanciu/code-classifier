class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        //Check if any of the lists is empty...
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        //Create tmp linkedList...
        ListNode tmp = new ListNode();
        //Store start of tmp linkedlist in head node...
        ListNode head = tmp;
        // Loop until any of the list becomes null...
        while (list1 != null && list2 != null) {
            //we compare if the value of list1 is less than the value of list2...
            if (list1.val < list2.val) {
                tmp.next = list1;
                list1 = list1.next;
                //if the value of list1 is equal or greater than the value of list2...
            } else {
                tmp.next = list2;
                list2 = list2.next;
            }
            tmp = tmp.next;
        }
        //if while loop terminated then , either list1 is finished traversing or list2 is finished traversing...
        //if list1 is finished traversing...
        if (list1 != null) {
            tmp.next = list1;
            //if list2 is finished traversing...
        } else {
            tmp.next = list2;
        }
        // return head of the result...
        return head.next;
    }
}
