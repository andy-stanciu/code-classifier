class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode retHeadNode = null;
        ListNode current = null;
        boolean changeNextOnList2 = false;
        if (list1.val < list2.val) {
            current = list1;
            list1 = list1.next;
        } else {
            current = list2;
            list2 = list2.next;
        }
        retHeadNode = current;
        while (true) {
            if (list1 == null) {
                changeNextOnList2 = true;
            } else if (list2 == null) {
                changeNextOnList2 = false;
            } else if (list1.val <= list2.val) {
                changeNextOnList2 = false;
            } else if (list1.val >= list2.val) {
                changeNextOnList2 = true;
            }
            if (changeNextOnList2) {
                current.next = list2;
                current = current.next;
                list2 = list2.next;
            } else {
                current.next = list1;
                current = current.next;
                list1 = list1.next;
            }
            if (list1 == null && list2 == null)
                break;
        }
        return retHeadNode;
    }
}
