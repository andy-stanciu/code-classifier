class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        // Dummy node to simplify handling the head of the list
        ListNode l = new ListNode(-1);
        ListNode node = l;
        // Merge both lists
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                node.next = list1;
                list1 = list1.next;
            } else {
                node.next = list2;
                list2 = list2.next;
            }
            node = node.next;
        }
        // Append the remaining nodes from list1 or list2
        if (list1 != null) {
            node.next = list1;
        } else if (list2 != null) {
            node.next = list2;
        }
        // Return the merged list, skipping the dummy node
        return l.next;
    }
}
