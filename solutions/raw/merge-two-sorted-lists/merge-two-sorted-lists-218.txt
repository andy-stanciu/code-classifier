class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        Boolean flag = false;
        ListNode temp1 = null;
        ListNode temp2 = null;
        if (list1.val > list2.val) {
            temp1 = list2;
            temp2 = list1;
            flag = true;
        } else {
            temp1 = list1;
            temp2 = list2;
        }
        while (temp2 != null) {
            ListNode node = temp2.next;
            while (temp1.next != null && temp2 != null && temp1.next.val <= temp2.val) temp1 = temp1.next;
            if (temp1.val <= temp2.val) {
                temp2.next = temp1.next;
                temp1.next = temp2;
                temp2 = node;
            }
        }
        if (flag)
            return list2;
        return list1;
    }
}
