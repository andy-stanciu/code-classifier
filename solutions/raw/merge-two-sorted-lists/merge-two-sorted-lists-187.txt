// TC - O(N)
// SC - O(1)
class Solution {

    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if (l1 == null)
            return l2;
        else if (l2 == null)
            return l1;
        if (l1.val > l2.val) {
            ListNode temp = l1;
            l1 = l2;
            l2 = temp;
        }
        ListNode newHead = l1;
        while (l1 != null && l2 != null) {
            ListNode cur = null;
            while (l1 != null && l1.val <= l2.val) {
                cur = l1;
                l1 = l1.next;
            }
            cur.next = l2;
            // swap
            ListNode temp = l1;
            l1 = l2;
            l2 = temp;
        }
        return newHead;
    }
}
