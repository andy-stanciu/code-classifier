class Solution {

    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        ListNode tail;
        tail = list1;
        if (list1 == null && list2 == null) {
            return list1;
        } else if (list1 == null && list2 != null) {
            return list2;
        } else if (list1 != null && list2 == null) {
            return list1;
        } else {
            while (tail.next != null) {
                tail = tail.next;
            }
            tail.next = list2;
        }
        ListNode i;
        ListNode j;
        for (i = list1; i != null; i = i.next) {
            for (j = i.next; j != null; j = j.next) {
                if (i.val > j.val) {
                    int temp = i.val;
                    i.val = j.val;
                    j.val = temp;
                }
            }
        }
        return list1;
    }
}
