public class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode slow = head, fptr = head;
        while (fptr != null && fptr.next != null) {
            slow = slow.next;
            fptr = fptr.next.next;
            if (fptr == slow)
                return true;
        }
        return false;
    }
}
