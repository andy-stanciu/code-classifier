/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean hasCycle(ListNode head) {
        ListNode tail = head;
        if (head == null || head.next == null)
            return false;
        while (head != null && head.next != null) {
            tail = tail.next;
            head = head.next.next;
            if (head == tail) {
                return true;
            }
        }
        return false;
    }
}
