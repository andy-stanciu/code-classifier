public class Solution {

    public boolean hasCycle(ListNode head) {
        ListNode n = head;
        Set<ListNode> set = new HashSet<>();
        while (n != null) {
            if (set.contains(n))
                return true;
            set.add(n);
            n = n.next;
        }
        return false;
    }
}
// TC: O(n), SC: O(n)
