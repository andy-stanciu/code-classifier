/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean hasCycle(ListNode head) {
        // HashSet<ListNode> set=new HashSet<>();
        // ListNode temp=head;
        // while(temp!=null){
        //     ListNode presentNode=temp;
        //     if(set.contains(presentNode)){
        //         return true;
        //     }set.add(presentNode);
        //     temp=temp.next;
        // }
        ListNode slow = head, fast = head;
        if (head == null || head.next == null) {
            return false;
        }
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                return true;
            }
        }
        return false;
    }
}
