public class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null)
            return false;
        ListNode aux = head.next;
        while (head != null && aux != null && aux.next != null) {
            head = head.next;
            aux = aux.next.next;
            if (head == aux)
                return true;
        }
        return false;
    }
}
