public class Solution {

    public boolean hasCycle(ListNode head) {
        // O(n) time | O(n) space
        HashSet<ListNode> seen = new HashSet<>();
        while (head != null) {
            if (seen.contains(head))
                return true;
            else {
                seen.add(head);
                head = head.next;
            }
        }
        return false;
    }
}
