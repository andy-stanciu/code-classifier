public class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null) {
            return false;
        }
        while (head != null) {
            if (head.val == Integer.MAX_VALUE) {
                // already visited
                return true;
            }
            //flagging the node
            head.val = Integer.MAX_VALUE;
            head = head.next;
        }
        return false;
    }
}
