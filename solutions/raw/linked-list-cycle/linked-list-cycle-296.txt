class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        while (head.next.next != null) {
            if (head.next.next == head) {
                return true;
            }
            ListNode temp = head.next;
            head.next = head.next.next;
            head = temp;
        }
        return false;
    }
}
