/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean hasCycle(ListNode head) {
        ListNode slow = head, fast = head;
        boolean ans = false;
        if (// there is no node
        head == null)
            return false;
        if (head.next == null) {
            // there is only one node
            return false;
        }
        slow = slow.next;
        fast = fast.next == null ? fast.next : fast.next.next;
        while (slow != null && fast != null && slow != fast) {
            slow = slow.next;
            if (fast.next == null)
                return false;
            fast = fast.next.next;
        }
        return slow == fast;
    }
}
