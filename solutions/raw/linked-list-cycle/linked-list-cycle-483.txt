/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean hasCycle(ListNode head) {
        Set<ListNode> s = new HashSet<>();
        ListNode dummy = head;
        while (dummy != null) {
            if (s.contains(dummy)) {
                return true;
            }
            s.add(dummy);
            dummy = dummy.next;
        }
        return false;
    }
}
