/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean hasCycle(ListNode head) {
        HashSet<ListNode> set = new HashSet<>();
        ListNode temp = head;
        while (temp != null) {
            ListNode presentNode = temp;
            if (set.contains(presentNode)) {
                return true;
            }
            set.add(presentNode);
            temp = temp.next;
        }
        return false;
    }
}
