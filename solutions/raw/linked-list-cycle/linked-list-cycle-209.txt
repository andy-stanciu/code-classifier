public class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        HashSet<ListNode> llSet = new HashSet<>();
        while (head.next != null) {
            if (llSet.contains(head)) {
                return true;
            } else {
                llSet.add(head);
            }
            head = head.next;
        }
        return false;
    }
}
