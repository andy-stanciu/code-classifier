/**
 * Definition for singly-linked list.
 * class ListNode {
 * int val; ListNode next;
 * ListNode(int x) {
 * val = x; next = null;}}
 */
public class Solution {

    public boolean hasCycle(ListNode head) {
        ListNode temp = head;
        if (temp == null) {
            return false;
        }
        HashSet<ListNode> set = new HashSet<>();
        //out of the loop if we find a value that is there in set
        //=> there\'s cycle
        while (!set.contains(temp)) {
            set.add(temp);
            temp = temp.next;
            if (temp == null) {
                return false;
            }
        }
        //to confirm that there\'s cycle
        //will check for next node and next-next node
        //if set contains => true
        //else false @line34
        if (temp.next != null && temp.next.next != null && set.contains(temp.next) && set.contains(temp.next.next)) {
            return true;
        }
        return false;
    }
}
