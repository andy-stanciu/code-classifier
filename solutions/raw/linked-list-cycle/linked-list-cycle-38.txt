//Approach #2
public class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null)
            return false;
        if (head.next == head)
            return true;
        ListNode curr = head;
        Set<ListNode> hs = new HashSet<ListNode>();
        while (!hs.contains(curr)) {
            if (curr == null) {
                return false;
            }
            hs.add(curr);
            curr = curr.next;
        }
        return true;
    }
}
