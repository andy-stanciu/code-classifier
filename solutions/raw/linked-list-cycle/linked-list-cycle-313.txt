public class Solution {

    public boolean hasCycle(ListNode head) {
        if (head == null || head.next == null)
            return false;
        Map<ListNode, Integer> mpp = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (mpp.containsKey(temp)) {
                return true;
            } else {
                mpp.put(temp, 1);
            }
            temp = temp.next;
        }
        return false;
    }
}
