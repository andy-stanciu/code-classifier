public class Solution {

    public boolean wordBreak(String s, List<String> dic) {
        int n = s.length();
        int[] memo = new int[n + 1];
        memo[n] = 1;
        for (int i = n - 1; i >= 0; i--) {
            boolean ans = false;
            for (String word : dic) {
                if (i + word.length() <= n && s.substring(i, i + word.length()).equals(word)) {
                    ans = ans || (memo[i + word.length()] == 1);
                }
            }
            memo[i] = ans ? 1 : 0;
        }
        return memo[0] == 1;
    }
}
