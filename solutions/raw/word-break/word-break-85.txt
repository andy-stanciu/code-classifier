class Solution {

    Set<String> notValidWords = new HashSet<>();

    public boolean wordBreak(String s, List<String> wordDict) {
        Set<String> words = new HashSet<>(wordDict);
        if (words.contains(s)) {
            return true;
        } else {
            notValidWords.add(s);
        }
        int len = s.length();
        for (int i = 0; i < len; i++) {
            String left = s.substring(0, i);
            String right = s.substring(i, len);
            if (notValidWords.contains(left) || notValidWords.contains(right)) {
                continue;
            }
            if (wordBreak(left, wordDict) && wordBreak(right, wordDict)) {
                return true;
            }
        }
        return false;
    }
}
