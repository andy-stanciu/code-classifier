class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        return helper(s, new HashSet<>(wordDict), new HashMap<>());
    }

    public boolean helper(String s, HashSet<String> word, HashMap<String, Boolean> map) {
        if (s.isEmpty())
            return true;
        if (map.containsKey(s))
            return map.get(s);
        for (String w : word) {
            if (s.startsWith(w) && helper(s.substring(w.length()), word, map)) {
                map.put(s, true);
                return true;
            }
        }
        map.put(s, false);
        return false;
    }
}
