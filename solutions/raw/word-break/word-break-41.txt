class Solution {

    HashMap<String, Boolean> dp = new HashMap();

    public boolean wordBreak(String s, List<String> wordDict) {
        if (s == "")
            return true;
        if (dp.containsKey(s))
            return dp.get(s);
        boolean ans = false;
        for (int i = 0; i < s.length(); i++) {
            if (wordDict.contains(s.substring(0, i + 1))) {
                ans = wordBreak(s.substring(i + 1), wordDict);
                if (ans)
                    break;
            }
        }
        dp.put(s, ans);
        return ans;
    }
}
