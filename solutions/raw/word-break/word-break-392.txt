class Solution {

    Set<String> set = new HashSet<String>();

    public boolean wordBreak(String s, List<String> wordDict) {
        if (s.length() == 0)
            return true;
        if (set.contains(s))
            return false;
        for (int i = 0; i < s.length(); i++) {
            if (wordDict.contains(s.substring(0, i + 1))) {
                boolean ans = wordBreak(s.substring(i + 1), wordDict);
                if (ans)
                    return true;
                //else set.add()
            }
        }
        set.add(s);
        return false;
    }
}
