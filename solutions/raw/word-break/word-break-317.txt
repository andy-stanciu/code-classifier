class Solution {

    int[] dp;

    public boolean wordBreak(String s, List<String> wordDict) {
        dp = new int[s.length()];
        Arrays.fill(dp, -1);
        return recursion(s, wordDict, 0);
    }

    public boolean recursion(String s, List<String> wordDict, int index) {
        // After we cross string length
        if (index == s.length()) {
            return true;
        }
        // If set to 0, means this path will not yield solution
        if (dp[index] == 0) {
            return false;
        }
        // Loop over the string.
        for (int i = index; i < s.length(); i++) {
            if (wordDict.contains(s.substring(index, i + 1))) {
                if (recursion(s, wordDict, i + 1)) {
                    return true;
                }
            }
        }
        // Note down the index which will not work, to avoid unnecessary iterations.
        dp[index] = 0;
        return false;
    }
}
