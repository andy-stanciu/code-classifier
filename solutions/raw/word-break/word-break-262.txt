class Solution {

    static boolean f(int idx, String s, List<String> wordDict, boolean ans) {
        int n = s.length();
        if (idx == n) {
            return true;
        }
        for (int i = idx; i < s.length(); i++) {
            String str = s.substring(idx, i + 1);
            if (wordDict.contains(str) && f(i + 1, s, wordDict, ans)) {
                ans = true;
            }
        }
        return ans;
    }

    public boolean wordBreak(String s, List<String> wordDict) {
        int n = s.length();
        return f(0, s, wordDict, false);
    }
}
