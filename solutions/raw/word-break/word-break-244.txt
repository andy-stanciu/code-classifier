class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        int[] dp = new int[s.length()];
        Arrays.fill(dp, -1);
        int ans = f(0, s, wordDict, dp);
        if (ans == 1)
            return true;
        else
            return false;
    }

    public int f(int i, String s, List<String> wordDict, int[] dp) {
        if (i == s.length()) {
            return 1;
        }
        if (dp[i] != -1)
            return dp[i];
        int ans = 0;
        for (int j = i; j < s.length(); j++) {
            if (wordDict.contains(s.substring(i, j + 1))) {
                int x = f(j + 1, s, wordDict, dp);
                ans = Math.max(x, ans);
            }
        }
        return dp[i] = ans;
    }
}
