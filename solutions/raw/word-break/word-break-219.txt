class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        Boolean[][] memo = new Boolean[s.length() + 1][s.length() + 1];
        Set<String> wordSet = new HashSet<String>();
        for (String word : wordDict) {
            wordSet.add(word);
        }
        for (int i = 1; i <= s.length(); i++) {
            if (wordBreak(s, wordSet, 0, i, memo)) {
                return true;
            }
        }
        return false;
    }

    private boolean wordBreak(String s, Set<String> wordSet, int start, int end, Boolean[][] memo) {
        if (memo[start][end] != null) {
            return memo[start][end];
        }
        String subS = s.substring(start, end);
        if (wordSet.contains(subS)) {
            if (end == s.length()) {
                return true;
            }
            for (int i = end + 1; i <= s.length(); i++) {
                if (wordBreak(s, wordSet, end, i, memo)) {
                    return true;
                }
            }
        }
        memo[start][end] = false;
        return false;
    }
}
