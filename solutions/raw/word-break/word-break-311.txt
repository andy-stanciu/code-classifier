class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        Boolean[] memo = new Boolean[s.length() + 1];
        return solve(s, wordDict, 0, memo);
    }

    private boolean solve(String s, List<String> wordDict, int k, Boolean[] memo) {
        if (k == s.length())
            return true;
        if (memo[k] != null)
            return memo[k];
        for (int i = 0; i < wordDict.size(); i++) {
            String word = wordDict.get(i);
            if (s.startsWith(word, k)) {
                if (solve(s, wordDict, k + word.length(), memo))
                    return memo[k] = true;
            }
        }
        return memo[k] = false;
    }
}
