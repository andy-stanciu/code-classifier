class Solution {

    private boolean solve(String s, List<String> wordDict, HashSet<String> set) {
        if (s.length() == 0)
            return true;
        if (set.contains(s))
            return false;
        for (int j = 0; j < wordDict.size(); j++) {
            String curr = wordDict.get(j);
            if (curr.length() <= s.length() && s.substring(0, curr.length()).equals(curr)) {
                boolean isTrue = solve(s.substring(curr.length()), wordDict, set);
                if (isTrue)
                    return true;
            }
        }
        set.add(s);
        return false;
    }

    public boolean wordBreak(String s, List<String> wordDict) {
        return solve(s, wordDict, new HashSet<>());
    }
}
