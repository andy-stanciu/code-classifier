class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        HashSet<String> set = new HashSet<>(wordDict);
        int len = set.size();
        HashMap<String, Boolean> map = new HashMap<>();
        return Find(s, set, map);
    }

    public boolean Find(String s, HashSet set, HashMap<String, Boolean> map) {
        if (s.length() == 0) {
            return true;
        }
        if (map.containsKey(s))
            return map.get(s);
        if (set.contains(s)) {
            map.put(s, true);
        }
        ;
        for (int i = 0; i < s.length(); i++) {
            String str = s.substring(0, i + 1);
            if (set.contains(str) && Find(s.substring(i + 1), set, map)) {
                map.put(str, true);
                map.put(s.substring(i + 1), true);
                return true;
            }
        }
        map.put(s, false);
        return false;
    }
}
