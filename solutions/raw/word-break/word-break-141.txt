class Solution {

    static int n;

    public static boolean solve(String s, int idx, List<String> wordDict, int[] dp) {
        if (idx == n) {
            return true;
        }
        if (dp[idx] != -1) {
            if (dp[idx] == 1) {
                return true;
            }
            return false;
        }
        for (int i = idx + 1; i <= n; i++) {
            String subs = s.substring(idx, i);
            System.out.println(subs);
            System.out.println(i);
            if (wordDict.contains(subs)) {
                System.out.println(subs);
                if (solve(s, i, wordDict, dp)) {
                    dp[idx] = 1;
                    return true;
                }
            }
        }
        dp[idx] = 0;
        return false;
    }

    public boolean wordBreak(String s, List<String> wordDict) {
        n = s.length();
        int[] dp = new int[s.length()];
        Arrays.fill(dp, -1);
        return solve(s, 0, wordDict, dp);
    }
}
