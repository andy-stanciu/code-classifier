class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        int[] dp = new int[s.length()];
        Arrays.fill(dp, -1);
        return f(s, wordDict, 0, dp);
    }

    public boolean f(String s, List<String> wordDict, int ci, int[] dp) {
        if (ci >= s.length())
            return true;
        if (dp[ci] != -1) {
            if (dp[ci] == 1)
                return true;
            else
                return false;
        }
        String ans = "";
        boolean fans = false;
        for (int i = ci; i < s.length(); i++) {
            ans += s.charAt(i);
            if (checkPresent(ans, wordDict)) {
                if (f(s, wordDict, i + 1, dp)) {
                    fans = true;
                }
            }
        }
        if (fans) {
            dp[ci] = 1;
        } else {
            dp[ci] = 0;
        }
        return fans;
    }

    public boolean checkPresent(String s1, List<String> wordDict) {
        for (String i : wordDict) {
            if (i.equals(s1))
                return true;
        }
        return false;
    }
}
