class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        boolean[] wm = new boolean[s.length() + 1];
        wm[0] = true;
        for (int i = 0; i <= s.length(); i++) {
            for (int j = 0; j < i; j++) {
                if (wm[j] == true && wordDict.contains(s.substring(j, i))) {
                    wm[i] = true;
                    break;
                }
            }
        }
        return wm[s.length()];
    }
}
