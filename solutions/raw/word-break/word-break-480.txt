class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        Boolean[] memo = new Boolean[s.length() + 1];
        return helper(0, s, wordDict, memo);
    }

    public boolean helper(int i, String s, List<String> dict, Boolean[] memo) {
        if (i == s.length())
            return true;
        if (memo[i] != null)
            return memo[i];
        for (int j = i; j < s.length(); j++) {
            String curr = s.substring(i, j + 1);
            if (dict.contains(curr) && helper(j + 1, s, dict, memo)) {
                return memo[i] = true;
            }
        }
        return memo[i] = false;
    }
}
