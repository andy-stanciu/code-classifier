class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        //wordDict
        HashMap<Integer, Boolean> set = new HashMap<>();
        return wordB(s, wordDict, 0, set);
    }

    public boolean wordB(String s, List<String> wordDict, int index, HashMap<Integer, Boolean> set) {
        if (index >= s.length()) {
            return true;
        }
        if (set.containsKey(index)) {
            return set.get(index);
        }
        for (String x : wordDict) {
            if (s.substring(index).startsWith(x)) {
                if (wordB(s, wordDict, index + x.length(), set)) {
                    set.put(index, true);
                    return true;
                }
            }
        }
        set.put(index, false);
        return false;
    }
}
