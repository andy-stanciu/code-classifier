class Solution {

    HashMap<String, Boolean> inDict;

    public Solution() {
        inDict = new HashMap<>();
    }

    public boolean wordBreak(String s, List<String> wordDict) {
        if (s.length() == 0) {
            return true;
        }
        if (inDict.containsKey(s)) {
            return inDict.get(s);
        }
        for (String word : wordDict) {
            if (s.startsWith(word)) {
                if (wordBreak(s.substring(word.length()), wordDict)) {
                    inDict.put(s, true);
                    return true;
                }
                // inDict.put(s, false);
                // return false;
            }
        }
        inDict.put(s, false);
        return false;
    }
}
