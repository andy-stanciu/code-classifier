class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        Set<String> wordSet = new HashSet<>(wordDict);
        int n = s.length();
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        // Find the maximum length of any word in the wordSet to limit inner loop range
        int maxWordLength = 0;
        for (String word : wordSet) {
            maxWordLength = Math.max(maxWordLength, word.length());
        }
        for (int i = 1; i <= n; i++) {
            // Only check up to maxWordLength characters before i to save time
            for (int j = Math.max(0, i - maxWordLength); j < i; j++) {
                if (dp[j] && wordSet.contains(s.substring(j, i))) {
                    dp[i] = true;
                    // No need to check further once dp[i] is true
                    break;
                }
            }
        }
        return dp[n];
    }
}
