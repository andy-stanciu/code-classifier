class Solution {

    public boolean wordBreak(String s, List<String> dict) {
        boolean[] dp = new boolean[s.length() + 1];
        dp[0] = true;
        for (int i = 1; i <= s.length(); i++) {
            for (int j = i; j <= s.length(); j++) {
                String curr = s.substring(i - 1, j);
                if (dict.contains(curr) && dp[i - 1]) {
                    dp[j] = true;
                }
            }
        }
        return dp[s.length()];
    }
}
