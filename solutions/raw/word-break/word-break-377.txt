class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        int n = s.length();
        // Create a boolean array dp to store if a substring can be segmented
        boolean[] dp = new boolean[n + 1];
        // Empty string is always valid
        dp[0] = true;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                // If the substring s[j:i] is in the wordDict and the previous substring s[0:j] is valid
                if (dp[j] && wordDict.contains(s.substring(j, i))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[n];
    }
}
