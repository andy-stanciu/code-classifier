class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        int n = s.length();
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        for (int i = 0; i <= n; i++) {
            for (String word : wordDict) {
                int start = i - word.length();
                if (start >= 0 && s.substring(start, i).equals(word) && dp[start]) {
                    dp[i] = true;
                }
            }
        }
        return dp[n];
    }
}
