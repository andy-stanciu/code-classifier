class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        HashSet<String> set = new HashSet<>();
        HashSet<String> dp = new HashSet<>();
        for (String obj : wordDict) set.add(obj);
        return getAns(s, set, dp);
    }

    public boolean getAns(String s, HashSet<String> set, HashSet<String> dp) {
        if (dp.contains(s))
            return false;
        if (s.length() == 0)
            return true;
        for (int i = 0; i < s.length(); i++) {
            if (set.contains(s.substring(0, i + 1))) {
                if (getAns(s.substring(i + 1), set, dp))
                    return true;
            }
        }
        dp.add(s);
        return false;
    }
}
