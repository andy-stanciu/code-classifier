class Solution {

    Boolean[] dp;

    public boolean wordBreak(String s, List<String> wordDict) {
        dp = new Boolean[s.length()];
        return solve(0, s, wordDict);
    }

    public boolean solve(int idx, String s, List<String> words) {
        if (idx == s.length()) {
            return true;
        }
        if (dp[idx] != null) {
            return dp[idx];
        }
        for (int i = idx; i < s.length(); i++) {
            String substr = s.substring(idx, i + 1);
            if (words.contains(substr)) {
                if (solve(i + 1, s, words)) {
                    return dp[idx] = true;
                }
            }
        }
        return dp[idx] = false;
    }
}
