class Solution {

    public boolean wordBreak(String s, List<String> wordDict) {
        return solve(s, wordDict, "");
    }

    private boolean solve(String s, List<String> wordDict, String temp) {
        if (s.equals(temp)) {
            return true;
        }
        if (temp.length() > s.length()) {
            return false;
        }
        for (String next : wordDict) {
            if (solve(s, wordDict, new StringBuilder(temp).append(next).toString())) {
                return true;
            }
        }
        return false;
    }
}
