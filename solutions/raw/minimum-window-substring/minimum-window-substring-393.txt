class Solution {

    public String minWindow(String sc, String tc) {
        // expand until you find some window matching the condition,
        // then shrink until you no longer match the condition
        // Record the size of the window just before you no longer match condition
        // Repeat this process until hitting end of string and take the min size of the window
        int m = sc.length();
        int n = tc.length();
        char[] s = sc.toCharArray();
        char[] t = tc.toCharArray();
        if (t.length > s.length) {
            // it is not possible to find a window such that everything in t is in a window of s
            return "";
        }
        Map<Character, Integer> tMap = buildMap(t);
        int left = 0;
        int right = 0;
        int numFound = 0;
        int minWindowSize = Integer.MAX_VALUE;
        int bestLeft = -1;
        int bestRight = -1;
        while (right < s.length) {
            // if the character of s is in t, then we have moved closer to an answer
            if (tMap.containsKey(s[right])) {
                if (tMap.get(s[right]) > 0) {
                    numFound++;
                }
                tMap.put(s[right], tMap.get(s[right]) - 1);
            }
            // shrink the window while this is true
            while (left <= right && numFound == t.length) {
                if (right - left + 1 < minWindowSize) {
                    minWindowSize = right - left + 1;
                    bestLeft = left;
                    bestRight = right;
                }
                if (tMap.containsKey(s[left])) {
                    tMap.put(s[left], tMap.get(s[left]) + 1);
                    if (tMap.get(s[left]) > 0) {
                        numFound--;
                    }
                }
                left++;
            }
            right++;
        }
        return bestLeft == -1 ? "" : sc.substring(bestLeft, bestRight + 1);
    }

    Map<Character, Integer> buildMap(char[] t) {
        Map<Character, Integer> map = new HashMap<>();
        for (char c : t) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        return map;
    }
}
