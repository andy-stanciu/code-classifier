class Solution {

    public String minWindow(String s, String t) {
        // Initialize two arrays to count characters in t and the current window.
        int[] tCount = new int[128];
        int[] windowCount = new int[128];
        // Initialize the character counts for string t.
        for (char c : t.toCharArray()) {
            tCount[c]++;
        }
        int left = 0;
        int right = 0;
        int minLen = Integer.MAX_VALUE;
        int minLeft = 0;
        int required = t.length();
        while (right < s.length()) {
            char currentChar = s.charAt(right);
            if (tCount[currentChar] > 0) {
                windowCount[currentChar]++;
                if (windowCount[currentChar] <= tCount[currentChar]) {
                    required--;
                }
            }
            while (required == 0) {
                if (right - left + 1 < minLen) {
                    minLen = right - left + 1;
                    minLeft = left;
                }
                char leftChar = s.charAt(left);
                if (tCount[leftChar] > 0) {
                    windowCount[leftChar]--;
                    if (windowCount[leftChar] < tCount[leftChar]) {
                        required++;
                    }
                }
                left++;
            }
            right++;
        }
        return minLen == Integer.MAX_VALUE ? "" : s.substring(minLeft, minLeft + minLen);
    }
}
