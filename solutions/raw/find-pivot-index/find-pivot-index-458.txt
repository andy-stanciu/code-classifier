class Solution {

    private static int sum(int[] arr, int start, int end) {
        int sum = 0;
        for (int i = start; i <= end; i++) {
            sum += arr[i];
        }
        return sum;
    }

    public int pivotIndex(int[] arr) {
        int start = 0;
        int end = arr.length - 1;
        if (arr.length == 0)
            return -1;
        if (arr.length == 1)
            return 0;
        for (int i = start; i <= end; i++) {
            int leftSum = sum(arr, start, i - 1);
            int rightSum = sum(arr, i + 1, end);
            if (leftSum == rightSum) {
                return i;
            }
        }
        // no pivot found
        return -1;
    }
}
