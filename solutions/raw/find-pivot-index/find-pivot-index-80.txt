class Solution {

    public int pivotIndex(int[] nums) {
        int leftSum = 0, pIndex;
        int rightSum = 0;
        for (int i = nums.length - 1; i >= 0; i--) rightSum += nums[i];
        for (pIndex = 0; pIndex < nums.length; pIndex++) {
            if (pIndex - 1 != -1)
                leftSum += nums[pIndex - 1];
            rightSum -= nums[pIndex];
            if (leftSum == rightSum)
                return pIndex;
        }
        return -1;
    }
}
