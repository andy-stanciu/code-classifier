class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        int sum2 = 0;
        int x = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            sum = sum - nums[i];
            if (sum == sum2) {
                return i;
            }
            sum2 = sum2 + nums[i];
        }
        return -1;
    }
}
