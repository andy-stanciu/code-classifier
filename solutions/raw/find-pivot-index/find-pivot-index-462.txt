class Solution {

    public int pivotIndex(int[] nums) {
        int sumLeft = 0;
        int totalSum = 0;
        int pivot = -1;
        for (int num : nums) {
            totalSum += num;
        }
        //right sum should be total - left sum - the number itself
        for (int i = 0; i < nums.length; i++) {
            // Right= Total-Left- Number
            //if Left=right
            //replacing right with left in equation
            //left=total-left-number
            //rearranging it
            //2*left=total-number
            //using that in equation
            if (2 * sumLeft == totalSum - nums[i]) {
                return i;
            }
            //calculate left sum
            sumLeft += nums[i];
        }
        return pivot;
    }
}
