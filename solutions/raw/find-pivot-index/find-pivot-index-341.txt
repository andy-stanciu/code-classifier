class Solution {

    public int pivotIndex(int[] nums) {
        int[] pre = new int[nums.length];
        int[] upre = new int[nums.length];
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            pre[i] = sum;
        }
        sum = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            sum += nums[i];
            upre[i] = sum;
        }
        for (int i = 0; i < nums.length; i++) {
            if (pre[i] == upre[i]) {
                return i;
            }
        }
        return -1;
    }
}
