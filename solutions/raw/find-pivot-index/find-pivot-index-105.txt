class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int[] prefix = new int[n];
        int[] suffix = new int[n];
        int res = -1;
        prefix[0] = 0;
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] + nums[i - 1];
        }
        suffix[n - 1] = 0;
        for (int i = n - 2; i >= 0; i--) {
            suffix[i] = suffix[i + 1] + nums[i + 1];
        }
        if (prefix[0] == suffix[0]) {
            res = 0;
        }
        for (int i = 0; i < n; i++) {
            if (prefix[i] == suffix[i]) {
                res = i;
                break;
            }
        }
        return res;
    }
}
