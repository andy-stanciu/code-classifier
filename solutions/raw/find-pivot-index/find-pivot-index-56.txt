class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum = sum + nums[i];
        }
        int sumL = 0;
        for (int i = 0; i < n; i++) {
            if (i != 0) {
                sumL = sumL + nums[i - 1];
            }
            if (sum - sumL - nums[i] == sumL) {
                return i;
            }
        }
        return -1;
    }
}
