class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int[] sumLeft = new int[n];
        int[] sumRight = new int[n];
        for (int i = 0; i < n; i++) {
            if (i == 0) {
                sumLeft[i] = 0;
                sumRight[n - i - 1] = 0;
            } else {
                sumLeft[i] = sumLeft[i - 1] + nums[i - 1];
                sumRight[n - i - 1] = sumRight[n - i] + nums[n - i];
            }
        }
        for (int i = 0; i < n; i++) {
            if (sumLeft[i] == sumRight[i]) {
                return i;
            }
        }
        return -1;
    }
}
