class Solution {

    public int pivotIndex(int[] nums) {
        int rightSum = 0;
        for (int value : nums) {
            rightSum += value;
        }
        int pivot = -1;
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            rightSum -= nums[i];
            if (leftSum == rightSum) {
                pivot = i;
                break;
            }
            leftSum += nums[i];
        }
        return pivot;
    }
}
