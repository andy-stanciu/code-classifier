class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int prefixSum = 0;
        int index = 0;
        while (index < nums.length) {
            if ((index == 0 || index == nums.length - 1) && (sum - nums[index] == 0)) {
                return index;
            }
            if (prefixSum == sum - (prefixSum + nums[index])) {
                return index;
            }
            prefixSum += nums[index];
            index++;
        }
        return -1;
    }
}
