class Solution {

    public int pivotIndex(int[] nums) {
        int leftSum = 0;
        int rightSum = 0;
        int index = 0;
        for (int i = 1; i < nums.length; i++) {
            rightSum = rightSum + nums[i];
        }
        if (leftSum == rightSum)
            return index;
        while (leftSum != rightSum && index < nums.length - 1) {
            leftSum = leftSum + nums[index];
            rightSum = rightSum - nums[index + 1];
            index++;
        }
        if (index == nums.length - 1 && leftSum != rightSum)
            return -1;
        return index;
    }
}
