class Solution {

    public int pivotIndex(int[] nums) {
        if (nums.length == 0)
            return -1;
        int left_Sum = 0;
        int right_Sum = 0;
        for (int i = 0; i < nums.length; i++) right_Sum += nums[i];
        for (int i = 0; i < nums.length; i++) {
            right_Sum -= nums[i];
            if (right_Sum == left_Sum)
                return i;
            left_Sum += nums[i];
        }
        return -1;
    }
}
