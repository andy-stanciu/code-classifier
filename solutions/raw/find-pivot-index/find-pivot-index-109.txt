class Solution {

    public int pivotIndex(int[] nums) {
        int total_sum = 0;
        int answer = -1;
        int sum_till_now = 0;
        for (int i = 0; i < nums.length; i++) {
            total_sum += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (sum_till_now == (total_sum - sum_till_now - nums[i])) {
                answer = i;
                break;
            }
            sum_till_now += nums[i];
        }
        return answer;
    }
}
