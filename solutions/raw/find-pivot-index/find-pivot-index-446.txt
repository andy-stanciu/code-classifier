class Solution {

    public int pivotIndex(int[] nums) {
        int[] prefix = new int[nums.length];
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            prefix[i] = sum;
            sum += nums[i];
        }
        for (int i = 0; i < prefix.length; i++) {
            if ((sum - nums[i]) % 2 == 0 && (sum - nums[i]) / 2 == prefix[i]) {
                return i;
            }
        }
        return -1;
    }
}
