class Solution {

    public int pivotIndex(int[] nums) {
        int rightSum = 0, leftSum = 0;
        for (var i = 1; i < nums.length; ++i) {
            rightSum += nums[i];
        }
        for (var i = 0; i < nums.length; ++i) {
            if (rightSum == leftSum) {
                return i;
            }
            leftSum += nums[i];
            if (i + 1 < nums.length) {
                rightSum -= nums[i + 1];
            }
        }
        return -1;
    }
}
