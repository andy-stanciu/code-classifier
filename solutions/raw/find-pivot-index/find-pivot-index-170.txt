class Solution {

    public int pivotIndex(int[] nums) {
        List<Integer> postfix = new ArrayList<>();
        int sum = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            postfix.add(sum);
            sum += nums[i];
        }
        sum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (sum == postfix.get(nums.length - 1 - i)) {
                return i;
            }
            sum += nums[i];
        }
        return -1;
    }
}
