class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        int left = 0;
        int result = -1;
        for (int i : nums) {
            sum += i;
        }
        for (int i = 0; i < nums.length; i++) {
            sum -= nums[i];
            if (sum == left) {
                result = i;
                break;
            }
            left += nums[i];
        }
        return result;
    }
}
