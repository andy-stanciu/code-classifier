class Solution {

    public int pivotIndex(int[] nums) {
        int totalSum = 0, sum = 0;
        for (int ele : nums) totalSum += ele;
        for (int i = 0; i < nums.length; i++) {
            if (sum == Math.ceil(totalSum - nums[i]) / 2)
                return i;
            sum += nums[i];
        }
        return -1;
    }
}
