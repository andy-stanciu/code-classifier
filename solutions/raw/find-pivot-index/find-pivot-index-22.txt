class Solution {

    public int pivotIndex(int[] nums) {
        int left = 0, right = 0;
        for (int i = 1; i < nums.length; i++) left += nums[i];
        for (int i = 0; i < nums.length; i++) {
            if (right == left) {
                return i;
            }
            System.out.println(right + "   " + left);
            if (i < nums.length - 1)
                left -= nums[i + 1];
            right += nums[i];
        }
        return -1;
    }
}
