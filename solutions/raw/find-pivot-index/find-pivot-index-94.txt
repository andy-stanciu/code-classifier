class Solution {

    public int pivotIndex(int[] nums) {
        //total sum
        int ts = 0;
        //left sum
        int ls = 0;
        for (int i : nums) {
            ts += i;
        }
        for (int i = 0; i < nums.length; i++) {
            if (ts - ls - nums[i] == ls) {
                // totalsum - leftsum - current element is
                //   basically right sum.
                return i;
            }
            ls += nums[i];
        }
        return -1;
    }
}
