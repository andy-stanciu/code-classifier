class Solution {

    public int pivotIndex(int[] nums) {
        if (nums == null || nums.length == 0)
            return -1;
        int totalAmount = 0;
        int sum = 0;
        for (int n : nums) totalAmount += n;
        for (int i = 0; i < nums.length; i++) {
            if (2 * sum == totalAmount - nums[i])
                return i;
            sum += nums[i];
        }
        return -1;
    }
}
