class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        int curr = 0;
        for (int i = 0; i < n; i++) {
            ans[i] += curr;
            curr += nums[i];
        }
        curr = 0;
        int idx = -1;
        for (int i = n - 1; i >= 0; i--) {
            if (curr == ans[i])
                idx = i;
            curr += nums[i];
        }
        return idx;
    }
}
