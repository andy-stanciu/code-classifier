class Solution {

    public int pivotIndex(int[] nums) {
        int totalSum = 0;
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalSum += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (leftSum == totalSum - nums[i]) {
                return i;
            } else {
                leftSum += nums[i];
                totalSum -= nums[i];
            }
        }
        return -1;
    }
}
