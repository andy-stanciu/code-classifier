class Solution {

    public int pivotIndex(int[] nums) {
        int left = 0, right = 0, k = -1;
        for (int n : nums) {
            right = right + n;
        }
        for (int i = 0; i < nums.length; i++) {
            right = right - nums[i];
            if (left == right) {
                k = i;
                break;
            }
            left = left + nums[i];
        }
        return k;
    }
}
