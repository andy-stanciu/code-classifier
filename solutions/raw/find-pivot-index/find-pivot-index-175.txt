class Solution {

    public int pivotIndex(int[] nums) {
        int[] r = new int[nums.length];
        int id1 = nums.length - 1;
        int[] l = new int[nums.length];
        int id2 = 0;
        int sr = 0;
        int sl = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            r[id1--] = sr;
            sr += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            l[id2++] = sl;
            sl += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (r[i] == l[i]) {
                return i;
            }
        }
        return -1;
    }
}
