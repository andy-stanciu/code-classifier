class Solution {

    public int pivotIndex(int[] nums) {
        int sumR = 0, sumL = 0;
        for (int i = 0; i < nums.length; i++) {
            sumR += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            sumR -= nums[i];
            if (sumL == sumR)
                return i;
            sumL += nums[i];
        }
        return -1;
    }
}
