class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int n : nums) {
            sum += n;
        }
        int current = 0;
        for (int i = 0; i < nums.length; i++) {
            if (2 * current == (sum - nums[i])) {
                return i;
            }
            current += nums[i];
        }
        return -1;
    }
}
