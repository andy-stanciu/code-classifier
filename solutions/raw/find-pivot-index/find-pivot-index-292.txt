class Solution {

    public int pivotIndex(int[] nums) {
        int left = 0;
        int right = 0;
        int atual = 0;
        int lim = nums.length - 1;
        for (int i = 0; i <= lim; i++) {
            right += nums[i];
        }
        for (int i = 0; i <= lim; i++) {
            right -= nums[i];
            if (left == right) {
                return i;
            }
            left += nums[i];
        }
        return -1;
    }
}
