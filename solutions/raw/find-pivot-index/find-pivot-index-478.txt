class Solution {

    public int pivotIndex(int[] nums) {
        int leftSum = 0;
        int rightSum = 0;
        // TotalSum
        for (int i : nums) {
            rightSum += i;
        }
        for (int i = 0; i < nums.length; i++) {
            // RightSum
            rightSum -= nums[i];
            // Pivot Index
            if (leftSum == rightSum)
                return i;
            // Left Sum
            leftSum += nums[i];
        }
        return -1;
    }
}
