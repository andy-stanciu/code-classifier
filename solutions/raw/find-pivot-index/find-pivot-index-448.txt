class Solution {

    public int pivotIndex(int[] nums) {
        long sum_left = 0;
        long sum_right = 0;
        for (int i = 1; i < nums.length; i++) {
            sum_right += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (sum_left == sum_right) {
                return i;
            }
            if (i < nums.length - 1) {
                sum_right -= nums[i + 1];
            }
            sum_left += nums[i];
        }
        return -1;
    }
}
