class Solution {

    public int pivotIndex(int[] nums) {
        int pivotIndex = 0;
        while (pivotIndex < nums.length) {
            int leftSum = 0;
            int rightSum = 0;
            for (int i = 0; i < pivotIndex; i++) {
                leftSum += nums[i];
            }
            if (pivotIndex != nums.length - 1) {
                for (int j = pivotIndex + 1; j < nums.length; j++) {
                    rightSum += nums[j];
                }
            }
            if (leftSum == rightSum) {
                return pivotIndex;
            } else {
                pivotIndex++;
            }
        }
        return -1;
    }
}
