class Solution {

    // faster algorithm - use two sliding windows (left and right)
    public int pivotIndex(int[] numbers) {
        int leftSum = 0;
        int rightSum = 0;
        if (numbers.length == 1) {
            return 0;
        }
        for (int i = 0; i < numbers.length; i++) {
            if (i == 0) {
                leftSum = 0;
                // we can confidently write i + 1 because len >= 2
                rightSum = sum(numbers, i + 1, numbers.length);
            } else {
                // sliding windows
                leftSum += numbers[i - 1];
                rightSum -= numbers[i];
            }
            if (leftSum == rightSum)
                return i;
        }
        return -1;
    }

    private int sum(int[] array, int startIndex, int finishIndex) {
        int sum = 0;
        for (int i = startIndex; i < finishIndex; i++) {
            sum += array[i];
        }
        return sum;
    }

    // algorithm 2 - slow (direct solution)
    public int pivotIndex1(int[] numbers) {
        int leftSum;
        int rightSum;
        for (int pivotIndex = 0; pivotIndex < numbers.length; pivotIndex++) {
            if (pivotIndex == 0)
                leftSum = 0;
            else
                leftSum = sum(numbers, 0, pivotIndex);
            if (pivotIndex == numbers.length - 1)
                rightSum = 0;
            else
                rightSum = sum(numbers, pivotIndex + 1, numbers.length);
            if (leftSum == rightSum)
                return pivotIndex;
        }
        return -1;
    }
}
