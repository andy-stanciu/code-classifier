class Solution {

    public int pivotIndex(int[] nums) {
        int pivotIndex = 0;
        int leftSum = 0;
        int rightSum = 0;
        int i = 0;
        // Calculate rightSum for Index0
        while (i < nums.length) {
            rightSum += nums[i];
            i++;
        }
        for (int j = 0; j < nums.length; j++) {
            // Since we need to avoid pivotIndex value in both the sums, we remove it from rightSum and add it to the leftSum
            if (j > 0)
                leftSum += nums[j - 1];
            rightSum -= nums[j];
            // Once leftSum and rightSum are equal, pivotIndex is found
            if (leftSum == rightSum) {
                pivotIndex = j;
                return j;
            }
        }
        // If no pivotIndex, then return -1 as index
        return -1;
    }
}
