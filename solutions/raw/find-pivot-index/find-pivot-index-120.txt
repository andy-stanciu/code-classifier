class Solution {

    public int pivotIndex(int[] nums) {
        int totalSum = 0, leftSum = 0, n = nums.length;
        for (int i = 0; i < n; i++) totalSum += nums[i];
        for (int i = 0; i < n; i++) {
            // leftSum == totalSum - leftSum - nums[i]
            if (leftSum << 1 == totalSum - nums[i])
                return i;
            leftSum += nums[i];
        }
        return -1;
    }
}
