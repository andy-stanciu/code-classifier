class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int i : nums) sum = sum + i;
        int i = 0;
        int n = nums.length;
        int k = 0;
        for (i = 0; i < n; ++i) {
            if (k == sum - nums[i] - k)
                return i;
            k = k + nums[i];
        }
        return -1;
    }
}
