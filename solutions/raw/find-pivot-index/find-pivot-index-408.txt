class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int[] sum = new int[n];
        int k = 0;
        int curr_sum = 0;
        for (int x : nums) {
            curr_sum += x;
            sum[k++] = curr_sum;
        }
        for (int i = 0; i < n; i++) {
            int ls, rs;
            ls = (i == 0) ? 0 : sum[i - 1];
            rs = (i == n - 1) ? 0 : (sum[n - 1] - sum[i]);
            if (ls == rs)
                return i;
        }
        return -1;
    }
}
