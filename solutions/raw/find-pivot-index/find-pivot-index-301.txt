class Solution {

    public int pivotIndex(int[] nums) {
        int postsum = 0;
        int presum = 0;
        for (int num : nums) postsum += num;
        for (int i = 0; i < nums.length; i++) {
            postsum -= nums[i];
            if (postsum == presum)
                return i;
            presum += nums[i];
        }
        return -1;
    }
}
