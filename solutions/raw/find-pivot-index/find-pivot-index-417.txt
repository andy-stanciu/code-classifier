class Solution {

    public int pivotIndex(int[] nums) {
        int rsum = 0;
        int lsum = 0;
        int n = nums.length;
        if (n == 1) {
            return 0;
        }
        for (int i = 0; i < n; i++) {
            rsum += nums[i];
        }
        for (int j = 0; j < n; j++) {
            rsum -= nums[j];
            if (lsum == rsum) {
                return j;
            }
            lsum += nums[j];
        }
        return -1;
    }
}
