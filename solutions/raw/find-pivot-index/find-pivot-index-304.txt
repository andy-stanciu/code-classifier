class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        int pivot = -1;
        for (int x : nums) {
            sum += x;
        }
        int leftSum = 0;
        int rightSum = sum;
        for (int i = 0; i < nums.length; i++) {
            if (i == 0) {
                leftSum = 0;
                rightSum -= nums[i];
            } else {
                leftSum += nums[i - 1];
                rightSum -= nums[i];
            }
            if (leftSum == rightSum) {
                pivot = i;
                break;
            }
        }
        return pivot;
    }
}
