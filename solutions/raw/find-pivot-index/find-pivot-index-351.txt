class Solution {

    public int pivotIndex(int[] arr) {
        int rs = 0;
        int n = arr.length;
        for (int i = 0; i < n; i++) rs += arr[i];
        int ls = 0;
        for (int i = 0; i < n; i++) {
            rs -= arr[i];
            if (ls == rs)
                return i;
            ls += arr[i];
        }
        return -1;
    }
}
