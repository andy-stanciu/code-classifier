class Solution {

    public int pivotIndex(int[] nums) {
        int totalSum = 0;
        int leftSum = 0;
        // Calculate the total sum of the array
        for (int num : nums) {
            totalSum += num;
        }
        // Iterate through the array to find the pivot index
        for (int i = 0; i < nums.length; i++) {
            // Right sum is total sum minus left sum and current element
            int rightSum = totalSum - leftSum - nums[i];
            // Check if left sum equals right sum
            if (leftSum == rightSum) {
                // Return the pivot index
                return i;
            }
            // Update left sum for the next iteration
            leftSum += nums[i];
        }
        // Return -1 if no pivot index is found
        return -1;
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        // Example test cases
        int[] nums1 = { 1, 7, 3, 6, 5, 6 };
        // Output: 3
        System.out.println(solution.pivotIndex(nums1));
        int[] nums2 = { 1, 2, 3 };
        // Output: -1
        System.out.println(solution.pivotIndex(nums2));
        int[] nums3 = { 2, 1, -1 };
        // Output: 0
        System.out.println(solution.pivotIndex(nums3));
    }
}
