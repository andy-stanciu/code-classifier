class Solution {

    public int pivotIndex(int[] arr) {
        int n = arr.length;
        int count = 0;
        for (int i = 0; i < n; i++) {
            int leftSum = 0;
            int rightSum = 0;
            if (i == 0)
                leftSum = 0;
            for (int j = 0; j < i; j++) {
                leftSum = leftSum + arr[j];
            }
            if (i == n - 1)
                rightSum = 0;
            for (int k = i + 1; k < n; k++) {
                rightSum = rightSum + arr[k];
            }
            if (leftSum == rightSum)
                return i;
        }
        return -1;
    }
}
