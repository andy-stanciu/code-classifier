class Solution {

    public int pivotIndex(int[] nums) {
        int pivotIndex = -1;
        int arrSum = 0;
        int currSum = 0;
        for (int num : nums) arrSum += num;
        for (int i = 0; i < nums.length; i++) {
            if ((arrSum - currSum - nums[i]) == currSum)
                return i;
            currSum += nums[i];
        }
        return -1;
    }
}
