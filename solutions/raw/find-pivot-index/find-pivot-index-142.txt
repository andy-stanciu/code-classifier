class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        int ls = 0;
        ;
        int rs = 0;
        for (int i = 0; i < nums.length; i++) {
            rs = sum - ls - nums[i];
            if (ls == rs) {
                return i;
            }
            ls = ls + nums[i];
        }
        return -1;
    }
}
