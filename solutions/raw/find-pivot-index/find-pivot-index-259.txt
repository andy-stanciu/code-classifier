class Solution {

    public int pivotIndex(int[] nums) {
        int tsum = 0;
        for (int i = 0; i < nums.length; i++) {
            tsum += nums[i];
        }
        int lsum = 0;
        int rsum = tsum;
        for (int i = 0; i < nums.length; i++) {
            rsum -= nums[i];
            if (rsum == lsum) {
                return i;
            }
            lsum += nums[i];
        }
        return -1;
    }
}
