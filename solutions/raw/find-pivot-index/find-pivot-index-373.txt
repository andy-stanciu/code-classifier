class Solution {

    public int pivotIndex(int[] nums) {
        int[] prefixSum = new int[nums.length];
        int sum = 0;
        int answer = -1;
        for (int i = 0; i < nums.length; i++) {
            prefixSum[i] = sum;
            sum += nums[i];
        }
        sum = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (sum == prefixSum[i])
                answer = i;
            sum += nums[i];
        }
        return answer;
    }
}
