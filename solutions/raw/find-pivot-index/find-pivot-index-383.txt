class Solution {

    public int pivotIndex(int[] nums) {
        if (nums.length == 0)
            return 0;
        else if (nums.length == 1)
            return 0;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        if (sum == nums[0])
            return 0;
        int pre = 0;
        ;
        int post = sum - nums[0];
        for (int i = 1; i < nums.length; i++) {
            post = post - nums[i];
            pre = pre + nums[i - 1];
            if (post == pre)
                return i;
        }
        return -1;
    }
}
