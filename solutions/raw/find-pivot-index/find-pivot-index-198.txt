class Solution {

    public int pivotIndex(int[] nums) {
        int sumNums = 0;
        for (int num : nums) sumNums += num;
        int tSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if ((sumNums - nums[i]) / 2f == (float) tSum)
                return i;
            tSum += nums[i];
        }
        return -1;
    }
}
