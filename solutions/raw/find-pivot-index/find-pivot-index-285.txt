class Solution {

    public int pivotIndex(int[] nums) {
        int[] pre = new int[nums.length];
        // pre[0] = 0;
        int[] suf = new int[nums.length];
        // suf[nums.length-1] = 0;
        int l = 0;
        int r = nums.length - 1;
        while (l <= nums.length - 1 && r >= 0) {
            if (l == 0) {
                pre[l] = 0;
            } else {
                pre[l] = pre[l - 1] + nums[l - 1];
            }
            if (r == nums.length - 1) {
                suf[r] = 0;
            } else {
                suf[r] = suf[r + 1] + nums[r + 1];
            }
            l++;
            r--;
        }
        for (int i = 0; i < nums.length; i++) {
            if (pre[i] == suf[i])
                return i;
        }
        return -1;
    }
}
