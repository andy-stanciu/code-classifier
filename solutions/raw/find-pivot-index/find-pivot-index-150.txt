class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0, curr = 0;
        for (int i = 0; i < nums.length; i++) sum += nums[i];
        for (int i = 0; i < nums.length; i++) {
            curr += nums[i];
            if (sum == curr)
                return i;
            else
                sum = sum - nums[i];
        }
        return -1;
    }
}
