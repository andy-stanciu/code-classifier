class Solution {

    public int pivotIndex(int[] nums) {
        int total = 0;
        for (int i = 0; i < nums.length; i++) {
            total += nums[i];
        }
        int half = 0;
        for (int i = 0; i < nums.length; i++) {
            half += nums[i];
            if (total == half) {
                return i;
            }
            total -= nums[i];
        }
        return -1;
    }
}
