class Solution {

    public int pivotIndex(int[] nums) {
        int returnVal = -1;
        int sum = 0;
        for (int i : nums) {
            sum += i;
        }
        int l = nums.length;
        int sumL = 0;
        int sumR = 0;
        for (int i = 0; i < l; i++) {
            sumR = sum - sumL;
            if (sumL == (sumR - nums[i])) {
                returnVal = i;
                break;
            }
            sumL += nums[i];
        }
        return returnVal;
    }
}
