class Solution {

    public int pivotIndex(int[] nums) {
        int leftSum = 0, rightSum = 0;
        for (int i : nums) rightSum += i;
        rightSum -= nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (leftSum == rightSum)
                return i;
            leftSum += nums[i];
            if (i < nums.length - 1)
                rightSum -= nums[i + 1];
        }
        return -1;
    }
}
