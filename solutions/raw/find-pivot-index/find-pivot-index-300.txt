class Solution {

    public int pivotIndex(int[] nums) {
        int fs = 0, sum = 0, sumall = 0, pi = -1;
        for (int x = 0; x < nums.length; x++) {
            sumall += nums[x];
        }
        for (int x = 0; x < nums.length; x++) {
            fs += nums[x];
            if (fs - nums[x] == sumall - fs) {
                pi = x;
                break;
            }
        }
        return pi;
    }
}
