class Solution {

    public int pivotIndex(int[] nums) {
        int totalSum = 0;
        int sum = 0;
        int leftSum = 0;
        int rightSum = 0;
        for (int el : nums) {
            totalSum += el;
        }
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            leftSum = sum - nums[i];
            rightSum = totalSum - sum;
            if (leftSum == rightSum)
                return i;
        }
        return -1;
    }
}
