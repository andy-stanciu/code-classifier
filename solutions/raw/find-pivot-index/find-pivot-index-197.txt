class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int sm = 0;
        for (int i = 0; i < n; i++) {
            sm += nums[i];
        }
        int ls = 0;
        int rs = 0;
        for (int i = 0; i < n; i++) {
            rs = sm - nums[i] - ls;
            if (ls == rs) {
                return i;
            } else {
                ls += nums[i];
            }
        }
        return -1;
    }
}
