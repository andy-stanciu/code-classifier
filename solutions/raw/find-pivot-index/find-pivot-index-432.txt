class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int[] pref = new int[n];
        int[] suff = new int[n];
        pref[0] = 0;
        for (int i = 1; i < n; i++) {
            pref[i] = pref[i - 1] + nums[i - 1];
        }
        suff[n - 1] = 0;
        for (int i = n - 2; i >= 0; i--) {
            suff[i] = nums[i + 1] + suff[i + 1];
        }
        int ans = -1;
        for (int i = 0; i < n; i++) {
            if (suff[i] == pref[i]) {
                ans = i;
                break;
            }
        }
        return ans;
    }
}
