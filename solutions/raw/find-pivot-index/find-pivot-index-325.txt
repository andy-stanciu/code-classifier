class Solution {

    public int pivotIndex(int[] nums) {
        int sum = Arrays.stream(nums).sum();
        int leftsum = 0;
        int rightsum = sum;
        for (int i = 0; i < nums.length; i++) {
            leftsum += nums[i];
            if (leftsum == rightsum) {
                return i;
            } else {
                rightsum -= nums[i];
            }
        }
        return -1;
    }
}
