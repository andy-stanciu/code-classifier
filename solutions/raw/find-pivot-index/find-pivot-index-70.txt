class Solution {

    public int pivotIndex(int[] nums) {
        if (nums.length < 2)
            return 0;
        int sum = 0;
        for (int i : nums) sum += i;
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (leftSum == sum - nums[i])
                return i;
            leftSum += nums[i];
            sum -= nums[i];
        }
        return -1;
    }
}
