class Solution {

    public int pivotIndex(int[] nums) {
        int lsum = 0;
        int rsum = -nums[0];
        for (int i : nums) rsum += i;
        if (lsum == rsum)
            return 0;
        for (int i = 1; i < nums.length; i++) {
            lsum += nums[i - 1];
            rsum -= nums[i];
            if (lsum == rsum)
                return i;
        }
        return -1;
    }
}
