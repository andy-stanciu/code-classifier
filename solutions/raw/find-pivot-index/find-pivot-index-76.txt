class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int Lsum = 0;
        int mid = 1;
        int pivot = -1;
        for (int j = 0; j < nums.length; j++) {
            sum -= nums[j];
            if (Lsum == sum) {
                pivot = j;
                break;
            }
            Lsum += nums[j];
        }
        return pivot;
    }
}
