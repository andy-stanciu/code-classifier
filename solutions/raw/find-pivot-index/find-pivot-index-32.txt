class Solution {

    public int pivotIndex(int[] nums) {
        int left_sum = 0, right_sum = 0;
        for (int i = 0; i < nums.length; i++) {
            left_sum += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            left_sum -= nums[i];
            if (right_sum == left_sum) {
                return i;
            }
            right_sum += nums[i];
        }
        return -1;
    }
}
