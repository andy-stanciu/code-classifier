class Solution {

    public int pivotIndex(int[] nums) {
        int pivot = -1;
        for (int x = 0; x < nums.length; x++) {
            int leftIndex = leftSum(nums, x);
            int rightIndex = rightSum(nums, x);
            if (leftIndex == rightIndex) {
                return x;
            }
        }
        return pivot;
    }

    private int leftSum(int[] nums, int n) {
        int leftIndex = 0;
        for (int x = n - 1; x >= 0; x--) {
            leftIndex += nums[x];
        }
        return leftIndex;
    }

    private int rightSum(int[] nums, int n) {
        int rightIndex = 0;
        for (int x = n + 1; x < nums.length; x++) {
            rightIndex += nums[x];
        }
        return rightIndex;
    }
}
