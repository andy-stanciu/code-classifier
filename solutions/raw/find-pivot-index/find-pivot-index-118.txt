class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        int left_sum = 0;
        int right_sum = sum - nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (left_sum == right_sum) {
                return i;
            } else {
                left_sum = left_sum + nums[i];
                if ((i + 1) < nums.length) {
                    right_sum = right_sum - nums[i + 1];
                }
            }
        }
        return -1;
    }
}
