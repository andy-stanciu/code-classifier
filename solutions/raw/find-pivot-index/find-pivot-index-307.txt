class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        for (int i = 1; i < n; i++) nums[i] += nums[i - 1];
        if (nums[0] == nums[n - 1])
            return 0;
        for (int i = 1; i < n - 1; i++) if (nums[i - 1] + nums[i] == nums[n - 1])
            return i;
        if (nums[n - 2] == 0)
            return n - 1;
        return -1;
    }
}
