class Solution {

    public int pivotIndex(int[] nums) {
        int totSum = 0;
        for (int i : nums) totSum += i;
        int leftSum = 0, rightSum = totSum;
        for (int i = 0; i < nums.length; i++) {
            rightSum -= nums[i];
            if (rightSum == leftSum)
                return i;
            leftSum += nums[i];
        }
        return -1;
    }
}
