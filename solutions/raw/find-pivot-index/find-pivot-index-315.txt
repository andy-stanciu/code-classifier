class Solution {

    public int pivotIndex(int[] nums) {
        int size = nums.length;
        int lsum = 0;
        int rsum = 0;
        for (int j = 0; j < size; j++) {
            rsum = rsum + nums[j];
        }
        for (int i = 0; i < size; i++) {
            rsum = rsum - nums[i];
            if (lsum == rsum)
                return i;
            lsum = lsum + nums[i];
        }
        return -1;
    }
}
