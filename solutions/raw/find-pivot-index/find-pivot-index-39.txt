class Solution {

    public int pivotIndex(int[] nums) {
        int leftSum = 0;
        int totalSum = 0;
        for (int n : nums) {
            totalSum += n;
        }
        for (int pivotIndex = 0; pivotIndex < nums.length; pivotIndex++) {
            if (leftSum * 2 == totalSum - nums[pivotIndex]) {
                return pivotIndex;
            }
            leftSum += nums[pivotIndex];
        }
        return -1;
    }
}
