class Solution {

    public int pivotIndex(int[] nums) {
        int right = 0;
        int left = 0;
        for (int i = 1; i < nums.length; i++) {
            right = right + nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (right == left)
                return i;
            if (i == nums.length - 1)
                break;
            right -= nums[i + 1];
            left += nums[i];
        }
        return -1;
    }
}
