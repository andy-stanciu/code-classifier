class Solution {

    public int pivotIndex(int[] nums) {
        int leftsum = 0;
        int rightsum = 0;
        // Calculate the total rightsum (sum of all elements)
        for (int num : nums) {
            rightsum += num;
        }
        // Iterate to find the pivot index
        for (int i = 0; i < nums.length; i++) {
            // rightsum now represents the sum of elements to the right of index i
            if (leftsum == rightsum - nums[i]) {
                // Return the index if the pivot condition is met
                return i;
            }
            // Update leftsum and rightsum for the next index
            leftsum += nums[i];
            rightsum -= nums[i];
        }
        // Return -1 if no pivot index is found
        return -1;
    }
}
