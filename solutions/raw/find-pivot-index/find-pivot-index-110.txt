class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        int leftsum = 0;
        int rightsum = sum;
        for (int i = 0; i < nums.length; i++) {
            rightsum = rightsum - nums[i];
            if (leftsum == rightsum) {
                return i;
            }
            leftsum = leftsum + nums[i];
        }
        return -1;
    }
}
