class Solution {

    public int pivotIndex(int[] nums) {
        int N = nums.length;
        int suffix = 0;
        for (int i = 1; i < N; i++) {
            suffix += nums[i];
        }
        int i = 0;
        int prefix = 0;
        while (i < N - 1 && suffix != prefix) {
            i++;
            suffix -= nums[i];
            prefix += nums[i - 1];
        }
        if (suffix == prefix)
            return i;
        else
            return -1;
    }
}
