class Solution {

    public int pivotIndex(int[] nums) {
        int sumLeft = 0, sumRight = 0;
        for (int i = 1; i < nums.length; i++) sumRight += nums[i];
        if (sumRight == 0)
            return 0;
        for (int i = 1; i < nums.length; i++) {
            if (sumLeft == sumRight)
                return i - 1;
            sumLeft += nums[i - 1];
            sumRight -= nums[i];
        }
        if (sumLeft == 0)
            return nums.length - 1;
        return -1;
    }
}
