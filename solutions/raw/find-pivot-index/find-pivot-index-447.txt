class Solution {

    public int pivotIndex(int[] nums) {
        int sum = 0, i = 0, cursum = 0;
        for (int n : nums) sum += n;
        for (i = 0; i < nums.length; i++) {
            if ((sum - nums[i]) % 2 == 0) {
                if (cursum == ((sum - nums[i]) / 2))
                    return i;
            }
            cursum += nums[i];
        }
        return -1;
    }
}
