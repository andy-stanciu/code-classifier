// Algo Used: Prefix Sum
// TC: O N , SC: O 1
class Solution {

    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int totalSum = 0;
        // Calculate the total sum of the array
        for (int num : nums) {
            totalSum += num;
        }
        int leftSum = 0;
        // Iterate through the array to find the pivot index
        for (int i = 0; i < n; i++) {
            // Right sum is totalSum - leftSum - nums[i]
            int rightSum = totalSum - leftSum - nums[i];
            if (leftSum == rightSum) {
                return i;
            }
            leftSum += nums[i];
        }
        // If no pivot index is found
        return -1;
    }
}
