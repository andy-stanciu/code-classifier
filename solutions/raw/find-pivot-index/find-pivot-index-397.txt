class Solution {

    public int pivotIndex(int[] nums) {
        int arrayLength = nums.length;
        int totalArraySum = 0;
        int leftArrayElementsSum = 0;
        if (arrayLength == 0)
            return -1;
        // while( arrayLength >= trav ){
        //      totalArraySum += nums[trav];
        // }
        for (int trav : nums) {
            totalArraySum += trav;
        }
        // int trav = 0;
        // while(arrayLength >= trav ){
        //     totalArraySum -= nums[trav];
        //     if( totalArraySum == leftArrayElementsSum )return trav;
        //     leftArrayElementsSum += nums[trav];
        // }
        for (int trav = 0; trav < arrayLength; trav++) {
            totalArraySum -= nums[trav];
            if (totalArraySum == leftArrayElementsSum)
                return trav;
            leftArrayElementsSum += nums[trav];
        }
        return -1;
    }
}
