class Solution {

    public int pivotIndex(int[] nums) {
        int sumL = 0;
        int sumR = 0;
        // store sum of whole array to sumR
        for (int i = 0; i < nums.length; i++) {
            sumR += nums[i];
        }
        // Loop over array to compare
        for (int i = 0; i < nums.length; i++) {
            sumR -= nums[i];
            if (sumL == sumR)
                return i;
            sumL += nums[i];
        }
        return -1;
    }
}
