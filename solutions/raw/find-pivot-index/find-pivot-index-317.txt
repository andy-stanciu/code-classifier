class Solution {

    public int pivotIndex(int[] nums) {
        int length = nums.length;
        if (length == 1) {
            return 0;
        }
        int totalSum = 0;
        int leftsum = 0;
        for (int num : nums) totalSum += num;
        //all we need to find is if the leftSum * 2 = totalSum - current value.
        for (int i = 0; i < length; leftsum += nums[i++]) if (leftsum * 2 == totalSum - nums[i])
            return i;
        //return -1 if no pivot found
        return -1;
    }
}
