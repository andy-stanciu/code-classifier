class Solution {

    public int pivotIndex(int[] nums) {
        if (nums.length == 1)
            return 0;
        int rightsum = 0;
        for (int i = 0; i < nums.length; i++) rightsum += nums[i];
        int lsum = 0;
        for (int i = 0; i < nums.length; i++) {
            rightsum -= nums[i];
            if (rightsum == lsum)
                return i;
            lsum += nums[i];
        }
        return -1;
    }
}
