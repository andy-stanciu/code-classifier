class Solution {

    public int pivotIndex(int[] nums) {
        // declare sum
        int sum = 0;
        // declare prefix sum
        int prefix = 0;
        // sum all nums in nums array
        for (int num : nums) sum += num;
        // iterate over nums array:
        for (int i = 0; i < nums.length; i++) {
            // make comparison of current prefix sum and result of subtractions of current prefix sum and ceurrent num from sum
            if (prefix == sum - prefix - nums[i])
                return i;
            // if we found equality, return current index
            prefix += nums[i];
        }
        // if nothing found return -1Prefi
        return -1;
    }
}
