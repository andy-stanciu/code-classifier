class Solution {

    public int pivotIndex(int[] nums) {
        var sum = 0;
        for (int num : nums) {
            sum += num;
        }
        var prefix = 0;
        for (int i = 0; i < nums.length; i++) {
            if (prefix == ((double) sum - nums[i]) / 2) {
                return i;
            }
            prefix += nums[i];
        }
        return -1;
    }
}
