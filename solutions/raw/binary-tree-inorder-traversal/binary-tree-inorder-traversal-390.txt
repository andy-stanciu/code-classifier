/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        inorderTraversalOfBT(root, result);
        return result;
    }

    public void inorderTraversalOfBT(TreeNode root, List<Integer> result) {
        //base condition
        if (root == null)
            return;
        //get the inorder of the LST
        inorderTraversalOfBT(root.left, result);
        //Add the root value to the result
        result.add(root.val);
        //get the inorder of the RST
        inorderTraversalOfBT(root.right, result);
    }
}
