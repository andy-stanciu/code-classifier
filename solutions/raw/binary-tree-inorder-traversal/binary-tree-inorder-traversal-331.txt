class Solution {

    public List<Integer> solve(TreeNode root, List<Integer> res) {
        if (root == null)
            return res;
        solve(root.left, res);
        res.add(root.val);
        solve(root.right, res);
        return res;
    }

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        return solve(root, res);
    }
}
