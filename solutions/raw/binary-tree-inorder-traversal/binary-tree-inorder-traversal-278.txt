class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        inorder(root, li);
        return li;
    }

    public void inorder(TreeNode root, List<Integer> li) {
        if (root != null) {
            inorder(root.left, li);
            li.add(root.val);
            inorder(root.right, li);
        }
    }
}
