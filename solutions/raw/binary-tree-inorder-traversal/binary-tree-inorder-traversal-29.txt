/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> inorder = new ArrayList<>();
        return inorderRecursion(root, inorder);
    }

    public List<Integer> inorderRecursion(TreeNode root, List<Integer> inorder) {
        if (root == null)
            return inorder;
        inorderRecursion(root.left, inorder);
        inorder.add(root.val);
        inorderRecursion(root.right, inorder);
        return inorder;
    }
}
