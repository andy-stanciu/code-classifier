/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> help = new ArrayList<>();
        helper(root, help);
        return help;
    }

    public void helper(TreeNode root, List<Integer> help) {
        if (root != null) {
            helper(root.left, help);
            help.add(root.val);
            helper(root.right, help);
        }
    }
}
