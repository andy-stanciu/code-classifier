class Solution {

    public void inorder(TreeNode node, List<Integer> inor) {
        if (node == null)
            return;
        inorder(node.left, inor);
        inor.add(node.val);
        inorder(node.right, inor);
    }

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> inor = new ArrayList<>();
        inorder(root, inor);
        return inor;
    }
}
