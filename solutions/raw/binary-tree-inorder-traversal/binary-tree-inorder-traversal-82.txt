class Solution {

    public ArrayList<Integer> list;

    public void fun(TreeNode root) {
        if (root == null) {
            return;
        }
        fun(root.left);
        list.add(root.val);
        fun(root.right);
    }

    public List<Integer> inorderTraversal(TreeNode root) {
        list = new ArrayList<>();
        fun(root);
        return list;
    }
}
