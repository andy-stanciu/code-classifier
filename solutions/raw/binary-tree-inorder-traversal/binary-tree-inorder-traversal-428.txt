/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> li = new ArrayList<Integer>();

    public List<Integer> inorderTraversal(TreeNode root) {
        inorderImpl(root, li);
        return li;
    }

    public void inorderImpl(TreeNode root, List<Integer> li) {
        if (root == null) {
            return;
        }
        inorderImpl(root.left, li);
        System.out.print(root.val + ",");
        li.add(root.val);
        inorderImpl(root.right, li);
    }
}
