class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        treeNode(root, list);
        return list;
    }

    public void treeNode(TreeNode root, ArrayList list) {
        if (root == null)
            return;
        treeNode(root.left, list);
        list.add(root.val);
        treeNode(root.right, list);
    }
}
