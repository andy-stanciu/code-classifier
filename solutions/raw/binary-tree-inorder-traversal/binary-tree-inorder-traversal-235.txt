/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public static List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        helpermethod(root, result);
        return result;
    }

    public static void helpermethod(TreeNode root, List<Integer> result) {
        if (root == null) {
            return;
        }
        helpermethod(root.left, result);
        result.add(root.val);
        helpermethod(root.right, result);
    }
}
