class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        inorder(root, ans);
        return ans;
    }

    public void inorder(TreeNode r, List<Integer> ans) {
        if (r == null)
            return;
        inorder(r.left, ans);
        ans.add(r.val);
        inorder(r.right, ans);
    }
}
