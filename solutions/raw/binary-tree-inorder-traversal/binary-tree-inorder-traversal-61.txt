/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        LinkedList<Integer> lt = new LinkedList<Integer>();
        inOrder(root, lt);
        return lt;
    }

    public static void inOrder(TreeNode r, LinkedList<Integer> l) {
        if (r == null)
            return;
        inOrder(r.left, l);
        l.add(r.val);
        inOrder(r.right, l);
    }
}
