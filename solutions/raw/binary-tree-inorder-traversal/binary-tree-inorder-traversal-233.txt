/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> concatenatedList = new ArrayList<>();
        if (root == null)
            return concatenatedList;
        if (root.left != null)
            concatenatedList.addAll(inorderTraversal(root.left));
        concatenatedList.add(root.val);
        if (root.right != null)
            concatenatedList.addAll(inorderTraversal(root.right));
        return concatenatedList;
    }
}
