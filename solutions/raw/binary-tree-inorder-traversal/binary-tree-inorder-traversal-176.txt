class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        TreeNode temp = root;
        List<Integer> list = new ArrayList<>();
        inorder(root, list);
        return list;
    }

    private void inorder(TreeNode root, List<Integer> list) {
        TreeNode temp = root;
        if (temp == null) {
            return;
        }
        inorder(temp.left, list);
        list.add(temp.val);
        inorder(temp.right, list);
    }
}
