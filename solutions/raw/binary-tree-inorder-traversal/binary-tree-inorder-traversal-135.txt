class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> listInOrder = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        TreeNode curr = root;
        while (curr != null || !stack.isEmpty()) {
            if (curr != null) {
                stack.push(curr);
                curr = curr.left;
            } else {
                curr = stack.pop();
                listInOrder.add(curr.val);
                curr = curr.right;
            }
        }
        return listInOrder;
    }
}
