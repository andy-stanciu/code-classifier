class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        inTra(root, ls);
        return ls;
    }

    public void inTra(TreeNode root, List<Integer> ls) {
        if (root != null) {
            inTra(root.left, ls);
            ls.add(root.val);
            inTra(root.right, ls);
        }
    }
}
