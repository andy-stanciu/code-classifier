class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null)
            return list;
        Stack<TreeNode> stk = new Stack<>();
        for (TreeNode i = root; root != null || !stk.isEmpty(); root = root.right) {
            for (TreeNode j = root; j != null; j = j.left) {
                stk.push(j);
            }
            root = stk.pop();
            list.add(root.val);
        }
        return list;
    }
}
