class Solution {

    List<Integer> ans = new ArrayList<Integer>();

    public List<Integer> inorderTraversal(TreeNode root) {
        inOrder(root);
        return ans;
    }

    void inOrder(TreeNode root) {
        if (root != null) {
            inOrder(root.left);
            ans.add(root.val);
            inOrder(root.right);
        }
    }
}
