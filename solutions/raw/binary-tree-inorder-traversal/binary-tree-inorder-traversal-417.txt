class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        // Morris Traversal
        TreeNode curr = root;
        List<Integer> arr = new ArrayList<>();
        while (curr != null) {
            if (curr.left != null) {
                // find Predecessor
                TreeNode pred = curr.left;
                while (pred.right != null && pred.right != curr) {
                    pred = pred.right;
                }
                if (pred.right == null) {
                    // Create the Link
                    pred.right = curr;
                    curr = curr.left;
                }
                if (pred.right == curr) {
                    //break the Link
                    pred.right = null;
                    arr.add(curr.val);
                    curr = curr.right;
                }
            } else {
                // will be used for the right side of traversal
                arr.add(curr.val);
                curr = curr.right;
            }
        }
        return arr;
    }
}
