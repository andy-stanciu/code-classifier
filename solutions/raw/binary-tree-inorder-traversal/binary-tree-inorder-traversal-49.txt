class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> ll = new ArrayList<>();
        inorder(root, ll);
        return ll;
    }

    public void inorder(TreeNode root, List<Integer> ll) {
        if (root == null)
            return;
        inorder(root.left, ll);
        ll.add(root.val);
        inorder(root.right, ll);
    }
}
