class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        inOrder(arr, root);
        return arr;
    }

    public static void inOrder(ArrayList<Integer> arr, TreeNode root) {
        if (root == null) {
            return;
        }
        //left value right
        inOrder(arr, root.left);
        arr.add(root.val);
        inOrder(arr, root.right);
    }
}
