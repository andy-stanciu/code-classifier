/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> inorderTraversal(TreeNode root) {
        return inorderTraversalHelper(root, new ArrayList<Integer>());
    }

    public List<Integer> inorderTraversalHelper(TreeNode root, List<Integer> l) {
        if (root == null) {
            return l;
        }
        inorderTraversalHelper(root.left, l);
        l.add(root.val);
        inorderTraversalHelper(root.right, l);
        return l;
    }
}
