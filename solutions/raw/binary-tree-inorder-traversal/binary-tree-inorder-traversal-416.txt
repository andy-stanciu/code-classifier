class Solution {

    private void inOrder(List<Integer> numbers, TreeNode root) {
        if (root == null)
            return;
        inOrder(numbers, root.left);
        numbers.add(root.val);
        inOrder(numbers, root.right);
    }

    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> numbers = new ArrayList<>();
        inOrder(numbers, root);
        return numbers;
    }
}
