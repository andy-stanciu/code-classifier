class Solution {

    public int singleNumber(int[] nums) {
        for (int i = 0; i <= nums.length - 1; i++) {
            int c = 1;
            for (int j = 0; j <= nums.length - 1; j++) {
                if (i != j && nums[i] == nums[j]) {
                    c++;
                }
            }
            if (c == 1) {
                return nums[i];
            }
        }
        return -1;
    }
}
