class Solution {

    public int singleNumber(int[] arr) {
        Arrays.sort(arr);
        for (int i = 0; i < arr.length - 1; i += 2) {
            if (arr[i] != arr[i + 1]) {
                //check pair wise if unique return
                return arr[i];
            }
        }
        // if largest element is unique
        return arr[arr.length - 1];
    }
}
