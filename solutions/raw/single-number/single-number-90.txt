class Solution {

    public int singleNumber(int[] nums) {
        List<Integer> list = new ArrayList<Integer>();
        for (int i = 0; i < nums.length; i++) {
            if (list.contains(nums[i])) {
                Integer ir = new Integer(nums[i]);
                list.remove(ir);
            } else
                list.add(nums[i]);
        }
        System.out.println(list);
        int out = list.get(0);
        return out;
    }
}
