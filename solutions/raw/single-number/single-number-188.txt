class Solution {

    public int singleNumber(int[] nums) {
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 1; i += 2) {
            if (nums[i] != nums[i + 1]) {
                //if the single number lies before the last element in the array
                return nums[i];
            }
        }
        // if the last element is single sumber in the array
        return nums[nums.length - 1];
    }
}
