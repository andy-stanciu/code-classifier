class Solution {

    public int singleNumber(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        // Traverse the array
        for (int num : nums) {
            if (!set.contains(num)) {
                // If number is not in the set, add it
                set.add(num);
            } else {
                // If number is already in the set, remove it (since it\'s a duplicate)
                set.remove(num);
            }
        }
        // The only element left in the set is the single number
        // Return the remaining element
        return set.iterator().next();
    }
}
