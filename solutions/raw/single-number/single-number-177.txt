class Solution {

    public int singleNumber(int[] nums) {
        Map<Integer, Integer> checkCount = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; i++) {
            if (checkCount.containsKey(nums[i])) {
                int c = checkCount.get(nums[i]);
                checkCount.put(nums[i], ++c);
            } else {
                checkCount.put(nums[i], 1);
            }
        }
        int op = 0;
        for (Map.Entry<Integer, Integer> entry : checkCount.entrySet()) {
            if (entry.getValue() == 1) {
                op = entry.getKey();
            }
        }
        return op;
    }
}
