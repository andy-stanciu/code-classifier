class Solution {

    public int singleNumber(int[] nums) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int s : nums) {
            if (hm.containsKey(s)) {
                hm.put(s, hm.get(s) + 1);
            } else {
                hm.put(s, 1);
            }
        }
        for (int k : hm.keySet()) {
            if (hm.get(k) == 1) {
                return k;
            }
        }
        return -1;
    }
}
