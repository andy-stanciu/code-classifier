public class Solution {

    public static int singleNumber(int[] arr) {
        int n = arr.length;
        HashMap<Integer, Integer> kvp = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int value = kvp.getOrDefault(arr[i], 0);
            kvp.put(arr[i], value + 1);
        }
        for (Map.Entry<Integer, Integer> it : kvp.entrySet()) {
            if (it.getValue() == 1) {
                return it.getKey();
            }
        }
        return -1;
    }
}
