/*
        * Using maps can help checking the value faster against a key
        * Populate the map with the values present in array
        * Return key whose value is one
        * */
class Solution {

    public int singleNumber(int[] nums) {
        int singleNumber = 0;
        for (int i = 0; i < nums.length; i++) {
            singleNumber ^= nums[i];
        }
        return singleNumber;
    }
}
