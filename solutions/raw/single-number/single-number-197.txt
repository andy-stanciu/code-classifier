class Solution {

    public int singleNumber(int[] nums) {
        Arrays.sort(nums);
        int k = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1])
                i++;
            else
                k = nums[i];
        }
        if (k == Integer.MIN_VALUE)
            k = nums[nums.length - 1];
        return k;
    }
}
