class Solution {

    public int singleNumber(int[] nums) {
        Map<Integer, Integer> m1 = new HashMap<>();
        for (int i : nums) {
            m1.put(i, m1.getOrDefault(i, 0) + 1);
        }
        for (int i : m1.keySet()) {
            if (m1.get(i) == 1) {
                return i;
            }
        }
        return -1;
    }
}
