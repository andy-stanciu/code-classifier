class Solution {

    public int singleNumber(int[] arr) {
        List<Integer> num = new ArrayList<>();
        for (int i = 0; i < arr.length; i++) {
            if (!num.contains(arr[i])) {
                num.add(arr[i]);
                continue;
            }
            if (num.contains(arr[i])) {
                num.remove(Integer.valueOf(arr[i]));
            }
        }
        return num.get(0);
    }
}
