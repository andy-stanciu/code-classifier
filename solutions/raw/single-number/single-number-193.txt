class Solution {

    public int singleNumber(int[] nums) {
        int n = nums.length;
        int x = nums[0];
        for (int i = 1; i < n; i++) {
            //since a XOR a=0, so every number with twice occurance results to zero, and a XOR 0=a hence we get the number which is only once appeared in the array.
            x = x ^ nums[i];
        }
        return x;
    }
}
