class Solution {

    public int singleNumber(int[] nums) {
        return singleNumberMini(nums, 0, new ArrayList<>(1));
    }

    private static int singleNumberMini(int[] nums, int index, List<Integer> res) {
        int a = nums[index];
        if (res.contains(a)) {
            res.remove((Integer) a);
        } else {
            res.add(a);
        }
        return index == nums.length - 1 ? res.get(0) : singleNumberMini(nums, index + 1, res);
    }
}
