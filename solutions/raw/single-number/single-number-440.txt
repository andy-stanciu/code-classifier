class Solution {

    public int singleNumber(int[] nums) {
        int flag = 0;
        int count = 0;
        for (int i : nums) {
            for (int j : nums) {
                if (i == j) {
                    flag++;
                }
            }
            if (flag == 1) {
                return i;
            } else
                flag = 0;
        }
        return 0;
    }
}
