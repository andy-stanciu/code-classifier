class Solution {

    public int singleNumber(int[] nums) {
        int result = 0;
        // 0 XOR (^) any number would return that number
        // If same numbers are XOR\'d it would cancel out and give 0
        // In the end only the single number would be in the result
        // 0^0 = 0 and 1^1 = 0, 0^1 = 1 and 1^0 = 1
        for (int n : nums) {
            result ^= n;
        }
        return result;
    }
}
