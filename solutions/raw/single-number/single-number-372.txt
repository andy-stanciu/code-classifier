class Solution {

    public int singleNumber(int[] nums) {
        int res = -1;
        ArrayList<Integer> a = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            a.add(nums[i]);
        }
        for (int i : a) {
            if (a.indexOf(i) == a.lastIndexOf(i)) {
                return i;
            }
        }
        return res;
    }
}
