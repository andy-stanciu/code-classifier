public class Solution {

    public String longestCommonPrefix(String[] strs) {
        if (strs.length == 0) {
            return "";
        }
        String str1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            while (!strs[i].startsWith(str1)) {
                str1 = str1.substring(0, str1.length() - 1);
            }
        }
        return str1;
    }
}
