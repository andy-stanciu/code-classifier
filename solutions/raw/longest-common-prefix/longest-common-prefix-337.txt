class Solution {

    public String longestCommonPrefix(String[] strs) {
        String output = strs[0];
        for (int i = 1; i < strs.length; i++) {
            String toCompare = strs[i];
            if (!output.equals("")) {
                if (output.equals(toCompare)) {
                    continue;
                } else {
                    StringBuilder commonPrefix = new StringBuilder();
                    int minLength = Math.min(output.length(), toCompare.length());
                    for (int j = 0; j < minLength; j++) {
                        if (output.charAt(j) == toCompare.charAt(j)) {
                            commonPrefix.append(output.charAt(j));
                        } else {
                            break;
                        }
                    }
                    output = commonPrefix.toString();
                }
            } else {
                break;
            }
        }
        return output;
    }
}
