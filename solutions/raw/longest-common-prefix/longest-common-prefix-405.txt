class Solution {

    public String longestCommonPrefix(String[] strs) {
        if (strs.length == 1) {
            return strs[0];
        }
        int numOfPerfix = 1000000, count = 0;
        for (int i = 0; i < strs.length - 1; i++) {
            count = 0;
            int len = strs[i].length() > strs[i + 1].length() ? strs[i + 1].length() : strs[i].length();
            for (int j = 0; j < len; j++) {
                if (strs[i].charAt(count) == strs[i + 1].charAt(count)) {
                    count++;
                }
            }
            if (numOfPerfix > count)
                numOfPerfix = count;
        }
        return strs[0].substring(0, numOfPerfix);
    }
}
