class Solution {

    public String longestCommonPrefix(String[] strs) {
        int count;
        Arrays.sort(strs);
        String str = "";
        String strCompare = strs[0];
        for (int i = strCompare.length(); i >= 0; i--) {
            count = 0;
            str = strCompare.substring(0, i);
            for (String str1 : strs) {
                if (str1.startsWith(str))
                    count = count + 1;
            }
            if (count == strs.length)
                return str;
        }
        return "";
    }
}
