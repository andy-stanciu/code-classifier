class Solution {

    public static String longestCommonPrefix(String[] strs) {
        if (strs == null || strs.length == 0) {
            // Return empty if input is null or empty
            return "";
        }
        String shortestStr = findShortestStr(strs);
        int min = shortestStr.length();
        for (String word : strs) {
            for (int i = 0; i < min; i++) {
                if (word.charAt(i) != shortestStr.charAt(i)) {
                    // Reduce min to the length of the common prefix
                    min = i;
                    break;
                }
            }
        }
        // Return the common prefix
        return shortestStr.substring(0, min);
    }

    // Helper method to find the shortest string
    public static String findShortestStr(String[] strs) {
        // Initialize with the first string
        String shortest = strs[0];
        for (String word : strs) {
            if (word.length() < shortest.length()) {
                shortest = word;
            }
        }
        return shortest;
    }
}
