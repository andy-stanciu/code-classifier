class Solution {

    public String longestCommonPrefix(String[] strs) {
        String prefix = "";
        for (String w : strs) {
            for (Character c : w.toCharArray()) {
                var prefixNew = prefix + c;
                if (!isPrefixInAllWords(strs, prefixNew))
                    return prefix;
                prefix = prefixNew;
            }
        }
        return prefix;
    }

    public Boolean isPrefixInAllWords(String[] words, String prefix) {
        for (String w : words) {
            if (!w.startsWith(prefix))
                return false;
        }
        return true;
    }
}
