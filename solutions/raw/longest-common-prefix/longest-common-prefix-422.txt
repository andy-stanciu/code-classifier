class Solution {

    public String longestCommonPrefix(String[] strs) {
        StringBuilder sb = new StringBuilder();
        int l = strs.length;
        Arrays.sort(strs);
        //storing the first and last element of sorted array in new string
        String f = strs[0];
        String s = strs[l - 1];
        int i = 0;
        while (i < f.length() && i < s.length() && f.charAt(i) == s.charAt(i)) {
            sb.append(f.charAt(i));
            i++;
        }
        return sb.toString();
    }
}
