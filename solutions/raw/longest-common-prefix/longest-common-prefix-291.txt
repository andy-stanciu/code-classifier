class Solution {

    public String longestCommonPrefix(String[] strs) {
        String maxPrefix = strs[0];
        for (int i = 1; i < strs.length; i++) {
            if (!strs[i].startsWith(maxPrefix)) {
                int newPrefixLen = 0;
                while (newPrefixLen < strs[i].length() && maxPrefix.charAt(newPrefixLen) == strs[i].charAt(newPrefixLen)) {
                    newPrefixLen++;
                }
                maxPrefix = strs[i].substring(0, newPrefixLen);
            }
        }
        return maxPrefix;
    }
}
