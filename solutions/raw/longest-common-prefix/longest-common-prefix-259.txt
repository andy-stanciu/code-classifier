class Solution {

    public String longestCommonPrefix(String[] strs) {
        StringBuilder sb = new StringBuilder();
        if (strs.length == 0 || strs == null) {
            return "";
        }
        String dasErsteWort = strs[0];
        for (int i = 0; i < dasErsteWort.length(); i++) {
            char buchstabeDerErsteWort = dasErsteWort.charAt(i);
            for (int j = 1; j < strs.length; j++) {
                if (i >= strs[j].length() || buchstabeDerErsteWort != strs[j].charAt(i)) {
                    return sb.toString();
                }
            }
            sb.append(buchstabeDerErsteWort);
        }
        return sb.toString();
    }
}
