class Solution {

    public String longestCommonPrefix(String[] strs) {
        String prefix = strs[0];
        int len = prefix.length();
        int i = 0;
        do {
            while (!strs[i].startsWith(prefix)) {
                prefix = prefix.substring(0, --len);
            }
            i++;
        } while (i < strs.length);
        return prefix;
    }
}
