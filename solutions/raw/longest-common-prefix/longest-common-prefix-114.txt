class Solution {

    public String longestCommonPrefix(String[] strs) {
        String target = strs[0];
        for (String str : strs) {
            for (int i = 0; i < str.length() && i < target.length(); i++) {
                if (str.charAt(i) != target.charAt(i)) {
                    target = str.substring(0, i);
                    break;
                }
            }
            if (str.length() < target.length())
                target = str;
        }
        return target;
    }
}
