class Solution {

    public String longestCommonPrefix(String[] strs) {
        int j = 0;
        boolean same = true;
        if (strs.length == 0)
            return "";
        if (strs.length == 1)
            return strs[0];
        while (same) {
            for (int i = 0; i < strs.length - 1; i++) {
                if (j < strs[i].length() && j < strs[i + 1].length() && strs[i].charAt(j) == strs[i + 1].charAt(j)) {
                    continue;
                } else {
                    same = false;
                    break;
                }
            }
            if (same) {
                j++;
            }
        }
        return strs[0].substring(0, j);
    }
}
