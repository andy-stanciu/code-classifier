class Solution {

    public String longestCommonPrefix(String[] strs) {
        int index = strs[0].length();
        for (int i = 1; index > 0 && i < strs.length; i++) {
            int j = 0;
            while (j < index && j < strs[i].length() && strs[i].charAt(j) == strs[0].charAt(j)) j++;
            index = j;
        }
        return strs[0].substring(0, index);
    }
}
