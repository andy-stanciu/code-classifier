class Solution {

    public String longestCommonPrefix(String[] strs) {
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < strs.length; i++) {
            if (strs[i].length() < min) {
                min = strs[i].length();
            }
        }
        String res = "";
        for (int i = 0; i < min; i++) {
            char x = strs[0].charAt(i);
            for (int j = 1; j < strs.length; j++) {
                if (x != strs[j].charAt(i)) {
                    return res;
                }
            }
            res += x;
        }
        return res;
    }
}
