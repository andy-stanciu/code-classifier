class Solution {

    public String longestCommonPrefix(String[] s) {
        int i = 0;
        Arrays.sort(s);
        for (String c : s) System.out.println(c);
        String s1 = s[0];
        String s2 = s[s.length - 1];
        while (i < s1.length() && i < s2.length()) {
            if (s1.charAt(i) == s2.charAt(i))
                i++;
            else
                break;
        }
        return s1.substring(0, i);
    }
}
