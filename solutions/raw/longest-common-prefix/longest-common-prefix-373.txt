class Solution {

    public String longestCommonPrefix(String[] strs) {
        String prefix = "";
        int minLen = strs[0].length();
        for (String s : strs) {
            if (minLen > s.length())
                minLen = s.length();
        }
        int prefixLen = minLen;
        boolean flag = false;
        while (prefixLen > 0) {
            prefix = strs[0].substring(0, prefixLen);
            for (String s1 : strs) {
                if (!startsWith(s1, prefix)) {
                    flag = false;
                    break;
                }
                flag = true;
            }
            if (flag)
                break;
            prefixLen--;
        }
        prefix = (flag) ? prefix : "";
        return prefix;
    }

    public static boolean startsWith(String s, String prefix) {
        if (s.startsWith(prefix))
            return true;
        return false;
    }
}
