class Solution {

    public String longestCommonPrefix(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        int len = strs[0].length();
        String result = "";
        for (int i = 0; i < len; i++) {
            char string1 = strs[0].charAt(i);
            for (int j = 1; j < strs.length; j++) {
                if (i >= strs[j].length() || strs[j].charAt(i) != string1) {
                    return result;
                }
            }
            result = result + string1;
        }
        return result;
    }
}
