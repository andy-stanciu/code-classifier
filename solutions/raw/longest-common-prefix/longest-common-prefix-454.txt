class Solution {

    public String longestCommonPrefix(String[] strs) {
        //verifico se a string esta vazia
        if (strs == null || strs.length == 0) {
            return "";
        }
        // Usa a primeira string como ponto de partida
        String prefix = strs[0];
        // Itera sobre as outras strings
        for (int i = 1; i < strs.length; i++) {
            // Enquanto o prefixo n\xE3o \xE9 encontrado, reduz o prefixo
            while (strs[i].indexOf(prefix) != 0) {
                prefix = prefix.substring(0, prefix.length() - 1);
                // Se o prefixo se tornar vazio, n\xE3o h\xE1 prefixo comum
                if (prefix.isEmpty()) {
                    return "";
                }
            }
        }
        return prefix;
    }
}
