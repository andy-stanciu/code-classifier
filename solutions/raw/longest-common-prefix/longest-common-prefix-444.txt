class Solution {

    public String longestCommonPrefix(String[] strs) {
        String s = "";
        int j = 0;
        outer: while (j < strs[0].length()) {
            int c = 1;
            for (int i = 1; i < strs.length && j < strs[i].length(); i++) {
                if (strs[i].charAt(j) == strs[0].charAt(j)) {
                    c++;
                } else {
                    break outer;
                }
            }
            if (c == strs.length) {
                s += String.valueOf(strs[0].charAt(j));
            }
            j++;
        }
        return s;
    }
}
