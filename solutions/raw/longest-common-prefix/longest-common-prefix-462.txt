class Solution {

    public String longestCommonPrefix(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        String strn = strs[0];
        String strn2 = "";
        int len = 0;
        for (int i = 1; i < strs.length; i++) {
            strn2 = "";
            char[] ch1 = strs[i].toCharArray();
            char[] ch2 = strn.toCharArray();
            if (strs[i].length() < strn.length()) {
                len = strs[i].length();
            } else {
                len = strn.length();
            }
            for (int j = 0; j < len; j++) {
                if (ch1[j] == ch2[j]) {
                    strn2 = strn2 + ch1[j];
                } else {
                    break;
                }
            }
            strn = strn2;
        }
        return strn;
    }
}
