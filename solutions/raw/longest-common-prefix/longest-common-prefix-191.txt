class Solution {

    public static Boolean isCommon(String[] strs, char curChar, int i) {
        Boolean isCommon = true;
        for (int j = 0; j < strs.length; j++) {
            if (strs[j].charAt(i) != curChar) {
                isCommon = false;
                break;
            }
        }
        return isCommon;
    }

    public static String shortestString(String[] strs, int idx, String shortestStr) {
        if (idx >= strs.length) {
            return shortestStr;
        }
        if (shortestStr.length() > strs[idx].length()) {
            shortestStr = strs[idx];
        }
        return shortestString(strs, idx + 1, shortestStr);
    }

    public String longestCommonPrefix(String[] strs) {
        String commonStr = "";
        String shortestStr = shortestString(strs, 0, strs[0]);
        if (strs.length == 1) {
            return strs[0];
        } else if (strs.length == 0) {
            return "";
        } else if (shortestStr.length() == 0) {
            return "";
        }
        for (int i = 0; i < shortestStr.length(); i++) {
            char curChar = strs[0].charAt(i);
            if (!isCommon(strs, curChar, i)) {
                break;
            } else {
                commonStr += curChar;
            }
        }
        return commonStr;
    }
}
