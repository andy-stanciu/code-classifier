class Solution {

    public String longestCommonPrefix(String[] strs) {
        String prefix = strs[0];
        for (String s : strs) {
            if (s.equals(prefix)) {
                continue;
            }
            int end = prefix.length();
            while (end >= 0) {
                prefix = prefix.substring(0, end);
                if (s.startsWith(prefix)) {
                    break;
                }
                end--;
            }
        }
        return prefix;
    }
}
