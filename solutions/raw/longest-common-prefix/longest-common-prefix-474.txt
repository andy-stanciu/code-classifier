class Solution {

    public String longestCommonPrefix(String[] strs) {
        String d = "";
        Arrays.sort(strs);
        String s1 = strs[0];
        String s2 = strs[strs.length - 1];
        int idx = 0;
        while (s1.length() > idx && s2.length() > idx) {
            if (s1.charAt(idx) == s2.charAt(idx)) {
                d = d + s1.charAt(idx);
                idx++;
            } else {
                break;
            }
        }
        return d;
    }
}
