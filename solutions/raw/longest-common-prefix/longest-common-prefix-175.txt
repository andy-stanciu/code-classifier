class Solution {

    public String longestCommonPrefix(String[] strs) {
        String str = strs[strs.length - 1];
        for (int i = 0; i < strs.length - 1; i++) {
            while (strs[i].indexOf(str) != 0) {
                str = str.substring(0, str.length() - 1);
                //System.out.println(str);
            }
        }
        return str;
    }
}
