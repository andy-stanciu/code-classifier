class Solution {

    public String longestCommonPrefix(String[] strs) {
        Arrays.sort(strs);
        String firstWord = strs[0], lastWord = strs[strs.length - 1];
        int idx = 0;
        while (idx < firstWord.length() && idx < lastWord.length()) {
            if (firstWord.charAt(idx) == lastWord.charAt(idx)) {
                idx++;
            } else {
                break;
            }
        }
        return firstWord.substring(0, idx);
    }
}
