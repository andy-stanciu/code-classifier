class Solution {

    public String longestCommonPrefix(String[] strs) {
        int length = strs[0].length();
        String result = "";
        for (int i = 0; i <= length; i++) {
            String prefix = strs[0].substring(0, i);
            for (int j = 0; j < strs.length; j++) {
                if (!strs[j].startsWith(prefix)) {
                    return result;
                }
            }
            result = prefix;
        }
        return result;
    }
}
