class Solution {

    public String longestCommonPrefix(String[] strs) {
        String pre = "";
        Arrays.sort(strs);
        String first = strs[0], last = strs[strs.length - 1];
        for (int i = 0; i < Math.min(first.length(), last.length()); i++) {
            if (first.charAt(i) != last.charAt(i))
                return pre;
            pre += first.charAt(i);
        }
        return pre;
    }
}
