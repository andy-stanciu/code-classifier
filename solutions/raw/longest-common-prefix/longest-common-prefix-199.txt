class Solution {

    public String longestCommonPrefix(String[] strs) {
        String s = strs[0];
        int end = s.length();
        for (int i = 0; i < strs.length; i++) {
            String temp = strs[i];
            while (end > 0 && !temp.startsWith(s.substring(0, end))) {
                end--;
            }
        }
        return s.substring(0, end);
    }
}
