class Solution {

    public String longestCommonPrefix(String[] strs) {
        Arrays.sort(strs);
        StringBuilder sb = new StringBuilder("");
        int count = 0;
        for (int i = 0; i < strs[0].length(); i++) {
            if (strs[0].charAt(i) != strs[strs.length - 1].charAt(i)) {
                break;
            } else {
                count++;
            }
        }
        if (count == 0) {
            return "";
        } else {
            return strs[0].substring(0, count);
        }
    }
}
