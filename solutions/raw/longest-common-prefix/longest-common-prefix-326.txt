class Solution {

    public String longestCommonPrefix(String[] strs) {
        StringBuilder ans1 = new StringBuilder(strs[0]);
        for (int i = 1; i < strs.length; i++) {
            StringBuilder temp = new StringBuilder();
            for (int j = 0; j < Math.min(ans1.length(), strs[i].length()); j++) {
                if (ans1.charAt(j) == strs[i].charAt(j)) {
                    temp.append(ans1.charAt(j));
                } else {
                    break;
                }
            }
            ans1 = temp;
            if (ans1.length() == 0) {
                return "";
            }
        }
        return ans1.toString();
    }
}
