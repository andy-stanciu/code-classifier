class Solution {

    public String longestCommonPrefix(String[] strs) {
        int i = 0, j = 0;
        for (String ans = ""; ; ) {
            for (j = 0; j + 1 < strs.length && i < Math.min(strs[j].length(), strs[j + 1].length()) && strs[j].charAt(i) == strs[j + 1].charAt(i); ++j) ;
            if (j + 1 == strs.length && strs[0].length() > i)
                ans += strs[0].charAt(i++);
            else
                return ans;
        }
    }
}
