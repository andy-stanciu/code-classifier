class Solution {

    public String longestCommonPrefix(String[] strs) {
        //Take prefix word
        String prefix = strs[0];
        int n = strs.length;
        // loop
        for (int i = 1; i < n; i++) {
            String word = strs[i];
            // deleting last character to find the prefix.
            while (!word.startsWith(prefix)) {
                if (prefix.length() == 1)
                    return "";
                prefix = prefix.substring(0, prefix.length() - 1);
            }
        }
        // return your prefix :/
        return prefix;
    }
}
