class Solution {

    private String running_prefix = "";

    // upon careful review, this variable is unnecessary :D
    private String temp_prefix = "";

    public String longestCommonPrefix(String[] strs) {
        String smallest_string = findSmallestString(strs);
        if (smallest_string.length() == 0) {
            return "";
        }
        for (int letter_index = smallest_string.length() - 1; letter_index >= 0; --letter_index) {
            temp_prefix = smallest_string;
            if (prefixIsShared(letter_index, strs)) {
                return smallest_string;
            }
            smallest_string = smallest_string.substring(0, letter_index);
        }
        return "";
    }

    private boolean prefixIsShared(int cur_index, String[] strs) {
        for (int word_index = 0; word_index < strs.length; ++word_index) {
            // guh very long condition.
            // this is just like if prefix in word matches our current prefix.
            // wait why couldn\'t i have just put running_prefix in here.
            if (!(strs[word_index].substring(0, cur_index + 1).equals(temp_prefix))) {
                return false;
            }
        }
        return true;
    }

    private String findSmallestString(String[] strs) {
        String cur_word = strs[0];
        for (String word : strs) {
            // look it\'s a for each loop!!!
            if (word.length() < cur_word.length()) {
                cur_word = word;
            }
        }
        return cur_word;
    }
}
