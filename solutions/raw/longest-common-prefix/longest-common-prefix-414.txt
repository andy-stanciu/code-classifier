class Solution {

    public String longestCommonPrefix(String[] strs) {
        String word = strs[0];
        int len = word.length();
        for (int i = 0; i < strs.length; i++) {
            if (!strs[i].startsWith(word)) {
                i = 0;
                word = word.substring(0, --len);
            }
        }
        return word;
    }
}
