class Solution {

    public String longestCommonPrefix(String[] strs) {
        // Sort the array of strings lexicographically
        Arrays.sort(strs);
        // Take the first string after sorting
        String s1 = strs[0];
        // Take the last string after sorting
        String s2 = strs[strs.length - 1];
        // Initialize an index to traverse the strings
        int index = 0;
        while (index < s1.length() && index < s2.length()) {
            if (s1.charAt(index) == s2.charAt(index)) {
                // Compare characters at the same index in s1 and s2
                // Move to the next character
                index++;
            } else {
                // If characters don\'t match, break out of the loop
                break;
            }
        }
        // Return the substring of s1 from index 0 to the index where characters stopped matching
        return s1.substring(0, index);
    }
}
