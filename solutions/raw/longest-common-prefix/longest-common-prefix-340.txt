class Solution {

    public String longestCommonPrefix(String[] strs) {
        int i = 0;
        Arrays.sort(strs);
        String A = strs[0];
        String B = strs[strs.length - 1];
        while (i < A.length() && i < B.length()) {
            if (A.charAt(i) == B.charAt(i)) {
                i++;
            } else {
                break;
            }
        }
        return A.substring(0, i);
    }
}
