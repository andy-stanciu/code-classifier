class Solution {

    public String longestCommonPrefix(String[] strs) {
        Arrays.sort(strs);
        String frst = strs[0];
        String last = strs[strs.length - 1];
        String ans = "";
        int a = Math.min(frst.length(), last.length());
        for (int i = 0; i < a; i++) {
            if (frst.charAt(i) != last.charAt(i)) {
                break;
            }
            ans += frst.charAt(i);
        }
        return ans;
    }
}
