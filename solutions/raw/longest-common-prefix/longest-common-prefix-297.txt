class Solution {

    public String longestCommonPrefix(String[] strs) {
        String prefix = strs[0];
        int m = strs.length;
        for (int a = 1; a < m; a++) {
            int n = prefix.length();
            for (int i = 0; i < n; i++) {
                if (i >= strs[a].length() || prefix.charAt(i) != strs[a].charAt(i)) {
                    prefix = prefix.substring(0, i);
                    break;
                }
            }
        }
        return prefix;
    }
}
