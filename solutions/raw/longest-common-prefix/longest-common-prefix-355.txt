class Solution {

    public String longestCommonPrefix(String[] strs) {
        String ans = "";
        if (strs.length == 1)
            return strs[0];
        int itr = 0;
        // lexicographically sorting would reduce number of comparisons,i.e.,
        // we can simply compare 2 extremes, the smallest and longest string.
        Arrays.sort(strs);
        String first = strs[0];
        String last = strs[strs.length - 1];
        int len = Math.min(first.length(), last.length());
        // iterate over the 1st word
        while (itr < len) {
            if (first.charAt(itr) == last.charAt(itr)) {
                ans += first.charAt(itr) + "";
            } else {
                return ans;
            }
            itr++;
        }
        return ans;
    }
}
