class Solution {

    public String longestCommonPrefix(String[] strs) {
        String s = strs[0];
        int l = s.length();
        for (int i = 0; i < strs.length; i++) {
            while (!strs[i].startsWith(s.substring(0, l))) {
                if (--l < 0)
                    break;
            }
        }
        return s.substring(0, l);
    }
}
