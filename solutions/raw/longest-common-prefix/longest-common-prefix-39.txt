class Solution {

    public String longestCommonPrefix(String[] s) {
        if (s == null || s.length == 0) {
            //This is in case the array is empty or one of its elements is empty
            return "";
        }
        String p = s[0];
        for (int i = 0; i < p.length(); i++) {
            char ch = p.charAt(i);
            for (int j = 1; j < s.length; j++) {
                if (i >= s[j].length() || s[j].charAt(i) != ch) {
                    //This is in case the array is empty or one of its elements is empty
                    return p.substring(0, i);
                }
            }
        }
        //This is in case there was no difference and the match was complete
        return p;
    }
}
