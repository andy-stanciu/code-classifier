class Solution {

    public int longestConsecutive(int[] nums) {
        Arrays.sort(nums);
        int currentcount = 1;
        int maxcount = 1;
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                continue;
            } else if (nums[i] + 1 == nums[i + 1]) {
                currentcount++;
            } else {
                if (currentcount > maxcount) {
                    maxcount = currentcount;
                }
                currentcount = 1;
            }
            if (currentcount > maxcount) {
                maxcount = currentcount;
            }
        }
        return maxcount;
    }
}
