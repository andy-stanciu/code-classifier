class Solution {

    public int longestConsecutive(int[] nums) {
        // Initialize a HashSet to store the unique elements
        HashSet<Integer> set = new HashSet<>();
        int max = 0;
        // Add all elements from the array to the HashSet
        for (int i : nums) {
            set.add(i);
        }
        // Iterate through each element in the HashSet
        for (int i : set) {
            // Check if the current element is the start of a sequence
            if (!set.contains(i - 1)) {
                // Start counting the length of the sequence
                int count = 1;
                // Continue to check for the next consecutive elements
                while (set.contains(i + 1)) {
                    count++;
                    // Move to the next element in the sequence
                    i++;
                }
                // Update the maximum length if the current sequence is longer
                max = Math.max(max, count);
            }
        }
        // Return the maximum length
        return max;
    }
}
