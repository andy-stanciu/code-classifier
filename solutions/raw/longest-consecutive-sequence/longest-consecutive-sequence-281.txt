class Solution {

    public int longestConsecutive(int[] nums) {
        int max = 0;
        HashSet<Integer> st = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            int x = nums[i];
            st.add(x);
        }
        for (int num : st) {
            int x = num;
            int count = 1;
            if (!st.contains(x - 1)) {
                while (st.contains(x + 1)) {
                    count++;
                    x++;
                }
                max = Math.max(max, count);
            }
        }
        return max;
    }
}
