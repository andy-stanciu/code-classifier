class Solution {

    public int longestConsecutive(int[] nums) {
        HashSet<Integer> s1 = new HashSet<>();
        int maxStrik = 0;
        for (int num : nums) s1.add(num);
        for (int n : s1) {
            if (!s1.contains(n - 1)) {
                int curr = n;
                int currStriek = 1;
                while (s1.contains(curr + 1)) {
                    curr++;
                    currStriek++;
                }
                maxStrik = Math.max(maxStrik, currStriek);
            }
        }
        return maxStrik;
    }
}
