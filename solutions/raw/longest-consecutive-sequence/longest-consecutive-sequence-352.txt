class Solution {

    public int longestConsecutive(int[] nums) {
        Arrays.sort(nums);
        int lastSmall = Integer.MIN_VALUE, count = 0, maxCount = 0;
        for (int i = 0; i < nums.length; i++) {
            if (lastSmall + 1 == nums[i]) {
                count++;
                lastSmall++;
            } else if (lastSmall != nums[i]) {
                count = 1;
                lastSmall = nums[i];
            }
            if (count > maxCount)
                maxCount = count;
        }
        return maxCount;
    }
}
