class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        int longest = 1;
        // create a set data structure
        HashSet<Integer> set = new HashSet<>();
        // put the elements in the set
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        for (int first : set) {
            // check if the previous element is present or not
            if (!set.contains(first - 1)) {
                //find the consecutive elements
                int cnt = 1;
                int x = first;
                while (set.contains(x + 1)) {
                    x = x + 1;
                    cnt = cnt + 1;
                }
                //compare longest with the current count
                longest = Math.max(longest, cnt);
            }
        }
        return longest;
    }
}
