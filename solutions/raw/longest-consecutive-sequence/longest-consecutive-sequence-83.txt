class Solution {

    public int longestConsecutive(int[] nums) {
        int[] sortedNums = Arrays.stream(nums).distinct().sorted().toArray();
        if (sortedNums.length == 1)
            return 1;
        int consecutiveCounter = 1, maxCount = 0;
        for (int i = 1; i < sortedNums.length; i++) {
            if (sortedNums[i - 1] + 1 == sortedNums[i])
                consecutiveCounter++;
            else
                consecutiveCounter = 1;
            maxCount = Math.max(maxCount, consecutiveCounter);
        }
        return maxCount;
    }
}
