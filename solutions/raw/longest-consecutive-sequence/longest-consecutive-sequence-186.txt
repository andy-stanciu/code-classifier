class Solution {

    public int longestConsecutive(int[] nums) {
        // Variable to store the length of the longest consecutive sequence
        int result = 0;
        if (nums.length > 0) {
            // Check if the array length is less than 10,000
            if (nums.length < 10000) {
                // Sort the array
                Arrays.sort(nums);
                // Counter for the current consecutive sequence
                int current = 0;
                for (int i = 1; i < nums.length; i++) {
                    // Check for unique elements
                    if (nums[i] != nums[i - 1]) {
                        // Check if the current number is consecutive
                        if (nums[i] - nums[i - 1] == 1) {
                            // Increment the current sequence length
                            current++;
                        } else {
                            // Update the result if the current sequence is longer
                            if (current + 1 > result) {
                                result = current + 1;
                            }
                            // Reset current counter
                            current = 0;
                        }
                    }
                }
                // Final check for the last sequence
                if (current + 1 > result) {
                    result = current + 1;
                }
            } else {
                // For larger arrays, find min and max
                int min = Integer.MAX_VALUE;
                int max = Integer.MIN_VALUE;
                for (int num : nums) {
                    if (num > max) {
                        // Update max
                        max = num;
                    }
                    if (num < min) {
                        // Update min
                        min = num;
                    }
                }
                // Create a bit array to track presence of numbers
                byte[] bits = new byte[max - min + 1];
                for (int num : nums) {
                    // Mark the presence of the number
                    bits[num - min] = 1;
                }
                // Counter for the current consecutive sequence
                int current = 0;
                for (byte bit : bits) {
                    if (bit > 0) {
                        // Increment for consecutive presence
                        current++;
                    } else {
                        // Update the result if the current sequence is longer
                        if (current > result) {
                            result = current;
                        }
                        // Reset current counter
                        current = 0;
                    }
                }
                // Final check for the last sequence
                if (current > result) {
                    result = current;
                }
            }
        }
        // Return the length of the longest consecutive sequence
        return result;
    }
}
