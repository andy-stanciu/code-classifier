class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        Arrays.sort(nums);
        int count = 1, max = 1;
        int j = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == j + 1) {
                count++;
                j = nums[i];
            } else if (nums[i] != j) {
                max = Math.max(count, max);
                count = 1;
                j = nums[i];
            }
        }
        return Math.max(max, count);
    }
}
