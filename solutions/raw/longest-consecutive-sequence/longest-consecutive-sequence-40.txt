class Solution {

    public int longestConsecutive(int[] nums) {
        int n = nums.length;
        if (//edge case
        n == 0)
            return 0;
        Arrays.sort(nums);
        //setting the initial subsequence length
        int count = 1, maxi = 1;
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1])
                //ignoring duplicates
                continue;
            if (nums[i] - nums[i + 1] == -1) {
                //checking consecutive elements
                count++;
            } else {
                //storing maximum length of the sequence
                maxi = Math.max(maxi, count);
                count = 1;
            }
        }
        maxi = Math.max(maxi, count);
        return maxi;
    }
}
