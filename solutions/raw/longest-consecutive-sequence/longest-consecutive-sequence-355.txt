class Solution {

    public int longestConsecutive(int[] nums) {
        int inputLength = nums.length;
        if (inputLength <= 1) {
            return inputLength;
        }
        // O (n)
        Set<Integer> uniquesNums = new HashSet<>();
        for (int index = 0; index < inputLength; index++) {
            int currentNum = nums[index];
            uniquesNums.add(currentNum);
        }
        int longestSequence = 1;
        // O(n)
        for (int i = 0; i < inputLength; i++) {
            int k = nums[i];
            if (uniquesNums.remove(k)) {
                int lowVal = k - 1;
                int upVal = k + 1;
                int currentSequenceLength = 1;
                while (uniquesNums.remove(lowVal)) {
                    currentSequenceLength += 1;
                    lowVal -= 1;
                }
                while (uniquesNums.remove(upVal)) {
                    currentSequenceLength += 1;
                    upVal += 1;
                }
                longestSequence = Math.max(longestSequence, currentSequenceLength);
            }
        }
        return longestSequence;
    }
}
