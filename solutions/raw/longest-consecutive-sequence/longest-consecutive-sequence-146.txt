class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0)
            return 0;
        Arrays.sort(nums);
        int currentLength = 1, longestLength = 1;
        int startLongest = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1])
                continue;
            else if (nums[i] == nums[i - 1] + 1) {
                currentLength++;
            } else {
                if (currentLength > longestLength) {
                    longestLength = currentLength;
                    startLongest = nums[i - currentLength];
                }
                currentLength = 1;
            }
        }
        if (currentLength > longestLength)
            longestLength = currentLength;
        return longestLength;
    }
}
