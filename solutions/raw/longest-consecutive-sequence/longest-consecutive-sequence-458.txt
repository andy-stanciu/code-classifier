class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length <= 1) {
            return nums.length;
        }
        Arrays.sort(nums);
        int n = 1, max = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] + 1 == nums[i + 1] || nums[i] == nums[i + 1]) {
                n++;
                if (nums[i] == nums[i + 1])
                    n--;
                if (n > max)
                    max = n;
            } else {
                n = 1;
            }
        }
        return max;
    }
}
