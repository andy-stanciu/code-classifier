class Solution {

    public int longestConsecutive(int[] nums) {
        // Create a set from the input array
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        //ans
        int maxi = 0;
        for (int num : nums) {
            // If the set still contains the number
            if (set.contains(num)) {
                // Remove the current number from the set
                set.remove(num);
                int cnt = 1;
                // Check and remove consecutive numbers less than num
                int currentNum = num;
                while (set.contains(currentNum - 1)) {
                    set.remove(currentNum - 1);
                    currentNum--;
                    cnt++;
                }
                // Reset currentNum to num and check and remove consecutive numbers greater than num
                currentNum = num;
                while (set.contains(currentNum + 1)) {
                    set.remove(currentNum + 1);
                    currentNum++;
                    cnt++;
                }
                // Update the maximum length found
                maxi = Math.max(maxi, cnt);
            }
        }
        return maxi;
    }
}
