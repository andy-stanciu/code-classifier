class Solution {

    Set<Integer> seen;

    Set<Integer> set;

    public int longestConsecutive(int[] nums) {
        set = new HashSet();
        seen = new HashSet();
        for (int num : nums) {
            set.add(num);
        }
        int max = 0;
        for (int i : set) {
            System.out.println(i);
            max = Math.max(max, dfs(i));
        }
        return max;
    }

    private int dfs(int i) {
        if (!set.contains(i) || seen.contains(i)) {
            return 0;
        }
        seen.add(i);
        return dfs(i + 1) + dfs(i - 1) + 1;
    }
}
