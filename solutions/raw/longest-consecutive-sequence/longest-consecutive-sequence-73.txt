class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        Arrays.sort(nums);
        int count = 1;
        int max = 1;
        for (int y = 0; y < nums.length - 1; y++) {
            if (nums[y + 1] - nums[y] == 1) {
                count++;
            } else if (nums[y + 1] - nums[y] != 0) {
                count = 1;
            }
            max = Math.max(max, count);
        }
        return max;
    }
}
