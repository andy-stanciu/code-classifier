class Solution {

    public int longestConsecutive(int[] nums) {
        int n = nums.length;
        if (n == 0 || n == 1) {
            return n;
        }
        Arrays.sort(nums);
        int count = 0, max = 0;
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1])
                continue;
            int curr = nums[i];
            int next = nums[i + 1];
            if ((curr + 1) == next) {
                count++;
            } else {
                count = 0;
            }
            max = Math.max(max, count);
        }
        return max + 1;
    }
}
