class Solution {

    public int longestConsecutive(int[] arr) {
        int n = arr.length;
        if (n == 0 || arr == null)
            return 0;
        Arrays.sort(arr);
        int curr = 1;
        int ans = 1;
        for (int i = 0; i < n - 1; i++) {
            if (arr[i] == arr[i + 1] - 1) {
                curr += 1;
            } else if (arr[i] == arr[i + 1]) {
                continue;
            } else {
                curr = 1;
            }
            ans = Math.max(ans, curr);
        }
        return ans;
    }
}
