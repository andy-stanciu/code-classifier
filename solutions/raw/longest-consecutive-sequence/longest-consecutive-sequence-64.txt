class Solution {

    public int longestConsecutive(int[] nums) {
        int size = nums.length;
        if (size == 0)
            return 0;
        // sorting the array in O(N log(N))
        Arrays.sort(nums);
        int longest = 0;
        int current_longest = 1;
        for (int i = 1; i < size; i++) {
            if (nums[i] == nums[i - 1]) {
                //same element as previous skip this
                continue;
            } else if (nums[i] == nums[i - 1] + 1) {
                //current element is consecutive so increase it.
                current_longest++;
            } else {
                longest = Math.max(longest, current_longest);
                // we can\'t extend so update longest as the maximum of the both and reset the current_longest.
                current_longest = 1;
            }
        }
        return Math.max(longest, current_longest);
    }
}
