class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length < 1)
            return 0;
        Arrays.sort(nums);
        int currentLength = 1;
        int maxLength = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1])
                continue;
            if (nums[i] - nums[i - 1] == 1)
                currentLength++;
            else {
                maxLength = Math.max(maxLength, currentLength);
                currentLength = 1;
            }
        }
        return Math.max(maxLength, currentLength);
    }
}
