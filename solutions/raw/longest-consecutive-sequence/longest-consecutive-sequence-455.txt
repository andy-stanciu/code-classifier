class Solution {

    public int longestConsecutive(int[] nums) {
        int size = nums.length;
        if (size == 0) {
            return 0;
        }
        Set<Integer> numbers = new HashSet<>();
        for (int i = 0; i < size; i++) {
            numbers.add(nums[i]);
        }
        int result = 0;
        for (int i = 0; i < size; i++) {
            int num = nums[i];
            int count = 1;
            while (numbers.contains(--num)) {
                numbers.remove(num);
                count++;
            }
            num = nums[i];
            while (numbers.contains(++num)) {
                numbers.remove(num);
                count++;
            }
            result = Math.max(result, count);
        }
        return result;
    }
}
