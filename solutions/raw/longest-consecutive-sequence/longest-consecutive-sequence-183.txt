class Solution {

    public int longestConsecutive(int[] numsarr) {
        Set<Integer> set = new HashSet<>();
        int maxsequence = 0;
        for (int num : numsarr) {
            set.add(num);
        }
        for (int num : set) {
            if (!set.contains(num - 1)) {
                int length = 1;
                while (set.contains(num + length)) {
                    length++;
                }
                maxsequence = Math.max(maxsequence, length);
            }
        }
        return maxsequence;
    }
}
