class Solution {

    public int longestConsecutive(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int longest = 0;
        for (int num : nums) {
            if (!set.contains(num))
                continue;
            int backward = num - 1;
            int forward = num + 1;
            while (set.contains(backward)) {
                set.remove(backward);
                backward--;
            }
            while (set.contains(forward)) {
                set.remove(forward);
                forward++;
            }
            longest = Math.max(longest, forward - backward - 1);
        }
        return longest;
    }
}
