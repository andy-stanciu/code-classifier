class Solution {

    public int longestConsecutive(int[] nums) {
        Arrays.sort(nums);
        if (nums.length == 1) {
            return 1;
        }
        int c = 1, maxc = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - 1 == nums[i - 1]) {
                c++;
            } else if (nums[i] != nums[i - 1]) {
                c = 1;
            }
            maxc = Math.max(c, maxc);
        }
        return maxc;
    }
}
