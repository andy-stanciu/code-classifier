class Solution {

    public int longestConsecutive(int[] nums) {
        Set<Integer> s = new HashSet<Integer>();
        int max = 0;
        for (int n : nums) {
            s.add(n);
        }
        for (int i = 0; i < nums.length; i++) {
            if (!s.contains(nums[i] - 1)) {
                int num = nums[i];
                int count = 1;
                while (s.contains(num + 1)) {
                    num += 1;
                    count++;
                }
                max = Math.max(max, count);
            }
        }
        return max;
    }
}
