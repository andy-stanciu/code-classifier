class Solution {

    public int longestConsecutive(int[] nums) {
        int maxSeq = 0;
        Set<Integer> s = new HashSet();
        for (int num : nums) {
            s.add(num);
        }
        int curr;
        int res = 0;
        for (int i : nums) {
            curr = 1;
            if (s.contains(i - 1) == false) {
                while (s.contains(i + curr)) curr++;
                res = Math.max(res, curr);
            }
        }
        return res;
    }
}
