class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums == null || nums.length == 0)
            return 0;
        Set<Integer> set = new HashSet<>();
        for (int x : nums) {
            set.add(x);
        }
        int maxStreak = 0;
        for (int num : set) {
            if (!set.contains(num - 1)) {
                int curr_num = num;
                int curr_streak = 1;
                while (set.contains(curr_num + 1)) {
                    curr_streak++;
                    curr_num++;
                }
                maxStreak = Math.max(maxStreak, curr_streak);
            }
        }
        return maxStreak;
    }
}
