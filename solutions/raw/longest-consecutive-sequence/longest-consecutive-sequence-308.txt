class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 1) {
            return 1;
        }
        Arrays.sort(nums);
        int max = 0;
        int count = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] + 1 == nums[i + 1]) {
                count++;
                max = Math.max(max, count);
            } else if (nums[i] == nums[i + 1]) {
                max = Math.max(max, count);
            } else {
                count = 1;
            }
        }
        return max;
    }
}
