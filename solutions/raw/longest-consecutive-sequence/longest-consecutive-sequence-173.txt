class Solution {

    public int longestConsecutive(int[] nums) {
        int count = 0;
        HashSet<Integer> set = new HashSet<>();
        for (int n : nums) {
            set.add(n);
        }
        for (int i = 0; i < nums.length; i++) {
            if (set.contains(nums[i] - 1)) {
                continue;
            }
            int currentlength = 0;
            while (set.contains(nums[i])) {
                nums[i]++;
                currentlength++;
            }
            count = Math.max(count, currentlength);
        }
        return count;
    }
}
