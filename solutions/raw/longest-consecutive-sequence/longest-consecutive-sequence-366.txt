class Solution {

    public int longestConsecutive(int[] nums) {
        HashSet<Integer> hSet = new HashSet<>();
        for (int num : nums) {
            hSet.add(num);
        }
        int longest = 0;
        for (int num : nums) {
            if (!hSet.contains(num - 1)) {
                int count = 1;
                while (hSet.contains(num + 1)) {
                    count++;
                    num++;
                }
                longest = Math.max(longest, count);
            }
            if (longest > nums.length / 2)
                break;
        }
        return longest;
    }
}
