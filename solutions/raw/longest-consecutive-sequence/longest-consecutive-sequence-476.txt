class Solution {

    public int longestConsecutive(int[] nums) {
        // add all elements to set
        HashSet<Integer> set = new HashSet<>();
        int longest = 0;
        if (nums.length == 0)
            return 0;
        for (int i : nums) {
            set.add(i);
        }
        //check prev and ince the num and cnt ++
        for (int it : set) {
            if (!set.contains(it - 1)) {
                int cnt = 1;
                int x = it;
                while (set.contains(x + 1)) {
                    x += 1;
                    cnt += 1;
                }
                longest = Math.max(longest, cnt);
            }
        }
        return longest;
    }
}
