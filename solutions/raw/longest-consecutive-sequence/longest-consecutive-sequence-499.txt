class Solution {

    public int longestConsecutive(int[] nums) {
        // use a set, put all the number into
        // the set, then check all num in nums to see
        // if they have neighbors
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int res = 0;
        for (int num : nums) {
            if (set.size() == 0) {
                break;
            }
            if (set.contains(num)) {
                int count = 1;
                int left = num - 1;
                int right = num + 1;
                set.remove(num);
                //System.out.println(num + " " + left + " " + right);
                while (set.contains(left)) {
                    count += 1;
                    set.remove(left);
                    left -= 1;
                }
                while (set.contains(right)) {
                    count += 1;
                    set.remove(right);
                    right += 1;
                }
                res = Math.max(res, count);
            }
        }
        return res;
    }
}
