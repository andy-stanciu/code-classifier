class Solution {

    public int longestIncreasingConsecutive(int[] nums) {
        if (nums.length == 0)
            return 0;
        int maxLen = 1;
        int currentLen = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1] + 1) {
                // Continue the sequence
                currentLen++;
            } else {
                // Reset if not consecutive
                currentLen = 1;
            }
            maxLen = Math.max(maxLen, currentLen);
        }
        return maxLen;
    }
}
