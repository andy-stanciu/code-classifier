class Solution {

    public int longestConsecutive(int[] nums) {
        int n = nums.length;
        int count = 0;
        int lastsmallest = Integer.MIN_VALUE;
        int longest = 0;
        Arrays.sort(nums);
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastsmallest) {
                count++;
                lastsmallest = nums[i];
            } else if (nums[i] != lastsmallest) {
                lastsmallest = nums[i];
                count = 1;
            }
            longest = Math.max(longest, count);
        }
        return longest;
    }
}
