class Solution {

    public int longestConsecutive(int[] nums) {
        Set<Integer> u = new HashSet<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            u.add(nums[i]);
        }
        int longest = 0;
        for (int i : u) {
            if (!u.contains(i - 1)) {
                int cnt = 1;
                int x = i;
                while (u.contains(x + 1)) {
                    x += 1;
                    cnt += 1;
                }
                longest = Math.max(longest, cnt);
            }
        }
        return longest;
    }
}
