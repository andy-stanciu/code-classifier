class Solution {

    public int longestConsecutive(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) set.add(num);
        int best = 0;
        for (int num : nums) {
            if (set.contains(num - 1))
                continue;
            int length = 0;
            for (int i = 0; set.contains(num + i); i++) length++;
            best = Math.max(best, length);
        }
        return best;
    }
}
