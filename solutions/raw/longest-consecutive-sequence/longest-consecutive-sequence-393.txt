class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0)
            return 0;
        Set<Integer> set = new HashSet<>();
        for (int a : nums) set.add(a);
        int max = 1;
        for (int i = 0; i < nums.length; i++) {
            if (!set.contains(nums[i] - 1)) {
                int next = nums[i] + 1;
                int len = 1;
                while (set.contains(next)) {
                    len++;
                    next++;
                    max = Math.max(max, len);
                }
            }
        }
        return max;
    }
}
