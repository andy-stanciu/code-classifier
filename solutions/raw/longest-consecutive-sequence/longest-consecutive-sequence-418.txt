class Solution {

    public int longestConsecutive(int[] nums) {
        HashSet<Integer> setOfNumbers = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            setOfNumbers.add(nums[i]);
        }
        int actualCount = 0;
        int currentElement = 0;
        for (Integer num : setOfNumbers) {
            if (!setOfNumbers.contains(num - 1)) {
                currentElement = num;
                int count = 1;
                while (setOfNumbers.contains(currentElement + 1)) {
                    currentElement++;
                    count++;
                }
                if (count > actualCount) {
                    actualCount = count;
                }
            }
        }
        return actualCount;
    }
}
