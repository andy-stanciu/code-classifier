class Solution {

    public int longestConsecutive(int[] nums) {
        int MaxLen = 0;
        Set<Integer> set = new HashSet<>();
        for (int element : nums) set.add(element);
        for (int SetElm : set) {
            if (!set.contains(SetElm - 1)) {
                int len = 1;
                while (set.contains(SetElm + len)) len++;
                MaxLen = Math.max(MaxLen, len);
            }
        }
        return MaxLen;
    }
}
