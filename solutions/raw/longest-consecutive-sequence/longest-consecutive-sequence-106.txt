class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0)
            return 0;
        Set<Integer> numsSet = new HashSet<Integer>();
        for (int i : nums) {
            numsSet.add(i);
        }
        int longest = 1;
        for (int num : nums) {
            if (!numsSet.contains(num - 1)) {
                int count = 1;
                while (numsSet.contains(num + 1)) {
                    //consecutive numbers; increment by 1
                    num++;
                    count++;
                }
                //get max from current count and longest so far
                longest = Math.max(longest, count);
            }
        }
        return longest;
    }
}
