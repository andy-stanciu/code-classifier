class Solution {

    public int longestConsecutive(int[] nums) {
        Map<Integer, Integer> mp = new HashMap<>();
        int longestConsecutive = 0;
        for (int i = 0; i < nums.length; i++) {
            mp.put(nums[i], mp.getOrDefault(nums[i], 0) + 1);
        }
        for (int i = 0; i < nums.length; i++) {
            if (!mp.containsKey(nums[i] - 1)) {
                int longestSubConsecutive = 1;
                int condition = 0;
                while (true) {
                    condition++;
                    if (mp.containsKey(nums[i] + condition)) {
                        longestSubConsecutive++;
                    } else {
                        if (longestSubConsecutive > longestConsecutive)
                            longestConsecutive = longestSubConsecutive;
                        break;
                    }
                }
            }
        }
        return longestConsecutive;
    }
}
