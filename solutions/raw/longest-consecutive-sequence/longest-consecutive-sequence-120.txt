class Solution {

    public int longestConsecutive(int[] nums) {
        int longCon = 0;
        //why set because of Union set
        Set<Integer> num_set = new HashSet<>();
        for (int num : nums) {
            num_set.add(num);
        }
        for (int i = 0; i < nums.length; i++) {
            int count = 1;
            //look left side of set
            int num = nums[i];
            while (num_set.contains(--num)) {
                count++;
                num_set.remove(num);
            }
            //look right side of set
            num = nums[i];
            while (num_set.contains(++num)) {
                count++;
                num_set.remove(num);
            }
            longCon = Math.max(longCon, count);
        }
        return longCon;
    }
}
