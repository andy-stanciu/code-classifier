class Solution {

    public int longestConsecutive(int[] arr) {
        int size = arr.length;
        if (size < 2)
            return size;
        // First, Simply sort the array
        Arrays.sort(arr);
        int count = 1;
        int max = 1;
        for (int i = 1; i < size; i++) {
            // If number is consecutive then increase the counter
            if (arr[i] - arr[i - 1] == 1)
                count++;
            else // if same number then ignore
            if (arr[i] - arr[i - 1] == 0)
                continue;
            else // if the sequence breaks then restart the counter
            {
                max = Math.max(max, count);
                count = 1;
            }
        }
        // In case if the longest sequence ends at last
        return Math.max(max, count);
    }
}
