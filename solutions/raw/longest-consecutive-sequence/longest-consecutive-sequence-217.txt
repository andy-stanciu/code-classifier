class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0)
            return 0;
        //creating map and insert into it
        HashMap<Integer, Boolean> map = new HashMap<>();
        int longestsubseq = 0;
        for (int i : nums) {
            map.put(i, true);
        }
        for (int i : nums) {
            if (!map.containsKey(i - 1)) {
                int currentNum = i;
                int seqlength = 1;
                while (map.containsKey(currentNum + 1)) {
                    currentNum += 1;
                    seqlength += 1;
                }
                longestsubseq = Math.max(longestsubseq, seqlength);
            }
        }
        return longestsubseq;
    }
}
