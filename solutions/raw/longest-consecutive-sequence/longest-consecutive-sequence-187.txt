// Google Interview Question
class Solution {

    public int longestConsecutive(int[] a) {
        Arrays.sort(a);
        int n = a.length;
        int maxCount = 1;
        int count = 1;
        if (//if the array is empty than return 0
        a == null || n == 0) {
            return 0;
        }
        for (int i = 0; i <= n - 2; i++) {
            if (a[i] == a[i + 1]) {
            } else if (a[i] + 1 == a[i + 1]) {
                count++;
                if (count > maxCount) {
                    maxCount = count;
                }
            } else {
                count = 1;
            }
        }
        return maxCount;
    }
}
