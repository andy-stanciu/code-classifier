class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0)
            return 0;
        Arrays.sort(nums);
        int longest = 1;
        int prev = Integer.MIN_VALUE;
        int currentLongest = 1;
        for (int i : nums) {
            if (i == prev)
                continue;
            if (i == prev + 1) {
                currentLongest++;
                if (currentLongest > longest)
                    longest = currentLongest;
            } else {
                currentLongest = 1;
            }
            prev = i;
        }
        return longest;
    }
}
