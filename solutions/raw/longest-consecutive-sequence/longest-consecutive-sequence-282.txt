class Solution {

    public int longestConsecutive(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int index = 0;
        int max = 0;
        boolean startSeq = false;
        while (index < nums.length) {
            int prevNum = nums[index] - 1;
            int count = 0;
            if (!set.contains(prevNum)) {
                prevNum++;
                while (true) {
                    if (set.contains(prevNum)) {
                        count++;
                        prevNum++;
                    } else {
                        break;
                    }
                }
                max = Math.max(max, count);
            }
            index++;
        }
        return max;
    }
}
