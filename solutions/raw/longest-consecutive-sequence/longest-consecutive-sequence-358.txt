class Solution {

    public int longestConsecutive(int[] nums) {
        Arrays.sort(nums);
        int maxSequence = 1;
        int currentSequence = 1;
        int n = nums.length;
        if (n == 0)
            return 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] == nums[i - 1])
                continue;
            if (nums[i] - 1 == nums[i - 1]) {
                currentSequence++;
                if (currentSequence > maxSequence)
                    maxSequence = currentSequence;
            } else
                currentSequence = 1;
        }
        return maxSequence;
    }
}
