class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        // it\'ll remove duplicates
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int maxLen = 0;
        for (int num : nums) {
            if (!set.contains(num - 1)) {
                int currEl = num;
                int len = 1;
                while (set.contains(currEl + 1)) {
                    len++;
                    currEl = currEl + 1;
                }
                maxLen = Math.max(len, maxLen);
            }
        }
        return maxLen;
    }
}
