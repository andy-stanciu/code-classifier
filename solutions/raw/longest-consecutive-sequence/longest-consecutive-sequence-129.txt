class Solution {

    public int longestConsecutive(int[] nums) {
        HashSet<Integer> graph = new HashSet(nums.length);
        for (int n : nums) graph.add(n);
        HashSet<Integer> visited = new HashSet(nums.length);
        int ans = 0;
        Iterator iter = graph.iterator();
        while (iter.hasNext()) {
            int temp = 1;
            int num = (int) iter.next();
            if (visited.contains(num))
                continue;
            int n_d = num;
            visited.add(num);
            while (graph.contains(++num)) {
                visited.add(num);
                temp++;
            }
            while (graph.contains(--n_d)) {
                visited.add(n_d);
                temp++;
            }
            ans = Math.max(ans, temp);
        }
        return ans;
    }
}
