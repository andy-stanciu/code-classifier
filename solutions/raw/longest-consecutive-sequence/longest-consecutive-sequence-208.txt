class Solution {

    public int longestConsecutive(int[] nums) {
        int numsSize = nums.length;
        if (numsSize == 0) {
            return 0;
        }
        int count = 0;
        Arrays.sort(nums);
        count = 1;
        int l = 1;
        for (int i = 0; i < numsSize - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                continue;
            }
            if (nums[i] == nums[i + 1] - 1) {
                count++;
            } else {
                if (count > l) {
                    l = count;
                }
                count = 1;
            }
        }
        if (count > l) {
            l = count;
        }
        return l;
    }
}
