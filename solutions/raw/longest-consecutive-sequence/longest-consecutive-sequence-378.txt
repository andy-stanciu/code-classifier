class Solution {

    public int longestConsecutive(int[] nums) {
        int n = nums.length;
        if (n <= 1) {
            return n;
        }
        Arrays.sort(nums);
        int maxSequence = 0;
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] == nums[i - 1] + 1) {
                count++;
            } else if (nums[i] == nums[i - 1]) {
                // Handle If there are multiple occurance of the same value
                continue;
            } else {
                maxSequence = Math.max(maxSequence, count);
                count = 1;
            }
        }
        return Math.max(maxSequence, count);
    }
}
