class Solution {

    public int longestConsecutive(int[] arr) {
        Arrays.sort(arr);
        int count = 1;
        int max = Integer.MIN_VALUE;
        int ls = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] - 1 == max) {
                count++;
                max = arr[i];
            } else if (arr[i] != max) {
                max = arr[i];
                count = 1;
            }
            ls = Math.max(ls, count);
        }
        return ls;
    }
}
