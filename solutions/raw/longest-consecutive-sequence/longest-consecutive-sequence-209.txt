class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        //1234100200
        Arrays.sort(nums);
        int count = 1;
        int maxLength = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1] - 1) {
                count++;
            } else if (nums[i] == nums[i + 1]) {
                continue;
            } else {
                if (count > maxLength) {
                    maxLength = count;
                }
                count = 1;
            }
        }
        if (count > maxLength) {
            maxLength = count;
        }
        return maxLength;
    }
}
