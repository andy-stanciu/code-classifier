class Solution {

    public int longestConsecutive(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return 0;
        Arrays.sort(nums);
        int longest = 1;
        // Start counting from 1 for the first element
        int cnt = 1;
        int lastsmaller = nums[0];
        for (int i = 1; i < n; i++) {
            if (nums[i] == lastsmaller + 1) {
                cnt++;
                lastsmaller = nums[i];
            } else if (nums[i] != lastsmaller) {
                // Skip duplicates
                cnt = 1;
                lastsmaller = nums[i];
            }
            longest = Math.max(longest, cnt);
        }
        return longest;
    }
}
