class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        Arrays.sort(nums);
        int longestSeq = 1;
        int currSeq = 1;
        int curr = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == curr + 1) {
                currSeq++;
                curr = nums[i];
            } else if (nums[i] == curr) {
                continue;
            } else {
                curr = nums[i];
                longestSeq = Math.max(currSeq, longestSeq);
                currSeq = 1;
            }
        }
        longestSeq = Math.max(currSeq, longestSeq);
        return longestSeq;
    }
}
