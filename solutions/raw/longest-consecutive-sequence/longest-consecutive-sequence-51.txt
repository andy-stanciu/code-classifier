class Solution {

    public int longestConsecutive(int[] nums) {
        Set<Long> set = new HashSet<>();
        for (int x : nums) {
            set.add(x * 1l);
        }
        int ans = 0;
        for (int x : nums) {
            long val = x;
            if (!set.contains(val))
                continue;
            long prev = val - 1;
            long next = val + 1;
            int len = 1;
            while (set.contains(prev)) {
                len++;
                set.remove(prev);
                prev--;
            }
            while (set.contains(next)) {
                len++;
                set.remove(next);
                next++;
            }
            ans = Math.max(ans, len);
        }
        return ans;
    }
}
