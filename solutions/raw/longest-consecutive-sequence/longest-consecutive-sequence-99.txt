class Solution {

    public int longestConsecutive(int[] nums) {
        if (nums.length <= 1)
            return nums.length;
        int localMax = 1, max = 1;
        Set<Integer> map = new TreeSet<Integer>();
        for (int i : nums) map.add(i);
        Integer[] arr = map.toArray(new Integer[0]);
        for (int i = 1; i < arr.length; i++) {
            if ((arr[i] - arr[i - 1]) == 1)
                localMax++;
            else {
                max = Math.max(localMax, max);
                localMax = 1;
            }
        }
        max = Math.max(localMax, max);
        return max;
    }
}
