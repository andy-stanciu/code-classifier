class Solution {

    public int longestConsecutive(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        if (n == 0)
            return 0;
        int longest = 1;
        int currentcount = 0;
        int lastsmaller = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastsmaller) {
                currentcount += 1;
                lastsmaller = nums[i];
            } else if (nums[i] != lastsmaller) {
                currentcount = 1;
                lastsmaller = nums[i];
            }
            longest = Math.max(longest, currentcount);
        }
        return longest;
    }
}
