class Solution {

    public int longestConsecutive(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (Integer i : nums) {
            set.add(i);
        }
        int maxlength = 0;
        for (int i = 0; i < nums.length; i++) {
            int currelem = nums[i];
            int currleng = 1;
            if (!set.contains(currelem + 1)) {
                while (set.contains(currelem - 1)) {
                    currelem -= 1;
                    currleng++;
                }
                maxlength = Math.max(currleng, maxlength);
            }
        }
        return maxlength;
    }
}
