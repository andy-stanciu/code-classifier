class Solution {

    public int longestConsecutive(int[] arr) {
        int size = arr.length;
        if (size < 2)
            return size;
        // put all the numbers in set
        Set<Integer> set = new HashSet<>();
        for (int e : arr) set.add(e);
        int count = 1, max = 1;
        for (int e : set) {
            // Move to the number which will be the start of the consecutive sequence
            if (!set.contains(e - 1)) {
                // and check for all the consecutive numbers whether available in set or not until the last number of consecutive sequence found
                int temp = e + 1;
                while (set.contains(temp)) {
                    count++;
                    temp++;
                }
                // save the longest sequence and restart the counter
                max = Math.max(max, count);
                count = 1;
            }
        }
        return Math.max(max, count);
    }
}
