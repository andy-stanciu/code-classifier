class Solution {

    public int longestConsecutive(int[] nums) {
        Set<Integer> setNums = new HashSet<>();
        for (int num : nums) {
            setNums.add(num);
        }
        int longestConsecutive = 0;
        for (int num : setNums) {
            if (!setNums.contains(num - 1)) {
                int currentNum = num;
                int lengthCurrentConsecutive = 1;
                while (setNums.contains(currentNum + 1)) {
                    lengthCurrentConsecutive++;
                    currentNum++;
                }
                longestConsecutive = Math.max(longestConsecutive, lengthCurrentConsecutive);
            }
        }
        return longestConsecutive;
    }
}
