class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        List<List<String>> God = new ArrayList<>();
        HashMap<String, String> Map = new HashMap<>();
        for (String x : strs) {
            char[] tem = x.toCharArray();
            Arrays.sort(tem);
            String temp = new String(tem);
            if (Map.containsKey(temp)) {
                String Key = Map.get(temp);
                for (List<String> list : God) {
                    for (String xx : list) {
                        if (xx.equals(Key)) {
                            list.add(x);
                            Map.put(temp, x);
                            break;
                        }
                    }
                }
            } else {
                Map.put(temp, x);
                List<String> templ = new ArrayList<>();
                templ.add(x);
                God.add(templ);
            }
        }
        return God;
    }
}
