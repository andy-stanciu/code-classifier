class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        List<List<String>> result = new ArrayList();
        Map<String, ArrayList<String>> hm = new HashMap<String, ArrayList<String>>();
        for (int i = 0; i < strs.length; i++) {
            char[] val = strs[i].toCharArray();
            Arrays.sort(val);
            String strVal = new String(val);
            ArrayList<String> ls = hm.getOrDefault(strVal, new ArrayList<String>());
            ls.add(strs[i]);
            hm.put(strVal, ls);
        }
        for (Map.Entry<String, ArrayList<String>> entry : hm.entrySet()) {
            result.add(entry.getValue());
        }
        return result;
    }
}
