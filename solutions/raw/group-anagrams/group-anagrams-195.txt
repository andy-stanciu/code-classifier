class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> mapSortedToActual = new HashMap<>();
        for (String str : strs) {
            char[] letters = str.toCharArray();
            Arrays.sort(letters);
            String sorted = new String(letters);
            mapSortedToActual.computeIfAbsent(sorted, (key) -> new ArrayList<>()).add(str);
        }
        return new ArrayList<>(mapSortedToActual.values());
    }
}
