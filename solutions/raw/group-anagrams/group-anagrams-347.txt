class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> hm = new HashMap<>();
        for (int i = 0; i < strs.length; i++) {
            String s = strs[i];
            //convert into charArray
            char[] arr = s.toCharArray();
            //sort that charArray
            Arrays.sort(arr);
            //again form a string
            String sortedstring = String.valueOf(arr);
            //if sortedstring is not present make a newlist and put into hashmap
            if (!hm.containsKey(sortedstring)) {
                hm.put(sortedstring, new ArrayList<>());
            }
            //if present we take key as sortedstring and add values as inputstring
            hm.get(sortedstring).add(s);
        }
        return new ArrayList<>(hm.values());
    }
}
