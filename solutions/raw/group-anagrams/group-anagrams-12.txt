class Solution {

    public List<List<String>> groupAnagrams(String[] ss) {
        Map<String, List<String>> map = new HashMap<>();
        String x;
        List<String> l;
        for (String s : ss) {
            x = getSortedStr(s);
            l = map.getOrDefault(x, new ArrayList<String>());
            l.add(s);
            map.put(x, l);
        }
        return new ArrayList<>(map.values());
    }

    private String getSortedStr(String s) {
        char[] c = s.toCharArray();
        Arrays.sort(c);
        return new String(c);
    }
}
