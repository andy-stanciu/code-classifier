class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> groupAnagrams = new HashMap<>();
        for (String str : strs) {
            char[] charArr = str.toCharArray();
            Arrays.sort(charArr);
            String sortStr = String.valueOf(charArr);
            if (!groupAnagrams.containsKey(sortStr)) {
                groupAnagrams.put(sortStr, new ArrayList<>());
            }
            groupAnagrams.get(sortStr).add(str);
        }
        return new ArrayList<>(groupAnagrams.values());
    }
}
