class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        // V.IMP concept of this question is the declaration of hashMap
        HashMap<HashMap<Character, Integer>, List<String>> mainMap = new HashMap<>();
        int n = strs.length;
        for (String s : strs) {
            HashMap<Character, Integer> map = new HashMap<>();
            for (int j = 0; j < s.length(); j++) {
                char ch = s.charAt(j);
                map.put(ch, map.getOrDefault(ch, 0) + 1);
            }
            if (mainMap.containsKey(map)) {
                mainMap.get(map).add(s);
            } else {
                List<String> val = new ArrayList<>();
                val.add(s);
                mainMap.put(map, val);
            }
        }
        List<List<String>> anagram = new ArrayList<>();
        for (List str : mainMap.values()) anagram.add(str);
        return anagram;
    }
}
