class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        if (strs.length == 0) {
            return new ArrayList();
        }
        Map<String, List<String>> map = new HashMap<>();
        for (String word : strs) {
            // Sort the current word
            char[] charArr = word.toCharArray();
            Arrays.sort(charArr);
            // Pass the sorted char array to string (For the map\'s key)
            String orderedW = new String(charArr);
            // Check if the key already exists
            if (!map.containsKey(orderedW))
                map.put(orderedW, new ArrayList<String>());
            // Add the new word to the map
            map.get(orderedW).add(word);
        }
        // Get the grouped anagrams from the map
        return new ArrayList<List<String>>(map.values());
    }
}
