class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        //Map the "sorted group of characters" to "all words made by this combination"
        HashMap<String, List<String>> anagramMap = new HashMap<>();
        for (int i = 0; i < strs.length; i++) {
            char[] charArr = strs[i].toCharArray();
            Arrays.sort(charArr);
            String sortedChars = new String(charArr);
            if (!anagramMap.containsKey(sortedChars)) {
                anagramMap.put(sortedChars, new ArrayList<>());
            }
            anagramMap.get(sortedChars).add(strs[i]);
        }
        return new ArrayList<>(anagramMap.values());
    }
}
