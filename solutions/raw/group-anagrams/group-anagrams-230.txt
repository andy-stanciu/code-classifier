class Solution {

    Map<String, List<String>> mp;

    public void clubAnagrams(String str) {
        char[] nStr = str.toCharArray();
        Arrays.sort(nStr);
        String sortedString = new String(nStr);
        if (mp.containsKey(sortedString)) {
            mp.get(sortedString).add(str);
        } else {
            List<String> aList = new ArrayList<>();
            aList.add(str);
            mp.put(sortedString, aList);
        }
    }

    public List<List<String>> groupAnagrams(String[] strs) {
        mp = new HashMap<>();
        for (String str : strs) {
            clubAnagrams(str);
        }
        System.out.println(mp);
        List<List<String>> answerList = new ArrayList<>();
        for (Map.Entry<String, List<String>> entry : mp.entrySet()) {
            answerList.add(entry.getValue());
        }
        return answerList;
    }
}
