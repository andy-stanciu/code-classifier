class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        int n = strs.length;
        List<List<String>> ls = new ArrayList<>();
        if (n == 0)
            return ls;
        Map<String, List<String>> lst = new HashMap<>();
        for (int i = 0; i < n; i++) {
            char[] temp = strs[i].toCharArray();
            Arrays.sort(temp);
            String sortedWord = new String(temp);
            if (!lst.containsKey(sortedWord)) {
                lst.put(sortedWord, new ArrayList<>());
            }
            lst.get(sortedWord).add(strs[i]);
        }
        ls.addAll(lst.values());
        return ls;
    }
}
