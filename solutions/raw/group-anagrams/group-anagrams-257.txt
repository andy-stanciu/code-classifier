class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        int n = strs.length;
        HashMap<String, List<String>> h = new HashMap<>();
        for (int i = 0; i < n; i++) {
            char[] ch = strs[i].toCharArray();
            Arrays.sort(ch);
            String s = new String(ch);
            if (!h.containsKey(s)) {
                h.put(s, new ArrayList<>());
            }
            h.get(s).add(strs[i]);
        }
        return new ArrayList<>(h.values());
    }
}
