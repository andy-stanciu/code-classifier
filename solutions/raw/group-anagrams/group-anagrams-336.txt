class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        var resMap = new HashMap<String, List<String>>();
        for (int i = 0; i < strs.length; i++) {
            var arr = strs[i].toCharArray();
            Arrays.sort(arr);
            var list = resMap.getOrDefault(String.valueOf(arr), new ArrayList<String>());
            list.add(strs[i]);
            resMap.put(String.valueOf(arr), list);
        }
        return resMap.values().stream().toList();
    }
}
