class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            //sort the given string, eg: ate --> aet
            char[] c = str.toCharArray();
            Arrays.sort(c);
            String sortedStr = new String(c);
            // get \'aet\' from map, if not present, then get new empty arraylist and add the new string to it.
            List<String> group = map.getOrDefault(sortedStr, new ArrayList<>());
            group.add(str);
            //keep the key as sorted string and update it with new arrayList value
            map.put(sortedStr, group);
        }
        return new ArrayList<>(map.values());
    }
}
