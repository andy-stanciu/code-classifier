class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> anagram = new HashMap<>();
        for (String s : strs) {
            //sorting string
            char[] c = s.toCharArray();
            Arrays.sort(c);
            String key = String.valueOf(c);
            // checking inside map key is available if not available insert new empty list
            if (!anagram.containsKey(key)) {
                anagram.put(key, new ArrayList<>());
            }
            // if already key is available fetch list and add original string
            anagram.get(key).add(s);
        }
        return new ArrayList<>(anagram.values());
    }
}
