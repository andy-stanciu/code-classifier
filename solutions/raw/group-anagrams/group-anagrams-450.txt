class Solution {

    public static List<List<String>> groupAnagrams(String[] strs) {
        ArrayList<List<String>> result = new ArrayList<>();
        HashMap<String, List<String>> anagram_map = new HashMap<>();
        for (int i = 0; i < strs.length; i++) {
            char[] str_arr = strs[i].toCharArray();
            Arrays.sort(str_arr);
            String key = new String(str_arr);
            if (!anagram_map.containsKey(key)) {
                List<String> val = new ArrayList<>();
                val.add(strs[i]);
                anagram_map.put(key, val);
            } else {
                List<String> val = anagram_map.get(key);
                val.add(strs[i]);
            }
        }
        for (List<String> val : anagram_map.values()) {
            result.add(val);
        }
        return result;
    }
}
