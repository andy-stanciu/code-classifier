class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> agrmMap = new HashMap<>();
        for (String word : strs) {
            char[] chars = word.toCharArray();
            Arrays.sort(chars);
            String sorted = new String(chars);
            if (!agrmMap.containsKey(sorted)) {
                agrmMap.put(sorted, new ArrayList<>());
            }
            agrmMap.get(sorted).add(word);
        }
        return new ArrayList<>(agrmMap.values());
    }
}
