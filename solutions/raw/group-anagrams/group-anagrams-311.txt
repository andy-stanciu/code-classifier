class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> n = new HashMap<>();
        for (String s : strs) {
            char[] ch = s.toCharArray();
            Arrays.sort(ch);
            String s1 = String.valueOf(ch);
            if (!n.containsKey(s1)) {
                n.put(s1, new ArrayList<>());
            }
            n.get(s1).add(s);
        }
        return new ArrayList<>(n.values());
    }
}
