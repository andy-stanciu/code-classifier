class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        //create the empty hashmap named anagramgroup
        Map<String, List<String>> anagramgroup = new HashMap<>();
        for (String string : strs) {
            //convert the string into character array and name it as chars
            char[] chars = string.toCharArray();
            //sort the chars
            Arrays.sort(chars);
            //create a new string name canonical to store the chars
            String canonical = new String(chars);
            //if anagramgroup contains canonical add string to the associated list
            if (anagramgroup.containsKey(canonical)) {
                anagramgroup.get(canonical).add(string);
            } else {
                //create a new list named group and add string to it, and put the pair of canonical list into anagram group
                List<String> group = new ArrayList<>();
                group.add(string);
                anagramgroup.put(canonical, group);
            }
        }
        //return the array list containing the values(list of anagrams) from anagramgroup
        return new ArrayList<>(anagramgroup.values());
    }
}
