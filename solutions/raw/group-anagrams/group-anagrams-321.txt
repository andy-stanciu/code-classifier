class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        String[] modifiedArr = new String[strs.length];
        for (int i = 0; i < strs.length; i++) {
            modifiedArr[i] = strs[i];
        }
        for (int i = 0; i < strs.length; i++) {
            String str = strs[i];
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            str = new String(chars);
            modifiedArr[i] = str;
        }
        int listIndex = 0;
        List<List<String>> output = new LinkedList<>();
        Map<String, Integer> map = new HashMap<>();
        for (int i = 0; i < modifiedArr.length; i++) {
            String str = modifiedArr[i];
            if (!map.containsKey(str)) {
                map.put(str, listIndex);
                List<String> list = new LinkedList<>();
                list.add(strs[i]);
                output.add(list);
                listIndex++;
            } else {
                int index = map.get(str);
                List<String> list = output.get(index);
                list.add(strs[i]);
            }
        }
        return output;
    }
}
