class Solution {

    public String sortString(String str) {
        char[] chars = str.toCharArray();
        Arrays.sort(chars);
        return new String(chars);
    }

    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String, List<String>> sortedStrings = new HashMap<String, List<String>>();
        for (String str : strs) {
            String sorted = this.sortString(str);
            List<String> elements = sortedStrings.get(sorted);
            elements = Optional.ofNullable(elements).orElse(new ArrayList<String>());
            elements.add(str);
            sortedStrings.put(sorted, elements);
        }
        return new ArrayList(sortedStrings.values());
    }
}
