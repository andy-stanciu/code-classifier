class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        List<List<String>> result = new ArrayList<>();
        Map<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            char[] arr = str.toCharArray();
            Arrays.sort(arr);
            String strResult = new String(arr);
            if (map.containsKey(strResult)) {
                List<String> list1 = map.get(strResult);
                list1.add(str);
                map.put(strResult, list1);
            } else {
                List<String> list1 = new ArrayList<>();
                list1.add(str);
                map.put(strResult, list1);
            }
        }
        for (List<String> str : map.values()) {
            result.add(str);
        }
        return result;
    }
}
