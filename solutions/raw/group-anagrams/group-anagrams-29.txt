class Solution {

    public List<List<String>> groupAnagrams(String[] word) {
        Map<String, List<String>> m = new HashMap<>();
        for (String s : word) {
            //[\'e\',\'a\',\'t\']
            char[] ch = s.toCharArray();
            //[\'a\',\'e\',\'t\']
            Arrays.sort(ch);
            //[aet]
            String sw = new String(ch);
            if (!m.containsKey(sw)) {
                //[aet, [ ]]
                m.put(sw, new ArrayList<>());
            }
            //[aet,[eat]]
            m.get(sw).add(s);
        }
        return new ArrayList<>(m.values());
    }
}
