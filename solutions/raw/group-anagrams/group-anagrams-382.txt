class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        //using hashmap to map each string with its anagrams
        //key= string   ;  value= list of valid anagrams
        HashMap<String, ArrayList<String>> map = new HashMap<>();
        //iterate over each string from string array
        for (String str : strs) {
            //converting respective string to its character array
            char[] sChar = str.toCharArray();
            //Sorting respective character array
            Arrays.sort(sChar);
            //storing new modified sorted string from each respective string
            String stringKey = new String(sChar);
            //checks if string-key is present in hashmap
            //if it doesn\'t, it creates new key-value pair of that string-key,
            //with string as key and empty list as value (for storing anagrams)
            if (!map.containsKey(stringKey)) {
                map.put(stringKey, new ArrayList<String>());
            }
            //if string is present in map it simply adds that original string,
            //to value of string-key pair
            map.get(stringKey).add(str);
        }
        return new ArrayList<>(map.values());
    }
}
