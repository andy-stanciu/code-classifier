class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        return new ArrayList<>(Arrays.stream(strs).collect(Collectors.groupingBy((s) -> {
            String t = new String(s);
            char[] ca = t.toCharArray();
            Arrays.sort(ca);
            return String.valueOf(ca);
        })).values());
    }
}
