class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        // First approach
        // Sort every string and store in it another array
        // If same elements, insert element at that index in a list
        // Second approach
        // Sort every string and store it as key in the HashMap
        // and value as the list of string
        // Lowercase English letters
        // hashMap - key, value as list of string
        Map<String, List<String>> result = new HashMap<>();
        for (String s : strs) {
            char[] charArray = s.toCharArray();
            Arrays.sort(charArray);
            String StringElem = String.valueOf(charArray);
            if (result.containsKey(StringElem)) {
                List<String> existingValue = result.get(StringElem);
                existingValue.add(s);
                result.put(StringElem, new ArrayList<>(existingValue));
            } else {
                result.put(StringElem, new ArrayList<>(Arrays.asList(s)));
            }
        }
        return new ArrayList<>(result.values());
    }
}
