class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        List<List<String>> ret = new ArrayList<>();
        if (strs == null || strs.length == 0) {
            return ret;
        }
        HashMap<String, Integer> arrIndex = new HashMap<>();
        for (String s : strs) {
            char[] letters = s.toCharArray();
            Arrays.sort(letters);
            String orderedWord = new String(letters);
            if (arrIndex.containsKey(orderedWord)) {
                ret.get(arrIndex.get(orderedWord)).add(s);
            } else {
                List<String> newSet = new ArrayList<>();
                newSet.add(s);
                int setIndex = ret.size();
                ret.add(newSet);
                arrIndex.put(orderedWord, setIndex);
            }
        }
        return ret;
    }
}
