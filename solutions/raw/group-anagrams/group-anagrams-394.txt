class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        //Initialize ArrayList to store groups of anagrams; Initialize HashMap to check if strings are anagrams and store them
        List<List<String>> groups = new ArrayList<>();
        HashMap<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            //Sort characters of the strings
            char[] strChars = str.toCharArray();
            Arrays.sort(strChars);
            String sortedStr = new String(strChars);
            //Create a new list in the HashMap and group anagrams together based on the sorted strings
            if (!map.containsKey(sortedStr)) {
                map.put(sortedStr, new ArrayList<>());
            }
            map.get(sortedStr).add(str);
        }
        //Add the groups of anagrams to the result list
        for (List<String> group : map.values()) {
            groups.add(group);
        }
        return groups;
    }
}
