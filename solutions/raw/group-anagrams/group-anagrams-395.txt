class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> anagramGroupsMap = new HashMap<>();
        for (String str : strs) {
            String strReverse = str;
            char[] reverseArr = strReverse.toCharArray();
            Arrays.sort(reverseArr);
            // System.out.println(str+" "+strReverse+" "+anagramGroupsMap);
            anagramGroupsMap.computeIfAbsent(new String(reverseArr), k -> new ArrayList<>()).add(str);
        }
        return anagramGroupsMap.values().stream().collect(Collectors.toList());
    }
}
