class Solution {

    private String stringSort(String str) {
        char[] temp = str.toCharArray();
        Arrays.sort(temp);
        return new String(temp);
    }

    public List<List<String>> groupAnagrams(String[] strs) {
        int n = strs.length;
        HashMap<String, List<String>> storeGroup = new HashMap<>();
        for (String str : strs) {
            String sortedString = stringSort(str);
            if (!storeGroup.containsKey(sortedString)) {
                storeGroup.put(sortedString, new ArrayList<>());
            }
            storeGroup.get(sortedString).add(str);
        }
        return new ArrayList<>(storeGroup.values());
    }
}
