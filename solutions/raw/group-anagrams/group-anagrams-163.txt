class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> stmap = new HashMap<>();
        for (String s : strs) {
            char[] st = s.toCharArray();
            Arrays.sort(st);
            //to convert arr to string
            String key = String.valueOf(st);
            if (!stmap.containsKey(key)) {
                stmap.put(key, new ArrayList<>());
            }
            stmap.get(key).add(s);
        }
        List<List<String>> res = new ArrayList<>();
        for (Map.Entry<String, List<String>> strin : stmap.entrySet()) {
            res.add(strin.getValue());
        }
        return res;
    }
}
