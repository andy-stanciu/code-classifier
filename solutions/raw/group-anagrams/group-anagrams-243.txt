class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String, ArrayList<String>> hashMap = new HashMap<>();
        List<List<String>> result = new ArrayList<>();
        for (String str : strs) {
            char[] tmpArray = str.toCharArray();
            Arrays.sort(tmpArray);
            String sortedString = new String(tmpArray);
            ArrayList<String> tmp;
            if (hashMap.get(sortedString) == null) {
                tmp = new ArrayList<String>();
            } else {
                tmp = hashMap.get(sortedString);
            }
            tmp.add(str);
            hashMap.put(sortedString, tmp);
        }
        for (ArrayList<String> l : hashMap.values()) {
            result.add(l);
        }
        return result;
    }
}
