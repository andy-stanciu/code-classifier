class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String, ArrayList<String>> sortedWordsToAnagrams = new HashMap<>();
        for (String word : strs) {
            char[] wordChars = word.toCharArray();
            Arrays.sort(wordChars);
            String sortedWord = new String(wordChars);
            if (!sortedWordsToAnagrams.containsKey(sortedWord)) {
                sortedWordsToAnagrams.put(sortedWord, new ArrayList<>());
            }
            sortedWordsToAnagrams.get(sortedWord).add(word);
        }
        List<List<String>> anagrams = new ArrayList<>();
        for (String key : sortedWordsToAnagrams.keySet()) {
            anagrams.add(sortedWordsToAnagrams.get(key));
        }
        return anagrams;
    }
}
