class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String, List> set = new HashMap<String, List>();
        for (String words : strs) {
            char[] chrs = words.toCharArray();
            Arrays.sort(chrs);
            String s = new String(chrs);
            System.out.println(s);
            if (set.containsKey(s)) {
                set.get(s).add(words);
            } else {
                set.put(s, new ArrayList<String>(Arrays.asList(words)));
            }
        }
        return new ArrayList(set.values());
    }
}
