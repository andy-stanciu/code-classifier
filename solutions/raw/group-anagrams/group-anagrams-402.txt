class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        // Create a map to store sorted strings and their corresponding anagrams
        Map<String, List<String>> map = new HashMap<>();
        // Iterate through each string in the array
        for (String str : strs) {
            // Sort the characters of the string
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            String sortedStr = new String(chars);
            // If the sorted string is already in the map, add the current string to its list of anagrams
            // Otherwise, create a new list and put it in the map
            if (map.containsKey(sortedStr)) {
                map.get(sortedStr).add(str);
            } else {
                List<String> list = new ArrayList<>();
                list.add(str);
                map.put(sortedStr, list);
            }
        }
        // Convert the values of the map to a list of lists and return
        return new ArrayList<>(map.values());
    }
}
