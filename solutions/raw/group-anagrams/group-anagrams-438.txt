class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        /*
            Option 1:
                1.sort each word O(N*mlogm) = O(N)
                2. add to map with original index
                3. convert map to list of lists by mapping via original index
            Option 2: is it possible to make a char order agnostic checksum?
        */
        Map<String, List<String>> anagrams = new HashMap<>();
        for (String word : strs) {
            char[] sortedChars = word.toCharArray();
            Arrays.sort(sortedChars);
            String wordKey = String.valueOf(sortedChars);
            List<String> anagramGroup = anagrams.getOrDefault(wordKey, new ArrayList<String>());
            anagramGroup.add(word);
            anagrams.put(wordKey, anagramGroup);
        }
        List<List<String>> anagramGroups = new ArrayList(anagrams.size());
        for (List<String> anagramGroup : anagrams.values()) {
            anagramGroups.add(anagramGroup);
        }
        return anagramGroups;
    }
}
