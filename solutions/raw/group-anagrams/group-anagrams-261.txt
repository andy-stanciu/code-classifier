class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> allSets = new HashMap<>();
        for (String str : strs) {
            char[] charArray = str.toCharArray();
            Arrays.sort(charArray);
            String sortedStr = new String(charArray);
            allSets.computeIfAbsent(sortedStr, x -> new ArrayList<>()).add(str);
        }
        return new ArrayList<>(allSets.values());
    }
}
