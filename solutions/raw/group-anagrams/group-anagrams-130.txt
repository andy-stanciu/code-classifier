class Solution {

    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> groupedanagram = new HashMap<>();
        for (String word : strs) {
            //convert to char array
            var charString = word.toCharArray();
            //now sort it
            Arrays.sort(charString);
            //now convert back after sorting
            var chartostring = new String(charString);
            //Now save this sorted string as key in our list
            groupedanagram.putIfAbsent(chartostring, new ArrayList<>());
            //now get the key
            groupedanagram.get(chartostring).add(word);
        }
        return new ArrayList<>(groupedanagram.values());
    }
}
