class Solution {

    boolean hoursKoko(int[] a, int H, int speed) {
        int hour = 0;
        for (int banana : a) {
            // ceil division
            hour += (banana + speed - 1) / speed;
        }
        return hour <= H;
    }

    public int minEatingSpeed(int[] piles, int h) {
        int n = piles.length;
        int ans = 0;
        int low = 1;
        int high = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (hoursKoko(piles, h, mid)) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return ans;
    }
}
