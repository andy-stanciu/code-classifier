class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int left = 1;
        int right = Integer.MAX_VALUE;
        while (left < right) {
            int mid = left + right >>> 1;
            if (canEat(piles, h, mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

    private boolean canEat(int[] piles, int h, int mid) {
        int time = 0;
        for (int pile : piles) {
            time += (pile + mid - 1) / mid;
        }
        return time <= h;
    }
}
