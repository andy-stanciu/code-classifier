class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int left = 1;
        int right = Arrays.stream(piles).max().getAsInt();
        while (left <= right) {
            int middle = (left + right) / 2;
            long bananasCouldEat = 0;
            for (int bananas : piles) {
                // ceilDiv();
                bananasCouldEat += (bananas + middle - 1) / middle;
            }
            if (bananasCouldEat > h) {
                left = middle + 1;
            } else {
                right = middle - 1;
            }
        }
        return left;
    }
}
