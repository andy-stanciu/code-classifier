class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int l = 1, hi = 0, max = 0;
        for (int e : piles) {
            hi = Math.max(hi, e);
        }
        while (l <= hi) {
            int mid = (l + hi) / 2;
            int ans = solve(piles, mid);
            if (ans <= h) {
                hi = mid - 1;
            } else
                l = mid + 1;
        }
        return l;
    }

    int solve(int[] piles, int time) {
        int value = 0;
        for (int i = 0; i < piles.length; i++) {
            value += Math.ceil((double) piles[i] / time);
        }
        return value;
    }
}
