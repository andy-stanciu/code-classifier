class Solution {

    private boolean canBeEaten(int[] piles, int value, int h) {
        int hours = 0;
        for (int i = 0; i < piles.length; i++) {
            hours += ((int) Math.ceil((double) piles[i] / value));
        }
        return hours <= h;
    }

    public int minEatingSpeed(int[] piles, int h) {
        Arrays.sort(piles);
        // binary search
        int low = 1, high = piles[piles.length - 1];
        // 3,4,5,6,7,8,9,10,11
        int mid = 0;
        while (low < high) {
            // mid = 4
            mid = (high + low) / 2;
            if (canBeEaten(piles, mid, h)) {
                // range can be up to high
                high = mid;
            } else {
                //
                low = mid + 1;
            }
        }
        return high;
    }
}
