class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int min = 1;
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < piles.length; i++) {
            max = Math.max(max, piles[i]);
        }
        int x = Integer.MAX_VALUE;
        while (min < max) {
            int mid = min + (max - min) / 2;
            int temp = 0;
            for (int i = 0; i < piles.length; i++) {
                double a = piles[i];
                double b = mid;
                temp += Math.ceil(a / b);
            }
            if (temp <= h) {
                max = mid;
            } else {
                min = mid + 1;
            }
        }
        return min;
    }
}
