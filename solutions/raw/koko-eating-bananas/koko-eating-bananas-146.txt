class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int low = 1;
        int high = findMax(piles);
        while (low <= high) {
            int mid = (low + high) / 2;
            int totalH = CalculateTotalHours(piles, mid);
            if (totalH <= h) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }

    private static int findMax(int[] a) {
        int maxi = Integer.MIN_VALUE;
        int n = a.length;
        for (int i = 0; i < n; i++) {
            maxi = Math.max(maxi, a[i]);
        }
        return maxi;
    }

    private static int CalculateTotalHours(int[] a, int hour) {
        int totalH = 0;
        int n = a.length;
        for (int i = 0; i < n; i++) {
            totalH += Math.ceil((double) (a[i]) / (double) (hour));
        }
        return totalH;
    }
}
