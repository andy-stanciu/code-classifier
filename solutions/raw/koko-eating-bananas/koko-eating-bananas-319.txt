class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int begin = 1, end = 1000000000;
        while (begin <= end) {
            int mid = begin + (end - begin) / 2;
            long time = 0;
            for (int pile : piles) {
                time += pile / mid + ((pile % mid == 0) ? 0 : 1);
            }
            if (time > h)
                begin = mid + 1;
            else
                end = mid - 1;
        }
        return begin;
    }
}
