class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int maxBanana = 0;
        for (int i = 0; i < piles.length; i++) {
            maxBanana = Math.max(maxBanana, piles[i]);
        }
        int low = 1;
        int high = maxBanana;
        int ans = 0;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (function(piles, mid, h) == true) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return ans;
    }

    public boolean function(int[] arr, int speed, int hours) {
        int hoursRequired = 0;
        for (int pile : arr) {
            hoursRequired += Math.ceil(pile * 1.0 / speed);
        }
        return hoursRequired <= hours;
    }
}
