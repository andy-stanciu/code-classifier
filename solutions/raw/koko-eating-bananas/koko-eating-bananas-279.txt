class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        Arrays.sort(piles);
        int start = 1;
        int end = piles[piles.length - 1];
        int res = Integer.MAX_VALUE;
        while (start <= end) {
            int i = (start + end) / 2;
            long timeTaken = 0;
            for (int j = 0; j < piles.length; j++) {
                timeTaken += (piles[j] % i == 0) ? piles[j] / i : piles[j] / i + 1;
            }
            if (timeTaken > h)
                start = i + 1;
            else if (timeTaken <= h) {
                res = Math.min(res, i);
                end = i - 1;
            }
        }
        return res;
    }
}
