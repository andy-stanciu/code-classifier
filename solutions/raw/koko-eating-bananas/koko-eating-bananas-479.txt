class Solution {

    public int calHr(int[] piles, int mid) {
        int hr = 0;
        for (int i : piles) {
            hr += Math.ceil(((double) i) / ((double) mid));
        }
        return hr;
    }

    public int minEatingSpeed(int[] piles, int h) {
        int low = 1;
        int high = piles[0];
        for (int i : piles) {
            high = high > i ? high : i;
        }
        while (low <= high) {
            int mid = low + (high - low) / 2;
            int d = calHr(piles, mid);
            if (d <= h) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
}
