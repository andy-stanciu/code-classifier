class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int max = Integer.MIN_VALUE;
        for (int val : piles) {
            //max bnanas koko can eat
            max = Math.max(max, val);
        }
        int low = 1;
        int high = max;
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (caneat(piles, mid, h)) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }

    public boolean caneat(int[] piles, int mid, int h) {
        int hours = 0;
        for (int val : piles) {
            //adding time to hours
            hours += val / mid;
            if (//bananas left
            val % mid != 0) {
                hours++;
            }
        }
        return hours <= h;
    }
}
