class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int left = 1;
        // right should contain max element from the array
        int right = getMaxElement(piles);
        while (left < right) {
            int currSpeed = left + (right - left) / 2;
            if (isFeasible(currSpeed, h, piles)) {
                // Don\'t do currSpeed-1 to lose the actual answer.
                right = currSpeed;
            } else {
                left = currSpeed + 1;
            }
        }
        return right;
    }

    // Time: O(N) - iterate over the array
    private boolean isFeasible(int speed, int hoursRemaining, int[] piles) {
        boolean isFeasible = true;
        for (int bananas : piles) {
            // ceil value
            int hoursNeeded = (int) Math.ceil((double) bananas / speed);
            hoursRemaining = hoursRemaining - hoursNeeded;
            if (hoursRemaining < 0) {
                // not feasible
                isFeasible = false;
                break;
            }
        }
        return isFeasible;
    }

    private int getMaxElement(int[] piles) {
        int right = piles[0];
        for (int pile : piles) {
            right = Math.max(right, pile);
        }
        return right;
    }
}
