class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int l = 1, r = max(piles), m = 0;
        while (l < r) {
            m = l + (r - l) / 2;
            int hours = 0;
            for (int p : piles) hours += (p + m - 1) / m;
            if (hours <= h)
                r = m;
            else
                l = m + 1;
        }
        return r;
    }

    int max(int[] A) {
        int max = A[0];
        for (int a : A) max = Math.max(a, max);
        return max;
    }
}
