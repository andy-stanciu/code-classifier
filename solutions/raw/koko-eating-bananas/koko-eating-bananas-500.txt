class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        //maximum value
        int right = Arrays.stream(piles).max().getAsInt();
        //min value
        int left = 1;
        int result = right;
        while (left <= right) {
            int k = (left + right) / 2;
            int hours = 0;
            for (int p : piles) {
                hours += Math.ceil((double) p / k);
            }
            if (hours <= h) {
                //updating result
                result = Math.min(result, k);
                right = k - 1;
            } else {
                left = k + 1;
            }
        }
        return result;
    }
}
