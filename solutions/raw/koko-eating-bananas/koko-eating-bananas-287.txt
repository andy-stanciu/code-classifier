class Solution {

    public int speed(int[] arr, int m) {
        int total = 0;
        for (int i = 0; i < arr.length; i++) {
            total += Math.ceil((double) arr[i] / m);
        }
        return total;
    }

    public int minEatingSpeed(int[] piles, int h) {
        int maxElement = Integer.MIN_VALUE;
        for (int i = 0; i < piles.length; i++) {
            maxElement = Math.max(maxElement, piles[i]);
        }
        int low = 1;
        int high = maxElement;
        // Initialize ans outside the loop
        int ans = 0;
        while (low <= high) {
            // Changed the condition to include equality
            int mid = low + (high - low) / 2;
            int totalHours = speed(piles, mid);
            if (totalHours <= h) {
                // Update ans if the current speed is acceptable
                ans = mid;
                high = mid - 1;
            } else
                low = mid + 1;
        }
        return ans;
    }
}
