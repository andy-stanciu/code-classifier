class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        Arrays.sort(piles);
        int low = 1;
        int high = 0;
        for (int n : piles) if (high < n)
            high = n;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            int tempH = 0;
            for (int n : piles) {
                tempH += n / mid;
                if (n % mid > 0)
                    tempH++;
            }
            if (tempH <= h && tempH > 0) {
                high = mid - 1;
            } else if (tempH > h || tempH < 0) {
                low = mid + 1;
            }
        }
        return low;
    }
}
