class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int low = 1, high = Arrays.stream(piles).max().getAsInt();
        //apply binary search:
        while (low <= high) {
            int mid_i = (low + high) / 2;
            int totalH = getTotalHrs_With_I_Bananas_HourlyRate(piles, mid_i);
            if (totalH <= h) {
                high = mid_i - 1;
            } else {
                low = mid_i + 1;
            }
        }
        return low;
    }

    public int getTotalHrs_With_I_Bananas_HourlyRate(int[] piles, int withIBananasHourly) {
        int totalHrs = 0;
        for (int i = 0; i < piles.length; i++) {
            totalHrs += Math.ceil(((double) piles[i]) / (double) withIBananasHourly);
        }
        return totalHrs;
    }
}
