class Solution {

    public int minEatingSpeed(int[] piles, int h) {
        int lval = 1;
        int rval = 0;
        int result = 0;
        for (int pile : piles) {
            if (pile > rval) {
                rval = pile;
            }
        }
        while (lval < rval) {
            int mid = (lval + rval) / 2;
            if (canFinish(piles, mid, h)) {
                rval = mid;
            } else {
                lval = mid + 1;
            }
        }
        return lval;
    }

    public boolean canFinish(int[] piles, int mid, int h) {
        int thours = 0;
        for (int pile : piles) {
            thours = thours + (pile + mid - 1) / mid;
        }
        return (thours <= h);
    }
}
