class Solution {

    private void preOrder(List<Integer> numbers, TreeNode root) {
        if (root == null)
            return;
        numbers.add(root.val);
        preOrder(numbers, root.left);
        preOrder(numbers, root.right);
    }

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> numbers = new ArrayList<>();
        preOrder(numbers, root);
        return numbers;
    }
}
