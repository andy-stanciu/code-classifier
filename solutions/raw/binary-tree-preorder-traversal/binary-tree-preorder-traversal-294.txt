/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> lst = new ArrayList<>();
        TreeNode node;
        if (root == null) {
            return lst;
        }
        Stack<TreeNode> stk = new Stack<>();
        stk.add(root);
        while (!stk.isEmpty()) {
            node = stk.pop();
            lst.add(node.val);
            if (node.right != null) {
                stk.add(node.right);
            }
            if (node.left != null) {
                stk.add(node.left);
            }
        }
        return lst;
    }
}
