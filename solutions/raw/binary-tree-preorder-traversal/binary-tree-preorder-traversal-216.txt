class Solution {

    List<Integer> list = new ArrayList<Integer>();

    public List<Integer> preorderTraversal(TreeNode root) {
        preOrder(root);
        return list;
    }

    public void preOrder(TreeNode root) {
        if (root == null)
            return;
        list.add(root.val);
        preOrder(root.left);
        preOrder(root.right);
    }
}
