class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> ans = new LinkedList<Integer>();
        TreeNode curr = root;
        Stack<TreeNode> s = new Stack<TreeNode>();
        while (!s.isEmpty() || curr != null) {
            while (curr != null) {
                ans.add(curr.val);
                s.push(curr);
                curr = curr.left;
            }
            curr = s.pop();
            curr = curr.right;
        }
        return ans;
    }
}
