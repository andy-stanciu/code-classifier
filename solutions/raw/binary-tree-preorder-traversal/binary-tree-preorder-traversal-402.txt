/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        return preorderHelper(root, list);
    }

    private List<Integer> preorderHelper(TreeNode curr, List<Integer> list) {
        if (curr == null) {
            return list;
        }
        list.add(curr.val);
        if (curr.left != null) {
            preorderHelper(curr.left, list);
        }
        if (curr.right != null) {
            preorderHelper(curr.right, list);
        }
        return list;
    }
}
