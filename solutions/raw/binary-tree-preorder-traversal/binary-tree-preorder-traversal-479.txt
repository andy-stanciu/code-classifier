class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        //Recursive
        return callPreOrderRecursive(root);
        //Iterative
        //return callPreOrderIterative(root);
    }

    private List<Integer> callPreOrderIterative(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        if (root == null)
            return result;
        stack.push(root);
        while (!stack.isEmpty()) {
            TreeNode current = stack.pop();
            if (current.right != null)
                stack.push(current.right);
            if (current.left != null)
                stack.push(current.left);
            result.add(current.val);
        }
        return result;
    }

    //Recursive
    private List<Integer> callPreOrderRecursive(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        callPreOrder(root, list);
        return list;
    }

    //preorder Root->Left->Right Recursive
    private void callPreOrder(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        callPreOrder(root.left, list);
        callPreOrder(root.right, list);
    }
    //Iterative
}
