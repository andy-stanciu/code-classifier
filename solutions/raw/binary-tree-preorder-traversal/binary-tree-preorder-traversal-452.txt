/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private static List<Integer> rec(TreeNode root, List<Integer> li) {
        if (root == null) {
            return li;
        }
        li.add(root.val);
        rec(root.left, li);
        rec(root.right, li);
        return li;
    }

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        return rec(root, li);
    }
}
