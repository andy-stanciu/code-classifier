class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Deque<TreeNode> stake = new ArrayDeque<>();
        stake.push(root);
        while (!stake.isEmpty()) {
            TreeNode current = stake.pop();
            result.add(current.val);
            if (current.right != null) {
                stake.push(current.right);
            }
            if (current.left != null) {
                stake.push(current.left);
            }
        }
        return result;
    }
}
