/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        TreeNode cur = root;
        List<Integer> list = new ArrayList<>();
        if (cur == null) {
            return list;
        }
        list.add(cur.val);
        list.addAll(preorderTraversal(cur.left));
        list.addAll(preorderTraversal(cur.right));
        return list;
    }
}
