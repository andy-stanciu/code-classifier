class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        preorder(root, list);
        return list;
    }

    private void preorder(TreeNode root, List<Integer> list) {
        TreeNode temp = root;
        if (temp == null) {
            return;
        }
        list.add(temp.val);
        preorder(temp.left, list);
        preorder(temp.right, list);
    }
}
