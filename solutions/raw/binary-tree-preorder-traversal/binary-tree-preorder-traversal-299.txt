class Solution {

    List<Integer> ans = new ArrayList<>();

    public List<Integer> inorderTraversal(TreeNode root) {
        inorderT(root);
        return ans;
    }

    private void inorderT(TreeNode node) {
        if (node == null) {
            return;
        }
        inorderT(node.left);
        ans.add(node.val);
        inorderT(node.right);
    }
}
