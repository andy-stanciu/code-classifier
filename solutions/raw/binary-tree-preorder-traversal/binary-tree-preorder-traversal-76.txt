class Solution {

    public void preorder(TreeNode node, List<Integer> pre) {
        if (node == null)
            return;
        pre.add(node.val);
        preorder(node.left, pre);
        preorder(node.right, pre);
    }

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        preorder(root, pre);
        return pre;
    }
}
