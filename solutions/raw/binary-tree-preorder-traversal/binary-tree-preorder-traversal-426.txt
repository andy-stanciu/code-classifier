class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        preorder(pre, root);
        return pre;
    }

    static void preorder(List<Integer> pre, TreeNode node) {
        if (node == null)
            return;
        TreeNode temp = node;
        pre.add(temp.val);
        preorder(pre, temp.left);
        preorder(pre, temp.right);
    }
}
