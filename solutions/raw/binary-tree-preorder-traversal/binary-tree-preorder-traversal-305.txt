class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> al = new ArrayList<>();
        preorder(root, al);
        return al;
    }

    private void preorder(TreeNode root, List<Integer> al) {
        if (root == null) {
            return;
        }
        al.add(root.val);
        preorder(root.left, al);
        preorder(root.right, al);
    }
}
