/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void amrit(TreeNode root, List<Integer> ans) {
        List<Integer> a = new ArrayList<>();
        if (root == null) {
            return;
        }
        ans.add(root.val);
        amrit(root.left, ans);
        amrit(root.right, ans);
    }

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        amrit(root, ans);
        return ans;
    }
}
