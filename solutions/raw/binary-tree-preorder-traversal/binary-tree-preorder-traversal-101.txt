class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        preorder(root, ans);
        return ans;
    }

    public void preorder(TreeNode r, List<Integer> ans) {
        if (r == null)
            return;
        ans.add(r.val);
        preorder(r.left, ans);
        preorder(r.right, ans);
    }
}
