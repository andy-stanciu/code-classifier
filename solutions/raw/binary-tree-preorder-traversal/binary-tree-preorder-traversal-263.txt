/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> traversal = new ArrayList<>();
        // preOrder(root, traversal);
        preOrderIter(root, traversal);
        return traversal;
    }

    private void preOrderIter(TreeNode node, List<Integer> traversal) {
        if (node == null)
            return;
        Deque<TreeNode> stack = new ArrayDeque<>();
        TreeNode tempNode;
        stack.push(node);
        while (!stack.isEmpty()) {
            tempNode = stack.pop();
            traversal.add(tempNode.val);
            if (tempNode.right != null) {
                stack.push(tempNode.right);
            }
            if (tempNode.left != null) {
                stack.push(tempNode.left);
            }
        }
    }

    private void preOrder(TreeNode node, List<Integer> traversal) {
        if (node == null)
            return;
        traversal.add(node.val);
        preOrder(node.left, traversal);
        preOrder(node.right, traversal);
    }
}
