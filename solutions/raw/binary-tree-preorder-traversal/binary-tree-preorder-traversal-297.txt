class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        preOrder(root);
        return ans;
    }

    List<Integer> ans = new ArrayList<>();

    public void preOrder(TreeNode node) {
        if (node == null) {
            return;
        }
        ans.add(node.val);
        preOrder(node.left);
        preOrder(node.right);
    }
}
