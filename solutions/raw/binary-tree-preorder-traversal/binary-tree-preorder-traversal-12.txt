/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list1 = new ArrayList<>();
        preorderHelper(root, list1);
        return list1;
    }

    private void preorderHelper(TreeNode root, List<Integer> list1) {
        if (root == null) {
            return;
        }
        list1.add(root.val);
        preorderHelper(root.left, list1);
        preorderHelper(root.right, list1);
    }
}
