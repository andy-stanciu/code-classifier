class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        if (root == null)
            return l;
        Stack<TreeNode> st = new Stack<>();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode t = st.pop();
            l.add(t.val);
            if (t.right != null)
                st.push(t.right);
            if (t.left != null)
                st.push(t.left);
        }
        return l;
    }
}
