/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private static List<Integer> result;

    public List<Integer> preorderTraversal(TreeNode root) {
        result = new ArrayList<>();
        if (root == null)
            return result;
        result.add(root.val);
        if (root.left != null) {
            preorderTraversalExecute(root.left);
        }
        if (root.right != null) {
            preorderTraversalExecute(root.right);
        }
        return result;
    }

    private void preorderTraversalExecute(TreeNode node) {
        result.add(node.val);
        if (node.left != null) {
            preorderTraversalExecute(node.left);
        }
        if (node.right != null) {
            preorderTraversalExecute(node.right);
        }
    }
}
