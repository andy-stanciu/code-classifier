/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList();
        // check whether tree is empty or not, because root is kind of pointer whch will access the tree
        if (root == null)
            return list;
        // PreOrder rule : Root Left Right
        list.add(root.val);
        //recursively call function to traverse through left subtree
        list.addAll(preorderTraversal(root.left));
        //recursively call function to traverse through right subtree
        list.addAll(preorderTraversal(root.right));
        return list;
    }
}
