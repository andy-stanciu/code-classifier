class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> a = new ArrayList<>();
        List<Integer> ans = preOrder(root, a);
        return ans;
    }

    public List<Integer> preOrder(TreeNode node, List<Integer> a) {
        if (node == null)
            return a;
        a.add(node.val);
        preOrder(node.left, a);
        preOrder(node.right, a);
        return a;
    }
}
