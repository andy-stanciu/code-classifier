/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> newList = new ArrayList<Integer>();
        if (root != null) {
            newList.add(root.val);
            if (root.left != null)
                newList.addAll(preorderTraversal(root.left));
            if (root.right != null)
                newList.addAll(preorderTraversal(root.right));
            return newList;
        }
        return newList;
    }
}
