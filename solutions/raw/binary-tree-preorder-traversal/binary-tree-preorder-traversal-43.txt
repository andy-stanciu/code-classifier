class Solution {

    List<Integer> list = new ArrayList<>();

    public List<Integer> preorderTraversal(TreeNode root) {
        if (root == null)
            return list;
        Stack<TreeNode> q = new Stack<>();
        q.add(root);
        while (!q.isEmpty()) {
            TreeNode node = q.pop();
            list.add(node.val);
            if (node.right != null)
                q.add(node.right);
            if (node.left != null)
                q.add(node.left);
        }
        return list;
    }
}
