/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> pretraverse = new ArrayList<>();
        if (root == null)
            return pretraverse;
        pretraversal(pretraverse, root);
        return pretraverse;
    }

    static void pretraversal(List pretraverse, TreeNode root) {
        if (root == null)
            return;
        pretraverse.add(root.val);
        pretraversal(pretraverse, root.left);
        pretraversal(pretraverse, root.right);
    }
}
