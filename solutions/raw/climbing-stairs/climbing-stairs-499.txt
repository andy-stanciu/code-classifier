class Solution {

    public int climbStairs(int n) {
        if (n == 1 || n == 2) {
            return n;
        }
        int prev2Sum = 1;
        int prevSum = 2;
        int sum = 0;
        for (int i = 3; i <= n; i++) {
            sum = prev2Sum + prevSum;
            prev2Sum = prevSum;
            prevSum = sum;
        }
        return sum;
    }
}
