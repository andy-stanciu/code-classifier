class Solution {

    public int findWays(int n, int[] ways) {
        if (n < 0) {
            return 0;
        }
        if (n == 0) {
            return 1;
        }
        if (ways[n] != 0) {
            return ways[n];
        }
        ways[n] = findWays(n - 1, ways) + findWays(n - 2, ways);
        return ways[n];
    }

    public int climbStairs(int n) {
        int[] ways = new int[n + 1];
        return findWays(n, ways);
    }
}
