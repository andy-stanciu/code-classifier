class Solution {

    Hashtable<Integer, Integer> steps = new Hashtable<>();

    public int climbStairs(int n) {
        if (steps.containsKey(n)) {
            return steps.get(n);
        }
        if (n == 1 || n == 0) {
            return 1;
        }
        steps.put(n, climbStairs(n - 1) + climbStairs(n - 2));
        return steps.get(n);
    }
}
