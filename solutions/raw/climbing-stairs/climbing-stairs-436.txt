class Solution {

    public int climbStairs(int n) {
        ArrayList<Integer> AA = new ArrayList<>();
        if (n == 1) {
            return 1;
        }
        AA.add(1);
        AA.add(0, 2);
        n = n - 2;
        while (n > 0) {
            AA.add(0, (AA.get(0) + AA.get(1)));
            AA.remove(2);
            n--;
        }
        return AA.get(0);
    }
}
