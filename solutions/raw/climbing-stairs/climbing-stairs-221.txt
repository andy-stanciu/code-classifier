class Solution {

    int first = 1;

    int second = 2;

    int ans;

    public int climbStairs(int n) {
        if (n == 1 || n == 2) {
            return n;
        }
        for (int i = 2; i < n; i++) {
            ans = first + second;
            first = second;
            second = ans;
        }
        return ans;
    }
}
