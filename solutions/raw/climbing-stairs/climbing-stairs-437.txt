class Solution {

    public int climbStairs(int n) {
        int allStep = 0;
        int previous = 1;
        int twoPrevious = 1;
        if (n == 0 || n == 1) {
            return 1;
        }
        for (int i = 2; i <= n; i++) {
            allStep = previous + twoPrevious;
            twoPrevious = previous;
            previous = allStep;
        }
        return allStep;
    }
}
