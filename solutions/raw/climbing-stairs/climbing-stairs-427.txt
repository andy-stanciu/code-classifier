class Solution {

    public int climbStairs(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        //got TLE
        // return climbStairs(n-2)+ climbStairs(n-1);
        //for n=0 and n=1
        int first = 1, sec = 1;
        for (int i = 2; i <= n; i++) {
            int temp = sec;
            sec = first + sec;
            first = temp;
        }
        return sec;
    }
}
