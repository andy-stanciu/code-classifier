//-----------Memorization-----------
class Solution {

    public int solve(int n, int[] dp) {
        if (n < 0) {
            return 0;
        }
        if (n == 0) {
            return 1;
        }
        if (dp[n] > 0) {
            return dp[n];
        }
        int ans = solve(n - 1, dp) + solve(n - 2, dp);
        dp[n] = ans;
        return ans;
    }

    public int climbStairs(int n) {
        int[] dp = new int[n + 1];
        return solve(n, dp);
    }
}
