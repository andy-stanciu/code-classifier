class Solution {

    public int climb(int n, int[] arr) {
        if (n == 0 || n == 1)
            return 1;
        if (arr[n] != 0)
            return arr[n];
        return arr[n] = climb(n - 1, arr) + climb(n - 2, arr);
    }

    public int climbStairs(int n) {
        int[] arr = new int[n + 1];
        return climb(n, arr);
    }
}
