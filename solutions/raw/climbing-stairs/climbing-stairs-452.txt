class Solution {

    public int climbStairs(int n) {
        return fibo(n, new HashMap<>());
    }

    public static int fibo(int n, HashMap<Integer, Integer> map) {
        if (n <= 1)
            return 1;
        if (map.containsKey(n))
            return map.get(n);
        map.put(n, fibo(n - 1, map) + fibo(n - 2, map));
        return map.get(n);
    }
}
