// Java Recursion with Memorization
class Solution {

    Map<Integer, Integer> map;

    private int helper(int n) {
        if (n == 1 || n == 0)
            return 1;
        if (map.get(n) != null)
            return map.get(n);
        int n1 = helper(n - 1);
        int n2 = helper(n - 2);
        map.put(n - 1, n1);
        map.put(n - 2, n2);
        return n1 + n2;
    }

    public int climbStairs(int n) {
        map = new HashMap<>();
        return helper(n);
    }
}
