class Solution {

    public int climbStairs(int n) {
        int[] memo = new int[n + 1];
        memo[0] = 1;
        return climbStairsUtil(n, memo);
    }

    private int climbStairsUtil(int n, int[] memo) {
        // path can be reached
        if (n == 0)
            return memo[0];
        // path cannot be reached
        if (n < 0)
            return 0;
        // if path already visited
        if (memo[n] != 0)
            return memo[n];
        memo[n] = climbStairsUtil(n - 1, memo) + climbStairsUtil(n - 2, memo);
        return memo[n];
    }
}
