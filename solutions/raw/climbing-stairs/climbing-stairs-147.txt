class Solution {

    public int climbStairs(int n) {
        if (n == 1)
            return 1;
        int prev = 0, curr = 1, sum;
        for (int idx = 0; idx < n; idx++) {
            //  idx  0   1   2   3
            //  sum  1   2   3   5
            sum = prev + curr;
            //  pre  1   1   2   3
            prev = curr;
            //  cur  1   2   3   5
            curr = sum;
        }
        return curr;
    }
}
