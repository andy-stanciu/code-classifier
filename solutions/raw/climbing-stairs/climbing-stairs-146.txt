class Solution {

    public int solve(int[] dp, int i, int n) {
        if (// We reach nth step
        i == n)
            // We reach nth step
            return 1;
        else // Out of bound
        if (i > n)
            return 0;
        if (dp[i] != -1)
            return dp[i];
        // 1 step
        int a = solve(dp, i + 1, n);
        // 2 steps
        int b = solve(dp, i + 2, n);
        // Total ways
        return dp[i] = a + b;
    }

    public int climbStairs(int n) {
        int[] dp = new int[n];
        Arrays.fill(dp, -1);
        return solve(dp, 0, n);
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        // Replace 4 with any desired input
        System.out.println(solution.climbStairs(4));
    }
}
