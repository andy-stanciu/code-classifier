class Solution {

    public int climbStairs(int n) {
        int fibonacciValue1 = 0;
        int fibonacciValue2 = 1;
        int actualValue = 0;
        //increment the actual value to the fibonacci numbers n times
        for (int i = 0; i < n; i++) {
            actualValue = fibonacciValue1 + fibonacciValue2;
            fibonacciValue1 = fibonacciValue2;
            fibonacciValue2 = actualValue;
        }
        return actualValue;
    }
}
