class Solution {

    public int fib(int num) {
        int first = 0, second = 1, third = first + second;
        for (int i = 1; i < num; i++) {
            first = second;
            second = third;
            third = first + second;
        }
        return third;
    }

    public int climbStairs(int n) {
        return (fib(n));
    }
}
