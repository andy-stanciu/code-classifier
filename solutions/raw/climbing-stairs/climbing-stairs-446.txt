class Solution {

    public int climbStairs(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        return recursive(1, 1, n, 2);
    }

    public int recursive(int prevprevTotal, int prevTotal, int n, int increment) {
        if (increment == n) {
            return prevTotal + prevprevTotal;
        } else {
            int newTotal = prevTotal + prevprevTotal;
            return recursive(prevTotal, newTotal, n, increment + 1);
        }
    }
}
