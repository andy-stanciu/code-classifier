class Solution {

    private int[] memo;

    public int climbStairs(int n) {
        if (n < 1 || n > 45) {
            throw new IllegalArgumentException("Parameter n must be between 1 and 45");
        }
        memo = new int[n + 1];
        return climbStairsMemo(n);
    }

    public int climbStairsMemo(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        if (memo[n] != 0) {
            return memo[n];
        }
        memo[n] = climbStairsMemo(n - 1) + climbStairsMemo(n - 2);
        return memo[n];
    }
}
