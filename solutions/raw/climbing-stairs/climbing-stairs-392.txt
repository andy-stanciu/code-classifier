class Solution {

    public int climbStairs(int n) {
        int[] dp = new int[n + 1];
        return f(n, dp);
    }

    public int f(int n, int[] dp) {
        if (n == 0 || n == 1 || n == 2) {
            return n;
        }
        if (dp[n] != 0)
            return dp[n];
        return dp[n] = f(n - 1, dp) + f(n - 2, dp);
    }
}
