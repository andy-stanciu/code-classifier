class Solution {

    public int climbStairs(int n) {
        int[] f = new int[n + 1];
        return climbStairs(n, f);
    }

    public static int climbStairs(int n, int[] f) {
        if (n < 0) {
            return 0;
        }
        if (n == 0) {
            return 1;
        }
        if (f[n] != 0) {
            return f[n];
        }
        f[n] = climbStairs(n - 1, f) + climbStairs(n - 2, f);
        return f[n];
    }
}
