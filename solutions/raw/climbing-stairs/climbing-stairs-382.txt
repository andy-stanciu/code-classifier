class Solution {

    public static int[] dp;

    public int climbStairs(int n) {
        dp = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            dp[i] = Integer.MIN_VALUE;
        }
        return helper(n);
    }

    public int helper(int n) {
        if (n == 0)
            return 0;
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        if (dp[n] == Integer.MIN_VALUE) {
            dp[n] = helper(n - 1) + helper(n - 2);
        }
        return dp[n];
    }
}
