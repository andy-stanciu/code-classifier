class Solution {

    public Map<Integer, Integer> hm = new HashMap<>();

    public int climbStairs(int n) {
        // essentially we want to "factorize" n in to sums involving 1 and 2. The correct term should be partition or something like that (factor * and partition +)
        if (n <= 0)
            return 0;
        else if (n == 2)
            return 2;
        else if (n == 1)
            return 1;
        else if (this.hm.get(n) != null)
            return this.hm.get(n);
        else {
            int first = climbStairs(n - 1);
            int second = climbStairs(n - 2);
            this.hm.put(n - 1, first);
            this.hm.put(n - 2, second);
            return first + second;
        }
    }
}
