class Solution {

    // Create new array based on size given
    int[] stairs = new int[46];

    public int climbStairs(int n) {
        // Base Condition
        if (n < 0)
            return 0;
        if (n == 0)
            return 1;
        if (stairs[n] != 0)
            return stairs[n];
        // Simple Concept (FIBONACCI)
        return stairs[n] = climbStairs(n - 1) + climbStairs(n - 2);
    }
}
