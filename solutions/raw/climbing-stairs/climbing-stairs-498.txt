class Solution {

    int[] dp = new int[46];

    public int climb(int n) {
        if (n == 2 || n == 1) {
            return n;
        }
        if (dp[n] != 0) {
            return dp[n];
        }
        dp[n] = climb(n - 2) + climb(n - 1);
        return climb(n - 2) + climb(n - 1);
    }

    public int climbStairs(int n) {
        return climb(n);
    }
}
