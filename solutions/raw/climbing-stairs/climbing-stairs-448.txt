class Solution {

    public int climbStairs(int n) {
        int[] memo = new int[n + 1];
        return countSteps(n, memo);
    }

    public int countSteps(int n, int[] memo) {
        if (n == 1) {
            return 1;
        }
        if (n == 2) {
            return 2;
        }
        if (memo[n] != 0)
            return memo[n];
        memo[n] = countSteps(n - 1, memo) + countSteps(n - 2, memo);
        return memo[n];
    }
}
