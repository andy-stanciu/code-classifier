class Solution {

    public int climbStairs(int n) {
        if (n <= 3) {
            return n;
        }
        int x = 1;
        int y = 2;
        int[] arr = new int[n + 1];
        arr[0] = 0;
        arr[1] = x;
        arr[2] = y;
        for (int i = 3; i <= n; i++) {
            arr[i] = arr[i - 1] + arr[i - 2];
        }
        return arr[n];
    }
}
