class Solution {

    public int climbStairs(int n) {
        int[] d = IntStream.range(1, n + 1).toArray();
        if (n <= 2) {
            return d[n - 1];
        } else {
            for (int i = 2; i < d.length; i++) {
                d[i] = d[i - 1] + d[i - 2];
            }
        }
        return d[n - 1];
    }
}
