class Solution {

    public int climbStairs(int n) {
        if (n <= 3) {
            return n;
        }
        int[] a = new int[n];
        a[0] = 1;
        a[1] = 2;
        a[2] = 3;
        for (int i = 3; i < n; i++) {
            a[i] = a[i - 1] + a[i - 2];
        }
        return a[n - 1];
    }
}
