class Solution {

    public int climbStairs(int n) {
        int[] dp = new int[n + 1];
        return Memo(n, dp);
    }

    public int Memo(int n, int[] dp) {
        if (n == 0 || n == 1) {
            dp[n] = 1;
            return dp[n];
        }
        if (dp[n] != 0)
            return dp[n];
        dp[n] = Memo(n - 1, dp) + Memo(n - 2, dp);
        return dp[n];
    }
}
