public class Solution {

    public int ClimbStairs(int n) {
        //the first example has tell us how many distinct way for 1 or 2 steps stairs
        int first = 1;
        int second = 2;
        int tmp = 0;
        if (//if n not more than 2 than just return the stpes it take
        n <= 2) {
            return n;
        } else {
            for (//every iteration,will add the last two steps together and save as \'second\'
            //every iteration,will add the last two steps together and save as \'second\'
            //every iteration,will add the last two steps together and save as \'second\'
            int i = 3; i <= n; i++) {
                tmp = second;
                //set second as now iteration needed steps
                second += first;
                //set first as the last time needed steps
                first = tmp;
            }
        }
        return second;
    }
}
