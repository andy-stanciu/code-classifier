class Solution {

    public int climbStairs(int n) {
        int[] arr = new int[3];
        arr[0] = 0;
        arr[1] = 1;
        arr[2] = 2;
        if (n < 3)
            return arr[n];
        int res = -1;
        for (int i = 3; i <= n; i++) {
            res = arr[2] + arr[1];
            arr[1] = arr[2];
            arr[2] = res;
        }
        return res;
    }
}
