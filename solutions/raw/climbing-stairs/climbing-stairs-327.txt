class Solution {

    public static int climbStairs(int n) {
        Integer[] dp = new Integer[n + 1];
        Arrays.fill(dp, -1);
        return helper(n, dp);
    }

    public static int helper(int n, Integer[] dp) {
        if (n == 0) {
            return 1;
        }
        if (dp[n] != -1) {
            return dp[n];
        }
        int left = helper(n - 1, dp);
        int right = 0;
        if (n - 2 >= 0) {
            right = helper(n - 2, dp);
        }
        return dp[n] = left + right;
    }
}
