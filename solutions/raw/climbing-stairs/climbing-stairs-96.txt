class Solution {

    public int climbStairs(int n) {
        int[] tab = new int[n + 1];
        if (tab.length > 0)
            tab[0] = 1;
        if (tab.length > 1)
            tab[1] = 1;
        for (int i = 2; i < tab.length; i++) tab[i] = tab[i - 1] + tab[i - 2];
        return tab[n];
    }
}
