class Solution {

    int[] arr;

    public int climbStairs(int n) {
        if (n <= 1)
            return 1;
        arr = new int[n + 1];
        return helper(n);
    }

    public int helper(int n) {
        if (n <= 1)
            return 1;
        if (arr[n] == 0) {
            int temp1 = helper(n - 1) + helper(n - 2);
            arr[n] = temp1;
            return temp1;
        }
        return arr[n];
    }
}
