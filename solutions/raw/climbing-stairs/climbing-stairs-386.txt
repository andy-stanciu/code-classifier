class Solution {

    public int climbStairs(int n) {
        int result = 0;
        if (n == 0) {
            return result;
        }
        int a = 0;
        int b = 1;
        int i = 0;
        while (i < n) {
            result = a + b;
            a = b;
            b = result;
            i++;
        }
        return result;
    }
}
