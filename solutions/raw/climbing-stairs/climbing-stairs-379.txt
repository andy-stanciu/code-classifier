class Solution {

    public int climbStairsHelper(int curr, int n, int count, int[] dp) {
        if (curr > n) {
            return count;
        }
        if (n == curr) {
            count++;
            return count;
        }
        if (dp[curr] != -1)
            return dp[curr];
        dp[curr] = climbStairsHelper(curr + 1, n, count, dp) + climbStairsHelper(curr + 2, n, count, dp);
        return dp[curr];
    }

    public int climbStairs(int n) {
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1);
        return climbStairsHelper(0, n, 0, dp);
    }
}
