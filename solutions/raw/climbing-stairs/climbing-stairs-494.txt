class Solution {

    public int climbStairs(int n) {
        int[] dp = new int[n + 1];
        //intitalize dp array
        Arrays.fill(dp, -1);
        return f(n, dp);
    }

    static int f(int n, int[] dp) {
        if (dp[n] != -1)
            return dp[n];
        //base cases
        if (n == 0)
            return 1;
        //base cases
        if (n == 1)
            return 1;
        return dp[n] = f(n - 1, dp) + f(n - 2, dp);
    }
}
