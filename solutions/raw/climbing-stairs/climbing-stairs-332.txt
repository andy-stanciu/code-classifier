class Solution {

    public int climbStairs(int n) {
        int[] memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return helper(n, memo);
    }

    public int helper(int n, int[] memo) {
        if (n == 0)
            return 1;
        if (memo[n] != -1)
            return memo[n];
        int left = helper(n - 1, memo);
        int right = 0;
        if (n > 1)
            right = helper(n - 2, memo);
        memo[n] = left + right;
        return memo[n];
    }
}
