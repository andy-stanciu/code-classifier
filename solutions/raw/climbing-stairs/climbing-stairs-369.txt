class Solution {

    public int climbStairs(int n) {
        // Array de tama\xF1o n+1
        int[] arr = new int[n + 1];
        // Caso base: 1 forma de quedarse en el escal\xF3n 0
        arr[0] = 1;
        // Caso base: 1 forma de llegar al escal\xF3n 1
        arr[1] = 1;
        for (int i = 2; i <= n; i++) {
            arr[i] = arr[i - 1] + arr[i - 2];
        }
        return arr[n];
    }
}
