class Solution {

    public int climbStairs(int n) {
        if (n <= 3)
            return n;
        int f = 0, s = 1, t = f + s;
        for (int i = 0; i < n; i++) {
            t = f + s;
            f = s;
            s = t;
        }
        return t;
    }
}
