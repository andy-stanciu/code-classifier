class Solution {

    public int climbStairs(int n) {
        if (// at 0 do nothing yes that\'s also a step:)
        n == 0)
            return 1;
        if (// when we are taking two steps back then from 2 to 0 is 1 way only
        n == 1)
            return 1;
        return climbStairs(n - 1) + climbStairs(n - 2);
    }
}
