class Solution {

    public int climbStairs(int n) {
        if (n == 0) {
            return 0;
        } else if (n == 1) {
            return 1;
        }
        int[] table = new int[n + 2];
        table[1] = 1;
        for (int i = 2; i <= n + 1; i++) {
            table[i] = table[i - 1] + table[i - 2];
        }
        return table[n + 1];
    }
}
