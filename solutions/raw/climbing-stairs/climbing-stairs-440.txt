// Java Iterative without Memorization
class Solution {

    public int climbStairs(int n) {
        int cache1 = 1;
        int cache2 = 1;
        for (int L = 2; L <= n; L++) {
            int tmp = cache1;
            cache1 = cache1 + cache2;
            cache2 = tmp;
        }
        return cache1;
    }
}
