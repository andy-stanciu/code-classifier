class Solution {

    public int climbStairs(int n) {
        HashMap<Integer, Integer> map = new HashMap<>();
        return memoclimb(map, n);
    }

    public int memoclimb(HashMap<Integer, Integer> memo, int n) {
        if (n <= 2) {
            return n;
        }
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        memo.put(n, memoclimb(memo, n - 1) + memoclimb(memo, n - 2));
        return memo.get(n);
    }
}
