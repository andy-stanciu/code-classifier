class Solution {

    int x = 0;

    int[] data = new int[46];

    public int climbStairs(int n) {
        x = n;
        return util(0);
    }

    private int util(int n) {
        if (n == x) {
            return 1;
        }
        if (n > x) {
            return 0;
        }
        if (data[n] == 0) {
            data[n] = util(n + 1) + util(n + 2);
        }
        return data[n];
    }
}
