class Solution {

    public int climbStairs(int n) {
        if (n <= 2)
            return n;
        int a = 1;
        int b = 2;
        int count = 0;
        for (int i = 3; i <= n; i++) {
            count = a + b;
            a = b;
            b = count;
        }
        return count;
    }
}
