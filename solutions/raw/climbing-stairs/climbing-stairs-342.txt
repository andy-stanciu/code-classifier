class Solution {

    public int climbStairs(int n) {
        int[] dp = new int[n + 2];
        dp[n] = 1;
        dp[n + 1] = 0;
        for (int steps = n - 1; steps >= 0; steps--) {
            int oneStep = dp[steps + 1];
            int twoSteps = dp[steps + 2];
            dp[steps] = oneStep + twoSteps;
        }
        return dp[0];
    }
}
