class Solution {

    public int climbStairs(int n) {
        if (n == 1) {
            return 1;
        }
        int onejump = 1;
        int twojump = 2;
        for (int i = 3; i <= n; i++) {
            int total = onejump + twojump;
            onejump = twojump;
            twojump = total;
        }
        return twojump;
    }
}
