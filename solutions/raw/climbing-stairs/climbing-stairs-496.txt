class Solution {

    public int climbStairs(int n) {
        int[] memo = new int[n + 1];
        Arrays.fill(memo, -1);
        memo[0] = 1;
        memo[1] = 1;
        return climbStairsHelper(n, memo);
    }

    public int climbStairsHelper(int n, int[] memo) {
        if (memo[n] != -1) {
            return memo[n];
        }
        memo[n] = climbStairsHelper(n - 1, memo) + climbStairsHelper(n - 2, memo);
        return memo[n];
    }
}
