class Solution {

    public int climb_Stairs(int n, HashMap<Integer, Integer> memo) {
        if (n <= 3) {
            return n;
        }
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        memo.put(n, climb_Stairs(n - 1, memo) + climb_Stairs(n - 2, memo));
        return memo.get(n);
    }

    public int climbStairs(int n) {
        HashMap<Integer, Integer> memo = new HashMap<Integer, Integer>();
        return climb_Stairs(n, memo);
    }
}
