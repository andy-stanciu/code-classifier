class Solution {

    public int climbStairs(int n) {
        HashMap<Integer, Integer> memo = new HashMap<>();
        return climb(0, n, memo);
    }

    private int climb(int current, int target, HashMap<Integer, Integer> memo) {
        if (current == target)
            return 1;
        if (current > target)
            return 0;
        int currentKey = current;
        if (memo.containsKey(currentKey))
            return memo.get(currentKey);
        int oneStep = climb(current + 1, target, memo);
        int twoStep = climb(current + 2, target, memo);
        memo.put(currentKey, oneStep + twoStep);
        return memo.get(currentKey);
    }
}
