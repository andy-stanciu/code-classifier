class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length <= 2) {
            return nums.length;
        }
        int j = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] == nums[j - 2]) {
                continue;
            }
            //Add the element instead of the duplicate.
            nums[j] = nums[i];
            //Next available position in the Array.
            j++;
        }
        //New length of the Array.
        return j;
    }
}
