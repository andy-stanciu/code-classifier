class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0)
            return 0;
        int count = 1;
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                count = 1;
                nums[k++] = nums[i];
            } else if (count < 2) {
                count++;
                nums[k++] = nums[i];
            }
        }
        return k;
    }
}
