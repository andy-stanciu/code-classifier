class Solution {

    public int removeDuplicates(int[] nums) {
        // Base case: If the length of the array is less than or equal to 2, return the length as is.
        if (nums.length <= 2) {
            return nums.length;
        }
        // Start from the third element
        int index = 2;
        // Iterate from the third element to the end
        for (int i = 2; i < nums.length; i++) {
            // If the current element is different from the element two places before it, we keep it
            if (nums[i] != nums[index - 2]) {
                nums[index] = nums[i];
                index++;
            }
        }
        // The index represents the length of the modified array
        return index;
    }
}
