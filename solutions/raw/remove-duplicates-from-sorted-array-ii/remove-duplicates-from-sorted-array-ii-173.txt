class Solution {

    public int removeDuplicates(int[] nums) {
        Map<Integer, Integer> hm = new HashMap<>();
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (hm.containsKey(nums[i])) {
                if (hm.get(nums[i]) < 2) {
                    hm.put(nums[i], hm.get(nums[i]) + 1);
                    nums[j++] = nums[i];
                }
            } else {
                hm.put(nums[i], 1);
                nums[j++] = nums[i];
            }
        }
        int k = 0;
        for (Map.Entry<Integer, Integer> e : hm.entrySet()) {
            k += e.getValue();
        }
        return k;
    }
}
