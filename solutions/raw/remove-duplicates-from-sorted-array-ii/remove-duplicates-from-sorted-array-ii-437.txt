class Solution {

    public int removeDuplicates(int[] nums) {
        int k = 2;
        for (int i = 1; i < nums.length - 1; i++) {
            if (nums[i] == nums[i - 1] && nums[i] == nums[i + 1]) {
                nums[i - 1] = Integer.MAX_VALUE;
            } else {
                k++;
            }
        }
        Arrays.sort(nums);
        return k;
    }
}
