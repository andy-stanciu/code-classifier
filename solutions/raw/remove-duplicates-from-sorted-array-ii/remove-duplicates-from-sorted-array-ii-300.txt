class Solution {

    public int removeDuplicates(int[] nums) {
        int curr = 0;
        int prev = Integer.MIN_VALUE;
        int ctr = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != prev) {
                prev = nums[i];
                ctr = 0;
                nums[curr++] = nums[i];
                ctr++;
            } else if (ctr < 2) {
                nums[curr++] = nums[i];
                ctr++;
            }
        }
        return curr;
    }
}
