public class Solution {

    public static int removeDuplicates(int[] nums) {
        if (nums.length <= 2) {
            return nums.length;
        }
        int k = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[k - 2]) {
                // Place the current element at index k
                nums[k] = nums[i];
                // Move k forward
                k++;
            }
        }
        // Return the new length
        return k;
    }

    public static void main(String[] args) {
        int[] nums = { 1, 1, 1, 2, 2, 3, 3, 4 };
        int k = removeDuplicates(nums);
        // Print the number of valid elements
        System.out.println("Number of elements after removing duplicates: " + k);
        // Print the modified array with allowed duplicates
        for (int i = 0; i < k; i++) {
            System.out.print(nums[i] + " ");
        }
    }
}
