class Solution {

    public int removeDuplicates(int[] nums) {
        Stack<Integer> st = new Stack();
        int count = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (st.isEmpty() || st.peek() == nums[i]) {
                st.add(nums[i]);
                count++;
                if (count > 2) {
                    st.pop();
                }
            } else {
                st.add(nums[i]);
                count = 1;
            }
        }
        int i = 0;
        while (!st.isEmpty()) {
            nums[i++] = st.pop();
        }
        return i;
    }
}
