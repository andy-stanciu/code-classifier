class Solution {

    public int removeDuplicates(int[] nums) {
        int[] a = new int[nums.length];
        if (nums.length == 0) {
            return 0;
        }
        int j = 0, c = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                c++;
            } else {
                c = 1;
            }
            if (c <= 2) {
                nums[++j] = nums[i];
            }
        }
        return j + 1;
    }
}
