class Solution {

    public int removeDuplicates(int[] nums) {
        int k = 1, counter = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                counter++;
            } else {
                counter = 1;
            }
            if (counter <= 2) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
