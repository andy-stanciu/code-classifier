class Solution {

    public int removeDuplicates(int[] nums) {
        int pointer = 0;
        int count = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[pointer]) {
                nums[++pointer] = nums[i];
                count = 0;
            } else if (count < 1) {
                nums[++pointer] = nums[i];
                count++;
            }
        }
        return pointer + 1;
    }
}
