class Solution {

    public int removeDuplicates(int[] nums) {
        int j = 0;
        int temp = 0;
        int diff = 0;
        if (nums.length > 2) {
            for (int i = 0; i < nums.length; i++) {
                if ((nums[i] > nums[j]) && ((i - j) > 1)) {
                    nums[j + 2] = nums[i];
                    j += 1;
                }
            }
            return j + 2;
        } else {
            return nums.length;
        }
    }
}
