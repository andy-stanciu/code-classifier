class Solution {

    public int removeDuplicates(int[] nums) {
        // Initialize a counter to track the position in the new array
        int i = 0;
        // Loop through each element in the array using an enhanced for loop
        for (int n : nums) {
            // If the current position is less than 2 or the current element is greater than the element at i-2,
            // assign the current element to the current position in the new array and increment the counter
            if (i < 2 || n > nums[i - 2])
                nums[i++] = n;
        }
        // Return the total count of elements in the new array
        return i;
    }
}
