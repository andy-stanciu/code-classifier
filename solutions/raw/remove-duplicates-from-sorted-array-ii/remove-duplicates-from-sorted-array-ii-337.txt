class Solution {

    public int removeDuplicates(int[] nums) {
        // Pointer for placing unique elements
        int left = 0;
        // Counter for occurrences of the current number
        int temp = 1;
        for (int i = 0; i < nums.length; i++) {
            // Check if the current number is the same as the next one
            if (i < nums.length - 1 && nums[i] == nums[i + 1]) {
                temp++;
            } else {
                // Place the number in the modified array based on its occurrence
                if (temp > 1) {
                    nums[left++] = nums[i];
                    nums[left++] = nums[i];
                } else {
                    nums[left++] = nums[i];
                }
                // Reset the count for the next number
                temp = 1;
            }
        }
        // Length of the modified array with at most two duplicates
        return left;
    }
}
