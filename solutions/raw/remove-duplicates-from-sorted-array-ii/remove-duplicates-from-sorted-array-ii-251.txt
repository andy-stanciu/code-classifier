class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0, freq = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] == nums[i]) {
                freq++;
                if (freq == 1) {
                    i++;
                    nums[i] = nums[j];
                }
            } else if (nums[i] != nums[j]) {
                freq = 0;
                i++;
                nums[i] = nums[j];
            }
        }
        return i + 1;
    }
}
