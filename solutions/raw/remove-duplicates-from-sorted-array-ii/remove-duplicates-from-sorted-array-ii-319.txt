class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0;
        int k = 1;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] == nums[i] && k < 2) {
                k++;
                i++;
                nums[i] = nums[j];
            } else if (nums[j] != nums[i]) {
                k = 1;
                i++;
                nums[i] = nums[j];
            }
        }
        return i + 1;
    }
}
