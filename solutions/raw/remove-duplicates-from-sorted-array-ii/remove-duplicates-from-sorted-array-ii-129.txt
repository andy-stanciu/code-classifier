class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length <= 2)
            return nums.length;
        int j = 0;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i == 0 || nums[i] != nums[i - 1]) {
                count = 1;
                nums[j++] = nums[i];
            } else if (count < 2) {
                count++;
                nums[j++] = nums[i];
            }
        }
        return j;
    }
}
