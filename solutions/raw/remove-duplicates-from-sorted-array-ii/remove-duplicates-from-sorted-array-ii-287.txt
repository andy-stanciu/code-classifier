class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        // j is the index of the element we should write to
        // (i always >= j, because we always increment i, but we only increment
        //  j for elements that are repeated no more than twice)
        int j = 1;
        int length = 1;
        int curr = nums[0];
        boolean repeated = false;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == curr) {
                if (!repeated) {
                    // we are seeing the current element for the second time
                    nums[j] = nums[i];
                    repeated = true;
                    j++;
                } else {
                    // we are seeing the current element for the third time.
                    // do nothing
                    continue;
                }
            } else {
                // we are seeing the current element for the first time
                curr = nums[i];
                nums[j] = nums[i];
                repeated = false;
                j++;
            }
        }
        return j;
    }
}
