class Solution {

    public static int removeDuplicates(int[] nums) {
        // i index always stores unique element
        // l and r used to expand window of duplicates
        int i = 0, l = 0, r = 0;
        while (r < nums.length) {
            // window expansion
            while (r < nums.length && nums[l] == nums[r]) {
                r++;
            }
            if (r - l >= 2) {
                // i and i+1 filled with next unique
                nums[i] = nums[l];
                nums[i + 1] = nums[l];
                // shift to next index supposed to fill with unique
                i = i + 2;
            } else {
                // i filled with next unique
                nums[i] = nums[l];
                // shift to  next index supposed to fill with unique
                i = i + 1;
            }
            l = r;
        }
        return i;
    }
}
