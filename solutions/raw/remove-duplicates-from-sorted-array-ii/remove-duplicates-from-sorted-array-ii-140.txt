class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        if (n <= 2) {
            return n;
        }
        int InsertIndex = 2;
        for (int i = 2; i < n; i++) {
            if (nums[i] != nums[InsertIndex - 2]) {
                nums[InsertIndex] = nums[i];
                InsertIndex++;
            }
        }
        return InsertIndex;
    }
}
