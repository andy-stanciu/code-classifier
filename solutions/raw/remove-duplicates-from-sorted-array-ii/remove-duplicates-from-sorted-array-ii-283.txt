class Solution {

    public int removeDuplicates(int[] nums) {
        int k = 0, count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                count++;
                if (count < 3) {
                    k++;
                }
            } else {
                count = 1;
                k++;
            }
            nums[k] = nums[i];
        }
        return k + 1;
    }
}
