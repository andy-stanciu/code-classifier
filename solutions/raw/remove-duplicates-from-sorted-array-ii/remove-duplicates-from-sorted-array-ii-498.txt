class Solution {

    public int removeDuplicates(int[] nums) {
        // put into hashmap with num and count
        //then go through keyset and start putting if >=2 -put it twice & if ==1 put it.
        LinkedHashMap<Integer, Integer> hm = new LinkedHashMap<>();
        for (int num : nums) {
            hm.put(num, hm.getOrDefault(num, 0) + 1);
        }
        int j = 0;
        for (int key : hm.keySet()) {
            if (hm.get(key) >= 2) {
                nums[j] = key;
                j++;
                nums[j] = key;
            } else if (hm.get(key) == 1) {
                nums[j] = key;
            }
            j++;
        }
        return j;
    }
}
