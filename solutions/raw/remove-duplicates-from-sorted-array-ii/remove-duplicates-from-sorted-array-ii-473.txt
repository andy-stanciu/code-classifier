class Solution {

    public int removeDuplicates(int[] nums) {
        int previous = nums[0];
        int previousB = nums[0] - 1;
        int counter = 0;
        for (int i = 1; i < nums.length; i++) {
            if (previous == nums[i]) {
                if (previousB == nums[i]) {
                    counter++;
                } else {
                    nums[i - counter] = nums[i];
                }
                previousB = nums[i];
            } else {
                previous = nums[i];
                nums[i - counter] = nums[i];
            }
        }
        return nums.length - counter;
    }
}
