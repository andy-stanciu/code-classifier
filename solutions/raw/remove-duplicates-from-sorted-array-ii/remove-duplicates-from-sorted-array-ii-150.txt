class Solution {

    public int removeDuplicates(int[] nums) {
        // tracking occurances
        int count = 0;
        // ensuring the first check has non matching integers
        int current = nums[0] + 1;
        // slow pointer
        int index = 0;
        // loop through the list
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != current) {
                // if(the value we are looking at has changed)
                // update the count
                count = 1;
                // swap the numbers
                nums[index] = nums[i];
                // update the slow pointer
                index++;
                // update the value we have just seen
                current = nums[i];
            } else if (nums[i] == current && count == 1) {
                // if we see the value for the second time
                // update the count
                count++;
                // swap the values
                nums[index] = nums[i];
                // update the slow pointer
                index++;
            }
        }
        return index;
    }
}
