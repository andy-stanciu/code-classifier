class Solution {

    public int removeDuplicates(int[] nums) {
        int j = 0;
        //as 1st element is alwats unique
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                //as no is unique set count to 1 as it will be added oce from here
                count = 1;
            } else {
                //duplicates element found
                count++;
            }
            //that is count more than 2 no will be ignored for a reason
            if (count <= 2) {
                //make place for next unique no
                j++;
                nums[j] = nums[i];
            }
        }
        return j + 1;
    }
}
