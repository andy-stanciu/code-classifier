class Solution {

    public int removeDuplicates(int[] nums) {
        int shift = 0;
        int prev = nums[0];
        int count = 1;
        int n = nums.length;
        for (int i = 1; i < n; i++) {
            if (nums[i] == prev)
                count++;
            else
                count = 1;
            prev = nums[i];
            if (count > 2)
                shift++;
            nums[i - shift] = nums[i];
        }
        return n - shift;
    }
}
