class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        int count = 0, cnt = 0, prev = nums[0];
        for (int i = 0; i < n; i++) {
            cnt++;
            if (cnt > 2 && prev == nums[i]) {
                // Mark for removal
                nums[i] = Integer.MAX_VALUE;
                continue;
            } else if (prev != nums[i]) {
                // Reset the count for new number
                cnt = 1;
            }
            count++;
            prev = nums[i];
        }
        // Move the marked elements to the end
        Arrays.sort(nums);
        return count;
    }
}
