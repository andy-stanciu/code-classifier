class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length <= 2) {
            return nums.length;
        }
        int insertIndex = 2;
        for (int current = 2; current < nums.length; current++) {
            if (nums[current] != nums[insertIndex - 2]) {
                nums[insertIndex] = nums[current];
                insertIndex++;
            }
        }
        return insertIndex;
    }
}
