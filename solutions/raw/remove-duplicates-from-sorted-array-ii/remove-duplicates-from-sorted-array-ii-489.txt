class Solution {

    public int removeDuplicates(int[] nums) {
        int ptr = 1;
        if (nums.length <= 2) {
            return nums.length;
        }
        for (int i = 2; i < nums.length; i++) {
            int curr = nums[i];
            if (curr != nums[ptr - 1]) {
                ++ptr;
                nums[ptr] = curr;
            }
        }
        return ptr + 1;
    }
}
