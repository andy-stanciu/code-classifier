class Solution {

    public int removeDuplicates(int[] nums) {
        int res = 0;
        boolean found = false;
        for (int i = 1; i < nums.length; i++) {
            // if duplicate past twice move on
            if (found && nums[i - 1] == nums[i])
                continue;
            if (// 1. if first duplicate, record boolean flag
            nums[i - 1] == nums[i])
                // 1. if first duplicate, record boolean flag
                found = true;
            else
                // 2. if not duplicate, reset boolean flag
                found = false;
            // overwrite for both cases
            nums[++res] = nums[i];
        }
        return ++res;
    }
}
