class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0;
        int k = 0;
        int prev = nums[0];
        int count = 0;
        while (i < nums.length) {
            if (nums[i] == prev) {
                if (count < 2) {
                    nums[k] = nums[i];
                    k++;
                    i++;
                    count++;
                } else {
                    nums[k] = nums[i];
                    i++;
                    count++;
                }
            } else {
                if (count <= 2) {
                    nums[k] = nums[i];
                    prev = nums[i];
                    count = 1;
                    k++;
                    i++;
                } else {
                    nums[k] = nums[i];
                    prev = nums[i];
                    count = 1;
                    i++;
                    k++;
                }
            }
        }
        // for (int j : nums) {
        //   System.out.print(" "+j);
        // }
        return k;
    }
}
