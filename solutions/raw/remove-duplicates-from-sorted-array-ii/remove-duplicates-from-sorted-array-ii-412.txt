class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length <= 2) {
            return nums.length;
        }
        // Index where the next element should be inserted
        int insertPos = 2;
        for (int i = 2; i < nums.length; i++) {
            // Check if the current element is not equal to the element at insertPos - 2
            if (nums[i] != nums[insertPos - 2]) {
                nums[insertPos] = nums[i];
                insertPos++;
            }
        }
        return insertPos;
    }
}
