class Solution {

    public int removeDuplicates(int[] a) {
        int p = 0, occ = 1;
        for (int i = 1; i < a.length; i++) {
            if (a[i] == a[p]) {
                if (occ > 0) {
                    occ--;
                    a[++p] = a[i];
                }
            } else {
                occ = 1;
                a[++p] = a[i];
            }
        }
        return p + 1;
    }
}
