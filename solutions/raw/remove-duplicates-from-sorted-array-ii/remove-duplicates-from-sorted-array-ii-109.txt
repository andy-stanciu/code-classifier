class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length == 2)
            return nums.length;
        int left = 2, i = 2;
        while (i < nums.length) {
            if (nums[i] != nums[left - 2]) {
                nums[left++] = nums[i];
            }
            i++;
        }
        return left;
    }
}
