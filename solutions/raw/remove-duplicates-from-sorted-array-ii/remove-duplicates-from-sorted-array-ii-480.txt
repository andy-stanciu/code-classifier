class Solution {

    public int removeDuplicates(int[] nums) {
        int insertIndex = 1, count = 1;
        for (int i = 1; i < nums.length; i++) {
            //increase the count when same element appears again
            if (nums[i] == nums[i - 1])
                count++;
            else
                //reset the count when diff element appears
                count = 1;
            //insert at the insertPos when count <= 2
            if (count <= 2)
                nums[insertIndex++] = nums[i];
        }
        //return the insertIndex
        return insertIndex;
    }
}
