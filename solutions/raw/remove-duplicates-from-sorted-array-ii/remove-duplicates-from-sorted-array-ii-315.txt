class Solution {

    public int removeDuplicates(int[] nums) {
        int pointer = 0, count = 1, allow = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[pointer]) {
                pointer++;
                nums[pointer] = nums[i];
                count++;
                allow = 1;
            } else if (allow < 2) {
                count++;
                allow++;
                pointer++;
                nums[pointer] = nums[i];
            }
        }
        return count;
    }
}
