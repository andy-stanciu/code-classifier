class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length <= 2) {
            return nums.length;
        }
        int pivot = 1;
        int curVal = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != curVal) {
                curVal = nums[i];
                count = 0;
            }
            if (count < 2) {
                nums[pivot] = nums[i];
                count++;
                pivot++;
            }
        }
        return pivot;
    }
}
