class Solution {

    public int removeDuplicates(int[] nums) {
        int index = 0;
        int count = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[index] != nums[i]) {
                count = 0;
                nums[++index] = nums[i];
                continue;
            }
            count++;
            if (count < 2) {
                nums[++index] = nums[i];
            }
        }
        return ++index;
    }
}
