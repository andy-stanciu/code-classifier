class Solution {

    public int removeDuplicates(int[] nums) {
        int slowPointer = 0;
        int count = 1;
        int fastPointer = 1;
        while (fastPointer < nums.length) {
            if (nums[slowPointer] != nums[fastPointer]) {
                slowPointer++;
                nums[slowPointer] = nums[fastPointer];
                fastPointer++;
                count = 1;
            } else if (count == 1) {
                slowPointer++;
                nums[slowPointer] = nums[fastPointer];
                fastPointer++;
                count++;
            } else {
                fastPointer++;
            }
        }
        return ++slowPointer;
    }
}
