class Solution {

    public int removeDuplicates(int[] nums) {
        Queue<Integer> queue = new LinkedList<Integer>();
        int currentCount = 1;
        int currentNumber = nums[0];
        queue.add(currentNumber);
        for (int i = 1; i < nums.length; i++) {
            if (currentNumber == nums[i]) {
                if (currentCount != 2) {
                    currentCount++;
                    queue.add(currentNumber);
                }
            } else {
                currentCount = 1;
                currentNumber = nums[i];
                queue.add(currentNumber);
            }
        }
        int i = 0;
        while (!queue.isEmpty()) {
            nums[i] = queue.remove();
            i++;
        }
        return i;
    }
}
