class Solution {

    public int removeDuplicates(int[] nums) {
        int min = Integer.MIN_VALUE;
        int t = 0;
        int v = 0;
        int n = 0;
        int l = nums.length;
        for (int i = 1; i < nums.length; i++) {
            if (nums[t] == nums[i]) {
                v++;
                if (v == 2) {
                    nums[t] = min;
                    t++;
                    n++;
                    l--;
                    v--;
                }
            } else {
                t = i;
                v = 0;
            }
        }
        for (int j = 0; j <= n; j++) {
            for (int i = 0; i < nums.length - 1; i++) {
                if (nums[i] == min) {
                    nums[i] = nums[i + 1];
                    nums[i + 1] = min;
                }
            }
        }
        return l;
    }
}
