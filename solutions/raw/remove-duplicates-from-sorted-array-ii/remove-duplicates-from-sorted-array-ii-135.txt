class Solution {

    public int removeDuplicates(int[] nums) {
        Map<Integer, Integer> counter = new HashMap<>();
        int count = 1;
        counter.put(nums[0], 1);
        for (int i = 1; i < nums.length; i++) {
            int num = nums[i];
            counter.put(num, counter.getOrDefault(num, 0) + 1);
            if (counter.get(num) <= 2) {
                nums[count] = num;
                count++;
            }
        }
        return count;
    }
}
