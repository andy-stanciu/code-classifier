class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 2, k = 2;
        while (i < nums.length) {
            if (nums[i] != nums[k - 2]) {
                nums[k] = nums[i];
                i++;
                k++;
            } else {
                i++;
            }
        }
        return k;
    }
}
