class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length < 2)
            return nums.length;
        int k = 2;
        int l = 0;
        for (int i = 2; i < nums.length; i++) {
            if (nums[k - 2] == nums[i] && nums[k - 1] == nums[i]) {
                continue;
            }
            nums[k] = nums[i];
            k++;
        }
        return k;
    }
}
