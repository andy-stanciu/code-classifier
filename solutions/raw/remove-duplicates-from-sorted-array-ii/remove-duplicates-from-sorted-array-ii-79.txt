class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0, count = 1, j = 1;
        while (j < nums.length) {
            if (nums[j] == nums[i]) {
                if (count < 2) {
                    i++;
                    nums[i] = nums[j];
                    count++;
                }
            } else {
                i++;
                nums[i] = nums[j];
                count = 1;
            }
            j++;
        }
        return i + 1;
    }
}
