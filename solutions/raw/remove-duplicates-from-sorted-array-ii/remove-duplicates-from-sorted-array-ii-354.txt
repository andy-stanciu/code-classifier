class Solution {

    public int removeDuplicates(int[] nums) {
        // since two consecutive numbers can be the same
        // we don\'t need to alter the first two elements
        int index = 1;
        // Start the loop from the third element
        for (int i = 2; i < nums.length; i++) {
            // If the current element is different from the element at index - 1,
            // it means it can be included in the final array
            if (nums[i] != nums[index - 1]) {
                // Move the index pointer to the next position
                index++;
                // Update the position with the current element
                nums[index] = nums[i];
            }
        }
        // Return the number of valid elements (index + 1)
        return index + 1;
    }
}
