class Solution {

    public int removeDuplicates(int[] nums) {
        int count = 1;
        int dup = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[count++] = nums[i];
                dup = 0;
            } else {
                dup++;
                if (dup > 0 && dup < 2) {
                    nums[count++] = nums[i];
                }
            }
        }
        return count;
    }
}
