class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 1, j = 0;
        int n = nums.length;
        int count = 1;
        while (i < n) {
            if (nums[i - 1] == nums[i]) {
                i++;
                count++;
            } else if (count >= 2) {
                nums[j] = nums[i - 1];
                nums[++j] = nums[i - 1];
                count = 1;
                i++;
                j++;
            } else {
                nums[j] = nums[i - 1];
                i++;
                j++;
            }
        }
        if (count >= 2) {
            nums[j] = nums[i - 1];
            nums[++j] = nums[i - 1];
        } else {
            nums[j] = nums[i - 1];
        }
        return j + 1;
    }
}
