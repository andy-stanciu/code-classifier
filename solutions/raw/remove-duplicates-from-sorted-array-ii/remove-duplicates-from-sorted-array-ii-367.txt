class Solution {

    public int removeDuplicates(int[] nums) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int num : nums) {
            if (map.containsKey(num)) {
                map.put(num, map.get(num) + 1);
            } else {
                map.put(num, 1);
            }
        }
        int idx = 0;
        for (int key : map.keySet()) {
            int cnt = 0;
            if (map.get(key) >= 2) {
                nums[idx] = key;
                idx++;
                nums[idx] = key;
                idx++;
            } else if (map.get(key) == 1) {
                nums[idx] = key;
                ;
                idx++;
            }
        }
        // int copy[]= new int[idx];
        // for(int i=0; i<idx; i++)
        // {
        //     copy[i]= nums[i];
        // }
        // Arrays.sort(copy);
        // for(int i=0; i<idx;i++)
        // {
        //     nums[i]= copy[i];
        // }
        return idx;
    }
}
