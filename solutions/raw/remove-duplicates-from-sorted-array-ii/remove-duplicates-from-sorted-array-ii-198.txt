class Solution {

    public int removeDuplicates(int[] nums) {
        int count = 0;
        for (int i = 0; i < nums.length - count; i++) {
            if ((i > 0 && i < nums.length - 1 - count) && (nums[i] == nums[i - 1]) && (nums[i + 1] == nums[i])) {
                for (int j = i; j < nums.length - 1; j++) {
                    nums[j] = nums[j + 1];
                }
                count++;
                i--;
            }
        }
        return (nums.length - count);
    }
}
