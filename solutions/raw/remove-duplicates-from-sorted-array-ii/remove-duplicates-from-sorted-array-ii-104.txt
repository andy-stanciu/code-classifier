class Solution {

    public int removeDuplicates(int[] arr) {
        int n = arr.length;
        // Length of the result array at any point
        int res = 1;
        // Count of last number added in the result array
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (arr[res - 1] != arr[i]) {
                // If current number is different than the previously inserted number
                // We insert it and set its count to 1
                arr[res++] = arr[i];
                count = 1;
            } else if (count < 2) {
                // If current number is same as the previously inserted number, but count < 2
                // We insert it and increment count by 1
                arr[res++] = arr[i];
                count++;
            }
        }
        return res;
    }
}
