class Solution {

    public int removeDuplicates(int[] nums) {
        //for j<2
        if (nums.length < 2)
            return nums.length;
        int i = 2;
        for (int j = 2; j < nums.length; j++) {
            if (nums[j] != nums[i - 2]) {
                nums[i++] = nums[j];
            }
        }
        return i;
    }
}
