class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 2;
        for (int j = 2; j < nums.length; j++) {
            if (nums[j] != nums[j - 1] || (nums[j] != nums[j - 2]) || (i == j - 1 && nums[j] == nums[j - 1] && nums[j] == nums[j - 2] && nums[j] != nums[j - 3])) {
                nums[i] = nums[j];
                i++;
            }
        }
        return i;
    }
}
