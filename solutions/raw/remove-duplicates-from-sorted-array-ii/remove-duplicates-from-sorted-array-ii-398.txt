class Solution {

    public int removeDuplicates(int[] nums) {
        int k = 1;
        if (nums.length == 1) {
            return k;
        }
        for (int i = 1; i < nums.length - 1; i++) {
            if (nums[i] != nums[i - 1] || nums[i] != nums[i + 1]) {
                nums[k] = nums[i];
                k++;
            }
        }
        nums[k] = nums[nums.length - 1];
        k++;
        return k;
    }
}
