class Solution {

    public int removeDuplicates(int[] nums) {
        // If the array is empty, return 0
        if (nums.length == 0)
            return 0;
        // Initialize k to 1, as the first element is always included
        int k = 1;
        // Initialize count to 1, as the first element is counted once
        int count = 1;
        // Iterate through the array starting from the second element
        for (int i = 1; i < nums.length; i++) {
            // If the current element is the same as the previous element
            if (nums[i] == nums[i - 1]) {
                count++;
            } else {
                // If the current element is different, reset count to 1
                count = 1;
            }
            // If count is less than or equal to 2, include the current element
            if (count <= 2) {
                nums[k] = nums[i];
                k++;
            }
        }
        // Return the number of elements in the modified array
        return k;
    }
}
