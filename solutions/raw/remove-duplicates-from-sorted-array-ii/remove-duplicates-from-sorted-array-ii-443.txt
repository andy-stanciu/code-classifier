class Solution {

    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        int count = 1;
        if (n == 1)
            return 1;
        int prev = nums[0];
        for (int i = 1; i < n; i++) {
            if (nums[i] == prev) {
                if (count >= 2)
                    nums[i] = Integer.MAX_VALUE;
                count++;
            } else {
                prev = nums[i];
                count = 1;
            }
        }
        Arrays.sort(nums);
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != Integer.MAX_VALUE)
                ans++;
            else
                return ans;
        }
        return nums.length;
    }
}
