class Solution {

    public int removeDuplicates(int[] nums) {
        if (nums.length < 2) {
            return 2;
        }
        int count = 2;
        for (int i = count; i < nums.length; i++) {
            if (nums[i] != nums[count - 2]) {
                nums[count] = nums[i];
                count++;
            }
        }
        return count;
    }
}
