class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 0;
        int j = 0;
        int last = -100001;
        int repeat = 0;
        int count = 0;
        while (j < nums.length) {
            if (last != nums[j]) {
                if (j > i) {
                    int temp = nums[j];
                    nums[j] = nums[i];
                    nums[i] = temp;
                }
                repeat = 0;
                last = nums[i];
                i++;
                j++;
            } else {
                if (repeat == 0) {
                    if (j > i) {
                        int temp = nums[j];
                        nums[j] = nums[i];
                        nums[i] = temp;
                    }
                    i++;
                    j++;
                    repeat++;
                } else {
                    j++;
                    count++;
                }
            }
        }
        return nums.length - count;
    }
}
