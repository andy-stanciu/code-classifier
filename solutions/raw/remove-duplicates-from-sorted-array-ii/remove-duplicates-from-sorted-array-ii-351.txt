class Solution {

    public int removeDuplicates(int[] nums) {
        // we start from index 1 as index 0 will be by default a unique element so start from index 1 and check if the second element is same ot not.
        int newArr = 1;
        //initalizes the number of occurences of an element
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            // we start from i=1 as the index 0 element would be a unique element and we can start from index 1
            if (nums[i] == nums[i - 1]) {
                // increment the count if current element is equal to previous element
                count++;
            } else {
                //we leave the count as it is
                count = 1;
            }
            if (count <= 2) {
                nums[newArr] = nums[i];
                newArr++;
            }
        }
        return newArr;
    }
}
