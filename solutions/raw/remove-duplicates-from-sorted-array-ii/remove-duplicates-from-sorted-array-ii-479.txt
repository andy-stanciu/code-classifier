class Solution {

    public int removeDuplicates(int[] nums) {
        int i = 1;
        // Count occurrences of the current element
        int count = 1;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] == nums[j - 1]) {
                count++;
            } else {
                // Reset the count for a new element
                count = 1;
            }
            if (count <= 2) {
                nums[i] = nums[j];
                i++;
            }
        }
        return i;
    }
}
