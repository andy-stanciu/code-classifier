class Solution {

    public int removeDuplicates(int[] nums) {
        boolean valid = true;
        int j = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1] && valid) {
                nums[j] = nums[i];
                j++;
                valid = false;
            }
            if (nums[i] != nums[i - 1]) {
                nums[j] = nums[i];
                j++;
                valid = true;
            }
        }
        return j;
    }
}
