class Solution {

    public int removeDuplicates(int[] nums) {
        int c = 1;
        int count = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[c++] = nums[i];
                count = 0;
            } else {
                if (count < 1) {
                    nums[c++] = nums[i];
                    count++;
                } else {
                    count++;
                }
            }
        }
        return c;
    }
}
