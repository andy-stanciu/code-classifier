class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        HashMap<Integer, Integer> hm = new HashMap<>();
        if (nums.length == 1) {
            ans.add(nums[0]);
            return ans;
        }
        for (int i = 0; i < nums.length; i++) {
            if (hm.containsKey(nums[i])) {
                hm.put(nums[i], hm.get(nums[i]) + 1);
                if (hm.get(nums[i]) > Math.floor(nums.length / 3) && !ans.contains(nums[i])) {
                    ans.add(nums[i]);
                }
            } else {
                hm.put(nums[i], 1);
                if (nums.length == 2 && !ans.contains(nums[i])) {
                    ans.add(nums[i]);
                }
            }
        }
        return ans;
    }
}
