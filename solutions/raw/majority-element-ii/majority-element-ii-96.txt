class Solution {

    public List<Integer> majorityElement(int[] nums) {
        Arrays.sort(nums);
        List<Integer> result = new ArrayList<>();
        if (nums.length == 0) {
            return result;
        }
        int length = nums.length;
        final int quota = length / 3;
        int lastVal = nums[0];
        int lastCount = 1;
        for (int index = 1; index < nums.length; index++) {
            if (nums[index] == lastVal) {
                lastCount++;
            } else {
                if (lastCount > quota) {
                    result.add(lastVal);
                }
                lastCount = 1;
                lastVal = nums[index];
            }
        }
        if (lastCount > quota) {
            result.add(lastVal);
        }
        return result;
    }
}
