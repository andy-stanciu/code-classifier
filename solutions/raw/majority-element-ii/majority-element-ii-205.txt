class Solution {

    public List<Integer> majorityElement(int[] nums) {
        Map<Integer, Integer> majors = new HashMap<>();
        for (int num : nums) {
            majors.put(num, majors.getOrDefault(num, 0) + 1);
        }
        List<Integer> res = new ArrayList<>();
        for (int key : majors.keySet()) {
            if (majors.get(key) > nums.length / 3)
                res.add(key);
        }
        return res;
    }
}
