class Solution {

    public List<Integer> majorityElement(int[] a) {
        int length = a.length;
        List<Integer> ls = new ArrayList<>();
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < length; i++) {
            hm.put(a[i], hm.getOrDefault(a[i], 0) + 1);
            if (hm.get(a[i]) > length / 3)
                if (!ls.contains(a[i]))
                    ls.add(a[i]);
        }
        return ls;
    }
}
