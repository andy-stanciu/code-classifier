class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int cnt1 = 0;
        int cnt2 = 0;
        int ele1 = Integer.MIN_VALUE;
        int ele2 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (cnt1 == 0 && ele2 != nums[i]) {
                cnt1++;
                ele1 = nums[i];
            } else if (cnt2 == 0 && ele1 != nums[i]) {
                cnt2++;
                ele2 = nums[i];
            } else if (ele1 == nums[i])
                cnt1++;
            else if (ele2 == nums[i])
                cnt2++;
            else {
                cnt1--;
                cnt2--;
            }
        }
        int cnt_1 = 0;
        int cnt_2 = 0;
        int mini = (int) (nums.length / 3) + 1;
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (ele1 == nums[i])
                cnt_1++;
            if (ele2 == nums[i])
                cnt_2++;
        }
        if (cnt_1 >= mini)
            ans.add(ele1);
        if (cnt_2 >= mini)
            ans.add(ele2);
        return ans;
    }
}
