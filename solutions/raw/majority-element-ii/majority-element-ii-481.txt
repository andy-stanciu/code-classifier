class Solution {

    public List<Integer> majorityElement(int[] nums) {
        ArrayList<Integer> arr = new ArrayList<Integer>();
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        int n = nums.length;
        int x = n / 3;
        for (int y : map.keySet()) {
            if (map.get(y) > x) {
                arr.add(y);
            }
        }
        return arr;
    }
}
