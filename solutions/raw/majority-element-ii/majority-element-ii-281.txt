class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int len = nums.length;
        int lowerCheck = len / 3;
        Map<Integer, Integer> arrList = new HashMap<>();
        //Set<Integer> str = new Set<>();
        List<Integer> arr = new ArrayList<>();
        for (int x : nums) {
            arrList.put(x, (arrList.getOrDefault(x, 0) + 1));
            // if(arrList.get(x) > lowerCheck)
            // {
            //  if(!arr.contains(x))
            // arr.add(x);
            // }
        }
        for (Map.Entry<Integer, Integer> entry : arrList.entrySet()) {
            int element = entry.getKey();
            int count = entry.getValue();
            if (count > lowerCheck) {
                arr.add(element);
            }
        }
        return arr;
    }
}
