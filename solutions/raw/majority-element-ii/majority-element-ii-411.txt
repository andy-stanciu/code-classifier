class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int c1 = 0, c2 = 0;
        int cnt1 = 0, cnt2 = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (cnt1 == 0 && nums[i] != c2) {
                cnt1 += 1;
                c1 = nums[i];
            } else if (cnt2 == 0 && nums[i] != c1) {
                cnt2 += 1;
                c2 = nums[i];
            } else if (nums[i] == c1)
                cnt1 += 1;
            else if (nums[i] == c2)
                cnt2 += 1;
            else {
                cnt1 -= 1;
                cnt2 -= 1;
            }
        }
        List<Integer> ans = new ArrayList<>();
        cnt1 = 0;
        cnt2 = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == c1)
                cnt1 += 1;
            else if (nums[i] == c2)
                cnt2 += 1;
        }
        if (cnt1 > n / 3)
            ans.add(c1);
        if (cnt2 > n / 3)
            ans.add(c2);
        return ans;
    }
}
