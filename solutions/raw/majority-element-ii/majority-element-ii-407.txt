class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int cnt1 = 0, cnt2 = 0;
        int elem1 = Integer.MIN_VALUE;
        int elem2 = Integer.MIN_VALUE;
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            if (cnt1 == 0 && nums[i] != elem2) {
                cnt1 = 1;
                elem1 = nums[i];
            } else if (cnt2 == 0 && nums[i] != elem1) {
                cnt2 = 1;
                elem2 = nums[i];
            } else if (elem1 == nums[i])
                cnt1++;
            else if (elem2 == nums[i])
                cnt2++;
            else {
                cnt1--;
                cnt2--;
            }
        }
        // manual check to add numbers in the list
        int c3 = 0;
        int c4 = 0;
        int boundry = (nums.length / 3) + 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == elem1) {
                c3++;
            } else if (nums[i] == elem2) {
                c4++;
            }
        }
        if (c3 >= boundry)
            list.add(elem1);
        if (c4 >= boundry)
            list.add(elem2);
        return list;
    }
}
