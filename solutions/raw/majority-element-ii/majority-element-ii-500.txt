class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int n = nums.length;
        List<Integer> ls = new ArrayList<>();
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < n; i++) {
            hm.put(nums[i], hm.getOrDefault(nums[i], 0) + 1);
        }
        for (Integer key : hm.keySet()) {
            if (hm.get(key) > n / 3) {
                ls.add(key);
            }
        }
        return ls;
    }
}
