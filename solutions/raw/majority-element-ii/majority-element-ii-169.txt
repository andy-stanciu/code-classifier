class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int n = nums.length / 3;
        List<Integer> res = new ArrayList<>();
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        for (int num : map.keySet()) {
            if (map.get(num) > n)
                res.add(num);
        }
        return res;
    }
}
