class Solution {

    public List<Integer> majorityElement(int[] nums) {
        Map<Integer, Integer> hp = new HashMap<>();
        for (int i : nums) {
            hp.put(i, hp.getOrDefault(i, 0) + 1);
        }
        int m = nums.length / 3;
        List<Integer> al = new LinkedList<>();
        for (int k : hp.keySet()) {
            if (hp.get(k) > m) {
                al.add(k);
            }
        }
        return al;
    }
}
