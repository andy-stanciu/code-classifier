class Solution {

    public List<Integer> majorityElement(int[] nums) {
        // Counters for the potential majority elements
        int count1 = 0, count2 = 0;
        // Potential majority element candidates
        int candidate1 = 0, candidate2 = 0;
        // First pass to find potential majority elements.
        for (int i = 0; i < nums.length; i++) {
            // If count1 is 0 and the current number is not equal to candidate2, update candidate1.
            if (count1 == 0 && nums[i] != candidate2) {
                count1 = 1;
                candidate1 = nums[i];
            } else // If count2 is 0 and the current number is not equal to candidate1, update candidate2.
            if (count2 == 0 && nums[i] != candidate1) {
                count2 = 1;
                candidate2 = nums[i];
            } else // Update counts for candidate1 and candidate2.
            if (candidate1 == nums[i]) {
                count1++;
            } else if (candidate2 == nums[i]) {
                count2++;
            } else // If the current number is different from both candidates, decrement their counts.
            {
                count1--;
                count2--;
            }
        }
        List<Integer> result = new ArrayList<>();
        // Threshold for majority element
        int threshold = nums.length / 3;
        // Second pass to count occurrences of the potential majority elements.
        count1 = 0;
        count2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (candidate1 == nums[i]) {
                count1++;
            } else if (candidate2 == nums[i]) {
                count2++;
            }
        }
        // Check if the counts of potential majority elements are greater than n/3 and add them to the result.
        if (count1 > threshold) {
            result.add(candidate1);
        }
        if (count2 > threshold) {
            result.add(candidate2);
        }
        return result;
    }
}
