class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int z = nums.length / 3;
        List<Integer> arr = new ArrayList<>();
        for (int i : nums) {
            int count = 0;
            for (int j : nums) {
                if (i == j)
                    count += 1;
            }
            if (count > z && !arr.contains(i))
                arr.add(i);
        }
        return arr;
    }
}
