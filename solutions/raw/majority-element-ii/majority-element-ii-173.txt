class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> ls = new ArrayList<>();
        int n = nums.length;
        if (n == 0)
            return ls;
        Arrays.sort(nums);
        int times = n / 3;
        int count = 1;
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                count++;
            } else {
                if (count > times) {
                    ls.add(nums[i]);
                }
                count = 1;
            }
        }
        if (count > times) {
            ls.add(nums[n - 1]);
        }
        return ls;
    }
}
