class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> distinctElements = new ArrayList<>();
        for (int num : nums) {
            distinctElements.add(num);
        }
        Map<Integer, Long> distinct = distinctElements.stream().collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        Set<Integer> finalValues = distinctElements.stream().filter(number -> distinct.get(number) > distinctElements.size() / 3).collect(Collectors.toSet());
        return new ArrayList<>(finalValues);
    }
}
