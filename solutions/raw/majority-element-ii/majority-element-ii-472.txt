class Solution {

    public List<Integer> majorityElement(int[] nums) {
        // As you know if number more than (n/3) occurence, and it can maximum 2;
        // Two Numbers
        int num1 = nums[0];
        int num2 = nums[0];
        int count1 = 0;
        int count2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == num1) {
                count1++;
            } else if (nums[i] == num2) {
                count2++;
            } else if (count1 == 0) {
                num1 = nums[i];
                count1 = 1;
            } else if (count2 == 0) {
                num2 = nums[i];
                count2 = 1;
            } else {
                count1--;
                count2--;
            }
        }
        // now check if num1 and num2 > n/3 then add to list;
        count1 = 0;
        count2 = 0;
        for (int i : nums) {
            if (i == num1)
                count1++;
            else if (i == num2)
                count2++;
        }
        List<Integer> list = new ArrayList<>();
        if (count1 > nums.length / 3) {
            list.add(num1);
        }
        if (count2 > nums.length / 3) {
            list.add(num2);
        }
        return list;
    }
}
