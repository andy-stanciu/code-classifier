class Solution {

    public List<Integer> majorityElement(int[] nums) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        ArrayList<Integer> ls = new ArrayList<>();
        for (int i : nums) {
            hm.put(i, hm.getOrDefault(i, 0) + 1);
        }
        int n = nums.length;
        int k = n / 3;
        for (int i : hm.keySet()) {
            if (hm.get(i) > k)
                ls.add(i);
        }
        return ls;
    }
}
