class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> ls = new ArrayList<>();
        Arrays.sort(nums);
        int count = 1;
        int n = nums.length;
        int floor = Math.floorDiv(n, 3);
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                count++;
            } else {
                if (count > floor) {
                    ls.add(nums[i]);
                }
                count = 1;
            }
        }
        if (count > floor) {
            ls.add(nums[n - 1]);
        }
        return ls;
    }
}
