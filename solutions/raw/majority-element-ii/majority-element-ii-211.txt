class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> a = new ArrayList<>();
        int el1 = 0, el2 = -1;
        int c1 = 0, c2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (c1 == 0 && nums[i] != el2) {
                el1 = nums[i];
                c1 = 1;
            } else if (c2 == 0 && nums[i] != el1) {
                el2 = nums[i];
                c2 = 1;
            } else if (nums[i] == el1) {
                c1++;
            } else if (nums[i] == el2) {
                c2++;
            } else {
                c1--;
                c2--;
            }
        }
        int temp1 = 0, temp2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == el1) {
                temp1++;
            } else if (nums[i] == el2) {
                temp2++;
            }
        }
        if (temp1 > (nums.length) / 3) {
            a.add(el1);
        }
        if (temp2 > (nums.length) / 3) {
            a.add(el2);
        }
        return a;
    }
}
