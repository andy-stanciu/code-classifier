class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int majority1 = 0;
        int majority2 = 0;
        int count1 = 0;
        int count2 = 0;
        List<Integer> list = new ArrayList<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (majority1 == nums[i]) {
                count1++;
            } else if (majority2 == nums[i]) {
                count2++;
            } else if (count1 == 0) {
                majority1 = nums[i];
                count1 = 1;
            } else if (count2 == 0) {
                majority2 = nums[i];
                count2 = 1;
            } else {
                count1--;
                count2--;
            }
        }
        int freq1 = 0;
        int freq2 = 0;
        for (int num : nums) {
            if (num == majority1) {
                freq1++;
            } else if (num == majority2) {
                freq2++;
            }
        }
        if (freq1 > (n / 3)) {
            list.add(majority1);
        }
        if (freq2 > (n / 3)) {
            list.add(majority2);
        }
        return list;
    }
}
