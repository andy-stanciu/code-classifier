class Solution {

    public List<Integer> majorityElement(int[] nums) {
        ArrayList<Integer> list = new ArrayList<>();
        Arrays.sort(nums);
        int n = nums.length;
        int count = 1;
        for (int i = 0; i < n; i++) {
            if (i < n - 1 && nums[i] == nums[i + 1]) {
                count++;
            } else {
                if (count > n / 3) {
                    list.add(nums[i]);
                }
                count = 1;
            }
        }
        return list;
    }
}
