class Solution {

    public List<Integer> majorityElement(int[] v) {
        int n = v.length;
        int n1 = -1, n2 = -1, c1 = 0, c2 = 0;
        for (int i = 0; i < n; i++) {
            if (n1 == v[i]) {
                c1++;
            } else if (n2 == v[i]) {
                c2++;
            } else if (c1 == 0) {
                n1 = v[i];
                c1 = 1;
            } else if (c2 == 0) {
                n2 = v[i];
                c2 = 1;
            } else {
                c1--;
                c2--;
            }
        }
        List<Integer> ans = new ArrayList<>();
        c1 = 0;
        c2 = 0;
        for (int i = 0; i < n; i++) {
            if (n1 == v[i]) {
                c1++;
            } else if (n2 == v[i]) {
                c2++;
            }
        }
        if (c1 > n / 3) {
            ans.add(n1);
        }
        if (c2 > n / 3) {
            ans.add(n2);
        }
        return ans;
    }
}
