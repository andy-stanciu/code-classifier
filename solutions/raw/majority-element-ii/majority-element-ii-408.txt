class Solution {

    public List<Integer> majorityElement(int[] nums) {
        /*
        HashMap
        O(N)

        traverse an check? O(N)

        space and time O(N)

        */
        Set<Integer> hs = new HashSet<>();
        int n = (int) Math.floor(nums.length / 3);
        Map<Integer, Integer> hm = new HashMap<>();
        for (int number : nums) {
            hm.put(number, hm.getOrDefault(number, 0) + 1);
            if (hm.get(number) > n) {
                hs.add(number);
            }
        }
        List<Integer> ls = new ArrayList<>();
        for (int v : hs) ls.add(v);
        return ls;
    }
}
