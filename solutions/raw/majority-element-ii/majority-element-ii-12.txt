class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int ct1 = 0, ct2 = 0;
        int el1 = Integer.MIN_VALUE, el2 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (ct1 == 0 && el2 != nums[i]) {
                ct1 = 1;
                el1 = nums[i];
            } else if (ct2 == 0 && el1 != nums[i]) {
                ct2 = 1;
                el2 = nums[i];
            } else if (el1 == nums[i])
                ct1++;
            else if (el2 == nums[i])
                ct2++;
            else {
                ct1--;
                ct2--;
            }
        }
        ct1 = 0;
        ct2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == el1)
                ct1++;
            if (nums[i] == el2)
                ct2++;
        }
        int gre = (int) nums.length / 3 + 1;
        ArrayList<Integer> list = new ArrayList<>();
        if (ct1 >= gre)
            list.add(el1);
        if (ct2 >= gre)
            list.add(el2);
        return list;
    }
}
