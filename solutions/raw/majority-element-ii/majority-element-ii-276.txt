class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int n = nums.length;
        int value = n / 3;
        Map<Integer, Integer> ElementCount = new HashMap<>();
        // iterate the input- array..
        for (int i = 0; i < n; i++) {
            ElementCount.put(nums[i], ElementCount.getOrDefault(nums[i], 0) + 1);
        }
        List<Integer> majorityElements = new ArrayList<>();
        // find the frequency of the  majority element
        for (var pair : ElementCount.entrySet()) {
            int element = pair.getKey();
            int count = pair.getValue();
            // cheak the count greater than n/3..
            if (count > value) {
                majorityElements.add(element);
            }
        }
        return majorityElements;
    }
}
