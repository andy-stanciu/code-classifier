class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> result = new ArrayList();
        HashMap<Integer, Integer> numberCountMap = new HashMap();
        for (int i = 0; i < nums.length; i++) {
            if (!numberCountMap.containsKey(nums[i])) {
                numberCountMap.put(nums[i], 1);
            } else {
                int increment = numberCountMap.get(nums[i]) + 1;
                numberCountMap.put(nums[i], increment);
            }
        }
        for (HashMap.Entry<Integer, Integer> entry : numberCountMap.entrySet()) {
            if (entry.getValue() > nums.length / 3) {
                result.add(entry.getKey());
            }
        }
        return result;
    }
}
