class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int count1 = 0, count2 = 0, candidate1 = 0, candidate2 = 0;
        for (int num : nums) {
            if (count1 == 0 && num != candidate2) {
                candidate1 = num;
                count1 = 1;
            } else if (count2 == 0 && num != candidate1) {
                candidate2 = num;
                count2 = 1;
            } else if (num == candidate1)
                count1++;
            else if (num == candidate2)
                count2++;
            else {
                count1--;
                count2--;
            }
        }
        List<Integer> res = new ArrayList<>();
        count1 = 0;
        count2 = 0;
        int thres = nums.length / 3;
        for (int num : nums) {
            if (candidate1 == num)
                count1++;
            else if (candidate2 == num)
                count2++;
        }
        if (count1 > thres)
            res.add(candidate1);
        if (count2 > thres)
            res.add(candidate2);
        return res;
    }
}
