class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> list = new ArrayList<Integer>();
        Arrays.sort(nums);
        int c = 1, n = nums.length;
        for (int i = 0; i < n; i++) {
            if (i < n - 1 && nums[i] == nums[i + 1]) {
                c++;
            } else {
                if (c > n / 3) {
                    list.add(nums[i]);
                }
                c = 1;
            }
        }
        return list;
    }
}
