class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        int n = nums.length;
        Map<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < n; i++) {
            hm.put(nums[i], hm.getOrDefault(nums[i], 0) + 1);
        }
        Set<Integer> s = hm.keySet();
        for (Integer x : s) {
            if (hm.get(x) > n / 3)
                ans.add(x);
        }
        return ans;
    }
}
