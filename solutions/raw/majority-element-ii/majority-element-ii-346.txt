class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int n = nums.length;
        int len = n / 3;
        HashMap<Integer, Integer> freq = new HashMap<>();
        for (int x : nums) {
            freq.put(x, freq.getOrDefault(x, 0) + 1);
        }
        List<Integer> list = new ArrayList<>();
        for (Map.Entry<Integer, Integer> x : freq.entrySet()) {
            if (x.getValue() > len) {
                list.add(x.getKey());
            }
        }
        return list;
    }
}
