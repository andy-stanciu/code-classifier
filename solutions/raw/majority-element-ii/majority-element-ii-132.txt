class Solution {

    public List<Integer> majorityElement(int[] nums) {
        List<Integer> list = new ArrayList<Integer>();
        int n = nums.length;
        Random rand = new Random();
        int el1 = Integer.MAX_VALUE;
        int el2 = Integer.MAX_VALUE;
        for (int j = 0; j < 7; j++) {
            int candidate = nums[rand.nextInt(n)];
            if (candidate == el1) {
                continue;
            }
            int count = 0;
            for (int num : nums) {
                if (num == candidate) {
                    count++;
                }
            }
            if (count * 3 > nums.length) {
                if (el1 == Integer.MAX_VALUE) {
                    el1 = candidate;
                    list.add(el1);
                } else {
                    el2 = candidate;
                    list.add(el2);
                    return list;
                }
            }
        }
        return list;
    }
}
