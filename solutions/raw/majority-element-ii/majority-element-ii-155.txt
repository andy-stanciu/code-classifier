class Solution {

    public List<Integer> majorityElement(int[] nums) {
        int majElem = nums.length / 3;
        ArrayList<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            int tempElem = nums[i];
            int count = 0;
            // Count occurrences of nums[i]
            for (int j = 0; j < nums.length; j++) {
                if (nums[j] == tempElem) {
                    count++;
                }
            }
            // If it\'s greater than n/3 and not already in the list
            if (count > majElem && !list.contains(tempElem)) {
                list.add(tempElem);
            }
        }
        return list;
    }
}
