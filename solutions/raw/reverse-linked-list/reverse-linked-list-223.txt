Solution {

    public ListNode reverseList(ListNode head) {
        if (head == null)
            return head;
        ListNode[] ans = new ListNode[1];
        reverse(head, ans).next = null;
        return ans[0];
    }

    private ListNode reverse(ListNode head, ListNode[] ans) {
        if (head.next == null) {
            ans[0] = head;
            return head;
        }
        reverse(head.next, ans).next = head;
        return head;
    }
}
