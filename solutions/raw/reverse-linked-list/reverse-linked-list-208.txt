class Solution {

    public ListNode reverseList(ListNode head) {
        Stack<ListNode> st = new Stack();
        ListNode l = new ListNode(0);
        ListNode temp = l;
        while (head != null) {
            st.push(head);
            head = head.next;
        }
        while (!st.isEmpty()) {
            temp.next = st.pop();
            temp = temp.next;
        }
        temp.next = null;
        return l.next;
    }
}
