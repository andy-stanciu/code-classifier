class Solution {

    public ListNode reverseList(ListNode head) {
        ListNode pre = null, cur = head, nxt;
        while (cur != null) {
            //to store next Node which is being DeLinked
            nxt = cur.next;
            cur.next = pre;
            pre = cur;
            cur = nxt;
        }
        head = pre;
        return head;
    }
}
