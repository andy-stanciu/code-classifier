class Solution {

    public ListNode reverseList(ListNode head) {
        ListNode curr = head;
        ListNode prev = null;
        while (curr != null) {
            //save reference in a temp Node
            ListNode temp = curr.next;
            // make next to current node as previous node
            curr.next = prev;
            //shift previous node to current node
            prev = curr;
            curr = temp;
        }
        return prev;
    }
}
