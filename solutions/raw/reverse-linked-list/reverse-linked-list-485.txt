class Solution {

    public ListNode reverseList(ListNode head) {
        //if only 1 node or no node no need to reverse
        if (head == null || head.next == null) {
            return head;
        }
        // here tempn is next node of temp
        ListNode pre = head, temp = head.next, tempn;
        //untill the last node
        while (temp != null) {
            tempn = temp.next;
            temp.next = pre;
            pre = temp;
            temp = tempn;
        }
        head.next = null;
        return pre;
    }
}
