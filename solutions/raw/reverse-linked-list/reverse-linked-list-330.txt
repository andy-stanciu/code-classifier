class Solution {

    public ListNode reverseList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode Current = head, Next, Previous = null;
        while (Current != null) {
            Next = Current.next;
            Current.next = Previous;
            Previous = Current;
            Current = Next;
        }
        return Previous;
    }
}
