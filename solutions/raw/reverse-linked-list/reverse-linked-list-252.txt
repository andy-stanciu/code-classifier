class Solution {

    ListNode reversedHead;

    public ListNode reverseList(ListNode head) {
        if (head == null)
            return null;
        reverse(head);
        head.next = null;
        return reversedHead;
    }

    private ListNode reverse(ListNode head) {
        if (head.next == null) {
            reversedHead = head;
            return head;
        }
        ListNode reversedTail = reverse(head.next);
        reversedTail.next = head;
        return head;
    }
}
