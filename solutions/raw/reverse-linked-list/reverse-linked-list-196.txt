class Solution {

    public ListNode reverseList(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        ListNode temp = head.next;
        ListNode prev = null;
        while (head != null) {
            head.next = prev;
            prev = head;
            head = temp;
            if (temp != null) {
                temp = temp.next;
            }
        }
        return prev;
    }
}
