class Solution {

    public ListNode reverseList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode prev = null;
        ListNode present = head;
        ListNode nxt = head.next;
        while (true) {
            present.next = prev;
            prev = present;
            present = nxt;
            if (present == null) {
                return prev;
            }
            nxt = nxt.next;
        }
    }
}
