class Solution {

    public ListNode reverseList(ListNode head) {
        //3 variables
        ListNode prev = null;
        ListNode curr = head;
        ListNode next;
        while (curr != null) {
            //4steps
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
}
