public class Solution {

    public static boolean isParentSum(Node root) {
        return reorder(root);
    }

    public static boolean reorder(Node root) {
        // Write your code here.
        if (root == null)
            return true;
        int child = 0;
        if (root.left != null) {
            child += root.left.data;
        }
        if (root.right != null) {
            child += root.right.data;
        }
        if ((root.left != null || root.right != null) && child != root.data) {
            return false;
        }
        boolean leftResult = reorder(root.left);
        boolean rightResult = reorder(root.right);
        return leftResult && rightResult;
    }
}
