class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> list = new ArrayList<>();
        zigzagLevelOrder(list, root, 0);
        return list;
    }

    public void zigzagLevelOrder(List<List<Integer>> list, TreeNode root, int height) {
        if (root == null)
            return;
        if (height >= list.size()) {
            list.add(new ArrayList<>());
        }
        if (height % 2 == 0) {
            list.get(height).add(root.val);
        } else {
            list.get(height).add(0, root.val);
        }
        zigzagLevelOrder(list, root.left, height + 1);
        zigzagLevelOrder(list, root.right, height + 1);
    }
}
