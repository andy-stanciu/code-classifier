class Solution {

    public ListNode reverseList(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode temp = head;
        while (temp.next != null) {
            temp = temp.next;
        }
        rev(head, head.next).next = null;
        return temp;
    }

    public ListNode rev(ListNode returnn, ListNode current) {
        if (current.next == null) {
            current.next = returnn;
            return returnn;
        }
        rev(current, current.next).next = returnn;
        return returnn;
    }
}
