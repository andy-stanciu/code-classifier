class Solution {

    public ListNode reverseList(ListNode head) {
        return reverseListRec(head, new ListNode(0)).next;
    }

    public ListNode reverseListRec(ListNode head, ListNode dummy) {
        if (head == null)
            return dummy;
        ListNode previous = dummy.next;
        dummy.next = head;
        ListNode next = head.next;
        head.next = previous;
        return reverseListRec(next, dummy);
    }
}
