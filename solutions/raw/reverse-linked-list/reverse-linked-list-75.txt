public class Solution {

    public static boolean isParentSum(Node root) {
        if (root == null)
            return true;
        Queue<Node> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            int sum = 0;
            Node cur = q.poll();
            if (cur.left != null) {
                sum += cur.left.data;
                q.offer(cur.left);
            }
            if (cur.right != null) {
                sum += cur.right.data;
                q.offer(cur.right);
            }
            if (sum == 0 && cur.left == null && cur.right == null)
                continue;
            if (sum != cur.data)
                return false;
        }
        return true;
    }
}
