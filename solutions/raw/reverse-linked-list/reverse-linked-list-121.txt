class Solution {

    static ListNode newHead = new ListNode(-1);

    ListNode temp;

    private void reverse(ListNode head) {
        if (head == null) {
            return;
        }
        reverse(head.next);
        temp.next = head;
        temp = temp.next;
    }

    public ListNode reverseList(ListNode head) {
        temp = newHead;
        reverse(head);
        temp.next = null;
        return newHead.next;
    }
}
