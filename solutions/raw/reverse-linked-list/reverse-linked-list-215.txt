class Solution {

    public ListNode reverseList(ListNode head) {
        ListNode curr = head, prev = null;
        while (curr != null) {
            //Base Case
            //1) Current Operation
            // Storing next node metadata
            ListNode next = curr.next;
            //Pointing to the previous node and breaking the previous link
            curr.next = prev;
            //2) Next Operation Preaparation
            //Moving the prev because it helps us to maingtain the previous node metadata
            prev = curr;
            //Moving the curr
            curr = next;
        }
        //returning the new head, since curr is null
        return prev;
    }
}
