class Solution {

    public ListNode reverseList(ListNode head) {
        ListNode dummy = new ListNode(0);
        ListNode previous = null;
        while (head != null) {
            // keep a copy of next
            ListNode next = head.next;
            // relink
            dummy.next = head;
            head.next = previous;
            // move the the next
            previous = head;
            head = next;
        }
        return dummy.next;
    }
}
