class Solution {

    public ListNode reverseList(ListNode head) {
        ListNode prev = null;
        ListNode curr = head;
        while (curr != null) {
            //preserve
            ListNode n = curr.next;
            //Build connection
            curr.next = prev;
            // move
            prev = curr;
            curr = n;
        }
        return prev;
    }
}
