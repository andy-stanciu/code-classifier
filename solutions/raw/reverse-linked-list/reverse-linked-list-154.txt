class Solution {

    public ListNode reverseList(ListNode head) {
        ListNode pre = null;
        ListNode curr = head;
        ListNode newNode = null;
        while (curr != null) {
            newNode = curr.next;
            curr.next = pre;
            pre = curr;
            curr = newNode;
        }
        head = pre;
        return head;
    }
}
