class Solution {

    ListNode tail = null;

    public ListNode reverseList(ListNode head) {
        if (head == null)
            return head;
        reverse(head);
        head.next = null;
        return tail;
    }

    public ListNode reverse(ListNode head) {
        if (head == null)
            return head;
        ListNode node = reverse(head.next);
        if (node == null)
            tail = head;
        else
            node.next = head;
        return head;
    }
}
