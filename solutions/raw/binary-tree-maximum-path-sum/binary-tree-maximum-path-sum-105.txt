/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int mxx = Integer.MIN_VALUE;

    public int getSol(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(getSol(root.left), 0);
        int right = Math.max(getSol(root.right), 0);
        int sum = root.val + left + right;
        mxx = Math.max(sum, mxx);
        return root.val + Math.max(left, right);
    }

    public int maxPathSum(TreeNode root) {
        getSol(root);
        return mxx;
    }
}
