/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int max;

    public int maxPathSum(TreeNode root) {
        max = Integer.MIN_VALUE;
        getMaxPathSum(root);
        return max;
    }

    private int getMaxPathSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lmax = Math.max(getMaxPathSum(root.left), 0);
        int rmax = Math.max(getMaxPathSum(root.right), 0);
        max = Math.max(max, lmax + rmax + root.val);
        return Math.max(lmax + root.val, rmax + root.val);
    }
}
