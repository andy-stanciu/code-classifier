class Solution {

    int maxSum = Integer.MIN_VALUE;

    int recurr(TreeNode root) {
        if (root == null)
            return 0;
        int left = Math.max(0, recurr(root.left));
        int right = Math.max(0, recurr(root.right));
        int currSum = root.val + left + right;
        maxSum = Math.max(maxSum, currSum);
        return root.val + Math.max(left, right);
    }

    public int maxPathSum(TreeNode root) {
        int temp = recurr(root);
        return maxSum;
    }
}
