class Solution {

    public static int solve(TreeNode node, int[] ans) {
        if (node == null) {
            return 0;
        }
        int left = solve(node.left, ans);
        int right = solve(node.right, ans);
        int max = Math.max(left, right);
        max = Math.max(node.val, max + node.val);
        ans[0] = Math.max(ans[0], max);
        ans[0] = Math.max(ans[0], left + right + node.val);
        return max;
    }

    public int maxPathSum(TreeNode root) {
        int[] ans = { root.val };
        solve(root, ans);
        return ans[0];
    }
}
