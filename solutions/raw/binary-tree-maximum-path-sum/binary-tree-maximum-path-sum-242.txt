/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root) {
        int[] arr = new int[] { Integer.MIN_VALUE };
        if (root.left == null && root.right == null) {
            return root.val;
        }
        getMaxSum(root, arr);
        return arr[0];
    }

    public int getMaxSum(TreeNode root, int[] arr) {
        if (root == null) {
            return 0;
        }
        int leftsum = Math.max(0, getMaxSum(root.left, arr));
        int rightsum = Math.max(0, getMaxSum(root.right, arr));
        arr[0] = Math.max(arr[0], leftsum + rightsum + root.val);
        return root.val + Math.max(leftsum, rightsum);
    }
}
