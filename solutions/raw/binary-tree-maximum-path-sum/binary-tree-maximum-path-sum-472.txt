/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        maxSumWithRoot(root);
        return maxSum;
    }

    private int maxSumWithRoot(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int maxPathSumleft = maxSumWithRoot(root.left);
        int maxPathSumRight = maxSumWithRoot(root.right);
        maxSum = Math.max(maxSum, maxPathSumleft + root.val + maxPathSumRight);
        int max = Math.max(maxPathSumleft, maxPathSumRight);
        if (max < 0) {
            max = 0;
        }
        max += root.val;
        maxSum = Math.max(maxSum, max);
        return max;
    }
}
