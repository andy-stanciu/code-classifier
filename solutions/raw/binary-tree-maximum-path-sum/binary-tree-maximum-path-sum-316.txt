/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root) {
        int[] max_path = new int[1];
        max_path[0] = Integer.MIN_VALUE;
        maxPathSumSolve(root, max_path);
        return max_path[0];
    }

    public int maxPathSumSolve(TreeNode root, int[] max_path) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(0, maxPathSumSolve(root.left, max_path));
        int right = Math.max(0, maxPathSumSolve(root.right, max_path));
        max_path[0] = Math.max(max_path[0], left + right + root.val);
        return root.val + Math.max(left, right);
    }
}
