/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxPathSum = Integer.MIN_VALUE;

    private int dfs(TreeNode root) {
        if (root == null)
            return 0;
        int l = dfs(root.left);
        int r = dfs(root.right);
        int maxToReturn = Math.max(l + root.val, Math.max(r + root.val, root.val));
        int max = Math.max(maxToReturn, l + root.val + r);
        maxPathSum = Math.max(maxPathSum, max);
        return maxToReturn;
    }

    public int maxPathSum(TreeNode root) {
        dfs(root);
        return maxPathSum;
    }
}
