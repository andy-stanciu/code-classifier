class Solution {

    int result = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        maxPath(root);
        return result;
    }

    public int maxPath(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftMaxSum = maxPath(root.left);
        int rightMaxSum = maxPath(root.right);
        result = Math.max(root.val, result);
        result = Math.max(result, leftMaxSum + root.val);
        result = Math.max(result, rightMaxSum + root.val);
        result = Math.max(result, leftMaxSum + rightMaxSum + root.val);
        int total = result;
        return Math.max(leftMaxSum + root.val, Math.max(rightMaxSum + root.val, root.val));
    }
}
