/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // Initialize with the smallest possible value
    private int sum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        maxiPath(root);
        return sum;
    }

    public int maxiPath(TreeNode root) {
        if (root == null)
            return 0;
        int leftsum = Math.max(0, maxiPath(root.left));
        int rightsum = Math.max(0, maxiPath(root.right));
        sum = Math.max(sum, root.val + leftsum + rightsum);
        return root.val + Math.max(leftsum, rightsum);
    }
}
