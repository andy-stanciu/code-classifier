class Solution {

    private int maxSum(TreeNode root, int[] ans) {
        if (root == null)
            return 0;
        int left = Math.max(maxSum(root.left, ans), 0);
        int right = Math.max(maxSum(root.right, ans), 0);
        ans[0] = Math.max(ans[0], root.val + left + right);
        return root.val + Math.max(left, right);
    }

    public int maxPathSum(TreeNode root) {
        int[] ans = { Integer.MIN_VALUE };
        maxSum(root, ans);
        return ans[0];
    }
}
