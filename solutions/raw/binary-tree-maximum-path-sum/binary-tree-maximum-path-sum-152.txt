/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root) {
        // array of size 1 to store the answer
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        maxsum(root, maxi);
        return maxi[0];
    }

    public int maxsum(TreeNode root, int[] maxi) {
        if (root == null)
            return 0;
        // calculate max sum on the left
        int l = Math.max(0, maxsum(root.left, maxi));
        // calculate max sum on the right
        int r = Math.max(0, maxsum(root.right, maxi));
        // update the max path sum including the current node
        maxi[0] = Math.max(maxi[0], l + r + root.val);
        // return the maximum path sum starting from the current node
        return Math.max(l, r) + root.val;
    }
}
