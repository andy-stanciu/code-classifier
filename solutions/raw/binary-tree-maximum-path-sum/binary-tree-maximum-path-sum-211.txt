/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int findSum(TreeNode root, int[] maxSum) {
        if (root == null)
            return 0;
        int ls = Math.max(0, findSum(root.left, maxSum));
        int rs = Math.max(0, findSum(root.right, maxSum));
        maxSum[0] = Math.max(maxSum[0], ls + rs + root.val);
        return Math.max(ls, rs) + root.val;
    }

    public int maxPathSum(TreeNode root) {
        int[] maxSum = { Integer.MIN_VALUE };
        findSum(root, maxSum);
        return maxSum[0];
    }
}
