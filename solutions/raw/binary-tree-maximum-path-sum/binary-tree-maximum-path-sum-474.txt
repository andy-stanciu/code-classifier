/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int max(TreeNode root, int[] arr) {
        if (root == null)
            return 0;
        int left = Math.max(0, max(root.left, arr));
        int right = Math.max(0, max(root.right, arr));
        int maxSumWithRoot = left + right + root.val;
        arr[0] = Math.max(arr[0], maxSumWithRoot);
        return Math.max(left, right) + root.val;
    }

    public int maxPathSum(TreeNode root) {
        int[] sum = new int[1];
        sum[0] = Integer.MIN_VALUE;
        max(root, sum);
        return sum[0];
    }
}
