/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // Initialize maxSum to the smallest possible integer value
    int maxSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        inorderTraversal(root);
        return maxSum;
    }

    private int inorderTraversal(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftSum = Math.max(0, inorderTraversal(root.left));
        int rightSum = Math.max(0, inorderTraversal(root.right));
        // Update maxSum by considering the current subtree\'s path sum
        maxSum = Math.max(maxSum, leftSum + rightSum + root.val);
        // Return the maximum path sum for the current subtree
        return Math.max(leftSum, rightSum) + root.val;
    }
}
