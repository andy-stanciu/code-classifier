/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxPathSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        dfs(root);
        return maxPathSum;
    }

    private int dfs(TreeNode root) {
        if (root == null)
            return 0;
        int leftPathSum = dfs(root.left);
        int rightPathSum = dfs(root.right);
        int rootPathSum = root.val + Math.max(leftPathSum, 0) + Math.max(rightPathSum, 0);
        maxPathSum = Math.max(maxPathSum, rootPathSum);
        return Math.max(leftPathSum, 0) > Math.max(rightPathSum, 0) ? root.val + Math.max(leftPathSum, 0) : root.val + Math.max(rightPathSum, 0);
    }
}
