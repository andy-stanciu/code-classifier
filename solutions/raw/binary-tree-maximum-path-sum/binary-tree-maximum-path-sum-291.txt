/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int ans;

    public static int rec(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int l = rec(root.left);
        int r = rec(root.right);
        int mx = Math.max(root.val + l, Math.max(root.val + r, root.val));
        ans = Math.max(ans, Math.max(mx, l + r + root.val));
        return mx;
    }

    public int maxPathSum(TreeNode root) {
        ans = root.val;
        int temp = rec(root);
        return ans;
    }
}
