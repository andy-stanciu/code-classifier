/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static List<Integer> sum;

    public static int dfs(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int left, right;
        left = dfs(node.left);
        right = dfs(node.right);
        int sumWithRoot = left + node.val + right;
        left = left > right ? left + node.val : right + node.val;
        left = node.val > left ? node.val : left;
        sumWithRoot = sumWithRoot > left ? sumWithRoot : left;
        sum.add(sumWithRoot);
        return left;
    }

    public int maxPathSum(TreeNode root) {
        sum = new ArrayList<>();
        dfs(root);
        int maxsum = Integer.MIN_VALUE;
        for (int i : sum) {
            if (i > maxsum) {
                maxsum = i;
            }
        }
        return maxsum;
    }
}
