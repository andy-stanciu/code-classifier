/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        dfs(root);
        return max;
    }

    int dfs(TreeNode node) {
        int leftNodeSum = Integer.MIN_VALUE;
        int rightNodeSum = Integer.MIN_VALUE;
        if (node.left != null) {
            leftNodeSum = Math.max(leftNodeSum, dfs(node.left));
        }
        if (node.right != null) {
            rightNodeSum = Math.max(rightNodeSum, dfs(node.right));
        }
        int value = node.val;
        leftNodeSum = Math.max(leftNodeSum, 0);
        rightNodeSum = Math.max(rightNodeSum, 0);
        max = Math.max(max, value + leftNodeSum + rightNodeSum);
        value += Math.max(leftNodeSum, rightNodeSum);
        max = Math.max(max, value);
        return value;
    }
}
