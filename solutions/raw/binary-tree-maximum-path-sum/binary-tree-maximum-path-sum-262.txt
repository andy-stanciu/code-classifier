class Solution {

    int maxSumAnywhere = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        int a = max(root);
        return Math.max(maxSumAnywhere, a);
    }

    private int max(TreeNode node) {
        if (node == null)
            return 0;
        int left = max(node.left);
        int right = max(node.right);
        maxSumAnywhere = Math.max(Math.max(maxSumAnywhere, Math.max(Math.max(left, right) + node.val, left + right + node.val)), node.val);
        return Math.max(Math.max(left, right) + node.val, node.val);
    }
}
