/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxSum(TreeNode root, int[] ans) {
        if (root == null)
            return 0;
        int lsum = maxSum(root.left, ans);
        int rsum = maxSum(root.right, ans);
        ans[0] = Math.max(ans[0], root.val + rsum + lsum);
        int val = root.val + Math.max(lsum, rsum);
        if (val < 0)
            return 0;
        return val;
    }

    public int maxPathSum(TreeNode root) {
        int[] ans = new int[1];
        ans[0] = Integer.MIN_VALUE;
        maxSum(root, ans);
        return ans[0];
    }
}
