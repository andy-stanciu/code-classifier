/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        //Setting to min as we have negative numbers
        postOrderTraversal(root);
        //If we have max as 0
        //[return 0 or root.val as value as per requirement]
        return max;
    }

    private int postOrderTraversal(TreeNode node) {
        //Base Condition
        if (node == null)
            return 0;
        int left = Math.max(postOrderTraversal(node.left), 0);
        int right = Math.max(postOrderTraversal(node.right), 0);
        max = Math.max(max, left + right + node.val);
        return Math.max(left, right) + node.val;
    }
}
