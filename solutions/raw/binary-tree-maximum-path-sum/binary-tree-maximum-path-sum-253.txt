class Solution {

    private int maxPathSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        findMaxSum(root);
        return maxPathSum;
    }

    private int findMaxSum(TreeNode root) {
        if (root == null)
            return 0;
        // Recursively find the maximum path sum for left and right subtrees
        int leftMax = Math.max(findMaxSum(root.left), 0);
        int rightMax = Math.max(findMaxSum(root.right), 0);
        // Calculate the maximum path sum passing through the current node
        int currMax = root.val + leftMax + rightMax;
        // Update the global maximum path sum
        maxPathSum = Math.max(maxPathSum, currMax);
        // Return the maximum path sum extending from the current node
        return root.val + Math.max(leftMax, rightMax);
    }
}
