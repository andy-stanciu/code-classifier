/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root, int[] res) {
        if (root == null) {
            return 0;
        }
        int leftMax = Math.max(maxPathSum(root.left, res), 0);
        int rightMax = Math.max(maxPathSum(root.right, res), 0);
        res[0] = Math.max(res[0], leftMax + root.val + rightMax);
        return root.val + Math.max(leftMax, rightMax);
    }

    public int maxPathSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] res = new int[1];
        res[0] = root.val;
        maxPathSum(root, res);
        return res[0];
    }
}
