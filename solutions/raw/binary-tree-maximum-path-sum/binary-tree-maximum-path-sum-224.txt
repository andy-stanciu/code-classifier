/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int result = 0;

    int summer(TreeNode root) {
        if (root == null)
            return 0;
        int l = summer(root.left);
        int r = summer(root.right);
        result = Math.max(result, Math.max(root.val, Math.max(root.val + l, Math.max(root.val + r, root.val + l + r))));
        return Math.max(root.val, Math.max(root.val + l, root.val + r));
    }

    public int maxPathSum(TreeNode root) {
        result = root.val;
        summer(root);
        return result;
    }
}
