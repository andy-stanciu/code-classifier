/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int pathSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        maxPathSumHelper(root);
        return pathSum;
    }

    public int maxPathSumHelper(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftHeightSum = Math.max(0, maxPathSumHelper(root.left));
        int rightHeightSum = Math.max(0, maxPathSumHelper(root.right));
        pathSum = Math.max(pathSum, root.val + leftHeightSum + rightHeightSum);
        return root.val + Math.max(leftHeightSum, rightHeightSum);
    }
}
