/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root) {
        int[] MaxV = new int[1];
        MaxV[0] = Integer.MIN_VALUE;
        MaxResult(root, MaxV);
        return MaxV[0];
    }

    private int MaxResult(TreeNode root, int[] MaxV) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(0, MaxResult(root.left, MaxV));
        int right = Math.max(0, MaxResult(root.right, MaxV));
        MaxV[0] = Math.max(MaxV[0], left + right + root.val);
        return Math.max(left, right) + root.val;
    }
}
