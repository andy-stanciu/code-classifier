class Solution {

    public int maxPathSum(TreeNode root) {
        dfs(root);
        return maxSum;
    }

    int maxSum = Integer.MIN_VALUE;

    int dfs(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int maxLeft = dfs(node.left);
        int maxRight = dfs(node.right);
        // update max through this node as root
        maxSum = Math.max(maxSum, node.val + maxLeft + maxRight);
        // return max of single branch, or 0
        return Math.max(0, node.val + Math.max(maxLeft, maxRight));
    }
}
