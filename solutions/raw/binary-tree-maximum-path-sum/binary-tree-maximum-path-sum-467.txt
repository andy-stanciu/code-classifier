/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        dfs(0, root);
        return max;
    }

    int dfs(int sum, TreeNode root) {
        if (root == null) {
            return 0;
        }
        max = Math.max(max, root.val);
        int totalSum = sum + root.val;
        int leftSum = dfs(totalSum, root.left);
        int rightSum = dfs(totalSum, root.right);
        max = Math.max(max, root.val + leftSum + rightSum);
        max = Math.max(max, root.val + leftSum);
        max = Math.max(max, root.val + rightSum);
        return Math.max(root.val, root.val + Math.max(leftSum, rightSum));
    }
}
