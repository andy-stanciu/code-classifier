class Solution {

    int ans = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        update(root);
        return ans;
    }

    private int update(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftsum = Math.max(update(root.left), 0);
        int rightsum = Math.max(update(root.right), 0);
        ans = Math.max(ans, leftsum + rightsum + root.val);
        return Math.max(leftsum + root.val, rightsum + root.val);
    }
}
