class Solution {

    public int maxPathSum(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] res = { Integer.MIN_VALUE };
        maxdown(root, res);
        return res[0];
    }

    public int maxdown(TreeNode root, int[] res) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(0, maxdown(root.left, res));
        int right = Math.max(0, maxdown(root.right, res));
        res[0] = Math.max(res[0], left + right + root.val);
        return Math.max(Math.max(left, right) + root.val, root.val);
    }
}
