/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxPathSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        maxPathSumCalculation(root);
        return maxPathSum;
    }

    private int maxPathSumCalculation(TreeNode root) {
        if (root == null)
            return 0;
        int leftSum = maxPathSumCalculation(root.left);
        int rightSum = maxPathSumCalculation(root.right);
        int result = root.val + Math.max(leftSum, rightSum);
        result = Math.max(result, root.val);
        maxPathSum = Math.max(maxPathSum, root.val + leftSum + rightSum);
        maxPathSum = Math.max(maxPathSum, result);
        return result;
    }
}
