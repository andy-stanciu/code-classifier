/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root) {
        //recursive
        int[] result = new int[1];
        result[0] = Integer.MIN_VALUE;
        maxPathSumRecursive(root, result);
        return result[0];
    }

    private int maxPathSumRecursive(TreeNode root, int[] result) {
        if (root == null)
            return 0;
        int leftSum = Math.max(0, maxPathSumRecursive(root.left, result));
        int rightSum = Math.max(0, maxPathSumRecursive(root.right, result));
        result[0] = Math.max(result[0], leftSum + rightSum + root.val);
        return root.val + Math.max(leftSum, rightSum);
    }
}
