/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int solver(TreeNode root, int[] sum) {
        if (root == null)
            return 0;
        int leftSum = Math.max(solver(root.left, sum), 0);
        int rightSum = Math.max(solver(root.right, sum), 0);
        sum[0] = Math.max(sum[0], leftSum + rightSum + root.val);
        return root.val + Math.max(leftSum, rightSum);
    }

    public int maxPathSum(TreeNode root) {
        int[] sum = new int[1];
        sum[0] = Integer.MIN_VALUE;
        solver(root, sum);
        return sum[0];
    }
}
