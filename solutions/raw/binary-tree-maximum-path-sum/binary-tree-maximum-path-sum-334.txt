/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxPathSum(TreeNode root) {
        int[] maxSum = { -1001 };
        func(root, maxSum);
        return maxSum[0];
    }

    public int func(TreeNode root, int[] maxSum) {
        if (root == null) {
            return 0;
        }
        int lH = Math.max(0, func(root.left, maxSum));
        int rH = Math.max(0, func(root.right, maxSum));
        maxSum[0] = Math.max(maxSum[0], root.val + lH + rH);
        return (root.val) + Math.max(lH, rH);
    }
}
