/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        Queue<TreeNode> que = new LinkedList<>();
        List<Double> li = new ArrayList<>();
        que.add(root);
        while (!que.isEmpty()) {
            int size = que.size();
            Double sum = 0.0;
            int count = 0;
            while (size > 0) {
                TreeNode node = que.poll();
                if (node != null) {
                    sum = sum + node.val;
                    count++;
                    if (node.left != null)
                        que.add(node.left);
                    if (node.right != null)
                        que.add(node.right);
                }
                size--;
            }
            Double avg = (double) sum / count;
            li.add(avg);
        }
        return li;
    }
}
