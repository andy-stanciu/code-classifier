/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        q.add(null);
        List<Double> ans = new ArrayList<>();
        while (q.peek() != null) {
            double sum = 0;
            int node = 0;
            while (q.peek() != null) {
                TreeNode curr = q.poll();
                sum = sum + curr.val;
                node++;
                if (curr.left != null) {
                    q.add(curr.left);
                }
                if (curr.right != null) {
                    q.add(curr.right);
                }
            }
            q.add(q.poll());
            ans.add(sum / node);
        }
        return ans;
    }
}
