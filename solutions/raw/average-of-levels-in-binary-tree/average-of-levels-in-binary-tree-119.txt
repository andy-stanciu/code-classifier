/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        double sum = 0, n = 0, average = 0;
        Queue<TreeNode> qu = new LinkedList<>();
        ArrayList<Double> list = new ArrayList<>();
        qu.add(root);
        qu.add(null);
        while (!qu.isEmpty()) {
            root = qu.poll();
            sum += root.val;
            n++;
            if (root.left != null)
                qu.add(root.left);
            if (root.right != null)
                qu.add(root.right);
            if (qu.peek() == null) {
                average = sum / n;
                list.add(average);
                average = 0;
                sum = 0;
                n = 0;
                qu.poll();
                if (qu.isEmpty())
                    break;
                qu.add(null);
            }
        }
        return list;
    }
}
