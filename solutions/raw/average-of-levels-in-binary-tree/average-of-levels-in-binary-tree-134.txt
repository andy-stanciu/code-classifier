/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {

        List<TreeNode> que = new ArrayList<>();
        que.add(root);

            for (int i = 0; i < x; i++) {
                TreeNode a = que.remove(0);
                temp += (double) a.val;
                if (a.left != null) {
                    que.add(a.left);
                }
                if (a.right != null) {
