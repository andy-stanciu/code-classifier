/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        Queue<TreeNode> qu = new LinkedList<>();
        qu.add(root);
        int c, cn;
        double sum;
        TreeNode node;
        List<Double> result = new ArrayList<>();
        while (!qu.isEmpty()) {
            c = qu.size();
            cn = c;
            sum = 0;
            while (c > 0) {
                node = qu.poll();
                sum += node.val;
                if (node.left != null) {
                    qu.add(node.left);
                }
                if (node.right != null) {
                    qu.add(node.right);
                }
                c--;
            }
            result.add(sum / cn);
        }
        return result;
    }
}
