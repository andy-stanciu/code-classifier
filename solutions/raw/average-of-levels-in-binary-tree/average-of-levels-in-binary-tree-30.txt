class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        List<Double> result = new ArrayList<>();
        while (!q.isEmpty()) {
            int count = q.size();
            int i = 0;
            double sum = 0;
            while (i++ < count) {
                TreeNode node = q.poll();
                if (node.left != null)
                    q.add(node.left);
                if (node.right != null)
                    q.add(node.right);
                sum += node.val;
            }
            result.add(sum / count);
        }
        return result;
    }
}
