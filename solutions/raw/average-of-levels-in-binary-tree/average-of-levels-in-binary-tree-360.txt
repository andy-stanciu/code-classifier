/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    class levelInfo {

        double sum;

        int count;

        levelInfo(int sum, int count) {
            this.sum = sum;
            this.count = count;
        }

        public double getSum() {
            return sum;
        }

        public void setSum(double sum) {
            this.sum = sum;
        }

        public int getCount() {
            return count;
        }

        public void setCount(int count) {
            this.count = count;
        }
    }

    public List<Double> averageOfLevels(TreeNode root) {
        Map<Integer, levelInfo> map = new HashMap<>();
        List<Double> result = new ArrayList<>();
        search(root, 0, map);
        for (int i = 0; i < map.size(); i++) {
            result.add((map.get(i).getSum() / map.get(i).getCount()));
        }
        return result;
    }

    public void search(TreeNode root, int level, Map<Integer, levelInfo> map) {
        if (root == null) {
            return;
        }
        if (!map.containsKey(level)) {
            map.put(level, new levelInfo(root.val, 1));
        } else {
            levelInfo levelInfo = map.get(level);
            levelInfo.setCount(levelInfo.getCount() + 1);
            levelInfo.setSum(levelInfo.getSum() + root.val);
            map.put(level, levelInfo);
        }
        search(root.left, level + 1, map);
        search(root.right, level + 1, map);
    }
}
