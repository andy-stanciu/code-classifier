/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode node) {
        List<Double> result = new ArrayList<Double>();
        if (node == null) {
            return result;
        }
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        queue.offer(node);
        while (!queue.isEmpty()) {
            int queueSize = queue.size();
            double sum = 0;
            for (int i = 1; i <= queueSize; i++) {
                TreeNode cureentNodeValue = queue.poll();
                sum += cureentNodeValue.val;
                if (cureentNodeValue.left != null) {
                    queue.offer(cureentNodeValue.left);
                }
                if (cureentNodeValue.right != null) {
                    queue.offer(cureentNodeValue.right);
                }
            }
            result.add(sum / queueSize);
        }
        return result;
    }
}
