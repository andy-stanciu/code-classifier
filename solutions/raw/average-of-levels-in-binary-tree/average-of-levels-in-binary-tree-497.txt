class Solution {

    Map<Integer, Double> sum = new HashMap();

    Map<Integer, Double> cnt = new HashMap();

    public List<Double> averageOfLevels(TreeNode root) {
        dfs(root, 0);
        List<Double> ret = new ArrayList();
        for (int k : sum.keySet()) {
            ret.add(sum.get(k) / cnt.get(k));
        }
        return ret;
    }

    private void dfs(TreeNode root, int depth) {
        if (root == null) {
            return;
        }
        depth++;
        sum.put(depth, sum.getOrDefault(depth, 0d) + root.val);
        cnt.put(depth, cnt.getOrDefault(depth, 0d) + 1);
        dfs(root.left, depth);
        dfs(root.right, depth);
    }
}
