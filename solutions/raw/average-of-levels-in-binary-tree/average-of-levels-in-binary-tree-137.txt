/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> avg = new LinkedList<>();
        avg.offer(root);
        while (!avg.isEmpty()) {
            int count = 0;
            double sum = 0;
            int levelSize = avg.size();
            for (int i = 0; i < levelSize; i++) {
                TreeNode currentNode = avg.poll();
                sum += currentNode.val;
                count++;
                if (currentNode.left != null) {
                    avg.offer(currentNode.left);
                }
                if (currentNode.right != null) {
                    avg.offer(currentNode.right);
                }
            }
            result.add(sum / count);
        }
        return result;
    }
}
