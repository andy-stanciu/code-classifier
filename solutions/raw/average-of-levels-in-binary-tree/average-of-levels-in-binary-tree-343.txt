/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        List<Double> al = new ArrayList();
        Queue<TreeNode> q = new ArrayDeque();
        q.add(root);
        while (!q.isEmpty()) {
            int numberOfNodesAtCurrentLevel = q.size();
            double levelSum = 0;
            for (int i = 0; i < numberOfNodesAtCurrentLevel; i++) {
                TreeNode curr = q.poll();
                levelSum = levelSum + (double) curr.val;
                if (curr.left != null) {
                    q.add(curr.left);
                }
                if (curr.right != null) {
                    q.add(curr.right);
                }
            }
            al.add(levelSum / numberOfNodesAtCurrentLevel);
        }
        return al;
    }
}
