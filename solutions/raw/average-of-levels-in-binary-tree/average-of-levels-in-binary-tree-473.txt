class Solution {

    public List<Double> averageOfLevels(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        Queue<TreeNode> que = new LinkedList<>();
        List<Double> res = new ArrayList<>();
        que.add(root);
        while (que.size() != 0) {
            int currSize = que.size();
            int div = que.size();
            double avg = 0.0;
            while (currSize-- > 0) {
                TreeNode node = que.remove();
                avg += node.val;
                if (node.left != null)
                    que.add(node.left);
                if (node.right != null)
                    que.add(node.right);
            }
            res.add(avg / div);
        }
        return res;
    }
}
