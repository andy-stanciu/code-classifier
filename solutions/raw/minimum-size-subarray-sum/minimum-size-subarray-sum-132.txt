class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int l = 0;
        int ans = Integer.MAX_VALUE;
        int temp = 0;
        for (int r = 0; r < n; r++) {
            temp += nums[r];
            while (temp >= target) {
                ans = Math.min(ans, r - l + 1);
                temp -= nums[l];
                l++;
            }
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
