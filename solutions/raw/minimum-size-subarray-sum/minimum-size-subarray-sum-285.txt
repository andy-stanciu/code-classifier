class Solution {

    public int minSubArrayLen(int k, int[] nums) {
        int j = 0, i = 0;
        int sum = 0;
        int min = Integer.MAX_VALUE;
        int n = nums.length;
        while (j < n) {
            sum += nums[j];
            while (sum >= k) {
                min = Math.min(min, (j - i + 1));
                sum -= nums[i];
                i++;
            }
            j++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
