class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = nums.length, current_min = 0;
        boolean reached = false;
        int sum = 0;
        int left = 0;
        for (int i = 0; i < nums.length; i++) {
            if (sum >= target)
                return 1;
            else {
                sum += nums[i];
                current_min++;
                while (sum >= target) {
                    reached = true;
                    sum -= nums[left];
                    left++;
                    current_min--;
                }
            }
            if (reached == true)
                min = Math.min(min, current_min + 1);
        }
        if (reached == false)
            return 0;
        return min;
    }
}
