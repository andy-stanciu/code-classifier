class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0, r = 0, sum = 0, ans = Integer.MAX_VALUE;
        boolean noAns = true;
        while (r < nums.length) {
            sum += nums[r];
            while (sum >= target) {
                ans = Math.min(r - l + 1, ans);
                sum -= nums[l++];
                noAns = false;
            }
            r++;
        }
        if (noAns)
            return 0;
        return ans;
    }
}
