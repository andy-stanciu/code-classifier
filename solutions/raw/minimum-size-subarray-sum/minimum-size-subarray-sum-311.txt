class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0;
        int current = 0;
        int min = Integer.MAX_VALUE;
        boolean found = false;
        for (int end = 0; end < nums.length; end++) {
            current += nums[end];
            while (current >= target) {
                min = Math.min(min, end - start + 1);
                current -= nums[start];
                start++;
                found = true;
            }
        }
        if (found) {
            return min;
        }
        return 0;
    }
}
