class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0, j = 0, sum = 0, minlen = 1000000;
        while (j < nums.length) {
            sum += nums[j];
            while (sum >= target) {
                minlen = (minlen > j - i + 1) ? (j - i + 1) : minlen;
                sum -= nums[i];
                i++;
            }
            j++;
        }
        if (minlen == 1000000) {
            return 0;
        }
        return minlen;
    }
}
