class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        //sliding windows
        int left = 0;
        int right = 0;
        int shortest = Integer.MAX_VALUE;
        int sum = 0;
        int numLen = nums.length;
        while (right < numLen) {
            sum = sum + nums[right];
            if (sum >= target) {
                while (sum >= target) {
                    sum = sum - nums[left];
                    left++;
                }
                shortest = Math.min(shortest, right - left + 2);
            }
            right++;
        }
        return shortest == Integer.MAX_VALUE ? 0 : shortest;
    }
}
