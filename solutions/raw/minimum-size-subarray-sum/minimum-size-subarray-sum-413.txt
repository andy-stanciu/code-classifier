class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0;
        int len = Integer.MAX_VALUE;
        int j = 0;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            count++;
            while (sum >= target) {
                len = Math.min(len, count);
                sum -= nums[j++];
                count--;
            }
        }
        if (len == Integer.MAX_VALUE)
            return 0;
        else
            return len;
    }
}
