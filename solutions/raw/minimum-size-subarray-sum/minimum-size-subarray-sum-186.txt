class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int len = 0;
        int minLen = Integer.MAX_VALUE;
        int sum = 0;
        int end = 0;
        int start = 0;
        while (end < nums.length) {
            sum = sum + nums[end];
            //move start.
            while (sum >= target) {
                len = end - start + 1;
                minLen = Math.min(len, minLen);
                sum = sum - nums[start];
                start++;
            }
            end++;
        }
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
