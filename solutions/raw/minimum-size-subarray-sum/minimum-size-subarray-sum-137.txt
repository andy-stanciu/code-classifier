class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int i = 0;
        int j = 0;
        int min = Integer.MAX_VALUE;
        int sum = 0;
        while (j < n) {
            sum += nums[j];
            while (sum > target) {
                min = Math.min(min, j - i + 1);
                sum = sum - nums[i];
                i++;
            }
            if (sum == target) {
                min = Math.min(min, j - i + 1);
            }
            j++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
