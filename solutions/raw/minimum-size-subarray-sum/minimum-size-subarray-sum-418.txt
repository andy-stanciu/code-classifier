class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int count = Integer.MAX_VALUE, sum = 0;
        for (int start = 0, end = 0; end < n; end++) {
            sum += nums[end];
            while (sum >= target) {
                count = Math.min(count, end - start + 1);
                sum = sum - nums[start++];
            }
        }
        return count == Integer.MAX_VALUE ? 0 : count;
    }
}
