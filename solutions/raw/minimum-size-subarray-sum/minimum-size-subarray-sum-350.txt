class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int n = nums.length;
        int sum = 0;
        while (j < n) {
            //calculations
            sum = sum + nums[j];
            while (sum >= target) {
                sum = sum - nums[i];
                // storing window size
                min = Math.min(min, j - i + 1);
                i++;
            }
            j++;
        }
        if (min == Integer.MAX_VALUE)
            return 0;
        return min;
    }
}
