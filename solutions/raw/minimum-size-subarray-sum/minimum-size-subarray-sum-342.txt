class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0;
        int j = 0;
        int min = nums.length + 1;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            while (sum >= target) {
                if (min > i - j + 1) {
                    min = i - j + 1;
                }
                sum -= nums[j++];
            }
            if (min == 1) {
                return min;
            }
        }
        if (min == nums.length + 1) {
            return 0;
        }
        return min;
    }
}
