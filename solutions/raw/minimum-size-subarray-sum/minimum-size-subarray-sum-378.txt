class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int end = nums.length;
        int start = 0;
        int currentsum = 0;
        int minumumLength = Integer.MAX_VALUE;
        while (left < end) {
            while (start < end && currentsum < target) {
                currentsum = currentsum + nums[start];
                start++;
            }
            if (currentsum >= target) {
                minumumLength = Math.min(minumumLength, start - left);
            }
            currentsum = currentsum - nums[left];
            left++;
        }
        if (minumumLength == Integer.MAX_VALUE)
            return 0;
        return minumumLength;
    }
}
