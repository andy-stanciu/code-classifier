class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length, window = 0, answer = n + 1, left = 0;
        for (int right = 0; right < n; right++) {
            window += nums[right];
            while (window >= target) {
                answer = Math.min(answer, right - left + 1);
                window -= nums[left++];
            }
        }
        return answer == n + 1 ? 0 : answer;
    }
}
