class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int winSum = 0, minSize = Integer.MAX_VALUE, left = 0, right = 0;
        while (right < nums.length) {
            winSum += nums[right];
            while (winSum >= target) {
                minSize = Math.min(minSize, (right - left + 1));
                winSum -= nums[left];
                left++;
            }
            right++;
        }
        if (minSize != Integer.MAX_VALUE) {
            return minSize;
        } else {
            return 0;
        }
    }
}
