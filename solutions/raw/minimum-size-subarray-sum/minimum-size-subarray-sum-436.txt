class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0, r = 0, n = nums.length, ans = n + 1;
        long tempSum = 0;
        for (; r < n; r++) {
            tempSum += nums[r];
            while (l <= r && tempSum >= target) {
                ans = Math.min(ans, r - l + 1);
                tempSum -= nums[l];
                l++;
            }
        }
        return ans == n + 1 ? 0 : ans;
    }
}
