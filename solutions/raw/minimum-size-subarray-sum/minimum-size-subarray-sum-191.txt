class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int sum = 0;
        int minSubArrayLength = Integer.MAX_VALUE;
        while (right < nums.length) {
            sum += nums[right];
            if (sum >= target) {
                minSubArrayLength = (right - left + 1) < minSubArrayLength ? (right - left + 1) : minSubArrayLength;
                do {
                    sum -= nums[left];
                    left++;
                    if (sum >= target)
                        minSubArrayLength = (right - left + 1) < minSubArrayLength ? (right - left + 1) : minSubArrayLength;
                } while (sum >= target);
            }
            right++;
        }
        return minSubArrayLength == Integer.MAX_VALUE ? 0 : minSubArrayLength;
    }
}
