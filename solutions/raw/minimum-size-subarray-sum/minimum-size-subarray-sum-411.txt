class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int s = 0;
        int sum = 0;
        int m = Integer.MAX_VALUE;
        int e = 0;
        int n = nums.length;
        while (e < n) {
            sum += nums[e++];
            while (sum >= target && s < e) {
                m = Math.min(m, e - s);
                sum -= nums[s++];
            }
        }
        return (m == Integer.MAX_VALUE) ? 0 : m;
    }
}
