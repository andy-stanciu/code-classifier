class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0, sum = 0, count = Integer.MAX_VALUE;
        for (int end = 0; end < nums.length; end++) {
            sum += nums[end];
            while (sum >= target) {
                count = Math.min(count, end - start + 1);
                sum -= nums[start++];
            }
        }
        return count == Integer.MAX_VALUE ? 0 : count;
    }
}
