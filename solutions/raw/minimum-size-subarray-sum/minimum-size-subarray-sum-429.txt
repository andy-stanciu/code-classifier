class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int sum = 0;
        int min = Integer.MAX_VALUE;
        while (left < nums.length) {
            while (right < nums.length && sum < target) {
                sum = sum + nums[right];
                right++;
            }
            if (sum >= target) {
                min = Math.min(min, right - left);
            }
            sum = sum - nums[left];
            left++;
        }
        return min != Integer.MAX_VALUE ? min : 0;
    }
}
