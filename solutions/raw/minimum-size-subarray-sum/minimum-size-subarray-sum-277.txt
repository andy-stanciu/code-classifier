class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int front = 0, end = 0;
        int min = Integer.MAX_VALUE;
        int sum = 0;
        while (end < nums.length) {
            sum += nums[end++];
            while (front < end && sum >= target) {
                min = Math.min(end - front, min);
                sum -= nums[front++];
            }
        }
        if (min == Integer.MAX_VALUE) {
            min = 0;
        }
        return min;
    }
}
