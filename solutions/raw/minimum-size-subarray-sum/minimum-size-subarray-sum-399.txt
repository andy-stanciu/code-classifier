class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int L = 0;
        int R = 0;
        int sum = 0;
        int min = Integer.MAX_VALUE;
        while (R < nums.length && L <= R) {
            while (R < nums.length && L <= R && sum < target) {
                sum += nums[R];
                R++;
            }
            while (R <= nums.length && L < R && sum >= target) {
                min = Math.min(min, R - L);
                sum -= nums[L];
                L++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
