class Solution {

    public int minSubArrayLen(int k, int[] arr) {
        int sum = 0;
        int start = 0;
        int len = 0;
        int minlenght = Integer.MAX_VALUE;
        for (int i = 0; i < arr.length; i++) {
            sum = sum + arr[i];
            while (sum >= k) {
                len = i - start + 1;
                minlenght = Math.min(minlenght, len);
                sum = sum - arr[start];
                start++;
            }
        }
        if (minlenght == Integer.MAX_VALUE) {
            return 0;
        }
        return minlenght;
    }
}
