class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0;
        int window = 0;
        int min = Integer.MAX_VALUE;
        for (int end = 0; end < nums.length; end++) {
            window += nums[end];
            while (window >= target) {
                min = Math.min(min, end - start + 1);
                window -= nums[start++];
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
