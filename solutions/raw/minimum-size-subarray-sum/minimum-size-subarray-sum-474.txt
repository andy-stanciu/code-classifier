class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int sum = 0;
        while (j < nums.length) {
            sum += nums[j];
            if (sum < target)
                j++;
            else if (sum == target) {
                min = Math.min(j - i + 1, min);
                j++;
            } else if (sum > target) {
                while (sum > target) {
                    min = Math.min(j - i + 1, min);
                    sum -= nums[i];
                    i++;
                }
                if (sum == target) {
                    min = Math.min(j - i + 1, min);
                }
                j++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
