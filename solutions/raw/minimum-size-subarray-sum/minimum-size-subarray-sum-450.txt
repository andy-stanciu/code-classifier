class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0, j = i, mini = Integer.MAX_VALUE, sum = 0;
        while (j < nums.length) {
            sum += nums[j];
            while (sum >= target) {
                mini = Math.min(mini, j - i + 1);
                sum -= nums[i++];
            }
            j++;
        }
        return mini == Integer.MAX_VALUE ? 0 : mini;
    }
}
