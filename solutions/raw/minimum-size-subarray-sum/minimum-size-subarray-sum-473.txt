class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int minimalLength = Integer.MAX_VALUE;
        int currentSum = 0;
        while (j < nums.length) {
            if (currentSum >= target) {
                minimalLength = Math.min(minimalLength, j - i);
                currentSum -= nums[i++];
            } else {
                currentSum += nums[j++];
            }
        }
        while (i < nums.length) {
            if (currentSum >= target) {
                minimalLength = Math.min(minimalLength, j - i);
                currentSum -= nums[i];
            }
            i++;
        }
        return minimalLength == Integer.MAX_VALUE ? 0 : minimalLength;
    }
}
