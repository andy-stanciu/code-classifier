class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int s = 0, e = -1, currentSum = 0;
        int length = nums.length;
        int minWindow = length + 1;
        while (e < length) {
            if (currentSum < target) {
                e++;
                if (e >= length)
                    break;
                currentSum += nums[e];
            } else {
                int winSize = e - s + 1;
                minWindow = winSize < minWindow ? winSize : minWindow;
                //                System.out.println("Min window: " + minWindow + ", s=" + s + ", e=" + e+" , curr sum:"+currentSum);
                currentSum -= nums[s];
                //                System.out.println("New curr sum:"+ currentSum);
                s++;
            }
        }
        return minWindow == length + 1 ? 0 : minWindow;
    }
}
