class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0;
        int end = 0;
        int minSize = nums.length + 1;
        int sum = nums[start];
        while (end < nums.length) {
            if (sum < target) {
                end = end + 1;
                if (end < nums.length) {
                    sum = sum + nums[end];
                }
            } else if (sum > target) {
                if (end == start)
                    minSize = Math.min(minSize, 1);
                else
                    minSize = Math.min(minSize, end - start + 1);
                sum = sum - nums[start];
                start = start + 1;
            } else {
                if (end == start)
                    minSize = Math.min(minSize, 1);
                else
                    minSize = Math.min(minSize, end - start + 1);
                end = end + 1;
                if (end < nums.length)
                    sum = sum + nums[end];
            }
        }
        if (minSize == nums.length + 1)
            return 0;
        return minSize;
    }
}
