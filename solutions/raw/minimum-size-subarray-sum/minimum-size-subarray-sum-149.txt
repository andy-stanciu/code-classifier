class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0, r = 0, n = nums.length;
        int minLen = 100001;
        int sum = 0;
        while (r < n) {
            sum += nums[r];
            while (sum >= target) {
                minLen = Math.min(minLen, r - l + 1);
                sum -= nums[l];
                l++;
            }
            r++;
        }
        return minLen == 100001 ? 0 : minLen;
    }
}
