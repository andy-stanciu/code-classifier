class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0, i = 0, j = 0, min = Integer.MAX_VALUE;
        while (i < nums.length) {
            sum += nums[i];
            while (sum >= target && j <= i) {
                min = Math.min(min, i - j + 1);
                sum -= nums[j];
                j++;
            }
            i++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
