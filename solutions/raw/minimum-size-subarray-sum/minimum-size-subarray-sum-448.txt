class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int subarraySum = 0;
        int lenSubarray = Integer.MAX_VALUE;
        for (int j = 0, i = 0; j < nums.length; ++j) {
            subarraySum += nums[j];
            while (subarraySum >= target) {
                subarraySum -= nums[i];
                lenSubarray = Math.min(lenSubarray, (j - i + 1));
                ++i;
            }
        }
        return lenSubarray == Integer.MAX_VALUE ? 0 : lenSubarray;
    }
}
