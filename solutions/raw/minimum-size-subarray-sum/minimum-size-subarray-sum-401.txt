class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0, end = 0, ans = Integer.MAX_VALUE;
        long sum = 0;
        while (end < nums.length) {
            sum += nums[end];
            if (sum >= target) {
                while (sum >= target) {
                    sum -= nums[start];
                    start++;
                }
                start--;
                sum += nums[start];
                ans = Math.min(ans, end - start + 1);
            }
            end++;
        }
        if (ans == Integer.MAX_VALUE)
            return 0;
        else
            return ans;
    }
}
