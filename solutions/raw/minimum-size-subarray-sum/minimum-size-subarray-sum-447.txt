class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int size = nums.length;
        int l = 0;
        int r = 0;
        int sum = 0;
        int minLength = Integer.MAX_VALUE;
        boolean minFound = false;
        while (l < size) {
            // move right pointer
            while (r < size && sum < target) {
                sum += nums[r++];
            }
            // update minimum
            if (sum >= target) {
                minLength = Math.min(minLength, r - l);
                minFound = true;
            }
            // move left
            sum -= nums[l++];
        }
        return minFound ? minLength : 0;
    }
}
