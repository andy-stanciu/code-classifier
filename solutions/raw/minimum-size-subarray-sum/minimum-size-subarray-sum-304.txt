class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int sum = 0;
        int lenSubArray = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            sum = sum + nums[i];
            while (sum >= target) {
                lenSubArray = Math.min(lenSubArray, i - left + 1);
                sum -= nums[left++];
            }
        }
        return lenSubArray == Integer.MAX_VALUE ? 0 : lenSubArray;
    }
}
