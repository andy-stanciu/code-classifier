class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = nums.length, sum = 0, k = 0, nb = 0;
        boolean reached = false;
        for (int i = 0; i < nums.length; i++) {
            if (sum + nums[i] < target) {
                sum += nums[i];
                nb++;
            } else {
                nb++;
                min = Math.min(min, nb);
                sum += nums[i];
                reached = true;
                while (sum >= target) {
                    sum -= nums[k];
                    k++;
                    nb--;
                }
                min = Math.min(min, nb + 1);
            }
        }
        if (reached == false)
            return 0;
        return min;
    }
}
