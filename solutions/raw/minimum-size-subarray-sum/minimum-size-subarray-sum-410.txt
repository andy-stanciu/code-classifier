class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int sum = 0;
        while (j < nums.length && i < nums.length) {
            if (nums[j] == target) {
                return 1;
            }
            sum = sum + nums[j];
            if (sum < target) {
                j++;
            }
            if (sum >= target) {
                while (sum >= target) {
                    sum = sum - nums[i];
                    ans = Math.min(ans, (j - i) + 1);
                    i++;
                }
                j++;
            }
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
