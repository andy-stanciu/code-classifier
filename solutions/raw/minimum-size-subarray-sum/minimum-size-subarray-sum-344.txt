class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int minLen = Integer.MAX_VALUE;
        int sum = 0;
        while (j < nums.length && i <= j) {
            sum += nums[j];
            if (sum >= target) {
                minLen = Math.min(minLen, j - i + 1);
                while (sum >= target) {
                    sum -= nums[i];
                    i++;
                    if (sum >= target) {
                        minLen = Math.min(minLen, j - i + 1);
                    }
                }
            }
            j++;
        }
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
