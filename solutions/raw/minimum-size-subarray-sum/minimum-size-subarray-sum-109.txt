class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0;
        int minlength = nums.length + 1;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            while (sum >= target) {
                if (i - start + 1 < minlength) {
                    minlength = i - start + 1;
                    System.out.print(minlength);
                }
                sum = sum - nums[start];
                start++;
            }
        }
        if (minlength > nums.length)
            return 0;
        return minlength;
    }
}
