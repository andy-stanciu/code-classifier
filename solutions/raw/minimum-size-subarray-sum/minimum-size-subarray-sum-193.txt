class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE, i = 0, j = 0, sum = nums[0];
        while (j < nums.length) {
            if (sum < target) {
                j++;
                if (j < nums.length)
                    sum += nums[j];
            } else {
                min = Math.min(j - i + 1, min);
                sum -= nums[i];
                i++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
