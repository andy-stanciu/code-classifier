class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int len = nums.length;
        int min_l = Integer.MAX_VALUE;
        int i = 0, j = 0;
        int sum = 0;
        while (j < len) {
            sum += nums[j];
            while (sum >= target) {
                min_l = Math.min(min_l, j - i + 1);
                sum -= nums[i];
                i++;
            }
            j++;
        }
        return min_l == Integer.MAX_VALUE ? 0 : min_l;
    }
}
