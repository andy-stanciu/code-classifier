class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int result = Integer.MAX_VALUE;
        int arraySum = 0;
        int left = 0;
        int right = 0;
        while (left < nums.length) {
            while (right < nums.length && arraySum < target) {
                arraySum = arraySum + nums[right];
                right++;
            }
            if (arraySum >= target) {
                result = Math.min(result, right - left);
            }
            arraySum = arraySum - nums[left];
            left++;
            if (result == Integer.MAX_VALUE) {
                return 0;
            }
        }
        return result;
    }
}
