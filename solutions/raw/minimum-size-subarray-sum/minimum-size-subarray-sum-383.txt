class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int j = -1;
        int sum = 0;
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            while (sum >= target) {
                j++;
                res = Math.min(res, i - j + 1);
                sum -= nums[j];
            }
        }
        return (res != Integer.MAX_VALUE) ? res : 0;
    }
}
