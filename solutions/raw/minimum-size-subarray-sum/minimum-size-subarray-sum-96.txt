class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int currSum = 0;
        int ind = Integer.MAX_VALUE;
        int left = 0;
        for (int right = 0; right < nums.length; right++) {
            currSum += nums[right];
            while (currSum >= target) {
                ind = Math.min(ind, right - left + 1);
                currSum -= nums[left];
                left++;
            }
        }
        return ind == Integer.MAX_VALUE ? 0 : ind;
    }
}
