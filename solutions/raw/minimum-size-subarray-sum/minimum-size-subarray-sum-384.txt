class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int fr = 0;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            while (sum >= target) {
                min = Math.min(min, i - fr + 1);
                sum -= nums[fr];
                fr++;
            }
        }
        if (min != Integer.MAX_VALUE) {
            return min;
        }
        return 0;
    }
}
