class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int minsub = Integer.MAX_VALUE;
        ;
        int sum = 0;
        while (r < nums.length) {
            sum += nums[r];
            if (sum >= target) {
                while (sum >= target) {
                    minsub = Math.min(minsub, r - l + 1);
                    sum -= nums[l];
                    l++;
                }
            }
            r++;
        }
        return minsub == Integer.MAX_VALUE ? 0 : minsub;
    }
}
