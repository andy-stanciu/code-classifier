class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int right = 0;
        int sum = 0;
        int minSubArrLen = Integer.MAX_VALUE;
        while (right < n) {
            sum += nums[right];
            while (sum >= target) {
                minSubArrLen = Math.min(minSubArrLen, right - left + 1);
                sum -= nums[left];
                left++;
            }
            right++;
        }
        return minSubArrLen != Integer.MAX_VALUE ? minSubArrLen : 0;
    }
}
