class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0, end = 0, runningSum = 0, lenSubArray = Integer.MAX_VALUE;
        while (start < nums.length && end < nums.length) {
            runningSum += nums[end];
            if (runningSum >= target) {
                while (start <= end) {
                    int diff = runningSum - nums[start];
                    if (diff >= target) {
                        runningSum = diff;
                        start++;
                    } else {
                        break;
                    }
                }
                lenSubArray = Math.min(lenSubArray, (end - start + 1));
                if (lenSubArray == 1) {
                    return lenSubArray;
                }
            }
            end++;
        }
        return lenSubArray == Integer.MAX_VALUE ? 0 : lenSubArray;
    }
}
