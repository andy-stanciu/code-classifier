class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int len = nums.length;
        int left = 0;
        int right = 0;
        int result = 0;
        int currSum = 0;
        while (right < len) {
            currSum += nums[right];
            while (currSum >= target) {
                int SubArrLen = right - left + 1;
                if (result == 0)
                    result = SubArrLen;
                else
                    result = result < SubArrLen ? result : SubArrLen;
                currSum -= nums[left];
                left++;
            }
            right++;
        }
        return result;
    }
}
