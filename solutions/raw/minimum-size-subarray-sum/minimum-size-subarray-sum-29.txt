class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int sum = 0, minLen = Integer.MAX_VALUE, l = 0, r = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            while (sum >= target) {
                minLen = Math.min(minLen, r - l + 1);
                sum -= nums[l];
                l++;
            }
            r++;
        }
        return (minLen == Integer.MAX_VALUE) ? 0 : minLen;
    }
}
