class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0, l = 0, r = 0;
        int minLen = Integer.MAX_VALUE;
        for (r = 0; r < nums.length; r++) {
            sum = sum + nums[r];
            while (sum >= target) {
                minLen = Math.min(minLen, r - l + 1);
                sum = sum - nums[l];
                l = l + 1;
            }
        }
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
