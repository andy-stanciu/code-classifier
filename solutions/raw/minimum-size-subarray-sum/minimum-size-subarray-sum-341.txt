class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int n = nums.length;
        int minimumValue = Integer.MAX_VALUE;
        int cursum = 0;
        while (r < n) {
            cursum += nums[r];
            while (cursum >= target) {
                minimumValue = Math.min(minimumValue, r - l + 1);
                cursum -= nums[l];
                l++;
            }
            r++;
        }
        if (minimumValue == Integer.MAX_VALUE) {
            minimumValue = 0;
        }
        return minimumValue;
    }
}
