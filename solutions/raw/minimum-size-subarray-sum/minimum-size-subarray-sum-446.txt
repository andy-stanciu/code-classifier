class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0, j = 0, n = nums.length, sum = 0, minL = n;
        // Claculating the sum if the total sum is less than target then it cant be calculated and we will return 0.
        for (i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        if (sum < target) {
            return 0;
        }
        sum = 0;
        i = 0;
        // Calculating the sum and increament j until sum > target
        while (j < nums.length) {
            sum += nums[j];
            // increament the i and subtract the value of i from it and calculate minL
            while (sum >= target) {
                minL = Math.min(minL, j - i + 1);
                sum -= nums[i];
                i++;
            }
            j++;
        }
        return minL;
    }
}
