class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0;
        int left = 0;
        int right = 0;
        int min = Integer.MAX_VALUE;
        int n = nums.length;
        while (left < n) {
            while (right < n && sum < target) {
                sum += nums[right];
                right++;
            }
            if (sum >= target) {
                min = Math.min(min, right - left);
            }
            sum -= nums[left];
            left++;
        }
        if (min == Integer.MAX_VALUE) {
            return 0;
        }
        return min;
    }
}
