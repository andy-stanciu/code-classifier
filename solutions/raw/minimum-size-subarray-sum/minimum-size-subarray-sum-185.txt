class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int first = 0;
        int second = 0;
        long sum = 0;
        int ans = Integer.MAX_VALUE;
        while (first <= second && second < nums.length) {
            sum += nums[second];
            if (sum >= target) {
                ans = Math.min(ans, second - first + 1);
                sum -= nums[first];
                sum -= nums[second];
                first++;
                continue;
            }
            second++;
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
