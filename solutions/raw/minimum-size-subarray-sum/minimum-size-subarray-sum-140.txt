class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        // it can be done by using prefix sum
        int left = 0;
        int min = Integer.MAX_VALUE;
        int sum = 0;
        int n = nums.length;
        for (int right = 0; right < n; right++) {
            sum += nums[right];
            while (sum >= target) {
                min = Math.min(min, right - left + 1);
                // removing the element from left
                sum -= nums[left];
                // increamenting the element to the next position
                left++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
