class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int n = nums.length;
        int ans = Integer.MAX_VALUE;
        int window = 0;
        for (int right = 0; right < n; right++) {
            window += nums[right];
            while (window >= target) {
                ans = Math.min(ans, right - left + 1);
                window -= nums[left];
                left++;
            }
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
