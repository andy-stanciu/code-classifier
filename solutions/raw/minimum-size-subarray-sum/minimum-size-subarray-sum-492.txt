class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int end = 0;
        int sum = 0;
        int min = Integer.MAX_VALUE;
        while (end < n) {
            sum += nums[end];
            //Finding shortest subarray
            while (sum >= target) {
                min = Math.min(min, end - left + 1);
                sum -= nums[left];
                left++;
            }
            end++;
        }
        //If there is no subarray
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
