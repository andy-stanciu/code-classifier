class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int minLength = Integer.MAX_VALUE;
        int i = 0, j = 0;
        int currSum = nums[j];
        while (j < nums.length) {
            if (currSum >= target) {
                minLength = Math.min(minLength, j - i + 1);
                currSum -= nums[i];
                i++;
            } else {
                j++;
                if (j < nums.length) {
                    currSum += nums[j];
                }
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
