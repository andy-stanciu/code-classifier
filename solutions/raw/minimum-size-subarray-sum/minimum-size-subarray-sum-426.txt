class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int n = nums.length;
        int len = Integer.MAX_VALUE, sum = 0;
        while (i < n) {
            sum += nums[i];
            while (sum >= target) {
                len = Math.min(len, i - j + 1);
                sum -= nums[j++];
            }
            i++;
        }
        if (len == Integer.MAX_VALUE) {
            return 0;
        }
        return len;
    }
}
