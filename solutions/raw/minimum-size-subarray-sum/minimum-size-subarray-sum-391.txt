class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int window = Integer.MAX_VALUE;
        int cursum = 0;
        int start = 0;
        for (int i = 0; i < nums.length; i++) {
            cursum += nums[i];
            while (cursum >= target) {
                window = Math.min(window, i - start + 1);
                cursum -= nums[start];
                start++;
            }
        }
        return window == Integer.MAX_VALUE ? 0 : window;
    }
}
