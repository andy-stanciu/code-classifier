class Solution {

    private final static int NO_SUBARRAY = Integer.MAX_VALUE;

    public int minSubArrayLen(int target, int[] nums) {
        final int len = nums.length;
        int s = 0, f = 0;
        int sum = 0, minLen = NO_SUBARRAY;
        while (f < len) {
            while (f < len && sum < target) {
                sum += nums[f++];
            }
            if (sum < target)
                return minLen == NO_SUBARRAY ? 0 : minLen;
            while (sum >= target) {
                sum -= nums[s++];
            }
            minLen = Math.min(minLen, f - s + 1);
        }
        return minLen == NO_SUBARRAY ? 0 : minLen;
    }
}
