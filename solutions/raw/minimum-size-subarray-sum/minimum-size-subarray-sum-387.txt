class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        // to
        int i = 0;
        // from
        int j = 0;
        int ret = Integer.MAX_VALUE;
        int sum = nums[0];
        while (true) {
            if (sum >= target) {
                ret = Math.min(ret, i - j + 1);
                if (ret == 1)
                    return 1;
                sum -= nums[j];
                j++;
                // System.out.println("j++ i " + i + " j " + j + " sum " + sum + " ret " + ret);
            } else {
                i++;
                if (i == nums.length)
                    break;
                sum += nums[i];
                // System.out.println("i++ i " + i + " j " + j + " sum " + sum + " ret " + ret);
            }
        }
        return ret == Integer.MAX_VALUE ? 0 : ret;
    }
}
