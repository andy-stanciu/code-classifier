class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int pfsum = 0;
        int minL = 0;
        int start = 0;
        int end = 0;
        while (end < nums.length) {
            pfsum += nums[end];
            if (pfsum < target)
                end++;
            else {
                pfsum = pfsum - nums[start] - nums[end];
                minL = minL == 0 ? end - start + 1 : Math.min(minL, end - start + 1);
                start++;
            }
        }
        return minL;
    }
}
