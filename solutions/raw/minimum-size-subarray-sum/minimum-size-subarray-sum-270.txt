class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int res = Integer.MAX_VALUE;
        int currSum = 0;
        while (j < nums.length) {
            currSum += nums[j];
            while (currSum >= target) {
                res = Math.min(res, j - i + 1);
                currSum -= nums[i];
                i += 1;
            }
            j += 1;
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}
