class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int result = Integer.MAX_VALUE;
        int sum = nums[0];
        while (true) {
            if (sum >= target) {
                result = Math.min(result, right - left + 1);
                left++;
                sum -= nums[left - 1];
            } else if (left >= right || sum < target) {
                right++;
                if (right < nums.length) {
                    sum += nums[right];
                } else {
                    break;
                }
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
