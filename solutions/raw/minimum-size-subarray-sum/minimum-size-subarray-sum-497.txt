class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int currsum = 0;
        int i = 0;
        int j = 0;
        int min = Integer.MAX_VALUE;
        while (j < nums.length) {
            currsum += nums[j];
            while (currsum >= target) {
                min = Math.min(min, j - i + 1);
                currsum -= nums[i];
                i++;
            }
            j++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
