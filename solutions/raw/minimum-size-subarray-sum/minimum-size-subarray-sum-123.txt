class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int s = 0, n = nums.length, sum = 0, result = Integer.MAX_VALUE;
        for (int e = 0; e < n; e++) {
            sum += nums[e];
            while (sum >= target && s <= e) {
                result = Math.min(result, e - s + 1);
                sum -= nums[s++];
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
