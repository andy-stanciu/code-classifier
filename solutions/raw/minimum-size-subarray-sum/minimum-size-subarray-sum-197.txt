class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int min_len = nums.length + 1;
        int cur_sum = 0;
        while (j < nums.length) {
            cur_sum += nums[j];
            j++;
            while (cur_sum >= target) {
                min_len = Math.min(min_len, (j - i));
                cur_sum -= nums[i];
                i++;
            }
        }
        return ((min_len == nums.length + 1) ? 0 : min_len);
    }
}
