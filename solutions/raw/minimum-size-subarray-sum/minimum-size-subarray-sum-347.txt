class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length, win = 0, ans = n + 1, l = 0;
        for (int r = 0; r < n; r++) {
            win += nums[r];
            while (win >= target) {
                ans = Math.min(ans, r - l + 1);
                win -= nums[l++];
            }
        }
        return ans == n + 1 ? 0 : ans;
    }
}
