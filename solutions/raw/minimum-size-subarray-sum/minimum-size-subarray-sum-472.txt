class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0;
        int sum = 0;
        int min = 100001;
        for (int r = 0; r < nums.length; r++) {
            sum += nums[r];
            while (sum >= target) {
                min = Math.min((r - l) + 1, min);
                sum -= nums[l];
                l++;
            }
        }
        if (l == 0) {
            return 0;
        } else {
            return min;
        }
    }
}
