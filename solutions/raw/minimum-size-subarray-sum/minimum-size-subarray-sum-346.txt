class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0;
        int min = Integer.MAX_VALUE;
        for (int i = 0, right = 1; i < nums.length; right++) {
            if (right >= nums.length + 1) {
                break;
            }
            sum += nums[right - 1];
            while (sum >= target) {
                min = Math.min(min, right - i);
                if (min == 1) {
                    return 1;
                }
                sum -= nums[i];
                i++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
