class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int sum = 0;
        int minLength = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            while (sum >= target) {
                if (i - left + 1 < minLength || minLength == 0) {
                    minLength = i - left + 1;
                }
                sum -= nums[left++];
            }
        }
        return minLength;
    }
}
