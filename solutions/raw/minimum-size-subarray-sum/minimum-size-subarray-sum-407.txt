class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int shortestLength = (int) Math.pow(10, 6), left = 0, right = 0, value = target;
        while (right < nums.length && left <= right) {
            target -= nums[right];
            while (target <= 0) {
                shortestLength = Math.min(shortestLength, right - left + 1);
                target += nums[left];
                left++;
            }
            right++;
        }
        return shortestLength == (int) Math.pow(10, 6) ? 0 : shortestLength;
    }
}
