class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int currsum = 0;
        int window = Integer.MAX_VALUE;
        int st = 0;
        for (int i = 0; i < nums.length; i++) {
            currsum += nums[i];
            while (currsum >= target) {
                window = Math.min(window, i - st + 1);
                currsum -= nums[st];
                st++;
            }
        }
        return window == Integer.MAX_VALUE ? 0 : window;
    }
}
