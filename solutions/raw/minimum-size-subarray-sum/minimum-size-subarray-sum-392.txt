class Solution {

    public int minSubArrayLen(int tg, int[] nums) {
        int sum = 0, n = nums.length, cnt = Integer.MAX_VALUE, j = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            while (sum >= tg) {
                cnt = Math.min(cnt, i - j + 1);
                sum -= nums[j++];
            }
        }
        return cnt == Integer.MAX_VALUE ? 0 : cnt;
    }
}
