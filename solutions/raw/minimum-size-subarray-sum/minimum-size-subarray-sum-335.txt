class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int[] psum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            psum[i + 1] = psum[i] + nums[i];
        }
        int ans = n + 1;
        int l = 0, r = 0;
        for (; r <= n; r++) {
            System.out.println("--" + r);
            while (r >= l) {
                if (psum[r] - psum[l] >= target) {
                    System.out.println("--" + l + " -" + r);
                    ans = Math.min(ans, r - l);
                    l++;
                } else {
                    break;
                }
            }
        }
        return ans == (n + 1) ? 0 : ans;
    }
}
