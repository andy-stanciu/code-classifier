class Solution {

    public int minSubArrayLen(int k, int[] arr) {
        int l = 0;
        int r;
        int minc = Integer.MAX_VALUE;
        int sum = 0;
        int n = arr.length;
        for (r = 0; r < n; r++) {
            sum += arr[r];
            while (sum >= k) {
                minc = Math.min(minc, r - l + 1);
                sum -= arr[l];
                l++;
            }
        }
        return minc == Integer.MAX_VALUE ? 0 : minc;
    }
}
