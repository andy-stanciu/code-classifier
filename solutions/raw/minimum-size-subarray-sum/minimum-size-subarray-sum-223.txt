class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int i = 0, j = 0;
        int min = Integer.MAX_VALUE, sum = 0;
        while (i <= j && j < n) {
            if (sum < target) {
                sum += nums[j];
            } else {
                min = Math.min(min, j - i + 1);
                sum -= nums[i++];
            }
            if (sum < target)
                j++;
        }
        if (min == Integer.MAX_VALUE)
            return 0;
        return min;
    }
}
