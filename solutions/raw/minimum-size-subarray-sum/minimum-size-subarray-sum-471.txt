class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int sum = 0;
        int start = 0;
        for (int i = 0; i < nums.length; ++i) {
            sum += nums[i];
            if (sum == target) {
                if (min > i - start) {
                    min = i - start + 1;
                    if (min == 1)
                        return min;
                }
            } else if (sum > target) {
                if (min > i - start) {
                    min = i - start + 1;
                    if (min == 1)
                        return min;
                }
                sum -= nums[start];
                sum -= nums[i];
                start++;
                i--;
            }
        }
        if (min == Integer.MAX_VALUE) {
            min = 0;
        }
        return min;
    }
}
