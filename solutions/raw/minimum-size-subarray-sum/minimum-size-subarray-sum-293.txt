class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int sol = Integer.MAX_VALUE;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            while (sum >= target) {
                sol = Math.min(sol, i + 1 - left);
                sum -= nums[left++];
            }
        }
        return sol == Integer.MAX_VALUE ? 0 : sol;
    }
}
