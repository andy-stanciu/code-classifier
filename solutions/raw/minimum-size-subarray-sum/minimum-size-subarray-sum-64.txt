class Solution {

    public int minSubArrayLen(int target, int[] arr) {
        int n = arr.length;
        int low = 0;
        int high = 0;
        int sum = 0;
        int result = Integer.MAX_VALUE;
        for (high = 0; high < n; high++) {
            sum += arr[high];
            while (sum >= target) {
                sum -= arr[low];
                result = Math.min(result, high - low + 1);
                low++;
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
