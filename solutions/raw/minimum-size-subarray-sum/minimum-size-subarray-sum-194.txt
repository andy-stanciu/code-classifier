class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0, end = 0, sum = 0;
        int count = Integer.MAX_VALUE;
        while (end < nums.length) {
            sum += nums[end];
            while (sum >= target) {
                count = Math.min(count, end - start + 1);
                sum -= nums[start++];
            }
            end++;
        }
        return (count == Integer.MAX_VALUE) ? 0 : count;
    }
}
