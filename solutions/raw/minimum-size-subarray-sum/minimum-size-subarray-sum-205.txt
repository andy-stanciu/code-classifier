class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int total = 0;
        int minLength = nums.length;
        int left = 0;
        int sum = 0;
        for (int right = 0; right < nums.length; right++) {
            total += nums[right];
            sum += nums[right];
            while (sum >= target) {
                minLength = Math.min(minLength, right - left + 1);
                sum -= nums[left];
                left++;
            }
        }
        if (total < target)
            return 0;
        return minLength;
    }
}
