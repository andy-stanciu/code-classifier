class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0;
        int sum = 0;
        int count = 0;
        int result = Integer.MAX_VALUE;
        int n = nums.length;
        int j = 0;
        for (i = 0; i < n; ++i) {
            sum = sum + nums[i];
            ++count;
            while (sum >= target) {
                result = Math.min(result, count);
                sum = sum - nums[j];
                --count;
                ++j;
            }
        }
        if (Integer.MAX_VALUE == result)
            return 0;
        return result;
    }
}
