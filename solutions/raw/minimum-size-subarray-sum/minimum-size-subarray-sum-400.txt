class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int sum = 0;
        int j = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            sum = sum + nums[i];
            while (sum >= target) {
                min = Math.min(min, i - j + 1);
                sum -= nums[j];
                j++;
            }
        }
        if (min > n) {
            return 0;
        }
        return min;
    }
}
