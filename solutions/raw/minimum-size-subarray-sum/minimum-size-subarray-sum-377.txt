class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int len = nums.length, left = 0, right = 0, sum = 0, minval = Integer.MAX_VALUE;
        for (right = 0; right < len; right++) {
            sum += nums[right];
            while (sum >= target) {
                minval = Math.min(minval, right - left + 1);
                sum -= nums[left];
                left++;
            }
        }
        if (minval == Integer.MAX_VALUE)
            return 0;
        else
            return minval;
    }
}
