class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int sum = 0;
        int end = 0;
        int minLen = Integer.MAX_VALUE;
        for (int start = 0; start < n; start++) {
            sum += nums[start];
            while (sum >= target) {
                minLen = Math.min(minLen, start - end + 1);
                sum -= nums[end];
                end++;
            }
        }
        return (minLen != Integer.MAX_VALUE) ? minLen : 0;
    }
}
