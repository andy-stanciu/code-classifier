class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = nums.length;
        int minLength = Integer.MAX_VALUE, sum = 0;
        int j = 0;
        for (int i = 0; i < l; i++) {
            sum += nums[i];
            while (sum >= target) {
                minLength = Math.min(minLength, i - j + 1);
                if (j < l)
                    sum -= nums[j++];
            }
        }
        return (minLength == Integer.MAX_VALUE) ? 0 : minLength;
    }
}
