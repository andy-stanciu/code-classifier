class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0;
        int ans = Integer.MAX_VALUE;
        int st = 0;
        for (int ed = 0; ed < nums.length; ed++) {
            sum = sum + nums[ed];
            while (sum >= target) {
                ans = Math.min(ans, ed - st + 1);
                sum = sum - nums[st];
                st++;
            }
        }
        if (ans == Integer.MAX_VALUE)
            return 0;
        return ans;
    }
}
