// easy optimal aproch
class Solution {

    public int minSubArrayLen(int target, int[] arr) {
        int n = arr.length;
        int i = 0, sum = 0, len = 0, minlen = Integer.MAX_VALUE;
        for (int j = 0; j < n; j++) {
            sum += arr[j];
            while (sum >= target) {
                len = j - i + 1;
                minlen = Math.min(len, minlen);
                sum -= arr[i];
                i++;
            }
        }
        if (minlen == Integer.MAX_VALUE)
            return 0;
        return minlen;
    }
}
