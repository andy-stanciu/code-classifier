class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int minWindow = Integer.MAX_VALUE;
        int currSum = 0;
        int low = 0;
        int high = 0;
        while (high < nums.length) {
            currSum += nums[high];
            high++;
            while (currSum >= target) {
                int window = high - low;
                minWindow = Math.min(minWindow, window);
                currSum -= nums[low];
                low++;
            }
        }
        return minWindow == Integer.MAX_VALUE ? 0 : minWindow;
    }
}
