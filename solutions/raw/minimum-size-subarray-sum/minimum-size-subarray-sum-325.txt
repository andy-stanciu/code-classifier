class Solution {

    public int minSubArrayLen(int target, int[] a) {
        int l = 0, r = 0, len = a.length, sum = 0, ans = Integer.MAX_VALUE;
        while (r < len) {
            sum += a[r];
            // System.out.println(sum);
            while (sum >= target) {
                ans = Math.min(ans, r - l + 1);
                sum -= a[l];
                l++;
            }
            r++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
