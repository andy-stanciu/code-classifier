class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int temp_ans = 0;
        int sum = 0;
        int first = 0;
        int last = 0;
        while (last < nums.length) {
            sum += nums[last];
            temp_ans = last - first + 1;
            while (sum >= target) {
                ans = Math.min(ans, temp_ans);
                sum -= nums[first];
                first++;
                temp_ans = last - first + 1;
            }
            last++;
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
