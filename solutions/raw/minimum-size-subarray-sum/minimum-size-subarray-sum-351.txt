class Solution {

    public int minSubArrayLen(int target, int[] a) {
        int sum = 0;
        int min = 1000000;
        int si = 0;
        for (int i = 0; i < a.length; i++) {
            sum += a[i];
            while (sum >= target && si <= i) {
                min = Math.min(min, i - si);
                sum -= a[si];
                si++;
            }
        }
        if (min == 1000000)
            return 0;
        return 1 + min;
    }
}
