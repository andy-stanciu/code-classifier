class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int minlen = nums.length + 1;
        int curSum = 0;
        int l = 0;
        int r = 0;
        while (r < nums.length) {
            curSum += nums[r];
            while (curSum >= target) {
                minlen = Math.min(minlen, r - l + 1);
                curSum -= nums[l];
                l++;
            }
            r++;
        }
        return (minlen == nums.length + 1) ? 0 : minlen;
    }
}
