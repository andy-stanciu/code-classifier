class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int total = 0;
        int sum = 0;
        int count = 0, minCount = Integer.MAX_VALUE;
        int left = 0, right = 0;
        for (int n : nums) total += n;
        if (target > total)
            return 0;
        while (right < nums.length) {
            sum += nums[right++];
            count++;
            while (sum >= target) {
                minCount = Math.min(minCount, count);
                sum -= nums[left++];
                count--;
            }
        }
        return minCount;
    }
}
