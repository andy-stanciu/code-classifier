class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int sum = 0;
        int i = 0;
        int length = Integer.MAX_VALUE;
        for (int j = 0; j < nums.length; j++) {
            sum += nums[j];
            if (sum == target) {
                length = Math.min(length, j - i + 1);
            } else if (sum > target) {
                length = Math.min(length, j - i + 1);
                while (sum > target) {
                    sum -= nums[i];
                    i++;
                    if (sum >= target) {
                        length = Math.min(length, j - i + 1);
                    }
                }
            }
        }
        if (length == Integer.MAX_VALUE) {
            return 0;
        }
        return length;
    }
}
