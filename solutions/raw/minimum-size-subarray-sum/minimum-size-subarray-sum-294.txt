class Solution {

    public int minSubArrayLen(int k, int[] arr) {
        int n = arr.length;
        int i = 0;
        int j = 0;
        int sum = 0;
        int min = Integer.MAX_VALUE;
        while (j < n) {
            sum = sum + arr[j];
            if (sum < k) {
                j++;
            } else {
                min = Math.min(min, j - i + 1);
                while (sum >= k) {
                    min = Math.min(min, j - i + 1);
                    sum = sum - arr[i];
                    i++;
                }
                j++;
            }
        }
        if (min == Integer.MAX_VALUE) {
            return 0;
        }
        return min;
    }
}
