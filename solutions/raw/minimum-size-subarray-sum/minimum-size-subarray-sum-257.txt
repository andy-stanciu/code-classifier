class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        //left
        int l = 0;
        //right
        int r = 0;
        //current sum
        int sum = 0;
        int minSize = Integer.MAX_VALUE;
        while (r < nums.length) {
            sum += nums[r];
            //if the window is valid then
            while (sum >= target) {
                //store the minimum window size
                minSize = Math.min(r - l + 1, minSize);
                //remove the left most element
                sum -= nums[l];
                //increment left
                l++;
            }
            r++;
        }
        System.gc();
        return minSize == Integer.MAX_VALUE ? 0 : minSize;
    }
}
