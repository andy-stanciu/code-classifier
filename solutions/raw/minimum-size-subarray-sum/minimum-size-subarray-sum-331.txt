class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int i = 0, j = 0, res = Integer.MAX_VALUE, curSum = 0;
        while (j < nums.length) {
            curSum += nums[j];
            while (curSum >= target) {
                res = Math.min(res, j - i + 1);
                curSum -= nums[i];
                i++;
            }
            j++;
        }
        if (res == Integer.MAX_VALUE)
            return 0;
        return res;
    }
}
