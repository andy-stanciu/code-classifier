class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length, min = Integer.MAX_VALUE, sum = 0;
        int i = 0, j = 0;
        while (i < n) {
            while (j < n && sum < target) sum += nums[j++];
            if (sum >= target)
                min = Math.min(min, j - i);
            sum -= nums[i++];
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
