class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int n = nums.length;
        int min = Integer.MAX_VALUE;
        int sum = 0;
        int stlength = -1;
        while (r < n) {
            sum += nums[r];
            while (sum >= target) {
                if (min > r - l + 1) {
                    min = r - l + 1;
                    stlength = l;
                }
                sum -= nums[l];
                l++;
            }
            r++;
        }
        if (stlength == -1) {
            return 0;
        }
        return min;
    }
}
