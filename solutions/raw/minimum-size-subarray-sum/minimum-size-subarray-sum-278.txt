class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int n = nums.length;
        int start = 0;
        int end = 0;
        int sum = 0;
        int minLength = Integer.MAX_VALUE;
        for (end = 0; end < n; end++) {
            sum = sum + nums[end];
            while (sum >= target) {
                minLength = Math.min(minLength, (end - start) + 1);
                sum = sum - nums[start];
                start++;
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
