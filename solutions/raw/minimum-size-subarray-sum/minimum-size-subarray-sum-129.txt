class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int start = 0;
        int end = 0;
        int length = nums.length;
        int minSub = length + 1;
        int sum = 0;
        while (end < length) {
            sum += nums[end++];
            while (sum >= target) {
                minSub = (end - start < minSub) ? end - start : minSub;
                sum -= nums[start++];
            }
        }
        if (minSub > length)
            return 0;
        return minSub;
    }
}
