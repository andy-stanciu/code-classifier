class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int start = 0;
        int end = 0;
        int currentSum = 0;
        while (start < nums.length) {
            if (currentSum >= target) {
                min = Math.min(min, end - start);
                currentSum -= nums[start++];
            } else {
                if (end < nums.length) {
                    currentSum += nums[end++];
                } else {
                    break;
                }
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
