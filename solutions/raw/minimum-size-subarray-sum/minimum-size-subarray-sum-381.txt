class Solution {

    public int minSubArrayLen(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int low = 0;
        int high = 0;
        int n = nums.length;
        int sum = 0;
        while (high < n) {
            sum += nums[high];
            if (sum >= target) {
                while (sum >= target) {
                    sum -= nums[low];
                    low++;
                }
                ans = Math.min(ans, high - low + 2);
            }
            high++;
        }
        return ans > n ? 0 : ans;
    }
}
