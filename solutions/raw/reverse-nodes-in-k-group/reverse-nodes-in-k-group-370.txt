class Solution {

    public ListNode reverseKGroup(ListNode head, int k) {
        ListNode dummy = head;
        for (int i = 0; i < k; i++) {
            if (dummy == null)
                return head;
            dummy = dummy.next;
        }
        ListNode curr = head;
        ListNode prev = null;
        ListNode nxt = null;
        for (int i = 0; i < k; i++) {
            nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
        }
        head.next = reverseKGroup(curr, k);
        return prev;
    }
}
