class Solution {

    public void findCombinations(int n, int k, ArrayList<Integer> temp, int count, int index, List<List<Integer>> ans) {
        if (count == k) {
            ans.add(new ArrayList<>(temp));
            return;
        }
        if (index > n) {
            return;
        }
        temp.add(index);
        findCombinations(n, k, temp, count + 1, index + 1, ans);
        temp.remove(temp.size() - 1);
        findCombinations(n, k, temp, count, index + 1, ans);
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        findCombinations(n, k, new ArrayList<>(), 0, 1, ans);
        return ans;
    }
}
