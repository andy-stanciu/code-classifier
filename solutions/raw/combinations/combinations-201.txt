class Solution {

    public void solve(int n, int k, List<Integer> temp, List<List<Integer>> sol, int index) {
        if (temp.size() == k) {
            sol.add(new ArrayList<>(temp));
        }
        for (int i = index; i <= n; i++) {
            temp.add(i);
            solve(n, k, temp, sol, i + 1);
            temp.remove(temp.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> sol = new ArrayList<>();
        solve(n, k, new ArrayList<>(), sol, 1);
        return sol;
    }
}
