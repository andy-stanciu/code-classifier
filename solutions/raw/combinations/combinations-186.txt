class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        dfs(n, k, 0, list, res, 1);
        return res;
    }

    public List<List<Integer>> dfs(int n, int k, int index, List<Integer> list, List<List<Integer>> res, int start) {
        if (index == k) {
            res.add(new ArrayList<>(list));
            return res;
        }
        for (int i = start; i <= n; i++) {
            list.add(i);
            dfs(n, k, index + 1, list, res, i + 1);
            list.remove(list.size() - 1);
        }
        return res;
    }
}
