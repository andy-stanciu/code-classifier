class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> combinations = new ArrayList<>();
        List<Integer> curr = new ArrayList<>();
        backtrack(combinations, k, n, curr, 1);
        return combinations;
    }

    public void backtrack(List<List<Integer>> combinations, int k, int n, List<Integer> l, int steps) {
        if (l.size() == k) {
            List<Integer> res = new ArrayList<>(l);
            combinations.add(res);
            return;
        }
        for (int i = steps; i <= n; i++) {
            l.add(i);
            backtrack(combinations, k, n, l, i + 1);
            l.remove(l.size() - 1);
        }
    }
}
