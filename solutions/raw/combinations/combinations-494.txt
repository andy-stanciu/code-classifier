class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<Integer> comb = new ArrayList<>();
        if (n == 1) {
            comb.add(1);
            combinations.add(comb);
            return combinations;
        }
        int[] nums = new int[n];
        for (int i = 0; i < n; i++) {
            nums[i] = i + 1;
        }
        findCombinations(nums, 0, k, comb);
        return combinations;
    }

    List<List<Integer>> combinations = new ArrayList<>();

    void findCombinations(int[] nums, int ind, int k, List<Integer> combination) {
        if (combination.size() == k) {
            List<Integer> comb = new ArrayList<>();
            comb.addAll(combination);
            combinations.add(comb);
            return;
        }
        if (ind >= nums.length)
            return;
        combination.add(nums[ind]);
        findCombinations(nums, ind + 1, k, combination);
        combination.remove(combination.size() - 1);
        findCombinations(nums, ind + 1, k, combination);
    }
}
