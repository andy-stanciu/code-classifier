class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ansList = new ArrayList<>();
        helper(n, k, 1, new ArrayList<Integer>(), ansList);
        return ansList;
    }

    private static void helper(int n, int k, int index, ArrayList<Integer> ans, List<List<Integer>> ansList) {
        if (ans.size() == k) {
            ansList.add(new ArrayList<>(ans));
            return;
        }
        for (int i = index; i <= n; i++) {
            ans.add(i);
            helper(n, k, i + 1, ans, ansList);
            ans.remove((Integer) i);
        }
    }
}
