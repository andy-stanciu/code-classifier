class Solution {

    List<List<Integer>> ans;

    public List<List<Integer>> combine(int n, int k) {
        ans = new ArrayList<>();
        generateCombination(n, k, 1, new ArrayList<Integer>());
        return ans;
    }

    private void generateCombination(int n, int k, int current, ArrayList<Integer> temp) {
        if (current > n) {
            if (temp.size() == k) {
                ans.add(new ArrayList<Integer>(temp));
            }
            return;
        }
        // take
        temp.add(current);
        generateCombination(n, k, current + 1, temp);
        temp.remove(temp.size() - 1);
        // doNotTake
        generateCombination(n, k, current + 1, temp);
    }
}
