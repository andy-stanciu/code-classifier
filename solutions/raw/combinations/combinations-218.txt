class Solution {

    void function(int start, int n, ArrayList<Integer> curr, List<List<Integer>> subsets, int k) {
        if (curr.size() == k) {
            subsets.add(new ArrayList(curr));
        }
        for (int i = start; i <= n; i++) {
            curr.add(i);
            function(i + 1, n, curr, subsets, k);
            curr.remove(curr.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> subsets = new ArrayList<>();
        function(1, n, new ArrayList(), subsets, k);
        return subsets;
    }
}
