class Solution {

    void rec(int index, int true_count, int k, int n, boolean[] nums, List<List<Integer>> combinations) {
        if (true_count == k) {
            List<Integer> combi = new ArrayList<>();
            for (int i = 1; i <= n; i++) {
                if (nums[i]) {
                    combi.add(i);
                }
            }
            combinations.add(combi);
            return;
        }
        if (index <= n) {
            nums[index] = true;
            rec(index + 1, true_count + 1, k, n, nums, combinations);
            nums[index] = false;
            rec(index + 1, true_count, k, n, nums, combinations);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> combinations = new ArrayList<>();
        boolean[] nums = new boolean[n + 1];
        rec(1, 0, k, n, nums, combinations);
        return combinations;
    }
}
