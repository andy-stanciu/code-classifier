class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> list = new ArrayList<>();
        boolean[] visi = new boolean[n + 1];
        combi(1, list, new ArrayList<>(), n, k);
        return list;
    }

    public static void combi(int i, List<List<Integer>> list, List<Integer> temp, int n, int k) {
        if (k == 0) {
            list.add(new ArrayList<>(temp));
            return;
        }
        if (i > n) {
            return;
        }
        temp.add(i);
        combi(i + 1, list, temp, n, k - 1);
        temp.remove(temp.size() - 1);
        combi(i + 1, list, temp, n, k);
    }
}
