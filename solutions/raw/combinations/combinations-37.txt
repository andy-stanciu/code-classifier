class Solution {

    List<List<Integer>> ans = new ArrayList();

    public void solve(int num, int k, int n, List<Integer> temp) {
        if (num > n) {
            if (temp.size() == k)
                ans.add(new ArrayList<>(temp));
            return;
        }
        // include num
        temp.add(num);
        solve(num + 1, k, n, temp);
        //  remove the number from temp to discard it
        temp.remove(temp.indexOf(num));
        solve(num + 1, k, n, temp);
        return;
    }

    public List<List<Integer>> combine(int n, int k) {
        solve(1, k, n, new ArrayList());
        return ans;
    }
}
