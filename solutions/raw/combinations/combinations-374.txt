class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> currentCombination = new ArrayList<>();
        // Initialize the first combination with k elements from 1 to k
        for (int i = 1; i <= k; i++) {
            currentCombination.add(i);
        }
        while (true) {
            result.add(new ArrayList<>(currentCombination));
            int i = k - 1;
            while (i >= 0 && currentCombination.get(i) == n - k + i + 1) {
                i--;
            }
            if (i < 0) {
                break;
            }
            currentCombination.set(i, currentCombination.get(i) + 1);
            for (int j = i + 1; j < k; j++) {
                currentCombination.set(j, currentCombination.get(j - 1) + 1);
            }
        }
        return result;
    }
}
