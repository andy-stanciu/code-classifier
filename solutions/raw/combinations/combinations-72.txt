class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> combinations = new ArrayList<>();
        findCombinations(1, n, k, combinations, new ArrayList<>());
        return combinations;
    }

    private void findCombinations(int index, int n, int k, List<List<Integer>> combinations, List<Integer> list) {
        if (list.size() == k) {
            combinations.add(new ArrayList<>(list));
            return;
        }
        for (int i = index; i <= n; i++) {
            list.add(i);
            findCombinations(i + 1, n, k, combinations, list);
            list.remove(list.size() - 1);
        }
    }
}
