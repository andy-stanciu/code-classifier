class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        combines(res, n, k, 1, new ArrayList<>(), 0);
        return res;
    }

    public void combines(List<List<Integer>> res, int n, int k, int x, List<Integer> ans, int len) {
        if (k == len) {
            res.add(new ArrayList<>(ans));
            return;
        }
        for (int i = x; i <= n; i++) {
            ans.add(i);
            combines(res, n, k, i + 1, ans, len + 1);
            ans.remove(ans.size() - 1);
        }
    }
}
