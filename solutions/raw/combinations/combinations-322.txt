class Solution {

    List<List<Integer>> ans;

    public List<List<Integer>> combine(int n, int k) {
        ans = new ArrayList<>();
        solve(n, k, 1, new ArrayList<Integer>());
        return ans;
    }

    public void solve(int n, int k, int start, List<Integer> li) {
        if (li.size() == k) {
            ans.add(new ArrayList<Integer>(li));
            return;
        }
        for (int i = start; i <= n; i++) {
            li.add(i);
            solve(n, k, i + 1, li);
            li.remove(li.size() - 1);
        }
    }
}
