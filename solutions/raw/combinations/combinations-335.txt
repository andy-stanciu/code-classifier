class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> combinations = new ArrayList();
        generateCombinations(1, n, k, new ArrayList(), combinations);
        return combinations;
    }

    void generateCombinations(int start, int n, int k, List<Integer> curr, List<List<Integer>> combinations) {
        if (curr.size() == k) {
            combinations.add(new ArrayList(curr));
            return;
        }
        for (int i = start; i <= n; i++) {
            curr.add(i);
            generateCombinations(i + 1, n, k, curr, combinations);
            curr.remove(curr.size() - 1);
        }
    }
}
