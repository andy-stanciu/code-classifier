class Solution {

    List<List<Integer>> combinations = new ArrayList<>();

    int n, k;

    public List<List<Integer>> combine(int n, int k) {
        this.n = n;
        this.k = k;
        backtrack(1, new ArrayList<>());
        return combinations;
    }

    void backtrack(int id, List<Integer> combs) {
        if (combs.size() == k) {
            combinations.add(new ArrayList<>(combs));
            return;
        }
        for (int i = id; i <= n; i++) {
            combs.add(i);
            backtrack(i + 1, combs);
            combs.remove(combs.size() - 1);
        }
    }
}
