class Solution {

    public List<List<Integer>> combine(int n, int k) {
        ArrayList<List<Integer>> result = new ArrayList<>();
        combine0(n, 1, k, new ArrayList<>(), result);
        return result;
    }

    private void combine0(int maxElement, int currentElement, int remaining, List<Integer> carry, List<List<Integer>> result) {
        if (remaining == 0) {
            result.add(new ArrayList<>(carry));
        } else if (currentElement + remaining - 1 <= maxElement) {
            carry.add(currentElement);
            combine0(maxElement, currentElement + 1, remaining - 1, carry, result);
            carry.remove(carry.size() - 1);
            combine0(maxElement, currentElement + 1, remaining, carry, result);
        }
    }
}
