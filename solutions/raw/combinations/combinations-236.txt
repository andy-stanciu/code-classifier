class Solution {

    List<List<Integer>> res = new ArrayList<List<Integer>>();

    public List<List<Integer>> combine(int n, int k) {
        combine(n, k, new ArrayList<Integer>(), 1);
        return res;
    }

    public void combine(int n, int k, List<Integer> currList, int currNum) {
        int currSize = currList.size();
        if (currSize == k) {
            res.add(new ArrayList(currList));
            return;
        }
        for (int i = currNum; i <= n - k + currSize + 1; i++) {
            currList.add(i);
            combine(n, k, currList, i + 1);
            currList.remove(currSize);
        }
    }
}
