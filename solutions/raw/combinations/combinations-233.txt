class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> arr = new ArrayList<>();
        for (int i = 1; i <= n; i++) helper(n, i, k, 1, new ArrayList<Integer>(), arr);
        return arr;
    }

    public void helper(int n, int si, int k, int ki, List<Integer> ans, List<List<Integer>> arr) {
        if (si > n) {
            return;
        }
        ans.add(si);
        if (ki == k) {
            arr.add(new ArrayList<Integer>(ans));
            ans.remove(ans.size() - 1);
            return;
        }
        for (int i = si + 1; i <= n; i++) helper(n, i, k, ki + 1, ans, arr);
        ans.remove(ans.size() - 1);
    }
}
