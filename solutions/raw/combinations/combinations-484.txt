class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> combs = new ArrayList<>();
        helper(1, n, k, new ArrayList<>(), combs);
        return combs;
    }

    private void helper(int start, int n, int k, ArrayList<Integer> comb, List<List<Integer>> combs) {
        if (comb.size() == k) {
            combs.add(new ArrayList<>(comb));
        } else {
            for (int i = start; i <= n; ++i) {
                comb.add(i);
                // If you write start+1 instead of i+1, something bad will happen because algorithm is like on the photo.
                // At the root iteration, start will always be 1, but i will change. If you write start+1, you will always
                // start adding at two no matter what because start is 1 at the root.
                helper(i + 1, n, k, comb, combs);
                comb.remove(comb.size() - 1);
            }
        }
    }
}
