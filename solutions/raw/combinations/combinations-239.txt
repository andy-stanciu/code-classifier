class Solution {

    public List<List<Integer>> combine(int n, int k) {
        ArrayList<List<Integer>> res = new ArrayList<List<Integer>>();
        ArrayList<Integer> numCom = new ArrayList<Integer>();
        rec(1, k, n, numCom, res);
        return res;
    }

    public void rec(int currentNumber, int k, int n, ArrayList<Integer> numCom, List<List<Integer>> res) {
        if (numCom.size() == k) {
            res.add(new ArrayList<Integer>(numCom));
        }
        for (int i = currentNumber; i <= n; i++) {
            numCom.add(i);
            rec(i + 1, k, n, numCom, res);
            numCom.remove(numCom.size() - 1);
        }
    }
}
