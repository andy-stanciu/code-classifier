class Solution {

    public void combineUtil(int n, int k, int num, int pos, boolean[] visited, List<List<Integer>> combinations) {
        if (pos == k) {
            List<Integer> combination = new ArrayList<>();
            for (int i = 1; i <= n; i++) {
                if (visited[i])
                    combination.add(i);
            }
            if (combination.size() > 0)
                combinations.add(combination);
            return;
        }
        if (num > n)
            return;
        visited[num] = true;
        combineUtil(n, k, num + 1, pos + 1, visited, combinations);
        visited[num] = false;
        combineUtil(n, k, num + 1, pos, visited, combinations);
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> combinations = new ArrayList<>();
        boolean[] visited = new boolean[n + 1];
        combineUtil(n, k, 1, 0, visited, combinations);
        return combinations;
    }
}
