class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        traverse(ans, n, k, new ArrayList<>(), 1);
        return ans;
    }

    private void traverse(List<List<Integer>> ans, int n, int k, List<Integer> temp, int num) {
        if (temp.size() == k) {
            // fucking pass by reference in java is sometimes a bitch!!
            ans.add(new ArrayList<>(temp));
            return;
        }
        for (int i = num; i <= n; i++) {
            temp.add(i);
            traverse(ans, n, k, temp, i + 1);
            temp.remove(temp.size() - 1);
        }
    }
}
