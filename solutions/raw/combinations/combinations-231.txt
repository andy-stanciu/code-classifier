class Solution {

    public static void helper(int n, int index, int k, List<Integer> output, List<List<Integer>> result) {
        if (output.size() == k) {
            result.add(new ArrayList<>(output));
            return;
        }
        for (int i = index; i <= n; i++) {
            output.add(i);
            helper(n, i + 1, k, output, result);
            output.remove(output.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<Integer> output = new ArrayList<>();
        List<List<Integer>> result = new ArrayList<>();
        helper(n, 1, k, output, result);
        return result;
    }
}
