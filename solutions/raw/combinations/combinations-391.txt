class Solution {

    public List<List<Integer>> combine(int n, int k) {
        int[] arr = new int[n];
        for (int i = 1, j = 0; i <= n; i++) {
            arr[j++] = i;
        }
        List<List<Integer>> ans = new ArrayList<>();
        ArrayList<Integer> li = new ArrayList<>();
        backCom(arr, 0, ans, li, k);
        return ans;
    }

    void backCom(int[] arr, int i, List<List<Integer>> ans, ArrayList<Integer> list, int k) {
        if (list.size() == k) {
            ans.add(new ArrayList<>(list));
            return;
        } else if (i == arr.length)
            return;
        else {
            list.add(arr[i]);
            backCom(arr, i + 1, ans, list, k);
            list.remove(list.size() - 1);
            backCom(arr, i + 1, ans, list, k);
        }
    }
}
