class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        combinations(1, n, k, new ArrayList<>(), res);
        return res;
    }

    public void combinations(int i, int n, int k, List<Integer> ans, List<List<Integer>> res) {
        if (ans.size() == k) {
            res.add(new ArrayList<>(ans));
            return;
        }
        for (int j = i; j <= n; j++) {
            ans.add(j);
            combinations(j + 1, n, k, ans, res);
            ans.remove(ans.size() - 1);
        }
    }
}
