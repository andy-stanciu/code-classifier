/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int index = 0;

    int res = 0;

    public int kthSmallest(TreeNode root, int k) {
        small(root, k);
        return res;
    }

    public void small(TreeNode root, int k) {
        if (root == null)
            return;
        small(root.left, k);
        index++;
        if (index == k)
            res = root.val;
        small(root.right, k);
    }
}
