/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int k;

    int ans;

    public int kthSmallest(TreeNode root, int k) {
        this.k = k;
        this.ans = 0;
        inOrder(root);
        return ans;
    }

    private void inOrder(TreeNode root) {
        if (k <= 0 || root == null) {
            return;
        }
        inOrder(root.left);
        if (--k == 0) {
            ans = root.val;
            return;
        }
        inOrder(root.right);
    }
}
