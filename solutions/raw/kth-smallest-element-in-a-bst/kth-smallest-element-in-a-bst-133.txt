/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = 0, c = 1, d = 0;

    public int kthSmallest(TreeNode root, int k) {
        f(root, k);
        return ans;
    }

    public void f(TreeNode root, int k) {
        if (root == null)
            return;
        f(root.left, k);
        if (c == k && d == 0) {
            ans = root.val;
            d = 1;
        } else
            c++;
        f(root.right, k);
        return;
    }
}
