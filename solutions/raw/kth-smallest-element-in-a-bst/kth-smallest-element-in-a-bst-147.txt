class Solution {

    int kth = 0;

    int mval = 0;

    public int kthSmallest(TreeNode root, int k) {
        inorder(root, k);
        return mval;
    }

    void inorder(TreeNode root, int k) {
        if (root == null)
            return;
        inorder(root.left, k);
        kth++;
        if (kth == k)
            mval = root.val;
        inorder(root.right, k);
    }
}
