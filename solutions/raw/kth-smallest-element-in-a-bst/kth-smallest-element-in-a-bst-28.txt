class Solution {

    int k;

    private int inOrder(TreeNode node) {
        if (node == null)
            return -1;
        var ans = inOrder(node.left);
        if (ans != -1 && --k <= 0)
            return ans;
        if (--k <= 0)
            return node.val;
        return inOrder(node.right);
    }

    public int kthSmallest(TreeNode root, int k) {
        this.k = k;
        return inOrder(root);
    }
}
