class Solution {

    public int kthSmallest(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        adding(root, list);
        Collections.sort(list);
        return list.get(k - 1);
    }

    public void adding(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        adding(root.left, list);
        adding(root.right, list);
    }
}
