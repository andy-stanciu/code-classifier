class Solution {

    private static int ans = -1, cnt = 0;

    private static void Helper(TreeNode root, int k) {
        if (root == null)
            return;
        Helper(root.left, k);
        cnt += 1;
        if (cnt == k) {
            ans = root.val;
            return;
        }
        Helper(root.right, k);
    }

    public int kthSmallest(TreeNode root, int k) {
        Helper(root, k);
        int res = ans;
        cnt = 0;
        return res;
    }
}
