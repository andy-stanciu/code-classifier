/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int kthSmallest(TreeNode root, int k) {
        int[] count = new int[1];
        int[] kthSmallest = new int[1];
        inorder(root, k, count, kthSmallest);
        return kthSmallest[0];
    }

    public void inorder(TreeNode root, int k, int[] count, int[] kthSmallest) {
        if (root == null || count[0] >= k)
            return;
        inorder(root.left, k, count, kthSmallest);
        count[0]++;
        if (count[0] == k) {
            kthSmallest[0] = root.val;
            return;
        }
        inorder(root.right, k, count, kthSmallest);
    }
}
