class Solution {

    //i is the counter for k-th and c is for storing answer.
    // taken c[] and i[] because of lack of pointers in java as we cannot
    //change variable values so have to store it in a 1 sized array.
    public void inorder(TreeNode root, int[] i, int[] c, int k) {
        if (root == null || c[0] >= k) {
            return;
        }
        // As inorder traversal is L-N-R.
        inorder(root.left, i, c, k);
        //increasing counter for counting k-th.
        i[0]++;
        if (i[0] == k) {
            // storing the k-th smallest.
            c[0] = root.val;
            return;
        }
        inorder(root.right, i, c, k);
    }

    public int kthSmallest(TreeNode root, int k) {
        int[] c = new int[1];
        inorder(root, new int[1], c, k);
        return c[0];
    }
}
