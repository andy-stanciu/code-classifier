class Solution {

    int count = 0;

    int kmin = 0;

    public int kthSmallest(TreeNode root, int k) {
        inorder(root, k);
        return kmin;
    }

    public void inorder(TreeNode root, int k) {
        if (root == null)
            return;
        inorder(root.left, k);
        count++;
        if (count == k) {
            kmin = root.val;
            return;
        }
        inorder(root.right, k);
    }
}
