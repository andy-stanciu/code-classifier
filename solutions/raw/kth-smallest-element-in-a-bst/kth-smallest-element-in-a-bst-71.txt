class Solution {

    public int kthSmallest(TreeNode root, int k) {
        int[] count = new int[1];
        return f(root, k, count);
    }

    static int f(TreeNode root, int k, int[] count) {
        if (root == null)
            return -1;
        int left = f(root.left, k, count);
        if (left != -1)
            return left;
        count[0]++;
        if (count[0] == k) {
            return root.val;
        }
        return f(root.right, k, count);
    }
}
