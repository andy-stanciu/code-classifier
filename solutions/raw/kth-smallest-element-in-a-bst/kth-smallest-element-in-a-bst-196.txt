class Solution {

    PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Collections.reverseOrder());

    public int kthSmallest(TreeNode root, int k) {
        inorder(root, k);
        return maxHeap.poll();
    }

    private void inorder(TreeNode root, int k) {
        if (root == null)
            return;
        inorder(root.left, k);
        maxHeap.add(root.val);
        if (maxHeap.size() > k)
            maxHeap.remove();
        inorder(root.right, k);
    }
}
