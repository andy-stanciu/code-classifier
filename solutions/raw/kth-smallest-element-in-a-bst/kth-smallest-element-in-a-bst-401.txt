class Solution {

    private int count = 1;

    public int kthSmallest(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        int res = kthSmallest(root.left, k);
        if (res != -1) {
            return res;
        } else if (count++ == k) {
            return root.val;
        }
        return kthSmallest(root.right, k);
    }
}
