/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public ArrayList kthSmall(TreeNode root, ArrayList<Integer> arr) {
        if (root == null)
            return arr;
        kthSmall(root.left, arr);
        arr.add(root.val);
        kthSmall(root.right, arr);
        return arr;
    }

    public int kthSmallest(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        ArrayList<Integer> finalarr = kthSmall(root, arr);
        return finalarr.get(k - 1);
    }
}
