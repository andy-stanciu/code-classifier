/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    int ans = 0;

    public int kthSmallest(TreeNode root, int k) {
        inOrder(root, k);
        return ans;
    }

    //since inorder of BST is sorted
    public void inOrder(TreeNode curr, int k) {
        if (curr == null)
            return;
        inOrder(curr.left, k);
        count++;
        if (count == k)
            ans = curr.val;
        inOrder(curr.right, k);
    }
}
