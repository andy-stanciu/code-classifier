/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int min = 0;

    public int kthSmallest(TreeNode root, int k) {
        func(root, k);
        return min;
    }

    int func(TreeNode root, int k) {
        if (root == null)
            return k;
        if (k > 0) {
            k = func(root.left, k);
            if (k > 0) {
                min = root.val;
                --k;
            }
        }
        if (k > 0) {
            k = func(root.right, k);
        }
        return k;
    }
}
