class Solution {

    List<Integer> result = new ArrayList<>();

    public int kthSmallest(TreeNode root, int k) {
        InOrder(root);
        return result.get(k - 1);
    }

    void InOrder(TreeNode root) {
        if (root == null)
            return;
        InOrder(root.left);
        result.add(root.val);
        InOrder(root.right);
    }
}
