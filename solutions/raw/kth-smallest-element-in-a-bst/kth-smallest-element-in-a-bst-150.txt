/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int kthSmallest(TreeNode root, int k) {
        int[] ans = new int[] { 0 };
        int[] kthSmallest = new int[] { 0 };
        find(root, ans, kthSmallest, k);
        return ans[0];
    }

    public void find(TreeNode root, int[] ans, int[] kthSmallest, int k) {
        if (kthSmallest[0] > k || root == null) {
            return;
        }
        find(root.left, ans, kthSmallest, k);
        kthSmallest[0]++;
        if (k == kthSmallest[0]) {
            ans[0] = root.val;
            return;
        }
        find(root.right, ans, kthSmallest, k);
    }
}
