class Solution {

    public int kthSmallest(TreeNode root, int k) {
        int i = 0;
        while (root != null) {
            if (root.left == null) {
                i++;
                if (i == k)
                    return root.val;
                root = root.right;
            } else {
                TreeNode prev = root.left;
                while (prev.right != null && prev.right != root) prev = prev.right;
                if (prev.right == root) {
                    i++;
                    if (i == k)
                        return root.val;
                    prev.right = null;
                    root = root.right;
                } else {
                    prev.right = root;
                    root = root.left;
                }
            }
        }
        return -1;
    }
}
