class Solution {

    public int kthSmallest(TreeNode root, int k) {
        Queue<TreeNode> q = new LinkedList<>();
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>();
        helper(pq, root);
        while (k > 1) {
            pq.poll();
            k--;
        }
        return pq.poll();
    }

    public void helper(PriorityQueue<Integer> pq, TreeNode r) {
        if (r == null)
            return;
        pq.add(r.val);
        helper(pq, r.left);
        helper(pq, r.right);
    }
}
