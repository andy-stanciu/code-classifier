class Solution {

    private int k;

    private int result;

    public int kthSmallest(TreeNode root, int k) {
        this.k = k;
        small(root);
        return result;
    }

    private void small(TreeNode root) {
        if (root == null)
            return;
        small(root.left);
        k--;
        if (k == 0) {
            result = root.val;
            return;
        }
        small(root.right);
    }
}
