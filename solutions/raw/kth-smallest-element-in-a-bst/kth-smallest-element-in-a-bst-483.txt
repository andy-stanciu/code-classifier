class Solution {

    public int kthSmallest(TreeNode root, int k) {
        List<Integer> values = new ArrayList<>();
        inOrder(root, values);
        return values.get(k - 1);
    }

    private void inOrder(TreeNode root, List<Integer> values) {
        // base case
        if (root == null) {
            return;
        }
        // recursive case
        inOrder(root.left, values);
        values.add(root.val);
        inOrder(root.right, values);
    }
}
