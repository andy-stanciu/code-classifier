/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    PriorityQueue<Integer> pq = new PriorityQueue<>();

    public int kthSmallest(TreeNode root, int k) {
        filler(root, pq);
        int cnt = 0;
        int ans = 0;
        while (!pq.isEmpty()) {
            ans = pq.remove();
            cnt++;
            if (cnt == k) {
                break;
            }
        }
        return ans;
    }

    public void filler(TreeNode root, PriorityQueue<Integer> pq) {
        if (root == null) {
            return;
        }
        pq.add(root.val);
        filler(root.left, pq);
        filler(root.right, pq);
    }
}
