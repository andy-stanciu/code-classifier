class Solution {

    int count = 0;

    public int kthSmallest(TreeNode root, int k) {
        int[] ans = new int[1];
        inorder(root, k, ans);
        return ans[0];
    }

    public void inorder(TreeNode root, int k, int[] ans) {
        if (root == null) {
            return;
        }
        inorder(root.left, k, ans);
        count++;
        if (k == count) {
            ans[0] = root.val;
            return;
        }
        inorder(root.right, k, ans);
    }
}
