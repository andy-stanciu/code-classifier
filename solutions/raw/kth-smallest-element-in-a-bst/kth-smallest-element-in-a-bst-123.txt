/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int kthSmallest(TreeNode root, int K) {
        if (root.left == null && root.right == null)
            return root.val;
        int[] ans = new int[1];
        smallest(root, K, ans);
        return ans[0];
    }

    int index = 0;

    public boolean smallest(TreeNode root, int k, int[] ans) {
        if (root == null)
            return false;
        if (smallest(root.left, k, ans))
            return true;
        index++;
        if (k - index == 0) {
            ans[0] = root.val;
            return true;
        }
        if (smallest(root.right, k, ans))
            return true;
        return false;
    }
}
