class Solution {

    public int kthSmallest(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        solution(root, list);
        return (list.get(k - 1));
    }

    private void solution(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        solution(root.left, list);
        list.add(root.val);
        solution(root.right, list);
    }
}
