/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int cnt;

    private int result;

    public int kthSmallest(TreeNode root, int k) {
        cnt = k;
        visit(root);
        return result;
    }

    private void visit(TreeNode root) {
        if (root == null) {
            return;
        }
        visit(root.left);
        cnt--;
        if (cnt == 0) {
            result = root.val;
            return;
        } else // You got what you want already. Why do you want to do any more???
        // All you want now is finish it right away!
        if (cnt < 0) {
            return;
        }
        visit(root.right);
    }
}
