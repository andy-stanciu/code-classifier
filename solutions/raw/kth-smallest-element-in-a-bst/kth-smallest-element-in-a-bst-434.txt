/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int min;

    private int curIndex;

    public void inOrderTraversal(TreeNode curNode, int k) {
        if (curNode == null) {
            return;
        }
        inOrderTraversal(curNode.left, k);
        curIndex++;
        if (curIndex == k) {
            min = curNode.val;
            return;
        }
        inOrderTraversal(curNode.right, k);
    }

    public int kthSmallest(TreeNode root, int k) {
        curIndex = 0;
        inOrderTraversal(root, k);
        return min;
    }
}
