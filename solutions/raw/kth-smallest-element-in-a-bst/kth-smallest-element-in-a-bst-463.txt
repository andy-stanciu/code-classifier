class Solution {

    public int ctr = 0;

    public int result = 0;

    public int kthSmallest(TreeNode root, int k) {
        traverse(root, k);
        return result;
    }

    public void traverse(TreeNode n, int k) {
        if (n == null)
            return;
        traverse(n.left, k);
        ctr++;
        if (ctr == k) {
            result = n.val;
            return;
        }
        traverse(n.right, k);
    }
}
