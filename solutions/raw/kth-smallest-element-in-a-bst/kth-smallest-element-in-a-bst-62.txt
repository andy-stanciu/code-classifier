class Solution {

    private int result = -1, k = 0;

    public int kthSmallest(TreeNode root, int k) {
        this.k = k;
        kthSmallest(root);
        return result;
    }

    private void kthSmallest(TreeNode root) {
        if (root == null) {
            return;
        }
        kthSmallest(root.left);
        if (--k == 0) {
            result = root.val;
            return;
        }
        kthSmallest(root.right);
    }
}
