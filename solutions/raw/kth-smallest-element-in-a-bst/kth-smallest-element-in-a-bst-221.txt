/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int counter = 0;

    private int kth = 0;

    public int kthSmallest(TreeNode root, int k) {
        recursive(root, k);
        return kth;
    }

    private void recursive(TreeNode node, int k) {
        if (node == null || counter >= k)
            return;
        recursive(node.left, k);
        if (++counter == k) {
            kth = node.val;
            return;
        }
        recursive(node.right, k);
    }
}
