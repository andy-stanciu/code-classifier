/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    private int counter = 0;

    private int result;

    public int kthSmallest(TreeNode root, int k) {
        in_order_traversal(root, k);
        return result;
    }

    // find kth smallest and early return
    public boolean in_order_traversal(TreeNode root, int k) {
        // base case
        if (root == null)
            return false;
        // general case
        if (in_order_traversal(root.left, k))
            return true;
        counter++;
        if (counter == k) {
            result = root.val;
            return true;
        }
        if (in_order_traversal(root.right, k))
            return true;
        return false;
    }
}
