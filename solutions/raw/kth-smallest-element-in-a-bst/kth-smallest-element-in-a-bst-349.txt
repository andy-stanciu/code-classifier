/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int kGlobal;

    public int kthSmallest(TreeNode root, int k) {
        kGlobal = k;
        return answer(root);
    }

    private int answer(TreeNode root) {
        if (root == null) {
            return -1;
        }
        int left = answer(root.left);
        if (left != -1) {
            return left;
        }
        kGlobal--;
        if (kGlobal == 0) {
            return root.val;
        }
        int right = answer((root.right));
        if (right != -1) {
            return right;
        }
        return -1;
    }
}
