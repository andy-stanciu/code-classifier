class Solution {

    private int i;

    private int getKthSmallest(TreeNode root, int k) {
        if (root == null)
            return -1;
        int left = getKthSmallest(root.left, k);
        if (left != -1)
            return left;
        i++;
        if (i == k)
            return root.val;
        int right = getKthSmallest(root.right, k);
        if (right != -1)
            return right;
        return -1;
    }

    public int kthSmallest(TreeNode root, int k) {
        i = 0;
        return getKthSmallest(root, k);
    }
}
