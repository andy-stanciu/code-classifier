class Solution {

    List<Integer> list = new ArrayList<>();

    public int kthSmallest(TreeNode root, int k) {
        in(root);
        return list.get(k - 1);
    }

    public void in(TreeNode root) {
        if (root == null) {
            return;
        }
        in(root.left);
        list.add(root.val);
        in(root.right);
    }
}
