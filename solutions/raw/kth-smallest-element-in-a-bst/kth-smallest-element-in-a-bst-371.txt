/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int value = -1;

    int i = 0;

    public int kthSmallest(TreeNode root, int k) {
        readTree(root, k);
        return value;
    }

    private void readTree(TreeNode node, int k) {
        if (node == null || value != -1)
            return;
        readTree(node.left, k);
        i++;
        if (i == k)
            value = node.val;
        readTree(node.right, k);
    }
}
