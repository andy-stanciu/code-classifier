/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    ArrayList<Integer> ans = new ArrayList<>();

    public int kthSmallest(TreeNode root, int k) {
        fillArrayList(root, ans);
        Integer[] arr = new Integer[ans.size()];
        arr = ans.toArray(arr);
        //Arrays.sort(arr);
        return arr[k - 1];
    }

    void fillArrayList(TreeNode root, ArrayList<Integer> ans) {
        if (root == null) {
            return;
        }
        fillArrayList(root.left, ans);
        ans.add(root.val);
        fillArrayList(root.right, ans);
    }
}
