/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int kthSmallest(TreeNode root, int k) {
        TreeNode curr = root;
        TreeNode pred = null;
        while (curr != null) {
            if (curr.left == null) {
                if (k == 1)
                    return curr.val;
                k--;
                curr = curr.right;
            } else {
                pred = getPredecessor(curr);
                if (pred.right == null) {
                    pred.right = curr;
                    curr = curr.left;
                } else {
                    pred.right = null;
                    if (k == 1)
                        return curr.val;
                    k--;
                    curr = curr.right;
                }
            }
        }
        return 0;
    }

    private TreeNode getPredecessor(TreeNode root) {
        TreeNode curr = root.left;
        while (curr.right != null && curr.right != root) curr = curr.right;
        return curr;
    }
}
