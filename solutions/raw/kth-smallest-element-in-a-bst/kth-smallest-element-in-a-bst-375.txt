/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    private int currIter = 0;

    private int ans = -1;

    private void inOrder(TreeNode root, int k) {
        if (root != null) {
            if (this.currIter != k) {
                inOrder(root.left, k);
            }
            if (this.currIter != k) {
                this.ans = root.val;
                this.currIter++;
            }
            if (this.currIter != k) {
                inOrder(root.right, k);
            }
        }
    }

    public int kthSmallest(TreeNode root, int k) {
        inOrder(root, k);
        return this.ans;
    }
}
