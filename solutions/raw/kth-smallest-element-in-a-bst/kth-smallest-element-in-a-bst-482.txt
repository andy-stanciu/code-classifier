class Solution {

    int value = -1;

    int cnt = 0;

    public int kthSmallest(TreeNode root, int k) {
        inOrder(root, k);
        return value;
    }

    private void inOrder(TreeNode root, int k) {
        // base case
        if (root == null || cnt == k) {
            return;
        }
        // recursive case
        inOrder(root.left, k);
        if (cnt < k) {
            value = root.val;
            cnt++;
        }
        inOrder(root.right, k);
    }
}
