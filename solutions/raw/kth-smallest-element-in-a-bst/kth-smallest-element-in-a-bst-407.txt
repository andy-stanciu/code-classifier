/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> pre(TreeNode root, List<Integer> l) {
        if (root != null) {
            l = pre(root.left, l);
            l.add(root.val);
            l = pre(root.right, l);
        }
        return l;
    }

    public int kthSmallest(TreeNode root, int k) {
        List<Integer> l = new ArrayList<>();
        l = pre(root, l);
        return l.get(k - 1);
    }
}
