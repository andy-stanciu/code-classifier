class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> library = new HashMap<>();
        if (ransomNote.length() > magazine.length()) {
            return false;
        } else {
            for (int i = 0; i < magazine.length(); i++) {
                char letter = magazine.charAt(i);
                if (!library.containsKey(letter)) {
                    library.put(letter, 1);
                } else {
                    library.put(letter, library.get(letter) + 1);
                }
            }
            for (int j = 0; j < ransomNote.length(); j++) {
                char letter2 = ransomNote.charAt(j);
                if (library.containsKey(letter2) && library.get(letter2) > 0) {
                    library.put(letter2, library.get(letter2) - 1);
                } else {
                    return false;
                }
            }
            return true;
        }
    }
}
