class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] letters = ransomNote.toCharArray();
        for (char note : letters) {
            if (magazine.indexOf(note) >= 0) {
                magazine = magazine.replaceFirst(note + "", "");
            } else {
                return false;
            }
        }
        return true;
    }
}
