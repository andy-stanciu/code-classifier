class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int[] c1 = new int[128];
        int[] c2 = new int[128];
        for (int i = 0; i < ransomNote.length(); i++) {
            c1[ransomNote.charAt(i)]++;
        }
        for (int i = 0; i < magazine.length(); i++) {
            c2[magazine.charAt(i)]++;
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            if (c2[ransomNote.charAt(i)] < c1[ransomNote.charAt(i)]) {
                return false;
            }
        }
        return true;
    }
}
