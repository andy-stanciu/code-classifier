/*
1.convert the string to the array of characters
2.loop through each character of letter if ransomNote is present in the mazagine
3. if the character is found, remove the first occurance of the found element from magazineCharacters array
*/
class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] ransomeNoteCharacters = ransomNote.toCharArray();
        char[] magazineCharacters = magazine.toCharArray();
        Boolean isAvailable = false;
        for (int i = 0; i < ransomeNoteCharacters.length; i++) {
            if (containsElement(ransomeNoteCharacters[i], magazineCharacters)) {
                isAvailable = true;
                magazineCharacters = removeFirstOccurrence(magazineCharacters, ransomeNoteCharacters[i]);
            } else {
                isAvailable = false;
                break;
            }
        }
        return isAvailable;
    }

    Boolean containsElement(char singleCharacter, char[] characters) {
        Boolean isAvailable = false;
        for (int i = 0; i < characters.length; i++) {
            if (characters[i] == singleCharacter) {
                isAvailable = true;
                return isAvailable;
            }
        }
        return isAvailable;
    }

    private static char[] removeFirstOccurrence(char[] array, char element) {
        int indexToRemove = -1;
        for (int i = 0; i < array.length; i++) {
            if (array[i] == element) {
                indexToRemove = i;
                break;
            }
        }
        if (indexToRemove != -1) {
            for (int i = indexToRemove; i < array.length - 1; i++) {
                array[i] = array[i + 1];
            }
            return Arrays.copyOf(array, array.length - 1);
        } else {
            // Return the original array if the element is not found
            return array;
        }
    }
}
