class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> hsh = new HashMap<>();
        Map<Character, Integer> hs2 = new HashMap<>();
        for (int j = 0; j < ransomNote.length(); j++) {
            char ch2 = ransomNote.charAt(j);
            if (hs2.containsKey(ch2)) {
                hs2.put(ch2, hs2.get(ch2) + 1);
            } else {
                hs2.put(ch2, 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            char ch = magazine.charAt(i);
            if ((hsh.containsKey(ch)) && (hs2.get(ch) > hsh.get(ch))) {
                hsh.put(ch, hsh.get(ch) + 1);
            } else if ((hsh.containsKey(ch)) && !((hs2.get(ch) > hsh.get(ch)))) {
                continue;
            } else if (hs2.containsKey(ch)) {
                hsh.put(ch, 1);
            }
        }
        return hsh.equals(hs2);
    }
}
