public class Solution {

    public boolean canConstruct(String ransMote, String magazine) {
        HashMap<Character, Integer> magazineLetters = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char m = magazine.charAt(i);
            int currentCountOfL = magazineLetters.getOrDefault(m, 0);
            magazineLetters.put(m, currentCountOfL + 1);
        }
        for (int i = 0; i < ransMote.length(); i++) {
            char r = ransMote.charAt(i);
            int currentCountOfL = magazineLetters.getOrDefault(r, 0);
            if (currentCountOfL == 0) {
                return false;
            }
            magazineLetters.put(r, currentCountOfL - 1);
        }
        return true;
    }
}
