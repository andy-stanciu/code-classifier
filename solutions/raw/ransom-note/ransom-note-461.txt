class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length())
            return false;
        var a = ransomNote.toCharArray();
        var b = magazine.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        var flag = false;
        for (int i = 0, j = 0; i < a.length; ) {
            if (j == b.length)
                return false;
            else if (a[i] == b[j]) {
                flag = true;
                i++;
            } else
                flag = false;
            j++;
        }
        return flag;
    }
}
