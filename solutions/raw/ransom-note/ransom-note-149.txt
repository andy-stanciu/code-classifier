class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        boolean result = false;
        String[] ransomNew = ransomNote.split("");
        String[] magazineNew = magazine.split("");
        int ransomCounter = 0;
        for (int x = 0; x < ransomNew.length; x++) {
            for (int i = 0; i < magazineNew.length; i++) {
                if (ransomNew[x].equals(magazineNew[i])) {
                    result = true;
                    magazineNew[i] = "";
                    ransomCounter += 1;
                    break;
                }
            }
        }
        if (ransomCounter >= ransomNew.length) {
            return result;
        } else {
            return false;
        }
    }
}
