class Solution {

    public boolean canConstruct(String s1, String s2) {
        StringBuilder str = new StringBuilder(s2);
        for (char c : s1.toCharArray()) {
            int ind = str.indexOf(String.valueOf(c));
            if (ind == -1) {
                return false;
            }
            str.deleteCharAt(ind);
        }
        return true;
    }
}
