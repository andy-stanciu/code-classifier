class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // Create a HashMap to store the frequency of each character in the magazine
        HashMap<Character, Integer> letterCount = new HashMap<>();
        // Count the frequency of each character in magazine
        for (char c : magazine.toCharArray()) {
            letterCount.put(c, letterCount.getOrDefault(c, 0) + 1);
        }
        // Check each character in ransomNote
        for (char c : ransomNote.toCharArray()) {
            // If character is not available or count is insufficient, return false
            if (!letterCount.containsKey(c) || letterCount.get(c) == 0) {
                return false;
            }
            // Decrease the count of the character in the map
            letterCount.put(c, letterCount.get(c) - 1);
        }
        // All characters in ransomNote are available in magazine
        return true;
    }
}
