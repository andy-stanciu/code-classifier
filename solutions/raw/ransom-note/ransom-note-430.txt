class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length()) {
            return false;
        }
        HashMap<Character, Integer> ransomMap = new HashMap<>();
        HashMap<Character, Integer> magzineMap = new HashMap<>();
        for (int i = 0; i < ransomNote.length(); i++) {
            if (ransomMap.containsKey(ransomNote.charAt(i))) {
                ransomMap.put(ransomNote.charAt(i), (ransomMap.get(ransomNote.charAt(i))) + 1);
            } else {
                ransomMap.put(ransomNote.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (magzineMap.containsKey(magazine.charAt(i))) {
                magzineMap.put(magazine.charAt(i), (magzineMap.get(magazine.charAt(i))) + 1);
            } else {
                magzineMap.put(magazine.charAt(i), 1);
            }
        }
        for (Character ch : ransomMap.keySet()) {
            if (magzineMap.get(ch) != null) {
                if ((Integer.valueOf(ransomMap.get(ch)) > Integer.valueOf(magzineMap.get(ch)))) {
                    return false;
                }
            } else if (magzineMap.get(ch) == null) {
                return false;
            }
        }
        return true;
    }
}
