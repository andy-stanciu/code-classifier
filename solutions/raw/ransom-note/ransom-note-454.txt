class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<String, Integer> magazineLetters = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            magazineLetters.compute(String.valueOf(c), (k, v) -> v == null ? 1 : ++v);
        }
        for (char c : ransomNote.toCharArray()) {
            if (magazineLetters.compute(String.valueOf(c), (k, v) -> v == null || v == 0 ? null : --v) == null) {
                return false;
            }
        }
        return true;
    }
}
