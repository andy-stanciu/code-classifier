class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<String, Integer> ransomNoteMap = new HashMap<>();
        String[] ransomNotesplit = ransomNote.split("");
        for (String a : ransomNotesplit) {
            ransomNoteMap.merge(a, 1, Integer::sum);
        }
        HashMap<String, Integer> magazineMap = new HashMap<>();
        String[] magazinesplit = magazine.split("");
        for (String a : magazinesplit) {
            magazineMap.merge(a, 1, Integer::sum);
        }
        boolean mission = false;
        for (String a : ransomNotesplit) {
            if (magazineMap.get(a) != null && magazineMap.get(a) >= ransomNoteMap.get(a)) {
                mission = true;
            } else {
                mission = false;
                break;
            }
        }
        return mission;
    }
}
