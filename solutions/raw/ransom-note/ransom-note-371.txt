class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        for (int i = 0; i < ransomNote.length(); i++) {
            String aChar = String.valueOf(ransomNote.charAt(i));
            if (!magazine.contains(aChar)) {
                return false;
            }
            magazine = magazine.replaceFirst(aChar, "");
        }
        return true;
    }
}
