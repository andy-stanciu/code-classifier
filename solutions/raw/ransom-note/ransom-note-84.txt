class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> charCountMap = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            charCountMap.put(c, charCountMap.getOrDefault(c, 0) + 1);
        }
        for (char m : ransomNote.toCharArray()) {
            // If character is not present or its count is zero, return false
            if (!charCountMap.containsKey(m) || charCountMap.get(m) == 0) {
                return false;
            } else {
                // Decrement the count of the character
                charCountMap.put(m, charCountMap.get(m) - 1);
            }
        }
        return true;
    }
}
