class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        for (int i = 0; i < ransomNote.length(); i++) {
            char r = ransomNote.charAt(i);
            int m = magazine.indexOf(r);
            if (m == -1) {
                return false;
            }
            magazine = magazine.substring(0, m) + magazine.substring(m + 1);
        }
        return true;
    }
}
