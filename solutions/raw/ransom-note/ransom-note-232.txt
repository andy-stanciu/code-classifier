class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        var ransomChars = toCharArr(ransomNote);
        var ransomList = toCharList();
        var ransomListCopy = toCharList();
        populateList(ransomList, ransomChars);
        populateList(ransomListCopy, ransomChars);
        var magazineChars = toCharArr(magazine);
        var magazineList = toCharList();
        populateList(magazineList, magazineChars);
        ransomListCopy.forEach(element -> {
            if (magazineList.contains(element)) {
                magazineList.remove(element);
                ransomList.remove(element);
            }
        });
        return ransomList.isEmpty();
    }

    private char[] toCharArr(String input) {
        return input.toCharArray();
    }

    private ArrayList<Character> toCharList() {
        return new ArrayList<>();
    }

    private void populateList(ArrayList<Character> list, char[] chars) {
        for (char ch : chars) {
            list.add(ch);
        }
    }
}
