class Solution {

    public String sortString(String s) {
        char[] tempS = s.toCharArray();
        Arrays.sort(tempS);
        return new String(tempS);
    }

    public boolean canConstruct(String ransomNote, String magazine) {
        String str1 = sortString(magazine);
        String str2 = sortString(ransomNote);
        int li = 0, ri = 0, le = ransomNote.length(), re = magazine.length();
        if (str1.indexOf(str2) != -1) {
            return true;
        } else if (li < le) {
            while (li < le && ri < re) {
                if (str2.charAt(li) == str1.charAt(ri)) {
                    li++;
                    if (li == le)
                        return true;
                }
                ri++;
            }
        }
        return false;
    }
}
