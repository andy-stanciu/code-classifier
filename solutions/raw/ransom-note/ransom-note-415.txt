class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> check = new HashMap<>();
        char[] check2 = ransomNote.toCharArray();
        char[] check3 = magazine.toCharArray();
        for (int i = 0; i < check3.length; i++) {
            if (check.containsKey(check3[i])) {
                check.put(check3[i], check.get(check3[i]) + 1);
            } else {
                check.put(check3[i], 1);
            }
        }
        for (int j = 0; j < check2.length; j++) {
            if (check.containsKey(check2[j])) {
                if ((check.get(check2[j])) > 0) {
                    check.put(check2[j], check.get(check2[j]) - 1);
                } else {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
