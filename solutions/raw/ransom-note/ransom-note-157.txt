class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // this is to map the last index of each character found in the "magazine"
        int[] alphabet = new int[26];
        for (char c : ransomNote.toCharArray()) {
            // this will return the index of c found starting at the index of "alphabet[c % 26]"
            // so the previous character used to match will not be included as the index will be move forward
            int i = magazine.indexOf(c, alphabet[c % 26]);
            // return false, if the specific character is not found
            if (i == -1)
                return false;
            // +1 to make sure the next search will not use current index
            alphabet[c % 26] = i + 1;
        }
        // if reach here means all the character in ransomNote is found then its true
        return true;
    }
}
