class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length()) {
            return false;
        }
        // create note map
        Map<Character, Integer> noteMap = new HashMap<>();
        for (int i = 0; i < ransomNote.length(); i++) {
            char c = ransomNote.charAt(i);
            if (noteMap.containsKey(c)) {
                noteMap.put(c, noteMap.get(c) + 1);
            } else {
                noteMap.put(c, 1);
            }
        }
        // remove chars contained in magazine
        for (int j = 0; j < magazine.length(); j++) {
            char c2 = magazine.charAt(j);
            if (noteMap.containsKey(c2)) {
                if (noteMap.get(c2) == 1) {
                    noteMap.remove(c2);
                } else {
                    noteMap.put(c2, noteMap.get(c2) - 1);
                }
            }
        }
        return noteMap.isEmpty();
    }
}
