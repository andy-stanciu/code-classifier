class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int[] freq = new int[26];
        for (char c : magazine.toCharArray()) {
            int index = c - 97;
            freq[index] += 1;
        }
        for (char c : ransomNote.toCharArray()) {
            int index = c - 97;
            if (freq[index] <= 0) {
                return false;
            } else {
                freq[index] -= 1;
            }
        }
        return true;
    }
}
