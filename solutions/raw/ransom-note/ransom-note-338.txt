class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] tempRansom = ransomNote.toCharArray();
        char[] tempMag = magazine.toCharArray();
        Arrays.sort(tempRansom);
        Arrays.sort(tempMag);
        String str1 = new String(tempMag);
        String str2 = new String(tempRansom);
        int li = 0, ri = 0, le = ransomNote.length(), re = magazine.length();
        if (str1.indexOf(str2) != -1) {
            return true;
        } else if (li < le) {
            while (li < le && ri < re) {
                if (str2.charAt(li) == str1.charAt(ri)) {
                    li++;
                    if (li == le)
                        return true;
                }
                ri++;
            }
            return false;
        }
        return false;
    }
}
