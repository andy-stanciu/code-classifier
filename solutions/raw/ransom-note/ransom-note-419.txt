class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int ransomNote_len = ransomNote.length();
        int magazine_len = magazine.length();
        HashMap<Character, Integer> map_ransomNote = new HashMap<>();
        HashMap<Character, Integer> map_magazine = new HashMap<>();
        boolean res = true;
        for (int i = 0; i < ransomNote_len; i++) {
            if (map_ransomNote.containsKey(ransomNote.charAt(i))) {
                map_ransomNote.put(ransomNote.charAt(i), map_ransomNote.get(ransomNote.charAt(i)) + 1);
            } else {
                map_ransomNote.put(ransomNote.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine_len; i++) {
            if (map_magazine.containsKey(magazine.charAt(i))) {
                map_magazine.put(magazine.charAt(i), map_magazine.get(magazine.charAt(i)) + 1);
            } else {
                map_magazine.put(magazine.charAt(i), 1);
            }
        }
        for (Character ch : map_ransomNote.keySet()) {
            if (!map_magazine.containsKey(ch)) {
                res = false;
                break;
            }
            if (map_ransomNote.get(ch) > map_magazine.get(ch)) {
                res = false;
                break;
            }
        }
        return res;
    }
}
