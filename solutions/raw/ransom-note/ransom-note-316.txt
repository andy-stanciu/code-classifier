class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> availableLetters = new HashMap<>(26);
        for (char letter : magazine.toCharArray()) {
            int num = availableLetters.getOrDefault(letter, 0);
            availableLetters.put(letter, num + 1);
        }
        for (char letter : ransomNote.toCharArray()) {
            Integer num = availableLetters.get(letter);
            if (num == null || num == 0) {
                return false;
            }
            availableLetters.put(letter, num - 1);
        }
        return true;
    }
}
