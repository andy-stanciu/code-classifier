class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] ransomChars = ransomNote.toCharArray();
        char[] magazineChars = magazine.toCharArray();
        Arrays.sort(magazineChars);
        Arrays.sort(ransomChars);
        int ransomIndex = 0;
        for (int i = 0; i < magazineChars.length && ransomIndex < ransomChars.length; i++) {
            char ch = magazineChars[i];
            if (ch == ransomChars[ransomIndex]) {
                ransomIndex++;
            }
        }
        return ransomIndex == ransomChars.length;
    }
}
