class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> map = new HashMap(magazine.length());
        // Look at every character in magazine.
        // Map characters to number of occurrences.
        for (int i = 0; i < magazine.length(); i++) {
            char current = magazine.charAt(i);
            if (map.containsKey(current)) {
                int occurrences = map.get(current);
                map.replace(current, ++occurrences);
            } else {
                map.put(current, 1);
            }
        }
        // Look at every character in ransomNote.
        // If it is in map, decrement the number of occurrences.
        for (int i = 0; i < ransomNote.length(); i++) {
            char current = ransomNote.charAt(i);
            if (map.containsKey(current) && map.get(current) > 0) {
                int occurrences = map.get(current);
                map.replace(current, --occurrences);
            } else {
                return false;
            }
        }
        return true;
    }
}
