class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        List<String> ransomNotes = Arrays.stream(ransomNote.split("")).toList();
        List<String> magazines = new ArrayList<>(Arrays.stream(magazine.split("")).toList());
        for (String r : ransomNotes) {
            if (magazines.contains(r)) {
                magazines.remove(r);
            } else {
                return false;
            }
        }
        return true;
    }
}
