class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> mOccur = new HashMap();
        for (char m : magazine.toCharArray()) {
            if (mOccur.get(m) != null) {
                mOccur.put(m, mOccur.get(m) + 1);
            } else {
                mOccur.put(m, 1);
            }
        }
        for (char r : ransomNote.toCharArray()) {
            int m = mOccur.getOrDefault(r, 0);
            if (m > 0) {
                mOccur.put(r, mOccur.get(r) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
