class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> words = new HashMap<>();
        //populate the Hashmap with characters from magazine, and calculate the frequency of repitition
        for (int i = 0; i < magazine.length(); i++) {
            if (words.containsKey(magazine.charAt(i))) {
                words.put(magazine.charAt(i), words.get(magazine.charAt(i)) + 1);
            } else {
                words.put(magazine.charAt(i), 1);
            }
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            //check if the character from ransom note exists, and itd frequency of repitition is greater than 0
            if (words.containsKey(ransomNote.charAt(i)) && words.get(ransomNote.charAt(i)) > 0) {
                //System.out.print(words);
                //If the Character was found reducy the frequency count
                words.put(ransomNote.charAt(i), words.get(ransomNote.charAt(i)) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
