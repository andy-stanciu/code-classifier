class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> ransomMap = new HashMap<>();
        Map<Character, Integer> magazineMap = new HashMap<>();
        for (int i = 0; i < ransomNote.length(); i++) {
            if (ransomMap.containsKey(ransomNote.charAt(i))) {
                Integer val = ransomMap.get(ransomNote.charAt(i));
                ransomMap.put(ransomNote.charAt(i), ++val);
            } else {
                ransomMap.put(ransomNote.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (magazineMap.containsKey(magazine.charAt(i))) {
                Integer val = magazineMap.get(magazine.charAt(i));
                magazineMap.put(magazine.charAt(i), ++val);
            } else {
                magazineMap.put(magazine.charAt(i), 1);
            }
        }
        for (Character key : ransomMap.keySet()) {
            Integer ransomVal = ransomMap.get(key);
            Integer magazineVal = magazineMap.get(key);
            ransomVal = (ransomVal == null) ? 0 : ransomVal;
            magazineVal = (magazineVal == null) ? 0 : magazineVal;
            if (ransomVal > magazineVal) {
                return false;
            }
        }
        return true;
    }
}
