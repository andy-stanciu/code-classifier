class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] r = ransomNote.toCharArray();
        char[] m = magazine.toCharArray();
        int i = 0, j = 0;
        Arrays.sort(r);
        Arrays.sort(m);
        if (Arrays.equals(r, m)) {
            return true;
        }
        if (r.length > m.length) {
            return false;
        }
        while (i < r.length && j < m.length) {
            if (r[i] == m[j]) {
                i++;
            }
            j++;
        }
        return i == r.length;
    }
}
