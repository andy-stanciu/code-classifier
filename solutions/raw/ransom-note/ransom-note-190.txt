class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] a = ransomNote.toCharArray();
        char[] b = magazine.toCharArray();
        int[] arr = new int[123];
        for (int i = 0; i < b.length; i++) {
            int cint = (int) b[i];
            arr[cint]++;
        }
        int cnt = 0;
        for (int i = 0; i < a.length; i++) {
            int cint = (int) a[i];
            if (arr[cint] > 0) {
                arr[cint]--;
                cnt++;
            }
        }
        return cnt == a.length;
    }
}
