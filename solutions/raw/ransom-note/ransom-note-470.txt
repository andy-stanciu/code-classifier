class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] charRansom = ransomNote.toCharArray();
        char[] charMagazine = magazine.toCharArray();
        Arrays.sort(charRansom);
        Arrays.sort(charMagazine);
        int flag = 0;
        int i = 0;
        int j = 0;
        while (i < charRansom.length && j < charMagazine.length) {
            if (charRansom[i] == charMagazine[j]) {
                // Move to the next character in charRansom
                i++;
            }
            // Always move to the next character in charMagazine
            j++;
        }
        // Return true if all characters in charRansom are matched
        return i == charRansom.length;
    }
}
