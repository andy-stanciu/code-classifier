class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        //create frequency map
        Map<Character, Integer> targetFMap = new HashMap<>();
        Map<Character, Integer> sourceFMap = new HashMap<>();
        for (int i = 0; i < ransomNote.length(); i++) {
            Character c = ransomNote.charAt(i);
            targetFMap.put(c, targetFMap.getOrDefault(c, 1) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            Character c = magazine.charAt(i);
            sourceFMap.put(c, sourceFMap.getOrDefault(c, 1) + 1);
        }
        for (Character key : targetFMap.keySet()) {
            int tCount = targetFMap.get(key);
            int sCount = sourceFMap.getOrDefault(key, 0);
            if (sCount < tCount)
                return false;
        }
        return true;
    }
}
