class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> map = new HashMap();
        // Put all chars from magazine and add/increment it\'s frequency.
        for (char m : magazine.toCharArray()) {
            map.put(m, map.getOrDefault(m, 0) + 1);
        }
        // Check map.
        for (char r : ransomNote.toCharArray()) {
            // if map contains the char, and the frquency is > 0, we reduce the frequency
            if (map.containsKey(r) && map.get(r) > 0) {
                map.put(r, map.get(r) - 1);
            } else {
                // char is not present in map, OR frequency is 0
                return false;
            }
        }
        return true;
    }
}
