class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] magazine1 = magazine.toCharArray();
        char[] ransomNote1 = ransomNote.toCharArray();
        Arrays.sort(magazine1);
        Arrays.sort(ransomNote1);
        int index = 0;
        for (int i = 0; i < magazine1.length; i++) {
            if (ransomNote1[index] == magazine1[i]) {
                index++;
                if (index == ransomNote1.length)
                    return true;
            }
        }
        return false;
    }
}
