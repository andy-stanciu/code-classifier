class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] ransome = ransomNote.toCharArray();
        List<Character> megalist = new ArrayList<Character>();
        for (int i = 0; i < magazine.length(); i++) {
            megalist.add(magazine.charAt(i));
        }
        for (char ch : ransome) {
            if (megalist.contains(ch)) {
                int index = megalist.indexOf(ch);
                megalist.remove(index);
            } else {
                return false;
            }
        }
        return true;
    }
}
