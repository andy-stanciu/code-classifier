class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // With Array
        int[] arr = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            int val = magazine.charAt(i) - 97;
            arr[val] = ++arr[val];
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            int val = ransomNote.charAt(i) - 97;
            if (arr[val] == 0) {
                return false;
            } else {
                arr[val] = --arr[val];
            }
        }
        return true;
