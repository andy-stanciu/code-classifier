class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> note = new HashMap<>();
        Map<Character, Integer> mag = new HashMap<>();
        char[] noteArr = ransomNote.toCharArray();
        char[] magArr = magazine.toCharArray();
        for (char i : noteArr) {
            note.put(i, note.getOrDefault(i, 0) + 1);
        }
        for (char i : magArr) {
            mag.put(i, mag.getOrDefault(i, 0) + 1);
        }
        for (Map.Entry<Character, Integer> entry : note.entrySet()) {
            Character key = entry.getKey();
            Integer value = entry.getValue();
            // check if magazine has letters required for creating note and the count of letters >= the ones in note, if not return false
            if (!mag.containsKey(key) || !(mag.get(key) >= value)) {
                return false;
            }
        }
        return true;
    }
}
