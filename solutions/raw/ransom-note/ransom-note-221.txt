class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<String, Integer> magazineMap = new HashMap<>();
        Map<String, Integer> ransomNoteMap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            String key = String.valueOf(magazine.charAt(i));
            magazineMap.put(key, magazineMap.getOrDefault(key, 0) + 1);
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            String key = String.valueOf(ransomNote.charAt(i));
            ransomNoteMap.put(key, ransomNoteMap.getOrDefault(key, 0) + 1);
        }
        for (Map.Entry<String, Integer> ransomNoteEntry : ransomNoteMap.entrySet()) {
            if (ransomNoteEntry.getValue() > 1) {
                if (!magazineMap.containsKey(ransomNoteEntry.getKey()) || magazineMap.get(ransomNoteEntry.getKey()) < ransomNoteEntry.getValue())
                    return false;
            } else {
                if (!magazineMap.containsKey(ransomNoteEntry.getKey())) {
                    return false;
                }
            }
        }
        return true;
    }
}
