class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // a -> 0, b -> 1, ...
        int[] noteChars = new int[26];
        for (int i = 0; i < ransomNote.length(); i++) {
            noteChars[ransomNote.charAt(i) - 97] += 1;
        }
        for (int j = 0; j < magazine.length(); j++) {
            noteChars[magazine.charAt(j) - 97] -= 1;
        }
        for (int k = 0; k < noteChars.length; k++) {
            if (noteChars[k] > 0) {
                return false;
            }
        }
        return true;
    }
}
