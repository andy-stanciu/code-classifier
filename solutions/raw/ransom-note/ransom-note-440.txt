class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        for (int i = 0; i < ransomNote.length(); i++) {
            int idx = magazine.indexOf(ransomNote.charAt(i));
            if (idx >= 0) {
                magazine = magazine.substring(0, idx) + magazine.substring(idx + 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
