class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        String ransomTemp = ransomNote;
        String magazineTemp = magazine;
        while (ransomTemp.length() > 0) {
            magazineTemp = hasThisChar(ransomTemp.charAt(0), magazineTemp);
            if (magazineTemp == null && ransomTemp.length() > 0) {
                return false;
            }
            ransomTemp = ransomTemp.substring(1);
        }
        return true;
    }

    private String hasThisChar(char charX, String currentMagazine) {
        String beginPart = "";
        String wordTemp = currentMagazine;
        while (wordTemp.length() > 0) {
            if (wordTemp.charAt(0) == charX) {
                return (beginPart + wordTemp.substring(1));
            }
            beginPart += wordTemp.charAt(0);
            wordTemp = wordTemp.substring(1);
        }
        return null;
    }
}
