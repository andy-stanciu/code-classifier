class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> magazineChars = new HashMap<>();
        for (var i = 0; i < magazine.length(); i++) {
            magazineChars.compute(magazine.charAt(i), (k, v) -> v == null ? 1 : v + 1);
        }
        for (var i = 0; i < ransomNote.length(); i++) {
            var c = ransomNote.charAt(i);
            var charCount = magazineChars.get(c);
            if (charCount == null || charCount == 0) {
                return false;
            }
            magazineChars.put(c, charCount - 1);
        }
        return true;
    }
}
