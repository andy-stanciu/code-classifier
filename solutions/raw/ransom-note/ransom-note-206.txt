class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> magazineChar = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            magazineChar.put(c, magazineChar.getOrDefault(c, 0) + 1);
        }
        for (char c : ransomNote.toCharArray()) {
            if (!magazineChar.containsKey(c) || magazineChar.get(c) == 0) {
                return false;
            }
            magazineChar.put(c, magazineChar.get(c) - 1);
        }
        return true;
    }
}
