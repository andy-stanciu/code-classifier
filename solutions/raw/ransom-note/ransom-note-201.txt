class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> rhm = new HashMap<>();
        HashMap<Character, Integer> mhm = new HashMap<>();
        char[] ran = ransomNote.toCharArray();
        char[] mag = magazine.toCharArray();
        for (char c : ran) {
            if (rhm.containsKey(c)) {
                rhm.put(c, rhm.get(c) + 1);
            } else {
                rhm.put(c, 1);
            }
        }
        for (char c : mag) {
            if (mhm.containsKey(c)) {
                mhm.put(c, mhm.get(c) + 1);
            } else {
                mhm.put(c, 1);
            }
        }
        for (int i = 0; i < ran.length; i++) {
            int x = 0;
            int y = 0;
            if (mhm.containsKey(ran[i])) {
                x = mhm.get(ran[i]);
                y = rhm.get(ran[i]);
                if (x < y) {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
