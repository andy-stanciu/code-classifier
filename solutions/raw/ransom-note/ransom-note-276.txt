class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int[] alphabet = new int[26];
        for (char c : ransomNote.toCharArray()) {
            int index = magazine.indexOf(c, alphabet[(int) c % 26]);
            if (index == -1)
                return false;
            alphabet[(int) c % 26] = index + 1;
        }
        return true;
    }
}
