class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length())
            return false;
        else if (ransomNote != null && magazine != null) {
            for (int i = 0; i < ransomNote.length(); i++) {
                char tempC = ransomNote.charAt(i);
                long char1Count = ransomNote.chars().filter(c -> c == tempC).count();
                long char2Count = magazine.chars().filter(c -> c == tempC).count();
                if (char1Count > char2Count)
                    return false;
            }
        }
        return true;
    }
}
