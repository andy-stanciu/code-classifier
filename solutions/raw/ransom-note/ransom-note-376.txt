class Solution {

    public boolean canConstruct(String r, String m) {
        StringBuilder sb = new StringBuilder(m);
        for (int i = 0; i < r.length(); i++) {
            int j = sb.indexOf("" + r.charAt(i));
            if (j != -1) {
                sb.delete(j, j + 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
