class Solution {

    public boolean canConstruct(String riansomNote, String magazine) {
        char[] rN = riansomNote.toCharArray();
        char[] mg = magazine.toCharArray();
        Arrays.sort(rN);
        Arrays.sort(mg);
        int a = 0;
        for (int i = 0; i < mg.length; i++) {
            if (a != rN.length) {
                if (rN[a] == mg[i]) {
                    a++;
                }
            }
        }
        if (a == rN.length) {
            return true;
        } else {
            return false;
        }
    }
}
