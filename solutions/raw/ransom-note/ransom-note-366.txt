class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        //put stuff into a map I guess and compare entries?
        String[] r = ransomNote.split("");
        String[] m = magazine.split("");
        //put the string arrays into a map
        HashMap<String, Integer> rMap = new HashMap<>();
        for (String x : r) {
            if (rMap.containsKey(x)) {
                rMap.put(x, rMap.get(x) + 1);
            } else {
                rMap.put(x, 1);
            }
        }
        //do for magazine
        HashMap<String, Integer> mMap = new HashMap<>();
        for (String x : m) {
            if (mMap.containsKey(x)) {
                mMap.put(x, mMap.get(x) + 1);
            } else {
                mMap.put(x, 1);
            }
        }
        for (Map.Entry<String, Integer> entry : rMap.entrySet()) {
            if (!mMap.containsKey(entry.getKey())) {
                return false;
            } else {
                //now the key exists in mMap, so check to see if the value is greater than or equal to the value of the key in rMap
                int mcount = mMap.get(entry.getKey());
                int rcount = entry.getValue();
                if (mcount < rcount) {
                    return false;
                }
            }
        }
        return true;
    }
}
