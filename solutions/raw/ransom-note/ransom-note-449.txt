class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> noteMap = mapCharacters(ransomNote);
        HashMap<Character, Integer> magazineMap = mapCharacters(magazine);
        for (Map.Entry<Character, Integer> entry : noteMap.entrySet()) {
            char character = entry.getKey();
            int characterCount = entry.getValue();
            if (!magazineMap.containsKey(character) || characterCount > magazineMap.get(character)) {
                return false;
            }
        }
        return true;
    }

    private HashMap<Character, Integer> mapCharacters(String document) {
        HashMap<Character, Integer> documentMap = new HashMap<>();
        char[] documentCharacters = document.toCharArray();
        for (int i = 0; i < documentCharacters.length; i++) {
            char character = documentCharacters[i];
            if (!documentMap.containsKey(character)) {
                documentMap.put(character, 1);
            } else {
                int characterCount = documentMap.get(character);
                characterCount++;
                documentMap.put(character, characterCount);
            }
        }
        return documentMap;
    }
}
