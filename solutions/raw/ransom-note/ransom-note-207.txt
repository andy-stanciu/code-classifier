class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> map = new HashMap<>();
        //Storing ransomNote values in map with their count
        for (char ch : ransomNote.toCharArray()) {
            map.put(ch, map.getOrDefault(ch, 0) + 1);
        }
        //checking the occurance and substracting them from the map
        for (char ch : magazine.toCharArray()) {
            map.put(ch, map.getOrDefault(ch, 0) - 1);
        }
        int count = 0;
        //if ransomNote values occure in magazine when values of that map be 0 or negative(occured may times in magazine)
        for (char ch : ransomNote.toCharArray()) {
            if (map.get(ch) <= 0) {
                //taking count to check all elements are occured not only single elment
                count++;
            }
        }
        System.out.println(map);
        //if count == ransomNote.length then all element is occured in magazine and outoput will be true
        if (count == ransomNote.length()) {
            return true;
        }
        return false;
    }
}
