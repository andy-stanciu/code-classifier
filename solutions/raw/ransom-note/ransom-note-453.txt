class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] ran = ransomNote.toCharArray();
        char[] mag = magazine.toCharArray();
        Arrays.sort(ran);
        Arrays.sort(mag);
        if (mag.length < ran.length) {
            return false;
        }
        Stack<Character> rann = new Stack<>();
        int i = 0;
        int j = 0;
        while (i < ran.length && j < mag.length) {
            if (ran[i] == mag[j]) {
                rann.push(ran[i]);
                i++;
            }
            j++;
        }
        if (rann.isEmpty() || i < ran.length) {
            return false;
        }
        while (!rann.isEmpty()) {
            char abb = rann.pop();
            if (ran[--i] != abb) {
                return false;
            }
        }
        return true;
    }
}
