class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char a;
        HashMap<Integer, Character> map = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            map.put(i, magazine.charAt(i));
        }
        int n, n1, k;
        k = 0;
        n = ransomNote.length();
        n1 = magazine.length();
        for (int j = 0; j < n; j++) {
            if (map.isEmpty()) {
                System.out.println("empty");
            }
            if (!(map.containsValue(ransomNote.charAt(j)))) {
                return false;
            } else {
                for (int l = 0; l < n1; l++) {
                    if (map.containsKey(l)) {
                        if ((map.get(l)) == (ransomNote.charAt(j))) {
                            map.remove(l);
                            break;
                        }
                    }
                }
            }
        }
        return true;
    }
}
