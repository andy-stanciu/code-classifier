class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // This is the recommended approach and it takes 18ms
        // Slower than the array approach **Bold**
        HashMap<Character, Integer> ht = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            // Similar to the array Approach by counting all the characters in the array ..
            char c = magazine.charAt(i);
            if (!ht.containsKey(c)) {
                ht.put(c, 1);
            } else {
                ht.put(c, ht.get(c) + 1);
            }
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            char c = ransomNote.charAt(i);
            if (ht.containsKey(c) && ht.get(c) > 0) {
                ht.put(c, ht.get(c) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
