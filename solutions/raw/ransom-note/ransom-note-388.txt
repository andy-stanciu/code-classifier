class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Long> ransome = new HashMap();
        for (int i = 0; i < ransomNote.length(); i++) {
            if (ransome.get(ransomNote.charAt(i)) != null) {
                ransome.put(ransomNote.charAt(i), ransome.get(ransomNote.charAt(i)) + 1);
            } else {
                ransome.put(ransomNote.charAt(i), 1L);
            }
        }
        HashMap<Character, Long> magazineMap = new HashMap();
        for (int i = 0; i < magazine.length(); i++) {
            if (magazineMap.get(magazine.charAt(i)) != null) {
                magazineMap.put(magazine.charAt(i), magazineMap.get(magazine.charAt(i)) + 1);
            } else {
                magazineMap.put(magazine.charAt(i), 1L);
            }
        }
        boolean result = true;
        for (Map.Entry<Character, Long> characterLongEntry : ransome.entrySet()) {
            Character key = characterLongEntry.getKey();
            Long value = characterLongEntry.getValue();
            if (magazineMap.get(key) == null || magazineMap.get(key) < value) {
                result = false;
            }
        }
        return result;
    }
}
