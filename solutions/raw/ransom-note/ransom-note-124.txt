class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        StringBuilder output = new StringBuilder();
        for (int i = 0; i < ransomNote.length(); i++) {
            if (magazine.contains(ransomNote.charAt(i) + "")) {
                String magazineChar = magazine.charAt(magazine.indexOf(ransomNote.charAt(i))) + "";
                output.append(magazineChar);
                magazine = magazine.replaceFirst(magazineChar, "");
            }
        }
        return output.toString().equals(ransomNote);
    }
}
