class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> magazineLetters = new HashMap<>();
        fillLetterMap(magazineLetters, magazine);
        return validateNote(magazineLetters, ransomNote);
    }

    private void populateLetterMap(Map<Character, Integer> map, String str) {
        for (char c : str.toCharArray()) {
            if (map.containsKey(c)) {
                map.put(c, map.get(c) + 1);
            } else {
                map.put(c, 1);
            }
        }
    }

    private boolean validateNote(Map<Character, Integer> map, String ransomNote) {
        for (char c : ransomNote.toCharArray()) {
            if (map.containsKey(c) && map.get(c) > 0) {
                map.put(c, map.get(c) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
