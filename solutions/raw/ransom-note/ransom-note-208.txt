class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> maps = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            maps.put(magazine.charAt(i), maps.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            if (maps.containsKey(ransomNote.charAt(i)) && maps.get(ransomNote.charAt(i)) > 0)
                maps.put(ransomNote.charAt(i), maps.get(ransomNote.charAt(i)) - 1);
            else {
                System.out.println(ransomNote.charAt(i) + "" + i);
                return false;
            }
        }
        return true;
    }
}
