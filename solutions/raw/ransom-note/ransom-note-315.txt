class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> letterToFreqMagMap = new HashMap<Character, Integer>();
        for (int i = 0; i < magazine.length(); i++) {
            char c = magazine.charAt(i);
            if (letterToFreqMagMap.containsKey(c)) {
                letterToFreqMagMap.put(c, letterToFreqMagMap.get(c) + 1);
            } else {
                letterToFreqMagMap.put(c, 1);
            }
        }
        HashMap<Character, Integer> letterToFreqNoteMap = new HashMap<Character, Integer>();
        for (int i = 0; i < ransomNote.length(); i++) {
            char c = ransomNote.charAt(i);
            if (letterToFreqNoteMap.containsKey(c)) {
                letterToFreqNoteMap.put(c, letterToFreqNoteMap.get(c) + 1);
            } else {
                letterToFreqNoteMap.put(c, 1);
            }
        }
        for (Map.Entry<Character, Integer> entry : letterToFreqNoteMap.entrySet()) {
            if (!(letterToFreqMagMap.containsKey(entry.getKey()) && entry.getValue() <= letterToFreqMagMap.get(entry.getKey()))) {
                return false;
            }
        }
        return true;
    }
}
