class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        boolean flag = false;
        for (int i = 0; i < ransomNote.length(); i++) {
            String str = String.valueOf(ransomNote.charAt(i));
            if (magazine.contains(str)) {
                flag = true;
                int drop = magazine.indexOf(ransomNote.charAt(i));
                String mag1 = "";
                if (drop == 1) {
                    mag1 = String.valueOf(magazine.charAt(0));
                } else if (drop == 0) {
                    mag1 = "";
                } else {
                    mag1 = magazine.substring(0, drop);
                }
                String mag2 = "";
                if ((drop + 1) == (magazine.length() + 1)) {
                    mag2 = String.valueOf(magazine.charAt(magazine.length() - 1));
                } else if ((drop + 1) < magazine.length()) {
                    mag2 = magazine.substring((drop + 1), magazine.length());
                } else {
                    mag2 = "";
                }
                String resmag = mag1 + mag2;
                magazine = resmag;
            } else {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
