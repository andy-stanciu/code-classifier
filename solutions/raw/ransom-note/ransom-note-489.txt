class Solution {

    public boolean canConstruct(String s, String t) {
        char[] a = s.replaceAll(" ", "").toCharArray();
        char[] b = t.replaceAll(" ", "").toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        if (t.length() < s.length())
            return false;
        int i = 0;
        for (int j = 0; j < b.length; j++) {
            if (a[i] == b[j]) {
                i++;
                if (i >= a.length) {
                    return true;
                }
            }
        }
        return false;
    }
}
