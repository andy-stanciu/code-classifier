class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote == null || magazine == null) {
            return false;
        }
        if (ransomNote.length() > magazine.length()) {
            return false;
        }
        char[] ransomArray = ransomNote.toCharArray();
        char[] magazineArray = magazine.toCharArray();
        Map<Character, Integer> magazineMap = new HashMap();
        for (int i = 0; i < magazineArray.length; i++) {
            magazineMap.put(magazineArray[i], magazineMap.getOrDefault(magazineArray[i], 0) + 1);
        }
        for (int i = 0; i < ransomArray.length; i++) {
            if (magazineMap.get(ransomArray[i]) == null || magazineMap.get(ransomArray[i]) == 0) {
                return false;
            } else {
                magazineMap.put(ransomArray[i], magazineMap.get(ransomArray[i]) - 1);
            }
        }
        return true;
    }
}
