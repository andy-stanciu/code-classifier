class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int lenRansom = ransomNote.length(), lenMaga = magazine.length();
        HashMap<Character, Integer> mapRansom = new HashMap<>();
        HashMap<Character, Integer> mapMaga = new HashMap<>();
        for (int i = 0; i < lenRansom; i++) {
            mapRansom.put(ransomNote.charAt(i), mapRansom.getOrDefault(ransomNote.charAt(i), 0) + 1);
        }
        for (int j = 0; j < lenMaga; j++) {
            mapMaga.put(magazine.charAt(j), mapMaga.getOrDefault(magazine.charAt(j), 0) + 1);
        }
        for (Map.Entry<Character, Integer> element : mapRansom.entrySet()) {
            if (mapMaga.containsKey(element.getKey())) {
                if (mapMaga.get(element.getKey()) >= element.getValue()) {
                    continue;
                } else {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
