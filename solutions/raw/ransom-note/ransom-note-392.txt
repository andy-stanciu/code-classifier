class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        List<Character> ransomNote_list = new ArrayList<>();
        List<Character> magazine_list = new ArrayList<>();
        boolean flag = false;
        for (char char_ran : ransomNote.toCharArray()) ransomNote_list.add(char_ran);
        for (char char_mag : magazine.toCharArray()) magazine_list.add(char_mag);
        for (int i = 0; i < magazine_list.size(); i++) {
            char check = magazine_list.get(i);
            int index_ran = ransomNote_list.indexOf(check);
            if (ransomNote_list.contains(check)) {
                ransomNote_list.remove(index_ran);
            }
        }
        return ransomNote_list.isEmpty();
    }
}
