class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() == 1 && magazine.length() == 1)
            return ransomNote.equalsIgnoreCase(magazine);
        HashMap<Character, Integer> letters = new HashMap<Character, Integer>();
        //Building the letters set O(n)
        for (int i = 0; i < magazine.length(); i++) {
            char l = magazine.charAt(i);
            letters.put(l, letters.getOrDefault(l, 0) + 1);
        }
        //going over ransomenote, and check if every letter exists in the letters map
        for (int i = 0; i < ransomNote.length(); i++) {
            char l = ransomNote.charAt(i);
            if (!letters.containsKey(l)) {
                return false;
            }
            int lettersCount = letters.get(l);
            lettersCount--;
            if (lettersCount < 0) {
                return false;
            }
            letters.put(l, lettersCount);
        }
        return true;
    }
}
