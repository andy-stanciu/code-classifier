class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        String x = ransomNote;
        String y = magazine;
        boolean z = false;
        for (int i = 0; i < x.length(); i++) {
            if (y.contains(x.substring(i, i + 1))) {
                z = true;
                int r = y.indexOf(x.substring(i, i + 1));
                y = y.substring(0, r) + y.substring(r + 1);
            } else {
                z = false;
                break;
            }
        }
        return z;
    }
}
