class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length())
            return false;
        Map<Character, Long> ransomNoteCharMap = ransomNote.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        Map<Character, Long> magazineCharMap = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        return ransomNoteCharMap.entrySet().stream().allMatch(e -> e.getValue() <= magazineCharMap.getOrDefault(e.getKey(), 0L));
    }
}
