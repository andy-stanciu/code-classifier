class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> ransomFreq = new HashMap();
        Map<Character, Integer> magazineFreq = new HashMap();
        for (int i = 0; i < ransomNote.length(); ++i) {
            ransomFreq.merge(ransomNote.charAt(i), 1, Integer::sum);
        }
        for (int i = 0; i < magazine.length(); ++i) {
            magazineFreq.merge(magazine.charAt(i), 1, Integer::sum);
        }
        for (int x = 0; x < ransomNote.length(); ++x) {
            if (!magazineFreq.containsKey(ransomNote.charAt(x)) || (magazineFreq.containsKey(ransomNote.charAt(x)) && magazineFreq.get(ransomNote.charAt(x)) < ransomFreq.get(ransomNote.charAt(x)))) {
                return false;
            }
        }
        return true;
    }
}
