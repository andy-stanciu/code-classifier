class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> hm1 = new HashMap<>();
        HashMap<Character, Integer> hm = new HashMap<>();
        for (int i = 0; i < ransomNote.length(); i++) {
            char pc = ransomNote.charAt(i);
            hm.put(pc, hm.getOrDefault(pc, 0) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            char cp = magazine.charAt(i);
            hm1.put(cp, hm1.getOrDefault(cp, 0) + 1);
        }
        for (int k = 0; k < ransomNote.length(); k++) {
            char ch = ransomNote.charAt(k);
            if (!hm1.containsKey(ch) || hm1.get(ch) < hm.get(ch)) {
                return false;
            }
        }
        return true;
    }
}
