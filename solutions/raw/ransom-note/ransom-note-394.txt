class Solution {

    public static boolean canConstruct(String ransomNote, String magazine) {
        if (magazine.length() < ransomNote.length())
            return false;
        int[] checkArray = new int[123];
        for (char ch : magazine.toCharArray()) {
            checkArray[ch]++;
        }
        for (char ch : ransomNote.toCharArray()) {
            --checkArray[ch];
            if (checkArray[ch] < 0)
                return false;
        }
        return true;
    }
}
