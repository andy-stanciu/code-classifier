class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> k = new HashMap<>();
        boolean returnb = true;
        for (int i = 0; i < magazine.length(); i++) {
            char g = magazine.charAt(i);
            if (!k.containsKey(g)) {
                k.put(g, 1);
            } else {
                int count = k.get(g);
                k.put(g, count + 1);
            }
        }
        System.out.println(k);
        for (int j = 0; j < ransomNote.length(); j++) {
            if (k.containsKey(ransomNote.charAt(j)) && k.get(ransomNote.charAt(j)) > 0) {
                int count = k.get(ransomNote.charAt(j));
                k.put(ransomNote.charAt(j), count - 1);
            } else {
                returnb = false;
            }
        }
        return returnb;
    }
}
