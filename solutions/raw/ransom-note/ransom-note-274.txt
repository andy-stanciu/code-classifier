class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> hp2 = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            if (hp2.get(magazine.charAt(i)) == null) {
                hp2.put(magazine.charAt(i), 1);
            } else {
                hp2.put(magazine.charAt(i), hp2.get(magazine.charAt(i)) + 1);
            }
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            if (hp2.get(ransomNote.charAt(i)) == null || hp2.get(ransomNote.charAt(i)) == 0) {
                return false;
            } else {
                hp2.put(ransomNote.charAt(i), hp2.get(ransomNote.charAt(i)) - 1);
            }
        }
        return true;
    }
}
