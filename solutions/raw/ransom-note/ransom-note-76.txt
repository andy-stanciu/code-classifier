class Solution {

    public static boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length())
            return false;
        List<Character> charList = new java.util.ArrayList<>(magazine.chars().mapToObj(c -> (char) c).toList());
        boolean flag = true;
        for (Character c : ransomNote.toCharArray()) {
            if (charList.contains(c)) {
                charList.remove(c);
            } else {
                flag = false;
            }
        }
        return flag;
    }
}
