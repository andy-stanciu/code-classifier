class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Queue<Character> ransomQueue = new LinkedList();
        //add to queue
        for (Character c : ransomNote.toCharArray()) {
            ransomQueue.add(c);
        }
        System.out.println(ransomQueue);
        //now iterate over magazine, take one letter at a time and remove from queue if found,
        for (Character c : magazine.toCharArray()) {
            if (ransomQueue.contains(c)) {
                ransomQueue.remove(c);
            }
        }
        if (ransomQueue.isEmpty()) {
            return true;
        } else {
            return false;
        }
    }
}
