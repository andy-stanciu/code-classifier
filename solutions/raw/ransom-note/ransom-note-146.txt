class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // Frequency map for characters in ransomNote
        Map<Character, Integer> map = new HashMap<>();
        // Count frequencies of characters in ransomNote
        for (char c : ransomNote.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        // Decrease frequencies based on characters in magazine
        for (char c : magazine.toCharArray()) {
            if (map.containsKey(c)) {
                map.put(c, map.get(c) - 1);
                // If count reaches zero, remove it from map
                if (map.get(c) == 0) {
                    map.remove(c);
                }
            }
        }
        // Check if all counts in map are zero
        return map.isEmpty();
    }
}
