class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int i = 0;
        int[] m = new int[26];
        for (char c : ransomNote.toCharArray()) {
            i = magazine.indexOf(c, m[c % 26]);
            if (i == -1)
                return false;
            m[c % 26] = i + 1;
        }
        return true;
    }
}
