class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> map = new HashMap();
        if (magazine.length() < ransomNote.length()) {
            return false;
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (map.containsKey(magazine.charAt(i))) {
                map.replace(magazine.charAt(i), map.get(magazine.charAt(i)), map.get(magazine.charAt(i)) + 1);
            } else {
                map.put(magazine.charAt(i), 1);
            }
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            if (!map.containsKey(ransomNote.charAt(i))) {
                return false;
            } else {
                map.replace(ransomNote.charAt(i), map.get(ransomNote.charAt(i)), map.get(ransomNote.charAt(i)) - 1);
                if (map.get(ransomNote.charAt(i)) == -1) {
                    return false;
                }
            }
        }
        return true;
    }
}
