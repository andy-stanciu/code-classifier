class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Long> collect = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        //Map<Character, Integer> collect = new HashMap<>();
        // for (char c : magazine.toCharArray()) {
        //       collect.put(c, collect.getOrDefault(c, 0) + 1);
        //   }
        for (char c : ransomNote.toCharArray()) {
            if (!collect.containsKey(c) || collect.get(c) <= 0) {
                return false;
            }
            collect.put(c, (collect.get(c) - 1));
        }
        return true;
    }
}
