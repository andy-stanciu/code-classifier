class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        //ransomNote = "aa", magazine = "aab"
        char[] ransomNotechar = ransomNote.toCharArray();
        char[] magazinechar = magazine.toCharArray();
        Map<Character, Integer> map1 = new HashMap<>();
        Map<Character, Integer> map2 = new HashMap<>();
        for (char ch1 : ransomNotechar) {
            map1.put(ch1, map1.getOrDefault(ch1, 0) + 1);
        }
        for (char ch2 : magazinechar) {
            map2.put(ch2, map2.getOrDefault(ch2, 0) + 1);
        }
        for (char ch3 : map1.keySet()) {
            if (map2.containsKey(ch3)) {
                int op = Integer.compare(map1.get(ch3), map2.get(ch3));
                if (op > 0) {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
