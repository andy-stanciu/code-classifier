class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> magazine_map = new HashMap<>();
        for (final char c : magazine.toCharArray()) {
            if (magazine_map.containsKey(c))
                magazine_map.put(c, magazine_map.get(c) + 1);
            else
                magazine_map.put(c, 1);
        }
        for (final char c : ransomNote.toCharArray()) {
            if (!magazine_map.containsKey(c) || magazine_map.get(c) == 0)
                return false;
            magazine_map.put(c, magazine_map.get(c) - 1);
        }
        return true;
    }
}
