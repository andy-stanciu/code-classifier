class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> magazine1 = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char currentChar = magazine.charAt(i);
            magazine1.put(currentChar, magazine1.getOrDefault(currentChar, 0) + 1);
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            char currentChar = ransomNote.charAt(i);
            if (!magazine1.containsKey(currentChar) || magazine1.get(currentChar) == 0) {
                return false;
            }
            magazine1.put(currentChar, magazine1.get(currentChar) - 1);
        }
        return true;
    }
}
