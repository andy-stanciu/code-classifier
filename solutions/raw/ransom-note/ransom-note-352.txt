class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Long> charToFreq = magazine.chars().mapToObj(codePoint -> (char) codePoint).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        return ransomNote.chars().mapToObj(codePoint -> (char) codePoint).allMatch(character -> charToFreq.merge(character, -1L, Long::sum) >= 0);
    }
}
