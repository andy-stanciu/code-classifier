class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int[] arr1 = new int[26];
        int[] arr2 = new int[26];
        for (char c : ransomNote.toCharArray()) {
            int index = c - 97;
            int val = arr1[index];
            arr1[index] = ++val;
        }
        for (char c : magazine.toCharArray()) {
            int index = c - 97;
            int val = arr2[index];
            arr2[index] = ++val;
        }
        for (int i = 0; i < 26; i++) {
            if (arr1[i] > 0 && arr1[i] > arr2[i]) {
                return false;
            }
        }
        return true;
    }
}
