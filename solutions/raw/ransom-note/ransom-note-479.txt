class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        // Approach1 - A simulation:
        // step-1: For each character c in the ransom note, find the index of the first occurance of c in the magazine
        // If there are none of c left in the magazine, i.e index == -1,  return false
        // step-2: once we check as c is within megazine, remove c from the megazine and create a new megazine, this is mainly due to the fact that we are not allowed to reuse characters
        for (char c : ransomNote.toCharArray()) {
            int index = magazine.indexOf(c);
            if (index == -1)
                return false;
            magazine = magazine.substring(0, index) + magazine.substring(index + 1);
        }
        return true;
    }
}
