class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int x = 0;
        char[] noteChars = ransomNote.toCharArray();
        while (x < ransomNote.length()) {
            char noteChar = noteChars[x];
            int index = magazine.indexOf(noteChar);
            if (index != -1) {
                // char is exist
                magazine = removeCharAtIndex(index, magazine);
            } else {
                // char is not exist
                return false;
            }
            x++;
        }
        return true;
    }

    private String removeCharAtIndex(int index, String magazine) {
        StringBuilder stringBuilder = new StringBuilder(magazine);
        stringBuilder.replace(index, index + 1, "$");
        return stringBuilder.toString();
    }
}
