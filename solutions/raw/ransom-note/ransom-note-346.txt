class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        if (ransomNote.length() > magazine.length())
            return false;
        Map<Character, Integer> availableCharacter = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            if (availableCharacter.containsKey(c))
                availableCharacter.put(c, availableCharacter.get(c) + 1);
            else
                availableCharacter.put(c, 1);
        }
        for (char c : ransomNote.toCharArray()) {
            if (availableCharacter.containsKey(c) && availableCharacter.get(c) > 0) {
                availableCharacter.put(c, availableCharacter.get(c) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
