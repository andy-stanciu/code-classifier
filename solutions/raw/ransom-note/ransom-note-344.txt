class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        boolean flag = true;
        StringBuilder magazineM = new StringBuilder(magazine);
        for (int i = 0; i < ransomNote.length(); i++) {
            int index = magazineM.indexOf(String.valueOf(ransomNote.charAt(i)));
            if (index >= 0) {
                magazineM.deleteCharAt(index);
            } else {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
