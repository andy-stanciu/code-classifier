class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        final var map = new int[26];
        for (var ch : magazine.toCharArray()) {
            map[Objects.hash(ch) % 26]++;
        }
        for (var ch : ransomNote.toCharArray()) {
            if ((map[Objects.hash(ch) % 26]--) == 0)
                return false;
        }
        return true;
    }
}
