class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> magazineLetterCount = new HashMap<>();
        if (magazine.length() == 0)
            return false;
        for (char c : magazine.toCharArray()) {
            if (magazineLetterCount.containsKey(c)) {
                magazineLetterCount.put(c, magazineLetterCount.get(c) + 1);
            } else {
                magazineLetterCount.put(c, 1);
            }
        }
        for (char c : ransomNote.toCharArray()) {
            if (!magazineLetterCount.containsKey(c)) {
                return false;
            } else {
                int count = magazineLetterCount.get(c);
                count--;
                if (count <= 0) {
                    magazineLetterCount.remove(c);
                } else {
                    magazineLetterCount.put(c, count);
                }
            }
        }
        return true;
    }
}
