class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        char[] arr = ransomNote.toCharArray();
        for (int c = 0; c < arr.length; c++) {
            if (!magazine.contains("" + arr[c]))
                return false;
            magazine = magazine.replaceFirst(("" + arr[c]), "");
        }
        System.gc();
        return true;
    }
}
