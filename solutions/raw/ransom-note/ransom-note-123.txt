class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        Map<Character, Integer> ransomMap = new HashMap<>();
        //update ransom map
        for (char r : ransomNote.toCharArray()) {
            ransomMap.put(r, ransomMap.getOrDefault(r, 0) + 1);
        }
        for (char m : magazine.toCharArray()) {
            if (ransomMap.containsKey(m)) {
                ransomMap.put(m, ransomMap.get(m) - 1);
                // remove if key has no value
                if (ransomMap.get(m) == 0) {
                    ransomMap.remove(m);
                }
            }
        }
        return ransomMap.isEmpty();
    }
}
