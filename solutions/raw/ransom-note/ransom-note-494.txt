class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        HashMap<Character, Integer> matchMake = new HashMap<>();
        /**
         * Here we populate the hashmap with the character as the key, and the amount
         *          of times that we see the character are the value
         */
        for (char c : magazine.toCharArray()) {
            matchMake.put(c, matchMake.getOrDefault(c, 0) + 1);
            //.getOrDefault works by checking if we even have that char as a key
            //if so use the current value associated with that key, then add one to it
            //or if there is no value currently,
            // set the value as 0 for that key, and then add 1 to it
        }
        for (char c : ransomNote.toCharArray()) {
            //if the key is not in the hash, or its value is less than 0 return false
            if (!matchMake.containsKey(c) || matchMake.get(c) <= 0) {
                return false;
            }
            matchMake.put(c, matchMake.get(c) - 1);
        }
        return true;
    }
}
