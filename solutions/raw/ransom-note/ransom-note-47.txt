class Solution {

    public boolean canConstruct(String ran, String magazine) {
        HashMap<Character, Integer> mag = new HashMap<Character, Integer>();
        putInMap(magazine, mag);
        for (int i = 0; i < ran.length(); i++) {
            char c = ran.charAt(i);
            if (!mag.containsKey(c)) {
                return false;
            } else {
                mag.put(c, mag.get(c) - 1);
                if (mag.get(c) == 0) {
                    mag.remove(c);
                }
            }
        }
        return true;
    }

    public void putInMap(String s, HashMap<Character, Integer> map) {
        for (int i = 0; i < s.length(); i++) {
            map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
        }
    }
}
