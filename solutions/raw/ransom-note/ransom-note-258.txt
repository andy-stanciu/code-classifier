class Solution {

    public boolean canConstruct(String ransomNote, String magazine) {
        int[] nums = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            nums[magazine.charAt(i) - 97]++;
        }
        for (int i = 0; i < ransomNote.length(); i++) {
            nums[ransomNote.charAt(i) - 97]--;
        }
        for (int i = 0; i < 26; i++) {
            if (nums[i] < 0)
                return false;
        }
        return true;
    }
}
