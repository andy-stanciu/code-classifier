class Solution {

    public boolean canConstruct(String r, String m) {
        // make array which is contains number of occurrence letters.
        // add occurrence a(0) to z(26)
        int[] a = new int[27];
        int[] b = new int[27];
        // add the occurrence letter on there position .
        for (int i = 0; i < r.length(); i++) {
            a[r.charAt(i) - 97] = a[r.charAt(i) - 97] + 1;
        }
        for (int i = 0; i < m.length(); i++) {
            b[m.charAt(i) - 97] = b[m.charAt(i) - 97] + 1;
        }
        // compare if number of letter in a letter is lesser than return false.
        for (int i = 0; i < r.length(); i++) {
            if (b[r.charAt(i) - 97] - a[r.charAt(i) - 97] < 0) {
                return false;
            }
        }
        return true;
    }
}
