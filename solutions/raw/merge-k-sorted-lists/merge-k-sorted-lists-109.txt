class Solution {

    public ListNode mergeKLists(ListNode[] lists) {
        if (lists.length == 0)
            return null;
        ListNode result = new ListNode(0);
        ListNode head = result;
        PriorityQueue<ListNode> pq = new PriorityQueue<>((l1, l2) -> Integer.compare(l1.val, l2.val));
        for (ListNode list : lists) {
            if (list != null)
                pq.add(list);
        }
        while (!pq.isEmpty()) {
            ListNode l = pq.poll();
            head.next = l;
            head = head.next;
            if (l.next != null)
                pq.add(l.next);
        }
        return result.next;
    }
}
