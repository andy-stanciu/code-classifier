/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    ListNode node = null;

    public ListNode mergeKLists(ListNode[] lists) {
        int ind = this.findTinyNode(lists);
        if (ind < 0)
            return null;
        ListNode node = lists[ind];
        lists[ind] = lists[ind].next;
        node.next = this.mergeKLists(lists);
        return node;
    }

    public int findTinyNode(ListNode[] lists) {
        int index = -1;
        ListNode temp = null;
        for (int i = 0; i < lists.length; i++) {
            if (lists[i] == null)
                continue;
            if (temp == null || temp.val > lists[i].val) {
                temp = lists[i];
                index = i;
            }
        }
        return index;
    }
}
