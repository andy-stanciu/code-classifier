class Solution {

    public ListNode mergeKLists(ListNode[] lists) {
        if (lists.length == 0 || lists == null) {
            return null;
        }
        if (lists.length == 1) {
            return lists[0];
        }
        ListNode answer = null;
        for (ListNode list : lists) {
            answer = merger(answer, list);
        }
        return answer;
    }

    private ListNode merger(ListNode list1, ListNode list2) {
        ListNode thing = new ListNode(0);
        ListNode curr = thing;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                curr.next = list1;
                list1 = list1.next;
                curr = curr.next;
            } else {
                curr.next = list2;
                list2 = list2.next;
                curr = curr.next;
            }
        }
        if (list1 != null) {
            curr.next = list1;
        }
        if (list2 != null) {
            curr.next = list2;
        }
        return thing.next;
    }
}
