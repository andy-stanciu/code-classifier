class Solution {

    public ListNode mergeKLists(ListNode[] lists) {
        if (lists == null) {
            return null;
        }
        if (lists.length == 1) {
            return lists[0];
        }
        PriorityQueue<ListNode> queue = new PriorityQueue<ListNode>((node1, node2) -> Integer.compare(node1.val, node2.val));
        for (ListNode node : lists) {
            while (node != null) {
                queue.offer(node);
                node = node.next;
            }
        }
        ListNode mergedLists = queue.poll();
        ListNode temp = mergedLists;
        while (!queue.isEmpty()) {
            ListNode listNode = queue.poll();
            temp.next = new ListNode(listNode.val);
            temp = temp.next;
        }
        return mergedLists;
    }
}
