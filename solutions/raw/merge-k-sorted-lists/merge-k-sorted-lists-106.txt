class Solution {

    public ListNode mergeKLists(ListNode[] lists) {
        if (lists.length == 0)
            return null;
        PriorityQueue<ListNode> heap = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode list : lists) {
            while (list != null) {
                heap.offer(list);
                list = list.next;
            }
        }
        ListNode head = heap.poll();
        ListNode temp = head;
        while (heap.size() > 0) {
            temp.next = heap.poll();
            temp = temp.next;
            temp.next = null;
        }
        return head;
    }
}
