class Solution {

    public ListNode mergeKLists(ListNode[] lists) {
        if (lists == null || lists.length == 0)
            return null;
        ListNode dummy = new ListNode(0), head = dummy;
        //counting total nodes
        int count = 0;
        for (ListNode list : lists) {
            ListNode currlist = list;
            while (currlist != null) {
                currlist = currlist.next;
                count++;
            }
        }
        //filling array with nodes values
        int[] arr = new int[count];
        int j = 0;
        for (ListNode list : lists) {
            while (list != null) {
                arr[j] = list.val;
                list = list.next;
                j++;
            }
        }
        //creating a sorted list from array values
        Arrays.sort(arr);
        for (int val : arr) {
            dummy.next = new ListNode(val);
            dummy = dummy.next;
        }
        return head.next;
    }
}
