class Solution {

    public ListNode mergeKLists(ListNode[] lists) {
        PriorityQueue<Integer> min = new PriorityQueue<>();
        for (int i = 0; i < lists.length; i++) {
            ListNode node = lists[i];
            while (node != null) {
                min.add(node.val);
                node = node.next;
            }
        }
        ListNode ans = new ListNode(-1);
        ListNode temp = ans;
        while (!min.isEmpty()) {
            int val = min.remove();
            ListNode node = new ListNode(val);
            temp.next = node;
            temp = temp.next;
        }
        return ans.next;
    }
}
