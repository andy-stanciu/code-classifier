class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int num : nums) {
            if (var1.size() < k) {
                var1.offer(num);
            } else if (num > var1.peek()) {
                var1.poll();
                var1.offer(num);
            }
        }
        return var1.peek();
    }
}
