class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            var1.offer(nums[i]);
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.size() == 0 ? 0 : var1.poll();
    }
}
