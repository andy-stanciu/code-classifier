class Solution {

    public int method1(int[] nums, int k) {
        int var1 = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (var1 > nums[i]) {
                var1 = nums[i];
            }
            if (max < nums[i]) {
                max = nums[i];
            }
        }
        int[] arr = new int[max - var1 + 1];
        for (int num : nums) {
            arr[num - var1]++;
        }
        int var2 = k;
        for (int j = arr.length - 1; j >= 0; j--) {
            var2 -= arr[j];
            if (var2 <= 0) {
                return j + var1;
            }
        }
        return -1;
    }
}
