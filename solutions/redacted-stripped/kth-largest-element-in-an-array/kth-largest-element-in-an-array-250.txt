class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            var1.add(nums[i]);
        }
        while (var1.size() > k) {
            var1.remove();
        }
        return var1.peek();
    }
}
