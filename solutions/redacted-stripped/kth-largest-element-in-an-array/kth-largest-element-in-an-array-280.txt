class Solution {

    public int method1(int[] nums, int k) {
        List<Integer> li = new ArrayList<>();
        for (int i : nums) {
            li.add(i);
        }
        PriorityQueue<Integer> pq = new PriorityQueue<>(li);
        int el = 0;
        while (pq.size() > k) {
            pq.poll();
        }
        return pq.peek();
    }
}
