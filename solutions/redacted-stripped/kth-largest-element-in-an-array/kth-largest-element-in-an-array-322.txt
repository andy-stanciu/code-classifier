class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>((a, b) -> b - a);
        for (int i : nums) {
            pq.add(i);
        }
        System.out.println(pq);
        while (k > 1) {
            pq.poll();
            k--;
        }
        return pq.peek();
    }
}
