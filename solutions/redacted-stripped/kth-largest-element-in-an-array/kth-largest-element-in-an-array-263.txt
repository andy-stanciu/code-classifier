class Solution {

    public int method1(int[] nums, int k) {
        int n = nums.length;
        PriorityQueue<Integer> var1 = new PriorityQueue<>((a, b) -> (b - a));
        for (int i = 0; i < n; i++) {
            var1.add(nums[i]);
        }
        while (k - 1 > 0) {
            var1.poll();
            k--;
        }
        return var1.peek();
    }
}
