class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            queue.add(nums[i]);
        }
        int var1 = nums.length - k;
        while (var1 > 0) {
            queue.poll();
            var1--;
        }
        return queue.peek();
    }
}
