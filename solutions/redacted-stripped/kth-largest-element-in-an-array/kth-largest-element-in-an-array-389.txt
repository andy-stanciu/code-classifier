class Solution {

    public int method1(int[] nums, int k) {
        Queue<Integer> var1 = new PriorityQueue<>();
        int var2 = 0;
        for (Integer e : nums) {
            var1.add(e);
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.poll();
    }
}
