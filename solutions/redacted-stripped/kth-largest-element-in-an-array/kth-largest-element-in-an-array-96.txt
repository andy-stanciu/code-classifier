class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            var1.add(nums[i]);
        }
        int head = var1.peek();
        for (int i = k; i < nums.length; i++) {
            if (nums[i] > head) {
                var1.poll();
                var1.offer(nums[i]);
                head = var1.peek();
            }
        }
        return var1.poll();
    }
}
