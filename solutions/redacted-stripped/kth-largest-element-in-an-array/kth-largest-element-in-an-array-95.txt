class Solution {

    public int method1(int[] arr, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < arr.length; i++) {
            if (var1.size() < k) {
                var1.add(arr[i]);
            } else {
                if (arr[i] > var1.peek()) {
                    var1.poll();
                    var1.add(arr[i]);
                }
            }
        }
        return var1.peek();
    }
}
