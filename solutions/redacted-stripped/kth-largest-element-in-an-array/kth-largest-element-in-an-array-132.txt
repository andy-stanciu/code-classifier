class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int t = k;
        int i = 0;
        while (t-- > 0) {
            pq.add(nums[i++]);
        }
        int var1 = 0;
        for (i = k; i < nums.length; i++) {
            var1 = pq.peek();
            if (var1 < nums[i]) {
                pq.poll();
                pq.add(nums[i]);
            }
        }
        return pq.peek();
    }
}
