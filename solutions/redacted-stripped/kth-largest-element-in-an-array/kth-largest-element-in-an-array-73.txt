class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (var num : nums) {
            if (var1.size() < k)
                var1.offer(num);
            else if (var1.peek() < num) {
                var1.poll();
                var1.offer(num);
            }
        }
        return var1.peek();
    }
}
