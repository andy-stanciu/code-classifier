class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            pq.add(nums[i]);
        }
        int f = k - 1;
        while (f > 0) {
            pq.poll();
            f--;
        }
        return pq.peek();
    }
}
