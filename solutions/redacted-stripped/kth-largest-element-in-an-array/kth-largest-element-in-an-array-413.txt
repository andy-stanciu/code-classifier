class Solution {

    public int method1(int[] nums, int k) {
        int BASE = 10000;
        int[] radix = new int[20001];
        for (int x : nums) {
            radix[x + BASE]++;
        }
        for (int i = 20000; i >= 0; i--) {
            k -= radix[i];
            if (k <= 0) {
                return i - BASE;
            }
        }
        return -1;
    }
}
