class Solution {

    public int method1(int[] nums, int k) {
        Queue<Integer> q = new PriorityQueue<Integer>(Collections.reverseOrder());
        for (int a : nums) {
            q.add(a);
        }
        while (k > 1) {
            int current = q.poll();
            k--;
        }
        return q.poll();
    }
}
