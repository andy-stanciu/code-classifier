class Solution {

    public int method1(int[] nums, int k) {
        int[] count = new int[20001];
        for (int i = 0; i < nums.length; i++) {
            count[nums[i] + 10000] += 1;
        }
        for (int i = 20000; i > -1; i--) {
            if (count[i] >= k) {
                return i - 10000;
            } else {
                k -= count[i];
            }
        }
        return -1;
    }
}
