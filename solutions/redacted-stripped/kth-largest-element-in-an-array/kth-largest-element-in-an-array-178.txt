class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Collections.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            maxHeap.add(nums[i]);
        }
        int var1 = 0;
        for (int i = 0; i < k; i++) {
            var1 = maxHeap.remove();
        }
        return var1;
    }
}
