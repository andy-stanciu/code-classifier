class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) pq.add(nums[i]);
        int var1 = nums.length - k;
        while (var1 > -1) {
            if (var1 == 0) {
                return pq.peek();
            }
            pq.poll();
            var1--;
        }
        return 0;
    }
}
