class Solution {

    public int method1(int[] nums, int k) {
        HashSet<Integer> set = new HashSet<>();
        for (int n : nums) {
            if (set.size() < k)
                set.add(n);
            else if (n > Collections.min(set)) {
                set.remove(Collections.min(set));
                set.add(n);
            }
        }
        return Collections.min(set);
    }
}
