class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>(Collections.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            pq.offer(nums[i]);
        }
        int i = 0;
        while (i < k - 1) {
            pq.poll();
            i++;
        }
        return pq.peek();
    }
}
