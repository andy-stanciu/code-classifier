class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> maxheap = new PriorityQueue<>(Collections.reverseOrder());
        for (int v : nums) {
            maxheap.add(v);
        }
        while (k != 1) {
            k--;
            maxheap.poll();
        }
        return maxheap.peek();
    }
}
