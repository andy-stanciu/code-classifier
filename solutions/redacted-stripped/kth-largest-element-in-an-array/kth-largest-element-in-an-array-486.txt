class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>(Comparator.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            q.add(nums[i]);
        }
        while (k - 1 != 0) {
            q.poll();
            k--;
        }
        return q.poll();
    }
}
