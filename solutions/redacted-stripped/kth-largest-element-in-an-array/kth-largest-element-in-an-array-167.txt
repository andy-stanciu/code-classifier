class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<Integer>((a, b) -> a - b);
        for (int i : nums) {
            var1.add(i);
            if (var1.size() > k)
                var1.poll();
        }
        return var1.poll();
    }
}
