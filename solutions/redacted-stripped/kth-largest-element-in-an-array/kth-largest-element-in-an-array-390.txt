class Solution {

    public int method1(int[] nums, int k) {
        Queue<Integer> maxHeap = new PriorityQueue<>();
        int var1 = 0;
        for (Integer e : nums) {
            maxHeap.add(-e);
        }
        int i = 1;
        while (i < k) {
            maxHeap.poll();
            i++;
        }
        return -1 * maxHeap.poll();
    }
}
