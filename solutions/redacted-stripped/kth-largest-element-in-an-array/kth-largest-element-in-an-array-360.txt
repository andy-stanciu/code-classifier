class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int n : nums) {
            if (pq.size() < k) {
                pq.add(n);
            } else if (n > pq.peek()) {
                pq.poll();
                pq.add(n);
            }
        }
        return pq.peek();
    }
}
