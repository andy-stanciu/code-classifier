class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> result = new PriorityQueue<>(Collections.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            result.add(nums[i]);
        }
        int length = 0;
        int num = 0;
        while (length < k) {
            num = result.poll();
            length++;
        }
        return num;
    }
}
