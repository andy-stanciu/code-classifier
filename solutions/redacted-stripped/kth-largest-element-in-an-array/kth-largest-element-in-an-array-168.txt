class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue();
        for (int i = 0; i < nums.length; i++) {
            if (i < k) {
                pq.add(nums[i]);
                continue;
            }
            int var1 = pq.peek();
            if (nums[i] > var1) {
                pq.poll();
                pq.add(nums[i]);
            }
        }
        return pq.peek();
    }
}
