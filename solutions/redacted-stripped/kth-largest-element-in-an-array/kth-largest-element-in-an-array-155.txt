class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>(k);
        for (int item : nums) {
            var1.offer(item);
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.peek();
    }
}
