class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        for (int i : nums) {
            pq.add(i);
        }
        while (k > 1) {
            pq.poll();
            --k;
        }
        return pq.poll();
    }
}
