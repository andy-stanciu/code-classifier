class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> set = new PriorityQueue<Integer>();
        for (int i = 0; i < k; i++) {
            set.add(nums[i]);
        }
        for (int i = k; i < nums.length; i++) {
            if (set.peek() < nums[i]) {
                set.poll();
                set.add(nums[i]);
            }
        }
        return set.peek();
    }
}
