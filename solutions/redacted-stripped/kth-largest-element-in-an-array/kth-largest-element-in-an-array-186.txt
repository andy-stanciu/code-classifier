class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> p1 = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            p1.add(nums[i]);
        }
        for (int i = 0; i < nums.length - k; i++) p1.poll();
        return p1.poll();
    }
}
