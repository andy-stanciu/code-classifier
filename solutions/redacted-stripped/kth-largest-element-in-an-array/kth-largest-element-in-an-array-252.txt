class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        int n = nums.length;
        for (int i = 0; i < k; i++) {
            var1.add(nums[i]);
        }
        for (int i = k; i < n; i++) {
            if (var1.peek() < nums[i]) {
                var1.poll();
                var1.add(nums[i]);
            }
        }
        return var1.peek();
    }
}
