class Solution {

    public int method1(int[] nums, int k) {
        Queue<Integer> queue = new PriorityQueue<>(Collections.reverseOrder());
        for (Integer n : nums) {
            queue.offer(n);
        }
        while (!queue.isEmpty()) {
            k -= 1;
            int n = queue.poll();
            if (k == 0) {
                return n;
            }
        }
        return -1;
    }
}
