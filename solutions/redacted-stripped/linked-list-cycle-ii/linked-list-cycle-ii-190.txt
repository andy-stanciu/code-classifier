public class Solution {

    public ListNode method1(ListNode head) {
        HashMap<ListNode, Integer> hash = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (hash.containsKey(temp)) {
                return temp;
            } else {
                hash.put(temp, 1);
            }
            temp = temp.next;
        }
        return null;
    }
}
