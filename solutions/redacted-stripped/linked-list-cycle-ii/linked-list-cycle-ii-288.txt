public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        ListNode t = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                while (t != slow) {
                    t = t.next;
                    slow = slow.next;
                }
                return slow;
            }
        }
        return null;
    }
}
