public class Solution {

    public ListNode method1(ListNode head) {
        ListNode fptr = head;
        ListNode sptr = head;
        while (fptr != null && fptr.next != null) {
            fptr = fptr.next.next;
            sptr = sptr.next;
            if (fptr == sptr) {
                sptr = head;
                while (sptr != fptr) {
                    fptr = fptr.next;
                    sptr = sptr.next;
                }
                return sptr;
            }
        }
        return null;
    }
}
