public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode slow = head;
        ListNode fast = method2(head);
        if (fast == null)
            return null;
        while (slow != fast) {
            slow = slow.next;
            fast = fast.next;
        }
        return slow;
    }

    public ListNode method2(ListNode head) {
        if (head == null)
            return null;
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                return slow;
            }
        }
        return null;
    }
}
