public class Solution {

    public ListNode method1(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        if (head == null) {
            return null;
        }
        while (fast != null) {
            if (slow == null)
                return null;
            slow = slow.next;
            if (fast.next == null)
                return null;
            fast = fast.next.next;
            if (fast == slow)
                break;
        }
        ListNode temp = head;
        while (temp != slow) {
            if (slow.next == null)
                return null;
            temp = temp.next;
            slow = slow.next;
        }
        return slow;
    }
}
