class Solution {

    public ListNode method1(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (slow == fast) {
                ListNode track = head;
                while (track != slow) {
                    track = track.next;
                    slow = slow.next;
                }
                return track;
            }
        }
        return null;
    }
}
