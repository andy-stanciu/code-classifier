public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        Set<ListNode> set = new HashSet<>();
        ListNode root = head;
        while (root.next != null) {
            if (set.contains(root.next))
                return root.next;
            else
                set.add(root);
            root = root.next;
        }
        return null;
    }
}
