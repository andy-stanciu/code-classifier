public class Solution {

    public ListNode method1(ListNode head) {
        HashSet<ListNode> result = new HashSet<>();
        ListNode temp = head;
        ListNode res = new ListNode(-1);
        while (temp != null) {
            if (result.contains(temp)) {
                return temp;
            } else {
                result.add(temp);
                temp = temp.next;
            }
        }
        return null;
    }
}
