public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (fast == slow)
                return firstNode(head, fast);
        }
        return null;
    }

    private ListNode firstNode(ListNode head, ListNode fast) {
        while (head != fast) {
            head = head.next;
            fast = fast.next;
        }
        return head;
    }
}
