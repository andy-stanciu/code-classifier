public class Solution {

    public ListNode method1(ListNode head) {
        int flag = 0;
        ListNode slow = head;
        ListNode fast = head;
        int pos = -1;
        if (head == null)
            return null;
        if (head.next == null)
            return null;
        while (slow.next != null) {
            if (slow == fast && pos != -1)
                break;
            slow = slow.next;
            if (fast == null || fast.next == null)
                return null;
            fast = fast.next.next;
            pos++;
        }
        if (slow != fast)
            return null;
        slow = head;
        while (slow != fast) {
            slow = slow.next;
            fast = fast.next;
        }
        return slow;
    }
}
