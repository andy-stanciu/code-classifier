public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                if (slow == head)
                    return head;
                slow = head;
                while (fast.next != slow.next) {
                    slow = slow.next;
                    fast = fast.next;
                }
                return slow.next;
            }
        }
        return null;
    }
}
