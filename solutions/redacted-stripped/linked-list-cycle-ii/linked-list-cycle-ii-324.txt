public class Solution {

    public ListNode method1(ListNode head) {
        Map<ListNode, Integer> map = new HashMap<>();
        ListNode curr = head;
        int i = 0;
        while (curr != null) {
            if (map.containsKey(curr))
                return curr;
            map.put(curr, i);
            curr = curr.next;
            i++;
        }
        return null;
    }
}
