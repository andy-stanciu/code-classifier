class Class1 {

    int val;

    Class1 next;

    Class1(int x) {
        val = x;
        next = null;
    }
}

public class Solution {

    public Class1 method1(Class1 head) {
        if (head == null || head.next == null) {
            return null;
        }
        Class1 slow = head;
        Class1 fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                break;
            }
        }
        if (fast == null || fast.next == null) {
            return null;
        }
        slow = head;
        while (slow != fast) {
            slow = slow.next;
            fast = fast.next;
        }
        return slow;
    }
}
