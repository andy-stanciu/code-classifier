public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                break;
            }
        }
        if (fast == null || fast.next == null)
            return null;
        while (head != slow) {
            slow = slow.next;
            head = head.next;
        }
        return head;
    }
}
