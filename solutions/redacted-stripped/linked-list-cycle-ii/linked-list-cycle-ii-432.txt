public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode slow = head;
        ListNode fast = head;
        while (fast.next != null && fast.next.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                ListNode s1 = head;
                while (s1 != slow) {
                    s1 = s1.next;
                    slow = slow.next;
                }
                return s1;
            }
        }
        return null;
    }
}
