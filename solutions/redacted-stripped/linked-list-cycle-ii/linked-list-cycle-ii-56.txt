public class Solution {

    public ListNode method1(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        int index = 0;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                index = method2(slow);
                break;
            }
        }
        if (index == 0)
            return null;
        ListNode f = head;
        ListNode s = head;
        while (index > 0) {
            s = s.next;
            index--;
        }
        while (f != s) {
            f = f.next;
            s = s.next;
        }
        return s;
    }

    public int method2(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                ListNode temp = slow;
                int len = 0;
                do {
                    temp = temp.next;
                    len++;
                } while (temp != slow);
                return len;
            }
        }
        return 0;
    }
}
