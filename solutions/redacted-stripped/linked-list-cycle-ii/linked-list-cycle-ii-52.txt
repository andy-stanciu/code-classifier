public class Solution {

    public ListNode method1(ListNode cabeza) {
        ListNode lento = cabeza;
        ListNode rapido = cabeza;
        while (rapido != null && rapido.next != null) {
            lento = lento.next;
            rapido = rapido.next.next;
            if (lento == rapido) {
                lento = cabeza;
                while (lento != rapido) {
                    lento = lento.next;
                    rapido = rapido.next;
                }
                return lento;
            }
        }
        return null;
    }
}
