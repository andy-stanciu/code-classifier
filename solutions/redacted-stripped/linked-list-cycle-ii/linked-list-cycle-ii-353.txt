public class Solution {

    public ListNode method1(ListNode head) {
        boolean ans = false;
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                ans = true;
                break;
            }
        }
        if (ans) {
            ListNode temp = head;
            while (temp != slow) {
                temp = temp.next;
                slow = slow.next;
            }
            return temp;
        } else
            return null;
    }
}
