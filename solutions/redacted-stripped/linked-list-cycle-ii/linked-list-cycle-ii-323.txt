public class Solution {

    public ListNode method1(ListNode head) {
        ListNode t = head;
        ListNode f = head;
        ListNode s = head;
        while (f != null && f.next != null) {
            s = s.next;
            f = f.next.next;
            if (f == s) {
                if (s == t)
                    return s;
                while (s != null && t != null) {
                    s = s.next;
                    t = t.next;
                    if (s == t)
                        return s;
                }
            }
        }
        return null;
    }
}
