public class Solution {

    public ListNode method1(ListNode head) {
        int l = countNodesinLoop(head);
        if (l == 0)
            return null;
        ListNode p1 = head;
        ListNode p2 = head;
        while (l != 0) {
            p1 = p1.next;
            l--;
        }
        while (p1 != p2) {
            p1 = p1.next;
            p2 = p2.next;
        }
        return p2;
    }

    int countNodesinLoop(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        int l = 0;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                do {
                    l++;
                    slow = slow.next;
                } while (slow != fast);
                return l;
            }
        }
        return l;
    }
}
