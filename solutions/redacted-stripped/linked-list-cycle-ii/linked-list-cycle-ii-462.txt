public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                break;
            }
        }
        if (slow == null || fast == null || fast.next == null)
            return null;
        while (slow != head) {
            slow = slow.next;
            head = head.next;
        }
        return slow;
    }
}
