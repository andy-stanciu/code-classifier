public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode rabbit = head;
        ListNode turtle = head;
        while (rabbit != null && rabbit.next != null) {
            rabbit = rabbit.next.next;
            turtle = turtle.next;
            if (rabbit == turtle) {
                break;
            }
        }
        turtle = head;
        if (rabbit == null || rabbit.next == null)
            return null;
        while (turtle != rabbit) {
            turtle = turtle.next;
            rabbit = rabbit.next;
        }
        return turtle;
    }
}
