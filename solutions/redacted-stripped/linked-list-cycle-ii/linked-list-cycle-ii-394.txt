public class Solution {

    public ListNode method1(ListNode head) {
        int len = 0;
        ListNode curr = head;
        while (curr != null) {
            int temp_len = 0;
            ListNode temp = head;
            while (temp != curr) {
                temp = temp.next;
                temp_len++;
            }
            if (temp_len != len)
                return curr;
            len++;
            curr = curr.next;
        }
        return null;
    }
}
