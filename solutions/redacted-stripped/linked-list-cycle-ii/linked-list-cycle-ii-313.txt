public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                return method2(slow, head);
            }
        }
        return null;
    }

    public ListNode method2(ListNode cyclicNode, ListNode start) {
        while (start != cyclicNode) {
            start = start.next;
            cyclicNode = cyclicNode.next;
        }
        return start;
    }
}
