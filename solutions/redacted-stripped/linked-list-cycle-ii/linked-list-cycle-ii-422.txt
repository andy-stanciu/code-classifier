public class Solution {

    public ListNode method1(ListNode head) {
        HashMap<ListNode, Integer> hashmap = new HashMap<>();
        ListNode temp = head;
        int m = 0;
        while (temp != null) {
            if (!hashmap.containsKey(temp)) {
                hashmap.put(temp, m);
                temp = temp.next;
            } else {
                break;
            }
        }
        return temp;
    }
}
