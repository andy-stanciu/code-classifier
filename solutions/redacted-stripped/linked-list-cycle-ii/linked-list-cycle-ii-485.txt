public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        boolean var1 = false;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                var1 = true;
                break;
            }
        }
        if (fast == null || fast.next == null) {
            return null;
        }
        slow = head;
        ListNode prev = null;
        while (slow != fast) {
            prev = fast;
            slow = slow.next;
            fast = fast.next;
        }
        return slow;
    }
}
