public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        ListNode fast = head;
        ListNode slow = head;
        ListNode first = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (fast == slow) {
                while (first != slow) {
                    first = first.next;
                    slow = slow.next;
                }
                return slow;
            }
        }
        return null;
    }
}
