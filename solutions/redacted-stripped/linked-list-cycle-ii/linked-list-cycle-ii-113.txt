public class Solution {

    public ListNode method1(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                ListNode newSlow = head;
                while (newSlow != slow) {
                    slow = slow.next;
                    newSlow = newSlow.next;
                }
                return slow;
            }
        }
        return null;
    }
}
