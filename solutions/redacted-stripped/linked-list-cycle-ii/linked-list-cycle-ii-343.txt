public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode low = head;
        ListNode high = head;
        while (high != null && high.next != null) {
            high = high.next.next;
            low = low.next;
            if (high == low) {
                high = head;
                while (low != high) {
                    low = low.next;
                    high = high.next;
                }
                return low;
            }
        }
        return null;
    }
}
