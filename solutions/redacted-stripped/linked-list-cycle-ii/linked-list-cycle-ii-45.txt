public class Solution {

    public ListNode method1(ListNode head) {
        int len = method2(head);
        if (len == 0)
            return null;
        ListNode f = head;
        ListNode s = head;
        while (len > 0) {
            s = s.next;
            len--;
        }
        while (s != f) {
            f = f.next;
            s = s.next;
        }
        return s;
    }

    public int method2(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (slow == fast) {
                int len = 0;
                ListNode temp = slow;
                do {
                    temp = temp.next;
                    len++;
                } while (temp != slow);
                return len;
            }
        }
        return 0;
    }
}
