public class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = method2(head);
        if (temp == null)
            return temp;
        while (true) {
            if (temp == head)
                return temp;
            head = head.next;
            temp = temp.next;
        }
        return null;
    }

    public ListNode method2(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (head != null) {
            if ((fast == null) || (fast.next == null && fast != null))
                return null;
            slow = slow.next;
            fast = fast.next.next;
            if (fast == slow)
                return slow;
        }
        return null;
    }
}
