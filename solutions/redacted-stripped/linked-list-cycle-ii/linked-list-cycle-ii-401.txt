public class Solution {

    public ListNode method1(ListNode head) {
        HashSet<ListNode> seenNodes = new HashSet<>();
        ListNode current = head;
        while (current != null) {
            if (seenNodes.contains(current)) {
                return current;
            }
            seenNodes.add(current);
            current = current.next;
        }
        return null;
    }
}
