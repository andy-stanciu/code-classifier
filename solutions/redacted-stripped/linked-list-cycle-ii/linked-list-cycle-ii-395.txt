public class Solution {

    public ListNode method1(ListNode head) {
        int k = method2(head);
        if (k == 0)
            return null;
        ListNode s = head;
        ListNode f = head;
        while (k > 0) {
            k--;
            s = s.next;
        }
        while (s != f) {
            s = s.next;
            f = f.next;
        }
        return s;
    }

    public int method2(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                int length = 0;
                do {
                    slow = slow.next;
                    length++;
                } while (slow != fast);
                return length;
            }
        }
        return 0;
    }
}
