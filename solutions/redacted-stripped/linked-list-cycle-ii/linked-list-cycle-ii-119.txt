public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode slow = head;
        ListNode fast = head;
        for (; fast != null && fast.next != null; ) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                break;
            }
        }
        if (fast == null || fast.next == null) {
            return null;
        }
        ListNode start = head;
        for (; start != slow; ) {
            start = start.next;
            slow = slow.next;
        }
        return slow;
    }
}
