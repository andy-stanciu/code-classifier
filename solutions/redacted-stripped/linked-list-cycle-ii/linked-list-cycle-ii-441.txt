public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                ListNode newslow = head;
                ListNode newfast = fast;
                while (newslow != newfast) {
                    newslow = newslow.next;
                    newfast = newfast.next;
                }
                return newslow;
            }
        }
        return null;
    }
}
