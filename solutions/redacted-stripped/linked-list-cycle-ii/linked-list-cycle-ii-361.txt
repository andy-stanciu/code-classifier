public class Solution {

    public ListNode method1(ListNode head) {
        ListNode f = head;
        ListNode s = head;
        while (f != null && f.next != null) {
            s = s.next;
            f = f.next.next;
            if (f == s) {
                ListNode t = head;
                while (t != s) {
                    s = s.next;
                    t = t.next;
                }
                return s;
            }
        }
        return null;
    }
}
