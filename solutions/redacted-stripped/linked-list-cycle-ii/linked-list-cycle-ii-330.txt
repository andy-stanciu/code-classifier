public class Solution {

    public ListNode method1(ListNode head) {
        int size = findLength(head);
        System.out.println("Size received from length method is " + size);
        if (size > 0) {
            ListNode first = head;
            ListNode second = head;
            for (int i = 0; i < size; i++) {
                second = second.next;
            }
            System.out.println("Current pos of second pointer is " + second.val);
            while (true) {
                if (first == second) {
                    return first;
                }
                first = first.next;
                second = second.next;
            }
        }
        return null;
    }

    private int findLength(ListNode head) {
        int length = 0;
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                System.out.println("Cycle found");
                do {
                    slow = slow.next;
                    length += 1;
                } while (slow != fast);
                System.out.println("Size - " + length);
                return length;
            }
        }
        return 0;
    }
}
