public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (slow == fast)
                break;
        }
        slow = head;
        if (fast == null || fast.next == null) {
            return null;
        } else {
            while (slow != fast) {
                slow = slow.next;
                fast = fast.next;
            }
        }
        return slow;
    }
}
