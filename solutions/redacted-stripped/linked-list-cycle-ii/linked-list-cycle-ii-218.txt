public class Solution {

    public ListNode method1(ListNode head) {
        HashSet<ListNode> set = new HashSet<>();
        ListNode temp = head;
        while (temp != null) {
            if (set.contains(temp.next)) {
                return temp.next;
            } else {
                set.add(temp);
            }
            temp = temp.next;
        }
        return null;
    }
}
