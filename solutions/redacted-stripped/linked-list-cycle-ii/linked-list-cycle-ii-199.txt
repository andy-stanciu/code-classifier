public class Solution {

    public ListNode method1(ListNode head) {
        HashMap<ListNode, Integer> mpp = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (mpp.containsKey(temp)) {
                return temp;
            }
            mpp.put(temp, 1);
            temp = temp.next;
        }
        return null;
    }
}
