class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        method2(n, n, "", result);
        return result;
    }

    private void method2(int leftCount, int rightCount, String current, List<String> result) {
        if (leftCount == 0 && rightCount == 0) {
            result.add(current);
            return;
        }
        if (leftCount > 0) {
            method2(leftCount - 1, rightCount, current + "(", result);
        }
        if (rightCount > leftCount) {
            method2(leftCount, rightCount - 1, current + ")", result);
        }
    }
}
