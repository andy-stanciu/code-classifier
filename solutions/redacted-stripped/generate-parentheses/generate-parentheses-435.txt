class Solution {

    List<String> result = new ArrayList<>();

    public List<String> method1(int n) {
        method1(n, n, "");
        return result;
    }

    public void method1(int open, int close, String s) {
        if (open == 0 && close == 0) {
            result.add(s);
            return;
        }
        if (open > 0)
            method1(open - 1, close, s + "(");
        if (close > open && close > 0)
            method1(open, close - 1, s + ")");
    }
}
