class Solution {

    public List<String> method1(int n) {
        List<String> list = new ArrayList<>();
        rec(0, 0, n, "", list);
        return list;
    }

    public static void rec(int o, int c, int n, String s, List<String> list) {
        if (o == c && c == n) {
            list.add(s);
            return;
        }
        if (o < n)
            rec(o + 1, c, n, s + "(", list);
        if (c < o)
            rec(o, c + 1, n, s + ")", list);
    }
}
