class Solution {

    List<String> res = new ArrayList<String>();

    public void backTrack(int open, int close, int n, String s) {
        if ((open == close) && (close == n)) {
            res.add(s);
        }
        if (open < n) {
            backTrack(open + 1, close, n, s + "(");
        }
        if (close < open) {
            backTrack(open, close + 1, n, s + ")");
        }
    }

    public List<String> method1(int n) {
        backTrack(0, 0, n, "");
        return res;
    }
}
