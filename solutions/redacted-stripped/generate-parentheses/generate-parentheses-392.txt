class Solution {

    public List<String> method1(int n) {
        List<String> list = new ArrayList<>();
        print(n, "", 0, 0, list);
        return list;
    }

    private static List<String> print(int n, String ans, int open, int close, List<String> list) {
        if (open == n && close == n) {
            list.add(ans);
        }
        if (open < n) {
            print(n, ans + "(", open + 1, close, list);
        }
        if (close < open) {
            print(n, ans + ")", open, close + 1, list);
        }
        return list;
    }
}
