class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        method2(result, "", n, n);
        return result;
    }

    public void method2(List<String> result, String s, int left, int right) {
        if (left == 0 && right == 0) {
            result.add(s);
            return;
        }
        if (left > 0) {
            method2(result, s + "(", left - 1, right);
        }
        if (right > left) {
            method2(result, s + ")", left, right - 1);
        }
    }
}
