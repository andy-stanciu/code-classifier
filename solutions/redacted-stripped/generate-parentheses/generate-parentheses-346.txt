class Solution {

    public void method1(List<String> param1, int n, int open, int close, StringBuilder str) {
        if (open == n && close == n) {
            param1.add(str.toString());
            return;
        }
        if (open < n) {
            method1(param1, n, open + 1, close, str.append("("));
            str.deleteCharAt(str.length() - 1);
        }
        if (close < open) {
            method1(param1, n, open, close + 1, str.append(")"));
            str.deleteCharAt(str.length() - 1);
        }
    }

    public List<String> method2(int n) {
        List<String> param1 = new ArrayList<>();
        method1(param1, n, 0, 0, new StringBuilder());
        return param1;
    }
}
