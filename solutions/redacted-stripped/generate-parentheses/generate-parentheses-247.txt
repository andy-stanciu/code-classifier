class Solution {

    List<String> ans = new ArrayList<>();

    public void backTrack(int n, int open, int close, String comb) {
        if (open == n && close == n) {
            ans.add(comb);
            return;
        }
        if (close < open) {
            backTrack(n, open, close + 1, comb + ")");
        }
        if (open < n) {
            backTrack(n, open + 1, close, comb + "(");
        }
    }

    public List<String> method1(int n) {
        List<String> answer = new ArrayList<>();
        int open = 0;
        int close = 0;
        backTrack(n, open, close, "");
        return ans;
    }
}
