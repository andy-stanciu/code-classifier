class Solution {

    public List<String> method1(int n) {
        List<String> res = new ArrayList<>();
        addParanthese("", 0, 0, n, res);
        return res;
    }

    public void addParanthese(String temp, int start, int end, int length, List<String> res) {
        if (length == start) {
            int n = temp.length();
            for (int i = n; i < length * 2; i++) {
                temp = temp + ")";
            }
            res.add(temp);
            return;
        }
        addParanthese(temp + "(", start + 1, end, length, res);
        if (start > 0 && start > end)
            addParanthese(temp + ")", start, end + 1, length, res);
    }
}
