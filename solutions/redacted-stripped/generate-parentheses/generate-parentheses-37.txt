class Solution {

    ArrayList<String> result = new ArrayList<>();

    public List<String> method1(int n) {
        method2(n, 0, 0, "");
        return result;
    }

    public void method2(int n, int open, int close, String str) {
        if (open == n && close == n) {
            result.add(str);
        }
        if (open < n) {
            method2(n, open + 1, close, str + "(");
        }
        if (close < open) {
            method2(n, open, close + 1, str + ")");
        }
    }
}
