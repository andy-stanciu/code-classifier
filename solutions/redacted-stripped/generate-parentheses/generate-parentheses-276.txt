class Solution {

    public List<String> method1(int n) {
        List<String> combinations = new ArrayList<>();
        helper(n, combinations, "", 0, 0);
        return combinations;
    }

    private void helper(int numParen, List<String> list, String accumulator, int numLeftParen, int numRightParen) {
        if (numLeftParen + numRightParen == numParen * 2) {
            list.add(accumulator);
            return;
        }
        if (numLeftParen < numParen) {
            helper(numParen, list, accumulator + "(", numLeftParen + 1, numRightParen);
        }
        if (numRightParen < numLeftParen) {
            helper(numParen, list, accumulator + ")", numLeftParen, numRightParen + 1);
        }
    }
}
