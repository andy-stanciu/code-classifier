class Solution {

    public List<String> method1(int n) {
        List<String> res = new ArrayList();
        genParen(n, "", 0, res);
        return res;
    }

    public void genParen(int n, String str, int open, List<String> res) {
        int strlen = str.length();
        if (open < 0 || strlen > (2 * n))
            return;
        if (open == 0 && strlen == (2 * n)) {
            res.add(str);
            return;
        }
        if (open < n)
            genParen(n, str + "(", open + 1, res);
        genParen(n, str + ")", open - 1, res);
    }
}
