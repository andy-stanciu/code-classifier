class Solution {

    public List<String> method1(int n) {
        var list = new ArrayList<String>();
        method2(list, new StringBuilder(), n, n);
        return list;
    }

    private void method2(List<String> list, StringBuilder seq, int opening, int closing) {
        if (opening == 0 && closing == 0) {
            list.add(seq.toString());
        }
        if (opening > 0) {
            seq.append("(");
            method2(list, seq, opening - 1, closing);
            seq.deleteCharAt(seq.length() - 1);
        }
        if (closing > opening) {
            seq.append(")");
            method2(list, seq, opening, closing - 1);
            seq.deleteCharAt(seq.length() - 1);
        }
    }
}
