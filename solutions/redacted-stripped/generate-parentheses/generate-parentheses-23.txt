class Solution {

    public List<String> method1(int n) {
        List<String> lt = new ArrayList<>();
        method2(lt, 0, 0, n, "");
        return lt;
    }

    public void method2(List<String> lt, int close, int open, int n, String ans) {
        if (close == n && open == n) {
            lt.add(ans);
            return;
        }
        if (open < n) {
            method2(lt, close, open + 1, n, ans + "(");
        }
        if (close < open) {
            method2(lt, close + 1, open, n, ans + ")");
        }
    }
}
