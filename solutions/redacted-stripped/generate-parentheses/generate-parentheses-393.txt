class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        int open = n - 1;
        int close = n;
        String output = "(";
        solve(open, close, output, result);
        return result;
    }

    void solve(int open, int close, String output, List<String> result) {
        if (open == 0 && close == 0) {
            result.add(output);
            return;
        }
        if (open != 0) {
            String output1 = output;
            output1 += "(";
            solve(open - 1, close, output1, result);
        }
        if (open < close) {
            String output2 = output;
            output2 += ")";
            solve(open, close - 1, output2, result);
        }
    }
}
