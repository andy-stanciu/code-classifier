class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        rec(n, n, new StringBuilder(), result);
        return result;
    }

    private void rec(int needOpen, int needClose, StringBuilder sb, List<String> result) {
        if (needOpen == 0 && needClose == 0) {
            result.add(sb.toString());
            return;
        }
        if (needClose > needOpen) {
            sb.append(")");
            rec(needOpen, needClose - 1, sb, result);
            sb.deleteCharAt(sb.length() - 1);
        }
        if (needOpen > 0) {
            sb.append("(");
            rec(needOpen - 1, needClose, sb, result);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
