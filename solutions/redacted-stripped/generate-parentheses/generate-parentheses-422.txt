class Solution {

    public List<String> method1(int n) {
        List<String> list = new ArrayList<>();
        Parenthesis(n, 0, 0, "", list);
        return list;
    }

    public static void Parenthesis(int n, int opening, int closing, String str, List<String> list) {
        if (opening == n && closing == n) {
            list.add(str);
            return;
        }
        if (opening < n) {
            Parenthesis(n, opening + 1, closing, str + "(", list);
        }
        if (closing < opening) {
            Parenthesis(n, opening, closing + 1, str + ")", list);
        }
    }
}
