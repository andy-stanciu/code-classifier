class Solution {

    List<String> res = new ArrayList<>();

    public List<String> method1(int n) {
        solve("", n, n);
        return this.res;
    }

    void solve(String ans, int open, int close) {
        if (open == 0 && close == 0) {
            this.res.add(ans);
            return;
        }
        if (open != 0) {
            solve(ans + "(", open - 1, close);
        }
        if (open < close) {
            solve(ans + ")", open, close - 1);
        }
    }
}
