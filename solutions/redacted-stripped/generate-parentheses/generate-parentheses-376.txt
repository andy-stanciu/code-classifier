class Solution {

    static void parenthesisHelper(String current, int open, int close, int n, List<String> result) {
        if (current.length() == 2 * n) {
            result.add(current);
            return;
        }
        if (open < n) {
            parenthesisHelper(current + "(", open + 1, close, n, result);
        }
        if (close < open) {
            parenthesisHelper(current + ")", open, close + 1, n, result);
        }
    }

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        parenthesisHelper("", 0, 0, n, result);
        return result;
    }
}
