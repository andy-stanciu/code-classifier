class Solution {

    static List<String> ansList;

    public List<String> method1(int n) {
        ansList = new ArrayList<>();
        method2("", n, n);
        return ansList;
    }

    private void method2(String op, int openBrackets, int closedBrackets) {
        if (openBrackets == 0 && closedBrackets == 0) {
            ansList.add(op);
            return;
        }
        if (openBrackets != 0) {
            method2(op + "(", openBrackets - 1, closedBrackets);
        }
        if (closedBrackets > openBrackets) {
            method2(op + ")", openBrackets, closedBrackets - 1);
        }
    }
}
