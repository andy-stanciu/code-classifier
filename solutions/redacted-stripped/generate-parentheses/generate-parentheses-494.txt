class Solution {

    List<String> res = new ArrayList<>();

    public List<String> method1(int n) {
        backTracking("", 0, 0, n);
        return res;
    }

    public void backTracking(String s, int cc, int co, int n) {
        if (cc == n && co == n) {
            res.add(s);
            return;
        }
        if (co < n)
            backTracking(s + "(", cc, co + 1, n);
        if (cc < co)
            backTracking(s + ")", cc + 1, co, n);
    }
}
