class Solution {

    private void solve(int i, int j, String str, List<String> list) {
        if (i == 0 && j == 0) {
            list.add(str);
            return;
        }
        if (i < 0 || j < 0 || i > j) {
            return;
        }
        solve(i - 1, j, str.concat("("), list);
        solve(i, j - 1, str.concat(")"), list);
    }

    public List<String> method1(int n) {
        List<String> res = new ArrayList<>();
        String str = "(";
        solve(n - 1, n, str, res);
        return res;
    }
}
