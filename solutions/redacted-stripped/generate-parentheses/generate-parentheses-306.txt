class Solution {

    private List<String> list = new ArrayList<>();

    public List<String> method1(int n) {
        genrateParenthesis(new String(), 0, 0, n);
        return list;
    }

    public void genrateParenthesis(String str, int open, int closed, int n) {
        if (n * 2 == str.length()) {
            list.add(str);
            return;
        }
        if (open < n) {
            genrateParenthesis(str + "(", open + 1, closed, n);
        }
        if (closed < open)
            genrateParenthesis(str + ")", open, closed + 1, n);
    }
}
