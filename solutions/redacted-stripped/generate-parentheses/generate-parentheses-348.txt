class Solution {

    public List<String> method1(int n) {
        List<String> strList = new ArrayList<String>();
        makeParComb(n, new StringBuilder(), 0, 0, strList);
        return strList;
    }

    public void makeParComb(int n, StringBuilder ans, int o, int c, List<String> strList) {
        if (ans.length() == n * 2) {
            strList.add(ans.toString());
            return;
        }
        if (o < n) {
            ans.append("(");
            makeParComb(n, ans, o + 1, c, strList);
            ans.deleteCharAt(ans.length() - 1);
        }
        if (c < o) {
            ans.append(")");
            makeParComb(n, ans, o, c + 1, strList);
            ans.deleteCharAt(ans.length() - 1);
        }
    }
}
