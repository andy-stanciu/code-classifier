class Solution {

    List<String> ans;

    public List<String> method1(int n) {
        ans = new ArrayList<>();
        method2(new StringBuilder(), 0, 0, n);
        return ans;
    }

    public void method2(StringBuilder sb, int open, int close, int max) {
        if (sb.length() == max * 2) {
            ans.add(sb.toString());
            return;
        }
        if (open < max) {
            sb.append("(");
            method2(sb, open + 1, close, max);
            sb.deleteCharAt(sb.length() - 1);
        }
        if (close < open) {
            sb.append(")");
            method2(sb, open, close + 1, max);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
