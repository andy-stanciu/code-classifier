class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        method2(2 * n, n, n, "", result);
        return result;
    }

    private void method2(int length, int left, int right, String currStr, List<String> result) {
        if (currStr.length() == length) {
            result.add(currStr);
            return;
        }
        if (left > 0)
            method2(length, left - 1, right, currStr + "(", result);
        if (right > left)
            method2(length, left, right - 1, currStr + ")", result);
    }
}
