class Solution {

    List<String> ans;

    void strgen(String p, int lp, int rp) {
        if (lp == 0 && rp == 0) {
            ans.add(p);
            return;
        }
        if (lp != 0) {
            strgen(p + "(", lp - 1, rp);
        }
        if (rp != 0 && rp > lp) {
            strgen(p + ")", lp, rp - 1);
        }
    }

    public List<String> method1(int n) {
        ans = new ArrayList<String>((int) Math.pow(2, n) - n);
        strgen("", n, n);
        return ans;
    }
}
