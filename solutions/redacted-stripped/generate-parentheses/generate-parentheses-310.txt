class Solution {

    public List<String> method1(int n) {
        List<String> combinations = new ArrayList<>();
        allCombinations(n, n, new StringBuilder(), combinations);
        return combinations;
    }

    private void allCombinations(int l, int r, final StringBuilder sb, List<String> combinations) {
        if (l == 0 && r == 0) {
            combinations.add(sb.toString());
            return;
        }
        if (l > 0) {
            sb.append("(");
            allCombinations(l - 1, r, sb, combinations);
            sb.deleteCharAt(sb.length() - 1);
        }
        if (l < r) {
            sb.append(")");
            allCombinations(l, r - 1, sb, combinations);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
