class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        simulate(n, 0, 0, new StringBuilder(), result);
        return result;
    }

    private void simulate(int n, int left, int right, StringBuilder str, List<String> result) {
        if (left == n && right == n) {
            result.add(str.toString());
            return;
        }
        if (left < n) {
            str.append("(");
            simulate(n, left + 1, right, str, result);
            str.deleteCharAt(str.length() - 1);
        }
        if (right < left) {
            str.append(")");
            simulate(n, left, right + 1, str, result);
            str.deleteCharAt(str.length() - 1);
        }
    }
}
