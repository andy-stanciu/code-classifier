class Solution {

    public List<String> method1(int n) {
        ArrayList<String> list = new ArrayList<>();
        getParanthesis(list, n, 0, 0, "");
        return list;
    }

    public void getParanthesis(ArrayList<String> list, int n, int open, int close, String str) {
        if (str.length() == n * 2) {
            list.add(str);
            return;
        }
        if (open < n) {
            getParanthesis(list, n, open + 1, close, str + "(");
        }
        if (close < open) {
            getParanthesis(list, n, open, close + 1, str + ")");
        }
    }
}
