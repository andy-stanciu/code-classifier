class Solution {

    public List<String> method1(int n) {
        List<String> outputList = new ArrayList();
        backtrack(outputList, "", 0, 0, n);
        return outputList;
    }

    public void backtrack(List<String> outputList, String currentString, int open, int close, int max) {
        if (currentString.length() == max * 2) {
            outputList.add(currentString);
            return;
        }
        if (open < max) {
            backtrack(outputList, currentString + "(", open + 1, close, max);
        }
        if (close < open) {
            backtrack(outputList, currentString + ")", open, close + 1, max);
        }
    }
}
