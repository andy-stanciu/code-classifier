class Solution {

    public List<String> method1(int n) {
        List<String> output = new ArrayList<>();
        permutationBalancedParenthesis(n, n, "", output);
        return output;
    }

    public void permutationBalancedParenthesis(int open, int close, String param1, List<String> output) {
        if (open == 0 && close == 0) {
            output.add(param1);
            return;
        }
        if (open > 0) {
            String openingBracket = param1 + "(";
            permutationBalancedParenthesis(open - 1, close, openingBracket, output);
        }
        if (close > open) {
            String closingBracket = param1 + ")";
            permutationBalancedParenthesis(open, close - 1, closingBracket, output);
        }
    }
}
