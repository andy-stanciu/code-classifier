class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        method2("", result, 0, 0, n);
        return result;
    }

    private static void method2(String current, List<String> result, int open, int close, int n) {
        if (current.length() == n * 2) {
            if (!result.contains(current))
                result.add(current);
            return;
        }
        if (open < n)
            method2(current + "(", result, open + 1, close, n);
        if (close < open)
            method2(current + ")", result, open, close + 1, n);
    }
}
