class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(int n) {
        genP(0, n, "");
        return ans;
    }

    public void genP(int sum, int avail, String res) {
        if (sum < 0 || avail < 0)
            return;
        if (avail == 0 && sum == 0) {
            ans.add(res);
            return;
        }
        genP(sum + 1, avail - 1, res + "(");
        genP(sum - 1, avail, res + ")");
        return;
    }
}
