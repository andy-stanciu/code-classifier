class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        backTrack(n, 0, 0, "", result);
        return result;
    }

    private void backTrack(int n, int openN, int closedN, String cur, List<String> result) {
        if (openN == n && closedN == n) {
            result.add(cur);
            return;
        }
        if (openN < n) {
            String s = cur.concat("(");
            backTrack(n, openN + 1, closedN, s, result);
        }
        if (closedN < openN) {
            String s = cur.concat(")");
            backTrack(n, openN, closedN + 1, s, result);
        }
    }
}
