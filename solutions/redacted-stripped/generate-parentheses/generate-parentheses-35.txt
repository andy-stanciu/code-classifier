class Solution {

    public List<String> method1(int n) {
        ArrayList<String> ans = new ArrayList<>();
        method2(ans, "", n, n);
        return ans;
    }

    public void method2(ArrayList ans, String s, int l, int r) {
        if (l == 0 && r == 0) {
            ans.add(s);
            return;
        }
        if (l > 0)
            method2(ans, s + "(", l - 1, r);
        if (r > l)
            method2(ans, s + ")", l, r - 1);
    }
}
