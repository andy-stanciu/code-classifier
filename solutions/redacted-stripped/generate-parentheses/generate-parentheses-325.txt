class Solution {

    static List<String> arr;

    public void print(int open, int close, int n, String ans) {
        if (ans.length() == 2 * n) {
            arr.add(ans);
            return;
        }
        if (open < n)
            print(open + 1, close, n, ans + "(");
        if (close < open)
            print(open, close + 1, n, ans + ")");
    }

    public List<String> method1(int n) {
        arr = new ArrayList<>();
        print(0, 0, n, "");
        return arr;
    }
}
