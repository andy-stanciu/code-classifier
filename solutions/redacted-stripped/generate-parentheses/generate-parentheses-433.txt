class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(int n) {
        getAns("", 0, 0, n);
        return ans;
    }

    public void getAns(String seq, int nOpen, int nClose, int n) {
        if (nClose > nOpen || nOpen > n) {
            return;
        } else if (nOpen == nClose && nOpen == n) {
            ans.add(seq);
        }
        getAns(seq + "(", nOpen + 1, nClose, n);
        getAns(seq + ")", nOpen, nClose + 1, n);
    }
}
