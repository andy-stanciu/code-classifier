class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.left == null && root.right == null)
                return null;
            else if (root.right != null) {
                TreeNode var2 = var2(root);
                root.val = var2.val;
                root.right = method1(root.right, var2.val);
                return root;
            } else {
                TreeNode var1 = var1(root);
                root.val = var1.val;
                root.left = method1(root.left, var1.val);
                return root;
            }
        } else {
            root.left = method1(root.left, key);
            root.right = method1(root.right, key);
            return root;
        }
    }

    private TreeNode var2(TreeNode root) {
        if (root.right != null) {
            TreeNode var3 = root.right;
            while (var3.left != null) {
                var3 = var3.left;
            }
            return var3;
        }
        return null;
    }

    private TreeNode var1(TreeNode root) {
        if (root.left != null) {
            TreeNode var3 = root.left;
            while (var3.right != null) {
                var3 = var3.right;
            }
            return var3;
        }
        return null;
    }
}
