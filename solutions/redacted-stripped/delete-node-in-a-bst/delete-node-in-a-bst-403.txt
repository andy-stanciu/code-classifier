class Solution {

    TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            if (root.right == null) {
                return root.left;
            } else if (root.left == null) {
                return root.right;
            }
            root.val = successor(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    int successor(TreeNode root) {
        if (root.left == null)
            return root.val;
        return successor(root.left);
    }

    public TreeNode method2(TreeNode root, int key) {
        root = method1(root, key);
        return root;
    }
}
