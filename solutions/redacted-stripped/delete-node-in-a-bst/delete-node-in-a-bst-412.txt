class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val)
            root.left = method1(root.left, key);
        else if (key > root.val)
            root.right = method1(root.right, key);
        else {
            if (root.left == null)
                return root.right;
            if (root.right == null)
                return root.left;
            root.val = method2(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    private int method2(TreeNode root) {
        int k = root.val;
        while (root.left != null) {
            k = root.left.val;
            root = root.left;
        }
        return k;
    }
}
