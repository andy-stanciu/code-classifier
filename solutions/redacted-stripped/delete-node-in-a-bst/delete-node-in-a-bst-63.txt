class Solution {

    boolean left = true;

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val)
            root.left = method1(root.left, key);
        if (key > root.val)
            root.right = method1(root.right, key);
        if (key == root.val) {
            if (root.left == null && root.right == null)
                return null;
            if (root.right == null)
                return root.left;
            else {
                TreeNode sub = root.right;
                TreeNode tmp = sub;
                while (sub.left != null) {
                    if (sub.left.left == null)
                        tmp = sub;
                    sub = sub.left;
                }
                tmp.left = sub.right;
                sub.left = root.left;
                if (sub != root.right)
                    sub.right = root.right;
                return sub;
            }
        }
        return root;
    }
}
