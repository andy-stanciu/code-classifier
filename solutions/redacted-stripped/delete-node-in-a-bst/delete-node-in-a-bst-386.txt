class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        else {
            root = del(root, key);
        }
        return root;
    }

    private TreeNode del(TreeNode r, int x) {
        if (r == null)
            return r;
        else {
            if (r.val > x)
                r.left = del(r.left, x);
            else if (r.val < x)
                r.right = del(r.right, x);
            else {
                if (r.right == null && r.left == null)
                    return null;
                else if (r.left == null && r.right != null)
                    return r.right;
                else if (r.right == null && r.left != null)
                    return r.left;
                else {
                    TreeNode c = r.right;
                    while (c.left != null) c = c.left;
                    r.val = c.val;
                    r.right = del(r.right, c.val);
                }
            }
        }
        return r;
    }
}
