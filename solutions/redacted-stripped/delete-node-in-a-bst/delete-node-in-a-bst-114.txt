class Solution {

    public TreeNode method1(TreeNode root, int key) {
        TreeNode temp = root;
        if (temp == null) {
            return null;
        }
        if (key < temp.val) {
            temp.left = method1(temp.left, key);
        } else if (key > temp.val) {
            temp.right = method1(temp.right, key);
        } else {
            if (temp.left == null && temp.right == null) {
                return null;
            } else if (temp.left == null) {
                temp = temp.right;
            } else if (temp.right == null) {
                temp = temp.left;
            } else {
                int var1 = method2(temp.right);
                temp.val = var1;
                temp.right = method1(temp.right, var1);
            }
        }
        return temp;
    }

    public int method2(TreeNode temp) {
        while (temp.left != null) {
            temp = temp.left;
        }
        return temp.val;
    }
}
