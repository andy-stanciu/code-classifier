class Solution {

    public TreeNode method1(TreeNode root) {
        TreeNode temp = root;
        while (temp.left != null) {
            temp = temp.left;
        }
        return temp;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return root;
        } else if (root.val == key) {
            if (root.left == null && root.right == null) {
                root = null;
                return null;
            }
            if (root.left != null && root.right == null) {
                TreeNode temp = root.left;
                root = null;
                return temp;
            }
            if (root.left == null && root.right != null) {
                TreeNode temp = root.right;
                root = null;
                return temp;
            }
            if (root.left != null && root.right != null) {
                int var1 = method1(root.right).val;
                root.val = var1;
                root.right = method2(root.right, var1);
                return root;
            }
        }
        if (root.val > key) {
            root.left = method2(root.left, key);
            return root;
        } else {
            root.right = method2(root.right, key);
            return root;
        }
    }
}
