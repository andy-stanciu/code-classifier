class Solution {

    private TreeNode method1(TreeNode root) {
        TreeNode curr = root;
        while (curr != null && curr.left != null) {
            curr = curr.left;
        }
        return curr;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key > root.val) {
            root.right = method2(root.right, key);
        } else if (key < root.val) {
            root.left = method2(root.left, key);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                TreeNode var1 = method1(root.right);
                root.val = var1.val;
                root.right = method2(root.right, var1.val);
            }
        }
        return root;
    }
}
