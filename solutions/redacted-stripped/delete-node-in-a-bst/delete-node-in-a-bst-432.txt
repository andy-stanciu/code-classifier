class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        }
        if (root.val > key) {
            root.left = method1(root.left, key);
        }
        if (root.val == key) {
            if (root.right == null && root.left == null) {
                return null;
            }
            if (root.left == null) {
                return root.right;
            }
            if (root.right == null) {
                return root.left;
            }
            int small = leftmost(root.right);
            root.val = small;
            root.right = method1(root.right, small);
        }
        return root;
    }

    public int leftmost(TreeNode root) {
        if (root.left == null) {
            return root.val;
        }
        return leftmost(root.left);
    }
}
