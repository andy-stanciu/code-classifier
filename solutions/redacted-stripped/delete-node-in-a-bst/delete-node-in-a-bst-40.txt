class Solution {

    public static int method1(TreeNode param1) {
        if (param1.left == null || param1 == null) {
            return param1.val;
        } else {
            return method1(param1.left);
        }
    }

    public TreeNode method2(TreeNode param1, int data) {
        if (param1 == null) {
            return null;
        }
        if (param1.val > data) {
            param1.left = method2(param1.left, data);
        } else if (param1.val < data) {
            param1.right = method2(param1.right, data);
        } else {
            if (param1.left == null && param1.right == null) {
                return null;
            } else if (param1.right == null) {
                return param1.left;
            } else if (param1.left == null) {
                return param1.right;
            } else {
                int var1 = method1(param1.right);
                param1.val = var1;
                param1.right = method2(param1.right, var1);
            }
        }
        return param1;
    }
}
