class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val > key) {
            root.left = method1(root.left, key);
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else {
            return method2(root.right, root.left);
        }
        return root;
    }

    private TreeNode method2(TreeNode param1, TreeNode param2) {
        if (param1 == null)
            return param2;
        if (param2 == null)
            return param1;
        TreeNode var1 = method3(param1);
        var1.left = param2;
        return param1;
    }

    private TreeNode method3(TreeNode param3) {
        while (param3.left != null) {
            param3 = param3.left;
        }
        return param3;
    }
}
