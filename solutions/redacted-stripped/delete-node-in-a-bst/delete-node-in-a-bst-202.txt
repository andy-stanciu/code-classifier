class Solution {

    public ArrayList<Integer> ar = new ArrayList<>();

    public void function(TreeNode root) {
        if (root == null)
            return;
        function(root.left);
        ar.add(root.val);
        function(root.right);
    }

    public TreeNode function2(int start, int end) {
        if (start > end)
            return null;
        int mid = (start + end) / 2;
        TreeNode root = new TreeNode(ar.get(mid));
        root.left = function2(start, mid - 1);
        root.right = function2(mid + 1, end);
        return root;
    }

    public TreeNode method1(TreeNode root, int key) {
        function(root);
        int var1 = ar.indexOf(key);
        if (var1 == -1)
            return root;
        ar.remove(var1);
        Collections.sort(ar);
        return function2(0, ar.size() - 1);
    }
}
