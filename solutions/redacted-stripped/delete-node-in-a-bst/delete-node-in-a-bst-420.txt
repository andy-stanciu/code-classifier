class Solution {

    public TreeNode method1(TreeNode root, int key) {
        TreeNode temp = del(root, key);
        return temp;
    }

    public TreeNode del(TreeNode root, int key) {
        if (root == null) {
            return null;
        } else if (key < root.val) {
            root.left = del(root.left, key);
        } else if (key > root.val) {
            root.right = del(root.right, key);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                TreeNode var1 = suc(root.right);
                root.val = var1.val;
                root.right = method1(root.right, var1.val);
            }
        }
        return root;
    }

    public TreeNode suc(TreeNode root) {
        TreeNode temp = root;
        while (temp.left != null) {
            temp = temp.left;
        }
        return temp;
    }
}
