class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key && (root.right == null && root.left == null)) {
            root = null;
            return root;
        }
        if (root.val < key)
            root.right = method1(root.right, key);
        else if (root.val > key)
            root.left = method1(root.left, key);
        else {
            if (root.right == null)
                return root.left;
            if (root.right == null)
                return root.left;
            else {
                root.val = method2(root.right);
                root.right = method1(root.right, root.val);
            }
        }
        return root;
    }

    public int method2(TreeNode root) {
        int var1 = root.val;
        if (root == null)
            return var1;
        while (root.left != null) {
            var1 = root.left.val;
            root = root.left;
        }
        return var1;
    }
}
