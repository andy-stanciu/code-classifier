class Solution {

    private TreeNode method1(TreeNode param1, TreeNode root) {
        if (param1 == null) {
            return root;
        }
        if (root == null) {
            return param1;
        } else {
            if (param1.val < root.val) {
                root.left = method1(param1, root.left);
            } else {
                root.right = method1(param1, root.right);
            }
            return root;
        }
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        } else {
            if (key == root.val) {
                root = method1(root.left, root.right);
            } else if (key < root.val) {
                root.left = method2(root.left, key);
            } else {
                root.right = method2(root.right, key);
            }
            return root;
        }
    }
}
