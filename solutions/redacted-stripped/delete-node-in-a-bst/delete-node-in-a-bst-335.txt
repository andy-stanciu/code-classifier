class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (root.val == key) {
            return merge(root.left, root.right);
        }
        if (root.val > key) {
            root.left = method1(root.left, key);
        } else {
            root.right = method1(root.right, key);
        }
        return root;
    }

    TreeNode merge(TreeNode left, TreeNode right) {
        if (left == null) {
            return right;
        }
        if (right == null) {
            return left;
        }
        if (left.right == null) {
            left.right = right;
            return left;
        }
        left.right = merge(left.right, right);
        return left;
    }
}
