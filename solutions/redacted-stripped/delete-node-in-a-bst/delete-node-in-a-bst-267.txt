class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (key < root.val)
            root.left = method1(root.left, key);
        else if (key > root.val)
            root.right = method1(root.right, key);
        else {
            if (root.right == null)
                return root.left;
            if (root.left == null)
                return root.right;
            TreeNode tmp = root;
            root = method2(root.right);
            root.right = method3(tmp.right);
            root.left = tmp.left;
        }
        return root;
    }

    private TreeNode method2(TreeNode root) {
        if (root == null)
            return null;
        if (root.left == null)
            return root;
        else
            return method2(root.left);
    }

    private TreeNode method3(TreeNode root) {
        if (root.left == null)
            return root.right;
        root.left = method3(root.left);
        return root;
    }
}
