class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.left == null)
                return root.right;
            if (root.right == null)
                return root.left;
            TreeNode temp = root.left;
            TreeNode prev = null;
            while (temp.right != null) {
                prev = temp;
                temp = temp.right;
            }
            if (prev == null) {
                temp.right = root.right;
                return temp;
            } else {
                prev.right = temp.left;
                temp.left = root.left;
                temp.right = root.right;
            }
            return temp;
        } else if (root.val > key)
            root.left = method1(root.left, key);
        else
            root.right = method1(root.right, key);
        return root;
    }
}
