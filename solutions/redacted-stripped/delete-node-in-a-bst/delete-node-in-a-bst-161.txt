class Solution {

    private int method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root.val;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key < root.val) {
            root.left = method2(root.left, key);
        } else if (key > root.val) {
            root.right = method2(root.right, key);
        } else {
            if (root.left == null && root.right == null) {
                return null;
            }
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                root.val = method1(root.right);
                root.right = method2(root.right, root.val);
            }
        }
        return root;
    }
}
