class Solution {

    public int max(TreeNode root) {
        if (root == null) {
            return -1;
        }
        if (root.right != null) {
            return max(root.right);
        } else {
            return root.val;
        }
    }

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                int m = max(root.left);
                root.val = m;
                root.left = method1(root.left, m);
            }
        }
        return root;
    }
}
