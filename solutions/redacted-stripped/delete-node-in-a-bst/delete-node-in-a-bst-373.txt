class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.right != null) {
                TreeNode temp = root.right;
                while (temp.left != null) temp = temp.left;
                temp.left = root.left;
                return root.right;
            } else
                return root.left;
        }
        if (root.val > key)
            root.left = method1(root.left, key);
        if (root.val < key)
            root.right = method1(root.right, key);
        return root;
    }
}
