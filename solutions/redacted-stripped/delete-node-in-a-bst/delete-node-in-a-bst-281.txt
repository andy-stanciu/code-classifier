class Solution {

    public static TreeNode method1(TreeNode r) {
        while (r.left != null) {
            r = r.left;
        }
        return r;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key > root.val) {
            root.right = method2(root.right, key);
        } else if (key < root.val) {
            root.left = method2(root.left, key);
        } else {
            if (root.left == null && root.right == null) {
                return null;
            }
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            TreeNode is = method1(root.right);
            root.val = is.val;
            root.right = method2(root.right, is.val);
        }
        return root;
    }
}
