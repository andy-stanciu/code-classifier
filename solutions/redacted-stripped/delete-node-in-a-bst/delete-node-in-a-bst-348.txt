class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            TreeNode var1 = root.right;
            while (var1 != null && var1.left != null) var1 = var1.left;
            root.val = var1.val;
            root.right = method1(root.right, var1.val);
        }
        if (root.val > key) {
            root.left = method1(root.left, key);
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        }
        return root;
    }
}
