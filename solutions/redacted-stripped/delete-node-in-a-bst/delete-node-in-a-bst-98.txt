class Solution {

    public TreeNode method1(TreeNode root, int val) {
        if (root == null) {
            return null;
        }
        if (val < root.val) {
            root.left = method1(root.left, val);
        } else if (val > root.val) {
            root.right = method1(root.right, val);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            root.val = method2(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    private int method2(TreeNode root) {
        int var1 = root.val;
        while (root.left != null) {
            root = root.left;
            var1 = root.val;
        }
        return var1;
    }
}
