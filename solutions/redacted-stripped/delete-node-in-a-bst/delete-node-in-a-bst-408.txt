class Solution {

    public TreeNode succ(TreeNode root) {
        root = root.right;
        TreeNode prev = root;
        while (root != null) {
            prev = root;
            root = root.left;
        }
        return prev;
    }

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.left == null && root.right == null)
                return null;
            else if (root.left == null && root.right != null)
                return root.right;
            else if (root.right == null && root.left != null)
                return root.left;
            else {
                TreeNode s = succ(root);
                root.val = s.val;
                root.right = method1(root.right, s.val);
            }
        } else if (root.val > key)
            root.left = method1(root.left, key);
        else
            root.right = method1(root.right, key);
        return root;
    }
}
