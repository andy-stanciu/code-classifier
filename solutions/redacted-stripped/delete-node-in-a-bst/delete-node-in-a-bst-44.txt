class Solution {

    public TreeNode method1(TreeNode param1, int key) {
        if (param1 == null)
            return null;
        if (param1.val > key)
            param1.left = method1(param1.left, key);
        else if (param1.val < key)
            param1.right = method1(param1.right, key);
        else {
            if (param1.left != null && param1.right != null) {
                int lmax = max(param1.left);
                param1.val = lmax;
                param1.left = method1(param1.left, lmax);
                return param1;
            } else if (param1.left != null)
                return param1.left;
            else if (param1.right != null)
                return param1.right;
            else
                return null;
        }
        return param1;
    }

    public int max(TreeNode param1) {
        if (param1.right != null)
            return max(param1.right);
        else
            return param1.val;
    }
}
