class Solution {

    TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        root.left = method1(root.left, key);
        root.right = method1(root.right, key);
        if (root.val == key) {
            TreeNode temp = root.right;
            if (temp == null)
                return root.left;
            if (root.left == null)
                return root.right;
            root = root.left;
            TreeNode gr = root;
            while (gr.right != null) {
                gr = gr.right;
            }
            gr.right = temp;
        }
        return root;
    }
}
