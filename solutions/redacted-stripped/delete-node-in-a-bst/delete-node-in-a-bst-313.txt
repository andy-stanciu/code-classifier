class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        } else if (root.val > key) {
            root.left = method1(root.left, key);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                int lmax = max1(root.left);
                root.left = method1(root.left, lmax);
                root.val = lmax;
            }
        }
        return root;
    }

    public static int max1(TreeNode root) {
        if (root == null) {
            return Integer.MIN_VALUE;
        }
        int r = max1(root.right);
        return Math.max(root.val, r);
    }
}
