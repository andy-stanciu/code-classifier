class Solution {

    public TreeNode method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method2(root.right, key);
        } else if (root.val > key) {
            root.left = method2(root.left, key);
        } else {
            if (root.left == null && root.right == null) {
                return null;
            }
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            TreeNode IS = method1(root.right);
            root.val = IS.val;
            root.right = method2(root.right, IS.val);
        }
        return root;
    }
}
