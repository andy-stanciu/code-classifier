class Solution {

    public TreeNode method1(TreeNode root, int key) {
        root = deletor(root, key);
        return root;
    }

    public static TreeNode method2(TreeNode given) {
        given = given.right;
        while (given != null && given.left != null) {
            given = given.left;
        }
        return given;
    }

    public static TreeNode deletor(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = deletor(root.right, key);
        } else if (root.val > key) {
            root.left = deletor(root.left, key);
        } else {
            if (root.left == null) {
                return root.right;
            }
            if (root.right == null) {
                return root.left;
            }
            TreeNode successor = method2(root);
            root.val = successor.val;
            root.right = deletor(root.right, successor.val);
        }
        return root;
    }
}
