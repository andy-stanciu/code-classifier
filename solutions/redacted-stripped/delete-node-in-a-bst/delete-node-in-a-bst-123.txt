class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        } else if (root.val > key) {
            root.left = method1(root.left, key);
        } else {
            if (root.left == null & root.right == null) {
                return null;
            } else if (root.left != null && root.right == null) {
                return root.left;
            } else if (root.left == null && root.right != null) {
                return root.right;
            } else {
                TreeNode max = func(root.left);
                root.val = max.val;
                root.left = method1(root.left, max.val);
            }
            return root;
        }
        return root;
    }

    public TreeNode func(TreeNode root) {
        if (root.right == null) {
            return root;
        }
        while (root.right != null) {
            root = root.right;
        }
        return root;
    }
}
