class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val > key) {
            root.left = method1(root.left, key);
            return root;
        } else if (root.val < key) {
            root.right = method1(root.right, key);
            return root;
        } else {
            TreeNode r = root.right;
            TreeNode l = root.left;
            if (l == null && r == null) {
                return null;
            }
            if (r == null)
                return l;
            TreeNode dummy = r;
            while (r.left != null) {
                r = r.left;
            }
            r.left = l;
            return dummy;
        }
    }
}
