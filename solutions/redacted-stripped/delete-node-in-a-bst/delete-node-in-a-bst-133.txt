class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val == key) {
            if (root.right == null)
                return root.left;
            if (root.left == null)
                return root.right;
            TreeNode temp = root.right;
            while (temp.left != null) {
                temp = temp.left;
            }
            temp.left = root.left;
            return root.right;
        }
        if (root.val < key)
            root.right = method1(root.right, key);
        else
            root.left = method1(root.left, key);
        return root;
    }
}
