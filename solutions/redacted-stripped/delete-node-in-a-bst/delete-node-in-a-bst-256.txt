class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (key > root.val)
            root.right = method1(root.right, key);
        else if (key < root.val)
            root.left = method1(root.left, key);
        else {
            if (root.right == null && root.left == null)
                return null;
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            TreeNode var1 = root.right;
            TreeNode lastRight = method2(root.left);
            lastRight.right = var1;
            return root.left;
        }
        return root;
    }

    private static TreeNode method2(TreeNode param1) {
        while (param1.right != null) {
            param1 = param1.right;
        }
        return param1;
    }
}
