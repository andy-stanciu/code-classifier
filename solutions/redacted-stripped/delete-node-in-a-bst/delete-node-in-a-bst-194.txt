class Solution {

    public TreeNode method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }

    private TreeNode getAns(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val) {
            root.left = getAns(root.left, key);
        } else if (key > root.val) {
            root.right = getAns(root.right, key);
        } else {
            if (root.left == null && root.right == null) {
                return null;
            } else if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            else {
                method1(root.right).left = root.left;
                return root.right;
            }
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        return getAns(root, key);
    }
}
