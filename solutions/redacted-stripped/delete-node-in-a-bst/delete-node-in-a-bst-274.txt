class Solution {

    TreeNode method1(TreeNode root) {
        TreeNode temp = root.left;
        while (temp.right != null) temp = temp.right;
        return temp;
    }

    TreeNode predParent(TreeNode root, TreeNode pred) {
        if (root.left == pred || root.right == pred)
            return root;
        TreeNode temp = root.left;
        while (temp.right != pred) temp = temp.right;
        return temp;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val > key)
            root.left = method2(root.left, key);
        if (root.val < key)
            root.right = method2(root.right, key);
        if (root.val == key) {
            if (root.left == null && root.right == null)
                return null;
            else if (root.left == null || root.right == null) {
                if (root.right == null)
                    return root.left;
                else
                    return root.right;
            } else if (root.left != null && root.right != null) {
                TreeNode pred = method1(root);
                TreeNode predParent = predParent(root, pred);
                if (root == predParent) {
                    pred.right = root.right;
                    return pred;
                }
                predParent.right = pred.left;
                pred.left = root.left;
                pred.right = root.right;
                return pred;
            }
        }
        return root;
    }
}
