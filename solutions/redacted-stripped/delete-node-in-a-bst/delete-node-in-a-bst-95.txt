class Solution {

    int var1 = Integer.MAX_VALUE;

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        root.left = method1(root.left, key);
        root.right = method1(root.right, key);
        if (root.val == key) {
            TreeNode Del = root;
            root = null;
            if (Del.right != null && Del.left != null) {
                method3(Del.right);
                Del.val = var1;
                Del.right = method2(Del.right, var1);
                root = Del;
            } else if (Del.left != null && Del.right == null) {
                Del = Del.left;
                root = Del;
            } else if (Del.left == null && Del.right != null) {
                Del = Del.right;
                root = Del;
            }
        }
        return root;
    }

    private TreeNode method2(TreeNode param1, int var1) {
        if (param1 == null) {
            return param1;
        }
        param1.left = method2(param1.left, var1);
        param1.right = method2(param1.right, var1);
        if (var1 == param1.val) {
            if (param1.right != null) {
                return param1.right;
            }
            if (param1.left != null) {
                return param1.left;
            }
            param1 = null;
        }
        return param1;
    }

    private void method3(TreeNode param1) {
        if (param1 == null) {
            return;
        }
        if (param1.val < var1) {
            var1 = param1.val;
        }
        method3(param1.left);
        method3(param1.right);
    }
}
