class Solution {

    int method1(TreeNode root) {
        if (root == null)
            return -1;
        while (root.right != null) {
            root = root.right;
        }
        return root.val;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            root.val = method1(root.left);
            if (root.val == -1)
                root = root.right;
            else
                root.left = method2(root.left, root.val);
        } else if (root.val < key)
            root.right = method2(root.right, key);
        else
            root.left = method2(root.left, key);
        return root;
    }
}
