class Solution {

    public TreeNode method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val > key) {
            root.left = method2(root.left, key);
        } else if (root.val < key) {
            root.right = method2(root.right, key);
        } else {
            if (root.left == null && root.right == null) {
                return null;
            }
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            if (root.left != null && root.right != null) {
                TreeNode successor = method1(root.right);
                root.val = successor.val;
                root.right = method2(root.right, successor.val);
            }
        }
        return root;
    }
}
