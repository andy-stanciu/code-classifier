class Solution {

    public static TreeNode method1(TreeNode root) {
        while (root.right != null) {
            root = root.right;
        }
        return root;
    }

    public static TreeNode deletion(TreeNode root) {
        if (root == null) {
            return null;
        }
        if (root.left == null) {
            return root.right;
        }
        if (root.right == null) {
            return root.left;
        }
        TreeNode r = root.right;
        TreeNode var1 = method1(root.left);
        var1.right = r;
        return root.left;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val == key) {
            return deletion(root);
        }
        TreeNode temp = root;
        while (root != null) {
            if (root.val > key) {
                if (root.left != null && root.left.val != key) {
                    root = root.left;
                } else {
                    root.left = deletion(root.left);
                    break;
                }
            } else {
                if (root.right != null && root.right.val != key) {
                    root = root.right;
                } else {
                    root.right = deletion(root.right);
                    break;
                }
            }
        }
        return temp;
    }
}
