class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            if (root.left == null) {
                return root.right;
            }
            if (root.right == null) {
                return root.left;
            }
            root.val = successor(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    private int successor(TreeNode param1) {
        while (param1 != null && param1.left != null) {
            param1 = param1.left;
        }
        return param1.val;
    }
}
