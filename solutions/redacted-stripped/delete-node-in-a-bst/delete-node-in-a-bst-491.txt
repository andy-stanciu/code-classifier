class Solution {

    private int method1(TreeNode root) {
        while (root.right != null) root = root.right;
        return root.val;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.left == null && root.right == null)
                return null;
            else if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            else {
                int successor = method1(root.left);
                root.val = successor;
                root.left = method2(root.left, successor);
                return root;
            }
        }
        if (key < root.val)
            root.left = method2(root.left, key);
        else
            root.right = method2(root.right, key);
        return root;
    }
}
