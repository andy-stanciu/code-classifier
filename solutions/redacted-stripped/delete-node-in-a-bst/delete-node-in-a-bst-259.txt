class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val > key) {
            root.left = method1(root.left, key);
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else {
            if (root.left == null)
                return root.right;
            if (root.right == null)
                return root.left;
            TreeNode var1 = method2(root.left);
            root.val = var1.val;
            root.left = method1(root.left, root.val);
        }
        return root;
    }

    private TreeNode method2(TreeNode root) {
        while (root.right != null) {
            root = root.right;
        }
        return root;
    }
}
