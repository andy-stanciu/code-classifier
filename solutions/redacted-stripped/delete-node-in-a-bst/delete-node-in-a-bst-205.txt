class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            TreeNode ptr = root.right;
            TreeNode prev = root;
            while (ptr.left != null) {
                prev = ptr;
                ptr = ptr.left;
            }
            root.val = ptr.val;
            if (root == prev) {
                prev.right = ptr.right;
            } else {
                prev.left = ptr.right;
            }
        }
        return root;
    }
}
