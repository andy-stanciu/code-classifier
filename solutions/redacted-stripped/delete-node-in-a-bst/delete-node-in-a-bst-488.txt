class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        } else if (root.val > key) {
            root.left = method1(root.left, key);
        } else {
            if (root.left != null && root.right != null) {
                int lmax = method2(root.left);
                root.val = lmax;
                root.left = method1(root.left, lmax);
                return root;
            } else if (root.left != null) {
                return root.left;
            } else if (root.right != null) {
                return root.right;
            } else {
                return null;
            }
        }
        return root;
    }

    public int method2(TreeNode param1) {
        while (param1.right != null) {
            param1 = param1.right;
        }
        return param1.val;
    }
}
