class Solution {

    public TreeNode method1(TreeNode param1) {
        TreeNode curr = param1;
        while (curr.left != null) {
            curr = curr.left;
        }
        return curr;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method2(root.right, key);
        } else if (root.val > key) {
            root.left = method2(root.left, key);
        } else {
            if (root.left == null) {
                TreeNode temp = root.right;
                return temp;
            } else if (root.right == null) {
                TreeNode temp = root.left;
                return temp;
            }
            TreeNode temp = method1(root.right);
            root.val = temp.val;
            root.right = method2(root.right, temp.val);
        }
        return root;
    }
}
