class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        } else {
            if (root.val == key) {
                if (root.right == null) {
                    return root.left;
                }
                if (root.right.left == null) {
                    root.right.left = root.left;
                    return root.right;
                }
                TreeNode prev = root.right;
                TreeNode curr = root.right.left;
                while (curr.left != null) {
                    prev = curr;
                    curr = curr.left;
                }
                prev.left = curr.right;
                root.val = curr.val;
                return root;
            }
            root.left = method1(root.left, key);
            root.right = method1(root.right, key);
            return root;
        }
    }
}
