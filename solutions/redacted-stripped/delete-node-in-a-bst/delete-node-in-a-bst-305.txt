class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val > key) {
            TreeNode left = method1(root.left, key);
            root.left = left;
            return root;
        } else if (root.val < key) {
            TreeNode right = method1(root.right, key);
            root.right = right;
            return root;
        }
        if (root.left == null && root.right == null)
            return null;
        if (root.left == null || root.right == null) {
            if (root.left != null)
                return root.left;
            else
                return root.right;
        }
        TreeNode mostRight = method2(root.left);
        mostRight.right = root.right;
        return root.left;
    }

    private TreeNode method2(TreeNode root) {
        if (root == null)
            return null;
        if (root.right != null)
            return method2(root.right);
        else
            return root;
    }
}
