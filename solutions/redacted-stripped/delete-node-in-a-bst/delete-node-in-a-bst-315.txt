class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key)
            return solver(root);
        TreeNode dummy = root;
        while (root != null) {
            if (key < root.val) {
                if (root.left != null && root.left.val == key) {
                    root.left = solver(root.left);
                    break;
                } else
                    root = root.left;
            } else {
                if (root.right != null && root.right.val == key) {
                    root.right = solver(root.right);
                    break;
                } else
                    root = root.right;
            }
        }
        return dummy;
    }

    public TreeNode solver(TreeNode root) {
        if (root.left == null)
            return root.right;
        else if (root.right == null)
            return root.left;
        TreeNode var1 = root.right;
        TreeNode ExtremeRightofLeft = method2(root.left);
        ExtremeRightofLeft.right = var1;
        return root.left;
    }

    public TreeNode method2(TreeNode root) {
        if (root.right == null)
            return root;
        return method2(root.right);
    }
}
