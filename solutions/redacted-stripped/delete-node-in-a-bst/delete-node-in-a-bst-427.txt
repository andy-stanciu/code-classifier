class Solution {

    public static int method1(TreeNode root) {
        int var1 = root.val;
        while (root.left != null) {
            var1 = root.left.val;
            root = root.left;
        }
        return var1;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return root;
        } else if (key > root.val) {
            root.right = method2(root.right, key);
        } else if (key < root.val) {
            root.left = method2(root.left, key);
        } else {
            if (root.left == null) {
                return root.right;
            }
            if (root.right == null) {
                return root.left;
            }
            root.val = method1(root.right);
            root.right = method2(root.right, root.val);
        }
        return root;
    }
}
