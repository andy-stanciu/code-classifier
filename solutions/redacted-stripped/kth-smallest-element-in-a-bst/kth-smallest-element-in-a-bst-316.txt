class Solution {

    int var1 = 0;

    public int method1(TreeNode root, int k) {
        func(root, k);
        return var1;
    }

    int func(TreeNode root, int k) {
        if (root == null)
            return k;
        if (k > 0) {
            k = func(root.left, k);
            if (k > 0) {
                var1 = root.val;
                --k;
            }
        }
        if (k > 0) {
            k = func(root.right, k);
        }
        return k;
    }
}
