class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<TreeNode> nodes = new ArrayList<>();
        dfs(root, nodes);
        return (nodes.get(k - 1).val);
    }

    public static void dfs(TreeNode root, ArrayList<TreeNode> nodes) {
        if (root == null)
            return;
        dfs(root.left, nodes);
        nodes.add(root);
        dfs(root.right, nodes);
    }
}
