class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> list = method2(root, new ArrayList<>());
        return list.get(k - 1);
    }

    public List<Integer> method2(TreeNode root, List<Integer> res) {
        if (root == null) {
            return res;
        }
        method2(root.left, res);
        res.add(root.val);
        method2(root.right, res);
        return res;
    }
}
