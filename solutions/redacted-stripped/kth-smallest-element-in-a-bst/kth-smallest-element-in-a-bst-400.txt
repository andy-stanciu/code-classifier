class Solution {

    int c = 1, ans = -1;

    public void method1(TreeNode root, int k) {
        if (root == null)
            return;
        method1(root.left, k);
        if (c++ == k) {
            ans = root.val;
            return;
        }
        method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return ans;
    }
}
