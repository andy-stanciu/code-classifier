class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> arr = new ArrayList<>();
        dfs(root, arr);
        arr.sort((a, b) -> a - b);
        if (k == 0) {
            return arr.get(0);
        }
        return arr.get(k - 1);
    }

    private void dfs(TreeNode root, List<Integer> arr) {
        if (root == null) {
            return;
        }
        arr.add(root.val);
        dfs(root.left, arr);
        dfs(root.right, arr);
    }
}
