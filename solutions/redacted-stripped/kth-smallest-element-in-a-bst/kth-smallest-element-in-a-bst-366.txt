class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> a = new ArrayList<>();
        small(root, a);
        return a.get(k - 1);
    }

    public void small(TreeNode root, List<Integer> a) {
        if (root == null)
            return;
        small(root.left, a);
        a.add(root.val);
        small(root.right, a);
    }
}
