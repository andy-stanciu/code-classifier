class Solution {

    public void preOrder(TreeNode root, List<Integer> al) {
        if (root == null)
            return;
        al.add(root.val);
        preOrder(root.left, al);
        preOrder(root.right, al);
    }

    public int method1(TreeNode root, int k) {
        List<Integer> al = new ArrayList<Integer>();
        preOrder(root, al);
        Collections.sort(al);
        return al.get(k - 1);
    }
}
