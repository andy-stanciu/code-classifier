class Solution {

    int i = 0;

    int ans = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode node, int k) {
        if (node == null)
            return;
        method2(node.left, k);
        i++;
        if (i == k) {
            ans = node.val;
            return;
        }
        method2(node.right, k);
    }
}
