class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> array = new ArrayList<>();
        method2(root, array);
        return array.get(k - 1);
    }

    public static void method2(TreeNode node, List<Integer> array) {
        if (node == null) {
            return;
        }
        method2(node.left, array);
        array.add(node.val);
        method2(node.right, array);
    }
}
