class Solution {

    int k;

    int ans;

    public int method1(TreeNode root, int k) {
        this.k = k;
        this.ans = 0;
        method2(root);
        return ans;
    }

    private void method2(TreeNode root) {
        if (k <= 0 || root == null) {
            return;
        }
        method2(root.left);
        if (--k == 0) {
            ans = root.val;
            return;
        }
        method2(root.right);
    }
}
