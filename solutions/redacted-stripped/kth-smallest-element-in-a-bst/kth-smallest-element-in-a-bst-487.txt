class Solution {

    private int m;

    private int n;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return n;
    }

    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        m++;
        if (m == k) {
            n = root.val;
            return;
        }
        method2(root.right, k);
    }
}
