class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> ls = new ArrayList<>();
        method2(root, ls);
        int z = ls.size();
        return ls.get(k - 1);
    }

    public void method2(TreeNode root, List<Integer> ls) {
        if (root != null) {
            method2(root.left, ls);
            ls.add(root.val);
            method2(root.right, ls);
        }
    }
}
