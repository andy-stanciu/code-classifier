class Solution {

    private static List<Integer> list;

    public int method1(TreeNode root, int k) {
        list = new ArrayList<>();
        method2(root);
        return list.get(k - 1);
    }

    private void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        list.add(root.val);
        method2(root.right);
    }
}
