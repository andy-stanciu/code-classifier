class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        Stack<TreeNode> st = new Stack<>();
        TreeNode current = root;
        while (current != null || !st.isEmpty()) {
            while (current != null) {
                st.push(current);
                current = current.left;
            }
            current = st.pop();
            k--;
            if (k == 0) {
                return current.val;
            }
            current = current.right;
        }
        return -1;
    }
}
