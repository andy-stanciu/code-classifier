class Solution {

    int kCount = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        if (root == null || kCount > k) {
            return ans;
        }
        method1(root.left, k);
        kCount++;
        if (k == kCount) {
            ans = root.val;
            return root.val;
        }
        method1(root.right, k);
        return ans;
    }
}
