class Solution {

    int ans = 0;

    int count = 1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        if (count == k) {
            ans = root.val;
        }
        count += 1;
        method2(root.right, k);
    }
}
