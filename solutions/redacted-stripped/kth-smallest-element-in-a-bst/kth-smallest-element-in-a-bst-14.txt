class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        method2(root, arr);
        return arr.get(k - 1);
    }

    public static void method2(TreeNode root, ArrayList<Integer> arr) {
        if (root == null) {
            return;
        }
        method2(root.left, arr);
        arr.add(root.val);
        method2(root.right, arr);
    }
}
