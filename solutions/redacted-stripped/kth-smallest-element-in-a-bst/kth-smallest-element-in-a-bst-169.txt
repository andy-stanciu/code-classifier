class Solution {

    int var1 = 0;

    int res = 0;

    public int method1(TreeNode root, int k) {
        small(root, k);
        return res;
    }

    public void small(TreeNode root, int k) {
        if (root == null)
            return;
        small(root.left, k);
        var1++;
        if (var1 == k)
            res = root.val;
        small(root.right, k);
    }
}
