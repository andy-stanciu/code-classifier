class Solution {

    int count = 0;

    int var1 = -1;

    public int var1(TreeNode root, int k) {
        method1(root, k);
        if (count < k)
            return -1;
        return var1;
    }

    public void method1(TreeNode root, int k) {
        if (root == null)
            return;
        method1(root.left, k);
        count++;
        if (count == k) {
            var1 = root.val;
            return;
        }
        method1(root.right, k);
    }
}
