class Solution {

    List<Integer> list = new ArrayList<>();

    public void method1(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method1(root.left, k);
        list.add(root.val);
        if (list.size() == k) {
            return;
        }
        method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return list.get(k - 1);
    }
}
