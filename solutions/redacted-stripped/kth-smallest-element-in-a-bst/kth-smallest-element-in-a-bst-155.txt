class Solution {

    int count = 0;

    public int method1(TreeNode root, int k) {
        int[] res = new int[1];
        method2(root, k, res);
        return res[0];
    }

    public void method2(TreeNode root, int k, int[] res) {
        if (root == null) {
            return;
        }
        method2(root.left, k, res);
        count++;
        if (k == count) {
            res[0] = root.val;
            return;
        }
        method2(root.right, k, res);
    }
}
