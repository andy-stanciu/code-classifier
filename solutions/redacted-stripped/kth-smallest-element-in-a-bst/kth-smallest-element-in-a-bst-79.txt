class Solution {

    public int count = 0;

    public int result = -1;

    public int method1(TreeNode root, int k) {
        traversal(root, k);
        return result;
    }

    public void traversal(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        traversal(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        traversal(root.right, k);
    }
}
