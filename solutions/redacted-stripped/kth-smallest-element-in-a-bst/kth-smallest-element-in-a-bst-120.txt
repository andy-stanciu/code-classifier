class Solution {

    public void method1(TreeNode root, List<Integer> l) {
        if (root == null)
            return;
        method1(root.left, l);
        l.add(root.val);
        method1(root.right, l);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> l = new ArrayList<>();
        method1(root, l);
        return l.get(k - 1);
    }
}
