class Solution {

    int ans = 0;

    int counter = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        ++counter;
        if (counter == k) {
            ans = root.val;
        }
        method2(root.right, k);
    }
}
