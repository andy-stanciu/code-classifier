class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        List<Integer> list = new ArrayList<>();
        method2(root, list);
        return list.get(k - 1);
    }

    private void method2(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        method2(root.left, list);
        list.add(root.val);
        method2(root.right, list);
    }
}
