class Solution {

    List<Integer> l = new ArrayList<>();

    public void method1(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        l.add(root.val);
        method1(root.right);
    }

    public int method2(TreeNode root, int k) {
        if (root == null)
            return 0;
        method1(root);
        return l.get(k - 1);
    }
}
