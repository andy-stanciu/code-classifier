class Solution {

    static int count;

    static int res;

    public int method1(TreeNode root, int k) {
        count = 0;
        res = 0;
        method2(root, k);
        return res;
    }

    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        count++;
        if (count == k) {
            res = root.val;
            return;
        }
        method2(root.right, k);
    }
}
