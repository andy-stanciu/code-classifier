class Solution {

    int count = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        int Left = method1(root.left, k);
        if (Left != (-1)) {
            return Left;
        }
        count++;
        if (count == k) {
            return root.val;
        }
        return method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        int ans = method1(root, k);
        return ans;
    }
}
