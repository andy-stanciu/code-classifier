class Solution {

    List<Integer> list = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        if (root == null || list.size() >= k)
            return 0;
        method1(root.left, k);
        list.add(root.val);
        method1(root.right, k);
        return list.size() >= k ? list.get(k - 1) : -1;
    }
}
