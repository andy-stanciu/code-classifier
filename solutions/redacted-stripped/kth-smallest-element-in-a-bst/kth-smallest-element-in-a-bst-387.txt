class Solution {

    public int method1(TreeNode root, int k) {
        return method2(root, k);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        if (root == null)
            return 0;
        method3(root, list);
        return list.get(k - 1);
    }

    private void method3(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        method3(node.left, list);
        list.add(node.val);
        method3(node.right, list);
    }
}
