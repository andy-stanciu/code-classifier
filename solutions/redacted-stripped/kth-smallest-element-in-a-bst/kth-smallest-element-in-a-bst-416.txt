class Solution {

    public ArrayList method1(TreeNode root, ArrayList<Integer> arr) {
        if (root == null)
            return arr;
        method1(root.left, arr);
        arr.add(root.val);
        method1(root.right, arr);
        return arr;
    }

    public int method2(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        ArrayList<Integer> var1 = method1(root, arr);
        return var1.get(k - 1);
    }
}
