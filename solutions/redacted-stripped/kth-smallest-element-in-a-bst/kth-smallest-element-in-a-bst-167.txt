class Solution {

    PriorityQueue<Integer> result = new PriorityQueue<>();

    public int method1(TreeNode root, int k) {
        dfs(root);
        int z = 1;
        while (z++ < k) {
            result.poll();
        }
        return result.peek();
    }

    void dfs(TreeNode root) {
        if (root == null)
            return;
        result.add(root.val);
        dfs(root.left);
        dfs(root.right);
    }
}
