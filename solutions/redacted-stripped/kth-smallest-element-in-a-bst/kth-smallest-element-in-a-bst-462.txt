class Solution {

    ArrayList<Integer> arr = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        helper(root);
        return arr.get(k - 1);
    }

    public void helper(TreeNode root) {
        if (root == null) {
            return;
        }
        helper(root.left);
        arr.add(root.val);
        helper(root.right);
    }
}
