class Solution {

    int value = -1;

    int i = 0;

    public int method1(TreeNode root, int k) {
        readTree(root, k);
        return value;
    }

    private void readTree(TreeNode node, int k) {
        if (node == null || value != -1)
            return;
        readTree(node.left, k);
        i++;
        if (i == k)
            value = node.val;
        readTree(node.right, k);
    }
}
