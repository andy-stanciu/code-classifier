class Solution {

    private int var1 = 0;

    private int var2 = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        method1(root.left, k);
        var2++;
        if (k == var2) {
            var1 = root.val;
        }
        method1(root.right, k);
        return var1;
    }
}
