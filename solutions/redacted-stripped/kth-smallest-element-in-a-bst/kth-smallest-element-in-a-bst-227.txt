class Solution {

    int var1 = -1;

    int count = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method1(root.left, k);
        if (k == count) {
            return var1;
        } else {
            var1 = root.val;
            count++;
        }
        method1(root.right, k);
        return var1;
    }
}
