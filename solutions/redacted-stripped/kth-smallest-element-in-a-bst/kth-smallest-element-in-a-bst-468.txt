class Solution {

    List<Integer> preOrder = new ArrayList<>();

    int position = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return preOrder.get(k - 1);
    }

    public void method2(TreeNode root, int k) {
        if (position == k || root == null) {
            return;
        }
        method2(root.left, k);
        preOrder.add(root.val);
        method2(root.right, k);
    }
}
