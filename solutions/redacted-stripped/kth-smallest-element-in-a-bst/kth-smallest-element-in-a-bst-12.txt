class Solution {

    int count = 0;

    public int method1(TreeNode root, int k) {
        TreeNode ans = method2(root, k);
        return ans.val;
    }

    TreeNode method2(TreeNode root, int k) {
        if (root == null) {
            return null;
        }
        TreeNode left = method2(root.left, k);
        if (left != null) {
            return left;
        }
        count++;
        if (count == k) {
            return root;
        }
        return method2(root.right, k);
    }
}
