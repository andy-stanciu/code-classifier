class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        dfs(root, k, list);
        return list.get(k - 1);
    }

    private void dfs(TreeNode root, int k, List<Integer> list) {
        if (root == null || list.size() == k) {
            return;
        }
        dfs(root.left, k, list);
        list.add(root.val);
        dfs(root.right, k, list);
    }
}
