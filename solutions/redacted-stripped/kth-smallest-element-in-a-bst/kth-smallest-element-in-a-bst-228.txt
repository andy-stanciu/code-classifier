class Solution {

    int i = -1;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        int left = method1(root.left, k);
        i++;
        if (i == k - 1) {
            return root.val;
        }
        int right = method1(root.right, k);
        if (left == -1) {
            return right;
        }
        return left;
    }
}
