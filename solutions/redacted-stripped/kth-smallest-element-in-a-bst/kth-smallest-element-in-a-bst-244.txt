class Solution {

    int count = 0;

    int result = -1;

    public int method1(TreeNode root, int k) {
        dfs(root, k);
        return result;
    }

    public void dfs(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        dfs(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        dfs(root.right, k);
    }
}
