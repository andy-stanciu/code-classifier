class Solution {

    int k;

    public int method1(TreeNode root, int k) {
        this.k = k;
        return traverse(root);
    }

    private int traverse(TreeNode root) {
        if (root == null)
            return -1;
        var res = traverse(root.left);
        if (res != -1)
            return res;
        if (k == 1)
            return root.val;
        k--;
        return traverse(root.right);
    }
}
