class Solution {

    int i, ans;

    public int method1(TreeNode root, int k) {
        i = 1;
        ans = -1;
        helper(root, k);
        return ans;
    }

    void helper(TreeNode root, int k) {
        if (root == null || ans != -1)
            return;
        helper(root.left, k);
        if (i == k) {
            ans = root.val;
        }
        i++;
        helper(root.right, k);
    }
}
