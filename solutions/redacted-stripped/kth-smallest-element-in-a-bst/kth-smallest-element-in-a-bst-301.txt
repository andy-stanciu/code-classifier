class Solution {

    int x = 0;

    int count = 0;

    public int method1(TreeNode root, int k) {
        count = k;
        method2(root);
        return x;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        count--;
        if (count == 0) {
            x = root.val;
            return;
        }
        method2(root.right);
    }
}
