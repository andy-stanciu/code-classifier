class Solution {

    public int method1(TreeNode root, int k) {
        LinkedList<Integer> ll = new LinkedList<>();
        dfs(root, ll);
        return ll.get(k - 1);
    }

    public void dfs(TreeNode root, LinkedList<Integer> ll) {
        if (root == null)
            return;
        dfs(root.left, ll);
        ll.add(root.val);
        dfs(root.right, ll);
    }
}
