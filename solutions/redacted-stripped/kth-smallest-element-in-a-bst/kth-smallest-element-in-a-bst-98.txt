class Solution {

    public static void method1(ArrayList<Integer> list, TreeNode root) {
        if (root == null) {
            return;
        }
        method1(list, root.left);
        list.add(root.val);
        method1(list, root.right);
    }

    public int method2(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        method1(list, root);
        return list.get(k - 1);
    }
}
