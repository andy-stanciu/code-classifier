class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> store = new ArrayList<>(2000);
        method2(root, store);
        return store.get(k - 1);
    }

    public static void method2(TreeNode root, List store) {
        if (root == null)
            return;
        method2(root.left, store);
        store.add(root.val);
        method2(root.right, store);
    }
}
