class Solution {

    private int count = 0;

    private int result = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return result;
    }

    private void method2(TreeNode node, int k) {
        if (node == null) {
            return;
        }
        method2(node.left, k);
        count++;
        if (count == k) {
            result = node.val;
            return;
        }
        method2(node.right, k);
    }
}
