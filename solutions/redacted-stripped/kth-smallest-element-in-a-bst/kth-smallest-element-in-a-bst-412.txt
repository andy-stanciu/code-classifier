class Solution {

    Queue<Integer> queue = new LinkedList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        while (k - 1 != 0) {
            queue.poll();
            k--;
        }
        return queue.poll();
    }

    private void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        queue.add(root.val);
        method2(root.right);
    }
}
