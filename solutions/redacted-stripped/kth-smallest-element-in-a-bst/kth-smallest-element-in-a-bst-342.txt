class Solution {

    int pos = 0;

    public int method1(TreeNode root, int k) {
        TreeNode node = method2(root, k);
        if (node == null) {
            return -1;
        }
        return node.val;
    }

    public TreeNode method2(TreeNode root, int k) {
        if (root == null) {
            return root;
        }
        TreeNode left = method2(root.left, k);
        if (left != null) {
            return left;
        }
        pos++;
        if (pos == k) {
            return root;
        }
        return method2(root.right, k);
    }
}
