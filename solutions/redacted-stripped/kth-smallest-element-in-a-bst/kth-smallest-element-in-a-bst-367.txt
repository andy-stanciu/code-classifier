class Solution {

    int n = 0;

    int k = 0;

    public int method1(TreeNode root, int k) {
        this.k = k;
        method2(root);
        return n;
    }

    public void method2(TreeNode node) {
        if (node == null)
            return;
        method2(node.left);
        if (--k == 0) {
            n = node.val;
            return;
        }
        method2(node.right);
    }
}
