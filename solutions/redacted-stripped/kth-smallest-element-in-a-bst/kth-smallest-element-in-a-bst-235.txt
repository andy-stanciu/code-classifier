class Solution {

    private int count = 0;

    private int ans = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        count++;
        if (count == k)
            ans = root.val;
        method2(root.right, k);
    }
}
