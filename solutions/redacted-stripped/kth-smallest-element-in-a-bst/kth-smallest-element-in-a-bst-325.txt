class Solution {

    ArrayList<Integer> result = new ArrayList();

    public int method1(TreeNode root, int k) {
        dfs(root);
        return result.get(k - 1);
    }

    void dfs(TreeNode root) {
        if (root == null)
            return;
        dfs(root.left);
        result.add(root.val);
        dfs(root.right);
    }
}
