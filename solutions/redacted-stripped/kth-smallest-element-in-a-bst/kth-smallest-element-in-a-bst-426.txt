class Solution {

    int c = 0, res;

    public int method1(TreeNode root, int k) {
        traverse(root, k);
        return res;
    }

    public void traverse(TreeNode root, int k) {
        if (root != null) {
            traverse(root.left, k);
            c++;
            if (k == c)
                res = root.val;
            traverse(root.right, k);
        }
    }
}
