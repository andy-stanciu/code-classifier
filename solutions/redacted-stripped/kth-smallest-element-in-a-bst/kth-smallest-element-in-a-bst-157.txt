class Solution {

    public static void solve(TreeNode root, int k, int[] ans) {
        if (root == null)
            return;
        solve(root.left, k, ans);
        ans[0]++;
        if (ans[0] == k) {
            ans[1] = root.val;
            return;
        }
        solve(root.right, k, ans);
    }

    public int method1(TreeNode root, int k) {
        int[] ans = new int[2];
        solve(root, k, ans);
        return ans[1];
    }
}
