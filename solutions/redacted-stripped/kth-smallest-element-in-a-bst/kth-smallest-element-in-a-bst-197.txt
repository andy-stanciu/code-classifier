class Solution {

    public int method1(TreeNode root, int k) {
        Stack<TreeNode> st = new Stack<>();
        while (!st.isEmpty() || root != null) {
            while (root != null) {
                st.add(root);
                root = root.left;
            }
            root = st.pop();
            k--;
            if (k == 0)
                return root.val;
            root = root.right;
        }
        return 0;
    }
}
