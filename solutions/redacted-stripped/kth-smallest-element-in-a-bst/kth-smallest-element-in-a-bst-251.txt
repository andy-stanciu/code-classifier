class Solution {

    Stack<TreeNode> s = new Stack<>();

    public int method1(TreeNode root, int k) {
        helper(root, k);
        return s.pop().val;
    }

    public void helper(TreeNode root, int k) {
        if (root == null || s.size() == k) {
            return;
        }
        helper(root.left, k);
        if (s.size() != k) {
            s.push(root);
        }
        helper(root.right, k);
    }
}
