class Solution {

    int count = 0;

    int result = 0;

    public int method1(TreeNode root, int k) {
        small(root, k);
        return result;
    }

    public void small(TreeNode root, int k) {
        if (root == null)
            return;
        small(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        small(root.right, k);
    }
}
