class Solution {

    int var1, count;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return var1;
    }

    public void method2(TreeNode root, int k) {
        if (root.left != null) {
            method2(root.left, k);
        }
        count++;
        if (count == k) {
            var1 = root.val;
        }
        if (root.right != null) {
            method2(root.right, k);
        }
    }
}
