class Solution {

    public List<Integer> list;

    public int method1(TreeNode root, int k) {
        list = new ArrayList<Integer>();
        solve(root);
        return list.get(k - 1);
    }

    public void solve(TreeNode root) {
        if (root == null)
            return;
        solve(root.left);
        list.add(root.val);
        solve(root.right);
    }
}
