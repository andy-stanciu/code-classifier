class Solution {

    int count = 0;

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root, int k) {
        traversal(root, k);
        return ans;
    }

    public void traversal(TreeNode root, int k) {
        if (root == null)
            return;
        traversal(root.left, k);
        count++;
        if (count == k) {
            ans = root.val;
        }
        traversal(root.right, k);
    }
}
