class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        List<Integer> list = new ArrayList<>();
        list = method2(root, list, k);
        return list.getLast();
    }

    public static List<Integer> method2(TreeNode node, List<Integer> list, int k) {
        if (node == null) {
            return list;
        }
        method2(node.left, list, k);
        if (list.size() < k) {
            list.add(node.val);
        } else {
            return list;
        }
        method2(node.right, list, k);
        return list;
    }
}
