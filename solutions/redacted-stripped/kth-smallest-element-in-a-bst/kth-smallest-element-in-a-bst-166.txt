class Solution {

    private int count = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        int left = method1(root.left, k);
        if (left != -1) {
            return left;
        }
        if (++count == k) {
            return root.val;
        }
        return method1(root.right, k);
    }
}
