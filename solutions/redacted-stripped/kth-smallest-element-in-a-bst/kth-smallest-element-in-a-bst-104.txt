class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> l = new ArrayList<>();
        method2(root, l);
        Collections.sort(l);
        return (l.size() >= k - 1) ? l.get(k - 1) : -1;
    }

    void method2(TreeNode root, List<Integer> l) {
        if (root == null)
            return;
        l.add(root.val);
        method2(root.left, l);
        method2(root.right, l);
    }
}
