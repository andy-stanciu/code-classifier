class Solution {

    public int method1(TreeNode root, int k) {
        int[] ans = { 0, 0 };
        method2(root, k, ans);
        return ans[1];
    }

    private void method2(TreeNode root, int k, int[] ans) {
        if (root == null)
            return;
        method2(root.left, k, ans);
        if (k == ++ans[0]) {
            ans[1] = root.val;
            return;
        }
        method2(root.right, k, ans);
    }
}
