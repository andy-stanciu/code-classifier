class Solution {

    int var1 = 0;

    int val = -1;

    public int method1(TreeNode root, int k) {
        call(root, k);
        return val;
    }

    public void call(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        call(root.left, k);
        var1++;
        if (var1 == k) {
            val = root.val;
        }
        call(root.right, k);
    }
}
