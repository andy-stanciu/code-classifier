class Solution {

    public void method1(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        method1(root.left, res);
        res.add(root.val);
        method1(root.right, res);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> res = new ArrayList<>();
        method1(root, res);
        Collections.sort(res);
        return res.get(k - 1);
    }
}
