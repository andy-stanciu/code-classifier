class Solution {

    int count = 0;

    int result = Integer.MIN_VALUE;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return result;
    }

    private void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        method2(root.right, k);
    }
}
