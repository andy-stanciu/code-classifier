class Solution {

    public static void method1(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        method1(root.left, list);
        list.add(root.val);
        method1(root.right, list);
    }

    public static int method2(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        method1(root, list);
        return list.get(k - 1);
    }
}
