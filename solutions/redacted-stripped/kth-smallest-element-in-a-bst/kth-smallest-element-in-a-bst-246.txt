class Solution {

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    int c = 0;

    int ans = 0;

    void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        c++;
        if (c == k) {
            ans = root.val;
            return;
        }
        method2(root.right, k);
    }
}
