class Solution {

    int poss;

    int ans;

    void value(TreeNode root) {
        if (root != null) {
            if (root.left != null) {
                value(root.left);
            }
            poss--;
            if (poss == 0)
                ans = root.val;
            if (root.right != null) {
                value(root.right);
            }
        }
        return;
    }

    public int method1(TreeNode root, int k) {
        poss = k;
        value(root);
        return ans;
    }
}
