class Solution {

    public int method1(TreeNode root, int k) {
        int cnt = 0;
        TreeNode curr = root;
        int res = -1;
        while (cnt != k) {
            if (curr.left == null) {
                res = curr.val;
                curr = curr.right;
                cnt++;
            } else {
                TreeNode prev = curr.left;
                while (prev.right != null && prev.right != curr) {
                    prev = prev.right;
                }
                if (prev.right == null) {
                    prev.right = curr;
                    curr = curr.left;
                } else {
                    prev.right = null;
                    res = curr.val;
                    curr = curr.right;
                    cnt++;
                }
            }
        }
        return res;
    }
}
