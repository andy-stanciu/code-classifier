class Solution {

    int i = 0;

    public int method1(TreeNode root, int k) {
        return solve(root, k);
    }

    private int solve(TreeNode root, int k) {
        if (root == null)
            return -1;
        int left = solve(root.left, k);
        if (left != -1)
            return left;
        i++;
        if (i == k)
            return root.val;
        return solve(root.right, k);
    }
}
