class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        int val = 0;
        method2(root, list);
        for (int i = 0; i < k; i++) {
            val = list.get(i);
        }
        return val;
    }

    public static void method2(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        method2(root.left, list);
        list.add(root.val);
        method2(root.right, list);
    }
}
