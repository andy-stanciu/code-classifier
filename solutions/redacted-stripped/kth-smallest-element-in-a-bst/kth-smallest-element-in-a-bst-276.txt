class Solution {

    public int method1(TreeNode root, int k) {
        int[] i = { 0 };
        int ans = solve(root, i, k);
        return ans;
    }

    public int solve(TreeNode root, int[] i, int k) {
        if (root == null)
            return -1;
        int l = solve(root.left, i, k);
        if (l != -1)
            return l;
        i[0]++;
        if (i[0] == k)
            return root.val;
        return solve(root.right, i, k);
    }
}
