class Solution {

    int count = 1, res = -1;

    public int method1(TreeNode root, int k) {
        if (root.left != null)
            method1(root.left, k);
        if (count++ == k)
            res = root.val;
        if (root.right != null)
            method1(root.right, k);
        return res;
    }
}
