class Solution {

    ArrayList<Integer> arr = new ArrayList<>();

    private void method1(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        arr.add(root.val);
        method1(root.right);
    }

    public int method2(TreeNode root, int k) {
        method1(root);
        return arr.get(k - 1);
    }
}
