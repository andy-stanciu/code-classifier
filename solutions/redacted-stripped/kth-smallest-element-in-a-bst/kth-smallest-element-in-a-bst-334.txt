class Solution {

    int res = -1;

    public int method1(TreeNode root, int k) {
        if (null == root.left && null == root.right)
            return root.val;
        method2(root, k, 0);
        return res;
    }

    int method2(TreeNode root, int k, int parcel) {
        if (root == null)
            return parcel;
        parcel = method2(root.left, k, parcel);
        parcel = parcel + 1;
        if (parcel == k) {
            res = root.val;
        }
        parcel = method2(root.right, k, parcel);
        return parcel;
    }
}
