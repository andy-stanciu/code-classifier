class Solution {

    int count = 0;

    int var1 = 0;

    public int method1(TreeNode root, int k) {
        DFS(root, k);
        return var1;
    }

    public void DFS(TreeNode root, int k) {
        if (root == null)
            return;
        DFS(root.left, k);
        count++;
        if (count == k) {
            var1 = root.val;
            return;
        }
        DFS(root.right, k);
    }
}
