class Solution {

    private int count;

    private int result = 0;

    public int method1(TreeNode root, int k) {
        count = k;
        method2(root);
        return result;
    }

    private void method2(TreeNode node) {
        if (node == null)
            return;
        method2(node.left);
        count--;
        if (count == 0) {
            result = node.val;
            return;
        }
        method2(node.right);
    }
}
