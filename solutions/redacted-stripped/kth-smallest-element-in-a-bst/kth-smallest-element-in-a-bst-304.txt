class Solution {

    int[] ans;

    int count;

    public int method1(TreeNode root, int k) {
        ans = new int[1];
        count = k;
        method2(root);
        return ans[0];
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        count--;
        if (count == 0) {
            ans[0] = root.val;
            return;
        }
        method2(root.right);
    }
}
