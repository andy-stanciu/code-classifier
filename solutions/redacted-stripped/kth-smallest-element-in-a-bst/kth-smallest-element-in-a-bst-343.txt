class Solution {

    public int method1(TreeNode root, int k) {
        TreeNode curr = root;
        TreeNode pred = null;
        while (curr != null) {
            if (curr.left == null) {
                if (k == 1)
                    return curr.val;
                k--;
                curr = curr.right;
            } else {
                pred = getPredecessor(curr);
                if (pred.right == null) {
                    pred.right = curr;
                    curr = curr.left;
                } else {
                    pred.right = null;
                    if (k == 1)
                        return curr.val;
                    k--;
                    curr = curr.right;
                }
            }
        }
        return 0;
    }

    private TreeNode getPredecessor(TreeNode root) {
        TreeNode curr = root.left;
        while (curr.right != null && curr.right != root) curr = curr.right;
        return curr;
    }
}
