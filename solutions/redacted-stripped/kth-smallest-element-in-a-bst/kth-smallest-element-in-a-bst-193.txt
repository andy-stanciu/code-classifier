class Solution {

    private void solve(TreeNode node, List<Integer> ans) {
        if (node == null)
            return;
        solve(node.left, ans);
        ans.add(node.val);
        solve(node.right, ans);
    }

    public int method1(TreeNode root, int k) {
        List<Integer> ans = new ArrayList<>();
        solve(root, ans);
        return ans.get(k - 1);
    }
}
