class Solution {

    int count = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return -1;
        int left = method1(root.left, k);
        if (left != -1)
            return left;
        count++;
        if (count == k)
            return root.val;
        int right = method1(root.right, k);
        if (right != -1)
            return right;
        return -1;
    }
}
