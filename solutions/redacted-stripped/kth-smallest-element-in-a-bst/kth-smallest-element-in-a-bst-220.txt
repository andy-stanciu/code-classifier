class Solution {

    int count = 0;

    int out = 0;

    public void method1(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method1(root.left, k);
        count++;
        if (count == k) {
            out = root.val;
            return;
        }
        method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return out;
    }
}
