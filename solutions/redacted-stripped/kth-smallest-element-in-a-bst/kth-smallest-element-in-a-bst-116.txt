class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> lst = new ArrayList<>();
        return method2(root, lst, k);
    }

    public int method2(TreeNode root, List<Integer> arr, int k) {
        if (root == null) {
            return -1;
        }
        int left = method2(root.left, arr, k);
        if (left != -1) {
            return left;
        }
        arr.add(root.val);
        if (arr.size() == k) {
            return root.val;
        }
        return method2(root.right, arr, k);
    }
}
