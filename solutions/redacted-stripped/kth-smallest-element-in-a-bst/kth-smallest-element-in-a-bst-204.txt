class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> l = new ArrayList<>();
        method2(root, l);
        return l.get(k - 1);
    }

    private void method2(TreeNode node, List<Integer> l) {
        if (node == null)
            return;
        method2(node.left, l);
        l.add(node.val);
        method2(node.right, l);
    }
}
