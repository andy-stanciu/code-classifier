class Solution {

    int kGlobal;

    public int method1(TreeNode root, int k) {
        kGlobal = k;
        return answer(root);
    }

    private int answer(TreeNode root) {
        if (root == null) {
            return -1;
        }
        int left = answer(root.left);
        if (left != -1) {
            return left;
        }
        kGlobal--;
        if (kGlobal == 0) {
            return root.val;
        }
        int right = answer((root.right));
        if (right != -1) {
            return right;
        }
        return -1;
    }
}
