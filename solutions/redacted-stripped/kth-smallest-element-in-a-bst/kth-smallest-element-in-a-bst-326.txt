class Solution {

    private int count = 0;

    private int result = -1;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        method1(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return result;
        }
        method1(root.right, k);
        return result;
    }
}
