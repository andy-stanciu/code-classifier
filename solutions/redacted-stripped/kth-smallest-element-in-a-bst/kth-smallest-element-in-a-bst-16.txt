class Solution {

    int ans = 0;

    int x;

    public int method1(TreeNode root, int k) {
        x = k;
        solve(root);
        return ans;
    }

    public void solve(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null)
            solve(root.left);
        x--;
        if (x == 0)
            ans = root.val;
        if (root.right != null)
            solve(root.right);
    }
}
