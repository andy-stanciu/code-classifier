class Solution {

    private void method1(TreeNode node, List<Integer> arr) {
        if (node == null) {
            return;
        }
        method1(node.left, arr);
        arr.add(node.val);
        method1(node.right, arr);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> arr = new ArrayList<>();
        method1(root, arr);
        return arr.get(k - 1);
    }
}
