class Solution {

    public int method1(TreeNode root, int k) {
        TreeNode current = root;
        Stack<TreeNode> stack = new Stack<TreeNode>();
        while (current != null || !stack.isEmpty()) {
            while (current != null) {
                stack.push(current);
                current = current.left;
            }
            current = stack.pop();
            if (--k == 0)
                return current.val;
            current = current.right;
        }
        return 0;
    }
}
