class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> var1 = new ArrayList<>();
        compute(root, var1);
        return var1.get(k - 1);
    }

    public void compute(TreeNode root, List<Integer> var1) {
        if (null == root) {
            return;
        }
        compute(root.left, var1);
        var1.add(root.val);
        compute(root.right, var1);
    }
}
