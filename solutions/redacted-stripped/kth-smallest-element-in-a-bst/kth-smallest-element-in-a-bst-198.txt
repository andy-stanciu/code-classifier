class Solution {

    int count = 0;

    int out = 0;

    public int method1(TreeNode root, int k) {
        recur(root, k);
        return out;
    }

    public void recur(TreeNode root, int k) {
        if (root.left != null) {
            recur(root.left, k);
        }
        count++;
        if (count == k) {
            out = root.val;
            return;
        }
        if (root.right != null) {
            recur(root.right, k);
        }
    }
}
