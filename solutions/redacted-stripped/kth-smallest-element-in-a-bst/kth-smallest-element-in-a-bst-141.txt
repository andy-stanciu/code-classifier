class Solution {

    int count = 0;

    int value;

    public int method1(TreeNode root, int k) {
        calc(root, k);
        return value;
    }

    public void calc(TreeNode node, int k) {
        if (node == null)
            return;
        calc(node.left, k);
        count++;
        if (count == k) {
            value = node.val;
        }
        calc(node.right, k);
    }
}
