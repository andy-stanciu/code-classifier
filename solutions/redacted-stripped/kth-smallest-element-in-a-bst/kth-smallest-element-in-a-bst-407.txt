class Solution {

    public List<Integer> pre(TreeNode root, List<Integer> l) {
        if (root != null) {
            l = pre(root.left, l);
            l.add(root.val);
            l = pre(root.right, l);
        }
        return l;
    }

    public int method1(TreeNode root, int k) {
        List<Integer> l = new ArrayList<>();
        l = pre(root, l);
        return l.get(k - 1);
    }
}
