class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        Stack<Integer> stackValues = new Stack<>();
        method2(stackValues, root);
        return stackValues.get(k - 1);
    }

    public void method2(Stack<Integer> stackValues, TreeNode root) {
        if (root == null)
            return;
        method2(stackValues, root.left);
        stackValues.push(root.val);
        method2(stackValues, root.right);
    }
}
