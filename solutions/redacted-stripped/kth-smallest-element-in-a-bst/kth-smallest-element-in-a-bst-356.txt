class Solution {

    int count = 0;

    private TreeNode helper(TreeNode root, int k) {
        if (root == null) {
            return null;
        }
        TreeNode leftNode = helper(root.left, k);
        if (leftNode != null) {
            return leftNode;
        }
        count++;
        if (k == count) {
            return root;
        }
        return helper(root.right, k);
    }

    public int method1(TreeNode root, int k) {
        return helper(root, k).val;
    }
}
