class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        traverse(root, pq);
        int val = k - 1;
        for (int i = 0; i < val; i++) {
            pq.poll();
        }
        return pq.poll();
    }

    void traverse(TreeNode node, PriorityQueue<Integer> pq) {
        if (node == null)
            return;
        pq.offer(node.val);
        traverse(node.left, pq);
        traverse(node.right, pq);
    }
}
