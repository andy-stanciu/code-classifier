class Solution {

    public int count = 0;

    public int res = -1;

    public int method1(TreeNode root, int k) {
        count = 0;
        res = -1;
        method2(root, k);
        return res;
    }

    public void method2(TreeNode root, int k) {
        if (root != null) {
            method2(root.left, k);
            count++;
            if (count == k) {
                res = root.val;
                return;
            }
            method2(root.right, k);
        }
    }
}
