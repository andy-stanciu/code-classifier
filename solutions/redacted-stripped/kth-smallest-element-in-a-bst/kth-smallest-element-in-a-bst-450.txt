class Solution {

    int count = 0;

    int var1 = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return var1;
    }

    public void method2(TreeNode root, int k) {
        if (root == null || count >= k) {
            return;
        }
        method2(root.left, k);
        count++;
        if (count == k) {
            var1 = root.val;
        }
        method2(root.right, k);
    }
}
