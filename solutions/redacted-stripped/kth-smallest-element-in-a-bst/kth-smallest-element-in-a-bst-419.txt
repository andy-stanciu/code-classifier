class Solution {

    ArrayList<Integer> list = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        int n = list.size();
        method2(root);
        return list.get(k - 1);
    }

    void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        list.add(root.val);
        method2(root.right);
    }
}
