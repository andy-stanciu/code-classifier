class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> ds = new ArrayList<>();
        method2(root, ds);
        return ds.get(k - 1);
    }

    void method2(TreeNode root, List<Integer> ds) {
        if (root == null)
            return;
        method2(root.left, ds);
        ds.add(root.val);
        method2(root.right, ds);
    }
}
