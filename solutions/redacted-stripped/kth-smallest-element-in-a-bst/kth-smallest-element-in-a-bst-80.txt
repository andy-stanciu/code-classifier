class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        traversal(root, list);
        return list.get(k - 1);
    }

    public void traversal(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        traversal(root.left, list);
        list.add(root.val);
        traversal(root.right, list);
    }
}
