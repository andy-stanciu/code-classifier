class Solution {

    public void method1(ArrayList<Integer> arr, TreeNode root) {
        if (root == null) {
            return;
        }
        method1(arr, root.left);
        arr.add(root.val);
        method1(arr, root.right);
    }

    public int method2(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        method1(arr, root);
        return arr.get(k - 1);
    }
}
