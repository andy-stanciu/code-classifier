class Solution {

    int count = 0, result = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method1(root.left, k);
        count++;
        if (count == k)
            result = root.val;
        else
            method1(root.right, k);
        return result;
    }
}
