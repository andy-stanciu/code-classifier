class Solution {

    int k;

    int ans = 0;

    public int method1(TreeNode root, int k) {
        this.k = k;
        helper(root);
        return ans;
    }

    public void helper(TreeNode root) {
        if (root == null) {
            return;
        }
        helper(root.left);
        k--;
        if (k == 0) {
            ans = root.val;
            return;
        }
        helper(root.right);
    }
}
