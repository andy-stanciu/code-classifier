class Solution {

    public void helper(TreeNode root, List<Integer> arr) {
        if (root == null)
            return;
        helper(root.left, arr);
        arr.add(root.val);
        helper(root.right, arr);
    }

    public int method1(TreeNode root, int k) {
        List<Integer> arr = new ArrayList<>();
        helper(root, arr);
        return arr.get(k - 1);
    }
}
