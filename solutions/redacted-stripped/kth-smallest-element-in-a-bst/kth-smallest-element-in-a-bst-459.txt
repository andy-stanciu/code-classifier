class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null)
            return -1;
        else {
            int ans = method1(root.left, k);
            if (ans != -1)
                return ans;
            this.count++;
            if (this.count == k)
                return root.val;
            else
                return method1(root.right, k);
        }
    }

    private int count = 0;
}
