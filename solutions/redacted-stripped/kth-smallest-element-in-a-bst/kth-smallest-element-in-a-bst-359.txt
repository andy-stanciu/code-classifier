class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans.get(k - 1);
    }

    public static void method2(TreeNode root, ArrayList<Integer> ans) {
        if (root == null) {
            return;
        }
        method2(root.left, ans);
        ans.add(root.val);
        method2(root.right, ans);
    }
}
