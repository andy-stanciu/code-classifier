class Solution {

    int globalk = 0;

    public int method1(TreeNode root, int k) {
        globalk = k;
        int value = method2(root);
        return value;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return -1;
        }
        int getLeft = method2(root.left);
        if (--globalk == 0) {
            return root.val;
        }
        int getRight = method2(root.right);
        if (getLeft != -1) {
            return getLeft;
        }
        if (getRight != -1) {
            return getRight;
        }
        return -1;
    }
}
