class Solution {

    private int k = 0;

    public int method1(TreeNode root, int k) {
        this.k = k;
        TreeNode var1 = method2(root);
        return var1.val;
    }

    private TreeNode method2(TreeNode node) {
        if (node.left != null) {
            TreeNode leftNode = method2(node.left);
            if (leftNode != null) {
                return leftNode;
            }
        }
        k--;
        if (k == 0) {
            return node;
        }
        if (node.right != null) {
            return method2(node.right);
        }
        return null;
    }
}
