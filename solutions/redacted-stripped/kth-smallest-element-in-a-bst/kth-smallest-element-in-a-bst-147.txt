class Solution {

    int var1 = 0;

    int mval = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return mval;
    }

    void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        var1++;
        if (var1 == k)
            mval = root.val;
        method2(root.right, k);
    }
}
