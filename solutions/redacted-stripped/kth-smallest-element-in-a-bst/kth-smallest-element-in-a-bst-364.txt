class Solution {

    public int dfs(TreeNode root) {
        if (root == null)
            return 0;
        return dfs(root.left) + dfs(root.right) + 1;
    }

    int i = 0;

    public void method1(TreeNode root, int[] arr) {
        if (root == null)
            return;
        arr[i++] = root.val;
        method1(root.left, arr);
        method1(root.right, arr);
    }

    public int method2(TreeNode root, int k) {
        int len = dfs(root);
        int[] arr = new int[len];
        method1(root, arr);
        Arrays.sort(arr);
        return arr[k - 1];
    }
}
