class Solution {

    public void method1(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        method1(root.left, ans);
        ans.add(root.val);
        method1(root.right, ans);
    }

    public int method2(TreeNode root, int k) {
        ArrayList<Integer> ans = new ArrayList();
        method1(root, ans);
        return ans.get(k - 1);
    }
}
