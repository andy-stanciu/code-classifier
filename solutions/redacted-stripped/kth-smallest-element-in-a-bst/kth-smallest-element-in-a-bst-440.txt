class Solution {

    private int cnt;

    private int result;

    public int method1(TreeNode root, int k) {
        cnt = k;
        visit(root);
        return result;
    }

    private void visit(TreeNode root) {
        if (root == null) {
            return;
        }
        visit(root.left);
        cnt--;
        if (cnt == 0) {
            result = root.val;
            return;
        } else if (cnt < 0) {
            return;
        }
        visit(root.right);
    }
}
