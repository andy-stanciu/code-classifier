class Solution {

    public int method1(TreeNode root, int k) {
        int count = 0;
        Stack<TreeNode> stack = new Stack<>();
        TreeNode curr = root;
        while (!stack.isEmpty() || curr != null) {
            if (curr != null) {
                stack.push(curr);
                curr = curr.left;
            } else {
                curr = stack.pop();
                count++;
                if (count == k) {
                    break;
                }
                curr = curr.right;
            }
        }
        return curr.val;
    }
}
