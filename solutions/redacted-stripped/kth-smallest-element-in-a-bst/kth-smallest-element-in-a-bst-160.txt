class Solution {

    public int method1(TreeNode root, int k) {
        int[] small = new int[1];
        f(root, k, small);
        return small[0];
    }

    int c = 0;

    public void f(TreeNode root, int k, int[] small) {
        if (root == null)
            return;
        f(root.left, k, small);
        c++;
        if (c == k) {
            small[0] = root.val;
        }
        f(root.right, k, small);
    }
}
