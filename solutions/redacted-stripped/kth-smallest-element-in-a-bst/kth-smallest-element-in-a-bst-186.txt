class Solution {

    int count = 0;

    int ans = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode curr, int k) {
        if (curr == null)
            return;
        method2(curr.left, k);
        count++;
        if (count == k)
            ans = curr.val;
        method2(curr.right, k);
    }
}
