class Solution {

    int ans = 0;

    int i = 0;

    public void method1(TreeNode root, int k) {
        if (root == null)
            return;
        method1(root.left, k);
        i++;
        if (i == k) {
            ans = root.val;
            return;
        }
        method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return ans;
    }
}
