class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        method2(root, list);
        int idx = k - 1;
        return list.get(idx);
    }

    public static void method2(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        method2(root.left, list);
        list.add(root.val);
        method2(root.right, list);
    }
}
