class Solution {

    private int var1;

    private int var2;

    public void method1(TreeNode curNode, int k) {
        if (curNode == null) {
            return;
        }
        method1(curNode.left, k);
        var2++;
        if (var2 == k) {
            var1 = curNode.val;
            return;
        }
        method1(curNode.right, k);
    }

    public int method2(TreeNode root, int k) {
        var2 = 0;
        method1(root, k);
        return var1;
    }
}
