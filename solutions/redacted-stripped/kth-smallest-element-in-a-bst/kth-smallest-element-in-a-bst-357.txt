class Solution {

    public int method1(TreeNode root, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        Queue<TreeNode> queue = new LinkedList<>();
        TreeNode temp = null;
        queue.offer(root);
        while (!queue.isEmpty()) {
            temp = queue.poll();
            pq.offer(temp.val);
            if (pq.size() > k)
                pq.poll();
            if (null != temp.left)
                queue.offer(temp.left);
            if (null != temp.right)
                queue.offer(temp.right);
        }
        return pq.poll();
    }
}
