class Solution {

    private int counter = 0;

    private int var1 = 0;

    public int method1(TreeNode root, int k) {
        recursive(root, k);
        return var1;
    }

    private void recursive(TreeNode node, int k) {
        if (node == null || counter >= k)
            return;
        recursive(node.left, k);
        if (++counter == k) {
            var1 = node.val;
            return;
        }
        recursive(node.right, k);
    }
}
