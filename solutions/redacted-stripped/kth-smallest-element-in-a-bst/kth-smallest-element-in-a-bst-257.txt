class Solution {

    int cnt = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return -1;
        int l = method1(root.left, k);
        if (l != -1)
            return l;
        cnt += 1;
        if (cnt == k)
            return root.val;
        int r = method1(root.right, k);
        if (r != -1)
            return r;
        return -1;
    }
}
