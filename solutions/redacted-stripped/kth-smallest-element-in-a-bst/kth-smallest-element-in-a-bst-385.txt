class Solution {

    int ans = -1;

    int i = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root.left != null)
            method2(root.left, k);
        i++;
        if (i == k) {
            ans = root.val;
            return;
        }
        if (root.right != null)
            method2(root.right, k);
    }
}
