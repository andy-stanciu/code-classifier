class Solution {

    Queue<TreeNode> q = new LinkedList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        for (int i = 0; i < k - 1; i++) {
            q.poll();
        }
        TreeNode ans = q.poll();
        return ans.val;
    }

    TreeNode method2(TreeNode node) {
        if (node == null)
            return null;
        TreeNode left = method2(node.left);
        q.offer(node);
        TreeNode right = method2(node.right);
        return node;
    }
}
