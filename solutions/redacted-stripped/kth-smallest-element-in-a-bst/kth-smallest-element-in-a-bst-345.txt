class Solution {

    public void method1(TreeNode root, List<Integer> result) {
        if (root != null) {
            method1(root.left, result);
            result.add(root.val);
            method1(root.right, result);
        }
    }

    public int method2(TreeNode root, int k) {
        List<Integer> result = new ArrayList<>();
        method1(root, result);
        return result.get(k - 1);
    }
}
