class Solution {

    public void method1(TreeNode root, ArrayList<Integer> al) {
        if (root == null)
            return;
        method1(root.left, al);
        al.add(root.val);
        method1(root.right, al);
    }

    public int method2(TreeNode root, int k) {
        ArrayList<Integer> al = new ArrayList();
        method1(root, al);
        return al.get(k - 1);
    }
}
