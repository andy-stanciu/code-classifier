class Solution {

    int cnt = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    void method2(TreeNode root, int k) {
        if (root != null) {
            method2(root.left, k);
            cnt++;
            if (cnt == k) {
                ans = root.val;
            }
            method2(root.right, k);
        }
    }
}
