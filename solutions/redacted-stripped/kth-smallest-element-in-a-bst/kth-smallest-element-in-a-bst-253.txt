class Solution {

    private int result = -1;

    private int count = 0;

    public void method1(TreeNode n, int k) {
        if (n == null) {
            return;
        }
        method1(n.left, k);
        count++;
        if (count == k) {
            result = n.val;
            return;
        }
        method1(n.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return result;
    }
}
