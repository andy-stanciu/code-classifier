class Solution {

    int count;

    int result;

    public int method1(TreeNode root, int k) {
        count = 0;
        result = 0;
        method2(root, k);
        return result;
    }

    public void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        count += 1;
        if (count == k) {
            result = root.val;
        }
        method2(root.right, k);
    }
}
