class Solution {

    int small = 0;

    int k;

    public int method1(TreeNode root, int k) {
        this.k = k;
        if (traverse(root, 1) >= k)
            return small;
        else
            return -1;
    }

    int traverse(TreeNode node, int param1) {
        if (node == null) {
            return param1;
        }
        if (param1 > k)
            return param1;
        param1 = traverse(node.left, param1);
        if (param1 == k) {
            small = node.val;
            return param1 + 1;
        }
        return traverse(node.right, param1 + 1);
    }
}
