class Solution {

    public int method1(TreeNode root, int k) {
        int[] ansArray = new int[2];
        method2(root, ansArray, k);
        ansArray[0] = 1;
        return ansArray[1];
    }

    public static void method2(TreeNode root, int[] ansArray, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, ansArray, k);
        ansArray[0] += 1;
        if (ansArray[0] == k) {
            ansArray[1] = root.val;
        }
        method2(root.right, ansArray, k);
    }
}
