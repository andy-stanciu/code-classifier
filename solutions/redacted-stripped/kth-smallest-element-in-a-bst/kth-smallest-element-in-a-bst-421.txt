class Solution {

    int count = 0;

    int ans;

    boolean found = false;

    public int method1(TreeNode root, int k) {
        util(root, k);
        return ans;
    }

    private void util(TreeNode root, int k) {
        if (!found && root != null) {
            util(root.left, k);
            count++;
            if (count == k) {
                ans = root.val;
                found = true;
            }
            util(root.right, k);
        }
    }
}
