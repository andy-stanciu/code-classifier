class Solution {

    ArrayList<Integer> list = new ArrayList<>();

    void helper(TreeNode root) {
        if (root == null)
            return;
        helper(root.left);
        list.add(root.val);
        helper(root.right);
    }

    public int method1(TreeNode root, int k) {
        helper(root);
        return list.get(k - 1);
    }
}
