class Solution {

    public static int result;

    public static int temp;

    public static void helper(TreeNode root) {
        if (root == null || temp == 0) {
            return;
        }
        helper(root.left);
        if (temp > 0) {
            result = root.val;
            temp--;
        }
        helper(root.right);
    }

    public int method1(TreeNode root, int k) {
        result = 0;
        temp = k;
        helper(root);
        return result;
    }
}
