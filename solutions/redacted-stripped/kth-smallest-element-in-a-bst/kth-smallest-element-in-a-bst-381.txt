class Solution {

    public int method1(TreeNode root, int k) {
        Stack<TreeNode> s = new Stack<>();
        TreeNode curr = root;
        while (curr != null || s.size() > 0) {
            while (curr != null) {
                s.push(curr);
                curr = curr.left;
            }
            curr = s.pop();
            k--;
            if (k == 0)
                return curr.val;
            curr = curr.right;
        }
        return -1;
    }
}
