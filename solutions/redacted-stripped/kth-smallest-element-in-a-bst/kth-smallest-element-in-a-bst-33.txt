class Solution {

    int ans = Integer.MAX_VALUE;

    int count = 1;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method1(root.left, k);
        if (count == k) {
            ans = root.val;
        }
        count++;
        method1(root.right, k);
        return ans;
    }
}
