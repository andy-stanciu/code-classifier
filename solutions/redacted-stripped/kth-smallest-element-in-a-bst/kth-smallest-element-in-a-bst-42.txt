class Solution {

    int kCounter = 0;

    int res = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return res;
    }

    public void method2(TreeNode node, int k) {
        if (node != null) {
            method2(node.left, k);
            kCounter++;
            if (kCounter == k) {
                res = node.val;
            }
            method2(node.right, k);
        }
    }
}
