class Solution {

    int count = 0;

    int ans = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        method1(root.left, k);
        count++;
        if (count == k) {
            ans = root.val;
        }
        method1(root.right, k);
        return ans;
    }

    public int method2(TreeNode root, int k) {
        return method1(root, k);
    }
}
