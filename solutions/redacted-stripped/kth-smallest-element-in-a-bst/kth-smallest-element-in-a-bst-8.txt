class Solution {

    List<Integer> values = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        return values.get(k - 1);
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        values.add(root.val);
        method2(root.right);
    }
}
