class Solution {

    int count;

    private int helper(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        int left = helper(root.left, k);
        if (left != 0) {
            return left;
        }
        count++;
        if (count == k) {
            return root.val;
        }
        return helper(root.right, k);
    }

    public int method1(TreeNode root, int k) {
        count = 0;
        return helper(root, k);
    }
}
