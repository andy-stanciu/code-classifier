class Solution {

    int ans = 0, count = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        count++;
        if (k == count) {
            ans = root.val;
            return;
        }
        method2(root.right, k);
    }
}
