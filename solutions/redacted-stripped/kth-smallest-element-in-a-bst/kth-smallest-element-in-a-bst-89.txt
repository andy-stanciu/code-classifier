class Solution {

    private int count = 0, ans = -1;

    public int method1(TreeNode root, int k) {
        count = 0;
        ans = -1;
        method2(root, k);
        return ans;
    }

    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        count++;
        if (count == k) {
            ans = root.val;
            return;
        }
        method2(root.right, k);
    }
}
