class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> h = new ArrayList<>();
        check(root, h);
        Collections.sort(h);
        for (int i : h) {
            if (k == 1) {
                return i;
            }
            k--;
        }
        return 0;
    }

    public void check(TreeNode root, List<Integer> h) {
        if (root == null) {
            return;
        }
        h.add(root.val);
        check(root.left, h);
        check(root.right, h);
        return;
    }
}
