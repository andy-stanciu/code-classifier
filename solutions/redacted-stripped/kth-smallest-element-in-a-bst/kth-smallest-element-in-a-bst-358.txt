class Solution {

    int i, res = Integer.MAX_VALUE;

    public int method1(TreeNode root, int k) {
        i = k;
        helper(root);
        return res;
    }

    private void helper(TreeNode root) {
        if (res != Integer.MAX_VALUE || root == null)
            return;
        helper(root.left);
        if (i-- == 1)
            res = root.val;
        helper(root.right);
    }
}
