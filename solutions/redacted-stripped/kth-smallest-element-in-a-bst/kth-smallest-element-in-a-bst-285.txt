class Solution {

    int count = 0;

    public int method1(TreeNode root, int k) {
        int left = -1, right = -1;
        if (root.left != null)
            left = method1(root.left, k);
        count++;
        if (count == k)
            return root.val;
        if (root.right != null)
            right = method1(root.right, k);
        if (left != -1)
            return left;
        if (right != -1)
            return right;
        return -1;
    }
}
