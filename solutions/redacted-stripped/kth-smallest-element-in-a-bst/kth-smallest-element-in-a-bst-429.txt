class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null)
            return -1;
        ArrayList<Integer> ans = new ArrayList<>();
        pre(root, ans);
        return ans.get(k - 1);
    }

    public void pre(TreeNode root, ArrayList<Integer> p) {
        if (root == null)
            return;
        pre(root.left, p);
        p.add(root.val);
        pre(root.right, p);
    }
}
