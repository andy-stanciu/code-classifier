class Solution {

    int c = 0, res = 0;

    public int method1(TreeNode root, int k) {
        return helper(root, k);
    }

    int helper(TreeNode root, int target) {
        if (root == null)
            return 0;
        helper(root.left, target);
        c++;
        if (c == target)
            res = root.val;
        helper(root.right, target);
        return res;
    }
}
