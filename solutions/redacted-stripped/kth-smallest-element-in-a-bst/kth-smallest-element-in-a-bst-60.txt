class Solution {

    public int count = 0;

    public boolean found;

    public int res;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        if (!found) {
            res = method1(root.left, k);
        }
        count = count + 1;
        if (count == k) {
            found = true;
            return root.val;
        }
        if (!found) {
            res = method1(root.right, k);
        }
        return res;
    }
}
