class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> li = new ArrayList<>();
        if (root == null)
            return 0;
        method2(root, li);
        return li.get(k - 1);
    }

    public void method2(TreeNode node, List<Integer> li) {
        if (node == null)
            return;
        method2(node.left, li);
        li.add(node.val);
        method2(node.right, li);
    }
}
