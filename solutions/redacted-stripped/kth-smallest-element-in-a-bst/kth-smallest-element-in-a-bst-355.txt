class Solution {

    int c = 0;

    int val = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return val;
    }

    public void method2(TreeNode root, int k) {
        if (root.left != null) {
            method1(root.left, k);
        }
        c += 1;
        if (k == c) {
            val = root.val;
            return;
        }
        if (root.right != null) {
            method1(root.right, k);
        }
    }
}
