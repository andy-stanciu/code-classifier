class Solution {

    int ans;

    int counter;

    void solve(TreeNode root, int k) {
        if (root == null)
            return;
        if (root.left != null)
            solve(root.left, k);
        counter++;
        if (counter == k)
            ans = root.val;
        if (root.right != null)
            solve(root.right, k);
    }

    public int method1(TreeNode root, int k) {
        counter = 0;
        ans = 0;
        solve(root, k);
        return ans;
    }
}
