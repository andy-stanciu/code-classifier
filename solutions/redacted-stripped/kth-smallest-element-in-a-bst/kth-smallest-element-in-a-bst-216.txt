class Solution {

    ArrayList<Integer> ans;

    public void method1(TreeNode root, int k) {
        if (root != null) {
            method1(root.left, k);
            ans.add(root.val);
            method1(root.right, k);
        }
    }

    public int method2(TreeNode root, int k) {
        ans = new ArrayList<Integer>();
        this.method1(root, k);
        return ans.get(k - 1);
    }
}
