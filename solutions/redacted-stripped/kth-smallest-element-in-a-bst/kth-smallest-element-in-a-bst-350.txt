class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> a = new ArrayList<>();
        method2(root, a);
        return a.get(k - 1);
    }

    private void method2(TreeNode node, ArrayList<Integer> a) {
        if (node == null) {
            return;
        }
        method2(node.left, a);
        a.add(node.val);
        method2(node.right, a);
    }
}
