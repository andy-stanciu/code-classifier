class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> arr = new ArrayList<>();
        trav(root, arr, k);
        return arr.get(k - 1);
    }

    public void trav(TreeNode root, List<Integer> arr, int k) {
        if (root == null) {
            return;
        }
        if (arr.size() <= k) {
            trav(root.left, arr, k);
            arr.add(root.val);
            trav(root.right, arr, k);
        }
    }
}
