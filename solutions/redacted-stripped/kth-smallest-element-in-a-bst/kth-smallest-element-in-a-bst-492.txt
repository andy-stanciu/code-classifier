class Solution {

    int result = 0;

    int k;

    public int method1(TreeNode root, int k) {
        this.k = k;
        method2(root);
        return result;
    }

    void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        if (--k == 0) {
            result = root.val;
            return;
        }
        method2(root.right);
    }
}
