class Solution {

    private int counter = 0;

    private int result;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return result;
    }

    public boolean method2(TreeNode root, int k) {
        if (root == null)
            return false;
        if (method2(root.left, k))
            return true;
        counter++;
        if (counter == k) {
            result = root.val;
            return true;
        }
        if (method2(root.right, k))
            return true;
        return false;
    }
}
