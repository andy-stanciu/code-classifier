class Solution {

    private int count = 0;

    private int result = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return result;
    }

    public void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        method2(root.right, k);
    }
}
