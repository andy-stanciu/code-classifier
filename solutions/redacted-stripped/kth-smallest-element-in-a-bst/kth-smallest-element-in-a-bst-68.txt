class Solution {

    private void method1(TreeNode root, ArrayList<Integer> arr, int k) {
        if (root == null) {
            return;
        }
        if (arr.size() >= k)
            return;
        method1(root.left, arr, k);
        arr.add(root.val);
        method1(root.right, arr, k);
    }

    public int method2(TreeNode root, int k) {
        if (root == null)
            return -1;
        ArrayList<Integer> arr = new ArrayList<>();
        method1(root, arr, k);
        return arr.get(k - 1);
    }
}
