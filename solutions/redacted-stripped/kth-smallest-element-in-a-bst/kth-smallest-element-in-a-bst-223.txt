class Solution {

    int c = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return ans;
        method1(root.left, k);
        c++;
        if (c == k)
            ans = root.val;
        return c < k ? method1(root.right, k) : ans;
    }
}
