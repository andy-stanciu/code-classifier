class Solution {

    int count = 0;

    int ans = 0;

    public void method1(TreeNode root, int k) {
        if (root.left != null) {
            method1(root.left, k);
        }
        count++;
        if (count == k) {
            ans = root.val;
            return;
        }
        if (root.right != null) {
            method1(root.right, k);
        }
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return ans;
    }
}
