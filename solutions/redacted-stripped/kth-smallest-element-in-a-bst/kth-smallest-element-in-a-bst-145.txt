class Solution {

    int ans = 0;

    int d = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root != null) {
            method2(root.left, k);
            d++;
            if (d == k)
                ans = root.val;
            method2(root.right, k);
        }
    }
}
