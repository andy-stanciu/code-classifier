class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        helper(root, list);
        Collections.sort(list);
        return list.get(k - 1);
    }

    public void helper(TreeNode node, ArrayList<Integer> list) {
        if (node == null)
            return;
        list.add(node.val);
        helper(node.left, list);
        helper(node.right, list);
    }
}
