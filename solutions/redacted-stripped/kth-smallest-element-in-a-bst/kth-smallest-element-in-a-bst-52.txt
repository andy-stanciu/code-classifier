class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> res = new ArrayList<>();
        method2(root, res);
        return res.get(k - 1);
    }

    public void method2(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        method2(root.left, res);
        res.add(root.val);
        method2(root.right, res);
    }
}
