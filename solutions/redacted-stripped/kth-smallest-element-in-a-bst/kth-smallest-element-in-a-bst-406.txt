class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        traversal(root, arr);
        int ans = arr.get(k - 1);
        return ans;
    }

    private static void traversal(TreeNode root, ArrayList<Integer> arr) {
        if (root == null)
            return;
        traversal(root.left, arr);
        arr.add(root.val);
        traversal(root.right, arr);
    }
}
