class Solution {

    public void method1(TreeNode root, List<Integer> arr) {
        if (root == null)
            return;
        method1(root.left, arr);
        arr.add(root.val);
        method1(root.right, arr);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> arr = new ArrayList<>();
        method1(root, arr);
        return arr.get(k - 1);
    }
}
