class Solution {

    private int val = 0;

    int count = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return val;
    }

    public void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        count++;
        if (count == k) {
            val = root.val;
        }
        method2(root.right, k);
    }
}
