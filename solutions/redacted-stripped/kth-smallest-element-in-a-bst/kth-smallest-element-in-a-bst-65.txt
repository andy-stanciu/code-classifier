class Solution {

    List<Integer> list = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        help(root, k);
        return list.get(k - 1);
    }

    public void help(TreeNode root, int k) {
        if (root == null)
            return;
        help(root.left, k);
        list.add(root.val);
        if (list.size() == k)
            return;
        help(root.right, k);
    }
}
