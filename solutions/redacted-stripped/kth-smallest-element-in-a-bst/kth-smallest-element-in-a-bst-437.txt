class Solution {

    ArrayList<Integer> arr = new ArrayList<>();

    void solve(TreeNode root) {
        if (root != null) {
            solve(root.left);
            arr.add(root.val);
            solve(root.right);
        }
    }

    public int method1(TreeNode root, int k) {
        solve(root);
        return arr.get(k - 1);
    }
}
