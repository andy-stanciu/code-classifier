class Solution {

    List<Integer> ans = new ArrayList<Integer>();

    public int method1(TreeNode root, int k) {
        helper(root);
        return ans.get(k - 1);
    }

    public void helper(TreeNode temp) {
        if (temp == null)
            return;
        helper(temp.left);
        ans.add(temp.val);
        helper(temp.right);
    }
}
