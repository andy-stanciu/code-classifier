class Solution {

    public int method1(TreeNode root, int k) {
        Stack<TreeNode> st = new Stack<>();
        while (root != null || !st.isEmpty()) {
            while (root != null) {
                st.push(root);
                root = root.left;
            }
            TreeNode n = st.pop();
            k--;
            if (k == 0)
                return n.val;
            root = n.right;
        }
        return -1;
    }
}
