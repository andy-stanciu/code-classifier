class Solution {

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        List<Integer> al = new ArrayList<>();
        method2(root, al);
        return al.get(k - 1);
    }

    public static void method2(TreeNode root, List<Integer> al) {
        if (root == null) {
            return;
        }
        method2(root.left, al);
        al.add(root.val);
        method2(root.right, al);
    }
}
