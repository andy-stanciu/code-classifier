class Solution {

    private static int k, result;

    static void method1(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        k--;
        if (k == 0)
            result = root.val;
        method1(root.right);
    }

    public int method2(TreeNode root, int k) {
        this.k = k;
        method1(root);
        return result;
    }
}
