class Solution {

    int val;

    int cnt;

    public int method1(TreeNode root, int k) {
        val = 0;
        cnt = 0;
        track(root, k);
        return val;
    }

    public void track(TreeNode node, int k) {
        if (node == null || cnt == k)
            return;
        track(node.left, k);
        if (++cnt == k)
            val = node.val;
        track(node.right, k);
    }
}
