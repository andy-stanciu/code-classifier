class Solution {

    int count = 0;

    int val = 0;

    private int helper(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        helper(root.left, k);
        count++;
        if (count == k) {
            val = root.val;
            return val;
        } else {
            helper(root.right, k);
        }
        return val;
    }

    public int method1(TreeNode root, int k) {
        return helper(root, k);
    }
}
