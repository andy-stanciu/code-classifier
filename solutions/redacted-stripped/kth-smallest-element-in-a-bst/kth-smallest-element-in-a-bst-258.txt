class Solution {

    public int method1(TreeNode root, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        dfs(root, pq, k);
        return pq.poll();
    }

    public void dfs(TreeNode root, PriorityQueue<Integer> pq, int k) {
        if (root == null) {
            return;
        }
        dfs(root.left, pq, k);
        pq.add(root.val);
        if (pq.size() > k) {
            pq.poll();
        }
        dfs(root.right, pq, k);
    }
}
