class Solution {

    int value = -1;

    int cnt = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return value;
    }

    private void method2(TreeNode root, int k) {
        if (root == null || cnt == k) {
            return;
        }
        method2(root.left, k);
        if (cnt < k) {
            value = root.val;
            cnt++;
        }
        method2(root.right, k);
    }
}
