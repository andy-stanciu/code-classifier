class Solution {

    int ans = 0;

    int count = 0;

    public int method1(TreeNode root, int k) {
        preOrder(root, k);
        return ans;
    }

    private void preOrder(TreeNode root, int k) {
        if (root == null)
            return;
        preOrder(root.left, k);
        count++;
        if (count == k) {
            ans = root.val;
            return;
        }
        preOrder(root.right, k);
    }
}
