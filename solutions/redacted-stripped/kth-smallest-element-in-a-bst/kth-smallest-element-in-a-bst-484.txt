class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    private void method2(TreeNode root, List<Integer> result) {
        if (root == null)
            return;
        method2(root.left, result);
        result.add(root.val);
        method2(root.right, result);
    }

    public int method3(TreeNode root, int k) {
        List<Integer> result = method1(root);
        if (k > 0 && k <= result.size()) {
            return result.get(k - 1);
        }
        return -1;
    }
}
