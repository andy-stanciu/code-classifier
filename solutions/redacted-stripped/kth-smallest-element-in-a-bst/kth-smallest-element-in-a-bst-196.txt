class Solution {

    PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Collections.reverseOrder());

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return maxHeap.poll();
    }

    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        maxHeap.add(root.val);
        if (maxHeap.size() > k)
            maxHeap.remove();
        method2(root.right, k);
    }
}
