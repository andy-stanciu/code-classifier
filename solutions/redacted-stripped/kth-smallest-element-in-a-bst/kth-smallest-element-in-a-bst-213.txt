class Solution {

    ArrayList<Integer> ll = new ArrayList<>();

    void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        ll.add(root.val);
        method1(root.right);
    }

    public int method2(TreeNode root, int k) {
        method1(root);
        return ll.get(k - 1);
    }
}
