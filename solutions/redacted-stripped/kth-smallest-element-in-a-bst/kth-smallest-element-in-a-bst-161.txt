class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> ordered = new ArrayList<>();
        method2(root, ordered);
        return ordered.get(k - 1);
    }

    private void method2(TreeNode root, List<Integer> ordered) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            method2(root.left, ordered);
        }
        ordered.add(root.val);
        if (root.right != null) {
            method2(root.right, ordered);
        }
    }
}
