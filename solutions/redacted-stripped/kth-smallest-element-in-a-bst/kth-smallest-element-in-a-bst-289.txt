class Solution {

    int var1 = 0;

    int result = -1;

    public int method1(TreeNode root, int k) {
        solve(root, k);
        return result;
    }

    void solve(TreeNode node, int k) {
        if (node == null)
            return;
        solve(node.left, k);
        var1++;
        if (var1 == k) {
            result = node.val;
            return;
        } else
            solve(node.right, k);
    }
}
