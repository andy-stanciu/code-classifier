class Solution {

    List<Integer> list = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        return list.get(k - 1);
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        list.add(root.val);
        method2(root.right);
    }
}
