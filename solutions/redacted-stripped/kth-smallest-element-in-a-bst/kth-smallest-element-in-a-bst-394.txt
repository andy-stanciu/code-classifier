class Solution {

    int val;

    int required_k;

    int num = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        required_k = k;
        method2(root);
        return val;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        num++;
        if (num == required_k)
            val = root.val;
        method2(root.right);
    }
}
