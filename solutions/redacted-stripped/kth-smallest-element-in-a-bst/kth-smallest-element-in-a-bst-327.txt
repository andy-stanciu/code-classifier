class Solution {

    int count = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        Recurssion(root, k);
        return ans;
    }

    public void Recurssion(TreeNode root, int k) {
        if (root == null)
            return;
        Recurssion(root.left, k);
        count++;
        if (count == k)
            ans = root.val;
        Recurssion(root.right, k);
    }
}
