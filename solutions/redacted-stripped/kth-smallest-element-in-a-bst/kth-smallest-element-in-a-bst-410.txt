class Solution {

    private int k;

    private int result;

    public int method1(TreeNode root, int k) {
        this.k = k;
        method2(root);
        return result;
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        k--;
        if (k == 0) {
            result = root.val;
            return;
        }
        method2(root.right);
    }
}
