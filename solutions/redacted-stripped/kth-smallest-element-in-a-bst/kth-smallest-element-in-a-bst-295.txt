class Solution {

    int var1 = -1;

    int counter = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return var1;
    }

    public void method2(TreeNode root, int k) {
        if (root == null || counter == k)
            return;
        method2(root.left, k);
        counter++;
        if (counter == k)
            var1 = root.val;
        method2(root.right, k);
    }
}
