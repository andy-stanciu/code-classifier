class Solution {

    private int count = 0;

    private int rootValue;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method1(root.left, k);
        this.count += 1;
        if (this.count == k)
            this.rootValue = root.val;
        method1(root.right, k);
        return this.rootValue;
    }
}
