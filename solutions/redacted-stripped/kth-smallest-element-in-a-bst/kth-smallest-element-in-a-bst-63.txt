class Solution {

    private int val = 0;

    private int count = 0;

    public int method1(TreeNode root, int k) {
        dfs(root, k);
        return val;
    }

    public void dfs(TreeNode node, int k) {
        if (node == null)
            return;
        dfs(node.left, k);
        if (++count == k) {
            this.val = node.val;
            return;
        }
        dfs(node.right, k);
    }
}
