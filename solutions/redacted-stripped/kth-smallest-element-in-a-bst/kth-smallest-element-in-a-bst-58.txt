class Solution {

    ArrayList<Integer> ans = new ArrayList<>();

    int c;

    public int method1(TreeNode root, int k) {
        c = k;
        if (root == null)
            return 0;
        method2(root);
        return ans.get(k - 1);
    }

    void method2(TreeNode root) {
        if (root == null || ans.size() > c)
            return;
        method2(root.left);
        if (ans.size() <= c)
            ans.add(root.val);
        method2(root.right);
    }
}
