class Solution {

    private List<Integer> result = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        return result.get(k - 1);
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        result.add(root.val);
        method2(root.right);
    }
}
