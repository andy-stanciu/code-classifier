class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> var1 = new ArrayList<>();
        helper(root, var1);
        return var1.get(k - 1);
    }

    void helper(TreeNode root, ArrayList<Integer> var1) {
        if (root == null) {
            return;
        }
        helper(root.left, var1);
        var1.add(root.val);
        helper(root.right, var1);
    }
}
