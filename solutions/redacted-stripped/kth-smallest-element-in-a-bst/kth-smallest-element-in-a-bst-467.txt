class Solution {

    int count = 0;

    int result = 0;

    public int method1(TreeNode root, int k) {
        count = k;
        method2(root);
        return result;
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        count--;
        if (count == 0) {
            result = root.val;
            return;
        }
        method2(root.right);
    }
}
