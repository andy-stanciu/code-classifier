class Solution {

    ArrayList<Integer> res = new ArrayList<>();

    private void method1(TreeNode root, int k) {
        if (root == null)
            return;
        method1(root.left, k);
        res.add(root.val);
        if (res.size() == k)
            return;
        method1(root.right, k);
        return;
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return res.get(k - 1);
    }
}
