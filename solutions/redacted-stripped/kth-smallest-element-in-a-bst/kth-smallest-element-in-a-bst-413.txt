class Solution {

    private int count;

    private int result;

    public int method1(TreeNode root, int k) {
        count = 0;
        result = 0;
        method2(root, k);
        return result;
    }

    private void method2(TreeNode node, int k) {
        if (node == null)
            return;
        method2(node.left, k);
        count++;
        if (count == k) {
            result = node.val;
            return;
        }
        method2(node.right, k);
    }
}
