class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        return var1.get(k - 1);
    }

    private static void method2(TreeNode node, List<Integer> var1) {
        if (node == null) {
            return;
        }
        method2(node.left, var1);
        var1.add(node.val);
        method2(node.right, var1);
    }
}
