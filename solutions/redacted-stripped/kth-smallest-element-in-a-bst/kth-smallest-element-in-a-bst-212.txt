class Solution {

    public int count = 0;

    public int res;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return res;
    }

    public void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        count++;
        if (count == k) {
            res = root.val;
            return;
        }
        method2(root.right, k);
    }
}
