class Solution {

    public void method1(TreeNode root, int[] i, int[] c, int k) {
        if (root == null || c[0] >= k) {
            return;
        }
        method1(root.left, i, c, k);
        i[0]++;
        if (i[0] == k) {
            c[0] = root.val;
            return;
        }
        method1(root.right, i, c, k);
    }

    public int method2(TreeNode root, int k) {
        int[] c = new int[1];
        method1(root, new int[1], c, k);
        return c[0];
    }
}
