class Solution {

    private int var1;

    private int count = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return var1;
        method1(root.left, k);
        if (++count == k)
            var1 = root.val;
        method1(root.right, k);
        return var1;
    }
}
