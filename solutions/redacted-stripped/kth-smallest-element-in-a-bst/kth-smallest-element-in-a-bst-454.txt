class Solution {

    public int method1(TreeNode root, int k) {
        int left = nodeCount(root.left);
        if (left + 1 == k) {
            return root.val;
        } else if (left + 1 < k) {
            return method1(root.right, k - left - 1);
        } else {
            return method1(root.left, k);
        }
    }

    private static int nodeCount(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + nodeCount(root.left) + nodeCount(root.right);
    }
}
