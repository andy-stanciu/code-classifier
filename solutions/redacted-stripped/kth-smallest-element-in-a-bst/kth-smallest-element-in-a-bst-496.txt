class Solution {

    int var1 = Integer.MAX_VALUE;

    int i = 1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return var1;
    }

    public void method2(TreeNode root, int k) {
        if (root == null || var1 != Integer.MAX_VALUE) {
            return;
        }
        method2(root.left, k);
        if (k == i) {
            var1 = root.val;
        }
        i += 1;
        method2(root.right, k);
    }
}
