class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans.get(k - 1);
    }

    public void method2(TreeNode node, List<Integer> ans) {
        if (node == null) {
            return;
        }
        method2(node.left, ans);
        ans.add(node.val);
        method2(node.right, ans);
    }
}
