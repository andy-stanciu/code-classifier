class Solution {

    private int result = -1;

    private int count = 0;

    public void method1(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method1(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return result;
    }
}
