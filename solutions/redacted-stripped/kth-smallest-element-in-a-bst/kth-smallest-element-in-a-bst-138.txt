class Solution {

    List<Integer> var1 = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        if (var1.size() != 0) {
            return var1.get(k - 1);
        }
        return 0;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        var1.add(root.val);
        method2(root.right);
    }
}
