class Solution {

    public int method1(TreeNode root, int k) {
        int[] stop = new int[1];
        stop[0] = k;
        int ans = traverse(root, stop);
        return ans;
    }

    public static int traverse(TreeNode root, int[] stop) {
        if (root == null)
            return -1;
        int left = traverse(root.left, stop);
        if (left != -1)
            return left;
        stop[0]--;
        if (stop[0] == 0)
            return root.val;
        return traverse(root.right, stop);
    }
}
