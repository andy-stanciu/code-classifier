class Solution {

    int cur = 0;

    int res = -1;

    int t;

    public int method1(TreeNode root, int k) {
        t = k;
        traverse(root);
        return res;
    }

    public void traverse(TreeNode root) {
        if (root == null)
            return;
        traverse(root.left);
        cur++;
        if (t == cur) {
            res = root.val;
            return;
        }
        traverse(root.right);
    }
}
