class Solution {

    private int ans = -1;

    private int counter = 0;

    public int method1(TreeNode root, int k) {
        helper(root, k);
        return ans;
    }

    private void helper(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        helper(root.left, k);
        counter++;
        if (counter == k) {
            ans = root.val;
            return;
        }
        helper(root.right, k);
    }
}
