class Solution {

    private int currIter = 0;

    private int ans = -1;

    private void method1(TreeNode root, int k) {
        if (root != null) {
            if (this.currIter != k) {
                method1(root.left, k);
            }
            if (this.currIter != k) {
                this.ans = root.val;
                this.currIter++;
            }
            if (this.currIter != k) {
                method1(root.right, k);
            }
        }
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return this.ans;
    }
}
