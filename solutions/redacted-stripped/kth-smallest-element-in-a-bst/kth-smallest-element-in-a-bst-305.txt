class Solution {

    int currentK = 0;

    int ans = 0;

    public int method1(TreeNode root, int k) {
        currentK = k;
        method2(root);
        return ans;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        if (currentK > 0) {
            method2(root.left);
            if (currentK > 0)
                ans = root.val;
            currentK--;
            method2(root.right);
        }
    }
}
