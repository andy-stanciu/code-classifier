class Solution {

    public void method1(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        method1(root.left, list);
        list.add(root.val);
        method1(root.right, list);
    }

    public int method2(TreeNode root, int k) {
        if (root == null) {
            return -1;
        }
        List<Integer> list = new ArrayList<>();
        method1(root, list);
        return list.get(k - 1);
    }
}
