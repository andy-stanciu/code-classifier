class Solution {

    public int method1(TreeNode root, int k) {
        Stack<TreeNode> nodeStack = new Stack<>();
        TreeNode iterator = root;
        while (iterator != null || !nodeStack.isEmpty()) {
            while (iterator != null) {
                nodeStack.push(iterator);
                iterator = iterator.left;
            }
            iterator = nodeStack.pop();
            k--;
            if (k == 0) {
                return iterator.val;
            }
            iterator = iterator.right;
        }
        return -1;
    }
}
