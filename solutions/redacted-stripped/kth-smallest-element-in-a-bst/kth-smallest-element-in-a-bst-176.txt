class Solution {

    private void method1(TreeNode node, List<Integer> res) {
        if (node == null) {
            return;
        }
        method1(node.left, res);
        res.add(node.val);
        method1(node.right, res);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> arr = new ArrayList<>();
        method1(root, arr);
        int t = arr.get(k - 1);
        return t;
    }
}
