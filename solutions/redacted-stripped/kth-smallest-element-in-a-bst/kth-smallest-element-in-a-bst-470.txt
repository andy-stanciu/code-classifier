class Solution {

    public int method1(TreeNode root, int k) {
        int cnt = 0;
        TreeNode curr = root;
        while (curr != null) {
            while (curr.left == null) {
                cnt++;
                if (cnt == k)
                    return curr.val;
                curr = curr.right;
            }
            TreeNode leftNode = curr.left;
            TreeNode rightMost = leftNode;
            while (rightMost.right != null && rightMost.right != curr) {
                rightMost = rightMost.right;
            }
            if (rightMost.right == curr) {
                rightMost.right = null;
                cnt++;
                if (cnt == k)
                    return curr.val;
                curr = curr.right;
            } else {
                rightMost.right = curr;
                curr = leftNode;
            }
        }
        return -1;
    }
}
