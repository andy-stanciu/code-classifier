class Solution {

    public int method1(TreeNode root, int k) {
        TreeNode ans = new TreeNode(0);
        method2(root, k, ans);
        return ans.left.val;
    }

    private void method2(TreeNode root, int k, TreeNode ans) {
        if (root == null) {
            return;
        }
        method2(root.left, k, ans);
        ans.val++;
        if (ans.val == k) {
            ans.left = root;
            return;
        }
        method2(root.right, k, ans);
    }
}
