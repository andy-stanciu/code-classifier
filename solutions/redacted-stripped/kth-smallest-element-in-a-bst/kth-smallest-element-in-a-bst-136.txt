class Solution {

    public void solve(TreeNode root, int[] ans, int k) {
        if (root == null) {
            return;
        }
        solve(root.left, ans, k);
        ans[0]++;
        if (ans[0] == k) {
            ans[1] = root.val;
        }
        solve(root.right, ans, k);
    }

    public int method1(TreeNode root, int k) {
        int[] ans = new int[2];
        solve(root, ans, k);
        return ans[1];
    }
}
