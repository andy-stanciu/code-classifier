class Solution {

    public int idx = 0;

    public int ans = 0;

    public void solve(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        solve(root.left, k);
        idx++;
        if (idx == k)
            ans = root.val;
        solve(root.right, k);
    }

    public int method1(TreeNode root, int k) {
        solve(root, k);
        return ans;
    }
}
