class Solution {

    private int var1 = Integer.MAX_VALUE;

    private int globalK;

    public int method1(TreeNode root, int k) {
        this.globalK = k;
        traverse(root);
        return var1;
    }

    private void traverse(TreeNode node) {
        if (node == null)
            return;
        traverse(node.left);
        if (globalK == 1) {
            var1 = node.val;
        }
        globalK--;
        traverse(node.right);
    }
}
