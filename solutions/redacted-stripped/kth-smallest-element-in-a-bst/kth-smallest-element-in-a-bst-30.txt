class Solution {

    int ans = -1;

    int x = 0;

    public void method1(TreeNode root, int k) {
        if (root == null)
            return;
        method1(root.left, k);
        x++;
        if (x == k) {
            ans = root.val;
        }
        method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        method1(root, k);
        return ans;
    }
}
