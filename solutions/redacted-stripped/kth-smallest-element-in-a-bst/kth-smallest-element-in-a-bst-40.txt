class Solution {

    ArrayList<Integer> ls = new ArrayList<>();

    private void method1(TreeNode root, ArrayList ls) {
        if (root == null) {
            return;
        }
        method1(root.left, ls);
        ls.add(root.val);
        method1(root.right, ls);
        return;
    }

    public int method2(TreeNode root, int k) {
        method1(root, ls);
        return ls.get(k - 1);
    }
}
