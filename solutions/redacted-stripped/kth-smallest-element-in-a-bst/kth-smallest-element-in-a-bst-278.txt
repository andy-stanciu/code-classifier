class Solution {

    int count = 0;

    int ele = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ele;
    }

    public void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        if (root != null)
            count++;
        if (count == k) {
            ele = root.val;
            return;
        }
        method2(root.right, k);
    }
}
