class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> nodes = new ArrayList<>();
        method2(root, nodes, k);
        if (nodes.size() == 0 && k > nodes.size())
            return -1;
        else
            return nodes.get(k - 1);
    }

    public void method2(TreeNode root, List<Integer> nodes, int k) {
        if (root == null)
            return;
        method2(root.left, nodes, k);
        if (nodes.size() <= k)
            nodes.add(root.val);
        method2(root.right, nodes, k);
    }
}
