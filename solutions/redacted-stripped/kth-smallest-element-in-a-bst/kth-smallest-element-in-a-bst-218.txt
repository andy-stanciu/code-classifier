class Solution {

    int var1 = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return -1;
        int left = method1(root.left, k);
        var1++;
        if (var1 == k)
            return root.val;
        int right = method1(root.right, k);
        return Math.max(left, right);
    }
}
