class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> a = new ArrayList<>();
        helper(root, a);
        return a.get(k - 1);
    }

    private void helper(TreeNode root, ArrayList<Integer> a) {
        if (root == null) {
            return;
        }
        helper(root.left, a);
        a.add(root.val);
        helper(root.right, a);
    }
}
