class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> var1 = new ArrayList<>();
        var1 = method2(root, var1);
        return var1.get(k - 1);
    }

    private List<Integer> method2(TreeNode root, List<Integer> var1) {
        if (root == null) {
            return var1;
        }
        method2(root.left, var1);
        var1.add(root.val);
        method2(root.right, var1);
        return var1;
    }
}
