class Solution {

    public void method1(int[] nums) {
        int redCount = 0, whiteCount = 0, blueCount = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                redCount++;
            else if (nums[i] == 1)
                whiteCount++;
            else
                blueCount++;
        }
        for (int i = 0; i < nums.length; i++) {
            if (i < redCount)
                nums[i] = 0;
            else if (i >= redCount && i < redCount + whiteCount)
                nums[i] = 1;
            else
                nums[i] = 2;
        }
    }
}
