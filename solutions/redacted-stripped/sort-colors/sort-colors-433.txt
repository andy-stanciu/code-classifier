class Solution {

    public void method1(int[] nums) {
        int one = 0;
        int two = 0;
        int zero = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zero++;
            } else if (nums[i] == 1) {
                one++;
            } else
                two++;
        }
        Arrays.fill(nums, 0, zero, 0);
        Arrays.fill(nums, zero, (zero + one), 1);
        Arrays.fill(nums, (zero + one), nums.length, 2);
    }
}
