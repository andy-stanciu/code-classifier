class Solution {

    public void method1(int[] nums) {
        int[] freq = new int[3];
        int i;
        for (i = 0; i < nums.length; i++) {
            freq[nums[i]]++;
        }
        int k = 0;
        for (i = 0; i < freq.length; i++) {
            while (freq[i]-- > 0) {
                nums[k++] = i;
            }
        }
    }
}
