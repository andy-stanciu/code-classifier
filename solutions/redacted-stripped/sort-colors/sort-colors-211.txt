class Solution {

    public void method1(int[] nums) {
        int n0 = 0, n1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                n0++;
            else if (nums[i] == 1)
                n1++;
        }
        for (int i = 0; i < nums.length; i++) {
            if (n0-- > 0)
                nums[i] = 0;
            else if (n1-- > 0)
                nums[i] = 1;
            else
                nums[i] = 2;
        }
    }
}
