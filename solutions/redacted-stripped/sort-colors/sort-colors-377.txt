class Solution {

    public void method1(int[] nums) {
        int ct0 = 0, ct1 = 0, ct2 = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                ct0++;
            } else if (nums[i] == 1) {
                ct1++;
            } else {
                ct2++;
            }
        }
        int j = 0;
        while (ct0 != 0) {
            nums[j++] = 0;
            ct0--;
        }
        while (ct1 != 0) {
            nums[j++] = 1;
            ct1--;
        }
        while (ct2 != 0) {
            nums[j++] = 2;
            ct2--;
        }
    }
}
