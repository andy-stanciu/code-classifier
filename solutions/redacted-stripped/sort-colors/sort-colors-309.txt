class Solution {

    public void method1(int[] nums) {
        int tem = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i; j < nums.length; j++) {
                if (nums[j] < nums[i]) {
                    tem = nums[i];
                    nums[i] = nums[j];
                    nums[j] = tem;
                }
            }
        }
    }
}
