class Solution {

    public void method1(int[] nums) {
        int[] n = new int[3];
        Arrays.fill(n, 0);
        for (int i = 0; i < nums.length; i++) n[nums[i]]++;
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            while (n[index] == 0) index++;
            nums[i] = index;
            n[index]--;
        }
    }
}
