class Solution {

    public void method1(int[] nums) {
        int[] counts = new int[3];
        for (int n : nums) {
            counts[n]++;
        }
        for (int i = 0, j = 0, cntIdx = 0; i < nums.length; i++, j++) {
            while (j == counts[cntIdx]) {
                j = 0;
                cntIdx++;
            }
            nums[i] = cntIdx;
        }
    }
}
