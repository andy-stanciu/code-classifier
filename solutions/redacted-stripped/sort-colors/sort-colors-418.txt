class Solution {

    public void method1(int[] arr) {
        int n = arr.length;
        int zero = 0;
        int ones = 0;
        for (int i = 0; i < n; i++) {
            if (arr[i] == 0)
                zero++;
            else if (arr[i] == 1)
                ones++;
        }
        int d = zero + ones;
        for (int i = 0; i < n; i++) {
            if (i < zero)
                arr[i] = 0;
            else if (i >= zero && i < d) {
                arr[i] = 1;
            }
            if (i >= d) {
                arr[i] = 2;
            }
        }
        for (int ele : arr) System.out.print(ele + " ");
    }
}
