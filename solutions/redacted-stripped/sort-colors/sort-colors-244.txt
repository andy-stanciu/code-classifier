class Solution {

    public void method1(int[] a) {
        int zeroes = 0, ones = 0, twos = 0;
        for (int i = 0; i < a.length; i++) {
            if (a[i] == 0)
                a[zeroes++] = 0;
            else if (a[i] == 1)
                ones++;
            else
                twos++;
        }
        for (int i = zeroes; i < zeroes + ones; i++) a[i] = 1;
        for (int i = zeroes + ones; i < a.length; i++) a[i] = 2;
    }
}
