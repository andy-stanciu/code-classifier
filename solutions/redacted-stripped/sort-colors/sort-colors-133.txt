class Solution {

    public void method1(int[] nums) {
        int z = 0;
        int o = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                z++;
            if (nums[i] == 1)
                o++;
        }
        for (int i = 0; i < z; i++) {
            nums[i] = 0;
        }
        for (int i = z; i < o + z; i++) {
            nums[i] = 1;
        }
        for (int i = o + z; i < nums.length; i++) {
            nums[i] = 2;
        }
    }
}
