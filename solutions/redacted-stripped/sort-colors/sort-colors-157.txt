class Solution {

    public void method1(int[] nums) {
        int lo = 0, hi = nums.length - 1, mid = 0;
        while (mid <= hi) {
            if (nums[mid] == 0) {
                nums[mid++] = nums[lo];
                nums[lo++] = 0;
            } else if (nums[mid] == 2) {
                nums[mid] = nums[hi];
                nums[hi--] = 2;
            } else
                mid++;
        }
    }
}
