class Solution {

    public void method1(int[] nums) {
        int l = 0, m = 0, n = nums.length - 1;
        while (m <= n) {
            if (nums[m] == 0) {
                int w = nums[l];
                nums[l] = nums[m];
                nums[m] = w;
                m++;
                l++;
            } else if (nums[m] == 1) {
                m++;
            } else if (nums[m] == 2) {
                int y = nums[n];
                nums[n] = nums[m];
                nums[m] = y;
                n--;
            }
        }
    }
}
