class Solution {

    public void method1(int[] nums) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            pq.offer(nums[i]);
        }
        for (int i = 0; i < nums.length; i++) {
            nums[i] = pq.remove();
        }
    }
}
