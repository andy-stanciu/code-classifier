class Solution {

    public void method1(int[] arr) {
        int n = arr.length;
        int l = 0, m = 0, h = n - 1;
        while (m <= h) {
            if (arr[m] == 0) {
                int temp = arr[l];
                arr[l] = arr[m];
                arr[m] = temp;
                l++;
                m++;
            } else if (arr[m] == 1) {
                m++;
            } else {
                int temp = arr[h];
                arr[h] = arr[m];
                arr[m] = temp;
                h--;
            }
        }
    }
}
