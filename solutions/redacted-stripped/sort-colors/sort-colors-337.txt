class Solution {

    public void method1(int[] arr) {
        int n = arr.length;
        int noOfZeroes = 0;
        int noOfOnes = 0;
        for (int i = 0; i < n; i++) {
            if (arr[i] == 0)
                noOfZeroes++;
            if (arr[i] == 1)
                noOfOnes++;
        }
        for (int i = 0; i <= n - 1; i++) {
            if (i < noOfZeroes)
                arr[i] = 0;
            else if (i < noOfZeroes + noOfOnes)
                arr[i] = 1;
            else
                arr[i] = 2;
        }
    }
}
