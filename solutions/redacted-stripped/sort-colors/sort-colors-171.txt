class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int[] countArray = new int[3];
        for (int i = 0; i < n; i++) {
            int element = nums[i];
            countArray[element]++;
        }
        int j = 0;
        for (int element = 0; element < 3; element++) {
            while (countArray[element] > 0) {
                nums[j] = element;
                j++;
                countArray[element]--;
            }
        }
    }
}
