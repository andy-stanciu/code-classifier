class Solution {

    public void method1(int[] nums) {
        int redCount = 0;
        int whiteCount = 0;
        int blueCount = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0)
                redCount++;
            if (nums[i] == 1)
                whiteCount++;
            if (nums[i] == 2)
                blueCount++;
        }
        for (int i = 0; i < nums.length; i++) {
            if (redCount > 0) {
                nums[i] = 0;
                redCount--;
            } else if (whiteCount > 0) {
                nums[i] = 1;
                whiteCount--;
            } else {
                nums[i] = 2;
                blueCount--;
            }
        }
    }
}
