class Solution {

    public void method1(int[] nums) {
        int[] var1 = new int[3];
        for (int num : nums) {
            var1[num]++;
        }
        int i = 0;
        for (int color = 0; color < var1.length; color++) {
            while (var1[color] > 0) {
                nums[i] = color;
                i++;
                var1[color]--;
            }
        }
    }
}
