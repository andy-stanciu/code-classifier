class Solution {

    public void method1(int[] nums) {
        int count_0s = 0;
        int count_1s = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0)
                count_0s++;
            if (nums[i] == 1)
                count_1s++;
        }
        for (int i = 0; i < n; i++) {
            if (i < count_0s)
                nums[i] = 0;
            else if (i < (count_0s + count_1s))
                nums[i] = 1;
            else
                nums[i] = 2;
        }
    }
}
