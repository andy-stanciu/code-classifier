class Solution {

    public void method1(int[] nums) {
        int s = 0, e = nums.length - 1;
        int i = 0;
        while (i <= e) {
            if (nums[i] == 0) {
                nums[i] = nums[s];
                nums[s] = 0;
                s++;
                i++;
            } else if (nums[i] == 2) {
                nums[i] = nums[e];
                nums[e] = 2;
                e--;
            } else {
                i++;
            }
        }
    }
}
