class Solution {

    public static int compress(char[] chars) {
        if (chars.length < 2) {
            return chars.length;
        }
        char curr = chars[0];
        int apprearance = 1;
        int lastIndex = 1;
        for (int i = 1; i < chars.length; i++) {
            if (chars[i] == curr) {
                apprearance++;
            }
            if (chars[i] != curr) {
                if (apprearance > 1) {
                    for (char num : String.valueOf(apprearance).toCharArray()) {
                        chars[lastIndex] = num;
                        lastIndex++;
                    }
                }
                curr = chars[i];
                apprearance = 1;
                chars[lastIndex] = curr;
                lastIndex++;
            }
            if (i == chars.length - 1) {
                if (apprearance > 1) {
                    for (char num : String.valueOf(apprearance).toCharArray()) {
                        chars[lastIndex] = num;
                        lastIndex++;
                    }
                }
            }
        }
        return lastIndex;
    }
}
