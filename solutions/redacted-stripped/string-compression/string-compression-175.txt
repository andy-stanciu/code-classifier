class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int index = 0;
        int i = 0;
        while (i < n) {
            char curr = chars[i];
            int count = 0;
            while (i < n && curr == chars[i]) {
                count++;
                i++;
            }
            chars[index++] = curr;
            if (count > 1) {
                String countstr = Integer.toString(count);
                for (char ch : countstr.toCharArray()) {
                    chars[index++] = ch;
                }
            }
        }
        return index;
    }
}
