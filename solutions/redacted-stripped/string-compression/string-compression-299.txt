class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1) {
            return chars.length;
        }
        int index = 0;
        char ele = chars[0];
        int count = 0;
        for (int i = 0; i < chars.length; i++) {
            if (ele == chars[i]) {
                count++;
            } else {
                chars[index++] = ele;
                if (count > 1) {
                    char[] countChars = String.valueOf(count).toCharArray();
                    for (char c : countChars) {
                        chars[index++] = c;
                    }
                }
                ele = chars[i];
                count = 1;
            }
        }
        chars[index++] = ele;
        if (count > 1) {
            char[] countchars = String.valueOf(count).toCharArray();
            for (char c : countchars) {
                chars[index++] = c;
            }
        }
        return index;
    }
}
