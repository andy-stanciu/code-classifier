public class Solution {

    public int compress(char[] A) {
        int n = A.length;
        if (n == 1)
            return 1;
        int l = 0;
        for (int r = 1; r < n; r++) {
            int len = 1;
            while (r < n && A[r] == A[r - 1]) {
                len++;
                r++;
            }
            if (len > 1) {
                String lenS = String.valueOf(len);
                for (int j = 0; j < lenS.length(); j++) A[++l] = lenS.charAt(j);
            }
            l++;
            if (r < n)
                A[l] = A[r];
            if (r == n - 1)
                l++;
        }
        return l;
    }
}
