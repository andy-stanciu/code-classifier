class Solution {

    public int compress(char[] ch) {
        StringBuilder newStr = new StringBuilder();
        for (int i = 0; i < ch.length; i++) {
            int count = 1;
            while (i < ch.length - 1 && ch[i] == ch[i + 1]) {
                count++;
                i++;
            }
            newStr.append(ch[i]);
            if (count > 1) {
                newStr.append(count);
            }
        }
        for (int j = 0; j < newStr.length(); j++) {
            ch[j] = newStr.charAt(j);
        }
        return newStr.length();
    }
}
