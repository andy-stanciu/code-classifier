class Solution {

    public int compress(char[] chars) {
        int count = 1;
        char current = chars[0];
        StringBuilder sb = new StringBuilder();
        if (chars.length == 1) {
            return 1;
        }
        for (int i = 1; i < chars.length; i++) {
            if (current == chars[i]) {
                count++;
            } else {
                append(sb, current, count);
                current = chars[i];
                count = 1;
            }
        }
        append(sb, current, count);
        for (int i = 0; i < sb.length(); i++) {
            chars[i] = sb.charAt(i);
        }
        return sb.length();
    }

    public void append(StringBuilder sb, char current, int count) {
        sb.append(current);
        if (count > 1) {
            String var1 = Integer.toString(count);
            for (int j = 0; j < var1.length(); j++) {
                char digit = var1.charAt(j);
                sb.append(digit);
            }
        }
    }
}
