class Solution {

    public int compress(char[] chars) {
        int currIndex = 0, modifiedIndex = 0, n = chars.length;
        while (currIndex < n) {
            chars[modifiedIndex++] = chars[currIndex];
            char currChar = chars[currIndex];
            int count = 0;
            while (currIndex < n && chars[currIndex] == currChar) {
                ++currIndex;
                ++count;
            }
            if (count != 1) {
                for (char c : Integer.toString(count).toCharArray()) {
                    chars[modifiedIndex++] = c;
                }
            }
        }
        return modifiedIndex;
    }
}
