class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        if (chars.length == 1)
            return 1;
        int count = 1;
        for (int i = 0; i < chars.length; i++) {
            char c = chars[i];
            if (i + 1 < chars.length && chars[i] == chars[i + 1]) {
                count++;
                if (i == chars.length - 1) {
                    sb.append(c);
                    if (count > 1)
                        sb.append(count);
                }
            } else {
                sb.append(c);
                if (count > 1)
                    sb.append(count);
                count = 1;
            }
        }
        System.arraycopy(sb.toString().toCharArray(), 0, chars, 0, sb.toString().length());
        return sb.toString().length();
    }
}
