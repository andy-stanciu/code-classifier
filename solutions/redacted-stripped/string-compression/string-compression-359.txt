class Solution {

    public static int compress(char[] chars) {
        StringBuilder var1 = new StringBuilder();
        char letter = chars[0];
        int counter = 1;
        for (int i = 1; i < chars.length; i++) {
            if (letter == chars[i]) {
                counter++;
            } else {
                var1.append(letter);
                if (counter > 1) {
                    var1.append(counter);
                }
                letter = chars[i];
                counter = 1;
            }
        }
        var1.append(letter);
        if (counter > 1) {
            var1.append(counter);
        }
        char[] c = var1.toString().toCharArray();
        int length = c.length;
        for (int i = 0; i < length; i++) {
            chars[i] = c[i];
        }
        return length;
    }
}
