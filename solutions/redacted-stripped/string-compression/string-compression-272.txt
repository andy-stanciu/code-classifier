class Solution {

    public int compress(char[] arr) {
        String ans = "";
        int i = 0, j = 0;
        while (j < arr.length) {
            if (arr[i] == arr[j])
                j++;
            else {
                ans += arr[i];
                int len = j - i;
                if (len > 1)
                    ans += len;
                i = j;
            }
        }
        ans += arr[i];
        int len = j - i;
        if (len > 1)
            ans += len;
        for (int k = 0; k < ans.length(); k++) {
            arr[k] = ans.charAt(k);
        }
        return ans.length();
    }
}
