class Solution {

    public int compress(char[] chars) {
        StringBuilder builder = new StringBuilder();
        char last = chars[0];
        int count = 0;
        for (char ch : chars) {
            if (ch != last) {
                builder.append(last);
                if (count > 1) {
                    builder.append(count);
                }
                last = ch;
                count = 1;
            } else {
                count++;
            }
        }
        builder.append(last);
        if (count > 1) {
            builder.append(count);
        }
        for (int i = 0; i < builder.length(); i++) {
            chars[i] = builder.charAt(i);
        }
        return builder.length();
    }
}
