class Solution {

    public int compress(char[] chars) {
        int size = chars.length;
        if (size == 1)
            return 1;
        int index = 0;
        int lastCount = 1;
        char lastChar = chars[0];
        for (int i = 1; i < size; i++) {
            if (lastChar == chars[i]) {
                lastCount++;
            } else {
                chars[index++] = lastChar;
                if (lastCount > 1) {
                    for (char cTemp : String.valueOf(lastCount).toCharArray()) {
                        chars[index++] = cTemp;
                    }
                }
                lastChar = chars[i];
                lastCount = 1;
            }
        }
        chars[index++] = lastChar;
        if (lastCount > 1) {
            for (char cTemp : String.valueOf(lastCount).toCharArray()) {
                chars[index++] = cTemp;
            }
        }
        return index;
    }
}
