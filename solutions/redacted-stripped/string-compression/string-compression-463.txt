class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int j = 0, i = 0;
        if (n == 1) {
            return 1;
        }
        while (i < n) {
            int c = 1;
            while (i < n - 1 && chars[i] == chars[i + 1]) {
                i++;
                c++;
            }
            chars[j++] = chars[i++];
            if (c > 1) {
                for (char co : String.valueOf(c).toCharArray()) {
                    chars[j++] = co;
                }
            }
        }
        return j;
    }
}
