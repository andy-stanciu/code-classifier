class Solution {

    public int compress(char[] chars) {
        if (chars.length == 0) {
            return 0;
        }
        Character lastChar = chars[0];
        int lastCharCount = 0;
        int index = 0;
        for (int i = 0; i < chars.length; i++) {
            if (lastChar == chars[i]) {
                lastCharCount++;
            } else {
                chars[index++] = lastChar;
                index = this.addCharCounts(chars, index, lastCharCount);
                lastChar = chars[i];
                lastCharCount = 1;
            }
        }
        chars[index++] = lastChar;
        index = this.addCharCounts(chars, index, lastCharCount);
        return index;
    }

    public int addCharCounts(char[] chars, int index, int charCount) {
        if (charCount > 1) {
            for (char c : Integer.toString(charCount).toCharArray()) {
                chars[index++] = c;
            }
        }
        return index;
    }
}
