class Solution {

    public int compress(char[] chars) {
        int currIndex = 0;
        int i = 0;
        while (i < chars.length) {
            int j = i;
            int count = 0;
            while (j < chars.length && chars[i] == chars[j]) {
                count++;
                j++;
            }
            chars[currIndex++] = chars[i];
            if (count > 1) {
                String s = String.valueOf(count);
                for (int x = 0; x < s.length(); x++) {
                    chars[currIndex++] = s.charAt(x);
                }
            }
            i = i + count;
        }
        return currIndex;
    }
}
