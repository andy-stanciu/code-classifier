class Solution {

    public int compress(char[] chars) {
        int freq = 1;
        int newArrayPointer = 0;
        for (int index = 1; index < chars.length; index++) {
            if (chars[index] == chars[index - 1]) {
                freq++;
            } else {
                chars[newArrayPointer++] = chars[index - 1];
                if (freq > 1) {
                    String frequency = String.valueOf(freq);
                    for (int i = 0; i < frequency.length(); i++) {
                        chars[newArrayPointer++] = frequency.charAt(i);
                    }
                }
                freq = 1;
            }
        }
        chars[newArrayPointer++] = chars[chars.length - 1];
        if (freq > 1) {
            String frequency = String.valueOf(freq);
            for (int i = 0; i < frequency.length(); i++) {
                chars[newArrayPointer++] = frequency.charAt(i);
            }
        }
        return newArrayPointer;
    }
}
