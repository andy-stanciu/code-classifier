class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1)
            return 1;
        char[] a = new char[chars.length];
        int c = 1, d = 0, f = 0;
        for (int i = 1; i < chars.length; i++) {
            if (chars[i] == chars[i - 1]) {
                c++;
                f = 1;
            } else {
                a[d] = chars[i - 1];
                d++;
                if (c > 1) {
                    String s = Integer.toString(c);
                    for (int j = 0; j < s.length(); j++) {
                        a[d] = s.charAt(j);
                        d++;
                    }
                }
                c = 1;
                f = 0;
            }
        }
        if (f == 1) {
            a[d] = chars[chars.length - 1];
            d++;
            if (c > 1) {
                String s = Integer.toString(c);
                for (int j = 0; j < s.length(); j++) {
                    a[d] = s.charAt(j);
                    d++;
                }
            }
        } else {
            a[d] = chars[chars.length - 1];
            d++;
        }
        for (int i = 0; i < d; i++) {
            chars[i] = a[i];
        }
        return d;
    }
}
