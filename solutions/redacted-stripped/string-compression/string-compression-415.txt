class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1) {
            return chars.length;
        }
        int groupSize;
        StringBuilder var1 = new StringBuilder();
        for (int i = 0; i < chars.length; i++) {
            groupSize = 1;
            var1.append(chars[i]);
            while (i + 1 < chars.length && chars[i + 1] == chars[i]) {
                i++;
                groupSize++;
            }
            if (groupSize > 10) {
                Arrays.stream(String.valueOf(groupSize).split("")).forEach(var1::append);
            } else if (groupSize > 1) {
                var1.append(groupSize);
            }
        }
        String s = var1.toString();
        char[] newChars = s.toCharArray();
        System.arraycopy(newChars, 0, chars, 0, newChars.length);
        return newChars.length;
    }
}
