class Solution {

    public int compress(char[] chars) {
        int ans = 0, s = 0, e = 0, n = chars.length;
        while (e != n) {
            while (e != n && chars[s] == chars[e]) {
                e++;
            }
            int t = e - s;
            chars[ans++] = chars[s];
            if (t > 1) {
                String val = String.valueOf(t);
                for (int i = 0; i < val.length(); i++) chars[ans++] = val.charAt(i);
            }
            s = e;
        }
        return ans;
    }
}
