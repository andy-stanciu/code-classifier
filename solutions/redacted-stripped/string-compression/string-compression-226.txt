class Solution {

    public int compress(char[] chars) {
        if (chars.length == 0)
            return 0;
        if (chars.length == 1)
            return 1;
        int write = 0;
        int i = 0;
        while (i < chars.length) {
            char currentChar = chars[i];
            int count = 0;
            while (i < chars.length && currentChar == chars[i]) {
                i++;
                count++;
            }
            chars[write++] = currentChar;
            if (count > 1) {
                for (char c : String.valueOf(count).toCharArray()) {
                    chars[write++] = c;
                }
            }
        }
        return write;
    }
}
