class Solution {

    public int compress(char[] chars) {
        int count = 1;
        int i = 0, idx = 0;
        while (i < chars.length) {
            char curch = chars[i];
            while (i + 1 < chars.length && curch == chars[i + 1]) {
                i++;
                count++;
            }
            chars[idx++] = curch;
            if (count > 1) {
                String len = String.valueOf(count);
                for (int j = 0; j < len.length(); j++) {
                    chars[idx++] = len.charAt(j);
                }
            }
            i++;
            count = 1;
        }
        return idx;
    }
}
