class Solution {

    public String method1(char[] chars) {
        StringBuilder str = new StringBuilder();
        str.append(chars[0]);
        int count = 1;
        for (int i = 1; i < chars.length; i++) {
            if (chars[i] == chars[i - 1]) {
                count++;
            } else {
                if (count > 1) {
                    str.append(count);
                }
                str.append(chars[i]);
                count = 1;
            }
        }
        if (count > 1) {
            str.append(count);
        }
        return str.toString();
    }

    public int compress(char[] chars) {
        String compressed = method1(chars);
        char[] compressedChars = compressed.toCharArray();
        System.arraycopy(compressedChars, 0, chars, 0, compressedChars.length);
        return compressedChars.length;
    }
}
