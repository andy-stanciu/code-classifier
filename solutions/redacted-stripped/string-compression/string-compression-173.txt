class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int indx = 0;
        int i = 0;
        while (i < n) {
            char currChar = chars[i];
            int count = 0;
            while (i < n && chars[i] == currChar) {
                count++;
                i++;
            }
            chars[indx++] = currChar;
            if (count > 1) {
                for (char c : Integer.toString(count).toCharArray()) {
                    chars[indx++] = c;
                }
            }
        }
        return indx;
    }
}
