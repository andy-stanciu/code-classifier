class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1)
            return 1;
        int left = 0, right = 1;
        int editIdx = 0;
        while (right <= chars.length) {
            chars[editIdx++] = chars[left];
            if (right < chars.length && chars[left] == chars[right]) {
                while (right < chars.length && chars[left] == chars[right]) {
                    right++;
                }
                String charsCount = Integer.toString(right - left);
                int idx = 0;
                while (idx < charsCount.length()) {
                    chars[editIdx++] = charsCount.charAt(idx++);
                }
            }
            left = right;
            right++;
        }
        return editIdx;
    }
}
