class Solution {

    public int compress(char[] ch) {
        StringBuilder sb = new StringBuilder();
        int count = 1;
        char prev = ch[0];
        for (int i = 1; i < ch.length; i++) {
            char cur = ch[i];
            if (cur == prev) {
                count++;
            } else {
                sb.append(prev);
                prev = cur;
                if (count > 1) {
                    sb.append(count);
                    count = 1;
                }
            }
        }
        sb.append(prev);
        if (count > 1) {
            sb.append(count);
            count = 1;
        }
        for (int i = 0; i < sb.length(); i++) {
            ch[i] = sb.charAt(i);
        }
        return sb.length();
    }
}
