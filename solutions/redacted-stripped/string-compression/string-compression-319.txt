class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        int i = 0;
        while (i < chars.length) {
            char c = chars[i];
            int count = 0;
            while (i < chars.length && chars[i] == c) {
                count++;
                i++;
            }
            sb.append(c);
            if (count > 1) {
                sb.append(count);
            }
        }
        int length = sb.length();
        for (int j = 0; j < length; j++) {
            chars[j] = sb.charAt(j);
        }
        return length;
    }
}
