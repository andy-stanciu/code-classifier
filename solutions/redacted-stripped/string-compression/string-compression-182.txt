class Solution {

    public int compress(char[] chars) {
        int c = 0, count = 0;
        String st = "";
        StringBuilder sb = new StringBuilder();
        int char_len = chars.length;
        if (char_len == 1)
            return 1;
        sb.append(chars[0]);
        for (int i = 0; i < char_len; i++) {
            if (chars[c] == chars[i]) {
                count++;
            } else {
                c = i;
                if (count > 1)
                    sb.append(count);
                count = 1;
                sb.append(chars[c]);
            }
        }
        if (count > 1)
            sb.append(count);
        st = sb.toString();
        int len = st.length();
        for (int i = 0; i < len; i++) {
            chars[i] = st.charAt(i);
        }
        return len;
    }
}
