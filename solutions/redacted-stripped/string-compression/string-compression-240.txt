class Solution {

    public int compress(char[] chars) {
        int c = 0;
        if (chars.length == 1) {
            return 1;
        }
        char ch = chars[0];
        StringBuilder str = new StringBuilder();
        for (int i = 0; i < chars.length; i++) {
            if (chars[i] == ch) {
                c++;
            } else {
                if (c > 1) {
                    str.append(ch);
                    str.append(c);
                } else {
                    str.append(ch);
                }
                ch = chars[i];
                c = 1;
            }
        }
        if (c > 1) {
            str.append(ch);
            str.append(c);
        } else {
            str.append(ch);
        }
        int char1 = str.length();
        for (int i = 0; i < str.length(); i++) {
            chars[i] = str.charAt(i);
        }
        return char1;
    }
}
