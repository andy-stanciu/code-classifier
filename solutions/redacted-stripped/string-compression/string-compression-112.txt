class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int i = 0;
        int index = 0;
        while (i < n) {
            char ch = chars[i];
            int count = 0;
            while (i < n && ch == chars[i]) {
                i++;
                count++;
            }
            chars[index] = ch;
            index++;
            if (count > 1) {
                String str = Integer.toString(count);
                for (char a : str.toCharArray()) {
                    chars[index++] = a;
                }
            }
        }
        return index;
    }
}
