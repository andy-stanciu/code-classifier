class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1) {
            return 1;
        }
        int k = 0;
        int count = 1;
        for (int i = 0; i < chars.length; i++) {
            if (i != chars.length - 1 && chars[i] == chars[i + 1]) {
                if (count == 1) {
                    chars[k] = chars[i];
                    k++;
                }
                count++;
            } else {
                if (count != 1) {
                    if (count > 9) {
                        String a = String.valueOf(count);
                        for (int j = 0; j < a.length(); j++) {
                            chars[k] = a.charAt(j);
                            k++;
                        }
                    } else {
                        chars[k] = Character.forDigit(count, 10);
                        k++;
                    }
                    count = 1;
                } else {
                    chars[k] = chars[i];
                    k++;
                }
            }
        }
        return k;
    }
}
