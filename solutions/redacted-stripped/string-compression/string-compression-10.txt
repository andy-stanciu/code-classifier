class Solution {

    public int compress(char[] chars) {
        int len1 = chars.length, i;
        String s = "";
        int count = 1;
        for (i = 0; i < len1; i++) {
            count = 1;
            s += chars[i];
            while (i + 1 < len1 && (chars[i] == chars[i + 1])) {
                count++;
                i++;
            }
            if (count > 1)
                s += count;
        }
        for (i = 0; i < s.length(); i++) chars[i] = s.charAt(i);
        return s.length();
    }
}
