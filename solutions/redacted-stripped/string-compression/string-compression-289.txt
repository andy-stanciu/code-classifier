class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int index = 0;
        int i = 0;
        while (i < n) {
            char ch = chars[i];
            int count = 0;
            chars[index++] = ch;
            while (i < n && chars[i] == ch) {
                i++;
                count++;
            }
            if (count > 1) {
                String str = Integer.toString(count);
                for (char c : str.toCharArray()) {
                    chars[index++] = c;
                }
            }
        }
        return index;
    }
}
