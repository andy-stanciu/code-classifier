class Solution {

    public int compress(char[] chars) {
        int k = 0;
        for (int i = 0; i < chars.length; i++) {
            Integer count = 1;
            chars[k++] = chars[i];
            while (i < chars.length - 1 && chars[i] == chars[i + 1]) {
                count++;
                i++;
            }
            if (count > 1) {
                String s = count.toString();
                for (int j = 0; j < s.length(); j++) {
                    chars[k++] = s.charAt(j);
                }
            }
        }
        return k;
    }
}
