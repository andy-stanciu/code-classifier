class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        if (n == 0)
            return 0;
        int index = 0;
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (chars[i] == chars[i - 1]) {
                count++;
            } else {
                chars[index++] = chars[i - 1];
                if (count > 1) {
                    for (char c : Integer.toString(count).toCharArray()) {
                        chars[index++] = c;
                    }
                }
                count = 1;
            }
        }
        chars[index++] = chars[n - 1];
        if (count > 1) {
            for (char c : Integer.toString(count).toCharArray()) {
                chars[index++] = c;
            }
        }
        return index;
    }
}
