class Solution {

    public int compress(char[] chars) {
        int index = 0, start = 0;
        while (start < chars.length) {
            int end = start;
            while (end < chars.length && chars[start] == chars[end]) {
                end++;
            }
            int count = end - start;
            chars[index++] = chars[start];
            if (count > 1) {
                char[] res = Integer.toString(count).toCharArray();
                for (char ch : res) {
                    chars[index++] = ch;
                }
            }
            start = end;
        }
        return index;
    }
}
