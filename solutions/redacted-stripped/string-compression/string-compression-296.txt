class Solution {

    public int compress(char[] chars) {
        int len = 0;
        if (chars.length == 1)
            return 1;
        for (int i = 0; i < chars.length; i++) {
            int count = 1;
            chars[len++] = chars[i];
            while (i < chars.length - 1 && chars[i] == chars[i + 1]) {
                count++;
                i++;
            }
            if (count > 1) {
                for (char c : String.valueOf(count).toCharArray()) {
                    chars[len++] = c;
                }
            }
        }
        return len;
    }
}
