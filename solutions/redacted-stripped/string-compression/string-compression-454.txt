class Solution {

    public int compress(char[] chars) {
        if (chars.length == 1) {
            return 1;
        }
        StringBuilder sb = new StringBuilder();
        sb.append(chars[0]);
        int count = 1;
        for (int i = 1; i < chars.length; i++) {
            if (chars[i] == chars[i - 1]) {
                count = count + 1;
            } else {
                if (count > 1) {
                    sb.append(count);
                    sb.append(chars[i]);
                    count = 1;
                } else {
                    sb.append(chars[i]);
                }
            }
        }
        if (count > 1) {
            sb.append(count);
        }
        String str = sb.toString();
        char[] ch = str.toCharArray();
        for (int i = 0; i < ch.length; i++) {
            chars[i] = ch[i];
        }
        int x = ch.length;
        return x;
    }
}
