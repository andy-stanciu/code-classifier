class Solution {

    public int method1(int[] param1, int amt) {
        int[] dp = new int[amt + 1];
        for (int i = 1; i < dp.length; i++) {
            int min = Integer.MAX_VALUE - 1;
            for (int j = 0; j < param1.length; j++) {
                if (i - param1[j] >= 0)
                    min = Math.min(dp[i - param1[j]], min);
            }
            dp[i] = min + 1;
        }
        if (dp[amt] != Integer.MAX_VALUE)
            return dp[amt];
        return -1;
    }
}
