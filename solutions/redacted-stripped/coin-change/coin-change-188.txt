class Solution {

    public int solveUsingMemo(int[] param1, int amount, int[] dp) {
        if (amount == 0) {
            return 0;
        }
        if (amount < 0) {
            return Integer.MAX_VALUE;
        }
        if (dp[amount] != 0) {
            return dp[amount];
        }
        int min = Integer.MAX_VALUE;
        for (int x : param1) {
            int ans = solveUsingMemo(param1, amount - x, dp);
            if (ans != Integer.MAX_VALUE) {
                min = Math.min(min, ans + 1);
            }
        }
        dp[amount] = min;
        return min;
    }

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        int ans = solveUsingMemo(param1, amount, dp);
        if (ans == Integer.MAX_VALUE) {
            return -1;
        }
        return ans;
    }
}
