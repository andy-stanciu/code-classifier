class Solution {

    public int method1(int[] arr, int k) {
        int[][] dp = new int[arr.length][k + 1];
        for (int i = 0; i <= k; i++) {
            if (i % arr[0] == 0) {
                dp[0][i] = i / arr[0];
            } else {
                dp[0][i] = 1000000;
            }
        }
        for (int i = 1; i < arr.length; i++) {
            for (int j = 0; j <= k; j++) {
                int nott = dp[i - 1][j];
                int take = 10000002;
                if (arr[i] <= j) {
                    take = 1 + dp[i][j - arr[i]];
                }
                dp[i][j] = Math.min(take, nott);
            }
        }
        int a = dp[arr.length - 1][k];
        if (a <= 100000)
            return a;
        return -1;
    }
}
