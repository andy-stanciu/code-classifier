class Solution {

    public int method1(int[] param1, int amount) {
        Map<Integer, Integer> dp = new HashMap<Integer, Integer>();
        return method1(param1, amount, dp);
    }

    public int method1(int[] param1, int amount, Map<Integer, Integer> dp) {
        if (dp.containsKey(amount))
            return dp.get(amount);
        if (amount < 0)
            return -1;
        if (amount == 0)
            return 0;
        int min = Integer.MAX_VALUE;
        for (int var1 : param1) {
            int remain = amount - var1;
            if (remain >= 0) {
                int newPath = method1(param1, remain, dp) + 1;
                if (newPath > 0 && newPath < min)
                    min = newPath;
            }
        }
        if (min == Integer.MAX_VALUE)
            dp.put(amount, -1);
        else
            dp.put(amount, min);
        return dp.get(amount);
    }
}
