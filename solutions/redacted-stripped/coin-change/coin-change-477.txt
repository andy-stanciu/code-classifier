class Solution {

    public int method1(int[] param1, int amounts) {
        long[] dp = new long[amounts + 1];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[0] = 0;
        for (int i = 1; i < amounts + 1; i++) {
            long min = Integer.MAX_VALUE;
            for (int j = 0; j < param1.length; j++) {
                if (i - param1[j] >= 0) {
                    min = Math.min(min, dp[i - param1[j]]);
                }
            }
            dp[i] = min + 1;
        }
        if (dp[amounts] >= Integer.MAX_VALUE) {
            return -1;
        }
        return (int) dp[amounts];
    }
}
