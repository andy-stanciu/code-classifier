class Solution {

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, -1);
        dp[0] = 0;
        for (int i = 0; i < param1.length; i++) {
            for (int j = 1; j <= amount; j++) {
                if (j - param1[i] >= 0 && dp[j - param1[i]] >= 0) {
                    dp[j] = dp[j] != -1 ? Math.min(dp[j], 1 + dp[j - param1[i]]) : 1 + dp[j - param1[i]];
                }
            }
        }
        return dp[amount];
    }
}
