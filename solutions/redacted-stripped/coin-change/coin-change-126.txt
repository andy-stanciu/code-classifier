class Solution {

    public int method1(int[] param1, int amount) {
        int[][] t = new int[param1.length + 1][amount + 1];
        for (int j = 0; j <= amount; j++) {
            t[0][j] = Integer.MAX_VALUE - 1;
            if (j > 0) {
                t[1][j] = j % param1[0] == 0 ? j / param1[0] : Integer.MAX_VALUE - 1;
            }
        }
        for (int i = 0; i <= param1.length; i++) {
            t[i][0] = 0;
        }
        for (int i = 2; i <= param1.length; i++) {
            for (int j = 1; j <= amount; j++) {
                if (param1[i - 1] <= j) {
                    t[i][j] = Math.min(t[i - 1][j], 1 + t[i][j - param1[i - 1]]);
                } else {
                    t[i][j] = t[i - 1][j];
                }
            }
        }
        return t[param1.length][amount] == Integer.MAX_VALUE - 1 ? -1 : t[param1.length][amount];
    }
}
