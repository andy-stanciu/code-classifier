class Solution {

    public int method1(int[] param1, int amount) {
        int n = param1.length;
        int[][] dp = new int[n + 1][amount + 1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= amount; j++) {
                if (j == 0)
                    dp[i][j] = 0;
                else if (i == 0)
                    dp[i][j] = 100000;
                else if (param1[i - 1] > j)
                    dp[i][j] = dp[i - 1][j];
                else
                    dp[i][j] = Math.min(1 + dp[i][j - param1[i - 1]], dp[i - 1][j]);
            }
        }
        return dp[n][amount] > 1e4 ? -1 : dp[n][amount];
    }
}
