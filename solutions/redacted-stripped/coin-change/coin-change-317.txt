class Solution {

    public int method1(int[] nums, int k) {
        int[][] dp = new int[nums.length][k + 1];
        for (int i = 0; i <= k; i++) {
            if (i % nums[0] == 0) {
                dp[0][i] = i / nums[0];
            } else {
                dp[0][i] = (int) 1e9;
            }
        }
        for (int i = 1; i < nums.length; i++) {
            for (int j = 0; j <= k; j++) {
                int take = (int) 1e9;
                if (nums[i] <= j) {
                    take = 1 + dp[i][j - nums[i]];
                }
                int notTake = 0 + dp[i - 1][j];
                dp[i][j] = Math.min(take, notTake);
            }
        }
        int ans = dp[nums.length - 1][k];
        if (ans >= (int) 1e9)
            return -1;
        else
            return ans;
    }
}
