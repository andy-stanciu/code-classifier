class Solution {

    public int method1(int[] param1, int amount) {
        int n = param1.length;
        int[][] dp = new int[n + 1][amount + 1];
        for (int i = 0; i <= n; i++) {
            dp[i][0] = 0;
        }
        for (int j = 0; j <= amount; j++) {
            dp[0][j] = Integer.MAX_VALUE - 10000;
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= amount; j++) {
                int taken = Integer.MAX_VALUE - 10000;
                int notTaken = dp[i - 1][j];
                if (j - param1[i - 1] >= 0) {
                    taken = 1 + dp[i][j - param1[i - 1]];
                }
                dp[i][j] = Math.min(taken, notTaken);
            }
        }
        return dp[n][amount] == Integer.MAX_VALUE - 10000 ? -1 : dp[n][amount];
    }
}
