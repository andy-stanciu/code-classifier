class Solution {

    private int[] var1, count;

    public int method1(int[] var1, int amount) {
        this.var1 = var1;
        this.count = new int[amount + 1];
        return dp(amount);
    }

    private int dp(int remainder) {
        if (remainder == 0)
            return 0;
        if (remainder < 0)
            return -1;
        if (count[remainder] != 0)
            return count[remainder];
        int min = Integer.MAX_VALUE;
        for (int var2 : var1) {
            int res = dp(remainder - var2);
            if (res >= 0 && res < min) {
                min = 1 + res;
            }
        }
        if (min == Integer.MAX_VALUE) {
            count[remainder] = -1;
        } else {
            count[remainder] = min;
        }
        return count[remainder];
    }
}
