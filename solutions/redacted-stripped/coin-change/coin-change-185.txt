class Solution {

    int[] dp;

    public int method1(int[] param1, int amount) {
        dp = new int[10005];
        Arrays.fill(dp, -1);
        int ans = method2(param1, amount, dp);
        if (ans == Integer.MAX_VALUE) {
            return -1;
        }
        return ans;
    }

    public int method2(int[] param1, int amount, int[] dp) {
        if (amount == 0) {
            return 0;
        }
        if (dp[amount] != -1) {
            return dp[amount];
        }
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < param1.length; i++) {
            if (amount - param1[i] >= 0) {
                int sub = method2(param1, amount - param1[i], dp);
                if (sub != Integer.MAX_VALUE) {
                    ans = Math.min(ans, 1 + sub);
                }
            }
        }
        return dp[amount] = ans;
    }
}
