class Solution {

    public int method1(int[] param1, int amount) {
        Queue<Integer> q = new LinkedList<>();
        q.add(0);
        int cs = 0;
        boolean[] vstd = new boolean[amount + 1];
        while (!q.isEmpty()) {
            int n = q.size();
            for (int i = 0; i < n; i++) {
                int sum = q.poll();
                if (sum == amount)
                    return cs;
                if (sum > amount || vstd[sum]) {
                    continue;
                }
                vstd[sum] = true;
                for (int var1 : param1) {
                    q.add(sum + var1);
                }
            }
            cs++;
        }
        return -1;
    }
}
