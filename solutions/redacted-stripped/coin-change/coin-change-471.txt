class Solution {

    public int method1(int[] param1, int amount) {
        int[][] dp = new int[param1.length][amount + 1];
        for (int i = 0; i < param1.length; i++) {
            for (int j = 0; j <= amount; j++) {
                if (i == 0) {
                    if (j % param1[i] == 0)
                        dp[i][j] = j / param1[i];
                    else
                        dp[i][j] = Integer.MAX_VALUE;
                } else {
                    if (j < param1[i])
                        dp[i][j] = dp[i - 1][j];
                    else {
                        dp[i][j] = Math.min(dp[i - 1][j], (dp[i][j - param1[i]] != Integer.MAX_VALUE) ? dp[i][j - param1[i]] + 1 : Integer.MAX_VALUE);
                    }
                }
            }
        }
        if (dp[param1.length - 1][amount] == Integer.MAX_VALUE)
            return -1;
        return dp[param1.length - 1][amount];
    }
}
