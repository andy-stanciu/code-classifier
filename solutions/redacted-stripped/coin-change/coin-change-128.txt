class Solution {

    public int method1(int[] param1, int amount) {
        int[] var1 = new int[amount + 1];
        Arrays.fill(var1, amount + 1);
        var1[0] = 0;
        for (int a = 1; a <= amount; a++) {
            for (int c : param1) {
                if (a - c >= 0) {
                    var1[a] = Math.min(var1[a], 1 + var1[a - c]);
                }
            }
        }
        return var1[amount] != amount + 1 ? var1[amount] : -1;
    }
}
