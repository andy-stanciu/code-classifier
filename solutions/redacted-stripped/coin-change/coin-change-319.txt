class Solution {

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        for (int i = 0; i < dp.length; i++) {
            dp[i] = Integer.MAX_VALUE;
        }
        int n = param1.length;
        dp[0] = 0;
        for (int i = 0; i < n; i++) {
            for (int j = param1[i]; j < amount + 1; j++) {
                if (dp[j - param1[i]] != Integer.MAX_VALUE) {
                    if (dp[j] != Integer.MAX_VALUE) {
                        if (dp[j] > dp[j - param1[i]] + 1) {
                            dp[j] = dp[j - param1[i]] + 1;
                        }
                    } else {
                        dp[j] = dp[j - param1[i]] + 1;
                    }
                }
            }
        }
        if (dp[amount] != Integer.MAX_VALUE)
            return dp[amount];
        else
            return -1;
    }
}
