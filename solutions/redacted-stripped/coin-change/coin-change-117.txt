class Solution {

    public int method1(int[] param1, int amount) {
        int[] counter = new int[amount + 1];
        Arrays.fill(counter, Integer.MAX_VALUE);
        counter[0] = 0;
        return method2(param1, counter, amount);
    }

    int method2(int[] param1, int[] counter, int amount) {
        if (amount < 0)
            return -1;
        if (counter[amount] != Integer.MAX_VALUE)
            return counter[amount];
        int result = 0;
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < param1.length; i++) {
            result = method2(param1, counter, amount - param1[i]);
            if (result >= 0 && result < min) {
                min = result + 1;
            }
        }
        if (min != Integer.MAX_VALUE)
            counter[amount] = min;
        else
            counter[amount] = -1;
        return counter[amount];
    }
}
