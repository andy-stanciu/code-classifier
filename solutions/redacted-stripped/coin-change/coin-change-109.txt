class Solution {

    public int method1(int[] param1, int amount) {
        if (amount < 1)
            return 0;
        int[] dp = new int[amount + 1];
        for (int currAmt = 1; currAmt <= amount; currAmt++) {
            for (int var2 : param1) {
                int diff = currAmt - var2;
                if (diff < 0)
                    continue;
                if (diff == 0)
                    dp[currAmt] = 1;
                else if (dp[diff] > 0) {
                    int var1 = dp[diff] + 1;
                    if (dp[currAmt] > 0)
                        dp[currAmt] = Math.min(var1, dp[currAmt]);
                    else
                        dp[currAmt] = var1;
                }
            }
        }
        return dp[amount] == 0 ? -1 : dp[amount];
    }
}
