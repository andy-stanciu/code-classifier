class Solution {

    public int method1(int[] param1, int value, int index) {
        if (index < 0 || value < 0)
            return Integer.MAX_VALUE;
        if (value == 0)
            return 0;
        int firstWay = method1(param1, value - param1[index], index);
        int secondWay = method1(param1, value, index - 1);
        if (firstWay < secondWay) {
            return 1 + firstWay;
        }
        return secondWay;
    }

    public int method1(int[] param1, int value) {
        int result = method1(param1, value, param1.length - 1);
        return result == Integer.MAX_VALUE ? -1 : result;
    }
}
