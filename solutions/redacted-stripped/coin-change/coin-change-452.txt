class Solution {

    public int method1(int[] param1, int amount) {
        if (amount == 0)
            return 0;
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, -1);
        int ans = rec(param1.length, param1, 0, amount, 0, dp);
        return ans == Integer.MAX_VALUE ? -1 : ans;
    }

    public static int rec(int n, int[] param1, int i, int remainingAmount, int noc, int[] dp) {
        if (remainingAmount == 0)
            return dp[remainingAmount] = noc;
        if (i == n || remainingAmount < 0)
            return Integer.MAX_VALUE;
        if (dp[remainingAmount] != -1)
            return dp[remainingAmount];
        return dp[remainingAmount] = Math.min(rec(n, param1, i + 1, remainingAmount, noc, dp), rec(n, param1, i, remainingAmount - param1[i], noc + 1, dp));
    }
}
