class Solution {

    public int method1(int[] param1, int amount) {
        Map<Integer, Integer> memo = new HashMap<>();
        memo.put(0, 0);
        for (int currentAmount = 1; currentAmount <= amount; currentAmount++) {
            for (int var3 : param1) {
                int previousAmount = currentAmount - var3;
                Integer var1 = memo.get(previousAmount);
                if (var1 == null)
                    continue;
                int var2 = memo.getOrDefault(currentAmount, Integer.MAX_VALUE);
                var2 = Integer.min(var2, var1 + 1);
                memo.put(currentAmount, var2);
            }
        }
        return memo.getOrDefault(amount, -1);
    }
}
