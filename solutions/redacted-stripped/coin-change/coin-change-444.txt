class Solution {

    private static final int var1 = Integer.MAX_VALUE;

    public int method1(int[] param1, int amount) {
        int[] var2 = new int[amount + 1];
        Arrays.fill(var2, var1);
        var2[0] = 0;
        for (int currAmt = 0; currAmt <= amount; currAmt++) {
            for (int var3 : param1) {
                if (var3 <= currAmt && var2[currAmt - var3] != var1)
                    var2[currAmt] = Math.min(var2[currAmt], var2[currAmt - var3] + 1);
            }
        }
        return (var2[amount] == var1) ? -1 : var2[amount];
    }
}
