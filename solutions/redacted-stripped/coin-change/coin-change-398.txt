class Solution {

    public static int method1(int[] param1, int amount, int[] dp) {
        if (amount == 0)
            return 0;
        int ans = Integer.MAX_VALUE - 1;
        for (int i = 0; i < param1.length; i++) {
            if (amount - param1[i] >= 0) {
                int subAns = 0;
                if (dp[amount - param1[i]] != -1) {
                    subAns = dp[amount - param1[i]];
                } else {
                    subAns = method1(param1, amount - param1[i], dp);
                }
                if (subAns != Integer.MAX_VALUE && subAns + 1 < ans) {
                    ans = subAns + 1;
                }
            }
        }
        return dp[amount] = ans;
    }

    public int method2(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, -1);
        dp[0] = 0;
        int ans = method1(param1, amount, dp);
        return (ans == Integer.MAX_VALUE - 1) ? -1 : ans;
    }
}
