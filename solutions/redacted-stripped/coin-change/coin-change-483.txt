class Solution {

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        int res = helper(param1, amount, dp);
        return (res == Integer.MAX_VALUE) ? -1 : res;
    }

    public int helper(int[] param1, int amount, int[] dp) {
        if (amount == 0) {
            return 0;
        } else if (dp[amount] != 0) {
            return dp[amount];
        } else {
            int res = Integer.MAX_VALUE;
            for (int i = 0; i < param1.length; i++) {
                if (amount - param1[i] >= 0)
                    res = Math.min(helper(param1, amount - param1[i], dp), res);
            }
            dp[amount] = (res == Integer.MAX_VALUE) ? Integer.MAX_VALUE : res + 1;
            return dp[amount];
        }
    }
}
