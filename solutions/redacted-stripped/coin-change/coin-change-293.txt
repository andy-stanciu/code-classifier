class Solution {

    public int method1(int[] param1, int amount) {
        int[] table = new int[amount + 1];
        for (int i = 0; i <= amount; i++) table[i] = 100000;
        table[0] = 0;
        for (int i = 1; i <= amount; i++) {
            for (int c : param1) {
                if (i - c >= 0)
                    table[i] = Math.min(table[i], 1 + table[i - c]);
            }
        }
        if (table[amount] == 100000)
            return -1;
        return table[amount];
    }
}
