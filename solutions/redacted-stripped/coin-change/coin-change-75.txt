class Solution {

    public int method1(int[] param1, int amount) {
        return helper(param1, amount, new Integer[amount + 1]);
    }

    int helper(int[] param1, int amount, Integer[] dp) {
        if (amount == 0) {
            dp[amount] = 0;
            return 0;
        }
        if (amount < 0)
            return -1;
        if (dp[amount] != null)
            return dp[amount];
        int result = Integer.MAX_VALUE;
        for (int var1 : param1) {
            int count = helper(param1, amount - var1, dp);
            if (count >= 0)
                result = Math.min(result, 1 + count);
        }
        dp[amount] = result == Integer.MAX_VALUE ? -1 : result;
        return dp[amount];
    }
}
