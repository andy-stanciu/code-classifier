class Solution {

    public int method1(int[] param1, int amount) {
        int n = param1.length;
        int[][] t = new int[n + 1][amount + 1];
        for (int i = 1; i <= amount; i++) t[0][i] = Integer.MAX_VALUE - 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= amount; j++) {
                if (j - param1[i - 1] >= 0)
                    t[i][j] = Math.min(t[i][j - param1[i - 1]] + 1, t[i - 1][j]);
                else
                    t[i][j] = t[i - 1][j];
            }
        }
        return t[n][amount] == Integer.MAX_VALUE - 1 ? -1 : t[n][amount];
    }
}
