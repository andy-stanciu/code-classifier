class Solution {

    public int method1(int[] param1, int amount) {
        int[][] cc = new int[param1.length + 1][amount + 1];
        for (int i = 0; i < param1.length + 1; i++) {
            for (int amt = 0; amt < amount + 1; amt++) {
                if (i == 0 || amt == 0)
                    cc[i][amt] = (amt == 0 ? 0 : amount + 1);
            }
        }
        for (int i = 1; i < param1.length + 1; i++) {
            for (int amt = 1; amt < amount + 1; amt++) {
                if (amt >= param1[i - 1]) {
                    cc[i][amt] = Math.min(1 + cc[i][amt - param1[i - 1]], cc[i - 1][amt]);
                } else {
                    cc[i][amt] = cc[i - 1][amt];
                }
            }
        }
        int ans = cc[param1.length][amount];
        return ans >= amount + 1 ? -1 : ans;
    }
}
