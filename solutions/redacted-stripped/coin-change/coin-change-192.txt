class Solution {

    public int method1(int[] param1, int amount) {
        int ans = method2(param1, amount, 0);
        return ans == Integer.MAX_VALUE - 1 ? -1 : ans;
    }

    int method2(int[] param1, int amount, int ind) {
        if (ind >= param1.length || amount <= 0) {
            return (amount == 0 ? 0 : Integer.MAX_VALUE - 1);
        }
        int res = -1;
        if (param1[ind] > amount) {
            int notTake = 0 + method2(param1, amount, ind + 1);
            res = notTake;
        } else {
            int take = 1 + method2(param1, amount - param1[ind], ind);
            int notTake = 0 + method2(param1, amount, ind + 1);
            res = Math.min(take, notTake);
        }
        return res;
    }
}
