class Solution {

    public int method1(int[] param1, int amount) {
        int[] t = new int[amount + 1];
        Arrays.fill(t, amount + 1);
        t[0] = 0;
        for (int i = 1; i <= amount; i++) {
            for (int var1 : param1) {
                if (i >= var1) {
                    t[i] = Math.min(t[i], 1 + t[i - var1]);
                }
            }
        }
        return t[amount] > amount ? -1 : t[amount];
    }
}
