class Solution {

    public int method1(int[] param1, int S) {
        if (S == 0)
            return 0;
        int n = param1.length;
        int[][] dp = new int[n + 1][S + 1];
        for (int j = 1; j <= S; j++) dp[0][j] = Integer.MAX_VALUE - 1;
        for (int i = 0; i <= n; i++) dp[i][0] = 0;
        for (int i = 1; i <= n; i++) for (int j = 1; j <= S; j++) {
            if (param1[i - 1] <= j)
                dp[i][j] = Math.min(1 + dp[i][j - param1[i - 1]], dp[i - 1][j]);
            else
                dp[i][j] = dp[i - 1][j];
        }
        return dp[n][S] != Integer.MAX_VALUE - 1 ? dp[n][S] : -1;
    }
}
