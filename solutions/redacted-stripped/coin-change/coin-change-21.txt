class Solution {

    public int method1(int[] param1, int amount) {
        int res = recursion(param1, amount, 0);
        return res == Integer.MAX_VALUE ? -1 : res;
    }

    public int recursion(int[] param1, int n, int number) {
        if (n == 0)
            return number;
        if (n < 0)
            return Integer.MAX_VALUE;
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < param1.length; i++) res = Math.min(res, recursion(param1, n - param1[i], number + 1));
        return res;
    }
}
