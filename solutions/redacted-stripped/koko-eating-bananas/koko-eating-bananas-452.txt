class Solution {

    public int method1(int[] piles, int h) {
        int left = 1;
        int right = Integer.MIN_VALUE;
        int mid = 0;
        for (int i : piles) {
            if (i > right)
                right = i;
        }
        while (left < right) {
            mid = (left + right) / 2;
            int totalpiles = 0;
            for (int i = 0; i < piles.length; i++) {
                totalpiles += (int) Math.ceil((double) piles[i] / mid);
            }
            if (totalpiles <= h) {
                right = mid;
            } else if (totalpiles > h) {
                left = mid + 1;
            }
        }
        return right;
    }
}
