class Solution {

    public int method1(int[] piles, int h) {
        int n = piles.length;
        int maxi = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            maxi = Math.max(maxi, piles[i]);
        }
        int low = 0;
        int high = maxi;
        while (low <= high) {
            int mid = (low + high) / 2;
            int ans = calculate(piles, mid);
            if (ans <= h)
                high = mid - 1;
            else
                low = mid + 1;
        }
        return low;
    }

    public static int calculate(int[] arr, int hrs) {
        int totalhrs = 0;
        for (int i = 0; i < arr.length; i++) {
            totalhrs += Math.ceil((double) (arr[i]) / (double) (hrs));
        }
        return totalhrs;
    }
}
