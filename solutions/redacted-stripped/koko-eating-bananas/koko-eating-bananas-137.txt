class Solution {

    static int method1(int[] piles, int h) {
        int max = getMax(piles);
        int s = 1;
        int e = max;
        while (s <= e) {
            long k = 0;
            int m = s + (e - s) / 2;
            for (int pile : piles) {
                k = k + (Math.ceilDiv(pile, m));
            }
            if (k > h) {
                s = m + 1;
            } else {
                e = m - 1;
            }
        }
        return s;
    }

    static int getMax(int[] piles) {
        int max1 = piles[0];
        for (int i = 1; i < piles.length; i++) {
            if (piles[i] > max1) {
                max1 = piles[i];
            }
        }
        return max1;
    }
}
