class Solution {

    public boolean isvalid(int mid, int h, int[] arr) {
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            count += arr[i] / mid;
            if (arr[i] % mid != 0) {
                count += 1;
            }
            if (count > h)
                return false;
        }
        return true;
    }

    public int method1(int[] arr, int h) {
        int n = arr.length;
        int max = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            if (arr[i] > max)
                max = arr[i];
        }
        int lo = 1, hi = max;
        int ans = -1;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (isvalid(mid, h, arr)) {
                ans = mid;
                hi = mid - 1;
            } else {
                lo = mid + 1;
            }
        }
        return ans;
    }
}
