class Solution {

    public int findMax(int[] piles) {
        int max = Integer.MIN_VALUE;
        int n = piles.length;
        for (int i = 0; i < n; i++) {
            max = Math.max(max, piles[i]);
        }
        return max;
    }

    public int totalHours(int[] piles, int h) {
        int total = 0;
        int n = piles.length;
        for (int i = 0; i < n; i++) {
            total += Math.ceil((double) (piles[i]) / (double) (h));
        }
        return total;
    }

    public int method1(int[] piles, int h) {
        int low = 1;
        int high = findMax(piles);
        while (low <= high) {
            int mid = (low + high) / 2;
            int total_hours = totalHours(piles, mid);
            if (total_hours <= h) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
}
