class Solution {

    public int method1(int[] piles, int h) {
        Arrays.sort(piles);
        int low = 0, high = piles[piles.length - 1], mid;
        while (low <= high) {
            mid = (low + high) / 2;
            int ans = this.countHours(piles, mid);
            if (ans <= h)
                high = mid - 1;
            else
                low = mid + 1;
        }
        return low;
    }

    public int countHours(int[] param1, int k) {
        int ans = 0;
        for (Integer i : param1) ans += (Math.ceil((double) i / (double) k));
        return ans;
    }
}
