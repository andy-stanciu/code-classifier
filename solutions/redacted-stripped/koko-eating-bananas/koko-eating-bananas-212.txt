class Solution {

    public int method1(int[] piles, int h) {
        int minspeed = 1;
        int maxspeed = 0;
        for (int i : piles) {
            maxspeed = Math.max(maxspeed, i);
        }
        while (minspeed < maxspeed) {
            int mid = minspeed + (maxspeed - minspeed) / 2;
            if (method2(piles, h, mid)) {
                maxspeed = mid;
            } else {
                minspeed = mid + 1;
            }
        }
        return minspeed;
    }

    private boolean method2(int[] piles, int h, int speed) {
        int hours = 0;
        for (int i : piles) {
            hours += (int) Math.ceil((double) i / speed);
        }
        return hours <= h;
    }
}
