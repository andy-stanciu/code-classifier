class Solution {

    public int findTotalHrs(int[] piles, int hourlyRate) {
        int totalHrs = 0;
        for (int i = 0; i < piles.length; i++) {
            totalHrs += Math.ceil((double) piles[i] / (double) hourlyRate);
        }
        return totalHrs;
    }

    public int method1(int[] piles, int h) {
        int low = 1, high = Arrays.stream(piles).max().getAsInt();
        while (low <= high) {
            int mid = (low + high) / 2;
            int totalHrs = findTotalHrs(piles, mid);
            if (totalHrs <= h)
                high = mid - 1;
            else
                low = mid + 1;
        }
        return low;
    }
}
