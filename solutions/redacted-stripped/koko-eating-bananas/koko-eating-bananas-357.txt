class Solution {

    public boolean bananaEat(int[] arr, int mid, int h) {
        int banana = 0;
        for (int i = 0; i < arr.length; i++) {
            banana += (arr[i] / mid);
            if (arr[i] % mid != 0) {
                banana++;
            }
        }
        return banana <= h;
    }

    public int method1(int[] piles, int h) {
        if (piles.length == 1) {
            return (int) (Math.ceil(piles[0] * 1.0 / h));
        }
        Arrays.sort(piles);
        int low = 1;
        int high = piles[piles.length - 1];
        while (low < high) {
            int mid = low + (high - low) / 2;
            if (bananaEat(piles, mid, h)) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
}
