class Solution {

    public int method1(int[] piles, int h) {
        int n = piles.length;
        int maxEle = piles[0];
        for (int i = 1; i < n; i++) {
            maxEle = Math.max(piles[i], maxEle);
        }
        int left = 1;
        int right = maxEle;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            int sum = 0;
            for (int i = 0; i < n; i++) {
                sum += Math.ceil((double) piles[i] / mid);
            }
            if (sum > h) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return left;
    }
}
