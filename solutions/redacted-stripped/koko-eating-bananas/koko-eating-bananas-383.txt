class Solution {

    public int method1(int[] piles, int h) {
        Arrays.sort(piles);
        int lo = 1;
        int hi = Integer.MAX_VALUE;
        int ans = 0;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (ispossible(piles, mid, h)) {
                hi = mid - 1;
                ans = mid;
            } else {
                lo = mid + 1;
            }
        }
        return ans;
    }

    public static boolean ispossible(int[] piles, int mid, int h) {
        int hours = 0;
        for (int pile : piles) {
            hours += (pile + mid - 1) / mid;
        }
        return hours <= h;
    }
}
