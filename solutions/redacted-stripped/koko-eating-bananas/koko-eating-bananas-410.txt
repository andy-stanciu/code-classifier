class Solution {

    public static int findMaximum(int[] arr) {
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < arr.length; i++) {
            max = Math.max(max, arr[i]);
        }
        return max;
    }

    public static int countHours(int[] arr, int hourly) {
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            count += Math.ceil((double) (arr[i]) / (double) (hourly));
        }
        return count;
    }

    public int method1(int[] piles, int h) {
        int low = 1;
        int high = findMaximum(piles);
        while (low <= high) {
            int mid = (low + high) / 2;
            int totalHours = countHours(piles, mid);
            if (totalHours <= h) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
}
