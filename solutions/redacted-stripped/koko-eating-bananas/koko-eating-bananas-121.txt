class Solution {

    public int method1(int[] piles, int h) {
        long l = 1, r = (long) (1e9);
        while (l < r) {
            long m = (l + r) / 2;
            if (canFinish(piles, m, (long) h) == true) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        return (int) l;
    }

    public boolean canFinish(int[] piles, long param1, long h) {
        long total_time = 0;
        for (int i = 0; i < piles.length; i++) {
            long cur_time = (long) piles[i] / param1;
            if ((long) piles[i] % param1 != 0)
                cur_time++;
            total_time += cur_time;
        }
        return total_time <= h;
    }
}
