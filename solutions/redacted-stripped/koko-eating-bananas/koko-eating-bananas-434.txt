class Solution {

    public int method1(int[] piles, int h) {
        int start = 1, end = 0;
        for (int x : piles) {
            end = Math.max(end, x);
        }
        while (start <= end) {
            int mid = (start + end) / 2;
            int hours = 0;
            for (int x : piles) {
                hours += Math.ceil((double) x / (double) mid);
            }
            if (hours <= h)
                end = mid - 1;
            else
                start = mid + 1;
        }
        return start;
    }
}
