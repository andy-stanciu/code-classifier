class Solution {

    public int method1(int[] piles, int h) {
        int first = 1;
        int last = Arrays.stream(piles).max().getAsInt();
        while (first < last) {
            int mid = (first + last) / 2;
            int eatH = 0;
            for (int pile : piles) {
                eatH += (pile + mid - 1) / mid;
            }
            if (eatH > h) {
                first = mid + 1;
            } else {
                last = mid;
            }
        }
        return first;
    }
}
