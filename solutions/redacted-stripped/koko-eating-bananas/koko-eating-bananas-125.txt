class Solution {

    public int method1(int[] piles, int h) {
        int minSpeed = 1;
        int maxSpeed = 0;
        for (int pile : piles) {
            maxSpeed = Math.max(maxSpeed, pile);
        }
        while (minSpeed < maxSpeed) {
            int mid = minSpeed + (maxSpeed - minSpeed) / 2;
            if (method2(piles, h, mid)) {
                maxSpeed = mid;
            } else {
                minSpeed = mid + 1;
            }
        }
        return minSpeed;
    }

    boolean method2(int[] piles, int h, int speed) {
        int totalSpeed = 0;
        for (int pile : piles) {
            totalSpeed += Math.ceil((double) pile / (double) speed);
        }
        return totalSpeed <= h;
    }
}
