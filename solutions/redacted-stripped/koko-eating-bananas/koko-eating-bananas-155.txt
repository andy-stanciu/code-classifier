class Solution {

    public int method1(int[] piles, int h) {
        int var1 = 0;
        for (int n : piles) if (var1 < n)
            var1 = n;
        int method1 = 1;
        while (method1 <= var1) {
            int guessSpeed = (method1 + var1) / 2;
            int minHrs = minHours(piles, guessSpeed);
            if (minHrs > h)
                method1 = guessSpeed + 1;
            else
                var1 = guessSpeed - 1;
        }
        return method1;
    }

    private static int minHours(int[] piles, int k) {
        int minHrs = 0;
        for (int ele : piles) {
            minHrs += Math.ceil((double) ele / k);
        }
        return minHrs;
    }
}
