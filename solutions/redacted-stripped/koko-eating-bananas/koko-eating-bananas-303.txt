class Solution {

    public int method1(int[] piles, int h) {
        int i;
        int n = piles.length;
        int minSpeed, maxSpeed = piles[0];
        for (i = 0; i < n; i++) {
            maxSpeed = Math.max(maxSpeed, piles[i]);
        }
        minSpeed = 1;
        while (minSpeed < maxSpeed) {
            int speed = minSpeed + (maxSpeed - minSpeed) / 2;
            int time = 0;
            for (i = 0; i < n; i++) {
                time += Math.ceil((double) piles[i] / speed);
                if (time > h) {
                    break;
                }
            }
            if (time <= h) {
                maxSpeed = speed;
            } else {
                minSpeed = speed + 1;
            }
        }
        return minSpeed;
    }
}
