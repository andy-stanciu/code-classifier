class Solution {

    public int method1(int[] piles, int h) {
        int maxPile = Arrays.stream(piles).max().getAsInt();
        int start = 1;
        int end = maxPile;
        while (start < end) {
            int mid = start + (end - start) / 2;
            int totalHours = 0;
            for (int pile : piles) {
                totalHours += (pile + mid - 1) / mid;
            }
            if (totalHours <= h) {
                end = mid;
            } else {
                start = mid + 1;
            }
        }
        return start;
    }
}
