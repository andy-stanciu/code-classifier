class Solution {

    public int method1(int[] piles, int h) {
        int max = Arrays.stream(piles).max().getAsInt();
        int high = max;
        int low = 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (cal(piles, mid) <= h) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }

    public static int cal(int[] arr, int mid) {
        int total = 0;
        for (int i = 0; i < arr.length; i++) {
            total += Math.ceil((double) (arr[i]) / (double) (mid));
        }
        return total;
    }
}
