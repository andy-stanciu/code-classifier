class Solution {

    public int method1(int[] p, int h) {
        int max = p[0];
        for (int n : p) {
            if (n > max) {
                max = n;
            }
        }
        int left = 1;
        int right = max;
        int res = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (is_eatable(p, mid, h)) {
                res = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return res;
    }

    private boolean is_eatable(int[] p, int param1, int hours) {
        int var1 = 0;
        for (int i = 0; i < p.length; i++) {
            var1 += p[i] / param1;
            if (p[i] % param1 != 0) {
                var1 += 1;
            }
            if (var1 > hours) {
                return false;
            }
        }
        return true;
    }
}
