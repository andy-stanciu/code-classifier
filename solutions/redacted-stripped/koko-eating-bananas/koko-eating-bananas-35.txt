class Solution {

    public int method1(int[] piles, int h) {
        int left = 1, right = (int) 1e9;
        while (left < right) {
            int mid = (left + right) / 2;
            if (method2(piles, h, mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

    private boolean method2(int[] piles, int h, int k) {
        int hours = 0;
        for (int var1 : piles) {
            int pileHours = var1 / k;
            if (var1 % k != 0) {
                pileHours++;
            }
            hours += pileHours;
            if (hours > h) {
                return false;
            }
        }
        return true;
    }
}
