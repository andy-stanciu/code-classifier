class Solution {

    public int method1(int[] piles, int h) {
        int high = 0;
        for (int pile : piles) high = Math.max(high, pile);
        int low = 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            long timeTaken = findTimeTaken(piles, mid);
            if (timeTaken == h && low == high)
                return mid;
            else if (timeTaken > h)
                low = mid + 1;
            else
                high = mid - 1;
        }
        return low;
    }

    private static long findTimeTaken(int[] piles, int midPile) {
        long totalTimeTaken = 0L;
        for (int pile : piles) {
            int timeTaken = Math.ceilDiv(pile, midPile);
            totalTimeTaken += timeTaken;
        }
        return totalTimeTaken;
    }
}
