class Solution {

    public int method1(int[] piles, int h) {
        return method2(piles, h);
    }

    public int method2(int[] piles, int hours) {
        int min = 1, max = Integer.MAX_VALUE;
        while (min <= max) {
            int middle = min + (max - min) / 2;
            if (method3(piles, middle, hours))
                max = middle - 1;
            else
                min = middle + 1;
        }
        return min;
    }

    public boolean method3(int[] piles, double speed, int hours) {
        int time = 0;
        for (int banana : piles) time += Math.ceil(banana / speed);
        return time <= hours;
    }
}
