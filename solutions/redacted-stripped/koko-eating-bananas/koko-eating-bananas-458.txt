class Solution {

    public int method1(int[] piles, int h) {
        int left = 1;
        int right = 0;
        for (int x : piles) right = Math.max(right, x);
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (condition(piles, mid, h)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

    public boolean condition(int[] piles, int mid, int h) {
        int count = 0;
        for (int x : piles) {
            if (x % mid == 0)
                count += x / mid;
            else {
                count += x / mid + 1;
            }
            if (count > h)
                return false;
        }
        return true;
    }
}
