class Solution {

    public int method1(int[] piles, int H) {
        int low = 1, high = 1000000000, k = 0;
        while (low <= high) {
            k = (low + high) / 2;
            int totalHours = 0;
            for (int i = 0; i < piles.length; i++) totalHours += Math.ceil(1.0 * piles[i] / k);
            if (totalHours > H)
                low = k + 1;
            else
                high = k - 1;
        }
        return low;
    }
}
