class Solution {

    public int method1(int[] piles, int H) {
        int left = 1, right = 1000000000;
        while (left < right) {
            int mid = (left + right) / 2, total = 0;
            for (int pile : piles) total += (pile + mid - 1) / mid;
            if (total > H)
                left = mid + 1;
            else
                right = mid;
        }
        return left;
    }
}
