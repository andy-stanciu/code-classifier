class Solution {

    public int method1(int[] piles, int H) {
        int maxValue = 0;
        for (int i : piles) {
            maxValue = Math.max(i, maxValue);
        }
        int lo = 1;
        int hi = maxValue;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            long hours = calculateHours(piles, mid);
            if (hours > H) {
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }
        return lo;
    }

    private long calculateHours(int[] piles, int speed) {
        long hours = 0;
        for (int pile : piles) {
            hours += ((pile - 1) / speed) + 1;
        }
        return hours;
    }
}
