class Solution {

    public int method1(int[] piles, int h) {
        int low = 1;
        int high = Integer.MIN_VALUE;
        for (int i = 0; i < piles.length; i++) {
            high = Math.max(high, piles[i]);
        }
        while (high > low) {
            int mid = low + (high - low) / 2;
            int currentDiv = 0;
            for (int i = 0; i < piles.length; i++) {
                currentDiv += piles[i] / mid;
                if (piles[i] % mid != 0) {
                    currentDiv++;
                }
            }
            if (currentDiv <= h) {
                high = mid;
            } else if (currentDiv > h) {
                low = mid + 1;
            }
        }
        return high;
    }
}
