class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        method2(candidates, 0, 0, target, new ArrayList<Integer>(), ans);
        return ans;
    }

    private void method2(int[] candidates, int idx, int param1, int target, List<Integer> ls, List<List<Integer>> ans) {
        if (param1 > target)
            return;
        if (param1 == target) {
            ans.add(new ArrayList<>(ls));
        } else {
            for (int i = idx; i < candidates.length; i++) {
                if (i != idx && candidates[i] == candidates[i - 1])
                    continue;
                ls.add(candidates[i]);
                param1 += candidates[i];
                method2(candidates, i + 1, param1, target, ls, ans);
                ls.remove(ls.size() - 1);
                param1 -= candidates[i];
            }
        }
    }
}
