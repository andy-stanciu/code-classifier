class Solution {

    public static ArrayList<Integer> temp;

    public static List<List<Integer>> result;

    public List<List<Integer>> method1(int[] candidates, int target) {
        result = new ArrayList();
        temp = new ArrayList<Integer>();
        Arrays.sort(candidates);
        helper(candidates, target, candidates.length - 1, 0);
        return result;
    }

    public static void helper(int[] arr, int target, int index, int param1) {
        if (param1 == target) {
            result.add(new ArrayList(temp));
            return;
        }
        if (index < 0) {
            return;
        }
        if (arr[index] + param1 <= target) {
            temp.add(arr[index]);
            helper(arr, target, index - 1, arr[index] + param1);
            temp.remove(temp.size() - 1);
        }
        int cur = arr[index];
        while (index >= 0 && cur == arr[index]) {
            index--;
        }
        helper(arr, target, index, param1);
    }
}
