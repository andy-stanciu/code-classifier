class Solution {

    List<List<Integer>> ans = null;

    public void method1(int[] candidates, int l, int param1, int target, List<Integer> list) {
        if (param1 == target) {
            ans.add(new ArrayList<>(list));
            return;
        }
        for (int i = l; i < candidates.length; i++) {
            if (param1 + candidates[i] > target)
                continue;
            if (i != 0 && candidates[i] == candidates[i - 1] && i > l) {
                continue;
            }
            list.add(candidates[i]);
            method1(candidates, i + 1, param1 + candidates[i], target, list);
            list.remove(list.size() - 1);
        }
    }

    public List<List<Integer>> method2(int[] candidates, int target) {
        ans = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        Arrays.sort(candidates);
        method1(candidates, 0, 0, target, list);
        return ans;
    }
}
