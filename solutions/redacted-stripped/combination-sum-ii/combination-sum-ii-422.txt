class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> answer = new ArrayList<>();
        Arrays.sort(candidates);
        helper(answer, new ArrayList<>(), candidates, target, 0);
        return answer;
    }

    public void helper(List<List<Integer>> answer, ArrayList<Integer> temp, int[] candidates, int target, int index) {
        if (target == 0) {
            answer.add(new ArrayList<>(temp));
            return;
        }
        for (int i = index; i < candidates.length; i++) {
            if (candidates[i] <= target) {
                if (i > index && candidates[i] == candidates[i - 1]) {
                    continue;
                }
                temp.add(candidates[i]);
                helper(answer, temp, candidates, target - candidates[i], i + 1);
                temp.remove(temp.size() - 1);
            }
        }
    }
}
