class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> answer = new ArrayList<>();
        Arrays.sort(candidates);
        method2(0, target, candidates, new ArrayList<>(), answer);
        return answer;
    }

    public static void method2(int index, int target, int[] candidates, List<Integer> list, List<List<Integer>> answer) {
        if (target == 0) {
            answer.add(new ArrayList<>(list));
            return;
        }
        for (int i = index; i < candidates.length; i++) {
            if (i > index && candidates[i] == candidates[i - 1])
                continue;
            if (candidates[i] > target)
                break;
            list.add(candidates[i]);
            method2(i + 1, target - candidates[i], candidates, list, answer);
            list.remove(list.size() - 1);
        }
    }
}
