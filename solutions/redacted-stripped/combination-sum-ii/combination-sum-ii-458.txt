class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(candidates);
        helper(candidates, target, result, new ArrayList<>(), 0);
        return result;
    }

    private void helper(int[] candidates, int target, List<List<Integer>> result, List<Integer> param1, int idx) {
        if (target <= 0) {
            if (target == 0) {
                result.add(new ArrayList<>(param1));
            }
            return;
        }
        for (int i = idx; i < candidates.length; i++) {
            if (i > idx && candidates[i] == candidates[i - 1]) {
                continue;
            }
            param1.add(candidates[i]);
            helper(candidates, target - candidates[i], result, param1, i + 1);
            param1.remove(param1.size() - 1);
        }
    }
}
