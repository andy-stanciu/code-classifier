class Solution {

    void helper(int[] candidate, int target, int index, List<Integer> al, List<List<Integer>> res, int param1) {
        if (param1 == target) {
            res.add(new ArrayList<>(al));
            return;
        }
        if (index >= candidate.length || param1 > target)
            return;
        al.add(candidate[index]);
        helper(candidate, target, index + 1, al, res, param1 + candidate[index]);
        al.remove(al.size() - 1);
        while (index < candidate.length - 1 && candidate[index] == candidate[index + 1]) index++;
        helper(candidate, target, index + 1, al, res, param1);
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> res = new ArrayList<>();
        helper(candidates, target, 0, new ArrayList<>(), res, 0);
        return res;
    }
}
