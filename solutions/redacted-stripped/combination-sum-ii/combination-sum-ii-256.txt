class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        ArrayList<List<Integer>> results = new ArrayList<>();
        ArrayList<Integer> var1 = new ArrayList<>();
        Arrays.sort(candidates);
        method2(candidates, target, 0, var1, results);
        return results;
    }

    public void method2(int[] candidates, int target, int startIndex, ArrayList<Integer> var1, ArrayList<List<Integer>> results) {
        if (target == 0) {
            results.add(new ArrayList<>(var1));
            return;
        }
        for (int i = startIndex; i < candidates.length; i++) {
            if (i != startIndex && candidates[i - 1] == candidates[i])
                continue;
            if (candidates[i] <= target) {
                var1.add(candidates[i]);
                method2(candidates, target - candidates[i], i + 1, var1, results);
                var1.remove(var1.size() - 1);
            }
        }
    }
}
