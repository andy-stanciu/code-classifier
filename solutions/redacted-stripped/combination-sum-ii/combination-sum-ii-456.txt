class Solution {

    public static void method1(int index, int[] arr, List<Integer> ds, List<List<Integer>> ans, int target) {
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = index; i < arr.length; i++) {
            if (i > index && arr[i] == arr[i - 1])
                continue;
            if (arr[i] > target)
                break;
            ds.add(arr[i]);
            method1(i + 1, arr, ds, ans, target - arr[i]);
            ds.remove(ds.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        method1(0, candidates, new ArrayList<>(), ans, target);
        return ans;
    }
}
