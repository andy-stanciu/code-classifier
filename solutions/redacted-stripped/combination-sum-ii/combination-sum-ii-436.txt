class Solution {

    public List<List<Integer>> l;

    public List<Integer> ll;

    public List<List<Integer>> method1(int[] candidates, int target) {
        l = new ArrayList<>();
        ll = new ArrayList<>();
        Arrays.sort(candidates);
        method2(candidates, target, 0, 0);
        return l;
    }

    public void method2(int[] candidates, int target, int i, int param1) {
        if (param1 == target) {
            l.add(new ArrayList<>(ll));
            return;
        }
        if (param1 > target) {
            return;
        }
        if (i == candidates.length) {
            return;
        }
        ll.add(candidates[i]);
        method2(candidates, target, i + 1, param1 + candidates[i]);
        ll.remove(ll.size() - 1);
        while (i + 1 < candidates.length && candidates[i] == candidates[i + 1]) {
            i++;
        }
        method2(candidates, target, i + 1, param1);
    }
}
