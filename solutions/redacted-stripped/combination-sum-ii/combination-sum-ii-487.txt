class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(candidates);
        List<Integer> var1 = new ArrayList<>();
        method2(candidates, target, 0, var1, result);
        return result;
    }

    public static void method2(int[] candidates, int target, int startIndex, List<Integer> var1, List<List<Integer>> result) {
        if (target == 0) {
            result.add(new ArrayList<>(var1));
            return;
        }
        for (int i = startIndex; i < candidates.length; i++) {
            if (i > startIndex && candidates[i] == candidates[i - 1]) {
                continue;
            }
            if (candidates[i] > target) {
                break;
            }
            var1.add(candidates[i]);
            method2(candidates, target - candidates[i], i + 1, var1, result);
            var1.remove(var1.size() - 1);
        }
    }
}
