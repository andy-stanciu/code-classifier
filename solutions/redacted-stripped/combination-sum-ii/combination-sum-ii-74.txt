class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public void helper(int[] candidates, int target, int li, ArrayList<Integer> asf) {
        if (target == 0) {
            ans.add(new ArrayList<>(asf));
            return;
        }
        for (int i = li; i < candidates.length; i++) {
            if (i > li && candidates[i] == candidates[i - 1])
                continue;
            if (target - candidates[i] < 0)
                break;
            asf.add(candidates[i]);
            helper(candidates, target - candidates[i], i + 1, asf);
            asf.remove(asf.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        helper(candidates, target, 0, new ArrayList<>());
        return ans;
    }
}
