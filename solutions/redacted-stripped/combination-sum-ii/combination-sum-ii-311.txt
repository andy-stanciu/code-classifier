class Solution {

    public void fun(List<List<Integer>> ls, List<Integer> temp, int[] candidates, int target, int idx, int param1) {
        if (param1 == target) {
            ls.add(new ArrayList<>(temp));
            return;
        }
        if (param1 > target)
            return;
        for (int i = idx; i < candidates.length; i++) {
            if (i > idx && candidates[i] == candidates[i - 1])
                continue;
            temp.add(candidates[i]);
            fun(ls, temp, candidates, target, i + 1, param1 + candidates[i]);
            temp.remove(temp.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ls = new ArrayList<>();
        fun(ls, new ArrayList<>(), candidates, target, 0, 0);
        return ls;
    }
}
