class Solution {

    public List<List<Integer>> method1(int k, int n) {
        List<List<Integer>> ans = new ArrayList<>();
        method2(ans, new ArrayList<Integer>(), k, 1, n);
        return ans;
    }

    private void method2(List<List<Integer>> ans, List<Integer> comb, int k, int start, int n) {
        if (comb.size() == k && n == 0) {
            ans.add(new ArrayList<Integer>(comb));
            return;
        }
        for (int i = start; i <= 9; i++) {
            comb.add(i);
            method2(ans, comb, k, i + 1, n - i);
            comb.remove(comb.size() - 1);
        }
    }
}
