class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ans = new ArrayList<List<Integer>>();
        method1(ans, candidates, target, 0, 0, new ArrayList<Integer>());
        return ans;
    }

    private void method1(List<List<Integer>> ans, int[] candidates, int target, int start, int param1, List<Integer> list) {
        if (param1 == target) {
            ans.add(new ArrayList<>(list));
        } else if (param1 > target) {
            return;
        }
        for (int j = start; j < candidates.length; j++) {
            if (j > start && candidates[j - 1] == candidates[j]) {
                continue;
            }
            list.add(candidates[j]);
            method1(ans, candidates, target, j + 1, param1 + candidates[j], list);
            list.remove(list.size() - 1);
        }
    }
}
