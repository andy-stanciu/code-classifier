class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> li = new ArrayList<>();
        recursive(candidates, new ArrayList<>(), li, 0, target);
        return li;
    }

    static void recursive(int[] cand, List<Integer> newli, List<List<Integer>> li, int index, int target) {
        if (target == 0) {
            li.add(new ArrayList<>(newli));
            return;
        } else {
            for (int i = index; i < cand.length; i++) {
                if (i > index && cand[i] == cand[i - 1])
                    continue;
                if (cand[i] > target)
                    break;
                newli.add(cand[i]);
                recursive(cand, newli, li, i + 1, target - cand[i]);
                newli.remove(newli.size() - 1);
            }
        }
    }
}
