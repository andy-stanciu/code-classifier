class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        ArrayList<Integer> list = new ArrayList<>();
        ArrayList<List<Integer>> result = new ArrayList<>();
        return method2(candidates, 0, 0, target, list, result);
    }

    static List<List<Integer>> method2(int[] candidates, int index, int method2, int target, List<Integer> list, List<List<Integer>> result) {
        if (method2 == target) {
            result.add(new ArrayList<>(list));
            return result;
        }
        for (int i = index; i < candidates.length; i++) {
            if (i > index && candidates[i] == candidates[i - 1]) {
                continue;
            }
            if (method2 > target) {
                return result;
            }
            list.add(candidates[i]);
            method2(candidates, i + 1, method2 + candidates[i], target, list, result);
            int num = list.size() - 1;
            list.remove(num);
        }
        return result;
    }
}
