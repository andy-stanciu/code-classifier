class Solution {

    public static void combi(int i, List<List<Integer>> list, List<Integer> temp, int[] candidates, int target) {
        if (target == 0) {
            list.add(new ArrayList<>(temp));
            return;
        }
        if (target < 0 || i == candidates.length) {
            return;
        }
        temp.add(candidates[i]);
        combi(i + 1, list, temp, candidates, target - candidates[i]);
        while (i + 1 < candidates.length && candidates[i] == candidates[i + 1]) {
            i++;
        }
        temp.remove(temp.size() - 1);
        combi(i + 1, list, temp, candidates, target);
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        combi(0, list, temp, candidates, target);
        return list;
    }
}
