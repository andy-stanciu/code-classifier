class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        Arrays.sort(candidates);
        solve(candidates, target, 0, 0, res, cur);
        return res;
    }

    private void solve(int[] candidates, int target, int param1, int i, List<List<Integer>> res, List<Integer> cur) {
        if (param1 == target) {
            res.add(new ArrayList<>(cur));
            return;
        }
        if (param1 > target || i == candidates.length)
            return;
        cur.add(candidates[i]);
        solve(candidates, target, param1 + candidates[i], i + 1, res, cur);
        cur.remove(cur.size() - 1);
        int nextIndex = i + 1;
        while (nextIndex < candidates.length && candidates[nextIndex] == candidates[i]) {
            nextIndex++;
        }
        solve(candidates, target, param1, nextIndex, res, cur);
    }
}
