class Solution {

    public void helper(int[] candidates, int target, List<List<Integer>> res, List<Integer> temp, int param1, int index) {
        for (int i = index; i < candidates.length; i++) {
            if (i > index && candidates[i] == candidates[i - 1])
                continue;
            temp.add(candidates[i]);
            param1 += candidates[i];
            if (param1 == target)
                res.add(new ArrayList<>(temp));
            if (param1 < target)
                helper(candidates, target, res, temp, param1, i + 1);
            temp.remove(Integer.valueOf(candidates[i]));
            param1 -= candidates[i];
            if (param1 >= target)
                break;
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> res = new ArrayList<>();
        helper(candidates, target, res, new ArrayList<>(), 0, 0);
        return res;
    }
}
