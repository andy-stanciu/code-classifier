class Solution {

    public int method1(int[] nums, int target) {
        int var1 = nums[0] + nums[1] + nums[2];
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target) {
                    return sum;
                }
                var1 = (Math.abs(target - sum) < Math.abs(target - var1)) ? sum : var1;
                if (sum < target) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return var1;
    }
}
