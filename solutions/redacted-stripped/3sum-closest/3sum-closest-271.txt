class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int initialSum = nums[0] + nums[1] + nums[2];
        int diff = Math.abs(target - initialSum);
        for (int i = 0; i < nums.length; i++) {
            int lft = i + 1;
            int rght = nums.length - 1;
            while (lft < rght) {
                int calcSum = nums[i] + nums[lft] + nums[rght];
                int calcDiff = Math.abs(calcSum - target);
                if (calcDiff < diff) {
                    initialSum = calcSum;
                    diff = calcDiff;
                }
                if (calcSum < target) {
                    lft++;
                } else if (calcSum > target) {
                    rght--;
                } else if (calcSum == target) {
                    return calcSum;
                }
            }
        }
        return initialSum;
    }
}
