class Solution {

    public int method1(int[] nums, int target) {
        int minDiff = Integer.MAX_VALUE;
        int res = -1;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            int s = i + 1;
            int e = nums.length - 1;
            while (s < e) {
                int sum = nums[i] + nums[s] + nums[e];
                int diff = Math.abs(target - sum);
                if (diff < minDiff) {
                    minDiff = diff;
                    res = sum;
                } else if (sum > target) {
                    e--;
                } else {
                    s++;
                }
            }
        }
        return res;
    }
}
