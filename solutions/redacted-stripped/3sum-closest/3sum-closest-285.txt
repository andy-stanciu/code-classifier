class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int output = nums[0] + nums[1] + nums[2];
        int difference = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 2; i++) {
            int leftPointer = i + 1;
            int rightPointer = nums.length - 1;
            while (leftPointer < rightPointer) {
                int sum = nums[i] + nums[leftPointer] + nums[rightPointer];
                if (sum == target) {
                    return target;
                } else if (sum < target) {
                    leftPointer++;
                } else {
                    rightPointer--;
                }
                if (difference > Math.abs(sum - target)) {
                    difference = Math.abs(sum - target);
                    output = sum;
                }
            }
        }
        return output;
    }
}
