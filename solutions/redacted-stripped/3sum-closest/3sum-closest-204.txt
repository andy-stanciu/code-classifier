class Solution {

    public int method1(int[] nums, int target) {
        int var1 = 0;
        int minDiff = Integer.MAX_VALUE;
        Arrays.sort(nums);
        int currSum;
        int currDiff;
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                currSum = nums[i] + nums[j] + nums[k];
                if (currSum == target) {
                    return target;
                }
                currDiff = Math.abs(currSum - target);
                if (currDiff < minDiff) {
                    minDiff = currDiff;
                    var1 = currSum;
                }
                if (currSum > target) {
                    k--;
                } else {
                    j++;
                }
            }
        }
        return var1;
    }
}
