class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int ans = Integer.MAX_VALUE;
        int diff = ans;
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                if (i == j || j == k)
                    break;
                int sum = nums[i] + nums[j] + nums[k];
                if (sum == target) {
                    return sum;
                } else if (sum < target) {
                    int curr_diff = Math.abs(sum - target);
                    if (curr_diff < diff) {
                        diff = curr_diff;
                        ans = sum;
                    }
                    j++;
                } else {
                    int curr_diff = Math.abs(sum - target);
                    if (curr_diff < diff) {
                        diff = curr_diff;
                        ans = sum;
                    }
                    k--;
                }
            }
        }
        return ans;
    }
}
