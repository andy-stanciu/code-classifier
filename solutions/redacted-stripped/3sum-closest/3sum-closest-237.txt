class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        Arrays.sort(nums);
        int var1 = Integer.MAX_VALUE;
        int var2 = Integer.MAX_VALUE;
        for (int i = 0; i < n - 2; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            int left = i + 1, right = n - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                int diff = Math.abs(sum - target);
                if (diff == 0) {
                    return sum;
                } else if (diff < var2) {
                    var1 = sum;
                    var2 = diff;
                }
                if (sum < target) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return var1;
    }
}
