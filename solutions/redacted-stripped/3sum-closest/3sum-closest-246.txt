class Solution {

    public int method1(int[] nums, int target) {
        final int n = nums.length;
        Arrays.sort(nums);
        int min = Integer.MAX_VALUE;
        int var1 = 0;
        for (int i = 0; i < n - 2; ++i) {
            int low = i + 1;
            int high = n - 1;
            while (low < high) {
                int sum = nums[i] + nums[low] + nums[high];
                if (sum == target) {
                    return sum;
                } else if (sum < target) {
                    ++low;
                } else {
                    --high;
                }
                int diff = Math.abs(sum - target);
                if (diff < min) {
                    min = diff;
                    var1 = sum;
                }
            }
        }
        return var1;
    }
}
