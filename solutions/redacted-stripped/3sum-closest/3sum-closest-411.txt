class Solution {

    public int method1(int[] nums, int target) {
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        Arrays.sort(nums);
        int length = nums.length;
        for (int i = 0; i < length - 2; i++) {
            int left = i + 1;
            int right = length - 1;
            while (left < length && right < length && left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target) {
                    return sum;
                } else if (sum < target) {
                    int differnce = target - sum;
                    if (differnce < var1) {
                        var2 = sum;
                        var1 = differnce;
                    }
                    left++;
                } else {
                    int differnce = sum - target;
                    if (differnce < var1) {
                        var2 = sum;
                        var1 = differnce;
                    }
                    right--;
                }
            }
        }
        return var2;
    }
}
