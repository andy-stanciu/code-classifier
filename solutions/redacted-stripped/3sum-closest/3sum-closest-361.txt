class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int min = (nums[0] + nums[1] + nums[2]) - (target);
        for (int i = 0; i < nums.length; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (Math.abs(sum - target) < Math.abs(min)) {
                    min = sum - target;
                }
                if (sum < target) {
                    j++;
                } else if (sum > target) {
                    k--;
                } else {
                    return sum;
                }
            }
        }
        return min + target;
    }
}
