class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int numsLength = nums.length;
        int var1 = 0;
        int maxDiff = Integer.MAX_VALUE;
        for (int counter = 0; counter < numsLength; counter++) {
            int val = nums[counter];
            int start = counter + 1;
            int end = numsLength - 1;
            int diff = 0;
            while (start < end) {
                int sum = val + nums[start] + nums[end];
                if (sum == target)
                    return sum;
                diff = Math.abs(target - sum);
                if (diff < maxDiff) {
                    maxDiff = diff;
                    var1 = sum;
                }
                if (sum > target)
                    end--;
                else
                    start++;
            }
        }
        return var1;
    }
}
