class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        if (nums.length == 3)
            return Arrays.stream(nums).sum();
        ;
        int minSum = 0;
        int minDiff = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 1; i++) {
            int sum = 0;
            int j = i + 1;
            int k = nums.length - 1;
            int diff = 0;
            if (i > 0 && nums[i] == nums[i - 1])
                continue;
            while (j < k) {
                sum = nums[i] + nums[j] + nums[k];
                diff = Math.abs(sum - target);
                if (sum == target)
                    return target;
                if (sum > target)
                    k--;
                else if (sum < target)
                    j++;
                if (diff < minDiff) {
                    minSum = sum;
                    minDiff = diff;
                }
            }
        }
        return minSum;
    }
}
