class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int min = Integer.MAX_VALUE;
        int result = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            int sIndex = i + 1;
            int eIndex = nums.length - 1;
            while (sIndex < eIndex) {
                int tSum = nums[i] + nums[sIndex] + nums[eIndex];
                if (tSum == target) {
                    return target;
                } else if (tSum < target) {
                    sIndex++;
                } else {
                    eIndex--;
                }
                int d = Math.abs(tSum - target);
                if (d < min) {
                    min = d;
                    result = tSum;
                }
            }
        }
        return result;
    }
}
