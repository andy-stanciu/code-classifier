class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int cls = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target)
                    return sum;
                if (Math.abs(sum - target) < Math.abs(cls - target)) {
                    cls = sum;
                }
                if (sum < target) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return cls;
    }
}
