class Solution {

    public int method1(int[] arr, int target) {
        int n = arr.length;
        int var1 = 100000;
        Arrays.sort(arr);
        for (int k = 0; k <= n - 3; k++) {
            int i = k + 1;
            int j = n - 1;
            while (i < j) {
                int sum = arr[k] + arr[i] + arr[j];
                if (Math.abs(target - sum) < Math.abs(target - var1)) {
                    var1 = sum;
                }
                if (sum < target) {
                    i++;
                } else {
                    j--;
                }
            }
        }
        return var1;
    }
}
