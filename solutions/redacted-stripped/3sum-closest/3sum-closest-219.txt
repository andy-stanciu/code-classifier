class Solution {

    public int method1(int[] a, int target) {
        Arrays.sort(a);
        int len = a.length;
        int diff = Integer.MAX_VALUE;
        int ans = 0;
        for (int i = 0; i < len - 2; i++) {
            if (i > 0 && a[i] == a[i - 1])
                continue;
            int j = i + 1;
            int k = len - 1;
            while (j < k) {
                int sum = a[i] + a[j] + a[k];
                if (Math.abs(target - sum) < diff) {
                    ans = sum;
                    diff = Math.abs(target - sum);
                }
                if (sum < target) {
                    j++;
                } else if (sum > target) {
                    k--;
                } else {
                    return sum;
                }
            }
        }
        return ans;
    }
}
