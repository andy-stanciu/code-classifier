class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int cloestSum = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int currentSum = nums[i] + nums[left] + nums[right];
                int diffCurrent = Math.abs(currentSum - target);
                int diffCloest = Math.abs(cloestSum - target);
                if (diffCurrent < diffCloest) {
                    cloestSum = currentSum;
                }
                if (currentSum < target) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return cloestSum;
    }
}
