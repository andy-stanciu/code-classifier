class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int closeDiff = Integer.MAX_VALUE;
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 1; i++) {
            int j = nums.length - 1;
            int k = i + 1;
            while (k < j) {
                int sum = nums[i] + nums[j] + nums[k];
                if (sum == target) {
                    return sum;
                }
                int diff = Math.max(target, sum) - Math.min(target, sum);
                if (closeDiff > diff) {
                    closeDiff = diff;
                    var1 = sum;
                }
                if (sum > target) {
                    j--;
                } else if (sum < target) {
                    k++;
                }
            }
        }
        return var1;
    }
}
