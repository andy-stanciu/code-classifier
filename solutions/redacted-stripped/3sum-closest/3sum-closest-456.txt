class Solution {

    public int method1(int[] a, int target) {
        Arrays.sort(a);
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < a.length - 2; i++) {
            if (i > 0 && a[i] == a[i - 1])
                continue;
            int l = i + 1;
            int h = a.length - 1;
            while (l < h) {
                int sum = a[i] + a[l] + a[h];
                if (sum == target)
                    return sum;
                else if (Math.abs(sum - target) < Math.abs(res - target)) {
                    res = sum;
                }
                if (sum < target) {
                    l++;
                } else {
                    h--;
                }
            }
        }
        return res;
    }
}
