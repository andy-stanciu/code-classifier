class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        Arrays.sort(nums);
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            int low = i + 1;
            int high = n - 1;
            if (i > 0 && nums[i] == nums[i - 1])
                continue;
            while (low < high) {
                int curr_sum = nums[i] + nums[low] + nums[high];
                if (Math.abs(target - curr_sum) < Math.abs(target - var1)) {
                    var1 = curr_sum;
                }
                if (curr_sum == target) {
                    return curr_sum;
                } else if (curr_sum < target) {
                    low++;
                } else {
                    high--;
                }
            }
        }
        return var1;
    }
}
