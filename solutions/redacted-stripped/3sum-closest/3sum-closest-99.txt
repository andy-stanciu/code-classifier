class Solution {

    public int method1(int[] nums, int target) {
        int left;
        int right;
        Arrays.sort(nums);
        int mindiff = Integer.MAX_VALUE;
        int closeSum = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            left = i + 1;
            right = nums.length - 1;
            int sum = 0;
            while (left < right) {
                sum = nums[i] + nums[left] + nums[right];
                if (target == sum) {
                    return target;
                } else {
                    int diff = Math.abs(sum - target);
                    if (diff < mindiff) {
                        mindiff = diff;
                        closeSum = sum;
                    }
                    if (sum > target) {
                        right--;
                    } else {
                        left++;
                    }
                }
            }
        }
        return closeSum;
    }
}
