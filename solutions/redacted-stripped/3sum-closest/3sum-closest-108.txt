class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int var1 = nums[0] + nums[1] + nums[2];
        int smallestDiffFromTarget = Math.abs(target - var1);
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (k > j) {
                int thisSum = nums[i] + nums[j] + nums[k];
                int thisDiffFromTarget = Math.abs(target - thisSum);
                if (thisDiffFromTarget < smallestDiffFromTarget) {
                    smallestDiffFromTarget = thisDiffFromTarget;
                    var1 = thisSum;
                }
                if (thisSum < target) {
                    j++;
                } else {
                    k--;
                }
            }
        }
        return var1;
    }
}
