class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int bestSum = 0, bestDiff = Integer.MAX_VALUE;
        for (int i = 0; i < n; ++i) {
            int j = i + 1, k = n - 1;
            while (j < k) {
                int currSum = nums[i] + nums[j] + nums[k];
                int currDiff = Math.abs(target - currSum);
                if (currDiff < bestDiff) {
                    bestDiff = currDiff;
                    bestSum = currSum;
                }
                if (currSum < target) {
                    j++;
                } else
                    k--;
            }
        }
        return bestSum;
    }
}
