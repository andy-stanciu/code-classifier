class Solution {

    public int method1(int[] nums, int target) {
        if (nums.length < 3)
            return 0;
        return method2(nums, target);
    }

    public static int method2(int[] nums, int target) {
        Arrays.sort(nums);
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target)
                    return sum;
                if (Math.abs(sum - target) < Math.abs(var1 - target))
                    var1 = sum;
                if (sum < target)
                    left++;
                else
                    right--;
            }
        }
        return var1;
    }
}
