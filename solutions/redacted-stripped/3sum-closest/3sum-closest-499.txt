class Solution {

    public int method1(int[] nums, int target) {
        int result = nums[0] + nums[1] + nums[nums.length - 1];
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            int a = i + 1;
            int b = nums.length - 1;
            while (a < b) {
                int sum = 0;
                sum = nums[i] + nums[a] + nums[b];
                if (sum < target)
                    a++;
                else
                    b--;
                if (Math.abs(sum - target) < Math.abs(result - target))
                    result = sum;
            }
        }
        return result;
    }
}
