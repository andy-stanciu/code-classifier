class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < n - 2; i++) {
            int start = i + 1;
            int end = n - 1;
            while (start < end) {
                int sum = nums[i] + nums[start] + nums[end];
                if (sum == target)
                    return sum;
                if (Math.abs(sum - target) < Math.abs(var1 - target))
                    var1 = sum;
                if (sum < target) {
                    start++;
                } else {
                    end--;
                }
            }
        }
        return var1;
    }
}
