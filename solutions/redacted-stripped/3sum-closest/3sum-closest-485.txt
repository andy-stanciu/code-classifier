class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int min = Integer.MAX_VALUE;
        int ans = 0;
        List<Integer> list1 = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            int sum = 0, diff = 0;
            while (j < k) {
                sum = nums[i] + nums[j] + nums[k];
                diff = Math.abs(target - sum);
                if (diff <= min) {
                    min = diff;
                    ans = sum;
                }
                if (sum < target) {
                    j++;
                } else {
                    k--;
                }
            }
        }
        return ans;
    }
}
