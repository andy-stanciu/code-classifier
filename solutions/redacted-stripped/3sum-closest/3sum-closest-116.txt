class Solution {

    public int method1(int[] nums, int target) {
        int result = MAX_VALUE;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            int l = i + 1;
            int r = nums.length - 1;
            while (l < r) {
                int triplet = nums[i] + nums[l] + nums[r];
                if (triplet == target) {
                    return triplet;
                }
                if (Math.abs(triplet - target) < Math.abs(result - target)) {
                    result = triplet;
                }
                if (triplet > target) {
                    r--;
                } else {
                    l++;
                }
            }
        }
        return result;
    }
}
