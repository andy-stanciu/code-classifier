class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int var1 = nums[0] + nums[1] + nums[nums.length - 1];
        for (int i = 0; i < nums.length - 2; i++) {
            int low = i + 1;
            int high = nums.length - 1;
            while (low < high) {
                int currSum = nums[i] + nums[low] + nums[high];
                if (currSum > target)
                    high--;
                else
                    low++;
                if (Math.abs(currSum - target) < Math.abs(var1 - target))
                    var1 = currSum;
            }
        }
        return var1;
    }
}
