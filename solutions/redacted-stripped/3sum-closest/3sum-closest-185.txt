class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 2; i++) {
            int var1 = nums[i] + method2(nums, i + 1, nums.length - 1, target - nums[i]);
            if (Math.abs(ans - target) > Math.abs(var1 - target)) {
                ans = var1;
            }
        }
        return ans;
    }

    private int method2(int[] nums, int startIndex, int endIndex, int target) {
        int ans = nums[startIndex] + nums[endIndex];
        while (startIndex < endIndex) {
            int sum = nums[startIndex] + nums[endIndex];
            if (Math.abs(sum - target) < Math.abs(ans - target))
                ans = sum;
            if (sum == target) {
                return target;
            }
            if (target > sum) {
                startIndex++;
            } else {
                endIndex--;
            }
        }
        return ans;
    }
}
