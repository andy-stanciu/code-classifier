class Solution {

    public boolean method1(String s) {
        int n = s.length();
        String str = new String("");
        str = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int len = str.length();
        int i = 0;
        int j = len - 1;
        if (n == 1)
            return true;
        while (i < j) {
            if (str.charAt(i) != str.charAt(j)) {
                return false;
            } else if (str.charAt(i) == str.charAt(j)) {
                i++;
                j--;
            }
        }
        return true;
    }
}
