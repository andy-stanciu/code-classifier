class Solution {

    public boolean method1(String s) {
        String reversStr = "";
        s = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int size = s.length();
        for (int i = size - 1; i >= 0; i--) {
            reversStr += s.charAt(i);
        }
        return s.equals(reversStr);
    }
}
