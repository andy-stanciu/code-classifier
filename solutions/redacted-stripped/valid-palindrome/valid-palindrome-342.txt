class Solution {

    public boolean method1(String s) {
        String sFormatted = s.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        int l = 0;
        int r = sFormatted.length() - 1;
        while (r >= l) {
            if (sFormatted.charAt(r) != sFormatted.charAt(l)) {
                return false;
            }
            r--;
            l++;
        }
        return true;
    }
}
