class Solution {

    public boolean method1(String s) {
        String s1 = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        return method1(s1, 0, s1.length() - 1);
    }

    private boolean method1(String s, int l, int r) {
        if (l >= r) {
            return true;
        }
        if (s.charAt(l) != s.charAt(r)) {
            return false;
        }
        return method1(s, l + 1, r - 1);
    }
}
