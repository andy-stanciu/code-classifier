class Solution {

    public boolean method1(String str) {
        str = str.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        if (str.length() == 0 || str.length() == 1) {
            return true;
        }
        for (int i = 0; i < str.length(); i++) {
            System.out.print(str.charAt(i) + " ");
        }
        int i = 0;
        int n = str.length();
        while (i <= n / 2) {
            if (str.charAt(i) != str.charAt(n - i - 1)) {
                return false;
            }
            i++;
        }
        return true;
    }
}
