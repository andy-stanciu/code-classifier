class Solution {

    static boolean method1(int i, String s) {
        int n = s.length();
        if (i >= n / 2) {
            return true;
        }
        if (s.charAt(i) != s.charAt(n - i - 1)) {
            return false;
        }
        return method1(i + 1, s);
    }

    public boolean method2(String s) {
        s = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        return method1(0, s);
    }
}
