class Solution {

    public boolean method1(String s) {
        int stx = 0;
        int etx = s.length() - 1;
        while (stx < etx) {
            while (stx < etx && !Character.isLetterOrDigit(s.charAt(stx))) stx++;
            while (stx < etx && !Character.isLetterOrDigit(s.charAt(etx))) etx--;
            if (Character.toLowerCase(s.charAt(stx)) != Character.toLowerCase(s.charAt(etx)))
                return false;
            stx++;
            etx--;
        }
        return true;
    }
}
