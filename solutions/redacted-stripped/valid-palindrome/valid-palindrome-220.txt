class Solution {

    public boolean method1(String s) {
        String[] ch = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < ch.length; i++) {
            for (int j = 0; j < ch[i].length(); j++) {
                if (Character.isLetter(ch[i].charAt(j)) || Character.isDigit(ch[i].charAt(j))) {
                    sb.append(String.valueOf(Character.toLowerCase(ch[i].charAt(j))));
                }
            }
        }
        String prev = sb.toString();
        return prev.equals(sb.reverse().toString());
    }
}
