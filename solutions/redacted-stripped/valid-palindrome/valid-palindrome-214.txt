class Solution {

    public boolean method1(String s) {
        if (s.isEmpty()) {
            return true;
        }
        String newString = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int n = newString.length();
        for (int i = 0; i < n / 2; i++) {
            if (newString.charAt(i) != newString.charAt(n - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
