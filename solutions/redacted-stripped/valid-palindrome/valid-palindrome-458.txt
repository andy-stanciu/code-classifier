class Solution {

    public boolean method1(String s) {
        int l = 0;
        int r = s.length() - 1;
        while (l <= r) {
            char getAtl = s.charAt(l);
            char getAtr = s.charAt(r);
            if (!Character.isLetterOrDigit(getAtl)) {
                l++;
            } else if (!Character.isLetterOrDigit(getAtr)) {
                r--;
            } else {
                if (Character.toLowerCase(getAtl) != Character.toLowerCase(getAtr)) {
                    return false;
                }
                l++;
                r--;
            }
        }
        return true;
    }
}
