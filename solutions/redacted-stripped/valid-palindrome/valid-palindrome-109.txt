class Solution {

    public boolean method1(String s) {
        StringBuffer res = new StringBuffer();
        for (char digit : s.toCharArray()) {
            if (cleanString(digit))
                res.append(digit);
        }
        String lowerCase = res.toString().toLowerCase();
        int start = 0, end = lowerCase.length() - 1;
        while (start < end) {
            if (lowerCase.charAt(start) != lowerCase.charAt(end))
                return false;
            start++;
            end--;
        }
        return true;
    }

    private boolean cleanString(char digit) {
        if ((48 <= digit && digit <= 57) || (65 <= digit && digit <= 90) || (97 <= digit && digit <= 122))
            return true;
        return false;
    }
}
