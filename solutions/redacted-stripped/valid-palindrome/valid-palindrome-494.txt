class Solution {

    public boolean method1(String s) {
        int j = s.length();
        StringBuilder s1 = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) >= 65 && s.charAt(i) <= 90) {
                char c = (char) ((int) s.charAt(i) + 32);
                s1.append(c);
            } else if (s.charAt(i) >= 97 && s.charAt(i) <= 122 || s.charAt(i) >= 48 && s.charAt(i) <= 57) {
                s1.append(s.charAt(i));
            }
        }
        int p = 0;
        int k = s1.length() - 1;
        while (p <= k) {
            if (s1.charAt(p) != s1.charAt(k)) {
                return false;
            } else {
                p++;
                k--;
            }
        }
        return true;
    }
}
