class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        s = s.toLowerCase();
        int i = 0;
        int j = s.length() - 1;
        int count = 0;
        while (i < j) {
            if (s.charAt(i) == s.charAt(j)) {
                count++;
            }
            i++;
            j--;
        }
        if (s.length() % 2 == 0) {
            if (count == ((s.length()) / 2)) {
                return true;
            } else {
                return false;
            }
        } else {
            if (count == ((s.length() - 1) / 2)) {
                return true;
            } else {
                return false;
            }
        }
    }
}
