class Solution {

    public static boolean method1(String s) {
        s = s.trim().toLowerCase();
        int i = 0, j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i) == s.charAt(j)) {
                i++;
                j--;
            } else {
                boolean left = isCharOrNumber(s.charAt(i));
                boolean right = isCharOrNumber(s.charAt(j));
                if (left && right)
                    return false;
                if (!left) {
                    i++;
                }
                if (!right) {
                    j--;
                }
            }
        }
        return true;
    }

    private static boolean isCharOrNumber(char s) {
        return (s >= 48 && s <= 57) || (s >= 65 && s <= 90) || (s >= 97 && s <= 122);
    }
}
