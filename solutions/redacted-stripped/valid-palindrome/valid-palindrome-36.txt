class Solution {

    private boolean method1(char c) {
        if (Character.isAlphabetic(c) || Character.isDigit(c))
            return true;
        return false;
    }

    public boolean method2(String s) {
        s = s.toLowerCase();
        for (int i = 0, j = s.length() - 1; i < j; ) {
            if (!method1(s.charAt(i)))
                i++;
            else if (!method1(s.charAt(j)))
                j--;
            else if (s.charAt(i) != s.charAt(j))
                return false;
            else {
                i++;
                j--;
            }
        }
        return true;
    }
}
