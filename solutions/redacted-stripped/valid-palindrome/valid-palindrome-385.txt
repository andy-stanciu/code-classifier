class Solution {

    public boolean method1(String s) {
        int i = 0, j = s.length() - 1;
        int a = 0, b = 0;
        while (i < j) {
            a = equivalent_val(s.charAt(i));
            if (a == -1) {
                i++;
                continue;
            }
            b = equivalent_val(s.charAt(j));
            if (b == -1) {
                j--;
                continue;
            }
            if (a != b)
                return false;
            i++;
            j--;
        }
        return true;
    }

    public static int equivalent_val(char c) {
        int val = c + 0;
        if (val > 47 && val < 58)
            return val;
        if (val > 64 && val < 91)
            return val + 32;
        if (val > 96 && val < 123)
            return val;
        return -1;
    }
}
