class Solution {

    public boolean method1(String s) {
        int left = 0;
        int right = s.length() - 1;
        while (left < right) {
            char onLeft = s.charAt(left);
            char onRight = s.charAt(right);
            boolean onLeftALetterOrNumber = Character.isLetterOrDigit(onLeft);
            boolean onRightALetterOrNumber = Character.isLetterOrDigit(onRight);
            if (!onLeftALetterOrNumber || !onRightALetterOrNumber) {
                if (!onLeftALetterOrNumber)
                    left++;
                if (!onRightALetterOrNumber)
                    right--;
            } else if (Character.toLowerCase(onLeft) == Character.toLowerCase(onRight)) {
                left++;
                right--;
            } else {
                return false;
            }
        }
        return true;
    }
}
