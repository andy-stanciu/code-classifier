class Solution {

    public boolean method1(String s) {
        int fIdx = 0;
        int lIdx = s.length() - 1;
        while (fIdx < lIdx) {
            if (!Character.isLetterOrDigit(s.charAt(fIdx))) {
                fIdx++;
            } else if (!Character.isLetterOrDigit(s.charAt(lIdx))) {
                lIdx--;
            } else {
                if (Character.toLowerCase(s.charAt(fIdx)) != Character.toLowerCase(s.charAt(lIdx))) {
                    return false;
                }
                fIdx++;
                lIdx--;
            }
        }
        return true;
    }
}
