class Solution {

    public boolean method1(String s) {
        if (s.isEmpty()) {
            return true;
        }
        String text = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int last = text.length() - 1;
        int start = 0;
        while (start < last) {
            if (text.charAt(start++) != text.charAt(last--)) {
                return false;
            }
        }
        return true;
    }
}
