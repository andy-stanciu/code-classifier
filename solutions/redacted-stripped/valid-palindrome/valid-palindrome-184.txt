class Solution {

    public StringBuffer nume(char[] ch) {
        StringBuffer te = new StringBuffer();
        for (int i = 0; i < ch.length; i++) {
            if (Character.isLetterOrDigit(ch[i])) {
                te = te.append(ch[i]);
            }
        }
        return te;
    }

    public boolean method1(String s) {
        StringBuffer s6 = new StringBuffer();
        StringBuffer s9;
        char[] ch;
        String[] arr = s.split(" ");
        for (int i = 0; i < arr.length; i++) {
            ch = arr[i].toCharArray();
            s9 = nume(ch);
            s6 = s6.append(s9);
        }
        StringBuffer org = new StringBuffer(s6);
        s6.reverse();
        if (org.toString().equalsIgnoreCase(s6.toString())) {
            return true;
        }
        return false;
    }
}
