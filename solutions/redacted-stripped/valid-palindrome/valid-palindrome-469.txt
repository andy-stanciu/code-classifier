public class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        char[] spl = new char[s.length()];
        int k = 0;
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (Character.isLetterOrDigit(ch)) {
                spl[k++] = ch;
            }
        }
        char[] temp = new char[k];
        System.arraycopy(spl, 0, temp, 0, k);
        return isP(temp, 0);
    }

    public boolean isP(char[] s, int k) {
        if (k >= s.length / 2) {
            return true;
        }
        if (s[k] != s[s.length - k - 1]) {
            return false;
        }
        return isP(s, k + 1);
    }
}
