class Solution {

    public boolean method1(String s) {
        return method1(s, 0, s.length() - 1);
    }

    public boolean method1(String s, int l, int r) {
        if (l >= r)
            return true;
        char sl = Character.toLowerCase(s.charAt(l));
        char sr = Character.toLowerCase(s.charAt(r));
        if (!Character.isLetterOrDigit(sl)) {
            return method1(s, l + 1, r);
        } else if (!Character.isLetterOrDigit(sr)) {
            return method1(s, l, r - 1);
        }
        return sl == sr && method1(s, l + 1, r - 1);
    }
}
