class Solution {

    public boolean method1(String s) {
        String s1 = s.replaceAll("[^A-Za-z0-9]", "");
        String s3 = "";
        for (int i = s1.length() - 1; i >= 0; i--) {
            s3 += s1.charAt(i);
        }
        if (s3.equalsIgnoreCase(s1)) {
            return true;
        } else {
            return false;
        }
    }
}
