class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        s = s.toLowerCase();
        System.out.println(s);
        String[] arr = s.split("");
        int l = 0, r = arr.length - 1;
        while (l <= r) {
            if (!Objects.equals(arr[l++], arr[r--])) {
                return false;
            }
        }
        return true;
    }
}
