class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        int L = 0;
        int R = s.length() - 1;
        while (!(L > R)) {
            int asciiCodeL = (int) s.charAt(L);
            int asciiCodeR = (int) s.charAt(R);
            if (!(asciiCodeL >= 48 && asciiCodeL <= 57) && !(asciiCodeL >= 97 && asciiCodeL <= 122)) {
                L++;
                continue;
            }
            if (!(asciiCodeR >= 48 && asciiCodeR <= 57) && !(asciiCodeR >= 97 && asciiCodeR <= 122)) {
                R--;
                continue;
            }
            if (s.charAt(L) == s.charAt(R)) {
                R--;
                L++;
            } else {
                return false;
            }
        }
        return true;
    }
}
