class Solution {

    public boolean method1(String s) {
        String regex = "[^a-zA-Z0-9]";
        String str = s.replaceAll(regex, "");
        str = str.toLowerCase();
        int start = 0, end = str.length() - 1;
        return isPal(start, end, str);
    }

    private boolean isPal(int start, int end, String str) {
        if (start > end)
            return true;
        if (str.charAt(start) == str.charAt(end))
            return true && isPal(start + 1, end - 1, str);
        else {
            System.out.println(str.charAt(start) == str.charAt(end));
            System.out.println(str.charAt(start) + ":" + str.charAt(end));
            System.out.println(start + ":" + end);
            return false;
        }
    }
}
