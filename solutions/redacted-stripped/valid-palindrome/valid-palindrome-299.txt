class Solution {

    public boolean method1(String s) {
        if (s.length() == 1) {
            return true;
        }
        String pattern = "[^a-zA-Z0-9]";
        String rsStr = s.replaceAll(pattern, "").toLowerCase();
        int left = 0;
        int right = rsStr.length() - 1;
        while (left < right) {
            if (rsStr.charAt(left) != rsStr.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
