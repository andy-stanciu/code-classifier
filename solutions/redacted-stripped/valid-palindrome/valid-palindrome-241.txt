class Solution {

    public boolean method1(String s) {
        String A = s.replaceAll("[^a-zA-Z0-9]", "");
        String B = A.toLowerCase();
        int left = 0;
        int right = B.length() - 1;
        while (left < right) {
            if (B.charAt(left) != B.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
