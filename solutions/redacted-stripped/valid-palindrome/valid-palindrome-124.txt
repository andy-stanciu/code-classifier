class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        return method2(s, 0, s.length() - 1);
    }

    public boolean method2(String s, int start, int end) {
        if (start >= end)
            return true;
        if (s.charAt(start) != s.charAt(end))
            return false;
        return method2(s, start + 1, end - 1);
    }
}
