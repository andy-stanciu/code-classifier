class Solution {

    public boolean method1(String s) {
        int endIndx = s.length() - 1;
        for (int i = 0; i < endIndx; i++) {
            if (!Character.isLetterOrDigit(s.charAt(i)))
                continue;
            while (!Character.isLetterOrDigit(s.charAt(endIndx))) endIndx--;
            if (!(Character.toLowerCase(s.charAt(i)) == Character.toLowerCase(s.charAt(endIndx))))
                return false;
            else
                endIndx--;
        }
        return true;
    }
}
