class Solution {

    public boolean method1(String s) {
        int start = 0;
        int end = s.length() - 1;
        while (start < end) {
            Character cStart = s.charAt(start);
            Character cEnd = s.charAt(end);
            if (!Character.isLetterOrDigit(cStart)) {
                start++;
                continue;
            }
            if (!Character.isLetterOrDigit(cEnd)) {
                end--;
                continue;
            }
            if (Character.toLowerCase(cStart) != Character.toLowerCase(cEnd)) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
