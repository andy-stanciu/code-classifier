class Solution {

    int i = 0;

    public boolean method1(String s) {
        s = s.replaceAll("[^A-Za-z0-9]", "").toLowerCase();
        return method2(s);
    }

    private boolean method2(String s) {
        if (i >= s.length() / 2) {
            return true;
        }
        if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
            return false;
        }
        i++;
        return method2(s);
    }
}
