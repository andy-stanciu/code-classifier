class Solution {

    public boolean method1(String s) {
        String Res = "";
        for (int i = 0; i < s.length(); i++) {
            if ((s.charAt(i) >= 48 && s.charAt(i) <= 57) || ((s.charAt(i) > 64 && s.charAt(i) <= 90) || (s.charAt(i) > 96 && s.charAt(i) <= 122))) {
                Res = Res.concat(String.valueOf(Character.toLowerCase(s.charAt(i))));
            }
        }
        System.out.println(Res);
        String Rev = "";
        for (int i = Res.length() - 1; i >= 0; i--) {
            Rev = Rev.concat(String.valueOf(Res.charAt(i)));
        }
        System.out.println(Rev);
        return Rev.equals(Res);
    }
}
