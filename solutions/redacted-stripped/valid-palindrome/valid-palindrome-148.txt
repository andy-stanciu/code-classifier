class Solution {

    public boolean method1(String s) {
        if (s.isEmpty()) {
            return true;
        }
        int st = 0;
        int last = s.length() - 1;
        while (st <= last) {
            char currSt = s.charAt(st);
            char currEnd = s.charAt(last);
            if (!Character.isLetterOrDigit(currSt)) {
                st++;
            } else if (!Character.isLetterOrDigit(currEnd)) {
                last--;
            } else {
                if (Character.toLowerCase(currSt) != Character.toLowerCase(currEnd)) {
                    return false;
                } else {
                    st++;
                    last--;
                }
            }
        }
        return true;
    }
}
