class Solution {

    public boolean method1(String s) {
        int start = 0;
        String abc = s.replaceAll("[^a-zA-Z0-9]", "");
        String str = abc.toLowerCase();
        int end = abc.length() - 1;
        while (start <= end) {
            if (str.charAt(start) != str.charAt(end)) {
                return false;
            } else {
                start++;
                end--;
            }
        }
        return true;
    }
}
