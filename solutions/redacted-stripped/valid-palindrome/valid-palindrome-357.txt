class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        int n = s.length();
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (Character.isDigit(s.charAt(i)) || Character.isLetter(s.charAt(i))) {
                sb.append(s.charAt(i));
            }
        }
        int m = sb.length();
        for (int i = 0; i < m / 2; i++) {
            if (sb.charAt(i) != sb.charAt(m - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
