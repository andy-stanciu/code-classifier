class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        int n = s.length();
        int lb = 0;
        int ub = n - 1;
        while (ub >= lb) {
            if (!Character.isLetterOrDigit(s.charAt(ub)))
                ub--;
            else if (!Character.isLetterOrDigit(s.charAt(lb)))
                lb++;
            else {
                if (s.charAt(lb) != s.charAt(ub))
                    return false;
                lb++;
                ub--;
            }
        }
        return true;
    }
}
