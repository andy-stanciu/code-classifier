class Solution {

    public boolean method1(String s) {
        String txt1 = s.toLowerCase();
        String txt2 = txt1.replaceAll("[^a-zA-Z0-9]", "");
        int left = 0;
        int right = txt2.length() - 1;
        while (left < right) {
            if (txt2.charAt(left) == txt2.charAt(right)) {
                left = left + 1;
                right = right - 1;
            } else {
                return false;
            }
        }
        return true;
    }
}
