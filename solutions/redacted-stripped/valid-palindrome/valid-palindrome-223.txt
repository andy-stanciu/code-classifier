class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        return check(0, s);
    }

    boolean check(int i, String s) {
        if (i >= s.length() / 2) {
            return true;
        } else if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
            return false;
        } else {
            return check(i + 1, s);
        }
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        String s = "madam";
        boolean result = solution.method1(s);
        System.out.println(result);
    }
}
