class Solution {

    public boolean method1(String s) {
        String s1 = s.toLowerCase();
        int i = 0;
        int j = s1.length() - 1;
        while (i < j) {
            char c1 = s1.charAt(i);
            char c2 = s1.charAt(j);
            int asc1 = c1;
            int asc2 = c2;
            if (((asc1 >= 97 && asc1 <= 122) || (asc1 >= 48 && asc1 <= 57)) && ((asc2 >= 97 && asc2 <= 122) || (asc2 >= 48 && asc2 <= 57))) {
                if (c1 == c2) {
                    i++;
                    j--;
                } else {
                    return false;
                }
            } else if ((asc1 > 57 && asc1 < 97) || asc1 > 122 || asc1 < 48) {
                i++;
            } else if ((asc2 > 57 && asc2 < 97) || asc2 > 122 || asc2 < 48) {
                j--;
            }
        }
        return true;
    }
}
