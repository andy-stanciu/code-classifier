class Solution {

    public boolean method1(String s) {
        StringBuilder sb = new StringBuilder();
        for (char c : s.toCharArray()) {
            if (c >= 65 && c <= 90)
                c = (char) (c + 32);
            if (c >= 97 && c <= 122)
                sb.append(c);
            if (c >= 48 && c <= 57)
                sb.append(c);
        }
        s = sb.toString();
        int low = 0, high = s.length() - 1;
        while (low < high) {
            if (s.charAt(low) == s.charAt(high)) {
                low++;
                high--;
            } else
                return false;
        }
        return true;
    }
}
