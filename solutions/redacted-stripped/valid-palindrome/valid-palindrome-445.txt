class Solution {

    public boolean method1(String s) {
        int st = 0;
        int l = s.length() - 1;
        while (st < l) {
            char currest = s.charAt(st);
            char currel = s.charAt(l);
            if (!Character.isLetterOrDigit(currest)) {
                st++;
            } else if (!Character.isLetterOrDigit(currel)) {
                l--;
            } else {
                if (Character.toLowerCase(currest) != Character.toLowerCase(currel)) {
                    return false;
                }
                st++;
                l--;
            }
        }
        return true;
    }
}
