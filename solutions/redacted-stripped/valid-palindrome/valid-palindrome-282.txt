class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        String lettersOnly = s.replaceAll("[^a-zA-Z0-9]", "");
        String lettersOnlyre;
        StringBuffer sb = new StringBuffer(lettersOnly);
        lettersOnlyre = sb.reverse().toString();
        if (lettersOnly.equals(lettersOnlyre)) {
            return true;
        } else {
            return false;
        }
    }
}
