class Solution {

    public boolean method1(String s) {
        StringBuilder originalStringCleaned = new StringBuilder();
        StringBuilder var1 = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (Character.isLetterOrDigit(s.charAt(i))) {
                originalStringCleaned.append(s.charAt(i));
            }
        }
        for (int i = originalStringCleaned.length() - 1; i >= 0; i--) {
            var1.append(originalStringCleaned.charAt(i));
        }
        return originalStringCleaned.toString().equalsIgnoreCase(var1.toString());
    }
}
