class Solution {

    public boolean method1(String s) {
        int left = 0;
        int right = s.length() - 1;
        s = s.toLowerCase();
        while (left <= right) {
            if (checkIfInRange(s.charAt(left))) {
                left++;
                continue;
            }
            if (checkIfInRange(s.charAt(right))) {
                right--;
                continue;
            }
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }

    public boolean checkIfInRange(char c) {
        return !(c >= 65 && c <= 90) && !(c >= 97 && c <= 122) && !(c >= 48 && c <= 57);
    }
}
