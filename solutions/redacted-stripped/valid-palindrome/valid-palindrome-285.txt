class Solution {

    public boolean method1(String s) {
        return helper(0, s.length() - 1, s.toLowerCase());
    }

    public boolean helper(int i, int j, String s) {
        if (i > j) {
            return true;
        }
        if (!Character.isLetterOrDigit(s.charAt(i))) {
            return helper(i + 1, j, s);
        }
        if (!Character.isLetterOrDigit(s.charAt(j))) {
            return helper(i, j - 1, s);
        }
        if (s.charAt(i) != s.charAt(j)) {
            return false;
        }
        return helper(i + 1, j - 1, s);
    }
}
