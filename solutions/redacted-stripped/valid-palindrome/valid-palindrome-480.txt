class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        char[] c = s.toCharArray();
        int start = 0;
        int end = c.length - 1;
        while (start < end) {
            char temp = c[end];
            c[end] = c[start];
            c[start] = temp;
            start++;
            end--;
        }
        String reverse = String.valueOf(c);
        return s.equalsIgnoreCase(reverse);
    }
}
