class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase().replace(" ", "").replaceAll("[^a-zA-Z0-9]", "");
        System.out.println(s);
        char[] arr = s.toCharArray();
        int orig_string = 0;
        int reverse_string = arr.length - 1;
        while (orig_string < reverse_string) {
            if (arr[orig_string] != arr[reverse_string]) {
                return false;
            }
            orig_string++;
            reverse_string--;
        }
        return true;
    }
}
