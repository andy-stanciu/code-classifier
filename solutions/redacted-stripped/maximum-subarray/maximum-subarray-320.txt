class Solution {

    public int method1(int[] nums) {
        int currSum = 0;
        int maxSum = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (currSum < 0) {
                currSum = 0;
            }
            currSum = currSum + nums[i];
            maxSum = Math.max(currSum, maxSum);
        }
        return maxSum;
    }
}
