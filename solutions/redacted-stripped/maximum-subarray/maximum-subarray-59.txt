class Solution {

    public int method1(int[] nums) {
        int max = Integer.MIN_VALUE;
        int subArr = 0;
        for (int num : nums) {
            subArr += num;
            if (subArr > max) {
                max = subArr;
            }
            if (subArr < 0) {
                subArr = 0;
            }
        }
        return max;
    }
}
