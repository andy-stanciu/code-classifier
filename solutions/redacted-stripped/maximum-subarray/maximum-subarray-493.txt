class Solution {

    public int method1(int[] nums) {
        int curr_sum = 0;
        int max_sum = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (curr_sum < 0) {
                curr_sum = 0;
            }
            curr_sum = curr_sum + nums[i];
            max_sum = Math.max(curr_sum, max_sum);
        }
        return max_sum;
    }
}
