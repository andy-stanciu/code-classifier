class Solution {

    public int method1(int[] nums) {
        int maxSub = nums[0];
        int cur = 0;
        for (int i : nums) {
            if (cur < 0) {
                cur = 0;
            }
            cur += i;
            maxSub = Math.max(cur, maxSub);
        }
        return maxSub;
    }
}
