class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int res = Integer.MIN_VALUE, currSum = 0;
        for (int i : nums) {
            currSum += i;
            res = Math.max(res, currSum);
            if (currSum < 0)
                currSum = 0;
        }
        return res;
    }
}
