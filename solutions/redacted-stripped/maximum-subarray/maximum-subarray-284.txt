class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int currSum = nums[0];
        int longestSum = nums[0];
        if (n == 1)
            return nums[0];
        for (int i = 1; i < n; i++) {
            if (currSum >= 0) {
                currSum += nums[i];
            } else {
                currSum = nums[i];
            }
            if (currSum > longestSum) {
                longestSum = currSum;
            }
        }
        return longestSum;
    }
}
