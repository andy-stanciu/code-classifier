class Solution {

    public int method1(int[] nums) {
        int max = nums[0];
        int currnetsum = nums[0];
        if (nums.length == 1) {
            return nums[0];
        }
        for (int i = 1; i < nums.length; i++) {
            if (currnetsum + nums[i] > nums[i]) {
                currnetsum += nums[i];
            } else {
                currnetsum = nums[i];
            }
            max = Math.max(max, currnetsum);
        }
        return max;
    }
}
