class Solution {

    public int method1(int[] nums) {
        int result = nums[0];
        int tempSum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            tempSum = tempSum > 0 ? tempSum + nums[i] : nums[i];
            result = Math.max(result, tempSum);
        }
        return result;
    }
}
