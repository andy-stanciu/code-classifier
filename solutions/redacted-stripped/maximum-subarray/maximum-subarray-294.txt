class Solution {

    public int method1(int[] nums) {
        int max = nums[0];
        int curr = 0;
        for (int i = 0; i < nums.length; i++) {
            curr += nums[i];
            if (curr < 0) {
                if (curr > max) {
                    max = curr;
                }
                curr = 0;
            } else if (curr > max) {
                max = curr;
            }
        }
        return max;
    }
}
