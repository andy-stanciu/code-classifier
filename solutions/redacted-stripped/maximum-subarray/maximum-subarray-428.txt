class Solution {

    public int method1(int[] nums) {
        int res = nums[0];
        int maxEnding = nums[0];
        for (int i = 1; i < nums.length; i++) {
            maxEnding = Math.max(nums[i], maxEnding + nums[i]);
            res = Math.max(res, maxEnding);
        }
        return res;
    }
}
