class Solution {

    public int method1(int[] nums) {
        int currMax = nums[0], overallMax = nums[0];
        for (int i = 1; i < nums.length; i++) {
            currMax += nums[i];
            if (currMax < nums[i])
                currMax = nums[i];
            if (overallMax < currMax)
                overallMax = currMax;
        }
        return overallMax;
    }
}
