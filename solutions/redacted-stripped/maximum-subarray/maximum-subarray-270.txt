class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = Integer.MIN_VALUE;
        int maxsum = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (sum < 0) {
                sum = nums[i];
            } else {
                sum += nums[i];
            }
            if (sum > maxsum) {
                maxsum = sum;
            }
        }
        return maxsum;
    }
}
