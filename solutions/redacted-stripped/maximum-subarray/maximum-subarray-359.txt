class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int Max = nums[0];
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (sum < 0) {
                sum = 0;
                Max = Math.max(Max, nums[i]);
            } else {
                Max = Math.max(Max, sum);
            }
        }
        return Max;
    }
}
