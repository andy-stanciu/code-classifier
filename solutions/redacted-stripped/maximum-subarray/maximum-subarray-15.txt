class Solution {

    public int method1(int[] nums) {
        int maxT = Integer.MIN_VALUE;
        int sumT = 0;
        for (int i = 0; i < nums.length; i++) {
            sumT += nums[i];
            if (maxT < sumT)
                maxT = sumT;
            if (sumT < 0)
                sumT = 0;
        }
        return maxT;
    }
}
