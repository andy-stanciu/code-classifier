class Solution {

    public int method1(int[] nums) {
        int last = nums[0];
        int highest = nums[0];
        if (nums.length == 1)
            return nums[0];
        for (int i = 1; i < nums.length; i++) {
            last = Math.max(nums[i], nums[i] + last);
            ;
            if (last > highest) {
                highest = last;
            }
        }
        return highest;
    }
}
