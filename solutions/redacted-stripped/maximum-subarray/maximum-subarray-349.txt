class Solution {

    public int method1(int[] nums) {
        int maxSum = nums[0], sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            maxSum = sum > maxSum ? sum : maxSum;
            if (sum < 0)
                sum = 0;
        }
        return maxSum;
    }
}
