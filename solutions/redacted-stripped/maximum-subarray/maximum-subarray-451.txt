class Solution {

    public int method1(int[] nums) {
        int total = 0;
        int ans = 0;
        int res = nums[0];
        for (int n : nums) {
            if (total < 0) {
                total = 0;
            }
            total += n;
            if (total > res) {
                res = total;
            }
            ans = Math.max(res, total);
        }
        return ans;
    }
}
