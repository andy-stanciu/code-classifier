class Solution {

    public int method1(int[] nums) {
        int maxsum = Integer.MIN_VALUE;
        int sum = 0;
        for (int x : nums) {
            if (sum < 0) {
                sum = 0;
            }
            sum += x;
            if (sum > maxsum) {
                maxsum = sum;
            }
        }
        return maxsum;
    }
}
