class Solution {

    public int method1(int[] nums) {
        int maxi = Integer.MIN_VALUE, n = nums.length, sum = 0;
        if (n == 1) {
            if (nums[0] < 0) {
                return -1;
            } else {
                return nums[0];
            }
        }
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            if (sum > maxi) {
                maxi = sum;
            }
            if (sum < 0) {
                sum = 0;
            }
        }
        return maxi;
    }
}
