class Solution {

    public int method1(int[] nums) {
        int max = nums[0];
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            max = Math.max(max, var1);
            if (var1 < 0) {
                var1 = 0;
            }
        }
        return max;
    }
}
