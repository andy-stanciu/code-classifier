class Solution {

    public int method1(int[] nums) {
        int sum = nums[0];
        int prevsum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (sum + nums[i] > nums[i]) {
                sum = sum + nums[i];
            } else {
                sum = nums[i];
            }
            if (sum > prevsum) {
                prevsum = sum;
            }
        }
        return prevsum;
    }
}
