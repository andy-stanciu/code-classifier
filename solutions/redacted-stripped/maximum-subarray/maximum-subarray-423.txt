class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int maxint = nums[0];
        for (int i = 0; i < nums.length; i++) {
            sum = Math.max(nums[i], sum + nums[i]);
            if (nums.length == 1) {
                return sum;
            }
            maxint = Math.max(sum, maxint);
        }
        return maxint;
    }
}
