class Solution {

    public int method1(int[] nums) {
        int maxSum = Integer.MIN_VALUE;
        int currSum = 0;
        String ele = "";
        for (int i = 0; i < nums.length; i++) {
            currSum += nums[i];
            maxSum = Math.max(currSum, maxSum);
            if (currSum < 0)
                currSum = 0;
        }
        return maxSum;
    }
}
