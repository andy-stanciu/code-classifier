class Solution {

    public int method1(int[] arr) {
        int num = 0;
        int max = -2147483647;
        for (int i = 0; i < arr.length; i++) {
            num += arr[i];
            if (num > max)
                max = num;
            if (num < 0)
                num = 0;
        }
        return max;
    }
}
