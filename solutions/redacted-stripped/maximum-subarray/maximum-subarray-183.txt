class Solution {

    public int method1(int[] nums) {
        int current_sum = 0;
        int var1 = nums[0];
        for (int i = 0; i < nums.length; i++) {
            current_sum += nums[i];
            var1 = Math.max(var1, current_sum);
            if (current_sum <= 0) {
                current_sum = 0;
            }
        }
        return var1;
    }
}
