class Solution {

    public int method1(int[] arr) {
        int max_sum_so_far = arr[0];
        int res = arr[0];
        for (int i = 1; i < arr.length; i++) {
            max_sum_so_far = Math.max(max_sum_so_far + arr[i], arr[i]);
            res = Math.max(max_sum_so_far, res);
        }
        return res;
    }
}
