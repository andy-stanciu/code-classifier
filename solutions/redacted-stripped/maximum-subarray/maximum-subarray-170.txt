class Solution {

    public int method1(int[] nums) {
        int maxSum = nums[0];
        int currentSum = maxSum;
        for (int k = 1; k < nums.length; k++) {
            currentSum = Math.max(currentSum + nums[k], nums[k]);
            maxSum = Math.max(maxSum, currentSum);
        }
        return maxSum;
    }
}
