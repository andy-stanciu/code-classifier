class Solution {

    public int method1(int[] nums) {
        int ans = nums[0];
        int[] dp = new int[nums.length];
        dp[0] = nums[0];
        for (int i = 1; i < dp.length; i++) {
            dp[i] = Math.max(nums[i], nums[i] + dp[i - 1]);
            if (ans < dp[i]) {
                ans = dp[i];
            }
        }
        return ans;
    }
}
