class Solution {

    public int method1(int[] nums) {
        int max_s = Integer.MIN_VALUE;
        int cs = 0;
        for (int i = 0; i < nums.length; i++) {
            cs += nums[i];
            if (cs > max_s) {
                max_s = cs;
            }
            if (cs < 0) {
            }
        }
        return max_s;
    }
}
