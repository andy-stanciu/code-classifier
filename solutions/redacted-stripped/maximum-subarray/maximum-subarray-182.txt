class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int maxSum = Integer.MIN_VALUE;
        int maxCurrent = 0;
        for (int num : nums) {
            int currentSum = num + maxCurrent;
            maxCurrent = Math.max(num, currentSum);
            maxSum = Math.max(maxSum, maxCurrent);
        }
        return maxSum;
    }
}
