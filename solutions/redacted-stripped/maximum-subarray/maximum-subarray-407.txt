class Solution {

    public int method1(int[] nums) {
        int res = Integer.MIN_VALUE;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
            res = Math.max(res, sum);
            if (sum <= 0)
                sum = 0;
        }
        return res;
    }
}
