class Solution {

    public int method1(int[] nums) {
        int largestSum = nums[0];
        int sum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (sum < 0) {
                sum = nums[i];
            } else {
                sum += nums[i];
            }
            largestSum = Math.max(largestSum, sum);
        }
        return largestSum;
    }
}
