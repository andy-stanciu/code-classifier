class Solution {

    public int method1(int[] nums) {
        int m = 0;
        int ans = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (m < 0) {
                m = 0;
            }
            m += nums[i];
            ans = Math.max(m, ans);
        }
        return ans;
    }
}
