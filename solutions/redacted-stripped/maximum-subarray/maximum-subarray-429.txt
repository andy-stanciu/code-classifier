class Solution {

    public int method1(int[] nums) {
        int currSum = 0;
        int maxSum = Integer.MIN_VALUE;
        for (int x : nums) {
            currSum = currSum + x;
            if (maxSum < currSum) {
                maxSum = currSum;
            }
            if (currSum < 0) {
                currSum = 0;
            }
        }
        return maxSum;
    }
}
