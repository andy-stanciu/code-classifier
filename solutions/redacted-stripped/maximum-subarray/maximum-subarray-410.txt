class Solution {

    public int method1(int[] nums) {
        int currsum = 0;
        int maxSum = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            currsum += nums[i];
            maxSum = Math.max(currsum, maxSum);
            if (currsum < 0) {
                currsum = 0;
            }
        }
        return maxSum;
    }
}
