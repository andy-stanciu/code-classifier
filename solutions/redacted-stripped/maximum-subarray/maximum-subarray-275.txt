class Solution {

    public int method1(int[] nums) {
        if (nums.length < 1)
            return -1;
        int max = nums[0];
        int cursum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            cursum = Math.max(nums[i], cursum + nums[i]);
            max = Math.max(cursum, max);
        }
        return max;
    }
}
