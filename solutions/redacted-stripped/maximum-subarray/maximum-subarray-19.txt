class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int var1 = Integer.MIN_VALUE, var2 = 0;
        for (int i = 0; i < n; i++) {
            var2 += nums[i];
            var1 = Math.max(var1, var2);
            var2 = Math.max(var2, 0);
        }
        return var1;
    }
}
