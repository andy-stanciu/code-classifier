class Solution {

    public int method1(int[] nums) {
        int currSum = 0;
        int subArr = nums[0];
        for (int i : nums) {
            if (currSum < 0) {
                currSum = 0;
            }
            currSum += i;
            subArr = Math.max(subArr, currSum);
        }
        return subArr;
    }
}
