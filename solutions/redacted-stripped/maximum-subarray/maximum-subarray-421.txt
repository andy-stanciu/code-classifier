class Solution {

    public int method1(int[] nums) {
        int maxSub = nums[0];
        int currSum = 0;
        for (int num : nums) {
            if (currSum < 0)
                currSum = 0;
            currSum += num;
            maxSub = Math.max(maxSub, currSum);
        }
        return maxSub;
    }
}
