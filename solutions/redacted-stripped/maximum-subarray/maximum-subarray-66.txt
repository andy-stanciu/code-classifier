class Solution {

    public int method1(int[] nums) {
        int current_sum = nums[0];
        int max_sum = current_sum;
        for (int i = 1; i < nums.length; ++i) {
            currentSum = Math.max(nums[i], currentSum + nums[i]);
            max_sum = Math.max(maxSum, currentSum);
        }
        return maxSum;
    }
}
