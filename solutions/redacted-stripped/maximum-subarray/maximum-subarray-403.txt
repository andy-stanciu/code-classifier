class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int max_sum = nums[0];
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            sum = sum + nums[i];
            if (sum > max_sum) {
                max_sum = sum;
            }
            if (sum < 0) {
                sum = 0;
            }
        }
        return max_sum;
    }
}
