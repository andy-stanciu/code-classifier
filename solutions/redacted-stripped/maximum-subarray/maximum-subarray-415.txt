class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = 0;
        int currSum = 0;
        int maxSum = Integer.MIN_VALUE;
        int len = nums.length;
        while (right < len) {
            currSum += nums[right];
            maxSum = Math.max(maxSum, currSum);
            if (currSum < 0) {
                left = right + 1;
                currSum = 0;
            }
            right++;
        }
        return maxSum;
    }
}
