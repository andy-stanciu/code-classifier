class Solution {

    public int method1(int[] arr) {
        int sum = 0, n = arr.length, maxSum = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            sum += arr[i];
            maxSum = maxSum < sum ? sum : maxSum;
            ;
            if (sum < 0)
                sum = 0;
        }
        return maxSum;
    }
}
