class Solution {

    public int count = 0;

    public int method1(int[] nums, int param1) {
        helper(nums, param1, 0, 0);
        return count;
    }

    public void helper(int[] nums, int param1, int index, int value) {
        if (index == nums.length) {
            if (value == param1) {
                count++;
            }
            return;
        }
        helper(nums, param1, index + 1, value + nums[index]);
        helper(nums, param1, index + 1, value - nums[index]);
    }
}
