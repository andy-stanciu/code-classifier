class Solution {

    public int method1(int[] nums, int param1) {
        return param1(nums, param1, 0);
    }

    private int param1(int[] nums, int param1, int index) {
        if (index == nums.length)
            return param1 == 0 ? 1 : 0;
        return param1(nums, param1 + nums[index], index + 1) + param1(nums, param1 - nums[index], index + 1);
    }
}
