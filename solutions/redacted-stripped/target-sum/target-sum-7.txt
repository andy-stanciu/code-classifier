class Solution {

    public int method1(int[] nums, int param1) {
        return solve(nums.length, 0, param1, nums);
    }

    public int solve(int n, int ind, int param1, int[] nums) {
        if (ind == n) {
            if (param1 == 0)
                return 1;
            else
                return 0;
        }
        return solve(n, ind + 1, param1 - nums[ind], nums) + solve(n, ind + 1, param1 + nums[ind], nums);
    }
}
