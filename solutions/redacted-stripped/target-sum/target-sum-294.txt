class Solution {

    public int method1(int[] nums, int param1) {
        return backtrack(nums, 0, param1, 0);
    }

    public static int backtrack(int[] nums, int param2, int param1, int i) {
        if (i == nums.length) {
            if (param2 == param1)
                return 1;
            return 0;
        }
        return backtrack(nums, param2 - nums[i], param1, i + 1) + backtrack(nums, param2 + nums[i], param1, i + 1);
    }
}
