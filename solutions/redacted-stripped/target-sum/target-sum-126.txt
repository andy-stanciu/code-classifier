class Solution {

    public int method1(int[] nums, int param1) {
        return param1(nums, 0, 0, param1);
    }

    public static int param1(int[] nums, int pos, int param2, int param1) {
        if (pos == nums.length) {
            return param2 == param1 ? 1 : 0;
        }
        int ans = param1(nums, pos + 1, param2 + nums[pos], param1) + param1(nums, pos + 1, param2 + (-nums[pos]), param1);
        return ans;
    }
}
