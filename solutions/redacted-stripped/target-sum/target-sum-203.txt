class Solution {

    public static int counter(int[] arr, int param1, int index, int param2, HashMap<String, Integer> hm) {
        if (index < 0) {
            return param2 == param1 ? 1 : 0;
        }
        String key = index + "," + param2;
        if (hm.containsKey(key))
            return hm.get(key);
        int plus = counter(arr, param1, index - 1, param2 + arr[index], hm);
        int minus = counter(arr, param1, index - 1, param2 - arr[index], hm);
        hm.put(key, plus + minus);
        return plus + minus;
    }

    public int method1(int[] arr, int param1) {
        HashMap<String, Integer> hm = new HashMap<>();
        return counter(arr, param1, arr.length - 1, 0, hm);
    }
}
