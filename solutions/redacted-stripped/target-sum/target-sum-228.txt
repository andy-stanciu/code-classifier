class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        int var2 = var1 + param1;
        if (var2 % 2 != 0 || Math.abs(param1) > var1) {
            return 0;
        }
        var2 /= 2;
        int[] dp = new int[var2 + 1];
        dp[0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = var2; j >= nums[i - 1]; j--) {
                dp[j] += dp[j - nums[i - 1]];
            }
        }
        return dp[var2];
    }
}
