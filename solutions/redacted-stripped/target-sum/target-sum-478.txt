class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, 0, -param1, new HashMap<String, Integer>());
    }

    private int helper(int[] nums, int index, int param1, Map<String, Integer> cache) {
        if (index == nums.length) {
            return param1 == 0 ? 1 : 0;
        }
        String key = new StringBuilder().append(index).append(",").append(param1).toString();
        if (!cache.containsKey(key)) {
            cache.put(key, helper(nums, index + 1, param1 - nums[index], cache) + helper(nums, index + 1, param1 + nums[index], cache));
        }
        return cache.get(key);
    }
}
