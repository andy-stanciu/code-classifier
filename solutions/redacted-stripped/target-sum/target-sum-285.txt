class Solution {

    int var1;

    public int method1(int[] nums, int param1) {
        var1 = param1;
        return no_of_ways(0, nums, 0);
    }

    int no_of_ways(int index, int[] nos, int op) {
        if (index == nos.length && op == var1) {
            return 1;
        } else if (nos.length == 0 || index >= nos.length) {
            return 0;
        } else {
            int p = no_of_ways(index + 1, nos, op + nos[index]);
            int n = no_of_ways(index + 1, nos, op - nos[index]);
            return p + n;
        }
    }
}
