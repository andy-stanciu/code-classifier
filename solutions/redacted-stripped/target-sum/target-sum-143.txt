class Solution {

    public int method1(int[] nums, int param1) {
        return rec(nums, 0, 0, param1);
    }

    public int rec(int[] nums, int idx, int param2, int param1) {
        if (idx == nums.length) {
            if (param2 == param1)
                return 1;
            return 0;
        }
        return rec(nums, idx + 1, param2 + nums[idx], param1) + rec(nums, idx + 1, param2 - nums[idx], param1);
    }
}
