class Solution {

    int count = 0;

    public int method1(int[] nums, int param1) {
        dfs(0, 0, nums, param1);
        return count;
    }

    public void dfs(int total, int index, int[] nums, int param1) {
        if (index == nums.length) {
            if (total == param1) {
                count++;
            }
            return;
        }
        dfs(total + nums[index], index + 1, nums, param1);
        dfs(total - nums[index], index + 1, nums, param1);
    }
}
