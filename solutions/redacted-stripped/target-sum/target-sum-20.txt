class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        if (n == 0)
            return 0;
        int var1 = 0;
        for (int i : nums) var1 += i;
        int var2 = (var1 - param1) / 2;
        if (param1 > var1 || (var1 - param1) % 2 != 0)
            return 0;
        int[][] dp = new int[n + 1][var2 + 1];
        dp[0][0] = 1;
        for (int i = 1; i <= n; i++) {
            if (nums[i - 1] == 0) {
                dp[i][0] = dp[i - 1][0] * 2;
            } else {
                dp[i][0] = 1;
            }
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= var2; j++) {
                if (j >= nums[i - 1]) {
                    dp[i][j] = dp[i - 1][j] + dp[i - 1][j - nums[i - 1]];
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][var2];
    }
}
