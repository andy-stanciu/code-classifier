class Solution {

    private int ways = 0;

    public int method1(int[] nums, int param1) {
        backtracking(nums, param1, 0);
        return ways;
    }

    private void backtracking(int[] nums, int param1, int index) {
        if (index == nums.length && param1 == 0) {
            ways++;
            return;
        }
        if (index == nums.length)
            return;
        backtracking(nums, param1 + nums[index], index + 1);
        backtracking(nums, param1 - nums[index], index + 1);
        return;
    }
}
