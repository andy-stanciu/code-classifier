class Solution {

    public int method1(int[] nums, int param1) {
        int ans = helper(nums, param1, 0);
        return ans;
    }

    private int helper(int[] nums, int param1, int i) {
        int ans = 0;
        if (i >= nums.length) {
            return param1 == 0 ? 1 : 0;
        }
        ans = ans + helper(nums, param1 - nums[i] * (-1), i + 1);
        ans = ans + helper(nums, param1 - nums[i], i + 1);
        return ans;
    }
}
