class Solution {

    int ans = 0;

    public int method1(int[] nums, int param1) {
        getWays(nums, param1, 0, 0);
        return ans;
    }

    public void getWays(int[] nums, int param1, int param2, int index) {
        if (index == nums.length) {
            if (param2 == param1)
                ans++;
            return;
        }
        getWays(nums, param1, param2 + nums[index], index + 1);
        getWays(nums, param1, param2 - nums[index], index + 1);
    }
}
