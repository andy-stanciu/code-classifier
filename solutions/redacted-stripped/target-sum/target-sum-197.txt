class Solution {

    public int method1(int[] nums, int param1) {
        for (int i = 0; i < nums.length; i++) {
            nums[i] = Math.abs(nums[i]);
        }
        param1 = Math.abs(param1);
        int n = nums.length;
        int var1 = 0;
        for (Integer i : nums) {
            var1 += i;
        }
        if (param1 > var1)
            return 0;
        if ((param1 + var1) % 2 != 0)
            return 0;
        int var2 = (param1 + var1) / 2;
        int[][] dp = new int[n + 1][var2 + 1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= var2; j++) {
                if (i == 0)
                    dp[i][j] = 0;
                if (j == 0)
                    dp[i][j] = 1;
            }
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= var2; j++) {
                if (nums[i - 1] <= j) {
                    dp[i][j] = (dp[i - 1][j - nums[i - 1]] + dp[i - 1][j]);
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][var2];
    }
}
