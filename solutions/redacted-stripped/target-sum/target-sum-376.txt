class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        if ((param1 > var1) || (var1 - param1) % 2 == 1) {
            return 0;
        }
        int var2 = (var1 - param1) / 2;
        int n = nums.length;
        int[][] dp = new int[n + 1][var2 + 1];
        dp[0][0] = 1;
        for (int k = 1; k <= n; k++) {
            for (int j = 0; j <= var2; j++) {
                if (j >= nums[k - 1]) {
                    dp[k][j] = dp[k - 1][j] + dp[k - 1][j - nums[k - 1]];
                } else {
                    dp[k][j] = dp[k - 1][j];
                }
            }
        }
        return dp[n][var2];
    }
}
