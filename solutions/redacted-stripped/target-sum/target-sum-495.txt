class Solution {

    public int method1(int[] nums, int param1) {
        return compute(0, 0, nums, param1);
    }

    public int compute(int i, int param2, int[] nums, int param1) {
        if (i == nums.length) {
            if (param2 == param1)
                return 1;
            return 0;
        }
        int minus = compute(i + 1, param2 - nums[i], nums, param1);
        int plus = compute(i + 1, param2 + nums[i], nums, param1);
        return minus + plus;
    }
}
