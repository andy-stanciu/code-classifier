class Solution {

    int count = 0;

    public int method1(int[] nums, int param1) {
        helper(nums, param1, 0, 0);
        return count;
    }

    private void helper(int[] nums, int param1, int param2, int index) {
        if (index == nums.length) {
            if (param2 == param1) {
                count++;
            }
            return;
        }
        helper(nums, param1, param2 + nums[index], index + 1);
        helper(nums, param1, param2 - nums[index], index + 1);
    }
}
