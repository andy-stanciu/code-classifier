class Solution {

    public int method1(int[] nums, int param1) {
        return rec(0, nums, param1);
    }

    public int rec(int idx, int[] nums, int param1) {
        if (idx == nums.length) {
            if (param1 == 0) {
                return 1;
            } else {
                return 0;
            }
        }
        int put = rec(idx + 1, nums, param1 - (0 - nums[idx]));
        int notPut = rec(idx + 1, nums, param1 - nums[idx]);
        return put + notPut;
    }
}
