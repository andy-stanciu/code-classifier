class Solution {

    public int method1(int[] nums, int param1) {
        return method1(0, nums, param1);
    }

    private int method1(int i, int[] nums, int param1) {
        if (i == nums.length) {
            if (param1 == 0)
                return 1;
            else
                return 0;
        }
        return method1(i + 1, nums, param1 - nums[i]) + method1(i + 1, nums, param1 + nums[i]);
    }
}
