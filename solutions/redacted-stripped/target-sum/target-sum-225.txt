class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, 0, 0, param1);
    }

    private int helper(int[] nums, int index, int param2, int param1) {
        if (index == nums.length) {
            if (param2 == param1) {
                return 1;
            } else {
                return 0;
            }
        }
        int left = helper(nums, index + 1, param2 + nums[index], param1);
        int right = helper(nums, index + 1, param2 - nums[index], param1);
        return left + right;
    }
}
