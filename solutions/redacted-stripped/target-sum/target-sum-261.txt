class Solution {

    public int method1(int[] nums, int param1) {
        return solve(0, nums, param1, 0);
    }

    private int solve(int curr, int[] nums, int param1, int param2) {
        if (curr > nums.length - 1) {
            if (param2 == param1) {
                return 1;
            }
            return 0;
        }
        return solve(curr + 1, nums, param1, param2 + nums[curr]) + solve(curr + 1, nums, param1, param2 - nums[curr]);
    }
}
