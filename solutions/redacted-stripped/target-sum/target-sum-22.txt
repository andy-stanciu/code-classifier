class Solution {

    int aux(int[] nums, int i, int param1, int S) {
        if (i == nums.length) {
            if (S == param1)
                return 1;
            return 0;
        }
        int pos = aux(nums, i + 1, param1 + nums[i], S);
        int neg = aux(nums, i + 1, param1 - nums[i], S);
        return pos + neg;
    }

    public int method1(int[] nums, int S) {
        return aux(nums, 0, 0, S);
    }
}
