class Solution {

    public int method1(int[] nums, int param1) {
        if (nums.length == 0)
            return 0;
        return helper(nums, param1, nums[0], 1) + helper(nums, param1, -nums[0], 1);
    }

    public int helper(int[] nums, int param1, int current, int index) {
        if (index == nums.length) {
            if (param1 == current)
                return 1;
            else
                return 0;
        }
        return helper(nums, param1, current + nums[index], index + 1) + helper(nums, param1, current - nums[index], index + 1);
    }
}
