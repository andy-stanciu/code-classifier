class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length, var1 = 0;
        for (int a : nums) {
            var1 += a;
        }
        int[][] dp = new int[n + 1][4 * var1 + 1];
        for (int i = n; i >= 0; i--) {
            for (int j = var1; j >= -var1; j--) {
                if (i == n && param1 == j)
                    dp[i][j + (2 * var1)] = 1;
                else if (i != n)
                    dp[i][j + (2 * var1)] = dp[i + 1][j + nums[i] + (2 * var1)] + dp[i + 1][j - nums[i] + (2 * var1)];
            }
        }
        return dp[0][2 * var1];
    }
}
