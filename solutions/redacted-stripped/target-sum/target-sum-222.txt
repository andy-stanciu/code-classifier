class Solution {

    public int solve(int ind, int[] nums, int param1, int param2) {
        if (ind < 0) {
            if (param2 == param1) {
                return 1;
            }
            return 0;
        }
        int pos = solve(ind - 1, nums, param1, param2 + nums[ind]);
        int neg = solve(ind - 1, nums, param1, param2 - nums[ind]);
        return pos + neg;
    }

    public int method1(int[] nums, int param1) {
        return solve(nums.length - 1, nums, param1, 0);
    }
}
