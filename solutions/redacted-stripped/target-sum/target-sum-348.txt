class Solution {

    public int method1(int[] nums, int param1) {
        return method1(nums, param1, 0);
    }

    public int method1(int[] nums, int param1, int start) {
        if (nums.length == start)
            return (param1 == 0) ? 1 : 0;
        return method1(nums, param1 + nums[start], start + 1) + method1(nums, param1 - nums[start], start + 1);
    }
}
