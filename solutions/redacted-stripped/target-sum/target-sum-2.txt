class Solution {

    int ans = 0;

    public int method1(int[] nums, int param1) {
        solve(0, 0, param1, nums);
        return ans;
    }

    public void solve(int index, int result, int param1, int[] nums) {
        if (index == nums.length) {
            if (result == param1) {
                ans++;
            }
            return;
        }
        solve(index + 1, result + nums[index], param1, nums);
        solve(index + 1, result - nums[index], param1, nums);
    }
}
