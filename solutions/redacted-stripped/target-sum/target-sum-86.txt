class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, param1, 0, 0);
    }

    public int helper(int[] nums, int param1, int param2, int ind) {
        if (ind == nums.length) {
            if (param2 == param1) {
                return 1;
            }
            return 0;
        }
        int add = helper(nums, param1, param2 + nums[ind], ind + 1);
        int subtract = helper(nums, param1, param2 - nums[ind], ind + 1);
        return add + subtract;
    }
}
