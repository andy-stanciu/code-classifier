class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, param1, nums.length);
    }

    public static int helper(int[] nums, int param1, int i) {
        if (i == 0)
            return param1 == 0 ? 1 : 0;
        int p = helper(nums, param1 - nums[i - 1], i - 1);
        int m = helper(nums, param1 + nums[i - 1], i - 1);
        return p + m;
    }
}
