class Solution {

    public int method1(int[] nums, int param1) {
        return countWays(param1, nums, nums.length - 1);
    }

    int countWays(int t, int[] nums, int i) {
        if (i < 0)
            return t == 0 ? 1 : 0;
        int P = countWays(t - nums[i], nums, i - 1);
        int N = countWays(t + nums[i], nums, i - 1);
        return N + P;
    }
}
