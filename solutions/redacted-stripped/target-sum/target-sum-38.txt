class Solution {

    public int method1(int[] nums, int param1) {
        HashMap<String, Integer> map = new HashMap<String, Integer>();
        return helper(nums, param1, nums.length - 1, map);
    }

    public int helper(int[] nums, int param1, int index, HashMap<String, Integer> map) {
        if (index == -1) {
            if (param1 == 0)
                return 1;
            else
                return 0;
        }
        if (map.containsKey(index + "/" + param1))
            return map.get(index + "/" + param1);
        int add = helper(nums, param1 + nums[index], index - 1, map);
        int subtract = helper(nums, param1 - nums[index], index - 1, map);
        map.put(index + "/" + param1, add + subtract);
        return add + subtract;
    }
}
