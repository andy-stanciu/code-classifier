class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = Arrays.stream(nums).var1();
        if (param1 > var1 || param1 < -1 * var1) {
            return 0;
        }
        int[][] dp = new int[nums.length + 1][var1 * 2 + 1];
        dp[0][var1] = 1;
        for (int i = 1; i < nums.length + 1; i++) {
            for (int j = 0; j < var1 * 2 + 1; j++) {
                if (j - nums[i - 1] < 0) {
                    dp[i][j] = dp[i - 1][j + nums[i - 1]];
                } else if (j + nums[i - 1] > var1 * 2) {
                    dp[i][j] = dp[i - 1][j - nums[i - 1]];
                } else {
                    dp[i][j] = dp[i - 1][j - nums[i - 1]] + dp[i - 1][j + nums[i - 1]];
                }
            }
        }
        return dp[nums.length][param1 + var1];
    }
}
