class Solution {

    public static void count(int[] nums, int param1, int[] c, int index, int form) {
        if (index == nums.length) {
            if (form == param1) {
                c[0]++;
            }
            return;
        }
        count(nums, param1, c, index + 1, form + nums[index]);
        count(nums, param1, c, index + 1, form - nums[index]);
    }

    public int method1(int[] nums, int param1) {
        if (nums.length == 1 && param1 == nums[0]) {
            return 1;
        }
        int form = 0;
        int[] c = { 0 };
        count(nums, param1, c, 0, form);
        return c[0];
    }
}
