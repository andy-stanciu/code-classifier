class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0, 0);
    }

    private int method2(int[] nums, int param1, int k, int param2) {
        if (k == nums.length) {
            return param2 == param1 ? 1 : 0;
        }
        int countPositive = method2(nums, param1, k + 1, param2 + nums[k]);
        int countNegative = method2(nums, param1, k + 1, param2 - nums[k]);
        return countPositive + countNegative;
    }
}
