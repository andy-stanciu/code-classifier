class Solution {

    public int method1(int[] nums, int param1) {
        int[] arr = new int[1];
        findExpressions(0, 0, nums, param1, arr);
        return arr[0];
    }

    private void findExpressions(int idx, int param2, int[] nums, int param1, int[] arr) {
        if (idx == nums.length) {
            if (param2 == param1)
                arr[0]++;
            return;
        }
        param2 -= nums[idx];
        findExpressions(idx + 1, param2, nums, param1, arr);
        param2 += nums[idx] * 2;
        findExpressions(idx + 1, param2, nums, param1, arr);
    }
}
