class Solution {

    int count;

    public int method1(int[] nums, int param1) {
        helper(nums, param1, 0, 0);
        return count;
    }

    private void helper(int[] nums, int param1, int index, int currentValue) {
        if (index == nums.length) {
            if (currentValue == param1) {
                count++;
            }
            return;
        }
        helper(nums, param1, index + 1, currentValue + nums[index]);
        helper(nums, param1, index + 1, currentValue - nums[index]);
    }
}
