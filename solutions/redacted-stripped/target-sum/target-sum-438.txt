class Solution {

    public int method1(int[] arr, int diff) {
        int var1 = 0;
        for (int i = 0; i < arr.length; i++) {
            var1 += arr[i];
        }
        int var2 = (diff + var1) / 2;
        if (var1 + diff < 0 || (var1 + diff) % 2 == 1) {
            return 0;
        }
        return method2(arr, var2);
    }

    public int method2(int[] arr, int var2) {
        int n = arr.length;
        int[][] t = new int[n + 1][var2 + 1];
        for (int i = 0; i < n + 1; i++) {
            for (int j = 0; j < var2 + 1; j++) {
                if (i == 0) {
                    t[i][j] = 0;
                }
                if (j == 0) {
                    t[i][j] = 1;
                }
            }
        }
        for (int i = 1; i < n + 1; i++) {
            for (int j = 0; j < var2 + 1; j++) {
                if (arr[i - 1] <= j) {
                    t[i][j] = t[i - 1][j] + t[i - 1][j - arr[i - 1]];
                } else {
                    t[i][j] = t[i - 1][j];
                }
            }
        }
        return t[n][var2];
    }
}
