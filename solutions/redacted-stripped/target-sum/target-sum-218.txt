class Solution {

    public int method1(int[] nums, int param1) {
        return fun(nums, 0, param1, 0);
    }

    public int fun(int[] nums, int index, int param1, int param2) {
        if (index == nums.length) {
            if (param2 == param1) {
                return 1;
            }
            return 0;
        }
        return fun(nums, index + 1, param1, param2 + nums[index]) + fun(nums, index + 1, param1, param2 - nums[index]);
    }
}
