class Solution {

    public int method1(int[] nums, int param1) {
        int s = 0;
        for (int x : nums) s += x;
        if (s < param1 || (s - param1) % 2 != 0)
            return 0;
        int n = (s - param1) / 2;
        int[] c = new int[n + 1];
        c[0] = 1;
        for (int x : nums) for (int i = n; i >= x; i--) c[i] += c[i - x];
        return c[n];
    }
}
