class Solution {

    public int method1(int[] nums, int param1) {
        return dfs(nums, param1, 0, 0);
    }

    private int dfs(int[] nums, int param1, int index, int param2) {
        if (index == nums.length) {
            return param2 == param1 ? 1 : 0;
        }
        int add = dfs(nums, param1, index + 1, param2 + nums[index]);
        int subtract = dfs(nums, param1, index + 1, param2 - nums[index]);
        return add + subtract;
    }
}
