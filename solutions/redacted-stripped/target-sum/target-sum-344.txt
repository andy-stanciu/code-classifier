class Solution {

    public int method1(int[] nums, int param1) {
        return DP(0, param1, nums, 0);
    }

    public int DP(int index, int param1, int[] nums, int myAmount) {
        if (index == nums.length) {
            if (myAmount == param1)
                return 1;
            return 0;
        }
        int positive = DP(index + 1, param1, nums, myAmount + nums[index]);
        int negative = DP(index + 1, param1, nums, myAmount - nums[index]);
        return positive + negative;
    }
}
