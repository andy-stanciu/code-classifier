class Solution {

    Integer n = null;

    public int method1(int[] nums, int param1) {
        n = nums.length;
        return findWays(nums, param1, 0, 0, 0);
    }

    private int findWays(int[] nums, int param1, int param2, int index, int count) {
        if (n == index && param2 == param1) {
            count++;
            return count;
        }
        if (n == index) {
            return 0;
        }
        return findWays(nums, param1, param2 + nums[index], index + 1, count) + findWays(nums, param1, param2 - nums[index], index + 1, count);
    }
}
