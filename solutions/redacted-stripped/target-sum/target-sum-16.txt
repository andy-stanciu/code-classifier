class Solution {

    int cnt = 0;

    public int method1(int[] nums, int param1) {
        DFS(nums, 0, 0, param1);
        return cnt;
    }

    private void DFS(int[] nums, int index, int param2, int param1) {
        if (index == nums.length) {
            if (param2 == param1)
                cnt++;
            return;
        }
        DFS(nums, index + 1, param2 + nums[index], param1);
        DFS(nums, index + 1, param2 - nums[index], param1);
    }
}
