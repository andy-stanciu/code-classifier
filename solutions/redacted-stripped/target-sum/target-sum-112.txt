class Solution {

    public int method1(int[] nums, int param1) {
        return solve(nums, nums.length - 1, param1, 0);
    }

    public int solve(int[] nums, int i, int param1, int totalNums) {
        if (i < 0 && param1 == totalNums) {
            return 1;
        }
        if (i < 0 && param1 != totalNums) {
            return 0;
        }
        return solve(nums, i - 1, param1, totalNums - nums[i]) + solve(nums, i - 1, param1, totalNums + nums[i]);
    }
}
