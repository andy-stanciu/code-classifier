public class Solution {

    int res;

    public int method1(int[] nums, int param1) {
        dfs(0, param1, 0, nums);
        return res;
    }

    public void dfs(int start, int param1, int curr, int[] nums) {
        if (curr == param1 && start == nums.length) {
            res++;
            return;
        }
        if (start == nums.length)
            return;
        dfs(start + 1, param1, curr + nums[start], nums);
        dfs(start + 1, param1, curr - nums[start], nums);
    }
}
