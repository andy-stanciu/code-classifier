class Solution {

    int count = 0;

    public void method1(int[] arr, int idx, int total, int param1) {
        if (idx == arr.length) {
            if (total == param1)
                count++;
            return;
        } else {
            method1(arr, idx + 1, total + arr[idx], param1);
            method1(arr, idx + 1, total - arr[idx], param1);
        }
    }

    public int method2(int[] nums, int param1) {
        method1(nums, 0, 0, param1);
        return count;
    }
}
