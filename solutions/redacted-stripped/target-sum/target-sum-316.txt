class Solution {

    int res = 0;

    public int method1(int[] nums, int param1) {
        backtracking(nums, 0, param1, 0);
        return res;
    }

    private void backtracking(int[] nums, int start, int param1, int param2) {
        if (start == nums.length) {
            if (param2 == param1) {
                res++;
            }
            return;
        }
        backtracking(nums, start + 1, param1, param2 + nums[start]);
        backtracking(nums, start + 1, param1, param2 - nums[start]);
    }
}
