class Solution {

    public int method1(int[] nums, int param1) {
        Map<String, Integer> memo = new HashMap<>();
        return helper(nums, 0, 0, param1, memo);
    }

    private int helper(int[] nums, int index, int param2, int param1, Map<String, Integer> memo) {
        if (index >= nums.length) {
            if (param2 == param1)
                return 1;
            else
                return 0;
        }
        String key = param2 + "|" + index;
        if (memo.containsKey(key))
            return memo.get(key);
        int count = 0;
        count += helper(nums, index + 1, param2 - nums[index], param1, memo);
        count += helper(nums, index + 1, param2 + nums[index], param1, memo);
        memo.put(key, count);
        return count;
    }
}
