class Solution {

    public int method1(int[] nums, int param1) {
        int total = 0;
        for (int n : nums) {
            total += n;
        }
        param1 = Math.abs(param1);
        if ((param1 + total) % 2 == 1) {
            return 0;
        }
        int s1 = (param1 + total) / 2;
        int[] dp = new int[s1 + 1];
        dp[0] = 1;
        for (int num : nums) {
            for (int i = s1; i >= num; i--) {
                dp[i] += dp[i - num];
            }
        }
        return dp[s1];
    }
}
