class Solution {

    public int method1(int[] nums, int param1) {
        return method1(nums, 0, 0, param1);
    }

    public static int method1(int[] nums, int index, int ans, int param1) {
        if (index == nums.length) {
            if (ans == param1)
                return 1;
            else
                return 0;
        }
        return method1(nums, index + 1, ans + nums[index], param1) + method1(nums, index + 1, ans - nums[index], param1);
    }
}
