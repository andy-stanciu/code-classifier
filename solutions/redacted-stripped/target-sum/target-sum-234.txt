class Solution {

    public int method1(int[] nums, int param1) {
        int N = nums.length;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 = var1 + nums[i];
        }
        if ((param1 + var1) % 2 != 0 || (param1 + var1) < 0)
            return 0;
        return method2(nums.length, nums, (param1 + var1) / 2);
    }

    static int method2(int N, int[] arr, int var1) {
        int[][] t = new int[N + 1][var1 + 1];
        for (int i = 0; i <= N; i++) {
            for (int j = 0; j <= var1; j++) {
                if (i == 0)
                    t[i][j] = 0;
                if (j == 0)
                    t[i][j] = 1;
            }
        }
        for (int i = 1; i <= N; i++) {
            for (int j = 0; j <= var1; j++) {
                if (arr[i - 1] > j)
                    t[i][j] = t[i - 1][j];
                else
                    t[i][j] = (t[i - 1][j] + t[i - 1][j - arr[i - 1]]);
            }
        }
        return t[N][var1];
    }
}
