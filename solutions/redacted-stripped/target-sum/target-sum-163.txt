class Solution {

    public int solve(int[] a, int s) {
        int n = a.length;
        int[][] dp = new int[n + 1][s + 1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= s; j++) {
                if (i == 0 && j == 0) {
                    dp[i][j] = 1;
                } else if (i == 0) {
                    dp[i][j] = 0;
                } else {
                    if (a[i - 1] <= j) {
                        dp[i][j] = dp[i - 1][j - a[i - 1]] + dp[i - 1][j];
                    } else {
                        dp[i][j] = dp[i - 1][j];
                    }
                }
            }
        }
        return dp[n][s];
    }

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int i : nums) {
            var1 += i;
        }
        int s1 = (param1 + var1) / 2;
        if ((var1 + param1) % 2 != 0 || s1 < 0) {
            return 0;
        }
        return solve(nums, s1);
    }
}
