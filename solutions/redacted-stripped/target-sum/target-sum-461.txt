class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) var1 += num;
        if (var1 < Math.abs(param1) || (var1 + param1) % 2 == 1)
            return 0;
        return subsets(nums, (var1 + param1) / 2);
    }

    public int subsets(int[] nums, int var1) {
        int n = nums.length;
        int[][] dp = new int[n + 1][var1 + 1];
        dp[0][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= var1; j++) {
                if (j >= nums[i - 1]) {
                    dp[i][j] = dp[i - 1][j] + dp[i - 1][j - nums[i - 1]];
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][var1];
    }
}
