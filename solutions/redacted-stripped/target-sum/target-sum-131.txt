class Solution {

    public int method1(int[] nums, int param1) {
        int[] count = { 0 };
        method2(nums, param1, 0, count, 0);
        return count[0];
    }

    public void method2(int[] nums, int param1, int ptr, int[] count, int value) {
        if (ptr == nums.length) {
            if (value == param1) {
                count[0]++;
            }
            return;
        }
        method2(nums, param1, ptr + 1, count, value - nums[ptr]);
        method2(nums, param1, ptr + 1, count, value + nums[ptr]);
    }
}
