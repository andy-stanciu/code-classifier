class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        if (var1 < Math.abs(param1) || (var1 + param1) % 2 != 0) {
            return 0;
        }
        int s1 = (var1 + param1) / 2;
        int[][] dp = new int[nums.length + 1][s1 + 1];
        dp[0][0] = 1;
        for (int i = 1; i < dp.length; i++) {
            for (int j = 0; j < dp[0].length; j++) {
                dp[i][j] = ((j - nums[i - 1]) < 0 ? 0 : dp[i - 1][j - nums[i - 1]]) + dp[i - 1][j];
            }
        }
        return dp[dp.length - 1][dp[0].length - 1];
    }
}
