class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        if (Math.abs(param1) > var1 || (var1 + param1) % 2 == 1) {
            return 0;
        }
        param1 = (var1 + param1) / 2;
        int n = nums.length;
        int[][] dp = new int[n + 1][param1 + 1];
        for (int i = 0; i <= n; i++) {
            dp[i][0] = 1;
        }
        int numOfZero = 0;
        for (int i = 1; i <= n; i++) {
            if (nums[i - 1] == 0) {
                numOfZero++;
            }
            dp[i][0] = (int) Math.pow(2, numOfZero);
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= param1; j++) {
                if (j < nums[i - 1]) {
                    dp[i][j] = dp[i - 1][j];
                } else {
                    dp[i][j] = dp[i - 1][j] + dp[i - 1][j - nums[i - 1]];
                }
            }
        }
        return dp[n][param1];
    }
}
