class Solution {

    public int method1(int[] arr, int param1) {
        return recurse(arr, 0, 0, param1);
    }

    private int recurse(int[] arr, int i, int param2, int param1) {
        int n = arr.length;
        if (i == n)
            return param2 == param1 ? 1 : 0;
        return recurse(arr, i + 1, param2 + arr[i], param1) + recurse(arr, i + 1, param2 - arr[i], param1);
    }
}
