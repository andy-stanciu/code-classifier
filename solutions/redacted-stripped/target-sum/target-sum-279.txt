class Solution {

    static int count(int[] nums, int t, int i, int s) {
        if (i == nums.length) {
            if (s == t) {
                return 1;
            } else {
                return 0;
            }
        }
        return count(nums, t, i + 1, s + nums[i]) + count(nums, t, i + 1, s - nums[i]);
    }

    public int method1(int[] nums, int param1) {
        return count(nums, param1, 0, 0);
    }
}
