class Solution {

    int output = 0;

    public void solve(int[] arr, int i, int param1, int param2) {
        if (i == arr.length) {
            if (param1 == param2) {
                output++;
            }
            return;
        }
        solve(arr, i + 1, param1 + arr[i], param2);
        solve(arr, i + 1, param1 - arr[i], param2);
    }

    public int method1(int[] nums, int param2) {
        solve(nums, 0, 0, param2);
        return output;
    }
}
