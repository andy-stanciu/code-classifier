class Solution {

    public int method1(int[] nums, int param1) {
        Map<Pair<Integer, Integer>, Integer> dp = new HashMap<>();
        return method2(nums, param1, 0, dp);
    }

    private int method2(int[] nums, int param1, int idx, Map<Pair<Integer, Integer>, Integer> dp) {
        if (idx == nums.length) {
            if (param1 == 0)
                return 1;
            return 0;
        }
        Pair<Integer, Integer> key = new Pair(idx, param1);
        Integer dpValue = dp.get(key);
        if (dpValue != null)
            return dpValue;
        int newValue = method2(nums, param1 - nums[idx], idx + 1, dp) + method2(nums, param1 + nums[idx], idx + 1, dp);
        dp.put(key, newValue);
        return newValue;
    }
}
