class Solution {

    public int method1(int[] nums, int param1, int i, int counter) {
        if (i == nums.length) {
            if (counter == param1) {
                return 1;
            } else {
                return 0;
            }
        }
        return method1(nums, param1, i + 1, counter + nums[i]) + method1(nums, param1, i + 1, counter - nums[i]);
    }

    public int method2(int[] nums, int param1) {
        return method1(nums, param1, 0, 0);
    }
}
