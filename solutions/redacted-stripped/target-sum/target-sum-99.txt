class Solution {

    public int method1(int[] nums, int param1) {
        return solution(nums, param1, 0, 0);
    }

    public int solution(int[] nums, int param1, int i, int param2) {
        if (i == nums.length) {
            return (param2 == param1 ? 1 : 0);
        }
        int add = solution(nums, param1, i + 1, param2 + nums[i]);
        int sub = solution(nums, param1, i + 1, param2 - nums[i]);
        return add + sub;
    }
}
