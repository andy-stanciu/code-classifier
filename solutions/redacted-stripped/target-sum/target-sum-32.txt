class Solution {

    public int method1(int[] a, int param1) {
        int var1 = 0;
        for (int i : a) var1 += i;
        if (param1 + var1 < 0)
            return 0;
        if ((param1 + var1) % 2 == 1)
            return 0;
        int n = a.length;
        int var2 = (param1 + var1) / 2;
        int[][] dp = new int[n][var2 + 1];
        if (a[0] == 0)
            dp[0][0] = 2;
        else
            dp[0][0] = 1;
        if (a[0] != 0 && a[0] <= var2)
            dp[0][a[0]] = 1;
        for (int i = 1; i < n; i++) {
            for (param1 = 0; param1 <= var2; param1++) {
                int notPick = dp[i - 1][param1];
                int pick = 0;
                if (param1 >= a[i])
                    pick = dp[i - 1][param1 - a[i]];
                dp[i][param1] = pick + notPick;
            }
        }
        return dp[n - 1][var2];
    }
}
