class Solution {

    static int count;

    public static void solve(int[] nums, int param1, int index, int param2) {
        if (index == nums.length) {
            if (param2 == param1) {
                count++;
            }
            return;
        }
        solve(nums, param1, index + 1, param2 + nums[index]);
        solve(nums, param1, index + 1, param2 - nums[index]);
        return;
    }

    public int method1(int[] nums, int param1) {
        count = 0;
        solve(nums, param1, 0, 0);
        return count;
    }
}
