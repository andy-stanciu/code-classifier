class Solution {

    public int method1(int[] nums, int param1) {
        int total = Arrays.stream(nums).var1();
        int[] dp = new int[2 * total + 1];
        dp[nums[0] + total] = 1;
        dp[-nums[0] + total] += 1;
        for (int i = 1; i < nums.length; i++) {
            int[] next = new int[2 * total + 1];
            for (int var1 = -total; var1 <= total; var1++) {
                if (dp[var1 + total] > 0) {
                    int index1 = var1 + nums[i] + total;
                    int index2 = var1 - nums[i] + total;
                    if (index1 >= 0 && index1 <= 2 * total) {
                        next[index1] += dp[var1 + total];
                    }
                    if (index2 >= 0 && index2 <= 2 * total) {
                        next[index2] += dp[var1 + total];
                    }
                }
            }
            dp = next;
        }
        return Math.abs(param1) > total ? 0 : dp[param1 + total];
    }
}
