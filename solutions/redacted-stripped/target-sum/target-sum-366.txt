class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        HashMap<String, Integer> map = new HashMap<String, Integer>();
        return recu(nums, n, 0, param1, map);
    }

    public int recu(int[] arr, int i, int param2, int param1, HashMap<String, Integer> map) {
        if (i == 0 && param2 == param1) {
            return 1;
        }
        if (i == 0 && param2 != param1) {
            return 0;
        }
        String s = param2 + "$" + i;
        if (map.containsKey(s)) {
            return map.get(s);
        }
        int x = recu(arr, i - 1, param2 - arr[i - 1], param1, map) + recu(arr, i - 1, param2 + arr[i - 1], param1, map);
        map.put(s, x);
        return x;
    }
}
