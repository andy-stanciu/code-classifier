class Solution {

    public int method1(int[] nums, int param1) {
        Map<String, Integer> memo = new HashMap<>();
        return helper(nums, param1, 0, memo);
    }

    private int helper(int[] nums, int param1, int index, Map<String, Integer> memo) {
        if (index == nums.length) {
            return param1 == 0 ? 1 : 0;
        }
        String key = index + "," + param1;
        if (memo.containsKey(key)) {
            return memo.get(key);
        }
        int ways = helper(nums, param1 - nums[index], index + 1, memo) + helper(nums, param1 + nums[index], index + 1, memo);
        memo.put(key, ways);
        return ways;
    }
}
