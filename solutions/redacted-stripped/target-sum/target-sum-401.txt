class Solution {

    int ans = 0;

    public void f(int t, int e, int[] nums, int id) {
        if (id == nums.length) {
            if (e == t) {
                ans++;
            }
            return;
        }
        f(t, e + nums[id], nums, id + 1);
        f(t, e - nums[id], nums, id + 1);
    }

    public int method1(int[] nums, int param1) {
        f(param1, 0, nums, 0);
        return ans;
    }
}
