class Solution {

    public int method1(int[] nums, int param1) {
        int s = 0;
        for (int i : nums) s += i;
        Arrays.sort(nums);
        int var1 = (param1 + s) / 2;
        int n = nums.length;
        if ((param1 + s) % 2 != 0)
            return 0;
        if (param1 + s < 0)
            return 0;
        int[][] dp = new int[n + 1][var1 + 1];
        for (int i = 0; i < n + 1; i++) {
            for (int j = 0; j < var1 + 1; j++) {
                if (i == 0)
                    dp[i][j] = 0;
                if (j == 0)
                    dp[i][j] = 1;
            }
        }
        for (int i = 1; i < n + 1; i++) {
            for (int j = 0; j < var1 + 1; j++) {
                if (nums[i - 1] <= j) {
                    dp[i][j] = dp[i - 1][j - nums[i - 1]] + dp[i - 1][j];
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][var1];
    }
}
