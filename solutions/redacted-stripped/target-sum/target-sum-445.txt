class Solution {

    public static int method1(int[] nums, int param1) {
        return method2(nums, param1, 0, 0);
    }

    private static int method2(int[] nums, int param1, int p, int param2) {
        if (p == nums.length) {
            return param1 == param2 ? 1 : 0;
        }
        return method2(nums, param1, p + 1, param2 + (nums[p])) + method2(nums, param1, p + 1, param2 + (-nums[p]));
    }
}
