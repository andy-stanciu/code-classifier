class Solution {

    public int method1(int[] nums, int param1) {
        return totalWays(nums, 0, param1);
    }

    private int totalWays(int[] nums, int currentIndex, int param1) {
        if (currentIndex == nums.length && param1 == 0)
            return 1;
        if (currentIndex == nums.length && param1 != 0)
            return 0;
        int posSign = totalWays(nums, currentIndex + 1, param1 - nums[currentIndex]);
        int negSign = totalWays(nums, currentIndex + 1, param1 + nums[currentIndex]);
        return posSign + negSign;
    }
}
