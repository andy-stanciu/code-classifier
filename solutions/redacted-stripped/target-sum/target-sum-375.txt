class Solution {

    public int method1(int[] nums, int param1) {
        HashMap<String, Integer> hash = new HashMap();
        return DFS(nums, param1, 0, 0, hash);
    }

    public int DFS(int[] nums, int param1, int param2, int index, HashMap<String, Integer> hash) {
        String key = index + "," + param2;
        if (index == nums.length) {
            if (param2 == param1)
                return 1;
            else
                return 0;
        }
        if (hash.containsKey(key))
            return hash.get(key);
        int add = DFS(nums, param1, param2 + nums[index], index + 1, hash);
        int subtract = DFS(nums, param1, param2 - nums[index], index + 1, hash);
        hash.put(key, add + subtract);
        return hash.get(key);
    }
}
