class Solution {

    int ans = 0;

    public int method1(int[] nums, int param1) {
        helper(nums, 0, 0, param1, nums.length);
        return ans;
    }

    public void helper(int[] arr, int index, int param2, int param1, int n) {
        if (index == n) {
            if (param2 == param1)
                ans++;
            return;
        }
        helper(arr, index + 1, param2 - arr[index], param1, n);
        helper(arr, index + 1, param2 + arr[index], param1, n);
    }
}
