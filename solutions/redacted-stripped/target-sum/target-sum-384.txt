class Solution {

    public int method1(int[] nums, int param1) {
        return method2(0, nums, param1);
    }

    private int method2(int curr, int[] nums, int param1) {
        if (nums.length == 0) {
            if (curr == param1) {
                return 1;
            }
            return 0;
        }
        int[] newNums = Arrays.copyOfRange(nums, 1, nums.length);
        return findTarget(curr + nums[0], newNums, param1) + findTarget(curr - nums[0], newNums, param1);
    }
}
