class Solution {

    public int getWays(int[] nums, int param1, int idx) {
        if (idx >= nums.length) {
            return param1 == 0 ? 1 : 0;
        }
        return getWays(nums, param1 - nums[idx], idx + 1) + getWays(nums, param1 + nums[idx], idx + 1);
    }

    public int method1(int[] nums, int param1) {
        return getWays(nums, param1, 0);
    }
}
