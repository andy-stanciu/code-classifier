class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] >= var1) {
                profit = Math.max(profit, prices[i] - var1);
            } else {
                var1 = prices[i];
            }
        }
        return profit;
    }
}
