class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int min = prices[0];
        for (int day = 1; day < prices.length; day++) {
            if (prices[day] < min) {
                min = prices[day];
            } else {
                profit = Math.max(profit, prices[day] - min);
            }
        }
        return profit;
    }
}
