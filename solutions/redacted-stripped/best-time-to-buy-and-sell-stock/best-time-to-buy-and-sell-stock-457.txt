class Solution {

    public int maxProfit(int[] a) {
        int maxprof = 0;
        int minsofar = a[0];
        for (int i = 0; i < a.length; i++) {
            minsofar = Math.min(minsofar, a[i]);
            int profit = a[i] - minsofar;
            maxprof = Math.max(maxprof, profit);
        }
        return maxprof;
    }
}
