class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 0)
            return 0;
        int profit = 0;
        int min = prices[0];
        int max = 0;
        for (int var1 : prices) {
            min = Math.min(var1, min);
            profit = var1 - min;
            max = Math.max(profit, max);
        }
        return max;
    }
}
