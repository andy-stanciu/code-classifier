class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0], var2 = prices[0], ans = 0;
        for (int i = 0; i < prices.length; i++) {
            if (var2 > prices[i]) {
                var2 = prices[i];
                var1 = prices[i];
            }
            var1 = Math.max(var1, prices[i]);
            ans = Math.max(var1 - var2, ans);
        }
        return ans;
    }
}
