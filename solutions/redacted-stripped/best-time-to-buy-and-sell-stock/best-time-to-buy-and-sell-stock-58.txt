class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length <= 1)
            return 0;
        int var1 = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            profit = Math.max(profit, prices[i] - var1);
        }
        return profit;
    }
}
