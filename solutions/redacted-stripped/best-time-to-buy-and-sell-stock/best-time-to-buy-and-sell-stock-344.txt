class Solution {

    public int maxProfit(int[] prices) {
        int maxCurrent = 0;
        int maxSoFar = 0;
        for (int i = 1; i < prices.length; i++) {
            int diff = prices[i] - prices[i - 1];
            maxCurrent = Math.max(0, maxCurrent + diff);
            maxSoFar = Math.max(maxSoFar, maxCurrent);
        }
        return maxSoFar;
    }
}
