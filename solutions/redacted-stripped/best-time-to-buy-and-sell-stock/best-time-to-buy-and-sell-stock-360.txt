class Solution {

    public int maxProfit(int[] prices) {
        int maxp = Integer.MIN_VALUE;
        int var1 = 0;
        int minv = prices[0];
        for (int a : prices) {
            minv = Math.min(minv, a);
            maxp = Math.max(maxp, a - minv);
        }
        return maxp;
    }
}
