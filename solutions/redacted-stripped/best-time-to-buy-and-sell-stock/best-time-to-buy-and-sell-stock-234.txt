class Solution {

    public int maxProfit(int[] prices) {
        int minprice = Integer.MAX_VALUE;
        int maxprofit = 0;
        for (int e : prices) {
            if (e < minprice) {
                minprice = e;
            } else if (e - minprice > maxprofit) {
                maxprofit = e - minprice;
            }
        }
        return maxprofit;
    }
}
