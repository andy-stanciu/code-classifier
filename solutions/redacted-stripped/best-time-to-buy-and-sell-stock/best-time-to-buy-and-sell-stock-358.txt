class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int maxD = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < min) {
                min = prices[i];
                System.out.println("min " + min);
            } else {
                int newMaxD = prices[i] - min;
                if (newMaxD > maxD) {
                    maxD = newMaxD;
                }
            }
        }
        return maxD;
    }
}
