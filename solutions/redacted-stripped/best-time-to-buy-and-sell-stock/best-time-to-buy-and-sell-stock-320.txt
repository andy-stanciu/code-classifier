class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = Integer.MAX_VALUE;
        for (int price : prices) {
            if (price < var1) {
                var1 = price;
            }
            maxProfit = Math.max(maxProfit, price - var1);
        }
        return maxProfit;
    }
}
