class Solution {

    public int maxProfit(int[] prices) {
        int least = Integer.MAX_VALUE;
        int var1 = 0;
        int var2 = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < least) {
                least = prices[i];
            } else {
                var2 = prices[i] - least;
                if (var1 < var2) {
                    var1 = var2;
                }
            }
        }
        return var1;
    }
}
