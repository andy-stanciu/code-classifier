class Solution {

    public int maxProfit(int[] prices) {
        int prof = 0;
        int f_prof = 0;
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < min) {
                min = prices[i];
            }
            prof = prices[i] - min;
            if (f_prof < prof) {
                f_prof = prof;
            }
        }
        return f_prof;
    }
}
