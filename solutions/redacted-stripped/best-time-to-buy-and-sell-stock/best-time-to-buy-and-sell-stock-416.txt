class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int prof = 0;
        for (int p : prices) {
            if (p < var1) {
                var1 = p;
            } else {
                prof = Math.max(prof, p - var1);
            }
        }
        return prof;
    }
}
