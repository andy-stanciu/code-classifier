class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (var1 < prices[i]) {
                profit = Math.max(prices[i] - var1, profit);
            } else {
                var1 = prices[i];
            }
        }
        return profit;
    }
}
