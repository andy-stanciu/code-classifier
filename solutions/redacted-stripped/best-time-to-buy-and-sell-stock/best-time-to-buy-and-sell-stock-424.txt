class Solution {

    public int maxProfit(int[] prices) {
        int maxi = 0;
        int mini = prices[0];
        int n = prices.length;
        for (int i = 0; i < n; i++) {
            int cost = prices[i] - mini;
            maxi = Math.max(maxi, cost);
            mini = Math.min(mini, prices[i]);
        }
        return maxi;
    }
}
