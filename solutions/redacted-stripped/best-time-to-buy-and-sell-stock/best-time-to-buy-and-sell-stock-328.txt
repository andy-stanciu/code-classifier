class Solution {

    public int maxProfit(int[] prices) {
        int mani = prices[0];
        int maxProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            int cost = prices[i] - mani;
            maxProfit = Math.max(maxProfit, cost);
            mani = Math.min(mani, prices[i]);
        }
        return maxProfit;
    }
}
