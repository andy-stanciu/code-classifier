class Solution {

    public int maxProfit(int[] prices) {
        int max = 0;
        int lowind = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[lowind] > prices[i])
                lowind = i;
            if (max < prices[i] - prices[lowind])
                max = prices[i] - prices[lowind];
        }
        return max;
    }
}
