class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            profit = Math.max(profit, prices[i] - var1);
        }
        return profit;
    }
}
