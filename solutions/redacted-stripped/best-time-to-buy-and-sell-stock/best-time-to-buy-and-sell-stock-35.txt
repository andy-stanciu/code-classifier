class Solution {

    public int maxProfit(int[] prices) {
        int len = prices.length;
        if (len == 1)
            return 0;
        int min = prices[0];
        int profit = 0;
        for (int i = 1; i < len; i++) {
            if (prices[i] < min) {
                min = prices[i];
                continue;
            }
            if ((prices[i] - min) > profit)
                profit = prices[i] - min;
        }
        return profit;
    }
}
