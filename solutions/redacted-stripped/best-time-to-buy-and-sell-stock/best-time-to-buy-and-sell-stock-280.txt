class Solution {

    public int maxProfit(int[] prices) {
        Stack<Integer> st = new Stack<>();
        int maxi = 0;
        for (int i = 0; i < prices.length; i++) {
            while (!st.isEmpty() && prices[i] <= st.peek()) {
                st.pop();
            }
            if (st.isEmpty()) {
                st.push(prices[i]);
            }
            if (prices[i] > st.peek()) {
                maxi = Math.max(maxi, prices[i] - st.peek());
            }
        }
        return maxi;
    }
}
