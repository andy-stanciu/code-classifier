class Solution {

    public int maxProfit(int[] prices) {
        int borrow = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < borrow) {
                borrow = prices[i];
            } else if (prices[i] - borrow > profit) {
                profit = prices[i] - borrow;
            }
        }
        return profit;
    }
}
