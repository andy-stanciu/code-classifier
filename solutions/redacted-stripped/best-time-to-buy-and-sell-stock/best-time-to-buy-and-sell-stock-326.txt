class Solution {

    public int maxProfit(int[] prices) {
        int cheapest = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (cheapest > prices[i]) {
                cheapest = prices[i];
                continue;
            }
            if (prices[i] - cheapest > profit) {
                profit = prices[i] - cheapest;
            }
        }
        return profit;
    }
}
