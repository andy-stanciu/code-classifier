class Solution {

    public int maxProfit(int[] prices) {
        int l = prices.length;
        if (l < 1)
            return -1;
        int minPrice = prices[0];
        int maxProfit = 0;
        for (int right = 1; right < l; right++) {
            if (minPrice > prices[right])
                minPrice = prices[right];
            if (maxProfit < (prices[right] - minPrice))
                maxProfit = prices[right] - minPrice;
        }
        return maxProfit;
    }
}
