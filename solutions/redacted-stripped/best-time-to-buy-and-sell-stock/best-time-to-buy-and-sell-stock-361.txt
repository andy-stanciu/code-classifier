class Solution {

    public int maxProfit(int[] prices) {
        int netProfit = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (var1 < prices[i]) {
                int currProfit = prices[i] - var1;
                netProfit = Integer.max(currProfit, netProfit);
            } else
                var1 = prices[i];
        }
        return netProfit;
    }
}
