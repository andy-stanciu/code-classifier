class Solution {

    public int maxProfit(int[] arr) {
        int mini = arr[0];
        int maxprofit = 0;
        for (int i = 1; i < arr.length; i++) {
            int cost = arr[i] - mini;
            maxprofit = Math.max(maxprofit, cost);
            mini = Math.min(mini, arr[i]);
        }
        return maxprofit;
    }
}
