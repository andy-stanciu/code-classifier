class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            } else if (profit < (prices[i] - var1)) {
                profit = prices[i] - var1;
            }
        }
        return profit;
    }
}
