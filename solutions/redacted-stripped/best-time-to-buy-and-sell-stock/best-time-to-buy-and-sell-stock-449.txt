class Solution {

    public int maxProfit(int[] prices) {
        int minprice = Integer.MAX_VALUE;
        int result = 0;
        int n = prices.length;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < minprice)
                minprice = prices[i];
            else {
                result = Math.max(result, prices[i] - minprice);
            }
        }
        return result;
    }
}
