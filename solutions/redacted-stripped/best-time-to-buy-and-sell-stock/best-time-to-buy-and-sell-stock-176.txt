class Solution {

    public int maxProfit(int[] prices) {
        int min_price = prices[0];
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            int temp_profit;
            if (prices[i] < min_price) {
                min_price = prices[i];
            }
            temp_profit = prices[i] - min_price;
            if (profit < temp_profit)
                profit = temp_profit;
        }
        if (profit > 0) {
            return profit;
        } else {
            return 0;
        }
    }
}
