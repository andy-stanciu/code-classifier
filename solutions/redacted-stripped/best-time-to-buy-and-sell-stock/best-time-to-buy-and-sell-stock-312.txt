class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int profit;
        int var1 = prices[0];
        for (int price : prices) {
            if (var1 < price) {
                profit = price - var1;
                if (profit > maxProfit)
                    maxProfit = profit;
            } else {
                var1 = price;
            }
        }
        return maxProfit;
    }
}
