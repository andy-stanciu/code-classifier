class Solution {

    public int maxProfit(int[] prices) {
        int lowestValue = prices[0];
        int highest = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < lowestValue) {
                lowestValue = prices[i];
            } else {
                int profit = prices[i] - lowestValue;
                if (profit > highest) {
                    highest = profit;
                }
            }
        }
        return highest;
    }
}
