class Solution {

    public int maxProfit(int[] prices) {
        int[] out = new int[2];
        int maxGl = prices[prices.length - 1];
        out[1] = 0;
        for (int i = prices.length - 2; i >= 0; i--) {
            out[0] = max(maxGl - prices[i], out[1]);
            maxGl = max(maxGl, prices[i]);
            out[1] = out[0];
        }
        return out[0];
    }

    private int max(int a, int b) {
        return a > b ? a : b;
    }
}
