class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length < 2) {
            return 0;
        }
        int maxProfit = 0;
        int var1 = 0;
        int var2 = 1;
        int n = prices.length;
        while (var2 < n) {
            int currentPrice = prices[var2];
            int var3 = prices[var1];
            if (var3 < currentPrice) {
                int currentProfit = currentPrice - var3;
                maxProfit = Math.max(maxProfit, currentProfit);
            } else {
                var1 = var2;
            }
            var2++;
        }
        return maxProfit;
    }
}
