class Solution {

    public int maxProfit(int[] prices) {
        int prof = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if ((var1 > prices[i])) {
                var1 = prices[i];
            }
            prof = Math.max(prof, prices[i] - var1);
        }
        return prof;
    }
}
