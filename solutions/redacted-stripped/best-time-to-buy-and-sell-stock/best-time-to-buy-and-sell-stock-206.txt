class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int var2 = 1;
        int profit = 0;
        while (var2 < prices.length) {
            if (prices[var1] >= prices[var2]) {
                var1 = var2;
            } else {
                int potentialProfit = prices[var2] - prices[var1];
                profit = Math.max(profit, potentialProfit);
            }
            var2++;
        }
        return profit;
    }
}
