class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int costPrice = prices[0];
        for (int i = 1; i < prices.length; i++) {
            maxProfit = Math.max(maxProfit, prices[i] - costPrice);
            if (prices[i] < costPrice)
                costPrice = prices[i];
        }
        return maxProfit;
    }
}
