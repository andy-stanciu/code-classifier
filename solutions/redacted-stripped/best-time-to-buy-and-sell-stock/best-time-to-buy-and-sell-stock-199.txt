class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length == 0)
            return 0;
        int maxProfit = 0;
        int maxEndingHere = prices[0];
        int minEndingHere = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < maxEndingHere) {
                maxEndingHere = prices[i];
            }
            maxEndingHere = Math.max(maxEndingHere, prices[i]);
            minEndingHere = Math.min(minEndingHere, prices[i]);
            maxProfit = Math.max(maxProfit, Math.abs(maxEndingHere - minEndingHere));
        }
        return maxProfit;
    }
}
