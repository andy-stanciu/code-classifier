class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int maxProfit = Integer.MIN_VALUE;
        for (int i : prices) {
            if (i < var1)
                var1 = i;
            if (maxProfit < i - var1)
                maxProfit = i - var1;
        }
        return maxProfit;
    }
}
