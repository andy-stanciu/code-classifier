class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int max = 0;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            profit = prices[i] - min;
            max = Math.max(profit, max);
            min = Math.min(prices[i], min);
        }
        return max;
    }
}
