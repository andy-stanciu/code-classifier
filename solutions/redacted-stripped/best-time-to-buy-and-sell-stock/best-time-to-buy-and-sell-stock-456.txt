class Solution {

    public int maxProfit(int[] prices) {
        int maxprofit = 0;
        int var1 = prices[0];
        for (int i = 0; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            int profit = prices[i] - var1;
            maxprofit = Math.max(maxprofit, profit);
        }
        return maxprofit;
    }
}
