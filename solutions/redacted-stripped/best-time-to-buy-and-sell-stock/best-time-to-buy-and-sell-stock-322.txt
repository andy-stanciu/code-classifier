class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0], maxprofit = 0;
        for (int i = 1; i < prices.length; i++) {
            min = Math.min(min, prices[i]);
            maxprofit = Math.max(maxprofit, prices[i] - min);
        }
        return maxprofit;
    }
}
