class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        for (int i = 0; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            var2 = Math.max(var2, prices[i] - var1);
        }
        return var2;
    }
}
