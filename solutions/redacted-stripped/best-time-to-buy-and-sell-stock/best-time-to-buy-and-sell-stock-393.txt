class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            }
            if (prices[i] - var1 > profit) {
                profit = prices[i] - var1;
            }
        }
        return profit;
    }
}
