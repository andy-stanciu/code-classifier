class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int maxm = Integer.MIN_VALUE, var1 = 0;
        for (int i = n - 1; i >= 0; i--) {
            maxm = Math.max(maxm, prices[i]);
            var1 = Math.max(var1, maxm - prices[i]);
        }
        return var1;
    }
}
