class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else {
                int profit = prices[i] - var1;
                maxProfit = Math.max(maxProfit, profit);
            }
        }
        return maxProfit;
    }
}
