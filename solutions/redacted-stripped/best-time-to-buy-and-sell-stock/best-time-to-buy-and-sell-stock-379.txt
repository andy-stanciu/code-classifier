class Solution {

    public int maxProfit(int[] prices) {
        int Max = 0;
        int var1 = prices[0];
        for (int i : prices) {
            if (var1 < i) {
                int profit = i - var1;
                Max = Math.max(Max, profit);
            } else {
                var1 = i;
            }
        }
        return Max;
    }
}
