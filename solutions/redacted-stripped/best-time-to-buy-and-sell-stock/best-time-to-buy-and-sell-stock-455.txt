class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            int var2 = prices[i];
            int diff = var2 - var1;
            if (diff > profit)
                profit = diff;
            if (var2 < var1)
                var1 = var2;
        }
        return profit;
    }
}
