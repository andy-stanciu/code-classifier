class Solution {

    public int maxProfit(int[] prices) {
        int minI = 0;
        int maxP = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] <= prices[minI]) {
                minI = i;
                for (int j = minI + 1; j < prices.length; j++) {
                    maxP = Math.max(prices[j] - prices[minI], maxP);
                }
            }
        }
        return maxP;
    }
}
