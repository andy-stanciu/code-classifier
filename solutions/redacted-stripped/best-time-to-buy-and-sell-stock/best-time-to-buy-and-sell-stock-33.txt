class Solution {

    public static int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        int var2 = 0;
        for (int i = 1; i < prices.length; ++i) {
            profit = prices[i] - var1;
            if (profit < 0 && prices[i] < var1) {
                var1 = prices[i];
            }
            var2 = Math.max(profit, var2);
        }
        if (var2 < 0) {
            return 0;
        }
        return var2;
    }

    public static void main(String[] args) {
        int[] prices = { 7, 1, 5, 3, 6, 4 };
        System.out.println(maxProfit(prices));
    }
}
