class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int maxpro = 0;
        for (int i = 0; i < prices.length; i++) {
            if (var1 < prices[i]) {
                int profit = prices[i] - var1;
                maxpro = Math.max(profit, maxpro);
            } else {
                var1 = prices[i];
            }
        }
        return maxpro;
    }
}
