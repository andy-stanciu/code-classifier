class Solution {

    public int maxProfit(int[] prices) {
        int b = 0;
        int s = 1;
        int profit = 0;
        while (s < prices.length) {
            if (prices[b] > prices[s]) {
                b = s;
                s++;
            } else {
                if (prices[s] - prices[b] > profit) {
                    profit = prices[s] - prices[b];
                    s++;
                } else {
                    s++;
                }
            }
        }
        return profit;
    }
}
