class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int max_profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            } else if (prices[i] - var1 > max_profit) {
                max_profit = prices[i] - var1;
            }
        }
        return max_profit;
    }
}
