class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int j = prices[0];
        int profit = 0;
        for (int i = 0; i < n; i++) {
            int temp = prices[i] - j;
            profit = Math.max(profit, temp);
            if (prices[i] < j) {
                j = prices[i];
            }
        }
        return profit;
    }
}
