class Solution {

    public int maxProfit(int[] prices) {
        int min = Integer.MAX_VALUE;
        int profit = 0, maxProfit = 0;
        for (int num : prices) {
            min = Math.min(num, min);
            profit = num - min;
            maxProfit = Math.max(maxProfit, profit);
        }
        return maxProfit;
    }
}
