class Solution {

    public int maxProfit(int[] arr) {
        if (arr == null || arr.length == 0) {
            return 0;
        }
        int minPrice = arr[0];
        int maxProfit = 0;
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] < minPrice) {
                minPrice = arr[i];
            }
            int potentialProfit = arr[i] - minPrice;
            if (potentialProfit > maxProfit) {
                maxProfit = potentialProfit;
            }
        }
        return maxProfit;
    }
}
