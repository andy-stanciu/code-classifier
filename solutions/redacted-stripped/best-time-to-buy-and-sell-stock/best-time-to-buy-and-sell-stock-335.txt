class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length == 0) {
            return 0;
        }
        int SmallN = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < SmallN) {
                SmallN = prices[i];
            } else if (prices[i] - SmallN > maxProfit) {
                maxProfit = prices[i] - SmallN;
            }
        }
        return maxProfit;
    }
}
