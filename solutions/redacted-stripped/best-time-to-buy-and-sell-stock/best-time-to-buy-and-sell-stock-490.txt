class Solution {

    public int maxProfit(int[] prices) {
        int cp = Integer.MAX_VALUE, max = 0;
        for (int i : prices) {
            if (i < cp) {
                cp = i;
            } else if (i > cp) {
                max = Math.max(max, i - cp);
            }
        }
        return max;
    }
}
