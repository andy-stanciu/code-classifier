class Solution {

    public int maxProfit(int[] arr) {
        int t = 0;
        int n = arr.length;
        int mi = 0;
        for (int i = mi + 1; i < n; i++) {
            if (arr[mi] < arr[i]) {
                t = Math.max(t, arr[i] - arr[mi]);
            } else {
                mi = i;
            }
        }
        return t;
    }
}
