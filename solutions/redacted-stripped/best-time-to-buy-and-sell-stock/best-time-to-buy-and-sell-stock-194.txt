class Solution {

    public int maxProfit(int[] prices) {
        int p = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            } else {
                p = Math.max(p, prices[i] - var1);
            }
        }
        return p;
    }
}
