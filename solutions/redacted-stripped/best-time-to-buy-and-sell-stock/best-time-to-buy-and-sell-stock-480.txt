class Solution {

    public int maxProfit(int[] prices) {
        int maxprofit = 0;
        int profit;
        int minsofar = prices[0];
        for (int i = 1; i < prices.length; i++) {
            minsofar = Math.min(prices[i], minsofar);
            profit = prices[i] - minsofar;
            maxprofit = Math.max(profit, maxprofit);
        }
        return maxprofit;
    }
}
