class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int smleft = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < smleft) {
                smleft = prices[i];
            }
            var1 = Math.max(var1, prices[i] - smleft);
        }
        return var1;
    }
}
