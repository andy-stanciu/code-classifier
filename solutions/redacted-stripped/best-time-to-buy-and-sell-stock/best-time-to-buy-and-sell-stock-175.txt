class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int minCost = Integer.MAX_VALUE;
        for (int price : prices) {
            minCost = Math.min(price, minCost);
            int currentProfit = price - minCost;
            maxProfit = Math.max(currentProfit, maxProfit);
        }
        return maxProfit;
    }
}
