class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = prices[0], var2 = prices[0];
        for (int n : prices) {
            if (n < var1) {
                var1 = n;
                var2 = n;
            } else if (n > var2) {
                var2 = n;
            }
            maxProfit = Math.max(maxProfit, var2 - var1);
        }
        return maxProfit;
    }
}
