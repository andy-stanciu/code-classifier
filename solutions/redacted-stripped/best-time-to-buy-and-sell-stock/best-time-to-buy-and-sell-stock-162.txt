class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0, min = prices[0], temp;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < min)
                min = prices[i];
            temp = prices[i] - min;
            if (temp > profit)
                profit = temp;
        }
        return profit;
    }
}
