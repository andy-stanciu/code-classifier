class Solution {

    public int maxProfit(int[] prices) {
        int max_profit = 0;
        int min_value = prices[0];
        for (int i = 1; i < prices.length; i++) {
            max_profit = Math.max(max_profit, prices[i] - min_value);
            min_value = Math.min(min_value, prices[i]);
        }
        return max_profit;
    }
}
