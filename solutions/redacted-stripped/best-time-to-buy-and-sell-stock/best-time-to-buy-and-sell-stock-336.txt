class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int max = 0;
        for (int i = 0; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            }
            max = Math.max(max, prices[i] - var1);
        }
        return max;
    }
}
