class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        if (n == 0) {
            return 0;
        }
        int var1 = prices[0];
        int profit = 0;
        for (int i = 0; i < n; i++) {
            var1 = Math.min(var1, prices[i]);
            profit = Math.max(profit, prices[i] - var1);
        }
        return profit;
    }
}
