class Solution {

    public int maxProfit(int[] prices) {
        int min = 100000, max = 0, val = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < min) {
                min = prices[i];
                max = prices[i];
            } else if (prices[i] > max) {
                max = prices[i];
                if (max - min > val) {
                    val = max - min;
                }
            }
        }
        return val;
    }
}
