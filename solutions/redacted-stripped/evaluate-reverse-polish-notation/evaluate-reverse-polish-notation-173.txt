class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> val = new Stack<>();
        for (int i = 0; i < tokens.length; i++) {
            int v1;
            int v2;
            switch(tokens[i]) {
                case "+":
                    v2 = val.pop();
                    v1 = val.pop();
                    val.push(v1 + v2);
                    break;
                case "-":
                    v2 = val.pop();
                    v1 = val.pop();
                    val.push(v1 - v2);
                    break;
                case "*":
                    v2 = val.pop();
                    v1 = val.pop();
                    val.push(v1 * v2);
                    break;
                case "/":
                    v2 = val.pop();
                    v1 = val.pop();
                    val.push(v1 / v2);
                    break;
                default:
                    val.push(Integer.parseInt(tokens[i]));
            }
        }
        return val.peek();
    }
}
