class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<Integer>();
        for (String symbol : tokens) {
            if (symbol.equals("+")) {
                int result = stack.pop() + stack.pop();
                stack.push(result);
            } else if (symbol.equals("-")) {
                int from = stack.pop();
                int result = stack.pop() - from;
                stack.push(result);
            } else if (symbol.equals("*")) {
                int result = stack.pop() * stack.pop();
                stack.push(result);
            } else if (symbol.equals("/")) {
                int dividend = stack.pop();
                int result = stack.pop() / dividend;
                stack.push(result);
            } else {
                stack.push(Integer.valueOf(symbol));
            }
        }
        return stack.pop();
    }
}
