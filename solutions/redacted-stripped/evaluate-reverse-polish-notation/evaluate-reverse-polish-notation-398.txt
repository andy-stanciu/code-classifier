class Solution {

    public int evalRPN(String[] tokens) {
        Deque<Integer> stack = new ArrayDeque<>();
        for (String s : tokens) {
            if (s.equals("+")) {
                int i = stack.pollFirst();
                int j = stack.pollFirst();
                stack.offerFirst(i + j);
            } else if (s.equals("-")) {
                int i = stack.pollFirst();
                int j = stack.pollFirst();
                stack.offerFirst(j - i);
            } else if (s.equals("*")) {
                int i = stack.pollFirst();
                int j = stack.pollFirst();
                stack.offerFirst(i * j);
            } else if (s.equals("/")) {
                int denom = stack.pollFirst();
                int nom = stack.pollFirst();
                stack.offerFirst(nom / denom);
            } else {
                int i = Integer.valueOf(s);
                stack.offerFirst(i);
            }
        }
        return stack.pollFirst();
    }
}
