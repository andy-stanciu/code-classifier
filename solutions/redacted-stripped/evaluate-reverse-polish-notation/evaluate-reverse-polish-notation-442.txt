class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < tokens.length; i++) {
            String s = tokens[i];
            int x;
            int y;
            switch(s) {
                case "+":
                    x = stack.pop();
                    y = stack.pop();
                    stack.push(x + y);
                    break;
                case "-":
                    x = stack.pop();
                    y = stack.pop();
                    stack.push(y - x);
                    break;
                case "/":
                    x = stack.pop();
                    y = stack.pop();
                    stack.push(y / x);
                    break;
                case "*":
                    x = stack.pop();
                    y = stack.pop();
                    stack.push(x * y);
                    break;
                default:
                    stack.push(Integer.parseInt(s));
                    break;
            }
        }
        return stack.pop();
    }
}
