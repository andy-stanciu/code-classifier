class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> st = new Stack<>();
        int n = tokens.length;
        for (int i = 0; i < n; i++) {
            String str = tokens[i];
            if (str.equals("+") || str.equals("-") || str.equals("*") || str.equals("/")) {
                if (!st.isEmpty()) {
                    int v1 = st.pop();
                    int v2 = st.pop();
                    if (str.equals("+")) {
                        st.push(v1 + v2);
                    } else if (str.equals("-")) {
                        st.push(v2 - v1);
                    } else if (str.equals("*")) {
                        st.push(v1 * v2);
                    } else if (str.equals("/")) {
                        st.push(v2 / v1);
                    }
                }
            } else {
                st.push(Integer.parseInt(str));
            }
        }
        return st.pop();
    }
}
