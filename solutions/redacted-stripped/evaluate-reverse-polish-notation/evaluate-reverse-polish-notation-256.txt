class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String s : tokens) {
            int firstNum;
            int secondNum;
            switch(s) {
                case "+" ->
                    {
                        secondNum = stack.pop();
                        firstNum = stack.pop();
                        stack.push(firstNum + secondNum);
                    }
                case "-" ->
                    {
                        secondNum = stack.pop();
                        firstNum = stack.pop();
                        stack.push(firstNum - secondNum);
                    }
                case "/" ->
                    {
                        secondNum = stack.pop();
                        firstNum = stack.pop();
                        stack.push(firstNum / secondNum);
                    }
                case "*" ->
                    {
                        secondNum = stack.pop();
                        firstNum = stack.pop();
                        stack.push(firstNum * secondNum);
                    }
                default ->
                    stack.push(Integer.parseInt(s));
            }
        }
        return stack.peek();
    }
}
