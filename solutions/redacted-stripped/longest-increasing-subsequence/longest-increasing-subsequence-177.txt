class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length + 1];
        for (int i = 0; i < nums.length + 1; i++) {
            dp[i] = 1;
        }
        int count = -1;
        for (int i = 1; i < nums.length; i++) {
            int temp_max = 0;
            for (int j = i - 1; j >= 0; j--) {
                if (nums[i] > nums[j] && dp[j] > temp_max) {
                    temp_max = dp[j];
                }
            }
            dp[i] += temp_max;
        }
        Arrays.sort(dp);
        count = dp[nums.length];
        return count;
    }
}
