class Solution {

    public int lengthOfLIS(int[] nums) {
        int max = 1;
        int n = nums.length;
        int[] arr = new int[n + 1];
        Arrays.fill(arr, 1);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] > nums[j]) {
                    arr[i] = Math.max(arr[i], 1 + arr[j]);
                }
                max = Math.max(max, arr[i]);
            }
        }
        return max;
    }
}
