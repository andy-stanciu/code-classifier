class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] tail = new int[nums.length];
        int size = 0;
        for (int i : nums) {
            int j = 0, k = size;
            while (j != k) {
                int m = (j + k) / 2;
                if (tail[m] < i) {
                    j = m + 1;
                } else {
                    k = m;
                }
            }
            tail[j] = i;
            if (j == size) {
                size++;
            }
        }
        return size;
    }
}
