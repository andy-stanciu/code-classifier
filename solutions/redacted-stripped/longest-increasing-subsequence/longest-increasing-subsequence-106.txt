class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        int maxCount = 0;
        for (int i = 0; i < nums.length; i++) {
            int count = 0;
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    count = Math.max(count, dp[j]);
                }
            }
            dp[i] = count + 1;
            maxCount = Math.max(maxCount, dp[i]);
        }
        return maxCount;
    }
}
