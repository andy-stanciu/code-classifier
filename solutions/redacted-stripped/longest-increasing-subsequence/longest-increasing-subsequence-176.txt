class Solution {

    public int lengthOfLIS(int[] nums) {
        TreeSet<Integer> ts = new TreeSet<>();
        ts.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] < ts.last()) {
                int idx = ts.ceiling(nums[i]);
                ts.remove(idx);
            }
            ts.add(nums[i]);
        }
        return ts.size();
    }
}
