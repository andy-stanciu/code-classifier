class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] var1 = new int[nums.length];
        var1[0] = 1;
        int maxLengthFound = 1;
        for (int curIdx = 1; curIdx < nums.length; curIdx++) {
            int curMax = 0;
            for (int prevIdx = curIdx - 1; prevIdx >= 0; prevIdx--) {
                if (nums[prevIdx] < nums[curIdx]) {
                    curMax = Math.max(var1[prevIdx], curMax);
                }
            }
            var1[curIdx] = curMax + 1;
            maxLengthFound = Math.max(var1[curIdx], maxLengthFound);
        }
        return maxLengthFound;
    }
}
