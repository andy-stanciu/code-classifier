class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] x = new int[n + 1];
        x[1] = nums[0];
        int size = 1;
        for (int i = 1; i < n; ++i) {
            if (nums[i] > x[size]) {
                x[++size] = nums[i];
            } else {
                int left = 1, right = size;
                while (left < right) {
                    int mid = (left + right) >> 1;
                    if (x[mid] >= nums[i]) {
                        right = mid;
                    } else {
                        left = mid + 1;
                    }
                }
                int p = x[left] >= nums[i] ? left : 1;
                x[p] = nums[i];
            }
        }
        return size;
    }
}
