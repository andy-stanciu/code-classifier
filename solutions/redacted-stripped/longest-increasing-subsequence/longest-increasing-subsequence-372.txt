class Solution {

    public int lengthOfLIS(int[] nums) {
        List<Integer> pile = new ArrayList<>();
        for (int num : nums) {
            int index = Collections.binarySearch(pile, num, (a, b) -> a - b);
            if (index < 0)
                index = -(index + 1);
            if (index == pile.size()) {
                pile.add(num);
            } else {
                pile.set(index, num);
            }
        }
        return pile.size();
    }
}
