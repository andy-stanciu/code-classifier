class Solution {

    public int lengthOfLIS(int[] nums) {
        int len = nums.length, pos = 0;
        int[] sizes = new int[len + 1];
        Arrays.fill(sizes, Integer.MIN_VALUE);
        for (int val : nums) {
            int left = 0, right = pos;
            while (left < right) {
                int mid = left + (right - left) / 2;
                if (val > sizes[mid])
                    left = mid + 1;
                else
                    right = mid;
            }
            if (sizes[left] == Integer.MIN_VALUE)
                pos++;
            sizes[left] = val;
        }
        return pos;
    }
}
