class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        List<Integer> dp = new ArrayList<>();
        dp.add(nums[0]);
        for (int i = 1; i < n; i++) {
            int l = 0, r = dp.size() - 1;
            while (l <= r) {
                int m = l + r >> 1;
                if (dp.get(m) < nums[i]) {
                    l = m + 1;
                } else {
                    r = m - 1;
                }
            }
            if (l == dp.size()) {
                dp.add(nums[i]);
            } else {
                dp.set(l, nums[i]);
            }
        }
        return dp.size();
    }
}
