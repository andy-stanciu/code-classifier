class Solution {

    public int lengthOfLIS(int[] a) {
        int n = a.length;
        return help(0, -1, a);
    }

    public static int help(int i, int prev, int[] a) {
        if (i == a.length)
            return 0;
        int notPick = help(i + 1, prev, a);
        int pick = -1;
        if (prev == -1 || a[i] > a[prev]) {
            pick = 1 + help(i + 1, i, a);
        }
        return Math.max(pick, notPick);
    }
}
