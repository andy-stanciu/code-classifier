class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        int size = 0;
        for (int current : nums) {
            int i = 0;
            int j = size;
            while (i != j) {
                int mid = (i + j) / 2;
                if (dp[mid] < current) {
                    i = mid + 1;
                } else {
                    j = mid;
                }
            }
            dp[i] = current;
            if (i == size) {
                size++;
            }
        }
        return size;
    }
}
