class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int max = 0;
        int[] t = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (nums[i] < nums[j] && t[i] >= t[j]) {
                    t[j]++;
                    if (t[j] > max)
                        max = t[j];
                }
            }
        }
        return max + 1;
    }
}
