class Solution {

    public int lengthOfLIS(int[] nums) {
        List<Integer> lis = new ArrayList<>();
        lis.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] > lis.get(lis.size() - 1)) {
                lis.add(nums[i]);
            } else {
                int index = 0;
                while (lis.get(index) < nums[i]) index++;
                lis.set(index, nums[i]);
            }
        }
        return lis.size();
    }
}
