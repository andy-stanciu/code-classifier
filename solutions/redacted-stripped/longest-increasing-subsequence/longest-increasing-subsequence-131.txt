class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] arr = new int[nums.length];
        arr[n - 1] = 1;
        for (int i = n - 2; i >= 0; i--) {
            int curr = 0;
            for (int j = i + 1; j < n; j++) {
                if (nums[j] > nums[i])
                    curr = Math.max(curr, arr[j]);
            }
            arr[i] = curr + 1;
        }
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i] + " ");
            max = Math.max(max, arr[i]);
        }
        return max;
    }
}
