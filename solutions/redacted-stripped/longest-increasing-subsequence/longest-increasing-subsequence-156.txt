class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] lis = new int[nums.length];
        Arrays.fill(lis, 1);
        for (int i = nums.length - 2; i >= 0; i--) {
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[i] < nums[j]) {
                    lis[i] = Math.max(lis[j] + 1, lis[i]);
                }
            }
        }
        int max = 1;
        for (int j = 0; j < nums.length; j++) {
            if (lis[j] > max)
                max = lis[j];
        }
        return max;
    }
}
