class Solution {

    int n;

    int[][] dp;

    public int lengthOfLIS(int[] nums) {
        n = nums.length;
        dp = new int[n][n + 1];
        for (int i = 0; i < n; i++) Arrays.fill(dp[i], -1);
        return f(nums, 0, -1);
    }

    public int f(int[] a, int idx, int preidx) {
        if (idx == n)
            return 0;
        if (dp[idx][preidx + 1] != -1)
            return dp[idx][preidx + 1];
        int len = 0 + f(a, idx + 1, preidx);
        if (preidx == -1 || a[idx] > a[preidx]) {
            len = Math.max(len, 1 + f(a, idx + 1, idx));
        }
        return dp[idx][preidx + 1] = len;
    }
}
