class Solution {

    static void print(Object... objs) {
        for (Object obj : objs) System.out.println(obj);
    }

    public int lengthOfLIS(int[] nums) {
        List<Integer> lis = new ArrayList<>();
        for (int num : nums) {
            int index = Collections.binarySearch(lis, num);
            print(index);
            if (index < 0) {
                index = -index - 1;
                if (index == lis.size()) {
                    lis.add(num);
                } else {
                    lis.set(index, num);
                }
            }
        }
        print(lis);
        return lis.size();
    }
}
