class Solution {

    public int lengthOfLIS(int[] nums) {
        if (nums.length == 1)
            return 1;
        int[] dp = new int[nums.length];
        int max = 0;
        Arrays.fill(dp, 1);
        for (int i = 1; i < nums.length; i++) {
            int curr = nums[i - 1];
            for (int j = i; j < nums.length; j++) {
                if (nums[j] > curr) {
                    dp[j] = Math.max(dp[j], dp[i - 1] + 1);
                    max = Math.max(max, dp[j]);
                }
            }
        }
        return max != 0 ? max : 1;
    }
}
