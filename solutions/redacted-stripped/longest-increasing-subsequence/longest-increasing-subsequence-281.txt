class Solution {

    public int lengthOfLIS(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) {
            Integer k = set.ceiling(num);
            if (k == null)
                set.add(num);
            else {
                set.remove(k);
                set.add(num);
            }
        }
        return set.size();
    }
}
