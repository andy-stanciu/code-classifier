class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length, ans = 0;
        int[] lengthOfLIS = new int[n];
        for (int i = 0; i < n; i++) {
            lengthOfLIS[i] = 1;
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    lengthOfLIS[i] = Math.max(lengthOfLIS[i], lengthOfLIS[j] + 1);
                }
            }
            ans = Math.max(ans, lengthOfLIS[i]);
        }
        return ans;
    }
}
