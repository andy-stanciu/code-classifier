class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        int max = Integer.MIN_VALUE;
        for (int curr = 0; curr < n; curr++) {
            for (int prev = 0; prev < curr; prev++) {
                if (nums[prev] < nums[curr])
                    dp[curr] = Math.max(1 + dp[prev], dp[curr]);
            }
            max = Math.max(dp[curr], max);
        }
        return max;
    }
}
