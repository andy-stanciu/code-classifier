class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        int sum = len1 + len2;
        int[] arr = new int[sum];
        System.arraycopy(nums1, 0, arr, 0, len1);
        System.arraycopy(nums2, 0, arr, len1, len2);
        Arrays.sort(arr);
        if (sum % 2 == 1) {
            return arr[sum / 2];
        } else {
            return (arr[sum / 2 - 1] + arr[sum / 2]) / 2.0;
        }
    }
}
