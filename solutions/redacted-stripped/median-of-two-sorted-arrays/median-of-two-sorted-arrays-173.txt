class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        int i = 0, j = 0, k = 0;
        int[] result = new int[len1 + len2];
        while (i < len1) {
            result[k] = nums1[i];
            i++;
            k++;
        }
        while (j < len2) {
            result[k] = nums2[j];
            j++;
            k++;
        }
        Arrays.sort(result);
        int resLen = result.length;
        if ((resLen % 2) == 0) {
            double sum = result[resLen / 2] + result[(resLen / 2) - 1];
            return (sum / 2.00);
        }
        return result[resLen / 2];
    }
}
