class Solution {

    public double method1(int[] nums1, int[] nums2) {
        List<Integer> ans = new ArrayList<>();
        for (int a : nums1) {
            ans.add(a);
        }
        for (int b : nums2) {
            ans.add(b);
        }
        Collections.sort(ans);
        int n = ans.size();
        if (n % 2 != 0) {
            return ans.get(((n + 1) / 2) - 1);
        } else if (n % 2 == 0) {
            double a = (ans.get((n / 2) - 1) + ans.get(n / 2)) / 2.0;
            return a;
        }
        return 0;
    }
}
