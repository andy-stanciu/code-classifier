class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        int[] nums = new int[m + n];
        System.arraycopy(nums1, 0, nums, 0, m);
        System.arraycopy(nums2, 0, nums, m, n);
        Arrays.sort(nums);
        int a = nums.length;
        if (a % 2 == 0)
            return (nums[(a / 2) - 1] + nums[a / 2]) / 2.0;
        else
            return nums[a / 2];
    }
}
