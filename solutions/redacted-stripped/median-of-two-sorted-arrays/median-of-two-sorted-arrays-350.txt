class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int a1 = nums1.length;
        int b1 = nums2.length;
        int c1 = a1 + b1;
        double v = 0;
        if (c1 != 0) {
            int[] margedarray = new int[c1];
            System.arraycopy(nums1, 0, margedarray, 0, a1);
            System.arraycopy(nums2, 0, margedarray, a1, b1);
            Arrays.sort(margedarray);
            if (c1 % 2 != 0) {
                int z = c1 / 2;
                v = margedarray[z];
                System.out.println(v);
                return v;
            } else {
                int a = (c1 - 1) / 2;
                double vn = (margedarray[a] + margedarray[a + 1]) / 2.0;
                return vn;
            }
        }
        return v;
    }
}
