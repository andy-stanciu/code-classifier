class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        int len = m + n;
        int[] arr = new int[len];
        int i = 0, j = 0, k = 0;
        while (k < len / 2 + 1) {
            if (i == m) {
                arr[k++] = nums2[j++];
            } else if (j == n) {
                arr[k++] = nums1[i++];
            } else {
                if (nums1[i] < nums2[j]) {
                    arr[k++] = nums1[i++];
                } else {
                    arr[k++] = nums2[j++];
                }
            }
        }
        return len % 2 == 0 ? (float) (arr[k - 1] + arr[k - 2]) / 2 : (float) arr[k - 1];
    }
}
