class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        long[] res = new long[n + m];
        int i = 0, j = 0, k = 0;
        while (i < m && j < n) {
            if (nums1[i] < nums2[j]) {
                res[k] = nums1[i];
                i++;
                k++;
            } else {
                res[k] = nums2[j];
                j++;
                k++;
            }
        }
        while (i < m) {
            res[k] = nums1[i];
            i++;
            k++;
        }
        while (j < n) {
            res[k] = nums2[j];
            j++;
            k++;
        }
        if ((m + n) % 2 != 0)
            return res[(m + n) / 2];
        else {
            double ans = res[(m + n) / 2] + res[(m + n) / 2 - 1];
            return (ans / 2);
        }
    }
}
