        int length2 = nums2.length;
        int[] answer = new int[length1 + length2];

        int i = 0, j = 0;
        for (i = 0, j = 0; i < length1 && j < length2; ) {
            if (nums1[i] < nums2[j])
                answer[index++] = nums1[i++];
            else
                answer[index++] = nums2[j++];
        }
        while (i < length1) {
            answer[index++] = nums1[i++];
        }
        while (j < length2) {
            answer[index++] = nums2[j++];
        }

            int indexM = (length1 + length2) / 2;
            result = (answer[indexM - 1] + answer[indexM]);
