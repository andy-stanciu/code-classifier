class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int i = 0, j = 0, k = 0;
        int[] arr = new int[nums1.length + nums2.length];
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] <= nums2[j])
                arr[k++] = nums1[i++];
            else
                arr[k++] = nums2[j++];
        }
        while (i < nums1.length) arr[k++] = nums1[i++];
        while (j < nums2.length) arr[k++] = nums2[j++];
        int mid1 = 0, mid2 = 0;
        if (arr.length % 2 == 0) {
            mid1 = arr.length / 2;
            mid2 = mid1 - 1;
            return (double) (arr[mid1] + arr[mid2]) / 2;
        } else {
            mid1 = arr.length / 2;
            return (double) arr[mid1];
        }
    }
}
