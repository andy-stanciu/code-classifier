class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        int i = 0, j = 0, k = 0;
        int[] tempArray = new int[m + n];
        while (i < m && j < n) {
            if (nums1[i] < nums2[j]) {
                tempArray[k++] = nums1[i++];
            } else {
                tempArray[k++] = nums2[j++];
            }
        }
        while (i < m) {
            tempArray[k++] = nums1[i++];
        }
        while (j < n) {
            tempArray[k++] = nums2[j++];
        }
        int size = m + n;
        if (size % 2 == 1) {
            return tempArray[size / 2];
        }
        return (double) (tempArray[size / 2] + tempArray[(size / 2) - 1]) / 2;
    }
}
