class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        int[] arr = new int[n1 + n2];
        int j = 0;
        for (int i = 0; i < n1; i++) {
            arr[j] = nums1[i];
            j++;
        }
        for (int i = 0; i < n2; i++) {
            arr[j] = nums2[i];
            j++;
        }
        java.util.Arrays.sort(arr);
        int size = arr.length;
        int mid = 0;
        if (size % 2 != 0) {
            return arr[size / 2];
        } else {
            return ((arr[(size - 1) / 2] + arr[size / 2]) / 2.0);
        }
    }
}
