class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            int idx = i;
            int down = 2 * (numRows - i - 1);
            int up = 2 * (i);
            boolean godown = true;
            while (idx < s.length()) {
                sb.append(s.charAt(idx));
                if (i == 0) {
                    idx += down;
                } else if (i == numRows - 1) {
                    idx += up;
                } else {
                    if (godown) {
                        idx += down;
                    } else {
                        idx += up;
                    }
                    godown = !godown;
                }
            }
        }
        return sb.toString();
    }
}
