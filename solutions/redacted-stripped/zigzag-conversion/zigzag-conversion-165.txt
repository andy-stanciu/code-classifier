class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        StringBuilder convertedStr = new StringBuilder();
        int i, j, k;
        for (i = 1; i <= numRows; i++) {
            j = i - 1;
            k = 0;
            while (j < s.length()) {
                convertedStr.append(s.charAt(j));
                k++;
                if (k * (numRows - 1) - j == 0)
                    k++;
                j += 2 * (k * (numRows - 1) - j);
            }
        }
        return String.valueOf(convertedStr);
    }
}
