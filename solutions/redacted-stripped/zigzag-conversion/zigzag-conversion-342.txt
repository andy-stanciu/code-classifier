class Solution {

    public String convert(String s, int numRows) {
        int length = s.length();
        if (length == numRows || numRows == 1) {
            return s;
        }
        char[] ans = new char[length];
        int count = 0;
        for (int i = 0; i < numRows; i++) {
            int j = i;
            int k = 0;
            int m = 2 * (numRows - 1);
            while (j < length) {
                ans[count++] = s.charAt(j);
                if (i == 0 || ((i + j) % m) != 0) {
                    k = 2 * (numRows - (i + 1));
                } else {
                    k = 2 * i;
                }
                j = j + k;
            }
        }
        return new String(ans);
    }
}
