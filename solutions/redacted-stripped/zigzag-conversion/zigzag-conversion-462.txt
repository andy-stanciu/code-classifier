class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1 || numRows >= s.length())
            return s;
        StringBuilder[] var1 = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            var1[i] = new StringBuilder();
        }
        int i = 1;
        int j = 0;
        var1[0].append(s.charAt(0));
        while (i < s.length()) {
            for (int k = 1; k < numRows; k++) {
                if (i < s.length()) {
                    var1[k].append(s.charAt(i));
                    i++;
                }
            }
            for (int k = numRows - 2; k >= 0; k--) {
                if (i < s.length()) {
                    var1[k].append(s.charAt(i));
                    i++;
                }
            }
        }
        StringBuilder sol = new StringBuilder();
        for (StringBuilder si : var1) {
            sol.append(si);
        }
        return sol.toString();
    }
}
