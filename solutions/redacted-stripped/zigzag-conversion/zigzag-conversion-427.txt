class Solution {

    public String convert(String s, int numRows) {
        if (s == null || s.length() <= 2 || numRows <= 1 || s.length() <= numRows)
            return s;
        int startIndex = 0;
        StringBuilder out = new StringBuilder();
        char[] ch = s.toCharArray();
        do {
            int i = startIndex;
            boolean isBtwIndex = true;
            int startBtwIndex = numRows + (numRows - 2);
            while (i < ch.length) {
                out.append(ch[i]);
                if (startIndex == 0 || startIndex == numRows - 1) {
                    i = i + startBtwIndex;
                } else {
                    if (isBtwIndex) {
                        i = i + startBtwIndex - startIndex * 2;
                        isBtwIndex = false;
                    } else {
                        i = i + startIndex * 2;
                        isBtwIndex = true;
                    }
                }
            }
            startIndex++;
        } while (startIndex < numRows);
        return out.toString();
    }
}
