class Solution {

    public String convert(String s, int n) {
        String[] ans = new String[n];
        for (int i = 0; i < n; i++) {
            ans[i] = "";
        }
        int i = 0;
        while (i < s.length()) {
            for (int index = 0; index < n && i < s.length(); index++) {
                ans[index] += s.charAt(i++);
            }
            for (int index = n - 2; index > 0 && i < s.length(); index--) {
                ans[index] += s.charAt(i++);
            }
        }
        String res = "";
        for (String str : ans) {
            res += str;
        }
        return res;
    }
}
