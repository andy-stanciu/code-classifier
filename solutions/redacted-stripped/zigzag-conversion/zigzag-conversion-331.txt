class Solution {

    public String convert(String s, int numRows) {
        int i = 0;
        int cal = -1;
        int length = s.length();
        List<String> rsList = new ArrayList<>(Collections.nCopies(numRows, ""));
        String result = "";
        if (numRows >= length || numRows == 1) {
            return s;
        }
        for (char v : s.toCharArray()) {
            rsList.set(i, rsList.get(i) + String.valueOf(v));
            if (i == 0 || i == numRows - 1) {
                cal *= -1;
            }
            i += cal;
        }
        for (int j = 0; j <= numRows - 1; j++) {
            result += rsList.get(j);
        }
        return result;
    }
}
