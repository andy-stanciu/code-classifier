class Solution {

    public String convert(String s, int numRows) {
        if (numRows <= 1)
            return s;
        StringBuilder result = new StringBuilder();
        for (int row = 0; row < numRows; row++) {
            int i = row;
            boolean even = true;
            while (i < s.length()) {
                result.append(s.charAt(i));
                int evenV = 2 * (numRows - row - 1);
                int oddV = 2 * row;
                int add = even ? evenV : oddV;
                if (add == 0)
                    add = !even ? evenV : oddV;
                i += add;
                even = !even;
            }
        }
        return result.toString();
    }
}
