class Solution {

    public String convert(String s, int rows) {
        int n = s.length();
        String[] mat = new String[rows];
        for (int i = 0; i < rows; i++) mat[i] = "";
        int k = 0;
        while (k < n) {
            for (int i = 0; k < n && i < rows; i++) mat[i] += s.charAt(k++);
            for (int i = rows - 2; k < n && i > 0; i--) mat[i] += s.charAt(k++);
        }
        String ans = "";
        for (int i = 0; i < rows; i++) {
            ans += mat[i];
        }
        return ans;
    }
}
