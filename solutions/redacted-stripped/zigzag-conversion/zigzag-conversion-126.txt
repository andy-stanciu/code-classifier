class Solution {

    public String convert(String s, int numRows) {
        int m = numRows - 1 + numRows - 2 + 1;
        if (m <= 0) {
            return s;
        }
        StringBuilder sb = new StringBuilder();
        for (int x = 0; x < numRows; x++) {
            for (int i = 0; i < s.length() + m; i += m) {
                if (x == 0) {
                    if (i < s.length())
                        sb.append(s.charAt(i));
                } else if (x == numRows - 1) {
                    if (i + x < s.length())
                        sb.append(s.charAt(i + x));
                } else {
                    if (i - x >= 0 && i - x < s.length())
                        sb.append(s.charAt(i - x));
                    if (i + x < s.length())
                        sb.append(s.charAt(i + x));
                }
            }
        }
        return sb.toString();
    }
}
