class Solution {

    public String convert(String s, int numRows) {
        if (numRows >= s.length() || numRows == 1) {
            return s;
        }
        char[] solution = new char[s.length()];
        int index = 0;
        int curr;
        int jump1;
        int jump2;
        for (int i = 0; i < numRows; i++) {
            curr = i;
            while (curr < s.length() && index < s.length()) {
                solution[index++] = s.charAt(curr);
                jump1 = (numRows - i - 1) * 2;
                jump2 = 2 * i;
                if (jump1 == 0 || jump2 == 0) {
                    curr += jump1;
                    curr += jump2;
                } else {
                    curr += jump1;
                    if (curr < s.length() && index < s.length()) {
                        solution[index++] = s.charAt(curr);
                    }
                    curr += jump2;
                }
            }
        }
        return String.valueOf(solution);
    }
}
