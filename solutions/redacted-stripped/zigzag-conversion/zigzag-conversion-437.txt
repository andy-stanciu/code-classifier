class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        StringBuffer[] stringPerRow = new StringBuffer[numRows];
        for (int i = 0; i < numRows; i++) {
            stringPerRow[i] = new StringBuffer();
        }
        int currentRow = 0;
        boolean goingUp = false;
        for (int i = 0; i < s.length(); i++) {
            stringPerRow[currentRow] = stringPerRow[currentRow].append(s.charAt(i));
            if (currentRow == 0)
                goingUp = false;
            if (currentRow == numRows - 1) {
                goingUp = true;
                currentRow--;
                continue;
            }
            if (goingUp)
                currentRow--;
            else
                currentRow++;
        }
        for (int i = 1; i < numRows; i++) {
            stringPerRow[i] = stringPerRow[i - 1].append(stringPerRow[i]);
        }
        return stringPerRow[numRows - 1].toString();
    }
}
