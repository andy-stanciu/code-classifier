class Solution {

    public String convert(String s, int n) {
        ArrayList<Character>[] a = new ArrayList[n];
        int i = 0;
        String news = "";
        for (i = 0; i < n; i++) {
            a[i] = new ArrayList<>();
        }
        i = 0;
        while (true) {
            for (int j = 0; j < n; j++) {
                if (i < s.length())
                    a[j].add(s.charAt(i++));
            }
            for (int j = n - 2; j > 0; j--) {
                if (i < s.length())
                    a[j].add(s.charAt(i++));
            }
            if (i >= s.length())
                break;
        }
        for (i = 0; i < n; i++) {
            for (int j = 0; j < a[i].size(); j++) news += a[i].get(j);
        }
        return news;
    }
}
