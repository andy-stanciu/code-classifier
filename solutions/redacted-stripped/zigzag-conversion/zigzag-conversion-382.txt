class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        StringBuilder answer = new StringBuilder();
        HashMap<Integer, List<Character>> h = new HashMap<>();
        for (int i = 0; i < numRows; i++) {
            h.put(i, new ArrayList<Character>());
        }
        int i = 0;
        int j = 0;
        while (i < s.length()) {
            while (j < numRows && i < s.length()) {
                h.get(j).add(s.charAt(i));
                j++;
                i++;
            }
            j--;
            j--;
            while (j >= 1 && i < s.length()) {
                h.get(j).add(s.charAt(i));
                j--;
                i++;
            }
        }
        for (List<Character> g : h.values()) {
            for (char c : g) {
                answer.append(c);
            }
        }
        return answer.toString();
    }
}
