class Solution {

    public String convert(String s, int n) {
        if (n == 1 || n >= s.length())
            return s;
        StringBuilder p = new StringBuilder();
        int j = 0;
        for (int i = 0; i < n; i++) {
            j = i;
            int diagdiff = n + (n - 2) - i;
            while (j < s.length()) {
                p.append(s.charAt(j));
                if (i != 0 && i != n - 1) {
                    if (diagdiff < s.length())
                        p.append(s.charAt(diagdiff));
                    diagdiff = diagdiff + n + n - 2;
                }
                j = j + n + (n - 2);
            }
        }
        return p.toString();
    }
}
