class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        StringBuilder result = new StringBuilder();
        int length = s.length();
        int totalDistance = 2 * (numRows - 1);
        for (int row = 0; row < numRows; row++) {
            int step = totalDistance - 2 * row;
            for (int index = row; index < length; index += totalDistance) {
                result.append(s.charAt(index));
                if (row > 0 && row < numRows - 1 && step < length) {
                    int middleIndex = index + step;
                    if (middleIndex < length) {
                        result.append(s.charAt(middleIndex));
                    }
                }
            }
        }
        return result.toString();
    }
}
