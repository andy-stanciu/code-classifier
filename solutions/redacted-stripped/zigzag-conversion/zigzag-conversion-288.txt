class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        StringBuilder convertedResult = new StringBuilder();
        int n = s.length(), charInSection = (numRows - 1) * 2;
        for (int currRow = 0; currRow < numRows; ++currRow) {
            int index = currRow;
            while (index < n) {
                convertedResult.append(s.charAt(index));
                if (currRow != 0 && currRow != numRows - 1) {
                    int charInBetween = charInSection - (currRow * 2);
                    if (index + charInBetween < n)
                        convertedResult.append(s.charAt(index + charInBetween));
                }
                index += charInSection;
            }
        }
        return convertedResult.toString();
    }
}
