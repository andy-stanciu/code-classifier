class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        StringBuilder res = new StringBuilder();
        int n = s.length();
        for (int row = 0; row < numRows; row++) {
            int step1 = 2 * (numRows - 1 - row);
            int step2 = 2 * row;
            int pos = row;
            while (pos < n) {
                if (step1 > 0) {
                    res.append(s.charAt(pos));
                    pos += step1;
                }
                if (pos >= n)
                    break;
                if (step2 > 0) {
                    res.append(s.charAt(pos));
                    pos += step2;
                }
            }
        }
        return res.toString();
    }
}
