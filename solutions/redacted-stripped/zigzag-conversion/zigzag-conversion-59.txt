class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1 || numRows >= s.length())
            return s;
        StringBuilder[] var1 = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            var1[i] = new StringBuilder();
        }
        int j = 0;
        boolean x = false;
        for (int i = 0; i < s.length(); i++) {
            if (j == 0)
                x = true;
            if (j == numRows - 1)
                x = false;
            var1[j].append(s.charAt(i));
            if (x == true)
                j++;
            else
                j--;
        }
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            ans.append(var1[i]);
        }
        return ans.toString();
    }
}
