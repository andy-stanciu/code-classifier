class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        String result = "";
        int size = s.length();
        int bigGap = 2 * (numRows - 1);
        int gap = 0;
        int index = 0;
        for (int rowIndex = 0; rowIndex < numRows; rowIndex++) {
            gap = bigGap - 2 * rowIndex;
            if (rowIndex == numRows - 1)
                gap = bigGap;
            index = rowIndex;
            while (index < size) {
                result += s.charAt(index);
                index += gap;
                if (gap != bigGap)
                    gap = bigGap - gap;
            }
        }
        return result;
    }
}
