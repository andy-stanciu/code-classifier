class Solution {

    public String convert(String s, int numRows) {
        String str = "";
        int n = numRows - 1;
        if (s.length() - 1 == 0 || n == 0) {
            return s;
        }
        for (int i = 0; i < numRows; i++) {
            int x = i;
            if (i == 0 || i == numRows - 1) {
                n = numRows - 1;
                while (x <= s.length() - 1) {
                    str += s.charAt(x);
                    x += n * 2;
                }
            } else {
                n -= 1;
                while (x <= s.length() - 1) {
                    str += s.charAt(x);
                    x += n * 2;
                    if (x <= s.length() - 1) {
                        str += s.charAt(x);
                        x += i * 2;
                    }
                }
            }
        }
        return str;
    }
}
