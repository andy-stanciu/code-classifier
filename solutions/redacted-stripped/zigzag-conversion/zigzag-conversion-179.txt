class Solution {

    public String convert(String s, int numRows) {
        if (numRows <= 1 || numRows >= s.length()) {
            return s;
        }
        int idx = 0, d = -1;
        String[] rows = new String[numRows];
        for (int i = 0; i < numRows; i++) {
            rows[i] = "";
        }
        for (int i = 0; i < s.length(); i++) {
            rows[idx] += s.charAt(i);
            if (idx == 0 || idx == (numRows - 1)) {
                d *= -1;
            }
            idx += d;
        }
        return String.join("", rows);
    }
}
