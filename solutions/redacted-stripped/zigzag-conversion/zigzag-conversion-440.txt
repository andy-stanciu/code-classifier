class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        String[] string = new String[numRows];
        for (int index = 0; index < numRows; index++) string[index] = "";
        boolean positiveIncrement = true;
        int counter = 0;
        for (int index = 0; index < s.length(); index++) {
            char character = s.charAt(index);
            string[counter] = string[counter] + character;
            if (positiveIncrement)
                counter++;
            else
                counter--;
            if (counter == numRows) {
                counter--;
                counter--;
                positiveIncrement = false;
            } else if (counter < 0) {
                counter = 0;
                counter++;
                positiveIncrement = true;
            }
        }
        s = "";
        for (String strings : string) s += strings;
        return s;
    }
}
