class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        List<StringBuilder> map = new ArrayList<>();
        for (int i = 0; i < numRows; i++) map.add(new StringBuilder());
        int index = 0;
        int r = 0;
        while (index < s.length()) {
            while (index < s.length() && r < numRows) map.get(r++).append(s.charAt(index++));
            r = r - 2;
            while (index < s.length() && r >= 0) map.get(r--).append(s.charAt(index++));
            r = r + 2;
        }
        StringBuilder ret = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            ret.append(map.get(i));
        }
        return ret.toString();
    }
}
