class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1 || numRows > s.length()) {
            return s;
        }
        StringBuilder ans = new StringBuilder();
        int i = 0, j = 0, itr = 0, sum1 = (2 * numRows - 2), sum2 = 0;
        while (j < numRows) {
            if (i >= s.length()) {
                itr = 0;
                i = ++j;
                sum1 -= 2;
                sum2 += 2;
                if (j == numRows) {
                    continue;
                }
            }
            ans.append(s.charAt(i));
            if (sum1 == 0 || sum2 == 0) {
                i += (sum1 > sum2) ? sum1 : sum2;
                continue;
            }
            if (itr % 2 == 0) {
                i += sum1;
            } else {
                i += sum2;
            }
            itr++;
        }
        return ans.toString();
    }
}
