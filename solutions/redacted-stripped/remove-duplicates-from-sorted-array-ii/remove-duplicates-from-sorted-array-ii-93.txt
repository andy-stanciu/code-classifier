class Solution {

    public int method1(int[] nums) {
        int res = 0;
        boolean found = false;
        for (int i = 1; i < nums.length; i++) {
            if (found && nums[i - 1] == nums[i])
                continue;
            if (nums[i - 1] == nums[i])
                found = true;
            else
                found = false;
            nums[++res] = nums[i];
        }
        return ++res;
    }
}
