class Solution {

    public int method1(int[] nums) {
        int indexForAns = 0;
        int tempElement = -1;
        int freqOfTemp = 0;
        for (int i = 0; i < nums.length; i++) {
            if (tempElement != nums[i]) {
                tempElement = nums[i];
                freqOfTemp = 1;
            } else {
                freqOfTemp++;
            }
            if (freqOfTemp <= 2) {
                nums[indexForAns++] = nums[i];
            }
        }
        return indexForAns;
    }
}
