class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (nums == null || n == 0)
            return 0;
        int count = 1, k = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] == nums[i - 1]) {
                count++;
            } else {
                count = 1;
            }
            if (count <= 2) {
                nums[k++] = nums[i];
            }
        }
        return k;
    }
}
