class Solution {

    public int method1(int[] nums) {
        int shiftCount = 0;
        int index = 1;
        int count = 1;
        int lastNum = nums[0];
        while (index < nums.length) {
            if (shiftCount > 0) {
                nums[index - shiftCount] = nums[index];
            }
            if (lastNum == nums[index]) {
                count++;
                if (count > 2) {
                    shiftCount++;
                }
            } else {
                count = 1;
            }
            lastNum = nums[index];
            index++;
        }
        return index - shiftCount;
    }
}
