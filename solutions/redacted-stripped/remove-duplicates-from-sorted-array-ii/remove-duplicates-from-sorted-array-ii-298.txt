class Solution {

    public int method1(int[] nums) {
        int j = 1;
        int allowDublicate = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[j] = nums[i];
                allowDublicate = 0;
                j++;
            } else if (nums[i] == nums[i - 1] && allowDublicate < 1) {
                nums[j] = nums[i];
                j++;
                allowDublicate++;
            }
        }
        return j;
    }
}
