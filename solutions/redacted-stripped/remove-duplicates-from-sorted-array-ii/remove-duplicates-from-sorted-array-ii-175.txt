class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int i = 1;
        int index = 1;
        while (i < nums.length) {
            if (nums[i] != nums[i - 1]) {
                count = 0;
                nums[index++] = nums[i];
            } else {
                count++;
                if (count == 1) {
                    nums[index++] = nums[i];
                }
            }
            i++;
        }
        return index;
    }
}
