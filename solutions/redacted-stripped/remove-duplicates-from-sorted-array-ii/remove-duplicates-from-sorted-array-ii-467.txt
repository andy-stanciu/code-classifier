class Solution {

    public int method1(int[] nums) {
        int q = nums[0];
        int count = 1;
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == q) {
                count++;
            } else {
                q = nums[i];
                count = 1;
            }
            if (count <= 2) {
                nums[k++] = nums[i];
            }
        }
        return k;
    }
}
