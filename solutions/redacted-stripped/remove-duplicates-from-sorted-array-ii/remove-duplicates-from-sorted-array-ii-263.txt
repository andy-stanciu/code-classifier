class Solution {

    public int method1(int[] nums) {
        int prevElement = nums[0], prevElemCount = 1, correctPos = 0;
        for (int i = 1; i < nums.length; i++) {
            if (prevElement == nums[i]) {
                prevElemCount++;
            }
            if (prevElement != nums[i]) {
                prevElement = nums[i];
                prevElemCount = 1;
            }
            if (i > correctPos && prevElemCount < 3) {
                correctPos++;
                nums[correctPos] = nums[i];
            }
        }
        return correctPos + 1;
    }
}
