class Solution {

    public int method1(int[] ar) {
        int size = ar.length;
        if (size <= 2)
            return size;
        int i = 0, j = 0;
        while (j < size) {
            if (ar[i] == ar[j]) {
                if (j - i <= 1)
                    j++;
                else {
                    int m = j, n = j + 1;
                    int gap = 1;
                    while (n < size && ar[n] == ar[n - 1]) {
                        gap++;
                        n++;
                    }
                    while (n < size) {
                        ar[m++] = ar[n++];
                    }
                    size -= gap;
                }
            } else {
                i = j;
            }
        }
        return size;
    }
}
