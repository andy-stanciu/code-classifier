class Solution {

    public int method1(int[] nums) {
        int count = 1;
        int occured = 1;
        int index = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                if (occured == 2)
                    continue;
                nums[index] = nums[i];
                count++;
                index++;
                occured++;
            } else {
                nums[index] = nums[i];
                occured = 1;
                index++;
                count++;
            }
        }
        return count;
    }
}
