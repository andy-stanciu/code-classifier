class Solution {

    public int method1(int[] nums) {
        if (nums.length < 3)
            return nums.length;
        int dup = 0;
        int n = nums.length;
        int k = 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] == nums[i - 1]) {
                dup++;
                if (dup > 1)
                    k++;
            } else
                dup = 0;
            nums[i - k] = nums[i];
        }
        return n - k;
    }
}
