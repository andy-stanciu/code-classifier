class Solution {

    public int method1(int[] nums) {
        int index = 2;
        if (nums.length <= 2)
            return nums.length;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[index - 2])
                nums[index++] = nums[i];
        }
        return index;
    }
}
