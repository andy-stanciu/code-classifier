class Solution {

    public int method1(int[] nums) {
        int prev = nums[0], count = 1, idx = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == prev) {
                count++;
            } else {
                count = 1;
                prev = nums[i];
            }
            if (count >= 3) {
                continue;
            }
            nums[idx++] = prev;
        }
        return idx;
    }
}
