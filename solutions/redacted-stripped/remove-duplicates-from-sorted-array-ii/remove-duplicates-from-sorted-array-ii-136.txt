class Solution {

    public int method1(int[] nums) {
        int j = 1;
        int count = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1] && count < 1) {
                count++;
                nums[j] = nums[i];
                j++;
            } else if (nums[i] != nums[i - 1]) {
                count = 0;
                nums[j] = nums[i];
                j++;
            }
        }
        return j;
    }
}
