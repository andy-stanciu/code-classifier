class Solution {

    public int method1(int[] nums) {
        int position = 1;
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1])
                count = 1;
            else
                count++;
            if (count <= 2)
                nums[position++] = nums[i];
        }
        return position;
    }
}
