class Solution {

    public int method1(int[] nums) {
        int index = 1;
        int tempNum = nums[0];
        int counter = 0;
        for (int i = 1; i < nums.length; i++) {
            int currentNum = nums[i];
            if (counter < 1 && currentNum == tempNum) {
                nums[index] = currentNum;
                counter++;
                index++;
            } else if (counter >= 1 && currentNum == tempNum) {
                counter++;
            } else if (currentNum != tempNum) {
                counter = 0;
                nums[index] = currentNum;
                tempNum = currentNum;
                index++;
            }
        }
        return index;
    }
}
