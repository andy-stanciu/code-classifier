class Solution {

    public int method1(int[] nums) {
        int currInd = 0;
        int fx = 0;
        int sl = 0;
        while (sl < nums.length) {
            if (nums[fx] != nums[sl])
                fx = sl;
            if (sl - fx <= 1 && nums[sl] == nums[fx])
                nums[currInd++] = nums[sl];
            sl++;
        }
        return currInd;
    }
}
