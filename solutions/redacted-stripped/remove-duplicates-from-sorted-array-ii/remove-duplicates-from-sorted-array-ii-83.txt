class Solution {

    public int method1(int[] nums) {
        int no1 = 0, count = 0, index = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[no1] != nums[i]) {
                no1 = i;
                nums[++index] = nums[i];
                count = 1;
            } else if (count < 2) {
                nums[++index] = nums[i];
                count++;
            }
        }
        return index + 1;
    }
}
