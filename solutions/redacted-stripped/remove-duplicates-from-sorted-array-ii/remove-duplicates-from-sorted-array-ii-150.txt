class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int current = nums[0] + 1;
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != current) {
                count = 1;
                nums[index] = nums[i];
                index++;
                current = nums[i];
            } else if (nums[i] == current && count == 1) {
                count++;
                nums[index] = nums[i];
                index++;
            }
        }
        return index;
    }
}
