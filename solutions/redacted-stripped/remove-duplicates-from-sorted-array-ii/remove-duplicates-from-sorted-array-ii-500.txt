class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int count = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i]) {
                i++;
                nums[i] = nums[j];
                count = 0;
            } else {
                count++;
                if (count < 2) {
                    i++;
                    nums[i] = nums[j];
                }
            }
        }
        return i + 1;
    }
}
