class Solution {

    public int method1(int[] nums) {
        int idx = 1;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[idx] || nums[i] != nums[idx - 1]) {
                idx++;
                nums[idx] = nums[i];
            }
        }
        return idx + 1;
    }
}
