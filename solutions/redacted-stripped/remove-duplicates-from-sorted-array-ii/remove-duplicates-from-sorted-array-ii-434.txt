class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int isp = 2;
        for (int i = 2; i < n; i++) {
            if (nums[i] != nums[isp - 2]) {
                nums[isp] = nums[i];
                isp++;
            }
        }
        return isp++;
    }
}
