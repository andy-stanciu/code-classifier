class Solution {

    public int method1(int[] nums) {
        int iterator = 0;
        boolean ok = false;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[iterator]) {
                nums[++iterator] = nums[i];
                ok = false;
            } else {
                if (!ok) {
                    nums[++iterator] = nums[i];
                    ok = true;
                }
            }
        }
        return (iterator + 1);
    }
}
