class Solution {

    public int method1(int[] arr) {
        int n = arr.length;
        int res = 1;
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (arr[res - 1] != arr[i]) {
                arr[res++] = arr[i];
                count = 1;
            } else if (count < 2) {
                arr[res++] = arr[i];
                count++;
            }
        }
        return res;
    }
}
