class Solution {

    public int method1(int[] nums) {
        int p1 = 0;
        int factor = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i - 1] == nums[i]) {
                if (factor++ < 2) {
                    p1++;
                }
            } else {
                factor = 1;
                p1++;
            }
            nums[p1] = nums[i];
        }
        return p1 + 1;
    }
}
