class Solution {

    public int method1(int[] arr) {
        int index = 0, count = 1;
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] == arr[index] && count < 2) {
                count++;
                index++;
                arr[index] = arr[i];
            } else if (arr[i] != arr[index]) {
                count = 1;
                index++;
                arr[index] = arr[i];
            }
        }
        return index + 1;
    }
}
