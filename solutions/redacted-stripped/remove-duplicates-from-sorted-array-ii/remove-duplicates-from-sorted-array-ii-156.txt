class Solution {

    public int method1(int[] nums) {
        int up = 2;
        int length = nums.length;
        int prevprev = nums[0];
        for (int i = 2; i < length; i++) {
            int currentprev = nums[i - 1];
            if (nums[i] != nums[i - 1] || nums[i] != prevprev) {
                nums[up++] = nums[i];
            }
            prevprev = currentprev;
        }
        return up;
    }
}
