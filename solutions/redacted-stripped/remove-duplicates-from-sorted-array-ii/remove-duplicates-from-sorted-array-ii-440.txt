class Solution {

    public int method1(int[] nums) {
        int k = 1;
        int i = 1;
        int flag = 1;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i - 1]) {
                flag = 1;
                nums[i] = nums[j];
                i++;
                k++;
            } else if (flag < 2) {
                flag = 2;
                nums[i] = nums[j];
                i++;
                k++;
            }
        }
        return k;
    }
}
