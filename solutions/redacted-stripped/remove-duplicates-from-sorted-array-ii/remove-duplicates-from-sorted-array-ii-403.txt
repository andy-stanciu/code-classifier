class Solution {

    public int method1(int[] nums) {
        int pointer = 1;
        int uniqueCount = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                uniqueCount++;
            } else {
                if (nums[i] != nums[i - 1]) {
                    uniqueCount = 1;
                }
            }
            if (uniqueCount <= 2) {
                nums[pointer] = nums[i];
                pointer++;
            }
        }
        return pointer;
    }
}
