class Solution {

    public int method1(int[] nums) {
        if (nums.length <= 2) {
            return nums.length;
        }
        int k = 2;
        for (int i = k; i < nums.length; i++) {
            if (nums[i] != nums[i - 2]) {
                nums[k] = nums[i];
                k++;
            } else if (nums[k] > nums[k - 2] && k < i) {
                k++;
            }
        }
        return k;
    }
}
