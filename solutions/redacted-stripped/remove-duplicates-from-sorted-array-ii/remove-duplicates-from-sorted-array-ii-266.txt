class Solution {

    public int method1(int[] nums) {
        int count = 1;
        int result = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[result] = nums[i];
                result++;
                count = 1;
            } else if (count < 2) {
                nums[result] = nums[i];
                count++;
                result++;
            }
        }
        return result;
    }
}
