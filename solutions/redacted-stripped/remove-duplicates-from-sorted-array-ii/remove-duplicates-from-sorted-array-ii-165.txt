class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int per = 1;
        for (int j = 1; j < nums.length; j++) {
            if (nums[i] == nums[j] && per < 2) {
                i++;
                nums[i] = nums[j];
                per++;
            } else if (nums[i] != nums[j]) {
                i++;
                nums[i] = nums[j];
                per = 1;
            }
        }
        return i + 1;
    }
}
