class Solution {

    public int method1(int[] nums) {
        int o = 1, in = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                o++;
            } else {
                o = 1;
            }
            if (o <= 2) {
                nums[in] = nums[i];
                in++;
            }
        }
        return in;
    }
}
