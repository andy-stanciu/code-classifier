class Solution {

    public int method1(int[] nums) {
        if (nums.length > 2) {
            int index = 1;
            for (int i = 2; i < nums.length; i++) {
                if (nums[index - 1] == nums[i]) {
                    continue;
                } else {
                    index++;
                }
                nums[index] = nums[i];
            }
            return index + 1;
        } else {
            return nums.length;
        }
    }
}
