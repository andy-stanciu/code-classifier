class Solution {

    public int method1(int[] nums) {
        int checkedPosition = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[checkedPosition - 2]) {
                nums[checkedPosition] = nums[i];
                checkedPosition++;
            }
        }
        return checkedPosition;
    }
}
