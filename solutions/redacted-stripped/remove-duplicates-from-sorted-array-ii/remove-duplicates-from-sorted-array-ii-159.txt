class Solution {

    public int method1(int[] nums) {
        int c = 0;
        int de = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                c++;
                if (c > 1) {
                    nums[i] = Integer.MAX_VALUE;
                    de++;
                }
            } else {
                c = 0;
            }
        }
        Arrays.sort(nums);
        return (nums.length - de);
    }
}
