class Solution {

    public int method1(int[] nums) {
        int k = 1;
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                if (count < 2) {
                    nums[k] = nums[i];
                    k++;
                }
                count++;
            } else {
                nums[k] = nums[i];
                k++;
                count = 1;
            }
        }
        return k;
    }
}
