class Solution {

    public int method1(int[] nums) {
        if (nums.length < 3) {
            return nums.length;
        }
        int len = nums.length;
        int count = 2;
        for (int i = 2; i < len; i++) {
            if (nums[i] != nums[count - 2]) {
                nums[count++] = nums[i];
            }
        }
        return count;
    }
}
