class Solution {

    public int method1(int[] nums) {
        if (nums.length < 2) {
            return nums.length;
        }
        int up = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[up - 2]) {
                nums[up] = nums[i];
                up++;
            }
        }
        return up;
    }
}
