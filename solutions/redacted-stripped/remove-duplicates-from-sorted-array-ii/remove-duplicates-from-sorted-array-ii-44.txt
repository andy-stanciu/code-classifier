class Solution {

    public int method1(int[] nums) {
        int index = 1;
        int freq = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1])
                freq++;
            else
                freq = 0;
            if (freq <= 1) {
                nums[index++] = nums[i];
            }
        }
        return index;
    }
}
