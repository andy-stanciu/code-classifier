class Solution {

    public int method1(int[] nums) {
        if (nums.length < 3) {
            return nums.length;
        }
        int firstOpen = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[firstOpen - 1] || nums[i] != nums[firstOpen - 2]) {
                nums[firstOpen++] = nums[i];
            }
        }
        return firstOpen;
    }
}
