class Solution {

    public int method1(int[] nums) {
        int j = 0;
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                count++;
                if (count == 1 || count == 2) {
                    nums[++j] = nums[i];
                }
            } else {
                count = 1;
                nums[++j] = nums[i];
            }
        }
        return j + 1;
    }
}
