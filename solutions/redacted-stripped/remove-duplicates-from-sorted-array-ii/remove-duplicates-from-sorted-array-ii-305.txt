class Solution {

    public int method1(int[] nums) {
        int k = 0;
        int n = nums.length;
        for (int i = 0; i < nums.length; ) {
            int temp = nums[i];
            nums[k++] = nums[i];
            if (i + 1 < n && nums[i + 1] == nums[i])
                nums[k++] = nums[i];
            while (i < n && temp == nums[i]) i++;
        }
        return k;
    }
}
