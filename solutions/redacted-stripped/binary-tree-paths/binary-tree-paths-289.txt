class Solution {

    List<String> ans = new ArrayList<>();

    public void solve(TreeNode root, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            str = str + Integer.toString(root.val);
            ans.add(str);
            return;
        }
        str = str + Integer.toString(root.val) + "->";
        solve(root.left, str);
        solve(root.right, str);
    }

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return ans;
        }
        solve(root, "");
        return ans;
    }
}
