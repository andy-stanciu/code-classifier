class Solution {

    public void method1(TreeNode root, StringBuilder sb, ArrayList<String> res) {
        int length = sb.length();
        if (root == null) {
            return;
        }
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            res.add(sb.toString());
        }
        sb.append("->");
        method1(root.left, sb, res);
        method1(root.right, sb, res);
        sb.setLength(length);
    }

    public List<String> method2(TreeNode root) {
        ArrayList<String> res = new ArrayList<>();
        method1(root, new StringBuilder(), res);
        return res;
    }
}
