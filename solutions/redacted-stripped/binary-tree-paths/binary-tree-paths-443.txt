class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ListOfPath = new ArrayList<>();
        String s = "";
        helper(ListOfPath, root, s);
        return ListOfPath;
    }

    public void helper(List<String> ListOfPath, TreeNode root, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            ListOfPath.add(s + root.val);
        }
        helper(ListOfPath, root.left, s + root.val + "->");
        helper(ListOfPath, root.right, s + root.val + "->");
        return;
    }
}
