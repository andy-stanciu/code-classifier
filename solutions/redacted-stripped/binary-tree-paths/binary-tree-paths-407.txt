class Solution {

    public List<String> result = new ArrayList<>();

    public void method1(TreeNode root, String str) {
        if (root.left == null && root.right == null) {
            result.add(str + root.val);
            str = "";
            return;
        }
        if (root.left != null)
            method1(root.left, str + root.val + "->");
        if (root.right != null)
            method1(root.right, str + root.val + "->");
    }

    public List<String> method2(TreeNode root) {
        method1(root, "");
        return result;
    }
}
