class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<String>();
        method2(list, root, "");
        return list;
    }

    public void method2(List<String> list, TreeNode root, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            list.add(s + root.val);
            return;
        }
        method2(list, root.left, s + root.val + "->");
        method2(list, root.right, s + root.val + "->");
    }
}
