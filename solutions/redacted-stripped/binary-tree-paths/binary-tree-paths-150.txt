class Solution {

    List<String> var1 = new ArrayList<>();

    private void helper(TreeNode root, String path) {
        if (root == null) {
            return;
        }
        path += root.val;
        if (root.left == null && root.right == null) {
            var1.add(path);
            return;
        }
        if (root.left != null) {
            helper(root.left, path + "->");
        }
        if (root.right != null) {
            helper(root.right, path + "->");
        }
    }

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return var1;
        }
        helper(root, "");
        return var1;
    }
}
