class Solution {

    public List<String> method1(TreeNode root) {
        List<String> results = new ArrayList<String>();
        path(root, results, "");
        return results;
    }

    private void path(TreeNode root, List<String> results, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            str += Integer.toString(root.val);
            results.add(str);
            str = "";
            return;
        } else {
            str += Integer.toString(root.val) + "->";
        }
        path(root.left, results, str);
        path(root.right, results, str);
    }
}
