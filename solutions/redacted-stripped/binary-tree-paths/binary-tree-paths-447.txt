class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        dfs(result, "", root);
        return result;
    }

    public void dfs(List<String> result, String temp, TreeNode node) {
        if (node == null)
            return;
        if (node.left == null && node.right == null) {
            result.add(temp + node.val);
            return;
        }
        temp = temp + node.val + "->";
        dfs(result, temp, node.left);
        dfs(result, temp, node.right);
    }
}
