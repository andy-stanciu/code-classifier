class Solution {

    List<String> result = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        StringBuilder sb = new StringBuilder();
        method1(root, sb);
        return result;
    }

    public void method1(TreeNode root, StringBuilder sb) {
        if (root == null)
            return;
        int len = sb.length();
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            result.add(sb.toString());
        } else {
            sb.append("->");
            method1(root.left, sb);
            method1(root.right, sb);
        }
        sb.setLength(len);
    }
}
