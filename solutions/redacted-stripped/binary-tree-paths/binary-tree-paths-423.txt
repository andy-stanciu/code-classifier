class Solution {

    ArrayList<String> ans = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        String path = "";
        getPath(root, path);
        return ans;
    }

    void getPath(TreeNode root, String path) {
        if (root == null)
            return;
        path = path + root.val;
        if (root.left == null && root.right == null) {
            ans.add(path);
            return;
        }
        path = path + "->";
        getPath(root.left, path);
        getPath(root.right, path);
    }
}
