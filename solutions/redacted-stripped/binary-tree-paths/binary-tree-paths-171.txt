class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<String>();
        if (root == null) {
            return result;
        }
        method2(root, "", result);
        return result;
    }

    private void method2(TreeNode root, String path, List<String> result) {
        if (root.left == null && root.right == null) {
            result.add(path + root.val);
        }
        if (root.left != null) {
            method2(root.left, path + root.val + "->", result);
        }
        if (root.right != null) {
            method2(root.right, path + root.val + "->", result);
        }
    }
}
