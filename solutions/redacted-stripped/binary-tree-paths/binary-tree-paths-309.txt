class Solution {

    public List<String> method1(TreeNode root) {
        List<String> li = new ArrayList<>();
        helper(root, li, new StringBuilder());
        return li;
    }

    public void helper(TreeNode root, List<String> li, StringBuilder sb) {
        if (root == null)
            return;
        if (!sb.isEmpty()) {
            sb.append("->");
        }
        sb.append(root.val);
        if (root.right == null && root.left == null) {
            li.add(sb.toString());
            return;
        }
        if (root.left != null)
            helper(root.left, li, new StringBuilder(sb));
        if (root.right != null)
            helper(root.right, li, new StringBuilder(sb));
    }
}
