class Solution {

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        method2(root, "", res);
        return res;
    }

    public void method2(TreeNode root, String path, List<String> res) {
        if (root == null) {
            return;
        }
        path += root.val;
        if (root.left == null && root.right == null) {
            res.add(path);
        } else {
            path += "->";
            method2(root.left, path, res);
            method2(root.right, path, res);
        }
    }
}
