class Solution {

    public List<String> method1(TreeNode root) {
        List<String> arr = new ArrayList<>();
        if (root != null) {
            helper(root, "", arr);
        }
        return arr;
    }

    void helper(TreeNode root, String curr, List<String> arr) {
        if (root == null)
            return;
        if (curr.isEmpty()) {
            curr += root.val;
        } else {
            curr += "->" + root.val;
        }
        if (root.left == null && root.right == null) {
            arr.add(curr);
            return;
        }
        helper(root.left, curr, arr);
        helper(root.right, curr, arr);
    }
}
