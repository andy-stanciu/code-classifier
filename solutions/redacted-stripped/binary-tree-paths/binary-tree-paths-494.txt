class Solution {

    public List<String> method1(TreeNode root) {
        List<String> forReturn = new ArrayList<String>();
        helper(root, "", forReturn);
        return forReturn;
    }

    public void helper(TreeNode root, String str, List<String> forReturn) {
        if (root.left == null && root.right == null)
            forReturn.add(str = str + root.val);
        if (root.left != null)
            helper(root.left, str + root.val + "->", forReturn);
        if (root.right != null)
            helper(root.right, str + root.val + "->", forReturn);
    }
}
