class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        if (root == null)
            return result;
        return method2(root, result, "");
    }

    private List<String> method2(TreeNode root, List<String> result, String path) {
        if (root.left == null && root.right == null)
            result.add(path + root.val);
        if (root.left != null)
            method2(root.left, result, path + root.val + "->");
        if (root.right != null)
            method2(root.right, result, path + root.val + "->");
        return result;
    }
}
