class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        String path = "";
        calpath(root, path, var1);
        return var1;
    }

    public static void calpath(TreeNode root, String path, List<String> var1) {
        if (root == null) {
            return;
        }
        path += root.val;
        if (root.left == null && root.right == null) {
            var1.add(path);
        }
        path += "->";
        calpath(root.left, path, var1);
        calpath(root.right, path, var1);
    }
}
