class Solution {

    public List<String> method1(TreeNode root) {
        ArrayList<String> var1 = new ArrayList<>();
        method1(root, var1, "");
        return var1;
    }

    private void method1(TreeNode root, ArrayList<String> var1, String currentPath) {
        if (root == null) {
            return;
        }
        currentPath = getPath(root, currentPath);
        if (root.left == null && root.right == null) {
            var1.add(currentPath);
            currentPath = "";
        }
        method1(root.left, var1, currentPath);
        method1(root.right, var1, currentPath);
    }

    private static String getPath(TreeNode root, String currentPath) {
        if (Objects.equals(currentPath, "")) {
            currentPath = String.valueOf(root.val);
        } else {
            currentPath = currentPath + "->" + root.val;
        }
        return currentPath;
    }
}
