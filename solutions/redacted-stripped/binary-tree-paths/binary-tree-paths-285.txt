class Solution {

    List<String> list = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return list;
        }
        helper(root, "");
        return list;
    }

    public void helper(TreeNode root, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            s = s + root.val;
            list.add(s);
        } else {
            s = s + root.val + "->";
        }
        helper(root.left, s);
        helper(root.right, s);
    }
}
