class Solution {

    public List<String> method1(TreeNode root) {
        List<String> l = new ArrayList<String>();
        if (root != null)
            SearchBT(root, "", l);
        return l;
    }

    public void SearchBT(TreeNode root, String path, List<String> l) {
        if (root.left == null && root.right == null)
            l.add(path + root.val);
        if (root.left != null)
            SearchBT(root.left, path + root.val + "->", l);
        if (root.right != null)
            SearchBT(root.right, path + root.val + "->", l);
    }
}
