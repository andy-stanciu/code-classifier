class Solution {

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        recursion(root, res, "");
        return res;
    }

    public void recursion(TreeNode root, List<String> res, String curr) {
        if (root == null) {
            return;
        }
        curr += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            res.add(curr);
            return;
        }
        curr += "->";
        recursion(root.left, res, curr);
        recursion(root.right, res, curr);
    }
}
