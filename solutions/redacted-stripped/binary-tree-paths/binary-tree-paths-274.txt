class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList();
        method2(root, "", result);
        return result;
    }

    public void method2(TreeNode node, String currentString, List<String> result) {
        if (node != null) {
            if (currentString.isEmpty())
                currentString += node.val;
            else
                currentString = currentString + "->" + node.val;
            if (node.left == null && node.right == null)
                result.add(currentString);
            method2(node.left, currentString, result);
            method2(node.right, currentString, result);
        }
    }
}
