class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        if (root == null)
            return var1;
        method2(root, "", var1);
        return var1;
    }

    public void method2(TreeNode root, String currPath, List<String> var1) {
        if (root == null)
            return;
        currPath = currPath + root.val;
        if (root.left == null && root.right == null) {
            var1.add(currPath);
        }
        currPath = currPath + "->";
        method2(root.left, currPath, var1);
        method2(root.right, currPath, var1);
    }
}
