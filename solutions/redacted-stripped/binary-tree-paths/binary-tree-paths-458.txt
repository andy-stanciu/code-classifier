class Solution {

    List<String> list = new ArrayList<>();

    StringBuilder sb = new StringBuilder();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        int tmp = sb.length();
        if (root.left == null && root.right == null) {
            sb.append(root.val);
            list.add(sb.toString());
            sb.delete(tmp, sb.length());
            return;
        } else {
            sb.append(root.val + "->");
        }
        TreeNode leftNode = root.left;
        if (leftNode != null) {
            method1(leftNode);
        }
        TreeNode rightNode = root.right;
        if (rightNode != null) {
            method1(rightNode);
        }
        sb.delete(tmp, sb.length());
        return;
    }

    public List<String> method2(TreeNode root) {
        method1(root);
        return list;
    }
}
