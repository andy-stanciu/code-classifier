class Solution {

    List<String> mc = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        StringBuilder str = new StringBuilder();
        help(root, str);
        return mc;
    }

    public void help(TreeNode root, StringBuilder str) {
        if (root == null) {
            return;
        }
        int len = str.length();
        String help1 = "" + root.val;
        str.append(help1);
        if (root.left == null && root.right == null) {
            mc.add(str.toString());
        } else {
            str.append("->");
        }
        help(root.left, str);
        help(root.right, str);
        str.setLength(len);
    }
}
