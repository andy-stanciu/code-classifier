class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<String>();
        rootToLeaf(root, "", ans);
        return ans;
    }

    private void rootToLeaf(TreeNode root, String curr, List<String> ans) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            curr += root.val;
            ans.add(curr);
            return;
        }
        curr += root.val;
        curr += "->";
        rootToLeaf(root.left, curr, ans);
        rootToLeaf(root.right, curr, ans);
        return;
    }
}
