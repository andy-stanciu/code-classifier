class Solution {

    public List<String> method1(TreeNode root) {
        List<String> idc = new ArrayList<String>();
        help(root, "", idc);
        return idc;
    }

    public void help(TreeNode root, String k, List<String> re) {
        if (root == null)
            return;
        StringBuilder s = new StringBuilder(k);
        if (root.left == null && root.right == null) {
            re.add(s.append(root.val).toString());
        } else {
            s = s.append(root.val).append("->");
        }
        help(root.left, s.toString(), re);
        help(root.right, s.toString(), re);
    }
}
