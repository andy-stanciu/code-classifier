class Solution {

    public List<String> list = new ArrayList<String>();

    public List<String> method1(TreeNode root) {
        method2(root, "");
        return list;
    }

    public void method2(TreeNode root, String str) {
        if (root == null)
            return;
        str += root.val + "->";
        method2(root.left, str);
        method2(root.right, str);
        if (root.left == null && root.right == null) {
            list.add(str.substring(0, str.length() - 2));
            return;
        }
    }
}
