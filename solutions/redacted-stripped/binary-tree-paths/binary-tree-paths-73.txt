class Solution {

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return null;
        }
        ArrayList<String> a = new ArrayList<String>();
        bin(root, "", a);
        return a;
    }

    public void bin(TreeNode root, String curr, List<String> param1) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            param1.add(curr + root.val);
            return;
        }
        bin(root.left, curr + root.val + "->", param1);
        bin(root.right, curr + root.val + "->", param1);
    }
}
