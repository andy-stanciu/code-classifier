class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        if (root == null)
            return ans;
        String temp = "";
        postOrder(root, temp, ans);
        return ans;
    }

    public void postOrder(TreeNode root, String temp, List<String> ans) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            temp += root.val;
            ans.add(temp);
            return;
        }
        temp += root.val + "->";
        if (root.left != null)
            postOrder(root.left, temp, ans);
        if (root.right != null)
            postOrder(root.right, temp, ans);
    }
}
