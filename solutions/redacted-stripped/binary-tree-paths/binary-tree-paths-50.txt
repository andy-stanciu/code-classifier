class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        path(root, "");
        return ans;
    }

    public void path(TreeNode root, String str) {
        if (root.left == null && root.right == null) {
            str += Integer.toString(root.val);
            ans.add(str);
            return;
        }
        String a = Integer.toString(root.val);
        str += a + "->";
        if (root.left != null) {
            path(root.left, str);
        }
        if (root.right != null) {
            path(root.right, str);
        }
    }
}
