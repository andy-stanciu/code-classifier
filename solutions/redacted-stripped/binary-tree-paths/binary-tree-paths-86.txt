class Solution {

    public List<String> method1(TreeNode root) {
        ArrayList<String> arr = new ArrayList<>();
        String str = "";
        func(root, arr, str);
        return arr;
    }

    public void func(TreeNode root, ArrayList<String> arr, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            str = str + root.val;
            arr.add(str);
            str = "";
            return;
        }
        str = str + root.val + "->";
        func(root.left, arr, str);
        func(root.right, arr, str);
    }
}
