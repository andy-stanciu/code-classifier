class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        method2(root, "", result);
        return result;
    }

    public void method2(TreeNode root, String path, List<String> result) {
        if (root == null) {
            return;
        }
        path += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            result.add(path);
        } else {
            path += "->";
            method2(root.left, path, result);
            method2(root.right, path, result);
        }
    }
}
