class Solution {

    public static List<String> method1(TreeNode root) {
        List<String> a = new ArrayList<>();
        helper(root, a, "");
        return a;
    }

    public static void helper(TreeNode root, List<String> h, String currentPath) {
        currentPath += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            h.add(currentPath);
            return;
        }
        currentPath += "->";
        if (root.left != null) {
            helper(root.left, h, currentPath);
        }
        if (root.right != null) {
            helper(root.right, h, currentPath);
        }
    }
}
