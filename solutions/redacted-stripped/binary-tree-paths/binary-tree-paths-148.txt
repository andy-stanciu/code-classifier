class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        btpUtil(root, result, "");
        return result;
    }

    public void btpUtil(TreeNode root, List<String> result, String path) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            path = path == "" ? root.val + "" : path + "->" + root.val;
            result.add(path);
        }
        path = path == "" ? root.val + "" : path + "->" + root.val;
        btpUtil(root.left, result, path);
        btpUtil(root.right, result, path);
    }
}
