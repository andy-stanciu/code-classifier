class Solution {

    public List<String> method1(TreeNode root) {
        List<String> pathList = new ArrayList<>();
        StringBuilder path = new StringBuilder();
        helper(root, pathList, path);
        return pathList;
    }

    public static void helper(TreeNode root, List<String> pathList, StringBuilder path) {
        if (root == null)
            return;
        path.append(root.val);
        if (root.left == null && root.right == null)
            pathList.add(path.toString());
        else {
            path.append("->");
            int curLength = path.length();
            helper(root.left, pathList, path);
            path.setLength(curLength);
            helper(root.right, pathList, path);
            path.setLength(curLength);
        }
    }
}
