class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        method1(root, ans, root.val + "");
        return ans;
    }

    public void method1(TreeNode root, List<String> ans, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            ans.add(str);
            return;
        }
        if (root.left != null)
            method1(root.left, ans, str + "->" + root.left.val);
        if (root.right != null)
            method1(root.right, ans, str + "->" + root.right.val);
    }
}
