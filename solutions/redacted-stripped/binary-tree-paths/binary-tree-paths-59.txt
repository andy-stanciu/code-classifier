class Solution {

    public void preOrder(TreeNode root, List<String> res, StringBuilder sb) {
        if (root == null) {
            return;
        }
        int len = sb.length();
        if (sb.length() > 0) {
            sb.append("->");
        }
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            res.add(sb.toString());
        } else {
            preOrder(root.left, res, sb);
            preOrder(root.right, res, sb);
        }
        sb.setLength(len);
    }

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<String>();
        StringBuilder sb = new StringBuilder();
        preOrder(root, res, sb);
        return res;
    }
}
