class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        sol(root, ans, "");
        return ans;
    }

    public void sol(TreeNode root, List<String> ans, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            s = s + root.val;
            ans.add(s);
            return;
        }
        sol(root.left, ans, s + root.val + "->");
        sol(root.right, ans, s + root.val + "->");
    }
}
