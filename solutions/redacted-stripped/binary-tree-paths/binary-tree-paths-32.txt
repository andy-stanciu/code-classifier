class Solution {

    List<String> ans;

    public List<String> method1(TreeNode root) {
        ans = new LinkedList<>();
        method2(root, "");
        return ans;
    }

    void method2(TreeNode root, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            ans.add(s.concat(String.valueOf(root.val)));
        }
        method2(root.left, s.concat(String.valueOf(root.val).concat("->")));
        method2(root.right, s.concat(String.valueOf(root.val).concat("->")));
    }
}
