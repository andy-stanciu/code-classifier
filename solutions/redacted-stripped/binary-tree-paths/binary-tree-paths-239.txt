class Solution {

    public List<String> method1(TreeNode root) {
        String s = "";
        List<String> result = new ArrayList<>();
        return method2(root, result, s);
    }

    public static List<String> method2(TreeNode root, List<String> result, String s) {
        if (root == null)
            return result;
        if (root.left == null && root.right == null) {
            result.add(s + root.val);
            return result;
        }
        s = s + root.val + "->";
        method2(root.left, result, s);
        method2(root.right, result, s);
        return result;
    }
}
