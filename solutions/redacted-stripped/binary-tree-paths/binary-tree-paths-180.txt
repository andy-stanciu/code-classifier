class Solution {

    public void method1(TreeNode root, String str, List<String> list) {
        if (root == null) {
            return;
        }
        str += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            list.add(str);
            return;
        }
        str += "->";
        method1(root.left, str, list);
        method1(root.right, str, list);
    }

    public List<String> method2(TreeNode root) {
        List<String> ans = new ArrayList<>();
        if (root.left == null && root.right == null) {
            ans.add(Integer.toString(root.val));
            return ans;
        }
        method1(root, "", ans);
        return ans;
    }
}
