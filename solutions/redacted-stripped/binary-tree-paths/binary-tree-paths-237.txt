class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        StringBuilder sb = new StringBuilder();
        recursion(root, sb, result);
        return result;
    }

    private void recursion(TreeNode root, StringBuilder sb, List<String> result) {
        if (root == null) {
            return;
        }
        int originalLength = sb.length();
        if (originalLength > 0) {
            sb.append("->");
        }
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            result.add(sb.toString());
        } else {
            recursion(root.left, sb, result);
            recursion(root.right, sb, result);
        }
        sb.setLength(originalLength);
    }
}
