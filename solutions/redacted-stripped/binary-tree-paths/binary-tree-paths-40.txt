class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        rootToNodePath(root, "", list);
        return list;
    }

    public static void rootToNodePath(TreeNode root, String curr, List<String> ans) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            curr += root.val;
            ans.add(curr);
            return;
        }
        rootToNodePath(root.left, curr + root.val + "->", ans);
        rootToNodePath(root.right, curr + root.val + "->", ans);
    }
}
