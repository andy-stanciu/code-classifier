class Solution {

    public List<String> method1(TreeNode root) {
        List<String> l = new ArrayList<>();
        ino(root, "", l);
        return l;
    }

    public void ino(TreeNode root, String s, List<String> l) {
        if (root == null) {
            return;
        }
        s += root.val + "->";
        if (root.left == null && root.right == null) {
            l.add(s.substring(0, s.length() - 2));
        }
        ino(root.left, s, l);
        ino(root.right, s, l);
    }
}
