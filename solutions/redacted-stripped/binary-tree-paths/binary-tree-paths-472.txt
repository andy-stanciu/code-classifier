class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        String path = "";
        method2(result, path, root);
        return result;
    }

    public void method2(List<String> result, String path, TreeNode root) {
        if (root.left == null && root.right == null) {
            path += root.val;
            result.add(path);
        }
        if (root.left != null) {
            method2(result, path + root.val + "->", root.left);
        }
        if (root.right != null) {
            method2(result, path + root.val + "->", root.right);
        }
    }
}
