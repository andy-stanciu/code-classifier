class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        if (root != null) {
            String currentPath = String.valueOf(root.val);
            if (root.left == null && root.right == null) {
                var1.add(currentPath);
                return var1;
            }
            method2(root.left, var1, new StringBuilder(currentPath));
            method2(root.right, var1, new StringBuilder(currentPath));
        }
        return var1;
    }

    private void method2(TreeNode root, List<String> var1, StringBuilder currentPath) {
        if (root == null) {
            return;
        }
        currentPath.append("->").append(root.val);
        if (root.left == null && root.right == null) {
            var1.add(currentPath.toString());
            return;
        }
        method2(root.left, var1, new StringBuilder(currentPath.toString()));
        method2(root.right, var1, currentPath);
        return;
    }
}
