class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        solve(root, list, "");
        return list;
    }

    public static void solve(TreeNode root, List<String> list, String ans) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            ans += root.val;
            list.add(ans);
            return;
        }
        solve(root.left, list, ans + root.val + "->");
        solve(root.right, list, ans + root.val + "->");
    }
}
