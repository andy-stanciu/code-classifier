class Solution {

    List<String> var1 = new ArrayList<>();

    public void dfs(TreeNode node, String path) {
        if (node == null)
            return;
        path += node.val;
        if (node.left == null && node.right == null) {
            var1.add(path);
        } else
            path += "->";
        dfs(node.left, path);
        dfs(node.right, path);
        path = path.substring(0, path.length());
    }

    public List<String> method1(TreeNode root) {
        dfs(root, "");
        return var1;
    }
}
