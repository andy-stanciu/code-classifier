class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<String>();
        if (root.left == null && root.right == null)
            result.add(root.val + "");
        else
            test("", root, result);
        return result;
    }

    public void test(String path, TreeNode root, List<String> result) {
        String currentPath = path == "" ? root.val + "" : path + "->" + root.val;
        if (root.left == null && root.right == null) {
            result.add(currentPath);
        }
        if (root.left != null) {
            test(currentPath, root.left, result);
        }
        if (root.right != null) {
            test(currentPath, root.right, result);
        }
    }
}
