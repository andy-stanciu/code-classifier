class Solution {

    public List<String> method1(TreeNode root) {
        ArrayList<String> al = new ArrayList<>();
        method2(root, al, "");
        return al;
    }

    public void method2(TreeNode root, List<String> al, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            str += root.val;
            al.add(str);
            return;
        }
        method2(root.left, al, str + root.val + "->");
        method2(root.right, al, str + root.val + "->");
        return;
    }
}
