class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        String path = "";
        method2(root, path, list);
        return list;
    }

    public static void method2(TreeNode root, String path, List<String> List) {
        if (root.left == null && root.right == null) {
            path = path + root.val + "";
            List.add(path);
        }
        if (root.left != null) {
            method2(root.left, path + root.val + "->", List);
        }
        if (root.right != null) {
            method2(root.right, path + root.val + "->", List);
        }
    }
}
