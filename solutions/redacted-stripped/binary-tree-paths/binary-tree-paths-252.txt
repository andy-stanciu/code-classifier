class Solution {

    List<String> results = new ArrayList();

    public List<String> method1(TreeNode root) {
        method2(root, new StringBuilder().append(root.val));
        return results;
    }

    private void method2(TreeNode root, StringBuilder leafPath) {
        if (root.left == null && root.right == null) {
            results.add(leafPath.toString());
            return;
        }
        if (root.left != null)
            method2(root.left, new StringBuilder(leafPath).append("->" + root.left.val));
        if (root.right != null)
            method2(root.right, new StringBuilder(leafPath).append("->" + root.right.val));
    }
}
