class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        List<Integer> l = new ArrayList<>();
        method2(root, result, l);
        return result;
    }

    public void method2(TreeNode node, List<String> result, List<Integer> l) {
        if (node == null) {
            return;
        }
        l.add(node.val);
        if (node.left == null && node.right == null) {
            StringBuilder a = new StringBuilder();
            for (int i = 0; i < l.size() - 1; i++) {
                a.append(l.get(i)).append("->");
            }
            a.append(l.get(l.size() - 1));
            result.add(a.toString());
        }
        method2(node.left, result, l);
        method2(node.right, result, l);
        l.remove(l.size() - 1);
    }
}
