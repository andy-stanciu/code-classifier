class Solution {

    List<String> list;

    public List<String> method1(TreeNode root) {
        list = new ArrayList<>();
        findPath(root, "");
        return list;
    }

    void findPath(TreeNode root, String s) {
        s += root.val;
        if (root.left == null && root.right == null) {
            list.add(s);
            return;
        }
        s += "->";
        if (root.left != null)
            findPath(root.left, s);
        if (root.right != null)
            findPath(root.right, s);
    }
}
