class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        solver(root, ans, "");
        return ans;
    }

    public void solver(TreeNode root, List<String> ans, String temp) {
        if (root == null)
            return;
        temp = temp + root.val + "->";
        if (root.right == null && root.left == null) {
            ans.add(temp.substring(0, temp.length() - 2));
        }
        solver(root.left, ans, temp);
        solver(root.right, ans, temp);
    }
}
