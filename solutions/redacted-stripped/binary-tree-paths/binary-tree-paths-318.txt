class Solution {

    List<String> res = new ArrayList<>();

    String s = "";

    public List<String> method1(TreeNode root) {
        helper(root, s);
        return res;
    }

    private void helper(TreeNode root, String s) {
        if (root == null)
            return;
        s += root.val;
        if (root.left == null && root.right == null) {
            res.add(s);
        } else {
            s += "->";
        }
        helper(root.right, s);
        helper(root.left, s);
    }
}
