class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        String path = "";
        return (findPath(root, result, path));
    }

    public List<String> findPath(TreeNode root, List<String> result, String path) {
        if (root == null) {
            return result;
        } else if (root.left == null && root.right == null) {
            if (path.isEmpty()) {
                path = "" + root.val;
            } else {
                path = path + "->" + root.val;
            }
            result.add(path);
        } else {
            if (path.isEmpty()) {
                path = "" + root.val;
            } else {
                path = path + "->" + root.val;
            }
        }
        result = findPath(root.left, result, path);
        result = findPath(root.right, result, path);
        return result;
    }
}
