class Solution {

    private List<String> ans = new ArrayList<>();

    private StringBuilder sb = new StringBuilder();

    public List<String> method1(TreeNode root) {
        path(root);
        return ans;
    }

    private void path(TreeNode root) {
        if (root == null) {
            return;
        }
        int n = sb.length();
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            ans.add(sb.toString());
        }
        sb.append("->");
        path(root.left);
        path(root.right);
        sb.setLength(n);
    }
}
