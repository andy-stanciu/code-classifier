class Solution {

    public static List<String> ans;

    public List<String> method1(TreeNode root) {
        ans = new ArrayList<>();
        findPath(root, "");
        return ans;
    }

    public static void findPath(TreeNode root, String path) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            ans.add(path + root.val);
            return;
        }
        findPath(root.left, path + root.val + "->");
        findPath(root.right, path + root.val + "->");
    }
}
