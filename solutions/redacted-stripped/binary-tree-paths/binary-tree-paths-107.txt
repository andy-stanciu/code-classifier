class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        method2(root, list, "");
        return list;
    }

    public void method2(TreeNode trav, List<String> list, String ans) {
        if (trav == null)
            return;
        if (trav.left == null && trav.right == null) {
            list.add(ans + trav.val);
            return;
        }
        method2(trav.left, list, ans + trav.val + "->");
        method2(trav.right, list, ans + trav.val + "->");
        return;
    }
}
