class Solution {

    ArrayList<String> path = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        String s = "";
        method2(root, s);
        return path;
    }

    public void method2(TreeNode root, String s) {
        if (root == null)
            return;
        if (root.left == null && root.right == null)
            path.add(s + root.val);
        method2(root.left, s + root.val + "->");
        method2(root.right, s + root.val + "->");
    }
}
