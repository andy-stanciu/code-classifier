class Solution {

    List<String> temp = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        String s = "";
        traversal(root, s);
        return temp;
    }

    public void traversal(TreeNode root, String s) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            s = s + root.val;
            temp.add(s);
        } else
            s = s + root.val + "->";
        traversal(root.left, s);
        traversal(root.right, s);
    }
}
