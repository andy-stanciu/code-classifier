class Solution {

    public List<String> method1(TreeNode root) {
        List<String> output = new ArrayList<>();
        path(root, output, "");
        return output;
    }

    private void path(TreeNode root, List<String> output, String path) {
        path = path + root.val;
        if (root.left == null && root.right == null) {
            output.add(path);
            return;
        }
        if (root.left != null)
            path(root.left, output, path + "->");
        if (root.right != null)
            path(root.right, output, path + "->");
    }
}
