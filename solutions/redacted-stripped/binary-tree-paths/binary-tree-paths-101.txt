class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        if (root == null)
            return ans;
        find(root, "");
        return ans;
    }

    void find(TreeNode root, String path) {
        if (root == null)
            return;
        if (path.length() == 0) {
            path += root.val;
        } else {
            path += "->" + root.val;
        }
        if (root.left == null && root.right == null) {
            ans.add(path);
            return;
        }
        find(root.left, path);
        find(root.right, path);
    }
}
