class Solution {

    public void solve(TreeNode root, List<String> list, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            String s = Integer.toString(root.val);
            str += s;
            list.add(str);
            return;
        }
        String s = Integer.toString(root.val);
        str += s;
        str += "->";
        solve(root.left, list, str);
        solve(root.right, list, str);
    }

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        solve(root, list, "");
        return list;
    }
}
