class Solution {

    List<String> list = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        Path(root, "");
        return list;
    }

    private void Path(TreeNode root, String str) {
        if (root == null)
            return;
        str = str + root.val + "->";
        Path(root.left, str);
        Path(root.right, str);
        if (root.left == null && root.right == null) {
            list.add(str.substring(0, (str.length() - 2)));
            return;
        }
    }
}
