class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        if (root == null)
            return result;
        method2(root, result, "");
        return result;
    }

    public void method2(TreeNode root, List<String> result, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            result.add(new String(str + root.val));
            return;
        }
        if (root.left != null) {
            method2(root.left, result, str + root.val + "->");
        }
        if (root.right != null) {
            method2(root.right, result, str + root.val + "->");
        }
    }
}
