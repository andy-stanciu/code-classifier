class Solution {

    List<String> l = new ArrayList<>();

    public boolean isLeaf(TreeNode node) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            return true;
        return false;
    }

    public void helper(TreeNode root, String s) {
        if (root == null)
            return;
        if (isLeaf(root)) {
            s += "->";
            s += (String.valueOf(root.val));
            l.add(s);
        } else {
            if (s.length() == 0) {
                s += (String.valueOf(root.val));
            } else {
                s += "->";
                s += (String.valueOf(root.val));
            }
        }
        helper(root.left, s);
        helper(root.right, s);
    }

    public List<String> method1(TreeNode root) {
        if (root == null)
            return l;
        if (root.left == null && root.right == null) {
            l.add(String.valueOf(root.val));
            return l;
        }
        String s = "";
        helper(root, s);
        return l;
    }
}
