class Solution {

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        if (root == null)
            return res;
        method2(root, "", res);
        return res;
    }

    private void method2(TreeNode node, String path, List<String> res) {
        if (node == null)
            return;
        path += Integer.toString(node.val);
        if (node.left == null && node.right == null)
            res.add(path);
        else {
            path += "->";
            method2(node.left, path, res);
            method2(node.right, path, res);
        }
    }
}
