class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        var1 = BFS(root);
        return var1;
    }

    public static List<String> BFS(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        Queue<TreeNode> nQueue = new LinkedList<>();
        Queue<String> sQueue = new LinkedList<>();
        if (root == null)
            return var1;
        nQueue.add(root);
        sQueue.add("");
        while (!nQueue.isEmpty()) {
            TreeNode nTemp = nQueue.poll();
            String sTemp = sQueue.poll();
            if (nTemp.left == null && nTemp.right == null)
                var1.add(sTemp + nTemp.val);
            if (nTemp.left != null) {
                nQueue.add(nTemp.left);
                sQueue.add(sTemp + nTemp.val + "->");
            }
            if (nTemp.right != null) {
                nQueue.add(nTemp.right);
                sQueue.add(sTemp + nTemp.val + "->");
            }
        }
        return var1;
    }
}
