class Solution {

    List<String> list = new ArrayList<>();

    public void Traverse(TreeNode root, String str) {
        if (root == null)
            return;
        str = str + root.val;
        if (root.left == null && root.right == null) {
            list.add(str);
            return;
        }
        str = str + "->";
        Traverse(root.left, str);
        Traverse(root.right, str);
    }

    public List<String> method1(TreeNode root) {
        Traverse(root, "");
        return list;
    }
}
