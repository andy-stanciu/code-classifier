class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        if (root == null)
            return ans;
        add(root, "");
        return ans;
    }

    public void add(TreeNode root, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            str += root.val;
            ans.add(str);
            return;
        } else {
            str += root.val;
            if (root.left != null || root.right != null)
                str += "->";
        }
        add(root.left, str);
        add(root.right, str);
        return;
    }
}
