class Solution {

    private final static String TAG_JOINER = "->";

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        Deque<String> var1 = new ArrayDeque<>();
        method2(root, result, var1);
        return result;
    }

    public void method2(TreeNode cur, List<String> result, Deque<String> var1) {
        var1.add(String.valueOf(cur.val));
        if (cur.left == null && cur.right == null) {
            result.add(var1.stream().collect(Collectors.joining(TAG_JOINER)));
            return;
        }
        if (cur.left != null) {
            method2(cur.left, result, var1);
            var1.removeLast();
        }
        if (cur.right != null) {
            method2(cur.right, result, var1);
            var1.removeLast();
        }
    }
}
