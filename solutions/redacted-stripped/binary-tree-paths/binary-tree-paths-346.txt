class Solution {

    public List<String> method1(TreeNode root) {
        List<String> stringList = new ArrayList<>();
        hasPath(root, "", stringList);
        return stringList;
    }

    private static void hasPath(TreeNode root, String s, List<String> stringList) {
        if (root == null)
            return;
        s += root.val + "->";
        if (root.left == null && root.right == null) {
            s = s.substring(0, s.length() - 2);
            stringList.add(s);
            return;
        }
        hasPath(root.left, s, stringList);
        hasPath(root.right, s, stringList);
    }
}
