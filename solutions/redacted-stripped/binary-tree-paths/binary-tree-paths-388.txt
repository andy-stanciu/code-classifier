class Solution {

    List<String> res = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        dfs(root, "");
        return res;
    }

    private void dfs(TreeNode node, String s) {
        if (node == null)
            return;
        if (node.left == null && node.right == null) {
            res.add(s + node.val);
            return;
        }
        dfs(node.left, s + node.val + "->");
        dfs(node.right, s + node.val + "->");
    }
}
