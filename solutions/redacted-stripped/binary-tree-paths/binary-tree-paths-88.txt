class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        path(root, ans, "");
        return ans;
    }

    public void path(TreeNode root, List<String> ans, String path) {
        if (root == null) {
            return;
        }
        path += (path.isEmpty() ? "" : "->") + root.val;
        if (root.left == null && root.right == null) {
            ans.add(path);
        } else {
            path(root.left, ans, path);
            path(root.right, ans, path);
        }
    }
}
