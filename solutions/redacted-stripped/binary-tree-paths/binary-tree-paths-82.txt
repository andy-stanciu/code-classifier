class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        method2(root, result, "");
        return result;
    }

    public void method2(TreeNode root, List<String> result, String stack) {
        if (root == null) {
            return;
        }
        stack += root.val;
        if (root.left == null && root.right == null) {
            result.add(new String(stack));
            return;
        }
        stack += "->";
        method2(root.left, result, stack);
        method2(root.right, result, stack);
    }
}
