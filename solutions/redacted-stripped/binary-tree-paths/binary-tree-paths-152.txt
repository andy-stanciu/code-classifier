class Solution {

    List<String> res = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return new ArrayList<>();
        }
        getResult(root, "", 0);
        return res;
    }

    public void getResult(TreeNode root, String path, int index) {
        if (root == null) {
            return;
        }
        if (index == 0) {
            path += String.valueOf(root.val);
        } else {
            path += "->" + String.valueOf(root.val);
        }
        if (root.left == null && root.right == null) {
            res.add(path);
            path = "";
            return;
        }
        index++;
        getResult(root.left, path, index);
        getResult(root.right, path, index);
    }
}
