class Solution {

    private List<String> var1 = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        method1(root, new StringBuilder());
        return var1;
    }

    private void method1(TreeNode node, StringBuilder currentPath) {
        if (node == null)
            return;
        int originalLength = currentPath.length();
        if (originalLength > 0)
            currentPath.append("->");
        currentPath.append(node.val);
        if (node.left == null && node.right == null)
            var1.add(currentPath.toString());
        else {
            method1(node.left, currentPath);
            method1(node.right, currentPath);
        }
        currentPath.setLength(originalLength);
    }
}
