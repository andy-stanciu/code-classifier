class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        method2(root, "", ans);
        return ans;
    }

    public void method2(TreeNode root, String s, List<String> ans) {
        if (root == null)
            return;
        s += root.val + "";
        if (root.left == null && root.right == null)
            ans.add(s);
        s += "->";
        method2(root.left, s, ans);
        method2(root.right, s, ans);
    }
}
