class Solution {

    List<String> s = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        add(root, s, "");
        return s;
    }

    public void add(TreeNode root, List<String> s, String st) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            st += root.val;
            s.add(st);
        }
        st += root.val + "->";
        add(root.left, s, st);
        add(root.right, s, st);
    }
}
