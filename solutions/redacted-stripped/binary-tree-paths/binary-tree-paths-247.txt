class Solution {

    private final List<String> result = new LinkedList<>();

    public List<String> method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return List.of(String.valueOf(root.val));
        }
        method2(root, new StringBuilder());
        return result;
    }

    public void method2(TreeNode node, StringBuilder sb) {
        if (node.left == null && node.right == null) {
            sb.append(node.val);
            result.add(sb.toString());
            return;
        }
        if (node.left != null) {
            method2(node.left, new StringBuilder(sb).append(node.val).append("->"));
        }
        if (node.right != null) {
            method2(node.right, new StringBuilder(sb).append(node.val).append("->"));
        }
    }
}
