class Solution {

    public List<String> result = new ArrayList();

    public void getPath(TreeNode root, StringBuilder sb) {
        if (root == null) {
            return;
        }
        sb.append(root.val).append("->");
        StringBuilder sb1 = new StringBuilder();
        sb1.append(root.val).append("->");
        int n = sb1.length();
        getPath(root.left, sb);
        getPath(root.right, sb);
        if (root.left == null && root.right == null) {
            String str = sb.toString();
            result.add(str.substring(0, str.length() - 2));
        }
        if (sb.length() != 0) {
            sb.delete(sb.length() - n, sb.length());
        }
    }

    public List<String> method1(TreeNode root) {
        getPath(root, new StringBuilder());
        return result;
    }
}
