class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        method2(result, root, "");
        return result;
    }

    private void method2(List<String> result, TreeNode root, String path) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            path += root.val;
            result.add(path);
            return;
        } else {
            path += root.val + "->";
        }
        method2(result, root.left, path);
        method2(result, root.right, path);
    }
}
