class Solution {

    public static List<String> l;

    public void path(TreeNode root, String s) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            s += Integer.toString(root.val);
            l.add(s);
            return;
        }
        s += Integer.toString(root.val);
        s += "->";
        path(root.left, s);
        path(root.right, s);
    }

    public List<String> method1(TreeNode root) {
        l = new ArrayList<>();
        String s = "";
        path(root, s);
        return l;
    }
}
