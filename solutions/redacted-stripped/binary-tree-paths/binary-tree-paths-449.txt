class Solution {

    public List<String> method1(TreeNode root) {
        List<String> lst = new ArrayList<>();
        method2(root, lst, new LinkedList<>());
        return lst;
    }

    public void method2(TreeNode node, List<String> lst, Deque<String> dq) {
        if (node == null)
            return;
        dq.addLast(Integer.toString(node.val));
        if (node.left == null && node.right == null) {
            lst.add(toString(dq));
        } else {
            dq.addLast("->");
            method2(node.left, lst, dq);
            method2(node.right, lst, dq);
            dq.removeLast();
        }
        dq.removeLast();
    }

    public String toString(Deque<String> dq) {
        StringBuilder sb = new StringBuilder();
        for (String s : dq) {
            sb.append(s);
        }
        return sb.toString();
    }
}
