class Solution {

    List<String> l = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        if (root == null) {
            return var1;
        }
        StringBuilder path = new StringBuilder();
        method2(root, path, var1);
        return var1;
    }

    private void method2(TreeNode node, StringBuilder path, List<String> var1) {
        if (node == null) {
            return;
        }
        int originalLength = path.length();
        if (originalLength > 0) {
            path.append("->");
        }
        path.append(node.val);
        if (node.left == null && node.right == null) {
            var1.add(path.toString());
        } else {
            method2(node.left, path, var1);
            method2(node.right, path, var1);
        }
        path.setLength(originalLength);
    }
}
