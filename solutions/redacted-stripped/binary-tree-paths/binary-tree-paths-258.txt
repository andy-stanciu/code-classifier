class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        find(list, root, "");
        return list;
    }

    void find(List<String> list, TreeNode root, String s) {
        if (root == null) {
            return;
        }
        find(list, root.left, s + root.val + "->");
        find(list, root.right, s + root.val + "->");
        if (root.left == null && root.right == null) {
            s = s + root.val;
            list.add(s);
            return;
        }
    }
}
