class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        List<Integer> tempList = new ArrayList<>();
        method2(list, tempList, root);
        return list;
    }

    public static void method2(List<String> list, List<Integer> tempList, TreeNode root) {
        if (root == null) {
            return;
        }
        tempList.add(root.val);
        if (root.left == null && root.right == null) {
            addTolist(list, tempList);
        } else {
            method2(list, tempList, root.left);
            method2(list, tempList, root.right);
        }
        tempList.remove(tempList.size() - 1);
    }

    public static void addTolist(List<String> list, List<Integer> tempList) {
        String str = "";
        System.out.println(tempList);
        for (int i = 0; i < tempList.size(); i++) {
            if (i == tempList.size() - 1) {
                str += String.valueOf(tempList.get(i));
            } else {
                str += String.valueOf(tempList.get(i)) + "->";
            }
        }
        list.add(str);
    }
}
