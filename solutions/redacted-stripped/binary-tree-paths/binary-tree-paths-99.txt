class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        method2(root, "", var1);
        return var1;
    }

    public static void method2(TreeNode root, String path, List<String> var1) {
        if (root == null)
            return;
        path = path + Integer.toString(root.val);
        if (root.left == null && root.right == null)
            var1.add(path);
        else {
            path += "->";
            method2(root.left, path, var1);
            method2(root.right, path, var1);
        }
    }
}
