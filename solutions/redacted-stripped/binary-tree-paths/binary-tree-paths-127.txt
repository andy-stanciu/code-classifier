class Solution {

    List<String> result = new ArrayList<String>();

    public List<String> method1(TreeNode root) {
        recusive(root, "");
        return result;
    }

    void recusive(TreeNode root, String path) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            result.add(path + root.val);
            return;
        }
        recusive(root.left, path + root.val + "->");
        recusive(root.right, path + root.val + "->");
    }
}
