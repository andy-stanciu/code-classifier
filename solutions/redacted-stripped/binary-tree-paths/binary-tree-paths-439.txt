class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList<>();
        StringBuilder slate = new StringBuilder();
        dfsHelper(root, result, slate);
        return result;
    }

    private void dfsHelper(TreeNode node, List<String> result, StringBuilder slate) {
        int temp = slate.length();
        if (node.left == null && node.right == null) {
            slate.append(Integer.valueOf(node.val));
            result.add(slate.toString());
            slate.delete(temp, slate.length());
        }
        slate.append(Integer.valueOf(node.val));
        slate.append("->");
        if (node.left != null) {
            dfsHelper(node.left, result, slate);
        }
        if (node.right != null) {
            dfsHelper(node.right, result, slate);
        }
        slate.delete(temp, slate.length());
    }
}
