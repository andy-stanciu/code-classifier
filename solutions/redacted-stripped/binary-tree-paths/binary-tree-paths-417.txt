class Solution {

    List<String> res = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        String ans = "";
        Path(root, ans);
        return res;
    }

    void Path(TreeNode root, String ans) {
        if (root == null)
            return;
        ans = ans + root.val;
        if (root.left == null && root.right == null) {
            res.add(ans);
            return;
        }
        ans = ans + "->";
        Path(root.left, ans);
        Path(root.right, ans);
    }
}
