class Solution {

    public List<String> method1(TreeNode root, String path, List<String> list) {
        if (root == null) {
            return list;
        }
        if (root.left == null && root.right == null) {
            list.add(path + root.val);
            return list;
        }
        method1(root.left, (path + root.val + "->"), list);
        method1(root.right, (path + root.val + "->"), list);
        return list;
    }

    public List<String> method2(TreeNode root) {
        List<String> list = new ArrayList<String>();
        method1(root, "", list);
        return list;
    }
}
