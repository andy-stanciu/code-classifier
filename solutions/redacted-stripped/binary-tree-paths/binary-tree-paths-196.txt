class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        helper(ans, "", root);
        return ans;
    }

    public void helper(List<String> ans, String curr, TreeNode node) {
        if (node.left == null && node.right == null) {
            ans.add(curr + node.val);
            return;
        }
        if (node.left != null) {
            helper(ans, curr + node.val + "->", node.left);
        }
        if (node.right != null) {
            helper(ans, curr + node.val + "->", node.right);
        }
    }
}
