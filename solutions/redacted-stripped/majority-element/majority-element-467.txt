class Solution {

    public int method1(int[] nums) {
        int maxcount = 1;
        int maxnum = nums[0];
        int count = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != maxnum) {
                if (maxcount == 1) {
                    maxnum = nums[i];
                } else
                    maxcount--;
            } else {
                maxcount++;
            }
        }
        return maxnum;
    }
}
