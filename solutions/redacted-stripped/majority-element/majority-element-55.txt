class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < n; i++) {
            hm.put(nums[i], hm.getOrDefault(nums[i], 0) + 1);
        }
        for (int k : hm.keySet()) {
            if (hm.get(k) > n / 2) {
                return k;
            }
        }
        return -1;
    }
}
