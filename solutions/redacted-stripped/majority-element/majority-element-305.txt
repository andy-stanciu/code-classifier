class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int var1 = nums[0];
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] == var1) {
                count++;
            } else {
                count--;
                if (count == 0) {
                    count = 1;
                    var1 = nums[i];
                }
            }
        }
        return var1;
    }
}
