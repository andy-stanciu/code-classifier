class Solution {

    public int method1(int[] nums) {
        int countBox = 0;
        int person = 0;
        for (int num : nums) {
            if (countBox == 0) {
                person = num;
            }
            if (num == person) {
                countBox++;
            } else {
                countBox--;
            }
        }
        return person;
    }
}
