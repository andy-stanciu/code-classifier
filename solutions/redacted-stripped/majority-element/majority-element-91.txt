class Solution {

    public int method1(int[] nums) {
        return method2(nums, 0, nums.length - 1);
    }

    public static int method2(int[] nums, int si, int ei) {
        if (si == ei) {
            return nums[si];
        }
        int mid = si + (ei - si) / 2;
        int var1 = method2(nums, si, mid);
        int var2 = method2(nums, mid + 1, ei);
        if (var1 == var2) {
            return var1;
        }
        int leftCount = countOccurrances(nums, si, mid, var1);
        int rightCount = countOccurrances(nums, mid + 1, ei, var2);
        return leftCount > rightCount ? var1 : var2;
    }

    public static int countOccurrances(int[] nums, int si, int ei, int target) {
        int count = 0;
        for (int i = si; i <= ei; i++) {
            if (nums[i] == target) {
                count++;
            }
        }
        return count;
    }
}
