class Solution {

    public int method1(int[] nums) {
        int ans = 0, freq = 0;
        for (int i = 0; i < nums.length; i++) {
            if (freq == 0)
                ans = nums[i];
            if (nums[i] == ans)
                freq++;
            else
                freq--;
        }
        return ans;
    }
}
