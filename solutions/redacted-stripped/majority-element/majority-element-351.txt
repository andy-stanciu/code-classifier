class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            map.put(nums[i], map.getOrDefault(nums[i], 0) + 1);
        }
        int ans = nums[0];
        for (Integer key : map.keySet()) {
            if (map.get(ans) < map.get(key)) {
                ans = key;
            }
        }
        return ans;
    }
}
