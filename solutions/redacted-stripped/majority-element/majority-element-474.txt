class Solution {

    public int method1(int[] nums) {
        for (int a = 0; a < nums.length; a++) {
            int repeatCount = 0;
            for (int b = 0; b < nums.length; b++) {
                if (nums[a] == nums[b]) {
                    repeatCount += 1;
                }
            }
            if (repeatCount > nums.length / 2) {
                return nums[a];
            }
        }
        return -1;
    }
}
