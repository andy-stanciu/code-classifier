class Solution {

    public int method1(int[] nums) {
        int cand = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == cand)
                count++;
            else {
                if (count == 0) {
                    cand = nums[i];
                    count++;
                } else {
                    count--;
                }
            }
        }
        return cand;
    }
}
