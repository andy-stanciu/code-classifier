class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        int res = 0;
        for (int i : nums) {
            if (map.get(i) > (int) nums.length / 2) {
                res = i;
            }
        }
        return res;
    }
}
