class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> hmap = new HashMap<>();
        for (int i : nums) {
            hmap.put(i, hmap.getOrDefault(i, 0) + 1);
            if (hmap.get(i) > nums.length / 2) {
                return i;
            }
        }
        return -1;
    }
}
