class Solution {

    public int method1(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            int c = 1;
            for (int j = 1 + i; j < nums.length; j++) {
                if (nums[i] == nums[j])
                    c++;
            }
            if (c > (nums.length / 2))
                return nums[i];
        }
        return 0;
    }
}
