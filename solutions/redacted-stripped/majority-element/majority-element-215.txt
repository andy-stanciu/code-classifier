class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int ele : nums) {
            if (map.containsKey(ele)) {
                map.put(ele, map.get(ele) + 1);
            } else {
                map.put(ele, 1);
            }
        }
        int var1 = 0;
        for (int ele : nums) {
            if (map.get(ele) > (nums.length / 2)) {
                var1 = ele;
                break;
            }
        }
        return var1;
    }
}
