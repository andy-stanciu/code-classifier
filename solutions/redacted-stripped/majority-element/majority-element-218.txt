class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            count += nums[i] == nums[var1] ? 1 : -1;
            if (count == 0) {
                var1 = ++i;
                count++;
            }
        }
        return nums[var1];
    }
}
