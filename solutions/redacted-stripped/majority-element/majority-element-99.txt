class Solution {

    public int method1(int[] nums) {
        int count = 0, var1 = 0;
        for (int num : nums) {
            if (count == 0) {
                var1 = num;
            }
            if (num == var1) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
