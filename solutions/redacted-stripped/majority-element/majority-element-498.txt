class Solution {

    public int method1(int[] nums) {
        int f = 0, ans = 0;
        for (int i = 0; i < nums.length; i++) {
            if (f == 0) {
                ans = nums[i];
            }
            if (ans == nums[i]) {
                f++;
            } else {
                f--;
            }
        }
        return ans;
    }
}
