class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int freq = 0;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (freq == 0) {
                ans = nums[i];
            }
            if (ans == nums[i]) {
                freq = freq + 1;
            } else {
                freq = freq - 1;
            }
        }
        return ans;
    }
}
