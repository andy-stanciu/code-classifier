class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                var1 = num;
                count = 1;
            } else if (num == var1) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
