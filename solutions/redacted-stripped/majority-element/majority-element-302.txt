class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0], vote = 1;
        for (int i = 1; i < nums.length; i++) {
            if (vote == 0) {
                vote++;
                var1 = nums[i];
            } else if (var1 == nums[i]) {
                vote++;
            } else {
                vote--;
            }
        }
        return var1;
    }
}
