class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int count = 0;
        int ele = nums[0];
        for (int i = 0; i < n; i++) {
            if (count == 0) {
                ele = nums[i];
            }
            if (ele == nums[i]) {
                count++;
            } else {
                count--;
            }
        }
        return ele;
    }
}
