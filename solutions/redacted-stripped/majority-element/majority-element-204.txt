class Solution {

    public static int method1(int[] nums) {
        int[] bit = new int[32];
        for (int n : nums) {
            for (int i = 31; i >= 0; i--) {
                if (((n >> i) & 1) == 1) {
                    bit[i]++;
                }
            }
        }
        int maj = 0;
        for (int i = 0; i < 32; i++) {
            if (bit[i] > (nums.length / 2)) {
                maj += (1 << i);
            }
        }
        return maj == 0 ? -1 : maj;
    }
}
