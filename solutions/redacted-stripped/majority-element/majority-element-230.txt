class Solution {

    public int method1(int[] nums) {
        int count = 1;
        int var1 = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (count == 0)
                var1 = nums[i];
            count = nums[i] == var1 ? count + 1 : count - 1;
        }
        return var1;
    }
}
