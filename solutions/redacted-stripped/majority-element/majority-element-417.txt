class Solution {

    static int method1(int[] nums) {
        int var1 = nums[0], votes = 1;
        for (int i = 1; i < nums.length; i++) {
            if (votes == 0) {
                votes++;
                var1 = nums[i];
            } else if (var1 == nums[i]) {
                votes++;
            } else {
                votes--;
            }
        }
        return var1;
    }
}
