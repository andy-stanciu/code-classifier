class Solution {

    public int method1(int[] nums) {
        int n = nums.length / 2;
        int r = 0;
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i <= nums.length - 1; i++) {
            if (hm.containsKey(nums[i])) {
                int p = hm.get(nums[i]);
                hm.put(nums[i], p + 1);
            } else {
                hm.put(nums[i], 1);
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (hm.get(nums[i]) > n) {
                r = nums[i];
            }
        }
        return r;
    }
}
