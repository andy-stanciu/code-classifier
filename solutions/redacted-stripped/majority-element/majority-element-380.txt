class Solution {

    public int method1(int[] nums) {
        int winner = nums[0], vote = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == winner)
                vote++;
            else
                vote--;
            if (vote == 0) {
                winner = nums[i];
                vote = 1;
            }
        }
        return winner;
    }
}
