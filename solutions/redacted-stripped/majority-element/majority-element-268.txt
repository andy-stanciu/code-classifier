class Solution {

    public int method1(int[] nums) {
        int curr = 0, count = 0, max = 0, maxNum = nums[0];
        Arrays.sort(nums);
        for (int x : nums) {
            if (curr != x) {
                curr = x;
                count = 1;
            } else {
                if (count > max) {
                    max = count;
                    maxNum = curr;
                }
                ++count;
            }
        }
        return maxNum;
    }
}
