class Solution {

    public int method1(int[] nums) {
        int el = -1;
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                el = num;
                count = 1;
            } else if (num == el) {
                count++;
            } else {
                count--;
            }
        }
        count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == el) {
                count++;
            }
        }
        if (count > (nums.length / 2))
            return el;
        return -1;
    }
}
