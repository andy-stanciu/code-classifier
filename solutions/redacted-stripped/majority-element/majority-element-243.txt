class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int max = 0;
        int cnt = 0;
        for (int i = 0; i < nums.length; i++) {
            int cc = 1;
            while (i + 1 < nums.length && nums[i] == nums[i + 1]) {
                cc++;
                i++;
            }
            if (cc > cnt) {
                max = nums[i];
                cnt = cc;
            }
        }
        return max;
    }
}
