class Solution {

    public int method1(int[] nums) {
        int i = 1;
        int m = nums[0];
        for (int k = 1; k < nums.length; k++) {
            if (i == 0 || nums[k] == m) {
                i++;
                m = nums[k];
            } else {
                i--;
            }
        }
        return m;
    }
}
