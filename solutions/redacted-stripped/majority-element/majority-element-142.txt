class Solution {

    public int method1(int[] nums) {
        int count = 0, var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0)
                var1 = nums[i];
            if (var1 == nums[i])
                count++;
            else
                count--;
        }
        return var1;
    }
}
