class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (count == 0) {
                var1 = nums[i];
            }
            if (var1 == nums[i]) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
