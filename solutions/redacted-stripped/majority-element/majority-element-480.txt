class Solution {

    public int method1(int[] nums) {
        int count = 1;
        int method1 = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (count == 0) {
                method1 = nums[i];
                count = 1;
            } else if (nums[i] == method1) {
                count++;
            } else {
                count--;
            }
        }
        return method1;
    }
}
