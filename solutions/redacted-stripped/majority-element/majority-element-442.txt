class Solution {

    public int method1(int[] nums) {
        int count = 1, var1 = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (var1 == nums[i]) {
                count++;
            } else {
                count--;
                if (count == 0) {
                    var1 = nums[i];
                    count++;
                }
            }
        }
        if (count >= 1) {
            return var1;
        } else
            return -1;
    }
}
