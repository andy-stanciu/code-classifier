class Solution {

    public int method1(int[] nums) {
        int voteCount = 0;
        int candidate = -1;
        for (int i = 0; i < nums.length; i++) {
            if (voteCount == 0) {
                candidate = nums[i];
                voteCount = 1;
            } else if (candidate == nums[i]) {
                voteCount++;
            } else {
                voteCount--;
            }
        }
        return candidate;
    }
}
