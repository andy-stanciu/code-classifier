class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> count = new HashMap<>();
        for (int num : nums) {
            count.put(num, count.getOrDefault(num, 0) + 1);
        }
        int lowBar = nums.length / 2;
        for (Integer key : count.keySet()) {
            if (count.get(key) > lowBar) {
                return key;
            }
        }
        return 0;
    }
}
