class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> cache = new HashMap();
        for (int num : nums) {
            cache.put(num, cache.getOrDefault(num, 0) + 1);
        }
        int curMax = 0;
        int ans = 0;
        for (Integer key : cache.keySet()) {
            if (curMax < cache.get(key)) {
                curMax = cache.get(key);
                ans = key;
            }
        }
        return ans;
    }
}
