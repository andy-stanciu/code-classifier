class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int var1 = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == var1) {
                count += 1;
            } else {
                count -= 1;
            }
            if (count == 0) {
                var1 = nums[i];
                count = 1;
            }
        }
        return var1;
    }
}
