class Solution {

    public int method1(int[] nums) {
        int count = 1, counter = 0;
        int ansIndex = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[ansIndex]) {
                count++;
            } else {
                count--;
            }
            if (count == 0) {
                ansIndex = i;
                count = 1;
            }
        }
        return nums[ansIndex];
    }
}
