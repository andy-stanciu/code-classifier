class Solution {

    public int method1(int[] nums) {
        int m = nums[0];
        int v = 1;
        for (int i = 1; i < nums.length; i++) {
            if (v == 0) {
                v = 1;
                m = nums[i];
            } else if (m == nums[i]) {
                v++;
            } else {
                v--;
            }
        }
        return m;
    }
}
