class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                count = 1;
                var1 = nums[i];
            } else if (nums[i] == var1) {
                count++;
            } else {
                count--;
            }
        }
        int counter = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == var1) {
                counter++;
            }
        }
        if (counter > nums.length / 2) {
            return var1;
        }
        return -1;
    }
}
