class Solution {

    public int method1(int[] nums) {
        int ans = 0;
        int var1 = 0;
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                var1 = num;
                count++;
            } else if (var1 == num) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
