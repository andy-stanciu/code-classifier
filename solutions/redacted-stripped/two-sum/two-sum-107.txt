class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        int[] method1 = new int[2];
        for (int i = 0; i < nums.length; i++) {
            int var1 = target - nums[i];
            if (hm.containsKey(var1)) {
                method1[0] = i;
                method1[1] = hm.get(var1);
                break;
            }
            hm.put(nums[i], i);
        }
        return method1;
    }
}
