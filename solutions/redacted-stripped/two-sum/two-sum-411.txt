class Solution {

    public int[] method1(int[] nums, int target) {
        for (int i = 0; i < nums.length; i++) {
            int num1 = nums[i];
            for (int j = i + 1; j < nums.length; j++) {
                int num2 = nums[j];
                int var1 = num1 + num2;
                if (var1 == target) {
                    return new int[] { i, j };
                }
            }
        }
        return null;
    }
}
