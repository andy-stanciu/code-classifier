class Solution {

    public int[] method1(int[] a, int target) {
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        for (int i = 0; i < a.length; i++) {
            if (map.containsKey(target - a[i]))
                return new int[] { i, map.get(target - a[i]) };
            map.put(a[i], i);
        }
        return new int[] { -1, -1 };
    }
}
