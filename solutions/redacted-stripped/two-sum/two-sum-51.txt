class Solution {

    public int[] method1(int[] nums, int target) {
        Map<Integer, Integer> HM = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int com = target - nums[i];
            if (HM.containsKey(com)) {
                return new int[] { i, HM.get(com) };
            } else {
                HM.put(nums[i], i);
            }
        }
        return null;
    }
}
