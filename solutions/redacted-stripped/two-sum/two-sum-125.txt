class Solution {

    public int[] method1(int[] nums, int target) {
        int count = nums.length - 1;
        int i = 0;
        while (i < nums.length) {
            if (nums[i] + nums[count] == target && i != count) {
                return new int[] { i, count };
            } else if (count == 0) {
                count = nums.length - 1;
                i++;
            } else {
                count--;
            }
        }
        return new int[] {};
    }
}
