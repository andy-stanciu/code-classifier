class Solution {

    public int[] method1(int[] num, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < num.length; i++) {
            int cur = num[i];
            int x = target - cur;
            if (map.containsKey(x)) {
                return new int[] { map.get(x), i };
            }
            map.put(cur, i);
        }
        return null;
    }
}
