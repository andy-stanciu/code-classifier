class Solution {

    public int[] method1(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int value = nums[i];
            map.put(value, i);
        }
        for (int i = 0; i < nums.length; i++) {
            int value = nums[i];
            int diff = target - value;
            if (map.containsKey(diff) && map.get(diff) != i) {
                return new int[] { i, map.get(diff) };
            }
        }
        return new int[0];
    }
}
