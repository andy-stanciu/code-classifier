class Solution {

    public int[] method1(int[] nums, int target) {
        int[] x = new int[2];
        outer: for (int i = 0; i < nums.length - 1; i++) {
            x[0] = i;
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[i] + nums[j] == target) {
                    x[1] = j;
                    break outer;
                }
            }
        }
        return x;
    }
}
