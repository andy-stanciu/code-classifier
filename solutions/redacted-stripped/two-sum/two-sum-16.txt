class Solution {

    public int[] method1(int[] numbers, int param1) {
        int length = numbers.length;
        for (int gap = 1; gap < length; gap++) {
            for (int right = gap; right < length; right++) {
                int left = right - gap;
                if (numbers[left] + numbers[right] == param1) {
                    return new int[] { left, right };
                }
            }
        }
        return new int[0];
    }
}
