class Solution {

    public int[] method1(int[] nums, int target) {
        int[] in = new int[2];
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (nums[i] + nums[j] == target) {
                    in[0] = i;
                    in[1] = j;
                    break;
                }
            }
        }
        return in;
    }
}
