class Solution {

    public int[] method1(int[] nums, int target) {
        int[][] matrix = new int[nums.length][2];
        for (int i = 0; i < nums.length; ++i) {
            matrix[i][0] = nums[i];
            matrix[i][1] = i;
        }
        Arrays.sort(matrix, (row1, row2) -> row1[0] - row2[0]);
        int left = 0, right = nums.length - 1;
        while (left <= right) {
            if (matrix[left][0] + matrix[right][0] == target)
                return new int[] { matrix[left][1], matrix[right][1] };
            else if (matrix[left][0] + matrix[right][0] > target)
                right--;
            else
                left++;
        }
        return null;
    }
}
