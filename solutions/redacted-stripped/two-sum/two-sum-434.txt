class Solution {

    public int[] method1(int[] nums, int target) {
        int[] a = new int[2];
        int i, j, n, val;
        n = nums.length;
        Map<Integer, Integer> m = new HashMap<>();
        for (i = 0; i < n; i++) {
            val = target - nums[i];
            if (m.containsKey(val)) {
                a[0] = m.get(val);
                a[1] = i;
                break;
            }
            m.put(nums[i], i);
        }
        return a;
    }
}
