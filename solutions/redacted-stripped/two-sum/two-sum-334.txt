class Solution {

    public int[] method1(int[] nums, int target) {
        for (int i = 1; i < nums.length; ++i) {
            for (int j = i; j < nums.length; ++j) {
                int var1 = nums[j - i] + nums[j];
                if (var1 == target) {
                    return new int[] { j - i, j };
                }
            }
        }
        return null;
    }
}
