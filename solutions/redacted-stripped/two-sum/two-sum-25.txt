class Solution {

    public int[] method1(int[] nums, int target) {
        int[] arr = new int[2];
        for (int i = 0; i < nums.length; i++) {
            arr[0] = i;
            for (int j = i + 1; j < nums.length; j++) {
                if (target == nums[j] + nums[i]) {
                    arr[1] = j;
                    return arr;
                }
            }
        }
        return arr;
    }
}
