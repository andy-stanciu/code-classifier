class Solution {

    public int[] method1(int[] nums, int target) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 = target - nums[i];
            for (int j = i + 1; j < nums.length; j++) {
                if (var1 == nums[j]) {
                    return new int[] { i, j };
                }
            }
        }
        return new int[] { -1, -1 };
    }
}
