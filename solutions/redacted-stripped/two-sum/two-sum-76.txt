class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> helper = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (helper.get(nums[i]) != null) {
                return new int[] { helper.get(nums[i]), i };
            }
            helper.put(target - nums[i], i);
        }
        return new int[] { -1, -1 };
    }
}
