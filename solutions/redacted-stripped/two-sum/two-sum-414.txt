class Solution {

    public int[] method1(int[] nums, int target) {
        int a[] = new int[2], i;
        HashMap<Integer, Integer> hm = new HashMap<Integer, Integer>();
        for (i = 0; i < nums.length; i++) if (hm.containsKey(target - nums[i])) {
            a[0] = i;
            a[1] = hm.get(target - nums[i]);
            break;
        } else
            hm.put(nums[i], i);
        return a;
    }
}
