class Solution {

    public int[] method1(int[] nums, int target) {
        int var1 = 0;
        int[] indices = new int[2];
        for (int i = 0; i < nums.length - 1; i++) {
            for (int j = i + 1; j < nums.length; j++) if (nums[i] + nums[j] == target) {
                indices[0] = i;
                indices[1] = j;
            }
        }
        return indices;
    }
}
