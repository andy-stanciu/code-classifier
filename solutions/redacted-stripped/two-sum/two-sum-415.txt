class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> map = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; i++) {
            int temp = target - nums[i];
            if (!map.containsKey(temp)) {
                map.put(nums[i], i);
            } else {
                return new int[] { i, map.get(temp) };
            }
        }
        return new int[] { -1, -1 };
    }
}
