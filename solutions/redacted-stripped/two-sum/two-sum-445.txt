class Solution {

    public int[] method1(int[] nums, int target) {
        int[] addNum = new int[2];
        int len = nums.length;
        for (int i = 0; i < len; i++) {
            for (int a = 0; a < len; a++) {
                if (((nums[i] + nums[a]) == target) && (i != a)) {
                    addNum[0] = i;
                    addNum[1] = a;
                }
            }
        }
        return addNum;
    }
}
