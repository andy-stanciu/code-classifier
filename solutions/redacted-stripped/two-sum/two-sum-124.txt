class Solution {

    public int[] method1(int[] nums, int target) {
        int[] result = new int[2];
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                var1 = nums[i] + nums[j];
                if (var1 == target) {
                    result[0] = i;
                    result[1] = j;
                    return result;
                }
            }
        }
        return result;
    }
}
