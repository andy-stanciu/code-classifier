class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int r = target - nums[i];
            if (mp.containsKey(r)) {
                return new int[] { mp.get(r), i };
            }
            mp.put(nums[i], i);
        }
        return new int[] {};
    }
}
