class Solution {

    public int[] method1(int[] nums, int target) {
        int[] a = new int[2];
        Map<Integer, Integer> hm = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; i++) {
            if (hm.containsKey(target - nums[i])) {
                a[0] = i;
                a[1] = hm.get(target - nums[i]);
                return a;
            } else {
                hm.put(nums[i], i);
            }
        }
        return a;
    }
}
