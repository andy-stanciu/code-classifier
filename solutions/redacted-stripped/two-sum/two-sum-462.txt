class Solution {

    public int[] method1(int[] nums, int target) {
        int[] res = new int[2];
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            hm.put(nums[i], i);
        }
        for (int i = 0; i < nums.length; i++) {
            int rem = target - nums[i];
            if (hm.containsKey(rem) && hm.get(rem) != i) {
                res[0] = i;
                res[1] = hm.get(rem);
                break;
            }
        }
        return res;
    }
}
