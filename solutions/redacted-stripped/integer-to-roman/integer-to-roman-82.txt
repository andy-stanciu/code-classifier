class Solution {

    private final Map<Integer, String> var1 = new LinkedHashMap<>();

    public Solution() {
        var1.put(1000, "M");
        var1.put(900, "CM");
        var1.put(500, "D");
        var1.put(400, "CD");
        var1.put(100, "C");
        var1.put(90, "XC");
        var1.put(50, "L");
        var1.put(40, "XL");
        var1.put(10, "X");
        var1.put(9, "IX");
        var1.put(5, "V");
        var1.put(4, "IV");
        var1.put(1, "I");
    }

    public String method1(int num) {
        StringBuilder sb = new StringBuilder();
        int number = num;
        for (Map.Entry<Integer, String> entry : var1.entrySet()) {
            int value = number / entry.getKey();
            if (value > 0)
                sb.append(entry.getValue().repeat(value));
            number = number % entry.getKey();
        }
        return sb.toString();
    }
}
