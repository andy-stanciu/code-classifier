class Solution {

    public String method1(int num) {
        int[] n = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] c = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        String op = "";
        for (int i = 0; i < n.length && num > 0; i++) {
            while (num >= n[i]) {
                num -= n[i];
                op += c[i];
            }
        }
        return op;
    }
}
