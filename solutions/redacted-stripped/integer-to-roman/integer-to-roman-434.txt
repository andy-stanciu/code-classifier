class Solution {

    public String method1(int num) {
        StringBuilder resultBuilder = new StringBuilder();
        String[] var1 = { "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        if (num >= 1000) {
            for (int i = 0; i < ((num - (num % 1000)) / 1000); i++) {
                resultBuilder.append("M");
            }
            num = num % 1000;
        }
        if (num >= 100) {
            resultBuilder.append(method2(num, var1, 100));
            num = num % 100;
        }
        if (num >= 10) {
            resultBuilder.append(method2(num, var1, 10));
            num = num % 10;
        }
        if (num >= 1) {
            resultBuilder.append(method2(num, var1, 1));
        }
        return resultBuilder.toString();
    }

    private String method2(int num, String[] var1, int k) {
        int index = 0;
        if (k == 10) {
            index = 4;
        }
        if (k == 1) {
            index = 8;
        }
        int tempVar;
        StringBuilder tempBuilder = new StringBuilder();
        if (num >= 9 * k) {
            return var1[0 + index];
        }
        if (num >= 5 * k) {
            tempBuilder.append(var1[1 + index]);
            tempVar = (num - 5 * k) / k;
            for (int i = 0; i < tempVar; i++) {
                tempBuilder.append(var1[3 + index]);
            }
            return tempBuilder.toString();
        }
        if (num >= 4 * k) {
            return var1[2 + index];
        }
        if (num >= 1 * k) {
            tempBuilder.append(var1[3 + index]);
            if (k == 100 || k == 10) {
                num = num - num % (1 * k);
            }
            tempVar = (num - 1 * k) / k;
            for (int i = 0; i < tempVar; i++) {
                tempBuilder.append(var1[3 + index]);
            }
        }
        return tempBuilder.toString();
    }
}
