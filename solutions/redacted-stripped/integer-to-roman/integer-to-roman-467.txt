class Solution {

    public String method1(int num) {
        ArrayList<Integer> al = new ArrayList<>();
        int temp = num;
        int place = 1;
        while (temp > 0) {
            int r = temp % 10;
            al.add(r * place);
            temp = temp / 10;
            place = place * 10;
        }
        Collections.reverse(al);
        String ans = "";
        for (int i = 0; i < al.size(); i++) {
            ans += solve(al.get(i));
        }
        return ans;
    }

    public static String solve(int num) {
        String ans = "";
        while (num > 0) {
            while (num >= 1000) {
                ans += "M";
                num = num - 1000;
            }
            if (num == 900) {
                ans += "CM";
                num = num - 900;
            }
            while (num >= 500) {
                ans += "D";
                num = num - 500;
            }
            if (num == 400) {
                ans += "CD";
                num = num - 400;
            }
            while (num >= 100) {
                ans += "C";
                num = num - 100;
            }
            if (num == 90) {
                ans += "XC";
                num = num - 90;
            }
            while (num >= 50) {
                ans += "L";
                num = num - 50;
            }
            if (num == 40) {
                ans += "XL";
                num = num - 40;
            }
            while (num >= 10) {
                ans += "X";
                num = num - 10;
            }
            if (num == 9) {
                ans += "IX";
                num = num - 9;
            }
            while (num >= 5) {
                ans += "V";
                num = num - 5;
            }
            if (num == 4) {
                ans += "IV";
                num = num - 4;
            }
            if (num <= 3) {
                for (int i = 0; i < num; i++) {
                    ans += "I";
                }
                num = 0;
            }
        }
        return ans;
    }
}
