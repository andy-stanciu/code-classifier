class Solution {

    public String method1(int num) {
        String[] s = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        String ans = "";
        int[] arr = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        for (int i = 0; num > 0; i++) {
            while (num >= arr[i]) {
                ans += s[i];
                num -= arr[i];
            }
        }
        return ans;
    }
}
