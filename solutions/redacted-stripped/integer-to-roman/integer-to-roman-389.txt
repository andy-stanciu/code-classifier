class Solution {

    public String method1(int num) {
        return helper(num);
    }

    String helper(int n) {
        if (n == 0) {
            return "";
        } else if (n >= 1000) {
            return "M" + helper(n - 1000);
        } else if (n < 1000 && n >= 900) {
            return "CM" + helper(n - 900);
        } else if (n < 900 && n >= 500) {
            return "D" + helper(n - 500);
        } else if (n < 500 && n >= 400) {
            return "CD" + helper(n - 400);
        } else if (n < 400 && n >= 100) {
            return "C" + helper(n - 100);
        } else if (n < 100 && n >= 90) {
            return "XC" + helper(n - 90);
        } else if (n < 90 && n >= 50) {
            return "L" + helper(n - 50);
        } else if (n < 50 && n >= 40) {
            return "XL" + helper(n - 40);
        } else if (n < 40 && n >= 10) {
            return "X" + helper(n - 10);
        } else if (n == 9) {
            return "IX" + helper(n - 9);
        } else if (n < 9 && n >= 5) {
            return "V" + helper(n - 5);
        } else if (n == 4) {
            return "IV" + helper(n - 4);
        } else if (n < 4 && n >= 1) {
            return "I" + helper(n - 1);
        } else {
            return "";
        }
    }
}
