class Solution {

    public String method1(int num) {
        Map<Integer, String> var1 = new HashMap<>();
        var1.put(1, "I");
        var1.put(4, "IV");
        var1.put(5, "V");
        var1.put(9, "IX");
        var1.put(10, "X");
        var1.put(40, "XL");
        var1.put(50, "L");
        var1.put(90, "XC");
        var1.put(100, "C");
        var1.put(400, "CD");
        var1.put(500, "D");
        var1.put(900, "CM");
        var1.put(1000, "M");
        String result = method2(num, var1);
        return result;
    }

    private String method2(int num, Map<Integer, String> var1) {
        StringBuilder var2 = new StringBuilder();
        for (int value : new int[] { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 }) {
            while (num >= value) {
                num -= value;
                var2.append(var1.get(value));
            }
        }
        return var2.toString();
    }
}
