class Solution {

    public String method1(int num) {
        int[] symbol = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] value = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        StringBuilder sb = new StringBuilder();
        int i = 0;
        while (num > 0) {
            if (num >= symbol[i]) {
                sb.append(value[i]);
                num = num - symbol[i];
            } else {
                i++;
            }
        }
        return sb.toString();
    }
}
