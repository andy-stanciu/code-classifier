class Solution {

    public String method1(int num) {
        String[] var1 = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] var2 = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String result = "";
        int i = 0;
        while (num > 0 && i < var2.length) {
            if (num >= var2[i]) {
                int n = num / var2[i];
                num = num % var2[i];
                while (n-- > 0) result += var1[i];
            }
            i++;
        }
        return result;
    }
}
