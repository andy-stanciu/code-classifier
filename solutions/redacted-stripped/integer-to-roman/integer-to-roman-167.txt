class Solution {

    public String method1(int num) {
        String[] value = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] valuen = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String s = "";
        for (int i = 0; i < valuen.length; i++) {
            while (num >= valuen[i]) {
                s += value[i];
                num -= valuen[i];
            }
        }
        return s;
    }
}
