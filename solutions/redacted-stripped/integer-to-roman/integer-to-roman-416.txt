class Solution {

    public String method1(int num) {
        String result = "";
        while (num > 0) {
            if (num >= 1000) {
                int temp = num / 1000;
                num %= 1000;
                result += "M".repeat(temp);
            } else if (num >= 500) {
                if (num >= 900) {
                    num %= 900;
                    result += "CM";
                } else {
                    int temp = num / 500;
                    num %= 500;
                    result += "D".repeat(temp);
                }
            } else if (num >= 100) {
                if (num >= 400) {
                    num %= 400;
                    result += "CD";
                } else {
                    int temp = num / 100;
                    num %= 100;
                    result += "C".repeat(temp);
                }
            } else if (num >= 50) {
                if (num >= 90) {
                    num %= 90;
                    result += "XC";
                } else {
                    int temp = num / 50;
                    num %= 50;
                    result += "L".repeat(temp);
                }
            } else if (num >= 10) {
                if (num >= 40) {
                    num %= 40;
                    result += "XL";
                } else {
                    int temp = num / 10;
                    num %= 10;
                    result += "X".repeat(temp);
                }
            } else if (num >= 5) {
                if (num == 9) {
                    num %= 9;
                    result += "IX";
                } else {
                    int temp = num / 5;
                    num %= 5;
                    result += "V".repeat(temp);
                }
            } else {
                if (num == 4) {
                    num %= 4;
                    result += "IV";
                } else {
                    int temp = num / 1;
                    num %= 1;
                    result += "I".repeat(temp);
                }
            }
        }
        return result;
    }
}
