class Solution {

    public String method1(int num) {
        int[] val = { 1000, 500, 100, 50, 10, 5, 1 };
        String[] notation = { "M", "D", "C", "L", "X", "V", "I" };
        StringBuilder strbld = new StringBuilder();
        for (int i = 0; i < val.length; i++) {
            while (num >= val[i]) {
                num -= val[i];
                strbld.append(notation[i]);
            }
            if (i % 2 == 0 && i < val.length - 2 && num >= val[i] - val[i + 2]) {
                num -= val[i] - val[i + 2];
                strbld.append(notation[i + 2]).append(notation[i]);
            } else if (i % 2 == 1 && i < val.length - 1 && num >= val[i] - val[i + 1]) {
                num -= val[i] - val[i + 1];
                strbld.append(notation[i + 1]).append(notation[i]);
            }
        }
        return strbld.toString();
    }
}
