class Solution {

    public String method1(int num) {
        String st = "";
        while (num > 0) {
            if (num >= 1000) {
                st = st + "M";
                num = num - 1000;
            } else if (num >= 500 && num < 1000) {
                if (num >= 900) {
                    st = st + "CM";
                    num = num - 900;
                } else {
                    st = st + "D";
                    num = num - 500;
                }
            } else if (num >= 100 && num < 500) {
                if (num >= 400) {
                    st = st + "CD";
                    num = num - 400;
                } else {
                    st = st + "C";
                    num = num - 100;
                }
            } else if (num >= 50 && num < 100) {
                if (num >= 90) {
                    st = st + "XC";
                    num = num - 90;
                } else {
                    st = st + "L";
                    num = num - 50;
                }
            } else if (num >= 10 && num < 50) {
                if (num >= 40) {
                    st = st + "XL";
                    num = num - 40;
                } else {
                    st = st + "X";
                    num = num - 10;
                }
            } else if (num >= 5 && num < 10) {
                if (num >= 9) {
                    st = st + "IX";
                    num = num - 9;
                } else {
                    st = st + "V";
                    num = num - 5;
                }
            } else if (num >= 1 && num < 5) {
                if (num >= 4 && num < 5) {
                    st = st + "IV";
                    num = num - num;
                } else {
                    st = st + "I";
                    num = num - 1;
                }
            }
        }
        return st;
    }
}
