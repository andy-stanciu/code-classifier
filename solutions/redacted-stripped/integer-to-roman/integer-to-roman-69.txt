class Solution {

    public String method1(int num) {
        if (num < 1 || num > 3999)
            return "";
        StringBuilder str = new StringBuilder();
        String[] var1 = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        int i = 0;
        while (num > 0) {
            while (num >= values[i]) {
                num -= values[i];
                str.append(var1[i]);
            }
            i++;
        }
        return str.toString();
    }
}
