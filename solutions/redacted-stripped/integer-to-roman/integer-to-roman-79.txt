class Solution {

    public String method1(int num) {
        String[] var1 = { "I", "IV", "V", "IX", "X", "XL", "L", "XC", "C", "CD", "D", "CM", "M" };
        int[] value = { 1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000 };
        int i = var1.length - 1;
        String sol = "";
        while (num > 0) {
            while (value[i] <= num) {
                sol += var1[i];
                num -= value[i];
            }
            i--;
        }
        return sol;
    }
}
