class Solution {

    public List toList(int[] nums) {
        ArrayList<Integer> l = new ArrayList();
        for (int x : nums) {
            l.add(x);
        }
        return l;
    }

    public void swap(int x, int y, int[] nums) {
        int temp = nums[x];
        nums[x] = nums[y];
        nums[y] = temp;
    }

    public void permutation(List<List<Integer>> ans, int[] nums, int index) {
        if (index >= nums.length) {
            List<Integer> l = toList(nums);
            if (!ans.contains(l))
                ans.add(l);
            return;
        }
        for (int j = index; j < nums.length; j++) {
            swap(index, j, nums);
            permutation(ans, nums, index + 1);
            swap(index, j, nums);
        }
    }

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        permutation(ans, nums, 0);
        return ans;
    }
}
