class Solution {

    public List<List<Integer>> permuteUnique(int[] nums) {
        int n = nums.length;
        List<List<Integer>> ans = new ArrayList<>();
        recursion(0, ans, nums, n);
        return ans;
    }

    void recursion(int start, List<List<Integer>> ans, int[] nums, int n) {
        if (start == n) {
            List<Integer> temp = new ArrayList<>();
            for (int it : nums) {
                temp.add(it);
            }
            if (!ans.contains(temp))
                ans.add(new ArrayList<>(temp));
            return;
        }
        for (int i = start; i < n; i++) {
            swap(i, start, nums);
            recursion(start + 1, ans, nums, n);
            swap(i, start, nums);
        }
    }

    void swap(int start, int end, int[] nums) {
        int temp = nums[start];
        nums[start] = nums[end];
        nums[end] = temp;
    }
}
