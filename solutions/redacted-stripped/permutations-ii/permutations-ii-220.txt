class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    Set<List<Integer>> set = new HashSet<>();

    void solve(int[] nums, int[] per, int c) {
        if (c == nums.length) {
            List<Integer> permutation = new ArrayList<>();
            for (int num : per) {
                permutation.add(num);
            }
            set.add(permutation);
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (per[i] == 11) {
                per[i] = nums[c];
                solve(nums, per, c + 1);
                per[i] = 11;
            }
        }
    }

    public List<List<Integer>> permuteUnique(int[] nums) {
        int n = nums.length;
        int[] per = new int[n];
        for (int i = 0; i < n; i++) {
            per[i] = 11;
        }
        solve(nums, per, 0);
        for (List<Integer> l : set) ans.add(l);
        return ans;
    }
}
