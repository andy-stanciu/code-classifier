class Solution {

    List<Integer>[] graph;

    boolean[] visited;

    boolean[] marked;

    public boolean canFinish(int param1, int[][] prerequisites) {
        graph = new ArrayList[param1];
        visited = new boolean[param1];
        marked = new boolean[param1];
        for (int i = 0; i < param1; i++) {
            graph[i] = new ArrayList<>();
        }
        for (int i = 0; i < prerequisites.length; i++) {
            graph[prerequisites[i][0]].add(prerequisites[i][1]);
        }
        for (int i = 0; i < param1; i++) {
            if (!visited[i]) {
                if (isCycle(i)) {
                    return false;
                }
            }
        }
        return true;
    }

    private boolean isCycle(int i) {
        visited[i] = true;
        for (Integer j : graph[i]) {
            if (!visited[j]) {
                if (isCycle(j)) {
                    return true;
                }
            } else if (!marked[j]) {
                return true;
            }
        }
        marked[i] = true;
        return false;
    }
}
