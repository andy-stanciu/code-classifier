class Solution {

    public boolean canFinish(int param1, int[][] prerequisites) {
        if (param1 == 0)
            return true;
        int[] var1 = new int[param1];
        for (int i = 0; i < prerequisites.length; i++) {
            var1[prerequisites[i][1]]++;
        }
        boolean[] visited = new boolean[prerequisites.length];
        boolean flag = true;
        while (flag) {
            flag = false;
            for (int i = 0; i < prerequisites.length; i++) {
                if (!visited[i]) {
                    if (var1[prerequisites[i][0]] == 0) {
                        var1[prerequisites[i][1]]--;
                        visited[i] = true;
                        flag = true;
                    }
                }
            }
        }
        for (int i = 0; i < param1; i++) {
            if (var1[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
