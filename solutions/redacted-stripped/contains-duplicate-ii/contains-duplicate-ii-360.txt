class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> indexValueMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (indexValueMap.containsKey(nums[i])) {
                int index = indexValueMap.get(nums[i]);
                if (i - index <= k) {
                    return true;
                } else {
                    indexValueMap.replace(nums[i], index, i);
                }
            } else {
                indexValueMap.put(nums[i], i);
            }
        }
        return false;
    }
}
