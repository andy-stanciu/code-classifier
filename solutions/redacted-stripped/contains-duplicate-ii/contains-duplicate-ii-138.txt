class Solution {

    public boolean method1(int[] nums, int k) {
        int left = 0;
        HashSet<Integer> set = new HashSet<>();
        while (left < nums.length) {
            if (set.contains(nums[left]))
                return true;
            set.add(nums[left]);
            if (set.size() > k) {
                set.remove(nums[left - k]);
            }
            left++;
        }
        return false;
    }
}
