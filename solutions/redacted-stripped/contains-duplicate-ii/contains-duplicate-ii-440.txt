class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (!map.containsKey(nums[i])) {
                map.put(nums[i], i);
            } else {
                int j;
                j = map.get(nums[i]);
                if (nums[i] == nums[j] && Math.abs(i - j) <= k) {
                    return true;
                }
                map.put(nums[i], i);
            }
        }
        System.out.println(map);
        return false;
    }
}
