class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> recentIndex = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (recentIndex.containsKey(nums[i]) && i - recentIndex.get(nums[i]) <= k) {
                return true;
            }
            recentIndex.put(nums[i], i);
        }
        return false;
    }
}
