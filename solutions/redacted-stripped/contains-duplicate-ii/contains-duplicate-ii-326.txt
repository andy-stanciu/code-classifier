class Solution {

    public boolean method1(int[] nums, int k) {
        int l = 0;
        HashSet<Integer> set = new HashSet();
        for (int r = 0; r < nums.length; r++) {
            while (set.contains(nums[r])) {
                if (nums[r] == nums[l] && (r - l) <= k)
                    return true;
                set.remove(nums[l]);
                l++;
            }
            set.add(nums[r]);
        }
        return false;
    }
}
