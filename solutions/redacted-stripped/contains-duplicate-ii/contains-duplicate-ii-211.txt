class Solution {

    public boolean method1(int[] nums, int k) {
        Set<Integer> arr = new HashSet<>();
        for (int i = 0; i < nums.length; ++i) {
            if (!arr.add(nums[i]))
                return true;
            if (i >= k)
                arr.remove(nums[i - k]);
        }
        return false;
    }
}
