class Solution {

    public boolean method1(int[] nums, int k) {
        if (k <= 0)
            return false;
        var se = new HashSet<Integer>(k, 1.0f);
        for (int i = 0; i < k && i < nums.length; i++) {
            if (!se.add(nums[i]))
                return true;
        }
        for (int i = k; i < nums.length; i++) {
            if (!se.add(nums[i]))
                return true;
            se.remove(nums[i - k]);
        }
        return false;
    }
}
