class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        int placeValue = -1;
        for (int i = 0; i < nums.length; i++) {
            placeValue = map.getOrDefault(nums[i], -1);
            if (placeValue != -1) {
                if (Math.abs(placeValue - i) <= k) {
                    return true;
                }
            }
            map.put(nums[i], i);
        }
        return false;
    }
}
