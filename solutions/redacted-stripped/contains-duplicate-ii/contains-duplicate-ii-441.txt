class Solution {

    public boolean method1(int[] nums, int k) {
        if (nums == null || nums.length == 0 || k <= 0) {
            return false;
        }
        for (int i = 0; i < nums.length; i++) {
            int lim = Math.min(i + k + 1, nums.length);
            for (int j = i + 1; j < lim; j++) {
                if (nums[i] == nums[j] && Math.abs(i - j) <= k) {
                    return true;
                }
            }
        }
        return false;
    }
}
