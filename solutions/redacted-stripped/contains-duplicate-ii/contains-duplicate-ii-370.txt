class Solution {

    public boolean method1(int[] nums, int k) {
        HashSet hsObj = new HashSet<Integer>();
        int j = k + 1;
        for (int i = 0; i < j; i++) {
            if (i == nums.length)
                return false;
            if (!hsObj.contains(nums[i]))
                hsObj.add(nums[i]);
            else
                return true;
        }
        for (int i = j; i < nums.length; i++) {
            hsObj.remove(nums[i - j]);
            if (!hsObj.contains(nums[i]))
                hsObj.add(nums[i]);
            else
                return true;
        }
        return false;
    }
}
