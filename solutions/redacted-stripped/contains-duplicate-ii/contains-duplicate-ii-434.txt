class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> a = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (!a.containsKey(nums[i])) {
                a.put(nums[i], i);
            } else if (a.containsKey(nums[i])) {
                int c = i;
                int d = a.get(nums[i]);
                int y = Math.abs(c - d);
                if (y <= k) {
                    return true;
                } else
                    a.put(nums[i], i);
            }
        }
        return false;
    }
}
