class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> h = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (h.containsKey(nums[i])) {
                int x = h.get(nums[i]);
                if (Math.abs(x - i) <= k)
                    return true;
            }
            h.put(nums[i], i);
        }
        return false;
    }
}
