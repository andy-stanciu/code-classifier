class Solution {

    public boolean method1(int[] nums, int k) {
        for (int i = 0; i < nums.length; i++) {
            int limit = Math.min(nums.length, i + k + 1);
            for (int j = i + 1; j < limit; j++) {
                if (nums[i] == nums[j])
                    return true;
            }
        }
        return false;
    }
}
