class Solution {

    public boolean method1(int[] nums, int k) {
        for (int i = 0; i < nums.length - 1; i++) {
            int j = i + 1;
            while (j < nums.length && Math.abs(i - j) <= k) {
                if (nums[i] == nums[j]) {
                    return true;
                }
                j++;
            }
        }
        return false;
    }
}
