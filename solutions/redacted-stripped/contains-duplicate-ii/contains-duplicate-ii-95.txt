class Solution {

    public boolean method1(int[] nums, int k) {
        if (k == 0)
            return false;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int key = nums[i];
            if (map.containsKey(key)) {
                int idx = map.get(key);
                if (i - idx <= k) {
                    return true;
                }
            }
            map.put(key, i);
        }
        return false;
    }
}
