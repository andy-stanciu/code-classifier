class Solution {

    public boolean method1(int[] nums, int k) {
        int i = 0;
        int j = 0;
        int n = nums.length;
        HashSet<Integer> set = new HashSet<>();
        while (i < n && j < n) {
            while (j < n && (j - i) <= k) {
                if (set.contains(nums[j]))
                    return true;
                set.add(nums[j++]);
            }
            set.remove(nums[i++]);
        }
        return false;
    }
}
