class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        int index = 0;
        for (var getTheElements : nums) {
            if (map.containsKey(getTheElements) && Math.abs(map.get(getTheElements) - index) <= k) {
                return true;
            } else {
                map.put(getTheElements, index);
                index += 1;
            }
        }
        return false;
    }
}
