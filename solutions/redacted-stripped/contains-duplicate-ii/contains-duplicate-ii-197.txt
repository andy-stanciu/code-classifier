class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> ans = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (ans.containsKey(nums[i])) {
                int past = ans.get(nums[i]);
                if (Math.abs(past - i) <= k) {
                    return true;
                }
            }
            ans.put(nums[i], i);
        }
        return false;
    }
}
