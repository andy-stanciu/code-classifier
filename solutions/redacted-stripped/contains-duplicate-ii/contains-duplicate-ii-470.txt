class Solution {

    public boolean method1(int[] nums, int k) {
        if (nums == null || k == 0 || nums.length < 2) {
            return false;
        }
        HashSet<Integer> hset = new HashSet<>();
        for (int j = 0; j < nums.length; j++) {
            if (!hset.add(nums[j])) {
                return true;
            }
            if (hset.size() >= k + 1) {
                hset.remove(nums[j - k]);
            }
        }
        return false;
    }
}
