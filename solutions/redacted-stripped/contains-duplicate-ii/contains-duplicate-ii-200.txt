class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> st = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (st.containsKey(nums[i])) {
                int diff = i - st.get(nums[i]);
                if (diff <= k)
                    return true;
                if (diff >= k)
                    st.put(nums[i], i);
            } else {
                st.put(nums[i], i);
            }
        }
        return false;
    }
}
