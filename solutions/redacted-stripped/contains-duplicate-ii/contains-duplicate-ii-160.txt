class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> hp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (hp.containsKey(nums[i])) {
                int value = hp.get(nums[i]);
                if (Math.abs(value - i) <= k) {
                    return true;
                }
            }
            hp.put(nums[i], i);
        }
        return false;
    }
}
