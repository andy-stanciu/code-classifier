class Solution {

    public boolean method1(int[] nums, int k) {
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n && j <= i + k; j++) {
                if (nums[i] == nums[j] && (Math.abs(i - j) <= k))
                    return true;
            }
        }
        return false;
    }
}
