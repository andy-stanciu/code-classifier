class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        HashSet<Integer> hs = new HashSet<>();
        int temp = nums[0];
        for (int i = 1; i < n; i++) {
            if (nums[i] > temp) {
                temp = nums[i];
            }
        }
        for (int i = 0; i < n; i++) {
            if (!hs.contains(nums[i])) {
                hs.add(nums[i]);
            }
        }
        n = Math.max(temp, n);
        for (int i = 1; i <= n; i++) {
            if (!hs.contains(i)) {
                return i;
            }
        }
        return temp + 1;
    }
}
