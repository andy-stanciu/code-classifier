class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int lastSeen = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] <= 0 || (i > 0 && nums[i] == nums[i - 1])) {
                continue;
            }
            if (nums[i] == lastSeen) {
                lastSeen++;
            } else
                break;
        }
        return lastSeen;
    }
}
