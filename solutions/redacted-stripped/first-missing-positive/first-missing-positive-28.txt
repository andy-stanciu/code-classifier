class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int small = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] <= 0)
                continue;
            if (nums[i] == small) {
                small = nums[i] + 1;
            }
        }
        return small;
    }
}
