class Solution {

    public int method1(int[] nums) {
        int n = nums.length, i = 0;
        while (i < n) {
            if (nums[i] > 0 && nums[i] <= n) {
                int pos = nums[i] - 1;
                if (nums[i] != nums[pos]) {
                    int temp = nums[i];
                    nums[i] = nums[pos];
                    nums[pos] = temp;
                    i--;
                }
            }
            i++;
        }
        for (i = 0; i < n; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        return n + 1;
    }
}
