class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int k = 1;
        for (int i : nums) {
            if (i < 0) {
                i = 0;
            }
            if (i > 0 && i == k) {
                i = 0;
                k++;
            }
        }
        return k;
    }
}
