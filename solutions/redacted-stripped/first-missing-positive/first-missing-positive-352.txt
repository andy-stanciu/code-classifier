class Solution {

    public int method1(int[] nums) {
        int min = 1;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; ++i) {
            if (min == nums[i])
                min = nums[i] + 1;
            else if (min < nums[i])
                return min;
        }
        return min;
    }
}
