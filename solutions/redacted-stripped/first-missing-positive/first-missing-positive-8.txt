class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] <= 0 || nums[i] > n) {
                nums[i] = n + 1;
            }
        }
        for (int i = 0; i < n; i++) {
            int num = nums[i];
            if (num > n || num <= 0 || Math.abs(num) == i + 1) {
                continue;
            }
            if (num == i + 1) {
                nums[i] = num * -1;
            }
            int existing = nums[num - 1];
            nums[num - 1] = -1 * num;
            nums[i] = existing;
        }
        for (int i = 0; i < n; i++) {
            if (Math.abs(nums[i]) != i + 1) {
                return i + 1;
            }
        }
        return n + 1;
    }
}
