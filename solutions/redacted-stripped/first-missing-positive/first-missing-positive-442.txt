class Solution {

    public int method1(int[] arr) {
        boolean onepresent = false;
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            if (arr[i] == 1) {
                onepresent = true;
            } else if (arr[i] < 1 || arr[i] > arr.length) {
                arr[i] = 1;
            }
        }
        if (onepresent == false) {
            return 1;
        }
        for (int i = 0; i < n; i++) {
            int index = Math.abs(arr[i]) - 1;
            if (arr[index] > 0) {
                arr[index] *= -1;
            }
        }
        for (int i = 0; i < n; i++) {
            if (arr[i] > 0) {
                return i + 1;
            }
        }
        return n + 1;
    }
}
