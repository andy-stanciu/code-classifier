class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        boolean oneExists = false;
        for (int i = 0; i < len; i++) {
            int curr = nums[i];
            if (curr == 1) {
                oneExists = true;
            } else if (curr <= 0 || curr > len) {
                nums[i] = 1;
            }
        }
        if (!oneExists) {
            return 1;
        }
        for (int i = 0; i < len; i++) {
            int curr = Math.abs(nums[i]);
            if (nums[curr - 1] > 0) {
                nums[curr - 1] = -nums[curr - 1];
            }
        }
        for (int i = 0; i < len; i++) {
            int curr = nums[i];
            if (curr > 0) {
                return i + 1;
            }
        }
        return len + 1;
    }
}
