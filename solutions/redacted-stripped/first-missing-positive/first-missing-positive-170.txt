class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        long max = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0)
                set.add(nums[i]);
            if (max < nums[i])
                max = nums[i];
        }
        if (max > 0) {
            for (int i = 1; i <= max + 1; i++) if (!set.contains(i))
                return i;
        }
        return 1;
    }
}
