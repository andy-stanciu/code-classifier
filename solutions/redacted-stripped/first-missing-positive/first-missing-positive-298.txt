class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        for (int i = 0; i <= n - 1; i++) {
            while (nums[i] >= 0 && nums[i] != i && nums[i] <= n - 1 && nums[nums[i]] != nums[i]) {
                swap(nums, i, nums[i]);
            }
        }
        for (int i = 1; i <= n - 1; i++) {
            int ki = nums[i];
            if (ki != i) {
                return i;
            }
        }
        return nums[0] == n ? n + 1 : n;
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
