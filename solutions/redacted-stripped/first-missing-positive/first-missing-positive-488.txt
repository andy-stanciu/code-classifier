class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        Boolean[] flag = new Boolean[n + 1];
        for (int num : nums) {
            if (num > 0 && num <= n) {
                flag[num] = true;
            }
        }
        for (int i = 1; i <= n; i++) {
            if (flag[i] == null) {
                return i;
            }
        }
        return n + 1;
    }
}
