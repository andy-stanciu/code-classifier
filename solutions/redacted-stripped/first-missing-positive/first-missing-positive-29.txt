class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int p = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                if (p == nums[i]) {
                    p++;
                }
            }
        }
        return p;
    }
}
