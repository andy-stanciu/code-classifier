class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        Boolean bool = false;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 1)
                bool = true;
            if (nums[i] < 1 || nums[i] > n)
                nums[i] = 1;
        }
        if (bool == false)
            return 1;
        for (int i = 0; i < n; i++) {
            int idx = Math.abs(nums[i]);
            nums[idx - 1] = -Math.abs(nums[idx - 1]);
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0)
                return i + 1;
        }
        return n + 1;
    }
}
