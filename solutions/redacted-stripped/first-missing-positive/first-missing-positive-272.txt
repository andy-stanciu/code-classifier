class Solution {

    public int method1(int[] nums) {
        boolean[] check = new boolean[nums.length];
        for (int i = 0; i < nums.length; i++) {
            int idx = nums[i] - 1;
            if (idx >= 0 && idx < nums.length) {
                check[idx] = true;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (check[i] == false)
                return i + 1;
        }
        return nums.length + 1;
    }
}
