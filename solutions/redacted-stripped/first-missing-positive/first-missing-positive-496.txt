class Solution {

    public static int method1(int[] nums) {
        int n = nums.length;
        Arrays.sort(nums);
        int i = 0, ind = 1;
        while (i < n) {
            if ((nums[i] > 0) && (ind == nums[i])) {
                ind++;
            }
            i++;
        }
        return ind;
    }
}
