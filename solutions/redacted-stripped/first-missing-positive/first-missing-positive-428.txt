class Solution {

    public int method1(int[] nums) {
        int var1 = 1;
        Arrays.sort(nums);
        for (int number : nums) {
            if (number > 0) {
                if (var1 == number) {
                    var1++;
                }
            } else {
                if (number > var1) {
                    break;
                }
            }
        }
        return var1;
    }
}
