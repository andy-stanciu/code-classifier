class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Boolean> map = new HashMap<>();
        for (int num : nums) {
            if (num > 0) {
                map.put(num, true);
            }
        }
        int var1 = 1;
        while (map.containsKey(var1)) {
            var1++;
        }
        return var1;
    }
}
