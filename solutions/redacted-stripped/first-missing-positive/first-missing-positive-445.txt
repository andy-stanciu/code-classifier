class Solution {

    public int method1(int[] A) {
        if (A.length == 1)
            return A[0] == 1 ? 2 : 1;
        for (int i = 0; i < A.length; i++) {
            while (A[i] >= 1 && A[i] <= A.length && A[i] != A[A[i] - 1]) {
                int temp = A[i];
                A[i] = A[A[i] - 1];
                A[temp - 1] = temp;
            }
        }
        for (int i = 0; i < A.length; i++) {
            if (A[i] != i + 1)
                return i + 1;
        }
        return A.length + 1;
    }
}
