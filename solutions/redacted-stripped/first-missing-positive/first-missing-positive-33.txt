class Solution {

    public int method1(int[] array) {
        int n = array.length;
        for (int i = 0; i < n; i++) {
            while (array[i] > 0 && array[i] <= n && array[i] != array[array[i] - 1]) {
                swap(array, i, array[i] - 1);
            }
        }
        for (int i = 0; i < n; i++) {
            if (array[i] != i + 1) {
                return i + 1;
            }
        }
        return n + 1;
    }

    static void swap(int[] array, int param1, int second) {
        int temp = array[param1];
        array[param1] = array[second];
        array[second] = temp;
    }
}
