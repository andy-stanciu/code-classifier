class Solution {

    public int method1(int[] nums) {
        for (int i = 0; i < nums.length; ) {
            int curr = nums[i];
            if (curr > 0 && curr < nums.length && curr != nums[curr - 1])
                swap(nums, i, curr - 1);
            else
                i++;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        return nums.length + 1;
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
