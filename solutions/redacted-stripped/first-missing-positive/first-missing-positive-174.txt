class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int check = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < check) {
                continue;
            } else if (nums[i] == check) {
                check++;
            } else {
                return check;
            }
        }
        return check;
    }
}
