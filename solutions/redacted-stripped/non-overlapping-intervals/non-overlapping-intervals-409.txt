class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (x, y) -> x[1] - y[1]);
        int n = param1.length;
        int count = 1, lastEnd = param1[0][1];
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= lastEnd) {
                count++;
                lastEnd = param1[i][1];
            }
        }
        return n - count;
    }
}
