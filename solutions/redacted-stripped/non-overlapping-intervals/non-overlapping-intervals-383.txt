class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 1) {
            return 0;
        }
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                if (a[0] == b[0]) {
                    return Integer.compare(a[1], b[1]);
                }
                return Integer.compare(a[0], b[0]);
            }
        });
        int start = 0;
        int end = 1;
        int ans = 0;
        while (end < param1.length) {
            if (param1[end][1] <= param1[start][1]) {
                ans++;
                if (param1[end][1] < param1[start][1]) {
                    start = end;
                }
            } else if (param1[end][0] < param1[start][1]) {
                ans++;
                if (param1[end][1] < param1[start][1]) {
                    start = end;
                }
            } else {
                start = end;
            }
            end++;
        }
        return ans;
    }
}
