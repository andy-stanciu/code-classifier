class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int remove = 0;
        int end = -100000;
        for (int[] interval : param1) {
            if (interval[0] >= end) {
                end = interval[1];
            } else {
                remove++;
            }
        }
        return remove;
    }
}
