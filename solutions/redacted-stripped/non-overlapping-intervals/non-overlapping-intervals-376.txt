class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length;
        int k = Integer.MIN_VALUE;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int start = param1[i][0];
            int end = param1[i][1];
            if (start >= k) {
                k = end;
                continue;
            } else {
                ans++;
            }
        }
        return ans;
    }
}
