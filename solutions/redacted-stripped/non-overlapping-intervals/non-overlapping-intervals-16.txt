class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 1)
            return 0;
        Arrays.sort(param1, (a, b) -> (a[0] - b[0]));
        int[] prev = param1[0];
        int count = 0;
        for (int i = 1; i < param1.length; i++) {
            int[] curr = param1[i];
            if (curr[0] < prev[1]) {
                count++;
                if (curr[1] < prev[1]) {
                    prev = curr;
                }
            } else {
                prev = curr;
            }
        }
        return count;
    }
}
