class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int i = 0, count = 1;
        for (int j = 1; j < param1.length; j++) {
            if (param1[j][0] >= param1[i][1]) {
                i = j;
                count++;
            }
        }
        return param1.length - count;
    }
}
