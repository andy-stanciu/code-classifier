class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int end = Integer.MIN_VALUE;
        for (int[] i : param1) {
            if (i[0] >= end) {
                end = i[1];
            } else {
                count++;
            }
        }
        return count;
    }
}
