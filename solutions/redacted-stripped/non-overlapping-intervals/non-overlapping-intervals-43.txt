class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int ans = 0;
        int prev = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (prev > param1[i][0]) {
                ans++;
            } else {
                prev = param1[i][1];
            }
        }
        return ans;
    }
}
