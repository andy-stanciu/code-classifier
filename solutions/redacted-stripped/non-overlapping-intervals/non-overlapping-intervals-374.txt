class Solution {

    public int method1(int[][] arr) {
        if (arr.length == 0)
            return 0;
        Arrays.sort(arr, (a, b) -> a[1] - b[1]);
        int end = arr[0][1], count = 1;
        for (int[] e : arr) {
            if (e[0] >= end) {
                end = e[1];
                count++;
            }
        }
        return arr.length - count;
    }
}
