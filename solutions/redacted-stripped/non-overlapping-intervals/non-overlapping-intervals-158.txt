class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int count = 0;
        int prevEnd = Integer.MIN_VALUE;
        for (int[] interval : param1) {
            int start = interval[0];
            int end = interval[1];
            if (start < prevEnd) {
                count++;
            } else {
                prevEnd = end;
            }
        }
        return count;
    }
}
