class Solution {

    public int method1(int[][] param1) {
        int count = 1;
        Arrays.sort(param1, Comparator.comparing(o -> o[1]));
        int prev = param1[0][1];
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= prev) {
                prev = param1[i][1];
                count++;
            }
        }
        return param1.length - count;
    }
}
