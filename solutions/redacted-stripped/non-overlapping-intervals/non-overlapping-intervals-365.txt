class Solution {

    public int method1(int[][] param1) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        for (int[] a : param1) pq.add(a);
        int[] prev = pq.poll();
        int needToRemove = 0;
        while (pq.size() > 0) {
            int[] temp = pq.poll();
            if (temp[0] < prev[1]) {
                needToRemove++;
            } else {
                prev = temp;
            }
        }
        return needToRemove;
    }
}
