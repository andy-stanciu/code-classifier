class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int count = 1, prev = 0;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= param1[prev][1]) {
                count++;
                prev = i;
            }
        }
        return n - count;
    }
}
