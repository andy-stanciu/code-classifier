class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        if (n == 0)
            return 0;
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        int res = 0;
        int index = 0;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= param1[index][1]) {
                index = i;
            } else
                res++;
        }
        return res;
    }
}
