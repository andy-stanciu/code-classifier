class Solution {

    public int method1(int[][] param1) {
        if (param1 == null || param1.length == 0)
            return 0;
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int count = 0;
        int[] pre = param1[0];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < pre[1])
                count++;
            else
                pre = param1[i];
        }
        return count;
    }
}
