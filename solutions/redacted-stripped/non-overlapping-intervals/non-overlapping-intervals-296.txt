class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int count = 0;
        int n = param1.length;
        int k = Integer.MIN_VALUE;
        int i = 0;
        while (i < n) {
            if (param1[i][0] >= k) {
                k = param1[i][1];
            } else {
                count++;
            }
            i++;
        }
        return count;
    }
}
