class Solution {

    public int method1(int[][] param1) {
        int count = 0;
        int end = Integer.MIN_VALUE;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        for (int[] interval : param1) {
            int start = interval[0];
            if (start < end) {
                count++;
            } else {
                end = interval[1];
            }
        }
        return count;
    }
}
