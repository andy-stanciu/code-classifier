class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int last = Integer.MIN_VALUE;
        int cnt = 0;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= last) {
                last = param1[i][1];
            } else {
                cnt++;
            }
        }
        return cnt;
    }
}
