class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> {
            if (a[1] > b[1])
                return 1;
            else if (a[1] == b[1])
                return 0;
            else
                return -1;
        });
        int c = 1, d = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= d) {
                c += 1;
                d = param1[i][1];
            }
        }
        return param1.length - c;
    }
}
