class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0) {
            return 0;
        }
        Arrays.sort(param1, (i1, i2) -> i1[1] - i2[1]);
        int d = 0;
        int curEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= curEnd) {
                curEnd = param1[i][1];
            } else {
                d++;
            }
        }
        return d;
    }
}
