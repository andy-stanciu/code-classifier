class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> (a[0] - b[0]));
        int count = 0;
        int start = param1[0][0];
        for (int[] nums : param1) {
            if (nums[0] < start) {
                count++;
                start = Math.min(start, nums[1]);
            } else {
                start = nums[1];
            }
        }
        return count;
    }
}
