class Solution {

    public int method1(int[][] param1) {
        ;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        Stack<int[]> stk = new Stack<>();
        int res = 0;
        for (int[] i : param1) {
            if (!stk.isEmpty() && i[0] < stk.peek()[1]) {
                res++;
            } else {
                stk.push(i);
            }
        }
        return res;
    }
}
