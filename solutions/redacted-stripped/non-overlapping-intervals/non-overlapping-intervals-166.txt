class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int prevEnd = param1[0][1], var1 = 1;
        for (int i = 1; i < param1.length; i++) {
            int currStart = param1[i][0];
            if (prevEnd <= currStart) {
                prevEnd = param1[i][1];
                var1++;
            }
        }
        return param1.length - var1;
    }
}
