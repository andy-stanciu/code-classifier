class Solution {

    public int method1(int[][] in) {
        Arrays.sort(in, (x, y) -> x[1] - y[1]);
        int last = in[0][1];
        int count = 0;
        for (int i = 1; i < in.length; i++) {
            if (in[i][0] < last) {
                count++;
            } else
                last = in[i][1];
        }
        return count;
    }
}
