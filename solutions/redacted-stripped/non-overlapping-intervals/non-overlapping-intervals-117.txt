class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int remove = 0;
        int endPos = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < endPos) {
                remove++;
                continue;
            }
            endPos = param1[i][1];
        }
        return remove;
    }
}
