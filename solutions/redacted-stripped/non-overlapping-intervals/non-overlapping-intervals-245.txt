class Solution {

    public int method1(int[][] param1) {
        int ans = 0;
        Arrays.sort(param1, new customComparator());
        int last = param1[0][1];
        int i = 1;
        while (i < param1.length) {
            int currentStart = param1[i][0];
            int currentEnd = param1[i][1];
            if (currentStart < last) {
                ans++;
                last = Math.min(last, currentEnd);
            } else {
                last = currentEnd;
            }
            i++;
        }
        return ans;
    }

    class customComparator implements Comparator<int[]> {

        public int compare(int[] a, int[] b) {
            int cmp = Integer.compare(a[0], b[0]);
            if (cmp == 0) {
                cmp = Integer.compare(a[1], b[1]);
            }
            return cmp;
        }
    }
}
