class Solution {

    public int method1(int[][] arr) {
        Arrays.sort(arr, (a, b) -> a[1] - b[1]);
        int[] temp = arr[0];
        int count = 0;
        for (int i = 1; i < arr.length; i++) if (temp[1] > arr[i][0])
            count++;
        else
            temp = arr[i];
        return count;
    }
}
