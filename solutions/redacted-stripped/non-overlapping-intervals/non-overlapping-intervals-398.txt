class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int prev = param1[0][1];
        int count = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < prev) {
                count++;
                prev = Math.min(prev, param1[i][1]);
            } else {
                prev = param1[i][1];
            }
        }
        return count;
    }
}
