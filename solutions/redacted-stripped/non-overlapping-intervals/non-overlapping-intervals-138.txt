class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int n = param1.length;
        int count = 0;
        int endtime = param1[0][1];
        for (int i = 1; i < n; i++) {
            int start = param1[i][0];
            if (start < endtime) {
                count++;
            } else {
                endtime = param1[i][1];
            }
        }
        return count;
    }
}
