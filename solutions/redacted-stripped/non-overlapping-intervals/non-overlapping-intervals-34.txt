class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length;
        int count = 1;
        int prev = 0;
        for (int i = 1; i < n; i++) {
            if (param1[prev][1] <= param1[i][0]) {
                prev = i;
                count++;
            }
        }
        return n - count;
    }
}
