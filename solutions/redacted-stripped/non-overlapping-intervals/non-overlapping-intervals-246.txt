class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> {
            return Integer.compare(a[0], b[0]);
        });
        int count = 1;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < param1[i - 1][1]) {
                param1[i][1] = Math.min(param1[i - 1][1], param1[i][1]);
                continue;
            } else
                count += 1;
        }
        return param1.length - count;
    }
}
