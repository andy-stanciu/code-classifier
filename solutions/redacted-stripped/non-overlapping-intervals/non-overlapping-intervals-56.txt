class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int end = param1[0][1];
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= end) {
                end = param1[i][1];
                count++;
            }
        }
        return n - count;
    }
}
