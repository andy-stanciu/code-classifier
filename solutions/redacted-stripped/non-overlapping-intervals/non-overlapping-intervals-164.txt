class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        List<int[]> ls = new ArrayList<>();
        int count = 0;
        int[] intFir = param1[0];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < intFir[1])
                count++;
            else
                intFir = param1[i];
        }
        return count;
    }
}
