class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, Comparator.comparingInt(o -> o[1]));
        int end = param1[0][1];
        int count = 0;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] < end) {
                count++;
            } else {
                end = param1[i][1];
            }
        }
        return count;
    }
}
