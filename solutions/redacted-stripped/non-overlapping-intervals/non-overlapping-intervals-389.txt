class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[1]));
        int ans = 0;
        int endTime = Integer.MIN_VALUE;
        for (int i = 0; i < param1.length; i++) {
            if (endTime <= param1[i][0]) {
                ans++;
                endTime = param1[i][1];
            }
        }
        return param1.length - ans;
    }
}
