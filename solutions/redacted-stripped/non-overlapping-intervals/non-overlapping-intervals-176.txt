class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int count = 0;
        int[] prev = param1[0];
        for (int i = 1; i < param1.length; i++) {
            if (prev[1] > param1[i][0]) {
                count++;
            } else {
                prev = param1[i];
            }
        }
        return count;
    }
}
