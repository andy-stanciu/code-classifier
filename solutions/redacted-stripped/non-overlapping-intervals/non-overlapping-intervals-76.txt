class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int count = 0;
        int prevIntervalMax = Integer.MIN_VALUE;
        for (int[] interval : param1) {
            if (interval[0] < prevIntervalMax) {
                count++;
                continue;
            }
            prevIntervalMax = interval[1];
        }
        return count;
    }
}
