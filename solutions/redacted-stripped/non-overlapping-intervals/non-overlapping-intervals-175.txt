class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        int count = 0;
        int i = 0, j = 1;
        while (j < param1.length) {
            if (param1[j][0] < param1[i][1])
                count++;
            else
                i = j;
            ++j;
        }
        return count;
    }
}
