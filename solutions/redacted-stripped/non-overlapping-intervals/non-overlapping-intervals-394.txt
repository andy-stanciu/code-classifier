class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length, ans = 0, last = param1[0][1];
        for (int i = 1; i < n; i++) {
            if (param1[i][0] < last)
                ans++;
            else
                last = param1[i][1];
        }
        return ans;
    }
}
