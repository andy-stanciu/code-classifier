public class Solution {

    public int method1(int[][] param1) {
        int toBeRemoved = 0;
        Arrays.sort(param1, new Comparator<int[]>() {

            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            int[] interval = param1[i];
            if (interval[0] < end) {
                toBeRemoved++;
                end = Math.min(end, interval[1]);
            } else {
                end = interval[1];
            }
        }
        return toBeRemoved;
    }
}
