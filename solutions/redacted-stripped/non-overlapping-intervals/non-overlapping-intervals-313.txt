class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int[] prev = param1[0];
        int res = -1;
        for (int[] x : param1) {
            if (x[0] < prev[1] && x[1] <= prev[1]) {
                prev = x;
                res++;
            } else if (x[0] < prev[1] && x[1] > prev[1])
                res++;
            else
                prev = x;
        }
        return res;
    }
}
