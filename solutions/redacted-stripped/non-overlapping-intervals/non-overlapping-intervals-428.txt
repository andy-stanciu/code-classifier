class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (i, j) -> i[1] - j[1]);
        int n = param1.length;
        int count = 1;
        int x_end = param1[0][1];
        for (int[] i : param1) {
            int start = i[0];
            if (start >= x_end) {
                x_end = i[1];
                count++;
            }
        }
        return n - count;
    }
}
