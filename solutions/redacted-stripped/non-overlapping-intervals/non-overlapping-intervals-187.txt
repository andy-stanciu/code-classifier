class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new ArrayComparator());
        int l = 1;
        int last = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < last) {
                continue;
            } else {
                l++;
                last = param1[i][1];
            }
        }
        return (param1.length - l);
    }

    public class ArrayComparator implements Comparator<int[]> {

        @Override
        public int compare(int[] row1, int[] row2) {
            if (row1[1] < row2[1]) {
                return (-1);
            } else if (row1[1] > row2[1]) {
                return (1);
            } else {
                return (row1[0] - row2[0]);
            }
        }
    }
}
