class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        int sol = 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int start, end, firstEnd = param1[0][1];
        for (int[] p : param1) {
            start = p[0];
            end = p[1];
            if (start < firstEnd) {
                sol++;
            } else {
                firstEnd = end;
            }
        }
        return sol - 1;
    }
}
