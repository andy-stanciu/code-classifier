class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (x, y) -> x[0] - y[0]);
        int start = param1[0][0];
        int end = param1[0][1];
        int minCount = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < end) {
                minCount++;
                if (end > param1[i][1]) {
                    start = param1[i][0];
                    end = param1[i][1];
                }
            } else {
                start = param1[i][0];
                end = param1[i][1];
            }
        }
        return minCount;
        Arrays.sort(param1, (x, y) -> x[1] - y[1]);
        int end = param1[0][1];
        int minCount = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < end)
                minCount++;
            else {
                end = param1[i][1];
            }
        }
        return minCount;
    }
}
