class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int count = 0;
        int[] prev = param1[0];
        for (int i = 1; i < param1.length; i++) {
            if (prev[1] > param1[i][0]) {
                prev[1] = Math.min(prev[1], param1[i][1]);
                count++;
            } else
                prev = param1[i];
        }
        return count;
    }
}
