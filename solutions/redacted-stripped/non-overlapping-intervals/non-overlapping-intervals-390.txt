class Solution {

    public int method1(int[][] param1) {
        int count = 0;
        int prev = 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= param1[prev][1]) {
                prev = i;
            } else {
                count++;
            }
        }
        return count;
    }
}
