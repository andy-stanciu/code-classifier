class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int cnt = param1.length;
        int curr = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (curr <= param1[i][0]) {
                cnt--;
                curr = param1[i][1];
            }
        }
        return cnt - 1;
    }
}
