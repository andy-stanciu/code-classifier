class Solution {

    public int method1(int[][] param1) {
        int ans = 0;
        int n = param1.length;
        Arrays.sort(param1, new IntervalComparator());
        int[] temp = param1[0];
        for (int i = 1; i < n; i++) {
            if (temp[1] > param1[i][0]) {
                ans++;
            } else {
                temp = param1[i];
            }
        }
        return ans;
    }

    private class IntervalComparator implements Comparator<int[]> {

        @Override
        public int compare(int[] a, int[] b) {
            return Integer.compare(a[1], b[1]);
        }
    }
}
