class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> {
            if (Integer.compare(a[1], b[1]) != 0)
                return Integer.compare(a[1], b[1]);
            else
                return Integer.compare(a[0], b[0]);
        });
        int start = 0, end = 1, res = 0;
        while (start < end && end < param1.length) {
            if (param1[start][1] <= param1[end][0]) {
                start = end;
                end++;
                continue;
            }
            res++;
            end++;
        }
        return res;
    }
}
