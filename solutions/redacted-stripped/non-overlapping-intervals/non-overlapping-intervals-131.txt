class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (arr1, arr2) -> Integer.compare(arr1[1], arr2[1]));
        int prevIntervalEnd = param1[0][1];
        int var1 = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < prevIntervalEnd) {
                var1 += 1;
            } else {
                prevIntervalEnd = param1[i][1];
            }
        }
        return var1;
    }
}
