class Solution {

    public int method1(int[][] param1) {
        int rs = 0;
        int n = param1.length;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int end = Integer.MIN_VALUE;
        for (int[] interval : param1) {
            if (interval[0] < end) {
                rs++;
            } else {
                end = interval[1];
            }
        }
        return rs;
    }
}
