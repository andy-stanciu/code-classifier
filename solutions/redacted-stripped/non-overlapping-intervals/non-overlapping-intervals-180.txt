class Solution {

    public int method1(int[][] param1) {
        return param1.length - method2(param1);
    }

    private static int method2(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int x_end = param1[0][1];
        int count = 1;
        for (int i = 1; i < param1.length; i++) {
            if (x_end <= param1[i][0]) {
                count++;
                x_end = param1[i][1];
            }
        }
        return count;
    }
}
