class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int p = 0;
        int c = 0;
        for (int i = 0; i < param1.length; i++) {
            if (param1[p][1] > param1[i][0]) {
                c++;
            } else {
                p = i;
            }
        }
        return c - 1;
    }
}
