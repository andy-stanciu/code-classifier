class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int var1 = 1;
        int lastEnd = param1[0][1];
        for (int[] interval : param1) {
            if (interval[0] >= lastEnd) {
                lastEnd = interval[1];
                var1 += 1;
            }
        }
        return param1.length - var1;
    }
}
