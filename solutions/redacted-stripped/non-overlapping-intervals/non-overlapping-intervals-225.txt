class Solution {

    public int method1(int[][] activities) {
        Arrays.sort(activities, Comparator.comparingDouble(o -> o[1]));
        int n = 1;
        int l = activities[0][1];
        for (int i = 1; i < activities.length; i++) {
            if (activities[i][0] >= l) {
                l = activities[i][1];
                n++;
            }
        }
        return activities.length - n;
    }
}
