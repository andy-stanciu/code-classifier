class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        Arrays.sort(param1, Comparator.comparingDouble(a -> a[1]));
        int cnt = 1;
        int endTime = param1[0][1];
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= endTime) {
                cnt++;
                endTime = param1[i][1];
            }
        }
        return param1.length - cnt;
    }
}
