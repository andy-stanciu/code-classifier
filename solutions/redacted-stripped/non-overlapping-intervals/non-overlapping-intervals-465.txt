class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        Arrays.sort(param1, (a, b) -> a[1] == b[1] ? b[0] - a[0] : a[1] - b[1]);
        int min = 0;
        int prevEnd = Integer.MIN_VALUE;
        for (int[] interval : param1) {
            if (prevEnd > interval[0])
                min++;
            else
                prevEnd = interval[1];
        }
        return min;
    }
}
