class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int c = 0, pre = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (pre > param1[i][0]) {
                c++;
                pre = Math.min(pre, param1[i][1]);
            } else
                pre = param1[i][1];
        }
        return c;
    }
}
