class Solution {

    public int method1(int[][] param1) {
        if (param1.length <= 1)
            return 0;
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int c = 0;
        int curr_end = param1[0][1];
        for (int[] next : param1) {
            if (curr_end <= next[0]) {
                curr_end = Math.max(curr_end, next[1]);
            } else {
                c++;
            }
        }
        return c - 1;
    }
}
