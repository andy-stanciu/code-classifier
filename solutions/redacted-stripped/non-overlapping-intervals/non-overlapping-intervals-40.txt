class Solution {

    public int method1(int[][] param1) {
        int count = 0;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int[] previous = param1[0];
        for (int i = 1; i < param1.length; i++) {
            int[] current = param1[i];
            if (previous[1] > current[0]) {
                count++;
                continue;
            }
            previous = current;
        }
        return count;
    }
}
