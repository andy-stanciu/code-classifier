class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (e1, e2) -> e1[1] - e2[1]);
        int prev = -(int) 1e6;
        int res = 0;
        for (int idx = 0; idx < param1.length; idx++) {
            if (param1[idx][0] >= prev) {
                prev = param1[idx][1];
            } else {
                res++;
            }
        }
        return res;
    }
}
