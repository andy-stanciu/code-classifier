class Solution {

    public int method1(int[][] param1) {
        int count = 0;
        if (param1.length == 0) {
            return count;
        }
        Arrays.sort(param1, (a, b) -> (a[0] - b[0]));
        count = 0;
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (end > param1[i][0]) {
                end = Math.min(end, param1[i][1]);
                count++;
            } else {
                end = param1[i][1];
            }
        }
        return count;
    }
}
