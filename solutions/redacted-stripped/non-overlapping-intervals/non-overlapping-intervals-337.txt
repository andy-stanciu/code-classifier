class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        if (n == 1)
            return 0;
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int ans = 0, prev = 0;
        for (int i = 1; i < n; ++i) {
            if (param1[i][0] < param1[prev][1])
                ans++;
            else
                prev = i;
        }
        return ans;
    }
}
