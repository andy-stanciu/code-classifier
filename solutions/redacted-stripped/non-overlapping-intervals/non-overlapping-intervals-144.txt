class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int lastEnd = param1[0][1];
        int deleted = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= lastEnd) {
                lastEnd = param1[i][1];
            } else {
                lastEnd = Math.min(lastEnd, param1[i][1]);
                deleted++;
            }
        }
        return deleted;
    }
}
