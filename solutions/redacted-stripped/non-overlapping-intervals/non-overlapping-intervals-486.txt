class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int reach = param1[0][1];
        int count = 1;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= reach) {
                reach = param1[i][1];
                count++;
            }
        }
        return param1.length - count;
    }
}
