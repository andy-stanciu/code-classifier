class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[1]));
        int count = 0;
        long last = Integer.MIN_VALUE;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= last) {
                count++;
                last = param1[i][1];
            }
        }
        int ans = param1.length - count;
        return ans;
    }
}
