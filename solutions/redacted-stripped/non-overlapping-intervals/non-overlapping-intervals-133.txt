class Solution {

    public int method1(int[][] param1) {
        int var1 = param1.length;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int var2 = 1;
        int var3 = 0;
        for (int index = 1; index < var1; index++) {
            if (param1[index][0] >= param1[var3][1]) {
                var2++;
                var3 = index;
            }
        }
        return var1 - var2;
    }
}
