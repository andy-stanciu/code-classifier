class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int res = 0;
        int prevEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (prevEnd > param1[i][0]) {
                prevEnd = Math.min(prevEnd, param1[i][1]);
                res++;
            } else {
                prevEnd = param1[i][1];
            }
        }
        return res;
    }
}
