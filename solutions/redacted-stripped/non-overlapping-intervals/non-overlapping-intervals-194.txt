class Solution {

    public int method1(int[][] res) {
        Arrays.sort(res, (a, b) -> a[1] - b[1]);
        Stack<int[]> stk = new Stack<>();
        int ress = 0;
        for (int[] i : res) {
            if (!stk.isEmpty() && i[0] < stk.peek()[1])
                ress++;
            else
                stk.push(i);
        }
        return ress;
    }
}
