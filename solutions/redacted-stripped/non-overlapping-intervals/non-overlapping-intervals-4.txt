class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (o1, o2) -> o1[1] - o2[1]);
        int count = 0;
        int prev = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= prev) {
                prev = param1[i][1];
            } else {
                count++;
            }
        }
        return count;
    }
}
