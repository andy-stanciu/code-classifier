class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] o1, int[] o2) {
                return o1[1] - o2[1];
            }
        });
        int count = 0;
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (end > param1[i][0]) {
                count++;
            } else {
                end = param1[i][1];
            }
        }
        return count;
    }
}
