class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (int[] a, int[] b) -> a[0] - b[0]);
        int count = 0;
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= end) {
                end = param1[i][1];
            } else {
                count++;
                end = Math.min(end, param1[i][1]);
            }
        }
        return count;
    }
}
