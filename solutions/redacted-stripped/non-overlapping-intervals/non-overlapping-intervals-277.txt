class Solution {

    public int method1(int[][] param1) {
        int removeCount = 0, size = param1.length;
        Arrays.sort(param1, new Comparator<int[]>() {

            @Override
            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int currentEnd = param1[0][1];
        for (int i = 1; i < size; i++) {
            if (param1[i][0] < currentEnd) {
                removeCount++;
            } else {
                currentEnd = param1[i][1];
            }
        }
        return removeCount;
    }
}
