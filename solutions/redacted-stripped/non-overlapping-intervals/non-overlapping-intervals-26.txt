class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        int ans = 0;
        int endTime = -60000;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] < endTime)
                ans++;
            else
                endTime = param1[i][1];
        }
        return ans;
    }
}
