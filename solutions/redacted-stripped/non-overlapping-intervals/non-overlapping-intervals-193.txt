class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int count = 0;
        int[] lastInterval = param1[0];
        for (int i = 1; i < param1.length; i++) {
            int[] interval = param1[i];
            if (hasOverlap(lastInterval, interval)) {
                count++;
            } else {
                lastInterval = interval;
            }
        }
        return count;
    }

    boolean hasOverlap(int[] before, int[] after) {
        if (before[1] <= after[0])
            return false;
        return true;
    }
}
