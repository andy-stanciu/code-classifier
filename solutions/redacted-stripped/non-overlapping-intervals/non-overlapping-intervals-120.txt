class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        List<int[]> list = new ArrayList<>();
        int count = 0;
        for (int[] interval : param1) {
            if (list.isEmpty() || list.getLast()[1] <= interval[0]) {
                list.add(interval);
            } else {
                count++;
            }
        }
        return count;
    }
}
