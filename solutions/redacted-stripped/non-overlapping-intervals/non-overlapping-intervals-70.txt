class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0 || param1 == null)
            return 0;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int rmv = 0, prv = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < prv)
                rmv++;
            else {
                prv = param1[i][1];
            }
        }
        return rmv;
    }
}
