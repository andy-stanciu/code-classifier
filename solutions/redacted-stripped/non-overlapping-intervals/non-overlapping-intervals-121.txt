class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int prev = Integer.MIN_VALUE;
        int ans = 0;
        for (int i = 0; i < param1.length; i++) {
            int s = param1[i][0];
            int e = param1[i][1];
            if (s < prev) {
                prev = Math.min(prev, e);
                ans++;
            } else {
                prev = e;
            }
        }
        return ans;
    }
}
