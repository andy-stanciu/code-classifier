class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 1)
            return 0;
        Arrays.sort(param1, new Comparator<int[]>() {

            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[1] - o2[1];
            }
        });
        int count = 0;
        int j = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < param1[j][1]) {
                count++;
            } else {
                j = i;
            }
        }
        return count;
    }
}
