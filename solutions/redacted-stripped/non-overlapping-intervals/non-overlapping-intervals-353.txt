class Solution {

    public int method1(int[][] param1) {
        int l = param1.length;
        if (l == 0) {
            return 0;
        }
        Arrays.sort(param1, (a, b) -> {
            return a[1] - b[1];
        });
        int ans = 1;
        int ind = 0;
        for (int i = 1; i < l; i++) {
            if (param1[i][0] >= param1[ind][1]) {
                ans++;
                ind = i;
            }
        }
        return l - ans;
    }
}
