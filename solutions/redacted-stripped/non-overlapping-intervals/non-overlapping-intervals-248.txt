class Solution {

    public int method1(int[][] param1) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a, b) -> (a[1] - b[1]));
        for (int[] one : param1) {
            pq.add(one);
        }
        int end = Integer.MIN_VALUE, res = 0;
        while (!pq.isEmpty()) {
            int[] temp = pq.remove();
            if (temp[0] >= end) {
                end = temp[1];
            } else
                res++;
        }
        return res;
    }
}
