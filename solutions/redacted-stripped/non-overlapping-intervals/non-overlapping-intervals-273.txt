class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] == b[1] ? b[0] - a[0] : a[1] - b[1]);
        int count = 0, left = 0, right = 1;
        while (right < param1.length) {
            if (param1[left][1] <= param1[right][0]) {
                left = right;
                right += 1;
            } else if (param1[left][1] <= param1[right][1]) {
                count += 1;
                right += 1;
            } else if (param1[left][1] > param1[right][1]) {
                count += 1;
                left = right;
                right += 1;
            }
        }
        return count;
    }
}
