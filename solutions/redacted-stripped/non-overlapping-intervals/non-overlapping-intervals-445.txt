class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new MySort());
        int overlapCount = 0;
        int i = 1;
        int[] newInterval = param1[0];
        while (i < param1.length) {
            int[] interval = param1[i];
            if (!(newInterval[1] <= interval[0] || newInterval[0] >= interval[1])) {
                overlapCount++;
                newInterval[1] = Math.min(newInterval[1], interval[1]);
            } else {
                newInterval = param1[i];
            }
            i++;
        }
        return overlapCount;
    }
}

class MySort implements Comparator<int[]> {

    @Override
    public int compare(int[] a, int[] b) {
        return a[0] - b[0];
    }

    @Override
    public boolean equals(Object obj) {
        return false;
    }
}
