class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(a -> a[0]));
        int count = 0;
        int[] first = param1[0];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= first[1]) {
                first = param1[i];
            } else {
                count++;
                first[1] = Math.min(first[1], param1[i][1]);
            }
        }
        return count;
    }
}
