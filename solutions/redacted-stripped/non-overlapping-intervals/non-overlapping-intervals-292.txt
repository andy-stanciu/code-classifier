class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int ans = 0;
        int currentEnd = param1[0][1];
        for (int i = 1; i < param1.length; ++i) if (param1[i][0] >= currentEnd)
            currentEnd = param1[i][1];
        else
            ++ans;
        return ans;
    }
}
