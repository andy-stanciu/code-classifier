class Solution {

    public int method1(int[][] param1) {
        int count = 0;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int min = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < min) {
                count++;
            } else if (param1[i][0] >= min) {
                min = param1[i][1];
            }
        }
        return count;
    }
}
