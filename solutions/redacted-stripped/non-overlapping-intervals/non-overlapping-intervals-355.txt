class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int ans = 0;
        int pend = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < pend) {
                ans++;
                pend = Math.min(pend, param1[i][1]);
            } else {
                pend = param1[i][1];
            }
        }
        return ans;
    }
}
