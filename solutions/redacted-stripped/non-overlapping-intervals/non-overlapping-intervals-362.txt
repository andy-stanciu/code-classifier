class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int prev = Integer.MIN_VALUE, c = 0;
        for (int[] arr : param1) {
            if (arr[0] >= prev)
                prev = arr[1];
            else {
                c++;
                prev = Math.min(prev, arr[1]);
            }
        }
        return c;
    }
}
