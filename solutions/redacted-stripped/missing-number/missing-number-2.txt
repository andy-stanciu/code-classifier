class Solution {

    public int method1(int[] nums) {
        int numsLength = nums.length;
        int i = 0;
        while (i < numsLength) {
            int valAtI = nums[i];
            boolean belongsInRange = valAtI < numsLength;
            if (belongsInRange && valAtI != nums[valAtI]) {
                int temp = nums[i];
                nums[i] = nums[valAtI];
                nums[valAtI] = temp;
            } else {
                i++;
            }
        }
        for (int x = 0; x < numsLength; x++) {
            if (x != nums[x]) {
                return x;
            }
        }
        return numsLength;
    }
}
