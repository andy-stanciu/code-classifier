class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int ans = -1;
        HashSet<Integer> hs = new HashSet<>();
        for (int i = 0; i < n; i++) {
            hs.add(nums[i]);
        }
        for (int i = 0; i <= n; i++) {
            if (!hs.contains(i)) {
                ans = i;
            }
        }
        return ans;
    }
}
