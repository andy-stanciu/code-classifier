class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        for (int i = 0; i <= nums.length; i++) {
            if (!set.contains(i))
                return i;
            else
                continue;
        }
        return 0;
    }
}
