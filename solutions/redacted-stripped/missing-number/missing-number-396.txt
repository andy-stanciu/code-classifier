class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        int calSum = 0;
        int missNo = 0;
        calSum = (n * (n + 1)) / 2;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        missNo = calSum - sum;
        return missNo;
    }
}
