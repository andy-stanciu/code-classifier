class Solution {

    public int method1(int[] nums) {
        int end = nums.length;
        int totalSum = getSum(0, end);
        int numsSum = 0;
        for (int i = 0; i < nums.length; i++) {
            numsSum += nums[i];
        }
        return (totalSum - numsSum);
    }

    public int getSum(int start, int end) {
        int sum = 0;
        for (int i = start; i <= end; i++) {
            sum += i;
        }
        return sum;
    }
}
