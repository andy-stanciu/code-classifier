class Solution {

    public int method1(int[] nums) {
        boolean[] arr = new boolean[nums.length + 1];
        for (int x : nums) {
            arr[x] = true;
        }
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == false) {
                return i;
            }
        }
        return 1;
    }
}
