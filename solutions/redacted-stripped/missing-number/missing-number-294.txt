class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int y = 0;
        for (int i = 0; i <= nums.length - 2; i++) {
            if (nums[i] + 1 != (nums[i + 1])) {
                y = nums[i] + 1;
                return y;
            }
        }
        if (y == 0) {
            if (nums[0] == 0) {
                y = nums[nums.length - 1] + 1;
            } else {
                y = 0;
            }
        }
        return y;
    }
}
