class Solution {

    public int method1(int[] nums) {
        int total_sum = 0;
        int array_sum = 0;
        for (int i = 0; i <= nums.length; i++) {
            total_sum += i;
        }
        for (int i = 0; i < nums.length; i++) {
            array_sum += nums[i];
        }
        return total_sum - array_sum;
    }
}
