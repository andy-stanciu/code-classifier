class Solution {

    public int method1(int[] nums) {
        if (nums == null || nums.length == 0)
            return 0;
        int n = nums.length;
        int expectedTotal = n * (n + 1) / 2;
        for (int i = 0; i < n; i++) {
            expectedTotal = expectedTotal - nums[i];
        }
        return expectedTotal;
    }
}
