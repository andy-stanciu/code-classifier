class Solution {

    public int method1(int[] nums) {
        int n = nums.length, sum = n / 2 * (1 + n);
        for (int i : nums) sum -= i;
        return sum;
    }
}

class Solution {

    public int method1(int[] nums) {
        int n = nums.length, sum = 0;
        for (int i = 0; i < n; i++) sum += nums[i];
        return (int) ((double) n / 2 * (1 + n) - sum);
    }
}
