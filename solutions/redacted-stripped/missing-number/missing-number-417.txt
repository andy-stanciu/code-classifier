class Solution {

    public int method1(int[] nums) {
        List<Integer> al = new ArrayList<>();
        for (int val : nums) {
            al.add(val);
        }
        int num = 0;
        for (int i = 0; i <= nums.length; i++) {
            if (al.contains(i) == false) {
                num = i;
                break;
            }
        }
        return num;
    }
}
