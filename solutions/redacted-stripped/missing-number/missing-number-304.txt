class Solution {

    public int method1(int[] nums) {
        int N = nums.length;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
        }
        int var2 = (nums.length * (nums.length + 1)) / 2;
        int method1 = var2 - var1;
        return method1;
    }
}
