class Solution {

    public int method1(int[] nums) {
        int arrayLength = nums.length;
        int unique = 0;
        for (int i = 0; i <= arrayLength; i++) {
            unique ^= i;
        }
        for (int i = 0; i < arrayLength; i++) {
            unique ^= nums[i];
        }
        return unique;
    }
}
