class Solution {

    public int method1(int[] nums) {
        int[] f = new int[10001];
        for (int i = 0; i < nums.length; i++) {
            f[nums[i]]++;
        }
        for (int i = 0; i <= 10001; i++) {
            if (f[i] == 0)
                return i;
        }
        return -1;
    }
}
