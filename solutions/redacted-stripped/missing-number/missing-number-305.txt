class Solution {

    public int method1(int[] nums) {
        int length = nums.length;
        int init_xor = 0;
        if (length % 4 == 0)
            init_xor = length;
        if (length % 4 == 1)
            init_xor = 1;
        if (length % 4 == 2)
            init_xor = length + 1;
        int array_xor = 0;
        for (int i = 0; i < length; i++) {
            array_xor ^= nums[i];
        }
        return array_xor ^ init_xor;
    }
}
