class Solution {

    public int method1(int[] nums) {
        int[] var1 = new int[nums.length];
        for (int a : nums) {
            if (a <= nums.length && a > 0)
                var1[a - 1]++;
        }
        for (int b = 0; b < var1.length; b++) {
            if (var1[b] == 0)
                return b + 1;
        }
        return 0;
    }
}
