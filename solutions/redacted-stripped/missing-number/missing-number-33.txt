class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            map.put(nums[i], nums[i]);
        }
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(i)) {
                continue;
            } else {
                return i;
            }
        }
        return nums.length;
    }
}
