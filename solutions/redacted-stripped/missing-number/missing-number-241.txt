class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        if (nums[0] != 0)
            return 0;
        int left = 0;
        int right = nums.length - 1;
        while (left < right) {
            if (nums[left] + 1 != nums[left + 1]) {
                return nums[left] + 1;
            }
            if (nums[right] - 1 != nums[right - 1]) {
                return nums[right] - 1;
            }
            left++;
            right--;
        }
        return nums[nums.length - 1] + 1;
    }
}
