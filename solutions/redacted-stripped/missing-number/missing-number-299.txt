class Solution {

    public int method1(int[] nums) {
        int t = 0;
        Arrays.sort(nums);
        for (int i = 0; i <= nums.length; i++) {
            try {
                if (nums[i] != i) {
                    return i;
                }
            } catch (Exception e) {
                return i;
            }
        }
        return 0;
    }
}
