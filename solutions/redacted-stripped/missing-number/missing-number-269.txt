class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int ans = 0;
        if (nums[0] != 0) {
            return 0;
        }
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] != 1) {
                return nums[i] - 1;
            }
        }
        return nums.length;
    }
}
