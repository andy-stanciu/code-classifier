class Solution {

    public int method1(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            if (nums[i] < nums.length && nums[i] != i) {
                int temp = nums[i];
                nums[i] = nums[temp];
                nums[temp] = temp;
            } else {
                i++;
            }
        }
        for (int c = 0; c < nums.length; c++) {
            if (nums[c] != c)
                return c;
        }
        return nums.length;
    }
}
