class Solution {

    public int method1(int[] nums) {
        int xor1 = 0;
        for (int i = 0; i <= nums.length; i++) {
            xor1 = xor1 ^ i;
        }
        int xor2 = 0;
        for (int i = 0; i < nums.length; i++) {
            xor2 = xor2 ^ nums[i];
        }
        return (xor1 ^ xor2);
    }
}
