class Solution {

    public int method1(int[] nums) {
        int index = 0;
        while (index < nums.length) {
            int currect = nums[index];
            if (currect < nums.length && nums[currect] != nums[index]) {
                swap(nums, currect, index);
            } else {
                index++;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != i)
                return i;
        }
        return nums.length;
    }

    static void swap(int[] arr, int currect, int index) {
        int temp = arr[currect];
        arr[currect] = arr[index];
        arr[index] = temp;
    }
}
