class Solution {

    public int method1(int[] nums) {
        int length = nums.length;
        int[] numsTrack = new int[length + 1];
        int ans = 0;
        Arrays.fill(numsTrack, 0);
        for (int i = 0; i < length; i++) {
            numsTrack[nums[i]] = 1;
        }
        for (int i = 0; i < length + 1; i++) {
            if (numsTrack[i] == 0) {
                ans = i;
                break;
            }
        }
        return ans;
    }
}
