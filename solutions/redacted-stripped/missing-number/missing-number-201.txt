class Solution {

    public int method1(int[] nums) {
        int range = nums.length;
        int numsSum = 0;
        int rangeSum = 0;
        for (int i = 0; i <= range; i++) {
            rangeSum += i;
        }
        for (int num : nums) {
            numsSum += num;
        }
        int var1 = rangeSum - numsSum;
        return var1;
    }
}
