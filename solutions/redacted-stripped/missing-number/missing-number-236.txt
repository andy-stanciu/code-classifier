class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int lo = 0;
        int hi = nums.length - 1;
        while (lo < hi) {
            int mid = (hi + lo) / 2;
            if (nums[mid] != mid) {
                if (nums[mid] > mid) {
                    hi = mid;
                } else {
                    lo = mid + 1;
                }
            } else {
                lo = mid + 1;
            }
        }
        if (lo == nums[lo]) {
            return lo + 1;
        }
        return lo;
    }
}
