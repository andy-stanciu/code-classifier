class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        int s = n * (n + 1) / 2;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        if (sum == s) {
            return 0;
        }
        return s - sum;
    }
}
