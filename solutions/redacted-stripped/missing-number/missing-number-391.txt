class Solution {

    public int method1(int[] nums) {
        int sumOfAllTillN = 0;
        int sumOfAllInArray = 0;
        for (int i = 0; i < nums.length; i++) {
            sumOfAllTillN += i;
            sumOfAllInArray += nums[i];
        }
        sumOfAllTillN += nums.length;
        return sumOfAllTillN - sumOfAllInArray;
    }
}
