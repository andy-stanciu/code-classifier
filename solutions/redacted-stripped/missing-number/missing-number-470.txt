class Solution {

    public int method1(int[] nums) {
        Map<Integer, Boolean> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            map.put(nums[i], true);
        }
        for (int i = 0; i <= nums.length; i++) {
            if (!(map.getOrDefault(i, false))) {
                return i;
            }
        }
        return -1;
    }
}
