class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum_of_given = 0;
        int sum_of_full = n * (n + 1) / 2;
        for (int i = 0; i < nums.length; i++) {
            sum_of_given += nums[i];
        }
        return sum_of_full - sum_of_given;
    }
}
