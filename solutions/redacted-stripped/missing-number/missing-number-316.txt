class Solution {

    public int method1(int[] nums) {
        int l = 0;
        int r = nums.length - 1;
        int m = l + (r - l) / 2;
        Arrays.sort(nums);
        if (nums[l] != l) {
            return l;
        }
        if (nums[r] == r) {
            return r + 1;
        }
        while (l <= r) {
            if (nums[m] == m) {
                if (nums[m] + 1 != m + 1) {
                    return m + 1;
                }
                l = m + 1;
            } else if (nums[m] != m) {
                r = m - 1;
            }
            m = l + (r - l) / 2;
        }
        return l;
    }
}
