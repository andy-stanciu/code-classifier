class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int arrSize = nums.length;
        int j = 0;
        int method1 = 0;
        for (int i = 0; i < arrSize; i++) {
            if (nums[i] == j) {
                j++;
            } else {
                return nums[i] - 1;
            }
        }
        return arrSize;
    }
}
