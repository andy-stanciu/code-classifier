class Solution {

    public int method1(int[] nums) {
        int res = nums.length;
        for (int i = 0; i < nums.length; i++) {
            res += i - nums[i];
        }
        return res;
    }

    public static void main(String[] args) {
        int[] nums = { 3, 0, 1 };
        Solution solution = new Solution();
        System.out.println(solution.method1(nums));
    }
}
