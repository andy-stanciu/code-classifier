class Solution {

    public int method1(int[] result, int index) {
        if (result[index] == -1) {
            return index;
        }
        return method1(result, index + 1);
    }

    public int method2(int[] nums, int[] result, int size, int index) {
        if (index >= size) {
            return method1(result, 0);
        }
        result[nums[index]] = nums[index];
        return method2(nums, result, size, index + 1);
    }

    public int method3(int[] nums) {
        int size = nums.length;
        int[] result = new int[size + 1];
        Arrays.fill(result, -1);
        return method2(nums, result, size, 0);
    }
}
