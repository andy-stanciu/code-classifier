class Solution {

    public int method1(int[] nums) {
        if (nums == null || nums.length == 0)
            return -1;
        int n = nums.length;
        int expSum = (n * (n + 1)) / 2;
        int actualSum = 0;
        for (int num : nums) {
            actualSum += num;
        }
        return expSum - actualSum;
    }
}
