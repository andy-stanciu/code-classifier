class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int sum2 = 0;
        int n = nums.length;
        for (int i = 1; i <= n; i++) {
            sum = sum + i;
        }
        for (int i = 0; i < nums.length; i++) {
            sum2 = sum2 + nums[i];
        }
        return sum - sum2;
    }
}
