class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int max = nums[nums.length - 1];
        int[] a = new int[10000];
        for (int i = 0; i < nums.length; i++) {
            a[nums[i]]++;
        }
        for (int i = 0; i <= max + 1; i++) {
            if (a[i] == 0)
                return i;
        }
        return 0;
    }
}
