class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        int n = nums.length;
        int expec = n * (n + 1) / 2;
        if (sum == expec) {
            return 0;
        } else {
            return expec - sum;
        }
    }
}
