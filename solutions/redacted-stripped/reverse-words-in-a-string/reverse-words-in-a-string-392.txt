class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        for (int i = 0; i < var1.length / 2; i++) {
            String temp = var1[i];
            var1[i] = var1[var1.length - i - 1];
            var1[var1.length - i - 1] = temp;
        }
        return String.join(" ", var1);
    }
}
