class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] str = s.split("\\\\s+");
        String newStr = "";
        for (int i = str.length - 1; i >= 0; i--) {
            if (i == 0)
                newStr += str[i];
            else
                newStr += str[i] + " ";
        }
        return newStr;
    }
}
