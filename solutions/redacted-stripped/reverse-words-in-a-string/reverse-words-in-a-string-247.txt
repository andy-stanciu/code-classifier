class Solution {

    public String method1(String s) {
        String[] var1 = s.split("\\\\s+");
        StringBuilder result = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].equals("\\\\s+")) {
                result.append(var1[i]).append(" ");
            }
        }
        return result.toString().trim();
    }
}
