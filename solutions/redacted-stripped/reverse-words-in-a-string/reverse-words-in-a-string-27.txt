class Solution {

    public String method1(String s) {
        StringBuilder sb = new StringBuilder();
        String[] var1 = s.split(" +");
        int n = var1.length;
        for (int i = n - 1; i >= 0; i--) {
            sb.append(var1[i]);
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
