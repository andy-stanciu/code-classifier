class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split(" ");
        StringBuilder response = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (var1[i].length() == 0)
                continue;
            response.append(var1[i]);
            if (i != 0)
                response.append(" ");
        }
        return response.toString();
    }
}
