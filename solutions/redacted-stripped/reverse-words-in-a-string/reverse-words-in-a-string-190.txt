class Solution {

    public String method1(String s) {
        String[] ss = s.trim().split("\\\\s+");
        StringBuilder sss = new StringBuilder();
        for (int i = ss.length - 1; i >= 1; i--) {
            sss.append(ss[i] + " ");
        }
        sss.append(ss[0]);
        return sss.toString();
    }
}
