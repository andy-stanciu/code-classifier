class Solution {

    public String method1(String s) {
        Pattern pattern = Pattern.compile("\\\\s+");
        String[] var1 = pattern.split(s);
        String result = "";
        for (int i = 0; i < var1.length; i++) {
            if (i == var1.length - 1) {
                result = var1[i] + result;
            } else {
                result = " " + var1[i] + result;
            }
        }
        return result.trim();
    }
}
