class Solution {

    public String method1(String s) {
        s.trim();
        String[] var1 = s.split("\\\\s+");
        StringBuilder var2 = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            var2.append(var1[i]);
        }
        return var2.toString().trim();
    }
}
