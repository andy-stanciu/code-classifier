class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder ans = new StringBuilder();
        for (int i = var1.length - 1; i > 0; i--) {
            ans.append(var1[i]);
            ans.append(" ");
        }
        ans.append(var1[0]);
        return ans.toString();
    }
}
