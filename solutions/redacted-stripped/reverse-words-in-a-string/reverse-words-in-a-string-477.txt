class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        int j = var1.length - 1;
        StringBuilder ans = new StringBuilder();
        while (j >= 0) {
            ans.append(var1[j]);
            if (j != 0) {
                ans.append(" ");
            }
            j--;
        }
        return ans.toString();
    }
}
