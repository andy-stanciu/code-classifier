class Solution {

    public String method1(String s) {
        String[] ward = s.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = ward.length - 1; i >= 0; i--) {
            sb.append(ward[i]);
            if (i > 0) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
