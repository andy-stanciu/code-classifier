class Solution {

    public static String method1(String s) {
        String[] var1 = s.split("\\\\s+");
        String var2 = new String();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (var1[i] != " ")
                var2 = var2 + var1[i] + " ";
        }
        return var2.trim();
    }

    public static void method2(String[] args) {
        System.out.println(method1("the sky is blue"));
    }
}
