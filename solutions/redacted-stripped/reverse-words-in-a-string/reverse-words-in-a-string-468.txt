class Solution {

    public String method1(String s) {
        StringTokenizer tokn = new StringTokenizer(s);
        Stack<String> stack = new Stack();
        while (tokn.hasMoreTokens()) {
            stack.push(tokn.nextToken());
        }
        StringBuilder ans = new StringBuilder();
        while (!stack.isEmpty()) {
            ans.append(stack.pop());
            if (!stack.isEmpty()) {
                ans.append(" ");
            }
        }
        return ans.toString();
    }
}
