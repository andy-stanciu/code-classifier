class Solution {

    public String method1(String s) {
        String[] arrr = s.split("\\\\s+");
        StringBuilder sb = new StringBuilder("");
        for (int i = arrr.length - 1; i >= 0; i--) {
            sb.append(arrr[i] + " ");
        }
        String ans = sb.toString();
        ans = ans.trim();
        return ans;
    }
}
