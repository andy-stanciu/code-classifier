class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] splited = s.split("\\\\s+");
        int len = splited.length;
        s = "";
        for (int i = len - 1; i > 0; i--) {
            s += splited[i];
            s += " ";
        }
        s += splited[0];
        return s;
    }
}
