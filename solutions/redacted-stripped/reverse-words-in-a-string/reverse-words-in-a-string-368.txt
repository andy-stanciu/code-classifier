class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] seperate = s.split("\\\\s+");
        int n = seperate.length;
        String[] var1 = new String[n];
        for (int i = 0; i < n; i++) {
            var1[i] = seperate[n - 1 - i];
        }
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < var1.length; i++) {
            ans.append(var1[i]);
            if (i < n - 1) {
                ans.append(" ");
            }
        }
        return ans.toString();
    }
}
