class Solution {

    public String method1(String s) {
        final StringBuilder sb = new StringBuilder();
        final String[] parts = s.split(" ");
        for (int i = parts.length - 1; i > -1; i--) {
            String part = parts[i];
            if (!part.isEmpty()) {
                sb.append(part).append(" ");
            }
        }
        return sb.toString().stripTrailing();
    }
}
