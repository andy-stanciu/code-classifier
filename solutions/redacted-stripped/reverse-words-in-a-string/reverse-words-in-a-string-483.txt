class Solution {

    public String method1(String s) {
        String[] st = s.trim().split("\\\\s+");
        int start = 0;
        int end = st.length - 1;
        while (start < end) {
            String temp = st[start];
            st[start] = st[end];
            st[end] = temp;
            start++;
            end--;
        }
        return String.join(" ", st);
    }
}
