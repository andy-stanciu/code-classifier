class Solution {

    public String method1(String s) {
        String[] str = s.trim().split("\\\\s+");
        int start = 0, end = str.length - 1;
        while (start < end) {
            String temp = str[end];
            str[end] = str[start];
            str[start] = temp;
            start++;
            end--;
        }
        return String.join(" ", str);
    }
}
