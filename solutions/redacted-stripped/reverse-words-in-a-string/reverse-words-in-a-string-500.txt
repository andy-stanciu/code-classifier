class Solution {

    public String method1(String s) {
        if (s.length() <= 1 || s == null)
            return s;
        String[] var1 = s.split("\\\\s+");
        StringBuilder ans = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            ans.append(var1[i]);
            ans.append(" ");
        }
        return ans.toString().trim();
    }
}
