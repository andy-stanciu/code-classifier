class Solution {

    public String method1(String s) {
        return Arrays.stream(s.split(" ")).map(String::trim).filter(word -> !word.isEmpty()).reduce((word1, word2) -> word2 + " " + word1).orElse("");
    }
}

class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split(" +");
        Collections.reverse(Arrays.asList(var1));
        return String.join(" ", var1);
    }
}
