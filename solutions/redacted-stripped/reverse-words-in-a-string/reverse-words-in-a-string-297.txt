class Solution {

    public String method1(String s) {
        String[] str_array = s.split(" +");
        String var1 = "";
        for (int i = str_array.length - 1; i >= 0; i--) {
            if (i != str_array.length - 1) {
                var1 = var1 + " " + str_array[i];
            } else {
                var1 = var1 + str_array[i];
            }
        }
        return var1.trim();
    }
}
