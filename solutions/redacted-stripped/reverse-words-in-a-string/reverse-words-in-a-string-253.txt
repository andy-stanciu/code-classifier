class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("[ |\\s]+");
        String retVal = "";
        System.out.println(var1.length);
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].equals(" ")) {
                retVal += var1[i];
                if (i != 0) {
                    retVal += " ";
                }
            }
        }
        return retVal;
    }
}
