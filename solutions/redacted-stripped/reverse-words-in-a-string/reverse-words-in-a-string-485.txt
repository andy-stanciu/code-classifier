class Solution {

    public String method1(String s) {
        String[] arr = s.trim().split("\\\\s+");
        int var1 = 0;
        int var2 = arr.length - 1;
        while (var1 < var2) {
            String temp = arr[var1];
            arr[var1] = arr[var2];
            arr[var2] = temp;
            var1++;
            var2--;
        }
        return String.join(" ", arr);
    }
}
