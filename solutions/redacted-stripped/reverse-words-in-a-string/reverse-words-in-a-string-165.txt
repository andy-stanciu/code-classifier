class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split(" ");
        StringJoiner var2 = new StringJoiner(" ");
        for (int i = var1.length - 1; i >= 0; i--) {
            if (var1[i].length() > 0)
                var2.add(var1[i]);
        }
        return var2.toString();
    }
}
