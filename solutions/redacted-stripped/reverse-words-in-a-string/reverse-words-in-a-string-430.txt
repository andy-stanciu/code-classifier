class Solution {

    public String method1(String s) {
        Stack<String> var1 = new Stack<>();
        String[] var2 = s.split(" ");
        StringBuilder output = new StringBuilder();
        for (String word : var2) {
            if (word.isEmpty())
                continue;
            var1.push(word);
        }
        while (!var1.isEmpty()) {
            output.append(var1.pop()).append(" ");
        }
        return output.toString().trim();
    }
}
