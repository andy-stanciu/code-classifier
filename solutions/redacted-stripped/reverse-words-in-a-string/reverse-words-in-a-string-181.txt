class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] var1 = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            String word = var1[i];
            if (word.trim().length() != 0) {
                sb.append(word).append(" ");
            }
        }
        return sb.toString().trim();
    }
}
