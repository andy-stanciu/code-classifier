class Solution {

    public String method1(String s) {
        Stack<String> stack = new Stack<>();
        String[] arr = s.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (String word : arr) {
            stack.push(word);
        }
        while (!stack.isEmpty()) {
            sb.append(stack.pop());
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
