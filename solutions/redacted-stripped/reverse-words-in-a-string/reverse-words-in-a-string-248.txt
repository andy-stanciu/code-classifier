class Solution {

    public String method1(String s) {
        String[] str = s.split(" ");
        StringBuilder rev = new StringBuilder();
        for (int i = str.length - 1; i >= 0; i--) {
            if (str[i].length() == 0) {
                continue;
            }
            if (rev.length() == 0) {
                rev.append(str[i]);
            } else {
                rev.append(" ");
                rev.append(str[i]);
            }
        }
        return rev.toString();
    }
}
