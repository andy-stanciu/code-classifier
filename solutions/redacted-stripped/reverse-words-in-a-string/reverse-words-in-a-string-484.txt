class Solution {

    public String method1(String s) {
        String[] sArr = s.trim().split(" ");
        StringBuilder ans = new StringBuilder();
        for (int i = sArr.length - 1; i >= 0; i--) {
            if (sArr[i].length() > 0) {
                ans.append(sArr[i]).append(" ");
            }
        }
        return ans.toString().trim();
    }
}
