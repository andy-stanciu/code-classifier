class Solution {

    public String method1(String s) {
        String x = s.replaceAll("\\\\s+", " ").trim();
        String[] ss = x.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int j = ss.length - 1; j >= 0; j--) {
            if (ss[j] != " ") {
                sb.append(ss[j]);
                sb.append(" ");
            }
        }
        String res = new String(sb);
        return res.trim();
    }
}
