class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] str = s.split("\\\\s+");
        String ans = "";
        for (int i = str.length - 1; i >= 0; i--) {
            ans = ans + str[i] + " ";
        }
        ans.substring(1, ans.length() - 1);
        ans = ans.trim();
        return ans;
    }
}
