class Solution {

    public String method1(String s) {
        s = s.trim().replaceAll(" +", " ");
        String[] arr = s.split(" ");
        StringBuilder var1 = new StringBuilder();
        for (int i = arr.length - 1; i >= 0; i--) {
            var1.append(arr[i]);
            if (i != 0) {
                var1.append(" ");
            }
        }
        return var1.toString();
    }
}
