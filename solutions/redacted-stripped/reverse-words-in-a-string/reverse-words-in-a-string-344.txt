class Solution {

    public String method1(String s) {
        s = s.replaceAll("\\\\s+", " ").trim();
        String[] var1 = s.split(" ");
        int j = var1.length - 1, i = 0;
        while (i < j) {
            String temp = var1[i];
            var1[i] = var1[j];
            var1[j] = temp;
            j--;
            i++;
        }
        s = String.join(" ", var1);
        return s;
    }
}
