class Solution {

    public String method1(String s) {
        String var1 = "";
        if (s != null) {
            s = s.replaceAll("\\\\s+", " ");
            String[] var2 = s.split(" ");
            for (int i = var2.length - 1; i >= 0; i--) var1 += var2[i] + " ";
            var1 = var1.trim();
        }
        return var1;
    }
}
