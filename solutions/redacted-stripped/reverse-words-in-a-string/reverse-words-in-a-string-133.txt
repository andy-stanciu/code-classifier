class Solution {

    public String method1(String s) {
        String[] arr = s.trim().split("\\\\s+");
        int n = arr.length;
        StringBuilder var1 = new StringBuilder();
        for (int i = n - 1; i >= 0; i--) {
            var1.append(arr[i]);
            if (i != 0) {
                var1.append(" ");
            }
        }
        return var1.toString();
    }
}
