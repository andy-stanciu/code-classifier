class Solution {

    public String method1(String s) {
        String[] str = s.trim().split("\\\\s+");
        StringBuilder newStr = new StringBuilder();
        for (int i = str.length - 1; i > 0; i--) {
            newStr.append(str[i]).append(" ");
        }
        newStr.append(str[0]);
        return newStr.toString();
    }
}
