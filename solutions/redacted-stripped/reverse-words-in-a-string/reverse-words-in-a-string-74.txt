class Solution {

    public String method1(String s) {
        String[] arr = s.split(" ");
        StringBuilder str = new StringBuilder();
        System.out.println(arr.length);
        for (int i = arr.length - 1; i >= 0; i--) {
            if (arr[i].length() > 0)
                str.append(arr[i] + " ");
        }
        return str.toString().trim();
    }
}
