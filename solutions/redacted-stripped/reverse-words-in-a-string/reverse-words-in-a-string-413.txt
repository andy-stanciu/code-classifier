class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        int n = var1.length;
        String res = "";
        for (int i = n - 1; i >= 0; i--) {
            if (i < n - 1 && i >= 0)
                res += " ";
            res += var1[i];
        }
        return res;
    }
}
