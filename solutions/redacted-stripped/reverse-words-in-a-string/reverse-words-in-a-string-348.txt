class Solution {

    public String method1(String s) {
        String[] m = s.trim().split("\\\\s+");
        StringBuilder var1 = new StringBuilder();
        for (int i = m.length - 1; i >= 0; i--) {
            var1.append(m[i]);
            if (i != 0) {
                var1.append(" ");
            }
        }
        return var1.toString();
    }
}
