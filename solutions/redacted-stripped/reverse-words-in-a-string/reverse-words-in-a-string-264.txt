class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder newstr = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            newstr.append(var1[i]).append(" ");
        }
        return newstr.toString().trim();
    }
}
