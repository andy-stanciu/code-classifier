class Solution {

    public String method1(String s) {
        s = s.trim();
        var var1 = s.split("\\\\s+");
        var sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            sb.append(var1[i]);
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
