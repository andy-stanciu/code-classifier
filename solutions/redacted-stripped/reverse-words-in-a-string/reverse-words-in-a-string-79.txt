class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" ");
        StringBuilder sb = new StringBuilder();
        int end = var1.length - 1;
        for (int i = 0; i <= end; i++) {
            if (!var1[i].isEmpty()) {
                sb.insert(0, var1[i]);
                if (i < end)
                    sb.insert(0, " ");
            }
        }
        return sb.toString();
    }
}
