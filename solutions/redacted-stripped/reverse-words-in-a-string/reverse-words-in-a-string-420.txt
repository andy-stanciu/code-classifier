class Solution {

    public static String method1(String param1) {
        String var1 = "";
        param1 = param1.trim();
        String[] var2 = param1.split(" ");
        for (int i = var2.length - 1; i >= 0; i--) {
            var1 += var2[i];
            if (i != 0 && !var1.endsWith(" "))
                var1 += " ";
        }
        return var1;
    }
}
