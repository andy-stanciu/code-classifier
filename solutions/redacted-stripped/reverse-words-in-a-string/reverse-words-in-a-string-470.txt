class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] var1 = s.split("\\\\s+");
        Stack<String> stack = new Stack<>();
        for (String w : var1) {
            stack.push(w);
        }
        StringBuilder sb = new StringBuilder();
        while (!stack.isEmpty()) {
            sb.append(stack.pop());
            if (!stack.isEmpty()) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
