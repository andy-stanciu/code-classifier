class Solution {

    public String method1(String s) {
        s = s.trim().replaceAll("\\\\s+", " ");
        String[] sa = s.split(" ");
        return String.join(" ", rev(sa));
    }

    private String[] rev(String[] s) {
        int l = 0;
        int r = s.length - 1;
        while (l < r) {
            String t = s[l];
            s[l] = s[r];
            s[r] = t;
            l++;
            r--;
        }
        return s;
    }
}
