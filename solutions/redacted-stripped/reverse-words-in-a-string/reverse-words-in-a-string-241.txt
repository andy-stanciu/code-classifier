class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] var1 = s.split("\\\\s+");
        int n = var1.length;
        StringBuilder sb = new StringBuilder();
        for (int i = n - 1; i >= 0; i--) {
            sb.append(var1[i]);
            if (i != 0)
                sb.append(" ");
        }
        return sb.toString();
    }
}
