class Solution {

    public String method1(String s) {
        StringBuilder sb = new StringBuilder();
        String[] splited = s.trim().split("\\\\s+");
        for (int i = splited.length - 1; i >= 0; i--) {
            sb.append(splited[i]);
            if (i > 0) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
