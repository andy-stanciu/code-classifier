class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" +");
        System.out.println(Arrays.toString(var1));
        StringBuilder sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            sb.append(var1[i]);
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
