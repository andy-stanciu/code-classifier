class Solution {

    public String method1(String s) {
        String[] str = s.trim().split("\\\\s+");
        int n = str.length - 1;
        String out = "";
        for (int i = n; i > 0; i--) {
            out = out + str[i] + " ";
        }
        out = out + str[0];
        return out;
    }
}
