class Solution {

    public String method1(String s) {
        String[] str = s.split(" ");
        StringBuilder sb = new StringBuilder();
        StringBuilder spaces = new StringBuilder();
        for (int i = str.length - 1; i >= 0; i--) {
            if (!str[i].isEmpty()) {
                if (i > 0)
                    sb.append(str[i] + " ");
                else
                    sb.append(str[i]);
            }
        }
        return sb.toString().trim();
    }
}
