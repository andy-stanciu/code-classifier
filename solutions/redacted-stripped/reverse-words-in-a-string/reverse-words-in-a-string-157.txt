class Solution {

    public String method1(String s) {
        Stack<String> sb = new Stack<String>();
        s = s.trim();
        StringTokenizer stk = new StringTokenizer(s, " ");
        while (stk.hasMoreTokens()) {
            String token = stk.nextToken();
            sb.push(token);
        }
        String op = "";
        while (!sb.isEmpty()) {
            op = op + sb.pop() + " ";
        }
        return op.trim();
    }
}
