class Solution {

    public String method1(String s) {
        s = s.trim().replaceAll("\\\\s+", " ");
        String[] var1 = s.split(" ");
        StringBuilder var2 = new StringBuilder();
        int n = var1.length;
        int i = 1;
        for (int j = n - i; j > 0; j--) {
            var2.append(var1[j]).append(" ");
        }
        var2.append(var1[0]);
        return var2.toString();
    }
}
