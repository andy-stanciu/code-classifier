class Solution {

    public String method1(String s) {
        Stack<String> c = new Stack<>();
        StringTokenizer b = new StringTokenizer(s, " ");
        while (b.hasMoreTokens()) {
            String d = b.nextToken();
            c.push(d);
        }
        StringBuilder res = new StringBuilder();
        while (!c.isEmpty()) {
            res.append(c.pop());
            if (!c.isEmpty()) {
                res.append(" ");
            }
        }
        return res.toString();
    }
}
