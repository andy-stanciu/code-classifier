class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] var1 = s.split("\\\\s+");
        int left = 0, right = var1.length - 1;
        while (left < right) {
            String temp = var1[left];
            var1[left] = var1[right];
            var1[right] = temp;
            left++;
            right--;
        }
        return String.join(" ", var1);
    }
}
