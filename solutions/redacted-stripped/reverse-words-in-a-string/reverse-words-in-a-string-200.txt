class Solution {

    public String method1(String s) {
        String[] sp = s.split(" ");
        String rev = "";
        for (int i = sp.length - 1; i >= 0; i--) {
            if (sp[i] != " ") {
                rev = rev + sp[i];
                rev = rev.trim();
                rev = rev + " ";
            }
        }
        rev = rev.trim();
        return rev;
    }
}
