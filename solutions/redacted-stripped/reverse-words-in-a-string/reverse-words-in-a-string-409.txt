class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] str = s.split(" ");
        String res = "";
        for (int i = str.length - 1; i >= 0; i--) {
            if (str[i].length() != 0)
                res += str[i] + " ";
        }
        return res.trim();
    }
}
