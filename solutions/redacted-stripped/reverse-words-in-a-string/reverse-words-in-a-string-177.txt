class Solution {

    public String method1(String s) {
        s = s.trim();
        StringBuilder ans = new StringBuilder();
        String[] var1 = s.split(" ");
        int r = var1.length - 1;
        while (r >= 0) {
            if (var1[r] != "") {
                ans.append(var1[r]);
                if (r != 0) {
                    ans.append(" ");
                }
            }
            r--;
        }
        return ans.toString();
    }
}
