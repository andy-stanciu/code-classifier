class Solution {

    public String method1(String s) {
        String[] str = s.trim().split(" ");
        StringBuilder result = new StringBuilder();
        for (int i = str.length - 1; i >= 0; --i) {
            if (str[i].trim().isEmpty())
                continue;
            result.append(str[i]);
            if (i > 0)
                result.append(" ");
        }
        return result.toString();
    }
}
