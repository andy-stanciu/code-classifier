class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] word = s.split("\\\\s+");
        StringBuilder var1 = new StringBuilder();
        for (int i = word.length - 1; i >= 0; i--) {
            var1.append(word[i]);
            if (i > 0) {
                var1.append(" ");
            }
        }
        return var1.toString();
    }
}
