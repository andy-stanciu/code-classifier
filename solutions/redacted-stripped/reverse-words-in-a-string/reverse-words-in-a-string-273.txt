class Solution {

    public String method1(String s) {
        StringBuilder str = new StringBuilder();
        s = s.trim();
        String[] arr = s.split("\\\\s+");
        for (int i = arr.length - 1; i >= 0; i--) {
            str.append(arr[i]);
            if (i != 0)
                str.append(" ");
        }
        return str.toString();
    }
}
