class Solution {

    public String method1(String s) {
        String[] str = s.trim().split("\\\\s+");
        String[] var1 = new String[str.length];
        int pos = 0;
        for (int i = str.length - 1; i >= 0; i--) {
            var1[pos] = str[i];
            pos++;
        }
        return String.join(" ", var1);
    }
}
