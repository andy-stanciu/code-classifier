class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] var1 = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].isEmpty()) {
                sb.append(var1[i]);
                if (i != 0) {
                    sb.append(" ");
                }
            }
        }
        return sb.toString().trim();
    }
}
