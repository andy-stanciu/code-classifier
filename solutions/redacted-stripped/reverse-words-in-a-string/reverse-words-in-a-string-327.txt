class Solution {

    public String method1(String s) {
        String[] strArr = s.trim().split("\\\\s+");
        String res = "";
        for (int i = strArr.length - 1; i >= 0; i--) {
            res += strArr[i];
            if (i != 0) {
                res += " ";
            }
        }
        return res;
    }
}
