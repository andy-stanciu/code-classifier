class Solution {

    public String method1(String s) {
        String[] str = s.split(" ");
        String result = "";
        for (int i = str.length - 1; i >= 0; i--) {
            if (str[i] != "") {
                result = result + str[i] + " ";
            }
        }
        result = result.trim();
        return result;
    }
}
