class Solution {

    public String method1(String s) {
        String[] arr = s.trim().split("\\\\s+");
        StringBuffer s2 = new StringBuffer();
        for (int i = arr.length - 1; i >= 0; i--) {
            s2.append(arr[i]).append(" ");
        }
        return s2.toString().trim();
    }
}
