class Solution {

    public String method1(String s) {
        if (s.length() == 0) {
            return "";
        }
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder var2 = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            var2.append(var1[i]);
            if (i > 0) {
                var2.append(" ");
            }
        }
        return var2.toString();
    }
}
