class Solution {

    public String method1(String s) {
        String[] wordArray = s.split("\\\\s+");
        int n = wordArray.length;
        StringBuilder sb = new StringBuilder("");
        for (int i = n - 1; i >= 0; i--) {
            sb.append(wordArray[i] + " ");
        }
        String ans = sb.toString();
        return ans.trim();
    }
}
