class Solution {

    public String method1(String s) {
        String[] str = s.trim().split(" ");
        String str1 = "", sp = "";
        for (int i = str.length - 1; i >= 0; i--) {
            sp = (i != 0) ? " " : "";
            if (str[i] != "")
                str1 += str[i] + sp;
        }
        return str1;
    }
}
