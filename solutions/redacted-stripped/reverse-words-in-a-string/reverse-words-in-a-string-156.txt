class Solution {

    public String method1(String s) {
        String[] str = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = str.length - 1; i >= 0; i--) {
            if (!str[i].equals("")) {
                sb.append(str[i]);
                sb.append(" ");
            }
        }
        return sb.toString().trim();
    }
}
