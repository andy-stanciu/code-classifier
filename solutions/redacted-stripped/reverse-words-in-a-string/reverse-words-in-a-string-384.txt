class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] strs = s.split(" ");
        List<String> strList = Arrays.asList(strs);
        Collections.reverse(strList);
        StringBuilder strb = new StringBuilder();
        for (String str : strList) {
            if (!str.isEmpty())
                strb.append(str + " ");
        }
        return strb.toString().trim();
    }
}
