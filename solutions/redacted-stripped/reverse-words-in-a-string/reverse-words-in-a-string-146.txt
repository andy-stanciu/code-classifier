class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        int st = 0;
        int e = var1.length - 1;
        while (st < e) {
            String temp = var1[st];
            var1[st] = var1[e];
            var1[e] = temp;
            st++;
            e--;
        }
        String str = String.join(" ", var1);
        return str;
    }
}
