class Solution {

    public String method1(String s) {
        s = s.trim();
        StringBuilder var1 = new StringBuilder();
        String[] var2 = s.split("\\\\s+");
        for (int i = var2.length - 1; i >= 0; i--) {
            var1.append(var2[i]);
            if (i > 0) {
                var1.append(" ");
            }
        }
        return var1.toString();
    }
}
