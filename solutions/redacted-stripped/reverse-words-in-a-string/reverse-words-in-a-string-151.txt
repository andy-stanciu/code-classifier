class Solution {

    public String method1(String s) {
        if (s.length() == 1) {
            return s;
        }
        String[] array = s.split("\\\\s+");
        StringBuilder result = new StringBuilder();
        for (int i = array.length - 1; i >= 0; i--) {
            result.append(array[i]);
            result.append(" ");
        }
        return result.toString().trim();
    }
}
