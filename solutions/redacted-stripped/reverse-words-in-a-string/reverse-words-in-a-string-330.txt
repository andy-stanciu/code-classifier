class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder sen = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            sen.append(var1[i]);
            if (i != 0) {
                sen.append(" ");
            }
        }
        return sen.toString();
    }
}
