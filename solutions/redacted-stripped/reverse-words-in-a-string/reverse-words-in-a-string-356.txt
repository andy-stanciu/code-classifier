class Solution {

    public String method1(String s) {
        int n = s.length();
        s = s.trim();
        String[] var1 = s.split("\\\\s+");
        int left = 0;
        int right = var1.length - 1;
        while (left < right) {
            String temp = var1[left];
            var1[left] = var1[right];
            var1[right] = temp;
            left++;
            right--;
        }
        String result = String.join(" ", var1);
        return result;
    }
}
