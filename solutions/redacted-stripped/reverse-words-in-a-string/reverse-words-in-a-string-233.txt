class Solution {

    public String method1(String s) {
        s = s.trim().replaceAll("\\\\s+", " ");
        String[] var1 = s.split(" ");
        StringBuilder strRev = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (i < var1.length - 1) {
                strRev.append(" ");
            }
            strRev.append(var1[i]);
        }
        return strRev.toString();
    }
}
