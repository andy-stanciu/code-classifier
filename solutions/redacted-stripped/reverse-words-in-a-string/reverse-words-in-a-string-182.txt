class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder var2 = new StringBuilder();
        for (int iterator = var1.length - 1; iterator >= 0; iterator--) {
            var2.append(var1[iterator]);
            if (iterator != 0) {
                var2.append(" ");
            }
        }
        return var2.toString();
    }
}
