class Solution {

    public String method1(String s) {
        s.trim();
        String[] var1 = s.split("\\\\s+");
        StringBuilder rev = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            rev.append(var1[i] + " ");
        }
        return rev.toString().trim();
    }
}
