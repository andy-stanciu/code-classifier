class Solution {

    public String method1(String s) {
        String[] str = s.trim().split("\\\\s+");
        String res = "";
        StringBuilder sb = new StringBuilder();
        for (int i = str.length - 1; i >= 0; i--) {
            res = sb.append(str[i] + " ").toString().trim();
        }
        return res;
    }
}
