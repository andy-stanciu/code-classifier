class Solution {

    public String method1(String s) {
        String[] palabras = s.split(" ");
        String resultado = "";
        for (int i = palabras.length - 1; i >= 0; i--) {
            if (palabras[i].isEmpty())
                continue;
            resultado = resultado.concat(palabras[i]).concat(" ");
        }
        return resultado.trim();
    }
}
