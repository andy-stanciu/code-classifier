class Solution {

    public String method1(String s) {
        String trim = s.trim();
        String[] arr = trim.split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = arr.length - 1; i >= 0; i--) {
            sb.append(arr[i]).append(" ");
        }
        return sb.toString().trim();
    }
}
