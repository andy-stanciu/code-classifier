class Solution {

    public String method1(String s) {
        String[] parts = s.split("\\\\s+");
        String var1 = parts[parts.length - 1];
        for (int i = parts.length - 2; i > -1; i--) {
            if (parts[i] == "") {
                continue;
            }
            var1 = var1 + " " + parts[i];
        }
        return var1;
    }
}
