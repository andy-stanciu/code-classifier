class Solution {

    public String method1(String s) {
        StringBuilder result = new StringBuilder();
        s = s.strip();
        String[] word_arr = s.split(" ");
        for (int i = word_arr.length - 1; i >= 0; i--) {
            word_arr[i] = word_arr[i].strip();
            if (word_arr[i].length() != 0) {
                if (i == 0)
                    result.append(word_arr[i]);
                else {
                    result.append(word_arr[i]);
                    result.append(" ");
                }
            }
        }
        return result.toString();
    }
}
