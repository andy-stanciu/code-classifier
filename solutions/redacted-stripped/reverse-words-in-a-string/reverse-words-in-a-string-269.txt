class Solution {

    public String method1(String s) {
        ;
        String[] modified = s.trim().split(" ");
        String result = "";
        for (int i = modified.length - 1; i >= 0; i--) {
            if (i != 0 && modified[i].isEmpty() == false) {
                result += modified[i] + " ";
            } else if (modified[i].isEmpty() == false) {
                result += modified[i];
            }
        }
        return result;
    }
}
