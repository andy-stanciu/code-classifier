class Solution {

    public String method1(String s) {
        s = s.trim();
        ArrayList<String> strArr = new ArrayList<>(Arrays.asList(s.split(" ")));
        int m = strArr.size();
        for (int i = m - 1; i >= 0; i--) {
            if (strArr.get(i).trim().isEmpty()) {
                strArr.remove(i);
            }
        }
        int n = strArr.size();
        String[] revStr = new String[n];
        for (int i = 0; i <= n - 1; i++) {
            revStr[i] = strArr.get(n - 1 - i);
        }
        String revS1 = String.join(" ", revStr);
        return revS1;
    }
}
