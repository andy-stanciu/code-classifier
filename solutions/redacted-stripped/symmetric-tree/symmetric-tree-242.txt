class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return isMirror(root.left, root.right);
    }

    public boolean isMirror(TreeNode p, TreeNode q) {
        if (p == q)
            return true;
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        if (p.val != q.val)
            return false;
        if (isMirror(p.left, q.right) == false)
            return false;
        if (isMirror(p.right, q.left) == false)
            return false;
        return true;
    }
}
