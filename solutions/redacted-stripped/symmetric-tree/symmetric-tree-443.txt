class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return method1(root.left, root.right);
    }

    public boolean method1(TreeNode node1, TreeNode node2) {
        if (node1 == null && node2 == null)
            return true;
        if (node1 == null || node2 == null)
            return false;
        return (node1.val == node2.val) && method1(node1.left, node2.right) && method1(node1.right, node2.left);
    }
}
