class Solution {

    public boolean method1(TreeNode root) {
        List<Integer> list1 = new ArrayList<>();
        List<Integer> list2 = new ArrayList<>();
        list1 = Test1(root.left, list1);
        list2 = Test2(root.right, list2);
        System.out.println(list1);
        System.out.println(list2);
        if (list1.equals(list2)) {
            return true;
        }
        return false;
    }

    public List<Integer> Test1(TreeNode p, List<Integer> list) {
        if (p == null) {
            list.add(null);
            return list;
        }
        if (p != null) {
            list.add(p.val);
            Test1(p.left, list);
            Test1(p.right, list);
        }
        return list;
    }

    public List<Integer> Test2(TreeNode p, List<Integer> list) {
        if (p == null) {
            list.add(null);
            return list;
        }
        if (p != null) {
            list.add(p.val);
            Test2(p.right, list);
            Test2(p.left, list);
        }
        return list;
    }
}
