class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return mirror(root.left, root.right);
    }

    public boolean mirror(TreeNode lroot, TreeNode rroot) {
        if (lroot == null || rroot == null)
            return lroot == rroot;
        if (lroot.val != rroot.val)
            return false;
        return mirror(lroot.left, rroot.right) && mirror(lroot.right, rroot.left);
    }
}
