class Solution {

    public boolean method1(TreeNode root) {
        if (root != null) {
            return ch(root.left, root.right);
        } else
            return true;
    }

    public boolean ch(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        return (p.val == q.val && ch(p.left, q.right) && ch(p.right, q.left));
    }
}
