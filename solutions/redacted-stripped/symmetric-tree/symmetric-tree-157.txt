class Solution {

    public boolean method1(TreeNode root) {
        boolean b = check(root.left, root.right);
        return b;
    }

    public boolean check(TreeNode l, TreeNode r) {
        if (l == null && r == null)
            return true;
        if (l == null || r == null)
            return false;
        return r.val == l.val && check(l.left, r.right) && check(l.right, r.left);
    }
}
