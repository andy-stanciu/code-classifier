class Solution {

    static boolean method1(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            if (left == null && right == null)
                return true;
            else
                return false;
        }
        if (left.val != right.val)
            return false;
        boolean a = method1(left.left, right.right);
        boolean b = method1(left.right, right.left);
        return a && b;
    }

    public boolean method2(TreeNode root) {
        return method1(root.left, root.right);
    }
}
