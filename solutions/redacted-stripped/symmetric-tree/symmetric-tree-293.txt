class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return check(root.left, root.right);
    }

    boolean check(TreeNode param1, TreeNode param2) {
        if (param1 == null && param2 == null)
            return true;
        if (param1 == null || param2 == null)
            return false;
        if (param1.val != param2.val)
            return false;
        return check(param1.left, param2.right) && check(param1.right, param2.left);
    }
}
