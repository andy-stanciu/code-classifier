class Solution {

    public boolean method1(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (p == null || q == null) {
            return false;
        }
        if (p.val != q.val) {
            return false;
        }
        return method1(p.left, q.right) && method1(p.right, q.left);
    }

    public boolean method2(TreeNode root) {
        if (root.left == null && root.right == null)
            return true;
        if (root.left == null || root.right == null)
            return false;
        boolean x = method1(root.right, root.left);
        return x;
    }
}
