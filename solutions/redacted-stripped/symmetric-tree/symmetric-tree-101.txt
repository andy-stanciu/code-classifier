class Solution {

    public boolean method1(TreeNode root) {
        return helper(root.left, root.right);
    }

    private boolean helper(TreeNode p, TreeNode q) {
        if (p == null || q == null)
            return p == q;
        return p.val == q.val && helper(p.left, q.right) && helper(p.right, q.left);
    }
}
