class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode t1, TreeNode t2) {
        if (t1 == null && t2 == null) {
            return true;
        }
        if (t1 == null || t2 == null) {
            return false;
        }
        if (t1.val == t2.val) {
            return method2(t1.left, t2.right) && method2(t1.right, t2.left);
        }
        return false;
    }
}
