class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return helper(root.left, root.right, true);
    }

    boolean helper(TreeNode left, TreeNode right, boolean result) {
        if (left == null && right == null)
            return result;
        if (left == null || right == null)
            return false;
        if (left.val == right.val) {
            result = helper(left.left, right.right, result);
            result = helper(left.right, right.left, result);
        } else {
            return false;
        }
        return result;
    }
}
