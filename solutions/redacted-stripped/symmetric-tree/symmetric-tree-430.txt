class Solution {

    public boolean method1(TreeNode root1, TreeNode root2) {
        if (root1 == null && root2 != null || root1 != null && root2 == null)
            return false;
        if (root1 == null && root2 == null)
            return true;
        if (root1.val != root2.val)
            return false;
        boolean a = method1(root1.left, root2.right);
        boolean b = method1(root1.right, root2.left);
        if (a == false || b == false)
            return false;
        return true;
    }

    public boolean method2(TreeNode root) {
        if (root == null)
            return true;
        return method1(root.left, root.right);
    }
}
