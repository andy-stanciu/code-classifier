class Solution {

    public boolean mirror(TreeNode param1, TreeNode param2) {
        if (param1 == null && param2 == null) {
            return true;
        }
        if (param1 == null || param2 == null) {
            return false;
        }
        return (param1.val == param2.val) && mirror(param1.left, param2.right) && mirror(param1.right, param2.left);
    }

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return mirror(root.left, root.right);
    }
}
