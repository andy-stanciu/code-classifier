class Solution {

    public boolean method1(TreeNode root) {
        return isSame(root.left, root.right);
    }

    private boolean isSame(TreeNode p, TreeNode q) {
        if (p == null || q == null) {
            return p == q;
        }
        return (p.val == q.val) && isSame(p.left, q.right) && isSame(p.right, q.left);
    }
}
