class Solution {

    public boolean method1(TreeNode root) {
        return isSymUtil(root.left, root.right);
    }

    public boolean isSymUtil(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            return left == right ? true : false;
        }
        return (left.val == right.val) && isSymUtil(left.left, right.right) && isSymUtil(right.left, left.right);
    }
}
