class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return helper(root.left, root.right);
    }

    boolean helper(TreeNode n1, TreeNode n2) {
        if (n1 == null && n2 == null) {
            return true;
        }
        if (n1 == null || n2 == null) {
            return false;
        }
        return (n1.val == n2.val) && helper(n1.left, n2.right) && helper(n1.right, n2.left);
    }
}
