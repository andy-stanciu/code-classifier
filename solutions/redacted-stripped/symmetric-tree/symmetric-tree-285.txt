class Solution {

    public boolean method1(TreeNode root) {
        return method1(root, root);
    }

    private boolean method1(TreeNode p, TreeNode q) {
        if (p == null || q == null)
            return p == q;
        return p.val == q.val && method1(p.left, q.right) && method1(p.right, q.left);
    }
}
