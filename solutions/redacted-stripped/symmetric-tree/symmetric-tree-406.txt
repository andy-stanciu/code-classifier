class Solution {

    boolean isSame = true;

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        inOrderTraversal(root.left, root.right);
        return isSame;
    }

    private void inOrderTraversal(TreeNode p, TreeNode q) {
        if (!isSame) {
            return;
        }
        if (p == null && q == null) {
            return;
        }
        if (p == null || q == null || p.val != q.val) {
            isSame = false;
            return;
        }
        inOrderTraversal(p.left, q.right);
        inOrderTraversal(p.right, q.left);
    }
}
