class Solution {

    public boolean method1(TreeNode root) {
        return root == null || isSymetricRecursive(root.left, root.right);
    }

    private boolean isSymetricRecursive(TreeNode root1, TreeNode root2) {
        if (root1 == null || root2 == null)
            return root1 == root2;
        if (root1.val != root2.val)
            return false;
        return isSymetricRecursive(root1.left, root2.right) && isSymetricRecursive(root1.right, root2.left);
    }
}
