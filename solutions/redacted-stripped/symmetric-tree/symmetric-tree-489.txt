class Solution {

    public boolean chekMirror(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null) {
            return false;
        }
        if (left.val == right.val) {
            return chekMirror(left.left, right.right) && chekMirror(left.right, right.left);
        } else {
            return false;
        }
    }

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return chekMirror(root.left, root.right);
    }
}
