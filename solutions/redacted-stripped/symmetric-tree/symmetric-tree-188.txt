class Solution {

    public boolean method1(TreeNode root) {
        return compare(root.left, root.right);
    }

    public boolean compare(TreeNode a, TreeNode b) {
        if ((a == null) != (b == null)) {
            return false;
        }
        if (a != null) {
            if (a.val != b.val) {
                return false;
            }
            return compare(a.left, b.right) && compare(a.right, b.left);
        }
        return true;
    }
}
