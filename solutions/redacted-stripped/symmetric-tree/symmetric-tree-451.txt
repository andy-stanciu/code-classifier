class Solution {

    public boolean isMirror(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p != null && q == null)
            return false;
        if (p == null && q != null)
            return false;
        if (p.val != q.val)
            return false;
        boolean l = isMirror(p.left, q.right);
        boolean r = isMirror(p.right, q.left);
        if (!l || !r)
            return false;
        return true;
    }

    public boolean method1(TreeNode root) {
        return isMirror(root.left, root.right);
    }
}
