class Solution {

    public static boolean fn(TreeNode r, TreeNode l) {
        if (r == null && l == null)
            return true;
        if (r == null || l == null)
            return false;
        if (r.val != l.val)
            return false;
        return (fn(r.left, l.right) && fn(r.right, l.left));
    }

    public boolean method1(TreeNode r) {
        return fn(r.left, r.right);
    }
}
