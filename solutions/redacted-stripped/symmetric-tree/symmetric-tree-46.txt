class Solution {

    public boolean method1(TreeNode root) {
        return task(root.left, root.right) && task(root.right, root.left);
    }

    public boolean task(TreeNode left, TreeNode right) {
        if (left == null && right == null)
            return true;
        if (left == null || right == null)
            return false;
        if (left.val != right.val)
            return false;
        return task(left.left, right.right) && task(right.right, left.left);
    }
}
