class Solution {

    public boolean method1(TreeNode rLeft, TreeNode rRight) {
        if (rLeft == null && rRight == null) {
            return true;
        }
        if (rLeft == null || rRight == null) {
            return false;
        }
        boolean left = method1(rLeft.left, rRight.right);
        boolean right = method1(rLeft.right, rRight.left);
        if (rLeft.val == rRight.val) {
            if (left && right) {
                return true;
            }
        }
        return false;
    }

    public boolean method2(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method1(root.left, root.right);
    }
}
