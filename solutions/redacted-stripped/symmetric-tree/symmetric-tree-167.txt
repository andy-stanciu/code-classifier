class Solution {

    public boolean method1(TreeNode root) {
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode leftNode, TreeNode rightNode) {
        if (leftNode == null && rightNode == null)
            return true;
        if ((leftNode == null && rightNode != null) || (leftNode != null && rightNode == null) || (leftNode.val != rightNode.val))
            return false;
        return method2(leftNode.left, rightNode.right) && method2(leftNode.right, rightNode.left);
    }
}
