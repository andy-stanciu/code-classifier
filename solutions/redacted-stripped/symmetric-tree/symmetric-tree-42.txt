class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode l, TreeNode r) {
        if (l == null || r == null)
            return (l == r);
        if (l.val != r.val) {
            return false;
        }
        return method2(l.left, r.right) && method2(l.right, r.left);
    }
}
