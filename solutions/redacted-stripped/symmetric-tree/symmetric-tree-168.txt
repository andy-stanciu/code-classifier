class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        if (root.left == null && root.right == null)
            return true;
        return method2(root.left, root.right);
    }

    static boolean method2(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        else if (p == null) {
            return false;
        } else if (q == null)
            return false;
        else if (p.val != q.val)
            return false;
        else
            return method2(p.left, q.right) && method2(p.right, q.left);
    }
}
