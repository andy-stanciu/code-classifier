class Solution {

    void invert(TreeNode root) {
        if (root == null)
            return;
        invert(root.left);
        invert(root.right);
        TreeNode temp = root.left;
        root.left = root.right;
        root.right = temp;
    }

    boolean isEqual(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        return isEqual(p.left, q.left) && isEqual(p.right, q.right) && (p.val == q.val);
    }

    public boolean method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return true;
        if (root.left == null || root.right == null)
            return false;
        invert(root.left);
        return isEqual(root.left, root.right);
    }
}
