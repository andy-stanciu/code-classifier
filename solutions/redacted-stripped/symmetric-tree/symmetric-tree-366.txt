class Solution {

    public boolean method1(TreeNode root) {
        TreeNode leftNode = root.left;
        TreeNode rightNode = root.right;
        return method2(leftNode, rightNode);
    }

    boolean method2(TreeNode leftNode, TreeNode rightNode) {
        if (leftNode == null && rightNode == null)
            return true;
        else if (leftNode == null || rightNode == null)
            return false;
        else if (leftNode.val != rightNode.val)
            return false;
        return method2(leftNode.left, rightNode.right) && method2(leftNode.right, rightNode.left);
    }
}
