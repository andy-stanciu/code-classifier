class Solution {

    public boolean method1(TreeNode root) {
        return isSame(root.left, root.right);
    }

    private boolean isSame(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (p == null || q == null) {
            return false;
        }
        if (p.val != q.val) {
            return false;
        }
        return isSame(p.left, q.right) && isSame(p.right, q.left);
    }
}
