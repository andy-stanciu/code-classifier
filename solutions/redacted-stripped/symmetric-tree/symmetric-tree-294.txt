class Solution {

    public boolean method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return true;
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode root1, TreeNode root2) {
        if (root1 == null && root2 == null)
            return true;
        if ((root1 != null && root2 == null) || (root1 == null && root2 != null))
            return false;
        boolean leftright = method2(root1.left, root2.right);
        boolean rightleft = method2(root1.right, root2.left);
        if (!(root1.val == root2.val) || !leftright || !rightleft)
            return false;
        return true;
    }
}
