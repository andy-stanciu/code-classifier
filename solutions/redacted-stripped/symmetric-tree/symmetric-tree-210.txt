class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        boolean ans = issame(root.left, root.right);
        return ans;
    }

    private boolean issame(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (q != null && p != null && p.val == q.val) {
            return issame(p.left, q.right) && issame(p.right, q.left);
        }
        return false;
    }
}
