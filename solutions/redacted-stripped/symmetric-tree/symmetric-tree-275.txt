class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return (method1(root.left, root.right));
    }

    private boolean method1(TreeNode leftRoot, TreeNode rightRoot) {
        if (leftRoot == null || rightRoot == null)
            return (leftRoot == rightRoot);
        return (leftRoot.val == rightRoot.val && method1(leftRoot.left, rightRoot.right) && method1(leftRoot.right, rightRoot.left));
    }
}
