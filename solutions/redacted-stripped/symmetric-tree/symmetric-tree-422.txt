class Solution {

    public boolean method1(TreeNode root) {
        return root == null || isMirror(root.right, root.left);
    }

    boolean isMirror(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (p == null || q == null) {
            return false;
        }
        return (p.val == q.val) && isMirror(p.left, q.right) && isMirror(p.right, q.left);
    }
}
