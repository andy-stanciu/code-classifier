class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return isEqual(root.left, root.right);
    }

    public Boolean isEqual(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (p == null || q == null) {
            return false;
        }
        if (p == q && isEqual(p.left, q.right) && isEqual(p.right, q.left)) {
            return true;
        }
        return p.val == q.val && isEqual(p.left, q.right) && isEqual(p.right, q.left);
    }
}
