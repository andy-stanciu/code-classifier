class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        if (isSame(root.left, root.right)) {
            return true;
        }
        return false;
    }

    public boolean isSame(TreeNode one, TreeNode two) {
        if (one == null && two == null) {
            return true;
        }
        if (one == null || two == null) {
            return false;
        }
        if (one.val != two.val) {
            return false;
        }
        return isSame(one.left, two.right) && isSame(one.right, two.left);
    }
}
