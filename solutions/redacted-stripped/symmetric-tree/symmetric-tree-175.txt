class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return fun(root.left, root.right);
    }

    static boolean fun(TreeNode lroot, TreeNode rroot) {
        if (lroot == null && rroot == null)
            return true;
        if (lroot == null || rroot == null || lroot.val != rroot.val)
            return false;
        return fun(lroot.left, rroot.right) && fun(lroot.right, rroot.left);
    }
}
