class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method2(root.left, root.right);
    }

    public static boolean method2(TreeNode l, TreeNode r) {
        if (l == null && r == null) {
            return true;
        }
        if (l == null || r == null) {
            return false;
        }
        return (l.val == r.val) && (method2(l.left, r.right)) && (method2(l.right, r.left));
    }
}
