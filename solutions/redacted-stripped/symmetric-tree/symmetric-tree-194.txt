class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return checkSum(root.left, root.right);
    }

    public boolean checkSum(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null || left.val != right.val) {
            return false;
        }
        return checkSum(left.left, right.right) && checkSum(left.right, right.left);
    }
}
