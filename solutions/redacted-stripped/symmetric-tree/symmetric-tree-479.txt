class Solution {

    public boolean method1(TreeNode root) {
        TreeNode root2 = root;
        return isTrue(root, root2);
    }

    public boolean isTrue(TreeNode root, TreeNode root2) {
        if (root == null && root2 == null)
            return true;
        if (root == null && root2 != null)
            return false;
        if (root != null && root2 == null)
            return false;
        if (root.val == root2.val) {
            return isTrue(root.left, root2.right) && isTrue(root.right, root2.left);
        }
        return false;
    }
}
