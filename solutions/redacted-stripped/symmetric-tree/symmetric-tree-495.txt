class Solution {

    public boolean checker(TreeNode x, TreeNode y) {
        if (x == null && y == null)
            return true;
        if (x == null || y == null || x.val != y.val)
            return false;
        return checker(x.left, y.right) && checker(y.left, x.right);
    }

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return checker(root.left, root.right);
    }
}
