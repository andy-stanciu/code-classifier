class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return isMirrorRoot(root.left, root.right);
    }

    public boolean isMirrorRoot(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null) {
            return false;
        }
        return (left.val == right.val) && isMirrorRoot(left.right, right.left) && isMirrorRoot(left.left, right.right);
    }
}
