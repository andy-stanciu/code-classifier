class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return method2(root.left, root.right);
    }

    private boolean method2(TreeNode root1, TreeNode root2) {
        if (root1 == null && root2 == null)
            return true;
        if (root1 == null || root2 == null)
            return false;
        if (root1.val != root2.val)
            return false;
        return method2(root1.left, root2.right) && method2(root1.right, root2.left);
    }
}
