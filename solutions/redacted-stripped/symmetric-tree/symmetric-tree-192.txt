class Solution {

    private boolean dfs(TreeNode l, TreeNode r) {
        if (l == null && r == null)
            return true;
        if (l == null || r == null || l.val != r.val)
            return false;
        return dfs(l.left, r.right) && dfs(l.right, r.left);
    }

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return dfs(root.left, root.right);
    }
}
