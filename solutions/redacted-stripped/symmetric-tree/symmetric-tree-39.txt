class Solution {

    public boolean method1(TreeNode root) {
        return root == null || helperFn(root.left, root.right);
    }

    public boolean helperFn(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            return left == right;
        }
        if (left.val != right.val)
            return false;
        return helperFn(left.left, right.right) && helperFn(left.right, right.left);
    }
}
