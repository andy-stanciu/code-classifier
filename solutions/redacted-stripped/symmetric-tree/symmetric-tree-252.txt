class Solution {

    public boolean method1(TreeNode root) {
        return helper(root.left, root.right);
    }

    private boolean helper(TreeNode l, TreeNode r) {
        if (l == null && r == null)
            return true;
        if (l == null && r != l)
            return false;
        if (r == null && l != r)
            return false;
        if (l.val != r.val)
            return false;
        boolean left = helper(l.left, r.right);
        boolean right = helper(l.right, r.left);
        return (left && right);
    }
}
