class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return false;
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode left1, TreeNode right1) {
        if (left1 == null && right1 == null)
            return true;
        else if (left1 == null || right1 == null)
            return false;
        else if (left1.val != right1.val)
            return false;
        else
            return method2(left1.left, right1.right) && method2(left1.right, right1.left);
        return;
    }
}
