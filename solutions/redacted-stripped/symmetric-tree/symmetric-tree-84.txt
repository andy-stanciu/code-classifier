class Solution {

    public boolean method1(TreeNode root) {
        return method1(root.left, root.right);
    }

    public boolean method1(TreeNode rootLeft, TreeNode rootRight) {
        if (rootLeft == null && rootRight == null) {
            return true;
        }
        if (rootLeft == null || rootRight == null) {
            return false;
        }
        if (rootLeft.val != rootRight.val) {
            return false;
        } else
            return method1(rootLeft.right, rootRight.left) && method1(rootLeft.left, rootRight.right);
    }
}
