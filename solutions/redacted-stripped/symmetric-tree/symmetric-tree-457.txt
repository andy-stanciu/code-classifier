class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return checkSymmetry(root.left, root.right);
    }

    public static boolean checkSymmetry(TreeNode n1, TreeNode n2) {
        if (n1 == null || n2 == null) {
            return n1 == n2;
        }
        return (n1.val == n2.val) && checkSymmetry(n1.left, n2.right) && checkSymmetry(n1.right, n2.left);
    }
}
