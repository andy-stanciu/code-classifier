class Solution {

    public boolean method1(TreeNode n1, TreeNode n2) {
        if (n1 == null || n2 == null) {
            return n1 == n2;
        }
        boolean c1 = method1(n1.left, n2.right);
        boolean c2 = method1(n1.right, n2.left);
        return (n1.val == n2.val) && c1 && c2;
    }

    public boolean method2(TreeNode root) {
        if (root == null) {
            return false;
        }
        boolean check = method1(root.left, root.right);
        return check;
    }
}
