class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return helper(root.left, root.right);
    }

    boolean check = false;

    private boolean helper(TreeNode rL, TreeNode rR) {
        if (rL == null && rR != null) {
            return check;
        }
        if (rR == null && rL != null) {
            return check;
        }
        if (rL != null && rR != null) {
            if (rL.val != rR.val) {
                return check;
            } else {
                return helper(rL.left, rR.right) && helper(rL.right, rR.left);
            }
        }
        return true;
    }
}
