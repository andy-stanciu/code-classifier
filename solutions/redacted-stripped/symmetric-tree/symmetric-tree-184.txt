class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method1(root.left, root.right);
    }

    private boolean method1(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null) {
            return false;
        }
        return left.val == right.val && method1(left.left, right.right) && method1(left.right, right.left);
    }
}
