class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode lef, TreeNode righ) {
        if (lef == null && righ == null)
            return true;
        if (lef == null || righ == null)
            return false;
        if (lef.val != righ.val)
            return false;
        return method2(lef.left, righ.right) && method2(lef.right, righ.left);
    }
}
