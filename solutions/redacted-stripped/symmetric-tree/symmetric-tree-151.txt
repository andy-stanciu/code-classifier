class Solution {

    public boolean method1(TreeNode root) {
        boolean ans = sameornot(root.left, root.right);
        return ans;
    }

    public boolean sameornot(TreeNode p, TreeNode q) {
        if (p == null || q == null) {
            return p == q;
        }
        return (p.val == q.val && sameornot(p.left, q.right) && sameornot(p.right, q.left));
    }
}
