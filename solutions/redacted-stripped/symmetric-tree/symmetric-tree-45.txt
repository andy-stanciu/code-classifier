class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return ans(root.left, root.right);
    }

    public boolean ans(TreeNode l, TreeNode r) {
        if (l == null && r == null)
            return true;
        if (l == null ^ r == null)
            return false;
        return (l.val == r.val) && ans(l.left, r.right) && ans(l.right, r.left);
    }
}
