class Solution {

    public boolean method1(TreeNode root) {
        return isEqual(root.left, root.right);
    }

    public boolean isEqual(TreeNode p, TreeNode q) {
        if (p == q) {
            return true;
        }
        if (p == null) {
            return false;
        }
        if (q == null) {
            return false;
        }
        if (p.val != q.val) {
            return false;
        }
        if (!isEqual(p.left, q.right)) {
            return false;
        }
        if (!isEqual(p.right, q.left)) {
            return false;
        }
        return true;
    }
}
