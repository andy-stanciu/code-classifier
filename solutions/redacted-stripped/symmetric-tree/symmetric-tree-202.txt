class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return sys(root.left, root.right);
    }

    public static boolean sys(TreeNode a, TreeNode b) {
        if (a == null && b == null) {
            return true;
        }
        if (a == null || b == null) {
            return false;
        }
        return (a.val == b.val) && sys(a.left, b.right) && sys(a.right, b.left);
    }
}
