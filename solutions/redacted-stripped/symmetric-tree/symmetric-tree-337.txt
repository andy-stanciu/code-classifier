class Solution {

    public boolean method1(TreeNode root) {
        return isMirror(root, root);
    }

    public boolean isMirror(TreeNode a, TreeNode b) {
        if (a == null && b == null) {
            return true;
        }
        if (a != null && b != null && a.val == b.val) {
            return isMirror(a.left, b.right) && isMirror(a.right, b.left);
        }
        return false;
    }
}
