class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return false;
        }
        return SymetricHelp(root.left, root.right);
    }

    boolean SymetricHelp(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            return left == right;
        }
        if (left.val != right.val)
            return false;
        return SymetricHelp(left.left, right.right) && SymetricHelp(left.right, right.left);
    }
}
