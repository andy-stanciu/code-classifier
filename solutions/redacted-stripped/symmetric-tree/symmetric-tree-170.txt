class Solution {

    public boolean method1(TreeNode root) {
        if (root != null) {
            return method2(root.left, root.right);
        } else {
            return true;
        }
    }

    public boolean method2(TreeNode left, TreeNode right) {
        if (left == null && right != null || left != null && right == null) {
            return false;
        } else if (left == null && right == null) {
            return true;
        } else {
            if (left.val == right.val) {
                boolean x = method2(left.left, right.right);
                boolean y = method2(left.right, right.left);
                return (x & y);
            } else {
                return false;
            }
        }
    }
}
