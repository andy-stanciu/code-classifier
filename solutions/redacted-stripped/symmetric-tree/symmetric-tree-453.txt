class Solution {

    public boolean method1(TreeNode root) {
        return symetric(root.left, root.right);
    }

    public boolean symetric(TreeNode left, TreeNode right) {
        if (left == null && right == null)
            return true;
        if (left == null || right == null)
            return false;
        if (left.val == right.val)
            return true && symetric(left.left, right.right) && symetric(left.right, right.left);
        else
            return false && symetric(left.left, right.right) && symetric(left.right, right.left);
    }
}
