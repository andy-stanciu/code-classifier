class Solution {

    private boolean method1(TreeNode node1, TreeNode node2) {
        if (node1 != null && node2 != null) {
            boolean a = method1(node1.left, node2.right);
            boolean b = method1(node1.right, node2.left);
            if (node1.val == node2.val && a && b) {
                return true;
            } else {
                return false;
            }
        }
        if (node1 == null && node2 == null) {
            return true;
        } else {
            return false;
        }
    }

    public boolean method2(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method1(root.left, root.right);
    }
}
