class Solution {

    public boolean method1(TreeNode rootL, TreeNode rootR) {
        if (rootL == null || rootR == null) {
            return rootL == rootR;
        }
        return (rootL.val == rootR.val && method1(rootL.left, rootR.right) && method1(rootL.right, rootR.left));
    }

    public boolean method2(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method1(root.left, root.right);
    }
}
