class Solution {

    public boolean method1(TreeNode root) {
        return isMirror(root.left, root.right);
    }

    public boolean isMirror(TreeNode T1, TreeNode T2) {
        if (T1 == null && T2 == null)
            return true;
        if (T1 == null || T2 == null)
            return false;
        if (T1.val != T2.val)
            return false;
        return isMirror(T1.left, T2.right) && isMirror(T1.right, T2.left);
    }
}
