class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return isMirror(root.left, root.right);
    }

    private static boolean isMirror(TreeNode leftNode, TreeNode rightNode) {
        if (leftNode == null ^ rightNode == null)
            return false;
        if (leftNode == null && rightNode == null)
            return true;
        if (leftNode.val != rightNode.val)
            return false;
        return true && isMirror(leftNode.right, rightNode.left) && isMirror(leftNode.left, rightNode.right);
    }
}
