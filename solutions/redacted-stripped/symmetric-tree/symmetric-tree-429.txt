class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method1(root.left, root.right);
    }

    public boolean method1(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null) {
            return false;
        }
        if (left.val != right.val) {
            return false;
        }
        final boolean var1 = method1(left.left, right.right);
        final boolean var2 = method1(left.right, right.left);
        return var1 && var2;
    }
}
