class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        else
            return ifMirror(root.left, root.right);
    }

    public boolean ifMirror(TreeNode leftNode, TreeNode rightNode) {
        if (leftNode == null && rightNode == null)
            return true;
        else if (leftNode == null && rightNode != null)
            return false;
        else if (leftNode != null && rightNode == null)
            return false;
        else {
            return (leftNode.val == rightNode.val) && ifMirror(leftNode.left, rightNode.right) && ifMirror(leftNode.right, rightNode.left);
        }
    }
}
