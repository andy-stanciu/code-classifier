class Solution {

    boolean sol = true;

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        if (root.left == null && root.right != null)
            return false;
        if (root.right == null && root.left != null)
            return false;
        if (root.left == null && root.right == null)
            return true;
        go(root.left, root.right);
        return sol;
    }

    public void go(TreeNode left, TreeNode right) {
        if (!sol)
            return;
        if (left.val != right.val) {
            sol = false;
            return;
        }
        if ((left.left == null && right.right != null) || (right.right == null && left.left != null) || (right.left == null && left.right != null) || (left.right == null && right.left != null)) {
            sol = false;
            return;
        }
        if (left.right != null && right.left != null)
            go(left.right, right.left);
        if (left.left != null && right.right != null)
            go(left.left, right.right);
    }
}
