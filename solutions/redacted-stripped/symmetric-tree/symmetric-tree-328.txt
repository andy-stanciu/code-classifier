class Solution {

    public boolean method1(TreeNode root) {
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode K, TreeNode L) {
        if ((K == null) && (L == null))
            return true;
        if (K == null || L == null)
            return false;
        if (K.val == L.val) {
            return method2(K.right, L.left) && method2(K.left, L.right);
        }
        return false;
    }
}
