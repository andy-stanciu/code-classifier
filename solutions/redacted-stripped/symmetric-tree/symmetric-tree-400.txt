class Solution {

    public boolean method1(TreeNode root) {
        TreeNode leftRoot = root.left;
        TreeNode rightRoot = root.right;
        return dfs(leftRoot, rightRoot);
    }

    public boolean dfs(TreeNode leftRoot, TreeNode rightRoot) {
        if (leftRoot == null && rightRoot == null) {
            return true;
        } else if (leftRoot == null) {
            return false;
        } else if (rightRoot == null) {
            return false;
        } else if (leftRoot.val != rightRoot.val) {
            return false;
        }
        boolean left = dfs(leftRoot.left, rightRoot.right);
        boolean right = dfs(leftRoot.right, rightRoot.left);
        return left && right;
    }
}
