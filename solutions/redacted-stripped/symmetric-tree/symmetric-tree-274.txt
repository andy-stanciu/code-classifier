class Solution {

    public boolean method1(TreeNode root) {
        return root == null || solve(root.left, root.right);
    }

    public boolean solve(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        if (p.val != q.val)
            return false;
        boolean left = solve(p.left, q.right);
        boolean right = solve(p.right, q.left);
        return left && right;
    }
}
