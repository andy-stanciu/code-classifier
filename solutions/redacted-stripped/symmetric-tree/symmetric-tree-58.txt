class Solution {

    public boolean method1(TreeNode root) {
        return method1(root, root);
    }

    public boolean method1(TreeNode t1, TreeNode t2) {
        if (t1 == null && t2 == null) {
            return true;
        }
        if (t1 == null || t2 == null) {
            return false;
        }
        return t1.val == t2.val && method1(t1.left, t2.right) && method1(t1.right, t2.left);
    }
}
