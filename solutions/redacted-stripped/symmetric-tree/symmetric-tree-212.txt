class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        } else {
            return confirm(root.left, root.right);
        }
    }

    public boolean confirm(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null) {
            return false;
        }
        if (left.val != right.val) {
            return false;
        }
        return confirm(left.left, right.right) && confirm(right.left, left.right);
    }
}
