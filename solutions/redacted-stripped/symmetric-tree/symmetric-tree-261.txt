class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int val) {
        this.val = val;
    }
}

class Solution {

    public boolean method1(Class1 root) {
        if (root == null) {
            return true;
        }
        return Symm(root.left, root.right);
    }

    public boolean Symm(Class1 r1, Class1 r2) {
        if (r1 == null && r2 == null)
            return true;
        if (r1 == null || r2 == null)
            return false;
        return r1.val == r2.val && Symm(r1.left, r2.right) && Symm(r2.left, r1.right);
    }
}

public class Practice {

    public static Class1 method2(Scanner scanner) {
        System.out.print("Enter node value (-1 for null): ");
        int val = scanner.nextInt();
        if (val == -1) {
            return null;
        }
        Class1 node = new Class1(val);
        System.out.println("Enter left child of " + val);
        node.left = method2(scanner);
        System.out.println("Enter right child of " + val);
        node.right = method2(scanner);
        return node;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Create the root of the binary tree:");
        Class1 root = method2(scanner);
        Solution solution = new Solution();
        boolean method1 = solution.method1(root);
        System.out.println("Is the tree symmetric? " + method1);
        scanner.close();
    }
}
