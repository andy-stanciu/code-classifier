class Solution {

    public boolean traversal(TreeNode p, TreeNode q) {
        if (p == null || q == null) {
            if (p == null && q == null)
                return true;
            return false;
        }
        boolean l = traversal(p.left, q.right);
        boolean r = traversal(p.right, q.left);
        return l && r && (p.val == q.val);
    }

    public boolean method1(TreeNode root) {
        return traversal(root, root);
    }
}
