class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return all(root.left, root.right);
    }

    public boolean all(TreeNode root1, TreeNode root2) {
        if (root1 == null && root2 == null)
            return true;
        if (root1 == null || root2 == null)
            return false;
        if (root1.val == root2.val)
            return all(root1.left, root2.right) && all(root1.right, root2.left);
        return false;
    }
}
