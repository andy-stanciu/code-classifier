class Solution {

    public boolean isMirrored(TreeNode root1, TreeNode root2) {
        if (root1 == null && root2 == null) {
            return true;
        } else if (root1 == null || root2 == null) {
            return false;
        } else if (root1.val != root2.val) {
            return false;
        } else {
            return (isMirrored(root1.right, root2.left) && isMirrored(root1.left, root2.right));
        }
    }

    public boolean method1(TreeNode root) {
        if (root == null) {
            return false;
        } else {
            return isMirrored(root.left, root.right);
        }
    }
}
