class Solution {

    public boolean method1(TreeNode root) {
        boolean ans = method2(root.left, root.right);
        return ans;
    }

    public boolean method2(TreeNode p, TreeNode q) {
        if (p == null || q == null) {
            return p == q;
        }
        return (p.val == q.val) && method2(p.left, q.right) && method2(p.right, q.left);
    }
}
