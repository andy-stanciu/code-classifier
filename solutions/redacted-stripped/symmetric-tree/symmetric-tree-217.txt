class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return check(root.left, root.right);
    }

    public boolean check(TreeNode l, TreeNode r) {
        if (l == null || r == null)
            return l == r;
        if (l.val != r.val)
            return false;
        boolean k = check(l.left, r.right);
        boolean kk = check(l.right, r.left);
        if (k == true && kk == true)
            return true;
        else
            return false;
    }
}
