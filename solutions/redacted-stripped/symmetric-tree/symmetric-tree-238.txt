class Solution {

    public boolean method1(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        if (p.val != q.val)
            return false;
        return method1(p.left, q.left) && method1(q.right, p.right);
    }

    public TreeNode method2(TreeNode root) {
        if (root == null)
            return root;
        TreeNode temp = root.left;
        root.left = root.right;
        root.right = temp;
        TreeNode l = root.left;
        TreeNode r = root.right;
        root.left = method2(l);
        root.right = method2(r);
        return root;
    }

    public boolean method3(TreeNode root) {
        if (root == null)
            return true;
        root.left = method2(root.left);
        return method1(root.left, root.right);
    }
}
