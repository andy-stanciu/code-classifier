class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return method1(root.left, root.right);
    }

    public boolean method1(TreeNode first, TreeNode second) {
        if (first == null && second == null)
            return true;
        if ((first != null && second == null) || (first == null && second != null) || (first.val != second.val))
            return false;
        if (method1(first.left, second.right) && method1(first.right, second.left))
            return true;
        return false;
    }
}
