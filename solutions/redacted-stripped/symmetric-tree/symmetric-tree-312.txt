class Solution {

    public boolean method1(TreeNode root) {
        return method2(root, root);
    }

    public boolean method2(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null || right == null) {
            return false;
        }
        return left.val == right.val && method2(left.left, right.right) && method2(left.right, right.left);
    }
}
