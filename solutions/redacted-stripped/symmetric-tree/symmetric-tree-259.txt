class Solution {

    public boolean method1(TreeNode root) {
        return helper(root.left, root.right);
    }

    public boolean helper(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            return true;
        }
        if (left == null && right != null) {
            return false;
        }
        if (left != null && right == null) {
            return false;
        }
        if (left.val != right.val) {
            return false;
        }
        boolean var1 = helper(left.left, right.right);
        if (!var1) {
            return false;
        }
        boolean var2 = helper(left.right, right.left);
        return var2;
    }
}
