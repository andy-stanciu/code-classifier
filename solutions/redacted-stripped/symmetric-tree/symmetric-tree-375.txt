class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        if (root.left == null && root.right == null)
            return true;
        return check(root.left, root.right, true);
    }

    public boolean check(TreeNode left, TreeNode right, boolean ans) {
        if (left == null) {
            if (right != null)
                return false;
            return ans;
        }
        if (right == null) {
            if (left != null)
                return false;
            return ans;
        }
        if (left.val != right.val) {
            ans = false;
            return ans;
        }
        ans = check(left.left, right.right, ans);
        ans = check(left.right, right.left, ans);
        return ans;
    }
}
