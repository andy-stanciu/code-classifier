class Solution {

    public boolean method1(TreeNode root) {
        return root == null || method1(root.left, root.right);
    }

    private boolean method1(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            return left == right;
        }
        if (right.val != left.val) {
            return false;
        }
        return method1(left.left, right.right) && method1(right.left, left.right);
    }
}
