class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        if (root.left == null || root.right == null) {
            return root.left == root.right;
        }
        return isMinorCheck(root.left, root.right);
    }

    public boolean isMinorCheck(TreeNode a, TreeNode b) {
        if (a == null || b == null) {
            return a == b;
        }
        return (a.val == b.val && isMinorCheck(a.left, b.right) && isMinorCheck(a.right, b.left));
    }
}
