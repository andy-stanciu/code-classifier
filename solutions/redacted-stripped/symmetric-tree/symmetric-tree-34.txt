class Solution {

    public boolean method1(TreeNode root) {
        return method2(root.right, root.left);
    }

    public boolean method2(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        if (p.val == q.val)
            return method2(p.left, q.right) && method2(p.right, q.left);
        return false;
    }
}
