class Solution {

    public boolean method1(TreeNode root) {
        return method1(root.left, root.right);
    }

    private boolean method1(TreeNode left, TreeNode right) {
        if ((left == null && right != null) || (left != null && right == null)) {
            return false;
        }
        if (left == null && right == null) {
            return true;
        }
        return left.val == right.val && method1(left.right, right.left) && method1(left.left, right.right);
    }
}
