class Solution {

    public int[] method1(int n) {
        int[] nums = new int[n + 1];
        nums[0] = 0;
        if (n == 0)
            return nums;
        nums[1] = 1;
        if (n == 1)
            return nums;
        nums[2] = 1;
        if (n == 2)
            return nums;
        for (int i = 3; i < nums.length; i++) {
            nums[i] = nums[i / 2] + nums[i % 2];
        }
        return nums;
    }
}
