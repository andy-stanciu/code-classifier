public class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int numIndex = 0; numIndex <= n; numIndex++) {
            int num = numIndex;
            while (num > 0) {
                ans[numIndex] += num & 0x1;
                num >>= 1;
            }
        }
        return ans;
    }
}
