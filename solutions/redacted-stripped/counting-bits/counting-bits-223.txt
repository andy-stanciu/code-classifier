class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int a = 0; a <= n; a++) {
            int c = 0;
            for (int i = 0; i < 32; i++) {
                int j = 1 << i;
                int k = a & j;
                if (k != 0) {
                    c++;
                }
            }
            arr[a] = c;
        }
        return arr;
    }
}
