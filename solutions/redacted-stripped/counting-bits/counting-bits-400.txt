class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        int i = 0;
        while (i != n + 1) {
            ans[i] = countSetBit(i);
            i++;
        }
        return ans;
    }

    public static int countSetBit(int i) {
        int count = 0;
        while (i != 0) {
            count++;
            i &= (i - 1);
        }
        return count;
    }
}
