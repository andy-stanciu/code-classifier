class Solution {

    public int[] method1(int n) {
        int[] dp = new int[n + 1];
        for (int i = 1, substract = 1; i <= n; i++) {
            if (i == substract * 2) {
                substract = i;
            }
            dp[i] = dp[i - substract] + 1;
        }
        return dp;
    }
}
