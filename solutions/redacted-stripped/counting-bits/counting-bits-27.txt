class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        for (int i = 1; i < n + 1; i++) {
            int remainder = 0;
            int quo = i;
            int count = 0;
            while (quo > 0) {
                remainder = quo % 2;
                quo = quo / 2;
                if (remainder == 1)
                    count++;
            }
            arr[i] = count;
        }
        return arr;
    }
}
