class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        res[0] = 0;
        for (int num = 1; num <= n; num++) {
            res[num] = countHelper(num);
        }
        return res;
    }

    int countHelper(int n) {
        int count = 0;
        for (int i = 0; i < 32; i++) {
            if ((n & 1) == 1)
                count++;
            n = n >> 1;
        }
        return count;
    }
}
