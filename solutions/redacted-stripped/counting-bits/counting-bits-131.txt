class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            int number = i;
            int var2 = 0;
            while (number > 0) {
                if ((number & 1) == 1)
                    var2++;
                number = number >> 1;
            }
            var1[i] = var2;
        }
        return var1;
    }
}
