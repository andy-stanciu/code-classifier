class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        n = 0;
        int u = 1;
        for (int i = 1; i < arr.length; i++) {
            int count = 0;
            n = u;
            while (n != 0) {
                if ((n & 1) == 1)
                    count++;
                n >>= 1;
            }
            u++;
            arr[i] = count;
        }
        return arr;
    }
}
