class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            arr[i] = counts(i);
        }
        return arr;
    }

    public static int counts(int i) {
        int sum = 0;
        while (i > 0) {
            int last = i & 1;
            sum = sum + last;
            i = i >> 1;
        }
        return sum;
    }
}
