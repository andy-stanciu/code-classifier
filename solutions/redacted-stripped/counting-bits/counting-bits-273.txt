class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            boolean isOddNumber = i % 2 == 1;
            int halfOfEvenCompanion = i / 2;
            int onesCountForHalf = ans[halfOfEvenCompanion];
            ans[i] = isOddNumber ? onesCountForHalf + 1 : onesCountForHalf;
        }
        return ans;
    }
}
