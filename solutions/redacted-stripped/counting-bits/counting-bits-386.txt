class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int var1 = 0;
            int num = i;
            while (num != 0) {
                num = num & (num - 1);
                var1++;
            }
            arr[i] = var1;
        }
        return arr;
    }
}
