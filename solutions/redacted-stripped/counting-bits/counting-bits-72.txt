class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            arr[i] = countsetbit(i);
        }
        return arr;
    }

    public int countsetbit(int x) {
        int count = 0;
        while (x != 0) {
            count++;
            x = x & (x - 1);
        }
        return count;
    }
}
