class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        for (int j = 0; j < n + 1; j++) {
            res[j] = set(j);
        }
        return res;
    }

    static int set(int n) {
        int c = 0;
        while (n != 0) {
            n = n & (n - 1);
            c++;
        }
        return c;
    }
}
