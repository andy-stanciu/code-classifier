class Solution {

    public int[] method1(int n) {
        int[] out = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int count = 0;
            int attender = i;
            while (attender > 0) {
                int temp = attender % 2;
                attender = attender / 2;
                if (temp == 1) {
                    count++;
                }
            }
            out[i] = count;
        }
        return out;
    }
}
