class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int var1 = 0;
            int num = i;
            while (num > 0) {
                var1 += num % 2;
                num /= 2;
            }
            arr[i] = var1;
        }
        return arr;
    }
}
