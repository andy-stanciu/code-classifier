class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        var1[0] = 0;
        int power = 0;
        double next2PowerNum = 1;
        int nearest2PowerNum = 0;
        for (int num = 1; num <= n; num++) {
            if (next2PowerNum == num) {
                var1[num] = 1;
                nearest2PowerNum = num;
                power++;
                next2PowerNum = Math.pow(2, power);
            } else {
                var1[num] = var1[num - nearest2PowerNum] + 1;
            }
        }
        return var1;
    }
}
