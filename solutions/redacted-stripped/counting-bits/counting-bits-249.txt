class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        int i = 1;
        while (i < n + 1) {
            int j = i;
            int count = 0;
            while (j > 0) {
                count += (j & 1);
                j >>= 1;
            }
            arr[i] = count;
            i++;
        }
        return arr;
    }
}
