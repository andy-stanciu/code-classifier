class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int tmp = i;
            int ans = 0;
            while (tmp != 0) {
                tmp &= (tmp - 1);
                ans++;
            }
            res[i] = ans;
        }
        return res;
    }
}
