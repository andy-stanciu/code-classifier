class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int count = 0;
            int j = i;
            if (i == 0) {
                ans[i] = 0;
            } else {
                while (j != 0) {
                    j = j & (j - 1);
                    count++;
                }
                ans[i] = count;
            }
        }
        return ans;
    }
}
