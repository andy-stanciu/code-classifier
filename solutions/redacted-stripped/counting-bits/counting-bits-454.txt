class Solution {

    public int[] method1(int n) {
        int[] d = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int g = i;
            int count = 0;
            while (g != 0) {
                g = g & g - 1;
                count++;
            }
            d[i] = count;
        }
        return d;
    }
}
