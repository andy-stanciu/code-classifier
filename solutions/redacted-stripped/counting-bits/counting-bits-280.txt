class Solution {

    public int[] method1(int n) {
        int[] r = new int[n + 1];
        r[0] = 0;
        for (int i = 1; i <= n; i++) {
            int x = i;
            int count = 0;
            while (x > 0) {
                count += (x & 1);
                x = x >> 1;
            }
            r[i] = count;
        }
        return r;
    }
}
