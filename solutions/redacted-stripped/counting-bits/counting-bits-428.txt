class Solution {

    public int[] method1(int n) {
        int[] integers = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            integers[i] = countHelper(i);
        }
        return integers;
    }

    public int countHelper(int n) {
        if (n == 0) {
            return 0;
        } else if (n == 1) {
            return 1;
        } else if (n % 2 == 1) {
            return 1 + countHelper((n - 1) / 2);
        } else {
            return countHelper(n / 2);
        }
    }
}
