class Solution {

    private static int count(int n) {
        int var1 = 0;
        while (n > 0) {
            if ((n & 1) != 0) {
                var1++;
            }
            n = n >> 1;
        }
        return var1;
    }

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = count(i);
        }
        return ans;
    }
}
