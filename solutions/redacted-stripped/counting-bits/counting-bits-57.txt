class Solution {

    int[] dp;

    public int[] method1(int n) {
        dp = new int[n + 1];
        Arrays.fill(dp, -1);
        for (int i = 0; i <= n; i++) dp[i] = f(i);
        return dp;
    }

    public int f(int n) {
        if (n == 0)
            return 0;
        if (n == 1)
            return 1;
        if (dp[n] != -1)
            return dp[n];
        int count = 0;
        count = ((n & 1) == 1) ? 1 : 0;
        count += f(n >> 1);
        return dp[n] = count;
    }
}
