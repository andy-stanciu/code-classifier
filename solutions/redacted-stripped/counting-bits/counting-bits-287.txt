class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        int pow = 0;
        for (int i = 1; i < n + 1; i++) {
            var1[i] = var1[i - (1 << pow)] + 1;
            if ((i & (i + 1)) == 0) {
                pow++;
            }
        }
        return var1;
    }
}
