class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        ans[0] = 0;
        for (int i = 1; i <= n; i++) {
            int c = 0;
            int temp = i;
            while (temp != 0) {
                temp = temp & (temp - 1);
                c++;
            }
            ans[i] = c;
        }
        return ans;
    }
}
