class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        res[0] = 0;
        if (n > 0) {
            res[1] = 1;
        }
        int lastP = 0;
        for (int i = 2; i <= n; i++) {
            int bitCount = i & (i - 1);
            if (bitCount == 0) {
                res[i] = 1;
                lastP = i;
            } else {
                res[i] = 1 + res[i - lastP];
            }
        }
        return res;
    }
}
