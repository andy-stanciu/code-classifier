class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        res[0] = 0;
        if (n > 0) {
            res[1] = 1;
        }
        int offset = 2;
        int toOffsetChange = 2;
        for (int i = 2; i <= n; i++) {
            if (toOffsetChange == 0) {
                offset *= 2;
                toOffsetChange = offset;
            }
            res[i] = res[i - offset] + 1;
            toOffsetChange--;
        }
        return res;
    }
}
