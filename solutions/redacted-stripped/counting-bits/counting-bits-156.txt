class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        res[0] = 0;
        if (n == 0)
            return res;
        res[1] = 1;
        int x = 2;
        for (int i = 2; i <= n; i++) {
            if (i == x * 2) {
                x *= 2;
            }
            res[i] = res[i - x] + 1;
        }
        return res;
    }
}
