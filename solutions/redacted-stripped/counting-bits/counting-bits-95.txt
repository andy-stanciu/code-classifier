class Solution {

    public int[] method1(int n) {
        int[] resarr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int temp = i;
            int counter = 0;
            while (temp != 0) {
                int rsbm = temp & -temp;
                temp -= rsbm;
                counter++;
            }
            resarr[i] = counter;
        }
        return resarr;
    }
}
