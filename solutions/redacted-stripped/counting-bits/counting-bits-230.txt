class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        ans[0] = 0;
        if (n <= 0)
            return ans;
        ans[1] = 1;
        if (n <= 1)
            return ans;
        int lastpowerOfTwo = 0;
        for (int i = 2; i <= n; i++) {
            if (checkPowerOf2(i)) {
                lastpowerOfTwo = i;
                ans[i] = 1;
            } else {
                ans[i] = ans[i - lastpowerOfTwo] + 1;
            }
        }
        return ans;
    }

    public boolean checkPowerOf2(int i) {
        return (i & (i - 1)) == 0;
    }
}
