class Solution {

    int n = 0;

    int s = 1;

    int[] arr = null;

    public int[] method1(int n) {
        this.arr = new int[n + 1];
        this.n = n;
        return helper(0, 0, 0);
    }

    public int[] helper(int x, int i, int _1s) {
        arr[x] = _1s;
        for (int k = i; (x | (s << k)) <= n; k++) {
            arr = helper(x | (s << k), k + 1, _1s + 1);
        }
        return arr;
    }
}
