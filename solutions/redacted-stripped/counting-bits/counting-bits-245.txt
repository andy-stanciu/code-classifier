class Solution {

    public int[] method1(int n) {
        int[] answer = new int[n + 1];
        int n2 = 1;
        for (int i = 1; i <= n; i++) {
            if (i == n2) {
                answer[i] = 1;
                n2 *= 2;
            } else {
                answer[i] = answer[i - n2 / 2] + 1;
            }
        }
        return answer;
    }
}
