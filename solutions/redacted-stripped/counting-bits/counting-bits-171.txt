class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        method1(ans, 1, 1);
        return ans;
    }

    public void method1(int[] ans, int n, int count) {
        if (n >= ans.length)
            return;
        ans[n] = count;
        int val = n << 1;
        method1(ans, val, count);
        method1(ans, val + 1, count + 1);
    }
}
