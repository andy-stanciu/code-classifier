class Solution {

    public static int bin(int n) {
        if (n == 0)
            return 0;
        else if (n % 2 != 0) {
            return (1 + bin(n / 2));
        } else
            return bin(n / 2);
    }

    public int[] method1(int n) {
        int[] A = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            A[i] = bin(i);
        }
        return A;
    }
}
