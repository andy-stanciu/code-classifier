class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        if (n == 0)
            return ans;
        ans[1] = 1;
        for (int i = 2; i < ans.length; i++) {
            if (i % 2 == 0) {
                ans[i] = ans[i / 2] + 0;
            } else {
                ans[i] = ans[i / 2] + 1;
            }
        }
        return ans;
    }
}
