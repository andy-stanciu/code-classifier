class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int res = 0, d = i;
            while (d > 0) {
                if ((d & 1) == 1)
                    res++;
                d >>= 1;
            }
            ans[i] = res;
        }
        return ans;
    }
}
