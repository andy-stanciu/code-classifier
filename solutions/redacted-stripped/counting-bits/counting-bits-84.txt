class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        for (int i = 1; i <= n; i++) {
            int count = 0;
            int new_i = i;
            while (new_i > 0) {
                count += (new_i & 1);
                new_i = new_i >> 1;
            }
            arr[i] = count;
        }
        return arr;
    }
}
