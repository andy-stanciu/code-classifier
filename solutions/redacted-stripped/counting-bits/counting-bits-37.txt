class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = method2(i);
        }
        return ans;
    }

    public int method2(int n) {
        if (n == 0)
            return 0;
        return (n & 1) + method2(n >> 1);
    }
}
