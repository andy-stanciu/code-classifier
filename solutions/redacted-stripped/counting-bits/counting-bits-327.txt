class Solution {

    public int[] method1(int n) {
        int[] result = new int[n + 1];
        for (int i = 1; i < n + 1; i++) {
            if (i % 2 == 0)
                result[i] = result[i / 2];
            else
                result[i] = result[i - 1] + 1;
        }
        return result;
    }
}
