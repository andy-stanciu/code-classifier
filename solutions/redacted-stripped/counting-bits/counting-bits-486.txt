class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        for (int i = 1; i <= n; i++) {
            int y = i / 2;
            if (i % 2 == 0) {
                arr[i] = arr[y];
            } else {
                arr[i] = arr[y] + 1;
            }
        }
        return arr;
    }
}
