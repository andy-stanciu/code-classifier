class Solution {

    public int[] method1(int n) {
        int len = n + 1;
        int[] arr = new int[len];
        int mask = 1;
        int count = 0;
        arr[0] = 0;
        for (int j = 1; j < len; j++) {
            count = 0;
            mask = 1;
            for (int i = 0; i < 33; i++) {
                if ((j & mask) != 0)
                    count++;
                mask <<= 1;
            }
            arr[j] = count;
        }
        return arr;
    }
}
