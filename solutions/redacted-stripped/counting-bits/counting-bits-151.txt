class Solution {

    public int[] method1(int n) {
        int[] array = new int[n + 1];
        array[0] = 0;
        for (int i = 1; i <= n; i++) {
            array[i] = binary(i);
        }
        return array;
    }

    public int binary(int i) {
        int total = 0;
        while (i > 0) {
            if (i % 2 == 1)
                total++;
            i /= 2;
        }
        return total;
    }
}
