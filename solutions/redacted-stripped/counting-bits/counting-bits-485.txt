class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        for (int num = 0; num <= n; num++) {
            res[num] = countBit(num);
        }
        return res;
    }

    private int countBit(int n) {
        int count = 0;
        while (n > 0) {
            n = n & (n - 1);
            count++;
        }
        return count;
    }
}
