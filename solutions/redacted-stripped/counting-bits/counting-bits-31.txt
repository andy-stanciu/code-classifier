class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        int var2 = 0;
        for (int i = 0; i <= n; i++) {
            int j = i;
            while (j != 0) {
                int d = j % 2;
                var2 += d;
                j = j / 2;
            }
            var1[i] = var2;
            var2 = 0;
        }
        return var1;
    }
}
