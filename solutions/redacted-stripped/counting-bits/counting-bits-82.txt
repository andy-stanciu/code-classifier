class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        int power = 1;
        for (int i = 1; i <= n; i++) {
            if (power * 2 == i) {
                power = power * 2;
            }
            res[i] = 1 + res[i - power];
        }
        return res;
    }
}
