class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        int count = 0;
        for (int i = 0; i <= n; ++i) {
            int curr = i;
            while (curr != 0) {
                curr &= curr - 1;
                count++;
            }
            ans[i] = count;
            count = 0;
        }
        return ans;
    }
}
