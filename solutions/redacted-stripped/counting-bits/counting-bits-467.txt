class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        arr[0] = 0;
        for (int i = 1; i <= n; i++) {
            int count = 0;
            int j = i;
            while (j != 0) {
                count++;
                j = j & (j - 1);
            }
            arr[i] = count;
        }
        return arr;
    }
}
