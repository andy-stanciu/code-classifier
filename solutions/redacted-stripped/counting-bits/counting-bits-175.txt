class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int num = i;
            for (int j = 0; j < 32; j++) {
                if ((num & 1) == 1)
                    ans[i]++;
                num = num >> 1;
            }
        }
        return ans;
    }
}
