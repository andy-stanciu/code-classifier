class Solution {

    public int[] method1(int n) {
        int[] ones = new int[n + 1];
        for (int i = 0; i < ones.length; i++) {
            if (i == 0)
                ones[i] = 0;
            else if (i == 1)
                ones[i] = 1;
            else {
                int var1 = (int) Math.floor(Math.log(i) / Math.log(2));
                var1 = 1 + ones[(int) (i - Math.pow(2, var1))];
                ones[i] = var1;
            }
        }
        return ones;
    }
}
