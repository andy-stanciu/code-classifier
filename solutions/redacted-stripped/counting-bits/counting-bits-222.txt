public class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int x = 1; x <= n; x++) {
            if (x % 2 == 0) {
                ans[x] = ans[x / 2];
            } else {
                ans[x] = 1 + ans[x / 2];
            }
        }
        return ans;
    }
}
