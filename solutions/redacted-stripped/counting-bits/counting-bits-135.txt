class Solution {

    public int hammingWeight(int n) {
        int var1 = (int) (Math.log(n) / Math.log(2)) + 1;
        int cnt = 0;
        for (int i = 0; i < var1; i++) {
            if ((n & (1 << i)) != 0)
                cnt++;
        }
        return cnt;
    }

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            res[i] = hammingWeight(i);
        }
        return res;
    }
}
