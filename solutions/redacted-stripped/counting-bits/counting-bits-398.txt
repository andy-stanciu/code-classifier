class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        res[0] = 0;
        for (int i = 1; i <= n; i++) {
            int realI = i;
            int count = 0;
            while (i != 0) {
                i = i & i - 1;
                count++;
            }
            res[realI] = count;
            i = realI;
        }
        return res;
    }
}
