class Solution {

    public int[] method1(int n) {
        int[] countsOfOne = new int[n + 1];
        int div, remainder;
        for (int i = 0; i < countsOfOne.length; i++) {
            div = i / 2;
            remainder = i % 2;
            countsOfOne[i] = countsOfOne[div] + remainder;
        }
        return countsOfOne;
    }
}
