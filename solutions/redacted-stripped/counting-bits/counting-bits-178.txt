class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 1; i < n + 1; i++) {
            if (i % 2 != 0) {
                arr[i] = arr[i >> 1] + 1;
            } else {
                arr[i] = arr[i >> 1];
            }
        }
        return arr;
    }
}
