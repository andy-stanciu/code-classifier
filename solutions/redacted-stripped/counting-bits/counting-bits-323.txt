class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int c = 0;
            int k = i;
            while (k > 0) {
                if (k % 2 == 1) {
                    c++;
                }
                k = k / 2;
            }
            arr[i] = c;
        }
        return arr;
    }
}
