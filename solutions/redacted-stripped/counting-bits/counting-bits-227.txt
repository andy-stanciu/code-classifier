class Solution {

    public int[] method1(int n) {
        int res = 0;
        int[] list = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j < 32; j++) {
                if (((i >> j) & 1) == 1) {
                    res++;
                }
            }
            list[i] = res;
            res = 0;
        }
        return list;
    }
}
