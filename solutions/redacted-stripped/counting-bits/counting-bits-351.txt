class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = countbit(i);
        }
        return ans;
    }

    public int countbit(int n) {
        int var1 = 0;
        while (n != 0) {
            var1 += (n & 1);
            n = n >>> 1;
        }
        return var1;
    }
}
