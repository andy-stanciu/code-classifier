class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            arr[i] = 0;
            int j = i;
            while (j > 0) {
                if (j % 2 != 0)
                    arr[i]++;
                j /= 2;
            }
        }
        return arr;
    }
}
