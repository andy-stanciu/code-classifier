class Solution {

    public static int countOne(int i) {
        int count = 0;
        while (i != 0) {
            int lastbit = i & 1;
            count += lastbit;
            i = i >>> 1;
        }
        return count;
    }

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = countOne(i);
        }
        return ans;
    }
}
