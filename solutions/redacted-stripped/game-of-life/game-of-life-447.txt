class Solution {

    public void method1(int[][] board) {
        int m = board.length;
        int n = board[0].length;
        int[][] result = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int liveNeibhors = 0;
                if (i - 1 >= 0) {
                    liveNeibhors += board[i - 1][j];
                    if (j - 1 >= 0)
                        liveNeibhors += board[i - 1][j - 1];
                    if (j + 1 < n)
                        liveNeibhors += board[i - 1][j + 1];
                }
                if (i + 1 < m) {
                    liveNeibhors += board[i + 1][j];
                    if (j - 1 >= 0)
                        liveNeibhors += board[i + 1][j - 1];
                    if (j + 1 < n)
                        liveNeibhors += board[i + 1][j + 1];
                }
                if (j - 1 >= 0)
                    liveNeibhors += board[i][j - 1];
                if (j + 1 < n)
                    liveNeibhors += board[i][j + 1];
                if (board[i][j] == 1) {
                    if (liveNeibhors == 2 || liveNeibhors == 3)
                        result[i][j] = 1;
                } else {
                    if (liveNeibhors == 3)
                        result[i][j] = 1;
                }
            }
        }
        for (int i = 0; i < m; i++) {
            System.arraycopy(result[i], 0, board[i], 0, n);
        }
    }
}
