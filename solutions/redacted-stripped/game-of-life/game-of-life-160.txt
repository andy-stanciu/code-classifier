class Solution {

    public void method1(int[][] board) {
        int[][] ans = new int[board.length][board[0].length];
        int n = board.length;
        int m = board[0].length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int zero = 0;
                int one = 0;
                for (int k = -1; k <= 1; k++) {
                    for (int l = -1; l <= 1; l++) {
                        if (k == 0 && l == 0)
                            continue;
                        if (k + i >= 0 && k + i < n && l + j >= 0 && l + j < m) {
                            if (board[k + i][l + j] == 0)
                                zero++;
                            else
                                one++;
                        }
                    }
                }
                if (board[i][j] == 1) {
                    if (one < 2)
                        ans[i][j] = 0;
                    else if (one == 3 || one == 2)
                        ans[i][j] = 1;
                    else if (one > 3)
                        ans[i][j] = 0;
                } else {
                    if (one == 3)
                        ans[i][j] = 1;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                board[i][j] = ans[i][j];
            }
        }
    }
}
