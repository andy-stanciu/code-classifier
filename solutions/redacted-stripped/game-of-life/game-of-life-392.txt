class Solution {

    public void method1(int[][] board) {
        int row = board.length;
        int col = board[0].length;
        int[][] result = new int[row][col];
        result = board;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                int temp = 0;
                int[] dx = new int[] { 0, 0, 1, 1, 1, -1, -1, -1 };
                int[] dy = new int[] { 1, -1, 0, 1, -1, 0, 1, -1 };
                for (int k = 0; k < 8; k++) {
                    int newx = i + dx[k];
                    int newy = j + dy[k];
                    if (newx >= 0 && newx < row && newy >= 0 && newy < col) {
                        if (Math.abs(result[newx][newy]) == 1)
                            temp++;
                    }
                }
                if (result[i][j] == 1 && temp < 2) {
                    board[i][j] = -1;
                }
                if (result[i][j] == 1 && temp > 3) {
                    board[i][j] = -1;
                }
                if (result[i][j] == 0 && temp == 3) {
                    board[i][j] = 2;
                }
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j] >= 1)
                    board[i][j] = 1;
                else
                    board[i][j] = 0;
            }
        }
    }
}
