class Solution {

    public void method1(int[][] board) {
        int n = board.length;
        int m = board[0].length;
        boolean[][] var1 = new boolean[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                check(i, j, board, n, m, var1);
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (var1[i][j])
                    board[i][j] = 1;
                else
                    board[i][j] = 0;
            }
        }
    }

    public void check(int i, int j, int[][] board, int n, int m, boolean[][] var1) {
        int[] ar = new int[8];
        Arrays.fill(ar, -1);
        if ((i - 1) >= 0)
            ar[0] = board[i - 1][j];
        if ((j - 1) >= 0)
            ar[1] = board[i][j - 1];
        if ((i - 1) >= 0 && (j - 1) >= 0)
            ar[2] = board[i - 1][j - 1];
        if ((i + 1) <= n - 1)
            ar[3] = board[i + 1][j];
        if ((j + 1) <= m - 1)
            ar[4] = board[i][j + 1];
        if ((i + 1) <= n - 1 && (j + 1) <= m - 1)
            ar[5] = board[i + 1][j + 1];
        if ((i - 1) >= 0 && (j + 1) <= m - 1)
            ar[6] = board[i - 1][j + 1];
        if ((i + 1) <= n - 1 && j - 1 >= 0)
            ar[7] = board[i + 1][j - 1];
        int one = 0;
        int zero = 0;
        for (int k : ar) {
            if (k == 1)
                one++;
            if (k == 0)
                zero++;
        }
        if ((one + zero) < 2 && board[i][j] == 1) {
            var1[i][j] = false;
            return;
        }
        if ((one == 2 || one == 3) && board[i][j] == 1) {
            var1[i][j] = true;
            return;
        }
        if (one > 3 && board[i][j] == 1) {
            var1[i][j] = false;
            return;
        }
        if (board[i][j] == 0 && one == 3) {
            var1[i][j] = true;
        }
    }
}
