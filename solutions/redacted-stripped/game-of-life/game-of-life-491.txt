class Solution {

    public void method1(int[][] board) {
        List<Integer> x = new ArrayList<>();
        List<Integer> y = new ArrayList<>();
        int m = board.length;
        int n = board[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int var1 = 0;
                if (i + 1 < m && board[i + 1][j] == 1)
                    var1++;
                if (i - 1 >= 0 && board[i - 1][j] == 1)
                    var1++;
                if (j + 1 < n && board[i][j + 1] == 1)
                    var1++;
                if (j - 1 >= 0 && board[i][j - 1] == 1)
                    var1++;
                if (i + 1 < m && j + 1 < n && board[i + 1][j + 1] == 1)
                    var1++;
                if (i - 1 >= 0 && j - 1 >= 0 && board[i - 1][j - 1] == 1)
                    var1++;
                if (i - 1 >= 0 && j + 1 < n && board[i - 1][j + 1] == 1)
                    var1++;
                if (i + 1 < m && j - 1 >= 0 && board[i + 1][j - 1] == 1)
                    var1++;
                if ((var1 == 3 && board[i][j] == 0) || (var1 < 2 && board[i][j] == 1) || (var1 > 3 && board[i][j] == 1)) {
                    x.add(i);
                    y.add(j);
                }
            }
        }
        for (int i = 0; i < x.size(); i++) {
            board[x.get(i)][y.get(i)] = board[x.get(i)][y.get(i)] == 0 ? 1 : 0;
        }
    }
}
