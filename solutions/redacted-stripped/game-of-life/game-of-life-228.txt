class Solution {

    public void method1(int[][] B) {
        int m = B.length - 1, n = B[0].length - 1;
        for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= n; j++) {
                int count = 0;
                if (i > 0) {
                    if (j > 0 && B[i - 1][j - 1] % 10 == 1)
                        count++;
                    if (j < n && B[i - 1][j + 1] % 10 == 1)
                        count++;
                    if (B[i - 1][j] % 10 == 1)
                        count++;
                }
                if (i < m) {
                    if (j > 0 && B[i + 1][j - 1] % 10 == 1)
                        count++;
                    if (j < n && B[i + 1][j + 1] == 1)
                        count++;
                    if (B[i + 1][j] == 1)
                        count++;
                }
                if (j > 0 && B[i][j - 1] % 10 == 1)
                    count++;
                if (j < n && B[i][j + 1] == 1)
                    count++;
                B[i][j] += count * 10;
            }
        }
        for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= n; j++) {
                if (B[i][j] / 10 == 2)
                    B[i][j] = B[i][j] % 10;
                else
                    B[i][j] = B[i][j] / 10 == 3 ? 1 : 0;
            }
        }
    }
}
