class Solution {

    public void method1(int[][] b) {
        int[][] a = new int[b.length][b[0].length];
        for (int i = 0; i < b.length; i++) {
            for (int j = 0; j < b[0].length; j++) {
                int c = count(b, i, j);
                if (b[i][j] == 0) {
                    if (c == 3) {
                        a[i][j] = 1;
                    }
                } else {
                    if (c == 2 || c == 3)
                        a[i][j] = 1;
                }
            }
        }
        for (int i = 0; i < b.length; i++) {
            for (int j = 0; j < b[0].length; j++) {
                b[i][j] = a[i][j];
            }
        }
    }

    public int count(int[][] b, int i, int j) {
        int c = 0;
        int m = b.length;
        int n = b[0].length;
        if (j - 1 >= 0 && b[i][j - 1] == 1)
            c++;
        if (i - 1 >= 0 && b[i - 1][j] == 1)
            c++;
        if (j + 1 < n && b[i][j + 1] == 1)
            c++;
        if (i + 1 < m && b[i + 1][j] == 1)
            c++;
        if (i - 1 >= 0 && j + 1 < n && b[i - 1][j + 1] == 1)
            c++;
        if (i - 1 >= 0 && j - 1 >= 0 && b[i - 1][j - 1] == 1)
            c++;
        if (i + 1 < m && j - 1 >= 0 && b[i + 1][j - 1] == 1)
            c++;
        if (i + 1 < m && j + 1 < n && b[i + 1][j + 1] == 1)
            c++;
        return c;
    }
}
