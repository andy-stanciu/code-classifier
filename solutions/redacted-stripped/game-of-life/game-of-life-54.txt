class Solution {

    public void method1(int[][] arr) {
        int m = arr.length;
        int n = arr[0].length;
        int[][] my = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int count = 0;
                if (i - 1 >= 0 && arr[i - 1][j] == 1) {
                    count++;
                }
                if (i - 1 >= 0 && j + 1 < n && arr[i - 1][j + 1] == 1) {
                    count++;
                }
                if (j + 1 < n && arr[i][j + 1] == 1) {
                    count++;
                }
                if (i + 1 < m && j + 1 < n && arr[i + 1][j + 1] == 1) {
                    count++;
                }
                if (i + 1 < m && arr[i + 1][j] == 1) {
                    count++;
                }
                if (i + 1 < m && j - 1 >= 0 && arr[i + 1][j - 1] == 1) {
                    count++;
                }
                if (j - 1 >= 0 && arr[i][j - 1] == 1) {
                    count++;
                }
                if (i - 1 >= 0 && j - 1 >= 0 && arr[i - 1][j - 1] == 1) {
                    count++;
                }
                if (count < 2) {
                    my[i][j] = 0;
                } else if (arr[i][j] == 1 && (count == 2 || count == 3)) {
                    my[i][j] = 1;
                    continue;
                } else if (arr[i][j] == 1 && count > 3) {
                    my[i][j] = 0;
                } else if (arr[i][j] == 0 && count == 3) {
                    my[i][j] = 1;
                }
            }
        }
        for (int k = 0; k < m; k++) {
            for (int kk = 0; kk < n; kk++) {
                arr[k][kk] = my[k][kk];
            }
        }
    }
}
