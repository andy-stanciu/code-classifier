class Solution {

    public void method1(int[][] board) {
        int[][] mat = new int[board.length][board[0].length];
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                int cnt_1 = 0;
                int cnt_0 = 0;
                if (j - 1 >= 0) {
                    if (board[i][j - 1] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (j + 1 < board[0].length) {
                    if (board[i][j + 1] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (i - 1 >= 0) {
                    if (board[i - 1][j] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (i + 1 < board.length) {
                    if (board[i + 1][j] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (j - 1 >= 0 && i - 1 >= 0) {
                    if (board[i - 1][j - 1] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (j + 1 < board[0].length && i + 1 < board.length) {
                    if (board[i + 1][j + 1] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (j + 1 < board[0].length && i - 1 >= 0) {
                    if (board[i - 1][j + 1] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (j - 1 >= 0 && i + 1 < board.length) {
                    if (board[i + 1][j - 1] == 1) {
                        cnt_1++;
                    } else {
                        cnt_0++;
                    }
                }
                if (board[i][j] == 1) {
                    if (cnt_1 < 2) {
                        mat[i][j] = 0;
                    } else if (cnt_1 == 2 || cnt_1 == 3) {
                        mat[i][j] = 1;
                    } else if (cnt_1 > 3) {
                        mat[i][j] = 0;
                    }
                } else {
                    if (cnt_1 == 3) {
                        mat[i][j] = 1;
                    }
                }
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                board[i][j] = mat[i][j];
            }
        }
    }
}
