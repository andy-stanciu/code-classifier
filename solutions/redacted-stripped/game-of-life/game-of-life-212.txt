class Solution {

    public void method1(int[][] board) {
        int m = board.length, n = board[0].length;
        int[] arr = new int[8];
        int ones = 0, zeros = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                ones = 0;
                zeros = 0;
                arr[0] = i - 1 >= 0 ? board[i - 1][j] : -1;
                arr[1] = i + 1 < m ? board[i + 1][j] : -1;
                arr[2] = j - 1 >= 0 ? board[i][j - 1] : -1;
                arr[3] = j + 1 < n ? board[i][j + 1] : -1;
                arr[4] = i - 1 >= 0 && j - 1 >= 0 ? board[i - 1][j - 1] : -1;
                arr[5] = i - 1 >= 0 && j + 1 < n ? board[i - 1][j + 1] : -1;
                arr[6] = i + 1 < m && j - 1 >= 0 ? board[i + 1][j - 1] : -1;
                arr[7] = i + 1 < m && j + 1 < n ? board[i + 1][j + 1] : -1;
                for (int k = 0; k < 8; k++) {
                    if (arr[k] == 1 || arr[k] == 3) {
                        ones++;
                    }
                    if (arr[k] == 0 || arr[k] == 2) {
                        zeros++;
                    }
                }
                if (board[i][j] == 0) {
                    if (ones == 3) {
                        board[i][j] = 2;
                    }
                } else {
                    if (ones < 2 || ones > 3) {
                        board[i][j] = 3;
                    }
                }
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 2) {
                    board[i][j] = 1;
                }
                if (board[i][j] == 3) {
                    board[i][j] = 0;
                }
            }
        }
    }
}
