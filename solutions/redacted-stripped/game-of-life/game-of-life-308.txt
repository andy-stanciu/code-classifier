class Solution {

    public void method1(int[][] board) {
        int rows = board.length;
        int cols = board[0].length;
        int l_cell = 0;
        int[][] matrix = new int[rows][cols];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (j - 1 > -1) {
                    if (board[i][j - 1] == 1) {
                        l_cell += 1;
                    }
                }
                if (j + 1 < cols) {
                    if (board[i][j + 1] == 1) {
                        l_cell++;
                    }
                }
                if (i - 1 >= 0) {
                    if (board[i - 1][j] == 1) {
                        l_cell++;
                    }
                }
                if (i + 1 < rows) {
                    if (board[i + 1][j] == 1) {
                        l_cell += 1;
                    }
                }
                if (i - 1 >= 0 && j - 1 >= 0) {
                    if (board[i - 1][j - 1] == 1) {
                        l_cell += 1;
                    }
                }
                if (i - 1 >= 0 && j + 1 < cols) {
                    if (board[i - 1][j + 1] == 1) {
                        l_cell++;
                    }
                }
                if (i + 1 < rows && j - 1 >= 0) {
                    if (board[i + 1][j - 1] == 1) {
                        l_cell += 1;
                    }
                }
                if (i + 1 < rows && j + 1 < cols) {
                    if (board[i + 1][j + 1] == 1) {
                        l_cell++;
                    }
                }
                if (board[i][j] == 0) {
                    if (l_cell == 3)
                        matrix[i][j] = 1;
                    else
                        matrix[i][j] = 0;
                } else {
                    if (l_cell == 2 || l_cell == 3)
                        matrix[i][j] = 1;
                    else
                        matrix[i][j] = 0;
                }
                l_cell = 0;
            }
        }
        for (int e = 0; e < rows; e++) {
            for (int f = 0; f < cols; f++) {
                board[e][f] = matrix[e][f];
            }
        }
    }
}
