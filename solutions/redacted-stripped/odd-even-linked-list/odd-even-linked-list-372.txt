class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode var1 = new ListNode();
        ListNode var2 = new ListNode();
        var1 = head;
        var2 = head.next;
        ListNode eHead = var2;
        while (var1 != null && var1.next != null && var2 != null && var2.next != null) {
            var1.next = var1.next.next;
            var2.next = var1.next.next;
            var1 = var1.next;
            var2 = var1.next;
        }
        var1.next = eHead;
        return head;
    }
}
