class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = var2;
        while (var3 != null && var3.next != null) {
            var1.next = var3.next;
            var1 = var1.next;
            var3.next = var1.next;
            var3 = var3.next;
        }
        var1.next = var2;
        return head;
    }
}
