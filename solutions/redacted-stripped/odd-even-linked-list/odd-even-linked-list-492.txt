class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode oh = head, eh = head.next, var1 = head.next;
        while (eh != null && eh.next != null) {
            oh.next = oh.next.next;
            eh.next = eh.next.next;
            oh = oh.next;
            eh = eh.next;
        }
        oh.next = var1;
        return head;
    }
}
