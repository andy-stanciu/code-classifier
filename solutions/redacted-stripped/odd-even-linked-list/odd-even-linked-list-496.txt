class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(0);
        ListNode var2 = new ListNode(0);
        ListNode var3 = var1;
        ListNode var4 = var2;
        ListNode temp = head;
        int i = 1;
        while (temp != null) {
            if (i % 2 == 0) {
                var4.next = temp;
                var4 = temp;
            } else {
                var3.next = temp;
                var3 = temp;
            }
            temp = temp.next;
            i++;
        }
        var4.next = null;
        var3.next = var2.next;
        head = var1.next;
        return head;
    }
}
