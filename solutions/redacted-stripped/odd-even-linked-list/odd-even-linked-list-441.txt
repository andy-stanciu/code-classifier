class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(-1);
        ListNode var2 = var1;
        ListNode var3 = new ListNode(-1);
        ListNode var4 = var3;
        ListNode curr = head;
        while (curr != null) {
            var1.next = curr;
            if (curr.next != null) {
                var3.next = curr.next;
                var3 = var3.next;
            }
            var1 = var1.next;
            if (curr.next != null)
                curr = curr.next.next;
            else
                curr = curr.next;
        }
        var3.next = null;
        var1.next = null;
        var1.next = var4.next;
        return var2.next;
    }
}
