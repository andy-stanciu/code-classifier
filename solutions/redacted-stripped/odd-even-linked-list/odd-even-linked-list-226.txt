class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode fast = head;
        ListNode slow = head;
        ListNode previous = null;
        ListNode dummyfast = new ListNode(0);
        ListNode temp1 = dummyfast;
        ListNode dummyslow = new ListNode(0);
        ListNode temp2 = dummyslow;
        while (fast != null) {
            slow = slow.next;
            previous = slow;
            temp1.next = fast;
            temp1 = fast;
            if (fast.next != null) {
                fast = fast.next.next;
            } else {
                fast = fast.next;
            }
            temp2.next = previous;
            temp2 = previous;
            if (slow != null)
                slow = slow.next;
        }
        temp1.next = null;
        temp1.next = dummyslow.next;
        return dummyfast.next;
    }
}
