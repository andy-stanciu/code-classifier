class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(0);
        ListNode var2 = new ListNode(0);
        ListNode ev = var1;
        ListNode od = var2;
        int size = 1;
        while (head != null) {
            if (size % 2 != 0) {
                od.next = head;
                od = head;
                head = head.next;
                size++;
            } else {
                ev.next = head;
                ev = head;
                head = head.next;
                size++;
            }
        }
        var2 = var2.next;
        var1 = var1.next;
        ev.next = null;
        od.next = var1;
        return var2;
    }
}
