class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode var1 = new ListNode(-1);
        ListNode var2 = new ListNode(-1);
        ListNode var3 = var1;
        ListNode var4 = var2;
        ListNode prev = null;
        int c = 1;
        ListNode curr = head;
        while (curr != null) {
            if (c % 2 != 0) {
                var1.next = new ListNode(curr.val);
                var1 = var1.next;
            } else {
                var2.next = new ListNode(curr.val);
                var2 = var2.next;
            }
            c++;
            curr = curr.next;
        }
        var1.next = var4.next;
        return var3.next;
    }
}
