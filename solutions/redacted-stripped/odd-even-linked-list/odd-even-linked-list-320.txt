class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode[] var1 = { head.next, head };
        ListNode[] var2 = { head.next, head };
        int index = 3;
        ListNode pointer = head.next.next;
        var1[0].next = null;
        var1[1].next = null;
        while (pointer != null) {
            int r = index++ % 2;
            var2[r].next = pointer;
            pointer = pointer.next;
            var2[r] = var2[r].next;
            var2[r].next = null;
        }
        var2[1].next = var1[0];
        return var1[1];
    }
}
