class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode slow = head;
        ListNode fast = head.next;
        ListNode fasthead = head.next;
        while (slow.next != null && fast.next != null) {
            slow.next = slow.next.next;
            slow = slow.next;
            fast.next = fast.next.next;
            fast = fast.next;
        }
        slow.next = fasthead;
        return head;
    }
}
