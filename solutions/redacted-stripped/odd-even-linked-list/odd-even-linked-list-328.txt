class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode var1 = head.next;
        ListNode first = head;
        ListNode second = head.next;
        ListNode third = head.next.next;
        int count = 0;
        while (second != null) {
            count++;
            first.next = third;
            if (third == null && count % 2 == 1)
                break;
            first = second;
            second = third;
            if (third != null)
                third = third.next;
        }
        first.next = var1;
        return head;
    }
}
