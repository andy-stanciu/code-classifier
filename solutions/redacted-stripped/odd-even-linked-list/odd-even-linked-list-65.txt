class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode s = head, e = head.next, p = e;
        while (e != null && e.next != null) {
            s.next = e.next;
            s = s.next;
            e.next = s.next;
            e = e.next;
        }
        s.next = p;
        return head;
    }
}
