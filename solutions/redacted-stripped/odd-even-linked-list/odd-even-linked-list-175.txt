class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(0);
        ListNode var2 = new ListNode(0);
        ListNode tempe = var2;
        ListNode tempo = var1;
        ListNode temp = head;
        while (temp != null) {
            tempo.next = temp;
            temp = temp.next;
            tempo = tempo.next;
            tempe.next = temp;
            if (temp != null)
                temp = temp.next;
            tempe = tempe.next;
        }
        tempo.next = var2.next;
        return var1.next;
    }
}
