class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode a = head;
        ListNode b = head.next;
        ListNode res = b;
        while (a.next != null && b.next != null) {
            a.next = a.next.next;
            b.next = b.next.next;
            a = a.next;
            b = b.next;
        }
        a.next = res;
        return head;
    }
}
