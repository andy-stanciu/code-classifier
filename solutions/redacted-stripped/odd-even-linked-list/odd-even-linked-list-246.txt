class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        int i = 0;
        ListNode prev = head;
        ListNode curr = head.next;
        ListNode shead = curr;
        while (curr != null && curr.next != null) {
            prev.next = curr.next;
            prev = curr;
            curr = prev.next;
            i++;
        }
        if (i % 2 != 0) {
            curr.next = shead;
            prev.next = null;
        } else {
            prev.next = shead;
            curr.next = null;
        }
        return head;
    }
}
