class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode p = head;
        ListNode temp = p.next;
        ListNode head2 = p.next;
        ListNode c = head;
        int count = 1;
        while (c.next != null) {
            count++;
            c = c.next;
        }
        ListNode prev = null;
        while (p.next != null) {
            p.next = temp.next;
            prev = p;
            p = temp;
            temp = temp.next;
        }
        if (count % 2 == 0)
            prev.next = head2;
        else
            p.next = head2;
        return head;
    }
}
