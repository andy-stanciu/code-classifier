class Solution {

    int var1 = 0;

    public ListNode lastNode(ListNode head) {
        int currLength = 1;
        ListNode prev = null;
        ListNode curr = head;
        while (curr.next != null) {
            prev = curr;
            currLength++;
            curr = curr.next;
        }
        var1 = currLength;
        if (currLength % 2 == 0) {
            return prev;
        }
        return curr;
    }

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode last = lastNode(head);
        boolean var2 = false;
        ListNode lastNext = last.next;
        if (var1 % 2 == 0) {
            lastNext = last.next;
            last.next = null;
            var1--;
            var2 = true;
        }
        ListNode prev = null;
        ListNode curr = head;
        for (int i = 1; i <= var1; i++) {
            if (i % 2 != 0) {
                prev = curr;
                curr = curr.next;
            } else {
                ListNode nextNode = curr.next;
                prev.next = nextNode;
                curr.next = null;
                last.next = curr;
                last = last.next;
                curr = prev.next;
            }
        }
        if (var2) {
            last.next = lastNext;
        }
        return head;
    }
}
