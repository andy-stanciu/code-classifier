class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = head.next;
        while (var1 != null && var2 != null && var1.next != null && var2.next != null) {
            ListNode var4 = var1.next.next;
            ListNode var5 = var2.next.next;
            var1.next = var4;
            var2.next = var5;
            var1 = var4;
            var2 = var5;
        }
        var1.next = var3;
        return head;
    }
}
