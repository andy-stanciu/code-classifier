class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        ListNode var1 = new ListNode(-1);
        ListNode var2 = new ListNode(-1);
        ListNode var3 = var1;
        ListNode var4 = var2;
        int c = 0;
        while (temp != null) {
            if (c % 2 == 0) {
                ListNode newnode = new ListNode(temp.val);
                var3.next = newnode;
                var3 = var3.next;
            } else {
                ListNode newnode = new ListNode(temp.val);
                var4.next = newnode;
                var4 = var4.next;
            }
            if (temp != null) {
                temp = temp.next;
                c++;
            }
        }
        var3.next = var2.next;
        return var1.next;
    }
}
