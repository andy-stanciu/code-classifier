class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode tempO = var1;
        ListNode var2 = head.next;
        ListNode tempE = var2;
        while (tempO.next != null) {
            tempO.next = tempO.next.next;
            if (tempE.next == null)
                break;
            tempE.next = tempE.next.next;
            tempO = tempO.next;
            tempE = tempE.next;
        }
        tempO.next = var2;
        return head;
    }
}
