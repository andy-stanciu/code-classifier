class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode curr = head.next, slow = head, fast = head.next;
        ;
        while (curr != null && curr.next != null) {
            slow.next = curr.next;
            curr.next = slow.next.next;
            slow.next.next = fast;
            curr = curr.next;
            slow = slow.next;
        }
        return head;
    }
}
