class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode res = null;
        ListNode tail = null;
        while (var1 != null) {
            ListNode newnode = new ListNode(var1.val);
            if (res == null) {
                res = newnode;
                tail = newnode;
            } else {
                tail.next = newnode;
                tail = newnode;
            }
            if (var1.next == null) {
                break;
            }
            var1 = var1.next.next;
        }
        while (var2 != null) {
            ListNode newnode = new ListNode(var2.val);
            tail.next = newnode;
            tail = newnode;
            if (var2.next == null) {
                break;
            }
            var2 = var2.next.next;
        }
        return res;
    }
}
