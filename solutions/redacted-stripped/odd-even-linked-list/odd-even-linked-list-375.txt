class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode iterator = head.next.next;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = head.next;
        int i = 1;
        while (iterator != null) {
            if (i % 2 != 0) {
                var1.next = iterator;
                var1 = iterator;
            } else {
                var3.next = iterator;
                var3 = iterator;
            }
            if (iterator.next != null) {
                iterator = iterator.next;
            } else {
                var1.next = var2;
                var3.next = null;
                break;
            }
            i++;
        }
        return head;
    }
}
