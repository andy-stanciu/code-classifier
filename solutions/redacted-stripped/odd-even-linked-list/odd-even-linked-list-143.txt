class Solution {

    public ListNode method1(ListNode head) {
        int i = 1;
        ListNode var1 = new ListNode(), var2 = new ListNode(), cursor = head, var3 = var1, var4 = var2;
        while (cursor != null) {
            if ((i & 1) == 1) {
                var3.next = cursor;
                var3 = cursor;
            } else {
                var4.next = cursor;
                var4 = cursor;
            }
            cursor = cursor.next;
            i++;
        }
        var4.next = null;
        var3.next = var2.next;
        return var1.next;
    }
}
