class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null || head.next.next == null)
            return head;
        ListNode head2 = null, temp = head, curr = head, node = null;
        while (curr.next != null && curr.next.next != null) {
            temp = curr.next;
            curr.next = temp.next;
            temp.next = null;
            if (head2 == null) {
                head2 = temp;
                node = temp;
            } else {
                node.next = temp;
                node = node.next;
            }
            curr = curr.next;
        }
        if (curr.next != null && curr.next.next == null)
            node.next = curr.next;
        curr.next = head2;
        return head;
    }
}
