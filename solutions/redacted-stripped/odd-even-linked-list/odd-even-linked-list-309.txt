class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode var1 = new ListNode(-1);
        ListNode oHead = var1;
        ListNode var2 = new ListNode(-1);
        ListNode eHead = var2;
        ListNode curr = head;
        int index = 1;
        while (curr != null) {
            if (index % 2 != 0) {
                oHead.next = curr;
                oHead = oHead.next;
            } else {
                eHead.next = curr;
                eHead = eHead.next;
            }
            curr = curr.next;
            index++;
        }
        eHead.next = null;
        oHead.next = var2.next;
        return var1.next;
    }
}
