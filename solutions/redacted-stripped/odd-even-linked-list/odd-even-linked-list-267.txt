class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ArrayList<Integer> arr = new ArrayList<>();
        ListNode temp = head;
        while (temp != null) {
            arr.add(temp.val);
            if (temp.next != null) {
                temp = temp.next.next;
            } else {
                break;
            }
        }
        temp = head.next;
        while (temp != null) {
            arr.add(temp.val);
            if (temp.next != null) {
                temp = temp.next.next;
            } else {
                break;
            }
        }
        temp = head;
        int i = 0;
        while (temp != null) {
            temp.val = arr.get(i);
            i++;
            temp = temp.next;
        }
        return head;
    }
}
