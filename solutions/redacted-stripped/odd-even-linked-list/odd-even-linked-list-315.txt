class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(1);
        ListNode var2 = new ListNode(1);
        ListNode tempe = var2;
        ListNode tempo = var1;
        ListNode temp = head;
        while (temp != null) {
            tempo.next = temp;
            tempo = tempo.next;
            temp = temp.next;
            tempe.next = temp;
            if (temp == null)
                break;
            tempe = tempe.next;
            temp = temp.next;
        }
        var1 = var1.next;
        var2 = var2.next;
        tempo.next = var2;
        return var1;
    }
}
