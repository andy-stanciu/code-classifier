class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode temp = new ListNode();
        temp = head;
        int k = 0;
        while (temp.next != null) {
            temp = temp.next;
            k++;
        }
        k = (k + 1) / 2;
        ListNode temph = new ListNode();
        temph = head;
        while ((k--) > 0) {
            ListNode tempN = new ListNode();
            tempN = temph.next;
            temph.next = temph.next.next;
            temp.next = tempN;
            tempN.next = null;
            temp = tempN;
            temph = temph.next;
        }
        return head;
    }
}
