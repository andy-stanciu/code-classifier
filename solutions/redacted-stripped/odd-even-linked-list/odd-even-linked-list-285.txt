class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode curr = head, next = head, newhead = head.next;
        int count = 1;
        while (curr.next != null) {
            next = curr.next;
            if (next.next == null) {
                if ((count & 1) == 0) {
                    curr.next = next.next;
                } else {
                    curr.next = newhead;
                }
            } else
                curr.next = next.next;
            curr = next;
            count++;
        }
        if ((count & 1) == 0)
            curr.next = null;
        else
            curr.next = newhead;
        return head;
    }
}
