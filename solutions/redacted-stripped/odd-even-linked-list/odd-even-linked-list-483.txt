class Solution {

    class Node {

        int data;

        Node next;

        Node(int data, Node next) {
            this.data = data;
            this.next = next;
        }

        Node(int data) {
            this.data = data;
            this.next = null;
        }
    }

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = head.next;
        while (var2 != null && var2.next != null) {
            var1.next = var1.next.next;
            var2.next = var2.next.next;
            var1 = var1.next;
            var2 = var2.next;
        }
        var1.next = var3;
        return head;
    }
}
