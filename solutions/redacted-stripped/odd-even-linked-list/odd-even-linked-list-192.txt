class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode eHead = head.next, p = head, ep = head.next, pre = null;
        while (p != null && ep != null) {
            p.next = ep.next;
            pre = p;
            p = p.next;
            if (ep.next != null) {
                ep.next = ep.next.next;
            }
            ep = ep.next;
        }
        if (p != null)
            p.next = eHead;
        else if (pre != null)
            pre.next = eHead;
        return head;
    }
}
