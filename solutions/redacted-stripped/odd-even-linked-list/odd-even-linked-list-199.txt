class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode tail = head;
        int numNodes = 1;
        while (tail.next != null) {
            tail = tail.next;
            numNodes++;
        }
        if (numNodes == 1 || numNodes == 2)
            return head;
        int n = numNodes / 2;
        ListNode dummy = head;
        for (int i = 0; i < n; i++) {
            ListNode save = head.next.next;
            tail.next = head.next;
            tail.next.next = null;
            head.next = save;
            head = head.next;
            tail = tail.next;
        }
        return dummy;
    }
}
