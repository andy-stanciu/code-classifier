class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode os = head;
        ListNode oe = head;
        ListNode es = head.next;
        ListNode ee = head.next;
        while (ee != null && ee.next != null) {
            oe.next = oe.next.next;
            ee.next = ee.next.next;
            oe = oe.next;
            ee = ee.next;
        }
        oe.next = es;
        return os;
    }
}
