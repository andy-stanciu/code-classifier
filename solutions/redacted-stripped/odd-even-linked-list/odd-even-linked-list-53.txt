class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode t1 = head, t2 = head.next, th = t2;
        while (t2 != null && t2.next != null) {
            t1.next = t1.next.next;
            t2.next = t2.next.next;
            t1 = t1.next;
            t2 = t2.next;
        }
        t1.next = th;
        return head;
    }
}
