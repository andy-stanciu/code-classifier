class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1, var2;
        var1 = new ListNode(head.val);
        var2 = new ListNode(head.next.val);
        ListNode var3 = var1;
        ListNode var4 = var2;
        head = head.next.next;
        while (head != null && head.next != null) {
            var3.next = new ListNode(head.val);
            var4.next = new ListNode(head.next.val);
            var4 = var4.next;
            var3 = var3.next;
            head = head.next.next;
        }
        if (head != null) {
            var3.next = new ListNode(head.val);
            var3 = var3.next;
        }
        var3.next = var2;
        return var1;
    }
}
