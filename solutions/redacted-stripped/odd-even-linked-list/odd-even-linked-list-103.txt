class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode var1 = head.next;
        ListNode var2 = head, var3 = head.next;
        while (var3 != null && var3.next != null) {
            var2.next = var3.next;
            var2 = var2.next;
            var3.next = var2.next;
            var3 = var3.next;
        }
        var2.next = var1;
        return head;
    }
}
