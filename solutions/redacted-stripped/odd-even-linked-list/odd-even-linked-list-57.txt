class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode var1 = new ListNode(head.val);
        ;
        ListNode var2 = new ListNode(head.next.val);
        ;
        ListNode var3 = var1;
        ListNode var4 = var2;
        ListNode ptr = head.next.next;
        int count = 2;
        while (ptr != null) {
            count++;
            if (count % 2 == 0) {
                ListNode n = new ListNode(ptr.val);
                var4.next = n;
                var4 = var4.next;
            } else {
                ListNode n = new ListNode(ptr.val);
                var3.next = n;
                var3 = var3.next;
            }
            if (ptr.next == null) {
                var3.next = var2;
            }
            ptr = ptr.next;
        }
        return var1;
    }
}
