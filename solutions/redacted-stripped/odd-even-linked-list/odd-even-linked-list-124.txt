class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode half = head.next;
        ListNode temp = head.next.next;
        while (temp != null) {
            var1.next = temp;
            var1 = var1.next;
            temp = temp.next;
            var2.next = temp;
            var2 = var2.next;
            ;
            if (temp != null)
                temp = temp.next;
        }
        var1.next = half;
        return head;
    }
}
