class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = new ListNode(-1);
        ListNode var2 = new ListNode(-1);
        ListNode var3 = var1;
        ListNode var4 = var2;
        ListNode current = head;
        int index = 1;
        while (current != null) {
            if (index % 2 == 1) {
                var3.next = current;
                var3 = var3.next;
            } else {
                var4.next = current;
                var4 = var4.next;
            }
            current = current.next;
            index++;
        }
        var3.next = null;
        var4.next = null;
        var3.next = var2.next;
        return var1.next;
    }
}
