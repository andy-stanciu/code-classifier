class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummyo = new ListNode(-1);
        ListNode dummyo1 = dummyo;
        ListNode dummye = new ListNode(-1);
        ListNode dummye1 = dummye;
        ListNode temp = head;
        while (temp != null) {
            dummye1.next = temp;
            dummye1 = dummye1.next;
            temp = temp.next;
            if (temp != null) {
                dummyo1.next = temp;
                dummyo1 = dummyo1.next;
                temp = temp.next;
            }
        }
        dummye1.next = dummyo.next;
        dummyo1.next = null;
        return dummye.next;
    }
}
