class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode var1 = head, var2 = head.next, var3 = head.next;
        while (var2 != null && var2.next != null) {
            var1.next = var2.next;
            ListNode temp = var2.next.next;
            var1.next.next = var3;
            var2.next = temp;
            var1 = var1.next;
            var2 = var2.next;
        }
        return head;
    }
}
