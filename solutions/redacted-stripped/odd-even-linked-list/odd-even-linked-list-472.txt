class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = head.next;
        while (var1.next != null && var1.next.next != null) {
            var1.next = var2.next;
            var1 = var2.next;
            var2.next = var1.next;
            var2 = var1.next;
        }
        var1.next = var3;
        return head;
    }
}
