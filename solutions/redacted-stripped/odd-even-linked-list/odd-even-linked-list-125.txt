class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        } else {
            ListNode var1 = head;
            ListNode var2 = head.next;
            ListNode ans2 = head.next;
            while (var1.next != null && var2.next != null) {
                var1.next = var1.next.next;
                var1 = var1.next;
                var2.next = var2.next.next;
                var2 = var2.next;
            }
            var1.next = ans2;
            return head;
        }
    }
}
