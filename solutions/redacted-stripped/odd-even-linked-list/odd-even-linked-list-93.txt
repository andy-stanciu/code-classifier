class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode fast = head;
        ListNode join = head.next;
        ListNode slow = join;
        while (fast.next != null && slow.next != null) {
            fast.next = slow.next;
            fast = fast.next;
            slow.next = fast.next;
            slow = slow.next;
        }
        fast.next = join;
        return head;
    }
}
