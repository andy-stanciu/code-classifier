class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode first = head;
        ListNode second = head.next;
        ListNode var1 = first;
        ListNode var2 = second;
        while (var1.next != null && var1.next.next != null) {
            var1.next = var1.next.next;
            var1 = var1.next;
            var2.next = var1.next;
            if (var2.next != null) {
                var2 = var2.next;
            }
        }
        var1.next = second;
        return head;
    }
}
