class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head;
        ListNode var3 = head.next;
        ListNode var4 = head.next;
        while (var3 != null && var3.next != null) {
            var1.next = var1.next.next;
            var1 = var1.next;
            var3.next = var3.next.next;
            var3 = var3.next;
        }
        var1.next = var4;
        return var2;
    }
}
