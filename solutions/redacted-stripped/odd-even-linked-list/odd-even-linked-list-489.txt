class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode();
        ListNode var2 = new ListNode();
        ListNode en = var1;
        ListNode od = var2;
        int count = 1;
        while (head != null) {
            if (count % 2 == 0) {
                var1.next = head;
                head = head.next;
                var1 = var1.next;
                var1.next = null;
            } else {
                var2.next = head;
                head = head.next;
                var2 = var2.next;
                var2.next = null;
            }
            count++;
        }
        var2.next = en.next;
        return od.next;
    }
}
