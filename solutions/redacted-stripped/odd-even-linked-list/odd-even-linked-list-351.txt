class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        int index = 1;
        int length = 1;
        ListNode tail = head;
        while (tail.next != null) {
            length++;
            tail = tail.next;
        }
        ListNode current = head;
        ListNode prev = null;
        while (current != null && index <= length) {
            if (index % 2 == 0 && current != tail) {
                ListNode next = current.next;
                current.next = null;
                tail.next = current;
                tail = current;
                prev.next = next;
                current = next;
            } else {
                prev = current;
                current = current.next;
            }
            index++;
        }
        return head;
    }
}
