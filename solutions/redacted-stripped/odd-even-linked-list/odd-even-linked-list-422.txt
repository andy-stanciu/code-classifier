class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode var1 = head.next;
        ListNode var2 = head.next;
        ListNode var3 = head;
        ListNode var4 = var2.next.next;
        ListNode var5 = var3.next.next;
        while (var5 != null || var4 != null) {
            var3.next = var5;
            var3 = var3.next;
            var3.next = var1;
            var2.next = var4;
            if (var4 == null)
                return head;
            var2 = var2.next;
            var5 = var2.next;
            if (var5 == null)
                return head;
            var4 = var5.next;
        }
        return head;
    }
}
