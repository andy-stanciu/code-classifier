class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode start = head;
        ListNode end = head;
        while (end.next != null) {
            end = end.next;
        }
        ListNode tempEnd = end;
        while (start != null && start != end) {
            ListNode next = start.next;
            if (next != null) {
                start.next = next.next;
                tempEnd.next = next;
                next.next = null;
                tempEnd = tempEnd.next;
                start = start.next;
                if (end == tempEnd) {
                    break;
                }
            } else {
                start.next = next;
            }
        }
        return head;
    }
}
