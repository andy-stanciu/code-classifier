class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = var1.next;
        ListNode prev = var2;
        ListNode nextNode = var2.next;
        while (var2 != null && nextNode != null) {
            var1.next = nextNode;
            var2.next = nextNode.next;
            var1 = var1.next;
            var2 = var2.next;
            if (var2 != null) {
                nextNode = var2.next;
            }
        }
        var1.next = prev;
        return head;
    }
}
