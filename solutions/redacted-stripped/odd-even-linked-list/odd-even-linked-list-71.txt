class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        ListNode var1 = head, var2 = head.next, q = head.next;
        while (var1.next != null && var2.next != null) {
            var1.next = var2.next;
            var1 = var1.next;
            var2.next = var1.next;
            var2 = var2.next;
        }
        var1.next = q;
        return head;
    }
}
