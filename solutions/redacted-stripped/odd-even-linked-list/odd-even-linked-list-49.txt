class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode e = head.next;
        ListNode i = e;
        ListNode j = i.next;
        ListNode last = j, prev = head;
        int k = 0;
        while (j != null) {
            prev.next = j;
            prev = i;
            i = i.next;
            last = j;
            j = j.next;
            k++;
        }
        if (k % 2 == 1) {
            prev.next = null;
            last.next = e;
            return head;
        }
        prev.next = e;
        return head;
    }
}
