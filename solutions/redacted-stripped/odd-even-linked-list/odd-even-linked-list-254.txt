class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode joiner = head.next;
        while (var2 != null && var2.next != null) {
            ListNode tempO = var1.next.next;
            ListNode tempE = var2.next != null ? var2.next.next : null;
            var1.next = tempO;
            var2.next = tempE;
            var1 = tempO;
            var2 = tempE;
        }
        var1.next = joiner;
        return head;
    }
}
