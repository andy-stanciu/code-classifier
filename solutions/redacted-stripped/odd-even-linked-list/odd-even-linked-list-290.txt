class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head, var2 = head.next, buffer = head, var3 = head.next;
        while (var2 != null && var2.next != null) {
            var1.next = var2.next;
            var1 = var1.next;
            if (var1 != null) {
                var2.next = var1.next;

    }
