class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = var1;
        ListNode var4 = var2;
        while (var4 != null && var4.next != null) {
            var3.next = var4.next;
            var3 = var3.next;
            var4.next = var3.next;
            var4 = var4.next;
        }
        var3.next = var2;
        return var1;
    }
}
