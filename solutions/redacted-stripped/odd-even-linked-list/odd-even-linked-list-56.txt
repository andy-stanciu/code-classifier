class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode fast1 = head, fast2 = head.next, fastHead = fast2;
        while (fast2 != null && fast2.next != null) {
            fast1.next = fast2.next;
            fast1 = fast1.next;
            fast2.next = fast1.next;
            fast2 = fast1.next;
        }
        fast1.next = fastHead;
        return head;
    }
}
