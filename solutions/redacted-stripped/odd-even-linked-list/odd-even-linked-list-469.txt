class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode slow = head;
        ListNode fast = slow;
        while (fast.next != null) {
            fast = fast.next;
        }
        ListNode mid = fast;
        while (slow != mid && slow != mid.next) {
            fast.next = new ListNode(slow.next.val);
            fast = fast.next;
            slow.next = slow.next.next;
            slow = slow.next;
        }
        return head;
    }
}
