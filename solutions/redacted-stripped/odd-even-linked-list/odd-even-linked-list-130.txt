class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode odTail = head;
        ListNode var1 = head.next, var2 = head.next;
        while (odTail.next != null && var2.next != null) {
            odTail.next = odTail.next.next;
            var2.next = var2.next.next;
            odTail = odTail.next;
            var2 = var2.next;
        }
        odTail.next = var1;
        return head;
    }
}
