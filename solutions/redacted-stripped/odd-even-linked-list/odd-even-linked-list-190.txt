class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode temp = null;
        while (var2 != null && var2.next != null) {
            temp = var2.next.next;
            var2.next.next = var1.next;
            var1.next = var2.next;
            var2.next = temp;
            var2 = var2.next;
            var1 = var1.next;
        }
        return head;
    }
}
