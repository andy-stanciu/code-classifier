class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode();
        ListNode var2 = new ListNode();
        ListNode od = var1;
        ListNode ev = var2;
        int i = 0;
        while (head != null) {
            i++;
            if (i % 2 != 0) {
                od.next = head;
                od = od.next;
            } else {
                ev.next = head;
                ev = ev.next;
            }
            head = head.next;
        }
        od.next = var2.next;
        ev.next = null;
        return var1.next;
    }
}
