class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode h = head, n = null, pre = null;
        ListNode dummy = new ListNode();
        ListNode d = dummy;
        while (head != null && head.next != null) {
            pre = head;
            dummy.next = head.next;
            head.next = head.next.next;
            head = head.next;
            dummy = dummy.next;
            dummy.next = n;
        }
        if (head == null) {
            pre.next = d.next;
        } else {
            head.next = d.next;
        }
        return h;
    }
}
