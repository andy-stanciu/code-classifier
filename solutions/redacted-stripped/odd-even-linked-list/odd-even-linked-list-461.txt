class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode temp = var2;
        while (var1.next != null && temp.next != null) {
            var1.next = temp.next;
            var1 = temp.next;
            temp.next = var1.next;
            temp = var1.next;
        }
        var1.next = var2;
        return head;
    }
}
