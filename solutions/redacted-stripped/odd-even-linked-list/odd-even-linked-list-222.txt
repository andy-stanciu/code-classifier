class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = new ListNode(0);
        ListNode var2 = var1;
        ListNode var3 = new ListNode(0);
        ListNode var4 = var3;
        while (head != null) {
            ListNode newNode = new ListNode(head.val);
            var2.next = newNode;
            var2 = newNode;
            head = head.next;
            if (head != null) {
                ListNode newNode1 = new ListNode(head.val);
                var4.next = newNode1;
                var4 = newNode1;
                head = head.next;
            }
        }
        ListNode res = new ListNode(0);
        ListNode temp = res;
        var1 = var1.next;
        while (var1 != null) {
            ListNode newNode = new ListNode(var1.val);
            temp.next = newNode;
            temp = newNode;
            var1 = var1.next;
        }
        var3 = var3.next;
        while (var3 != null) {
            ListNode newNode = new ListNode(var3.val);
            temp.next = newNode;
            temp = newNode;
            var3 = var3.next;
        }
        return res.next;
    }
}
