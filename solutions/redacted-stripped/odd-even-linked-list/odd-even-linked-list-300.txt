class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode evehead = head.next;
        while (var1.next != null && var2.next != null) {
            var1.next = var1.next.next;
            var2.next = var2.next.next;
            var1 = var1.next;
            var2 = var2.next;
        }
        var1.next = evehead;
        return head;
    }
}
