class Solution {

    public ListNode method1(ListNode head) {
        ListNode tail = head;
        while (tail != null && tail.next != null) {
            tail = tail.next;
        }
        ListNode temp = head;
        int count = 1;
        ListNode prev = null;
        int var1 = method2(head);
        while (temp != null && count <= var1) {
            if (count % 2 == 0) {
                tail.next = temp;
                prev.next = temp.next;
                temp.next = null;
                tail = temp;
                temp = prev.next;
            } else {
                prev = temp;
                temp = temp.next;
            }
            count++;
        }
        return head;
    }

    public static int method2(ListNode head) {
        ListNode temp = head;
        int size = 0;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        return size;
    }
}
