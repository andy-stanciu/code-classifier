class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode h1 = head;
        ListNode h2 = h1.next;
        ListNode h3 = h2;
        while (h1 != null && h2 != null && h1.next != null && h2.next != null) {
            h1.next = h2.next;
            h2.next = null;
            h1 = h1.next;
            h2.next = h1.next;
            h2 = h2.next;
        }
        h1.next = h3;
        return head;
    }
}
