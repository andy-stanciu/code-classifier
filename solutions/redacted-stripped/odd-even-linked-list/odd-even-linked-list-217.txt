class Solution {

    public ListNode method1(ListNode A) {
        if (A == null || A.next == null) {
            return A;
        }
        ListNode head1 = new ListNode(0);
        ListNode head2 = new ListNode(0);
        ListNode cur1 = head1;
        ListNode cur2 = head2;
        ListNode cur = A;
        int count = 1;
        while (cur != null) {
            ListNode next = cur.next;
            cur.next = null;
            if (count % 2 != 0) {
                cur1.next = cur;
                cur1 = cur;
            } else {
                cur2.next = cur;
                cur2 = cur;
            }
            cur = next;
            count++;
        }
        cur1.next = head2.next;
        return head1.next;
    }
}
