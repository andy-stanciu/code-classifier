class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = new ListNode(var1.val);
        ListNode var3 = var2;
        ListNode var4 = head.next;
        ListNode var5 = new ListNode(var4.val);
        ListNode var6 = var5;
        while (var1.next != null && var4.next != null) {
            var1 = var1.next.next;
            if (var1 != null) {
                var3.next = new ListNode(var1.val);
                var3 = var3.next;
            }
            var4 = var4.next.next;
            if (var4 != null) {
                var6.next = new ListNode(var4.val);
                var6 = var6.next;
            }
        }
        var3.next = var5;
        return var2;
    }
}
