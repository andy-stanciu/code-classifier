class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode h1 = head.next, t1 = head, t2 = h1;
        while (t1.next != null && t2.next != null) {
            ListNode t = t2.next;
            t2.next = t2.next.next;
            t1.next = t;
            t1 = t1.next;
            t2 = t2.next;
        }
        t1.next = h1;
        return head;
    }
}
