class Solution {

    public static ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        int cnt = 1;
        ListNode crnt = head;
        ListNode cnect = head;
        ListNode var1 = head.next;
        while (crnt.next != null) {
            ListNode temp = crnt;
            if (cnt % 2 == 1) {
                cnect = crnt;
            }
            cnt++;
            crnt = crnt.next;
            temp.next = temp.next.next;
        }
        if (cnt % 2 == 0)
            cnect.next = var1;
        else
            crnt.next = var1;
        return head;
    }
}
