class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head.next;
        ListNode var2 = head;
        ListNode var3 = var1;
        while (var1 != null && var1.next != null) {
            var2.next = var1.next;
            var2 = var2.next;
            var1.next = var2.next;
            var1 = var1.next;
        }
        var2.next = var3;
        return head;
    }
}
