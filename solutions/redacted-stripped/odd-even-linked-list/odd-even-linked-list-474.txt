class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode var1 = null, var2 = null;
        ListNode var3 = null, var4 = null;
        ListNode current = head;
        int currentIndex = 1;
        while (current != null) {
            ListNode nodeToBeAdded = current;
            current = current.next;
            nodeToBeAdded.next = null;
            if (currentIndex % 2 == 0) {
                if (var3 == null) {
                    var3 = nodeToBeAdded;
                    var4 = nodeToBeAdded;
                } else {
                    var4.next = nodeToBeAdded;
                    var4 = var4.next;
                }
            } else {
                if (var1 == null) {
                    var1 = nodeToBeAdded;
                    var2 = nodeToBeAdded;
                } else {
                    var2.next = nodeToBeAdded;
                    var2 = var2.next;
                }
            }
            currentIndex++;
        }
        var2.next = var3;
        return head;
    }
}
