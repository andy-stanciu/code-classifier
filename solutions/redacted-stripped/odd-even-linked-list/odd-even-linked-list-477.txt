class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode temp = head;
        List<Integer> li = new ArrayList<>();
        while (temp != null && temp.next != null) {
            li.add(temp.val);
            temp = temp.next.next;
        }
        if (temp != null)
            li.add(temp.val);
        temp = head.next;
        while (temp != null && temp.next != null) {
            li.add(temp.val);
            temp = temp.next.next;
        }
        if (temp != null)
            li.add(temp.val);
        temp = head;
        int i = 0;
        while (temp != null) {
            temp.val = li.get(i);
            i++;
            temp = temp.next;
        }
        return head;
    }
}
