class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode estart = null;
        ListNode eend = null;
        ListNode ostart = null;
        ListNode oend = null;
        ListNode curr = head;
        int x = 1;
        while (curr != null) {
            if (x % 2 == 0) {
                if (estart == null) {
                    estart = curr;
                    eend = estart;
                } else {
                    eend.next = curr;
                    eend = eend.next;
                }
            } else {
                if (ostart == null) {
                    ostart = curr;
                    oend = ostart;
                } else {
                    oend.next = curr;
                    oend = oend.next;
                }
            }
            curr = curr.next;
            x++;
        }
        if (estart == null || ostart == null) {
            return head;
        }
        oend.next = estart;
        eend.next = null;
        return ostart;
    }
}
