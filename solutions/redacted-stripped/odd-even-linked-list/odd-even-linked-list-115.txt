class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode ans = head;
        ListNode semi = head.next;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode traverse = var2.next;
        int index = 0;
        while (traverse != null) {
            if (index % 2 == 0) {
                var1.next = traverse;
                var1 = traverse;
            } else {
                var2.next = traverse;
                var2 = traverse;
            }
            traverse = traverse.next;
            index++;
        }
        var2.next = null;
        var1.next = semi;
        return ans;
    }
}
