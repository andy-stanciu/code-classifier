class Solution {

    static ListNode ans;

    public static ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ans = null;
        ListNode first = head;
        ListNode second = head.next;
        while (first != null) {
            addLast(first.val);
            first = (first.next != null) ? first.next.next : null;
        }
        while (second != null) {
            addLast(second.val);
            second = (second.next != null) ? second.next.next : null;
        }
        return ans;
    }

    public static void addLast(int val) {
        ListNode newNode = new ListNode(val);
        if (ans == null) {
            ans = newNode;
            return;
        }
        ListNode temp = ans;
        while (temp.next != null) {
            temp = temp.next;
        }
        temp.next = newNode;
        newNode.next = null;
    }
}
