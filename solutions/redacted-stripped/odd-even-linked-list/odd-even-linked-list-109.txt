class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode temp1 = head;
        ListNode temp2 = head.next;
        ListNode dummy = temp2;
        while (temp2 != null && temp2.next != null) {
            temp1.next = temp2.next;
            temp1 = temp1.next;
            temp2.next = temp1.next;
            temp2 = temp2.next;
        }
        temp1.next = dummy;
        return head;
    }
}
