class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode temp = head.next;
        ListNode prev = var1;
        while (var1 != null && var2 != null) {
            var1.next = var2.next;
            var1 = var1.next;
            if (var1 != null) {
                prev = var1;
                var2.next = var1.next;
                var2 = var2.next;
            }
        }
        prev.next = temp;
        return head;
    }
}
