class Solution {

    public ListNode method1(ListNode head) {
        ListNode head2 = new ListNode();
        ListNode temp = head2;
        ListNode temp1 = head;
        while (temp1 != null) {
            temp.next = temp1.next;
            if (temp1.next == null) {
                temp1.next = head2.next;
                break;
            }
            if (temp1.next.next == null) {
                temp1.next = head2.next;
                break;
            }
            temp1.next = temp1.next.next;
            temp = temp.next;
            temp1 = temp1.next;
        }
        return head;
    }
}
