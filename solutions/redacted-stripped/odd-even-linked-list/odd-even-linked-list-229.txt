class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = new ListNode(-1);
        ListNode var2 = var1;
        ListNode var3 = new ListNode(-1);
        ListNode var4 = var3;
        ListNode curr = head;
        ListNode temp;
        boolean flag = true;
        while (curr != null) {
            temp = curr;
            curr = curr.next;
            temp.next = null;
            if (flag == true) {
                var2.next = temp;
                var2 = temp;
                flag = false;
            } else {
                var4.next = temp;
                var4 = temp;
                flag = true;
            }
        }
        var2.next = var3.next;
        return var1.next;
    }
}
