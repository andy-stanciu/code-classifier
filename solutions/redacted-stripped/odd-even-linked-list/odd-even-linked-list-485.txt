class Solution {

    public ListNode method1(ListNode head) {
        ListNode current = head;
        ListNode prev = null;
        Queue<ListNode> var1 = new LinkedList<>();
        Queue<ListNode> var2 = new LinkedList<>();
        int nodeCounter = 1;
        while (current != null) {
            if (nodeCounter % 2 == 0) {
                var2.add(new ListNode(current.val));
            } else {
                var1.add(new ListNode(current.val));
            }
            current = current.next;
            nodeCounter += 1;
        }
        ListNode newHead = var1.poll();
        ListNode cur2 = newHead;
        while (!var1.isEmpty()) {
            cur2.next = var1.poll();
            cur2 = cur2.next;
        }
        while (!var2.isEmpty()) {
            cur2.next = var2.poll();
            cur2 = cur2.next;
        }
        return newHead;
    }
}
