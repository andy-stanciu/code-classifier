class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head.next;
        ListNode var2 = head.next;
        ListNode var3 = head;
        while (var1 != null && var1.next != null) {
            var3.next = var3.next.next;
            var1.next = var1.next.next;
            var3 = var3.next;
            var1 = var1.next;
        }
        var3.next = var2;
        return head;
    }
}
