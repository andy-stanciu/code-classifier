class Solution {

    public List<Integer> method1(int[] arr) {
        HashMap<Integer, Integer> hp = new HashMap<>();
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            hp.put(arr[i], hp.getOrDefault(arr[i], 0) + 1);
        }
        int x = n / 3;
        List<Integer> ans = new ArrayList<>();
        for (var a : hp.keySet()) {
            if (hp.get(a) > x)
                ans.add(a);
        }
        return ans;
    }
}
