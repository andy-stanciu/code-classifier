class Solution {

    public List<Integer> method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        ArrayList<Integer> list = new ArrayList<>();
        int N = nums.length;
        for (int i : nums) {
            if (map.containsKey(i)) {
                map.put(i, map.get(i) + 1);
            } else {
                map.put(i, 1);
            }
        }
        for (int key : map.keySet()) {
            if (map.get(key) > N / 3) {
                list.add(key);
            } else {
                continue;
            }
        }
        return list;
    }
}
