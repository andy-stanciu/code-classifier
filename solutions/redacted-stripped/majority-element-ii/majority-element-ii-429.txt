class Solution {

    public List<Integer> method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        ArrayList<Integer> list = new ArrayList<>();
        int n = nums.length;
        int currvalue;
        for (int i = 0; i < nums.length; i++) {
            currvalue = nums[i];
            if (!map.containsKey(currvalue)) {
                map.put(currvalue, 1);
            } else {
                int a = map.get(currvalue);
                map.put(currvalue, a + 1);
            }
        }
        Set<Integer> set = map.keySet();
        for (Integer j : set) {
            if (map.get(j) > (n / 3)) {
                list.add(j);
            }
        }
        return list;
    }
}
