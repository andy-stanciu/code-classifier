class Solution {

    public List<Integer> method1(int[] nums) {
        HashMap<Integer, Integer> h = new HashMap<>();
        for (int i : nums) {
            h.put(i, h.getOrDefault(i, 0) + 1);
        }
        List<Integer> l = new ArrayList<>();
        for (Map.Entry<Integer, Integer> m : h.entrySet()) {
            if (m.getValue() > nums.length / 3)
                l.add(m.getKey());
        }
        return l;
    }
}
