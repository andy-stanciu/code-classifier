class Solution {

    public static List<Integer> method1(int[] nums) {
        Set<Integer> list = new HashSet<>();
        int count = 0;
        int n = nums.length;
        Arrays.sort(nums);
        int num = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == num) {
                count++;
            } else {
                num = nums[i];
                count = 1;
            }
            if (count > n / 3) {
                list.add(nums[i]);
            }
        }
        ArrayList<Integer> ans = new ArrayList<>(list);
        return ans;
    }
}
