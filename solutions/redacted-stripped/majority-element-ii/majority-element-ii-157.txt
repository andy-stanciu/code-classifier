class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        Map<Integer, Integer> map = new HashMap<>();
        for (int ele : nums) {
            if (map.containsKey(ele))
                map.put(ele, map.get(ele) + 1);
            else
                map.put(ele, 1);
        }
        for (int ele : map.keySet()) {
            if (map.get(ele) > (nums.length / 3))
                ans.add(ele);
        }
        return ans;
    }
}
