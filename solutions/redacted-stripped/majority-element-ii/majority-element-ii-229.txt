class Solution {

    public List<Integer> method1(int[] nums) {
        ArrayList<Integer> list = new ArrayList<>();
        int a = nums.length / 3;
        HashMap<Integer, Integer> mp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            mp.put(nums[i], mp.getOrDefault(nums[i], 0) + 1);
        }
        for (int x : mp.keySet()) {
            if (mp.get(x) > a)
                list.add(x);
        }
        return list;
    }
}
