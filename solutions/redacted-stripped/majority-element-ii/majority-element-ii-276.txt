class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        int value = n / 3;
        Map<Integer, Integer> var1 = new HashMap<>();
        for (int i = 0; i < n; i++) {
            var1.put(nums[i], var1.getOrDefault(nums[i], 0) + 1);
        }
        List<Integer> var2 = new ArrayList<>();
        for (var pair : var1.entrySet()) {
            int var3 = pair.getKey();
            int count = pair.getValue();
            if (count > value) {
                var2.add(var3);
            }
        }
        return var2;
    }
}
