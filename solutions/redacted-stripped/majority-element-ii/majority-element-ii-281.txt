class Solution {

    public List<Integer> method1(int[] nums) {
        int len = nums.length;
        int lowerCheck = len / 3;
        Map<Integer, Integer> arrList = new HashMap<>();
        List<Integer> arr = new ArrayList<>();
        for (int x : nums) {
            arrList.put(x, (arrList.getOrDefault(x, 0) + 1));
        }
        for (Map.Entry<Integer, Integer> entry : arrList.entrySet()) {
            int var1 = entry.getKey();
            int count = entry.getValue();
            if (count > lowerCheck) {
                arr.add(var1);
            }
        }
        return arr;
    }
}
