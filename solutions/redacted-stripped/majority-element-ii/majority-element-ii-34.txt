class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        List<Integer> res = new ArrayList<>();
        HashMap<Integer, Integer> mpp = new HashMap<>();
        int min = (n / 3) + 1;
        for (int i = 0; i < n; i++) {
            int value = mpp.getOrDefault(nums[i], 0);
            mpp.put(nums[i], value + 1);
            if (mpp.get(nums[i]) == min) {
                res.add(nums[i]);
            }
            if (res.size() == 2) {
                break;
            }
        }
        return res;
    }
}
