class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        int el1 = 0;
        int el2 = 0;
        int c1 = 0;
        int c2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (c1 == 0 && nums[i] != el2) {
                c1 = 1;
                el1 = nums[i];
            } else if (c2 == 0 && nums[i] != el1) {
                c2 = 1;
                el2 = nums[i];
            } else if (nums[i] == el1) {
                c1++;
            } else if (nums[i] == el2) {
                c2++;
            } else {
                c1--;
                c2--;
            }
        }
        int ct1 = 0;
        int ct2 = 0;
        for (int n : nums) {
            if (n == el1) {
                ct1++;
            } else if (n == el2) {
                ct2++;
            }
        }
        if (ct1 > nums.length / 3) {
            ans.add(el1);
        }
        if (ct2 > nums.length / 3) {
            ans.add(el2);
        }
        return ans;
    }
}
