class Solution {

    public List<Integer> method1(int[] nums) {
        int el1 = Integer.MIN_VALUE, el2 = Integer.MIN_VALUE, cnt1 = 0, cnt2 = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (cnt1 == 0 && el2 != nums[i]) {
                cnt1 = 1;
                el1 = nums[i];
            } else if (cnt2 == 0 && el1 != nums[i]) {
                cnt2 = 1;
                el2 = nums[i];
            } else if (el1 == nums[i])
                cnt1++;
            else if (el2 == nums[i])
                cnt2++;
            else {
                cnt1--;
                cnt2--;
            }
        }
        List<Integer> array = new ArrayList<>();
        cnt1 = 0;
        cnt2 = 0;
        int threshold = (n / 3) + 1;
        for (int i = 0; i < n; i++) {
            if (el1 == nums[i]) {
                cnt1++;
                if (cnt1 == threshold) {
                    array.add(el1);
                }
            }
            if (el2 == nums[i]) {
                cnt2++;
                if (cnt2 == threshold) {
                    array.add(el2);
                }
            }
        }
        Collections.sort(array);
        return array;
    }
}
