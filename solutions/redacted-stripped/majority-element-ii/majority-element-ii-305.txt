class Solution {

    public List<Integer> method1(int[] nums) {
        int c1 = 0, e1 = Integer.MAX_VALUE, e2 = Integer.MAX_VALUE, c2 = 0, i, l = nums.length;
        for (i = 0; i < l; i++) {
            if (c1 == 0 && e2 != nums[i]) {
                e1 = nums[i];
                c1 = 1;
            } else if (c2 == 0 && e1 != nums[i]) {
                e2 = nums[i];
                c2 = 1;
            } else if (e1 == nums[i])
                c1++;
            else if (e2 == nums[i])
                c2++;
            else {
                c1--;
                c2--;
            }
        }
        List<Integer> li = new ArrayList<>();
        c1 = 0;
        c2 = 0;
        for (i = 0; i < l; i++) {
            if (nums[i] == e1)
                c1++;
            if (nums[i] == e2)
                c2++;
        }
        int x = l / 3 + 1;
        if (c1 >= x)
            li.add(e1);
        if (c2 >= x)
            li.add(e2);
        return li;
    }
}
