class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            int count = 0;
            if (ans.size() == 0 || ans.get(0) != nums[i]) {
                for (int j = i; j < nums.length; j++) {
                    if (nums[j] == nums[i])
                        count++;
                }
            }
            if (count > nums.length / 3) {
                ans.add(nums[i]);
            }
            if (ans.size() == 2)
                break;
        }
        return ans;
    }
}
