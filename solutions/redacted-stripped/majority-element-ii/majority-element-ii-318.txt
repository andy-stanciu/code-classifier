class Solution {

    public List<Integer> method1(int[] nums) {
        int var1 = 0;
        int var2 = 0;
        int count1 = 0;
        int count2 = 0;
        for (int i = 0; i < nums.length; ++i) {
            int num = nums[i];
            if (num == var1) {
                count1++;
            } else if (num == var2) {
                count2++;
            } else if (count1 == 0) {
                var1 = num;
                count1 = 1;
            } else if (count2 == 0) {
                var2 = num;
                count2 = 1;
            } else {
                count1--;
                count2--;
            }
        }
        count1 = 0;
        count2 = 0;
        for (int n : nums) {
            if (n == var1)
                count1++;
            if (n == var2)
                count2++;
        }
        Set<Integer> result = new HashSet<>();
        if (count1 > nums.length / 3)
            result.add(var1);
        if (count2 > nums.length / 3)
            result.add(var2);
        return new ArrayList<>(result);
    }
}
