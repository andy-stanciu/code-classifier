class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new LinkedList<>();
        int n = nums.length;
        int res1 = Integer.MIN_VALUE, res2 = Integer.MIN_VALUE, c1 = 0, c2 = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == res1)
                c1++;
            else if (nums[i] == res2)
                c2++;
            else if (c1 == 0) {
                res1 = nums[i];
                c1 = 1;
            } else if (c2 == 0) {
                res2 = nums[i];
                c2 = 1;
            } else {
                c1--;
                c2--;
            }
        }
        int x1 = 0, x2 = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == res1)
                x1++;
            if (nums[i] == res2)
                x2++;
        }
        if (x1 > n / 3)
            ans.add(res1);
        if (x2 > n / 3)
            ans.add(res2);
        return ans;
    }
}
