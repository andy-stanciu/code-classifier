class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Integer> hs = new HashMap<>();
        for (int i : nums) {
            hs.put(i, hs.getOrDefault(i, 0) + 1);
        }
        List<Integer> result = new ArrayList<>();
        for (int i : hs.keySet()) {
            int value = hs.get(i);
            if (value > (n / 3))
                result.add(i);
        }
        return result;
    }
}
