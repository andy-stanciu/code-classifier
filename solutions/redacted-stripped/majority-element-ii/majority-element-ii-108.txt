class Solution {

    int i, j;

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList();
        Arrays.sort(nums);
        for (i = 0; i < nums.length; i = j + 1) {
            for (j = i; j + 1 < nums.length && nums[j + 1] == nums[i]; ++j) ;
            if (j - i + 1 > nums.length / 3)
                ans.add(nums[i]);
        }
        return ans;
    }
}
