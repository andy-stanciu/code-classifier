class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        int cnt1 = 0;
        int cnt2 = 0;
        int el1 = 0;
        int el2 = 0;
        for (int i = 0; i < n; i++) {
            if (cnt1 == 0 && nums[i] != el2) {
                cnt1 = 1;
                el1 = nums[i];
            } else if (cnt2 == 0 && nums[i] != el1) {
                cnt2 = 1;
                el2 = nums[i];
            } else if (el1 == nums[i]) {
                cnt1++;
            } else if (el2 == nums[i]) {
                cnt2++;
            } else {
                cnt1--;
                cnt2--;
            }
        }
        List<Integer> ls = new ArrayList<>();
        int c1 = 0;
        int c2 = 0;
        int mini = (int) (n / 3) + 1;
        for (int i = 0; i < n; i++) {
            if (nums[i] == el1) {
                c1++;
            }
            if (nums[i] == el2) {
                c2++;
            }
        }
        if (c1 >= mini) {
            ls.add(el1);
        }
        if (c2 >= mini && el2 != el1) {
            ls.add(el2);
        }
        return ls;
    }
}
