class Solution {

    public List<Integer> method1(int[] nums) {
        Map<Integer, Integer> myDict = new HashMap<>();
        for (int num : nums) {
            myDict.put(num, myDict.getOrDefault(num, 0) + 1);
        }
        List<Integer> ans = new ArrayList<>();
        int n = nums.length / 3;
        for (Map.Entry<Integer, Integer> entry : myDict.entrySet()) {
            if (entry.getValue() > n) {
                ans.add(entry.getKey());
            }
        }
        return ans;
    }
}
