class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> res = new ArrayList<>();
        int n1 = 0, n2 = 1;
        int count1 = 0, count2 = 0;
        for (int a : nums) {
            if (a == n1)
                count1++;
            else if (a == n2)
                count2++;
            else if (count1 == 0) {
                n1 = a;
                count1++;
            } else if (count2 == 0) {
                n2 = a;
                count2++;
            } else {
                count1--;
                count2--;
            }
        }
        count1 = 0;
        count2 = 0;
        for (int a : nums) {
            if (a == n1)
                count1++;
            else if (a == n2)
                count2++;
        }
        if (count1 > nums.length / 3)
            res.add(n1);
        if (count2 > nums.length / 3)
            res.add(n2);
        return res;
    }
}
