class Solution {

    public List<Integer> method1(int[] nums) {
        int candidate = 0, candidate1 = 0;
        int count = 0, count1 = 0;
        for (int n : nums) {
            if (n == candidate) {
                count++;
            } else if (n == candidate1) {
                count1++;
            } else if (count == 0) {
                candidate = n;
                count++;
            } else if (count1 == 0) {
                candidate1 = n;
                count1++;
            } else {
                count--;
                count1--;
            }
        }
        count = 0;
        count1 = 0;
        for (int num : nums) {
            if (num == candidate) {
                count++;
            } else if (num == candidate1) {
                count1++;
            }
        }
        List<Integer> list = new ArrayList<>();
        if (count > nums.length / 3) {
            list.add(candidate);
        }
        if (count1 > nums.length / 3) {
            list.add(candidate1);
        }
        return list;
    }
}
