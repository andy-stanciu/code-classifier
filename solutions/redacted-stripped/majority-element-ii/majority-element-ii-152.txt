class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Integer> a = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (a.containsKey(nums[i])) {
                int d = a.get(nums[i]);
                a.put(nums[i], d + 1);
            } else
                a.put(nums[i], 1);
        }
        ArrayList<Integer> k = new ArrayList<>();
        for (Entry<Integer, Integer> e : a.entrySet()) {
            if (e.getValue() > n / 3)
                k.add(e.getKey());
        }
        return k;
    }
}
