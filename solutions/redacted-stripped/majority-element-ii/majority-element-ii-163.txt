class Solution {

    public List<Integer> method1(int[] nums) {
        ArrayList<Integer> ans = new ArrayList<>();
        int cou1 = 0;
        int cou2 = 0;
        int ele1 = Integer.MIN_VALUE;
        int ele2 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (cou1 == 0 && ele2 != nums[i]) {
                cou1 = 1;
                ele1 = nums[i];
            } else if (cou2 == 0 && ele1 != nums[i]) {
                cou2 = 1;
                ele2 = nums[i];
            } else if (nums[i] == ele1) {
                cou1++;
            } else if (nums[i] == ele2) {
                cou2++;
            } else {
                cou1--;
                cou2--;
            }
        }
        int c1 = 0;
        int c2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == ele1) {
                c1++;
            }
            if (nums[i] == ele2) {
                c2++;
            }
        }
        int mini = (int) (nums.length / 3) + 1;
        if (c1 >= mini) {
            ans.add(ele1);
        }
        if (c2 >= mini) {
            ans.add(ele2);
        }
        return ans;
    }
}
