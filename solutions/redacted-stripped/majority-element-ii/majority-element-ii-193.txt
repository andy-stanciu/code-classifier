class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        HashMap<Integer, Integer> mp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int cnt = 0;
            if (mp.containsKey(nums[i])) {
                cnt = mp.get(nums[i]);
            }
            cnt++;
            mp.put(nums[i], cnt);
        }
        int n = nums.length / 3;
        for (Integer key : mp.keySet()) {
            if (mp.get(key) > n) {
                ans.add(key);
            }
        }
        return ans;
    }
}
