class Solution {

    public List<Integer> method1(int[] nums) {
        Integer var1 = 0;
        Integer var2 = 0;
        int counter1 = 0;
        int counter2 = 0;
        for (int n : nums) {
            if (n == var1) {
                counter1++;
            } else if (n == var2) {
                counter2++;
            } else if (counter1 == 0) {
                var1 = n;
                counter1++;
            } else if (counter2 == 0) {
                var2 = n;
                counter2++;
            } else {
                counter1--;
                counter2--;
            }
        }
        counter1 = 0;
        counter2 = 0;
        for (int n : nums) {
            if (n == var1) {
                counter1++;
            } else if (n == var2) {
                counter2++;
            }
        }
        List<Integer> res = new ArrayList<>();
        if (counter1 > nums.length / 3) {
            res.add(var1);
        }
        if (counter2 > nums.length / 3) {
            res.add(var2);
        }
        return res;
    }
}
