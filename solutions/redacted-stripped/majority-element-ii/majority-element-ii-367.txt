class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        int candidate1 = Integer.MAX_VALUE;
        int candidate2 = Integer.MIN_VALUE;
        int count1 = 0;
        int count2 = 0;
        List<Integer> ans = new ArrayList<>();
        for (int i : nums) {
            if (i == candidate1) {
                count1++;
            } else if (i == candidate2) {
                count2++;
            } else if (c1 == 0) {
                candidate1 = i;
                count1++;
            } else if (c2 == 0) {
                candidate2 = i;
                count2++;
            } else {
                count1--;
                count2--;
            }
        }
        count1 = 0;
        count2 = 0;
        for (int i : nums) {
            count1 += (i == candidate1) ? 1 : 0;
            count2 += (i == candidate2) ? 1 : 0;
        }
        if (count1 > n / 3) {
            ans.add(candidate1);
        }
        if (count2 > n / 3) {
            ans.add(candidate2);
        }
        return ans;
    }
}
