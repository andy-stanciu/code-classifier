class Solution {

    public List<Integer> method1(int[] nums) {
        int el1 = 0, el2 = 0, count1 = 0, count2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count1 == 0 && nums[i] != el2) {
                el1 = nums[i];
                count1 = 1;
            } else if (count2 == 0 && nums[i] != el1) {
                el2 = nums[i];
                count2 = 1;
            } else if (el1 == nums[i])
                count1++;
            else if (el2 == nums[i])
                count2++;
            else {
                count1--;
                count2--;
            }
        }
        count1 = 0;
        count2 = 0;
        int min = (int) (nums.length / 3) + 1;
        for (int data : nums) {
            if (data == el1)
                count1++;
            if (data == el2)
                count2++;
        }
        List<Integer> ans = new ArrayList<>();
        if (count1 >= min)
            ans.add(el1);
        if (count2 >= min && el1 != el2)
            ans.add(el2);
        return ans;
    }
}
