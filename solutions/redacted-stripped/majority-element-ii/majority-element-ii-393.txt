class Solution {

    public List<Integer> method1(int[] nums) {
        Map<Integer, Integer> dict = new HashMap<>();
        for (var keyValuePair : nums) if (dict.containsKey(keyValuePair))
            dict.put(keyValuePair, dict.get(keyValuePair) + 1);
        else
            dict.put(keyValuePair, 1);
        List<Integer> output = new ArrayList<>();
        for (var keyValuePair : dict.entrySet()) if (keyValuePair.getValue() > nums.length / 3)
            output.add(keyValuePair.getKey());
        return output;
    }
}
