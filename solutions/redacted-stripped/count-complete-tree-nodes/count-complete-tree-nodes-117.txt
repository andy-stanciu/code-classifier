class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        TreeNode left = root.left;
        TreeNode right = root.right;
        int lc = 0, rc = 0;
        while (left != null) {
            left = left.left;
            lc++;
        }
        while (right != null) {
            right = right.right;
            rc++;
        }
        if (lc == rc) {
            return (int) Math.pow(2, lc + 1) - 1;
        }
        return 1 + method1(root.left) + method1(root.right);
    }
}
