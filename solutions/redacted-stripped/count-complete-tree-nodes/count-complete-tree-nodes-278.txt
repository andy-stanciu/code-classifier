class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = method2(root);
        int rh = method3(root);
        if (lh == rh) {
            return ((2 << lh) - 1);
        } else {
            return method1(root.left) + method1(root.right) + 1;
        }
    }

    public int method2(TreeNode root) {
        int height = 0;
        while (root.left != null) {
            height++;
            root = root.left;
        }
        return height;
    }

    public int method3(TreeNode root) {
        int height = 0;
        while (root.right != null) {
            height++;
            root = root.right;
        }
        return height;
    }
}
