class Solution {

    private void method1(TreeNode root, int[] cnt) {
        if (root == null) {
            return;
        }
        cnt[0]++;
        method1(root.left, cnt);
        method1(root.right, cnt);
    }

    public int method2(TreeNode root) {
        int[] cnt = { 0 };
        method1(root, cnt);
        return cnt[0];
    }
}
