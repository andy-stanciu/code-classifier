class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int lh = height(root.left);
        int rh = height(root.right);
        if (rh != lh)
            return (int) Math.pow(2, rh) + method1(root.left);
        else
            return (int) Math.pow(2, lh) + method1(root.right);
    }

    private int height(TreeNode node) {
        if (node == null)
            return 0;
        else
            return height(node.left) + 1;
    }
}
