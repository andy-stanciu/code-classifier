class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        } else if (root.right == null) {
            return method1(root.left) + 1;
        } else if (root.left == null) {
            return method1(root.right) + 1;
        } else {
            return method1(root.left) + method1(root.right) + 1;
        }
    }
}
