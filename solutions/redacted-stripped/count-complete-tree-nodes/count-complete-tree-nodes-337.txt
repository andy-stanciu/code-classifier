class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        return f(root);
    }

    public static int f(TreeNode root) {
        if (root == null)
            return 0;
        int left = f(root.left);
        int right = f(root.right);
        return 1 + left + right;
    }
}
