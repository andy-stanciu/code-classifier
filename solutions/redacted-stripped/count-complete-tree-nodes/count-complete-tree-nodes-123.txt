class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int l = depth(root, true);
        int r = depth(root, false);
        if (l == r) {
            return (int) Math.pow(2, l) - 1;
        }
        return 1 + method1(root.left) + method1(root.right);
    }

    public int depth(TreeNode root, boolean seekLeft) {
        int res = 0;
        while (root != null) {
            res++;
            if (seekLeft) {
                root = root.left;
            } else {
                root = root.right;
            }
        }
        return res;
    }
}
