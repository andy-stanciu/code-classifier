class Solution {

    static int var1 = 0;

    static void Postorder(TreeNode root) {
        if (root == null) {
            return;
        }
        Postorder(root.left);
        Postorder(root.right);
        var1++;
    }

    public int method1(TreeNode root) {
        var1 = 0;
        Postorder(root);
        return var1;
    }
}
