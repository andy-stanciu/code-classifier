class Solution {

    public int method1(TreeNode root) {
        int[] sum = { 0 };
        method2(root, sum);
        return sum[0];
    }

    public void method2(TreeNode root, int[] arr) {
        if (root == null)
            return;
        method2(root.left, arr);
        method2(root.right, arr);
        arr[0] += 1;
    }
}
