class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        var1(root);
        return var1;
    }

    public void var1(TreeNode node) {
        if (node == null)
            return;
        var1(node.left);
        var1++;
        var1(node.right);
    }
}
