class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int l = lHeight(root);
        int r = rHeight(root);
        if (l == r)
            return (1 << l) - 1;
        return 1 + method1(root.left) + method1(root.right);
    }

    int lHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.left;
        }
        return height;
    }

    int rHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.right;
        }
        return height;
    }
}
