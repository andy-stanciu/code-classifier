class Solution {

    int helper(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftC = 0;
        int rightC = 0;
        if (root.left != null) {
            leftC = helper(root.left);
        }
        if (root.right != null) {
            rightC = helper(root.right);
        }
        return 1 + leftC + rightC;
    }

    public int method1(TreeNode root) {
        return helper(root);
    }
}
