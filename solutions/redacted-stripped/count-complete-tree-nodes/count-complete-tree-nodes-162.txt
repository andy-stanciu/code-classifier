class Solution {

    int c = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        c++;
        if (root.right == null && root.left == null) {
            return c;
        }
        method1(root.left);
        method1(root.right);
        return c;
    }
}
