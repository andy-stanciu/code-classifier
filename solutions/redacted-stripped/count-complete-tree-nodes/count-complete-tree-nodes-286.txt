class Solution {

    private int num = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return num;
        if (root.left != null) {
            method1(root.left);
        }
        num += 1;
        if (root.right != null) {
            method1(root.right);
        }
        return num;
    }
}
