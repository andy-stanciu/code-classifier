class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int res = solve(root);
        return res;
    }

    public int solve(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = solve(root.left);
        int right = solve(root.right);
        return left + right + 1;
    }
}
