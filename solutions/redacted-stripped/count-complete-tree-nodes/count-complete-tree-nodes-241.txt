class Solution {

    public static int method1(TreeNode root) {
        return method2(root);
    }

    public static int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = method2(root.left);
        int rh = method2(root.right);
        return lh + rh + 1;
    }
}
