class Solution {

    public int method1(TreeNode root) {
        return method2(root);
    }

    public static int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lc = method2(root.left);
        int lr = method2(root.right);
        return lc + lr + 1;
    }
}
