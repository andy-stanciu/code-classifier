class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = getLeft(root);
        int rh = getRight(root);
        if (lh == rh) {
            return (int) Math.pow(2, lh) - 1;
        }
        return 1 + method1(root.left) + method1(root.right);
    }

    public int getLeft(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.left;
        }
        return height;
    }

    public int getRight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.right;
        }
        return height;
    }
}
