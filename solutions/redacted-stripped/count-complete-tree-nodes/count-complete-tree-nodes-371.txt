class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = findLeftHeight(root.left);
        int right = findRightHeight(root.right);
        if (left == right)
            return (2 << left) - 1;
        else
            return 1 + method1(root.left) + method1(root.right);
    }

    public int findLeftHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.left;
        }
        return height;
    }

    public int findRightHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.right;
        }
        return height;
    }
}
