class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int var1 = 0;
        while (!q.isEmpty()) {
            TreeNode curr = q.poll();
            var1++;
            if (curr.left != null)
                q.add(curr.left);
            if (curr.right != null)
                q.add(curr.right);
        }
        return var1;
    }
}
