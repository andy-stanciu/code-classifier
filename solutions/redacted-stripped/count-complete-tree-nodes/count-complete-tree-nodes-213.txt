class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int leftHeight = method2(root, true);
        int rightHeight = method2(root, false);
        if (leftHeight == rightHeight) {
            return (1 << leftHeight) - 1;
        } else {
            return method1(root.left) + method1(root.right) + 1;
        }
    }

    private int method2(TreeNode node, boolean left) {
        int height = 0;
        while (node != null) {
            height++;
            node = left ? node.left : node.right;
        }
        return height;
    }
}
