class Solution {

    static class Node {

        int data;

        Node left;

        Node right;

        Node(int data) {
            this.data = data;
            this.left = null;
            this.right = null;
        }
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lroot = method1(root.left);
        int rroot = method1(root.right);
        return lroot + rroot + 1;
    }
}
