class Solution {

    public int method1(TreeNode root) {
        int result = 0;
        if (root == null)
            return result;
        if (root.left == null && root.right == null)
            return 1;
        return sum(root);
    }

    public int sum(TreeNode node) {
        if (node == null)
            return 0;
        int sumLeft = sum(node.left);
        int sumRight = sum(node.right);
        return sumLeft + sumRight + 1;
    }
}
