class Solution {

    public int method1(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int leftNode = method1(node.left);
        int rightNode = method1(node.right);
        return leftNode + rightNode + 1;
    }

    public int method2(TreeNode root) {
        return method1(root);
    }
}
