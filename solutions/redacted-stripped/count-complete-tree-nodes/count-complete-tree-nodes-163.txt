class Solution {

    public int nodeTraversal(TreeNode root) {
        if (root == null) {
            return 0;
        } else if (root.left == null && root.right == null) {
            return 1;
        } else if (root.left == null) {
            return 1 + nodeTraversal(root.right);
        } else if (root.right == null) {
            return 1 + nodeTraversal(root.left);
        } else {
            return 1 + nodeTraversal(root.left) + nodeTraversal(root.right);
        }
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + nodeTraversal(root.left) + nodeTraversal(root.right);
    }
}
