class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        else if (root != null && root.left == null && root.right == null) {
            return 1;
        }
        int var1 = 0;
        var1 += method1(root.left);
        var1 = var1 + 1;
        var1 += method1(root.right);
        return var1;
    }
}
