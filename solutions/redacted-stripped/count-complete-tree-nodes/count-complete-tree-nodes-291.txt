class Solution {

    private int sum = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        sum = 0;
        helper(root);
        return sum;
    }

    private void helper(TreeNode root) {
        if (root == null) {
            return;
        }
        sum++;
        helper(root.left);
        helper(root.right);
    }
}
