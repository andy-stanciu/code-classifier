class Solution {

    int res = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int leftH = getHeight(root.left);
        int rightH = getHeight(root.right);
        if (leftH == rightH) {
            res += Math.pow(2, leftH);
            method1(root.right);
        } else {
            res += Math.pow(2, rightH);
            method1(root.left);
        }
        return res;
    }

    int getHeight(TreeNode root) {
        if (root == null)
            return 0;
        int h = 1;
        while (root.left != null) {
            h++;
            root = root.left;
        }
        return h;
    }
}
