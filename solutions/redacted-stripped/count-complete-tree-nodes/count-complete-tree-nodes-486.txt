class Solution {

    static int max = 0;

    public int method1(TreeNode r) {
        max = 0;
        if (r != null) {
            cmax(r);
        }
        return max;
    }

    public void cmax(TreeNode r) {
        if (r == null)
            return;
        max++;
        cmax(r.left);
        cmax(r.right);
    }
}
