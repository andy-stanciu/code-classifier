class Solution {

    public int method1(TreeNode root) {
        int var1 = method2(root);
        return var1;
    }

    public int method2(TreeNode node) {
        if (node == null)
            return 0;
        int l = method2(node.left);
        int e = method2(node.right);
        return l + e + 1;
    }
}
