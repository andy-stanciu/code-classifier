class Solution {

    public int method1(TreeNode root) {
        return trav(root) - 1;
    }

    public static int trav(TreeNode root) {
        if (root == null) {
            return 1;
        }
        int left = trav(root.left);
        int right = trav(root.right);
        return (left + right);
    }
}
