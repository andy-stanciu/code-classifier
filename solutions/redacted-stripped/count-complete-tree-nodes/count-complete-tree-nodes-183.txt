class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        int var2 = 0;
        if (root.left != null || root.right != null) {
            var1 = method1(root.left);
            var2 = method1(root.right);
        }
        return var1 + var2 + 1;
    }
}
