class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            return 1;
        }
        int l = method1(root.left);
        int r = method1(root.right);
        return l + r + 1;
    }
}
