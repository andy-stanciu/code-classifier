class Solution {

    public void method1(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        method1(root.left, list);
        method1(root.right, list);
        list.add(root.val);
    }

    public int method2(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        if (root == null) {
            return list.size();
        }
        method1(root, list);
        return list.size();
    }
}
