class Solution {

    public int findMaxHeightLeft(TreeNode root) {
        if (root == null)
            return 0;
        TreeNode temp1 = root;
        int cntLeft = 0;
        while (temp1 != null) {
            cntLeft++;
            temp1 = temp1.left;
        }
        return cntLeft;
    }

    public int findMaxHeightRight(TreeNode root) {
        if (root == null)
            return 0;
        TreeNode temp2 = root;
        int cntRight = 0;
        while (temp2 != null) {
            cntRight++;
            temp2 = temp2.right;
        }
        return cntRight;
    }

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int lh = findMaxHeightLeft(root.left);
        int rh = findMaxHeightRight(root.right);
        if (lh == rh)
            return (2 << lh) - 1;
        return 1 + method1(root.left) + method1(root.right);
    }
}
