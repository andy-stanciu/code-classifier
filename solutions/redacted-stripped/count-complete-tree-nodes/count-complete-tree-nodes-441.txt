class Solution {

    public int method1(TreeNode root) {
        int[] ans = new int[1];
        inorder(root, ans);
        return ans[0];
    }

    void inorder(TreeNode root, int[] ans) {
        if (root == null)
            return;
        ans[0] += 1;
        inorder(root.left, ans);
        inorder(root.right, ans);
    }
}
