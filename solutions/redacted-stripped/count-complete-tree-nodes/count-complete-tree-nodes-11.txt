class Solution {

    public int method1(TreeNode root) {
        ArrayList<Integer> var1 = new ArrayList<Integer>();
        cnt(root, var1);
        return var1.size();
    }

    public static void cnt(TreeNode root, ArrayList<Integer> var1) {
        if (root == null)
            return;
        var1.add(1);
        cnt(root.left, var1);
        cnt(root.right, var1);
    }
}
