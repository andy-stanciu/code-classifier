class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = method2(root);
        int right = method3(root);
        if (left == right)
            return ((2 << (left)) - 1);
        else
            return 1 + method1(root.left) + method1(root.right);
    }

    int method2(TreeNode root) {
        if (root == null)
            return 0;
        int ans = 0;
        while (root.left != null) {
            root = root.left;
            ans++;
        }
        return ans;
    }

    int method3(TreeNode root) {
        if (root == null)
            return 0;
        int ans = 0;
        while (root.right != null) {
            root = root.right;
            ans++;
        }
        return ans;
    }
}
