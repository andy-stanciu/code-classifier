class Solution {

    public int method1(TreeNode root) {
        int left = leftdepth(root);
        int right = rightdepth(root);
        if (left == right) {
            return (int) Math.pow(2, left) - 1;
        } else {
            return 1 + method1(root.left) + method1(root.right);
        }
    }

    public int leftdepth(TreeNode node) {
        int depth = 0;
        while (node != null) {
            node = node.left;
            depth++;
        }
        return depth;
    }

    public int rightdepth(TreeNode node) {
        int depth = 0;
        while (node != null) {
            node = node.right;
            depth++;
        }
        return depth;
    }
}
