class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int ans = method2(root);
        return ans;
    }

    int method2(TreeNode root) {
        if (root == null)
            return 0;
        int lh = method2(root.left);
        int rh = method2(root.right);
        return lh + rh + 1;
    }
}
