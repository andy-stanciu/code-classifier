class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int leftH = 1;
        TreeNode left = root.left;
        while (left != null) {
            left = left.left;
            leftH++;
        }
        int rightH = 1;
        TreeNode right = root.right;
        while (right != null) {
            right = right.right;
            rightH++;
        }
        if (leftH == rightH)
            return (int) (Math.pow(2, leftH) - 1);
        return 1 + method1(root.left) + method1(root.right);
    }
}
