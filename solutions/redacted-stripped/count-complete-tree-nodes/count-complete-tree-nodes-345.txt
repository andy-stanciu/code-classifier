class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null)
            return 1;
        if (root.left == null)
            return 1 + method1(root.right);
        if (root.right == null)
            return 1 + method1(root.left);
        return 1 + method1(root.left) + method1(root.right);
    }
}
