class Solution {

    public int method1(TreeNode root) {
        return method1(root, 1);
    }

    public int method1(TreeNode root, int n) {
        if (root == null)
            return n - 1;
        int max = method1(root.left, ++n);
        max = method1(root.right, ++max);
        return max;
    }
}
