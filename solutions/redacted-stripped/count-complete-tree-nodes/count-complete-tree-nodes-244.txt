class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        traversal(root);
        return var1;
    }

    public void traversal(TreeNode root) {
        if (root == null)
            return;
        else {
            var1 += 1;
            traversal(root.left);
            traversal(root.right);
        }
    }
}
