class Solution {

    public int method1(TreeNode root) {
        return helper(root, 0);
    }

    private int helper(TreeNode current, int param1) {
        if (current == null) {
            return param1;
        }
        param1++;
        if (current.left != null) {
            param1 = helper(current.left, param1);
        }
        if (current.right != null) {
            param1 = helper(current.right, param1);
        }
        return param1;
    }
}
