class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + cnt(root);
    }

    public int cnt(TreeNode root) {
        if (root == null) {
            return -1;
        }
        int a = 1 + cnt(root.left);
        int b = 1 + cnt(root.right);
        return a + b;
    }
}
