class Solution {

    public int findl(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.left;
        }
        System.out.println(height);
        return height;
    }

    public int findR(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.right;
        }
        return height;
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = findl(root);
        int rh = findR(root);
        if (lh == rh) {
            return (1 << lh) - 1;
        }
        return 1 + method1(root.left) + method1(root.right);
    }
}
