class Solution {

    public int method1(TreeNode root) {
        int left = checkLeftHeight(root);
        int right = checkRightHeight(root);
        if (left == right) {
            return (int) Math.pow(2, left) - 1;
        }
        int left_ = method1(root.left);
        int right_ = method1(root.right);
        return left_ + right_ + 1;
    }

    static int checkLeftHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.left;
        }
        return height;
    }

    static int checkRightHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            height++;
            root = root.right;
        }
        return height;
    }
}
