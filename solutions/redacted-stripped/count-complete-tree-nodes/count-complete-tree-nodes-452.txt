class Solution {

    public int method1(TreeNode root) {
        return noofnode(root, 0);
    }

    int c = 0;

    public int noofnode(TreeNode root, int param1) {
        if (root == null)
            return 0;
        noofnode(root.left, param1);
        noofnode(root.right, param1);
        c++;
        return c;
    }
}
