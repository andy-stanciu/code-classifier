class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return var1;
        }
        method2(root);
        return var1;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        var1++;
        method2(root.left);
        method2(root.right);
        return;
    }
}
