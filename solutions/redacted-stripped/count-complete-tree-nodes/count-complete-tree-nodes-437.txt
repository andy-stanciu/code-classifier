class Solution {

    private int findLh(TreeNode root) {
        int h = 0;
        while (root != null) {
            h++;
            root = root.left;
        }
        return h;
    }

    private int findRh(TreeNode root) {
        int h = 0;
        while (root != null) {
            h++;
            root = root.right;
        }
        return h;
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = findLh(root);
        int rh = findRh(root);
        if (lh == rh)
            return (1 << lh) - 1;
        int left = method1(root.left);
        int right = method1(root.right);
        return 1 + left + right;
    }
}
