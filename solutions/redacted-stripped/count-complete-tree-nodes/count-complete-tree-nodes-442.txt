class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int hLeft = height(root.left);
        int hRight = height(root.right);
        if (hLeft == hRight) {
            return (1 << hLeft) + method1(root.right);
        } else {
            return method1(root.left) + method1(root.right) + 1;
        }
    }

    private int height(TreeNode node) {
        int h = 0;
        while (node != null) {
            h++;
            node = node.left;
        }
        return h;
    }
}
