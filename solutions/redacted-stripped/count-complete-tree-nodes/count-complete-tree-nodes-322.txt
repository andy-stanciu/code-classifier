class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lh = flh(root.left);
        int rh = frh(root.right);
        if (lh == rh) {
            return (int) Math.pow(2, lh + 1) - 1;
        }
        return method1(root.left) + method1(root.right) + 1;
    }

    public int flh(TreeNode node) {
        int ans = 0;
        TreeNode temp = node;
        while (temp != null) {
            ans++;
            temp = temp.left;
        }
        return ans;
    }

    public int frh(TreeNode node) {
        int ans = 0;
        TreeNode temp = node;
        while (temp != null) {
            ans++;
            temp = temp.right;
        }
        return ans;
    }
}
