class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        PreOrder(root, var1);
        return var1[0];
    }

    public static void PreOrder(TreeNode root, int[] var1) {
        if (root == null) {
            return;
        }
        var1[0] += 1;
        PreOrder(root.left, var1);
        PreOrder(root.right, var1);
    }
}
