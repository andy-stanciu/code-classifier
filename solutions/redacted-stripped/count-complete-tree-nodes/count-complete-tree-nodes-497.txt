class Solution {

    int method1(TreeNode root, int c) {
        if (root == null) {
            return c;
        } else {
            c++;
        }
        c = method1(root.left, c);
        c = method1(root.right, c);
        return c;
    }

    public int method2(TreeNode root) {
        int ans = method1(root, 0);
        return ans;
    }
}
