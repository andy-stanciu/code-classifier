class Solution {

    public int method1(TreeNode root) {
        return method2(root);
    }

    private int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftHeight = getLeftHeight(root.left);
        int rightHeight = getRightHeight(root.right);
        if (leftHeight == rightHeight) {
            int var1 = (2 << leftHeight) - 1;
            return var1;
        }
        int var2 = method2(root.left);
        int var3 = method2(root.right);
        return 1 + var2 + var3;
    }

    private int getLeftHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            root = root.left;
            height++;
        }
        return height;
    }

    private int getRightHeight(TreeNode root) {
        int height = 0;
        while (root != null) {
            root = root.right;
            height++;
        }
        return height;
    }
}
