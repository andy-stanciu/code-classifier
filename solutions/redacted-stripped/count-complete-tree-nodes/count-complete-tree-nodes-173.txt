class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lHeight = 0, rHeight = 0;
        TreeNode left = root, right = root;
        while (left != null) {
            lHeight++;
            left = left.left;
        }
        while (right != null) {
            rHeight++;
            right = right.right;
        }
        if (lHeight == rHeight) {
            return (int) Math.pow(2, lHeight) - 1;
        }
        return 1 + method1(root.left) + method1(root.right);
    }
}
