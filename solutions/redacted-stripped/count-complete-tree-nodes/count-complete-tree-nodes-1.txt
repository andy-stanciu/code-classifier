class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return var1;
        var1(root);
        return var1;
    }

    void var1(TreeNode root) {
        if (root == null)
            return;
        var1++;
        var1(root.left);
        var1(root.right);
    }
}
