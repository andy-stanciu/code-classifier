class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return Number(root);
    }

    public int Number(TreeNode p) {
        if (p == null) {
            return 0;
        }
        var1++;
        Number(p.left);
        Number(p.right);
        return var1;
    }
}
