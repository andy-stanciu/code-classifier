class Solution {

    List<Integer> res = new ArrayList<>();

    public int method1(TreeNode root) {
        inorderTraversal(root);
        return res.size();
    }

    private void inorderTraversal(TreeNode root) {
        if (root == null) {
            return;
        }
        inorderTraversal(root.left);
        res.add(root.val);
        inorderTraversal(root.right);
    }
}
