class Solution {

    ArrayList<Integer> ls = new ArrayList<>();

    public int method1(TreeNode root) {
        method2(root);
        return ls.size();
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        ls.add(root.val);
        method2(root.left);
        method2(root.right);
    }
}
