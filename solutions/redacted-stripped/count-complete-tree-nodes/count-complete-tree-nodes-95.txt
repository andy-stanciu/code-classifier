class Solution {

    public static int lefthight(TreeNode root) {
        int var1 = 0;
        while (root.left != null) {
            var1++;
            root = root.left;
        }
        return var1;
    }

    public static int righthight(TreeNode root) {
        int var1 = 0;
        while (root.right != null) {
            var1++;
            root = root.right;
        }
        return var1;
    }

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = lefthight(root);
        int right = righthight(root);
        if (left == right) {
            return ((2 << (left)) - 1);
        } else {
            return method1(root.left) + method1(root.right) + 1;
        }
    }
}
