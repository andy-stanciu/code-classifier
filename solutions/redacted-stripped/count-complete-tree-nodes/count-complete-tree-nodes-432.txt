class Solution {

    public int method1(TreeNode root) {
        ArrayList<TreeNode> list = new ArrayList<>();
        list = preOrder(root);
        return list.size();
    }

    public ArrayList<TreeNode> preOrder(TreeNode root) {
        ArrayList<TreeNode> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        list.add(root);
        list.addAll(preOrder(root.left));
        list.addAll(preOrder(root.right));
        return list;
    }
}
