class Solution {

    int c = 0;

    public int method1(TreeNode root) {
        method2(root);
        return c;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        c = c + 1;
        method2(root.left);
        method2(root.right);
    }
}
