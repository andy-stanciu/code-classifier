class Solution {

    ArrayList<Integer> list;

    public int method1(TreeNode root) {
        this.list = new ArrayList<>();
        DFS(root);
        return list.size();
    }

    public void DFS(TreeNode root) {
        if (root != null) {
            list.add(root.val);
            if (root.left != null)
                DFS(root.left);
            if (root.right != null)
                DFS(root.right);
        }
    }
}
