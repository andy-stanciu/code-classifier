class Solution {

    Set<String> cache;

    public boolean method1(String s, List<String> param1) {
        cache = new HashSet<>();
        cache.addAll(param1);
        Boolean[][] dp = new Boolean[s.length()][s.length()];
        return method1(s, 0, 0, dp);
    }

    private boolean method1(String s, int i, int j, Boolean[][] dp) {
        if (i == s.length() && j == s.length()) {
            return true;
        }
        if (i >= s.length() || j >= s.length())
            return false;
        if (dp[i][j] != null) {
            return dp[i][j];
        }
        boolean not = method1(s, i, j + 1, dp);
        boolean has = false;
        if (cache.contains(s.substring(i, j + 1))) {
            has = method1(s, j + 1, j + 1, dp);
        }
        dp[i][j] = not || has;
        return dp[i][j];
    }
}
