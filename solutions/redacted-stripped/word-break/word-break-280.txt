class Solution {

    private int method1(String s, List<String> param1, int ind, int[] dp) {
        if (ind >= s.length())
            return 1;
        if (dp[ind] != -1)
            return dp[ind];
        int ans = 0;
        for (int i = ind; i < s.length(); i++) {
            String chk = s.substring(ind, i + 1);
            if (param1.contains(chk) && method1(s, param1, i + 1, dp) == 1) {
                ans = 1;
            }
        }
        return dp[ind] = ans;
    }

    public boolean method2(String s, List<String> param1) {
        int[] dp = new int[s.length()];
        Arrays.fill(dp, -1);
        if (method1(s, param1, 0, dp) == 1)
            return true;
        return false;
    }
}
