class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> var1 = new HashSet<>(param1);
        Map<String, Boolean> memo = new HashMap<>();
        return method2(s, var1, memo);
    }

    private boolean method2(String s, Set<String> var1, Map<String, Boolean> memo) {
        if (s.isEmpty()) {
            return true;
        }
        if (memo.containsKey(s)) {
            return memo.get(s);
        }
        for (String var3 : var1) {
            if (s.startsWith(var3)) {
                boolean var2 = method2(s.substring(var3.length()), var1, memo);
                if (var2) {
                    memo.put(s, true);
                    return true;
                }
            }
        }
        memo.put(s, false);
        return false;
    }
}
