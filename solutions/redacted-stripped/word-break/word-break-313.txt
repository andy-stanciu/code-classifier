class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> dict = new HashSet<>();
        for (String var1 : param1) dict.add(var1);
        int n = s.length();
        int[][] dp = new int[n][n];
        for (int[] row : dp) Arrays.fill(row, -1);
        return check(s, 0, s.length() - 1, dict, dp);
    }

    public boolean isInDict(String s, Set<String> param2) {
        return param2.contains(s);
    }

    public boolean check(String s, int start, int end, Set<String> param2, int[][] dp) {
        if (start > end)
            return false;
        if (isInDict(s.substring(start, end + 1), param2))
            return true;
        if (dp[start][end] != -1)
            return dp[start][end] == 1 ? true : false;
        for (int i = start; i < end; i++) {
            boolean left = check(s, start, i, param2, dp);
            boolean right = check(s, i + 1, end, param2, dp);
            if (left == true && right == true) {
                dp[start][end] = 1;
                return true;
            }
        }
        dp[start][end] = 0;
        return false;
    }
}
