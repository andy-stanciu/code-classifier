class Solution {

    Set<String> set = null;

    public boolean method1(String s, List<String> param1) {
        set = new HashSet(param1);
        return recur(s, s.length(), 0);
    }

    public boolean recur(String s, int n, int ind) {
        if (ind == n)
            return true;
        int i = ind;
        String sub = "";
        while (i < n) {
            sub += s.charAt(i);
            if (set.contains(sub)) {
                if (recur(s, n, i + 1))
                    return true;
            }
            i++;
        }
        return false;
    }
}
