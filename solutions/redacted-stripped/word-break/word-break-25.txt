public class Solution {

    public boolean method1(String s, List<String> dic) {
        int n = s.length();
        int[] memo = new int[n + 1];
        memo[n] = 1;
        for (int i = n - 1; i >= 0; i--) {
            boolean ans = false;
            for (String var1 : dic) {
                if (i + var1.length() <= n && s.substring(i, i + var1.length()).equals(var1)) {
                    ans = ans || (memo[i + var1.length()] == 1);
                }
            }
            memo[i] = ans ? 1 : 0;
        }
        return memo[0] == 1;
    }
}
