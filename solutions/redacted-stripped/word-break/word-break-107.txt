class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean[] dp = new boolean[s.length() + 1];
        dp[s.length()] = true;
        int start = s.length() - 1;
        while (start >= 0) {
            boolean result = false;
            for (int index = 0; index < param1.size(); ++index) {
                String currentString = param1.get(index);
                if ((s.length() - start) >= currentString.length() && s.substring(start, start + currentString.length()).equals(currentString)) {
                    result = dp[start + currentString.length()];
                }
                if (result) {
                    break;
                }
            }
            dp[start] = result;
            --start;
        }
        return dp[0];
    }

    }
}
