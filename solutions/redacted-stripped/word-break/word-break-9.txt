class Solution {

    private Map<String, Boolean> memo = new HashMap<>();

    public boolean helper(String s, Set<String> param1) {
        if (memo.containsKey(s)) {
            return memo.get(s);
        }
        if (s.length() == 0)
            return true;
        for (int j = 1; j <= s.length(); j++) {
            String substring = s.substring(0, j);
            if (param1.contains(substring)) {
                memo.put(substring, true);
                String rSubstring = s.substring(j);
                if (helper(rSubstring, param1)) {
                    return true;
                }
            }
        }
        memo.put(s, false);
        return false;
    }

    public boolean method1(String s, List<String> param2) {
        Set<String> param1 = new HashSet<>(param2);
        return helper(s, param1);
    }
}
