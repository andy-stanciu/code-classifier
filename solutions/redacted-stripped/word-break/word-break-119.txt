class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean[] dp = new boolean[s.length()];
        for (int i = s.length() - 1; i >= 0; i--) {
            for (final var var1 : param1) {
                if (s.startsWith(var1, i)) {
                    dp[i] = i + var1.length() == s.length() ? true : dp[i + var1.length()];
                    if (dp[i])
                        break;
                }
            }
        }
        return dp[0];
    }
}
