class Solution {

    String s;

    List<String> var1;

    int[] memo;

    public boolean method1(String s, List<String> var1) {
        this.s = s;
        this.var1 = var1;
        memo = new int[s.length()];
        Arrays.fill(memo, -1);
        return dp(0);
    }

    public boolean dp(int index) {
        if (index == s.length()) {
            return true;
        }
        if (memo[index] != -1) {
            return memo[index] == 1 ? true : false;
        }
        boolean res = false;
        for (String var2 : var1) {
            if (index + var2.length() <= s.length()) {
                String sub = s.substring(index, index + var2.length());
                if (sub.equals(var2)) {
                    if (dp(index + sub.length())) {
                        res = true;
                    }
                }
            }
        }
        if (res == true) {
            memo[index] = 1;
        } else {
            memo[index] = 0;
        }
        return res;
    }
}
