class Solution {

    public int method1(int i, int[] dp, Set<String> dict, String s) {
        if (i == s.length()) {
            return 1;
        }
        if (dp[i] != -1) {
            return dp[i];
        }
        for (int j = i; j < s.length(); j++) {
            if (dict.contains(s.substring(i, j + 1))) {
                int subVal = method1(j + 1, dp, dict, s);
                if (subVal == 1) {
                    dp[i] = 1;
                    return dp[i];
                }
            }
        }
        dp[i] = 0;
        return dp[i];
    }

    public boolean method1(String s, List<String> param1) {
        Set<String> dict = new HashSet<>(param1);
        int[] dp = new int[s.length()];
        Arrays.fill(dp, -1);
        return method1(0, dp, dict, s) == 1;
    }
}
