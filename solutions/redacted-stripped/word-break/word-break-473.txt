class Solution {

    public Boolean[] dp;

    public boolean method1(String s, List<String> param1) {
        dp = new Boolean[s.length() + 1];
        return solve(0, s, param1);
    }

    private boolean solve(int start, String s, List<String> param1) {
        if (start >= s.length()) {
            return true;
        }
        if (dp[start] != null) {
            return dp[start];
        }
        for (int end = start + 1; end <= s.length(); end++) {
            if (param1.contains(s.substring(start, end)) && solve(end, s, param1)) {
                return dp[start] = true;
            }
        }
        return dp[start] = false;
    }
}
