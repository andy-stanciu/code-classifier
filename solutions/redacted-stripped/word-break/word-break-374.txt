class Solution {

    public boolean method1(String s, List<String> param1) {
        return helper(s, new HashSet<>(param1), new HashMap<>());
    }

    public boolean helper(String s, HashSet<String> param2, HashMap<String, Boolean> map) {
        if (s.isEmpty())
            return true;
        if (map.containsKey(s))
            return map.get(s);
        for (String w : param2) {
            if (s.startsWith(w) && helper(s.substring(w.length()), param2, map)) {
                map.put(s, true);
                return true;
            }
        }
        map.put(s, false);
        return false;
    }
}
