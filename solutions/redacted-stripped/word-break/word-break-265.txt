class Solution {

    public boolean method1(String s, List<String> param1) {
        Map<String, Boolean> dp = new HashMap<>();
        return method2(s, param1, dp);
    }

    public static boolean method2(String s, List<String> param1, Map<String, Boolean> dp) {
        if (dp.containsKey(s)) {
            return dp.get(s);
        }
        if (s.isEmpty()) {
            return true;
        }
        for (String str : param1) {
            if (s.indexOf(str) == 0) {
                String suffix = s.substring(str.length());
                if (method2(suffix, param1, dp)) {
                    dp.put(s, true);
                    return true;
                }
            }
        }
        dp.put(s, false);
        return false;
    }
}
