class Solution {

    private boolean dfs(String s, int start, List<String> param1, Boolean[] memo) {
        if (start == s.length()) {
            return true;
        }
        if (memo[start] != null) {
            return memo[start];
        }
        for (String var1 : param1) {
            if (s.startsWith(var1, start)) {
                if (dfs(s, start + var1.length(), param1, memo)) {
                    memo[start] = true;
                    return true;
                }
            }
        }
        return memo[start] = false;
    }

    public boolean method1(String s, List<String> param1) {
        Boolean[] memo = new Boolean[s.length()];
        return dfs(s, 0, param1, memo);
    }
}
