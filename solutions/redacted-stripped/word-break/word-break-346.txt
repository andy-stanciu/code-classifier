class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        HashSet<String> set = new HashSet<>(param1);
        int[] dp = new int[n + 1];
        dp[0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                if (set.contains(s.substring(j, i)) && dp[j] == 1) {
                    dp[i] = 1;
                }
            }
        }
        return dp[n] == 1 ? true : false;
    }
}
