class Solution {

    public boolean method1(String s, List<String> param1) {
        Boolean[] dp = new Boolean[s.length() + 1];
        return backtrack(s, param1, dp, 0);
    }

    private boolean backtrack(String s, List<String> param1, Boolean[] dp, int index) {
        if (index == s.length())
            return true;
        if (dp[index] != null)
            return dp[index];
        for (int i = 0; i < param1.size(); i++) {
            String var1 = param1.get(i);
            if (s.startsWith(var1, index))
                if (backtrack(s, param1, dp, index + var1.length()))
                    return true;
        }
        return dp[index] = false;
    }
}
