class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean[] arr = new boolean[s.length() + 1];
        arr[0] = true;
        for (int i = 1; i <= s.length(); i++) {
            for (int j = 0; j < i; j++) {
                if (param1.contains(s.substring(j, i)) && arr[j] == true)
                    arr[i] = true;
            }
        }
        return arr[s.length()];
    }
}
