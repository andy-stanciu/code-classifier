class Solution {

    boolean check = false;

    HashMap<String, Boolean> dp = new HashMap<>();

    boolean solve(String s, List<String> param1) {
        if (s.length() == 0)
            return true;
        if (dp.containsKey(s))
            return dp.get(s);
        for (String var1 : param1) {
            if (s.startsWith(var1)) {
                check = check || solve(s.substring(var1.length()), param1);
            }
        }
        dp.put(s, check);
        return dp.get(s);
    }

    public boolean method1(String s, List<String> param1) {
        return solve(s, param1);
    }
}
