class Solution {

    public boolean method1(String s, List<String> param1) {
        return method2(s, param1, new HashMap<>());
    }

    private boolean method2(String s, List<String> param1, Map<String, Boolean> memo) {
        if (s.isEmpty())
            return true;
        if (memo.containsKey(s))
            return memo.get(s);
        for (String var1 : param1) {
            if (s.startsWith(var1) && method2(s.substring(var1.length()), param1, memo)) {
                memo.put(s, true);
                return true;
            }
        }
        memo.put(s, false);
        return false;
    }
}
