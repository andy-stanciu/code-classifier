class Solution {

    public boolean method1(String s, List<String> param1) {
        return method1(s, new HashSet<>(param1), 0, new Boolean[s.length()]);
    }

    private boolean method1(String s, Set<String> param1, int start, Boolean[] memo) {
        if (start == s.length()) {
            return true;
        }
        if (null != memo[start]) {
            return memo[start];
        }
        for (int end = start + 1; end <= s.length(); ++end) {
            if (param1.contains(s.substring(start, end)) && method1(s, param1, end, memo)) {
                memo[start] = true;
                return true;
            }
        }
        memo[start] = false;
        return false;
    }
}
