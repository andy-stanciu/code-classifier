class Solution {

    HashMap<String, Boolean> map = new HashMap<>();

    public boolean method1(String s, List ls) {
        if (ls.contains(s))
            return true;
        if (map.containsKey(s))
            return map.get(s);
        for (int i = 1; i < s.length(); i++) {
            String left = s.substring(0, i);
            if (ls.contains(left) && method1(s.substring(i), ls)) {
                map.put(s, true);
                return true;
            }
        }
        map.put(s, false);
        return false;
    }
}
