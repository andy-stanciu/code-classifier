class Solution {

    List<String> var1 = null;

    String s = null;

    Map<Integer, Boolean> memo = new HashMap();

    public boolean method1(String s, List<String> var1) {
        this.var1 = var1;
        this.s = s;
        return method1(0);
    }

    public boolean method1(int i) {
        boolean res = false;
        if (memo.containsKey(i)) {
            return memo.get(i);
        }
        if (i == s.length()) {
            return true;
        }
        for (String var2 : var1) {
            if (s.indexOf(var2, i) == i) {
                res = res || method1(i + var2.length());
            }
        }
        memo.put(i, res);
        return res;
    }
}
