class Solution {

    public boolean method1(String s, List<String> param1) {
        return helper(s.toCharArray(), 0, param1, new HashSet<>());
    }

    public boolean helper(char[] arr, int index, List<String> param1, HashSet<Integer> set) {
        if (index == arr.length)
            return true;
        if (set.contains(index))
            return false;
        set.add(index);
        for (String item : param1) {
            int i = 0;
            while (i < item.length() && index + i < arr.length && item.charAt(i) == arr[index + i]) {
                i++;
            }
            if (i == item.length() && helper(arr, index + item.length(), param1, set))
                return true;
        }
        return false;
    }
}
