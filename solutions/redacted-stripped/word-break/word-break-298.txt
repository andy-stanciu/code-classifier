class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> strSet = new HashSet<>(param1);
        int n = s.length();
        boolean[] isPossible = new boolean[n + 1];
        isPossible[0] = true;
        for (int i = 1; i <= n; i++) {
            for (int j = i - 1; j >= 0; j--) {
                if (isPossible[j] && strSet.contains(s.substring(j, i))) {
                    isPossible[i] = true;
                    break;
                }
            }
        }
        return isPossible[n];
    }
}
