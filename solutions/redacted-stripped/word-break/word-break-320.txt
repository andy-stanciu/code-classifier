class Solution {

    public boolean method1(String s, List<String> param1) {
        int size = s.length();
        List<Integer> trueIndexes = new ArrayList<Integer>();
        HashSet<String> var1 = new HashSet<>();
        int[] dp = new int[size + 1];
        trueIndexes.add(0);
        dp[0] = 0;
        for (String str : param1) {
            var1.add(str);
        }
        for (int i = 1; i <= size; i++) {
            for (int k : trueIndexes) {
                if (var1.contains(s.substring(k, i))) {
                    dp[i] = 1;
                    trueIndexes.add(i);
                    break;
                }
            }
        }
        return dp[size] == 1 ? true : false;
    }
}
