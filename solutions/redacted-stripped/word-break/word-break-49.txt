class Solution {

    public boolean method1(String str, List<String> dict) {
        int size = str.length();
        if (size == 0)
            return true;
        boolean[] dp = new boolean[size + 1];
        for (int i = 1; i <= size; i++) {
            if (dp[i] == false && dict.contains(str.substring(0, i)))
                dp[i] = true;
            if (dp[i] == true) {
                if (i == size)
                    return true;
                for (int j = i + 1; j <= size; j++) {
                    if (dp[j] == false && dict.contains(str.substring(i, j)))
                        dp[j] = true;
                    if (j == size && dp[j] == true)
                        return true;
                }
            }
        }
        return false;
    }
}
