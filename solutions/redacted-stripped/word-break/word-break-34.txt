class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean[] table = new boolean[s.length()];
        for (int i = 0; i < table.length; i++) {
            for (int j = 0; j <= i; j++) {
                if (j == 0 || table[j - 1]) {
                    if (param1.contains(s.substring(j, i + 1))) {
                        table[i] = true;
                        break;
                    }
                }
            }
        }
        return table[table.length - 1];
    }
}
