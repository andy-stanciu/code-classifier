class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<Integer> a = new HashSet<>();
        a.add(0);
        for (int i = 0; i < s.length(); i++) {
            for (Integer n : a) {
                String s1 = s.substring(n, i + 1);
                if (param1.contains(s1)) {
                    a.add(i + 1);
                    break;
                }
            }
        }
        return a.contains(s.length());
    }
}
