class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> dicts = new HashSet<>(param1);
        return method2(s, dicts);
    }

    private boolean method2(String s, Set<String> dicts) {
        for (int i = 1; i <= s.length(); i++) {
            String var1 = s.substring(0, i);
            if (dicts.contains(var1) && method2(s.substring(i, s.length()), dicts)) {
                return true;
            }
        }
        return false;
    }
}
