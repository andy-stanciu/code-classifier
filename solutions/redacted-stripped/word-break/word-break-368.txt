class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        int maxLen = 0;
        for (String var1 : param1) {
            maxLen = Math.max(maxLen, var1.length());
        }
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        Set<String> var2 = new HashSet<>(param1);
        for (int i = 1; i <= n; i++) {
            for (int j = i - 1; j >= 0 && i - j <= maxLen; j--) {
                if (dp[j] && var2.contains(s.substring(j, i))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[n];
    }
}
