class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean[] seen = new boolean[s.length() + 1];
        seen[0] = true;
        for (int i = 1; i <= s.length(); i++) {
            for (int j = 0; j < i; j++) {
                if (seen[j] && param1.contains(s.substring(j, i))) {
                    seen[i] = true;
                }
            }
        }
        return seen[s.length()];
    }
}
