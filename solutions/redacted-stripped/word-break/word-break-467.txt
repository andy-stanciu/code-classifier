class Solution {

    String s;

    List<String> dict;

    int[] dp;

    public boolean method1(String s, List<String> param1) {
        this.s = s;
        this.dict = param1;
        dp = new int[s.length()];
        Arrays.fill(dp, -1);
        return dfs(0) == 1;
    }

    public int dfs(int i) {
        if (i == s.length())
            return 1;
        if (dp[i] != -1)
            return dp[i];
        int p = 0;
        for (String var1 : dict) {
            int matchedChars = equalCharsCount(i, var1);
            if (matchedChars == var1.length())
                p = dfs(i + var1.length());
            if (p == 1)
                break;
        }
        return dp[i] = p;
    }

    public int equalCharsCount(int i, String var1) {
        int j = 0;
        for (j; i < s.length() && j < var1.length(); j++, i++) {
            if (var1.charAt(j) != s.charAt(i))
                break;
        }
        return j;
    }
}
