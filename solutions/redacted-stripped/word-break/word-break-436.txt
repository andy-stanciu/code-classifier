class Solution {

    public boolean method1(String s, List<String> param1, int start, Map<Integer, Boolean> map) {
        if (start == s.length()) {
            return true;
        }
        if (map.containsKey(start)) {
            return map.get(start);
        }
        for (String var1 : param1) {
            int len = var1.length();
            if (start + len <= s.length() && s.substring(start, start + len).equals(var1)) {
                if (method1(s, param1, start + len, map)) {
                    map.put(start, true);
                    return true;
                }
            }
        }
        map.put(start, false);
        return false;
    }

    public boolean method2(String s, List<String> param1) {
        Map<Integer, Boolean> map = new HashMap<>();
        return method1(s, param1, 0, map);
    }
}
