class Solution {

    public boolean method1(String s, List<String> param1) {
        int slen = s.length();
        boolean[] dp = new boolean[slen + 1];
        dp[0] = true;
        for (int i = 1; i <= slen; i++) {
            for (int j = 0; j < i; j++) {
                if (dp[j] && param1.contains(s.substring(j, i))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[slen];
    }
}
