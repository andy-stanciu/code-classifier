class Solution {

    String s;

    List<String> dict;

    Boolean[] memo;

    HashSet<String> set;

    int n = 0;

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        this.s = s;
        this.dict = param1;
        this.memo = new Boolean[n];
        this.set = new HashSet<>(param1);
        this.n = s.length();
        return dp(0);
    }

    public boolean dp(int start) {
        if (start == n)
            return true;
        if (memo[start] != null)
            return memo[start];
        for (int end = start + 1; end <= n; end++) {
            if (set.contains(s.substring(start, end)) && dp(end)) {
                return memo[start] = true;
            }
        }
        return memo[start] = false;
    }
}
