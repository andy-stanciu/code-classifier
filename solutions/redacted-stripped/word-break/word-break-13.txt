class Solution {

    public boolean method1(String s, List<String> param1) {
        return helper(s, 0, new HashSet<>(param1));
    }

    public boolean helper(String s, int startIdx, Set<String> param1) {
        if (startIdx == s.length())
            return true;
        for (String var1 : param1) {
            int end = startIdx + var1.length();
            if (end <= s.length() && s.substring(startIdx, end).equals(var1) && helper(s, end, param1))
                return true;
        }
        return false;
    }
}
