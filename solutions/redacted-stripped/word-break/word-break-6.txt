class Solution {

    public boolean method1(String s, List<String> param1) {
        return helper(s, s.length(), param1, new int[s.length() + 1]);
    }

    public boolean helper(String s, int n, List<String> param1, int[] dp) {
        if (s.isEmpty())
            return true;
        if (dp[n - s.length()] != 0)
            return dp[n - s.length()] == 1;
        for (String var1 : param1) {
            int l = var1.length();
            if (s.length() >= l) {
                if (s.startsWith(var1)) {
                    boolean b = helper(s.substring(l), n, param1, dp);
                    if (b) {
                        dp[n - s.length()] = 1;
                        return b;
                    }
                }
            }
        }
        dp[n - s.length()] = -1;
        return false;
    }
}
