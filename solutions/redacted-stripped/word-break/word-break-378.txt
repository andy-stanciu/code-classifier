class Solution {

    public boolean method1(String s, List<String> param1) {
        return method1(s, new HashSet<>(param1), new HashMap<>());
    }

    private boolean method1(final String s, Set<String> param2, Map<String, Boolean> memo) {
        if (memo.containsKey(s))
            return memo.get(s);
        if (param2.contains(s)) {
            memo.put(s, true);
            return true;
        }
        for (int i = 1; i < s.length(); ++i) {
            final String prefix = s.substring(0, i);
            final String suffix = s.substring(i);
            if (param2.contains(prefix) && method1(suffix, param2, memo)) {
                memo.put(s, true);
                return true;
            }
        }
        memo.put(s, false);
        return false;
    }
}
