class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        for (int i = 0; i <= n; i++) {
            for (String var1 : param1) {
                int start = i - var1.length();
                if (start >= 0 && s.substring(start, i).equals(var1) && dp[start]) {
                    dp[i] = true;
                }
            }
        }
        return dp[n];
    }
}
