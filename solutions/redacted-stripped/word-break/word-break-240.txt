class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean ans = canConstruct(s, param1, new HashMap<>());
        return ans;
    }

    public boolean canConstruct(String target, List<String> param2, HashMap<String, Boolean> memo) {
        if (memo.containsKey(target)) {
            return memo.get(target);
        }
        if (target.equals("")) {
            return true;
        }
        for (String var1 : param2) {
            if (target.startsWith(var1)) {
                String suffix = target.substring(var1.length());
                if (canConstruct(suffix, param2, memo)) {
                    memo.put(suffix, true);
                    return true;
                }
            }
        }
        memo.put(target, false);
        return false;
    }
}
