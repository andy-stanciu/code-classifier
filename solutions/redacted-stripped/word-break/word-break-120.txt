class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> var1 = new HashSet<>(param1);
        int[] dp = new int[s.length()];
        Arrays.fill(dp, -1);
        return operations(0, s, var1, dp);
    }

    public static boolean operations(int startIndex, String s, Set<String> var1, int[] dp) {
        if (startIndex == s.length())
            return true;
        if (dp[startIndex] != -1)
            return dp[startIndex] == 1 ? true : false;
        boolean result = false;
        for (int i = startIndex; i < s.length(); i++) {
            if (var1.contains(s.substring(startIndex, i + 1))) {
                result = result || (operations(i + 1, s, var1, dp));
            }
            if (result == true) {
                dp[startIndex] = 1;
                return true;
            }
        }
        dp[startIndex] = 0;
        return false;
    }
}
