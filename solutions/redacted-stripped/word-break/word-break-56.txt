class Solution {

    public boolean method1(String s, List<String> param1) {
        boolean[] var1 = new boolean[s.length() + 1];
        var1[0] = true;
        for (int i = 0; i <= s.length(); i++) {
            for (int j = 0; j < i; j++) {
                if (var1[j] && param1.contains(s.substring(j, i))) {
                    var1[i] = true;
                    break;
                }
            }
        }
        return var1[s.length()];
    }
}
