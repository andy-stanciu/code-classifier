class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        boolean[] dp = new boolean[n];
        for (int i = n - 1; i >= 0; i--) {
            for (String str : param1) {
                if (s.startsWith(str, i)) {
                    if (i + str.length() >= n) {
                        dp[i] = true;
                        break;
                    }
                    if (i + str.length() < n && dp[i + str.length()]) {
                        dp[i] = true;
                        break;
                    }
                }
            }
        }
        return dp[0];
    }
}
