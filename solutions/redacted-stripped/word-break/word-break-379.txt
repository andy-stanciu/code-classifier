class Solution {

    static boolean method1(String s, List<String> param1) {
        return method2(s, 0, new HashSet<>(param1), new Boolean[s.length()]);
    }

    static boolean method2(String s, int i, Set<String> Dict, Boolean[] record) {
        if (i == s.length()) {
            return true;
        }
        if (record[i] != null) {
            return record[i];
        }
        for (int j = i; j < s.length(); j++) {
            if (Dict.contains(s.substring(i, j + 1)) && method2(s, j + 1, Dict, record)) {
                return record[i] = true;
            }
        }
        return record[i] = false;
    }
}
