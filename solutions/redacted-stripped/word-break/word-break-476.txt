class Solution {

    Set<String> v = new HashSet();

    public boolean method1(String s, List<String> dict) {
        if (v.contains(s))
            return false;
        for (String var1 : dict) {
            if (s.startsWith(var1)) {
                int len = var1.length();
                if (s.length() == len)
                    return true;
                else {
                    if (method1(s.substring(len), dict))
                        return true;
                }
            }
        }
        v.add(s);
        return false;
    }
}
