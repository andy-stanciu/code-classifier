class Solution {

    Map<String, Boolean> map = new HashMap<String, Boolean>();

    public boolean method1(String s, List<String> param1) {
        if (s.length() == 0) {
            return true;
        }
        if (map.containsKey(s) && !map.get(s)) {
            return false;
        }
        for (String str : param1) {
            int len = str.length();
            if (s.startsWith(str)) {
                if (method1(s.substring(len), param1)) {
                    return true;
                }
            }
        }
        map.put(s, false);
        return false;
    }
}
