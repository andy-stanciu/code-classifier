class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1);
        return method2(n, dp);
    }

    public int method2(int n, int[] dp) {
        if (n == 1 || n == 2) {
            return n;
        }
        if (dp[n] != -1) {
            return dp[n];
        }
        int x = method2(n - 1, dp);
        int y = method2(n - 2, dp);
        dp[n] = x + y;
        return x + y;
    }
}
