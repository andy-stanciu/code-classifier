class Solution {

    public static void main(String[] args) {
        Solution solution = new Solution();
        System.out.println(solution.method1(6));
    }

    public int method1(int n) {
        if (n <= 3) {
            return n;
        }
        int a = 3, b = 2;
        for (int i = 4; i <= n; i++) {
            a = a + b;
            b = a - b;
        }
        return a;
    }
}
