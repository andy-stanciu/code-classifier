class Solution {

    public int method1(int n) {
        int cnt = 0;
        int[] dp = new int[n];
        dp[0] = 1;
        if (n == 1)
            return dp[0];
        dp[1] = 2;
        for (int i = 2; i < dp.length; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n - 1];
    }
}
