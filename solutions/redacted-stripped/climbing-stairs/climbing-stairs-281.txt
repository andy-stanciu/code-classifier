class Solution {

    public int method1(int n) {
        if (n <= 3) {
            return n;
        }
        int prev1step = 2;
        int prev2step = 1;
        int current = 0;
        for (int i = 3; i <= n; ++i) {
            current = prev1step + prev2step;
            prev2step = prev1step;
            prev1step = current;
        }
        return current;
    }
}
