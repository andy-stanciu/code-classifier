class Solution {

    public int method1(int n) {
        if (n == 1) {
            return 1;
        }
        if (n == 2) {
            return 2;
        }
        int current = 0;
        int num1 = 1;
        int num2 = 2;
        for (int i = 3; i <= n; i++) {
            current = num1 + num2;
            num1 = num2;
            num2 = current;
        }
        return current;
    }
}
