class Solution {

    public int method1(int n) {
        int i = 0;
        int[] contador = new int[n + 1];
        contador[0] = 1;
        contador[1] = 1;
        if (n <= 3) {
            return n;
        }
        for (i = 2; i <= n; i++) {
            contador[i] = contador[i - 1] + contador[i - 2];
        }
        return contador[n];
    }
}
