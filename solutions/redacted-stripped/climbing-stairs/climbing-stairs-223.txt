class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1);
        return compute(n, dp);
    }

    public int compute(int n, int[] dp) {
        if (n == 0 || n == 1)
            return 1;
        if (dp[n] != -1)
            return dp[n];
        int oneStep = compute(n - 1, dp);
        int twoSteps = compute(n - 2, dp);
        return dp[n] = oneStep + twoSteps;
    }
}
