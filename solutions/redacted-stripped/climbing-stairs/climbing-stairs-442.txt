class Solution {

    public int method1(int n) {
        int odd = 1, eve = 2;
        if (n == 1 || n == 2)
            return n;
        for (int i = 3; i <= n; i++) {
            if (i % 2 == 1)
                odd += eve;
            else
                eve += odd;
        }
        return n % 2 == 0 ? eve : odd;
    }
}
