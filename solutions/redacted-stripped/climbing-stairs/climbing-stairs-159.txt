class Solution {

    int[] memo;

    public int method1(int n) {
        memo = new int[n + 1];
        for (int i = 0; i <= n; i++) memo[i] = -1;
        return method2(n);
    }

    public int method2(int n) {
        if (memo[n] == -1) {
            int r;
            if (n == 0 || n == 1 || n == 2)
                r = n;
            else {
                r = method2(n - 1) + method2(n - 2);
            }
            memo[n] = r;
        }
        return memo[n];
    }
}
