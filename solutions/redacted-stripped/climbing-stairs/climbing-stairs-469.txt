class Solution {

    public int method1(int n) {
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        int n2Call = 1;
        int n1Call = 2;
        int current = 0;
        for (int i = 3; i <= n; i++) {
            current = n1Call + n2Call;
            n2Call = n1Call;
            n1Call = current;
        }
        return current;
    }
}
