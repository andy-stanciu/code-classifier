class Solution {

    public int method1(int n) {
        int[] memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return cli(n - 1, memo) + cli(n - 2, memo);
    }

    private int cli(int n, int[] memo) {
        if (n < 0) {
            return 0;
        }
        if (n == 0 || n == 1) {
            memo[n] = 1;
            return 1;
        }
        if (memo[n] != -1) {
            return memo[n];
        }
        memo[n] = cli(n - 1, memo) + cli(n - 2, memo);
        return memo[n];
    }
}
