class Solution {

    public int method1(int n) {
        if (n == 1)
            return 1;
        int first = 0;
        int second = 1;
        int num = second;
        for (int i = 2; i <= n; i++) {
            num = first + second;
            first = second;
            second = num;
        }
        return num + first;
    }
}
