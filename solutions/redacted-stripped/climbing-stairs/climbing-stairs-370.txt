class Solution {

    Map<Integer, Integer> memory = new HashMap();

    public int method1(int n) {
        if (memory.containsKey(n))
            return memory.get(n);
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        int res = method1(n - 1) + method1(n - 2);
        memory.put(n, res);
        return res;
    }
}
