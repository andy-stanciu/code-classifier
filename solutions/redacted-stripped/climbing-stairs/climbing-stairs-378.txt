class Solution {

    public int method1(int n) {
        int[] cache = new int[n + 2];
        int res = memoization(n, cache);
        return res;
    }

    public int memoization(int n, int[] cache) {
        if (n <= 1) {
            return 1;
        }
        if (cache[n] != 0) {
            return cache[n];
        }
        cache[n] = memoization(n - 1, cache) + memoization(n - 2, cache);
        return cache[n];
    }
}
