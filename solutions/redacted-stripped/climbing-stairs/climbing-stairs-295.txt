class Solution {

    public int method1(int n) {
        int[] arr = new int[n + 1];
        Arrays.fill(arr, -1);
        return climb(n, arr);
    }

    private int climb(int n, int[] arr) {
        if (n == 1)
            return 1;
        if (n == 0)
            return 1;
        if (arr[n - 1] == -1) {
            arr[n - 1] = climb(n - 1, arr);
        }
        if (arr[n - 2] == -1) {
            arr[n - 2] = climb(n - 2, arr);
        }
        arr[n] = arr[n - 1] + arr[n - 2];
        return arr[n];
    }
}
