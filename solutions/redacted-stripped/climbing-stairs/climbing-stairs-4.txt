class Solution {

    public int method1(int n) {
        Map<Integer, Integer> memo = new HashMap<>();
        return method1(n, memo);
    }

    private int method1(int n, Map<Integer, Integer> memo) {
        if (n == 0 || n == 1) {
            return 1;
        }
        if (!memo.containsKey(n)) {
            memo.put(n, method1(n - 1, memo) + method1(n - 2, memo));
        }
        return memo.get(n);
    }
}
