class Solution {

    Map<Integer, Integer> map = new HashMap<Integer, Integer>();

    int temp;

    public int method1(int n) {
        if (map.containsKey(n)) {
            return map.get(n);
        }
        if (n == 2) {
            return 2;
        }
        if (n == 1) {
            return 1;
        }
        temp = method1(n - 1) + method1(n - 2);
        map.put(n, temp);
        return temp;
    }
}
