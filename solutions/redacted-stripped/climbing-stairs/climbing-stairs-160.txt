public class Solution {

    public int method1(int n) {
        int first = 1;
        int second = 2;
        int tmp = 0;
        if (n <= 2) {
            return n;
        } else {
            for (//every iteration,will add the last two steps together and save as \'second\'
            int i = 3; //every iteration,will add the last two steps together and save as \'second\'
            i <= n; i++) {
                tmp = second;
                second += first;
                first = tmp;
            }
        }
        return second;
    }
}
