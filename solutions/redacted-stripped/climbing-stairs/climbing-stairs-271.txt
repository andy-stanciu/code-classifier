class Solution {

    public int method1(int n) {
        if (n == 1 || n == 0)
            return 1;
        int prev1 = 1, prev2 = 1, temp = 0;
        for (int i = 2; i <= n; i++) {
            temp = prev1 + prev2;
            prev2 = prev1;
            prev1 = temp;
        }
        return temp;
    }
}
