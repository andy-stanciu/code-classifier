class Solution {

    public int ways(int n) {
        if (n == 0) {
            return 1;
        }
        int take = 0;
        if (n >= 2) {
            take = ways(n - 2);
        }
        int nottake = ways(n - 1);
        return take + nottake;
    }

    public int method1(int n) {
        return ways(n);
    }
}
