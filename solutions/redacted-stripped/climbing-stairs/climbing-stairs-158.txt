class Solution {

    public int method1(int n) {
        if (n <= 1) {
            return n;
        }
        int current = 0;
        int prev1 = 1;
        int prev2 = 1;
        for (int i = 2; i <= n; i++) {
            current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        return current;
    }
}
