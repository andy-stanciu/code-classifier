class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 1;
        return countSteps(n, dp);
    }

    static int countSteps(int n, int[] dp) {
        for (int i = 1; i <= n; i++) {
            if (//1-2 = -1 out of bounds of dp array
            i == 1)
                dp[i] = dp[i - 1];
            else
                dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }
}
