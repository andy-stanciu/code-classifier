class Solution {

    public int method1(int n) {
        int r = 0, r1 = 1, r2 = 2;
        if (n == 1)
            return r1;
        else if (n == 2)
            return r2;
        for (int i = 2; i < n; i++) {
            r = r1 + r2;
            r1 = r2;
            r2 = r;
        }
        return r;
    }
}
