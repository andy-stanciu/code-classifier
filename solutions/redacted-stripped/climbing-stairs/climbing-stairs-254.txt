class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        return countSteps(n, dp);
    }

    static int countSteps(int n, int[] dp) {
        if (n == 0) {
            return 1;
        }
        if (n == 1) {
            return 1;
        }
        if (dp[n] != 0) {
            return dp[n];
        }
        int lh = countSteps(n - 1, dp);
        int rh = countSteps(n - 2, dp);
        return dp[n] = lh + rh;
    }
}
