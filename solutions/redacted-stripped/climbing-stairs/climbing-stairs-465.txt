class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1);
        return method2(n, dp);
    }

    private int method2(int n, int[] dp) {
        if (n == 0)
            return 1;
        if (n == 1)
            return 1;
        if (dp[n] != -1)
            return dp[n];
        dp[n] = method2(n - 1, dp) + method2(n - 2, dp);
        return dp[n];
    }
}
