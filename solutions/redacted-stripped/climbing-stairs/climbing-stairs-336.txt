class Solution {

    private int getAll(int i, int n, int[] t) {
        if (i == n)
            return 1;
        if (i > n)
            return 0;
        if (t[i] != -1)
            return t[i];
        t[i] = getAll(i + 1, n, t) + getAll(i + 2, n, t);
        return t[i];
    }

    public int method1(int n) {
        int[] t = new int[n + 1];
        for (int i = 0; i <= n; i++) t[i] = -1;
        return getAll(0, n, t);
    }
}
