class Solution {

    public int solve(int[] dp, int n) {
        dp[0] = 1;
        dp[1] = 1;
        if (dp[n] != -1) {
            return dp[n];
        }
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }

    public int method1(int n) {
        int[] dp = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            dp[i] = -1;
        }
        int ans = solve(dp, n);
        return ans;
    }
}
