class Solution {

    public int method1(int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return 1;
        int[] var1 = new int[n];
        var1[0] = 1;
        var1[1] = 2;
        for (int i = 2; i < n; i++) var1[i] = var1[i - 1] + var1[i - 2];
        return var1[n - 1];
    }
}
