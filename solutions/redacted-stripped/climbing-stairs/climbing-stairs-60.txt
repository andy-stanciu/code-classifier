class Solution {

    public int method1(int n) {
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        int l = 2;
        int pl = 1;
        int ans = 0;
        int i;
        for (i = 3; i <= n; i++) {
            ans = l + pl;
            pl = l;
            l = ans;
        }
        return ans;
    }
}
