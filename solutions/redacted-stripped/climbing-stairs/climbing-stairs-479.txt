class Solution {

    public int method1(int n) {
        List<Integer> dpArr = new ArrayList();
        int i = 2;
        dpArr.add(1);
        dpArr.add(2);
        while (i < n) {
            dpArr.add(dpArr.get(i - 1) + dpArr.get(i - 2));
            i++;
        }
        return dpArr.get(n - 1);
    }
}
