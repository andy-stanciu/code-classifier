class Solution {

    int[] dp = new int[46];

    public int method1(int n) {
        Arrays.fill(dp, -1);
        int i = 0;
        int x = climb(n, 0);
        return x;
    }

    public int climb(int n, int i) {
        if (i > n)
            return 0;
        if (i == n)
            return 1;
        if (dp[i] != -1)
            return dp[i];
        return dp[i] = climb(n, i + 1) + climb(n, i + 2);
    }
}
