class Solution {

    public int method1(int n) {
        if (n == 0) {
            return 1;
        }
        if (n <= 0) {
            return 0;
        }
        int[] dp = new int[n + 1];
        if (dp[n] != 0) {
            return dp[n];
        }
        dp[n] = method1(n - 1) + method1(n - 2);
        return dp[n];
    }
}
