class Solution {

    public int method1(int n) {
        if (n <= 3)
            return n;
        int first = 0;
        int second = 1;
        int third = first + second;
        for (int i = 0; i < n; i++) {
            third = first + second;
            first = second;
            second = third;
        }
        return third;
    }
}
