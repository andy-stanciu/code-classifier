class Solution {

    public int method1(int n) {
        if (n <= 1)
            return 1;
        int[] dp = new int[n + 1];
        int prev1 = 1;
        int prev2 = 1;
        for (int i = 2; i <= n; i++) {
            int curr = prev1 + prev2;
            prev2 = prev1;
            prev1 = curr;
        }
        return prev1;
    }
}
