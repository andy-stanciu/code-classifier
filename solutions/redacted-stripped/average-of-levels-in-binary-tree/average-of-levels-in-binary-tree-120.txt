class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            double var1 = q.size();
            double levelTotal = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode curr = q.remove();
                levelTotal = levelTotal + curr.val;
                if (curr.left != null) {
                    q.add(curr.left);
                }
                if (curr.right != null) {
                    q.add(curr.right);
                }
            }
            double var2 = levelTotal / var1;
            result.add(var2);
        }
        return result;
    }
}
