class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> qu = new LinkedList<>();
        List<Double> l = new ArrayList<>();
        qu.add(root);
        while (!qu.isEmpty()) {
            int var1 = qu.size();
            double nodes = var1;
            double sum = 0;
            double avg = 0;
            while (var1 > 0) {
                TreeNode curr = qu.poll();
                if (curr.left != null)
                    qu.add(curr.left);
                if (curr.right != null)
                    qu.add(curr.right);
                sum += curr.val;
                var1--;
            }
            avg = (sum) / nodes;
            l.add(avg);
        }
        return l;
    }
}
