class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            int var1 = queue.size();
            double avg = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode ans = queue.poll();
                avg += ans.val;
                if (ans.left != null) {
                    queue.offer(ans.left);
                }
                if (ans.right != null) {
                    queue.offer(ans.right);
                }
            }
            avg = avg / var1;
            result.add(avg);
        }
        return result;
    }
}
