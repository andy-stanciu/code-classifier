class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            double size = (double) q.size();
            long sum = 0;
            for (int i = 0; i < size; i++) {
                TreeNode front = q.poll();
                sum += front.val;
                if (front.left != null) {
                    q.add(front.left);
                }
                if (front.right != null) {
                    q.add(front.right);
                }
            }
            ans.add(sum / size);
        }
        return ans;
    }
}
