class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            int size = queue.size();
            double level = 0;
            for (int i = 0; i < size; i++) {
                TreeNode front = queue.poll();
                level += front.val;
                if (front.left != null) {
                    queue.add(front.left);
                }
                if (front.right != null) {
                    queue.add(front.right);
                }
            }
            double var1 = level / size;
            result.add(var1);
        }
        return result;
    }
}
