class Solution {

    public List<Double> method1(TreeNode node) {
        List<Double> result = new ArrayList<Double>();
        if (node == null) {
            return result;
        }
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        queue.offer(node);
        while (!queue.isEmpty()) {
            int queueSize = queue.size();
            double sum = 0;
            for (int i = 1; i <= queueSize; i++) {
                TreeNode cureentNodeValue = queue.poll();
                sum += cureentNodeValue.val;
                if (cureentNodeValue.left != null) {
                    queue.offer(cureentNodeValue.left);
                }
                if (cureentNodeValue.right != null) {
                    queue.offer(cureentNodeValue.right);
                }
            }
            result.add(sum / queueSize);
        }
        return result;
    }
}
