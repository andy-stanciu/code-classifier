class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int n = q.size();
            double sum = 0;
            double count = 0;
            while (n > 0) {
                TreeNode curr = q.remove();
                sum += curr.val;
                count++;
                if (curr.left != null)
                    q.add(curr.left);
                if (curr.right != null)
                    q.add(curr.right);
                n--;
            }
            list.add(sum / count);
        }
        return list;
    }
}
