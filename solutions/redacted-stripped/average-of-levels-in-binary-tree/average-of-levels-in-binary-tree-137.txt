class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> avg = new LinkedList<>();
        avg.offer(root);
        while (!avg.isEmpty()) {
            int count = 0;
            double sum = 0;
            int var1 = avg.size();
            for (int i = 0; i < var1; i++) {
                TreeNode currentNode = avg.poll();
                sum += currentNode.val;
                count++;
                if (currentNode.left != null) {
                    avg.offer(currentNode.left);
                }
                if (currentNode.right != null) {
                    avg.offer(currentNode.right);
                }
            }
            result.add(sum / count);
        }
        return result;
    }
}
