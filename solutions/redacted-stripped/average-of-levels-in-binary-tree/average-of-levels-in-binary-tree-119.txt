class Solution {

    public List<Double> method1(TreeNode root) {
        double sum = 0, n = 0, var1 = 0;
        Queue<TreeNode> qu = new LinkedList<>();
        ArrayList<Double> list = new ArrayList<>();
        qu.add(root);
        qu.add(null);
        while (!qu.isEmpty()) {
            root = qu.poll();
            sum += root.val;
            n++;
            if (root.left != null)
                qu.add(root.left);
            if (root.right != null)
                qu.add(root.right);
            if (qu.peek() == null) {
                var1 = sum / n;
                list.add(var1);
                var1 = 0;
                sum = 0;
                n = 0;
                qu.poll();
                if (qu.isEmpty())
                    break;
                qu.add(null);
            }
        }
        return list;
    }
}
