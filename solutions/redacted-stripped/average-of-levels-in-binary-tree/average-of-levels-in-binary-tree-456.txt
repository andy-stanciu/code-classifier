class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<Double>();
        Queue<TreeNode> queue = new LinkedList<>();
        if (root == null)
            return ans;
        queue.add(root);
        while (!queue.isEmpty()) {
            Double sum = 0.0;
            int count = 0;
            int n = queue.size();
            while (n-- > 0) {
                TreeNode curr = queue.poll();
                sum += curr.val;
                count++;
                if (curr.left != null)
                    queue.add(curr.left);
                if (curr.right != null)
                    queue.add(curr.right);
            }
            ans.add(sum / count);
            count = 0;
        }
        return ans;
    }
}
