class Solution {

    public List<Double> method1(TreeNode root) {
        Double sum = 0d;
        int var1 = 0;
        ArrayList<TreeNode> pred = new ArrayList<>();
        ArrayList<TreeNode> next = null;
        ArrayList<Double> avgList = new ArrayList<>();
        pred.add(root);
        while (!pred.isEmpty()) {
            sum = 0d;
            var1 = 0;
            next = new ArrayList<>();
            for (TreeNode node : pred) {
                sum += node.val;
                var1++;
                if (node.left != null)
                    next.add(node.left);
                if (node.right != null)
                    next.add(node.right);
            }
            Double avg = sum / var1;
            avgList.add(avg);
            pred = next;
        }
        return avgList;
    }
}
