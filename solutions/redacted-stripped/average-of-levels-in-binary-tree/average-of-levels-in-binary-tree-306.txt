class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int var1 = q.size();
            long sum = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode node = q.remove();
                sum += (long) node.val;
                if (node.left != null) {
                    q.add(node.left);
                }
                if (node.right != null) {
                    q.add(node.right);
                }
            }
            double a = ((double) sum / (double) var1);
            ans.add(a);
        }
        return ans;
    }
}
