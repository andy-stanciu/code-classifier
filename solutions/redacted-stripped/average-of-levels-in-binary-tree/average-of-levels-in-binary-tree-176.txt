class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> al = new ArrayList();
        Double sum = 0.0;
        Queue<TreeNode> q = new LinkedList();
        q.add(root);
        while (q.isEmpty() == false) {
            sum = 0.0;
            int size = q.size();
            for (int i = 0; i < size; i++) {
                TreeNode node = q.poll();
                sum = sum + node.val;
                if (node.left != null)
                    q.add(node.left);
                if (node.right != null)
                    q.add(node.right);
            }
            al.add(sum / size);
        }
        return al;
    }
}
