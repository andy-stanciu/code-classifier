class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        ArrayList<Double> ans = new ArrayList<>();
        q.add(root);
        double size = 1;
        while (!q.isEmpty()) {
            size = q.size();
            double num = size;
            double sum = 0;
            while (size-- > 0) {
                TreeNode currNode = q.remove();
                sum += currNode.val;
                if (currNode.left != null)
                    q.add(currNode.left);
                if (currNode.right != null)
                    q.add(currNode.right);
            }
            ans.add(sum / num);
        }
        return ans;
    }
}
