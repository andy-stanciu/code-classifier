class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            int var1 = queue.size();
            double avg = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode node = queue.poll();
                avg += node.val;
                if (node.left != null) {
                    queue.offer(node.left);
                }
                if (node.right != null) {
                    queue.offer(node.right);
                }
            }
            avg = avg / var1;
            ans.add(avg);
        }
        return ans;
    }
}
