class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            int var1 = queue.size();
            double sum = 0.00;
            for (int i = 0; i < var1; i++) {
                TreeNode curr = queue.poll();
                sum += (double) curr.val;
                if (curr.left != null)
                    queue.add(curr.left);
                if (curr.right != null)
                    queue.add(curr.right);
            }
            double avg = sum / var1;
            ans.add(avg);
        }
        return ans;
    }
}
