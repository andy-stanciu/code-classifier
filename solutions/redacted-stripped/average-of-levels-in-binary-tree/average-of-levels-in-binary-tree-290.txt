class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> avgList = new ArrayList<>();
        if (root == null)
            return avgList;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            double avgValue = 0;
            int level = q.size();
            for (int i = 0; i < level; i++) {
                TreeNode node = q.poll();
                avgValue += node.val;
                if (node.left != null)
                    q.offer(node.left);
                if (node.right != null)
                    q.offer(node.right);
            }
            avgList.add(avgValue / level);
        }
        return avgList;
    }
}
