class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> out = new LinkedList<>();
        if (root == null) {
            return out;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            double sum = 0;
            int var1 = q.size();
            for (int i = 0; i < var1; i++) {
                if (q.peek().left != null) {
                    q.add(q.peek().left);
                }
                if (q.peek().right != null) {
                    q.add(q.peek().right);
                }
                sum += q.remove().val;
            }
            out.add(sum / var1);
        }
        return out;
    }
}
