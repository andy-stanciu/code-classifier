class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        Queue<TreeNode> q = new LinkedList<>();
        int elementCount = 1;
        q.add(root);
        while (!q.isEmpty()) {
            int newElementCount = 0;
            double sum = 0;
            for (int i = 0; i < elementCount; i++) {
                TreeNode node = q.poll();
                sum += node.val;
                if (node.left != null) {
                    q.add(node.left);
                    newElementCount++;
                }
                if (node.right != null) {
                    q.add(node.right);
                    newElementCount++;
                }
            }
            res.add(sum / elementCount);
            elementCount = newElementCount;
        }
        return res;
    }
}
