class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        Queue<TreeNode> que = new LinkedList<>();
        que.add(root);
        int size = 1;
        while (size > 0) {
            double sum = 0;
            double oldSize = size;
            for (int i = size; i > 0; i--) {
                TreeNode temp = que.remove();
                sum += temp.val;
                if (temp.left != null) {
                    size++;
                    que.add(temp.left);
                }
                if (temp.right != null) {
                    size++;
                    que.add(temp.right);
                }
            }
            size -= oldSize;
            list.add(sum / oldSize);
        }
        return list;
    }
}
