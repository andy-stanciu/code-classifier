class Solution {

    public List<Double> method1(TreeNode root) {
        ArrayList<Double> ans = new ArrayList<>();
        if (root == null)
            return ans;
        Queue<TreeNode> qt = new LinkedList<>();
        qt.add(root);
        double sum = 0.0;
        while (!qt.isEmpty()) {
            int var1 = qt.size();
            sum = 0.0;
            for (int i = 0; i < var1; i++) {
                TreeNode temp = qt.remove();
                if (temp.left != null)
                    qt.add(temp.left);
                if (temp.right != null)
                    qt.add(temp.right);
                sum += temp.val;
            }
            double avg = sum / var1;
            ans.add(avg);
        }
        return ans;
    }
}
