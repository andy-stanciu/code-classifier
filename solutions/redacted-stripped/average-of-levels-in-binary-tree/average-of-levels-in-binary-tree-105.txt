class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        List<Integer> num = new ArrayList<>();
        getAvgs(root, list, num, 0);
        for (int i = 0; i < list.size(); i++) {
            list.set(i, list.get(i) / num.get(i));
        }
        return list;
    }

    public void getAvgs(TreeNode root, List<Double> list, List<Integer> num, int current) {
        if (root == null) {
            return;
        }
        if (current < list.size()) {
            list.set(current, list.get(current) + root.val);
            num.set(current, num.get(current) + 1);
        } else {
            list.add((double) root.val);
            num.add(1);
        }
        getAvgs(root.left, list, num, current + 1);
        getAvgs(root.right, list, num, current + 1);
        return;
    }
}
