class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            int var1 = q.size();
            long a = 0;
            double b = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode curr = q.poll();
                a += curr.val;
                b++;
                if (curr.left != null) {
                    q.offer(curr.left);
                }
                if (curr.right != null) {
                    q.offer(curr.right);
                }
            }
            list.add(a / b);
        }
        return list;
    }
}
