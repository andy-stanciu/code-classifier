class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        if (root == null) {
            return ans;
        }
        q.offer(root);
        while (!q.isEmpty()) {
            int size = q.size();
            double va = 0.0;
            int s = size;
            while (size-- > 0) {
                TreeNode temp = q.poll();
                va = va + temp.val;
                if (temp.left != null) {
                    q.offer(temp.left);
                }
                if (temp.right != null) {
                    q.offer(temp.right);
                }
            }
            ans.add(va / s);
        }
        return ans;
    }
}
