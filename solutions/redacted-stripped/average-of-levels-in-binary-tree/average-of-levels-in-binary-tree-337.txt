class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList();
        Queue<TreeNode> q = new LinkedList();
        q.add(root);
        int size = 0;
        Double sum = Double.valueOf(root.val);
        while (!q.isEmpty()) {
            size = q.size();
            Double value = Double.valueOf(sum / size);
            list.add(value);
            sum = Double.valueOf(0);
            while (size > 0) {
                TreeNode temp = q.peek();
                q.remove();
                if (temp.right != null) {
                    sum = sum + Double.valueOf(temp.right.val);
                    q.add(temp.right);
                }
                if (temp.left != null) {
                    sum = sum + Double.valueOf(temp.left.val);
                    q.add(temp.left);
                }
                size--;
            }
        }
        return list;
    }
}
