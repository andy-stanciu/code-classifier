class Solution {

    public List<Double> method1(TreeNode root) {
        if (root == null)
            return null;
        List<Double> list = new ArrayList<>();
        if (root.left == null && root.right == null) {
            list.add((double) root.val);
            return list;
        }
        Queue<TreeNode> queue = new ArrayDeque<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            int size = queue.size();
            Double var1 = 0D;
            Double count = 0D;
            while (size-- > 0) {
                count++;
                TreeNode node = queue.remove();
                var1 += node.val;
                if (node.left != null) {
                    queue.add(node.left);
                }
                if (node.right != null) {
                    queue.add(node.right);
                }
            }
            list.add(var1 / count);
        }
        return list;
    }
}
