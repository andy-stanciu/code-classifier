class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        q.add(null);
        List<Double> ans = new ArrayList<>();
        while (q.peek() != null) {
            double sum = 0;
            int node = 0;
            while (q.peek() != null) {
                TreeNode curr = q.poll();
                sum = sum + curr.val;
                node++;
                if (curr.left != null) {
                    q.add(curr.left);
                }
                if (curr.right != null) {
                    q.add(curr.right);
                }
            }
            q.add(q.poll());
            ans.add(sum / node);
        }
        return ans;
    }
}
