class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        TreeNode node;
        double sum = 0, count = 0;
        List<Double> lst = new ArrayList<>();
        queue.add(root);
        queue.add(null);
        while (!queue.isEmpty()) {
            node = queue.remove();
            if (node == null) {
                lst.add(sum / count);
                sum = 0;
                count = 0;
                if (queue.size() > 0)
                    queue.add(null);
                continue;
            }
            sum += node.val;
            count = count + 1.0;
            if (node.left != null)
                queue.add(node.left);
            if (node.right != null)
                queue.add(node.right);
        }
        return lst;
    }
}
