class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ansList = new ArrayList<Double>();
        if (root == null)
            return ansList;
        Queue<TreeNode> q = new LinkedList<TreeNode>();
        q.offer(root);
        while (!q.isEmpty()) {
            int s = q.size();
            double levelavg = 0;
            for (int i = 0; i < s; i++) {
                TreeNode cur = q.poll();
                levelavg += (double) cur.val;
                if (cur.left != null)
                    q.offer(cur.left);
                if (cur.right != null)
                    q.offer(cur.right);
            }
            levelavg = levelavg / s;
            ansList.add(levelavg);
        }
        return ansList;
    }
}
