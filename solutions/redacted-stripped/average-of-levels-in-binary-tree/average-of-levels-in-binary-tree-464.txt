class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> var1 = new LinkedList<>();
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        while (queue.size() > 0) {
            int n = queue.size();
            Double sum = 0.0;
            for (int i = 1; i <= n; ++i) {
                TreeNode node = queue.poll();
                sum += node.val;
                if (node.left != null)
                    queue.add(node.left);
                if (node.right != null)
                    queue.add(node.right);
            }
            var1.add(sum / (double) n);
        }
        return var1;
    }
}
