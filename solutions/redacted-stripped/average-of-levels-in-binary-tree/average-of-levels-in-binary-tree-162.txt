class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        Queue<TreeNode> level = new LinkedList<>();
        level.offer(root);
        while (!level.isEmpty()) {
            int var1 = level.size();
            double levelAvg = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode node = level.poll();
                levelAvg += node.val;
                if (node.left != null) {
                    level.offer(node.left);
                }
                if (node.right != null) {
                    level.offer(node.right);
                }
            }
            levelAvg /= var1;
            ans.add(levelAvg);
        }
        return ans;
    }
}
