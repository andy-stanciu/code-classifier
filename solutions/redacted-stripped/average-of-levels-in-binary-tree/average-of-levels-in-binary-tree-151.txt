class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        if (root == null) {
            return new ArrayList<>();
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        q.add(null);
        while (q.peek() != null) {
            int nodes = 0;
            Double sum = 0.0;
            while (q.peek() != null) {
                TreeNode temp = q.poll();
                nodes++;
                sum += temp.val;
                if (temp.left != null) {
                    q.add(temp.left);
                }
                if (temp.right != null) {
                    q.add(temp.right);
                }
            }
            q.add(q.poll());
            list.add(sum / nodes);
        }
        return list;
    }
}
