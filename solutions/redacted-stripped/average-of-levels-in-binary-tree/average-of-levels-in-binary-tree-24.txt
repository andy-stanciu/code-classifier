class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> var1 = new ArrayList<Double>();
        if (root == null)
            return var1;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            int var2 = queue.size();
            double sum = 0.0;
            for (int i = 0; i < var2; i++) {
                TreeNode node = queue.poll();
                sum += node.val;
                if (node.left != null)
                    queue.add(node.left);
                if (node.right != null)
                    queue.add(node.right);
            }
            var1.add(sum / var2);
        }
        return var1;
    }
}
