class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        List<Double> result = new ArrayList<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            int num = queue.size();
            int numbkp = num;
            long sum = 0;
            while (num != 0) {
                TreeNode node = queue.remove();
                sum = sum + (long) node.val;
                if (node.left != null) {
                    queue.add(node.left);
                }
                if (node.right != null) {
                    queue.add(node.right);
                }
                num--;
            }
            result.add((double) sum / numbkp);
        }
        return result;
    }
}
