class Solution {

    public List<Double> method1(TreeNode root) {
        ArrayList<Double> ans = new ArrayList<Double>();
        Queue<TreeNode> arr = new LinkedList<TreeNode>();
        arr.add(root);
        while (arr.isEmpty() == false) {
            int size = arr.size();
            double avg = 0;
            for (int i = 0; i < size; i++) {
                TreeNode curr = arr.poll();
                avg = avg + curr.val;
                if (curr.left != null)
                    arr.add(curr.left);
                if (curr.right != null)
                    arr.add(curr.right);
            }
            ans.add(avg / size);
        }
        return ans;
    }
}
