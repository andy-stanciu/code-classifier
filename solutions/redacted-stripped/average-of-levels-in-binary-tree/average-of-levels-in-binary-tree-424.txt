class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> li = new ArrayList<Double>();
        Queue<TreeNode> q = new LinkedList<TreeNode>();
        q.add(root);
        while (!q.isEmpty()) {
            double size = q.size();
            double s = size;
            double sum = 0;
            if (size == 0)
                return li;
            while (size > 0) {
                TreeNode temp = q.remove();
                sum = sum + temp.val;
                if (temp.left != null) {
                    q.add(temp.left);
                }
                if (temp.right != null) {
                    q.add(temp.right);
                }
                size--;
            }
            li.add(sum / s);
        }
        return li;
    }
}
