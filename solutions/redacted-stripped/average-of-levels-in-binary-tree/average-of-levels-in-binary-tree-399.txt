class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> anslist = new ArrayList();
        Queue<TreeNode> queue = new LinkedList<>();
        if (root == null)
            return anslist;
        queue.add(root);
        while (!queue.isEmpty()) {
            int size = queue.size();
            double sum = 0.0;
            for (int i = 0; i < size; i++) {
                TreeNode node = queue.poll();
                sum += node.val;
                if (node.left != null)
                    queue.add(node.left);
                if (node.right != null)
                    queue.add(node.right);
            }
            anslist.add(sum / size);
        }
        return anslist;
    }
}
