class Solution {

    public List<Double> method1(TreeNode root) {
        Deque<TreeNode> deque = new ArrayDeque<>();
        List<Double> res = new ArrayList<>();
        if (root != null) {
            deque.addLast(root);
        }
        while (!deque.isEmpty()) {
            int size = deque.size();
            double sum = 0d;
            for (int i = 0; i < size; i++) {
                TreeNode cur = deque.removeFirst();
                sum += cur.val;
                if (cur.left != null) {
                    deque.addLast(cur.left);
                }
                if (cur.right != null) {
                    deque.addLast(cur.right);
                }
            }
            res.add(sum / size);
        }
        return res;
    }
}
