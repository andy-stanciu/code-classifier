class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        List<Double> list = new ArrayList<>();
        if (root == null)
            return list;
        queue.offer(root);
        while (queue.isEmpty() == false) {
            double avg = 0;
            int l = queue.size();
            for (int i = 0; i < l; i++) {
                TreeNode t = queue.poll();
                if (t.left != null)
                    queue.offer(t.left);
                if (t.right != null)
                    queue.offer(t.right);
                avg += t.val;
            }
            list.add(avg / l);
        }
        return list;
    }
}
