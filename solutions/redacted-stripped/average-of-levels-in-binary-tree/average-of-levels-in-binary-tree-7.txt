class Solution {

    public List<Double> method1(TreeNode root) {
        ArrayList<Double> ans = new ArrayList<>();
        Queue<TreeNode> q1 = new LinkedList<>();
        q1.add(root);
        while (q1.size() > 0) {
            int n = q1.size();
            double sum = 0.000;
            for (int i = 0; i < n; i++) {
                TreeNode node = q1.poll();
                sum = sum + node.val;
                if (node.left != null)
                    q1.offer(node.left);
                if (node.right != null)
                    q1.offer(node.right);
            }
            ans.add(sum / n);
        }
        return ans;
    }
}
