class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> arr = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            double sum = 0;
            if (root == null) {
                return arr;
            }
            int size = q.size();
            for (int i = 0; i < size; i++) {
                TreeNode curr = q.remove();
                sum += curr.val;
                if (curr.left != null) {
                    q.add(curr.left);
                }
                if (curr.right != null) {
                    q.add(curr.right);
                }
            }
            double d = sum / (double) size;
            arr.add(d);
        }
        return arr;
    }
}
