class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int currSize = q.size();
            double sum = 0.00000;
            int i = currSize;
            while (i > 0) {
                TreeNode curr = q.poll();
                sum += curr.val;
                if (curr.left != null)
                    q.add(curr.left);
                if (curr.right != null)
                    q.add(curr.right);
                i--;
            }
            ans.add(sum / currSize);
        }
        return ans;
    }
}
