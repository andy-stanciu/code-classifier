class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> solution = new ArrayList<>();
        if (root == null) {
            return solution;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            int n = queue.size();
            Double sum = 0.0;
            int i = 0;
            for (i = 0; i < n; i++) {
                TreeNode temp = queue.poll();
                sum = sum + temp.val;
                if (temp.left != null) {
                    queue.offer(temp.left);
                }
                if (temp.right != null) {
                    queue.offer(temp.right);
                }
            }
            solution.add(sum / n);
        }
        return solution;
    }
}
