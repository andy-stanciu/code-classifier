class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        List<Integer> count = new ArrayList<>();
        method1(root, 0, list, count);
        for (int i = 0; i < list.size(); i++) list.set(i, list.get(i) / count.get(i));
        return list;
    }

    private void method1(TreeNode root, int level, List<Double> list, List<Integer> count) {
        if (root == null)
            return;
        if (list.size() == level) {
            list.add(level, (double) root.val);
            count.add(level, 1);
        } else {
            list.set(level, root.val + list.get(level));
            count.set(level, count.get(level) + 1);
        }
        level++;
        method1(root.left, level, list, count);
        method1(root.right, level, list, count);
    }
}
