class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> res = new ArrayList<>();
        if (root == null)
            return res;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int var1 = q.size();
            double sum = 0;
            int count = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode node = q.poll();
                sum += node.val;
                count++;
                if (node.left != null)
                    q.add(node.left);
                if (node.right != null)
                    q.add(node.right);
            }
            res.add(sum / count);
        }
        return res;
    }
}
