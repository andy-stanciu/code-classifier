class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            int var1 = queue.size();
            double var2 = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode node = queue.poll();
                var2 += node.val;
                if (node.left != null) {
                    queue.offer(node.left);
                }
                if (node.right != null) {
                    queue.offer(node.right);
                }
            }
            double levelAvg = var2 / var1;
            result.add(levelAvg);
        }
        return result;
    }
}
