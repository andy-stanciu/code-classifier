class Solution {

    private static final Deque<TreeNode> q = new ArrayDeque<>(5000);

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        if (root == null)
            return list;
        q.clear();
        q.add(root);
        int n;
        while ((n = q.size()) != 0) {
            long sum = 0;
            for (int i = 0; i < n; i++) {
                TreeNode node = q.removeFirst();
                sum += node.val;
                if (node.left != null)
                    q.addLast(node.left);
                if (node.right != null)
                    q.addLast(node.right);
            }
            list.add((double) sum / n);
        }
        return list;
    }
}
