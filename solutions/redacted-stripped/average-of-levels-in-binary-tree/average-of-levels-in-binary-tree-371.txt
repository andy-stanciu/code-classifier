class Solution {

    public List<Double> method1(TreeNode root) {
        final var result = new ArrayList<Double>();
        if (root == null)
            return result;
        final var list = new LinkedList<TreeNode>();
        list.add(root);
        while (!list.isEmpty()) {
            var avgSum = 0.0;
            final var n = list.size();
            for (var i = 0; i < n; i++) {
                final var node = list.poll();
                avgSum += node.val;
                if (node.left != null)
                    list.add(node.left);
                if (node.right != null)
                    list.add(node.right);
            }
            result.add(avgSum / n);
        }
        return result;
    }
}
