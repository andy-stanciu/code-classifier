class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList();
        Queue<TreeNode> q = new LinkedList();
        q.add(root);
        avg(q, ans);
        return ans;
    }

    static void avg(Queue<TreeNode> q, List<Double> ans) {
        if (q.isEmpty())
            return;
        int var1 = q.size();
        int count = var1;
        long sum = 0;
        while (count > 0) {
            TreeNode node = q.poll();
            sum += node.val;
            if (node.left != null)
                q.add(node.left);
            if (node.right != null)
                q.add(node.right);
            count--;
        }
        ans.add((double) sum / var1);
        avg(q, ans);
    }
}
