class Solution {

    public String method1(String s, String t) {
        int sLen = s.length();
        int tLen = t.length();
        int[] tFreqs = new int[58];
        for (int i = 0; i < tLen; i++) {
            tFreqs[t.charAt(i) - 65]++;
        }
        int start = 0;
        int end = 0;
        int method1 = sLen + 1;
        int minStart = -1;
        int count = 0;
        while (end < sLen) {
            if (tFreqs[s.charAt(end++) - 65]-- > 0) {
                count++;
            }
            while (count == tLen) {
                if (end - start < method1) {
                    method1 = end - start;
                    minStart = start;
                }
                if (++tFreqs[s.charAt(start++) - 65] > 0) {
                    count--;
                }
            }
        }
        return minStart == -1 ? "" : s.substring(minStart, minStart + method1);
    }
}
