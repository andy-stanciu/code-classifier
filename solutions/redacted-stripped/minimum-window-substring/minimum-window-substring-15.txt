class Solution {

    public String method1(String s, String t) {
        Map<Character, Integer> targetFreq = new HashMap<>();
        Map<Character, Integer> var1 = new HashMap<>();
        for (char c : t.toCharArray()) {
            targetFreq.put(c, targetFreq.getOrDefault(c, 0) + 1);
        }
        int left = 0, right = 0, minLength = Integer.MAX_VALUE, minLeft = 0, requiredChars = targetFreq.size();
        while (right < s.length()) {
            char currentChar = s.charAt(right);
            var1.put(currentChar, var1.getOrDefault(currentChar, 0) + 1);
            if (targetFreq.containsKey(currentChar) && var1.get(currentChar).equals(targetFreq.get(currentChar))) {
                requiredChars--;
            }
            while (requiredChars == 0) {
                if (right - left + 1 < minLength) {
                    minLength = right - left + 1;
                    minLeft = left;
                }
                char leftChar = s.charAt(left);
                var1.put(leftChar, var1.get(leftChar) - 1);
                if (targetFreq.containsKey(leftChar) && var1.get(leftChar) < targetFreq.get(leftChar)) {
                    requiredChars++;
                }
                left++;
            }
            right++;
        }
        if (minLength == Integer.MAX_VALUE) {
            return "";
        }
        return s.substring(minLeft, minLeft + minLength);
    }
}
