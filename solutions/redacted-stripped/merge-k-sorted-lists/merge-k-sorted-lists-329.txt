class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> ans = new PriorityQueue<>((a, b) -> a.val - b.val);
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            ListNode temp = param1[i];
            while (temp != null) {
                ans.add(temp);
                temp = temp.next;
            }
        }
        ListNode result = ans.poll();
        ListNode temp = result;
        while (ans.size() > 0) {
            temp.next = ans.poll();
            temp = temp.next;
        }
        if (temp != null) {
            temp.next = null;
        }
        return result;
    }
}
