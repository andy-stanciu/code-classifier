class Solution {

    List<Integer> ans = new ArrayList<>();

    ListNode ptr = null;

    public ListNode method1(ListNode[] param1) {
        for (ListNode list : param1) {
            while (list != null) {
                ans.add(list.val);
                list = list.next;
            }
        }
        Collections.sort(ans);
        ListNode dum = new ListNode(0);
        ptr = dum;
        for (int i : ans) {
            ptr.next = new ListNode(i);
            ptr = ptr.next;
        }
        return dum.next;
    }
}
