class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode res = new ListNode(0);
        for (ListNode list : param1) {
            res.next = method2(res.next, list);
        }
        return res.next;
    }

    public ListNode method2(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode res = new ListNode(0);
        ListNode tail = res;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                tail.next = list1;
                list1 = list1.next;
            } else {
                tail.next = list2;
                list2 = list2.next;
            }
            tail = tail.next;
        }
        if (list1 == null)
            tail.next = list2;
        else
            tail.next = list1;
        return res.next;
    }
}
