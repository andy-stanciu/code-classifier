class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode result = null;
        for (ListNode temp : param1) {
            result = method2(result, temp);
        }
        return result;
    }

    public ListNode method2(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        if (list1.val < list2.val)
            return new ListNode(list1.val, method2(list1.next, list2));
        else
            return new ListNode(list2.val, method2(list1, list2.next));
    }
}
