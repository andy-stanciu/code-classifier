class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0)
            return null;
        int interval = 1;
        while (interval < param1.length) {
            for (int i = 0; i < param1.length - interval; i += 2 * interval) {
                param1[i] = method2(param1[i], param1[i + interval]);
            }
            interval *= 2;
        }
        return param1[0];
    }

    private ListNode method2(ListNode list1, ListNode list2) {
        ListNode prehead = new ListNode(-1);
        ListNode current = prehead;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                current.next = list1;
                list1 = list1.next;
            } else {
                current.next = list2;
                list2 = list2.next;
            }
            current = current.next;
        }
        current.next = list1 != null ? list1 : list2;
        return prehead.next;
    }
}
