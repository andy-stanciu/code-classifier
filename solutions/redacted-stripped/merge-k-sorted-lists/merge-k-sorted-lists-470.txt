class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1 == null || param1.length == 0)
            return null;
        if (param1.length == 1)
            return param1[0];
        if (param1.length == 2)
            return method2(param1[0], param1[1]);
        int n = param1.length;
        int i = 0;
        int j = n - 1;
        while (i != j && i < j) {
            param1[i] = method2(param1[i], param1[j]);
            j--;
        }
        return param1[0];
    }

    public ListNode method2(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        if (list1.val <= list2.val) {
            list1.next = method2(list1.next, list2);
            return list1;
        } else {
            list2.next = method2(list1, list2.next);
            return list2;
        }
    }
}
