class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0)
            return null;
        PriorityQueue<Integer> pq = new PriorityQueue();
        for (ListNode node : param1) {
            while (node != null) {
                pq.add(node.val);
                node = node.next;
            }
        }
        if (pq.isEmpty()) {
            return null;
        }
        ListNode head = new ListNode(pq.poll());
        ListNode prev = head;
        while (!pq.isEmpty()) {
            ListNode current = new ListNode(pq.poll());
            prev.next = current;
            prev = current;
        }
        return head;
    }
}
