class Solution {

    public ListNode method1(ListNode[] param1) {
        ArrayList<Integer> list = new ArrayList<Integer>();
        for (ListNode root : param1) {
            while (root != null) {
                list.add(root.val);
                root = root.next;
            }
        }
        Collections.sort(list);
        ListNode last = null, start = null;
        for (Integer i : list) {
            if (start == null) {
                start = last = new ListNode(i);
            } else {
                last.next = new ListNode(i);
                last = last.next;
            }
        }
        return start;
    }
}
