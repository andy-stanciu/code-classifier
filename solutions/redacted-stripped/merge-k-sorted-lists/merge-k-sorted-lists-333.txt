class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        ListNode ans = new ListNode(-1);
        ListNode temp = ans;
        for (int i = 0; i < param1.length; i++) {
            ListNode head = param1[i];
            if (head != null) {
                pq.add(head);
            }
        }
        while (!pq.isEmpty()) {
            ListNode curr = pq.remove();
            temp.next = curr;
            temp = temp.next;
            if (curr.next != null) {
                pq.add(curr.next);
            }
        }
        return ans.next;
    }
}
