class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> maxHeap = new PriorityQueue<>((a, b) -> b.val - a.val);
        for (ListNode head : param1) {
            while (head != null) {
                maxHeap.add(head);
                head = head.next;
            }
        }
        ListNode resHead = null;
        while (!maxHeap.isEmpty()) {
            ListNode popped = maxHeap.poll();
            popped.next = resHead;
            resHead = popped;
        }
        return resHead;
    }
}
