class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < param1.length; i++) {
            ListNode temp = param1[i];
            while (temp != null) {
                list.add(temp.val);
                temp = temp.next;
            }
        }
        Collections.sort(list);
        ListNode temp = null;
        ListNode head = null;
        for (int i = 0; i < list.size(); i++) {
            ListNode nn = new ListNode(list.get(i));
            if (head == null) {
                head = nn;
                temp = head;
            } else {
                temp.next = nn;
                temp = temp.next;
            }
        }
        return head;
    }
}
