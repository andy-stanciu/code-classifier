class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        if (l1 == null)
            return l2;
        if (l2 == null)
            return l1;
        ListNode dummy = new ListNode(0);
        ListNode curr = dummy;
        while (l1 != null && l2 != null) {
            if (l1.val <= l2.val) {
                curr.next = l1;
                l1 = l1.next;
            } else {
                curr.next = l2;
                l2 = l2.next;
            }
            curr = curr.next;
        }
        if (l1 != null)
            curr.next = l1;
        if (l2 != null)
            curr.next = l2;
        return dummy.next;
    }

    public ListNode method2(ListNode[] param1) {
        if (param1.length < 1)
            return null;
        int interval = 1;
        while (interval < param1.length) {
            for (int i = 0; i < param1.length - interval; i += 2 * interval) {
                param1[i] = method1(param1[i], param1[i + interval]);
            }
            interval *= 2;
        }
        return param1[0];
    }
}
