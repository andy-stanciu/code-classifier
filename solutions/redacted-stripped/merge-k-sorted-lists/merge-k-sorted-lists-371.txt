class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode lis : param1) {
            if (lis != null) {
                pq.add(lis);
            }
        }
        ListNode temp = new ListNode(0);
        ListNode curr = temp;
        while (!pq.isEmpty()) {
            ListNode node = pq.poll();
            curr.next = node;
            curr = curr.next;
            if (node.next != null) {
                pq.add(node.next);
            }
        }
        return temp.next;
    }
}
