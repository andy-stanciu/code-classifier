class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> arr = new ArrayList<>();
        for (int i = 0; i < param1.length; i++) {
            ListNode head = param1[i];
            while (head != null) {
                arr.add(head.val);
                head = head.next;
            }
        }
        Collections.sort(arr);
        ListNode list = new ListNode(0);
        ListNode head = list;
        for (Integer it : arr) {
            head.next = new ListNode(it);
            head = head.next;
        }
        return list.next;
    }
}
