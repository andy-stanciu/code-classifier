class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        ListNode head = null;
        if (param1.length == 0) {
            return null;
        }
        for (int i = 0; i < param1.length; i++) {
            ListNode t = param1[i];
            while (t != null) {
                queue.offer(t.val);
                t = t.next;
            }
        }
        if (!queue.isEmpty())
            head = new ListNode(queue.poll());
        ListNode temp = head;
        while (!queue.isEmpty()) {
            temp.next = new ListNode(queue.poll());
            temp = temp.next;
        }
        return head;
    }
}
