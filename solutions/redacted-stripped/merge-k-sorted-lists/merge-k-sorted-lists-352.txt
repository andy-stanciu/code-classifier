class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> l = new ArrayList<>();
        for (ListNode list : param1) {
            ListNode temp = list;
            while (temp != null) {
                l.add(temp.val);
                temp = temp.next;
            }
        }
        Collections.sort(l);
        ListNode dummy = new ListNode(0);
        ListNode iter = dummy;
        for (int i = 0; i < l.size(); i++) {
            dummy.next = new ListNode(l.get(i));
            dummy = dummy.next;
        }
        return iter.next;
    }
}
