class Solution {

    public ListNode method1(ListNode[] param1) {
        final int length = param1.length;
        if (length == 0)
            return null;
        ListNode curr = param1[0];
        for (int i = 1; i < length; i++) {
            curr = method2(curr, param1[i]);
        }
        return curr;
    }

    public ListNode method2(ListNode list1, ListNode list2) {
        final ListNode temp = new ListNode();
        ListNode curr = temp;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                curr.next = list1;
                list1 = list1.next;
            } else {
                curr.next = list2;
                list2 = list2.next;
            }
            curr = curr.next;
        }
        if (list1 != null) {
            curr.next = list1;
        } else {
            curr.next = list2;
        }
        return temp.next;
    }
}
