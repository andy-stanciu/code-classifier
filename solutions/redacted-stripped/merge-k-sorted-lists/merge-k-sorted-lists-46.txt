class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode node : param1) {
            while (node != null) {
                pq.offer(node);
                node = node.next;
            }
        }
        if (pq.isEmpty())
            return null;
        ListNode result = new ListNode(0);
        ListNode temp = result;
        while (!pq.isEmpty()) {
            temp.next = pq.poll();
            temp = temp.next;
        }
        temp.next = null;
        return result.next;
    }
}
