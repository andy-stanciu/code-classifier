class Solution {

    public ListNode method1(ListNode[] param1) {
        Queue<ListNode> pq = new PriorityQueue<ListNode>((a, b) -> a.val - b.val);
        for (ListNode node : param1) {
            if (null != node) {
                pq.add(node);
            }
        }
        ListNode head = new ListNode(-1);
        ListNode curr = head;
        while (!pq.isEmpty()) {
            curr.next = pq.poll();
            curr = curr.next;
            if (null != curr.next) {
                pq.add(curr.next);
            }
        }
        return head.next;
    }
}
