class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1 == null || param1.length == 0) {
            return null;
        }
        ListNode head = new ListNode(0);
        List<Integer> list = new ArrayList<>();
        ListNode temp = head;
        for (ListNode l : param1) {
            while (l != null) {
                list.add(l.val);
                l = l.next;
            }
        }
        Collections.sort(list);
        for (int i : list) {
            temp.next = new ListNode(i);
            temp = temp.next;
        }
        return head.next;
    }
}
