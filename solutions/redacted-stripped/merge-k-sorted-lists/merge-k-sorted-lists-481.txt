class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0)
            return null;
        ListNode res = new ListNode();
        ListNode head = res;
        PriorityQueue<ListNode> pq = new PriorityQueue<>((l1, l2) -> l1.val - l2.val);
        for (ListNode list : param1) {
            if (list != null) {
                pq.add(list);
            }
        }
        while (!pq.isEmpty()) {
            head.next = pq.poll();
            head = head.next;
            if (head.next != null) {
                pq.add(head.next);
            }
        }
        return res.next;
    }
}
