class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> al = new ArrayList<>();
        for (ListNode i : param1) {
            ListNode temp = i;
            while (temp != null) {
                al.add(temp.val);
                temp = temp.next;
            }
        }
        Collections.sort(al);
        ListNode ans = new ListNode(0);
        ListNode temp = ans;
        for (int i : al) {
            ListNode curr = new ListNode(i);
            temp.next = curr;
            temp = temp.next;
        }
        return ans.next;
    }
}
