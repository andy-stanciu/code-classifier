class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (ListNode head : param1) {
            while (head != null) {
                pq.add(head.val);
                head = head.next;
            }
        }
        ListNode result = new ListNode();
        ListNode dummy = result;
        while (!pq.isEmpty()) {
            int value = pq.poll();
            ListNode newnode = new ListNode(value);
            dummy.next = newnode;
            dummy = dummy.next;
        }
        return result.next;
    }
}
