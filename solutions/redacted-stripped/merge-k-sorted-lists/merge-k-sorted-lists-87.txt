class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((o1, o2) -> o1.val - o2.val);
        for (ListNode list : param1) {
            if (list == null) {
                continue;
            }
            pq.add(list);
        }
        ListNode ans = new ListNode(0);
        ListNode dummy = ans;
        while (!pq.isEmpty()) {
            ListNode curr = pq.poll();
            dummy.next = curr;
            if (curr.next != null) {
                pq.add(curr.next);
            }
            dummy = dummy.next;
        }
        return ans.next;
    }
}
