class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> pq = new PriorityQueue();
        for (// traverse in all lists
        ListNode l : param1) {
            while (l != null) {
                pq.add(l.val);
                l = l.next;
            }
        }
        ListNode list = new ListNode(-1);
        ListNode ans = list;
        while (!pq.isEmpty()) {
            ListNode p = new ListNode(pq.poll());
            list.next = p;
            list = list.next;
        }
        return ans.next;
    }
}
