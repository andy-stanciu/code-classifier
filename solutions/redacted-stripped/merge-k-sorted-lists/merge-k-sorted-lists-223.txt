class Solution {

    public ListNode method1(ListNode first, ListNode second) {
        ListNode dummy = new ListNode(-1);
        ListNode temp = dummy;
        while (first != null && second != null) {
            if (first.val <= second.val) {
                temp.next = first;
                temp = temp.next;
                first = first.next;
            } else {
                temp.next = second;
                temp = temp.next;
                second = second.next;
            }
        }
        if (first != null) {
            temp.next = first;
        } else {
            temp.next = second;
        }
        return dummy.next;
    }

    public ListNode devide(ListNode[] list, int l, int h) {
        if (l > h) {
            return null;
        }
        if (l == h) {
            return list[h];
        }
        int mid = (l + h) / 2;
        ListNode first = devide(list, l, mid);
        ListNode second = devide(list, mid + 1, h);
        ListNode finalList = method1(first, second);
        return finalList;
    }

    public ListNode method2(ListNode[] param1) {
        return devide(param1, 0, param1.length - 1);
    }
}
