class Solution {

    public ListNode method1(ListNode[] param1) {
        int n = param1.length;
        ListNode dummy = new ListNode(0);
        ListNode cur = dummy;
        ArrayList<Integer> l = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            ListNode temp = param1[i];
            while (temp != null) {
                l.add(temp.val);
                temp = temp.next;
            }
        }
        Collections.sort(l);
        for (int i : l) {
            cur.next = new ListNode(i);
            cur = cur.next;
        }
        return dummy.next;
    }
}
