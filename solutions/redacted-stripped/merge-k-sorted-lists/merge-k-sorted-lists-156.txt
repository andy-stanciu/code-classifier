class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> minHeap = new PriorityQueue<ListNode>((a, b) -> {
            return Integer.compare(a.val, b.val);
        });
        ListNode var1 = null;
        ListNode var2 = null;
        for (ListNode node : param1) {
            if (node != null) {
                minHeap.add(node);
            }
        }
        while (!minHeap.isEmpty()) {
            ListNode temp = minHeap.poll();
            if (temp.next != null) {
                minHeap.add(temp.next);
            }
            if (var1 == null) {
                temp.next = null;
                var1 = temp;
                var2 = temp;
            } else {
                temp.next = null;
                var2.next = temp;
                var2 = var2.next;
            }
        }
        return var1;
    }
}
