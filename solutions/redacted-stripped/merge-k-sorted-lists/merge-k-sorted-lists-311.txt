class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> list = new ArrayList<>();
        for (ListNode head : param1) {
            while (head != null) {
                list.add(head.val);
                head = head.next;
            }
        }
        ListNode res = new ListNode();
        Collections.sort(list);
        ListNode temp = res;
        for (int ele : list) {
            ListNode node = new ListNode(ele);
            temp.next = node;
            temp = temp.next;
        }
        return res.next;
    }
}
