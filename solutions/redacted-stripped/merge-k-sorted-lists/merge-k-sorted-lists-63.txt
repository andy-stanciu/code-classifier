class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (ListNode a : param1) {
            while (a != null) {
                pq.add(a.val);
                a = a.next;
            }
        }
        ListNode ans = new ListNode();
        if (pq.size() == 0) {
            return null;
        } else {
            ans.val = pq.poll();
            ListNode temp = ans;
            while (pq.size() != 0) {
                temp.next = new ListNode(pq.poll());
                temp = temp.next;
            }
            return ans;
        }
    }
}
