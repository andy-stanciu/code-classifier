class Solution {

    public ListNode method1(ListNode[] param1) {
        Vector<Integer> vector = new Vector<>();
        for (int i = 0; i < param1.length; i++) {
            ListNode temp = param1[i];
            while (temp != null) {
                vector.add(temp.val);
                temp = temp.next;
            }
        }
        vector.sort(null);
        ListNode temp = new ListNode(0);
        ListNode head = temp;
        for (int i : vector) {
            temp.next = new ListNode(i);
            temp = temp.next;
        }
        return head.next;
    }
}
