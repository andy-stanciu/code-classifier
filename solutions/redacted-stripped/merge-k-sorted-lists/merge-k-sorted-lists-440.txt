class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0) {
            return null;
        }
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> (a.val - b.val));
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] != null) {
                pq.offer(param1[i]);
            }
        }
        ListNode dummy = new ListNode(-1);
        ListNode temp = dummy;
        while (!pq.isEmpty()) {
            ListNode curr = pq.poll();
            temp.next = curr;
            if (curr.next != null) {
                pq.offer(curr.next);
            }
            temp = temp.next;
        }
        return dummy.next;
    }
}
