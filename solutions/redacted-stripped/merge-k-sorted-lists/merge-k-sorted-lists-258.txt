class Solution {

    public ListNode method1(ListNode[] param1) {
        int[] nums = new int[10000 * 2 + 1];
        for (ListNode cur : param1) {
            while (cur != null) {
                nums[cur.val + 10000]++;
                cur = cur.next;
            }
        }
        ListNode head = new ListNode();
        ListNode cur = head;
        for (int index = 0; index < nums.length; index++) {
            for (int k = 0; k < nums[index]; k++) {
                cur.next = new ListNode(index - 10000);
                cur = cur.next;
            }
        }
        return head.next;
    }
}
