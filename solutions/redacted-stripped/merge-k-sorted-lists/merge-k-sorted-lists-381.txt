class Solution {

    public ListNode method1(ListNode[] param1) {
        ArrayList<Integer> al = new ArrayList<>();
        for (ListNode start : param1) {
            while (start != null) {
                al.add(start.val);
                start = start.next;
            }
        }
        System.out.println(al);
        Collections.sort(al);
        ListNode root = new ListNode();
        ListNode start = root;
        for (int i = 0; i < al.size(); i++) {
            root.next = new ListNode(al.get(i));
            root = root.next;
        }
        return start.next;
    }
}
