class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0)
            return null;
        PriorityQueue<ListNode> pq = new PriorityQueue<>((o1, o2) -> o1.val - o2.val);
        for (ListNode l : param1) if (l != null)
            pq.add(l);
        ListNode dummy = new ListNode();
        ListNode prev = dummy;
        while (!pq.isEmpty()) {
            ListNode pop = pq.poll();
            dummy.next = pop;
            if (pop.next != null)
                pq.add(pop.next);
            dummy = pop;
            pop.next = null;
        }
        return prev.next;
    }
}
