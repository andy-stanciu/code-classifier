class Solution {

    public ListNode method1(ListNode[] arr) {
        if (arr.length == 0 || arr == null)
            return null;
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode i : arr) {
            if (i != null)
                pq.add(i);
        }
        ListNode dummy = new ListNode(-1);
        ListNode temp = dummy;
        while (!pq.isEmpty()) {
            ListNode small = pq.peek();
            temp.next = small;
            temp = temp.next;
            pq.poll();
            if (small.next != null)
                pq.add(small.next);
        }
        return dummy.next;
    }
}
