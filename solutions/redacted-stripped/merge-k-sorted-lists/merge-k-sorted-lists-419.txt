class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> {
            return a.val - b.val;
        });
        for (ListNode node : param1) {
            while (node != null) {
                pq.add(node);
                node = node.next;
            }
        }
        ListNode head = new ListNode(-1);
        ListNode curr = head;
        while (!pq.isEmpty()) {
            curr.next = pq.poll();
            curr = curr.next;
            curr.next = null;
        }
        return head.next;
    }
}
