class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode ans = null;
        for (ListNode list : param1) {
            if (ans == null) {
                ans = list;
                continue;
            }
            ListNode temp = new ListNode(0);
            ListNode p = temp;
            while (list != null && ans != null) {
                if (list.val < ans.val) {
                    temp.next = new ListNode(list.val);
                    list = list.next;
                } else {
                    temp.next = new ListNode(ans.val);
                    ans = ans.next;
                }
                temp = temp.next;
            }
            if (list != null) {
                temp.next = list;
            } else if (ans != null) {
                temp.next = ans;
            }
            ans = p.next;
        }
        return ans;
    }
}
