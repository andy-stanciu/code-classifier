class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode itr = null;
        if (param1 == null)
            return itr;
        List<Integer> ls = new ArrayList<Integer>();
        for (ListNode list : param1) {
            while (list != null) {
                ls.add(list.val);
                list = list.next;
            }
        }
        Collections.sort(ls);
        if (ls != null && ls.size() > 0) {
            itr = new ListNode(ls.get(0));
        }
        ListNode result = itr;
        for (int i = 1; i < ls.size(); i++) {
            ListNode tempNode = new ListNode(ls.get(i));
            itr.next = tempNode;
            itr = itr.next;
        }
        return result;
    }
}
