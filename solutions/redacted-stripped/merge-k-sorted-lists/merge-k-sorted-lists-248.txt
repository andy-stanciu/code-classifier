class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (ListNode l : param1) {
            while (l != null) {
                pq.add(l.val);
                l = l.next;
            }
        }
        if (pq.isEmpty())
            return null;
        ListNode head = new ListNode(pq.poll());
        ListNode temp = head;
        while (!pq.isEmpty()) {
            temp.next = new ListNode(pq.poll());
            temp = temp.next;
        }
        return head;
    }
}
