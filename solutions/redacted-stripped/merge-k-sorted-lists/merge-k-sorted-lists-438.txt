class Solution {

    public ListNode method1(ListNode[] param1) {
        int len = param1.length;
        if (len == 0)
            return null;
        ListNode head = param1[0];
        for (int i = 1; i < len; i++) {
            head = method2(head, param1[i]);
        }
        return head;
    }

    public static ListNode method2(ListNode l1, ListNode l2) {
        if (l1 == null)
            return l2;
        if (l2 == null)
            return l1;
        if (l1.val < l2.val) {
            l1.next = method2(l1.next, l2);
            return l1;
        } else {
            l2.next = method2(l1, l2.next);
            return l2;
        }
    }
}
