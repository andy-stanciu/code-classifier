class Solution {

    public ListNode method1(ListNode[] param1) {
        int a = param1.length;
        if (a == 0) {
            return null;
        }
        ArrayList<Integer> arr = new ArrayList<>();
        for (ListNode b : param1) {
            ListNode curr = b;
            while (curr != null) {
                arr.add(curr.val);
                curr = curr.next;
            }
        }
        Collections.sort(arr);
        if (arr.size() == 0) {
            return null;
        }
        ListNode res = new ListNode(arr.get(0));
        ListNode pres = res;
        for (int i = 1; i < arr.size(); i++) {
            pres.next = new ListNode(arr.get(i));
            pres = pres.next;
        }
        return res;
    }
}
