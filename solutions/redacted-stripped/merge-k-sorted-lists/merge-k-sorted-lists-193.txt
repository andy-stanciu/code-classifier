class Solution {

    public ListNode method1(ListNode[] param1) {
        Queue<ListNode> q = new PriorityQueue<ListNode>((a, b) -> a.val - b.val);
        for (ListNode l : param1) {
            if (l != null) {
                q.add(l);
            }
        }
        ListNode head = new ListNode(-1);
        ListNode point = head;
        while (!q.isEmpty()) {
            point.next = q.poll();
            point = point.next;
            if (point.next != null) {
                q.add(point.next);
            }
        }
        return head.next;
    }
}
