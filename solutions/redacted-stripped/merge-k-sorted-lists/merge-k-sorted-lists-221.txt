class Solution {

    ListNode node = null;

    public ListNode method1(ListNode[] param1) {
        int ind = this.findTinyNode(param1);
        if (ind < 0)
            return null;
        ListNode node = param1[ind];
        param1[ind] = param1[ind].next;
        node.next = this.method1(param1);
        return node;
    }

    public int findTinyNode(ListNode[] param1) {
        int index = -1;
        ListNode temp = null;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] == null)
                continue;
            if (temp == null || temp.val > param1[i].val) {
                temp = param1[i];
                index = i;
            }
        }
        return index;
    }
}
