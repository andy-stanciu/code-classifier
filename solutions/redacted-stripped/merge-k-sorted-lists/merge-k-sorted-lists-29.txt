class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> x = new ArrayList<>();
        for (int i = 0; i < param1.length; i++) {
            ListNode temp = param1[i];
            while (temp != null) {
                x.add(temp.val);
                temp = temp.next;
            }
        }
        Collections.sort(x);
        ListNode f = new ListNode(0);
        ListNode t2 = f;
        for (int i : x) {
            ListNode m = new ListNode(i);
            t2.next = m;
            t2 = t2.next;
        }
        return f.next;
    }
}
