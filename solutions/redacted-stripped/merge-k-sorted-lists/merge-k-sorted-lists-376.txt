class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1 == null) {
            return null;
        }
        if (param1.length == 1) {
            return param1[0];
        }
        PriorityQueue<ListNode> queue = new PriorityQueue<ListNode>((node1, node2) -> Integer.compare(node1.val, node2.val));
        for (ListNode node : param1) {
            while (node != null) {
                queue.offer(node);
                node = node.next;
            }
        }
        ListNode var1 = queue.poll();
        ListNode temp = var1;
        while (!queue.isEmpty()) {
            ListNode listNode = queue.poll();
            temp.next = new ListNode(listNode.val);
            temp = temp.next;
        }
        return var1;
    }
}
