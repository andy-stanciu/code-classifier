class Solution {

    public ListNode method1(ListNode[] param1) {
        List<Integer> ls = new ArrayList<>();
        int j = 0;
        for (ListNode i : param1) {
            while (i != null) {
                ls.add(i.val);
                i = i.next;
            }
        }
        Collections.sort(ls);
        ListNode ans = new ListNode();
        ListNode temp = ans;
        for (int i : ls) {
            temp.next = new ListNode(i);
            temp = temp.next;
        }
        return ans.next;
    }
}
