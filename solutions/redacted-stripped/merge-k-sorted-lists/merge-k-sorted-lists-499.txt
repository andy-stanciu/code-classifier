class Solution {

    public ListNode method1(ListNode[] param1) {
        int k = param1.length;
        ListNode head = null;
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {

            public int compare(Integer i1, Integer i2) {
                return i2 - i1;
            }
        });
        int idx, min;
        for (int i = 0; i < k; i++) while (param1[i] != null) {
            pq.add(param1[i].val);
            param1[i] = param1[i].next;
        }
        while (!pq.isEmpty()) {
            head = new ListNode(pq.poll(), head);
        }
        return head;
    }
}
