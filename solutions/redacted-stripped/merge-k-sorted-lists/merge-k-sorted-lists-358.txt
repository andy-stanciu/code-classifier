class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1 == null || param1.length == 0) {
            return null;
        }
        int step = 1;
        while (step < param1.length) {
            for (int i = 0; i < param1.length - step; i += step * 2) {
                param1[i] = method2(param1[i], param1[i + step]);
            }
            step *= 2;
        }
        return param1[0];
    }

    private ListNode method2(ListNode head1, ListNode head2) {
        ListNode dummy = new ListNode(-1);
        ListNode prev = dummy;
        while (head1 != null && head2 != null) {
            if (head1.val <= head2.val) {
                prev.next = head1;
                head1 = head1.next;
            } else {
                prev.next = head2;
                head2 = head2.next;
            }
            prev = prev.next;
        }
        if (head1 == null) {
            prev.next = head2;
        } else {
            prev.next = head1;
        }
        return dummy.next;
    }
}
