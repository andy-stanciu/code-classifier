class Solution {

    public ListNode method1(ListNode[] param1) {
        int min = Integer.MAX_VALUE;
        int idx = -1;
        for (int i = 0; i < param1.length; i++) {
            ListNode list = param1[i];
            if (list != null && list.val < min) {
                min = list.val;
                idx = i;
            }
        }
        if (idx == -1)
            return null;
        ListNode head = param1[idx];
        param1[idx] = param1[idx].next;
        head.next = method1(param1);
        return head;
    }
}
