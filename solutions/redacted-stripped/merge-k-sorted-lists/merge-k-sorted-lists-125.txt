class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode node : param1) if (node != null)
            pq.add(node);
        ListNode dummy = new ListNode(-1);
        ListNode curr = dummy;
        while (!pq.isEmpty()) {
            ListNode node = pq.poll();
            curr.next = node;
            if (node.next != null)
                pq.add(node.next);
            curr = curr.next;
        }
        return dummy.next;
    }
}
