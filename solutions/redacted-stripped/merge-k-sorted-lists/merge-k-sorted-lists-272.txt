class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode dummy = new ListNode();
        ListNode curr = dummy;
        PriorityQueue<ListNode> pq = new PriorityQueue<>(Comparator.comparingInt(a -> a.val));
        for (ListNode node : param1) {
            if (node != null)
                pq.offer(node);
        }
        while (!pq.isEmpty()) {
            ListNode li = pq.poll();
            curr.next = li;
            li = li.next;
            if (li != null)
                pq.offer(li);
            curr = curr.next;
        }
        return dummy.next;
    }
}
