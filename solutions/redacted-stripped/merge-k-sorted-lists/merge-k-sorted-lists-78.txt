class Solution {

    public ListNode method1(ListNode[] param1) {
        ArrayList<Integer> arr = new ArrayList<>();
        for (ListNode l : param1) {
            while (l != null) {
                arr.add(l.val);
                l = l.next;
            }
        }
        Collections.sort(arr);
        ListNode ans = new ListNode();
        ListNode list = ans;
        for (int i : arr) {
            list.next = new ListNode(i);
            list = list.next;
        }
        return ans.next;
    }
}
