class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1 == null || param1.length == 0)
            return null;
        ListNode result = null;
        ListNode temp = null;
        while (true) {
            var min = Integer.MAX_VALUE;
            for (var head : param1) {
                if (head != null) {
                    if (head.val <= min) {
                        min = head.val;
                    }
                }
            }
            if (min == Integer.MAX_VALUE) {
                break;
            }
            if (result == null) {
                result = new ListNode(min);
                temp = result;
            } else {
                temp.next = new ListNode(min);
                temp = temp.next;
            }
            for (int i = 0; i < param1.length; i++) {
                if (param1[i] != null) {
                    if (param1[i].val == min) {
                        param1[i] = param1[i].next;
                        break;
                    }
                }
            }
        }
        return result;
    }
}
