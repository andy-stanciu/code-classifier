class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> q = new PriorityQueue<>();
        for (int i = 0; i < param1.length; i++) {
            ListNode h = param1[i];
            while (h != null) {
                q.add(h.val);
                h = h.next;
            }
        }
        ListNode ans = new ListNode(-1);
        ListNode head = ans;
        while (!q.isEmpty()) {
            ListNode n = new ListNode(q.remove());
            ans.next = n;
            ans = n;
        }
        return head.next;
    }
}
