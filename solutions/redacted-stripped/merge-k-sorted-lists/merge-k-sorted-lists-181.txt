class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode temp = null;
        List<Integer> ans = new ArrayList<>();
        for (ListNode l1 : param1) {
            while (l1 != null) {
                ans.add(l1.val);
                l1 = l1.next;
            }
        }
        Collections.sort(ans, Collections.reverseOrder());
        for (int i : ans) {
            ListNode curr = new ListNode(i);
            curr.next = temp;
            temp = curr;
        }
        return temp;
    }
}
