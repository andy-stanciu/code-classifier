class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<ListNode> pq = new PriorityQueue<ListNode>((a, b) -> a.val - b.val);
        if (param1 == null || param1.length == 0)
            return null;
        for (ListNode x : param1) {
            if (x != null)
                pq.offer(x);
        }
        ListNode ans = new ListNode(-1);
        ListNode curr = ans;
        while (!pq.isEmpty()) {
            curr.next = pq.poll();
            curr = curr.next;
            if (curr.next != null)
                pq.offer(curr.next);
        }
        return ans.next;
    }
}
