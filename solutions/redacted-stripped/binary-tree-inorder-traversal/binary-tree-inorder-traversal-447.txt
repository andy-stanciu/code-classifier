class Solution {

    List<Integer> temp1 = new ArrayList<Integer>();

    void TravList(TreeNode temp) {
        if (temp != null) {
            TravList(temp.left);
            temp1.add(temp.val);
            TravList(temp.right);
        }
    }

    public List<Integer> method1(TreeNode root) {
        TravList(root);
        return temp1;
    }
}
