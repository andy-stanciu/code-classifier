class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans;
    }

    public void method2(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        method2(root.left, ans);
        ans.add(root.val);
        method2(root.right, ans);
    }
}
