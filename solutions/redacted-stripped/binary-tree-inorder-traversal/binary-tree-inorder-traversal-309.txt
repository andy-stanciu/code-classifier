class Solution {

    List<Integer> ans = new ArrayList<Integer>();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        ans.add(root.val);
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return ans;
    }
}
