class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        TreeNode temp = root;
        while (true) {
            if (temp != null) {
                st.push(temp);
                temp = temp.left;
            } else {
                if (st.isEmpty())
                    break;
                temp = st.pop();
                l.add(temp.val);
                temp = temp.right;
            }
        }
        return l;
    }
}
