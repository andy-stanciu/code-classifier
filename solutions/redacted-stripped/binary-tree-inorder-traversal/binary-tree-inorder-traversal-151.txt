class Solution {

    List<Integer> ans;

    public List<Integer> method1(TreeNode root) {
        ans = new ArrayList<>();
        if (root == null)
            return ans;
        helper(root);
        return ans;
    }

    private void helper(TreeNode temp) {
        if (temp == null)
            return;
        helper(temp.left);
        ans.add(temp.val);
        helper(temp.right);
    }
}
