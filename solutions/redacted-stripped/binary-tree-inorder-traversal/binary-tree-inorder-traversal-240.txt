class Solution {

    public List<Integer> solve(TreeNode root, List<Integer> l) {
        if (root == null) {
            return l;
        }
        l = solve(root.left, l);
        l.add(root.val);
        l = solve(root.right, l);
        return l;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        l.clear();
        l = solve(root, l);
        return l;
    }
}
