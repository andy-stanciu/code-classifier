class Solution {

    public void method1(TreeNode root, ArrayList<Integer> ans) {
        if (root == null) {
            return;
        }
        method1(root.left, ans);
        ans.add(root.val);
        method1(root.right, ans);
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<>();
        method1(root, ans);
        return ans;
    }
}
