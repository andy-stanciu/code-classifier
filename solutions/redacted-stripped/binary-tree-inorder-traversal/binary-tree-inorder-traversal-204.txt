class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    public static List<Integer> method2(TreeNode node, ArrayList<Integer> list) {
        if (node == null) {
            return list;
        }
        method2(node.left, list);
        list.add(node.val);
        method2(node.right, list);
        return list;
    }
}
