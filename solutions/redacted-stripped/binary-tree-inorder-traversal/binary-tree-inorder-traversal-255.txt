class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> b = new ArrayList<>();
        while (root != null) {
            List<Integer> a = new ArrayList<>();
            a.addAll(method1(root.left));
            a.add(root.val);
            a.addAll(method1(root.right));
            return a;
        }
        return b;
    }
}
