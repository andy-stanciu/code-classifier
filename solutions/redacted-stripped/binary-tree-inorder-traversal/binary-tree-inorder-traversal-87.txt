class Solution {

    public void method1(TreeNode node, List<Integer> inor) {
        if (node == null)
            return;
        method1(node.left, inor);
        inor.add(node.val);
        method1(node.right, inor);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> inor = new ArrayList<>();
        method1(root, inor);
        return inor;
    }
}
