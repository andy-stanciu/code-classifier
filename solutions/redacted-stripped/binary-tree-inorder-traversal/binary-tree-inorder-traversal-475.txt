class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ll = new ArrayList<>();
        helper(root, ll);
        return ll;
    }

    public static void helper(TreeNode root, ArrayList<Integer> ll) {
        if (root == null) {
            return;
        }
        helper(root.left, ll);
        ll.add(root.val);
        helper(root.right, ll);
    }
}
