class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<>();
        List<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        TreeNode temp = root;
        while (true) {
            if (temp != null) {
                st.push(temp);
                temp = temp.left;
            } else {
                if (st.isEmpty())
                    break;
                temp = st.pop();
                ans.add(temp.val);
                temp = temp.right;
            }
        }
        return ans;
    }
}
