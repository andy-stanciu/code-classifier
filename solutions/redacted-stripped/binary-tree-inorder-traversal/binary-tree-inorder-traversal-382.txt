class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> l = new ArrayList<>();
        return method2(root, l);
    }

    public ArrayList<Integer> method2(TreeNode root, ArrayList<Integer> l) {
        if (root != null) {
            method2(root.left, l);
            l.add(root.val);
            method2(root.right, l);
        }
        return l;
    }
}
