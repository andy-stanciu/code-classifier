class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<Integer>();
        if (root == null)
            return res;
        res = method2(root, res);
        return res;
    }

    public List<Integer> method2(TreeNode temp, List<Integer> inlist) {
        if (temp.left != null)
            method2(temp.left, inlist);
        inlist.add(temp.val);
        if (temp.right != null)
            method2(temp.right, inlist);
        return inlist;
    }
}
