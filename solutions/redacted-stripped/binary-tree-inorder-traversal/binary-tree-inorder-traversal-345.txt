class Solution {

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    public void method2(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        method2(node.left, list);
        list.add(node.val);
        method2(node.right, list);
    }
}
