class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        helper(root, ans);
        return ans;
    }

    public TreeNode helper(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return null;
        }
        root.left = helper(root.left, ans);
        ans.add(root.val);
        root.right = helper(root.right, ans);
        return root;
    }
}
