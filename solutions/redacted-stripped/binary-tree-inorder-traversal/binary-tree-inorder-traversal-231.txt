class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        if (root == null)
            return li;
        method2(li, root);
        return li;
    }

    public void method2(List<Integer> li, TreeNode root) {
        if (root.left != null)
            method2(li, root.left);
        li.add(root.val);
        if (root.right != null)
            method2(li, root.right);
    }
}
