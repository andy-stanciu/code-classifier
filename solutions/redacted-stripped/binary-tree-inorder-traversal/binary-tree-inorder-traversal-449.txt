class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList();
        helper(res, root);
        return res;
    }

    void helper(List<Integer> res, TreeNode node) {
        if (node == null) {
            return;
        }
        helper(res, node.left);
        res.add(node.val);
        helper(res, node.right);
    }
}
