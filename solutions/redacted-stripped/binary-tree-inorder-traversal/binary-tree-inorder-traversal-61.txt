class Solution {

    public List<Integer> method1(TreeNode root) {
        LinkedList<Integer> lt = new LinkedList<Integer>();
        method2(root, lt);
        return lt;
    }

    public static void method2(TreeNode r, LinkedList<Integer> l) {
        if (r == null)
            return;
        method2(r.left, l);
        l.add(r.val);
        method2(r.right, l);
    }
}
