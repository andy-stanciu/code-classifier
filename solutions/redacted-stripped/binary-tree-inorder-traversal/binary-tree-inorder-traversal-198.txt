class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        morris(root, ans);
        return ans;
    }

    public void morris(TreeNode root, List<Integer> ans) {
        TreeNode curr = root;
        while (curr != null) {
            if (curr.left == null) {
                ans.add(curr.val);
                curr = curr.right;
            } else {
                TreeNode p = curr.left;
                while (p.right != null && p.right != curr) {
                    p = p.right;
                }
                if (p.right == null) {
                    p.right = curr;
                    curr = curr.left;
                } else {
                    p.right = null;
                    ans.add(curr.val);
                    curr = curr.right;
                }
            }
        }
    }
}
