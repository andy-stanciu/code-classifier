class Solution {

    static List<Integer> lst;

    public List<Integer> method1(TreeNode root) {
        lst = new ArrayList<>();
        traverse(root);
        return lst;
    }

    static void traverse(TreeNode root) {
        if (root == null) {
            return;
        }
        traverse(root.left);
        lst.add(root.val);
        traverse(root.right);
    }
}
