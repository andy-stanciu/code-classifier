class Solution {

    List<Integer> root_lst;

    public Solution() {
        this.root_lst = new ArrayList<>();
    }

    public List<Integer> method1(TreeNode root) {
        if (root != null) {
            method1(root.left);
            root_lst.add(root.val);
            method1(root.right);
        }
        return root_lst;
    }
}
