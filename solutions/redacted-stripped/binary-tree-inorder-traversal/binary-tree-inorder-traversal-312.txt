class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method2(root, res);
        return res;
    }

    public void method2(TreeNode root, List<Integer> res) {
        if (root != null) {
            method2(root.left, res);
            res.add(root.val);
            method2(root.right, res);
        }
    }
}
