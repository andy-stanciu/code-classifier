class Solution {

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return list;
        helper(root);
        return list;
    }

    public void helper(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        list.add(root.val);
        method1(root.right);
    }
}
