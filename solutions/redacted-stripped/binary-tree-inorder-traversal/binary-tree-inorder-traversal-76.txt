class Solution {

    List<Integer> list = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return list;
        TreeNode ptr = root;
        method2(ptr);
        return list;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        list.add(root.val);
        method2(root.right);
    }
}
