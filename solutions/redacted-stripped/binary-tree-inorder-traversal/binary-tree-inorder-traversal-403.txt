class Solution {

    List<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return ans;
    }

    public void method2(TreeNode root1) {
        if (root1 == null) {
            return;
        }
        method2(root1.left);
        ans.add(root1.val);
        method2(root1.right);
    }
}
