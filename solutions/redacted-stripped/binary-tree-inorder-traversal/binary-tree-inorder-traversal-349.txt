class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> a1 = new ArrayList<>();
        trav(root, a1);
        return a1;
    }

    public static void trav(TreeNode root, List<Integer> a1) {
        if (root == null)
            return;
        if (root != null) {
            trav(root.left, a1);
            a1.add(root.val);
            trav(root.right, a1);
        }
    }
}
