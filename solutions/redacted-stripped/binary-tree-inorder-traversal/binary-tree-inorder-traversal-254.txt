class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> var1 = new ArrayList<Integer>();
        method2(var1, root);
        return var1;
    }

    public void method2(List<Integer> var1, TreeNode node) {
        if (node != null) {
            if (node.left != null) {
                method2(var1, node.left);
            }
            var1.add(node.val);
            if (node.right != null) {
                method2(var1, node.right);
            }
        }
    }
}
