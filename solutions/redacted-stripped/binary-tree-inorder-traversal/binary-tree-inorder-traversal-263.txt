class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> var1 = new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        TreeNode node = root;
        while (true) {
            if (node != null) {
                st.push(node);
                node = node.left;
            } else {
                if (st.isEmpty()) {
                    break;
                }
                node = st.pop();
                var1.add(node.val);
                node = node.right;
            }
        }
        return var1;
    }
}
