class Solution {

    List<Integer> list;

    public void method1(TreeNode node) {
        if (node == null)
            return;
        if (node.left != null)
            method1(node.left);
        list.add(node.val);
        if (node.right != null)
            method1(node.right);
    }

    public List<Integer> method2(TreeNode root) {
        list = new ArrayList<>();
        method1(root);
        return list;
    }
}
