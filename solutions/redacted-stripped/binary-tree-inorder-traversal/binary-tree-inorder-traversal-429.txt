class Solution {

    public List<Integer> HelperIn(TreeNode root, List<Integer> ls) {
        if (root == null)
            return ls;
        HelperIn(root.left, ls);
        ls.add(root.val);
        HelperIn(root.right, ls);
        return ls;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList();
        return HelperIn(root, ls);
    }
}
