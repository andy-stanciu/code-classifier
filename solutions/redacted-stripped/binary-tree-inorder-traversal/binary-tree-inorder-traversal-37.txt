class Solution {

    private void method1(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        method1(node.left, list);
        list.add(node.val);
        method1(node.right, list);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method1(root, list);
        return list;
    }
}
