class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        generateArray(root, res);
        return res;
    }

    void generateArray(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        generateArray(root.left, res);
        res.add(root.val);
        generateArray(root.right, res);
    }
}
