class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<Integer>();
        if (root != null) {
            List<Integer> left = method1(root.left);
            list.addAll(left);
            list.add(root.val);
            List<Integer> right = method1(root.right);
            list.addAll(right);
        }
        return list;
    }
}
