class Solution {

    public void method1(TreeNode root, ArrayList<Integer> ll) {
        if (root == null) {
            return;
        }
        method1(root.left, ll);
        ll.add(root.val);
        method1(root.right, ll);
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> ll = new ArrayList<>();
        method1(root, ll);
        return ll;
    }
}
