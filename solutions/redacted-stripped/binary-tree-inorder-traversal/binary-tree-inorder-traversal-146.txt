class Solution {

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return list;
        method2(root);
        return list;
    }

    void method2(TreeNode r) {
        if (r != null) {
            method2(r.left);
            list.add(r.val);
            method2(r.right);
        }
    }
}
