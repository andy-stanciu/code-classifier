class Solution {

    void fun(TreeNode root, List<Integer> li) {
        if (root != null) {
            fun(root.left, li);
            li.add(root.val);
            fun(root.right, li);
        }
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        fun(root, li);
        return li;
    }
}
