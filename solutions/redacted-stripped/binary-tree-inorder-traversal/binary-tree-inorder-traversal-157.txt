class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> a = new ArrayList<>();
        Stack<TreeNode> s = new Stack<TreeNode>();
        if (root == null) {
            return a;
        }
        TreeNode temp = root;
        while (!s.isEmpty() || temp != null) {
            if (temp != null) {
                s.push(temp);
                temp = temp.left;
            } else {
                temp = s.pop();
                a.add(temp.val);
                temp = temp.right;
            }
        }
        return a;
    }
}
