class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> lst = new ArrayList<>();
        helper(root, lst);
        return lst;
    }

    public void helper(TreeNode root, List<Integer> lst) {
        if (root != null) {
            helper(root.left, lst);
            lst.add(root.val);
            helper(root.right, lst);
        }
    }
}
