class Solution {

    public List<Integer> method1(TreeNode root) {
        TreeNode node = root;
        List<Integer> list = new ArrayList<Integer>();
        list = method2(node, list);
        return list;
    }

    private List<Integer> method2(TreeNode node, List<Integer> list) {
        if (node == null) {
            return list;
        }
        list = method2(node.left, list);
        list.add(node.val);
        list = method2(node.right, list);
        return list;
    }
}
