class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null)
            return res;
        method2(root, res);
        return res;
    }

    private void method2(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        method2(root.left, res);
        res.add(root.val);
        method2(root.right, res);
    }
}
