class Solution {

    public static void method1(TreeNode root, List<Integer> l1) {
        if (root == null) {
            return;
        }
        method1(root.left, l1);
        l1.add(root.val);
        method1(root.right, l1);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> l1 = new ArrayList<>();
        method1(root, l1);
        return l1;
    }
}
