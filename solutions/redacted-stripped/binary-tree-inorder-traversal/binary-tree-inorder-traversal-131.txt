class Solution {

    List<Integer> res = new LinkedList();

    public List<Integer> method1(TreeNode root) {
        helper(root);
        return res;
    }

    private void helper(TreeNode root) {
        if (root != null) {
            res.add(root.val);
            helper(root.left);
            helper(root.right);
        }
    }
}

class Solution {

    List<Integer> res = new LinkedList();

    public List<Integer> method2(TreeNode root) {
        helper(root);
        return res;
    }

    private void helper(TreeNode root) {
        if (root != null) {
            helper(root.left);
            res.add(root.val);
            helper(root.right);
        }
    }
}

class Solution {

    List<Integer> res = new LinkedList();

    public List<Integer> method3(TreeNode root) {
        helper(root);
        return res;
    }

    private void helper(TreeNode root) {
        if (root != null) {
            helper(root.left);
            helper(root.right);
            res.add(root.val);
        }
    }
}
