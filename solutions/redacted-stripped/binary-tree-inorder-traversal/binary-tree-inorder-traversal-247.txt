class Solution {

    List<Integer> list;

    public List<Integer> method1(TreeNode root) {
        list = new ArrayList<>();
        find(root);
        return list;
    }

    public void find(TreeNode root) {
        if (root == null)
            return;
        find(root.left);
        list.add(root.val);
        find(root.right);
    }
}
