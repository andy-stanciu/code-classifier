class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new LinkedList<>();
        if (root == null) {
            return output;
        } else {
            method1(root, output);
        }
        return output;
    }

    public void method1(TreeNode root, List<Integer> output) {
        if (root.left != null) {
            method1(root.left, output);
        }
        output.add(root.val);
        if (root.right != null) {
            method1(root.right, output);
        }
    }
}
