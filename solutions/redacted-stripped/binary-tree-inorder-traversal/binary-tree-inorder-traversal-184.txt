class Solution {

    List<Integer> li = new ArrayList();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return li;
        if (root.left != null)
            method1(root.left);
        li.add(root.val);
        if (root.right != null)
            method1(root.right);
        return li;
    }
}
