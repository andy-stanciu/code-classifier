class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        Stack<TreeNode> s = new Stack<>();
        if (root == null) {
            return ls;
        }
        TreeNode cur = root;
        while (cur != null || !s.isEmpty()) {
            while (cur != null) {
                s.push(cur);
                cur = cur.left;
            }
            cur = s.pop();
            ls.add(cur.val);
            cur = cur.right;
        }
        return ls;
    }
}
