class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        if (root == null) {
            return ls;
        }
        Stack<TreeNode> s = new Stack<>();
        TreeNode temp = root;
        while (true) {
            if (temp != null) {
                s.push(temp);
                temp = temp.left;
            } else {
                if (s.isEmpty())
                    break;
                temp = s.pop();
                ls.add(temp.val);
                temp = temp.right;
            }
        }
        return ls;
    }
}
