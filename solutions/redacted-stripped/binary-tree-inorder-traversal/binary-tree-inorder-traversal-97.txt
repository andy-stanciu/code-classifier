class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<>();
        ArrayList<Integer> ls = new ArrayList<>();
        TreeNode curr = root;
        while (true) {
            if (curr != null) {
                st.push(curr);
                curr = curr.left;
            } else {
                if (st.isEmpty())
                    break;
                curr = st.pop();
                ls.add(curr.val);
                curr = curr.right;
            }
        }
        return ls;
    }
}
