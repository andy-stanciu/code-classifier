class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> s = new Stack<>();
        List<Integer> ans = new ArrayList<>();
        TreeNode temp = root;
        while (temp != null || !s.isEmpty()) {
            while (temp != null) {
                s.push(temp);
                temp = temp.left;
            }
            temp = s.pop();
            ans.add(temp.val);
            temp = temp.right;
        }
        return ans;
    }
}
