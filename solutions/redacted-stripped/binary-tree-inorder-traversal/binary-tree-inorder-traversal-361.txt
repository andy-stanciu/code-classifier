class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        traverse(root, var1);
        return var1;
    }

    public static void traverse(TreeNode root, List<Integer> var1) {
        if (root == null)
            return;
        traverse(root.left, var1);
        var1.add(root.val);
        traverse(root.right, var1);
    }
}
