class Solution {

    List<Integer> inTraverse(TreeNode root, List<Integer> list) {
        if (root == null)
            return list;
        inTraverse(root.left, list);
        list.add(root.val);
        inTraverse(root.right, list);
        return list;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        list = inTraverse(root, list);
        return list;
    }
}
