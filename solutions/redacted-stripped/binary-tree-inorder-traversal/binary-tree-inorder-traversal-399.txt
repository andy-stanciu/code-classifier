class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        method2(root, li);
        return li;
    }

    public void method2(TreeNode root, List<Integer> li) {
        if (root == null) {
            return;
        }
        method2(root.left, li);
        li.add(root.val);
        method2(root.right, li);
    }
}
