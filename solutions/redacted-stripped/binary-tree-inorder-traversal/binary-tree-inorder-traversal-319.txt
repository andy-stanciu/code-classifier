class Solution {

    List<Integer> ans = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return ans;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        ans.add(root.val);
        method1(root.right);
    }
}
