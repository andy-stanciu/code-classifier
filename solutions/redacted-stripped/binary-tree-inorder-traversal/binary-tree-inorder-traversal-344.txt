class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> a = new ArrayList<>();
        method2(root, a);
        return a;
    }

    private void method2(TreeNode root, ArrayList<Integer> a) {
        if (root == null)
            return;
        method2(root.left, a);
        a.add(root.val);
        method2(root.right, a);
    }
}
