class Solution {

    public static List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    public static void method2(TreeNode root, List<Integer> result) {
        if (root == null)
            return;
        if (root.left != null) {
            method2(root.left, result);
        }
        result.add(root.val);
        if (root.right != null) {
            method2(root.right, result);
        }
    }
}
