class Solution {

    List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return result;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        result.add(root.val);
        method2(root.right);
    }
}
