class Solution {

    public List<Integer> method1(TreeNode root) {
        List var1 = new ArrayList<Integer>();
        traverse(var1, root);
        return var1;
    }

    public void traverse(List<Integer> list, TreeNode root) {
        if (root == null)
            return;
        traverse(list, root.left);
        list.add(root.val);
        traverse(list, root.right);
    }
}
