class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        Stack<TreeNode> stk = new Stack<>();
        TreeNode curr = root;
        while (curr != null || !stk.isEmpty()) {
            while (curr != null) {
                stk.push(curr);
                curr = curr.left;
            }
            curr = stk.pop();
            res.add(curr.val);
            curr = curr.right;
        }
        return res;
    }

    public void method2(TreeNode curr, List<Integer> list) {
        if (curr == null)
            return;
        method2(curr.left, list);
        list.add(curr.val);
        method2(curr.right, list);
    }
}
