class Solution {

    List<Integer> res = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return res;
        if (root.left != null)
            method1(root.left);
        res.add(root.val);
        if (root.right != null)
            method1(root.right);
        return res;
    }
}
