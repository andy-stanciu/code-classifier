class Solution {

    public List<Integer> solve(TreeNode root, List<Integer> l) {
        if (root == null) {
            return l;
        }
        solve(root.left, l);
        l.add(root.val);
        solve(root.right, l);
        return l;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        return solve(root, l);
    }
}
