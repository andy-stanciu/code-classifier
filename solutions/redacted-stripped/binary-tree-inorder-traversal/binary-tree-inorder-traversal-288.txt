class Solution {

    public List<Integer> method1(TreeNode root) {
        final var list = new ArrayList<Integer>();
        inTra(root, list);
        return list;
    }

    void inTra(TreeNode node, ArrayList<Integer> list) {
        if (node == null)
            return;
        inTra(node.left, list);
        list.add(node.val);
        inTra(node.right, list);
    }
}
