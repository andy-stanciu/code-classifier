class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        list = addNodesInList(list, root);
        return list;
    }

    public List<Integer> addNodesInList(List<Integer> list, TreeNode root) {
        if (root == null) {
            return list;
        }
        addNodesInList(list, root.left);
        list.add(root.val);
        addNodesInList(list, root.right);
        return list;
    }
}
