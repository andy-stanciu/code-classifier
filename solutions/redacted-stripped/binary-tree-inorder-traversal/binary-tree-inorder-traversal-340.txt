class Solution {

    ArrayList<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> s = new Stack<>();
        ArrayList<Integer> ans = new ArrayList<>();
        TreeNode temp = root;
        while (temp != null || s.size() > 0) {
            while (temp != null) {
                s.add(temp);
                temp = temp.left;
            }
            temp = s.pop();
            ans.add(temp.val);
            temp = temp.right;
        }
        return ans;
    }
}
