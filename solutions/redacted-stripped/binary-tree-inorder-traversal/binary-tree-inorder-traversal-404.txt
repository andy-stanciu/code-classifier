class Solution {

    List<Integer> li = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        helper(root);
        return li;
    }

    void helper(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        li.add(root.val);
        method1(root.right);
        return;
    }
}
