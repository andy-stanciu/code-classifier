class Solution {

    public static List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        helpermethod(root, result);
        return result;
    }

    public static void helpermethod(TreeNode root, List<Integer> result) {
        if (root == null) {
            return;
        }
        helpermethod(root.left, result);
        result.add(root.val);
        helpermethod(root.right, result);
    }
}
