class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<Integer>();
        return method2(root, list);
    }

    public List<Integer> method2(TreeNode root, List<Integer> list) {
        if (root != null) {
            method2(root.left, list);
            list.add(root.val);
            method2(root.right, list);
        }
        return list;
    }
}
