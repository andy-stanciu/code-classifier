class Solution {

    List<Integer> li = new LinkedList<Integer>();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            List<Integer> li = new LinkedList<Integer>();
            return li;
        }
        method1(root.left);
        li.add(root.val);
        method1(root.right);
        return li;
    }
}
