class Solution {

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return var1;
    }

    private List<Integer> var1 = new ArrayList<>();

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        var1.add(root.val);
        method2(root.right);
    }
}
