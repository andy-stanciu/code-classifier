class Solution {

    private List<Integer> li = new ArrayList<Integer>();

    private void travers(TreeNode root) {
        if (root == null) {
            return;
        }
        travers(root.left);
        li.add(root.val);
        travers(root.right);
    }

    public List<Integer> method1(TreeNode root) {
        travers(root);
        return li;
    }
}
