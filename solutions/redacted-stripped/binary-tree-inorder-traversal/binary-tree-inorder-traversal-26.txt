class Solution {

    LinkedList<Integer> res = new LinkedList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        System.out.println(res);
        return res;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        method2(root.left);
        res.add(root.val);
        method2(root.right);
        return 1;
    }
}
