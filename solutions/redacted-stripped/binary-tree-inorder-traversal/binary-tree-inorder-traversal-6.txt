class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList();
        if (root == null)
            return ls;
        if (root.left == null) {
            ls.add(root.val);
            ls.addAll(method1(root.right));
            ls.addAll(method1(root.left));
            return ls;
        }
        if (root.right == null) {
            ls.addAll(method1(root.right));
            ls.addAll(method1(root.left));
            ls.add(root.val);
            return ls;
        }
        ls.addAll(method1(root.left));
        ls.add(root.val);
        ls.addAll(method1(root.right));
        return ls;
    }
}
