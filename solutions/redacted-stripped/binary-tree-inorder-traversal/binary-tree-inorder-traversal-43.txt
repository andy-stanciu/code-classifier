class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> var1 = new ArrayList<>();
        result(root, var1);
        return var1;
    }

    void result(TreeNode root, ArrayList<Integer> var1) {
        if (root == null) {
            return;
        }
        result(root.left, var1);
        var1.add(root.val);
        result(root.right, var1);
    }
}
