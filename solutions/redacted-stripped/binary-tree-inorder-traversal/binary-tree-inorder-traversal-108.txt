class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<>();
        List<Integer> list = new ArrayList<>();
        TreeNode temp = root;
        while (true) {
            if (temp != null) {
                st.push(temp);
                temp = temp.left;
            } else {
                if (st.size() == 0) {
                    break;
                }
                temp = st.pop();
                list.add(temp.val);
                temp = temp.right;
            }
        }
        return list;
    }
}
