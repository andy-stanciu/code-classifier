class Solution {

    public void helper(TreeNode root, List<Integer> lt) {
        if (root == null)
            return;
        helper(root.left, lt);
        lt.add(root.val);
        helper(root.right, lt);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> lt = new ArrayList<>();
        helper(root, lt);
        return lt;
    }
}
