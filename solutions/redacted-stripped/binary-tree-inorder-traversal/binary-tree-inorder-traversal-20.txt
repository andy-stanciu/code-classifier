class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int val) {
        this.val = val;
    }
}

public class Solution {

    public boolean method1(Class1 p, Class1 q) {
        return method2(q).equals(method2(p));
    }

    public List<Integer> method2(Class1 root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        result.addAll(method2(root.left));
        result.add(root.val);
        result.addAll(method2(root.right));
        return result;
    }
}
