class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        help(root, list);
        return list;
    }

    public void help(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        help(root.left, list);
        list.add(root.val);
        help(root.right, list);
    }
}
