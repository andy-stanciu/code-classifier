class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> s = new Stack<>();
        List<Integer> l = new LinkedList<>();
        while (root != null || !s.empty()) {
            while (root != null) {
                s.push(root);
                root = root.left;
            }
            TreeNode temp = s.pop();
            l.add(temp.val);
            root = temp.right;
        }
        return l;
    }
}
