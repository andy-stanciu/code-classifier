class Solution {

    List<Integer> result;

    public List<Integer> method1(TreeNode root) {
        result = new ArrayList<Integer>();
        method2(root);
        return result;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        result.add(root.val);
        method2(root.right);
    }
}
