class Solution {

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return List.of();
        ArrayList<Integer> var1 = new ArrayList<>();
        method1(root, var1);
        return var1;
    }

    private void method1(TreeNode node, List<Integer> var1) {
        if (node.left != null)
            method1(node.left, var1);
        var1.add(node.val);
        if (node.right != null)
            method1(node.right, var1);
    }
}
