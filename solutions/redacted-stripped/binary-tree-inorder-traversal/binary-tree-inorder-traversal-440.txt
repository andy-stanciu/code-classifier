class Solution {

    List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root != null)
            in(root, result);
        return result;
    }

    void in(TreeNode root, List<Integer> result) {
        if (root == null)
            return;
        in(root.left, result);
        result.add(root.val);
        in(root.right, result);
    }
}
