class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> arr = new ArrayList<>();
        method2(arr, root);
        return arr;
    }

    public void method2(List<Integer> arr, TreeNode root) {
        if (root != null) {
            method2(arr, root.left);
            arr.add(root.val);
            method2(arr, root.right);
        }
    }
}
