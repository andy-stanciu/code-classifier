class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        helpfun(root, ans);
        return ans;
    }

    void helpfun(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        helpfun(root.left, ans);
        ans.add(root.val);
        helpfun(root.right, ans);
    }
}
