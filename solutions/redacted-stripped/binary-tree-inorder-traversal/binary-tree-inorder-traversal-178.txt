class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        Stack<TreeNode> st = new Stack<>();
        TreeNode node = root;
        while (!st.isEmpty() || node != null) {
            if (node != null) {
                st.push(node);
                node = node.left;
            } else {
                node = st.pop();
                ans.add(node.val);
                node = node.right;
            }
        }
        return ans;
    }
}
