class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null)
            return res;
        Stack<TreeNode> st = new Stack<>();
        st.push(root);
        TreeNode node = root;
        while (!st.isEmpty()) {
            while (node.left != null) {
                node = node.left;
                st.push(node);
            }
            TreeNode temp = st.pop();
            res.add(temp.val);
            if (temp.right != null) {
                node = temp.right;
                st.push(temp.right);
            }
        }
        return res;
    }
}
