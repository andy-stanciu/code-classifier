class Solution {

    public void method1(List<Integer> list, TreeNode root) {
        if (root != null) {
            method1(list, root.left);
            list.add(root.val);
            method1(list, root.right);
        }
        return;
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method1(list, root);
        return list;
    }
}
