class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<Integer>();
        if (root == null)
            return result;
        if (root.left == null && root.right == null) {
            result.add(root.val);
            return result;
        }
        if (root.left != null) {
            result.addAll(method1(root.left));
        }
        result.add(root.val);
        if (root.right != null)
            result.addAll(method1(root.right));
        return result;
    }
}
