class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        return method2(root, result);
    }

    public List<Integer> method2(TreeNode root, List<Integer> result) {
        if (root.left != null) {
            method2(root.left, result);
        }
        result.add(root.val);
        if (root.right != null) {
            method2(root.right, result);
        }
        return result;
    }
}
