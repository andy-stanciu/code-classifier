class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> stack = new Stack<>();
        ArrayList<Integer> in = new ArrayList<Integer>();
        if (root == null)
            return new ArrayList<Integer>();
        TreeNode curr = root;
        while (curr != null) {
            while (curr != null) {
                stack.push(curr);
                curr = curr.left;
            }
            while (curr == null && !stack.empty()) {
                curr = stack.pop();
                in.add(curr.val);
                curr = curr.right;
            }
        }
        return in;
    }
}
