class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> array = new ArrayList<>();
        method2(root, array);
        return array;
    }

    private static void method2(TreeNode root, List<Integer> array) {
        if (root == null)
            return;
        method2(root.left, array);
        array.add(root.val);
        method2(root.right, array);
    }
}
