class Solution {

    public static List<Integer> io(TreeNode root, List<Integer> list) {
        if (root == null) {
            return list;
        }
        io(root.left, list);
        list.add(root.val);
        io(root.right, list);
        return list;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        return io(root, result);
    }
}
