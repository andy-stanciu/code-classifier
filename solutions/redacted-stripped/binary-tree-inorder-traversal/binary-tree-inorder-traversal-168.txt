class Solution {

    List<Integer> list;

    public void method1(TreeNode node) {
        if (node == null)
            return;
        if (node.left != null) {
            if (node.left.left == null && node.left.right == null) {
                list.add(node.left.val);
            } else {
                method1(node.left);
            }
        }
        list.add(node.val);
        method1(node.right);
    }

    public List<Integer> method2(TreeNode root) {
        list = new ArrayList<>();
        method1(root);
        return list;
    }
}
