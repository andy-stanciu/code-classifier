class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method2(root, res);
        return res;
    }

    static void method2(TreeNode node, List<Integer> res) {
        Stack<TreeNode> st = new Stack<>();
        while (node != null || !st.isEmpty()) {
            while (node != null) {
                st.push(node);
                node = node.left;
            }
            node = st.pop();
            res.add(node.val);
            node = node.right;
        }
    }
}
