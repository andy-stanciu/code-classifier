class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int x) {
        val = x;
    }
}

class Solution {

    public List<Integer> method1(Class1 root) {
        List<Integer> result = new ArrayList<>();
        traverse(root, result);
        return result;
    }

    private void traverse(Class1 root, List<Integer> result) {
        if (root != null) {
            traverse(root.left, result);
            result.add(root.val);
            traverse(root.right, result);
        }
    }
}
