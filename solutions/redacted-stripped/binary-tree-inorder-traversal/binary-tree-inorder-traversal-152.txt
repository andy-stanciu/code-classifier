class Solution {

    public void solve(TreeNode root, List<Integer> result) {
        if (root != null) {
            solve(root.left, result);
            result.add(root.val);
            solve(root.right, result);
        }
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        solve(root, result);
        return result;
    }
}
