class Solution {

    List<Integer> res = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        run(root);
        return res;
    }

    private void run(TreeNode root) {
        if (root == null) {
            return;
        }
        run(root.left);
        res.add(root.val);
        run(root.right);
    }
}
