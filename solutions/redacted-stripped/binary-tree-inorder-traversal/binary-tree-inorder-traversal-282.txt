class Solution {

    List<Integer> result = new ArrayList();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            return result;
        }
        method2(root);
        return result;
    }

    public void method2(TreeNode root) {
        if (root.left != null) {
            method2(root.left);
        }
        result.add(root.val);
        if (root.right != null) {
            method2(root.right);
        }
    }
}
