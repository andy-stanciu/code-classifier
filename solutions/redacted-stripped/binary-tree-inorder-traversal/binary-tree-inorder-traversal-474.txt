class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        helper(ans, root);
        return ans;
    }

    public void helper(List<Integer> ans, TreeNode root) {
        if (root == null) {
            return;
        }
        helper(ans, root.left);
        ans.add(root.val);
        helper(ans, root.right);
    }
}
