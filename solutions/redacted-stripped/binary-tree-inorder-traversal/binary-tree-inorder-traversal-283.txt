class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root != null) {
            method1(root, res);
        }
        return res;
    }

    public void method1(TreeNode root, List<Integer> res) {
        if (root.left != null) {
            method1(root.left, res);
        }
        res.add(root.val);
        if (root.right != null) {
            method1(root.right, res);
        }
    }
}
