class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        ans(list, root);
        return list;
    }

    public void ans(List<Integer> list, TreeNode root) {
        if (root == null) {
            return;
        }
        ans(list, root.left);
        list.add(root.val);
        ans(list, root.right);
    }
}
