class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> answer = new ArrayList<Integer>();
        Stack<TreeNode> st = new Stack<TreeNode>();
        TreeNode ptr = root;
        while (true) {
            if (ptr != null) {
                st.push(ptr);
                ptr = ptr.left;
            } else {
                if (st.isEmpty())
                    break;
                ptr = st.pop();
                answer.add(ptr.val);
                ptr = ptr.right;
            }
        }
        return answer;
    }
}
