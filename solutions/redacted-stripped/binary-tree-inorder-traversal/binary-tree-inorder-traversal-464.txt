class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        return var1;
    }

    static void method2(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        method2(node.left, list);
        list.add(node.val);
        method2(node.right, list);
    }
}
