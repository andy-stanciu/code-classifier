class Solution {

    public void method1(TreeNode root, List<Integer> param1) {
        if (root.left != null) {
            method1(root.left, param1);
        }
        param1.add(root.val);
        if (root.right != null) {
            method1(root.right, param1);
        }
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> param1 = new ArrayList<>();
        if (root == null) {
            return param1;
        }
        method1(root, param1);
        return param1;
    }
}
