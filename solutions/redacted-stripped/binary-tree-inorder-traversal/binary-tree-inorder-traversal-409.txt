class Solution {

    static List<Integer> res = new ArrayList<>();

    public static void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        res.add(root.val);
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        res = new ArrayList<>();
        method1(root);
        return res;
    }
}
