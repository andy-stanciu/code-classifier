class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        find(root, ans);
        return ans;
    }

    public static void find(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        find(root.left, ans);
        ans.add(root.val);
        find(root.right, ans);
    }
}
