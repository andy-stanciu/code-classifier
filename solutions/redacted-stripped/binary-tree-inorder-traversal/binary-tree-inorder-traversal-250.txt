class Solution {

    public List<Integer> method1(TreeNode root) {
        return method2(root, new ArrayList<Integer>());
    }

    public List<Integer> method2(TreeNode root, List<Integer> l) {
        if (root == null) {
            return l;
        }
        method2(root.left, l);
        l.add(root.val);
        method2(root.right, l);
        return l;
    }
}
