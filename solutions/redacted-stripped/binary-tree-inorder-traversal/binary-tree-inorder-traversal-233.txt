class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> concatenatedList = new ArrayList<>();
        if (root == null)
            return concatenatedList;
        if (root.left != null)
            concatenatedList.addAll(method1(root.left));
        concatenatedList.add(root.val);
        if (root.right != null)
            concatenatedList.addAll(method1(root.right));
        return concatenatedList;
    }
}
