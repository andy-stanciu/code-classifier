class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        bypass(list, root);
        return list;
    }

    public void bypass(ArrayList<Integer> list, TreeNode root) {
        if (root != null) {
            bypass(list, root.left);
            list.add(root.val);
            bypass(list, root.right);
        }
    }
}
