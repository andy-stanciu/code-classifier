class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new LinkedList<>();
        if (root == null)
            return output;
        method1(output, root);
        return output;
    }

    public void method1(List<Integer> output, TreeNode root) {
        if (root == null)
            return;
        method1(output, root.left);
        output.add(root.val);
        method1(output, root.right);
    }
}
