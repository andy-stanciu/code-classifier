class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        if (root == null)
            return ls;
        Stack<TreeNode> st = new Stack();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode temp = st.peek();
            if (temp.left != null) {
                st.push(temp.left);
                temp.left = null;
            } else {
                st.pop();
                ls.add(temp.val);
                if (temp.right != null)
                    st.push(temp.right);
            }
        }
        return ls;
    }
}
