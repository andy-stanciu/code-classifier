class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList();
        method2(list, root);
        return list;
    }

    public void method2(List<Integer> list, TreeNode root) {
        if (root == null)
            return;
        Stack<TreeNode> st = new Stack();
        st.push(root);
        while (!st.isEmpty()) {
            if (root.left != null) {
                st.push(root.left);
                root = root.left;
            } else {
                TreeNode node = st.pop();
                list.add(node.val);
                if (node.right != null) {
                    st.push(node.right);
                    root = node.right;
                }
            }
        }
    }
}
