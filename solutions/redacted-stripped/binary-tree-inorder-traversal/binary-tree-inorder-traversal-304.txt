class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> a = new ArrayList<Integer>();
        if (root == null)
            return a;
        method2(root, a);
        return a;
    }

    void method2(TreeNode root, List<Integer> a) {
        if (root == null)
            return;
        method2(root.left, a);
        a.add(root.val);
        method2(root.right, a);
    }
}
