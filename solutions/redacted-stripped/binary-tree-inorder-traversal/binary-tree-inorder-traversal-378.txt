class Pair {

    TreeNode node;

    boolean flag;

    public Pair(TreeNode node, boolean flag) {
        this.node = node;
        this.flag = flag;
    }
}

class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        if (root == null) {
            return var1;
        }
        Stack<Pair> stack = new Stack<>();
        stack.push(new Pair(root, false));
        Pair pair = null;
        TreeNode node = null;
        boolean flag = false;
        while (!stack.isEmpty()) {
            pair = stack.pop();
            node = pair.node;
            flag = pair.flag;
            if (flag == true) {
                var1.add(node.val);
                continue;
            }
            if (node.right != null) {
                stack.push(new Pair(node.right, false));
            }
            stack.push(new Pair(node, true));
            if (node.left != null) {
                stack.push(new Pair(node.left, false));
            }
        }
        return var1;
    }
}
