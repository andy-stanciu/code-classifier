class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans;
    }

    void method2(TreeNode node, List<Integer> list) {
        if (node == null) {
            return;
        }
        method2(node.left, list);
        list.add(node.val);
        method2(node.right, list);
    }
}
