class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        result(root, list);
        return list;
    }

    public void result(TreeNode root, ArrayList<Integer> list) {
        if (root == null)
            return;
        result(root.left, list);
        list.add(root.val);
        result(root.right, list);
    }
}
