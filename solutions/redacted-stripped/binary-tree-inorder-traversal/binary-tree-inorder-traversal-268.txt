class Solution {

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        List<Integer> ans = new ArrayList<>();
        List<Integer> left = method1(root.left);
        for (int n : left) {
            ans.add(n);
        }
        ans.add(root.val);
        List<Integer> right = method1(root.right);
        for (int n : right) {
            ans.add(n);
        }
        return ans;
    }
}
