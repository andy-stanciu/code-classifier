class Solution {

    private ListNode temp = null;

    public ListNode addLast(ListNode list, ListNode list3) {
        if (list3 == null) {
            list3 = list;
            temp = list;
            return list3;
        }
        temp.next = list;
        temp = temp.next;
        return list3;
    }

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode list3 = null;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                list3 = addLast(list1, list3);
                list1 = list1.next;
            } else {
                list3 = addLast(list2, list3);
                list2 = list2.next;
            }
        }
        if (list1 != null) {
            temp.next = list1;
        } else if (list2 != null) {
            temp.next = list2;
        }
        return list3;
    }
}
