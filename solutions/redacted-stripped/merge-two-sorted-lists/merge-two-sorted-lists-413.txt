class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode finalhead = new ListNode();
        ListNode currentfinal = finalhead;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                ListNode newNode = new ListNode(list1.val);
                currentfinal.next = newNode;
                currentfinal = newNode;
                list1 = list1.next;
            } else {
                ListNode newNode = new ListNode(list2.val);
                currentfinal.next = newNode;
                currentfinal = newNode;
                list2 = list2.next;
            }
        }
        while (list1 != null) {
            ListNode newNode = new ListNode(list1.val);
            currentfinal.next = newNode;
            currentfinal = newNode;
            list1 = list1.next;
        }
        while (list2 != null) {
            ListNode newNode = new ListNode(list2.val);
            currentfinal.next = newNode;
            currentfinal = newNode;
            list2 = list2.next;
        }
        finalhead = finalhead.next;
        return finalhead;
    }
}
