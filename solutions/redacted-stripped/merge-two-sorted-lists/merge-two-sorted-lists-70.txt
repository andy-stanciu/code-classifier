class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode var1 = new ListNode();
        ListNode currentNode = var1;
        while (l1 != null && l2 != null) {
            if (l1.val < l2.val) {
                currentNode.next = l1;
                l1 = l1.next;
            } else {
                currentNode.next = l2;
                l2 = l2.next;
            }
            currentNode = currentNode.next;
        }
        if (l1 != null)
            currentNode.next = l1;
        if (l2 != null)
            currentNode.next = l2;
        return var1.next;
    }
}
