class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode solvebase = new ListNode();
        ListNode pointer = solvebase;
        while (true) {
            if (list1 == null) {
                return list2;
            }
            if (list2 == null) {
                return list1;
            }
            if (list1.val < list2.val) {
                pointer.val = list1.val;
                if (!(list1.next instanceof ListNode)) {
                    pointer.next = list2;
                    return solvebase;
                }
                list1 = list1.next;
            } else {
                pointer.val = list2.val;
                if (!(list2.next instanceof ListNode)) {
                    pointer.next = list1;
                    return solvebase;
                }
                list2 = list2.next;
            }
            pointer.next = new ListNode();
            pointer = pointer.next;
        }
    }
}
