class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null && list2 == null) {
            return null;
        }
        if (list1 == null) {
            return list2;
        }
        if (list2 == null) {
            return list1;
        }
        if (list1.val <= list2.val) {
            ListNode var1 = method1(list1.next, list2);
            list1.next = var1;
            return list1;
        }
        ListNode var1 = method1(list1, list2.next);
        list2.next = var1;
        return list2;
    }
}
