class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode dummy = new ListNode(-1);
        ListNode result = dummy;
        while (list1 != null && list2 != null) {
            if (list1.val > list2.val) {
                ListNode ls = new ListNode(list2.val);
                list2 = list2.next;
                result.next = ls;
                result = ls;
            } else {
                ListNode ls = new ListNode(list1.val);
                list1 = list1.next;
                result.next = ls;
                result = ls;
            }
        }
        while (list1 != null) {
            ListNode ls = new ListNode(list1.val);
            list1 = list1.next;
            result.next = ls;
            result = ls;
        }
        while (list2 != null) {
            ListNode ls = new ListNode(list2.val);
            list2 = list2.next;
            result.next = ls;
            result = ls;
        }
        return dummy.next;
    }
}
