class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null && list2 == null)
            return null;
        else if (list1 == null)
            return list2;
        else if (list2 == null)
            return list1;
        ListNode curr1 = list1;
        ListNode curr2 = list2;
        ListNode head;
        if (curr1.val <= curr2.val) {
            head = curr1;
            curr1 = curr1.next;
        } else {
            head = curr2;
            curr2 = curr2.next;
        }
        ListNode curr = head;
        while (curr1 != null && curr2 != null) {
            if (curr1.val <= curr2.val) {
                curr.next = curr1;
                curr1 = curr1.next;
            } else {
                curr.next = curr2;
                curr2 = curr2.next;
            }
            curr = curr.next;
        }
        if (curr1 != null)
            curr.next = curr1;
        if (curr2 != null)
            curr.next = curr2;
        return head;
    }
}
