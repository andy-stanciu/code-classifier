class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode head = new ListNode();
        if (list1 == null && list2 == null)
            return list1;
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode curr = head;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                curr.next = list1;
                list1 = list1.next;
            } else {
                curr.next = list2;
                list2 = list2.next;
            }
            curr = curr.next;
            System.out.print(curr.val);
        }
        if (list1 != null) {
            curr.next = list1;
        } else
            curr.next = list2;
        return head.next;
    }
}
