class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list2 == null)
            return list1;
        if (list1 == null)
            return list2;
        if (list1.val < list2.val)
            return new ListNode(list1.val, method1(list1.next, list2));
        return new ListNode(list2.val, method1(list1, list2.next));
    }
}
