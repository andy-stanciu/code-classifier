class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode result = null;
        ListNode resultHead = null;
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                if (result == null) {
                    result = list1;
                    resultHead = result;
                } else {
                    result.next = list1;
                    result = result.next;
                }
                list1 = list1.next;
            } else {
                if (result == null) {
                    result = list2;
                    resultHead = result;
                } else {
                    result.next = list2;
                    result = result.next;
                }
                list2 = list2.next;
            }
        }
        if (list1 != null) {
            result.next = list1;
        }
        if (list2 != null) {
            result.next = list2;
        }
        return resultHead;
    }
}
