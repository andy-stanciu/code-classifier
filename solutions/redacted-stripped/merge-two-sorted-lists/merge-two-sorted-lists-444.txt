class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode dummy = new ListNode(-1);
        ListNode temp1 = dummy;
        ListNode temp2 = list1;
        ListNode temp3 = list2;
        while (temp2 != null && temp3 != null) {
            if (temp2.val <= temp3.val) {
                temp1.next = temp2;
                temp2 = temp2.next;
            } else {
                temp1.next = temp3;
                temp3 = temp3.next;
            }
            temp1 = temp1.next;
        }
        if (temp2 != null) {
            temp1.next = temp2;
        } else {
            temp1.next = temp3;
        }
        return dummy.next;
    }
}
