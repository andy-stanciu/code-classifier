class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode curr = new ListNode(-1);
        ListNode dummy = curr;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                ListNode nn = new ListNode(list1.val);
                dummy.next = nn;
                dummy = dummy.next;
                list1 = list1.next;
            } else {
                ListNode nn = new ListNode(list2.val);
                dummy.next = nn;
                dummy = dummy.next;
                list2 = list2.next;
            }
        }
        while (list1 != null) {
            ListNode nn = new ListNode(list1.val);
            dummy.next = nn;
            dummy = dummy.next;
            list1 = list1.next;
        }
        while (list2 != null) {
            ListNode nn = new ListNode(list2.val);
            dummy.next = nn;
            dummy = dummy.next;
            list2 = list2.next;
        }
        return curr.next;
    }
}
