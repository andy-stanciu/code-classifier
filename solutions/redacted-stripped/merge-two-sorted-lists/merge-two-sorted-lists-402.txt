class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode left = list1;
        ListNode right = list2;
        if (left == null)
            return right;
        if (right == null)
            return left;
        ListNode result;
        if (left.val <= right.val) {
            result = left;
            result.next = method1(left.next, right);
        } else {
            result = right;
            result.next = method1(left, right.next);
        }
        return result;
    }
}
