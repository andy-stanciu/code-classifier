class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        }
        if (list2 == null) {
            return list1;
        }
        ListNode newlist = new ListNode(-1);
        ListNode pointer = newlist;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                pointer.next = list1;
                list1 = list1.next;
            } else {
                pointer.next = list2;
                list2 = list2.next;
            }
            pointer = pointer.next;
        }
        if (list1 != null) {
            pointer.next = list1;
        } else if (list2 != null) {
            pointer.next = list2;
        }
        return newlist.next;
    }
}
