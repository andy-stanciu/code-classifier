class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode Head = new ListNode();
        ListNode tail = Head;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                tail.next = list1;
                list1 = list1.next;
                tail = tail.next;
            } else {
                tail.next = list2;
                list2 = list2.next;
                tail = tail.next;
            }
        }
        tail.next = (list1 != null) ? list1 : list2;
        return Head.next;
    }
}
