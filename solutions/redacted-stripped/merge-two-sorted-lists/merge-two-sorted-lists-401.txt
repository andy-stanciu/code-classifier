class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode result = new ListNode(0);
        ListNode tempR = result;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                tempR.next = new ListNode(list1.val);
                list1 = list1.next;
            } else {
                tempR.next = new ListNode(list2.val);
                list2 = list2.next;
            }
            tempR = tempR.next;
        }
        if (list1 != null) {
            tempR.next = list1;
        } else if (list2 != null) {
            tempR.next = list2;
        }
        return result.next;
    }
}
