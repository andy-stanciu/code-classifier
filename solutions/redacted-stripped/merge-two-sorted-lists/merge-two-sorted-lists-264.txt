class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode result = new ListNode();
        ListNode resultIter = result;
        while (list1 != null && list2 != null) {
            int first = list1.val;
            int second = list2.val;
            if (first < second) {
                resultIter = push(resultIter, first);
                list1 = list1.next;
            } else {
                resultIter = push(resultIter, second);
                list2 = list2.next;
            }
        }
        ListNode iter = (list1 != null) ? list1 : list2;
        while (iter != null) {
            resultIter = push(resultIter, iter.val);
            iter = iter.next;
        }
        return result.next;
    }

    private ListNode push(ListNode targetList, int what) {
        ListNode newNode = new ListNode(what);
        targetList.next = newNode;
        return newNode;
    }
}
