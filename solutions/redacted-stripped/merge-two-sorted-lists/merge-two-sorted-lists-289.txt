class Solution {

    public ListNode solve(ListNode list1, ListNode list2) {
        ListNode newhead = new ListNode();
        ListNode cur1 = list1.next;
        ListNode cur2 = list2;
        newhead = list1;
        ListNode newcur = newhead;
        while (cur1 != null && cur2 != null) {
            if (cur1.val < cur2.val) {
                newcur.next = cur1;
                cur1 = cur1.next;
                newcur = newcur.next;
            } else {
                newcur.next = cur2;
                cur2 = cur2.next;
                newcur = newcur.next;
            }
        }
        while (cur1 != null) {
            newcur.next = cur1;
            cur1 = cur1.next;
            newcur = newcur.next;
        }
        while (cur2 != null) {
            newcur.next = cur2;
            cur2 = cur2.next;
            newcur = newcur.next;
        }
        return newhead;
    }

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        }
        if (list2 == null) {
            return list1;
        }
        if (list1.val > list2.val) {
            return solve(list2, list1);
        }
        return solve(list1, list2);
    }
}
