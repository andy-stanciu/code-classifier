class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode dummy = new ListNode(0);
        ListNode ptr = dummy;
        ListNode left = list1;
        ListNode right = list2;
        while (left != null && right != null) {
            if (left.val <= right.val) {
                ptr.next = left;
                ptr = left;
                left = left.next;
            } else {
                ptr.next = right;
                ptr = right;
                right = right.next;
            }
        }
        if (left != null)
            ptr.next = left;
        if (right != null)
            ptr.next = right;
        return dummy.next;
    }
}
