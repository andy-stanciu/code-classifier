class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        List<Integer> lst = new ArrayList();
        ListNode dummy = new ListNode();
        ListNode res = dummy;
        while (list1 != null || list2 != null) {
            if (list1 != null) {
                lst.add(list1.val);
                list1 = list1.next;
            }
            if (list2 != null) {
                lst.add(list2.val);
                list2 = list2.next;
            }
        }
        Collections.sort(lst);
        for (Integer o : lst) {
            dummy.next = new ListNode(o);
            dummy = dummy.next;
        }
        return res.next;
    }
}
