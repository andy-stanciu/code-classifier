class Solution {

    private static ListNode getNodes(ListNode list) {
        list.next = new ListNode();
        return list.next;
    }

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null && list2 == null)
            return list1;
        ListNode list = new ListNode();
        ListNode head = list;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                list.val = list1.val;
                list1 = list1.next;
            } else {
                list.val = list2.val;
                list2 = list2.next;
            }
            list = getNodes(list);
        }
        while (list1 != null) {
            if (list1.next != null) {
                list.val = list1.val;
                list = getNodes(list);
            } else
                list.val = list1.val;
            list1 = list1.next;
        }
        while (list2 != null) {
            if (list2.next != null) {
                list.val = list2.val;
                list = getNodes(list);
            } else
                list.val = list2.val;
            list2 = list2.next;
        }
        return head;
    }
}
