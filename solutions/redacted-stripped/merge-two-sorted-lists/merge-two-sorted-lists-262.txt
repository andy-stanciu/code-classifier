class Solution {

    public ListNode method1(ListNode a, ListNode b) {
        ListNode r = new ListNode(-1);
        ListNode p = r;
        while (a != null && b != null) {
            if (a.val <= b.val) {
                p.next = a;
                a = a.next;
            } else {
                p.next = b;
                b = b.next;
            }
            p = p.next;
        }
        if (a != null) {
            p.next = a;
        } else {
            p.next = b;
        }
        return r.next;
    }
}
