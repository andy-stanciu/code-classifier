class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        List<Integer> cse = new ArrayList<>();
        ListNode curr = list1;
        ListNode curr1 = list2;
        while (curr != null) {
            cse.add(curr.val);
            curr = curr.next;
        }
        while (curr1 != null) {
            cse.add(curr1.val);
            curr1 = curr1.next;
        }
        Collections.sort(cse);
        ListNode dummy = new ListNode(0);
        ListNode tail = dummy;
        for (int val : cse) {
            tail.next = new ListNode(val);
            tail = tail.next;
        }
        return dummy.next;
    }
}
