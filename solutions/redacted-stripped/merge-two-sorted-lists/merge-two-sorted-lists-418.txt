class Solution {

    public ListNode method1(ListNode f, ListNode s) {
        ListNode ans = new ListNode();
        ListNode tail = ans;
        while (f != null && s != null) {
            if (f.val < s.val) {
                tail.next = f;
                f = f.next;
            } else {
                tail.next = s;
                s = s.next;
            }
            tail = tail.next;
        }
        tail.next = (f != null) ? f : s;
        return ans.next;
    }
}
