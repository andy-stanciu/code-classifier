class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null && list2 == null)
            return null;
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode newNode = new ListNode();
        ListNode temp = new ListNode();
        if (list1.val > list2.val) {
            newNode = list2;
            list2 = list2.next;
        } else {
            newNode = list1;
            list1 = list1.next;
        }
        temp = newNode;
        newNode.next = method1(list1, list2);
        return temp;
    }
}
