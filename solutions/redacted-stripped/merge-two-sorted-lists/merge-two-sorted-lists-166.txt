class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode result = null;
        if (list1 == null)
            return (list2);
        if (list2 == null)
            return (list1);
        if ((list1.val) <= (list2.val)) {
            result = list1;
            result.next = method1(list1.next, list2);
        } else {
            result = list2;
            result.next = method1(list1, list2.next);
        }
        return (result);
    }
}
