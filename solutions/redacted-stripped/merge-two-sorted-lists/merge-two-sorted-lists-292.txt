class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode newNode = new ListNode();
        ListNode cur = newNode;
        while (list1 != null && list2 != null) {
            if (list1.val > list2.val) {
                cur.next = list2;
                list2 = list2.next;
            } else {
                cur.next = list1;
                list1 = list1.next;
            }
            cur = cur.next;
        }
        if (list1 == null) {
            cur.next = list2;
        }
        if (list2 == null) {
            cur.next = list1;
        }
        return newNode.next;
    }
}
