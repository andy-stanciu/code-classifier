class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode list3 = new ListNode(0);
        ListNode ans = list3;
        while (list1 != null && list2 != null) {
            if (list1.val > list2.val) {
                ans.next = list2;
                list2 = list2.next;
            } else {
                ans.next = list1;
                list1 = list1.next;
            }
            ans = ans.next;
        }
        if (list1 != null) {
            ans.next = list1;
            list1 = list1.next;
        }
        if (list2 != null) {
            ans.next = list2;
            list2 = list2.next;
        }
        return list3.next;
    }
}
