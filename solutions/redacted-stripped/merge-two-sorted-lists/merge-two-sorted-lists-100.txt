class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode nlist = new ListNode();
        ListNode cur1 = l1, cur2 = l2, curr = nlist;
        while (cur1 != null && cur2 != null) {
            ListNode nnode = new ListNode();
            if (cur1.val < cur2.val) {
                nnode.val = cur1.val;
                cur1 = cur1.next;
                curr.next = nnode;
                curr = nnode;
            } else if (cur1.val > cur2.val) {
                nnode.val = cur2.val;
                cur2 = cur2.next;
                curr.next = nnode;
                curr = nnode;
            } else {
                nnode.val = cur1.val;
                cur1 = cur1.next;
                curr.next = nnode;
                curr = nnode;
            }
        }
        nlist = nlist.next;
        if (cur1 != null)
            curr.next = cur1;
        if (cur2 != null)
            curr.next = cur2;
        return nlist;
    }

    public ListNode method2(ListNode list1, ListNode list2) {
        if (list1 == null && list2 == null)
            return null;
        else if (list1 == null && list2 != null)
            return list2;
        else if (list2 == null && list1 != null)
            return list1;
        ListNode nlist = method1(list1, list2);
        return nlist;
    }
}
