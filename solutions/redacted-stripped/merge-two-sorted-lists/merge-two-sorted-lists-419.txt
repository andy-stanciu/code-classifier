class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        } else if (list2 == null) {
            return list1;
        }
        if (list1.val <= list2.val) {
            list1.next = method1(list1.next, list2);
            return list1;
        } else {
            list2.next = method1(list2.next, list1);
            return list2;
        }
    }
}
