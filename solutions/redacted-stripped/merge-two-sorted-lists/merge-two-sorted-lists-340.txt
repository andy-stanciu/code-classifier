class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ArrayList<Integer> al1 = new ArrayList<>();
        ArrayList<Integer> al2 = new ArrayList<>();
        while (list1 != null) {
            al1.add(list1.val);
            list1 = list1.next;
        }
        while (list2 != null) {
            al2.add(list2.val);
            list2 = list2.next;
        }
        al1.addAll(al2);
        Collections.sort(al1);
        ListNode temp = new ListNode(-1);
        ListNode head = temp;
        for (int data : al1) {
            temp.next = new ListNode(data);
            temp = temp.next;
        }
        head = head.next;
        return head;
    }
}
