class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        if (l1 == null || l2 == null)
            return l1 == null ? l2 : l1;
        ListNode first = (l1.val < l2.val) ? l1 : l2;
        first.next = method1(first.next, first == l1 ? l2 : l1);
        return first;
    }
}
