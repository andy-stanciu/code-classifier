class Solution {

    public ListNode method1(ListNode left, ListNode right) {
        ListNode newHead = new ListNode(-1);
        ListNode temp = newHead;
        while (left != null && right != null) {
            if (left.val > right.val) {
                temp.next = right;
                temp = temp.next;
                right = right.next;
            } else {
                temp.next = left;
                temp = temp.next;
                left = left.next;
            }
        }
        while (left != null) {
            temp.next = left;
            left = left.next;
            temp = temp.next;
        }
        while (right != null) {
            temp.next = right;
            right = right.next;
            temp = temp.next;
        }
        return newHead.next;
    }
}
