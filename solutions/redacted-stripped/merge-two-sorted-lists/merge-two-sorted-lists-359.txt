class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        }
        if (list2 == null) {
            return list1;
        }
        ListNode returnNode = null;
        if (list1.val <= list2.val) {
            returnNode = list1;
            returnNode.next = method1(list1.next, list2);
        } else {
            returnNode = list2;
            returnNode.next = method1(list1, list2.next);
        }
        return returnNode;
    }
}
