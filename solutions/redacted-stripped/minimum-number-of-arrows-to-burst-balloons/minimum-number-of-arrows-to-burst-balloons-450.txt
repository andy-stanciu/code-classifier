class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int var1 = 1;
        int limit = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (points[i][0] > limit) {
                var1++;
                limit = points[i][1];
            } else {
                limit = Math.min(limit, points[i][1]);
            }
        }
        return var1;
    }
}
