class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> a[1] - b[1]);
        int arrow = 1;
        int prev = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (prev >= points[i][0] && prev <= points[i][1]) {
                continue;
            }
            arrow++;
            prev = points[i][1];
        }
        return arrow;
    }
}
