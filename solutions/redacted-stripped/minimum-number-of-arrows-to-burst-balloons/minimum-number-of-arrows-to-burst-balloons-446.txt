class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, new Class1());
        int ans = 1;
        int lastEnd = points[0][1];
        int i = 1;
        while (i < points.length) {
            int currentStart = points[i][0];
            int currentEnd = points[i][1];
            if (currentStart > lastEnd) {
                ans++;
                lastEnd = currentEnd;
            } else {
                lastEnd = Math.min(lastEnd, currentEnd);
            }
            i++;
        }
        return ans;
    }

    class Class1 implements Comparator<int[]> {

        public int compare(int[] a, int[] b) {
            int cmp = Integer.compare(a[0], b[0]);
            if (cmp == 0) {
                cmp = Integer.compare(a[1], b[1]);
            }
            return cmp;
        }
    }
}
