class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int i = 0;
        int ans = 0;
        int last = points[0][1];
        ans += 1;
        while (i < points.length) {
            if (last < points[i][0]) {
                last = points[i][1];
                ans += 1;
            }
            i++;
        }
        return ans;
    }
}
