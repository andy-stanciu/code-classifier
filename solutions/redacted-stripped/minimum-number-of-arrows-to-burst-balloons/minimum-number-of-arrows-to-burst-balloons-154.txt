class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int lastArrow = points[0][1];
        int count = 1;
        for (int i = 1; i < points.length; i++) {
            if (lastArrow >= points[i][0])
                continue;
            else {
                lastArrow = points[i][1];
                count++;
            }
        }
        return count;
    }
}
