class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int count = 0;
        int commonSurface = points[0][1];
        int i = 1;
        while (i < points.length) {
            while (i < points.length && commonSurface >= points[i][0]) {
                i++;
            }
            if (i < points.length) {
                commonSurface = points[i][1];
            }
            i++;
            count++;
        }
        if (i == points.length) {
            count++;
        }
        return count;
    }
}
