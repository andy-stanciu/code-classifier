class Solution {

    public int method1(int[][] points) {
        int var1 = 1;
        Arrays.sort(points, Comparator.comparingInt(arr -> arr[1]));
        int last_num = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > last_num) {
                var1++;
                last_num = points[i][1];
            }
        }
        return var1;
    }
}
