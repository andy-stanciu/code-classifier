class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> a[1] - b[1]);
        int var1 = 1, idx = 1;
        int sp = points[0][1];
        while (idx < points.length) {
            if (points[idx][0] > sp || points[idx][1] < sp) {
                var1++;
                sp = points[idx][1];
            }
            idx++;
        }
        return var1;
    }
}
