class Solution {

    public int method1(int[][] points) {
        if (points.length == 0) {
            return 0;
        }
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int res = 1, endX = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (endX < points[i][0]) {
                res++;
                endX = points[i][1];
            }
        }
        return res;
    }
}
