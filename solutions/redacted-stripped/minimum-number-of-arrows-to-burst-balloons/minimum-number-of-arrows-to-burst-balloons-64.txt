class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int arrowCount = 1;
        int previous = 0;
        for (int current = 0; current < points.length; current++) {
            if (points[current][0] > points[previous][1]) {
                arrowCount++;
                previous = current;
            }
        }
        return arrowCount;
    }
}
