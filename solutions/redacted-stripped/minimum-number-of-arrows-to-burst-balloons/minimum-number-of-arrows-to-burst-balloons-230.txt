class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> {
            if (a[0] == b[0]) {
                return Integer.compare(a[1], b[1]);
            }
            return Integer.compare(a[0], b[0]);
        });
        int count = 1;
        int point = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > point) {
                count++;
                point = points[i][1];
            } else {
                point = Math.min(point, points[i][1]);
            }
        }
        return count;
    }
}
