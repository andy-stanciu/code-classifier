class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(point -> point[0]));
        int var1 = 1;
        int end = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (end < points[i][0]) {
                var1++;
                end = points[i][1];
            } else {
                end = Math.min(end, points[i][1]);
            }
        }
        return var1;
    }
}
