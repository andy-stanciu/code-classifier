class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return Integer.compare(a[1], b[1]);
            }
        });
        int i = 0;
        int j = i + 1;
        int var1 = 1;
        while (i < points.length && j < points.length) {
            if (points[i][1] >= points[j][0]) {
                j++;
            } else {
                i = j;
                j++;
                var1++;
            }
        }
        return var1;
    }
}
