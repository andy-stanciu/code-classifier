class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int n = points.length, cnt = 0, start = points[0][0], end = points[0][1];
        for (int i = 1; i < n; i++) {
            int currStart = points[i][0], currEnd = points[i][1];
            if (end >= currStart) {
                end = Math.min(end, currEnd);
            } else {
                cnt++;
                start = currStart;
                end = currEnd;
            }
        }
        cnt++;
        return cnt;
    }
}
