class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, new Comparator<int[]>() {

            @Override
            public int compare(int[] a, int[] b) {
                return Integer.compare(a[0], b[0]);
            }
        });
        int result = 1;
        int limit = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > limit) {
                result++;
                limit = points[i][1];
            } else if (limit > points[i][1])
                limit = points[i][1];
        }
        return result;
    }
}
