class Solution {

    public int method1(int[][] points) {
        int result = 0;
        int var1 = -1;
        Arrays.sort(points, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                if (a[1] == b[1]) {
                    return Integer.compare(a[0], b[0]);
                }
                return Integer.compare(a[1], b[1]);
            }
        });
        for (int i = 0; i < points.length; i++) {
            if (var1 == -1) {
                ++result;
                var1 = points[i][1];
            } else {
                if (var1 >= points[i][0]) {
                    continue;
                } else {
                    ++result;
                    var1 = points[i][1];
                }
            }
        }
        return result;
    }
}
