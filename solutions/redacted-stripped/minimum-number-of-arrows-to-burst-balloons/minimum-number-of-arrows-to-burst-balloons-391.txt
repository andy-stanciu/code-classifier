public class Solution {

    public int method1(int[][] points) {
        if (points == null || points.length == 0) {
            return 0;
        }
        Arrays.sort(points, Comparator.comparingInt(interval -> interval[1]));
        int arrow = 1;
        int k = points[0][1];
        for (int i = 1; i < points.length; ++i) {
            if (k < points[i][0]) {
                ++arrow;
                k = points[i][1];
            }
        }
        return arrow;
    }
}
