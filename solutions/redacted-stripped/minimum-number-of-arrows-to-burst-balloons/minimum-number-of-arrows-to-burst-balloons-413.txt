class Solution {

    public int method1(int[][] points) {
        long p = Long.MIN_VALUE;
        int count = 0;
        Arrays.sort(points, Comparator.comparingDouble(o -> o[0]));
        for (int i = 0; i < points.length; i++) {
            if (points[i][0] > p) {
                p = points[i][1];
                count++;
            } else
                p = Math.min(p, points[i][1]);
        }
        return count;
    }
}
