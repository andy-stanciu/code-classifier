class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (int[] a, int[] b) -> Integer.compare(a[0], b[0]));
        int count = 0;
        for (int i = 0; i < points.length; i++) {
            int end = points[i][1];
            while (i + 1 < points.length && points[i + 1][0] <= end) {
                count++;
                end = Math.min(end, points[i + 1][1]);
                i++;
            }
        }
        return points.length - count;
    }
}
