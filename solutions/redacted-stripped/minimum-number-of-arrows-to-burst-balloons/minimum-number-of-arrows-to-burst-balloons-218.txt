class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> a[0] - b[0]);
        int start = points[0][0], end = points[0][1];
        int count = 1;
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] >= start && points[i][0] <= end) {
                end = end < points[i][1] ? end : points[i][1];
            } else {
                start = points[i][0];
                end = points[i][1];
                count++;
            }
        }
        return count;
    }
}
