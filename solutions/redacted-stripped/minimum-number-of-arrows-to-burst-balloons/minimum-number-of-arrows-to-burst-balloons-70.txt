class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(o -> o[1]));
        int ans = 1;
        int bound = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > bound) {
                bound = points[i][1];
                ans++;
            }
        }
        return ans;
    }
}
