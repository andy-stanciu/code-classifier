class Solution {

    public int method1(int[][] points) {
        if (points.length == 0)
            return 0;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int arrow = 1;
        int end_point = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] < end_point)
                continue;
            if (points[i][0] > end_point) {
                arrow++;
                end_point = points[i][1];
            }
        }
        return arrow;
    }
}
