class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int maxAct = 1;
        int last = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (last < points[i][0]) {
                maxAct++;
                last = points[i][1];
            }
        }
        return maxAct;
    }
}
