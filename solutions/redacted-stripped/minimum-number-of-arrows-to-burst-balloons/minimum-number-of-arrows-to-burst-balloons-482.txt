class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> a[0] == b[0] ? Integer.compare(a[1], b[1]) : Integer.compare(a[0], b[0]));
        int ct = 1;
        int[] last = new int[] { points[0][0], points[0][1] };
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] <= last[1]) {
                last[0] = points[i][0];
                last[1] = Math.min(last[1], points[i][1]);
            } else {
                ct++;
                last[0] = points[i][0];
                last[1] = points[i][1];
            }
        }
        return ct;
    }
}
