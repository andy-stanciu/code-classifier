class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int var1 = 1;
        int prev = points[0][1];
        for (int i = 1; i < points.length; i++) {
            System.out.println(points[i][0] + " - " + points[i][1]);
            if (points[i][0] > prev) {
                var1++;
                prev = points[i][1];
            } else
                prev = Math.min(prev, points[i][1]);
        }
        return var1;
    }
}
