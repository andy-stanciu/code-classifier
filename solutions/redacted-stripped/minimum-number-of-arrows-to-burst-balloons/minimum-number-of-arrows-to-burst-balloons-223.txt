class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> a[1] - b[1]);
        int x = points[0][1], cnt = 1;
        for (int[] point : points) {
            if (point[0] <= x && point[1] >= x)
                continue;
            else {
                x = point[1];
                cnt++;
            }
        }
        return cnt;
    }
}
