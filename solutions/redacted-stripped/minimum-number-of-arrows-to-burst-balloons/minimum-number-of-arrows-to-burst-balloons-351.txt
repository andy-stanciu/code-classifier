class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(a -> a[1]));
        int count = 1;
        int temp = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (points[i][0] > temp) {
                count++;
                temp = points[i][1];
            }
        }
        return count;
    }
}
