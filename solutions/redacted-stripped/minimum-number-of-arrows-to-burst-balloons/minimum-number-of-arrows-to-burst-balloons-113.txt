class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (o1, o2) -> Integer.compare(o1[1], o2[1]));
        int var1 = 1;
        int reach = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (reach < points[i][0]) {
                var1++;
                reach = points[i][1];
            } else {
            }
        }
        return var1;
    }
}
