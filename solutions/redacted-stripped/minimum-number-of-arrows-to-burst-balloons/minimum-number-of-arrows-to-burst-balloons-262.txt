class Solution {

    public int method1(int[][] points) {
        int l = points.length;
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int[] prev = points[0];
        int cnt = 1;
        for (int i = 1; i < l; i++) {
            int currentStartPoint = points[i][0];
            int currentEndPoint = points[i][1];
            int previousStartPoint = prev[0];
            int previousEndPoint = prev[1];
            if (currentStartPoint > previousEndPoint) {
                cnt++;
                prev = points[i];
            } else {
                prev[0] = Math.max(currentStartPoint, previousStartPoint);
                prev[1] = Math.min(currentEndPoint, previousEndPoint);
            }
        }
        return cnt;
    }
}
