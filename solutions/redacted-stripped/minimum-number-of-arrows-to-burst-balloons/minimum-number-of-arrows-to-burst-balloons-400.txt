class Solution {

    public int method1(int[][] points) {
        int currMax = 0;
        int count = 0;
        if (points.length == 0) {
            return 0;
        }
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        currMax = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (currMax < points[i][0]) {
                count++;
                currMax = points[i][1];
            }
        }
        return count + 1;
    }
}
