class Solution {

    public int method1(int[][] points) {
        int arrowCount = 1;
        Arrays.sort(points, (o1, o2) -> {
            if (o1[1] == o2[1])
                return 0;
            if (o1[1] < o2[1])
                return -1;
            return 1;
        });
        int basePos = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (basePos >= points[i][0]) {
                continue;
            }
            arrowCount++;
            basePos = points[i][1];
        }
        return arrowCount;
    }
}
