class Solution {

    public int method1(int[][] points) {
        if (points.length == 1)
            return 1;
        Arrays.sort(points, Comparator.comparingInt(point -> point[0]));
        int prev = points[0][1];
        int count = 1;
        for (int i = 1; i < points.length; i++) {
            int a = points[i][0];
            int b = points[i][1];
            if (a > prev) {
                count++;
                prev = points[i][1];
            } else {
                prev = Math.min(prev, points[i][1]);
            }
        }
        return count;
    }
}
