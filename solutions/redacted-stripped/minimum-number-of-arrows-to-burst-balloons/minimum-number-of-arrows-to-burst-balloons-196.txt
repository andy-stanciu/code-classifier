class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int res = 1;
        int curEnd = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int xStart = points[i][0], xEnd = points[i][1];
            if (xStart <= curEnd)
                continue;
            res++;
            curEnd = xEnd;
        }
        return res;
    }
}
