class Solution {

    public int method1(int[][] points) {
        if (points == null || points.length == 0) {
            return 0;
        }
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int cnt = 1;
        int end = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int[] cur = points[i];
            if (cur[0] > end) {
                cnt++;
                end = cur[1];
            } else {
                end = Math.min(end, cur[1]);
            }
        }
        return cnt;
    }
}
