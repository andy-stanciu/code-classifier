class Solution {

    public int method1(int[][] points) {
        if (points.length == 0)
            return 0;
        Arrays.sort(points, (x, y) -> Integer.compare(x[1], y[1]));
        int var1 = 1;
        int endBoundary = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > endBoundary) {
                var1++;
                endBoundary = points[i][1];
            }
        }
        return var1;
    }
}
