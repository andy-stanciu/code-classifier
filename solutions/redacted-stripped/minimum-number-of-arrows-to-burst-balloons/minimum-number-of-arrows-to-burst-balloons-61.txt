class Solution {

    public int method1(int[][] arr) {
        Arrays.sort(arr, Comparator.comparingInt(i -> i[0]));
        int prev = arr[0][1];
        int count = 1;
        for (int i = 1; i < arr.length; i++) {
            if (arr[i][0] <= prev) {
                prev = Math.min(prev, arr[i][1]);
            } else {
                count++;
                prev = arr[i][1];
            }
        }
        return count;
    }
}
