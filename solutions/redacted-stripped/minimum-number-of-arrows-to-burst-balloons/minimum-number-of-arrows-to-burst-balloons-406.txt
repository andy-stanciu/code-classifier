class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int prevEnd = points[0][1];
        int ans = 1;
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > prevEnd) {
                ans++;
                prevEnd = points[i][1];
            } else {
                prevEnd = Math.min(prevEnd, points[i][1]);
            }
        }
        return ans;
    }
}
