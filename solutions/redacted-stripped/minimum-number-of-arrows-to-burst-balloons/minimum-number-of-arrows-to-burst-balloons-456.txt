class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(a -> a[0]));
        int i = 0;
        int count = 0;
        while (i < points.length) {
            long end = points[i][1];
            i = i + 1;
            while (i < points.length && points[i][0] <= end) {
                end = Math.min(end, points[i][1]);
                i++;
            }
            count++;
        }
        return count;
    }
}
