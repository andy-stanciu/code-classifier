class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int n = points.length;
        int prevEnd = points[0][1];
        int count = 1;
        for (int i = 1; i < n; i++) {
            int currStart = points[i][0];
            if (currStart > prevEnd) {
                count++;
                prevEnd = points[i][1];
            }
        }
        return count;
    }
}
