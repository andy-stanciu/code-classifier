class Solution {

    public int method1(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[1], b[1]));
        int arrow = 1;
        int end = intervals[0][1];
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] > end) {
                arrow++;
                end = intervals[i][1];
            }
        }
        return arrow;
    }
}
