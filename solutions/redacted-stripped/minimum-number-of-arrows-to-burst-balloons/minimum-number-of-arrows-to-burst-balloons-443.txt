class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        Stack<Integer> stack = new Stack<>();
        int len = points.length;
        int ans = 0;
        stack.push(points[0][1]);
        for (int i = 1; i < len; i++) {
            if (stack.peek() >= points[i][0]) {
                continue;
            }
            stack.push(points[i][1]);
        }
        return stack.size();
    }
}
