class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> (a[1] < b[1] ? -1 : 1));
        int cnt = 0;
        for (int i = 0; i < points.length; ) {
            int j = i + 1;
            while (j < points.length && points[j][0] <= points[i][1]) {
                j++;
            }
            i = j;
            cnt++;
        }
        return cnt;
    }
}
