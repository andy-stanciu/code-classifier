class Solution {

    public int method1(int[][] points) {
        if (points.length == 0)
            return 0;
        Arrays.sort(points, Comparator.comparingInt(a -> a[0]));
        int k = points[0][1], var1 = 1;
        for (int[] pt : points) {
            if (pt[0] > k) {
                var1++;
                k = pt[1];
            } else {
                k = Math.min(k, pt[1]);
            }
        }
        return var1;
    }
}
