class Solution {

    public int method1(int[][] ps) {
        Arrays.sort(ps, (a, b) -> Integer.compare(a[0], b[0]));
        int[] prev = ps[0];
        int count = 1;
        for (int[] p : ps) {
            if (prev[1] >= p[0]) {
                prev[1] = Math.min(prev[1], p[1]);
            } else {
                count++;
                prev = p;
            }
        }
        return count;
    }
}
