class Solution {

    public int method1(int[][] points) {
        int res = 0, n = points.length;
        Integer prevEnd = null;
        Arrays.sort(points, (a, b) -> (a[0] < b[0]) ? -1 : 1);
        for (int[] point : points) {
            if (prevEnd == null || prevEnd < point[0]) {
                res++;
                prevEnd = point[1];
            } else
                prevEnd = Math.min(prevEnd, point[1]);
        }
        return res;
    }
}
