class Solution {

    public int method1(int[][] points) {
        int len = points.length;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int arrow = 1, start = points[0][1];
        for (int i = 1; i < len; i++) {
            if (points[i][0] <= start) {
                continue;
            } else {
                arrow++;
                start = points[i][1];
            }
        }
        return arrow;
    }
}
