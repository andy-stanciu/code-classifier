class Solution {

    public int method1(int[][] points) {
        int var1 = 1, prev = 0;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        for (int cur = 1; cur < points.length; cur++) {
            if (points[prev][1] < points[cur][0]) {
                var1++;
                prev = cur;
            }
        }
        return var1;
    }
}
