class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int count = 1;
        int prev = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > prev) {
                count++;
                prev = points[i][1];
            } else {
                prev = Math.min(points[i][1], prev);
            }
        }
        return count;
    }
}
