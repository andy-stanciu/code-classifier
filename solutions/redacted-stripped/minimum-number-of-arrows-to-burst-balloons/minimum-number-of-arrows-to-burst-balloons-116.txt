class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (p1, p2) -> Integer.compare(p1[1], p2[1]));
        int var1 = 1;
        int end = points[0][1];
        for (int[] point : points) {
            if (point[0] > end) {
                var1++;
                end = point[1];
            }
        }
        return var1;
    }
}
