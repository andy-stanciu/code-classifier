class Solution {

    public boolean method1(int[] nums) {
        int currentEnd = 0;
        int farthest = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            farthest = Math.max(farthest, nums[i] + i);
            if (i == currentEnd) {
                if (currentEnd >= farthest) {
                    return false;
                }
                currentEnd = farthest;
            }
        }
        return currentEnd >= nums.length - 1;
    }
}
