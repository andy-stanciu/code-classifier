class Solution {

    public boolean method1(int[] nums) {
        int target = nums.length - 1;
        boolean canReach = true;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] + i >= target) {
                target = i;
                canReach = true;
            } else {
                canReach = false;
            }
        }
        return canReach;
    }
}
