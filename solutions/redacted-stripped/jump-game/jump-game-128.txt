class Solution {

    public boolean method1(int[] nums) {
        int max = 0;
        int dp = 0;
        for (int i = 0; i <= max; i++) {
            if (i != 0)
                dp = Math.max(i + nums[i], dp);
            else
                dp = nums[i];
            max = Math.max(max, dp);
            if (dp >= nums.length - 1)
                return true;
        }
        return false;
    }
}
