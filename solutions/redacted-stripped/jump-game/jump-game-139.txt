class Solution {

    public boolean method1(int[] nums) {
        if (nums.length <= 1)
            return true;
        int steps = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] >= steps) {
                steps = 1;
            } else
                steps++;
        }
        return (steps <= 1);
    }
}
