class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1)
            return true;
        int p = 1;
        for (int i = 0; i < nums.length; i++) {
            if (p < 1)
                return false;
            p = Math.max(--p, nums[i]);
        }
        return true;
    }
}
