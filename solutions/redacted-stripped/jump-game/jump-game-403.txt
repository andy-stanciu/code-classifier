class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int stepsLeft = n - 1;
        for (int i = n - 2; i >= 0; i--) {
            if (i + nums[i] >= stepsLeft) {
                stepsLeft = i;
            }
        }
        return stepsLeft == 0;
    }
}
