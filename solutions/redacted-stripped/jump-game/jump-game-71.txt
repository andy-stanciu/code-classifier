class Solution {

    public boolean method1(int[] nums) {
        int goal = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            int reachableIndex = i + nums[i];
            if (reachableIndex >= goal) {
                goal = i;
            }
        }
        return goal == 0 ? true : false;
    }
}
