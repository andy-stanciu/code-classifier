class Solution {

    public boolean method1(int[] nums) {
        int reachAtLeast = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] + i >= reachAtLeast) {
                reachAtLeast = i;
            }
        }
        return reachAtLeast == 0;
    }
}
