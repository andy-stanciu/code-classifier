class Solution {

    public boolean method1(int[] nums) {
        int maxPos = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > maxPos)
                return false;
            maxPos = Math.max(i + nums[i], maxPos);
            if (maxPos >= nums.length - 1)
                return true;
        }
        return true;
    }
}
