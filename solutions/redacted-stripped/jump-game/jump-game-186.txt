class Solution {

    public boolean method1(int[] nums) {
        int var1 = 1;
        int i = nums.length - 2;
        while (i >= 0) {
            if (nums[i] >= var1) {
                var1 = 1;
            } else {
                var1++;
            }
            i--;
        }
        return var1 == 1;
    }
}
