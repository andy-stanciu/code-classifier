class Solution {

    public boolean method1(int[] nums) {
        int maxindex = nums[0];
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (maxindex >= n - 1)
                return true;
            if (nums[i] == 0 && maxindex == i)
                return false;
            maxindex = Math.max(maxindex, i + nums[i]);
        }
        return true;
    }
}
