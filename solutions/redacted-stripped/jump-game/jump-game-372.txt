class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1 || nums[0] >= nums.length) {
            return true;
        }
        int max = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                max = Math.max(max, i + nums[i]);
            } else if (nums[i] == 0 && max <= i && i != nums.length - 1) {
                return false;
            }
            if (max >= nums.length)
                return true;
        }
        return true;
    }
}
