class Solution {

    public boolean method1(int[] nums) {
        int size = nums.length;
        int goal = size - 1;
        for (int i = size - 1; i >= 0; i--) {
            if (i + nums[i] >= goal) {
                goal = i;
            }
        }
        if (goal == 0) {
            return true;
        }
        return false;
    }
}
