class Solution {

    public boolean method1(int[] nums) {
        boolean var1 = true;
        int indexZero;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] == 0) {
                indexZero = i;
                var1 = false;
                for (int j = i - 1; j >= 0; j--) {
                    if (nums[j] > indexZero - j) {
                        var1 = true;
                        i = j;
                        break;
                    }
                }
                if (!var1)
                    return false;
            }
        }
        return true;
    }
}
