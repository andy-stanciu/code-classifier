class Solution {

    public boolean method1(int[] nums) {
        int leftPointer = 0;
        int rightPointer = 0;
        int farthest = 0;
        while (rightPointer < nums.length - 1) {
            while (leftPointer <= rightPointer) {
                farthest = Math.max(farthest, leftPointer + nums[leftPointer]);
                leftPointer++;
            }
            if (farthest <= rightPointer)
                return false;
            rightPointer = farthest;
        }
        return true;
    }
}
