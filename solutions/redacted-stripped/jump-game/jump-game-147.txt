class Solution {

    public boolean method1(int[] nums) {
        int len = nums.length;
        if (len == 1) {
            return true;
        }
        boolean[] isReachableFromZero = new boolean[len];
        isReachableFromZero[0] = true;
        for (int i = 1; i < len; i++) {
            for (int j = i - 1; j >= 0; j--) {
                if (isReachableFromZero[j] && j + nums[j] >= i) {
                    isReachableFromZero[i] = true;
                    break;
                }
            }
        }
        return isReachableFromZero[len - 1];
    }
}
