class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        boolean[] reach = new boolean[n];
        reach[n - 1] = true;
        for (int i = n - 2; i >= 0; i--) {
            int curr_len = nums[i];
            for (int j = i + 1; j <= Math.min(i + curr_len, n - 1); j++) {
                if (reach[j] == true) {
                    reach[i] = true;
                }
            }
        }
        return reach[0];
    }
}
