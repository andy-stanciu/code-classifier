class Solution {

    public boolean method1(int[] nums) {
        int size = nums.length;
        if (size == 1)
            return true;
        int var1 = 0;
        for (int i = 0; i < size; i++) {
            if (var1 < nums[i])
                var1 = nums[i];
            if (var1 == 0)
                return false;
            if (var1 + i >= size - 1)
                return true;
            var1--;
        }
        return false;
    }
}
