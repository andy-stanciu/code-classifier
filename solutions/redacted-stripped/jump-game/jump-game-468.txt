class Solution {

    public boolean method1(int[] nums) {
        int var1 = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i == 0 && nums[i] < var1) {
                return false;
            }
            if (nums[i] < var1) {
                var1++;
            } else {
                var1 = 1;
            }
        }
        return true;
    }
}
