class Solution {

    public boolean method1(int[] nums) {
        int goal = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] >= goal) {
                goal = 1;
            } else {
                goal++;
            }
        }
        return (goal == 1) ? true : false;
    }
}
