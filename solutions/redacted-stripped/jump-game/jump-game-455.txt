class Solution {

    public boolean method1(int[] nums) {
        if (nums == null || nums.length <= 1)
            return true;
        int maxSoFar = 1;
        for (int i = 0; i < nums.length; i++) {
            maxSoFar--;
            maxSoFar = Math.max(maxSoFar, nums[i]);
            if (maxSoFar == 0 && i != nums.length - 1)
                return false;
        }
        return true;
    }
}
