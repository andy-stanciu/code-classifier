class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int maxind = 0;
        for (int i = 0; i < n; i++) {
            if (i > maxind) {
                return false;
            }
            maxind = Math.max(maxind, i + nums[i]);
        }
        return true;
    }
}
