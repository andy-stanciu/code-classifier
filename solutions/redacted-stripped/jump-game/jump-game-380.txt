class Solution {

    public boolean method1(int[] nums) {
        int max = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (max < i)
                return false;
            max = Math.max(max, nums[i] + i);
            if (max >= n - 1)
                return true;
        }
        return false;
    }
}
