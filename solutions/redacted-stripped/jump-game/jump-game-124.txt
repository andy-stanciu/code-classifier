class Solution {

    public boolean method1(int[] arr) {
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            if (count < i)
                return false;
            if (i + arr[i] > count) {
                count = i + arr[i];
            } else {
                count = count;
            }
        }
        return true;
    }
}
