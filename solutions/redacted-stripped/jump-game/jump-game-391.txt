class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (var1 <= 0 && nums[i] <= 0) {
                return false;
            }
            var1 = Math.max(var1, nums[i]);
            var1--;
        }
        return true;
    }
}
