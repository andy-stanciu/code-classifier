class Solution {

    public boolean method1(int[] nums) {
        int ind = 0;
        int n = nums.length;
        for (int i = 0; i < nums.length; i++) {
            if (i > ind) {
                return false;
            }
            ind = Math.max(ind, i + nums[i]);
            if (ind >= nums.length - 1) {
                return true;
            }
        }
        return false;
    }
}
