class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (var1 >= nums.length - 1)
                return true;
            if (i > var1)
                return false;
            if (i + nums[i] > var1) {
                var1 = nums[i] + i;
            }
        }
        return true;
    }
}
