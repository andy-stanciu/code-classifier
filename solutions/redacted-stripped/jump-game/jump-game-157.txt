class Solution {

    public boolean method1(int[] nums) {
        int i = 0;
        int maxInd = 0;
        int n = nums.length;
        while (i < n - 1) {
            maxInd = Math.max(maxInd, i + nums[i]);
            if (maxInd >= n - 1)
                return true;
            if (maxInd == i)
                return false;
            i++;
        }
        return true;
    }
}
