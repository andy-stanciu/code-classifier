class Solution {

    boolean[] visited;

    public boolean method1(int[] nums) {
        int len = nums.length;
        visited = new boolean[len];
        return method1(0, len, nums, visited);
    }

    private boolean method1(int ind, int len, int[] nums, boolean[] visited) {
        if (ind >= len - 1) {
            return true;
        }
        if (visited[ind]) {
            return false;
        }
        visited[ind] = true;
        int var1 = nums[ind];
        if (var1 == 0) {
            return false;
        }
        for (int i = var1; i >= 1; i--) {
            if (method1(ind + i, len, nums, visited)) {
                return true;
            }
        }
        return false;
    }
}
