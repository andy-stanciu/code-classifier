class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        int var1 = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < var1) {
                var1++;
            } else {
                var1 = 1;
            }
        }
        return nums[0] >= var1;
    }
}
