class Solution {

    public boolean method1(int[] nums) {
        if (nums.length <= 1)
            return true;
        int currentMax = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            boolean canReachI = currentMax >= i;
            if (!canReachI)
                return false;
            currentMax = Math.max(currentMax, i + nums[i]);
            boolean reachedEnd = currentMax >= nums.length - 1;
            if (reachedEnd)
                return true;
        }
        return false;
    }
}
