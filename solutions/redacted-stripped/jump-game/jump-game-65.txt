class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        return method2(0, nums);
    }

    public static boolean method2(int index, int[] nums) {
        if (index >= nums.length - 1)
            return true;
        for (int i = 1; i <= nums[index]; i++) {
            if (method2(index + i, nums) == true)
                return true;
        }
        return false;
    }
}
