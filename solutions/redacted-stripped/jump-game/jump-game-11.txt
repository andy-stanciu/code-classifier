class Solution {

    public boolean method1(int[] nums) {
        int maxReach = 0;
        for (int i = 0; i < nums.length; i++) {
            if (maxReach >= nums.length - 1)
                return true;
            if (i > maxReach)
                return false;
            if (i + nums[i] > maxReach) {
                maxReach = nums[i] + i;
            }
        }
        return true;
    }
}
