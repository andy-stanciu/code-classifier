class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            int var2 = nums[i] + i;
            var1 = Math.max(var1, var2);
            if (var1 < i + 1) {
                return false;
            }
        }
        return true;
    }
}
