class Solution {

    public boolean method1(int[] nums) {
        int val = 0, len = nums.length - 1;
        for (int i = 0; i <= len; i++) {
            if (val < i)
                return false;
            val = Math.max(val, nums[i] + i);
            if (val >= len)
                break;
        }
        return true;
    }
}
