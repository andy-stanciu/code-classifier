class Solution {

    public boolean method1(int[] nums) {
        return helper(nums, 0);
    }

    private boolean helper(int[] nums, int idx) {
        if (idx == nums.length - 1) {
            return true;
        }
        int distance = nums[idx];
        for (int i = 1; i <= distance; i++) {
            if (helper(nums, idx + i)) {
                return true;
            }
        }
        return false;
    }
}
