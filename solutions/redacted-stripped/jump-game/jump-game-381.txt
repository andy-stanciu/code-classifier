class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        int maxRight = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if ((nums[i] + i) >= (nums.length - 1))
                return true;
            maxRight = Math.max(maxRight, nums[i] + i);
            if (nums[i] == 0 && maxRight <= i) {
                return false;
            }
        }
        return false;
    }
}
