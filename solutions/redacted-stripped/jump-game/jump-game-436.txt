class Solution {

    public boolean method1(int[] nums) {
        int finalPosition = nums.length - 1;
        for (int index = nums.length - 2; index >= 0; index--) {
            if (index + nums[index] >= finalPosition)
                finalPosition = index;
        }
        if (finalPosition == 0)
            return true;
        return false;
    }
}
