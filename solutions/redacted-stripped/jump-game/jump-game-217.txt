class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > var1) {
                return false;
            }
            var1 = (var1 >= i + nums[i]) ? var1 : i + nums[i];
        }
        return true;
    }
}
