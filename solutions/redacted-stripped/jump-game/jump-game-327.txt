class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] >= var1) {
                var1 = 1;
            } else {
                var1++;
            }
        }
        return nums[0] >= var1;
    }
}
