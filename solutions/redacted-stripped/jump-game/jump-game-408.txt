class Solution {

    public boolean method1(int[] nums) {
        int goal_post = nums.length - 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            int var1 = i + nums[i];
            if (var1 >= goal_post) {
                goal_post = i;
            }
        }
        return (goal_post == 0) ? true : false;
    }
}
