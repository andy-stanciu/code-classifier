class Solution {

    public boolean method1(int[] nums) {
        int m = nums[0];
        int i = 0;
        while (m > 0 && i < nums.length - 1) {
            i++;
            m--;
            if (nums[i] > m) {
                m = nums[i];
            }
        }
        if (i == nums.length - 1) {
            return true;
        } else {
            return false;
        }
    }
}
