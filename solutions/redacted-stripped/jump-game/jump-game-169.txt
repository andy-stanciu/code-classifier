class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int i = 0;
        int max = 0;
        for (i = 0; i < n; i++) {
            max = Math.max(nums[i], max - 1);
            if (max < 1) {
                break;
            }
        }
        if (i >= n - 1)
            return true;
        else
            return false;
    }
}
