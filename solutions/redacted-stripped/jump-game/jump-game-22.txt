class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0, n = nums.length, i = 0;
        if (n <= 1)
            return true;
        while (var1 <= n && i < n - 1) {
            var1--;
            if (nums[i] == 0 && var1 <= 0)
                return false;
            if (nums[i] > var1)
                var1 = nums[i];
            i++;
        }
        return true;
    }
}
