class Solution {

    public boolean method1(int[] nums) {
        int reachableIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > reachableIndex) {
                return false;
            }
            reachableIndex = Math.max(reachableIndex, i + nums[i]);
            if (reachableIndex >= nums.length - 1) {
                return true;
            }
        }
        return true;
    }
}
