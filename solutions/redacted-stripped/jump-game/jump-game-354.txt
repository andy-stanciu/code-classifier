class Solution {

    public boolean method1(int[] nums) {
        int var1 = nums.length - 1;
        boolean found = false;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= var1) {
                var1 = i;
            }
        }
        if (var1 == 0) {
            return true;
        } else {
            found = false;
        }
        return found;
    }
}
