class Solution {

    public boolean method1(int[] nums) {
        int maxReach = 0;
        int lastIndex = -1;
        while (lastIndex < nums.length && maxReach > lastIndex) {
            lastIndex++;
            maxReach = Math.max(maxReach, lastIndex + nums[lastIndex]);
            if (maxReach >= nums.length - 1) {
                return true;
            }
        }
        return false;
    }
}
