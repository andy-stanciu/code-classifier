class Solution {

    public boolean method1(int[] nums) {
        int len = nums.length;
        int lastIndex, i;
        for (i = 0; i < len; i++) {
            nums[i] += i;
        }
        lastIndex = nums[0];
        for (i = 0; i <= lastIndex; i++) {
            if (nums[i] > lastIndex)
                lastIndex = nums[i];
            if (i == len - 1)
                return true;
        }
        return false;
    }
}
