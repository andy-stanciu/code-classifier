class Solution {

    public boolean method1(int[] nums) {
        int movesLeft = 0;
        int i = 0;
        while (i < nums.length - 1) {
            movesLeft--;
            int rechargableMoves = nums[i];
            movesLeft = Math.max(movesLeft, rechargableMoves);
            if (movesLeft <= 0)
                return false;
            i++;
        }
        return (i == nums.length - 1);
    }
}
