class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int final_pos = n - 1;
        for (int i = n - 2; i >= 0; i--) {
            if (i + nums[i] >= final_pos) {
                final_pos = i;
            }
        }
        return final_pos == 0;
    }
}
