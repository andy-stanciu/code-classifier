class Solution {

    public boolean method1(int[] nums) {
        int arrayLength = nums.length;
        int oneElementBeforelast = arrayLength - 1;
        if (arrayLength == 1) {
            return true;
        }
        if (nums[0] == 0) {
            return false;
        }
        int var1 = nums[0] + 0;
        for (int i = 1; i < oneElementBeforelast; i++) {
            if (i <= var1) {
                var1 = Math.max(i + nums[i], var1);
            }
        }
        return var1 >= oneElementBeforelast;
    }
}
