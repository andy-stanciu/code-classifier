class Solution {

    public boolean method1(int[] nums) {
        int reach = 0;
        for (int i = 0; i <= reach; i++) {
            if (nums[i] + i > reach) {
                reach = nums[i] + i;
            }
            if (reach >= nums.length - 1) {
                return true;
            }
        }
        return false;
    }
}
