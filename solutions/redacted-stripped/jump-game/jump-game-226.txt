class Solution {

    public boolean method1(int[] nums) {
        int lastReachableInd = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            int distanceFromLastReachableInd = lastReachableInd - i;
            if (distanceFromLastReachableInd <= nums[i]) {
                lastReachableInd = i;
            }
        }
        return lastReachableInd == 0;
    }
}
