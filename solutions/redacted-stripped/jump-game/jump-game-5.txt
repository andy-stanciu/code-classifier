class Solution {

    public boolean method1(int[] nums) {
        int covrage = 0, var1 = 0;
        if (nums.length == 1) {
            return true;
        }
        for (int i = 0; i < nums.length; i++) {
            covrage = Math.max(covrage, i + nums[i]);
            if (i == var1) {
                var1 = covrage;
                if (covrage >= nums.length - 1) {
                    return true;
                }
            }
        }
        return false;
    }
}
