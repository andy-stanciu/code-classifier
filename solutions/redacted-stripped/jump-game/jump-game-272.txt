class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1)
            return true;
        int var1 = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (var1 <= 0)
                return false;
            var1 = Math.max(var1 - 1, nums[i]);
        }
        return true;
    }
}
