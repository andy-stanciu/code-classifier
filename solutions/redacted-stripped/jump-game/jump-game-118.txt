class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1)
            return true;
        int n = nums.length;
        for (int i = 1; i < n; i++) {
            if (nums[i - 1] < i)
                return false;
            nums[i] = Math.max(nums[i - 1], i + nums[i]);
            if (nums[i] >= n - 1)
                return true;
        }
        return false;
    }
}
