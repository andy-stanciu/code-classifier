class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        if (nums.length == 1)
            return true;
        for (int i = 0; i < nums.length; i++) {
            int var2 = nums[i] + i;
            if (i <= var1) {
                var1 = Math.max(var1, var2);
                if (var1 >= nums.length - 1)
                    return true;
            }
        }
        return false;
    }
}
