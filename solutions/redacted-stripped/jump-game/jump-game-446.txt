class Solution {

    public boolean method1(int[] nums) {
        Stack<Integer> st = new Stack<>();
        st.push(nums.length - 1);
        for (int i = nums.length - 2; i >= 0; i--) {
            if (st.peek() - i <= nums[i]) {
                st.push(i);
            }
        }
        return (st.peek() == 0);
    }
}
