class Solution {

    public boolean method1(int[] nums) {
        boolean[] dp = new boolean[nums.length];
        dp[nums.length - 1] = true;
        for (int k = nums.length - 2; k >= 0; k--) {
            int var1 = Math.min(k + nums[k], nums.length - 1);
            for (int j = k + 1; j <= var1; j++) {
                if (dp[j] == true) {
                    dp[k] = true;
                    break;
                }
            }
        }
        return dp[0];
    }
}
