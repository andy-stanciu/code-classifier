class Solution {

    public boolean method1(int[] nums) {
        int reachable = nums.length - 1;
        int var1 = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            var1 = Math.max(nums[i] + i, var1);
            if (nums[i] == 0 && var1 == i) {
                return false;
            }
        }
        if (var1 < reachable) {
            return false;
        }
        return true;
    }
}
