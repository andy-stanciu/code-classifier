class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int reachable = 0;
        for (int i = 0; i < n; i++) {
            if (i > reachable) {
                return false;
            }
            reachable = Math.max(reachable, i + nums[i]);
            if (reachable >= n - 1) {
                return true;
            }
        }
        return false;
    }
}
