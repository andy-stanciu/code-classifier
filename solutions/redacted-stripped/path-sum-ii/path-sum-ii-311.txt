class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        solve(ans, root, param1, new ArrayList<>());
        return ans;
    }

    void solve(List<List<Integer>> ans, TreeNode root, int param1, List<Integer> currentList) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (param1 == root.val) {
                currentList.add(root.val);
                ans.add(new ArrayList<>(currentList));
                currentList.remove(currentList.size() - 1);
            }
            return;
        }
        currentList.add(root.val);
        solve(ans, root.left, param1 - root.val, currentList);
        solve(ans, root.right, param1 - root.val, currentList);
        currentList.remove(currentList.size() - 1);
    }
}
