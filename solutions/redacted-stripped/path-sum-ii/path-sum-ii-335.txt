class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> param2) {
        if (root == null) {
            return new LinkedList<>();
        }
        param2.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            ans.add(new LinkedList<>(param2));
        }
        if (root.left != null) {
            method1(root.left, param1 - root.val, ans, param2);
        }
        if (root.right != null) {
            method1(root.right, param1 - root.val, ans, param2);
        }
        param2.remove(param2.size() - 1);
        return ans;
    }

    public List<List<Integer>> method2(TreeNode root, int param1) {
        return method1(root, param1, new LinkedList<>(), new LinkedList<>());
    }
}
