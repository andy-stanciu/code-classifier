class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> l = new ArrayList<>();
        helper(root, param1, l);
        return ans;
    }

    public void helper(TreeNode node, int param2, List<Integer> l) {
        if (node == null) {
            return;
        }
        l.add(node.val);
        if (node.left == null && node.right == null && node.val == param2) {
            ans.add(new ArrayList<>(l));
        }
        this.helper(node.left, param2 - node.val, l);
        this.helper(node.right, param2 - node.val, l);
        l.remove(l.size() - 1);
    }
}
