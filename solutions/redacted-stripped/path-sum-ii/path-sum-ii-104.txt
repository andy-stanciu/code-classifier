class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return res;
        helper(root, param1, new ArrayList<>());
        return res;
    }

    void helper(TreeNode root, int param1, ArrayList<Integer> temp) {
        if (root == null && param1 == 0) {
            res.add(new ArrayList<>(temp));
            return;
        }
        if (root == null)
            return;
        temp.add(root.val);
        if (root.left == null)
            helper(root.right, param1 - root.val, temp);
        else if (root.right == null)
            helper(root.left, param1 - root.val, temp);
        else {
            helper(root.left, param1 - root.val, temp);
            helper(root.right, param1 - root.val, temp);
        }
        temp.remove(temp.size() - 1);
    }
}
