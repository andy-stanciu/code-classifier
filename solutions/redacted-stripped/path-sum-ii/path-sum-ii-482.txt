class Solution {

    List<List<Integer>> lists;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        lists = new ArrayList<>();
        recursion(root, 0, new ArrayList(), param1);
        return lists;
    }

    void recursion(TreeNode root, int param2, List<Integer> temp, int target) {
        if (root == null)
            return;
        if (param2 + root.val == target && (root != null && root.right == null && root.left == null)) {
            temp.add(root.val);
            lists.add(new ArrayList(temp));
            temp.remove(temp.size() - 1);
            return;
        }
        temp.add(root.val);
        recursion(root.left, param2 + root.val, temp, target);
        recursion(root.right, param2 + root.val, temp, target);
        temp.remove(temp.size() - 1);
    }
}
