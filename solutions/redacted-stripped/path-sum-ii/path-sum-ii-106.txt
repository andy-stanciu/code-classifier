class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> local = new ArrayList<>();
        helper(root, param1, result, local);
        return result;
    }

    public static void helper(TreeNode root, int param1, List<List<Integer>> result, List<Integer> local) {
        if (root == null) {
            return;
        }
        local.add(root.val);
        if ((root.left == null && root.right == null) && param1 == root.val) {
            result.add(new ArrayList<>(local));
        }
        helper(root.left, param1 - root.val, result, local);
        helper(root.right, param1 - root.val, result, local);
        local.remove(local.size() - 1);
    }
}
