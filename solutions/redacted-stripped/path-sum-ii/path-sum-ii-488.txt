class Solution {

    List<List<Integer>> lists = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(root, param1, 0, new ArrayList<>());
        return lists;
    }

    public void method2(TreeNode root, int param1, int initial, List<Integer> list) {
        if (root == null) {
            return;
        }
        initial = initial + root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && initial == param1)
            lists.add(new ArrayList<>(list));
        method2(root.left, param1, initial, list);
        method2(root.right, param1, initial, list);
        list.remove(list.size() - 1);
    }
}
