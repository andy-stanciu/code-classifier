class Solution {

    public void helper(TreeNode root, int param1, List<Integer> temp, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        param1 -= root.val;
        temp.add(root.val);
        if (root.left == null && root.right == null && param1 == 0) {
            res.add(new ArrayList<>(temp));
        }
        helper(root.left, param1, temp, res);
        helper(root.right, param1, temp, res);
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        helper(root, param1, temp, res);
        return res;
    }
}
