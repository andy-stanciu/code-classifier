class Solution {

    static List<List<Integer>> res;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        res = new ArrayList<>();
        method2(root, param1, new ArrayList<>());
        return res;
    }

    public void method2(TreeNode curr, int param2, List<Integer> list) {
        if (curr == null)
            return;
        if (curr.left == null && curr.right == null && param2 - curr.val == 0) {
            List<Integer> l = new ArrayList<>(list);
            l.add(curr.val);
            res.add(l);
        } else {
            list.add(curr.val);
            if (curr.left != null)
                method2(curr.left, param2 - curr.val, list);
            if (curr.right != null)
                method2(curr.right, param2 - curr.val, list);
            list.removeLast();
        }
    }
}
