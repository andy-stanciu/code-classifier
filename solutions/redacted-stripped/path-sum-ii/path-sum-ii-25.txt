class Solution {

    public void method1(TreeNode root, int param1, List<Integer> param2, List<List<Integer>> list) {
        if (root == null) {
            return;
        }
        param2.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            list.add(new ArrayList<Integer>(param2));
        }
        method1(root.left, (param1 - root.val), param2, list);
        method1(root.right, (param1 - root.val), param2, list);
        param2.remove(param2.size() - 1);
    }

    public List<List<Integer>> method2(TreeNode root, int param1) {
        List<List<Integer>> list = new ArrayList<>();
        method1(root, param1, new ArrayList<Integer>(), list);
        return list;
    }
}
