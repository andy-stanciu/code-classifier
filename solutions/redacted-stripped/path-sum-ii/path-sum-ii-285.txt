class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var1 = new ArrayList<Integer>();
        List<List<Integer>> ansr = new ArrayList<>();
        method2(root, param1, var1, ansr);
        return ansr;
    }

    public void method2(TreeNode root, int ts, List<Integer> var1, List<List<Integer>> ansr) {
        if (root == null)
            return;
        var1.add(root.val);
        if (root.val == ts && root.left == null && root.right == null) {
            ansr.add(new ArrayList<Integer>(var1));
        } else {
            method2(root.left, ts - root.val, var1, ansr);
            method2(root.right, ts - root.val, var1, ansr);
        }
        var1.remove(var1.size() - 1);
    }
}
