class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public void sol(TreeNode root, int param1, List<Integer> temp) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if (param1 == root.val) {
                List<Integer> newList = new ArrayList<>(temp);
                newList.add(root.val);
                ans.add(newList);
                return;
            }
        }
        temp.add(root.val);
        sol(root.left, param1 - root.val, temp);
        sol(root.right, param1 - root.val, temp);
        temp.remove(temp.size() - 1);
        return;
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> temp = new ArrayList<>();
        sol(root, param1, temp);
        return ans;
    }
}
