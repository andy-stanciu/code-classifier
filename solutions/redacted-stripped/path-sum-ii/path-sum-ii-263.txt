class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        recur(root, ans, param1, 0, temp);
        return ans;
    }

    public void recur(TreeNode root, List<List<Integer>> ans, int param1, int param2, List<Integer> temp) {
        if (root == null)
            return;
        param2 += root.val;
        temp.add(root.val);
        if (param2 == param1 && root.left == null && root.right == null) {
            List<Integer> list = new ArrayList<>();
            list.addAll(temp);
            ans.add(list);
        }
        recur(root.left, ans, param1, param2, temp);
        recur(root.right, ans, param1, param2, temp);
        temp.remove(temp.size() - 1);
    }
}
