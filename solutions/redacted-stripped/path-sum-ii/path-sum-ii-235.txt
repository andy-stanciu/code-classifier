class Solution {

    List<List<Integer>> ans;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        ans = new ArrayList<>();
        method1(root, param1, new ArrayList<>());
        return ans;
    }

    public void method1(TreeNode root, int param1, List<Integer> temp) {
        if (root == null)
            return;
        temp.add(root.val);
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            ans.add(new ArrayList<>(temp));
        }
        method1(root.left, param1 - root.val, new ArrayList<>(temp));
        method1(root.right, param1 - root.val, new ArrayList<>(temp));
    }
}
