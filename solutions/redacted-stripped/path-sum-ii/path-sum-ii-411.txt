class Solution {

    private List<List<Integer>> list;

    private int var1 = 0;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        list = new ArrayList();
        var1 = 0;
        method2(root, new LinkedList<Integer>(), param1);
        return list;
    }

    private void method2(TreeNode root, List<Integer> param2, int target) {
        if (root == null)
            return;
        var1 += root.val;
        param2.add(root.val);
        if (root.left == null && root.right == null && target == var1) {
            List<Integer> l = new ArrayList();
            for (int k : param2) l.add(k);
            list.add(l);
        }
        method2(root.left, param2, target);
        method2(root.right, param2, target);
        var1 -= root.val;
        param2.remove(param2.size() - 1);
    }
}
