class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        method1(root, param1, result, new ArrayList<>(), 0);
        return result;
    }

    public void method1(TreeNode root, int param1, List<List<Integer>> result, List<Integer> current, int param2) {
        if (root == null) {
            return;
        }
        current.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 + root.val == param1) {
                result.add(new ArrayList<>(current));
            }
        }
        method1(root.left, param1, result, current, param2 + root.val);
        method1(root.right, param1, result, current, param2 + root.val);
        current.remove(current.size() - 1);
    }
}
