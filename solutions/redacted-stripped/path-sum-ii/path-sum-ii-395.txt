class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(root, new ArrayList<Integer>(), 0, param1);
        return res;
    }

    private void method2(TreeNode node, List<Integer> list, int current, int param1) {
        if (node == null)
            return;
        current += node.val;
        list.add(node.val);
        if (node.left == null && node.right == null) {
            if (current == param1) {
                res.add(list);
            } else {
                return;
            }
        }
        if (node.left != null) {
            method2(node.left, new ArrayList<Integer>(list), current, param1);
        }
        if (node.right != null) {
            method2(node.right, new ArrayList<Integer>(list), current, param1);
        }
    }
}
