class Solution {

    List<List<Integer>> res = new ArrayList();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        backtrack(root, param1, new ArrayList<>(), 0);
        return res;
    }

    public void backtrack(TreeNode root, int s, ArrayList<Integer> al, int param2) {
        if (root == null)
            return;
        param2 += root.val;
        al.add(root.val);
        if (param2 == s && root.left == null && root.right == null) {
            res.add(new ArrayList<>(al));
        }
        backtrack(root.left, s, al, param2);
        backtrack(root.right, s, al, param2);
        al.remove(al.size() - 1);
    }
}
