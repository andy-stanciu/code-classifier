class Solution {

    private List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result = new ArrayList<>();
        helper(root, 0, param1, new ArrayList<>());
        return result;
    }

    private void helper(TreeNode root, int param2, int param1, List<Integer> param3) {
        if (root == null)
            return;
        param2 += root.val;
        param3.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                result.add(new ArrayList<>(param3));
            }
        }
        helper(root.left, param2, param1, param3);
        helper(root.right, param2, param1, param3);
        param3.remove(param3.size() - 1);
    }
}
