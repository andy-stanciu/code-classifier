class Solution {

    List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result = new ArrayList<>();
        dfs(root, param1, new ArrayList<>());
        return result;
    }

    void dfs(TreeNode root, int param1, List<Integer> param2) {
        if (root == null)
            return;
        param2.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            result.add(new ArrayList<>(param2));
        }
        dfs(root.left, param1 - root.val, param2);
        dfs(root.right, param1 - root.val, param2);
        param2.remove(param2.size() - 1);
    }
}
