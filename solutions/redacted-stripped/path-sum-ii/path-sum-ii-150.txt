class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        solve(root, param1, ans, cur);
        return ans;
    }

    public static void solve(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> cur) {
        if (root == null)
            return;
        cur.add(root.val);
        param1 -= root.val;
        if (param1 == 0 && root.left == null && root.right == null) {
            ans.add(new ArrayList<>(cur));
        }
        solve(root.left, param1, ans, cur);
        solve(root.right, param1, ans, cur);
        cur.remove(cur.size() - 1);
    }
}
