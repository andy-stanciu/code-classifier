class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        find(root, 0, param1, cur, res);
        return res;
    }

    void find(TreeNode root, int param2, int t, List<Integer> cur, List<List<Integer>> res) {
        if (root == null)
            return;
        param2 += root.val;
        cur.add(root.val);
        if (param2 == t && root.left == null && root.right == null)
            res.add(new ArrayList<>(cur));
        find(root.left, param2, t, cur, res);
        find(root.right, param2, t, cur, res);
        cur.remove(cur.size() - 1);
    }
}
