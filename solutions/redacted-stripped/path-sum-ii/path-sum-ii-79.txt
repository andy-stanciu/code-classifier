class Solution {

    private void method1(TreeNode root, List<List<Integer>> ans, List<Integer> list, int param1, int param2) {
        if (root == null) {
            return;
        }
        param2 = param2 + root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && param2 == param1) {
            ArrayList<Integer> l = new ArrayList<>(list);
            ans.add(l);
        }
        method1(root.left, ans, list, param1, param2);
        method1(root.right, ans, list, param1, param2);
        list.remove(list.size() - 1);
    }

    public List<List<Integer>> method2(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        method1(root, ans, list, param1, 0);
        return ans;
    }
}
