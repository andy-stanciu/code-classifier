class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        method2(root, param1, ans, new ArrayList<Integer>());
        return ans;
    }

    public void method2(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> temp) {
        if (root == null) {
            return;
        }
        temp.add(root.val);
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            ans.add(new ArrayList<>(temp));
        }
        method2(root.left, param1 - root.val, ans, temp);
        method2(root.right, param1 - root.val, ans, temp);
        temp.remove(temp.size() - 1);
    }
}
