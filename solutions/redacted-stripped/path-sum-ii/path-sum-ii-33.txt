class Solution {

    List<List<Integer>> list = new ArrayList<List<Integer>>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        go(root, 0, param1, new ArrayList<Integer>());
        return list;
    }

    private void go(TreeNode node, int param2, int target, List<Integer> lis) {
        if (node == null)
            return;
        List<Integer> newl = new ArrayList<Integer>(lis);
        newl.add(node.val);
        if (node.left != null)
            go(node.left, param2 + node.val, target, newl);
        if (node.right != null)
            go(node.right, param2 + node.val, target, newl);
        if (node.left == null && node.right == null && node.val + param2 == target)
            list.add(newl);
    }
}
