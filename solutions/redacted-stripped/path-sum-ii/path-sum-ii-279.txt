class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList();
        method1(root, param1, new ArrayList(), result);
        return result;
    }

    private void method1(TreeNode root, int param1, List<Integer> param2, List<List<Integer>> result) {
        if (root == null)
            return;
        param2.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            result.add(new ArrayList(param2));
        }
        method1(root.left, param1 - root.val, param2, result);
        method1(root.right, param1 - root.val, param2, result);
        param2.remove(param2.size() - 1);
    }
}
