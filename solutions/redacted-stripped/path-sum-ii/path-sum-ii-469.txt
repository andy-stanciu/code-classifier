class Solution {

    List<List<Integer>> result = new ArrayList();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result.clear();
        traverse(root, new ArrayList(), 0, param1);
        return result;
    }

    private void traverse(TreeNode root, List<Integer> list, int param2, int param1) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && (param2 + root.val) == param1) {
            result.add(new ArrayList(list));
            list.remove(list.size() - 1);
            return;
        }
        traverse(root.left, list, param2 + root.val, param1);
        traverse(root.right, list, param2 + root.val, param1);
        list.remove(list.size() - 1);
    }
}
