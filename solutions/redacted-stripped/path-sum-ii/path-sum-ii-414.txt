class Solution {

    public void helper(TreeNode root, int target, List<Integer> param1, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        param1.add(root.val);
        if (root.left == null && root.right == null && target - root.val == 0) {
            res.add(new ArrayList<>(param1));
            return;
        }
        if (root.left != null) {
            helper(root.left, target - root.val, param1, res);
            param1.remove(param1.size() - 1);
        }
        if (root.right != null) {
            helper(root.right, target - root.val, param1, res);
            param1.remove(param1.size() - 1);
        }
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> param1 = new ArrayList<>();
        helper(root, param2, param1, res);
        return res;
    }
}
