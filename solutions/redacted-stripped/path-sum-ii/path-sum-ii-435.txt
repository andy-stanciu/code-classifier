class Solution {

    List<List<Integer>> listOfList;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        listOfList = new ArrayList<List<Integer>>();
        List<Integer> list = new ArrayList<Integer>();
        if (root == null) {
            return listOfList;
        }
        method2(root, param1, 0, list);
        return listOfList;
    }

    public void method2(TreeNode root, int param1, int param2, List<Integer> list) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            int lc = param2 + root.val;
            if (lc == param1) {
                List<Integer> l = new ArrayList<Integer>();
                l.addAll(list);
                l.add(root.val);
                listOfList.add(l);
            }
        }
        list.add(root.val);
        method2(root.left, param1, param2 + root.val, list);
        method2(root.right, param1, param2 + root.val, list);
        list.remove(list.size() - 1);
    }
}
