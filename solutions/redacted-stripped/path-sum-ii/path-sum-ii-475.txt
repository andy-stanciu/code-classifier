class Solution {

    private List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(root, param1, new ArrayList<>());
        return ans;
    }

    private void method2(TreeNode root, int param1, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        param1 -= root.val;
        if (root.left == null && root.right == null) {
            if (param1 == 0)
                ans.add(list);
            return;
        }
        method2(root.left, param1, new ArrayList<>(list));
        method2(root.right, param1, new ArrayList<>(list));
    }
}
