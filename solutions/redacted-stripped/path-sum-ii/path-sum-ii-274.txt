class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> list = new ArrayList<>();
        method1(root, param1, 0, new ArrayList<>(), list);
        return list;
    }

    private void method1(TreeNode root, int param1, int total, List<Integer> current, List<List<Integer>> list) {
        if (root == null)
            return;
        if (root.val + total == param1 && root.left == null && root.right == null) {
            current.add(root.val);
            list.add(current);
            return;
        }
        current.add(root.val);
        method1(root.left, param1, total + root.val, new ArrayList<>(current), list);
        method1(root.right, param1, total + root.val, new ArrayList<>(current), list);
    }
}
