class Solution {

    public List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> list = new ArrayList<>();
        solve(root, param1, 0, list);
        return ans;
    }

    public void solve(TreeNode root, int k, int param2, List<Integer> list) {
        if (root == null)
            return;
        int m = root.val;
        list.add(m);
        param2 += m;
        if (root.left == null && root.right == null) {
            if (param2 == k) {
                ans.add(new ArrayList<>(list));
            }
        }
        solve(root.left, k, param2, list);
        solve(root.right, k, param2, list);
        list.remove(list.size() - 1);
        param2 -= m;
        return;
    }
}
