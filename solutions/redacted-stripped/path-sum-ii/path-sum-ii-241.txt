class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        method1(root, new LinkedList<Integer>(), res, param1);
        return res;
    }

    public void method1(TreeNode node, LinkedList<Integer> param2, List<List<Integer>> res, int target) {
        if (node == null)
            return;
        if (node.val == target && node.left == null && node.right == null) {
            LinkedList<Integer> var1 = new LinkedList<Integer>(param2);
            var1.add(node.val);
            res.add(var1);
        } else {
            param2.add(node.val);
            method1(node.left, param2, res, target - node.val);
            method1(node.right, param2, res, target - node.val);
            param2.removeLast();
        }
    }
}
