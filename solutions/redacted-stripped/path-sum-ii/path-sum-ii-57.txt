class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var1 = new ArrayList<>();
        List<List<Integer>> var2 = new ArrayList<>();
        method1(root, param1, var1, var2);
        return var2;
    }

    public void method1(TreeNode root, int param1, List<Integer> var1, List<List<Integer>> var2) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            var1.add(root.val);
            var2.add(new ArrayList<>(var1));
            var1.remove(var1.size() - 1);
            return;
        }
        var1.add(root.val);
        method1(root.left, param1 - root.val, var1, var2);
        method1(root.right, param1 - root.val, var1, var2);
        var1.remove(var1.size() - 1);
    }
}
