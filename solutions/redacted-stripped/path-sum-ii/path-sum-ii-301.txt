class Solution {

    public void solve(TreeNode root, int param1, int target, List<Integer> arr, List<List<Integer>> ans) {
        if (root == null)
            return;
        arr.add(root.val);
        if (root.left == null && root.right == null && param1 + root.val == target) {
            ans.add(new ArrayList<>(arr));
        }
        solve(root.left, param1 + root.val, target, arr, ans);
        solve(root.right, param1 + root.val, target, arr, ans);
        arr.remove(arr.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        List<List<Integer>> result = new ArrayList<>();
        solve(root, 0, param2, new ArrayList<>(), result);
        return result;
    }
}
