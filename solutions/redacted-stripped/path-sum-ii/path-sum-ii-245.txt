class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> list = new ArrayList<>();
        helper(root, 0, param1, new ArrayList<Integer>(), list);
        return list;
    }

    private void helper(TreeNode root, int param2, int target, List<Integer> curr, List<List<Integer>> list) {
        if (root == null)
            return;
        param2 += root.val;
        curr.add(root.val);
        if (param2 == target && root.left == root.right) {
            list.add(new ArrayList<>(curr));
            curr.remove(curr.size() - 1);
            return;
        }
        helper(root.left, param2, target, curr, list);
        helper(root.right, param2, target, curr, list);
        curr.remove(curr.size() - 1);
    }
}
