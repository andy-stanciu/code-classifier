class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        helper(new ArrayList<>(), root, 0, param1, result);
        return result;
    }

    public void helper(List<Integer> param2, TreeNode node, int param3, int target, List<List<Integer>> result) {
        if (node == null) {
            return;
        }
        param2.add(node.val);
        param3 += node.val;
        if (node.left == null && node.right == null && param3 == target) {
            result.add(new ArrayList<>(param2));
        }
        helper(param2, node.left, param3, target, result);
        helper(param2, node.right, param3, target, result);
        param2.remove(param2.size() - 1);
    }
}
