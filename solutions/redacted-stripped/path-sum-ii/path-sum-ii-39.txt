class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> ls = new ArrayList<>();
        helper(root, param1, ls, res);
        return res;
    }

    void helper(TreeNode root, int param2, List<Integer> ls, List<List<Integer>> res) {
        if (root == null)
            return;
        ls.add(root.val);
        if (root.val == param2 && root.left == null && root.right == null) {
            res.add(new ArrayList<>(ls));
        }
        helper(root.left, param2 - root.val, ls, res);
        helper(root.right, param2 - root.val, ls, res);
        ls.remove(ls.size() - 1);
    }
}
