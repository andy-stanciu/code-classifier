class Solution {

    static List<List<Integer>> ans;

    public static void solve(TreeNode root, int param1, List<Integer> temp, int param2) {
        if (root == null) {
            return;
        }
        param2 = param2 + root.val;
        temp.add(root.val);
        solve(root.left, param1, temp, param2);
        solve(root.right, param1, temp, param2);
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                ans.add(new ArrayList<>(temp));
            }
        }
        param2 = param2 - root.val;
        temp.remove(temp.size() - 1);
        return;
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        int param2 = 0;
        solve(root, param1, temp, param2);
        return ans;
    }
}
