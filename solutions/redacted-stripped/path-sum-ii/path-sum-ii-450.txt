class Solution {

    public void findAns(TreeNode root, List<List<Integer>> res, ArrayList<Integer> list, int target) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            list.add(root.val);
            int var1 = 0;
            for (Integer ele : list) {
                var1 += ele;
            }
            if (var1 == target) {
                res.add(new ArrayList<>(list));
            }
            list.remove(list.size() - 1);
            return;
        }
        list.add(root.val);
        findAns(root.left, res, list, target);
        findAns(root.right, res, list, target);
        list.remove(list.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        ArrayList<Integer> list = new ArrayList<>();
        findAns(root, res, list, param1);
        return res;
    }
}
