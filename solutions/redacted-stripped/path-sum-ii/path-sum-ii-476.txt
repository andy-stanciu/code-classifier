class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null)
            return res;
        recur(res, new ArrayList<Integer>(), 0, param1, root);
        return res;
    }

    private void recur(List<List<Integer>> res, ArrayList<Integer> al, int param2, int param1, TreeNode root) {
        if (root == null)
            return;
        al.add(root.val);
        param2 += root.val;
        if (param2 == param1 && root.left == null && root.right == null)
            res.add(new ArrayList<>(al));
        recur(res, al, param2, param1, root.left);
        recur(res, al, param2, param1, root.right);
        al.remove(al.size() - 1);
    }
}
