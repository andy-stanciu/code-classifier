class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        method1(root, param1, new ArrayList<Integer>(), res);
        return res;
    }

    void method1(TreeNode root, int param2, List<Integer> sol, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        sol.add(root.val);
        if (root.left == null && root.right == null && param2 == root.val) {
            res.add(new ArrayList<Integer>(sol));
        } else {
            method1(root.left, param2 - root.val, sol, res);
            method1(root.right, param2 - root.val, sol, res);
        }
        sol.remove(sol.size() - 1);
    }
}
