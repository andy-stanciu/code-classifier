class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return res;
        preorder(root, param1, new ArrayList<>());
        return res;
    }

    void preorder(TreeNode root, int target, List<Integer> list) {
        list.add(root.val);
        if (root.left == null && root.right == null && root.val == target)
            res.add(list);
        if (root.left != null)
            preorder(root.left, target - root.val, new ArrayList<>(list));
        if (root.right != null)
            preorder(root.right, target - root.val, new ArrayList<>(list));
    }
}
