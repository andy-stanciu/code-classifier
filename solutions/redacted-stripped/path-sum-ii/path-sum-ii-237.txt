class Solution {

    public void method1(List<List<Integer>> list, List<Integer> subList, TreeNode root, int target) {
        if (root == null)
            return;
        subList.add(root.val);
        if (target == root.val && root.left == null && root.right == null) {
            list.add(new ArrayList<>(subList));
        } else {
            method1(list, subList, root.left, target - root.val);
            method1(list, subList, root.right, target - root.val);
        }
        subList.remove(subList.size() - 1);
    }

    public List<List<Integer>> method2(TreeNode root, int param1) {
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> subList = new ArrayList<>();
        method1(list, subList, root, param1);
        return list;
    }
}
