class Solution {

    List<List<Integer>> fal = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return fal;
        helper(new ArrayList<Integer>(), root, param1);
        return fal;
    }

    private void helper(ArrayList<Integer> al, TreeNode root, int param1) {
        if (root == null)
            return;
        al.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            fal.add(new ArrayList<Integer>(al));
        } else {
            helper(al, root.left, param1 - root.val);
            helper(al, root.right, param1 - root.val);
        }
        al.remove(al.size() - 1);
    }
}
