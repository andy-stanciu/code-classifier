class Solution {

    List<List<Integer>> var1 = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return var1;
        dfs(root, param1, new ArrayList<>());
        return var1;
    }

    void dfs(TreeNode node, int param1, ArrayList<Integer> param2) {
        if (node == null)
            return;
        param2.add(node.val);
        param1 -= node.val;
        if (param1 == 0 && node.left == null && node.right == null)
            var1.add(param2);
        dfs(node.left, param1, new ArrayList<>(param2));
        dfs(node.right, param1, new ArrayList<>(param2));
    }
}
