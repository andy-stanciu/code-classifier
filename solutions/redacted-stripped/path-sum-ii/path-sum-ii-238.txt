class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        method2(root, result, var1, param1, 0);
        return result;
    }

    private void method2(TreeNode root, List<List<Integer>> result, List<Integer> var1, int target, int method2) {
        if (root == null)
            return;
        var1.add(root.val);
        method2 += root.val;
        if (root.left == null && root.right == null && method2 == target)
            result.add(new ArrayList<>(var1));
        method2(root.left, result, var1, target, method2);
        method2(root.right, result, var1, target, method2);
        var1.removeLast();
    }
}
