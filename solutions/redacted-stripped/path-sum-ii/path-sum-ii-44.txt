class Solution {

    List<List<Integer>> l = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> list = new ArrayList<>();
        helper(root, param1, 0, list);
        return l;
    }

    public void helper(TreeNode root, int k, int param2, List<Integer> list) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        param2 += root.val;
        if (root.left == null && root.right == null) {
            if (param2 == k) {
                l.add(new ArrayList<>(list));
            }
        }
        helper(root.left, k, param2, list);
        helper(root.right, k, param2, list);
        list.remove(list.size() - 1);
    }
}
