class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> lt = new ArrayList<>();
        int var1 = 0;
        if (root != null) {
            var1 = root.val;
            lt.add(root.val);
            method2(root, param1, var1, ans, lt);
        }
        return ans;
    }

    public void method2(TreeNode root, int target, int var1, List<List<Integer>> ans, List<Integer> lt) {
        if (var1 == target && root.left == null && root.right == null) {
            ans.add(new ArrayList<>(lt));
            return;
        }
        if (root.left != null) {
            var1 += root.left.val;
            lt.add(root.left.val);
            method2(root.left, target, var1, ans, lt);
            var1 -= root.left.val;
            lt.remove(lt.size() - 1);
        }
        if (root.right != null) {
            var1 += root.right.val;
            lt.add(root.right.val);
            method2(root.right, target, var1, ans, lt);
            var1 -= root.right.val;
            lt.remove(lt.size() - 1);
        }
    }
}
