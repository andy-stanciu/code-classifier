class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        ArrayList<Integer> results = new ArrayList<Integer>();
        List<List<Integer>> finalResults = new ArrayList<List<Integer>>();
        return method2(root, param1, 0, results, finalResults);
    }

    public List<List<Integer>> method2(TreeNode root, int param1, int param2, ArrayList<Integer> results, List<List<Integer>> finalResults) {
        if (root == null) {
            return new ArrayList<List<Integer>>();
        }
        results.add(root.val);
        param2 += root.val;
        if (param2 == param1 && root.left == null && root.right == null) {
            finalResults.add(new ArrayList<Integer>(results));
        }
        method2(root.left, param1, param2, new ArrayList<Integer>(results), finalResults);
        method2(root.right, param1, param2, new ArrayList<Integer>(results), finalResults);
        return finalResults;
    }
}
