class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> list = new ArrayList<>();
        List<List<Integer>> listOfLists = new ArrayList<>();
        method2(root, param1, list, listOfLists);
        return listOfLists;
    }

    public void method2(TreeNode root, int param1, List<Integer> list, List<List<Integer>> listOfLists) {
        if (root == null)
            return;
        param1 -= root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && param1 == 0) {
            listOfLists.add(list);
            return;
        }
        method2(root.left, param1, new ArrayList<>(list), listOfLists);
        method2(root.right, param1, new ArrayList<>(list), listOfLists);
    }
}
