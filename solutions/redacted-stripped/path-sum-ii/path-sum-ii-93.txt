class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null)
            return ans;
        method1(root, ans, new ArrayList<>(), param1);
        return ans;
    }

    public void method1(TreeNode root, List<List<Integer>> ans, List<Integer> tempSet, int param1) {
        if (root == null)
            return;
        tempSet.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            ans.add(new ArrayList<>(tempSet));
        } else {
            method1(root.left, ans, tempSet, param1 - root.val);
            method1(root.right, ans, tempSet, param1 - root.val);
        }
        tempSet.remove(tempSet.size() - 1);
    }
}
