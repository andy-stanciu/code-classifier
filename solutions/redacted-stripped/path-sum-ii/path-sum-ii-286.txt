class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        helper(root, param1, new ArrayList<Integer>(), result);
        return result;
    }

    void helper(TreeNode root, int target, List<Integer> list, List<List<Integer>> result) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && target == root.val) {
            result.add(new ArrayList<>(list));
            result.get(result.size() - 1).add(root.val);
            return;
        }
        list.add(root.val);
        helper(root.left, target - root.val, list, result);
        helper(root.right, target - root.val, list, result);
        list.remove(list.size() - 1);
    }
}
