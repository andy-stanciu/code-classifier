class Solution {

    int var1 = 0;

    Stack<Integer> stack = new Stack();

    List<List<Integer>> result = new ArrayList();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return result;
        method2(root, param1);
        return result;
    }

    private void method2(TreeNode root, int param1) {
        if (root.left == null && root.right == null) {
            stack.push(root.val);
            var1 += root.val;
            if (var1 == param1) {
                List<Integer> list = new ArrayList(stack);
                result.add(list);
            }
            return;
        }
        var1 += root.val;
        stack.push(root.val);
        if (root.left != null) {
            method2(root.left, param1);
            var1 -= stack.pop();
        }
        if (root.right != null) {
            method2(root.right, param1);
            var1 -= stack.pop();
        }
    }
}
