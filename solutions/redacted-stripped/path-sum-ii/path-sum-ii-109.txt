class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> lis = new ArrayList<>();
        helper(root, param1, lis, 0);
        return res;
    }

    public void helper(TreeNode root, int target, List<Integer> arr, int param2) {
        if (root == null) {
            return;
        }
        arr.add(root.val);
        param2 += root.val;
        if (root.left == null && root.right == null && param2 == target) {
            res.add(arr);
        }
        helper(root.left, target, new ArrayList(arr), param2);
        helper(root.right, target, new ArrayList(arr), param2);
    }
}
