class Solution {

    public static void PthToTarget(List<List<Integer>> arr, List<Integer> ar, TreeNode root, int target, int param1) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if (root.val + param1 == target) {
                ar.add(root.val);
                List<Integer> l = new ArrayList<>();
                l.addAll(ar);
                arr.add(l);
                ar.remove(ar.size() - 1);
            }
            return;
        }
        ar.add(root.val);
        PthToTarget(arr, ar, root.left, target, param1 + root.val);
        PthToTarget(arr, ar, root.right, target, param1 + root.val);
        ar.remove(ar.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        List<List<Integer>> arr = new ArrayList<>();
        List<Integer> ar = new ArrayList<>();
        PthToTarget(arr, ar, root, param2, 0);
        return arr;
    }
}
