class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<List<Integer>>();
        bt(root, param1, new ArrayList<Integer>(), ans);
        return ans;
    }

    public void bt(TreeNode root, int param2, List<Integer> param3, List<List<Integer>> ans) {
        if (root == null) {
            return;
        }
        param3.add(root.val);
        if (root.left == null && root.right == null && root.val == param2) {
            ans.add(new ArrayList<>(param3));
        } else {
            bt(root.left, param2 - root.val, param3, ans);
            bt(root.right, param2 - root.val, param3, ans);
        }
        param3.remove(param3.size() - 1);
    }
}
