class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        List<Integer> tmp = new ArrayList<>();
        method2(root, result, tmp, 0, param1);
        return result;
    }

    public static void method2(TreeNode root, List<List<Integer>> result, List<Integer> tmp, int param2, int param1) {
        if (root == null) {
            return;
        }
        tmp.add(root.val);
        if (root.left == null && root.right == null && root.val + param2 == param1) {
            result.add(new ArrayList<>(tmp));
        }
        method2(root.left, result, tmp, param2 + root.val, param1);
        method2(root.right, result, tmp, param2 + root.val, param1);
        tmp.remove(tmp.size() - 1);
        return;
    }
}
