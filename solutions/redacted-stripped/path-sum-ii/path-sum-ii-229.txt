class Solution {

    List<List<Integer>> var1 = new ArrayList<>();

    int i = 0;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        int ans = 0;
        List<Integer> var2 = new ArrayList<>();
        method2(root, param1, ans, var2);
        return var1;
    }

    public void method2(TreeNode root, int target, int ans, List<Integer> var2) {
        if (root == null)
            return;
        var2.add(root.val);
        if (root.left == null && root.right == null) {
            ans += root.val;
            if (ans == target) {
                var1.add(new ArrayList<>(var2));
            }
        }
        method2(root.left, target, ans + root.val, var2);
        method2(root.right, target, ans + root.val, var2);
        var2.remove(var2.size() - 1);
    }
}
