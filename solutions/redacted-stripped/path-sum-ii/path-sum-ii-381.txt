class Solution {

    private List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        dfs(root, new ArrayList<>(), param1);
        return result;
    }

    private void dfs(TreeNode root, List<Integer> list, int param2) {
        if (root == null)
            return;
        param2 -= root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && param2 == 0) {
            result.add(new ArrayList<>(list));
        }
        dfs(root.left, list, param2);
        dfs(root.right, list, param2);
        list.remove(list.size() - 1);
    }
}
