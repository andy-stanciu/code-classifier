class Solution {

    List<List<Integer>> answer = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(root, param1, 0, new ArrayList<>());
        return answer;
    }

    void helper(TreeNode root, int param1, int param2, List<Integer> param3) {
        if (root == null)
            return;
        param2 += root.val;
        param3.add(root.val);
        if (root.left == null && root.right == null && param1 == param2) {
            answer.add(new ArrayList<>(param3));
        }
        helper(root.left, param1, param2, param3);
        helper(root.right, param1, param2, param3);
        param3.remove(param3.size() - 1);
    }
}
