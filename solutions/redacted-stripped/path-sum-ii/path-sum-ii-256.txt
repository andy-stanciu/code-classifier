class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        method1(root, param1, list, res, 0);
        return res;
    }

    public void method1(TreeNode root, int target, List<Integer> list, List<List<Integer>> res, int param2) {
        if (root == null) {
            return;
        }
        param2 += root.val;
        list.add(root.val);
        if (param2 == target && root.left == null && root.right == null) {
            res.add(new ArrayList<>(list));
        }
        method1(root.left, target, list, res, param2);
        method1(root.right, target, list, res, param2);
        list.remove(list.size() - 1);
    }
}
