class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        method2(root, param1, new ArrayList<Integer>(), res);
        return res;
    }

    void method2(TreeNode root, int param1, List<Integer> sol, List<List<Integer>> res) {
        if (root == null)
            return;
        sol.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            res.add(new ArrayList<Integer>(sol));
        } else {
            method2(root.left, param1 - root.val, sol, res);
            method2(root.right, param1 - root.val, sol, res);
        }
        sol.remove(sol.size() - 1);
    }
}
