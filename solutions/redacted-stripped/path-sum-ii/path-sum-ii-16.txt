class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        method1(root, param1, new ArrayList(), res);
        return res;
    }

    public void method1(TreeNode root, int param2, List<Integer> param3, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        param3.add(root.val);
        if (root.left == null && root.right == null && param2 == root.val) {
            res.add(new ArrayList<>(param3));
        }
        method1(root.left, param2 - root.val, param3, res);
        method1(root.right, param2 - root.val, param3, res);
        param3.remove(param3.size() - 1);
    }
}
