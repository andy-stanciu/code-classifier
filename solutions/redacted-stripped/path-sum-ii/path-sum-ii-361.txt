class Solution {

    private List<List<Integer>> res = new LinkedList<>();

    private LinkedList<Integer> var1 = new LinkedList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return res;
        var1.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            res.add(new LinkedList<>(var1));
        }
        method1(root.left, param1 - root.val);
        method1(root.right, param1 - root.val);
        var1.removeLast();
        return res;
    }
}
