class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null)
            return ans;
        calculate(root, param1, ans, new ArrayList<>());
        return ans;
    }

    public void calculate(TreeNode root, int param2, List<List<Integer>> ans, List<Integer> ds) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (param2 == root.val) {
                ds.add(root.val);
                ans.add(new ArrayList<>(ds));
                ds.remove(ds.size() - 1);
            }
            return;
        }
        ds.add(root.val);
        calculate(root.left, param2 - root.val, ans, ds);
        calculate(root.right, param2 - root.val, ans, ds);
        ds.remove(ds.size() - 1);
    }
}
