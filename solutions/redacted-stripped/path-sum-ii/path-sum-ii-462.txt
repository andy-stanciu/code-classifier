class Solution {

    List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result = new ArrayList<>();
        List<TreeNode> l1 = new ArrayList<>();
        preorder(root, param1, 0, l1);
        return result;
    }

    public void preorder(TreeNode root, int param1, int temp, List<TreeNode> l1) {
        if (root == null) {
            return;
        }
        temp += root.val;
        l1.add(root);
        if (temp == param1 && l1.get(l1.size() - 1).left == null && l1.get(l1.size() - 1).right == null) {
            List<Integer> var1 = new ArrayList<>();
            for (TreeNode n : l1) {
                var1.add(n.val);
            }
            result.add(var1);
        }
        preorder(root.left, param1, temp, l1);
        preorder(root.right, param1, temp, l1);
        l1.remove(l1.size() - 1);
    }
}
