class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        dfs(root, param1, res, new ArrayList<>());
        return res;
    }

    private void dfs(TreeNode root, int param1, List<List<Integer>> res, List<Integer> tempList) {
        if (root == null) {
            return;
        }
        param1 = param1 - root.val;
        if (root.left == null && root.right == null) {
            if (param1 == 0) {
                tempList.add(root.val);
                res.add(new ArrayList<>(tempList));
                tempList.remove(tempList.size() - 1);
            }
        }
        tempList.add(root.val);
        dfs(root.left, param1, res, tempList);
        dfs(root.right, param1, res, tempList);
        tempList.remove(tempList.size() - 1);
    }
}
