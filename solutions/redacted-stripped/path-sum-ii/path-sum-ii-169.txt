class Solution {

    private List<List<Integer>> var1 = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        backtracking(root, param1, 0, new LinkedList<>());
        return var1;
    }

    private void backtracking(TreeNode root, int param1, int param2, List<Integer> param3) {
        if (root == null) {
            return;
        }
        param2 += root.val;
        param3.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                var1.add(new ArrayList<>(param3));
            }
        }
        backtracking(root.left, param1, param2, param3);
        backtracking(root.right, param1, param2, param3);
        param3.removeLast();
    }
}
