class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        solve(root, param1, 0, result, list);
        return result;
    }

    public static void solve(TreeNode root, int param1, int param2, List<List<Integer>> result, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        param2 += root.val;
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                result.add(new ArrayList<>(list));
            }
        }
        solve(root.left, param1, param2, result, list);
        solve(root.right, param1, param2, result, list);
        list.remove(list.size() - 1);
    }
}
