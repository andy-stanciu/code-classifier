class Solution {

    List<List<Integer>> res = new ArrayList<List<Integer>>();

    int target = 0;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        target = param1;
        traverse(root, 0, new ArrayList<Integer>());
        return res;
    }

    public void traverse(TreeNode node, int param2, ArrayList<Integer> nums) {
        if (node == null) {
            return;
        }
        nums.add(node.val);
        if (node.left == null && node.right == null) {
            if (param2 + node.val == target) {
                res.add(nums);
            }
            return;
        }
        if (node.left != null) {
            traverse(node.left, param2 + node.val, new ArrayList<Integer>(nums));
        }
        if (node.right != null) {
            traverse(node.right, param2 + node.val, new ArrayList<Integer>(nums));
        }
    }
}
