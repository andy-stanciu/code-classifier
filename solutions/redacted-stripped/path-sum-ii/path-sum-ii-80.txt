class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        dfs(root, param1, var1, result);
        return result;
    }

    private void dfs(TreeNode node, int param2, List<Integer> var1, List<List<Integer>> result) {
        if (node == null) {
            return;
        }
        var1.add(node.val);
        if (node.left == null && node.right == null && param2 == node.val) {
            result.add(new ArrayList<>(var1));
        } else {
            dfs(node.left, param2 - node.val, var1, result);
            dfs(node.right, param2 - node.val, var1, result);
        }
        var1.remove(var1.size() - 1);
    }
}
