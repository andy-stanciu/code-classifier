class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(root, 0, param1, new ArrayList<>());
        return result;
    }

    public void helper(TreeNode root, int param2, int param1, List<Integer> param3) {
        if (root == null) {
            return;
        }
        param3.add(root.val);
        if (root.left == null && root.right == null && (param2 + root.val) == param1) {
            result.add(param3);
        } else {
            helper(root.left, param2 + root.val, param1, new ArrayList<>(param3));
            helper(root.right, param2 + root.val, param1, new ArrayList<>(param3));
        }
    }
}
