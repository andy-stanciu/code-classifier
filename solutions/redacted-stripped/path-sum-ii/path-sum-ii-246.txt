class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null) {
            return ans;
        }
        method2(root, param1, new ArrayList<>());
        return ans;
    }

    public void method2(TreeNode node, int targetRemaining, List<Integer> list) {
        if (node == null) {
            return;
        }
        list.add(node.val);
        if (node.left == null && node.right == null && targetRemaining == node.val) {
            ans.add(new ArrayList<>(list));
        } else {
            method2(node.left, targetRemaining - node.val, list);
            method2(node.right, targetRemaining - node.val, list);
        }
        list.remove(list.size() - 1);
    }
}
