class Solution {

    private void method1(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> arr) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (param1 - root.val == 0) {
                arr.add(root.val);
                ans.add(new ArrayList(arr));
                arr.remove(arr.size() - 1);
            }
            return;
        }
        arr.add(root.val);
        method1(root.left, param1 - root.val, ans, arr);
        method1(root.right, param1 - root.val, ans, arr);
        arr.remove(arr.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> arr = new ArrayList<>();
        method1(root, param1, ans, arr);
        return ans;
    }
}
