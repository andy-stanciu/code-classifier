class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<List<Integer>>();
        if (root == null)
            return ans;
        method2(root, ans, param1, new ArrayList<Integer>());
        return ans;
    }

    public void method2(TreeNode root, List<List<Integer>> ans, int param1, List<Integer> sub) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            sub.add(root.val);
            int var1 = 0;
            System.out.println(sub);
            for (int i = 0; i < sub.size(); i++) {
                var1 = var1 + sub.get(i);
            }
            System.out.println(var1);
            if (var1 == param1)
                ans.add(new ArrayList<Integer>(sub));
        }
        sub.add(root.val);
        method2(root.left, ans, param1, new ArrayList<Integer>(sub));
        method2(root.right, ans, param1, new ArrayList<Integer>(sub));
    }
}
