class Solution {

    public static void method1(TreeNode root, int param1, List<List<Integer>> result, List<Integer> temp) {
        if (root == null)
            return;
        temp.add(root.val);
        if (root.left == null && root.right == null) {
            param1 -= root.val;
            if (param1 == 0) {
                result.add(new ArrayList<>(temp));
            }
        }
        method1(root.left, param1 - root.val, result, temp);
        method1(root.right, param1 - root.val, result, temp);
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> method2(TreeNode root, int param1) {
        if (root == null)
            return new ArrayList<>();
        List<List<Integer>> result = new ArrayList<>();
        method1(root, param1, result, new ArrayList<>());
        return result;
    }
}
