class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        method2(root, param1, result, var1, 0);
        return result;
    }

    private void method2(TreeNode root, int param1, List<List<Integer>> result, List<Integer> var1, int param2) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param2 + root.val == param1) {
            result.add(var1);
        }
        var1.add(root.val);
        method2(root.left, param1, result, new ArrayList<>(var1), param2 + root.val);
        method2(root.right, param1, result, new ArrayList<>(var1), param2 + root.val);
    }
}
