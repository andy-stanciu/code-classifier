class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    int target;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        target = param1;
        if (root == null) {
            return ans;
        }
        helper(root, new ArrayList<>(), root.val);
        return ans;
    }

    public void helper(TreeNode root, ArrayList<Integer> al, int param2) {
        ArrayList<Integer> temp = new ArrayList<>(al);
        temp.add(root.val);
        if (target == param2 && root.left == null && root.right == null) {
            System.out.println(temp + " " + param2);
            ans.add(new ArrayList<Integer>(temp));
        }
        if (root.left != null) {
            helper(root.left, temp, param2 + root.left.val);
        }
        if (root.right != null) {
            helper(root.right, temp, param2 + root.right.val);
        }
    }
}
