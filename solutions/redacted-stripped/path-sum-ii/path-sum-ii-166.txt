class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> output = new ArrayList<>();
        method1(output, root, param1, new ArrayList<>(), 0);
        return output;
    }

    public void method1(List<List<Integer>> output, TreeNode root, int param1, List<Integer> temp, int param2) {
        if (root == null)
            return;
        temp.add(root.val);
        param2 += root.val;
        if (root.left == null && root.right == null && param2 == param1)
            output.add(new ArrayList<>(temp));
        method1(output, root.left, param1, temp, param2);
        method1(output, root.right, param1, temp, param2);
        temp.remove(temp.size() - 1);
    }
}
