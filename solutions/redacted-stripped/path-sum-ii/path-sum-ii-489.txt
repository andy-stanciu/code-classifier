class Solution {

    List<List<Integer>> ans = new LinkedList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var1 = new LinkedList<>();
        if (root == null)
            return ans;
        helper(root, param1, var1);
        return ans;
    }

    public void helper(TreeNode node, int target, List<Integer> var1) {
        if (node == null)
            return;
        var1.add(node.val);
        if (node.left == null && node.right == null && node.val == target) {
            ans.add(new LinkedList<>(var1));
        }
        if (node.left != null) {
            helper(node.left, target - node.val, var1);
        }
        if (node.right != null) {
            helper(node.right, target - node.val, var1);
        }
        var1.remove(var1.size() - 1);
    }
}
