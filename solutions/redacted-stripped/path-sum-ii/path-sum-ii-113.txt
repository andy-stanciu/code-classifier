class Solution {

    List<List<Integer>> ans;

    List<Integer> res;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        ans = new ArrayList<>();
        res = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        method2(root, param1);
        return ans;
    }

    void method2(TreeNode root, int param1) {
        if (root == null) {
            return;
        }
        res.add(root.val);
        if (param1 == root.val && root.left == null && root.right == null) {
            ans.add(new ArrayList<>(res));
        } else {
            method2(root.left, param1 - root.val);
            method2(root.right, param1 - root.val);
        }
        res.remove(res.size() - 1);
    }
}
