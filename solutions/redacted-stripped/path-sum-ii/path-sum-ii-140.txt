class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> li = new ArrayList<>();
        dfs(root, param1, li);
        return list;
    }

    public void dfs(TreeNode root, int param1, List<Integer> li) {
        if (root == null) {
            return;
        }
        li.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == root.val) {
                list.add(new ArrayList<>(li));
            }
        }
        dfs(root.left, param1 - root.val, li);
        dfs(root.right, param1 - root.val, li);
        li.remove(li.size() - 1);
    }
}
