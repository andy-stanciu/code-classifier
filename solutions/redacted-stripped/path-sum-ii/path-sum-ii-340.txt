class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        solve(root, param1, new ArrayList<>(), ans);
        return ans;
    }

    public void solve(TreeNode root, int param1, List<Integer> list, List<List<Integer>> ans) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.val == param1 && root.left == null && root.right == null)
            ans.add(new ArrayList<>(list));
        solve(root.left, (param1 - root.val), list, ans);
        solve(root.right, (param1 - root.val), list, ans);
        list.remove(list.size() - 1);
    }
}
