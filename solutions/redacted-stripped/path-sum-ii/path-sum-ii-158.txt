class Solution {

    List<List<Integer>> lli = new ArrayList<List<Integer>>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> li = new ArrayList<Integer>();
        recursive(root, param1, 0, "");
        return lli;
    }

    void recursive(TreeNode root, int param1, int param2, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if ((param2 + root.val) == param1) {
                str = str + root.val;
                List<Integer> li = new ArrayList<Integer>();
                String[] arrOfStr = str.split("->");
                for (String a : arrOfStr) {
                    int val = Integer.parseInt(a);
                    li.add(val);
                }
                lli.add(li);
            }
            return;
        }
        recursive(root.left, param1, param2 + root.val, str + root.val + "->");
        recursive(root.right, param1, param2 + root.val, str + root.val + "->");
    }
}
