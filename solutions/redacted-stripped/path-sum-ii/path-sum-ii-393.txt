class Solution {

    List<List<Integer>> result = new LinkedList<List<Integer>>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return result;
        helper(root, param1, 0, new LinkedList<Integer>());
        return result;
    }

    public void helper(TreeNode root, int param1, int param2, LinkedList<Integer> current) {
        if (root == null)
            return;
        param2 += root.val;
        current.add(root.val);
        if (param1 == param2 && root.left == null && root.right == null) {
            List<Integer> newItem = new LinkedList<>(current);
            result.add(newItem);
        } else {
            helper(root.left, param1, param2, current);
            helper(root.right, param1, param2, current);
        }
        param2 -= current.getLast();
        current.removeLast();
    }
}
