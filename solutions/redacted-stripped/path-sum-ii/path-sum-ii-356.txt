class Solution {

    public static List<List<Integer>> res;

    public List<List<Integer>> method1(TreeNode root, int target) {
        res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        List<Integer> child = new ArrayList<>();
        child.add(root.val);
        helper(root, target, root.val, child);
        return res;
    }

    public static void helper(TreeNode root, int target, int param1, List<Integer> child) {
        if (root == null) {
            return;
        }
        if (param1 == target && root.left == null && root.right == null) {
            res.add(new ArrayList<>(child));
        }
        if (root.left != null) {
            child.add(root.left.val);
            helper(root.left, target, param1 + root.left.val, child);
            child.remove(child.size() - 1);
        }
        if (root.right != null) {
            child.add(root.right.val);
            helper(root.right, target, param1 + root.right.val, child);
            child.remove(child.size() - 1);
        }
    }
}
