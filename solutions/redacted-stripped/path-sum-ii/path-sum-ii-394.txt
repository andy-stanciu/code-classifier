class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> l = new ArrayList<>();
        solve(root, param1, 0, ans, l);
        return ans;
    }

    private void solve(TreeNode root, int k, int param2, List<List<Integer>> ans, List<Integer> l) {
        if (root == null)
            return;
        param2 += root.val;
        l.add(root.val);
        if (root.left == null && root.right == null && param2 == k) {
            ans.add(new ArrayList<>(l));
            return;
        }
        solve(root.left, k, param2, ans, l);
        solve(root.right, k, param2, ans, l);
        l.remove(l.size() - 1);
    }
}
