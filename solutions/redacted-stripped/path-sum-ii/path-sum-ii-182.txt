class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        solve(root, param1, 0, new ArrayList<Integer>());
        return ans;
    }

    public void solve(TreeNode root, int target, int param2, List<Integer> values) {
        if (root == null)
            return;
        param2 += root.val;
        values.add(root.val);
        if (param2 == target && root.left == null && root.right == null) {
            ans.add(new ArrayList<Integer>(values));
        }
        solve(root.left, target, param2, values);
        solve(root.right, target, param2, values);
        values.remove(values.size() - 1);
    }
}
