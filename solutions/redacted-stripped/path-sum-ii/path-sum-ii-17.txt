class Solution {

    List<List<Integer>> result = new ArrayList<>();

    List<Integer> temp = new ArrayList<>();

    int var1;

    public List<List<Integer>> method1(TreeNode root, int var1) {
        this.var1 = var1;
        if (root == null)
            return result;
        recur(root, 0);
        return result;
    }

    void recur(TreeNode root, int param1) {
        int var2 = param1 + root.val;
        temp.add(root.val);
        if (root.left != null)
            recur(root.left, var2);
        if (root.right != null)
            recur(root.right, var2);
        if (var2 == var1 && root.left == null && root.right == null)
            result.add(new ArrayList<>(temp));
        temp.remove(temp.size() - 1);
    }
}
