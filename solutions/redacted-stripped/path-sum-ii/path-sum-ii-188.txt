class Solution {

    List<List<Integer>> result = new ArrayList<>();

    int target;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> list = new ArrayList<>();
        target = param1;
        if (root != null)
            traverse(root, 0, list);
        return result;
    }

    void traverse(TreeNode node, int param2, List<Integer> list) {
        list.add(node.val);
        int var1 = param2 + node.val;
        if (var1 == target && node.left == null & node.right == null) {
            result.add(new ArrayList(list));
        } else {
            if (node.left != null)
                traverse(node.left, var1, list);
            if (node.right != null)
                traverse(node.right, var1, list);
        }
        list.remove(list.size() - 1);
    }
}
