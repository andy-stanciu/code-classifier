class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        method2(result, new ArrayList<Integer>(), root, param1);
        return result;
    }

    private void method2(List<List<Integer>> result, List<Integer> param2, TreeNode node, int param1) {
        if (node == null)
            return;
        param2.add(node.val);
        if (node.left == null && node.right == null && param1 - node.val == 0) {
            result.add(new ArrayList<Integer>(param2));
        }
        method2(result, param2, node.left, param1 - node.val);
        method2(result, param2, node.right, param1 - node.val);
        param2.remove(param2.size() - 1);
    }
}
