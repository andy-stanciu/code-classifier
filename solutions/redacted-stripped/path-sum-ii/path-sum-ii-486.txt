class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ret = new ArrayList<List<Integer>>();
        List<Integer> cur = new ArrayList<Integer>();
        method1(root, param1, cur, ret);
        return ret;
    }

    public void method1(TreeNode root, int param1, List<Integer> cur, List<List<Integer>> ret) {
        if (root == null) {
            return;
        }
        cur.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            ret.add(new ArrayList(cur));
        } else {
            method1(root.left, param1 - root.val, cur, ret);
            method1(root.right, param1 - root.val, cur, ret);
        }
        cur.remove(cur.size() - 1);
    }
}
