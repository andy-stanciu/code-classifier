class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        dfs(root, param1, new ArrayList<>(), res);
        return res;
    }

    public void dfs(TreeNode root, int target, List<Integer> param2, List<List<Integer>> res) {
        if (root == null) {
            return;
        }
        param2.add(root.val);
        if (root.right == null && root.left == null) {
            if (target - root.val == 0) {
                res.add(new ArrayList<>(param2));
            }
            param2.remove(param2.size() - 1);
            return;
        }
        dfs(root.left, target - root.val, param2, res);
        dfs(root.right, target - root.val, param2, res);
        param2.remove(param2.size() - 1);
    }
}
