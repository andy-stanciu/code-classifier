class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return res;
        List<Integer> curr = new ArrayList<>();
        backtrack(root, curr, param1);
        return res;
    }

    void backtrack(TreeNode root, List<Integer> curr, int param1) {
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            curr.add(root.val);
            res.add(new ArrayList<>(curr));
            curr.remove(curr.size() - 1);
            return;
        }
        curr.add(root.val);
        if (root.left != null)
            backtrack(root.left, curr, param1 - root.val);
        if (root.right != null)
            backtrack(root.right, curr, param1 - root.val);
        curr.remove(curr.size() - 1);
    }
}
