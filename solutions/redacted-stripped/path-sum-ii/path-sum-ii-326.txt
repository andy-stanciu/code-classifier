class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(root, new ArrayList<Integer>(), param1);
        return list;
    }

    private void helper(TreeNode root, List<Integer> lis, int param1) {
        if (root == null)
            return;
        lis.add(root.val);
        param1 -= root.val;
        if (root.left == null && root.right == null && param1 == 0) {
            list.add(new ArrayList(lis));
        }
        helper(root.left, new ArrayList(lis), param1);
        helper(root.right, new ArrayList(lis), param1);
    }
}
