class Solution {

    List<List<Integer>> list = new ArrayList<List<Integer>>();

    int[] var1 = new int[5000];

    private void helper(TreeNode root, int index, int param1) {
        if (root == null)
            return;
        var1[index] = root.val;
        if (root.left == null && root.right == null && root.val == param1) {
            List<Integer> l = new ArrayList<>();
            for (int i = 0; i <= index; i++) {
                l.add(var1[i]);
            }
            list.add(l);
            return;
        }
        helper(root.left, index + 1, param1 - root.val);
        helper(root.right, index + 1, param1 - root.val);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(root, 0, param1);
        return list;
    }
}
