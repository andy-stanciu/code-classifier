class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public void DFS(TreeNode root, int param1, int target, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && param1 + root.val == target) {
            res.add(new ArrayList<Integer>(list));
        } else {
            DFS(root.left, param1 + root.val, target, list);
            DFS(root.right, param1 + root.val, target, list);
        }
        list.remove(list.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        if (root == null)
            return res;
        List<Integer> list = new ArrayList<>();
        DFS(root, 0, param2, list);
        return res;
    }
}
