class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null) {
            return result;
        }
        helper(root, param1, new ArrayList<>());
        return result;
    }

    public void helper(TreeNode root, int param1, List<Integer> param2) {
        param2.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            result.add(new ArrayList<>(param2));
        } else {
            if (root.left != null) {
                helper(root.left, param1 - root.val, param2);
            }
            if (root.right != null) {
                helper(root.right, param1 - root.val, param2);
            }
        }
        param2.remove(param2.size() - 1);
    }
}
