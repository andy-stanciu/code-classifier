class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(param1, 0, root, new ArrayList<>());
        return res;
    }

    void method2(int target, int param2, TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        param2 += node.val;
        list.add(node.val);
        if (node.left == null && node.right == null) {
            if (target == param2)
                res.add(list);
            return;
        }
        method2(target, param2, node.left, new ArrayList<>(list));
        method2(target, param2, node.right, new ArrayList<>(list));
    }
}
