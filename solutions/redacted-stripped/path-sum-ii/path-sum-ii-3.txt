class Solution {

    List<List<Integer>> result = new ArrayList<List<Integer>>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        target(root, param1, new ArrayList<>());
        return result;
    }

    public void target(TreeNode root, int param1, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1 == root.val) {
            list.add(root.val);
            result.add(new ArrayList<>(list));
            list.remove(list.size() - 1);
        }
        list.add(root.val);
        target(root.left, param1 - root.val, list);
        target(root.right, param1 - root.val, list);
        list.remove(list.size() - 1);
    }
}
