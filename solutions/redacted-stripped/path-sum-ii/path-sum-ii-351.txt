class Solution {

    public static void helper(TreeNode root, int tgt, List<List<Integer>> ans, List<Integer> temp) {
        if (root == null) {
            return;
        }
        temp.add(root.val);
        if (root.left == null && root.right == null && root.val == tgt) {
            ans.add(new ArrayList<>(temp));
        } else {
            helper(root.left, tgt - root.val, ans, temp);
            helper(root.right, tgt - root.val, ans, temp);
        }
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        helper(root, param1, ans, temp);
        return ans;
    }
}
