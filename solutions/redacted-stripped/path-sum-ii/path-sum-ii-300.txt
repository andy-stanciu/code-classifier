class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> mc = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        helper(root, mc, temp, 0, param1);
        return mc;
    }

    public void helper(TreeNode root, List<List<Integer>> mc, List<Integer> temp, int param2, int target) {
        if (root == null) {
            return;
        }
        param2 = param2 + root.val;
        temp.add(root.val);
        if (root.left == null && root.right == null && param2 == target) {
            mc.add(new ArrayList<>(temp));
        }
        helper(root.left, mc, temp, param2, target);
        helper(root.right, mc, temp, param2, target);
        param2 = param2 - temp.remove(temp.size() - 1);
    }
}
