class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return ans;
        preorder(ans, var1, root, param1);
        return ans;
    }

    void preorder(List<List<Integer>> ans, List<Integer> var1, TreeNode root, int param2) {
        if (root == null)
            return;
        param2 -= root.val;
        var1.add(root.val);
        if (root.right == null && root.left == null && param2 == 0) {
            ans.add(var1);
        }
        preorder(ans, new ArrayList<>(var1), root.left, param2);
        preorder(ans, new ArrayList<>(var1), root.right, param2);
    }
}
