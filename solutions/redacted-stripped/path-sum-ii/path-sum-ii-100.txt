class Solution {

    public void rec(TreeNode root, int param1, List<List<Integer>> list, List<Integer> innerList) {
        if (root == null)
            return;
        List<Integer> newInnerList = new ArrayList<>(innerList);
        newInnerList.add(root.val);
        param1 -= root.val;
        if (root.left == null && root.right == null && param1 == 0)
            list.add(newInnerList);
        else {
            rec(root.left, param1, list, newInnerList);
            rec(root.right, param1, list, newInnerList);
        }
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> innerList = new ArrayList<>();
        rec(root, param1, list, innerList);
        return list;
    }
}
