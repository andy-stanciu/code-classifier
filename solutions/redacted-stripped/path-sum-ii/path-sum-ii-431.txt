class Solution {

    public static List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return new ArrayList<>();
        List<List<Integer>> result = new ArrayList<>();
        return helper(root, param1, new ArrayList<>(), result);
    }

    public static List<List<Integer>> helper(TreeNode root, int param1, ArrayList<Integer> param2, List<List<Integer>> result) {
        if (root == null)
            return new ArrayList<>();
        param1 -= root.val;
        param2.add(root.val);
        if (param1 == 0 && root.left == null && root.right == null) {
            result.add(param2);
            return result;
        }
        helper(root.left, param1, new ArrayList<>(param2), result);
        helper(root.right, param1, new ArrayList<>(param2), result);
        return result;
    }
}
