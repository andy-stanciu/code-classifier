class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> temp = new ArrayList<>();
        fill(root, param1, 0, temp);
        return result;
    }

    public void fill(TreeNode root, int target, int param2, List<Integer> temp) {
        if (root == null)
            return;
        param2 = param2 + root.val;
        temp.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 == target) {
                result.add(new ArrayList<>(temp));
            }
        } else {
            fill(root.left, target, param2, temp);
            fill(root.right, target, param2, temp);
        }
        temp.remove(temp.size() - 1);
    }
}
