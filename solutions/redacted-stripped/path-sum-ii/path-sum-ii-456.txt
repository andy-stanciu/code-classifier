class Solution {

    List<List<Integer>> res;

    public void helper(TreeNode root, int param1, List<Integer> ans) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            ans.add(root.val);
            if (param1 - root.val == 0) {
                res.add(new ArrayList<>(ans));
            }
            ans.remove(ans.size() - 1);
            return;
        }
        ans.add(root.val);
        helper(root.left, param1 - root.val, ans);
        helper(root.right, param1 - root.val, ans);
        ans.remove(ans.size() - 1);
        return;
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        res = new ArrayList<>();
        helper(root, param1, new ArrayList<>());
        return res;
    }
}
