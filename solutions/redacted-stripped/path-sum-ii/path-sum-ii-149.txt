class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        if (root == null)
            return result;
        method2(root, param1, new ArrayList<Integer>(), result);
        return result;
    }

    public void method2(TreeNode root, int param1, List<Integer> curr, List<List<Integer>> result) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if (root.val == param1) {
                curr.add(root.val);
                result.add(new ArrayList<>(curr));
                curr.remove(curr.size() - 1);
            }
            return;
        }
        curr.add(root.val);
        method2(root.left, param1 - root.val, curr, result);
        method2(root.right, param1 - root.val, curr, result);
        curr.remove(curr.size() - 1);
    }
}
