class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        int var1 = 0;
        method2(root, param1, var1, temp, ans);
        return ans;
    }

    public void method2(TreeNode root, int target, int var1, List<Integer> temp, List<List<Integer>> ans) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            var1 = var1 + root.val;
            temp.add(root.val);
            if (var1 == target) {
                List<Integer> temp2 = new ArrayList<>(temp);
                ans.add(temp2);
            }
            temp.remove(temp.size() - 1);
            return;
        }
        temp.add(root.val);
        method2(root.left, target, var1 + root.val, temp, ans);
        method2(root.right, target, var1 + root.val, temp, ans);
        temp.remove(temp.size() - 1);
    }
}
