class Solution {

    public void method1(int[] nums) {
        int i = 0;
        int j = 0;
        while (i < nums.length) {
            if (nums[i] == 0) {
                i++;
            } else {
                nums[j] = nums[i];
                j++;
                i++;
            }
        }
        while (j < nums.length) {
            nums[j] = 0;
            j++;
        }
    }
}
