class Solution {

    public void method1(int[] nums) {
        int dst = -1;
        while (++dst < nums.length && nums[dst] != 0) ;
        int src = dst;
        while (++src < nums.length) if (nums[src] != 0)
            nums[dst++] = nums[src];
        while (dst < nums.length) nums[dst++] = 0;
    }
}
