class Solution {

    public void method1(int[] nums) {
        int c = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[c] = nums[i];
                c++;
            }
        }
        for (int i = c; i < n; i++) {
            nums[i] = 0;
        }
    }
}
