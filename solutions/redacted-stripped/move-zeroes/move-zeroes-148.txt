class Solution {

    public void method1(int[] nums) {
        int i = 0;
        int j = 0;
        while (i < nums.length && j < nums.length) {
            if (nums[j] != 0) {
                nums[i] = nums[j];
                i++;
                j++;
            } else {
                j++;
            }
        }
        while (i < nums.length) {
            nums[i] = 0;
            i++;
        }
    }
}
