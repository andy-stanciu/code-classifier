class Solution {

    public void method1(int[] nums) {
        int nonZeroPos = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                if (i != nonZeroPos) {
                    int temp = nums[i];
                    nums[i] = nums[nonZeroPos];
                    nums[nonZeroPos] = temp;
                }
                nonZeroPos++;
            }
        }
        for (int e : nums) {
            System.out.println(e);
        }
    }
}
