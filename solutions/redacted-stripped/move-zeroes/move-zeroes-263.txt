class Solution {

    public void method1(int[] nums) {
        int z = 0;
        int[] arr = new int[nums.length];
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                arr[j] = nums[i];
                j++;
            } else
                z++;
        }
        for (int i = 0; i < z; i++) {
            arr[j] = 0;
            j++;
        }
        for (int i = 0; i < nums.length; i++) nums[i] = arr[i];
    }
}
