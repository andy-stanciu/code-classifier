class Solution {

    public void method1(int[] nums) {
        int numsLength = nums.length;
        int index = 0;
        for (int i = 0; i < numsLength; i++) {
            if (nums[i] != 0) {
                nums[index] = nums[i];
                index++;
            }
        }
        for (int i = index; i < numsLength; i++) {
            nums[i] = 0;
        }
    }
}
