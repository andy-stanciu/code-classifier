class Solution {

    public void method1(int[] nums) {
        int s1 = 0;
        int s2 = 0;
        while (s2 < nums.length && s1 < nums.length) {
            if (nums[s2] != 0) {
                swap(nums, s1, s2);
                s1++;
            }
            s2++;
        }
    }

    private void swap(int[] nums, int s1, int s2) {
        int first = nums[s1];
        nums[s1] = nums[s2];
        nums[s2] = first;
    }
}
