class Solution {

    public void method1(int[] nums) {
        if (nums.length < 2) {
            return;
        }
        int l = 0;
        int r = 1;
        while (r < nums.length) {
            if (nums[l] != 0) {
                l++;
                r++;
            } else if (nums[r] == 0) {
                r++;
            } else {
                int temp = nums[r];
                nums[r] = nums[l];
                nums[l] = temp;
            }
        }
    }
}
