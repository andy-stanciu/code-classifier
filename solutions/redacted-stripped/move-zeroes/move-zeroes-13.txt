class Solution {

    public void method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                var1++;
            } else if (var1 > 0) {
                int temp = nums[i];
                nums[i] = 0;
                nums[i - var1] = temp;
            }
        }
    }
}
