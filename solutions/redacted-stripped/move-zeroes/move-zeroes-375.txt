class Solution {

    public void method1(int[] nums) {
        int[] temp = new int[nums.length];
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                temp[j] = nums[i];
                j++;
            }
        }
        int n = temp.length;
        for (int i = 0; i < n; i++) {
            nums[i] = temp[i];
        }
        for (int i = n; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
