class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        ArrayList<Integer> lst = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                lst.add(nums[i]);
            }
        }
        int m = lst.size();
        for (int i = 0; i < m; i++) {
            nums[i] = lst.get(i);
        }
        for (int i = m; i < n; i++) {
            nums[i] = 0;
        }
    }
}
