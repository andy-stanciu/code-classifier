class Solution {

    public void method1(int[] nums) {
        int curr = -1;
        int j = 0;
        while (j < nums.length) {
            if (nums[j] == 0) {
                if (curr == -1)
                    curr = j;
            } else if (curr != -1) {
                nums[curr] = nums[j];
                nums[j] = 0;
                curr++;
            }
            j++;
        }
        return;
    }
}
