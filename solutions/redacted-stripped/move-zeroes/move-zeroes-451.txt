class Solution {

    public void method1(int[] nums) {
        int lastNonZeroIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[lastNonZeroIndex] = nums[i];
                lastNonZeroIndex++;
            }
        }
        for (int j = lastNonZeroIndex; j < nums.length; j++) {
            nums[j] = 0;
        }
    }
}
