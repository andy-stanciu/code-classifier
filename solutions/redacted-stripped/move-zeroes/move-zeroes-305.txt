class Solution {

    public void method1(int[] nums) {
        int leftSpace = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[leftSpace++] = nums[i];
            }
        }
        for (int i = leftSpace; i < n; i++) {
            nums[i] = 0;
        }
    }
}
