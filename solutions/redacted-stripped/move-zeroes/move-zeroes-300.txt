class Solution {

    public void method1(int[] nums) {
        int len = nums.length;
        int iPointer = 0;
        int jPointer = 0;
        while (iPointer < len) {
            if (nums[iPointer] != 0) {
                if (iPointer != jPointer) {
                    nums[jPointer] = nums[iPointer];
                    nums[iPointer] = 0;
                    jPointer++;
                } else {
                    jPointer++;
                }
            }
            iPointer++;
        }
    }
}
