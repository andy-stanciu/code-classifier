class Solution {

    public void method1(int[] nums) {
        int n = nums.length, j = 0;
        if (n <= 1)
            return;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[j++] = nums[i];
            }
        }
        for (int i = j; i < n; i++) nums[i] = 0;
    }
}
