class Solution {

    public void method1(int[] nums) {
        ArrayList<Integer> newArr = new ArrayList<>();
        ArrayList<Integer> ZeroArr = new ArrayList<>();
        for (int num : nums) {
            if (num == 0) {
                ZeroArr.add(num);
            } else {
                newArr.add(num);
            }
        }
        newArr.addAll(ZeroArr);
        for (int i = 0; i < nums.length; i++) {
            nums[i] = newArr.get(i);
        }
        System.out.println(Arrays.toString(nums));
    }
}
