class Solution {

    public void method1(int[] nums) {
        int[] arr = new int[nums.length];
        int x = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                arr[x++] = nums[i];
            }
        }
        for (int i = x; i < nums.length; i++) {
            arr[x++] = 0;
        }
        for (int i = 0; i < nums.length; i++) {
            nums[i] = arr[i];
        }
    }
}
