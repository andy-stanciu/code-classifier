class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        Queue<Integer> q = new LinkedList<>();
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                q.add(nums[i]);
            }
        }
        int i = 0;
        while (!q.isEmpty()) {
            nums[i] = q.poll();
            i++;
        }
        while (i < n) {
            nums[i] = 0;
            i++;
        }
    }
}
