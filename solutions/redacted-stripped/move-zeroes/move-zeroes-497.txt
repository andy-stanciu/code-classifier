class Solution {

    public void method1(int[] nums) {
        if (nums.length == 1) {
            return;
        }
        int i = 0;
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] != 0) {
                int temp = nums[j];
                nums[j] = 0;
                nums[i] = temp;
                i++;
            }
        }
    }
}
