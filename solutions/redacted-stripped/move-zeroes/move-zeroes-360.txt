class Solution {

    public void method1(int[] nums) {
        int x = 0, count = 0, l = 0, k = nums.length;
        int[] p = new int[k];
        for (int i = 0; i < k; i++) {
            if (nums[i] != 0) {
                p[x] = nums[i];
                x++;
            }
        }
        for (int j = 0; j < k; j++) {
            nums[j] = p[l];
            l++;
        }
    }
}
