class Solution {

    public void method1(int[] nums) {
        int[] ans = new int[nums.length];
        int counter = 0;
        for (int count = 0; count < nums.length; count++) {
            if (nums[count] != 0) {
                ans[counter] = nums[count];
                counter++;
            }
        }
        for (int count = 0; count < nums.length; count++) {
            nums[count] = ans[count];
        }
    }
}
