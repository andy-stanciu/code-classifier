class Solution {

    public void method1(int[] nums) {
        int count = 0;
        int index = 0;
        for (int i : nums) {
            if (i == 0)
                count++;
            if (i != 0) {
                nums[index] = i;
                index++;
            }
        }
        for (int i = nums.length - count; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
