class Solution {

    public void method1(int[] nums) {
        int count = 0;
        for (int item : nums) {
            if (item != 0) {
                nums[count] = item;
                count++;
            }
        }
        for (int i = count; i < nums.length; i++) {
            nums[count] = 0;
            count++;
        }
    }
}
