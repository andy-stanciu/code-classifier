class Solution {

    public void method1(int[] nums) {
        int inspos = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[inspos] = nums[i];
                inspos++;
            }
        }
        while (inspos < nums.length) {
            nums[inspos++] = 0;
        }
    }
}
