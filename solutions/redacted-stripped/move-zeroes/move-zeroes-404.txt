class Solution {

    public void method1(int[] nums) {
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            while (i < nums.length - 1 && nums[i] == 0) {
                i++;
            }
            nums[count] = nums[i];
            count++;
        }
        while (count < nums.length) {
            nums[count] = 0;
            count++;
        }
    }
}
