class Solution {

    public void method1(int[] arr) {
        int nonZeroIndex = 0, n = arr.length;
        for (int i = 0; i < n; i++) {
            if (arr[i] != 0) {
                int tmp = arr[i];
                arr[i] = arr[nonZeroIndex];
                arr[nonZeroIndex] = tmp;
                nonZeroIndex++;
            }
        }
    }
}
