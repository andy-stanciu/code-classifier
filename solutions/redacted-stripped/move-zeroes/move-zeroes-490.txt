class Solution {

    public void method1(int[] nums) {
        int idx = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[idx] = nums[i];
                idx++;
            }
        }
        if (idx != 0) {
            for (int i = idx; i < n; i++) {
                nums[i] = 0;
            }
        }
    }
}
