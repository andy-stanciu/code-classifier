class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        if (n < 2) {
            return;
        }
        int left = 0;
        for (int right = 0; right < n; right++) {
            if (nums[right] != 0) {
                int temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
                left++;
            }
        }
    }
}
