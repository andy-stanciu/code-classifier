class Solution {

    public void method1(int[] nums) {
        int len = nums.length;
        if (len > 1) {
            for (int i = 0; i < len - 1; i++) {
                if (nums[i] == 0) {
                    for (int j = i + 1; j < len; j++) {
                        if ((nums[i] == 0 && nums[j] != 0)) {
                            int temp = nums[i];
                            nums[i] = nums[j];
                            nums[j] = temp;
                            break;
                        }
                    }
                }
            }
        }
    }
}
