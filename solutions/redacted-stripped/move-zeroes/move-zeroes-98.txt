class Solution {

    public void method1(int[] nums) {
        int j = 0;
        int nl = nums.length;
        for (int i = 0; i < nl; i++) {
            if (nums[i] != 0) {
                nums[j++] = nums[i];
            }
        }
        for (int i = 0; i < nl - j; i++) {
            nums[nl - 1 - i] = 0;
        }
    }
}
