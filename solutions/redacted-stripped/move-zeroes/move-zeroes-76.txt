class Solution {

    public void method1(int[] nums) {
        int zeros = 0;
        int l = 0, r = 0;
        while (r < nums.length) {
            while (r < nums.length - 1 && nums[r] == 0) {
                ++r;
            }
            if (nums[l] == 0) {
                nums[l] = nums[r];
                nums[r] = 0;
            }
            ++l;
            ++r;
        }
    }
}
