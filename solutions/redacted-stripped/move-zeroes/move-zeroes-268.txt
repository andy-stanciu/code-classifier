class Solution {

    public void method1(int[] nums) {
        int z = 0;
        int n = 0;
        while (z < nums.length && n < nums.length) {
            if (nums[n] != 0 && n > z) {
                nums[z] = nums[n];
                nums[n] = 0;
            }
            while (z < nums.length && nums[z] != 0) {
                z++;
            }
            n++;
        }
        return;
    }
}
