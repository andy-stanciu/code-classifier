class Solution {

    public void method1(int[] nums) {
        int len = nums.length;
        int newI = 0;
        int zeroCount = 0;
        for (int i = 0; i < len; i++) {
            if (nums[i] != 0) {
                nums[newI] = nums[i];
                newI++;
            } else
                zeroCount++;
        }
        for (int i = newI; i < newI + zeroCount; i++) {
            nums[i] = 0;
        }
    }
}
