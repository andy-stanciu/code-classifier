class Solution {

    public void method1(int[] arr) {
        int x = 0;
        boolean found = false;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != 0) {
                arr[x] = arr[i];
                x++;
            } else {
                found = true;
            }
        }
        if (found == false) {
            return;
        }
        for (int i = x; i < arr.length; i++) {
            arr[i] = 0;
        }
    }
}
