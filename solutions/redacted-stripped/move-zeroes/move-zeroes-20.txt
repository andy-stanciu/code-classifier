class Solution {

    public void method1(int[] nums) {
        int left = 0;
        int right = 0;
        while (left < nums.length) {
            if (nums[left] != 0) {
                int temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
                right++;
            }
            left++;
        }
    }
}
