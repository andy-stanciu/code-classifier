class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int j = 0, count = 0;
        int[] temp = new int[n];
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                count++;
                temp[j++] = nums[i];
            }
        }
        for (int i = count; i < n; i++) {
            temp[i] = 0;
        }
        for (int i = 0; i < n; i++) {
            nums[i] = temp[i];
        }
    }
}
