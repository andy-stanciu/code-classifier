class Solution {

    public void method1(int[] array) {
        int indx = 0;
        for (int i = 0; i < array.length; i++) {
            if (array[i] != 0) {
                if (i != indx) {
                    array[indx] = array[i];
                    array[i] = 0;
                }
                indx++;
            }
        }
    }
}
