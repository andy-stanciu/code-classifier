class Solution {

    public void method1(int[] nums) {
        int L = 0;
        int R = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                L++;
            } else {
                int temp = nums[L];
                nums[L] = nums[R];
                nums[R] = temp;
                L++;
                R++;
            }
        }
    }
}
