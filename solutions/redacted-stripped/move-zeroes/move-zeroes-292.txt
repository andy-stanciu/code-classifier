class Solution {

    public void method1(int[] arr) {
        int l = 0;
        int r = 0;
        while (r < arr.length) {
            if (arr[r] == 0) {
                r++;
            } else {
                if (l != r) {
                    int temp = arr[l];
                    arr[l] = arr[r];
                    arr[r] = temp;
                }
                l++;
                r++;
            }
        }
    }
}
