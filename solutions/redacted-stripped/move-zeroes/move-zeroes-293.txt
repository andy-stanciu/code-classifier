class Solution {

    public void method1(int[] nums) {
        int i = 0, j = 0, n = nums.length;
        while (i < n && j < n) {
            if (nums[i] == 0) {
                j++;
                while (j < n && nums[j] == 0) {
                    j++;
                }
                if (i < n && j < n)
                    swap(i, j, nums);
                j = i;
            }
            i++;
            j++;
        }
    }

    public void swap(int i, int j, int[] nums) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
