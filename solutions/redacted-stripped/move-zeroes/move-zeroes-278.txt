class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int k = 0;
        int[] a = new int[n];
        Arrays.fill(a, 0);
        for (int i = 0; i < n; i++) if (nums[i] != 0)
            a[k++] = nums[i];
        for (int i = 0; i < n; i++) nums[i] = a[i];
    }
}
