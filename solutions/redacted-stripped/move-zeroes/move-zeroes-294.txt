class Solution {

    public void method1(int[] nums) {
        int zeroCount = 0;
        for (int n : nums) {
            if (n == 0) {
                zeroCount++;
            }
        }
        int mustSwipe = nums.length - zeroCount;
        int tracker = 0;
        for (int i = 0; i < nums.length; i++) {
            if (mustSwipe == 0) {
                break;
            }
            if (nums[i] != 0) {
                nums[tracker] = nums[i];
                tracker++;
                mustSwipe--;
            }
        }
        for (int i = nums.length - zeroCount; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
