class Solution {

    public boolean method1(String s, String t) {
        char[] arr1 = s.toCharArray();
        char[] arr2 = t.toCharArray();
        int n = s.length();
        Arrays.sort(arr1);
        Arrays.sort(arr2);
        if (arr1.length != arr2.length) {
            return false;
        }
        for (int i = 0; i < n; i++) {
            if (arr1[i] != arr2[i]) {
                return false;
            }
        }
        return true;
    }
}
