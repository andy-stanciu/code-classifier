class Solution {

    public boolean method1(String s, String t) {
        String str1 = sortStr(s);
        String str2 = sortStr(t);
        return str1.equals(str2);
    }

    public String sortStr(String s) {
        String sorted = s.chars().sorted().collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append).toString();
        return sorted;
    }
}
