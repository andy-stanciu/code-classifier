class Solution {

    public boolean method1(String s, String t) {
        char[] c = s.toCharArray();
        char[] e = t.toCharArray();
        Arrays.sort(c);
        Arrays.sort(e);
        boolean value = false;
        if (c.length == e.length) {
            for (int i = 0; i < c.length; i++) {
                if (c[i] == e[i]) {
                    value = true;
                } else {
                    value = false;
                    return value;
                }
            }
        }
        return value;
    }
}
