class Solution {

    public boolean method1(String s, String t) {
        Map<Character, Integer> charCountMap = new HashMap<>();
        if (s.length() != s.length())
            return false;
        for (char c : t.toCharArray()) {
            charCountMap.put(c, charCountMap.getOrDefault(c, 0) + 1);
        }
        for (char k : s.toCharArray()) {
            if (!charCountMap.containsKey(k)) {
                return false;
            } else {
                int currentValue = charCountMap.get(k);
                charCountMap.put(k, currentValue - 1);
                if (charCountMap.get(k) == 0) {
                    charCountMap.remove(k);
                }
            }
        }
        if (charCountMap.isEmpty())
            return true;
        return false;
    }
}
