class Solution {

    public boolean method1(String s, String t) {
        char[] arr1 = s.toCharArray();
        char[] arr2 = t.toCharArray();
        Arrays.sort(arr1);
        Arrays.sort(arr2);
        String q = Arrays.toString(arr1);
        String w = Arrays.toString(arr2);
        if (q.equals(w)) {
            return true;
        } else {
            return false;
        }
    }
}
