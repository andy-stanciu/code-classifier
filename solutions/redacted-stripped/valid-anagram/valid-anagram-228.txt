class Solution {

    public boolean method1(String s, String t) {
        char[] sArray = s.toCharArray();
        char[] tArray = t.toCharArray();
        Arrays.sort(sArray);
        Arrays.sort(tArray);
        String newSString = new String(sArray);
        String newTString = new String(tArray);
        return (newSString.matches(newTString));
    }
}
