class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] arr = new int[26];
        for (char ch : s.toCharArray()) {
            arr[ch - 97] = arr[ch - 97] + 1;
        }
        for (char ch : t.toCharArray()) {
            arr[ch - 97] = arr[ch - 97] - 1;
        }
        for (int k : arr) {
            if (k > 0) {
                return false;
            }
        }
        return true;
    }
}
