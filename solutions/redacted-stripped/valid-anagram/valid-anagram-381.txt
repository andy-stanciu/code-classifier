class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        ArrayList<Character> list = new ArrayList<>();
        ArrayList<Character> listt = new ArrayList<>();
        for (int c = 0; c < s.length(); c++) {
            list.add(s.charAt(c));
        }
        ;
        for (int c = 0; c < t.length(); c++) {
            listt.add(t.charAt(c));
        }
        ;
        for (int i = 0; i < t.length(); i++) {
            if (list.contains(t.charAt(i))) {
                list.remove(Character.valueOf(t.charAt(i)));
                listt.remove(Character.valueOf(t.charAt(i)));
            } else {
                return false;
            }
        }
        if (list.size() != 0 || listt.size() != 0) {
            return false;
        }
        return true;
    }
}
