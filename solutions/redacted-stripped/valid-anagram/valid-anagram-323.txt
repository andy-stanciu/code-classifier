class Solution {

    public boolean method1(String s, String t) {
        int[] arr = new int[26];
        int[] arr2 = new int[26];
        for (int i = 0; i < s.length(); i++) {
            int n = (int) (s.charAt(i)) - 97;
            arr[n]++;
        }
        for (int i = 0; i < t.length(); i++) {
            int n = (int) (t.charAt(i)) - 97;
            arr2[n]++;
        }
        for (int i = 0; i < 26; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
        for (int i = 0; i < 26; i++) {
            System.out.print(arr2[i] + " ");
        }
        if (Arrays.equals(arr, arr2)) {
            return true;
        }
        return false;
    }
}
