class Solution {

    public String sortString(String s) {
        char[] chr = s.toCharArray();
        Arrays.sort(chr);
        return new String(chr);
    }

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        s = sortString(s);
        t = sortString(t);
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != t.charAt(i)) {
                return false;
            }
        }
        return true;
    }
}
