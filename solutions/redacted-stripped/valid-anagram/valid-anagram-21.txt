class Solution {

    public boolean method1(String s, String t) {
        int len = s.length();
        int len1 = t.length();
        int count = 0;
        int inj = 0;
        char[] arr = s.toCharArray();
        char[] arr1 = t.toCharArray();
        Arrays.sort(arr);
        Arrays.sort(arr1);
        if (len != len1) {
            return false;
        }
        for (int i = 0; i < len; i++) {
            if (arr[i] == arr1[i]) {
                count++;
            }
        }
        if (count == len || count == len1) {
            return true;
        }
        return false;
    }
}
