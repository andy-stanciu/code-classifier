class Solution {

    public boolean method1(String s, String t) {
        int sl = s.length();
        int freq = 0;
        if (sl != t.length()) {
            return false;
        }
        HashMap<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < sl; i++) {
            char chs = s.charAt(i);
            freq = 1;
            if (map.containsKey(chs)) {
                freq = map.get(chs) + 1;
            }
            map.put(chs, freq);
        }
        for (int i = 0; i < sl; i++) {
            char cht = t.charAt(i);
            if (map.containsKey(cht) && map.get(cht) > 0) {
                freq = map.get(cht) - 1;
                map.put(cht, freq);
            } else {
                return false;
            }
        }
        return true;
    }
}
