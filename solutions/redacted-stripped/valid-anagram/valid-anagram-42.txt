class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        } else {
            char[] s1 = s.toLowerCase().toCharArray();
            char[] t1 = t.toLowerCase().toCharArray();
            Arrays.sort(s1);
            Arrays.sort(t1);
            if (Arrays.equals(s1, t1)) {
                return true;
            } else {
                return false;
            }
        }
    }
}
