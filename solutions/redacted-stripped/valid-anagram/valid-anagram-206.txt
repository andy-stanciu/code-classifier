class Solution {

    public boolean method1(String s, String t) {
        char[] c1 = s.toCharArray();
        char[] c2 = t.toCharArray();
        Arrays.sort(c1);
        Arrays.sort(c2);
        String a = new String(c1);
        String b = new String(c2);
        if (a.equals(b))
            return true;
        return false;
    }
}
