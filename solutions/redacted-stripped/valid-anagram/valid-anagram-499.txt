class Solution {

    public static boolean method1(String s, String t) {
        char[] word1 = s.toCharArray();
        char[] word2 = t.toCharArray();
        Arrays.sort(word1);
        Arrays.sort(word2);
        if (Arrays.compare(word1, word2) == 0) {
            return true;
        }
        return false;
    }
}
