class Solution {

    public boolean method1(String s, String t) {
        int m = s.length();
        int n = t.length();
        if (m != n) {
            return false;
        }
        char[] arr = s.toCharArray();
        char[] brr = t.toCharArray();
        Arrays.sort(arr);
        Arrays.sort(brr);
        for (int i = 0; i <= arr.length - 1; i++) {
            if (arr[i] != brr[i]) {
                return false;
            }
        }
        return true;
    }
}
