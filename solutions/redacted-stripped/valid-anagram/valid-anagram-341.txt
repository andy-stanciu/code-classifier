class Solution {

    public boolean method1(String s, String t) {
        return hash(s) == hash(t);
    }

    int hash(String toHash) {
        char[] symbols = toHash.toCharArray();
        int hash = 0;
        for (char symbol : symbols) {
            hash += symbol;
        }
        for (char symbol : symbols) {
            hash *= symbol;
        }
        for (char symbol : symbols) {
            hash ^= symbol;
        }
        return (hash) + 31 ^ toHash.length();
    }
}
