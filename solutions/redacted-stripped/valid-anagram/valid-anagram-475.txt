class Solution {

    public boolean method1(String s1, String t1) {
        char[] s = s1.toCharArray();
        char[] t = t1.toCharArray();
        Arrays.sort(s);
        Arrays.sort(t);
        if (s.length != t.length) {
            return false;
        }
        for (int i = 0; i < s.length; i++) {
            if (s[i] != t[i])
                return false;
        }
        return true;
    }
}
