class Solution {

    public boolean method1(String s, String t) {
        ArrayList<String> arr = new ArrayList<String>(Arrays.asList(t.split("")));
        for (int i = 0; i < s.length(); i++) {
            if (arr.contains(s.charAt(i) + "")) {
                arr.remove(arr.indexOf(s.charAt(i) + ""));
                continue;
            }
            return false;
        }
        if (arr.size() == 0) {
            return true;
        }
        return false;
    }
}
