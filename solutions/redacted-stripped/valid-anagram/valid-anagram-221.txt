class Solution {

    public boolean method1(String s, String t) {
        char[] sArray = s.toCharArray();
        char[] tArray = t.toCharArray();
        Arrays.sort(sArray);
        Arrays.sort(tArray);
        String sString = new String(sArray);
        String tString = new String(tArray);
        if (sString.equals(tString)) {
            return true;
        } else {
            return false;
        }
    }
}
