class Solution {

    public boolean method1(String s, String t) {
        int a = s.length();
        int b = t.length();
        int[] sf = new int[26];
        int[] tf = new int[26];
        int at, as;
        int f = 0;
        for (int i = 0; i < 26; i++) {
            tf[i] = 0;
            sf[i] = 0;
        }
        if (b == a) {
            for (int i = 0; i < b; i++) {
                at = (int) (t.charAt(i));
                as = (int) (s.charAt(i));
                tf[at - 97] = tf[at - 97] + 1;
                sf[as - 97] = sf[as - 97] + 1;
            }
            for (int i = 0; i < 26; i++) {
                if (tf[i] == sf[i]) {
                    f = f + 1;
                }
            }
            if (f == 26)
                return true;
        }
        return false;
    }
}
