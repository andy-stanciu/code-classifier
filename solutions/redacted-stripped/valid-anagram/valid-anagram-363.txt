class Solution {

    public boolean method1(String s, String t) {
        char[] sArr = s.toCharArray();
        char[] tArr = t.toCharArray();
        Arrays.sort(sArr);
        Arrays.sort(tArr);
        String stringS = String.valueOf(sArr);
        String stringT = String.valueOf(tArr);
        return stringS.equals(stringT);
    }
}
