class Solution {

    public boolean method1(String s, String t) {
        String val1 = sortString(s);
        String val2 = sortString(t);
        if (val1.equals(val2)) {
            return true;
        }
        return false;
    }

    public String sortString(String str) {
        char[] c = str.toCharArray();
        Arrays.sort(c);
        String returnString = new String(c);
        return returnString;
    }
}
