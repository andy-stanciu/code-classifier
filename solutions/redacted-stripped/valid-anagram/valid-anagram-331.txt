class Solution {

    boolean method1(String s, String t) {
        return occurrences(s).equals(occurrences(t));
    }

    private Hashtable<Character, Integer> occurrences(String s) {
        Hashtable<Character, Integer> response = new Hashtable<>();
        for (char c : s.toCharArray()) {
            if (response.get(c) == null) {
                response.put(c, 1);
            } else {
                int count = response.get(c);
                response.put(c, ++count);
            }
        }
        return response;
    }
}
