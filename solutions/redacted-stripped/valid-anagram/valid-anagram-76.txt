class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        int n = s.length();
        int[] freqS = new int[27];
        int[] freqT = new int[27];
        for (int i = 0; i < n; i++) {
            freqS[s.charAt(i) - 97]++;
            freqT[t.charAt(i) - 97]++;
        }
        for (int i = 0; i < 27; i++) {
            if (freqS[i] != freqT[i])
                return false;
        }
        return true;
    }
}
