class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int i = 0;
        while (i < t.length()) {
            if (s.contains(t.charAt(i) + "")) {
                s = s.replaceFirst(t.charAt(i) + "", "");
                i++;
            } else {
                return false;
            }
        }
        return true;
    }
}
