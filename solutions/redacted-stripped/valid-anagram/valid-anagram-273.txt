class Solution {

    public boolean method1(String s, String t) {
        char[] so = s.toCharArray();
        char[] to = t.toCharArray();
        Arrays.sort(so);
        Arrays.sort(to);
        if (s.length() != t.length()) {
            return false;
        }
        for (int i = 0; i < s.length(); i++) {
            if (so[i] == to[i]) {
                continue;
            } else {
                return false;
            }
        }
        return true;
    }
}
