class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        char[] temp1 = s.toCharArray();
        char[] temp2 = t.toCharArray();
        Arrays.sort(temp1);
        Arrays.sort(temp2);
        boolean result = Arrays.equals(temp1, temp2);
        if (result == true) {
            return true;
        } else {
            return false;
        }
    }
}
