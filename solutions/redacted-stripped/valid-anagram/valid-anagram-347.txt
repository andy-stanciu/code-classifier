class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        HashMap<Character, Integer> map = new HashMap<>();
        char curr;
        for (int i = 0; i < t.length(); i++) {
            curr = t.charAt(i);
            if (map.containsKey(curr)) {
                map.put(curr, map.get(curr) + 1);
            } else {
                map.put(curr, 1);
            }
        }
        for (int i = 0; i < s.length(); i++) {
            curr = s.charAt(i);
            if (!map.containsKey(curr))
                return false;
            if (map.get(curr) > 1) {
                map.put(curr, map.get(curr) - 1);
            } else {
                map.remove(curr);
            }
        }
        return true;
    }
}
