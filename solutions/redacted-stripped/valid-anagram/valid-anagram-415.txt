class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        HashMap<Character, Integer> arrList = new HashMap<>();
        boolean flag = true;
        for (int i = 0; i < s.length(); i++) {
            arrList.put(s.charAt(i), arrList.getOrDefault(s.charAt(i), 0) + 1);
        }
        for (int j = 0; j < t.length(); j++) {
            if (arrList.getOrDefault(t.charAt(j), 0) > 0)
                arrList.put(t.charAt(j), arrList.get(t.charAt(j)) - 1);
            else
                flag = false;
        }
        return flag;
    }
}
