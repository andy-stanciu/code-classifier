class Solution {

    public boolean method1(String s, String t) {
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        int c = 0;
        if (a.length == b.length) {
            for (int i = 0; i < a.length; i++) {
                if (a[i] == b[i]) {
                    c++;
                }
            }
        } else
            return false;
        if (c == s.length())
            return true;
        else
            return false;
    }
}
