class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        Map<Character, Integer> check = new HashMap<>();
        for (Character c : s.toCharArray()) {
            if (check.get(c) != null) {
                check.put(c, check.get(c) + 1);
            } else {
                check.put(c, 0);
            }
        }
        for (Character c : t.toCharArray()) {
            if (check.containsKey(c)) {
                if (check.get(c) > 0) {
                    check.put(c, check.get(c) - 1);
                } else {
                    check.remove(c, 0);
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
