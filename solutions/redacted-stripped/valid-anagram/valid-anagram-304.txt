class Solution {

    public boolean method1(String s, String t) {
        char[] c = s.toCharArray();
        char[] b = t.toCharArray();
        if (c.length != b.length) {
            return false;
        }
        int n = c.length;
        Arrays.sort(c);
        Arrays.sort(b);
        for (int i = 0; i < n; i++) {
            if (c[i] != b[i]) {
                return false;
            }
        }
        return true;
    }
}
