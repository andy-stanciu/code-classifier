class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        List<Character> list = new ArrayList<Character>();
        for (int i = 0; i < s.length(); i++) {
            list.add(s.charAt(i));
        }
        for (int i = 0; i < t.length(); i++) {
            list.remove(new Character(t.charAt(i)));
        }
        if (list.size() == 0) {
            return true;
        }
        return false;
    }
}
