class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        int[] smap = new int[26];
        for (int i = 0; i < s.length(); i++) ++smap[s.charAt(i) - 97];
        for (int i = 0; i < t.length(); i++) {
            char ch = t.charAt(i);
            if (smap[ch - 97] == 0)
                return false;
            --smap[ch - 97];
        }
        return true;
    }
}
