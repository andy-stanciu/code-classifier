class Solution {

    public boolean method1(String s, String t) {
        if (t.length() == s.length()) {
            ArrayList<String> sLetters = new ArrayList<String>();
            for (int i = 0; i < s.length(); i++) {
                sLetters.add(s.substring(i, i + 1));
            }
            for (int i = 0; i < t.length(); i++) {
                if (sLetters.indexOf(t.substring(i, i + 1)) == -1) {
                    return false;
                } else {
                    sLetters.remove(sLetters.indexOf(t.substring(i, i + 1)));
                }
            }
            return true;
        } else {
            return false;
        }
    }
}
