class Solution {

    public boolean method1(String s, String t) {
        int n1 = s.length();
        int n2 = t.length();
        if (n1 != n2) {
            return false;
        }
        int[] arr = new int[26];
        for (int i = 0; i < n1; i++) {
            char ch = s.charAt(i);
            int j = ch - 97;
            arr[j]++;
        }
        for (int i = 0; i < n2; i++) {
            char ch = t.charAt(i);
            int j = ch - 97;
            arr[j]--;
        }
        for (int i = 0; i < 26; i++) {
            if (arr[i] != 0)
                return false;
        }
        return true;
    }
}
