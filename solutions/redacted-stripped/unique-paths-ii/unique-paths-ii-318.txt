class Solution {

    public int method1(int[][] o) {
        int[][] dp = new int[o.length][o[0].length];
        for (int[] num : dp) {
            Arrays.fill(num, -1);
        }
        return helper(o, o.length - 1, o[0].length - 1, o.length, o[0].length, dp);
    }

    private int helper(int[][] o, int i, int j, int n, int m, int[][] dp) {
        if (i < 0 || j < 0 || o[i][j] == 1)
            return 0;
        if (i == 0 && j == 0)
            return 1;
        if (dp[i][j] != -1)
            return dp[i][j];
        return dp[i][j] = (helper(o, i - 1, j, n, m, dp) + helper(o, i, j - 1, n, m, dp));
    }
}
