class Solution {

    public int solve(int row, int col, int[][] a, int[][] dp) {
        int m = a.length;
        int n = a[0].length;
        if (row > m - 1 || col > n - 1)
            return 0;
        if (a[row][col] == 1)
            return 0;
        if (row == m - 1 && col == n - 1)
            return 1;
        if (dp[row][col] != 0)
            return dp[row][col];
        else
            return dp[row][col] = solve(row + 1, col, a, dp) + solve(row, col + 1, a, dp);
    }

    public int method1(int[][] a) {
        int m = a.length;
        int n = a[0].length;
        int[][] dp = new int[m + 1][n + 1];
        dp[m][n] = 1;
        return solve(0, 0, a, dp);
    }
}
