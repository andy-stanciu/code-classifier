class Solution {

    public int method1(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        int[][] dp = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] = -1;
            }
        }
        return grid(obstacleGrid, m - 1, n - 1, dp);
    }

    public int grid(int[][] a, int i, int j, int[][] dp) {
        if (i < 0 || j < 0) {
            return 0;
        }
        if (a[i][j] == 1) {
            return 0;
        }
        if (i == 0 && j == 0) {
            return 1;
        }
        if (dp[i][j] != -1)
            return dp[i][j];
        int up = grid(a, i - 1, j, dp);
        int left = grid(a, i, j - 1, dp);
        dp[i][j] = up + left;
        return dp[i][j];
    }
}
