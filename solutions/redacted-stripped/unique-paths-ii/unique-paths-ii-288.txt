class Solution {

    public int method1(int[][] grid) {
        if (grid[0][0] == 1)
            return 0;
        int m = grid.length;
        int n = grid[0].length;
        int[][] dp = new int[m + 1][n + 1];
        for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= n; j++) {
                if (j == 1) {
                    dp[i][j] = 1;
                }
            }
        }
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = dp[i][j - 1] + dp[i - 1][j];
                if (grid[i - 1][j - 1] == 1) {
                    dp[i][j] = 0;
                }
            }
        }
        return dp[m][n];
    }
}
