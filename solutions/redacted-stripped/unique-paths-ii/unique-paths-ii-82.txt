class Solution {

    public int method1(int[][] obstacleGrid) {
        int n = obstacleGrid[0].length;
        int m = obstacleGrid.length;
        int[][] dp = new int[m + 1][n + 1];
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (i == 1 && j == 1 && obstacleGrid[i - 1][j - 1] != 1) {
                    dp[i][j] = 1;
                    continue;
                }
                dp[i][j] = (obstacleGrid[i - 1][j - 1] != 1) ? dp[i - 1][j] + dp[i][j - 1] : 0;
            }
        }
        return dp[m][n];
    }
}
