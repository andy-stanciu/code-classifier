class Solution {

    public int method1(int[][] grid) {
        int r = grid.length;
        int c = grid[0].length;
        if (grid[r - 1][c - 1] == 1)
            return 0;
        grid[r - 1][c - 1] = 1;
        for (int i = c - 2; i >= 0; i--) {
            if (grid[r - 1][i] == 1)
                grid[r - 1][i] = 0;
            else if (grid[r - 1][i + 1] == 1)
                grid[r - 1][i] = 1;
        }
        for (int i = r - 2; i >= 0; i--) {
            if (grid[i][c - 1] == 1)
                grid[i][c - 1] = 0;
            else if (grid[i + 1][c - 1] == 1)
                grid[i][c - 1] = 1;
        }
        for (int i = r - 2; i >= 0; i--) {
            for (int j = c - 2; j >= 0; j--) {
                if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                } else {
                    grid[i][j] = grid[i + 1][j] + grid[i][j + 1];
                }
            }
        }
        return grid[0][0];
    }
}
