class Solution {

    public int method1(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        if (grid[n - 1][m - 1] == 1) {
            return 0;
        }
        int[][] dp = new int[n + 1][m + 1];
        for (int i = n - 1; i >= 0; --i) {
            for (int j = m - 1; j >= 0; --j) {
                if (i == n - 1 && j == m - 1) {
                    dp[i][j] = 1;
                    continue;
                }
                if (grid[i][j] == 0) {
                    dp[i][j] = dp[i][j + 1] + dp[i + 1][j];
                }
            }
        }
        return dp[0][0];
    }
}
