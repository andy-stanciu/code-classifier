class Solution {

    public int method1(int[][] obstacleGrid) {
        m = obstacleGrid.length;
        n = obstacleGrid[0].length;
        var1 = new Integer[m][n];
        return method2(0, 0, obstacleGrid);
    }

    private int m;

    private int n;

    private Integer[][] var1;

    private int method2(int x, int y, int[][] obstacleGrid) {
        if (x >= m || y >= n || obstacleGrid[x][y] == 1)
            return 0;
        else if (x == m - 1 && y == n - 1)
            return 1;
        else
            return var1[x][y] == null ? var1[x][y] = method2(x + 1, y, obstacleGrid) + method2(x, y + 1, obstacleGrid) : var1[x][y];
    }
}
