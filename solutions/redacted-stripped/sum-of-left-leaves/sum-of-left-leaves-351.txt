class Solution {

    public int solve(TreeNode root) {
        if (root == null)
            return 0;
        int add = 0;
        if (root.left != null && root.left.left == null && root.left.right == null)
            add += root.left.val;
        add += solve(root.left);
        add += solve(root.right);
        return add;
    }

    public int method1(TreeNode root) {
        return solve(root);
    }
}
