class Solution {

    public int method1(TreeNode root, TreeNode parent) {
        if (root == null)
            return 0;
        if (root.right == null && root.left == null && parent.right != root)
            return root.val;
        return method1(root.left, root) + method1(root.right, root);
    }

    public int method2(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        return method1(root.left, root) + method1(root.right, root);
    }
}
