class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        cal(root.left);
        cal1(root.right);
        return var1;
    }

    public void cal(TreeNode root) {
        if (root == null)
            return;
        if (root.left == null && root.right == null)
            var1 = var1 + root.val;
        cal(root.left);
        cal1(root.right);
    }

    public void cal1(TreeNode root) {
        if (root == null)
            return;
        cal(root.left);
        cal1(root.right);
    }
}
