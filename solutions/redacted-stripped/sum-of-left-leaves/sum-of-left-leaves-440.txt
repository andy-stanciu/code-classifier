class Solution {

    public int var1 = 0;

    public boolean isRight = false;

    public int method1(TreeNode root) {
        if (root == null || (root.left == null && root.right == null))
            return 0;
        return var1(root);
    }

    public int var1(TreeNode root) {
        if (root == null)
            return 0;
        if (root.left != null) {
            isRight = false;
            var1(root.left);
        }
        if (root.left == null && root.right == null && isRight == false) {
            var1 += root.val;
        }
        if (root.right != null) {
            isRight = true;
            var1(root.right);
        }
        return var1;
    }
}
