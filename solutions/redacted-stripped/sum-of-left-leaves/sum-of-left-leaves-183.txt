class Solution {

    private int travail(TreeNode root) {
        int count = 0;
        if (root == null) {
            return 0;
        } else if (root.left != null && root.left.left == null && root.left.right == null) {
            return root.left.val + travail(root.right);
        } else {
            count = travail(root.left) + travail(root.right);
            return count;
        }
    }

    public int method1(TreeNode root) {
        return travail(root);
    }
}
