class Solution {

    public int method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        if ((root.left == null && root.right == null) || root == null)
            return 0;
        int var1 = 0;
        q.offer(root);
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                TreeNode node = q.poll();
                if (node.left != null) {
                    q.offer(node.left);
                    if (node.left.left == null && node.left.right == null) {
                        var1 += node.left.val;
                    }
                }
                if (node.right != null) {
                    q.offer(node.right);
                }
            }
        }
        return var1;
    }
}
