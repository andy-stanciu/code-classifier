class Solution {

    public int check(TreeNode root, int ans, int flag) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            if (flag == 0)
                return root.val;
            else
                return 0;
        }
        ans = check(root.left, ans, 0);
        return ans + check(root.right, ans, 1);
    }

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        return check(root, 0, 0);
    }
}
