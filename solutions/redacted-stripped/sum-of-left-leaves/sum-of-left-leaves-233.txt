class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int res = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                res += root.left.val;
            } else {
                res += method1(root.left);
            }
        }
        res += method1(root.right);
        return res;
    }
}
