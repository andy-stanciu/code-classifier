class Solution {

    public int method1(TreeNode root) {
        return dfs(root, 0);
    }

    private int dfs(TreeNode node, int param1) {
        if (node == null)
            return 0;
        if (node.param1 == null && node.right == null) {
            return (param1 == 1) ? node.val : 0;
        }
        int var1 = dfs(node.param1, 1);
        int var2 = dfs(node.right, 0);
        return var1 + var2;
    }
}
