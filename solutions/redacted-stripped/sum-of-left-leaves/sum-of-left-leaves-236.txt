class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        helper(root, false);
        return var1;
    }

    private void helper(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        helper(root.left, true);
        if (param1 == true && root.left == null && root.right == null) {
            var1 += root.val;
        }
        helper(root.right, false);
    }
}
