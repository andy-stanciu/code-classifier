class Solution {

    private int var1;

    public int method1(TreeNode root) {
        var1 = 0;
        dfs(root, false);
        return var1;
    }

    public void dfs(TreeNode node, boolean param1) {
        if (node == null) {
            return;
        }
        if (param1 && node.left == null && node.right == null) {
            var1 += node.val;
        }
        dfs(node.left, true);
        dfs(node.right, false);
    }
}
