class Solution {

    public int checkleaf(TreeNode root) {
        if (root != null && root.left == null && root.right == null)
            return 1;
        return 0;
    }

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (checkleaf(root.left) == 1)
            return root.left.val + method1(root.left) + method1(root.right);
        else
            return method1(root.left) + method1(root.right);
    }

    public int method2(TreeNode root) {
        int t = method1(root);
        return t;
    }
}
