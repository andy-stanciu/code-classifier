class Solution {

    int count = 0;

    public int method1(TreeNode root) {
        recursive(root);
        return count;
    }

    private void recursive(TreeNode node) {
        if (node.left != null) {
            if (node.left.left == null && node.left.right == null) {
                count = count + node.left.val;
            }
            recursive(node.left);
        }
        if (node.right != null) {
            recursive(node.right);
        }
    }
}
