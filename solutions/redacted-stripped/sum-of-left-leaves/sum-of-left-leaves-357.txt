class Solution {

    int ans;

    public int method1(TreeNode root) {
        ans = 0;
        if (root == null)
            return 0;
        method2(root, 0);
        return ans;
    }

    public void method2(TreeNode root, int counter) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && counter < 0) {
            ans += root.val;
            return;
        }
        method2(root.left, -1);
        method2(root.right, 1);
    }
}
