class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        return helper(root, false);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root.left == null && root.right == null) {
            return param1 ? root.val : 0;
        }
        int var1 = 0;
        if (root.left != null) {
            var1 = var1 + helper(root.left, true);
        }
        if (root.right != null) {
            var1 = var1 + helper(root.right, false);
        }
        return var1;
    }
}
