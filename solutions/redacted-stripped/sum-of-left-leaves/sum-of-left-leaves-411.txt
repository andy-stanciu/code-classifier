class Solution {

    int count = 0;

    public int method1(TreeNode root) {
        method2(root, false);
        return count;
    }

    public void method2(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1) {
            count += root.val;
        }
        method2(root.left, true);
        method2(root.right, false);
    }
}
