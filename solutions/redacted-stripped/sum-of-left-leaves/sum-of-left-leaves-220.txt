class Solution {

    public boolean isLeave(TreeNode node) {
        return node.left == null && node.right == null;
    }

    public int method1(TreeNode root) {
        return method1(root.left, true) + method1(root.right, false);
    }

    public int method1(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (isLeave(node) && param1) {
            return node.val;
        }
        return method1(node.left, true) + method1(node.right, false);
    }
}
