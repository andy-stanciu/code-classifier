class Solution {

    public int method1(TreeNode root) {
        int result = 0;
        if (root == null) {
            return 0;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            result += root.left.val;
        }
        result += method1(root.left);
        result += method1(root.right);
        return result;
    }
}
