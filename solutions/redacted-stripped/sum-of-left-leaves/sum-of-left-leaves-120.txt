class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return helper(root.left, true) + helper(root.right, false);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        } else if (root.left == null && root.right == null) {
            return param1 ? root.val : 0;
        } else {
            return helper(root.left, true) + helper(root.right, false);
        }
    }
}
