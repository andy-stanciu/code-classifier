class Solution {

    public int dfs(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null && param1) {
            return root.val;
        }
        return dfs(root.left, true) + dfs(root.right, false);
    }

    public int method1(TreeNode root) {
        return dfs(root, false);
    }
}
