class Solution {

    private int ans = 0;

    public int method1(TreeNode root) {
        dfs(root);
        return ans;
    }

    private void dfs(TreeNode node) {
        if (node == null) {
            return;
        }
        dfs(node.left);
        dfs(node.right);
        if (node.left != null) {
            node = node.left;
            if (node.left == null && node.right == null) {
                ans += node.val;
            }
        }
        return;
    }
}
