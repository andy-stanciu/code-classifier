class Solution {

    static int helper(TreeNode root) {
        if (root == null)
            return 0;
        int a = 0;
        if (root.left != null && root.left.left == null && root.left.right == null)
            a = root.left.val;
        return a + helper(root.left) + helper(root.right);
    }

    public int method1(TreeNode root) {
        return helper(root);
    }
}
