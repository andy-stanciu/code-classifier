class Solution {

    private boolean isLeaf(TreeNode node) {
        return node.left == null && node.right == null;
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        if (root.left != null && isLeaf(root.left)) {
            var1 += root.left.val;
        }
        var1 += method1(root.left);
        var1 += method1(root.right);
        return var1;
    }
}
