class Solution {

    public int method1(TreeNode root, int f) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            if (f == 1)
                return root.val;
            else
                return 0;
        }
        int a = method1(root.left, 1);
        int b = method1(root.right, 0);
        return a + b;
    }

    public int method2(TreeNode root) {
        return method1(root, 0);
    }
}
