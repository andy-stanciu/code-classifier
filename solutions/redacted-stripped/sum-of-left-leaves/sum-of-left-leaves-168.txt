class Solution {

    public int method1(TreeNode root) {
        return solve(root, false);
    }

    public int solve(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && param1 == true)
            return root.val;
        return solve(root.left, true) + solve(root.right, false);
    }
}
