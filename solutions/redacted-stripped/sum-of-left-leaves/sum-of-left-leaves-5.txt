class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int ans = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                ans += root.left.val;
            else
                ans += method1(root.left);
        }
        ans += method1(root.right);
        return ans;
    }
}
