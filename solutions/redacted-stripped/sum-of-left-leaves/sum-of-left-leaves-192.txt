class Solution {

    public int method1(TreeNode root) {
        return dfs(root, false);
    }

    int dfs(TreeNode node, boolean param1) {
        if (node == null)
            return 0;
        if (node.left == null && node.right == null)
            return param1 ? node.val : 0;
        int var1 = dfs(node.left, true);
        var1 += dfs(node.right, false);
        return var1;
    }
}
