class Solution {

    int var1 = 0;

    public void solve(TreeNode root, int flag) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && flag == -1) {
            var1 += root.val;
        }
        solve(root.left, -1);
        solve(root.right, 1);
    }

    public int method1(TreeNode root) {
        solve(root, 0);
        return var1;
    }
}
