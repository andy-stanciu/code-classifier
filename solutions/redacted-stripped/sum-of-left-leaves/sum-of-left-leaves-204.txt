class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        dfs(root, false);
        return var1;
    }

    public void dfs(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        if (param1 == true && root.left == null && root.right == null) {
            var1 += root.val;
            return;
        }
        dfs(root.right, false);
        dfs(root.left, true);
    }
}
