class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return ans;
        }
        solve(root);
        return ans;
    }

    public void solve(TreeNode node) {
        if (node == null) {
            return;
        }
        if (node.left != null && node.left.left == null && node.left.right == null) {
            ans += node.left.val;
            solve(node.left);
        } else {
            solve(node.left);
        }
        if (node.right != null) {
            solve(node.right);
        }
    }
}
