class Solution {

    public static int var1;

    public static int mdepth;

    static void method1(TreeNode root, int depth) {
        if (root == null)
            return;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                var1 += root.left.val;
            }
        }
        method1(root.left, depth + 1);
        method1(root.right, depth + 1);
    }

    public int method2(TreeNode root) {
        var1 = 0;
        mdepth = 0;
        method1(root, 0);
        return var1;
    }
}
