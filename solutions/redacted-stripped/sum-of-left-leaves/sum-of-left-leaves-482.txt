class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        method2(root);
        return ans;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                ans += root.left.val;
            }
        }
        method2(root.left);
        method2(root.right);
    }
}
