class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        if (root != null) {
            if (isLeafNode(root.left)) {
                var1 += root.left.val;
            } else {
                var1 += method1(root.left);
            }
            var1 += method1(root.right);
        }
        return var1;
    }

    public boolean isLeafNode(TreeNode node) {
        if (node == null) {
            return false;
        } else if (node.left == null && node.right == null) {
            return true;
        } else {
            return false;
        }
    }
}
