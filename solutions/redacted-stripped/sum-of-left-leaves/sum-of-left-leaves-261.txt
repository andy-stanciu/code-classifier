class Solution {

    private int method1(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null) {
            return param1 ? node.val : 0;
        }
        int method1 = method1(node.left, true);
        int var1 = method1(node.right, false);
        return method1 + var1;
    }

    public int method2(TreeNode root) {
        return method1(root, false);
    }
}
