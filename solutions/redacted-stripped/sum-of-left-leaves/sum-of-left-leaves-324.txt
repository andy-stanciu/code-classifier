class Solution {

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        int[] arr = new int[1];
        boolean flag = false;
        check(root, arr, flag);
        return arr[0];
    }

    public void check(TreeNode root, int[] param1, boolean flag) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            check(root.left, param1, true);
        }
        if (root.left == null && root.right == null && flag == true) {
            param1[0] += root.val;
            return;
        }
        if (root.right != null) {
            check(root.right, param1, false);
        }
    }
}
