class Solution {

    public void solve(TreeNode node, int[] param1, boolean flag) {
        if (Objects.isNull(node))
            return;
        if (!Objects.nonNull(node.left) && !Objects.nonNull(node.right) && flag)
            param1[0] += node.val;
        solve(node.left, param1, true);
        solve(node.right, param1, false);
    }

    public int method1(TreeNode root) {
        int[] param1 = { 0 };
        solve(root, param1, false);
        return param1[0];
    }
}
