class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return helper(root, false);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        return (param1 && root.left == null && root.right == null ? root.val : 0) + helper(root.left, true) + helper(root.right, false);
    }
}
