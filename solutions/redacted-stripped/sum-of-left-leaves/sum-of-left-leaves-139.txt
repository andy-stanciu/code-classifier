class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    public int method2(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && param1)
            return root.val;
        int var1 = method2(root.left, true);
        int var2 = method2(root.right, false);
        return var1 + var2;
    }
}
