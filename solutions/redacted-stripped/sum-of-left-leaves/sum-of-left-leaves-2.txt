class Pair {

    TreeNode first;

    boolean var1;

    Pair(TreeNode p, boolean l) {
        first = p;
        var1 = l;
    }
}

class Solution {

    public int method1(TreeNode root) {
        int ans = 0;
        Queue<Pair> q = new LinkedList<>();
        q.add(new Pair(root, false));
        while (!q.isEmpty()) {
            Pair removed = q.poll();
            if (removed.first.var1 != null) {
                q.add(new Pair(removed.first.var1, true));
            }
            if (removed.first.right != null) {
                q.add(new Pair(removed.first.right, false));
            }
            if (removed.first.var1 == null && removed.first.right == null) {
                if (removed.var1) {
                    ans = ans + removed.first.val;
                }
            }
        }
        return ans;
    }
}
