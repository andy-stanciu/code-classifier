class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        solve(root);
        return ans;
    }

    public void solve(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            ans += root.left.val;
        else
            solve(root.left);
        solve(root.right);
    }
}
