class Solution {

    public int method1(TreeNode root) {
        return method2(root);
    }

    public int method2(TreeNode root) {
        int var1 = 0;
        if (root == null) {
            return var1;
        }
        if (root.left != null && (root.left.right == null && root.left.left == null)) {
            var1 += root.left.val;
        }
        var1 += method2(root.right);
        var1 += method2(root.left);
        return var1;
    }
}
