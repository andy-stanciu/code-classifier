class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        helper(root, false);
        return var1;
    }

    void helper(TreeNode node, boolean param1) {
        if (node.left == null && node.right == null) {
            if (param1)
                var1 = var1 + node.val;
            return;
        } else {
            if (node.left != null)
                helper(node.left, true);
            if (node.right != null)
                helper(node.right, false);
        }
    }
}
