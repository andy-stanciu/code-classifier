class Solution {

    public int moon(TreeNode root, boolean flag) {
        if (root == null)
            return 0;
        if (flag && root.left == null && root.right == null)
            return root.val;
        return moon(root.left, true) + moon(root.right, false);
    }

    public int method1(TreeNode root) {
        return moon(root, false);
    }
}
