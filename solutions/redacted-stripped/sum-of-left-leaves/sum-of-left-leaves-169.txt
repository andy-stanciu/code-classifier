class Solution {

    private void method1(TreeNode root, boolean flag, int[] param1) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && flag) {
            param1[0] += root.val;
            return;
        }
        method1(root.left, true, param1);
        method1(root.right, false, param1);
    }

    public int method2(TreeNode root) {
        boolean flag = false;
        int[] param1 = { 0 };
        method1(root, flag, param1);
        return param1[0];
    }
}
