class Solution {

    public int method1(TreeNode root) {
        int rightRtn = 0;
        int var1 = 0;
        int var2 = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                var2 = root.left.val;
            } else {
                var1 = method1(root.left);
            }
        }
        if (root.right != null) {
            rightRtn = method1(root.right);
        }
        return var1 + rightRtn + var2;
    }
}
