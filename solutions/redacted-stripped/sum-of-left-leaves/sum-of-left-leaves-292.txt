class Solution {

    public int method1(TreeNode root) {
        return method2(root);
    }

    public int method2(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int var1 = 0;
        if (node.left != null && node.left.left == null && node.left.right == null) {
            var1 += node.left.val;
        }
        var1 += method2(node.left);
        var1 += method2(node.right);
        return var1;
    }
}
