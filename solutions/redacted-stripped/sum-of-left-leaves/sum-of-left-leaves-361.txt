class Solution {

    public int method1(TreeNode root) {
        return Solve(root, null);
    }

    public int Solve(TreeNode curr, TreeNode parent) {
        if (curr == null)
            return 0;
        if (curr.var1 == null && curr.right == null) {
            if (parent != null && parent.var1 == curr) {
                return curr.val;
            }
        }
        int var1 = Solve(curr.var1, curr);
        int right = Solve(curr.right, curr);
        return var1 + right;
    }
}
