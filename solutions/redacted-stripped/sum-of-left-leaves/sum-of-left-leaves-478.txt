class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        helper(root);
        return ans;
    }

    void helper(TreeNode node) {
        if (node == null) {
            return;
        }
        if (node.left != null && node.left.left == null && node.left.right == null) {
            ans += node.left.val;
        }
        helper(node.left);
        helper(node.right);
    }
}
