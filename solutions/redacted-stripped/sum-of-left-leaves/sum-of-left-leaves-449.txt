class Solution {

    int s = 0;

    public int method1(TreeNode root) {
        helper(root);
        return s;
    }

    void helper(TreeNode r) {
        if (r == null)
            return;
        if (r.left != null && r.left.left == null && r.left.right == null) {
            s = s + r.left.val;
        }
        helper(r.left);
        helper(r.right);
    }
}
