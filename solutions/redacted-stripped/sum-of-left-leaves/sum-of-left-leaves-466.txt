class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    private int method2(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null && param1) {
            return node.val;
        }
        int var1 = method2(node.left, true);
        int var2 = method2(node.right, false);
        return var1 + var2;
    }
}
