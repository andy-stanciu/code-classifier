class Solution {

    static int var1 = 0;

    public int method1(TreeNode root) {
        var1 = 0;
        leaf(root);
        return var1;
    }

    public void leaf(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null && root.left.right == null && root.left.left == null) {
            var1 += root.left.val;
        }
        leaf(root.left);
        leaf(root.right);
    }
}
