class Solution {

    public int var1;

    public void helper(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                var1 += root.left.val;
            } else {
                helper(root.left);
            }
        }
        helper(root.right);
    }

    public int method1(TreeNode root) {
        var1 = 0;
        helper(root);
        return var1;
    }
}
