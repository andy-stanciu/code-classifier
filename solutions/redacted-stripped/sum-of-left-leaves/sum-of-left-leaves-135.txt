class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    public int method2(TreeNode node, boolean flag) {
        if (node == null) {
            return 0;
        }
        if (flag && node.left == null && node.right == null) {
            return node.val;
        }
        return method2(node.left, true) + method2(node.right, false);
    }
}
