class Solution {

    public int var1 = 0;

    public int method1(TreeNode root) {
        class Traverse {

            Traverse(TreeNode root) {
                TreeNode temp = root;
                if (temp.var1 != null) {
                    if (temp.var1.var1 == null && temp.var1.right == null)
                        var1 += temp.var1.val;
                    new Traverse(temp.var1);
                }
                if (temp.right != null) {
                    new Traverse(temp.right);
                }
            }
        }
        new Traverse(root);
        return var1;
    }
}
