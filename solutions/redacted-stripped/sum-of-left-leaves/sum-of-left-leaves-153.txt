class Solution {

    public int method1(TreeNode root) {
        return method2(root, false, 0);
    }

    public int method2(TreeNode root, boolean param1, int param2) {
        if (root == null)
            return param2;
        if (root.left == null && root.right == null) {
            if (param1)
                return param2 + root.val;
            else
                return param2;
        }
        return method2(root.right, false, param2) + method2(root.left, true, param2);
    }
}
