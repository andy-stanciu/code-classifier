class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return 0;
        }
        helper(root, false);
        return var1;
    }

    void helper(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        if (param1 && root.left == null && root.right == null) {
            var1 += root.val;
            return;
        }
        helper(root.left, true);
        helper(root.right, false);
    }
}
