class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            var1 += root.left.val;
        }
        if (root.left != null)
            method1(root.left);
        if (root.right != null)
            method1(root.right);
        return var1;
    }
}
