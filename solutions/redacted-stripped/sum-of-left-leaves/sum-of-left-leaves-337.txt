class Solution {

    public int var1 = 0;

    public void method1(TreeNode root, boolean param1) {
        if (param1 && root.left == null && root.right == null) {
            var1 = var1 + root.val;
            return;
        }
        if (root.left != null) {
            method1(root.left, true);
        }
        if (root.right != null) {
            method1(root.right, false);
        }
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        method1(root, false);
        return var1;
    }
}
