class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        dfs(root, false);
        return var1;
    }

    public void dfs(TreeNode node, boolean param1) {
        if (node == null)
            return;
        if ((node.left == null && node.right == null) && param1) {
            var1 += node.val;
            return;
        }
        dfs(node.left, true);
        dfs(node.right, false);
    }
}
