class Solution {

    int res = 0;

    public void traverse(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            return;
        }
        if (root.left != null) {
            traverse(root.left);
            if (root.left.left == null && root.left.right == null) {
                res += root.left.val;
            }
        }
        if (root.right != null) {
            traverse(root.right);
        }
    }

    public int method1(TreeNode root) {
        traverse(root);
        return res;
    }
}
