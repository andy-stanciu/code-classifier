class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int val = 0;
        if (root.left != null && root.left.left == null && root.left.right == null)
            val = root.left.val;
        System.out.println(val + " " + root.val);
        int var1 = method1(root.left);
        int var2 = method1(root.right);
        return val + var1 + var2;
    }
}
