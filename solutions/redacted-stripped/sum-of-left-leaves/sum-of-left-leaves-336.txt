class Solution {

    private boolean isLeaf(TreeNode root) {
        return root.left == null && root.right == null;
    }

    private int dfs(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        int var1 = 0;
        if (isLeaf(root)) {
            if (param1) {
                var1 += root.val;
            }
        }
        var1 += dfs(root.left, true);
        var1 += dfs(root.right, false);
        return var1;
    }

    public int method1(TreeNode root) {
        return dfs(root, false);
    }
}
