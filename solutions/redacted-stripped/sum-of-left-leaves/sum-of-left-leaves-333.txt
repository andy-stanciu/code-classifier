class Solution {

    public int method1(TreeNode root) {
        int answer = 0;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            TreeNode cur = queue.poll();
            if (cur.left != null) {
                TreeNode var1 = cur.left;
                if (var1.left == null && var1.right == null) {
                    answer += var1.val;
                }
                queue.add(var1);
            }
            if (cur.right != null)
                queue.add(cur.right);
        }
        return answer;
    }
}
