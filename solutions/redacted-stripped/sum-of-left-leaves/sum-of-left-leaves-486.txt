class Solution {

    public int method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        int var1 = 0;
        q.add(root);
        while (!q.isEmpty()) {
            TreeNode node = q.poll();
            if (node.left != null && node.left.left == null && node.left.right == null) {
                var1 += node.left.val;
            }
            if (node.left != null) {
                q.add(node.left);
            }
            if (node.right != null) {
                q.add(node.right);
            }
        }
        return var1;
    }
}
