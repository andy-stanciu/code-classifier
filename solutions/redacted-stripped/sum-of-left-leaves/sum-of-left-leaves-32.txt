class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        dfs(root, var1);
        return var1[0];
    }

    public void dfs(TreeNode root, int[] var1) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            var1[0] += root.left.val;
        if (root.left != null)
            dfs(root.left, var1);
        if (root.right != null)
            dfs(root.right, var1);
    }
}
