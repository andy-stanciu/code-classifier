class Solution {

    private int method1(TreeNode node, String type) {
        if (node == null)
            return 0;
        int var1 = 0;
        if ((node.left == null && node.right == null) && type == "left")
            return node.val;
        return method1(node.left, "left") + method1(node.right, "right");
    }

    public int method2(TreeNode root) {
        return method1(root, "root");
    }
}
