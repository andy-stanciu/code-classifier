class Solution {

    int var1;

    public int method1(TreeNode root) {
        var1 = 0;
        traverseBST(root);
        return var1;
    }

    private void traverseBST(TreeNode node) {
        if (node == null)
            return;
        if (node.left != null && node.left.left == null && node.left.right == null)
            var1 += node.left.val;
        traverseBST(node.left);
        traverseBST(node.right);
    }
}
