class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        method2(root, false);
        return var1;
    }

    private void method2(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (param1 && root.left == null && root.right == null)
            var1 += root.val;
        method2(root.left, true);
        method2(root.right, false);
    }
}
