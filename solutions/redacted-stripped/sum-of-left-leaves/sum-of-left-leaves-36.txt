class Solution {

    public int method1(TreeNode node) {
        if (node == null)
            return 0;
        int var1 = 0;
        if (node.left != null && node.left.left == null && node.left.right == null)
            var1 = node.left.val;
        return var1 + method1(node.left) + method1(node.right);
    }
}
