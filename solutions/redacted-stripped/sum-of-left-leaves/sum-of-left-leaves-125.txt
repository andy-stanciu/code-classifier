class Solution {

    public int method1(TreeNode root) {
        return method2(root, root);
    }

    public int method2(TreeNode root, TreeNode prev) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            if (prev.left == root)
                return root.val;
        } else {
            return method2(root.left, root) + method2(root.right, root);
        }
        return 0;
    }
}
