class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        return recursive(root, false);
    }

    int recursive(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        int var1 = 0;
        if (param1 && node.left == null && node.right == null) {
            var1 += node.val;
            return var1;
        }
        var1 += recursive(node.left, true);
        var1 += recursive(node.right, false);
        return var1;
    }
}
