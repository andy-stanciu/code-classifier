class Solution {

    public int method1(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        int val = 0;
        if (root.left == null && root.right == null && param1) {
            val = root.val;
        }
        return val + method1(root.left, true) + method1(root.right, false);
    }

    public int method2(TreeNode root) {
        return method1(root, false);
    }
}
