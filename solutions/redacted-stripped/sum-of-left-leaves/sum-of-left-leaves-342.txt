class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int result = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                result += root.left.val;
            } else {
                result += method1(root.left);
            }
        }
        result += method1(root.right);
        return result;
    }
}
