class Solution {

    public int method1(TreeNode root) {
        return dfs(root, false);
    }

    public int dfs(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            return param1 ? root.val : 0;
        }
        int var1 = dfs(root.left, true);
        int var2 = dfs(root.right, false);
        return var1 + var2;
    }
}
