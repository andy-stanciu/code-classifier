class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        if (root.var1 != null) {
            if (root.var1.var1 == null && root.var1.right == null) {
                var1 = root.var1.val;
            } else {
                var1 = method1(root.var1);
            }
        }
        int right = method1(root.right);
        return var1 + right;
    }
}
