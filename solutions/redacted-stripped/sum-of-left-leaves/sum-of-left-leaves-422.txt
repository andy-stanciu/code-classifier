class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        if (root.left != null && root.left.right == null && root.left.left == null) {
            return root.left.val + method1(root.right);
        }
        return method1(root.right) + method1(root.left);
    }
}
