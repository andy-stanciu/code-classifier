class Solution {

    public int method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        findleaf(root, list, false);
        int var1 = 0;
        for (int num : list) {
            var1 += num;
        }
        return var1;
    }

    public TreeNode findleaf(TreeNode root, List<Integer> list, boolean val) {
        if (root == null) {
            return null;
        }
        TreeNode var2 = findleaf(root.var2, list, true);
        TreeNode right = findleaf(root.right, list, false);
        if (var2 == null && right == null && val == true) {
            list.add(root.val);
        }
        return root;
    }
}
