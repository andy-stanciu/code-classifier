class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    public int method2(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (param1 && root.left == null && root.right == null) {
            return root.val;
        } else if (!param1 && root.left == null && root.right == null) {
            return 0;
        } else {
            return method2(root.left, true) + method2(root.right, false);
        }
    }
}
