class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        helper(root, false);
        return var1;
    }

    public void helper(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (param1)
                var1 += root.val;
        }
        helper(root.left, true);
        helper(root.right, false);
    }
}
