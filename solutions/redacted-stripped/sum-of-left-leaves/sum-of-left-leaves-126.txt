class Solution {

    public int method1(TreeNode root) {
        if (root == null || (root.left == null && root.right == null)) {
            return 0;
        }
        return helper(root, false);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            if (param1) {
                return root.val;
            }
        }
        if (root.left == null) {
            return helper(root.right, false);
        }
        if (root.right == null) {
            return helper(root.left, true);
        }
        int a = helper(root.left, true);
        int b = helper(root.right, false);
        return a + b;
    }
}
