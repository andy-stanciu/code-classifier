class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        traverse(root.left, 1);
        traverse(root.right, 2);
        return var1;
    }

    private void traverse(TreeNode root, int leave) {
        if (root == null)
            return;
        traverse(root.left, 1);
        if (root.left == null && root.right == null && leave == 1)
            var1 += root.val;
        traverse(root.right, 2);
    }
}
