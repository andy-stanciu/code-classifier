class Solution {

    int s = 0;

    public void f(TreeNode root) {
        if (root != null) {
            if (root.left != null && root.left.left == null && root.left.right == null)
                s += root.left.val;
            f(root.left);
            f(root.right);
        }
    }

    public int method1(TreeNode root) {
        if (root.right == null && root.left == null)
            return 0;
        f(root);
        return s;
    }
}
