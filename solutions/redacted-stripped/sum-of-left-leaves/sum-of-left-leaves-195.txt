class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        evaluateTree(root.left, true);
        evaluateTree(root.right, false);
        return var1;
    }

    private void evaluateTree(TreeNode node, boolean param1) {
        if (node == null)
            return;
        if (node.left == null && node.right == null && param1) {
            var1 += node.val;
            return;
        }
        evaluateTree(node.left, true);
        evaluateTree(node.right, false);
    }
}
