class Solution {

    public int method1(TreeNode root) {
        return helper(root, false);
    }

    private int helper(TreeNode root, boolean flag) {
        if (root == null)
            return 0;
        int var1 = 0;
        if (flag && root.left == null && root.right == null)
            var1 += root.val;
        return var1 + helper(root.left, true) + helper(root.right, false);
    }
}
