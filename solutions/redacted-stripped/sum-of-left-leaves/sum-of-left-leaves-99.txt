class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int[] res = new int[1];
        find(root, res, false);
        return res[0];
    }

    void find(TreeNode root, int[] res, boolean leaf) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && leaf) {
            res[0] += root.val;
        }
        find(root.left, res, true);
        find(root.right, res, false);
    }
}
