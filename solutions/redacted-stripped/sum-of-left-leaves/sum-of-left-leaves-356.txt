class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        int var1 = 0;
        while (!queue.isEmpty()) {
            TreeNode node = queue.poll();
            if (node.left != null) {
                queue.add(node.left);
                if (node.left.left == null && node.left.right == null) {
                    var1 += node.left.val;
                }
            }
            if (node.right != null) {
                queue.add(node.right);
            }
        }
        return var1;
    }
}
