class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int res = 0;
        if (isLeaf(root.left))
            res += root.left.val;
        res += method1(root.left);
        res += method1(root.right);
        return res;
    }

    public boolean isLeaf(TreeNode node) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            return true;
        return false;
    }
}
