class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        solve(root);
        return var1;
    }

    public void solve(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            var1 += root.left.val;
        solve(root.left);
        solve(root.right);
    }
}
