class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        if (root.left == null)
            ;
        else if (root.left.left != null || root.left.right != null)
            var1 += method1(root.left);
        else
            var1 += root.left.val;
        if (root.right == null)
            ;
        else if (root.right.left != null || root.right.right != null)
            var1 += method1(root.right);
        return var1;
    }
}
