class Solution {

    public int method1(TreeNode root, boolean flag) {
        if (root == null)
            return 0;
        ;
        if (root.left == null && root.right == null && flag == true)
            return root.val;
        if (root.left == null && root.right == null && flag == false)
            return 0;
        return method1(root.left, true) + method1(root.right, false);
    }

    public int method2(TreeNode root) {
        return method1(root, false);
    }
}
