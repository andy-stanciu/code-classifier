class Solution {

    public int method1(TreeNode root, int flag) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null && flag == 1) {
            return root.val;
        } else if (root.left == null && root.right == null && flag != 1) {
            return 0;
        }
        return method1(root.left, 1) + method1(root.right, 0);
    }

    public int method2(TreeNode root) {
        return method1(root, 0);
    }
}
