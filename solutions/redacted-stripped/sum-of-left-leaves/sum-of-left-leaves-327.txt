class Solution {

    static int var1;

    public static void method1(TreeNode root, int flag) {
        if (root == null)
            return;
        ;
        if (flag == 1 && root.left == null && root.right == null) {
            var1 += root.val;
        }
        method1(root.left, 1);
        method1(root.right, 0);
    }

    public int method2(TreeNode root) {
        var1 = 0;
        method1(root, 0);
        return var1;
    }
}
