class Solution {

    public int method1(TreeNode root) {
        return dfs(root.left, true) + dfs(root.right, false);
    }

    private int dfs(TreeNode root, boolean isL) {
        if (root == null)
            return 0;
        if ((root.left == null && root.right == null) && isL)
            return root.val;
        if ((root.left == null && root.right == null) && !isL)
            return 0;
        return dfs(root.left, true) + dfs(root.right, false);
    }
}
