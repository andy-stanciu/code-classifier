class Solution {

    public int getAns(TreeNode root, boolean flag) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null && flag) {
            return root.val;
        }
        int var1 = getAns(root.left, true);
        int var2 = getAns(root.right, false);
        return var1 + var2;
    }

    public int method1(TreeNode root) {
        return getAns(root, false);
    }
}
