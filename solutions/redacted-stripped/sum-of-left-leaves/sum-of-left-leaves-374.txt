class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int l = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                l = root.left.val;
        }
        if (l == 0)
            l = method1(root.left);
        int r = method1(root.right);
        return l + r;
    }
}
