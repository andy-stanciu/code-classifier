class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        helper(root);
        return var1;
    }

    void helper(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                var1 += root.left.val;
        }
        helper(root.left);
        helper(root.right);
    }
}
