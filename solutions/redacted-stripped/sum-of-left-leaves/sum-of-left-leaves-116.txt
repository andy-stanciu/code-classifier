class Solution {

    int var1 = 0;

    public void method1(TreeNode root) {
        if (root == null)
            return;
        if (isLeafNode(root.left))
            var1 += root.left.val;
        method1(root.left);
        method1(root.right);
    }

    private boolean isLeafNode(TreeNode root) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return true;
        return false;
    }

    public int method2(TreeNode root) {
        method1(root);
        return var1;
    }
}
