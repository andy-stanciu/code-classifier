class Solution {

    public int method1(TreeNode root, boolean isRightPath) {
        int var1 = 0;
        int var2 = 0;
        if (root == null)
            return 0;
        if (isRightPath && root.left == null && root.right == null)
            return 0;
        if (root.left == null && root.right == null)
            return root.val;
        if (root.right != null && (root.right.left != null || root.right.right != null))
            var2 = var2 + method1(root.right, true);
        var1 = var1 + method1(root.left, false);
        return var2 + var1;
    }

    public int method2(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        return method1(root, false);
    }
}
