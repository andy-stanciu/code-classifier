class Solution {

    public int var1;

    public void helper(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (param1 && root.left == null && root.right == null) {
            var1 += root.val;
        }
        helper(root.left, true);
        helper(root.right, false);
    }

    public int method1(TreeNode root) {
        var1 = 0;
        helper(root, false);
        return var1;
    }
}
