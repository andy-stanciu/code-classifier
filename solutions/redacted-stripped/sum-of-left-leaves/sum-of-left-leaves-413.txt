class Solution {

    public int method1(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.param1 == null && root.right == null) {
            if (param1) {
                return root.val;
            }
            return 0;
        }
        int a = method1(root.param1, true);
        int b = method1(root.right, false);
        return a + b;
    }

    public int method2(TreeNode root) {
        int a = method1(root, false);
        return a;
    }
}
