class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        preOrderBinaryTree(root, var1);
        return var1[0];
    }

    public static void preOrderBinaryTree(TreeNode root, int[] var1) {
        if (root == null) {
            return;
        }
        if (root.left != null && (root.left.left == null && root.left.right == null)) {
            var1[0] += root.left.val;
        }
        preOrderBinaryTree(root.left, var1);
        preOrderBinaryTree(root.right, var1);
    }
}
