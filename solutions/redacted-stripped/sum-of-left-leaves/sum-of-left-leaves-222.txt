class Solution {

    int var1 = 0;

    public void dfs(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                var1 += root.left.val;
            }
            dfs(root.left);
        }
        if (root.right != null) {
            dfs(root.right);
        }
    }

    public int method1(TreeNode root) {
        dfs(root);
        return var1;
    }
}
