class Solution {

    private int method1(TreeNode root) {
        if (root.left != null && root.left.left == null && root.left.right == null)
            return root.left.val;
        return 0;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = 0;
        var1 += method2(root.left);
        var1 += method1(root);
        var1 += method2(root.right);
        return var1;
    }
}
