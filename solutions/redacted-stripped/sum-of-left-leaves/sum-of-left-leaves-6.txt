class Solution {

    public int fres = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return fres;
        preorder(root, false);
        return fres;
    }

    public void preorder(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if ((root.left == null && root.right == null) && param1) {
            fres += root.val;
        }
        preorder(root.left, true);
        preorder(root.right, false);
    }
}
