class Solution {

    public int method1(TreeNode root) {
        int[] var1 = { 0 };
        dfs(root, var1);
        return var1[0];
    }

    private void dfs(TreeNode root, int[] var1) {
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                var1[0] += root.left.val;
            }
            dfs(root.left, var1);
        }
        if (root.right != null) {
            dfs(root.right, var1);
        }
    }
}
