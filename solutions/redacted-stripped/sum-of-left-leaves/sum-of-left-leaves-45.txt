class Solution {

    public int method1(TreeNode root) {
        int s = 0;
        if (root == null)
            return 0;
        if (root.left != null && root.left.right == null && root.left.left == null)
            s += root.left.val;
        s += method1(root.left);
        s += method1(root.right);
        return s;
    }
}
