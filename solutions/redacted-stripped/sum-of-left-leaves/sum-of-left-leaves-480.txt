class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    private int method2(TreeNode node, boolean param1) {
        if (node.param1 == null && node.right == null && param1) {
            return node.val;
        }
        int var1 = 0;
        int var2 = 0;
        if (node.param1 != null) {
            var1 = method2(node.param1, true);
        }
        if (node.right != null) {
            var2 = method2(node.right, false);
        }
        return var1 + var2;
    }
}
