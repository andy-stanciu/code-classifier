class Solution {

    public int dfs(TreeNode node, Boolean param1) {
        int var1 = 0;
        if (node == null) {
            return 0;
        }
        if (node.param1 == null && node.right == null && param1) {
            return node.val;
        }
        if (node.param1 != null) {
            var1 += dfs(node.param1, true);
        }
        if (node.right != null) {
            var1 += dfs(node.right, false);
        }
        return var1;
    }

    public int method1(TreeNode root) {
        return dfs(root, false);
    }
}
