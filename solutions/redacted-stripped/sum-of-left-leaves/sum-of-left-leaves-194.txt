class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        pre(root);
        return var1;
    }

    void pre(TreeNode root) {
        if (root != null) {
            if (root.left != null && root.left.left == null && root.left.right == null)
                var1 += root.left.val;
            pre(root.left);
            pre(root.right);
        }
    }
}
