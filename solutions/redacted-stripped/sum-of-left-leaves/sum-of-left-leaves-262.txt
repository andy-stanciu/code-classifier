class Solution {

    public int method1(TreeNode root) {
        return helper(root);
    }

    int helper(TreeNode root) {
        if (root == null) {
            return 0;
        }
        if (null != root.left && null == root.left.left && null == root.left.right) {
            return root.left.val + helper(root.right);
        }
        return helper(root.left) + helper(root.right);
    }
}
