class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        else {
            if (root.left != null)
                return method1(root.left) + method1(root.right) + ((root.left.left == null && root.left.right == null) ? root.left.val : 0);
            else
                return method1(root.left) + method1(root.right);
        }
    }
}
