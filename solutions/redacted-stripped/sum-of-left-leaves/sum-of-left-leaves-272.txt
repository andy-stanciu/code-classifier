class Solution {

    int count = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                count += root.left.val;
        }
        method1(root.left);
        method1(root.right);
        return count;
    }
}
