class Solution {

    int method1(TreeNode root) {
        if (root == null)
            return 0;
        int ans = 0;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            ans += root.left.val;
        }
        ans += method1(root.left);
        ans += method1(root.right);
        return ans;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        return method1(root);
    }
}
