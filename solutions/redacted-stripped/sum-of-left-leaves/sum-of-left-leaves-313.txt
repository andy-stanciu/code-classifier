class Solution {

    int res = 0;

    public int method1(TreeNode root) {
        method2(root);
        return res;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            System.out.println(root.left.val + " ");
            res += root.left.val;
        }
        method2(root.left);
        method2(root.right);
    }
}
