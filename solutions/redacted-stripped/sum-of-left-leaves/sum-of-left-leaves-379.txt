class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        help(root, 0);
        return var1;
    }

    void help(TreeNode root, int f) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && f == 1) {
            var1 += root.val;
            return;
        }
        help(root.left, 1);
        help(root.right, 0);
    }
}
