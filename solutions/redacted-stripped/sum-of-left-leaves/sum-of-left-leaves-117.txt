class Solution {

    public int method1(TreeNode root) {
        int lr = 0;
        if (root.left != null) {
            lr += method1(root.left);
            if ((root.left.left == null) && (root.left.right == null)) {
                lr += root.left.val;
            }
        }
        if (root.right != null) {
            lr += method1(root.right);
        }
        return lr;
    }
}
