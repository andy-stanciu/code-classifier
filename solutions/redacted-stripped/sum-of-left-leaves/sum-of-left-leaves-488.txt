class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        helper(root, false);
        return var1;
    }

    public void helper(TreeNode node, boolean flag) {
        if (node == null) {
            return;
        }
        if (node.left == null && node.right == null && flag) {
            var1 += node.val;
        }
        helper(node.left, true);
        helper(node.right, false);
    }
}
