class Solution {

    public int method1(TreeNode root) {
        if (root == null || (root.left == null && root.right == null))
            return 0;
        return method1(root, "left");
    }

    public int method1(TreeNode root, String s) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && s.equals("left")) {
            System.out.println(root.val);
            return root.val;
        }
        return method1(root.left, "left") + method1(root.right, "right");
    }
}
