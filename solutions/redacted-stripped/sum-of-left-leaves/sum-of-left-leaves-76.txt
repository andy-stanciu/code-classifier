class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        int ans = 0;
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            return 0;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            ans = ans + root.left.val;
        }
        return method1(root.left) + method1(root.right) + ans;
    }
}
