class Solution {

    public int method1(TreeNode root) {
        return method1(root, false);
    }

    private int method1(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null) {
            if (param1) {
                return node.val;
            } else {
                return 0;
            }
        }
        return method1(node.left, true) + method1(node.right, false);
    }
}
