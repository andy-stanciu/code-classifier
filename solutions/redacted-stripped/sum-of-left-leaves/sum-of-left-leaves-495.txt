class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root != null) {
            traverse(root);
        }
        return var1;
    }

    private void traverse(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            if (isLeafNode(root.left)) {
                var1 += root.left.val;
            } else {
                traverse(root.left);
            }
        }
        if (root.right != null) {
            traverse(root.right);
        }
    }

    private boolean isLeafNode(TreeNode node) {
        return node != null && node.left == null && node.right == null;
    }
}
