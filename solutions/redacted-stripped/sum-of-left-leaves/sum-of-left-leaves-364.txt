class Solution {

    private void Magic(int[] res, TreeNode root) {
        if (root == null)
            return;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                res[0] += root.left.val;
        }
        Magic(res, root.left);
        Magic(res, root.right);
        return;
    }

    public int method1(TreeNode root) {
        int[] res = { 0 };
        Magic(res, root);
        return res[0];
    }
}
