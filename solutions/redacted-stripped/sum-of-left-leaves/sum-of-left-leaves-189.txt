class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return var1;
        }
        preorderHelper(root, false);
        return var1;
    }

    void preorderHelper(TreeNode node, boolean param1) {
        if (node == null) {
            return;
        }
        if (node.left == null && node.right == null && param1) {
            var1 += node.val;
        }
        preorderHelper(node.left, true);
        preorderHelper(node.right, false);
    }
}
