class Solution {

    public int method1(TreeNode root) {
        return method2(root, 0);
    }

    int method2(TreeNode t, int d) {
        if (t == null)
            return 0;
        if (t.left == null && t.right == null && d == -1)
            return t.val;
        return method2(t.left, -1) + method2(t.right, 1);
    }
}
