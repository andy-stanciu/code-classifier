class Solution {

    private int count;

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            count += root.left.val;
        }
        method1(root.left);
        method1(root.right);
    }

    public int method2(TreeNode root) {
        count = 0;
        method1(root);
        return count;
    }
}
