class Solution {

    public int method1(TreeNode root) {
        tree(root);
        return number;
    }

    public int number = 0;

    public void tree(TreeNode x) {
        if (x == null)
            return;
        if (x.left != null) {
            TreeNode y = x.left;
            if (y.left == null && y.right == null)
                number += y.val;
        }
        tree(x.left);
        tree(x.right);
    }
}
