class Solution {

    public int method1(TreeNode root) {
        return helper(root, "");
    }

    private int helper(TreeNode node, String direction) {
        if (node.left == null && node.right == null && direction == "left")
            return node.val;
        int var1 = 0;
        if (node.left != null)
            var1 += helper(node.left, "left");
        if (node.right != null)
            var1 += helper(node.right, "right");
        return var1;
    }
}
