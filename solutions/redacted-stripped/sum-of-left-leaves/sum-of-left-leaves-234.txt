class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return check(root, "");
    }

    public int check(TreeNode root, String s) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            if (s.equals("l")) {
                return root.val;
            }
            return 0;
        }
        return check(root.left, "l") + check(root.right, "r");
    }
}
