class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        if (root.left != null) {
            if (isLeaf(root.left)) {
                var1 += root.left.val;
            } else {
                var1 += method1(root.left);
            }
        }
        if (root.right != null) {
            var1 += method1(root.right);
        }
        return var1;
    }

    public boolean isLeaf(TreeNode root) {
        return root.left == null && root.right == null;
    }
}
