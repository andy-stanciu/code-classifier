class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                var1 += root.left.val;
            method1(root.left);
        }
        if (root.right != null)
            method1(root.right);
        return var1;
    }
}
