class Solution {

    public int method1(TreeNode root) {
        boolean k = false;
        return helper(root, k);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        if (param1 && root.left == null && root.right == null) {
            var1 += root.val;
            return var1;
        }
        var1 += helper(root.left, true);
        var1 += helper(root.right, false);
        return var1;
    }
}
