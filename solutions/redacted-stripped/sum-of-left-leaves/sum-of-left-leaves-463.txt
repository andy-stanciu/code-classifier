class Solution {

    int solve(TreeNode root, int k) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && k == 1)
            return root.val;
        int a = solve(root.left, 1);
        int b = solve(root.right, 0);
        return a + b;
    }

    public int method1(TreeNode root) {
        return solve(root, 0);
    }
}
