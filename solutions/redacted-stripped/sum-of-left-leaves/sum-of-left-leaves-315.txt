class Solution {

    int var1 = 0;

    public boolean check(TreeNode root) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return true;
        else
            return false;
    }

    public void find(TreeNode root) {
        if (root == null)
            return;
        if (check(root.left))
            var1 += root.left.val;
        find(root.left);
        find(root.right);
    }

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        find(root);
        return var1;
    }
}
