class Solution {

    public int method1(TreeNode root) {
        return traverse(root);
    }

    public int traverse(TreeNode root) {
        if (root == null)
            return 0;
        int ans = 0;
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null)
                ans = root.left.val;
        }
        return traverse(root.left) + traverse(root.right) + ans;
    }
}
