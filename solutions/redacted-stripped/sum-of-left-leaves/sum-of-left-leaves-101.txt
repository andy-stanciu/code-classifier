class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return var1;
        }
        dfs(root);
        return var1;
    }

    void dfs(TreeNode node) {
        if (node == null) {
            return;
        }
        if (node.left != null && node.left.left == null && node.left.right == null) {
            var1 += node.left.val;
        }
        dfs(node.left);
        dfs(node.right);
        return;
    }
}
