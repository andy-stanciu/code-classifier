class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        Stack<TreeNode> stack = new Stack<>();
        boolean var2 = false;
        stack.add(root);
        while (!stack.isEmpty()) {
            TreeNode curr = stack.pop();
            if (curr.left == null && curr.right == null && var2) {
                var1 += curr.val;
                var2 = false;
            }
            if (curr.right != null) {
                stack.push(curr.right);
                var2 = false;
            }
            if (curr.left != null) {
                stack.push(curr.left);
                var2 = true;
            }
        }
        return var1;
    }
}
