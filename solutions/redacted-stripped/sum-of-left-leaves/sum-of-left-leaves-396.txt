class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (isSafe(root.left)) {
            ans += root.left.val;
        }
        method1(root.left);
        method1(root.right);
        return ans;
    }

    public boolean isSafe(TreeNode root) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return true;
        else
            return false;
    }
}
