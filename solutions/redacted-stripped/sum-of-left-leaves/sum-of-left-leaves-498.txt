class Solution {

    public int var1 = 0;

    public int method1(TreeNode root) {
        helper(root, -1);
        return var1;
    }

    public int helper(TreeNode root, int idx) {
        if (root == null) {
            return 0;
        }
        if (idx == 1 && root.var2 == null && root.right == null) {
            var1 += root.val;
        }
        int var2 = helper(root.var2, 1);
        int right = helper(root.right, 0);
        return var2 + right;
    }
}
