class Solution {

    public int method1(TreeNode root) {
        return inorder(root, false);
    }

    private int inorder(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (param1 && root.left == null && root.right == null)
            return root.val;
        return inorder(root.left, true) + inorder(root.right, false);
    }
}
