class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        display(root, false);
        return var1;
    }

    public void display(TreeNode node, boolean param1) {
        if (node == null) {
            return;
        }
        display(node.left, true);
        if (node.left == null && node.right == null && param1) {
            var1 = var1 + node.val;
        }
        display(node.right, false);
    }
}
