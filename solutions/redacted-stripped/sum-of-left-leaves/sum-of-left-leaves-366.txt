class Solution {

    int res = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return res;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            res += root.left.val;
        }
        method1(root.left);
        method1(root.right);
        return res;
    }
}
