class Solution {

    public int method1(TreeNode root) {
        return dfs(root, false);
    }

    public int dfs(TreeNode node, Boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null) {
            if (param1) {
                return node.val;
            }
            return 0;
        }
        return dfs(node.left, true) + dfs(node.right, false);
    }
}
