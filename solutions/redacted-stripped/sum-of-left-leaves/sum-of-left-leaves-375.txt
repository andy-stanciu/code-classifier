class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        if (root != null) {
            if (root.left != null) {
                if (root.left.left == null && root.left.right == null) {
                    var1 += root.left.val;
                }
            }
            var1 += method1(root.left);
            var1 += method1(root.right);
        }
        return var1;
    }
}
