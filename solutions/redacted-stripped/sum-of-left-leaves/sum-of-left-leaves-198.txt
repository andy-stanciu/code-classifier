class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        dfs(root);
        return ans;
    }

    public void dfs(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                ans = ans + root.left.val;
            }
        }
        dfs(root.left);
        dfs(root.right);
    }
}
