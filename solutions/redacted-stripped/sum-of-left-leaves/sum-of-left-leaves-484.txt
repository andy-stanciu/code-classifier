class Solution {

    static int var1 = 0;

    public int method1(TreeNode root) {
        var1 = 0;
        method2(root, -1);
        return var1;
    }

    public static void method2(TreeNode root, int v) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && v == 0) {
            var1 = var1 + root.val;
            return;
        }
        method2(root.left, 0);
        method2(root.right, 1);
    }
}
