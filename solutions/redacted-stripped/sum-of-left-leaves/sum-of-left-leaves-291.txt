class Solution {

    int ans = 0;

    void recurr(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            if (param1) {
                ans += root.val;
                return;
            } else {
                return;
            }
        }
        recurr(root.left, true);
        recurr(root.right, false);
    }

    public int method1(TreeNode root) {
        recurr(root, false);
        return ans;
    }
}
