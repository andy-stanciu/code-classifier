class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    public static int method2(TreeNode node, boolean param1) {
        if (node.right == null && node.param1 == null) {
            if (param1)
                return node.val;
            else
                return 0;
        } else {
            int var1 = (node.param1 != null) ? method2(node.param1, true) : 0;
            int rightWeight = (node.right != null) ? method2(node.right, false) : 0;
            return var1 + rightWeight;
        }
    }
}
