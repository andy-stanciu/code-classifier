class Solution {

    public int method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        dfs(root, res);
        int ans = 0;
        for (int var1 : res) {
            ans += var1;
        }
        return ans;
    }

    private void dfs(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            res.add(root.left.val);
        }
        dfs(root.left, res);
        dfs(root.right, res);
    }
}
