class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        dfs(root, false);
        return var1;
    }

    public void dfs(TreeNode node, boolean param1) {
        if (node.left == null && node.right == null && param1) {
            var1 += node.val;
        }
        if (node.left != null)
            dfs(node.left, true);
        if (node.right != null)
            dfs(node.right, false);
    }
}
