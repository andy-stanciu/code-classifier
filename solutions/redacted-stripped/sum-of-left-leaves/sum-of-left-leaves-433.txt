class Solution {

    public int helper(TreeNode curr, TreeNode par) {
        if (curr == null)
            return 0;
        if (curr.var1 == null && curr.right == null) {
            if (par != null && par.var1 == curr) {
                return curr.val;
            }
        }
        int var1 = helper(curr.var1, curr);
        int right = helper(curr.right, curr);
        return var1 + right;
    }

    public int method1(TreeNode root) {
        return helper(root, null);
    }
}
