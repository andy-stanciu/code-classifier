class Solution {

    public int recur(TreeNode root, int param1) {
        if (root == null) {
            return 0;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            param1 = root.left.val;
        } else {
            param1 = 0;
        }
        param1 += recur(root.left, param1);
        param1 += recur(root.right, param1);
        return param1;
    }

    public int method1(TreeNode root) {
        int param1 = 0;
        return recur(root, param1);
    }
}
