class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null || (root.left == null && root.right == null)) {
            return 0;
        }
        if (root.left != null && root.left.left == null && root.left.right == null) {
            var1 += root.left.val;
        }
        method1(root.left);
        method1(root.right);
        return var1;
    }
}
