class Solution {

    public int method1(TreeNode root) {
        return solve(root, false);
    }

    int solve(TreeNode cur, boolean param1) {
        if (cur == null)
            return 0;
        if (cur.left == null && cur.right == null && param1) {
            return cur.val;
        }
        return solve(cur.left, true) + solve(cur.right, false);
    }
}
