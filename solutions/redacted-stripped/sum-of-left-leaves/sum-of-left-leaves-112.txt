class Solution {

    int var1 = 0;

    public int helper(TreeNode root, boolean check) {
        if (root == null)
            return var1;
        if (root.left == null && root.right == null) {
            if (check)
                var1 += root.val;
            return var1;
        }
        helper(root.left, true);
        helper(root.right, false);
        return var1;
    }

    public int method1(TreeNode root) {
        return helper(root, false);
    }
}
