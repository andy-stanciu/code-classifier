class Solution {

    private int helper(TreeNode node, int side) {
        if (node.left == null && node.right == null && side == 0) {
            return node.val;
        }
        int ans = 0;
        if (node.left != null) {
            ans += helper(node.left, 0);
        }
        if (node.right != null) {
            ans += helper(node.right, 1);
        }
        return ans;
    }

    public int method1(TreeNode root) {
        return helper(root, -1);
    }
}
