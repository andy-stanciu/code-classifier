class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (q.size() > 0) {
            TreeNode curr = q.remove();
            if (curr.left != null) {
                if (isLeaf(curr.left))
                    var1 += curr.left.val;
                q.add(curr.left);
            }
            if (curr.right != null)
                q.add(curr.right);
        }
        return var1;
    }

    boolean isLeaf(TreeNode curr) {
        if (curr.left == null && curr.right == null)
            return true;
        return false;
    }
}
