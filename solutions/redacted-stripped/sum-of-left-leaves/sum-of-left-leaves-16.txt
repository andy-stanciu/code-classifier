class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        traversal(root);
        return var1;
    }

    public TreeNode traversal(TreeNode root) {
        if (root == null) {
            return null;
        }
        if (root.left != null) {
            if (root.left.left == null && root.left.right == null) {
                var1 += root.left.val;
            }
        }
        traversal(root.left);
        traversal(root.right);
        return null;
    }
}
