class Solution {

    public int method1(TreeNode root) {
        return method1(root, false);
    }

    public int method1(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (param1 == true && root.param1 == null && root.right == null)
            return root.val;
        return method1(root.param1, true) + method1(root.right, false);
    }
}
