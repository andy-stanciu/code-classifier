class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    public static int method2(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null && param1) {
            return node.val;
        }
        return method2(node.left, true) + method2(node.right, false);
    }
}
