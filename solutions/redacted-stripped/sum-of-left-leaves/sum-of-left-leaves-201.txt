class Solution {

    private int calculate(TreeNode root, boolean param1, int param2) {
        if (root == null) {
            return param2;
        }
        if (root.left == null && root.right == null) {
            if (param1) {
                param2 = param2 + root.val;
            }
            return param2;
        }
        param2 = calculate(root.left, true, param2);
        param2 = calculate(root.right, false, param2);
        return param2;
    }

    public int method1(TreeNode root) {
        return calculate(root, false, 0);
    }
}
