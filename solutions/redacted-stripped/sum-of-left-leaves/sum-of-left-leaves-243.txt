class Solution {

    public int method1(TreeNode root) {
        return solve(root, false);
    }

    public int solve(TreeNode root, boolean check) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            if (check) {
                return root.val;
            } else {
                return 0;
            }
        }
        int l = solve(root.left, true);
        int r = solve(root.right, false);
        return l + r;
    }
}
