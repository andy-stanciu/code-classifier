class Solution {

    int var1 = 0;

    private void find(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && param1)
            var1 += root.val;
        find(root.left, true);
        find(root.right, false);
    }

    public int method1(TreeNode root) {
        find(root, false);
        return var1;
    }
}
