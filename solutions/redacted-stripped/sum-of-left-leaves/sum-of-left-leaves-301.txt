class Solution {

    public int method1(TreeNode root) {
        return helper(root, false);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (root.param1 == null && root.right == null && param1)
            return root.val;
        return helper(root.param1, true) + helper(root.right, false);
    }
}
