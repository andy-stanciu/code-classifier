class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        else
            return method2(null, root);
    }

    public int method2(TreeNode prev, TreeNode trav) {
        if (trav == null)
            return 0;
        if (trav.left == null && trav.right == null && prev != null && prev.left == trav)
            return trav.val;
        if (trav.left != null && trav.right != null)
            return method2(trav, trav.left) + method2(trav, trav.right);
        else if (trav.left != null)
            return method2(trav, trav.left);
        else
            return method2(trav, trav.right);
    }
}
