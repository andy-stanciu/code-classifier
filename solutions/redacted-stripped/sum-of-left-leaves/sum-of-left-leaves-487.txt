class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        traverse(root, false);
        return this.var1;
    }

    void traverse(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1) {
            this.var1 += root.val;
            return;
        }
        traverse(root.left, true);
        traverse(root.right, false);
    }
}
