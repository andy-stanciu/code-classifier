class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if ((root.left != null) && (root.left.left == null) && (root.left.right == null)) {
            ans += root.left.val;
        }
        method1(root.left);
        method1(root.right);
        return ans;
    }
}
