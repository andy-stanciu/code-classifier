class Solution {

    public int method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int var1 = 0;
        while (!q.isEmpty()) {
            int n = q.size();
            for (int i = 0; i < n; i++) {
                TreeNode current = q.poll();
                if (current.left != null) {
                    if (current.left.left == null && current.left.right == null) {
                        var1 += current.left.val;
                    }
                    q.add(current.left);
                }
                if (current.right != null) {
                    q.add(current.right);
                }
            }
        }
        return var1;
    }
}
