class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        var1(root, var1);
        return var1[0];
    }

    public void var1(TreeNode root, int[] s) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            s[0] += root.left.val;
        }
        var1(root.left, s);
        var1(root.right, s);
    }
}
