class Solution {

    public int method1(TreeNode root) {
        return cal(root, false);
    }

    public int cal(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null) {
            return param1 ? node.val : 0;
        }
        return cal(node.left, true) + cal(node.right, false);
    }
}
