class Solution {

    public int method1(int[] a) {
        int n = a.length;
        int st = 0;
        int end = n - 1;
        int ans = (int) 1e9;
        while (st <= end) {
            int mid = st + (end - st) / 2;
            if (a[st] <= a[mid]) {
                ans = Math.min(ans, a[st]);
                st = mid + 1;
            } else {
                ans = Math.min(ans, a[mid]);
                end = mid - 1;
            }
        }
        return ans;
    }
}
