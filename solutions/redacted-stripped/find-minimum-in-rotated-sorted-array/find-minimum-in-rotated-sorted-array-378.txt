class Solution {

    public int method1(int[] arr) {
        int n = arr.length;
        int low = 0;
        int high = n - 1;
        int ans = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (arr[low] <= arr[high]) {
                ans = Math.min(ans, arr[low]);
                break;
            }
            if (arr[low] <= arr[mid]) {
                low = low + 1;
                ans = Math.min(ans, arr[low]);
            } else {
                high = high - 1;
                ans = Math.min(ans, arr[mid]);
            }
        }
        return ans;
    }
}
