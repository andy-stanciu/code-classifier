class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        int j = nums.length - 1;
        int i = 0;
        if (nums.length == 1)
            return nums[0];
        while (i < j) {
            if (nums[i] < nums[j]) {
                var1 = nums[i];
                j--;
            } else {
                var1 = nums[j];
                i++;
            }
        }
        return var1;
    }
}
