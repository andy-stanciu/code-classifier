class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return 0;
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] < var1) {
                var1 = nums[i];
            }
        }
        return var1;
    }
}
