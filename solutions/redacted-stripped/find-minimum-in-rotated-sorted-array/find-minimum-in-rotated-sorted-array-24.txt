class Solution {

    public int method1(int[] nums) {
        return perform(nums, 0, nums.length - 1);
    }

    public int perform(int[] arr, int s, int e) {
        if (s == e || arr[s] < arr[e]) {
            return arr[s];
        }
        int m = s + (e - s) / 2;
        if (arr[m] >= arr[s]) {
            return perform(arr, m + 1, e);
        } else {
            return perform(arr, s, m);
        }
    }
}
