class Solution {

    public int method1(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int var1 = nums[0];
        while (low <= high) {
            int mid = low + (high - low) / 2;
            var1 = Math.var1(var1, nums[mid]);
            if (nums[low] <= nums[mid]) {
                var1 = Math.var1(nums[low], var1);
                low = mid + 1;
            } else {
                var1 = Math.var1(nums[high], var1);
                high = mid - 1;
            }
        }
        return var1;
    }
}
