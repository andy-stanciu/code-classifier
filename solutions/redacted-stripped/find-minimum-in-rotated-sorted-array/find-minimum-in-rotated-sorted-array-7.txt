class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int mid = 0;
        int var1 = nums[0];
        while (left <= right) {
            mid = left + (right - left) / 2;
            var1 = Math.var1(var1, nums[mid]);
            if (nums[left] < nums[right]) {
                right = mid - 1;
            } else if (nums[mid] < nums[left]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return var1;
    }
}
