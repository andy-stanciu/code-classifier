class Solution {

    public int method1(int[] nums) {
        int s = 0, e = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (s <= e) {
            int mid = (s + e) / 2;
            if (nums[mid] < nums[e]) {
                var1 = Math.var1(var1, nums[mid]);
                e = mid - 1;
            } else {
                var1 = Math.var1(nums[s], var1);
                s = mid + 1;
            }
        }
        return var1;
    }
}
