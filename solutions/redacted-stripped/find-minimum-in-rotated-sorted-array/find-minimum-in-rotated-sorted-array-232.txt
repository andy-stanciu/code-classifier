class Solution {

    public int method1(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        int len = nums.length;
        int mid = 0;
        while (start <= end) {
            mid = start + (end - start) / 2;
            if (mid != 0 && mid != len - 1 && nums[mid] < nums[mid - 1] && nums[mid] < nums[mid + 1])
                return nums[mid];
            else if (nums[mid] < nums[end])
                end = mid - 1;
            else
                start = mid + 1;
        }
        return nums[mid];
    }
}
