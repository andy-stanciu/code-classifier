class Solution {

    public int method1(int[] nums) {
        int pivot = -1;
        int var1 = nums[0];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i + 1] < nums[i]) {
                pivot = i + 1;
                break;
            }
        }
        if (pivot == -1) {
            return nums[0];
        }
        if (nums[pivot] < var1) {
            var1 = nums[pivot];
        }
        return var1;
    }
}
