class Solution {

    public int method1(int[] nums) {
        int l = 0;
        int r = nums.length - 1;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (isTrue(mid, nums, r)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return nums[l];
    }

    public boolean isTrue(int param1, int[] arr, int r) {
        return param1 < arr.length && arr[param1] < arr[r];
    }
}
