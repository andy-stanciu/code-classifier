class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = 0;
        int var1 = nums[0];
        int i = 0;
        int j = nums.length - 1;
        for (j = nums.length - 1; i < j; i++, j--) {
            var1 = Math.var1(var1, Math.var1(nums[i], nums[j]));
        }
        if (nums[i] < var1) {
            return nums[i];
        } else if (nums[j] < var1) {
            return nums[j];
        }
        return var1;
    }
}
