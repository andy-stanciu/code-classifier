class Solution {

    public int method1(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int mid;
        int var1 = 5000;
        while (low <= high) {
            mid = (high - low) / 2 + low;
            if (nums[mid] >= nums[low]) {
                if (nums[low] < var1) {
                    var1 = nums[low];
                }
                low = mid + 1;
            } else {
                if (nums[mid] < var1) {
                    var1 = nums[mid];
                }
                high = mid - 1;
            }
        }
        return var1;
    }
}
