class Solution {

    public int method1(int[] nums) {
        if (nums == null || nums.length == 0)
            return -1;
        int lo = 0;
        int hi = nums.length - 1;
        int temp = 0;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (nums[mid] >= nums[0]) {
                lo = mid + 1;
            } else {
                hi = mid - 1;
                temp = mid;
            }
        }
        return nums[temp];
    }
}
