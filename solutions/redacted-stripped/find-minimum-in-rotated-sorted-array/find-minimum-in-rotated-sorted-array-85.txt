class Solution {

    public int method1(int[] nums) {
        int i = 0, j = nums.length - 1;
        int mid;
        while (i < j) {
            mid = (i + j) / 2;
            if (nums[mid] > nums[j])
                i = mid + 1;
            else
                j = mid;
        }
        return nums[j];
    }
}
