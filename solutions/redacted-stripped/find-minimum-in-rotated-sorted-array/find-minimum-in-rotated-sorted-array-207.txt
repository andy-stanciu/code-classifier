class Solution {

    public int method1(int[] nums) {
        int var1 = method2(nums);
        return nums[var1];
    }

    public int method2(int[] nums) {
        int lo = 0;
        int hi = nums.length - 1;
        int mid = -1;
        while (lo <= hi) {
            mid = lo + (hi - lo) / 2;
            if (nums[mid] < nums[hi]) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return mid;
    }
}
