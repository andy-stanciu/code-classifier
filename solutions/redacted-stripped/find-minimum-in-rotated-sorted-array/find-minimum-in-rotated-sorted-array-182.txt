class Solution {

    public static int method1(int[] arr, int l, int r) {
        if (r >= l) {
            int m;
            m = l + (r - l) / 2;
            if (m < r && l < m && arr[m] < arr[m - 1] && arr[m] < arr[m + 1])
                return arr[m];
            if (r == l)
                return arr[m];
            if (arr[m] > arr[m + 1] && m < r)
                return arr[m + 1];
            if (arr[m] < arr[m - 1] && m > l)
                return arr[m - 1];
            if (arr[m] > arr[r])
                return method1(arr, m + 1, r);
            if (arr[m] < arr[l])
                return method1(arr, l, m - 1);
        }
        return arr[0];
    }

    public int method2(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        if (nums.length == 2)
            return Math.min(nums[0], nums[1]);
        return method1(nums, 0, nums.length - 1);
    }
}
