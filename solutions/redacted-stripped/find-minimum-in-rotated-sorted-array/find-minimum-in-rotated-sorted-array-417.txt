class Solution {

    public int method1(int[] nums) {
        int s = 0;
        int e = nums.length - 1;
        while (s <= e) {
            int m = (s + e) / 2;
            if ((m - 1 < 0 || nums[m] > nums[m - 1]) && (m + 1 == nums.length || nums[m] > nums[m + 1]))
                return nums[(m + 1) % nums.length];
            else if (nums[s] <= nums[m])
                s = m + 1;
            else
                e = m - 1;
        }
        return -1;
    }
}
