class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int var1 = Integer.MAX_VALUE;
        int si = 0;
        int ei = n - 1;
        while (si <= ei) {
            int mid = si + (ei - si) / 2;
            if (var1 > nums[mid])
                var1 = nums[mid];
            if (nums[si] <= nums[mid]) {
                if (nums[si] < var1)
                    var1 = nums[si];
                si = mid + 1;
            } else if (nums[ei] >= nums[mid]) {
                ei = mid - 1;
            }
        }
        return var1;
    }
}
