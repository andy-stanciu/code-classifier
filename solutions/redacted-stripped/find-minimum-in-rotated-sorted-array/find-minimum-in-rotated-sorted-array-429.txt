class Solution {

    public int method1(int[] nums) {
        int st = 0;
        int ed = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (st <= ed) {
            int mid = st + (ed - st) / 2;
            var1 = Math.var1(var1, nums[mid]);
            if (nums[st] <= nums[mid]) {
                if (nums[st] <= nums[ed]) {
                    ed = mid - 1;
                } else {
                    st = mid + 1;
                }
            } else {
                ed = mid - 1;
            }
        }
        return var1;
    }
}
