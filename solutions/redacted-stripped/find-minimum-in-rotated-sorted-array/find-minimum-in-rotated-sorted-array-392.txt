class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        int[] var1 = { Integer.MAX_VALUE };
        method2(nums, 0, len - 1, var1);
        return var1[0];
    }

    public void method2(int[] nums, int s, int e, int[] var1) {
        if (e < s)
            return;
        int mid = (e + s) / 2;
        var1[0] = Math.var1(var1[0], nums[mid]);
        if (mid == 0) {
            method2(nums, mid + 1, e, var1);
            return;
        }
        if (nums[mid - 1] < nums[e])
            method2(nums, s, mid - 1, var1);
        else
            method2(nums, mid + 1, e, var1);
        return;
    }
}
