class Solution {

    public int method1(int[] nums) {
        int left = 0, right = (nums.length) - 1;
        int small = Integer.MAX_VALUE;
        int mid = 0;
        while (left <= right) {
            mid = left + (right - left) / 2;
            if (nums[mid] >= nums[left]) {
                small = Math.min(small, nums[left]);
                left = mid + 1;
            } else {
                small = Math.min(small, nums[mid]);
                right = mid - 1;
            }
        }
        return small;
    }
}
