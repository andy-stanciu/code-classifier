class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int var1 = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (right > mid && nums[mid] > nums[mid + 1]) {
                var1 = nums[mid + 1];
            }
            if (left < mid && nums[mid - 1] > nums[mid]) {
                var1 = nums[mid];
            }
            if (nums[mid] >= nums[left]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        if (var1 == -1) {
            var1 = nums[0];
        }
        return var1;
    }
}
