class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int n = right;
        if (nums[0] <= nums[right])
            return nums[0];
        while (left <= right) {
            int mid = (left + right) / 2;
            if (nums[mid] > nums[n])
                left = mid + 1;
            else
                right = mid - 1;
        }
        return nums[left];
    }
}
