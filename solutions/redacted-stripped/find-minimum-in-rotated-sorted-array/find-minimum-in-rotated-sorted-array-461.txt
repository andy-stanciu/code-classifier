class Solution {

    public int method1(int[] nums) {
        int i = 0, j = nums.length - 1;
        if (nums[i] < nums[j]) {
            return nums[i];
        }
        while (i < j - 1) {
            int m = (i + j) / 2;
            if (nums[i] > nums[m]) {
                j = m;
            } else {
                i = m;
            }
        }
        return nums[j];
    }
}
