class Solution {

    public int method1(int[] nums) {
        int mid = (0 + nums.length) / 2;
        int var1 = nums[mid];
        if (nums[mid] > nums[0] && nums[mid] < nums[nums.length - 1]) {
            return nums[0];
        }
        if (nums[mid] > nums[0]) {
            var1 = nums[0];
            for (int i = mid + 1; i < nums.length; i++) {
                var1 = Math.var1(var1, nums[i]);
            }
        } else {
            var1 = nums[mid];
            for (int i = mid - 1; i >= 0; i--) {
                var1 = Math.var1(var1, nums[i]);
            }
        }
        return var1;
    }
}
