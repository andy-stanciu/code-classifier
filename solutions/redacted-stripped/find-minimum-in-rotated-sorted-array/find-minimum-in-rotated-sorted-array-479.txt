class Solution {

    public int method1(int[] nums) {
        int pivot = method2(nums);
        return nums[pivot + 1];
    }

    public int method2(int[] arr) {
        int s = 0, e = arr.length - 1;
        while (s <= e) {
            int m = s + (e - s);
            if (m < e && arr[m] > arr[m + 1])
                return m;
            if (m > s && arr[m] < arr[m - 1])
                return m - 1;
            if (arr[m] <= arr[s])
                e = m - 1;
            else
                s = m + 1;
        }
        return -1;
    }
}
