class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int j = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (i <= j) {
            int mid = i + (j - i) / 2;
            var1 = Math.var1(nums[mid], var1);
            if (nums[i] > nums[j]) {
                if (nums[mid] > nums[j]) {
                    i = mid + 1;
                } else {
                    j = mid - 1;
                }
            } else {
                var1 = Math.var1(nums[i], var1);
                j = mid - 1;
            }
        }
        return var1;
    }
}
