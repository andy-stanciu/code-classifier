class Solution {

    private boolean isValid(int mid, int[] nums) {
        return nums[mid] <= nums[nums.length - 1];
    }

    public int method1(int[] nums) {
        int var1 = -1;
        int valid = nums.length - 1;
        while (Math.abs(valid - var1) > 1) {
            int mid = (valid + var1) / 2;
            if (isValid(mid, nums)) {
                valid = mid;
            } else {
                var1 = mid;
            }
        }
        return nums[valid];
    }
}
