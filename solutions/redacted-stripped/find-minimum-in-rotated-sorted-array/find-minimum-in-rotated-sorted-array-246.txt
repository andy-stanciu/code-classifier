class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int s = 0;
        int e = nums.length - 1;
        int m;
        while (e >= s) {
            m = s + (e - s) / 2;
            if (m == 0) {
                if (nums[m] < nums[m + 1])
                    return nums[0];
                else
                    return nums[1];
            }
            if (m == nums.length - 1) {
                if (nums[m] < nums[m - 1])
                    return nums[m];
                else
                    return nums[m - 1];
            }
            if (nums[m] < nums[m - 1] && nums[m] < nums[m + 1])
                return nums[m];
            else if (nums[m] > nums[nums.length - 1])
                s = m + 1;
            else
                e = m - 1;
        }
        return -1;
    }
}
