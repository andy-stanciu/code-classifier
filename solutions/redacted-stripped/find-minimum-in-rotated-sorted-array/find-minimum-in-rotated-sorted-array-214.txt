class Solution {

    public int method1(int[] nums) {
        return method2(nums, 0, nums.length - 1, Integer.MAX_VALUE);
    }

    public static int method2(int[] nums, int low, int high, int param1) {
        if (low > high)
            return param1;
        int mid = (low + high) / 2;
        if (nums[mid] < nums[high]) {
            param1 = Math.param1(param1, nums[mid]);
            high = mid - 1;
        } else if (nums[mid] >= nums[low]) {
            param1 = Math.param1(param1, nums[low]);
            low = mid + 1;
        }
        return method2(nums, low, high, param1);
    }
}
