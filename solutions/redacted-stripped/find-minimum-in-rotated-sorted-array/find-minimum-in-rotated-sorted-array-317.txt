class Solution {

    public int method1(int[] nums) {
        int l = 0;
        int r = nums.length - 1;
        int res = nums[0];
        while (l <= r) {
            if (nums[l] < nums[r]) {
                res = Math.min(res, nums[l]);
            }
            int mid = (l + r) / 2;
            res = Math.min(res, nums[mid]);
            if (nums[mid] >= nums[l]) {
                l = mid + 1;
            } else if (nums[mid] < nums[l]) {
                r = mid - 1;
            }
        }
        return res;
    }
}
