class Solution {

    public int method1(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int mid = 0;
        while (low <= high) {
            mid = low + (high - low) / 2;
            if (nums[low] <= nums[high]) {
                return nums[low];
            }
            if (nums[mid] > nums[high]) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        return 1;
    }
}
