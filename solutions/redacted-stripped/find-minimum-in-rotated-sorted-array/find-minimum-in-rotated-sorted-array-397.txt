class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        int var1 = 1000000000;
        int start = 0;
        int end = len - 1;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (nums[start] <= nums[mid]) {
                var1 = Math.min(var1, nums[start]);
                start = mid + 1;
            } else {
                var1 = Math.min(var1, nums[mid]);
                end = mid - 1;
            }
        }
        return var1;
    }
}
