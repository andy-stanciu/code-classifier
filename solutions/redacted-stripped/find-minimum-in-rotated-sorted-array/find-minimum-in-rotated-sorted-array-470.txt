class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int var1 = nums[0];
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                var1 = nums[i + 1];
                break;
            }
        }
        return var1;
    }
}
