class Solution {

    public int method1(int[] arr) {
        int var1 = Integer.MAX_VALUE;
        int low = 0;
        int high = arr.length - 1;
        while (low <= high) {
            int mid = (low + high) >> 1;
            if (arr[low] <= arr[mid] && arr[mid] >= arr[high]) {
                var1 = Math.var1(var1, arr[low]);
                low = mid + 1;
            } else {
                var1 = Math.var1(var1, arr[mid]);
                high = mid - 1;
            }
        }
        return var1;
    }
}
