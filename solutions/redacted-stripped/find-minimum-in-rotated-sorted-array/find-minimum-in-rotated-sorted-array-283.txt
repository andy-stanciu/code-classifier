class Solution {

    public int method1(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        int result = 5000;
        while (start <= end) {
            int mid = (end - start) / 2 + start;
            result = Math.min(result, nums[mid]);
            if (nums[mid] > nums[end]) {
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return result;
    }
}
