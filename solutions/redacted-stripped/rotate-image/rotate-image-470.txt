class Solution {

    public void method1(int[][] matrix) {
        int lastPos = matrix[0].length - 1;
        int iterationCount = matrix[0].length / 2;
        for (int it = 0; it < iterationCount; it++) {
            for (int offset = it; offset < lastPos - it; offset++) {
                int vFirstRow = matrix[it][offset];
                int vLastRow = matrix[lastPos - it][lastPos - offset];
                int vFirstColumn = matrix[lastPos - offset][it];
                int vLastColumn = matrix[offset][lastPos - it];
                matrix[it][offset] = vFirstColumn;
                matrix[offset][lastPos - it] = vFirstRow;
                matrix[lastPos - it][lastPos - offset] = vLastColumn;
                matrix[lastPos - offset][it] = vLastRow;
            }
        }
    }
}
