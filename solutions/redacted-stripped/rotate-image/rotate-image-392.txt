class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        int i = 0, j = 0, temp;
        while (i < n / 2) {
            temp = matrix[i][j];
            matrix[i][j] = matrix[n - 1 - j][i];
            matrix[n - 1 - j][i] = matrix[n - 1 - i][n - 1 - j];
            matrix[n - 1 - i][n - 1 - j] = matrix[j][n - 1 - i];
            matrix[j][n - 1 - i] = temp;
            if (++j == n - 1 - i)
                j = ++i;
        }
    }
}
