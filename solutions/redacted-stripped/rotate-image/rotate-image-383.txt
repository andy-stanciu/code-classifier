class Solution {

    public void method1(int[][] matrix) {
        int rowlen = matrix.length;
        int collen = matrix[0].length;
        for (int i = 0; i < rowlen; i++) {
            for (int j = i + 1; j < rowlen; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        for (int k = 0; k < rowlen; k++) {
            int i = 0;
            int j = rowlen - 1;
            while (i < j) {
                int temp = matrix[k][i];
                matrix[k][i] = matrix[k][j];
                matrix[k][j] = temp;
                i++;
                j--;
            }
        }
    }
}
