class Solution {

    public void method1(int[][] matrix) {
        int row = matrix.length;
        int column = matrix[0].length;
        for (int i = 0; i < row; i++) {
            for (int j = i; j < column; j++) {
                int tmp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = tmp;
            }
        }
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < column / 2; j++) {
                int tmp = matrix[i][j];
                matrix[i][j] = matrix[i][column - j - 1];
                matrix[i][column - j - 1] = tmp;
            }
        }
    }
}
