class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        int l = 0, s = m - 1;
        while (l < s) {
            int[] temp2 = matrix[l];
            matrix[l] = matrix[s];
            matrix[s] = temp2;
            l++;
            s--;
        }
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < m; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
    }
}
