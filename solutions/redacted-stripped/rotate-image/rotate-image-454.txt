class Solution {

    public void method1(int[][] matrix) {
        for (int i = 0; i < matrix.length / 2; i++) {
            for (int j = 0; j < Math.ceil(matrix.length / 2.0); j++) {
                method1(matrix, i, j);
            }
        }
    }

    public void method1(int[][] matrix, int i, int j) {
        int temp = matrix[i][j], temp2 = matrix[i][j];
        int ni = i, nj = j;
        for (int it = 0; it < 4; it++) {
            ni = j;
            nj = matrix.length - 1 - i;
            temp2 = matrix[ni][nj];
            matrix[ni][nj] = temp;
            temp = temp2;
            i = ni;
            j = nj;
        }
    }
}
