class Solution {

    public void method1(int[][] matrix) {
        Stack<Integer> qu = new Stack<>();
        int n = matrix.length, m = matrix[0].length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int temp = matrix[i][j];
                qu.add(temp);
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = m - 1; j >= 0; j--) {
                matrix[j][i] = qu.pop();
            }
        }
    }
}
