class Solution {

    public void method1(int[][] matrix) {
        int countLevels = matrix.length / 2;
        int upWall = 0;
        int righthWall = matrix.length - 1;
        ;
        int leftWall = 0;
        int downWall = matrix.length - 1;
        for (int level = 0; level < countLevels; level++) {
            int currentMatrixLength = matrix.length - (level * 2);
            int countIterations = currentMatrixLength - 1;
            int lengthToZero = matrix.length - level - 1;
            for (int i = level; i < countIterations + level; i++) {
                int s1 = matrix[upWall][i];
                matrix[upWall][i] = matrix[lengthToZero][leftWall];
                matrix[lengthToZero][leftWall] = matrix[downWall][lengthToZero];
                matrix[downWall][lengthToZero] = matrix[i][righthWall];
                matrix[i][righthWall] = s1;
                lengthToZero--;
            }
            upWall += 1;
            righthWall -= 1;
            leftWall += 1;
            downWall -= 1;
        }
    }
}
