class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                swap(matrix, i, j);
            }
        }
        for (int i = 0; i < n; i++) {
            reverse(matrix[i]);
        }
    }

    public void swap(int[][] matrix, int i, int j) {
        int temp = matrix[i][j];
        matrix[i][j] = matrix[j][i];
        matrix[j][i] = temp;
    }

    public void reverse(int[] matrix) {
        int l = 0, h = matrix.length - 1;
        while (l <= h) {
            int temp = matrix[l];
            matrix[l] = matrix[h];
            matrix[h] = temp;
            l++;
            h--;
        }
    }
}
