class Solution {

    public void method1(int[][] M) {
        int n = M.length - 1, m = (n + 1) / 2;
        for (int i = 0; i < m; i++, n--) {
            for (int j = i; j < n; j++) {
                int xJ = M.length - j - 1;
                int temp = M[i][j];
                M[i][j] = M[xJ][i];
                M[xJ][i] = M[n][xJ];
                M[n][xJ] = M[j][n];
                M[j][n] = temp;
            }
        }
    }
}
