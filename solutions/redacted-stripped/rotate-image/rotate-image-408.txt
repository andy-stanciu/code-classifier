class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        int temp = 0;
        int k = n - 1;
        for (int i = 0; i < k; i++) {
            for (int j = 0; j < k - i; j++) {
                temp = matrix[i][j];
                matrix[i][j] = matrix[k - j][k - i];
                matrix[k - j][k - i] = temp;
            }
        }
        for (int i = 0; i < n / 2; i++) {
            int[] row = matrix[i];
            matrix[i] = matrix[k - i];
            matrix[k - i] = row;
        }
    }
}
