class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i < n / 2) {
                    int temp = matrix[j][i];
                    matrix[j][i] = matrix[j][(n - 1) - i];
                    matrix[j][(n - 1) - i] = temp;
                }
            }
        }
    }
}
