class Solution {

    static void transposeInPlace(int[][] matrix, int n) {
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
    }

    static void reverseArray(int[] arr) {
        int i = 0, j = arr.length - 1;
        while (i < j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
    }

    public void method1(int[][] matrix) {
        int n = matrix.length;
        transposeInPlace(matrix, n);
        for (int i = 0; i < n; i++) {
            reverseArray(matrix[i]);
        }
    }
}
