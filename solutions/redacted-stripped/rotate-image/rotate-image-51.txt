class Solution {

    public static void method1(int[][] matrix) {
        for (int deep = 0; deep < matrix.length / 2; deep++) {
            for (int offset = deep; offset < matrix[deep].length - 1 - deep; offset++) {
                int holdFirst = matrix[deep][offset];
                matrix[deep][offset] = matrix[matrix.length - 1 - offset][deep];
                matrix[matrix.length - 1 - offset][deep] = matrix[matrix.length - 1 - deep][matrix[deep].length - 1 - offset];
                matrix[matrix.length - 1 - deep][matrix[deep].length - 1 - offset] = matrix[offset][matrix[deep].length - 1 - deep];
                matrix[offset][matrix[deep].length - 1 - deep] = holdFirst;
            }
        }
    }
}
