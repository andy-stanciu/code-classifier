class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length - 1;
        int m = matrix[0].length - 1;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j <= m; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        for (int i = 0; i <= n; i++) {
            reverse(matrix[i]);
        }
    }

    public void reverse(int[] arr) {
        int left = 0;
        int right = arr.length - 1;
        while (left < right) {
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        }
    }
}
