class Solution {

    public void method1(int[][] matrix) {
        int l = matrix.length;
        for (int i = 0; i < l; i++) {
            for (int j = 0; j < i; j++) {
                swap(matrix, i, j);
            }
        }
        for (int[] col : matrix) {
            reverse(col);
        }
    }

    public void swap(int[][] mat, int i, int j) {
        int temp = mat[i][j];
        mat[i][j] = mat[j][i];
        mat[j][i] = temp;
    }

    public void reverse(int[] col) {
        int i = 0;
        int j = col.length - 1;
        while (i < j) {
            int temp = col[i];
            col[i] = col[j];
            col[j] = temp;
            i++;
            j--;
        }
    }
}
