class Solution {

    public void method1(int[][] arr) {
        int r = arr.length;
        int c = arr[0].length;
        for (int i = 0; i < r; i++) {
            for (int j = i; j < c; j++) {
                int t = arr[i][j];
                arr[i][j] = arr[j][i];
                arr[j][i] = t;
            }
        }
        int f = 0, l = c - 1;
        while (f < l) {
            for (int i = 0; i < r; i++) {
                int t = arr[i][f];
                arr[i][f] = arr[i][l];
                arr[i][l] = t;
            }
            f++;
            l--;
        }
    }
}
