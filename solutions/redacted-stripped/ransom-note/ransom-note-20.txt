class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length())
            return false;
        int[] count = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            if (i < param1.length())
                count[(int) param1.charAt(i) - 97] += 1;
            count[(int) magazine.charAt(i) - 97] -= 1;
        }
        for (int i = 0; i < 26; i++) if (count[i] > 0)
            return false;
        return true;
    }
}
