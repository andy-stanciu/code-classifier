class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] m = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            m[magazine.charAt(i) - 97]++;
        }
        for (int i = 0; i < param1.length(); i++) {
            m[param1.charAt(i) - 97]--;
        }
        for (int i = 0; i < 26; i++) {
            if (m[i] < 0) {
                return false;
            }
        }
        return true;
    }
}
