class Solution {

    Map<Character, Integer> genHashMap(String str) {
        Map<Character, Integer> strMap = new HashMap<>();
        for (int i = 0; i < str.length(); i++) {
            Integer in = strMap.get(str.charAt(i));
            if (Objects.isNull(in)) {
                strMap.put(str.charAt(i), 1);
            } else {
                strMap.put(str.charAt(i), in + 1);
            }
        }
        return strMap;
    }

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> str1 = genHashMap(magazine);
        for (int i = 0; i < param1.length(); i++) {
            Character c = param1.charAt(i);
            Integer freqCount = str1.get(c);
            if (!Objects.isNull(freqCount) && freqCount != 0) {
                str1.put(c, freqCount - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
