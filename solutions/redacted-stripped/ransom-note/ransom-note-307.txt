class Solution {

    public boolean canConstruct(String param1, String magazine) {
        StringBuilder str = new StringBuilder(magazine);
        for (char c : param1.toCharArray()) {
            int index = str.indexOf(String.valueOf(c));
            if (index == -1) {
                return false;
            } else {
                str.deleteCharAt(index);
            }
        }
        return true;
    }
}
