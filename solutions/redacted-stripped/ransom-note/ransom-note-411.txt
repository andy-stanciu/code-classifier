class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> mMap = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            mMap.put(c, mMap.getOrDefault(c, 0) + 1);
        }
        Map<Character, Integer> rMap = new HashMap<>();
        for (char c : param1.toCharArray()) {
            rMap.put(c, rMap.getOrDefault(c, 0) + 1);
            if (mMap.containsKey(c) && rMap.get(c) > mMap.get(c)) {
                return false;
            }
        }
        for (char c : rMap.keySet()) {
            if (!mMap.containsKey(c) || rMap.get(c) > mMap.get(c)) {
                return false;
            }
        }
        return true;
    }
}
