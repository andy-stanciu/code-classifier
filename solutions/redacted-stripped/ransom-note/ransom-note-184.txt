class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int b = 0;
        for (int i = 0; i < magazine.length(); i++) {
            if (param1.charAt(b) == magazine.charAt(i)) {
                magazine = magazine.substring(0, i) + " " + magazine.substring(i + 1);
                b++;
                if (b == param1.length()) {
                    return true;
                }
                i = -1;
            }
        }
        return false;
    }
}
