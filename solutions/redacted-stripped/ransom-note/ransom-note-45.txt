class Solution {

    public boolean canConstruct(String r, String m) {
        if (r.length() > m.length())
            return false;
        HashMap<Character, Integer> rh = mapper(r);
        HashMap<Character, Integer> mh = mapper(m);
        for (Character i : rh.keySet()) {
            if (mh.containsKey(i)) {
                if (mh.get(i) < rh.get(i))
                    return false;
            } else
                return false;
        }
        return true;
    }

    HashMap<Character, Integer> mapper(String s) {
        HashMap<Character, Integer> h = new HashMap<>();
        for (char i : s.toCharArray()) {
            if (h.containsKey(i)) {
                h.put(i, h.get(i) + 1);
            } else
                h.put(i, 1);
        }
        return h;
    }
}
