class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = getMap(param1);
        Map<Character, Integer> magazineMap = getMap(magazine);
        for (Map.Entry<Character, Integer> m : var1.entrySet()) if (!magazineMap.containsKey(m.getKey()) || magazineMap.get(m.getKey()) < m.getValue())
            return false;
        return true;
    }

    private Map<Character, Integer> getMap(String s) {
        Map<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
        return map;
    }
}
