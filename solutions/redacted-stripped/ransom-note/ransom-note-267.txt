class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            return false;
        }
        StringBuilder s = new StringBuilder(magazine);
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            int in = s.indexOf(String.valueOf(c));
            if (in != -1) {
                s.deleteCharAt(in);
            } else {
                return false;
            }
        }
        return true;
    }
}
