class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> hp = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char c = magazine.charAt(i);
            if (!hp.containsKey(c)) {
                hp.put(c, 1);
            } else {
                hp.put(c, hp.get(c) + 1);
            }
        }
        for (int j = 0; j < param1.length(); j++) {
            if (hp.containsKey(param1.charAt(j))) {
                if (hp.get(param1.charAt(j)) > 0) {
                    hp.put(param1.charAt(j), hp.get(param1.charAt(j)) - 1);
                } else {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
