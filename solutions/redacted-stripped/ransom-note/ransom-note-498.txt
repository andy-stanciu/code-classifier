class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> toStore = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            if (toStore.containsKey(magazine.charAt(i))) {
                toStore.put(magazine.charAt(i), toStore.get(magazine.charAt(i)) + 1);
            } else {
                toStore.put(magazine.charAt(i), 1);
            }
        }
        for (int j = 0; j < param1.length(); j++) {
            if ((!toStore.containsKey(param1.charAt(j))) || (toStore.get(param1.charAt(j)) < 1)) {
                return false;
            }
            toStore.put(param1.charAt(j), toStore.get(param1.charAt(j)) - 1);
        }
        return true;
    }
}
