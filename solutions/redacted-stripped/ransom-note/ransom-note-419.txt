class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int var1 = param1.length();
        int magazine_len = magazine.length();
        HashMap<Character, Integer> var2 = new HashMap<>();
        HashMap<Character, Integer> map_magazine = new HashMap<>();
        boolean res = true;
        for (int i = 0; i < var1; i++) {
            if (var2.containsKey(param1.charAt(i))) {
                var2.put(param1.charAt(i), var2.get(param1.charAt(i)) + 1);
            } else {
                var2.put(param1.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine_len; i++) {
            if (map_magazine.containsKey(magazine.charAt(i))) {
                map_magazine.put(magazine.charAt(i), map_magazine.get(magazine.charAt(i)) + 1);
            } else {
                map_magazine.put(magazine.charAt(i), 1);
            }
        }
        for (Character ch : var2.keySet()) {
            if (!map_magazine.containsKey(ch)) {
                res = false;
                break;
            }
            if (var2.get(ch) > map_magazine.get(ch)) {
                res = false;
                break;
            }
        }
        return res;
    }
}
