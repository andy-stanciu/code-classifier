class Solution {

    public boolean canConstruct(String ran, String mag) {
        HashMap<Character, Integer> com = new HashMap<>();
        for (char c : mag.toCharArray()) {
            if (!com.containsKey(c)) {
                com.put(c, 1);
            } else {
                com.put(c, com.get(c) + 1);
            }
        }
        for (char c : ran.toCharArray()) {
            if (com.containsKey(c) && com.get(c) > 0) {
                com.put(c, com.get(c) - 1);
            } else {
                return false;
            }
        }
        System.out.print(com);
        return true;
    }
}
