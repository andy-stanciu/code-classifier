class Solution {

    public boolean canConstruct(String a, String b) {
        Map<Character, Integer> mp = new HashMap();
        Map<Character, Integer> mp1 = new HashMap();
        for (int i = 0; i < b.length(); i++) {
            mp.put(b.charAt(i), mp.getOrDefault(b.charAt(i), 0) + 1);
        }
        for (int i = 0; i < a.length(); i++) {
            mp1.put(a.charAt(i), mp1.getOrDefault(a.charAt(i), 0) + 1);
        }
        for (int i = 0; i < a.length(); i++) {
            if (mp.get(a.charAt(i)) == null || mp.get(a.charAt(i)) < mp1.get(a.charAt(i)))
                return false;
        }
        return true;
    }
}
