class Solution {

    public boolean canConstruct(String param1, String magazine) {
        boolean result = true;
        Map<Character, Integer> magazineLibrary = new HashMap<>();
        for (var ch : magazine.toCharArray()) {
            var size = magazineLibrary.getOrDefault(ch, 0);
            magazineLibrary.put(ch, ++size);
        }
        for (var ch : param1.toCharArray()) {
            if (!magazineLibrary.containsKey(ch)) {
                result = false;
                break;
            }
            var size = magazineLibrary.get(ch);
            if (size - 1 < 0) {
                result = false;
                break;
            }
            magazineLibrary.put(ch, --size);
        }
        return result;
    }
}
