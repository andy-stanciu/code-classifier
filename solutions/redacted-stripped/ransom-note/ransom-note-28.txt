class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        int[] alm = new int[26];
        for (int letter : magazine.toCharArray()) ++alm[letter - 97];
        for (int letter : param1.toCharArray()) if (--alm[letter - 97] < 0)
            return false;
        return true;
    }
}
