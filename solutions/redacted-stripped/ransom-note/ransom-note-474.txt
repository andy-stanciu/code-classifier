class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int pos;
        int flag = 0;
        for (int i = 0; i < param1.length(); i++) {
            pos = magazine.indexOf(param1.charAt(i));
            if (pos != -1) {
                magazine = magazine.replaceFirst("" + magazine.charAt(pos), " ");
            } else {
                flag = 1;
                break;
            }
        }
        if (flag == 0) {
            return true;
        } else {
            return false;
        }
    }
}
