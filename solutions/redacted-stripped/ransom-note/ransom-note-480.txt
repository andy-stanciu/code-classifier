class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<String, Integer> h = new HashMap();
        boolean flag = true;
        for (String c : magazine.split("")) {
            h.put(c, h.getOrDefault(c, 0) + 1);
        }
        for (String c : param1.split("")) {
            h.put(c, h.getOrDefault(c, 0) - 1);
        }
        for (Integer i : h.values()) {
            if (i < 0) {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
