class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        StringBuilder sb = new StringBuilder(magazine);
        int index = 0;
        while (index < param1.length()) {
            int lastIndex = index;
            for (int i = 0; i < sb.length(); i++) {
                if (param1.charAt(index) == sb.charAt(i)) {
                    if (index == param1.length() - 1)
                        return true;
                    sb.deleteCharAt(i);
                    index++;
                }
            }
            if (lastIndex == index)
                return false;
        }
        return true;
    }
}
