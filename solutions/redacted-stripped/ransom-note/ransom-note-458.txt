class Solution {

    public boolean canConstruct(String r, String m) {
        int i = 0;
        int j = 0;
        char[] arr1 = r.toCharArray();
        Arrays.sort(arr1);
        char[] arr2 = m.toCharArray();
        Arrays.sort(arr2);
        while (i < r.length() && j < m.length()) {
            if (arr1[i] == arr2[j]) {
                i++;
            }
            j++;
        }
        return i == r.length();
    }
}
