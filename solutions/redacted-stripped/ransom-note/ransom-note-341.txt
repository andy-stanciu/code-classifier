class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] var1 = param1.toCharArray();
        char[] magazineChars = magazine.toCharArray();
        int charCounter = 0;
        for (char rChar : var1) {
            for (int i = 0; i < magazineChars.length; i++) {
                if (magazineChars[i] == rChar) {
                    magazineChars[i] = Character.MIN_VALUE;
                    charCounter++;
                    break;
                }
            }
        }
        return charCounter == var1.length;
    }
}
