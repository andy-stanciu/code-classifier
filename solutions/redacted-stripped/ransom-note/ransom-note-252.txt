class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        else if (param1 != null && magazine != null) {
            for (int i = 0; i < param1.length(); i++) {
                char tempC = param1.charAt(i);
                long char1Count = param1.chars().filter(c -> c == tempC).count();
                long char2Count = magazine.chars().filter(c -> c == tempC).count();
                if (char1Count > char2Count)
                    return false;
            }
        }
        return true;
    }
}
