class Solution {

    public boolean canConstruct(String param1, String magazine) {
        List<String> rList = Arrays.asList(param1.split("")).stream().sorted().toList();
        List<String> mList = Arrays.asList(magazine.split("")).stream().sorted().toList();
        int count = 0;
        for (int i = 0, j = 0; i < rList.size() && j < mList.size(); ) {
            if (rList.get(i).equals(mList.get(j))) {
                count++;
                i++;
            }
            j++;
        }
        if (count == rList.size()) {
            return true;
        }
        return false;
    }
}
