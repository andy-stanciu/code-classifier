class Solution {

    public boolean canConstruct(String r, String m) {
        List<Character> Li = new ArrayList();
        for (int i = 0; i < m.length(); i++) {
            Li.add(m.charAt(i));
        }
        for (int i = 0; i < r.length(); i++) {
            char ch = r.charAt(i);
            if (Li.contains(ch)) {
                Li.remove((Character) ch);
            } else {
                return false;
            }
        }
        return true;
    }
}
