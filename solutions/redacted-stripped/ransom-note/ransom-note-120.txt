class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Long> magazineCharCount = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        return param1.chars().mapToObj(c -> (char) c).allMatch(c -> magazineCharCount.getOrDefault(c, 0L) > 0 && magazineCharCount.put(c, magazineCharCount.get(c) - 1) != null);
    }
}
