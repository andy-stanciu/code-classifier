class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Long> collect = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        for (char c : param1.toCharArray()) {
            if (!collect.containsKey(c) || collect.get(c) <= 0) {
                return false;
            }
            collect.put(c, (collect.get(c) - 1));
        }
        return true;
    }
}
