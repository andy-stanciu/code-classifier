class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> words = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            if (words.containsKey(magazine.charAt(i))) {
                words.put(magazine.charAt(i), words.get(magazine.charAt(i)) + 1);
            } else {
                words.put(magazine.charAt(i), 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (words.containsKey(param1.charAt(i)) && words.get(param1.charAt(i)) > 0) {
                words.put(param1.charAt(i), words.get(param1.charAt(i)) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
