class Solution {

    public boolean canConstruct(String rN, String m) {
        char[] cr = rN.toCharArray();
        char[] cm = m.toCharArray();
        Arrays.sort(cr);
        Arrays.sort(cm);
        int i = 0, j = 0;
        while (i < cr.length && j < cm.length) {
            if (cm[j] == cr[i]) {
                i++;
                j++;
            } else
                j++;
        }
        if (i == cr.length)
            return true;
        else
            return false;
    }
}
