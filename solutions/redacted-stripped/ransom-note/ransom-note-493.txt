class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (Objects.equals(param1, magazine)) {
            return true;
        }
        final String[] arr = param1.split("");
        for (String s : arr) {
            final int indexOf = magazine.indexOf(s);
            if (indexOf == -1) {
                return false;
            }
            magazine = magazine.substring(0, indexOf) + magazine.substring(indexOf + 1);
        }
        return true;
    }
}
