class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> hm = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char c = magazine.charAt(i);
            if (hm.size() == 0) {
                hm.put(c, 1);
            } else if (hm.containsKey(c)) {
                hm.put(c, hm.get(c) + 1);
            } else {
                hm.put(c, 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            if (!hm.containsKey(c)) {
                return false;
            } else {
                if (hm.get(c) > 1) {
                    hm.put(c, hm.get(c) - 1);
                } else {
                    hm.remove(c);
                }
            }
        }
        return true;
    }
}
