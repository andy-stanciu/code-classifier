class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Long> charCountMap = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        for (char ch : param1.toCharArray()) {
            if (!charCountMap.containsKey(ch) || charCountMap.get(ch) == 0) {
                return false;
            }
            charCountMap.put(ch, charCountMap.get(ch) - 1);
        }
        return true;
    }
}
