class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Long> var1;
        Map<Character, Long> mag;
        mag = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        var1 = param1.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        return var1.entrySet().stream().allMatch(entry -> entry.getValue() <= mag.getOrDefault(entry.getKey(), 0L));
    }
}
