class Solution {

    public boolean canConstruct(String param1, String magazine) {
        StringBuilder str = new StringBuilder(magazine);
        int n = param1.length();
        if (n > magazine.length())
            return false;
        for (int i = 0; i < n; i++) {
            int index = str.toString().indexOf(param1.charAt(i));
            if (index == -1)
                return false;
            else
                str = str.deleteCharAt(index);
        }
        return true;
    }
}
