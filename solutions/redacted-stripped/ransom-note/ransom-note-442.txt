class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length())
            return false;
        Map<Character, Integer> hash_rN = new HashMap<>();
        Map<Character, Integer> hash_mG = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            Character c = param1.charAt(i);
            hash_rN.putIfAbsent(c, 0);
            hash_rN.put(c, hash_rN.get(c) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            Character c = magazine.charAt(i);
            hash_mG.putIfAbsent(c, 0);
            hash_mG.put(c, hash_mG.get(c) + 1);
        }
        for (Map.Entry<Character, Integer> entry : hash_rN.entrySet()) {
            if (hash_mG.containsKey(entry.getKey())) {
                if (hash_mG.get(entry.getKey()) < entry.getValue())
                    return false;
            } else
                return false;
        }
        return true;
    }
}
