class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> mp = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            mp.put(param1.charAt(i), mp.getOrDefault(param1.charAt(i), 0) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (mp.containsKey(magazine.charAt(i))) {
                int g = mp.get(magazine.charAt(i));
                if (g == 1)
                    mp.remove(magazine.charAt(i));
                else {
                    mp.put(magazine.charAt(i), g - 1);
                }
            }
        }
        return mp.isEmpty();
    }
}
