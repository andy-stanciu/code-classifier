class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int i = 0;
        int j = 0;
        char[] arr1 = param1.toCharArray();
        Arrays.sort(arr1);
        char[] arr2 = magazine.toCharArray();
        Arrays.sort(arr2);
        while (j < arr2.length) {
            if (arr1[i] == arr2[j]) {
                if (i != arr1.length - 1) {
                    i++;
                    j++;
                } else {
                    return true;
                }
            } else {
                j++;
            }
        }
        return false;
    }
}
