class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        for (int i = 0; i < param1.length(); i++) {
            if (magazine.contains(param1.charAt(i) + "")) {
                int j = magazine.indexOf(param1.charAt(i));
                StringBuilder sb = new StringBuilder(magazine);
                sb.deleteCharAt(j);
                magazine = sb.toString();
            } else {
                return false;
            }
        }
        return true;
    }
}
