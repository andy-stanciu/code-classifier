class Solution {

    public boolean canConstruct(String param1, String magazine) {
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            if (magazine.indexOf(ch) == -1) {
                return false;
            }
            magazine = magazine.replaceFirst(String.valueOf(ch), "");
        }
        return true;
    }
}
