class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> hm1 = getCharacterMap(param1);
        Map<Character, Integer> hm2 = getCharacterMap(magazine);
        boolean res = true;
        for (Map.Entry<Character, Integer> entry1 : hm1.entrySet()) {
            Character ch = entry1.getKey();
            if (hm2.containsKey(ch) && hm2.get(ch) >= entry1.getValue()) {
                continue;
            }
            res = false;
            break;
        }
        return res;
    }

    public static Map<Character, Integer> getCharacterMap(String s) {
        int len = s.length();
        Map<Character, Integer> hm1 = new HashMap<>();
        for (int i = 0; i < len; i++) {
            char ch = s.charAt(i);
            if (hm1.containsKey(ch)) {
                hm1.put(ch, hm1.get(ch) + 1);
            } else {
                hm1.put(ch, 1);
            }
        }
        return hm1;
    }
}
