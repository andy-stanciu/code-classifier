class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> map1 = new HashMap<>();
        Map<Character, Integer> map2 = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (map1.containsKey(param1.charAt(i))) {
                map1.put(param1.charAt(i), map1.getOrDefault(param1.charAt(i), 0) + 1);
            } else {
                map1.put(param1.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (map2.containsKey(magazine.charAt(i))) {
                map2.put(magazine.charAt(i), map2.getOrDefault(magazine.charAt(i), 0) + 1);
            } else {
                map2.put(magazine.charAt(i), 1);
            }
        }
        for (Character key : map1.keySet()) {
            Integer value1 = map1.get(key);
            Integer value2 = map2.getOrDefault(key, 0);
            if (value1 > value2) {
                return false;
            }
        }
        return true;
    }
}
