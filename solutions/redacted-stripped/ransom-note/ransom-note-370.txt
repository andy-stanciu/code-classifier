class Solution {

    private final int[] count = new int[26];

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            return false;
        }
        for (char c : param1.toCharArray()) {
            count[c % 26]--;
        }
        for (char c : magazine.toCharArray()) {
            count[c % 26]++;
        }
        for (int value : count) {
            if (value < 0) {
                return false;
            }
        }
        return true;
    }
}
