class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] map1 = new int[26];
        int[] map2 = new int[26];
        for (int i = 0; i < param1.length(); i++) map1[param1.charAt(i) - 97]++;
        for (int i = 0; i < magazine.length(); i++) map2[magazine.charAt(i) - 97]++;
        for (int i = 0; i < 26; i++) if ((map2[i] != 0 && map2[i] < map1[i]) || map1[i] > map2[i])
            return false;
        return true;
    }
}
