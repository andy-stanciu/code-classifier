class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> m = new HashMap<>();
        char[] t = param1.toCharArray();
        for (char e : t) {
            if (m.containsKey(e))
                m.put(e, m.get(e) + 1);
            else
                m.put(e, 1);
        }
        char[] w = magazine.toCharArray();
        for (char u : w) {
            if (m.containsKey(u)) {
                m.put(u, m.get(u) - 1);
                if (m.get(u) == 0)
                    m.remove(u);
            }
        }
        return m.isEmpty() ? true : false;
    }
}
