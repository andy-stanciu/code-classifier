class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] ra = param1.toCharArray();
        char[] ma = magazine.toCharArray();
        HashMap<Character, Integer> checklist = new HashMap<>();
        for (char c : ma) {
            if (checklist.containsKey(c)) {
                int val = checklist.get(c);
                checklist.put(c, ++val);
            } else {
                checklist.put(c, 1);
            }
        }
        for (int i = 0; i < ra.length; i++) {
            char candidate = ra[i];
            if (checklist.containsKey(candidate)) {
                int val = checklist.get(candidate);
                if (val == 0)
                    return false;
                checklist.put(candidate, --val);
            } else {
                return false;
            }
        }
        return true;
    }
}
