class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.isEmpty() || magazine.isEmpty())
            return false;
        else if (param1.length() > magazine.length())
            return false;
        HashMap<Character, Integer> mgMAP = new HashMap<>();
        char[] mgARR = magazine.toCharArray();
        for (char c : mgARR) {
            if (mgMAP.containsKey(c))
                mgMAP.put(c, mgMAP.get(c) + 1);
            else
                mgMAP.put(c, 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            if (mgMAP.containsKey(param1.charAt(i))) {
                mgMAP.put(param1.charAt(i), mgMAP.get(param1.charAt(i)) - 1);
                if (mgMAP.get(param1.charAt(i)) < 0)
                    return false;
            } else
                return false;
        }
        return true;
    }
}
