class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] one = param1.toCharArray();
        Arrays.sort(one);
        param1 = new String(one);
        char[] two = magazine.toCharArray();
        Arrays.sort(two);
        magazine = new String(two);
        int i = 0, j = 0, count = 0;
        while (j < magazine.length() && i < param1.length()) {
            if (param1.charAt(i) == magazine.charAt(j)) {
                count++;
                i++;
                j++;
            } else {
                j++;
            }
        }
        return count == param1.length();
    }
}
