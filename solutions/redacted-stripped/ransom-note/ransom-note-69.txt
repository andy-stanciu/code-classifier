class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<String, Integer> var1 = new HashMap<>();
        String[] var2 = param1.split("");
        for (String a : var2) {
            var1.merge(a, 1, Integer::sum);
        }
        HashMap<String, Integer> magazineMap = new HashMap<>();
        String[] magazinesplit = magazine.split("");
        for (String a : magazinesplit) {
            magazineMap.merge(a, 1, Integer::sum);
        }
        boolean mission = false;
        for (String a : var2) {
            if (magazineMap.get(a) != null && magazineMap.get(a) >= var1.get(a)) {
                mission = true;
            } else {
                mission = false;
                break;
            }
        }
        return mission;
    }
}
