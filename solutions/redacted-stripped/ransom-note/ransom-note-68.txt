class Solution {

    public boolean canConstruct(String param1, String magazine) {
        List<Character> rlist = new ArrayList<>();
        List<Character> mlist = new ArrayList<>();
        for (int i = 0; i < param1.length(); i++) {
            rlist.add(param1.charAt(i));
        }
        for (int i = 0; i < magazine.length(); i++) {
            mlist.add(magazine.charAt(i));
        }
        for (Character letter : rlist) {
            if (mlist.contains(letter)) {
                mlist.remove(letter);
            } else {
                return false;
            }
        }
        return true;
    }
}
