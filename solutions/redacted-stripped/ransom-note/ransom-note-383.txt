class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> mp = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            mp.put(magazine.charAt(i), mp.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            if (mp.containsKey(param1.charAt(i)) && mp.get(param1.charAt(i)) > 0) {
                mp.put(param1.charAt(i), mp.get(param1.charAt(i)) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
