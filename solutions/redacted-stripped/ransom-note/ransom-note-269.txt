class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] freq = new int[26];
        int l1 = magazine.length();
        int l2 = param1.length();
        if (l2 > l1)
            return false;
        for (int i = 0; i < l1; i++) {
            freq[magazine.charAt(i) - 97]++;
        }
        for (int i = 0; i < l2; i++) {
            if (freq[param1.charAt(i) - 97] < 1)
                return false;
            freq[param1.charAt(i) - 97]--;
        }
        return true;
    }
}
