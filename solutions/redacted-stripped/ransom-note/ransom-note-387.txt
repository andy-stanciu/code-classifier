class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            map.put(param1.charAt(i), map.getOrDefault(param1.charAt(i), 0) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (map.containsKey(magazine.charAt(i))) {
                if (map.get(magazine.charAt(i)) == 1) {
                    map.remove(magazine.charAt(i));
                } else {
                    int cur = map.get(magazine.charAt(i));
                    map.put(magazine.charAt(i), cur - 1);
                }
            } else {
                continue;
            }
        }
        return map.isEmpty();
    }
}
