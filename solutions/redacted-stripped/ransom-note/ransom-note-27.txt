class Solution {

    public boolean canConstruct(String param1, String magazine) {
        for (int i = 0; i < param1.length(); i++) {
            int index = magazine.indexOf(param1.charAt(i));
            if (index == -1)
                return false;
            else
                magazine = magazine.substring(0, index) + magazine.substring(index + 1);
        }
        return true;
    }
}
