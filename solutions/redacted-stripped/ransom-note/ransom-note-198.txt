class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            return false;
        }
        Map<Character, Integer> var1 = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            if (var1.containsKey(c)) {
                var1.put(c, var1.get(c) + 1);
            } else {
                var1.put(c, 1);
            }
        }
        for (int j = 0; j < magazine.length(); j++) {
            char c2 = magazine.charAt(j);
            if (var1.containsKey(c2)) {
                if (var1.get(c2) == 1) {
                    var1.remove(c2);
                } else {
                    var1.put(c2, var1.get(c2) - 1);
                }
            }
        }
        return var1.isEmpty();
    }
}
