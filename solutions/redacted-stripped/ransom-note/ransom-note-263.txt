class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = new HashMap<>();
        Map<Character, Integer> magazineMap = new HashMap<>();
        for (char ch : param1.toCharArray()) {
            var1.put(ch, var1.getOrDefault(ch, 0) + 1);
        }
        for (char ch : magazine.toCharArray()) {
            magazineMap.put(ch, magazineMap.getOrDefault(ch, 0) + 1);
        }
        for (char ch : var1.keySet()) {
            if (!magazineMap.containsKey(ch)) {
                return false;
            }
            if (var1.get(ch) > magazineMap.get(ch)) {
                return false;
            }
        }
        return true;
    }
}
