class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] arr1 = new int[26];
        int[] arr2 = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            arr1[magazine.charAt(i) - 97]++;
        }
        for (int i = 0; i < param1.length(); i++) {
            arr2[param1.charAt(i) - 97]++;
        }
        for (int i = 0; i < arr1.length; i++) {
            if (arr1[i] < arr2[i]) {
                return false;
            }
        }
        return true;
    }
}
