class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> charsIndex = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            int lastCharIndex = charsIndex.containsKey(c) ? charsIndex.get(c) + 1 : 0;
            int charIndex = magazine.indexOf(c, lastCharIndex);
            if (charIndex != -1) {
                charsIndex.put(c, charIndex);
            } else {
                return false;
            }
        }
        return true;
    }
}
