class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = new HashMap();
        Map<Character, Integer> magazineFreq = new HashMap();
        for (int i = 0; i < param1.length(); ++i) {
            var1.merge(param1.charAt(i), 1, Integer::sum);
        }
        for (int i = 0; i < magazine.length(); ++i) {
            magazineFreq.merge(magazine.charAt(i), 1, Integer::sum);
        }
        for (int x = 0; x < param1.length(); ++x) {
            if (!magazineFreq.containsKey(param1.charAt(x)) || (magazineFreq.containsKey(param1.charAt(x)) && magazineFreq.get(param1.charAt(x)) < var1.get(param1.charAt(x)))) {
                return false;
            }
        }
        return true;
    }
}
