class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] m1 = new int[256];
        for (char c : magazine.toCharArray()) m1[c]++;
        for (char c : param1.toCharArray()) {
            if (m1[c] <= 0)
                return false;
            m1[c]--;
        }
        return true;
    }
}
