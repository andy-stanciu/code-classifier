class Solution {

    public boolean canConstruct(String param1, String magazine) {
        boolean[] b = new boolean[magazine.length()];
        int cnt = 0;
        for (int i = 0; i < param1.length(); i++) {
            for (int j = 0; j < magazine.length(); j++) {
                if (param1.charAt(i) == magazine.charAt(j) && b[j] == false) {
                    cnt++;
                    b[j] = true;
                    break;
                }
            }
        }
        if (cnt == param1.length()) {
            return true;
        }
        return false;
    }
}
