class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = new HashMap<>();
        Map<Character, Integer> mag = new HashMap<>();
        char[] var2 = param1.toCharArray();
        char[] magArr = magazine.toCharArray();
        for (char i : var2) {
            var1.put(i, var1.getOrDefault(i, 0) + 1);
        }
        for (char i : magArr) {
            mag.put(i, mag.getOrDefault(i, 0) + 1);
        }
        for (Map.Entry<Character, Integer> entry : var1.entrySet()) {
            Character key = entry.getKey();
            Integer value = entry.getValue();
            if (!mag.containsKey(key) || !(mag.get(key) >= value)) {
                return false;
            }
        }
        return true;
    }
}
