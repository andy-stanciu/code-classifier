class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> charCountMagazine = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            charCountMagazine.put(c, charCountMagazine.getOrDefault(c, 0) + 1);
        }
        for (char c : param1.toCharArray()) {
            if (charCountMagazine.getOrDefault(c, 0) == 0) {
                return false;
            }
            charCountMagazine.put(c, charCountMagazine.get(c) - 1);
        }
        return true;
    }
}
