class Solution {

    public int count(String s, char ch) {
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == ch) {
                count++;
            }
        }
        return count;
    }

    public boolean canConstruct(String param1, String magazine) {
        Set<Character> set1 = new HashSet<Character>();
        for (char ch : magazine.toCharArray()) {
            set1.add(ch);
        }
        char[] c = param1.toCharArray();
        for (int i = 0; i < param1.length(); i++) {
            if (!set1.contains(c[i]) || count(magazine, param1.charAt(i)) < count(param1, param1.charAt(i))) {
                return false;
            }
        }
        return true;
    }
}
