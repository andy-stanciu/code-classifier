class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = getFrequencies(param1);
        Map<Character, Integer> mag = getFrequencies(magazine);
        for (Character key : var1.keySet()) {
            if (!mag.containsKey(key)) {
                return false;
            } else {
                if (!(mag.get(key) >= var1.get(key))) {
                    return false;
                }
            }
        }
        return true;
    }

    public Map<Character, Integer> getFrequencies(String s) {
        Map<Character, Integer> hm = new HashMap<>();
        char[] ch = s.toCharArray();
        int len = ch.length;
        while (len - 1 >= 0) {
            hm.put(ch[len - 1], hm.getOrDefault(ch[len - 1], 0) + 1);
            len--;
        }
        return hm;
    }
}
