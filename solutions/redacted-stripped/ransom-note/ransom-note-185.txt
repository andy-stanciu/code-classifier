class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> map = new HashMap<>();
        for (int x = 0; x < magazine.length(); x++) {
            map.putIfAbsent(magazine.charAt(x), 0);
            map.computeIfPresent(magazine.charAt(x), (key, value) -> {
                return value = value + 1;
            });
        }
        for (int x = 0; x < param1.length(); x++) {
            if (map.get(param1.charAt(x)) != null && map.get(param1.charAt(x)) > 0) {
                map.computeIfPresent(param1.charAt(x), (key, value) -> {
                    return value = value - 1;
                });
            } else {
                return false;
            }
        }
        return true;
    }
}
