class Solution {

    public static boolean canConstruct(String param1, String magazine) {
        char[] ran = param1.toCharArray();
        for (Character ch : ran) {
            if (!magazine.contains(ch.toString())) {
                return false;
            } else {
                magazine = magazine.replaceFirst(ch.toString(), "");
            }
        }
        return true;
    }
}
