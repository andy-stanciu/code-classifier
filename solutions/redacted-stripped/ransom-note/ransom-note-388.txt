class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Long> var1 = new HashMap();
        for (int i = 0; i < param1.length(); i++) {
            if (var1.get(param1.charAt(i)) != null) {
                var1.put(param1.charAt(i), var1.get(param1.charAt(i)) + 1);
            } else {
                var1.put(param1.charAt(i), 1L);
            }
        }
        HashMap<Character, Long> magazineMap = new HashMap();
        for (int i = 0; i < magazine.length(); i++) {
            if (magazineMap.get(magazine.charAt(i)) != null) {
                magazineMap.put(magazine.charAt(i), magazineMap.get(magazine.charAt(i)) + 1);
            } else {
                magazineMap.put(magazine.charAt(i), 1L);
            }
        }
        boolean result = true;
        for (Map.Entry<Character, Long> characterLongEntry : var1.entrySet()) {
            Character key = characterLongEntry.getKey();
            Long value = characterLongEntry.getValue();
            if (magazineMap.get(key) == null || magazineMap.get(key) < value) {
                result = false;
            }
        }
        return result;
    }
}
