class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> n = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char ch = magazine.charAt(i);
            if (!(n.containsKey(ch))) {
                n.put(ch, 1);
            } else {
                n.put(ch, n.get(ch) + 1);
            }
        }
        for (int j = 0; j < param1.length(); j++) {
            char ad = param1.charAt(j);
            if (n.containsKey(ad) && n.get(ad) > 0) {
                n.put(ad, n.get(ad) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
