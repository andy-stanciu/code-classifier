class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int n1 = param1.length();
        int n2 = magazine.length();
        int count = 0;
        ArrayList<Character> ans = new ArrayList<>();
        for (int i = 0; i < n1; i++) {
            ans.add(param1.charAt(i));
        }
        for (int i = 0; i < n2; i++) {
            for (int j = 0; j < ans.size(); j++) {
                if (ans.get(j) == magazine.charAt(i)) {
                    count++;
                    ans.remove(j);
                    break;
                }
            }
        }
        if (count == n1) {
            return true;
        }
        return false;
    }
}
