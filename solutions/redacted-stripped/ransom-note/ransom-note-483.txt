class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> ran = new HashMap<Character, Integer>();
        HashMap<Character, Integer> mag = new HashMap<Character, Integer>();
        for (int i = 0; i < param1.length(); i++) {
            ran.put(param1.charAt(i), ran.getOrDefault(param1.charAt(i), 0) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            mag.put(magazine.charAt(i), mag.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            if (mag.get(param1.charAt(i)) == null)
                return false;
            int a = (Integer) (ran.get(param1.charAt(i)));
            int b = (Integer) (mag.get(param1.charAt(i)));
            if (a > b)
                return false;
        }
        return true;
    }
}
