class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] arr = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            int val = magazine.charAt(i) - 97;
            arr[val] = ++arr[val];
        }
        for (int i = 0; i < param1.length(); i++) {
            int val = param1.charAt(i) - 97;
            if (arr[val] == 0) {
                return false;
            } else {
                arr[val] = --arr[val];
            }
        }
        return true;
    }
}
