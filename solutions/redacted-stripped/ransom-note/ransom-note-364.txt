class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> magMap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            if (magMap.containsKey(magazine.charAt(i))) {
                magMap.put(magazine.charAt(i), magMap.get(magazine.charAt(i)) + 1);
            } else {
                magMap.put(magazine.charAt(i), 1);
            }
        }
        for (int j = 0; j < param1.length(); j++) {
            if (magMap.containsKey(param1.charAt(j))) {
                magMap.put(param1.charAt(j), magMap.get(param1.charAt(j)) - 1);
                if (magMap.get(param1.charAt(j)) < 0) {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
