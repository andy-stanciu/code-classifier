class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> count = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            count.put(magazine.charAt(i), count.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            count.put(param1.charAt(i), count.getOrDefault(param1.charAt(i), 0) - 1);
            if (count.get(param1.charAt(i)) < 0) {
                return false;
            }
        }
        return true;
    }
}
