class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> hashMap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            if (hashMap.get(magazine.charAt(i)) == null)
                hashMap.put(magazine.charAt(i), 1);
            else {
                hashMap.put(magazine.charAt(i), hashMap.get(magazine.charAt(i)) + 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (hashMap.get(param1.charAt(i)) == null || hashMap.get(param1.charAt(i)) <= 0)
                return false;
            else {
                hashMap.put(param1.charAt(i), hashMap.get(param1.charAt(i)) - 1);
            }
        }
        return true;
    }
}
