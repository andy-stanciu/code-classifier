class Solution {

    public boolean canConstruct(String param1, String magazine) {
        StringBuilder output = new StringBuilder();
        for (int i = 0; i < param1.length(); i++) {
            if (magazine.contains(param1.charAt(i) + "")) {
                String magazineChar = magazine.charAt(magazine.indexOf(param1.charAt(i))) + "";
                output.append(magazineChar);
                magazine = magazine.replaceFirst(magazineChar, "");
            }
        }
        return output.toString().equals(param1);
    }
}
