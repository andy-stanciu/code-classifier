class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] arrRan = param1.toCharArray();
        char[] arrmag = magazine.toCharArray();
        Map<Integer, String> mapObj = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            int index = magazine.indexOf(arrRan[i]);
            if (index == -1) {
                break;
            } else {
                arrmag[index] = 0;
                magazine = String.valueOf(arrmag);
                mapObj.put(index, String.valueOf(arrRan[i]));
            }
        }
        return mapObj.size() == param1.length() ? true : false;
    }
}
