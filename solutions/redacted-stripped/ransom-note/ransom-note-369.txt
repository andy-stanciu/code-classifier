class Solution {

    public boolean canConstruct(String param1, String magazine) {
        var indexes = new HashMap<Character, Integer>();
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            int index = magazine.indexOf(ch, indexes.getOrDefault(ch, 0));
            if (index < 0) {
                return false;
            }
            indexes.put(ch, index + 1);
        }
        return true;
    }
}
