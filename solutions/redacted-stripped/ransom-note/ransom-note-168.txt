class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> hm = new HashMap<>();
        int n = param1.length();
        int m = magazine.length();
        if (m < n)
            return false;
        for (int i = 0; i < m; i++) {
            hm.put(magazine.charAt(i), hm.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < n; i++) {
            if (!hm.containsKey(param1.charAt(i))) {
                return false;
            }
            if (hm.get(param1.charAt(i)) == 1) {
                hm.remove(param1.charAt(i));
            } else
                hm.put(param1.charAt(i), hm.get(param1.charAt(i)) - 1);
        }
        return true;
    }
}
