class Solution {

    public boolean canConstruct(String r, String m) {
        Hashtable<Character, Integer> ht1 = new Hashtable<>();
        Hashtable<Character, Integer> ht2 = new Hashtable<>();
        for (int x = 0; x < r.length(); x++) {
            if (ht1.containsKey(r.charAt(x)))
                ht1.put(r.charAt(x), ht1.get(r.charAt(x)) + 1);
            else
                ht1.put(r.charAt(x), 1);
        }
        for (int x = 0; x < m.length(); x++) {
            if (ht2.containsKey(m.charAt(x)))
                ht2.put(m.charAt(x), ht2.get(m.charAt(x)) + 1);
            else
                ht2.put(m.charAt(x), 1);
        }
        for (char x : ht1.keySet()) {
            if (!(ht2.containsKey(x) && ht2.get(x) >= ht1.get(x)))
                return false;
        }
        return true;
    }
}
