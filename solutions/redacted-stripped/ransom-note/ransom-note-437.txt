class Solution {

    public boolean canConstruct(String param1, String magazine) {
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            int index = magazine.indexOf(c);
            if (index == -1)
                return false;
            magazine = magazine.substring(0, index) + magazine.substring(index + 1);
        }
        return true;
    }
}
