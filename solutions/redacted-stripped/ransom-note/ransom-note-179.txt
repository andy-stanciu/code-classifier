class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> ranMap = new HashMap<>();
        HashMap<Character, Integer> magMap = new HashMap<>();
        int i = 0;
        for (i = 0; i < magazine.length(); i++) {
            if (!magMap.containsKey(magazine.charAt(i))) {
                magMap.put(magazine.charAt(i), 1);
            } else {
                magMap.put(magazine.charAt(i), magMap.get(magazine.charAt(i)) + 1);
            }
        }
        for (i = 0; i < param1.length(); i++) {
            if (!ranMap.containsKey(param1.charAt(i))) {
                ranMap.put(param1.charAt(i), 1);
            } else {
                ranMap.put(param1.charAt(i), ranMap.get(param1.charAt(i)) + 1);
            }
        }
        for (i = 0; i < param1.length(); i++) {
            if (!magMap.containsKey(param1.charAt(i))) {
                return false;
            }
            if (!(ranMap.get(param1.charAt(i)) <= magMap.get(param1.charAt(i)))) {
                return false;
            }
        }
        return true;
    }
}
