class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> map = new HashMap(magazine.length());
        for (int i = 0; i < magazine.length(); i++) {
            char current = magazine.charAt(i);
            if (map.containsKey(current)) {
                int occurrences = map.get(current);
                map.replace(current, ++occurrences);
            } else {
                map.put(current, 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            char current = param1.charAt(i);
            if (map.containsKey(current) && map.get(current) > 0) {
                int occurrences = map.get(current);
                map.replace(current, --occurrences);
            } else {
                return false;
            }
        }
        return true;
    }
}
