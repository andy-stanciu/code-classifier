class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> hm = new HashMap<>();
        char[] strArray = magazine.toCharArray();
        for (char c : strArray) {
            if (hm.containsKey(c)) {
                hm.put(c, hm.get(c) + 1);
            } else {
                hm.put(c, 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            boolean exists = hm.containsKey(ch);
            if (!exists) {
                return false;
            }
            int freq = stringFrequency(ch, param1);
            int val = hm.get(ch);
            if (exists && freq > val) {
                return false;
            }
        }
        return true;
    }

    int stringFrequency(char c, String str) {
        int count = 0;
        for (int i = 0; i < str.length(); i++) {
            if (c == str.charAt(i)) {
                count++;
            }
        }
        return count;
    }
}
