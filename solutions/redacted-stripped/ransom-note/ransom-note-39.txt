class Solution {

    public boolean canConstruct(String param1, String magazine) {
        for (int i = 0; i < param1.length(); ++i) {
            String t = param1.substring(i, i + 1);
            int idx = magazine.indexOf(t);
            if (idx == -1)
                return false;
            magazine = magazine.substring(0, idx) + magazine.substring(idx + 1);
        }
        return true;
    }
}
