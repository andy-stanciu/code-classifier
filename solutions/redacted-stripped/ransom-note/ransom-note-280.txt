class Solution {

    public boolean canConstruct(String param1, String magazine) {
        boolean canConstruct = false;
        if (param1.length() <= magazine.length()) {
            HashMap<Character, Integer> dict = new HashMap<>();
            for (int i = 0; i < param1.length(); i++) {
                Character ch = param1.charAt(i);
                Integer value = dict.get(ch);
                if (value == null) {
                    value = Integer.valueOf(1);
                } else {
                    value = Integer.valueOf(value.intValue() + 1);
                }
                dict.put(ch, value);
            }
            for (int i = 0; i < magazine.length() && !dict.isEmpty(); i++) {
                Character ch = magazine.charAt(i);
                Integer value = dict.get(ch);
                if (value != null) {
                    value = Integer.valueOf(value.intValue() - 1);
                    if (value.intValue() == 0) {
                        dict.remove(ch);
                    } else {
                        dict.put(ch, value);
                    }
                }
            }
            canConstruct = dict.isEmpty();
        }
        return canConstruct;
    }
}
