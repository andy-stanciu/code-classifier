class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = new HashMap<>();
        Map<Character, Integer> magazineMap = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (var1.containsKey(param1.charAt(i))) {
                Integer val = var1.get(param1.charAt(i));
                var1.put(param1.charAt(i), ++val);
            } else {
                var1.put(param1.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (magazineMap.containsKey(magazine.charAt(i))) {
                Integer val = magazineMap.get(magazine.charAt(i));
                magazineMap.put(magazine.charAt(i), ++val);
            } else {
                magazineMap.put(magazine.charAt(i), 1);
            }
        }
        for (Character key : var1.keySet()) {
            Integer var2 = var1.get(key);
            Integer magazineVal = magazineMap.get(key);
            var2 = (var2 == null) ? 0 : var2;
            magazineVal = (magazineVal == null) ? 0 : magazineVal;
            if (var2 > magazineVal) {
                return false;
            }
        }
        return true;
    }
}
