class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> remainingLetters = new HashMap<>();
        for (Character letter : param1.toCharArray()) {
            if (remainingLetters.containsKey(letter)) {
                remainingLetters.put(letter, remainingLetters.get(letter) + 1);
            } else {
                remainingLetters.put(letter, 1);
            }
        }
        for (Character letter : magazine.toCharArray()) {
            if (remainingLetters.containsKey(letter)) {
                int remainingCount = remainingLetters.get(letter);
                if (remainingCount == 1) {
                    remainingLetters.remove(letter);
                    if (remainingLetters.isEmpty()) {
                        return true;
                    }
                } else {
                    remainingLetters.put(letter, --remainingCount);
                }
            }
        }
        return remainingLetters.isEmpty();
    }
}
