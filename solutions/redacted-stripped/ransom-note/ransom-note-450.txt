class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        int[] arr = new int[26];
        for (char c : magazine.toCharArray()) arr[c - 97]++;
        for (char c : param1.toCharArray()) {
            if (arr[c - 97] == 0)
                return false;
            arr[c - 97]--;
        }
        return true;
    }
}
