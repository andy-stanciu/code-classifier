class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] map = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            char ch = magazine.charAt(i);
            map[ch - 97]++;
        }
        for (int i = 0; i < param1.length(); i++) {
            char curr = param1.charAt(i);
            if (map[curr - 97] > 0)
                map[curr - 97]--;
            else
                return false;
        }
        return true;
    }
}
