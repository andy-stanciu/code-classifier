class Solution {

    public boolean canConstruct(String param1, String magazine) {
        List<Character> var1 = new ArrayList<>();
        List<Character> magazine_list = new ArrayList<>();
        boolean flag = false;
        for (char char_ran : param1.toCharArray()) var1.add(char_ran);
        for (char char_mag : magazine.toCharArray()) magazine_list.add(char_mag);
        for (int i = 0; i < magazine_list.size(); i++) {
            char check = magazine_list.get(i);
            int index_ran = var1.indexOf(check);
            if (var1.contains(check)) {
                var1.remove(index_ran);
            }
        }
        return var1.isEmpty();
    }
}
