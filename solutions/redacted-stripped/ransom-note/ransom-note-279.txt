class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] var1 = param1.toCharArray();
        char[] magazineChars = magazine.toCharArray();
        Arrays.sort(var1);
        Arrays.sort(magazineChars);
        int i = 0, j = 0;
        while (i < var1.length && j < magazineChars.length) {
            if (var1[i] == magazineChars[j]) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        return i == var1.length;
    }
}
