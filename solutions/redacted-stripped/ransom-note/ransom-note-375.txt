class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int magazineLength = magazine.length();
        int var1 = param1.length();
        if (var1 > magazineLength)
            return false;
        Map<Character, Integer> charCount = new HashMap<>();
        int gap = 0;
        for (int i = 0; i < magazineLength; i++) {
            char magazineChar = magazine.charAt(i);
            int magazineCharCount = charCount.getOrDefault(magazineChar, 0);
            if (magazineCharCount < 0)
                gap--;
            charCount.put(magazineChar, magazineCharCount + 1);
            if (i >= var1)
                continue;
            char var2 = param1.charAt(i);
            int var3 = charCount.getOrDefault(var2, 0);
            if (var3 <= 0)
                gap++;
            charCount.put(var2, var3 - 1);
        }
        return gap <= 0;
    }
}
