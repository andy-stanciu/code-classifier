class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] alphabet = new int[26];
        for (char c : param1.toCharArray()) {
            int i = magazine.indexOf(c, alphabet[c % 26]);
            if (i == -1) {
                return false;
            }
            alphabet[c % 26] = i + 1;
        }
        return true;
    }
}
