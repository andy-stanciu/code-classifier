class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int foundIndex = -1;
        for (int i = 0; i < param1.length(); i++) {
            foundIndex = magazine.indexOf(param1.charAt(i));
            if (foundIndex == -1)
                return false;
            magazine = magazine.replaceFirst(String.valueOf(param1.charAt(i)), "");
        }
        return true;
    }
}
