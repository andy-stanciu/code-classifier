class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] r = param1.toCharArray();
        char[] m = magazine.toCharArray();
        Arrays.sort(r);
        Arrays.sort(m);
        int i = 0;
        int j = 0;
        while (j < m.length) {
            if (r[i] == m[j])
                i++;
            if (r.length == i)
                return true;
            j++;
        }
        return false;
    }
}
