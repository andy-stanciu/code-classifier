class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> matchMake = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            matchMake.put(c, matchMake.getOrDefault(c, 0) + 1);
        }
        for (char c : param1.toCharArray()) {
            if (!matchMake.containsKey(c) || matchMake.get(c) <= 0) {
                return false;
            }
            matchMake.put(c, matchMake.get(c) - 1);
        }
        return true;
    }
}
