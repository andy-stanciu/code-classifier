class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] arr = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            int index = magazine.charAt(i) - 97;
            if (arr[index] > 0)
                arr[index] = arr[index] + 1;
            else
                arr[index] = 1;
        }
        for (int i = 0; i < param1.length(); i++) {
            int index = param1.charAt(i) - 97;
            if (arr[index] == 0)
                return false;
            else
                arr[index] = arr[index] - 1;
        }
        return true;
    }
}
