class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> var1 = new HashMap<>();
        HashMap<Character, Integer> mag = new HashMap<>();
        for (int i = 0; i < param1.length(); ++i) {
            var1.put(param1.charAt(i), var1.getOrDefault(param1.charAt(i), 0) + 1);
        }
        for (int i = 0; i < magazine.length(); ++i) {
            mag.put(magazine.charAt(i), mag.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (char c : param1.toCharArray()) {
            if (!mag.containsKey(c) || mag.get(c) < var1.get(c))
                return false;
        }
        return true;
    }
}
