class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        Deque<TreeNode> queue = new ArrayDeque<>();
        queue.offer(root);
        int depth = 0;
        while (!queue.isEmpty()) {
            int len = queue.size();
            List<Integer> var1 = new LinkedList<>();
            for (int i = 0; i < len; i++) {
                TreeNode node = queue.pollFirst();
                boolean reverse = depth % 2 != 0;
                if (!reverse) {
                    var1.add(node.val);
                } else {
                    var1.add(0, node.val);
                }
                if (node.left != null)
                    queue.offer(node.left);
                if (node.right != null)
                    queue.offer(node.right);
            }
            result.add(var1);
            depth++;
        }
        return result;
    }
}
