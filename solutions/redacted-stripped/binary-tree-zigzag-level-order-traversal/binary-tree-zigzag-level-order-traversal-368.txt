class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        traverse(root, 0, result);
        return result;
    }

    private void traverse(TreeNode node, int param1, List<List<Integer>> result) {
        if (node == null)
            return;
        if (result.size() == param1) {
            result.add(new ArrayList<>());
        }
        List<Integer> layer = result.get(param1);
        if (param1 % 2 == 0) {
            layer.addLast(node.val);
        } else {
            layer.addFirst(node.val);
        }
        traverse(node.left, param1 + 1, result);
        traverse(node.right, param1 + 1, result);
    }
}
