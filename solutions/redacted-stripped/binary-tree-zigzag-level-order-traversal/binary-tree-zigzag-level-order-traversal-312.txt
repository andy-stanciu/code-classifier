class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<TreeNode> deque = new ArrayList<>();
        List<TreeNode> temp = null;
        List<List<Integer>> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        deque.add(root);
        boolean isOdd = false;
        while (!deque.isEmpty()) {
            temp = new ArrayList<>();
            List<Integer> values = new ArrayList<>();
            for (TreeNode cur : deque) {
                if (cur.left != null) {
                    temp.add(cur.left);
                }
                if (cur.right != null) {
                    temp.add(cur.right);
                }
                values.add(cur.val);
            }
            if (isOdd) {
                Collections.reverse(values);
            }
            result.add(values);
            isOdd = !isOdd;
            deque = temp;
        }
        return result;
    }
}
