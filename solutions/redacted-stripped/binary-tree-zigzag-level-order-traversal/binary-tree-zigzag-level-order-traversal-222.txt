class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        Queue<TreeNode> que = new LinkedList<>();
        boolean flag = true;
        if (root == null) {
            return ans;
        }
        que.add(root);
        while (!que.isEmpty()) {
            int size = que.size();
            ArrayList<Integer> ll = new ArrayList<>();
            for (int i = 0; i < size; i++) {
                TreeNode node = que.poll();
                if (flag) {
                    ll.add(node.val);
                } else {
                    ll.add(0, node.val);
                }
                if (node.left != null) {
                    que.add(node.left);
                }
                if (node.right != null) {
                    que.add(node.right);
                }
            }
            ans.add(ll);
            flag = !flag;
        }
        return ans;
    }
}
