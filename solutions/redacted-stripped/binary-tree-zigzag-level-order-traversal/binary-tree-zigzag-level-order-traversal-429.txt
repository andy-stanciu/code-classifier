class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        List<List<Integer>> result = new ArrayList<>();
        if (root != null) {
            queue.add(root);
        }
        boolean left = true;
        while (!queue.isEmpty()) {
            int var1 = queue.size();
            List<Integer> var2 = new LinkedList<>();
            for (int i = 0; i < var1; i++) {
                TreeNode node = queue.remove();
                if (left) {
                    var2.addLast(node.val);
                } else {
                    var2.addFirst(node.val);
                }
                if (node.left != null)
                    queue.add(node.left);
                if (node.right != null)
                    queue.add(node.right);
            }
            left = !left;
            result.add(var2);
        }
        return result;
    }
}
