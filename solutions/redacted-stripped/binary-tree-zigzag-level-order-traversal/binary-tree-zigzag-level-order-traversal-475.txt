class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        return method1(0, root, new ArrayList<>());
    }

    public static List<List<Integer>> method1(int param1, TreeNode root, List<List<Integer>> listOfLists) {
        if (root == null)
            return listOfLists;
        if (param1 == listOfLists.size()) {
            listOfLists.add(new LinkedList<>());
        }
        List<Integer> list = listOfLists.get(param1);
        if (param1 % 2 == 0) {
            list.add(root.val);
        } else {
            list.add(0, root.val);
        }
        method1(param1 + 1, root.left, listOfLists);
        method1(param1 + 1, root.right, listOfLists);
        return listOfLists;
    }
}
