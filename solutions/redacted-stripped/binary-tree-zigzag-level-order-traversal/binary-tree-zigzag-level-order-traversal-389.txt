class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> list = new LinkedList<>();
        Queue<TreeNode> q = new LinkedList<>();
        if (root != null)
            q.add(root);
        boolean inverse = false;
        while (!q.isEmpty()) {
            List<Integer> l = new LinkedList<>();
            int n = q.size();
            for (int i = 0; i < n; i++) {
                TreeNode node = q.remove();
                if (node.left != null)
                    q.add(node.left);
                if (node.right != null)
                    q.add(node.right);
                if (inverse == true)
                    l.addFirst(node.val);
                else
                    l.add(node.val);
            }
            list.add(l);
            inverse = !inverse;
        }
        return list;
    }
}
