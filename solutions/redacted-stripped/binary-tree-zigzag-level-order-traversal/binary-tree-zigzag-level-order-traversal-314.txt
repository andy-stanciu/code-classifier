class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        List<List<Integer>> list = new ArrayList<>();
        if (root == null)
            return list;
        q.add(root);
        int index = 0;
        while (!q.isEmpty()) {
            List<Integer> ll = new LinkedList<>();
            index++;
            int len = q.size();
            for (int i = 0; i < len; i++) {
                TreeNode curr = q.poll();
                if (index % 2 != 0)
                    ll.add(curr.val);
                else
                    ll.addFirst(curr.val);
                if (curr.left != null) {
                    q.add(curr.left);
                }
                if (curr.right != null) {
                    q.add(curr.right);
                }
            }
            list.add(ll);
        }
        return list;
    }
}
