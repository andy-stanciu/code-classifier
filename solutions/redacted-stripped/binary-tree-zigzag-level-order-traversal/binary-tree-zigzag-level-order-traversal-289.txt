class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        boolean rev = false;
        Deque<TreeNode> dq = new LinkedList<>();
        dq.offer(root);
        while (!dq.isEmpty()) {
            int var1 = dq.size();
            List<Integer> var2 = new ArrayList<>(var1);
            for (int i = 0; i < var1; i++) {
                if (!rev) {
                    TreeNode currentNode = dq.pollFirst();
                    var2.add(currentNode.val);
                    if (currentNode.left != null) {
                        dq.addLast(currentNode.left);
                    }
                    if (currentNode.right != null) {
                        dq.addLast(currentNode.right);
                    }
                } else {
                    TreeNode currentNode = dq.pollLast();
                    var2.add(currentNode.val);
                    if (currentNode.right != null) {
                        dq.addFirst(currentNode.right);
                    }
                    if (currentNode.left != null) {
                        dq.addFirst(currentNode.left);
                    }
                }
            }
            rev = !rev;
            result.add(var2);
        }
        return result;
    }
}
