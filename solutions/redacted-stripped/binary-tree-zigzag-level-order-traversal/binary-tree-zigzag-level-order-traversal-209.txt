class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        int var1 = 0;
        while (!q.isEmpty()) {
            int var2 = q.size();
            List<Integer> var3 = new ArrayList<>();
            for (int i = 0; i < var2; i++) {
                TreeNode curr = q.poll();
                var3.add(curr.val);
                if (curr.left != null) {
                    q.offer(curr.left);
                }
                if (curr.right != null) {
                    q.offer(curr.right);
                }
            }
            if (var1 % 2 == 0) {
                result.add(var3);
            } else {
                result.add(var3.reversed());
            }
            var1++;
        }
        return result;
    }
}
