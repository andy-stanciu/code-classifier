class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int var1 = 1;
        while (!q.isEmpty()) {
            int s = q.size();
            List<Integer> list = new ArrayList<>();
            for (int i = 0; i < s; i++) {
                TreeNode node = q.poll();
                list.add(node.val);
                if (node.left != null) {
                    q.add(node.left);
                }
                if (node.right != null) {
                    q.add(node.right);
                }
            }
            if (var1 % 2 == 0) {
                Collections.reverse(list);
            }
            ans.add(list);
            var1++;
        }
        return ans;
    }
}
