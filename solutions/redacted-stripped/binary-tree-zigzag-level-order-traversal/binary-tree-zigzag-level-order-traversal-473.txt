class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        if (root == null)
            return List.of();
        List<List<Integer>> result = new ArrayList<>();
        method2(root, 0, result);
        return result;
    }

    private void method2(TreeNode root, int param1, List<List<Integer>> acc) {
        if (root == null)
            return;
        if (param1 == acc.size()) {
            acc.add(new LinkedList<>());
        }
        LinkedList<Integer> list = (LinkedList<Integer>) acc.get(param1);
        if (param1 % 2 == 0) {
            list.addLast(root.val);
        } else {
            list.addFirst(root.val);
        }
        method2(root.left, param1 + 1, acc);
        method2(root.right, param1 + 1, acc);
    }
}
