class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        List<List<Integer>> var1 = new ArrayList<>();
        int var2 = 2;
        if (null != root) {
            while (!queue.isEmpty()) {
                List<TreeNode> var3 = new ArrayList<>();
                List<Integer> nodeDataList = new ArrayList<>();
                while (!queue.isEmpty()) {
                    TreeNode latestNode = queue.poll();
                    nodeDataList.add(latestNode.val);
                    var3.add(latestNode);
                }
                var1.add(nodeDataList);
                Collections.reverse(var3);
                for (TreeNode node : var3) {
                    if (var2 % 2 == 0) {
                        if (null != node.right) {
                            queue.add(node.right);
                        }
                        if (null != node.left) {
                            queue.add(node.left);
                        }
                    } else {
                        if (null != node.left) {
                            queue.add(node.left);
                        }
                        if (null != node.right) {
                            queue.add(node.right);
                        }
                    }
                }
                var2++;
            }
        }
        return var1;
    }
}
