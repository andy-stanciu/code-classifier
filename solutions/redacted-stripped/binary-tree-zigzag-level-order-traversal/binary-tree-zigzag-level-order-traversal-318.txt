class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        boolean reverse = false;
        while (!q.isEmpty()) {
            ArrayList<Integer> var1 = new ArrayList<>();
            int size = q.size();
            while (size-- > 0) {
                TreeNode curr = q.poll();
                var1.add(curr.val);
                if (curr.left != null)
                    q.add(curr.left);
                if (curr.right != null)
                    q.add(curr.right);
            }
            if (reverse)
                Collections.reverse(var1);
            reverse = !reverse;
            result.add(var1);
        }
        return result;
    }
}
