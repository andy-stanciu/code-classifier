class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        boolean direction = true;
        while (!q.isEmpty()) {
            int size = q.size();
            List<Integer> row = new LinkedList<>();
            for (int i = 0; i < size; i++) {
                TreeNode node = q.poll();
                if (direction) {
                    row.add(node.val);
                } else {
                    row.add(0, node.val);
                }
                if (node.left != null) {
                    q.add(node.left);
                }
                if (node.right != null) {
                    q.add(node.right);
                }
            }
            direction = !direction;
            result.add(row);
        }
        return result;
    }
}
