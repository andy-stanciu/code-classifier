class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        f(root, ans, 1);
        for (int i = 1; i < ans.size(); i = i + 2) {
            Collections.reverse(ans.get(i));
        }
        return ans;
    }

    private void f(TreeNode root, List<List<Integer>> ans, int param1) {
        if (root == null)
            return;
        if (param1 > ans.size())
            ans.add(new ArrayList<>());
        ans.get(param1 - 1).add(root.val);
        f(root.left, ans, param1 + 1);
        f(root.right, ans, param1 + 1);
    }
}
