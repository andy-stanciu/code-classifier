class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        dfs(root, 0, result);
        return result;
    }

    private void dfs(TreeNode root, int param1, List<List<Integer>> result) {
        if (root == null)
            return;
        if (result.size() <= param1) {
            result.add(new ArrayList<>());
        }
        if (param1 % 2 == 0) {
            result.get(param1).add(root.val);
        } else {
            result.get(param1).add(0, root.val);
        }
        dfs(root.left, param1 + 1, result);
        dfs(root.right, param1 + 1, result);
    }
}
