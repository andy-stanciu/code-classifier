class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        if (root == null)
            return new LinkedList<>();
        List<List<Integer>> lst = new LinkedList<>();
        boolean flag = true;
        LinkedList<TreeNode> q = new LinkedList<>();
        q.add(root);
        TreeNode dummy = null;
        while (!q.isEmpty()) {
            List<Integer> temp = new LinkedList<>();
            int n = q.size();
            while (n > 0) {
                dummy = q.remove();
                if (dummy.left != null)
                    q.add(dummy.left);
                if (dummy.right != null)
                    q.add(dummy.right);
                if (flag == true) {
                    temp.add(dummy.val);
                } else {
                    temp.addFirst(dummy.val);
                }
                n--;
            }
            flag = flag == true ? false : true;
            lst.add(temp);
        }
        return lst;
    }
}
