class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int pointer1 = m - 1, pointer2 = n - 1, put = m + n - 1;
        while (pointer1 >= 0 && pointer2 >= 0) nums1[put--] = nums1[pointer1] > nums2[pointer2] ? nums1[pointer1--] : nums2[pointer2--];
        while (pointer2 >= 0) nums1[put--] = nums2[pointer2--];
    }
}
