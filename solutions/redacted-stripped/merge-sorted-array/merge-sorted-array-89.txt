class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int j = 0;
        for (int i = 0; i < m; i++) {
            if (j < m) {
                nums1[i] = nums1[i];
                j++;
            }
        }
        for (int i = 0; i < n; i++) {
            nums1[j] = nums2[i];
            j++;
        }
        Arrays.sort(nums1);
    }
}
