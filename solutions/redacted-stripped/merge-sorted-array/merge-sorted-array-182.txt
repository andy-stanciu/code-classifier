class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] temp = Arrays.copyOfRange(nums1, 0, m);
        method2(nums1, temp, nums2, 0, 0);
    }

    private void method2(int[] arr, int[] first, int[] second, int i, int j) {
        if (i + j >= arr.length)
            return;
        if (i < first.length && j < second.length) {
            if (first[i] < second[j]) {
                arr[i + j] = first[i];
                method2(arr, first, second, i + 1, j);
            } else {
                arr[i + j] = second[j];
                method2(arr, first, second, i, j + 1);
            }
        } else if (i < first.length) {
            arr[i + j] = first[i];
            method2(arr, first, second, i + 1, j);
        } else if (j < second.length) {
            arr[i + j] = second[j];
            method2(arr, first, second, i, j + 1);
        }
    }
}
