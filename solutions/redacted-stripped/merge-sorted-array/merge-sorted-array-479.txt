class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        for (int i = m, j = 0; i < nums1.length; i++, j++) {
            if (nums1[i] == 0) {
                nums1[i] = nums2[j];
            } else
                nums1[i] = nums2[j];
        }
        Arrays.sort(nums1);
    }
}
