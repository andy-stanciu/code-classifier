class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int i = 0;
        int j = 0;
        int[] res = new int[m + n];
        int index = 0;
        while (i < m && j < n) {
            if (nums1[i] < nums2[j]) {
                res[index] = nums1[i];
                i++;
            } else {
                res[index] = nums2[j];
                j++;
            }
            index++;
        }
        while (i < m) {
            res[index++] = nums1[i++];
        }
        while (j < n) {
            res[index++] = nums2[j++];
        }
        for (int k = 0; k < res.length; k++) {
            nums1[k] = res[k];
        }
    }
}
