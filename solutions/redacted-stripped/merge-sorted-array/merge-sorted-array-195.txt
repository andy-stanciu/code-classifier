class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        PriorityQueue<Integer> var1 = new PriorityQueue<Integer>();
        for (int i = 0; i < m; i++) {
            var1.add(nums1[i]);
        }
        for (int i = 0; i < n; i++) {
            var1.add(nums2[i]);
        }
        for (int i = 0; i < m + n; i++) {
            nums1[i] = var1.poll();
        }
    }
}
