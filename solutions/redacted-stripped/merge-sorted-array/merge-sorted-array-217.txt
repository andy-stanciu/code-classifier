class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int k = m - 1, j = n - 1, r = n + m - 1;
        while (k >= 0 && j >= 0) {
            if (nums1[k] > nums2[j]) {
                nums1[r--] = nums1[k--];
            } else {
                nums1[r--] = nums2[j--];
            }
        }
        while (r >= 0 && j >= 0) {
            nums1[r--] = nums2[j--];
        }
    }
}
