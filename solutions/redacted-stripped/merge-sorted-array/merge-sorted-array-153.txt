class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int right = nums1.length - 1;
        m--;
        n--;
        while (m >= 0 && n >= 0) {
            nums1[right--] = nums1[m] > nums2[n] ? nums1[m--] : nums2[n--];
        }
        while (n >= 0) {
            nums1[right--] = nums2[n--];
        }
    }
}
