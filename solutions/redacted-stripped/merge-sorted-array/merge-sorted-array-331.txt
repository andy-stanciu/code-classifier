class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] arr = new int[m + n];
        int indexOfNums1 = 0;
        int indexOfNums2 = 0;
        int indexOfResult = 0;
        while (indexOfNums1 < m && indexOfNums2 < n) {
            if (nums1[indexOfNums1] < nums2[indexOfNums2]) {
                arr[indexOfResult] = nums1[indexOfNums1];
                indexOfNums1++;
            } else {
                arr[indexOfResult] = nums2[indexOfNums2];
                indexOfNums2++;
            }
            indexOfResult++;
        }
        while (indexOfNums1 < m) {
            arr[indexOfResult] = nums1[indexOfNums1];
            indexOfNums1++;
            indexOfResult++;
        }
        while (indexOfNums2 < n) {
            arr[indexOfResult] = nums2[indexOfNums2];
            indexOfNums2++;
            indexOfResult++;
        }
        for (int i = 0; i < arr.length; i++) {
            nums1[i] = arr[i];
        }
    }
}
