class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int a = 0;
        int b = 0;
        int[] newnums1 = new int[nums1.length];
        for (int q = 0; q < nums1.length; q++) {
            newnums1[q] = nums1[q];
        }
        for (int l = 0; l < m + n; l++) {
            if (a >= m) {
                nums1[l] = nums2[b];
                b++;
            } else if (b >= n) {
                if (a <= nums1.length) {
                    nums1[l] = newnums1[a];
                    a++;
                } else {
                    break;
                }
            } else if (newnums1[a] <= nums2[b]) {
                nums1[l] = newnums1[a];
                a++;
            } else {
                nums1[l] = nums2[b];
                b++;
            }
        }
    }
}
