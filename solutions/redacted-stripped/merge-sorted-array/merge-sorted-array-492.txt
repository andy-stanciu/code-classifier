class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int k = m;
        int l = 0;
        if (n == 0) {
            l = 0;
        } else if (m == 0) {
            System.arraycopy(nums2, 0, nums1, 0, n);
        } else {
            while (k < m + n) {
                nums1[k] = nums2[l];
                k++;
                l++;
            }
        }
        for (int i = 0; i < m + n; i++) {
            for (int j = i + 1; j < m + n; j++) {
                int temp = 0;
                if (nums1[j] < nums1[i]) {
                    temp = nums1[i];
                    nums1[i] = nums1[j];
                    nums1[j] = temp;
                }
            }
        }
    }
}
