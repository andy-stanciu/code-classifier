class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int last = m + n - 1;
        int nums1Size = m - 1;
        int nums2Size = n - 1;
        while (nums2Size >= 0) {
            if (nums1Size >= 0 && nums1[nums1Size] >= nums2[nums2Size]) {
                nums1[last] = nums1[nums1Size];
                nums1Size--;
                last--;
            } else {
                nums1[last] = nums2[nums2Size];
                nums2Size--;
                last--;
            }
        }
    }
}
