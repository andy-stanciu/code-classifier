class Solution {

    public void method1(int[] arr, int n, int[] num, int m) {
        int i = n - 1;
        int j = m - 1;
        int k = n + m - 1;
        while (i >= 0 && j >= 0) {
            if (arr[i] > num[j]) {
                arr[k--] = arr[i--];
            } else {
                arr[k--] = num[j--];
            }
        }
        while (i >= 0) {
            arr[k--] = arr[i--];
        }
        while (j >= 0) {
            arr[k--] = num[j--];
        }
    }
}
