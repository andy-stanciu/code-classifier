class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int m_tracker = m - 1, n_tracker = n - 1, fin_length = m + n - 1;
        while (m_tracker >= 0 && n_tracker >= 0) {
            if (nums1[m_tracker] >= nums2[n_tracker]) {
                nums1[fin_length--] = nums1[m_tracker--];
            } else {
                nums1[fin_length--] = nums2[n_tracker--];
            }
        }
        while (m_tracker >= 0) {
            nums1[fin_length--] = nums1[m_tracker--];
        }
        while (n_tracker >= 0) {
            nums1[fin_length--] = nums2[n_tracker--];
        }
    }
}
