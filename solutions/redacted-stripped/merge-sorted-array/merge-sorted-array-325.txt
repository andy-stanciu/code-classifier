class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int k = 0;
        for (int i = 0; i < nums1.length; i++) {
            if (nums1[i] == 0) {
                if (k < nums2.length) {
                    nums1[i] = nums2[k];
                    k++;
                }
            }
        }
        Arrays.sort(nums1);
        System.out.println(Arrays.toString(nums1));
    }
}
