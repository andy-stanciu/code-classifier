class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] nums1Temp = nums1.clone();
        int i = 0;
        int j = 0;
        while (i + j < m + n) {
            if (i < m && j < n && nums1Temp[i] <= nums2[j]) {
                nums1[i + j] = nums1Temp[i];
                i++;
            } else if (j < n) {
                nums1[i + j] = nums2[j];
                j++;
            } else {
                nums1[i + j] = nums1Temp[i];
                i++;
            }
        }
    }
}
