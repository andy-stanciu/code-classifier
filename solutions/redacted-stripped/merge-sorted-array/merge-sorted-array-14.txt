class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] var1 = new int[m + n];
        int i = 0, j = 0, k = 0;
        while (i < m) {
            var1[k] = nums1[i];
            i++;
            k++;
        }
        while (j < n) {
            var1[k] = nums2[j];
            j++;
            k++;
        }
        Arrays.sort(var1);
        for (int x = 0; x < m + n; x++) {
            nums1[x] = var1[x];
        }
    }
}
