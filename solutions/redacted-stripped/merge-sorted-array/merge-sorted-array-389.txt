class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int middle = m - 1;
        int last = n - 1;
        int right = m + n - 1;
        while (last >= 0) {
            if (middle >= 0 && nums1[middle] > nums2[last]) {
                nums1[right] = nums1[middle];
                middle--;
            } else {
                nums1[right] = nums2[last];
                last--;
            }
            right--;
        }
    }
}
