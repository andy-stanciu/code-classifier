class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        if (n == 0) {
            return;
        }
        int[] nums3 = new int[m + n];
        int insertPointer = 0;
        int onePointer = 0;
        int twoPointer = 0;
        while (onePointer < m || twoPointer < n) {
            if (twoPointer == n) {
                nums3[insertPointer] = nums1[onePointer];
                onePointer++;
            } else if (onePointer == m) {
                nums3[insertPointer] = nums2[twoPointer];
                twoPointer++;
            } else if (nums1[onePointer] == nums2[twoPointer] || nums1[onePointer] < nums2[twoPointer]) {
                nums3[insertPointer] = nums1[onePointer];
                onePointer++;
            } else {
                nums3[insertPointer] = nums2[twoPointer];
                twoPointer++;
            }
            insertPointer++;
        }
        for (int i = 0; i < nums1.length; i++) {
            nums1[i] = nums3[i];
        }
    }
}
