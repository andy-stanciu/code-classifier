class Solution {

    public void method1(int[] arr1, int m, int[] arr2, int n) {
        int last = m + n - 1;
        int i = m - 1;
        int j = n - 1;
        while (i >= 0 && j >= 0) {
            if (arr1[i] > arr2[j]) {
                arr1[last--] = arr1[i--];
            } else {
                arr1[last--] = arr2[j--];
            }
        }
        while (j >= 0) {
            arr1[last--] = arr2[j--];
        }
    }
}
