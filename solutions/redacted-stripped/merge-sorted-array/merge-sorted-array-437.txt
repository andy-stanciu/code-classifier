class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        if (n == 0)
            return;
        m--;
        n--;
        int c = nums1.length - 1;
        while (m >= 0 && n >= 0) {
            if (nums1[m] >= nums2[n]) {
                nums1[c] = nums1[m];
                m--;
            } else {
                nums1[c] = nums2[n];
                n--;
            }
            c--;
        }
        while (n >= 0) {
            nums1[c] = nums2[n];
            n--;
            c--;
        }
    }
}
