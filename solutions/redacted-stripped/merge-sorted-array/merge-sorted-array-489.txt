class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int p1 = 0;
        int p2 = 0;
        if (n == 0)
            return;
        while (p1 < m && p2 < n) {
            if (nums1[p1] <= nums2[p2]) {
                p1++;
            } else {
                for (int i = m - 1; i >= p1; i--) {
                    nums1[i + 1] = nums1[i];
                }
                nums1[p1] = nums2[p2];
                p1++;
                p2++;
                m++;
            }
        }
        while (p2 < n) {
            nums1[p1] = nums2[p2];
            p1++;
            p2++;
        }
    }
}
