class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int ctr = m + n - 1;
        m--;
        n--;
        while (m >= 0 && n >= 0) {
            if (nums1[m] >= nums2[n]) {
                nums1[ctr--] = nums1[m--];
            } else if (nums2[n] > nums1[m]) {
                nums1[ctr--] = nums2[n--];
            }
        }
        while (n >= 0) {
            nums1[ctr--] = nums2[n--];
        }
    }
}
