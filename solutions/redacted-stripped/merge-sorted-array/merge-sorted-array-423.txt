class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int l = m - 1;
        int r = n - 1;
        for (int i = nums1.length - 1; i >= 0; i--) {
            if (l < 0)
                nums1[i] = nums2[r--];
            else if (r < 0)
                nums1[i] = nums1[l--];
            else if (nums1[l] > nums2[r])
                nums1[i] = nums1[l--];
            else
                nums1[i] = nums2[r--];
        }
    }
}
