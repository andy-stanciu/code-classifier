class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int a = m - 1;
        int b = n - 1;
        int x = m + n - 1;
        while (b >= 0) {
            if (a >= 0 && nums1[a] > nums2[b]) {
                nums1[x--] = nums1[a--];
            } else {
                nums1[x--] = nums2[b--];
            }
        }
    }
}
