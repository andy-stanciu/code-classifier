class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int flag = m + n - 1;
        int flag1 = m - 1;
        int flag2 = n - 1;
        while (flag >= 0) {
            if (flag1 == -1) {
                while (flag2 >= 0) {
                    nums1[flag] = nums2[flag2];
                    flag2--;
                    flag--;
                }
            } else if (flag2 == -1) {
                break;
            } else if (nums2[flag2] > nums1[flag1]) {
                nums1[flag] = nums2[flag2];
                flag2--;
                flag--;
            } else {
                nums1[flag] = nums1[flag1];
                flag1--;
                flag--;
            }
        }
    }
}
