class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet();
        for (int i = 0; i < nums.length; i++) set.add(nums[i]);
        if (set.size() == 1)
            return nums[0];
        if (set.size() == 2)
            return set.last();
        set.remove(set.last());
        set.remove(set.last());
        return set.last();
    }
}
