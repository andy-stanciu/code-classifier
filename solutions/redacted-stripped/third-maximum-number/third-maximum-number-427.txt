class Solution {

    private static final int TOP_N = 3;

    public int method1(int[] nums) {
        Set<Integer> visited = new HashSet<>(nums.length);
        Queue<Integer> largest = new PriorityQueue<>();
        for (int num : nums) {
            if (largest.size() < TOP_N) {
                if (visited.add(num)) {
                    largest.add(num);
                }
            } else {
                if (visited.add(num)) {
                    if (num > largest.element()) {
                        largest.remove();
                        largest.add(num);
                    }
                }
            }
        }
        if (largest.size() < TOP_N) {
            while (largest.size() > 1) {
                largest.remove();
            }
            return largest.remove();
        } else {
            return largest.remove();
        }
    }
}
