class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        if (nums.length < 3)
            return nums[nums.length - 1];
        int counter = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1] && counter == 1)
                return nums[i - 1];
            if (nums[i] != nums[i - 1])
                counter++;
        }
        return nums[nums.length - 1];
    }
}
