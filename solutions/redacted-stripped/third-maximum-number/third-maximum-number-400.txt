class Solution {

    public int method1(int[] nums) {
        long firstMax = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int v : nums) {
            if (v > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = v;
            } else if (v > secondMax && v != firstMax) {
                method1 = secondMax;
                secondMax = v;
            } else if (v > method1 && v != firstMax && v != secondMax) {
                method1 = v;
            }
        }
        if (method1 == Long.MIN_VALUE) {
            method1 = firstMax;
        }
        return (int) method1;
    }
}
