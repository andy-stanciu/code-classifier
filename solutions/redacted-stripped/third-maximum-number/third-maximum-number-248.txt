class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int uniqueIndex = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[uniqueIndex++] = nums[i];
            }
        }
        if (uniqueIndex >= 3) {
            return nums[uniqueIndex - 3];
        } else {
            return nums[uniqueIndex - 1];
        }
    }
}
