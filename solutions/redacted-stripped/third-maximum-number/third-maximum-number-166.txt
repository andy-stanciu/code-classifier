class Solution {

    public int method1(int[] nums) {
        List<Integer> Inner = new ArrayList<>();
        Set<Integer> Setting = new HashSet<>();
        for (int i : nums) {
            Setting.add(i);
        }
        for (int i : Setting) {
            Inner.add(i);
        }
        if (Inner.size() <= 2) {
            int a = Collections.max(Inner);
            return a;
        } else {
            Collections.sort(Inner);
            Collections.reverse(Inner);
            return Inner.get(2);
        }
    }
}
