class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int count = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (count == 3)
                return nums[i];
            if (i == 0)
                return nums[nums.length - 1];
            if (nums[i] == nums[i - 1])
                continue;
            else
                count++;
        }
        return count;
    }
}
