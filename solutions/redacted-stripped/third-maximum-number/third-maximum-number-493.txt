class Solution {

    public int method1(int[] nums) {
        long max = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int num : nums) {
            if (num > max) {
                method1 = secondMax;
                secondMax = max;
                max = num;
            } else if (num > secondMax && num < max) {
                method1 = secondMax;
                secondMax = num;
            } else if (num > method1 && num < secondMax) {
                method1 = num;
            }
        }
        int result = (method1 == Long.MIN_VALUE) ? (int) max : (int) method1;
        return result;
    }
}
