class Solution {

    public int method1(int[] nums) {
        Integer max = null;
        Integer secondMax = null;
        Integer method1 = null;
        for (Integer num : nums) {
            if (num.equals(max) || num.equals(secondMax) || num.equals(method1)) {
                continue;
            }
            if (max == null || num > max) {
                method1 = secondMax;
                secondMax = max;
                max = num;
            } else if (secondMax == null || num > secondMax) {
                method1 = secondMax;
                secondMax = num;
            } else if (method1 == null || num > method1) {
                method1 = num;
            }
        }
        if (method1 == null) {
            return max;
        }
        return method1;
    }
}
