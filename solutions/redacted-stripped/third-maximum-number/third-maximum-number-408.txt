class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        int[] arr = new int[set.size()];
        int index = 0;
        for (int i : set) {
            arr[index] = i;
            index++;
        }
        Arrays.sort(arr);
        int n = arr.length;
        if (arr.length < 3)
            return arr[n - 1];
        return arr[n - 3];
    }
}
