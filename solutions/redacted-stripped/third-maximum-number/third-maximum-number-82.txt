public class Solution {

    public int method1(int[] nums) {
        Integer max1 = null;
        for (Integer num : nums) {
            if (max1 == null || num > max1) {
                max1 = num;
            }
        }
        Integer max2 = null;
        for (Integer num : nums) {
            if ((max2 == null || num > max2) && !num.equals(max1)) {
                max2 = num;
            }
        }
        Integer max3 = null;
        for (Integer num : nums) {
            if ((max3 == null || num > max3) && !num.equals(max1) && !num.equals(max2)) {
                max3 = num;
            }
        }
        return max3 == null ? max1 : max3;
    }
}
