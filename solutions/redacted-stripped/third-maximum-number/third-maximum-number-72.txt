class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> s = new TreeSet<>();
        for (int i = 0; i < nums.length; i++) {
            s.add(nums[i]);
        }
        if (s.size() < 3) {
            return s.last();
        }
        s.pollLast();
        s.pollLast();
        return s.last();
    }
}
