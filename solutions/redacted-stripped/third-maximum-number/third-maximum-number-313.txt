class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        TreeSet<Integer> set = new TreeSet<>();
        for (int n : nums) {
            set.add(n);
        }
        if (set.size() <= 2) {
            return set.last();
        } else {
            return set.lower(set.lower(set.last()));
        }
    }
}
