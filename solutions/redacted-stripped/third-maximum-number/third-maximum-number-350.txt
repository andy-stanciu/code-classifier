class Solution {

    private void removeElement(HashSet<Integer> set, int num) {
        if (set.size() > 3) {
            set.remove(num);
        }
    }

    public int method1(int[] nums) {
        int largest = Integer.MIN_VALUE;
        int secondLargest = Integer.MIN_VALUE;
        int var1 = Integer.MIN_VALUE;
        ;
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            if (set.contains(num)) {
                continue;
            }
            if (largest <= num) {
                removeElement(set, var1);
                var1 = secondLargest;
                secondLargest = largest;
                largest = num;
                set.add(num);
            } else if (largest > num && secondLargest <= num) {
                removeElement(set, var1);
                var1 = secondLargest;
                secondLargest = num;
                set.add(num);
            } else if (secondLargest > num && var1 <= num) {
                removeElement(set, var1);
                var1 = num;
                set.add(num);
            } else {
                continue;
            }
        }
        if (set.size() < 3) {
            return largest;
        } else {
            return var1;
        }
    }
}
