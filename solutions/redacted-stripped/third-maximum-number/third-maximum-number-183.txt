class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int a : nums) {
            set.add(a);
        }
        Integer[] arr = set.toArray(new Integer[set.size()]);
        Arrays.sort(arr);
        int z = arr.length - 3;
        if (z >= 0) {
            return arr[z];
        } else
            return arr[arr.length - 1];
    }
}
