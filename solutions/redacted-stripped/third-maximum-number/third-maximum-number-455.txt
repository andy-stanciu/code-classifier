class Solution {

    public int method1(int[] nums) {
        int firstMax = Integer.MIN_VALUE;
        int secondMax = Integer.MIN_VALUE;
        int method1 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > firstMax) {
                firstMax = nums[i];
            }
        }
        if (nums.length < 3) {
            return firstMax;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > secondMax && nums[i] != firstMax) {
                secondMax = nums[i];
            }
        }
        boolean flag = true;
        for (int num : nums) {
            if (num >= method1 && num != firstMax && num != secondMax) {
                method1 = num;
                flag = false;
            }
        }
        if (flag) {
            return firstMax;
        }
        return method1;
    }
}
