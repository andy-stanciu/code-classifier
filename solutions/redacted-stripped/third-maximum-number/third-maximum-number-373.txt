class Solution {

    public int method1(int[] nums) {
        float fm = Float.NEGATIVE_INFINITY;
        float sm = Float.NEGATIVE_INFINITY;
        float tm = Float.NEGATIVE_INFINITY;
        for (int num : nums) {
            if ((num == fm || num == sm || num == tm))
                continue;
            if (num > fm) {
                float tfm = fm;
                fm = num;
                float tsm = sm;
                sm = tfm;
                tm = tsm;
            } else if (num > sm) {
                float tsm = sm;
                sm = num;
                tm = tsm;
            } else if (num > tm) {
                tm = num;
            }
        }
        return (tm == Float.NEGATIVE_INFINITY) ? (int) fm : (int) tm;
    }
}
