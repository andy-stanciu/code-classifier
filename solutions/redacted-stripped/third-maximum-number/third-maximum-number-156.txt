class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int distinctCount = 1;
        int currentMax = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] != currentMax) {
                distinctCount++;
                currentMax = nums[i];
            }
            if (distinctCount == 3) {
                return nums[i];
            }
        }
        return nums[nums.length - 1];
    }
}
