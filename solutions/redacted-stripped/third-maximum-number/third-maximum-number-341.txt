class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> numSet = new HashSet<>();
        Integer max = null;
        Integer second = null;
        Integer var1 = null;
        for (int n : nums) {
            boolean inSet = numSet.contains(n);
            if (inSet)
                continue;
            numSet.add(n);
            if (max == null || n > max) {
                var1 = second;
                second = max;
                max = n;
                continue;
            }
            if (second == null || n > second) {
                var1 = second;
                second = n;
                continue;
            }
            if (var1 == null || n > var1) {
                var1 = n;
            }
        }
        return var1 == null ? max : var1;
    }
}
