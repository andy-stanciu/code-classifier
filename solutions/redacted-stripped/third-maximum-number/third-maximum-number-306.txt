class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int result = nums[nums.length - 1], updated = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (updated == 3)
                break;
            if (result > nums[i]) {
                result = nums[i];
                updated++;
            }
            if (i == 0 && updated == 2)
                return nums[nums.length - 1];
        }
        return result;
    }
}
