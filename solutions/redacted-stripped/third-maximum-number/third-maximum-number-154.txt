class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int c = 0;
        int n = nums.length;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                c++;
            }
            if (c == 2) {
                return nums[i - 1];
            }
        }
        return nums[n - 1];
    }
}
