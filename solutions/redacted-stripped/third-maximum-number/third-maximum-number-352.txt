class Solution {

    public int method1(int[] nums) {
        long firstMax = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int num : nums) {
            if (num >= firstMax) {
                if (firstMax != num) {
                    method1 = secondMax;
                    secondMax = firstMax;
                }
                firstMax = num;
            } else if (num >= secondMax) {
                method1 = secondMax == num ? method1 : secondMax;
                secondMax = num;
            } else if (num >= method1) {
                method1 = num;
            }
        }
        return (int) (method1 == Long.MIN_VALUE ? firstMax : method1);
    }
}
