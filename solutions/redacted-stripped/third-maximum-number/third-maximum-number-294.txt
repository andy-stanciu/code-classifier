class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int count = 2;
        int ans = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] != nums[i + 1]) {
                count--;
            }
            if (count == 0) {
                return ans = nums[i];
            }
        }
        return ans;
    }
}
