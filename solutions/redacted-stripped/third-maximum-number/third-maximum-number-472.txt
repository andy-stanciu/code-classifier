class Solution {

    public int method1(int[] nums) {
        int ft = Integer.MIN_VALUE, st = Integer.MIN_VALUE, tt = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > ft) {
                tt = st;
                st = ft;
                ft = nums[i];
            } else if (nums[i] > st && nums[i] != ft) {
                tt = st;
                st = nums[i];
            } else if (nums[i] > tt && nums[i] != ft && nums[i] != st) {
                tt = nums[i];
            }
        }
        if (tt == Integer.MIN_VALUE && st == Integer.MIN_VALUE) {
            tt = ft;
        } else if (tt == Integer.MIN_VALUE && !contains(nums, tt)) {
            tt = ft;
        }
        return tt;
    }

    public static boolean contains(int[] arr, int n) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == n) {
                return true;
            }
        }
        return false;
    }
}
