class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> uniqueNums = new TreeSet<>();
        for (int num : nums) {
            uniqueNums.add(num);
        }
        if (uniqueNums.size() < 3) {
            return uniqueNums.last();
        } else {
            return uniqueNums.lower(uniqueNums.lower(uniqueNums.last()));
        }
    }
}
