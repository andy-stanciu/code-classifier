class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) {
            set.add(num);
        }
        if (set.size() < 3)
            return set.getLast();
        set.removeLast();
        set.removeLast();
        return set.getLast();
    }
}
