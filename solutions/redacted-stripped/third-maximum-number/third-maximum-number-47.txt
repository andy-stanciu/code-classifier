class Solution {

    public int method1(int[] nums) {
        Integer first = null, second = null, var1 = null;
        for (Integer num : nums) {
            if (num.equals(first) || num.equals(second) || num.equals(var1))
                continue;
            if (first == null || num > first) {
                var1 = second;
                second = first;
                first = num;
                continue;
            }
            if (second == null || num > second) {
                var1 = second;
                second = num;
                continue;
            }
            if (var1 == null || num > var1) {
                var1 = num;
            }
        }
        return (var1 != null) ? var1 : first;
    }
}
