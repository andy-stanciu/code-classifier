class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> a = new HashSet<>();
        for (int num : nums) {
            a.add(num);
        }
        List<Integer> list = new ArrayList<>(a);
        Collections.sort(list, Collections.reverseOrder());
        if (list.size() >= 3) {
            return list.get(2);
        } else
            return list.get(0);
    }
}
