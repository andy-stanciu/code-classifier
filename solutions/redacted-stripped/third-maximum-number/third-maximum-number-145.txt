class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new LinkedHashSet();
        Arrays.sort(nums);
        for (int i = nums.length - 1; i >= 0; i--) {
            set.add(nums[i]);
        }
        int[] arr = new int[set.size()];
        int k = 0;
        for (int e : set) {
            System.out.println(e);
            arr[k] = e;
            k++;
        }
        if (arr.length < 3) {
            return arr[0];
        }
        return arr[2];
    }
}
