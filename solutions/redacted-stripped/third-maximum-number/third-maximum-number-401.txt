class Solution {

    public int method1(int[] nums) {
        long first_max = Long.MIN_VALUE;
        long second_max = Long.MIN_VALUE;
        long var1 = Long.MIN_VALUE;
        for (int num : nums) {
            if (num > first_max) {
                var1 = second_max;
                second_max = first_max;
                first_max = num;
            } else if (num < first_max && num > second_max) {
                var1 = second_max;
                second_max = num;
            } else if (num < second_max && num > var1) {
                var1 = num;
            }
        }
        if (var1 != Long.MIN_VALUE)
            return (int) var1;
        else
            return (int) first_max;
    }
}
