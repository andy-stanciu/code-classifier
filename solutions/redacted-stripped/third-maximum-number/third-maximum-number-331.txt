class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> hs = new HashSet<Integer>();
        for (int a : nums) {
            hs.add(a);
        }
        ArrayList<Integer> al = new ArrayList<Integer>(hs);
        Collections.sort(al);
        if (al.size() < 3)
            return al.get(al.size() - 1);
        else
            return al.get(al.size() - 3);
    }
}
