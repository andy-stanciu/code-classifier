class Solution {

    public int method1(int[] nums) {
        List<Integer> list = Arrays.stream(nums).distinct().boxed().sorted(Comparator.reverseOrder()).collect(Collectors.toList());
        int index = 2;
        if (index < list.size()) {
            return list.get(index);
        } else {
            return list.get(0);
        }
    }
}
