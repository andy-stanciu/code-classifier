class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        long max, sec, thrd;
        max = sec = thrd = Long.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if ((nums[i] == max) || (nums[i] == sec) || (nums[i] == thrd))
                continue;
            if (nums[i] > max) {
                thrd = sec;
                sec = max;
                max = nums[i];
            } else if (nums[i] > sec) {
                thrd = sec;
                sec = nums[i];
            } else if (nums[i] > thrd) {
                thrd = nums[i];
            }
        }
        return thrd == Long.MIN_VALUE ? (int) max : (int) thrd;
    }
}
