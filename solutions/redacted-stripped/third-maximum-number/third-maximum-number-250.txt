class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int count = 1, max = nums[nums.length - 1], var1 = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (count < 3 && nums[i] != nums[i - 1]) {
                var1 = nums[i - 1];
                count++;
            }
        }
        return (count == 3) ? var1 : max;
    }
}
