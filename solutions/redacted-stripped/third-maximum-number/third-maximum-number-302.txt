class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> p = new HashSet<>();
        for (int y : nums) {
            p.add(y);
        }
        int[] d = new int[p.size()];
        int u = 0;
        for (int y : p) {
            d[u] = y;
            u++;
        }
        Arrays.sort(d);
        if (u <= 2)
            return d[u - 1];
        return d[u - 3];
    }
}
