class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int num = nums[nums.length - 1];
        int var1 = 1;
        int i = nums.length - 1;
        while (var1 < 3 && i > 0) {
            if (nums[i] != nums[i - 1]) {
                var1++;
                if (var1 == 3)
                    return nums[i - 1];
            }
            i--;
        }
        return num;
    }
}
