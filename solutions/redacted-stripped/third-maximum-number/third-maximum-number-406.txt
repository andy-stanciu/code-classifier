class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        if (nums.length < 3) {
            return nums[nums.length - 1];
        }
        int j = 0;
        double min = Double.MAX_VALUE;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] < min) {
                min = nums[i];
                j++;
            }
            if (j == 3) {
                return (int) min;
            }
        }
        return nums[nums.length - 1];
    }
}
