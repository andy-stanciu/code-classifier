class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> h = new TreeSet<Integer>();
        for (int e : nums) {
            h.add(e);
        }
        if (h.size() <= 2) {
            return h.last();
        }
        h.pollLast();
        h.pollLast();
        return h.last();
    }
}
