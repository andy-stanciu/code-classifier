class Solution {

    public int method1(int[] nums) {
        long first = Long.MIN_VALUE;
        long second = Long.MIN_VALUE;
        long var1 = Long.MIN_VALUE;
        for (int num : nums) {
            if (num == first || num == second || num == var1)
                continue;
            if (num > first) {
                var1 = second;
                second = first;
                first = num;
            } else if (num > second) {
                var1 = second;
                second = num;
            } else if (num > var1) {
                var1 = num;
            }
        }
        return var1 != Long.MIN_VALUE ? (int) var1 : (int) first;
    }
}
