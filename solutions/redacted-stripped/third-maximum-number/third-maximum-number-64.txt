class Solution {

    public int method1(int[] nums) {
        int firstMax = Integer.MIN_VALUE;
        int secondMax = Integer.MIN_VALUE;
        int method1 = Integer.MIN_VALUE;
        Set<Integer> seen = new HashSet<>();
        int i = 0;
        while (i < nums.length) {
            if (nums[i] > firstMax && !seen.contains(nums[i])) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = nums[i];
                seen.add(nums[i]);
            } else if (nums[i] > secondMax && !seen.contains(nums[i])) {
                method1 = secondMax;
                secondMax = nums[i];
                seen.add(nums[i]);
            } else if (nums[i] >= method1 && !seen.contains(nums[i])) {
                method1 = nums[i];
                seen.add(nums[i]);
            }
            i++;
        }
        if (seen.size() < 3) {
            return firstMax;
        }
        return method1;
    }
}
