class Solution {

    public int method1(int[] nums) {
        long max1 = Long.MIN_VALUE;
        long max2 = Long.MIN_VALUE;
        long max3 = Long.MIN_VALUE;
        for (int n : nums) {
            max1 = Math.max(max1, n);
        }
        for (int n : nums) {
            if (n > max2 && n < max1) {
                max2 = n;
            }
        }
        for (int n : nums) {
            if (n > max3 && n < max2) {
                max3 = n;
            }
        }
        return max3 == Long.MIN_VALUE ? (int) max1 : (int) max3;
    }
}
