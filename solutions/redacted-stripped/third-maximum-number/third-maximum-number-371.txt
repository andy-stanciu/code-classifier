class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> hash = new HashSet<Integer>();
        Arrays.sort(nums);
        int cnt = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (hash.add(nums[i])) {
                cnt++;
            }
            if (cnt == 3) {
                return nums[i];
            }
        }
        return nums[nums.length - 1];
    }
}
