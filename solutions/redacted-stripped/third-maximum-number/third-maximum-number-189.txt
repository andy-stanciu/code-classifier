class Solution {

    public int method1(int[] nums) {
        Set<Integer> hashSet = new HashSet<Integer>();
        for (int num : nums) {
            hashSet.add(num);
        }
        List<Integer> list = new ArrayList<Integer>(hashSet);
        Collections.sort(list, Collections.reverseOrder());
        if (list.size() >= 3) {
            return list.get(2);
        }
        return list.get(0);
    }
}
