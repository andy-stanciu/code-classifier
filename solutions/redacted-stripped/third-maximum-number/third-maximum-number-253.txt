class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int i, j = 0, len;
        Arrays.sort(nums);
        for (i = 0; i < n; i++) {
            if (nums[j] != nums[i]) {
                j++;
                nums[j] = nums[i];
            }
        }
        len = j + 1;
        if (len > 2) {
            return nums[len - 3];
        } else {
            return nums[len - 1];
        }
    }
}
