class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>((a, b) -> b.compareTo(a));
        for (int num : nums) {
            set.add(num);
            if (set.size() > 3)
                set.pollLast();
        }
        if (set.size() < 3) {
            return set.pollFirst();
        } else {
            return set.pollLast();
        }
    }
}
