class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int item : nums) {
            set.add(item);
        }
        if (set.size() > 2) {
            return (int) set.descendingSet().toArray()[2];
        } else {
            return (int) set.descendingSet().toArray()[0];
        }
    }
}
