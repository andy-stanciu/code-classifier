class Solution {

    public int method1(int[] nums) {
        long max = Long.MIN_VALUE;
        long max2 = Long.MIN_VALUE;
        long max3 = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int n = nums[i];
            if (n > max) {
                max3 = max2;
                max2 = max;
                max = n;
            } else if (n > max2 && n != max) {
                max3 = max2;
                max2 = n;
            } else if (n > max3 && n != max && n != max2) {
                max3 = n;
            }
        }
        if (max3 == Long.MIN_VALUE)
            return (int) max;
        return (int) max3;
    }
}
