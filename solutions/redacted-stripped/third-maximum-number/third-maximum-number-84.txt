class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int j = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[j++] = nums[i];
            }
        }
        if (j < 3) {
            return nums[j - 1];
        }
        return nums[j - 3];
    }
}
