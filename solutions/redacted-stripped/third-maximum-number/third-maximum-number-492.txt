class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        ArrayList<Integer> arr = new ArrayList<Integer>();
        for (int i = 0; i < nums.length; i++) {
            if (arr.contains(nums[i]) == false)
                arr.add(nums[i]);
        }
        int k = 3;
        if (k > arr.size())
            return arr.get(arr.size() - 1);
        else
            return arr.get(arr.size() - 3);
    }
}
