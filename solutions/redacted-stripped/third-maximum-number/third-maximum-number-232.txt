class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        if (set.size() < 3) {
            int max = Collections.max(set);
            return max;
        }
        int firstMax = Integer.MIN_VALUE;
        int secondMax = Integer.MIN_VALUE;
        int method1 = Integer.MIN_VALUE;
        for (int num : set) {
            if (num > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = num;
            } else if (num > secondMax && num < firstMax) {
                method1 = secondMax;
                secondMax = num;
            } else if (num > method1 && num < secondMax) {
                method1 = num;
            }
        }
        return method1;
    }
}
