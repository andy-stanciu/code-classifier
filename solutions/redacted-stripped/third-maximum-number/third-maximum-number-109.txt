class Solution {

    public int method1(int[] nums) {
        long max = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int n : nums) {
            if (max == n || secondMax == n || method1 == n) {
                continue;
            }
            if (n > max) {
                method1 = secondMax;
                secondMax = max;
                max = n;
            } else if (n > secondMax) {
                method1 = secondMax;
                secondMax = n;
            } else if (n > method1) {
                method1 = n;
            }
        }
        return method1 != Long.MIN_VALUE ? (int) method1 : (int) max;
    }
}
