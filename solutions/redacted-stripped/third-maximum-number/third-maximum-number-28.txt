class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int min = Integer.MAX_VALUE + 1;
        int i = nums.length - 1;
        int count = 0;
        while (i >= 0) {
            if (min != nums[i]) {
                count++;
                if (count == 3) {
                    return nums[i];
                }
                min = nums[i];
            }
            i--;
        }
        return nums[nums.length - 1];
    }
}
