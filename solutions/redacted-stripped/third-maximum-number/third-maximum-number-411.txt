class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        Set<Integer> set = new LinkedHashSet();
        for (int i = nums.length - 1; i >= 0; i--) {
            if (set.size() >= 3)
                break;
            set.add(nums[i]);
        }
        int k = 0, ans = -1;
        for (int x : set) {
            k++;
            if (k == 1)
                ans = x;
            if (k == 3) {
                ans = x;
                break;
            }
        }
        return ans;
    }
}
