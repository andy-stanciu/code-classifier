class Solution {

    public int method1(int[] nums) {
        List<Integer> a = new ArrayList<>();
        for (int n : nums) {
            if (!a.contains(n))
                a.add(n);
        }
        Collections.sort(a);
        if (a.size() < 3)
            return a.get(a.size() - 1);
        return a.get(a.size() - 3);
    }
}
