class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        List<Integer> set = new ArrayList<>();
        for (int i : nums) {
            if (!set.contains(i)) {
                set.add(i);
            }
        }
        if (set.size() < 3)
            return set.get(set.size() - 1);
        else
            return set.get(set.size() - 3);
    }
}
