class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> b = new TreeSet<>();
        ArrayList<Integer> a = new ArrayList<>();
        for (int i : nums) {
            b.add(i);
        }
        a.addAll(b);
        return a.size() > 2 ? a.get(a.size() - 3) : a.size() < 3 ? a.get(a.size() - 1) : 0;
    }
}
