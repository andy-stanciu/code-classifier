class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n < 3) {
            Arrays.sort(nums);
            return nums[n - 1];
        } else {
            Arrays.sort(nums);
            int distinctCount = 1;
            int var1 = nums[n - 1];
            for (int i = n - 2; i >= 0; i--) {
                if (nums[i] < nums[i + 1]) {
                    distinctCount++;
                }
                if (distinctCount == 3) {
                    var1 = nums[i];
                    break;
                }
            }
            return var1;
        }
    }
}
