class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int i : nums) set.add(i);
        List<Integer> list = new ArrayList<>(set);
        if (list.size() == 1 || list.size() == 2)
            return list.get(list.size() - 1);
        else
            return list.get(list.size() - 3);
    }
}
