class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        if (set.size() < 3) {
            return Collections.max(set);
        }
        ArrayList<Integer> list = new ArrayList<>(set);
        Collections.sort(list);
        return list.get(list.size() - 3);
    }
}
