class Solution {

    public int method1(int[] nums) {
        long var1 = Long.MIN_VALUE;
        long secondmax = Long.MIN_VALUE;
        long firstmax = Long.MIN_VALUE;
        for (int n : nums) {
            if (n > firstmax) {
                var1 = secondmax;
                secondmax = firstmax;
                firstmax = n;
            } else if (n < firstmax && n > secondmax) {
                var1 = secondmax;
                secondmax = n;
            } else if (n < secondmax && n > var1) {
                var1 = n;
            }
        }
        return (var1 != Long.MIN_VALUE) ? (int) var1 : (int) firstmax;
    }
}
