class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        Arrays.sort(nums);
        for (int i = nums.length - 1; i >= 0; i--) {
            set.add(nums[i]);
            if (set.size() == 3) {
                return nums[i];
            }
        }
        return nums[nums.length - 1];
    }
}
