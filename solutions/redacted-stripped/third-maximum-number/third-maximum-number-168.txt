class Solution {

    public int method1(int[] a) {
        Arrays.sort(a);
        int c = 1;
        for (int i = a.length - 1; i > 0; i--) {
            if (a[i] != a[i - 1]) {
                c++;
                if (c == 3) {
                    return a[i - 1];
                }
            }
        }
        return a[a.length - 1];
    }
}
