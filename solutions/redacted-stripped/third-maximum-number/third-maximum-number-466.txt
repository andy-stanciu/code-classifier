class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        ArrayList<Integer> list = new ArrayList<>();
        int x = 0;
        for (int i = 0; i < nums.length; i++) {
            if (list.contains(nums[i])) {
                continue;
            } else {
                list.add(nums[i]);
            }
        }
        if (list.size() == 1) {
            return list.get(0);
        } else if (list.size() == 2) {
            x = list.get(list.size() - 1);
            return x;
        } else if (list.size() == 3) {
            x = list.get(0);
            return x;
        } else {
            x = list.get(list.size() - 3);
            return x;
        }
    }
}
