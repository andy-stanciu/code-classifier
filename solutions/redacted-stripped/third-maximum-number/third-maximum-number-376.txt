class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int c = 1;
        int n = nums.length - 1;
        while (n > 0 && c < 3) {
            if (nums[n] != nums[n - 1]) {
                c++;
            }
            n--;
        }
        if (c == 3) {
            return nums[n];
        }
        return nums[nums.length - 1];
    }
}
