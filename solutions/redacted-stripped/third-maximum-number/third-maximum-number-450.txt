class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        HashSet<Integer> set1 = new HashSet<>();
        for (int i = nums.length - 1; i >= 0; i--) {
            set1.add(nums[i]);
            if (set1.size() == 3) {
                return nums[i];
            }
        }
        return nums[nums.length - 1];
    }
}
