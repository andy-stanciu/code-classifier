class Solution {

    public int method1(int[] nums) {
        Set<Integer> dc = new HashSet<>();
        for (int num : nums) {
            dc.add(num);
        }
        List<Integer> sorted = new ArrayList<>(dc);
        Collections.sort(sorted, Collections.reverseOrder());
        if (sorted.size() >= 3) {
            return sorted.get(2);
        } else {
            return sorted.get(0);
        }
    }
}
