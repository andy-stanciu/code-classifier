class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> ans = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            ans.add(nums[i]);
        }
        int[] arr = ans.stream().mapToInt(Integer::intValue).toArray();
        Arrays.sort(arr);
        if (arr.length < 3) {
            return arr[arr.length - 1];
        }
        return arr[arr.length - 3];
    }
}
