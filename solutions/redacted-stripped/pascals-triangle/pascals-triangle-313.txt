class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ls = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            List<Integer> innerList = new ArrayList<>(List.of(1));
            int ans = 1;
            for (int j = 1; j < i; j++) {
                ans = ans * (i - j);
                ans = ans / j;
                innerList.add(ans);
            }
            ls.add(innerList);
        }
        return ls;
    }
}
