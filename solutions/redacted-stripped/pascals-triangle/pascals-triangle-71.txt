class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> var1 = new ArrayList<>();
        var1.add(new ArrayList<>());
        var1.get(0).add(1);
        for (int currentRow = 1; currentRow < numRows; currentRow++) {
            var1.add(new ArrayList<>());
            List<Integer> currentRowList = var1.get(currentRow);
            List<Integer> previousRowList = var1.get(currentRow - 1);
            currentRowList.add(1);
            for (int j = 1; j < previousRowList.size(); j++) {
                int sum = previousRowList.get(j) + previousRowList.get(j - 1);
                currentRowList.add(sum);
            }
            currentRowList.add(1);
        }
        return var1;
    }
}
