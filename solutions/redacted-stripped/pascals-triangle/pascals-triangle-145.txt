class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> al = new ArrayList<>();
        al.add(new ArrayList<>());
        al.get(0).add(1);
        for (int i = 1; i < numRows; i++) {
            al.add(new ArrayList<>());
            al.get(i).add(1);
            for (int j = 0; j < al.get(i - 1).size() - 1; j++) al.get(i).add(al.get(i - 1).get(j) + al.get(i - 1).get(j + 1));
            al.get(i).add(1);
        }
        return al;
    }
}
