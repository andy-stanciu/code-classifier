class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> x = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            x.add(new ArrayList<Integer>());
            for (int j = 0; j < i + 1; j++) {
                if ((j == 0) || (j == i)) {
                    x.get(i).add(j, 1);
                } else {
                    int s = x.get(i - 1).get(j - 1) + x.get(i - 1).get(j);
                    x.get(i).add(s);
                }
            }
        }
        return x;
    }
}
