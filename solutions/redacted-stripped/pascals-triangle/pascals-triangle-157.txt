class Solution {

    public List<List<Integer>> generate(int n) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int c = 1;
            List<Integer> temp = new ArrayList<>();
            for (int j = 0; j <= i; j++) {
                temp.add(c);
                c = c * (i - j) / (j + 1);
            }
            ans.add(temp);
        }
        return ans;
    }
}
