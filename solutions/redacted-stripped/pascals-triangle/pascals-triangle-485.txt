class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            List<Integer> list = new ArrayList<>();
            int nCr = 1;
            for (int j = 0; j <= i; j++) {
                list.add(nCr);
                nCr = ((i - j) * nCr) / (j + 1);
            }
            ans.add(list);
        }
        return ans;
    }
}
