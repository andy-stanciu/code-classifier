class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            ArrayList<Integer> res = new ArrayList<>();
            for (int j = 0; j <= i; j++) {
                if (j == 0 || j == i) {
                    res.add(1);
                } else {
                    int value = ans.get(i - 1).get(j - 1) + ans.get(i - 1).get(j);
                    res.add(value);
                }
            }
            ans.add(res);
        }
        return ans;
    }
}
