class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            List<Integer> ls = new ArrayList<>();
            if (i == 0) {
                ls.add(1);
                ans.add(ls);
                continue;
            }
            ls.add(1);
            for (int j = 1; j < i; j++) {
                ls.add(ans.get(i - 1).get(j) + ans.get(i - 1).get(j - 1));
            }
            ls.add(1);
            ans.add(ls);
        }
        return ans;
    }
}
