class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        int N = numRows;
        for (int n = 0; n < N; n++) {
            int nCr = 1;
            List<Integer> list = new ArrayList<>();
            for (int r = 0; r <= n; r++) {
                list.add(nCr);
                nCr = ((n - r) * nCr) / (r + 1);
            }
            ans.add(list);
        }
        return ans;
    }
}
