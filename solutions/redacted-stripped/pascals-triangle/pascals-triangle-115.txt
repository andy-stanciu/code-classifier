class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> var1 = new ArrayList<>();
        var1.add(Arrays.asList(1));
        for (int i = 1; i < numRows; i++) {
            var1.add(new ArrayList<>());
            for (int j = 0; j <= i; j++) {
                if (j == 0 || i == 0 || j == i) {
                    var1.get(i).add(1);
                } else {
                    var1.get(i).add(var1.get(i - 1).get(j) + var1.get(i - 1).get(j - 1));
                }
            }
        }
        return var1;
    }
}
