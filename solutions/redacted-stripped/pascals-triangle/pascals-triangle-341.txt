class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        if (numRows == 0) {
            return ans;
        }
        for (int i = 0; i < numRows; i++) {
            List<Integer> row = new ArrayList<>();
            row.add(1);
            for (int j = 1; j < i; j++) {
                int value = ans.get(i - 1).get(j) + ans.get(i - 1).get(j - 1);
                row.add(value);
            }
            if (i > 0) {
                row.add(1);
            }
            ans.add(row);
        }
        return ans;
    }
}
