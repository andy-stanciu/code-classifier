class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> output = new ArrayList<>();
        output.add(List.of(1));
        for (int i = 1; i < numRows; i++) {
            List<Integer> curr = new ArrayList<>();
            curr.add(1);
            if (output.size() >= 2) {
                int last = output.get(i - 1).get(0);
                for (int j = 1; j < output.get(i - 1).size(); j++) {
                    curr.add(last + output.get(i - 1).get(j));
                    last = output.get(i - 1).get(j);
                }
            }
            curr.add(1);
            output.add(curr);
        }
        return output;
    }
}
