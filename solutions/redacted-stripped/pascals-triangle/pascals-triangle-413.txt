class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> pascal = new ArrayList<>();
        int ans = 1;
        List<Integer> row1 = new ArrayList<>();
        row1.add(ans);
        pascal.add(row1);
        for (int i = 2; i <= numRows; i++) {
            List<Integer> row = new ArrayList<>();
            row.add(ans);
            for (int j = 1; j < i; j++) {
                ans = ans * (i - j);
                ans = ans / j;
                row.add(ans);
            }
            ans = 1;
            pascal.add(row);
        }
        return pascal;
    }
}
