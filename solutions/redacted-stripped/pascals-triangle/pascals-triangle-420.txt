class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> answer = new ArrayList<>();
        List<Integer> initArray = Arrays.asList(1);
        answer.add(initArray);
        if (numRows == 1) {
            return answer;
        }
        for (int i = 1; i < numRows; i++) {
            int initIndex = 0;
            List<Integer> tempArray = new ArrayList<>();
            tempArray.add(1);
            while (initIndex < initArray.size() - 1) {
                tempArray.add(initArray.get(initIndex) + initArray.get(initIndex + 1));
                initIndex++;
            }
            tempArray.add(1);
            answer.add(tempArray);
            initArray = tempArray;
        }
        return answer;
    }
}
