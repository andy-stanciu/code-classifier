class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            List<Integer> val = new ArrayList<>();
            int combination = 1;
            for (int j = 0; j <= i; j++) {
                val.add(combination);
                combination = combination * (i - j) / (j + 1);
            }
            ans.add(val);
        }
        return ans;
    }
}
