class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            List<Integer> list = new ArrayList<>();
            if (i == 0) {
                list.add(1);
                ans.add(list);
            } else if (i == 1) {
                list.add(1);
                list.add(1);
                ans.add(list);
            } else {
                list.add(1);
                List<Integer> previous = ans.getLast();
                for (int j = 1; j < i; j++) {
                    int current = previous.get(j - 1) + previous.get(j);
                    list.add(current);
                }
                list.add(1);
                ans.add(list);
            }
        }
        return ans;
    }
}
