class Solution {

    public List<List<Integer>> generate(int numRows) {
        if (numRows == 0)
            return new ArrayList<>();
        List<List<Integer>> res = new ArrayList<>();
        int k = 1;
        while (k <= numRows) {
            if (k == 1) {
                List<Integer> temp = new ArrayList<>();
                temp.add(1);
                res.add(temp);
            } else if (k == 2) {
                List<Integer> temp = new ArrayList<>();
                temp.add(1);
                temp.add(1);
                res.add(temp);
            } else {
                List<Integer> temp = new ArrayList<>();
                temp.add(1);
                List<Integer> prev = res.get(k - 2);
                int j = 1;
                for (int i = 0; i < prev.size() - 1 && j < prev.size(); i++) {
                    int a = prev.get(i);
                    int b = prev.get(j++);
                    temp.add(a + b);
                }
                temp.add(1);
                res.add(temp);
            }
            k++;
        }
        return res;
    }
}
