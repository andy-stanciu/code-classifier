class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> var1 = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            List<Integer> row = new ArrayList<>();
            for (int j = 0; j <= i; j++) {
                row.add(1);
            }
            for (int j = 1; j < i; j++) {
                int value = var1.get(i - 1).get(j - 1) + var1.get(i - 1).get(j);
                row.set(j, value);
            }
            var1.add(row);
        }
        return var1;
    }
}
