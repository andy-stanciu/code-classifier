class Solution {

    public List<List<Integer>> generate(int n) {
        List<List<Integer>> ans = new ArrayList<>();
        if (n == 0) {
            return new ArrayList<>();
        }
        for (int i = 1; i <= n; i++) {
            List<Integer> row = new ArrayList<>();
            for (int j = 0; j < i; j++) {
                if (j == 0 || j == i - 1) {
                    row.add(1);
                } else {
                    int sum = ans.get(i - 2).get(j) + ans.get(i - 2).get(j - 1);
                    row.add(sum);
                }
            }
            ans.add(row);
        }
        return ans;
    }
}
