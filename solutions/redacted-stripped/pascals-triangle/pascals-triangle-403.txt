class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> currentList = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            if (i == 0) {
                currentList = new ArrayList<>();
                currentList.add(1);
            } else if (i == 1) {
                currentList = new ArrayList<>();
                currentList.add(1);
                currentList.add(1);
            } else {
                List<Integer> middleList = new ArrayList<>();
                middleList.add(1);
                int left = 0;
                int right = 1;
                while (right < currentList.size()) {
                    int sum = currentList.get(left) + currentList.get(right);
                    middleList.add(sum);
                    left++;
                    right++;
                }
                middleList.add(1);
                currentList = new ArrayList<>(middleList);
            }
            result.add(currentList);
        }
        return result;
    }
}
