class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> results = new ArrayList<>();
        List<Integer> prev = new ArrayList();
        prev.add(1);
        results.add(prev);
        for (int x = 1; x < numRows; x++) {
            List<Integer> row = new ArrayList();
            results.add(row);
            row.add(1);
            for (int y = 1; y < x; y++) {
                row.add(prev.get(y) + prev.get(y - 1));
            }
            row.add(1);
            prev = row;
        }
        return results;
    }
}
