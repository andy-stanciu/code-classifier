class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> result = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            List<Integer> first = new ArrayList<>();
            int value = 1;
            for (int j = 1; j <= i; j++) {
                first.add(value);
                value = value * (i - j) / j;
            }
            result.add(first);
        }
        return result;
    }
}
