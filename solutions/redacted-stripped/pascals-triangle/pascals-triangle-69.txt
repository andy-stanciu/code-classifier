class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> listOfList = new ArrayList<List<Integer>>();
        for (int i = 0; i < numRows; i++) {
            List<Integer> l = new ArrayList<Integer>();
            for (int j = 0; j <= i; j++) {
                if (j == 0 || j == i) {
                    l.add(1);
                } else {
                    int a = listOfList.get(i - 1).get(j - 1);
                    int b = listOfList.get(i - 1).get(j);
                    int c = a + b;
                    l.add(c);
                }
            }
            listOfList.add(l);
        }
        return listOfList;
    }
}
