class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> list = new ArrayList<>();
        for (int row = 0; row < numRows; row++) {
            List<Integer> current_row = Arrays.asList(new Integer[row + 1]);
            current_row.set(0, 1);
            current_row.set(row, 1);
            for (int col = 1; col < row; col++) {
                int value = list.get(row - 1).get(col) + list.get(row - 1).get(col - 1);
                current_row.set(col, value);
            }
            list.add(current_row);
        }
        return list;
    }
}
