class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> list = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            ArrayList<Integer> newList = new ArrayList<>();
            long ans = 1;
            newList.add((int) ans);
            for (int j = 1; j < i; j++) {
                ans = ans * (i - j) / j;
                newList.add((int) ans);
            }
            list.add(newList);
            ans = 1;
        }
        return list;
    }
}
