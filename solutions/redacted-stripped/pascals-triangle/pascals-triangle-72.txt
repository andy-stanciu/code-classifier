class Solution {

    public int calculateCombination(int n, int r) {
        int result = 1;
        for (int i = 0; i < r; i++) {
            result = result * (n - i) / (i + 1);
        }
        return result;
    }

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> var1 = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            var1.add(new ArrayList<>());
            List<Integer> currentRow = var1.get(i);
            for (int j = 0; j <= i; j++) {
                currentRow.add(calculateCombination(i, j));
            }
        }
        return var1;
    }
}
