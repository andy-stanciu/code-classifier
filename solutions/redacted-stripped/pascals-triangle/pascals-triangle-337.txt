class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            ans.add(method1(i));
        }
        return ans;
    }

    public List<Integer> method1(int c) {
        List<Integer> li = new ArrayList<>();
        int ans = 1;
        li.add(ans);
        for (int i = 1; i < c; i++) {
            ans = ans * (c - i);
            ans = ans / i;
            li.add(ans);
        }
        return li;
    }
}
