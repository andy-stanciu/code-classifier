class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> rows = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            rows.add(pascal(i));
        }
        return rows;
    }

    private static List<Integer> pascal(int n) {
        long ans = 1;
        List<Integer> c = new ArrayList<>();
        c.add(1);
        for (int col = 1; col < n; col++) {
            ans = ans * (n - col);
            ans = ans / col;
            c.add((int) ans);
        }
        return c;
    }
}
