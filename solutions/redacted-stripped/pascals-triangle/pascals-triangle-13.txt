class Solution {

    public static List<Integer> GeneratorFun(int R) {
        List<Integer> Ans = new ArrayList<Integer>();
        int ans = 1;
        Ans.add(1);
        for (int i = 1; i <= R; i++) {
            ans = ans * (R - i + 1) / i;
            Ans.add(ans);
        }
        return Ans;
    }

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> Fans = new ArrayList<List<Integer>>();
        for (int i = 0; i < numRows; i++) {
            Fans.add(GeneratorFun(i));
        }
        return Fans;
    }
}
