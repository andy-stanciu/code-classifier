class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> res = new LinkedList<>();
        for (int i = 0; i < nums.length - 2; i++) {
            if (i != 0 && nums[i] == nums[i - 1])
                continue;
            int num1 = nums[i];
            int strt = i + 1, end = nums.length - 1;
            while (strt < end) {
                int sum = num1 + nums[strt] + nums[end];
                if (sum == 0) {
                    List<Integer> lst = new LinkedList<>();
                    lst.add(num1);
                    lst.add(nums[strt]);
                    lst.add(nums[end]);
                    res.add(lst);
                    strt++;
                    while (strt < end && nums[strt] == nums[strt - 1]) {
                        strt++;
                    }
                } else if (sum > 0) {
                    end--;
                } else {
                    strt++;
                }
            }
        }
        return res;
    }
}
