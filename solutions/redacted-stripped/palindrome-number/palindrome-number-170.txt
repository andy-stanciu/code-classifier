class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        } else {
            int original = x;
            int reverse = 0;
            while (x != 0) {
                int remainder = x % 10;
                reverse = reverse * 10 + remainder;
                x = x / 10;
            }
            if (original == reverse) {
                return true;
            } else {
                return false;
            }
        }
    }
}
