class Solution {

    public boolean method1(int x) {
        String s = Integer.toString(x);
        int end = s.length() - 1;
        int start = 0;
        while (start < end) {
            if (s.charAt(start) != s.charAt(end)) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
