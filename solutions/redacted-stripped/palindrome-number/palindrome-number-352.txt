class Solution {

    public boolean method1(int x) {
        int temp = x, rem = 0, res = 0;
        if (x < 0) {
            return false;
        }
        while (x != 0) {
            rem = x % 10;
            res = res * 10 + rem;
            x /= 10;
        }
        if (res == temp) {
            return true;
        }
        return false;
    }
}
