class Solution {

    public static boolean method1(int x) {
        int rem, sum = 0, temp = x;
        while (x > 0) {
            rem = x % 10;
            sum = (sum * 10) + rem;
            x = x / 10;
        }
        if (temp == sum) {
            return true;
        } else {
            return false;
        }
    }

    public static void main(String[] args) {
        method1(151);
    }
}
