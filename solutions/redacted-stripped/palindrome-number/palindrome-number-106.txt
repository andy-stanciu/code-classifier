class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        String old = Integer.toString(x);
        for (int i = 0; i < old.length() / 2; i++) {
            if (old.charAt(i) != old.charAt(old.length() - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
