class Solution {

    public boolean method1(int x) {
        int var1 = x;
        if (x == 0)
            return true;
        if (x < 0 || x % 10 == 0)
            return false;
        int rev = 0;
        while (x > 0) {
            int digit = x % 10;
            rev = rev * 10 + digit;
            x = x / 10;
        }
        if (var1 == rev) {
            return true;
        } else {
            return false;
        }
    }
}
