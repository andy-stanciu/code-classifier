class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int r;
        int b = 0;
        int t = x;
        while (t != 0) {
            r = t % 10;
            b = b * 10 + r;
            t = t / 10;
        }
        return b == x;
    }
}
