class Solution {

    public boolean method1(int x) {
        int count = 0;
        String str = Integer.toString(x);
        int len = str.length();
        if (x < 0) {
            return false;
        }
        for (int i = 0; i < len / 2; i++) {
            if (str.charAt(i) == str.charAt(len - i - 1)) {
                count++;
            }
        }
        if (count == len / 2) {
            return true;
        } else {
            return false;
        }
    }
}
