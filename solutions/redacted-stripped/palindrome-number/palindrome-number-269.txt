class Solution {

    public boolean method1(int x) {
        int reminder = 0;
        int num = x;
        int var1 = 0;
        if (num < 0) {
            return false;
        } else {
            while (num != 0) {
                reminder = num % 10;
                var1 = (var1 * 10) + reminder;
                num /= 10;
            }
        }
        if (x == var1) {
            return true;
        } else {
            return false;
        }
    }
}
