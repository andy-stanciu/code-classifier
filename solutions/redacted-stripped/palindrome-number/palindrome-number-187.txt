class Solution {

    public boolean method1(int x) {
        char[] var1 = String.valueOf(x).toCharArray();
        for (int i = 0; i < var1.length / 2; i++) {
            if (var1[i] != var1[var1.length - 1 - i]) {
                return false;
            }
        }
        return true;
    }
}
