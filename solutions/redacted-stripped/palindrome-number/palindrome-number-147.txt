class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        } else {
            int temp = x, n = 0;
            while (temp != 0) {
                n = (n * 10) + temp % 10;
                temp = temp / 10;
            }
            if (n == x) {
                return true;
            } else {
                return false;
            }
        }
    }
}
