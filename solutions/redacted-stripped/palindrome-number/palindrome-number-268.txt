class Solution {

    public boolean method1(int x) {
        int reversedX = 0;
        int copiedX = x;
        if (x < 0) {
            return false;
        }
        while (copiedX != 0) {
            int digit = copiedX % 10;
            reversedX = reversedX * 10 + digit;
            copiedX = copiedX / 10;
        }
        if (x == reversedX) {
            return true;
        } else {
            return false;
        }
    }
}
