class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int num = reverse(x, 0);
        return num == x;
    }

    public int reverse(int x, int res) {
        if (x == 0) {
            return res;
        }
        return reverse(x / 10, (x % 10) + res * 10);
    }
}
