class Solution {

    public boolean method1(int x) {
        boolean check = false;
        String s1 = Integer.toString(x);
        if (s1.length() == 1)
            return true;
        for (int i = 0, j = s1.length() - 1; i < (s1.length()) / 2 && j > 0; i++, j--) {
            if (s1.charAt(i) != s1.charAt(j)) {
                return false;
            } else {
                check = true;
            }
        }
        return check;
    }
}
