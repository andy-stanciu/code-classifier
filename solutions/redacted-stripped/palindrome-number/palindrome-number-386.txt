class Solution {

    public boolean method1(int x) {
        int mul = 0, rem, temp = x;
        while (x > 0) {
            rem = x % 10;
            mul = (mul * 10) + rem;
            x = x / 10;
        }
        if (mul == temp)
            return true;
        else
            return false;
    }
}
