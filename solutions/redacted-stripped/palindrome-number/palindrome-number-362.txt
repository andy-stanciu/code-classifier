class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int n = x;
        int num = 0;
        while (n != 0) {
            int t = n % 10;
            num = num * 10 + t;
            n = n / 10;
        }
        return num == x;
    }
}
