class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int rem = 0, sum = 0;
        int num = x;
        while (x != 0) {
            rem = x % 10;
            sum = sum * 10 + rem;
            x = x / 10;
        }
        if (num == sum) {
            return true;
        } else {
            return false;
        }
    }
}
