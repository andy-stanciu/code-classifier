class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int var1 = x;
        int var2 = 0;
        while (x != 0) {
            int digit = x % 10;
            var2 = var2 * 10 + digit;
            x /= 10;
        }
        return var1 == var2;
    }
}
