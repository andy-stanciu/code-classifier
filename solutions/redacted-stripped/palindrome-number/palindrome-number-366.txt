class Solution {

    public boolean method1(int x) {
        String var1 = Integer.toString(x);
        String var2 = "";
        char letter;
        for (int i = 0; i < var1.length(); i++) {
            letter = var1.charAt(i);
            var2 = letter + var2;
        }
        for (int i = 0; i < var1.length(); i++) {
            if (!(var1.charAt(i) == var2.charAt(i))) {
                return false;
            }
        }
        return true;
    }
}
