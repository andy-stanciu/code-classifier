class Solution {

    public boolean method1(int x) {
        int var1 = 0;
        int dup = x;
        while (x > 0) {
            int ld = x % 10;
            var1 = var1 * 10 + ld;
            x = x / 10;
        }
        if (var1 == dup) {
            return true;
        } else {
            return false;
        }
    }
}
