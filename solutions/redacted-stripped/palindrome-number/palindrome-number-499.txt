class Solution {

    public boolean method1(int x) {
        boolean i;
        int n = 0;
        int r = x;
        if (x < 0) {
            i = false;
        } else {
            while (x != 0) {
                n = n * 10 + x % 10;
                x = x / 10;
            }
            if (r == n) {
                i = true;
            } else {
                i = false;
            }
        }
        return i;
    }
}
