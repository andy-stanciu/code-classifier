class Solution {

    public boolean method1(int x) {
        String word = String.valueOf(x);
        char[] charArr = word.toCharArray();
        int start = 0;
        int end = word.length() - 1;
        while (start < end) {
            if (charArr[start] != charArr[end]) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
