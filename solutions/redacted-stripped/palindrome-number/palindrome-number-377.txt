class Solution {

    public boolean method1(int x) {
        int original = x;
        if (x < 0) {
            return false;
        }
        int new_num = 0;
        int remainder = 0;
        int quote = 0;
        while (x != 0) {
            remainder = x % 10;
            new_num = new_num * 10 + remainder;
            x = x / 10;
        }
        if (new_num == original) {
            return true;
        }
        return false;
    }
}
