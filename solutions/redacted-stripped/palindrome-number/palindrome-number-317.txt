class Solution {

    public boolean method1(int x) {
        if (x == 0) {
            return true;
        } else if (x < 0) {
            return false;
        } else if (x == reverse(x)) {
            return true;
        } else {
            return false;
        }
    }

    int reverse(int x) {
        int rev = 0;
        while (x != 0) {
            int rem = x % 10;
            rev = rev * 10 + rem;
            x = x / 10;
        }
        return rev;
    }
}
