class Solution {

    int rev(int n, int s) {
        if (n == 0)
            return s;
        int sum = s * 10 + n % 10;
        return rev(n / 10, sum);
    }

    public boolean method1(int x) {
        if (x < 0)
            return false;
        if (rev(x, 0) == x)
            return true;
        else
            return false;
    }
}
