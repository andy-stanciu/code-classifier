class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int original = x;
        int invertido = 0;
        int resto;
        while (x != 0) {
            resto = x % 10;
            invertido = invertido * 10 + resto;
            x /= 10;
        }
        return original == invertido;
    }
}
