class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int rem = 0;
        int num = 0;
        int x1 = x;
        while (x > 0) {
            rem = x % 10;
            num = num * 10 + rem;
            x = x / 10;
        }
        return x1 == num;
    }
}
