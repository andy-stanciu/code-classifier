class Solution {

    public boolean method1(int x) {
        int n = x, rem = 0, dig;
        while (x != 0) {
            dig = x % 10;
            rem = rem * 10 + dig;
            x = x / 10;
        }
        if (n < 0) {
            return false;
        } else if (rem == n) {
            return true;
        } else {
            return false;
        }
    }
}
