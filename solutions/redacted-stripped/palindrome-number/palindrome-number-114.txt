class Solution {

    public int check(long n) {
        long rev = 0;
        long i = n;
        while (i > 0) {
            int c = (int) (i % 10);
            rev = rev * 10 + c;
            i = i / 10;
        }
        if (rev == n)
            return 1;
        else
            return 0;
    }

    public boolean method1(int x) {
        if (x < 0)
            return false;
        else {
            int p = check(x);
            return p == 1;
        }
    }
}
