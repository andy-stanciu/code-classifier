class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int var1 = 0;
        while (x > var1) {
            var1 = var1 * 10 + x % 10;
            x /= 10;
        }
        return x == var1 || x == (var1 / 10);
    }
}
