class Solution {

    public boolean method1(int x) {
        long n = x;
        long ans = 0;
        while (x > 0) {
            ans += x % 10;
            ans = ans * 10;
            x = x / 10;
        }
        ans = ans / 10;
        if (ans == n)
            return true;
        return false;
    }
}
