class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int original = x;
        int rev = 0;
        while (x != 0) {
            int i = x % 10;
            rev = rev * 10 + i;
            x = x / 10;
        }
        return rev == original;
    }
}
