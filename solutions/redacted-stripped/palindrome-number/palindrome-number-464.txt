class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int revN = 0;
        int oriN = x;
        while (x > 0) {
            int lastD = x % 10;
            x = x / 10;
            revN = revN * 10 + lastD;
        }
        return oriN == revN;
    }
}
