class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int originalNum = x;
        int reverseNum = 0;
        while (x != 0) {
            int lastdigit = x % 10;
            reverseNum = reverseNum * 10 + lastdigit;
            x /= 10;
        }
        return originalNum == reverseNum;
    }
}
