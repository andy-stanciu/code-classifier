class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int var1 = x;
        int reverse = 0;
        while (var1 > 0) {
            reverse = reverse * 10 + var1 % 10;
            var1 = var1 / 10;
        }
        return x == reverse;
    }
}
