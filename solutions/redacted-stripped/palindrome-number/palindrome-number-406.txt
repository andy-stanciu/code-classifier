class Solution {

    public boolean method1(int x) {
        String text = Integer.toString(x);
        int last = text.length() - 1;
        int start = 0;
        while (start < last) {
            if (text.charAt(start) == text.charAt(last)) {
                start++;
                last--;
            } else {
                return false;
            }
        }
        return true;
    }
}
