public class Solution {

    public static boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int num = x;
        int invertido = 0;
        while (x != 0) {
            int digito = x % 10;
            invertido = invertido * 10 + digito;
            x = x / 10;
        }
        return num == invertido;
    }
}
