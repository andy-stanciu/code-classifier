class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        return helper(String.valueOf(x), 0);
    }

    public boolean helper(String str, int num) {
        if (num >= str.length() / 2)
            return true;
        if (str.charAt(num) != str.charAt(str.length() - num - 1))
            return false;
        return helper(str, num + 1);
    }
}
