class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        String num = String.valueOf(x);
        int length = num.length();
        for (int i = 0; i < length / 2; i++) {
            if (num.charAt(i) != num.charAt(length - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
