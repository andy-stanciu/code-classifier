class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int var1 = 0;
        int var2 = x;
        while (x != 0) {
            int digit = x % 10;
            var1 = var1 * 10 + digit;
            x /= 10;
        }
        return var2 == var1;
    }
}
