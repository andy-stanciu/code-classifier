class Solution {

    public boolean method1(int n) {
        {
            int rev = 0;
            int temp = n;
            if (n < 0)
                return false;
            else {
                while (temp != 0) {
                    rev = rev * 10 + temp % 10;
                    temp /= 10;
                }
                if (rev == n)
                    return true;
                else
                    return false;
            }
        }
    }
}
