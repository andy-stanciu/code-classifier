class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        String m = String.valueOf(x);
        for (int i = 0; i < m.length() - 1; i++) {
            if (m.charAt(i) != m.charAt((m.length() - i) - 1)) {
                return false;
            }
        }
        return true;
    }
}
