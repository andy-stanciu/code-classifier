class Solution {

    public boolean method1(int x) {
        int reverse = 0;
        int num = x;
        if (x < 0 || (x != 0 && x % 10 == 0)) {
            return false;
        } else if (x % 10 == x) {
            return true;
        } else {
            while (num != 0) {
                int last = num % 10;
                reverse = reverse * 10 + last;
                num = num / 10;
            }
            return x == reverse;
        }
    }
}
