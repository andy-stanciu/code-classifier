class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int O = x;
        int r = 0;
        int d;
        while (O != 0) {
            d = O % 10;
            r = r * 10 + d;
            O = O / 10;
        }
        return r == x;
    }
}
