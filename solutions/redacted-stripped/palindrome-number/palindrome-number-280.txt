class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        List<Integer> var1 = new ArrayList();
        while (x > 0) {
            var1.add(x % 10);
            x = x / 10;
        }
        for (int i = 0; i < var1.size(); i++) {
            if (var1.get(i) != var1.get(var1.size() - (i + 1))) {
                return false;
            }
        }
        return true;
    }
}
