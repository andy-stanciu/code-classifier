class Solution {

    public boolean method1(int x) {
        int rev = 0;
        int n;
        n = x;
        while (x > 0) {
            rev = rev * 10 + x % 10;
            x = x / 10;
        }
        if (rev == n) {
            return true;
        } else {
            return false;
        }
    }
}
