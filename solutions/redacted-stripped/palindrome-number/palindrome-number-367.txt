class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        String s = Integer.toString(x);
        int i = 0, j = s.length() - 1;
        while (i != s.length() / 2) {
            if (s.charAt(i) != s.charAt(j))
                return false;
            i++;
            j--;
        }
        return true;
    }
}
