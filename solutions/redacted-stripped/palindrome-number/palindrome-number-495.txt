public class Solution {

    public boolean method1(long x) {
        if (x < 0) {
            return false;
        }
        long rem = 0;
        long x1 = x;
        long newNum = 0;
        while (x1 > 0) {
            rem = x1 % 10;
            x1 = x1 / 10;
            newNum = newNum * 10 + rem;
        }
        return newNum == x;
    }
}
