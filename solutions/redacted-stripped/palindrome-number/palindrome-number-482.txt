class Solution {

    public boolean method1(int x) {
        int rev = 0;
        int num = x;
        while (x != 0) {
            int rem = x % 10;
            rev = rev * 10 + rem;
            x /= 10;
        }
        if (num == rev) {
            if (num >= 0) {
                return true;
            }
        }
        return false;
    }
}
