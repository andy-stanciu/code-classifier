class Solution {

    public boolean method1(int x) {
        int rev = 0;
        int nx = x;
        while (x > 0) {
            int last = x % 10;
            rev = rev * 10 + last;
            x = x / 10;
        }
        if (rev == nx) {
            return true;
        } else {
            return false;
        }
    }
}
