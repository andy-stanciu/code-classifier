class Solution {

    public boolean method1(int x) {
        int k = x;
        int num = 0;
        if (x < 0) {
            return false;
        }
        while (x > 0) {
            int rem = x % 10;
            num = rem + (num * 10);
            x = x / 10;
        }
        if (num == k) {
            return true;
        }
        return false;
    }
}
