class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int num = 0;
        int a = x;
        int s;
        while (x > 0) {
            s = x % 10;
            num = num * 10 + s;
            x /= 10;
        }
        if (a == num) {
            return true;
        } else {
            return false;
        }
    }
}
