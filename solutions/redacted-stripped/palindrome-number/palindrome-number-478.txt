class Solution {

    public boolean method1(int x) {
        int rev = 0, rem = 0;
        int m = Math.abs(x);
        while (x != 0) {
            rem = x % 10;
            rev = rev * 10 + rem;
            x = x / 10;
        }
        if (m == rev) {
            return true;
        } else {
            return false;
        }
    }
}
