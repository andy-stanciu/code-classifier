class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int c = x;
        int rev = 0;
        while (x != 0) {
            int r = x % 10;
            rev = (rev * 10) + r;
            x = x / 10;
        }
        System.out.print(rev);
        System.out.print(c);
        return rev == c;
    }
}
