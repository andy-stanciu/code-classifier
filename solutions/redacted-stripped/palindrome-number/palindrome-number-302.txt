class Solution {

    public boolean method1(int x) {
        int r;
        int sum = 0;
        int temp = x;
        if (x < 0) {
            return false;
        }
        while (x != 0) {
            r = x % 10;
            sum = (sum * 10) + r;
            x = x / 10;
        }
        return (sum == temp);
    }
}
