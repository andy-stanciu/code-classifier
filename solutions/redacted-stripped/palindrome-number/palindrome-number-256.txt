class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int rem = 0;
        int copy = x;
        while (x != 0) {
            int digit = x % 10;
            x = x / 10;
            rem = (rem * 10) + digit;
        }
        return (rem == copy);
    }
}
