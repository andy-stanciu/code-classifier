class Solution {

    public boolean method1(int x) {
        int num = 0, original = x;
        if (x < 10 && x >= 0)
            return true;
        if (x < 0)
            return false;
        while (x > 0) {
            num = (num + x % 10) * 10;
            x = x / 10;
            if (x < 10)
                break;
        }
        num += x;
        return num == original;
    }
}
