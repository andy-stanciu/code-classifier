class Solution {

    public boolean method1(int x) {
        int nums = reverseInt(x);
        if (x < 0) {
            return false;
        } else if (x == nums) {
            return true;
        } else {
            return false;
        }
    }

    public int reverseInt(int param1) {
        int reverse = 0;
        while (param1 != 0) {
            int remainder = param1 % 10;
            reverse = reverse * 10 + remainder;
            param1 = param1 / 10;
        }
        return reverse;
    }
}
