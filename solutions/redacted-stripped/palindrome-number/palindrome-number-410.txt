class Solution {

    public boolean method1(int x) {
        int num = x;
        int sum = 0;
        if (x < 0) {
            return false;
        }
        while (num > 0) {
            int rem = num % 10;
            sum = sum * 10 + rem;
            num /= 10;
        }
        return x == sum ? true : false;
    }
}
