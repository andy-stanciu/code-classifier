class Solution {

    public boolean method1(int x) {
        int unit = 0;
        int temp = x;
        while (temp > 0) {
            int rem = temp % 10;
            unit = (unit * 10) + rem;
            temp = temp / 10;
        }
        if (x == unit)
            return true;
        else
            return false;
    }
}
