class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        String s = String.valueOf(x);
        return palin(s, 0, s.length() - 1);
    }

    public static boolean palin(String s, int start, int end) {
        if (start >= end) {
            return true;
        }
        return (s.charAt(start) == s.charAt(end) && palin(s, start + 1, end - 1));
    }
}
