class Solution {

    public ListNode method1(ListNode head) {
        ListNode ans = recursive(head);
        return ans;
    }

    static ListNode recursive(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode newhead = recursive(head.next);
        head.next.next = head;
        head.next = null;
        return newhead;
    }
}
