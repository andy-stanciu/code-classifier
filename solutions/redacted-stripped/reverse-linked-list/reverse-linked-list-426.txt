class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode res = new ListNode(head.val);
        head = head.next;
        while (head != null) {
            ListNode newNode = new ListNode(head.val);
            head = head.next;
            newNode.next = res;
            res = newNode;
        }
        return res;
    }
}
