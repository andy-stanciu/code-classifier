class Solution {

    public ListNode method1(ListNode head) {
        ArrayList<Integer> var1 = new ArrayList<>();
        while (head != null) {
            var1.add(head.val);
            head = head.next;
        }
        ListNode tail = convert(var1);
        return tail;
    }

    public static ListNode convert(ArrayList<Integer> var1) {
        if (var1.isEmpty())
            return null;
        int n = var1.size();
        ListNode tail = new ListNode(var1.get(n - 1));
        ListNode mover = tail;
        for (int i = 1; i < n; i++) {
            ListNode temp = new ListNode(var1.get(n - 1 - i));
            mover.next = temp;
            mover = temp;
        }
        return tail;
    }
}
