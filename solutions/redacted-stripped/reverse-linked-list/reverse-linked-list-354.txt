class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode curr = head.next;
        head.next = null;
        while (curr.next != null) {
            ListNode post = curr.next;
            curr.next = head;
            head = curr;
            curr = post;
        }
        curr.next = head;
        return curr;
    }
}
