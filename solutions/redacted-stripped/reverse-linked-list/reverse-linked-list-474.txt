class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        ListNode previous = null;
        while (head != null) {
            ListNode next = head.next;
            dummy.next = head;
            head.next = previous;
            previous = head;
            head = next;
        }
        return dummy.next;
    }
}
