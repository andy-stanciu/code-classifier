class Solution {

    ListNode init = null;

    public void method1(ListNode node, ListNode prev, ListNode next) {
        if (node == null) {
            return;
        }
        if (node.next == null) {
            init = node;
        }
        node.next = prev;
        if (next != null) {
            method1(next, node, next.next);
        }
        return;
    }

    public ListNode method2(ListNode head) {
        if ((head == null) || (head.next == null)) {
            return head;
        }
        ListNode next = head.next;
        ListNode curr = head;
        ListNode prev = null;
        method1(curr, prev, next);
        return init;
    }
}
