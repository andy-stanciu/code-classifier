class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode result = null;
        ListNode tem;
        while (head != null) {
            tem = new ListNode(head.val, result);
            head = head.next;
            result = tem;
        }
        return result;
    }
}
