class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        } else {
            return method2(null, head);
        }
    }

    public ListNode method2(ListNode prev, ListNode next) {
        if (next == null) {
            return prev;
        } else {
            ListNode node = next.next;
            next.next = prev;
            return method2(next, node);
        }
    }
}
