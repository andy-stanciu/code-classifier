class Solution {

    ListNode var1;

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        method2(head);
        head.next = null;
        return var1;
    }

    private ListNode method2(ListNode head) {
        if (head.next == null) {
            var1 = head;
            return head;
        }
        ListNode var2 = method2(head.next);
        var2.next = head;
        return head;
    }
}
