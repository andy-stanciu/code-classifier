class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode ptr1 = null;
        ListNode ptr2 = head;
        while (ptr2.next != null) {
            ListNode temp = ptr2.next;
            ptr2.next = ptr1;
            ptr1 = ptr2;
            ptr2 = temp;
        }
        ptr2.next = ptr1;
        head = ptr2;
        return head;
    }
}
