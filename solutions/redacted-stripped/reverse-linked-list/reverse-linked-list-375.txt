class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode currnode = head;
        ListNode prevnode = null;
        while (currnode != null) {
            ListNode nextnode = currnode.next;
            currnode.next = prevnode;
            prevnode = currnode;
            currnode = nextnode;
        }
        head.next = null;
        return prevnode;
    }
}
