class Solution {

    ListNode newHead;

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        method2(head);
        head.next = null;
        return newHead;
    }

    public ListNode method2(ListNode head) {
        if (head.next == null)
            return newHead = head;
        ListNode nextNode = method2(head.next);
        nextNode.next = head;
        return head;
    }
}
