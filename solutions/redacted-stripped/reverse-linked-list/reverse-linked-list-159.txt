class Solution {

    ListNode method1(ListNode curr, ListNode prev, ListNode forw) {
        curr.next = prev;
        if (forw == null)
            return curr;
        return (method1(forw, curr, forw.next));
    }

    public ListNode method2(ListNode head) {
        ListNode curr, prev, forw;
        if (head == null || head.next == null)
            return (head);
        prev = null;
        curr = head;
        forw = curr.next;
        return (method1(curr, prev, forw));
    }
}
