class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        Stack<Integer> stack = new Stack<>();
        ListNode var1 = new ListNode(0);
        ListNode temp = head;
        while (temp != null) {
            stack.push(temp.val);
            temp = temp.next;
        }
        temp = var1;
        while (!stack.isEmpty()) {
            temp.next = new ListNode(stack.pop());
            temp = temp.next;
        }
        return var1.next;
    }
}
