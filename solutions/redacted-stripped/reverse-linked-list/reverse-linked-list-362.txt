class Solution {

    public ListNode method1(ListNode head) {
        Stack<Integer> s = new Stack<>();
        ListNode n = head;
        while (n != null) {
            s.push(n.val);
            n = n.next;
        }
        n = head;
        while (n != null) {
            n.val = s.peek();
            s.pop();
            n = n.next;
        }
        return head;
    }
}
