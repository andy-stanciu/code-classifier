class Solution {

    public ListNode method1(ListNode head) {
        ListNode currNode = head;
        if (head == null)
            return head;
        ListNode nextNode = head.next;
        currNode.next = null;
        while (nextNode != null) {
            ListNode lastNode = currNode;
            currNode = nextNode;
            nextNode = currNode.next;
            currNode.next = lastNode;
        }
        return currNode;
    }
}
