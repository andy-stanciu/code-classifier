class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode newHead = rev(head, null);
        head.next = null;
        return newHead;
    }

    public ListNode rev(ListNode head, ListNode tail) {
        if (head.next == null)
            return head;
        tail = rev(head.next, null);
        head.next.next = head;
        return tail;
    }
}
