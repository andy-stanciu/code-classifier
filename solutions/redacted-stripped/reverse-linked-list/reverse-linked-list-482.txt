class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode prenode = null;
        ListNode currentnode = head;
        while (currentnode != null) {
            ListNode nextnode = currentnode.next;
            currentnode.next = prenode;
            prenode = currentnode;
            currentnode = nextnode;
        }
        head = prenode;
        return head;
    }
}
