class Solution {

    public static ListNode newhead = null;

    public ListNode method1(ListNode head) {
        ListNode tem = head;
        newhead = null;
        while (tem != null) {
            ListNode send = new ListNode(tem.val);
            insert(send);
            tem = tem.next;
        }
        return newhead;
    }

    public static void insert(ListNode newnode) {
        if (newhead == null) {
            newhead = newnode;
        } else {
            newnode.next = newhead;
            newhead = newnode;
        }
    }
}
