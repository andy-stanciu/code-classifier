class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        ListNode prev = null;
        ListNode next = null;
        if (head == null) {
            return head;
        }
        while (temp != null) {
            next = temp.next;
            temp.next = prev;
            prev = temp;
            temp = next;
        }
        return prev;
    }
}
