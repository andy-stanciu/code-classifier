class Solution {

    public ListNode method1(ListNode head) {
        return method2(head, null);
    }

    private ListNode method2(ListNode cur, ListNode pre) {
        if (cur == null) {
            return pre;
        }
        ListNode temp = cur.next;
        cur.next = pre;
        pre = cur;
        cur = temp;
        return method2(cur, pre);
    }
}
