class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode revHead = method1(head.next);
        ListNode temp = head.next;
        temp.next = head;
        head.next = null;
        return revHead;
    }
}

class Solution {

    public ListNode method1(ListNode head) {
        ListNode prev = null;
        ListNode curr = head;
        ListNode forw = null;
        while (curr != null) {
            forw = curr.next;
            curr.next = prev;
            prev = curr;
            curr = forw;
        }
        return prev;
    }
}
