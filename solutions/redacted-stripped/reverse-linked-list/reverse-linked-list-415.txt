class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode previous = null;
        ListNode present = head;
        ListNode next = present.next;
        while (present != null) {
            present.next = previous;
            previous = present;
            present = next;
            if (next != null) {
                next = next.next;
            }
        }
        head = previous;
        return head;
    }
}
