class Solution {

    public ListNode method1(ListNode head) {
        ListNode tempHead = new ListNode();
        ListNode temp = new ListNode();
        ListNode h = head;
        if (head == null)
            return null;
        while (head != null) {
            temp = head.next;
            head.next = tempHead;
            tempHead = head;
            head = temp;
        }
        if (h != null)
            h.next = null;
        return tempHead;
    }
}
