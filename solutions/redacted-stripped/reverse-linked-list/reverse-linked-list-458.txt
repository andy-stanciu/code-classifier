class Solution {

    public ListNode method1(ListNode head) {
        ListNode p = head;
        ListNode q = new ListNode();
        q.next = null;
        while (p != null) {
            ListNode tmp = new ListNode();
            tmp.val = p.val;
            tmp.next = q.next;
            q.next = tmp;
            p = p.next;
        }
        return q.next;
    }
}
