class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode node = head;
        ListNode temp = node.next;
        ListNode temp2 = temp.next;
        while (temp != null) {
            if (node == head)
                node.next = null;
            temp.next = node;
            node = temp;
            temp = temp2;
            temp2 = temp2 == null ? temp2 : temp2.next;
        }
        return node;
    }
}
