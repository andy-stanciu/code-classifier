class Solution {

    public ListNode method1(ListNode head) {
        return recursive(head);
    }

    public ListNode recursive(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode newHead = recursive(head.next);
        head.next.next = head;
        head.next = null;
        return newHead;
    }
}
