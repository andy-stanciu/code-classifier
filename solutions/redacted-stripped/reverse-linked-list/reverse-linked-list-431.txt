class Solution {

    ListNode front;

    ListNode rev_helper(ListNode tmp) {
        if (tmp.next == null) {
            front = tmp;
            return tmp;
        }
        ListNode r = rev_helper(tmp.next);
        r.next = tmp;
        return tmp;
    }

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode f = rev_helper(head);
        f.next = null;
        head = front;
        return head;
    }
}
