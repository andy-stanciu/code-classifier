class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode curr = new ListNode();
        ListNode prev = new ListNode();
        curr = head;
        prev = null;
        while (curr != null) {
            ListNode temp = new ListNode();
            temp = curr;
            curr = curr.next;
            temp.next = prev;
            prev = temp;
        }
        return prev;
    }
}
