class Solution {

    public static ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode node = head, prev = null, var1 = null;
        while (node != null) {
            final ListNode next = node.next;
            if (node.next == null) {
                var1 = node;
            }
            node.next = prev;
            prev = node;
            node = next;
        }
        return var1;
    }
}
