class Solution {

    ListNode tail = null;

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        method2(head);
        head.next = null;
        return tail;
    }

    public ListNode method2(ListNode head) {
        if (head == null)
            return head;
        ListNode node = method2(head.next);
        if (node == null)
            tail = head;
        else
            node.next = head;
        return head;
    }
}
