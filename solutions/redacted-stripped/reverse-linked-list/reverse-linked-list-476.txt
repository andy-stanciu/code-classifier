class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode pre = head;
        ListNode curr = head.next;
        while (curr != null) {
            ListNode nextnext = curr.next;
            curr.next = pre;
            pre = curr;
            curr = nextnext;
        }
        head.next = null;
        head = pre;
        return head;
    }
}
