class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode consequentNode = head.next;
        ListNode veryLastNode = method1(consequentNode);
        consequentNode.next = head;
        head.next = null;
        return veryLastNode;
    }
}
