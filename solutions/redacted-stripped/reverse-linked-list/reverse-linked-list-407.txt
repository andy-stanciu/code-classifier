class Solution {

    ListNode temp;

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
        }
        temp = head;
        method2(slow);
        return head;
    }

    public void method2(ListNode mid) {
        if (mid != null && mid.next != null)
            method2(mid.next);
        int val = mid.val;
        mid.val = temp.val;
        temp.val = val;
        temp = temp.next;
    }
}
