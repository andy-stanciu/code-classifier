class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode p = null, c = head, n = head.next;
        while (c != null) {
            c.next = p;
            p = c;
            c = n;
            if (n != null) {
                n = n.next;
            }
        }
        return p;
    }
}
