class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        ListNode temp = null;
        ListNode temp2 = null;
        while (curr != null) {
            temp = new ListNode(curr.val);
            temp.next = temp2;
            temp2 = temp;
            curr = curr.next;
        }
        return temp2;
    }
}
