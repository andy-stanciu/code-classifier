class Solution {

    public ListNode method1(ListNode head) {
        ListNode previousNode = null;
        ListNode traverseNode = head;
        while (traverseNode != null) {
            ListNode next = traverseNode.next;
            traverseNode.next = previousNode;
            previousNode = traverseNode;
            traverseNode = next;
        }
        return previousNode;
    }
}
