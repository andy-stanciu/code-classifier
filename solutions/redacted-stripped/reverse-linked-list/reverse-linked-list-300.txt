class Solution {

    public ListNode recc(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode res = recc(head.next);
        head.next.next = head;
        head.next = null;
        return res;
    }

    public ListNode iter(ListNode head) {
        ListNode curr = head;
        ListNode prev = null;
        while (curr != null) {
            ListNode nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
        }
        return prev;
    }

    public ListNode method1(ListNode head) {
        return recc(head);
    }
}
