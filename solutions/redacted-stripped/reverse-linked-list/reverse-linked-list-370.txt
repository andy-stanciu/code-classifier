class Solution {

    public ListNode method1(ListNode head) {
        Stack<Integer> myStack = new Stack<>();
        ListNode temp = head;
        if (head == null) {
            return head;
        }
        while (temp != null) {
            myStack.push(temp.val);
            temp = temp.next;
        }
        ListNode lf = new ListNode((myStack.peek()));
        ListNode kp = lf;
        ListNode tr = head;
        tr.val = myStack.peek();
        myStack.pop();
        while (tr != null && !myStack.isEmpty()) {
            tr = tr.next;
            tr.val = myStack.pop();
        }
        return head;
    }
}
