class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = null;
        if (head == null) {
            return null;
        }
        ListNode after = head.next;
        while (head.next != null) {
            head.next = dummy;
            dummy = head;
            head = after;
            after = after.next;
        }
        head.next = dummy;
        return head;
    }
}
