class Solution {

    public ListNode method1(ListNode head) {
        Stack<Integer> st = new Stack<>();
        ListNode ptr = head;
        while (ptr != null) {
            st.push(ptr.val);
            ptr = ptr.next;
        }
        ListNode dummy = new ListNode();
        ListNode h = dummy;
        ListNode result = dummy;
        while (!st.isEmpty()) {
            int temp = st.pop();
            ListNode node = new ListNode();
            h.next = node;
            node.val = temp;
            node.next = null;
            h = h.next;
        }
        return result.next;
    }
}
