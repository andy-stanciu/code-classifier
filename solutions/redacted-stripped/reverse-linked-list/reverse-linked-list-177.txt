class Solution {

    public ListNode method1(ListNode head) {
        ListNode prev = null;
        ListNode curr = head;
        ListNode temp;
        if (head == null) {
            return head;
        }
        while (curr != null) {
            temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
        }
        return prev;
    }
}
