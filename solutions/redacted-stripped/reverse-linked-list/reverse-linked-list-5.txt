class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode before = null;
        ListNode current = head;
        ListNode after = current.next;
        while (current != null) {
            current.next = before;
            before = current;
            current = after;
            if (after != null) {
                after = after.next;
            }
        }
        return before;
    }
}
