class Class1 {

    int val;

    Class1 next;

    Class1(int val) {
        this.val = val;
    }
}

class Solution {

    public Class1 method1(Class1 head) {
        Class1 prev = null;
        while (head != null) {
            Class1 nextNode = head.next;
            head.next = prev;
            prev = head;
            head = nextNode;
        }
        return prev;
    }
}
