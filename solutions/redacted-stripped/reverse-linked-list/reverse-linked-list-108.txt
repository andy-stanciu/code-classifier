class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp1 = null;
        ListNode current = head;
        ListNode temp2 = null;
        while (current != null) {
            temp2 = current.next;
            current.next = temp1;
            temp1 = current;
            current = temp2;
        }
        current = temp1;
        return current;
    }
}
