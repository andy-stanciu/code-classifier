class Solution {

    public ListNode method1(ListNode head) {
        ArrayList<Integer> arr = new ArrayList<>();
        if (head == null || head.next == null) {
            return head;
        }
        ListNode current = head;
        while (current != null) {
            arr.add(current.val);
            current = current.next;
        }
        current = head;
        int i = arr.size() - 1;
        while (i >= 0) {
            current.val = arr.get(i);
            current = current.next;
            i--;
        }
        return head;
    }
}
