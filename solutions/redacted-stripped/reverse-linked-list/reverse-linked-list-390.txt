class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode[] arr = new ListNode[5000];
        int cnt = 0;
        ListNode cur = head;
        while (cur != null) {
            arr[cnt] = cur;
            cnt++;
            cur = cur.next;
        }
        ListNode result = null;
        for (int i = cnt - 1; i >= 0; i--) {
            if (result == null) {
                cur = new ListNode(arr[i].val);
                result = cur;
            } else {
                cur.next = new ListNode(arr[i].val);
                cur = cur.next;
            }
        }
        return result;
    }
}
