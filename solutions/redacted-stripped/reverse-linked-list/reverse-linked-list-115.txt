class Solution {

    ListNode newHead;

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        method2(null, head);
        return newHead;
    }

    public void method2(ListNode prev, ListNode head) {
        if (head == null)
            return;
        if (head.next == null) {
            newHead = head;
        }
        ListNode next = head.next;
        head.next = prev;
        method2(head, next);
    }
}
