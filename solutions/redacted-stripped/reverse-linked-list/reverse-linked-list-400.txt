class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        ListNode prev = null;
        ListNode forward = head;
        if (curr == null)
            return null;
        while (curr != null) {
            forward = curr.next;
            curr.next = prev;
            prev = curr;
            curr = forward;
        }
        return prev;
    }
}
