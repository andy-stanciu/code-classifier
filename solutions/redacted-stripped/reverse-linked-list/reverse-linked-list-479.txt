class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode prev = null;
        ListNode curr = head;
        ListNode later = head.next;
        while (later != null) {
            curr.next = prev;
            prev = curr;
            curr = later;
            later = curr.next;
        }
        curr.next = prev;
        return curr;
    }
}
