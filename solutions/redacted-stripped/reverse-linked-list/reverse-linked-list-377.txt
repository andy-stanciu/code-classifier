class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode previous = head;
        ListNode current = head.next;
        while (current != null) {
            previous.next = current.next;
            current.next = head;
            head = current;
            current = previous.next;
        }
        return head;
    }
}
