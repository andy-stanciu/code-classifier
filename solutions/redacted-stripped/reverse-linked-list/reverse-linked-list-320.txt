class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        check(head);
        return temp;
    }

    ListNode ans = new ListNode();

    ListNode temp;

    public void check(ListNode head) {
        if (head.next != null) {
            check(head.next);
            ans.next = new ListNode(head.val);
            ans = ans.next;
        } else if (head.next == null) {
            ans.val = head.val;
            temp = ans;
        }
    }
}
