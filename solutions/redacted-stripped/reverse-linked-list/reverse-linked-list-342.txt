class Solution {

    public ListNode method1(ListNode head) {
        ListNode curent = head;
        ListNode prev = null;
        ListNode agla = null;
        while (curent != null) {
            agla = curent.next;
            curent.next = prev;
            prev = curent;
            curent = agla;
        }
        return prev;
    }
}
