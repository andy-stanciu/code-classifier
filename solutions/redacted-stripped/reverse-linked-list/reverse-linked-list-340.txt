class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode prev = head, temp = head.next, newHead = head;
        while (temp.next != prev) {
            while (temp.next != null && prev.next != temp.next.next) {
                prev = temp;
                temp = temp.next;
            }
            if (temp.next == null) {
                newHead = temp;
            }
            temp.next = prev;
            prev = head;
            temp = head.next;
        }
        head.next = null;
        ListNode print = newHead;
        while (print != null) {
            print = print.next;
        }
        return newHead;
    }
}
