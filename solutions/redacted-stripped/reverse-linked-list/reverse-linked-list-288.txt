class Solution {

    public ListNode method1(ListNode head) {
        return method1(head, null);
    }

    private ListNode method1(ListNode head, ListNode prev) {
        if (head != null) {
            ListNode next = head.next;
            head.next = prev;
            prev = next;
            prev = method1(next, head);
        }
        return prev;
    }
}
