class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode pre = head, temp = head.next, tempn;
        while (temp != null) {
            tempn = temp.next;
            temp.next = pre;
            pre = temp;
            temp = tempn;
        }
        head.next = null;
        return pre;
    }
}
