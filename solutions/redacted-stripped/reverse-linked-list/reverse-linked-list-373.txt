class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode res = head.next, prev = head;
        head = head.next;
        head = head.next;
        res.next = prev;
        prev.next = null;
        while (head != null) {
            prev = res;
            res = head;
            head = head.next;
            res.next = prev;
        }
        return res;
    }
}
