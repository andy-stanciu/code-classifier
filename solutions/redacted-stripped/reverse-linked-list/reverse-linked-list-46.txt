class Solution {

    public ListNode method1(ListNode head) {
        ListNode currentNode = head;
        ListNode var1 = new ListNode(0);
        if (head == null) {
            return null;
        }
        while (currentNode != null) {
            ListNode var2 = new ListNode();
            if (currentNode == head) {
                var2.next = null;
            } else {
                var2.next = var1;
            }
            var2.val = currentNode.val;
            var1 = var2;
            currentNode = currentNode.next;
        }
        return var1;
    }
}
