class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = null;
        ListNode temp2 = head.next;
        head.next = temp;
        temp = head;
        head = temp2;
        temp2 = temp2.next;
        while (temp2 != null) {
            head.next = temp;
            temp = head;
            head = temp2;
            temp2 = temp2.next;
        }
        head.next = temp;
        return head;
    }
}
