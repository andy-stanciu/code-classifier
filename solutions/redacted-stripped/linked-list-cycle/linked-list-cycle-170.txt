public class Solution {

    public boolean method1(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        boolean ans = false;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                ans = true;
                break;
            }
        }
        return ans;
    }
}
