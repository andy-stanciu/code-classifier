public class Solution {

    public boolean method1(ListNode head) {
        ListNode sp = head;
        ListNode fp = head;
        if (head == null) {
            return false;
        }
        while (fp.next != null && fp.next.next != null) {
            sp = sp.next;
            fp = fp.next.next;
            if (fp == sp) {
                return true;
            }
        }
        return false;
    }
}
