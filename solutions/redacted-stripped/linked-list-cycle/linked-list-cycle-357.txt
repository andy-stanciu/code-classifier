public class Solution {

    public boolean method1(ListNode head) {
        HashSet<ListNode> set = new HashSet<>();
        ListNode temp = head;
        while (temp != null) {
            ListNode presentNode = temp;
            if (set.contains(presentNode)) {
                return true;
            }
            set.add(presentNode);
            temp = temp.next;
        }
        return false;
    }
}
