public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        ListNode p = head;
        ListNode t = head.next;
        while (t != p) {
            if (t == null || t.next == null) {
                return false;
            }
            p = p.next;
            t = t.next.next;
        }
        return true;
    }
}
