public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        ListNode sp = head;
        ListNode fp = sp.next;
        while (sp != null && fp != null && sp != fp) {
            sp = sp.next;
            if (fp.next == null)
                return false;
            else
                fp = fp.next.next;
        }
        if (sp == null || fp == null) {
            return false;
        } else
            return true;
    }
}
