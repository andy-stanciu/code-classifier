public class Solution {

    public boolean method1(ListNode head) {
        if (head == null) {
            return false;
        }
        ListNode t = head;
        while (t.next != null) {
            t.val = Integer.MAX_VALUE;
            t = t.next;
            if (t.val == Integer.MAX_VALUE) {
                return true;
            }
        }
        return false;
    }
}
