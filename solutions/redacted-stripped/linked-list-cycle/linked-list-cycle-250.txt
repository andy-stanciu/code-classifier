public class Solution {

    public boolean method1(ListNode head) {
        ListNode n1 = head;
        ListNode n2 = head;
        if (head == null) {
            return false;
        }
        do {
            if (n1.next == null || n2.next == null || n2.next.next == null) {
                return false;
            }
            n1 = n1.next;
            n2 = n2.next.next;
        } while (n1 != n2);
        return true;
    }
}
