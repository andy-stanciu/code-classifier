public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return false;
        }
        ListNode ptr1 = head;
        ListNode ptr2 = head;
        while (ptr2.next != null) {
            ptr1 = ptr1.next;
            ptr2 = ptr2.next.next;
            if (ptr1 == ptr2) {
                return true;
            }
            if (ptr2 == null) {
                return false;
            }
        }
        return false;
    }
}
