public class Solution {

    public boolean method1(ListNode head) {
        ListNode a = head, b = head;
        while (a != null && a.next != null) {
            a = a.next.next;
            b = b.next;
            if (a == b && a.val == b.val)
                break;
        }
        return a != null && a.next != null;
    }
}
