public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode p1 = head, p2 = head;
        while (p1 != null && p2 != null) {
            p1 = p1.next;
            p2 = p2.next != null ? p2.next.next : null;
            if (p1 == p2)
                return true;
        }
        return false;
    }
}
