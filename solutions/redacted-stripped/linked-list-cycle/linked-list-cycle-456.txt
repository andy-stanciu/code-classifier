public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode fast = head;
        while (head != null && fast != null && fast.next != null) {
            fast = fast.next.next;
            head = head.next;
            if (head == fast) {
                return true;
            }
        }
        return false;
    }
}
