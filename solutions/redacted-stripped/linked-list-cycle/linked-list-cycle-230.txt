public class Solution {

    public boolean method1(ListNode head) {
        ListNode temp1 = head, temp2 = head;
        while (temp2 != null && temp2.next != null) {
            temp1 = temp1.next;
            temp2 = temp2.next.next;
            if (temp1 == temp2)
                return true;
        }
        return false;
    }
}
