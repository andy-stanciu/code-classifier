public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        ListNode two = head;
        ListNode one = head;
        while (two != null && two.next != null) {
            two = two.next.next;
            one = one.next;
            if (two == one) {
                return true;
            }
        }
        return false;
    }
}
