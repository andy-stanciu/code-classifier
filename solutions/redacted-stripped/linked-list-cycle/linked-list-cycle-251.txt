public class Solution {

    public boolean method1(ListNode head) {
        Set<ListNode> nodeSet = new HashSet<>();
        ListNode curr = head;
        while (curr != null) {
            if (nodeSet.contains(curr)) {
                return true;
            } else {
                nodeSet.add(curr);
            }
            curr = curr.next;
        }
        return false;
    }
}
