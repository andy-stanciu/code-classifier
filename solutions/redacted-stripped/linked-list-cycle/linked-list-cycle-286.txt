public class Solution {

    public boolean method1(ListNode head) {
        if (head == null) {
            return false;
        }
        ListNode hair = head;
        ListNode turtle = head;
        while (hair != null && hair.next != null) {
            hair = hair.next.next;
            turtle = turtle.next;
            if (turtle == hair) {
                return true;
            }
        }
        return false;
    }
}
