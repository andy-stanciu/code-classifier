public class Solution {

    public boolean method1(ListNode head) {
        ListNode slower = head;
        ListNode faster = head;
        while (slower != null && faster != null && faster.next != null) {
            slower = slower.next;
            faster = faster.next.next;
            if (slower == faster) {
                return true;
            }
        }
        return false;
    }
}
