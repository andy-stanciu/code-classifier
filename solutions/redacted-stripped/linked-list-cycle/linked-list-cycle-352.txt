public class Solution {

    public boolean method1(ListNode slow, ListNode fast, ListNode head) {
        if (head == null || fast == null || fast.next == null) {
            return false;
        }
        if (slow == fast) {
            return true;
        }
        return method1(slow.next, fast.next.next, head);
    }

    public boolean method2(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        return method1(head, head.next, head.next.next);
    }
}
