public class Solution {

    public boolean method1(ListNode head) {
        ListNode slowPntr;
        ListNode fastPntr;
        slowPntr = head;
        fastPntr = head;
        while (slowPntr != null && fastPntr != null && fastPntr.next != null) {
            slowPntr = slowPntr.next;
            fastPntr = fastPntr.next.next;
            if (slowPntr == fastPntr) {
                return true;
            }
        }
        return false;
    }
}
