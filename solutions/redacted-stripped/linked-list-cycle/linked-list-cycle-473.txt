public class Solution {

    public boolean method1(ListNode head) {
        int MAX = 100001;
        if (head == null) {
            return false;
        }
        while (head.next != null) {
            if (head.val == MAX) {
                return true;
            }
            head.val = MAX;
            head = head.next;
        }
        return false;
    }
}
