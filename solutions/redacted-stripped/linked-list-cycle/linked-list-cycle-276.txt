public class Solution {

    public boolean method1(ListNode head) {
        HashMap<ListNode, Integer> map = new HashMap<ListNode, Integer>();
        ListNode temp = head;
        int pos = 0;
        while (!map.containsKey(temp) && temp != null) {
            map.put(temp, pos);
            pos++;
            temp = temp.next;
        }
        if (map.containsKey(temp))
            return true;
        return false;
    }
}
