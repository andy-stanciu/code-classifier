public class Solution {

    public boolean method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        if (head == null) {
            return false;
        }
        if (slow.next == null) {
            return false;
        }
        while (fast.next != null && fast.next.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow.next == fast.next) {
                return true;
            }
        }
        return false;
    }
}
