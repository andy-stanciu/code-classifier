public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode sl, ft;
        sl = head;
        ft = head.next;
        while (ft != null && ft.next != null) {
            sl = sl.next;
            ft = ft.next.next;
            if (sl == ft) {
                return true;
            }
        }
        return false;
    }
}
