public class Solution {

    public boolean method1(ListNode head) {
        ListNode Fast = head;
        ListNode Slow = head;
        while (Fast != null && Fast.next != null) {
            Fast = Fast.next.next;
            Slow = Slow.next;
            if (Fast == Slow) {
                return true;
            }
        }
        return false;
    }
}
