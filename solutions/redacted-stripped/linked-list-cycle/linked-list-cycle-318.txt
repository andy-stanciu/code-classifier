public class Solution {

    public boolean method1(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        if (head == null)
            return false;
        do {
            if (fast == null || fast.next == null)
                return false;
            slow = slow.next;
            fast = fast.next.next;
        } while (fast != slow);
        return true;
    }
}
