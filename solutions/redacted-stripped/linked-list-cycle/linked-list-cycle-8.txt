public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode slow = head, fast = head.next;
        while (true) {
            if (fast.next == null || fast.next.next == null)
                return false;
            else {
                if (slow == fast)
                    return true;
                slow = slow.next;
                fast = fast.next.next;
            }
        }
    }
}
