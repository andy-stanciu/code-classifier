public class Solution {

    public boolean method1(ListNode head) {
        ListNode fast = head, slow = head;
        boolean flg = false;
        while (fast != null && fast.next != null && slow != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                return true;
            }
        }
        return false;
    }
}
