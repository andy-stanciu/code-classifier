public class Solution {

    public boolean method1(ListNode head) {
        HashMap<ListNode, Integer> hash1 = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (hash1.containsKey(temp)) {
                return true;
            }
            hash1.put(temp, 1);
            temp = temp.next;
        }
        return false;
    }
}
