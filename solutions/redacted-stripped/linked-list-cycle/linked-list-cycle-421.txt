public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode temp = head;
        boolean c = false;
        while (temp.next != null) {
            if (iscyclic(temp) == true) {
                return true;
            } else {
                temp = temp.next;
            }
        }
        return c;
    }

    private boolean iscyclic(ListNode head) {
        ListNode temp = head;
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                return true;
            }
        }
        return false;
    }
}
