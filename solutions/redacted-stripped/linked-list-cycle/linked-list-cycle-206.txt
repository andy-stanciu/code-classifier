public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        HashSet<ListNode> s = new HashSet<>();
        ListNode ptr = head;
        while (ptr != null) {
            if (s.contains(ptr))
                return true;
            s.add(ptr);
            ptr = ptr.next;
        }
        return false;
    }
}
