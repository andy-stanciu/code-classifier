public class Solution {

    public boolean method1(ListNode head) {
        if (head == null) {
            return false;
        }
        ListNode tada = new ListNode(-1);
        ListNode temp = head;
        ListNode temp2;
        while (true) {
            temp2 = temp.next;
            temp.next = tada;
            temp = temp2;
            if (temp == null) {
                return false;
            }
            if (temp == tada || temp.next == tada) {
                return true;
            }
        }
    }
}
