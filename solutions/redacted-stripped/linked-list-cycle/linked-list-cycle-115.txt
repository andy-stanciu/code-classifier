public class Solution {

    public boolean method1(ListNode head) {
        HashSet<ListNode> set = new HashSet<>();
        ListNode curr = head;
        while (curr != null) {
            if (set.contains(curr)) {
                return true;
            } else {
                set.add(curr);
                curr = curr.next;
            }
        }
        return false;
    }
}
