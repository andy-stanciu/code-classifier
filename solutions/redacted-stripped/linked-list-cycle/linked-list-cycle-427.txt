public class Solution {

    public boolean method1(ListNode head) {
        if (head == null)
            return false;
        ListNode hare = head;
        ListNode tortoise = head;
        while (hare.next != null && hare.next.next != null) {
            tortoise = tortoise.next;
            hare = hare.next.next;
            if (hare == tortoise)
                return true;
        }
        return false;
    }
}
