class Solution {

    public int strStr(String haystack, String needle) {
        if (needle.isEmpty()) {
            return 0;
        }
        int hLength = haystack.length();
        int nLength = needle.length();
        for (int i = 0; i <= hLength - nLength; i++) {
            int j;
            for (j = 0; j < nLength; j++) {
                if (haystack.charAt(i + j) != needle.charAt(j)) {
                    break;
                }
            }
            if (j == nLength) {
                return i;
            }
        }
        return -1;
    }
}
