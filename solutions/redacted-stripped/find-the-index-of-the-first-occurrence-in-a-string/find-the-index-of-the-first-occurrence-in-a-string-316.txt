class Solution {

    public int strStr(String haystack, String needle) {
        int result = -1;
        int var1 = 0;
        int right = needle.length();
        while (right <= haystack.length()) {
            if (haystack.substring(var1, right).equals(needle)) {
                result = var1;
                break;
            } else {
                var1++;
                right++;
            }
        }
        return result;
    }
}
