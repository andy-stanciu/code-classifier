class Solution {

    public int strStr(String haystack, String needle) {
        int var1 = 0;
        String var2 = haystack;
        for (int i = 0; i < haystack.length(); i++) {
            if (var2.startsWith(needle)) {
                return var1;
            } else {
                var2 = var2.substring(1);
                var1++;
            }
        }
        return var1 == haystack.length() ? -1 : var1;
    }
}
