class Solution {

    public int strStr(String haystack, String needle) {
        var m = haystack.length();
        var n = needle.length();
        for (var i = 0; i < m; i++) {
            var j = 0;
            while (i + j < m && j < n && needle.charAt(j) == haystack.charAt(i + j)) j++;
            if (j == n)
                return i;
        }
        return -1;
    }
}
