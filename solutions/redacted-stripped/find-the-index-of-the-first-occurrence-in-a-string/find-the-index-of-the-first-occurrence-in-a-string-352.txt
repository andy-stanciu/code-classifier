class Solution {

    boolean comp(String haystack, String needle, int param1) {
        for (int j = 0; j < needle.length(); j++) {
            if (param1 >= haystack.length()) {
                return false;
            }
            if (haystack.charAt(param1++) != needle.charAt(j)) {
                return false;
            }
        }
        return true;
    }

    public int strStr(String haystack, String needle) {
        for (int i = 0; i < haystack.length(); i++) {
            if (haystack.charAt(i) == needle.charAt(0)) {
                if (comp(haystack, needle, i) == true) {
                    return i;
                }
            }
        }
        return -1;
    }
}
