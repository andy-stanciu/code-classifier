class Solution {

    public int strStr(String haystack, String needle) {
        int f = 0;
        char[] haystackchar = haystack.toCharArray();
        char[] needlechar = needle.toCharArray();
        if (needlechar.length > haystackchar.length) {
            return -1;
        }
        for (int i = 0; i < haystackchar.length; i++) {
            if (needlechar[0] != haystackchar[i]) {
            } else {
                f = i;
                for (int j = 0; j < needlechar.length; j++) {
                    if (i + j >= haystackchar.length) {
                        return -1;
                    }
                    if (needlechar[j] != haystackchar[i + j]) {
                        f = 0;
                        break;
                    } else if (j == needlechar.length - 1) {
                        return f;
                    }
                }
            }
        }
        return -1;
    }
}
