class Solution {

    public int strStr(String haystack, String needle) {
        int i = 0, j = 0, k = 0;
        int n1 = haystack.length(), n2 = needle.length();
        while (i < n1 && j < n2) {
            if (haystack.charAt(i) == needle.charAt(j)) {
                i++;
                j++;
            } else {
                i = ++k;
                j = 0;
            }
        }
        return (j == n2) ? k : -1;
    }
}
