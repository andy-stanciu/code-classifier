class Solution {

    public int strStr(String haystack, String needle) {
        int hl = haystack.length(), nl = needle.length();
        for (int i = 0; i < hl; i++) {
            if (haystack.charAt(i) == needle.charAt(0) && (i + nl) <= hl && haystack.substring(i, i + nl).equals(needle)) {
                return i;
            }
        }
        return -1;
    }
}
