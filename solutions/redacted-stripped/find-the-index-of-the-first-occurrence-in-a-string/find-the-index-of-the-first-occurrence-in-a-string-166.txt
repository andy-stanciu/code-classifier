class Solution {

    public int strStr(String haystack, String needle) {
        if (needle == null)
            return 0;
        if (haystack == null || haystack.length() < needle.length())
            return -1;
        int x = haystack.length();
        int y = needle.length();
        int i = 0, j = 0, var1 = 0;
        while (i < x && j < y) {
            if (haystack.charAt(i) == needle.charAt(j)) {
                i++;
                j++;
            } else {
                i -= j;
                j = 0;
                var1 = ++i;
            }
        }
        if (j == y)
            return var1;
        else
            return -1;
    }
}
