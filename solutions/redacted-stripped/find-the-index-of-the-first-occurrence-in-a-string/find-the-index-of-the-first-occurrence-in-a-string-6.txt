class Solution {

    public int strStr(String haystack, String needle) {
        if (needle.isEmpty()) {
            return 0;
        }
        int haylength = haystack.length();
        int needlelength = needle.length();
        for (int i = 0; i <= haylength - needlelength; i++) {
            if (haystack.substring(i, i + needlelength).equals(needle)) {
                return i;
            }
        }
        return -1;
    }
}
