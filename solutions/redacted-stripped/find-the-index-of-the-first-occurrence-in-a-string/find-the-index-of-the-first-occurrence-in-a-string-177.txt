class Solution {

    public int strStr(String haystack, String needle) {
        int l1 = haystack.length();
        int l2 = needle.length();
        int occurence = -1;
        if (l2 > l1)
            return -1;
        for (int i = 0; i < l1 - l2 + 1; i++) {
            if (haystack.substring(i, i + l2).equals(needle)) {
                occurence = i;
                break;
            }
        }
        return occurence;
    }
}
