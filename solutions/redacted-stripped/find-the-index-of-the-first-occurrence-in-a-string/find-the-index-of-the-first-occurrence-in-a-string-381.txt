class Solution {

    public int strStr(String haystack, String needle) {
        int m = haystack.length();
        int n = needle.length();
        for (int i = 0; i <= m - n; i++) {
            int var1 = i, second = 0;
            while (second < n) {
                if (haystack.charAt(var1) != needle.charAt(second)) {
                    break;
                } else {
                    var1++;
                    second++;
                }
                if (second == needle.length())
                    return var1 - second;
            }
        }
        return -1;
    }
}
