class Solution {

    public int strStr(String haystack, String needle) {
        int len1 = haystack.length();
        int len2 = needle.length();
        int j = 0;
        for (int i = 0; i < len1; i++) {
            if (needle.charAt(j) == haystack.charAt(i))
                j++;
            else {
                i = i - j;
                j = 0;
            }
            if (j == len2)
                return i - len2 + 1;
        }
        return -1;
    }
}
