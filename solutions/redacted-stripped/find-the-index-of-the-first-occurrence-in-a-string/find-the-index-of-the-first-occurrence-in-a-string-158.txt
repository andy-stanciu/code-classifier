class Solution {

    public int strStr(String haystack, String needle) {
        if (haystack.equals(needle)) {
            return 0;
        }
        int var1 = 0;
        int var2 = needle.length();
        while (var2 <= haystack.length()) {
            if (haystack.substring(var1, var2).equals(needle)) {
                return var1;
            }
            var1++;
            var2++;
        }
        return -1;
    }
}
