class Solution {

    public int strStr(String haystack, String needle) {
        int l = haystack.length();
        int m = needle.length();
        if (l < m)
            return -1;
        for (// return -1.
        int i = 0; // return -1.
        i < l - m + 1; i++) {
            if (haystack.substring(i, m + i).equals(needle))
                return i;
        }
        return -1;
    }
}
