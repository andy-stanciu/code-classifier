class Solution {

    public static int strStr(String haystack, String needle) {
        if (needle.isEmpty())
            return -1;
        for (int i = 0; i < haystack.length(); i++) {
            if (haystack.charAt(i) == needle.charAt(0)) {
                boolean sonuc = kalan_ifade(haystack.substring(i), needle);
                if (sonuc == true) {
                    return i;
                }
            }
        }
        return -1;
    }

    public static boolean kalan_ifade(String kalan, String needle) {
        boolean flag = true;
        if (kalan.length() < needle.length())
            return false;
        for (int i = 0; i < needle.length(); i++) {
            if (kalan.charAt(i) != needle.charAt(i)) {
                flag = false;
                return flag;
            }
        }
        return true;
    }
}
