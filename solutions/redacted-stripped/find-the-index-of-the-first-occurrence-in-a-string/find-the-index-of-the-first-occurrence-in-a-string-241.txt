class Solution {

    public int strStr(String haystack, String needle) {
        int n = haystack.length();
        int m = needle.length();
        if (needle.isEmpty())
            return 0;
        for (int i = 0; i < n; ++i) {
            if (haystack.charAt(i) == needle.charAt(0)) {
                int l = i, k = 0;
                while (k < m && l < n) {
                    if (haystack.charAt(l) == needle.charAt(k)) {
                        l++;
                        k++;
                    } else {
                        break;
                    }
                }
                if (k == m)
                    return i;
            }
        }
        return -1;
    }
}
