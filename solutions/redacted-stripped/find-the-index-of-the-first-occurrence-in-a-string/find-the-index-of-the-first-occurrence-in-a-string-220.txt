class Solution {

    public int strStr(String s, String t) {
        int m = s.length(), n = t.length();
        if (n == 0)
            return 0;
        for (int i = 0, j; i <= m - n; i++) {
            if (s.charAt(i + n - 1) != t.charAt(n - 1))
                continue;
            for (j = 0; j < n - 1; j++) if (s.charAt(i + j) != t.charAt(j))
                break;
            if (j == n - 1)
                return i;
        }
        return -1;
    }
}
