class Solution {

    public int[] method1(int[] digits) {
        int i = digits.length - 1;
        do digits[i] = (digits[i] + 1) % 10; while (--i >= 0 && digits[i + 1] == 0);
        return digits[0] == 0 ? createRes(digits) : digits;
    }

    public int[] createRes(int[] digits) {
        int[] res = new int[digits.length + 1];
        res[0] = 1;
        return res;
    }
}
