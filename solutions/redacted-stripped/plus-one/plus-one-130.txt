class Solution {

    public int[] method1(int[] digits) {
        if (digits[0] == 9) {
            boolean allNines = true;
            for (int i = 1; i < digits.length; i++) {
                if (digits[i] != 9) {
                    allNines = false;
                    break;
                }
            }
            if (allNines) {
                int[] num = new int[digits.length + 1];
                Arrays.fill(num, 0);
                num[0] = 1;
                return num;
            }
        }
        int i = digits.length - 1;
        digits[i] = digits[i] + 1;
        while (digits[i] == 10) {
            digits[i] = 0;
            digits[i - 1] = digits[i - 1] + 1;
            i--;
        }
        return digits;
    }
}
