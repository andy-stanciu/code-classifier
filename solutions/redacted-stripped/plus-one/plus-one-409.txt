class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        int i = n - 1;
        int carry = 0;
        while (i >= 0) {
            if (digits[i] != 9) {
                digits[i] = digits[i] + 1;
                return digits;
            } else {
                carry = 1;
                digits[i] = 0;
                i--;
            }
        }
        int[] arr = new int[n + 1];
        arr[0] = 1;
        for (int j = 1; j < n; j++) {
            arr[j] = digits[j];
        }
        return arr;
    }
}
