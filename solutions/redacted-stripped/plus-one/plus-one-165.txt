class Solution {

    public int[] method1(int[] digits) {
        return approach1(digits);
    }

    private int[] approach1(int[] digits) {
        int carry = 1;
        ArrayList<Integer> result = new ArrayList<>();
        for (int i = digits.length - 1; i >= 0; i--) {
            int last_value = digits[i];
            result.add((last_value + carry) % 10);
            carry = (last_value + carry) / 10;
        }
        if (carry != 0)
            result.add(carry);
        Collections.reverse(result);
        int[] intArray = result.stream().mapToInt(Integer::intValue).toArray();
        return intArray;
    }
}
