class Solution {

    public int[] method1(int[] digits) {
        if (digits.length == 1 && digits[0] == 9) {
            int[] r = { 1, 0 };
            return r;
        }
        int l = digits.length;
        if (digits[l - 1] != 9) {
            digits[l - 1] = digits[l - 1] + 1;
            return digits;
        }
        ArrayList<Integer> list = new ArrayList<>();
        for (int num : digits) {
            list.add(num);
        }
        int count = 0;
        for (int j = 0; j < list.size(); j++) {
            if (list.get(j) == 9) {
                count++;
            }
        }
        if (count == list.size()) {
            Collections.fill(list, 0);
            list.add(0, 1);
            int[] result = new int[list.size()];
            for (int j = 0; j < list.size(); j++) {
                result[j] = list.get(j);
            }
            return result;
        }
        int i = l - 1;
        while (list.get(i) == 9) {
            i--;
        }
        System.out.println(i);
        int k = list.get(i);
        list.set(i, k + 1);
        for (int m = i + 1; m < list.size(); m++) {
            list.set(m, 0);
        }
        int[] result = new int[list.size()];
        for (int j = 0; j < list.size(); j++) {
            result[j] = list.get(j);
        }
        return result;
    }
}
