class Solution {

    public int[] method1(int[] digits) {
        for (int j = digits.length - 1; j >= 0; j--) {
            if (digits[j] < 9) {
                digits[j]++;
                return digits;
            }
            digits[j] = 0;
        }
        int[] number = new int[digits.length + 1];
        number[0] = 1;
        return number;
    }
}
