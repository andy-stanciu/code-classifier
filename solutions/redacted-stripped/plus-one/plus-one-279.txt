class Solution {

    public int[] method1(int[] digits) {
        int j = digits.length;
        for (int i = j - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        int[] k = new int[j + 1];
        k[0] = 1;
        return k;
    }
}
