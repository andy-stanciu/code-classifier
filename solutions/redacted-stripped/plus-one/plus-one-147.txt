class Solution {

    public int[] method1(int[] d) {
        if (d[d.length - 1] == 9) {
            int tn = 0;
            for (int i = d.length - 1; i >= 0; i--) {
                if (d[i] == 9) {
                    tn++;
                } else {
                    break;
                }
            }
            if (tn == d.length) {
                int[] ans = new int[d.length + 1];
                ans[0] = 1;
                return ans;
            } else {
                int idx = d.length - 1;
                while (tn != 0) {
                    d[idx] = 0;
                    idx--;
                    tn--;
                }
                d[idx]++;
                return d;
            }
        } else {
            d[d.length - 1]++;
            return d;
        }
    }
}
