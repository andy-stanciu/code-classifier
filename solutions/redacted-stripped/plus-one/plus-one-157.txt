class Solution {

    public int[] method1(int[] digits) {
        int carry = 1;
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i] += carry;
                carry = 0;
                return digits;
            } else {
                digits[i] = 0;
            }
        }
        int[] ans = new int[digits.length + 1];
        if (carry != 0) {
            ans[0] = 1;
            int k = 1;
            for (int i = 0; i < digits.length; i++) {
                ans[k] = 0;
                k++;
            }
        }
        return ans;
    }
}
