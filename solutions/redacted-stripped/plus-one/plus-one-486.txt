class Solution {

    public int[] method1(int[] digits) {
        int last = digits.length - 1;
        digits[last] += 1;
        for (int i = last; i > 0; i--) {
            if (digits[i] == 10) {
                digits[i] = 0;
                digits[i - 1] += 1;
            }
        }
        if (digits[0] == 10) {
            digits = new int[digits.length + 1];
            digits[0] = 1;
        }
        return digits;
    }
}
