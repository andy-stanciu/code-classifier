class Solution {

    public int[] method1(int[] digits) {
        int i = digits.length - 1;
        int summ = digits[i] + 1;
        digits[i] = summ % 10;
        int c = summ / 10;
        i--;
        while (c > 0 && i >= 0) {
            summ = digits[i] + c;
            digits[i] = summ % 10;
            c = summ / 10;
            i--;
        }
        if (c > 0) {
            int[] arr = new int[digits.length + 1];
            arr[0] = c;
            for (int j = 0; j < digits.length; j++) {
                arr[j + 1] = digits[j];
            }
            return arr;
        } else {
            return digits;
        }
    }
}
