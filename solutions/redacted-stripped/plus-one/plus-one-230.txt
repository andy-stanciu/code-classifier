class Solution {

    public int[] method1(int[] digits) {
        if (digits[digits.length - 1] != 9) {
            digits[digits.length - 1] = (digits[digits.length - 1] + 1);
            return digits;
        }
        for (int i = digits.length - 1; i > 0; i--) {
            digits[i] = 0;
            if (digits[i - 1] != 9) {
                digits[i - 1] += 1;
                return digits;
            }
        }
        int[] newArr = new int[digits.length + 1];
        newArr[0] = 1;
        for (int i = 1; i < newArr.length; i++) {
            newArr[i] = 0;
        }
        return newArr;
    }
}
