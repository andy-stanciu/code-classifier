class Solution {

    public int[] method1(int[] digit) {
        int length = digit.length;
        for (int i = length - 1; i >= 0; i--) {
            if (digit[i] < 9) {
                digit[i]++;
                return digit;
            }
            digit[i] = 0;
        }
        int[] finalArray = new int[length + 1];
        finalArray[0] = 1;
        return finalArray;
    }
}
