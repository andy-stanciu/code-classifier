class Solution {

    public int[] method1(int[] digits) {
        int nines = 0;
        for (int n : digits) if (n == 9)
            nines++;
        int[] ans;
        if (nines == digits.length)
            ans = new int[digits.length + 1];
        else
            ans = digits.clone();
        int len = digits.length - 1;
        while (len > 0 && digits[len] == 9) {
            ans[len] = 0;
            len--;
        }
        ans[len] += 1;
        return ans;
    }
}
