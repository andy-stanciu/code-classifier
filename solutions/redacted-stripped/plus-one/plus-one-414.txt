class Solution {

    public int[] method1(int[] n) {
        int length = n.length;
        for (int i = length - 1; i >= 0; i--) {
            if (n[i] < 9) {
                n[i]++;
                return n;
            }
            n[i] = 0;
        }
        int[] newNumber = new int[length + 1];
        newNumber[0] = 1;
        return newNumber;
    }
}
