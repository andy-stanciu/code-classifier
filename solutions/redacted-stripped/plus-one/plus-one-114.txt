class Solution {

    public int[] method1(int[] digits) {
        final int len = digits.length;
        for (int index = len - 1; index >= 0; --index) {
            if (digits[index] != 9) {
                digits[index] += 1;
                return digits;
            }
            digits[index] = 0;
        }
        final int[] result = new int[len + 1];
        result[0] = 1;
        return result;
    }
}
