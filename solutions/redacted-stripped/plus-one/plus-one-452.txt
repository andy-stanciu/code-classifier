class Solution {

    public int[] method1(int[] a) {
        for (int i = a.length - 1; i >= 0; i--) {
            if (a[i] < 9) {
                a[i]++;
                return a;
            }
            a[i] = 0;
        }
        int[] newarray = new int[a.length + 1];
        newarray[0] = 1;
        return newarray;
    }
}
