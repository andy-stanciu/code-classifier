class Solution {

    public int[] method1(int[] digits) {
        int len = digits.length;
        int i = len - 1;
        while (i >= 0) {
            if (digits[i] < 9) {
                digits[i] += 1;
                return digits;
            } else {
                digits[i] = 0;
            }
            i--;
        }
        if (i < 0) {
            int[] res = new int[len + 1];
            res[0] = 1;
            for (int x = 1; x <= len; x++) {
                res[x] = 0;
            }
            return res;
        }
        return digits;
    }
}
