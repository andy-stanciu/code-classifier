class Solution {

    public int[] method1(int[] digits) {
        int carry = 1;
        for (int i = digits.length - 1; i >= 0; i--) {
            int x = digits[i] + carry;
            digits[i] = x % 10;
            carry = x / 10;
        }
        if (carry == 0)
            return digits;
        int[] ans = new int[digits.length + 1];
        ans[0] = carry;
        for (int i = 1; i < ans.length; i++) {
            ans[i] = digits[i - 1];
        }
        return ans;
    }
}
