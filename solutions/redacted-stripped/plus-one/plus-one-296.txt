class Solution {

    public int[] method1(int[] digits) {
        boolean allNines = true;
        for (int i = 0; i < digits.length && allNines; i++) {
            if (digits[i] != 9) {
                allNines = false;
            }
        }
        if (allNines) {
            int[] res = new int[digits.length + 1];
            res[0] = 1;
            return res;
        } else {
            int[] res = digits;
            int index = res.length - 1;
            while (res[index] == 9) {
                res[index] = 0;
                index--;
            }
            res[index]++;
            return res;
        }
    }
}
