class Solution {

    public int[] method1(int[] arr) {
        int i = arr.length - 1;
        boolean flag = true;
        while (i >= 0) {
            if (arr[i] + 1 != 10) {
                arr[i] = arr[i] + 1;
                return arr;
            } else {
                arr[i] = 0;
            }
            i--;
        }
        int[] newArr = new int[arr.length + 1];
        newArr[0] = 1;
        return newArr;
    }
}
