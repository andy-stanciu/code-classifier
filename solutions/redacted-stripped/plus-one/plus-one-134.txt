class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            } else
                digits[i] = 0;
        }
        int[] var1 = new int[digits.length + 1];
        var1[0] = 1;
        return var1;
    }
}
