class Solution {

    public int[] method1(int[] digits) {
        if (digits[digits.length - 1] != 9) {
            digits[digits.length - 1]++;
            return digits;
        } else {
            int x, c = 1, a[] = new int[digits.length + 1], y = digits.length - 1;
            for (x = a.length - 1; x >= 0; x--) {
                if (y < digits.length && y >= 0) {
                    if (digits[y] == 9 && c == 1) {
                        c = 1;
                        a[x] = 0;
                    } else {
                        a[x] = digits[y] + c;
                        c = 0;
                    }
                    y--;
                } else {
                    a[0] = c;
                }
            }
            if (a[0] == 0) {
                for (x = 0; x < digits.length; x++) digits[x] = a[x + 1];
                return digits;
            }
            return a;
        }
    }
}
