class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] == 9) {
                digits[i] = 0;
            } else {
                digits[i]++;
                break;
            }
        }
        if (digits[0] == 0) {
            int[] ans = new int[digits.length + 1];
            ans[0] = 1;
            ans[1] = 0;
            for (int i = 2; i < digits.length; i++) {
                ans[i] = digits[i];
            }
            return ans;
        }
        return digits;
    }
}
