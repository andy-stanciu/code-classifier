class Solution {

    public int[] method1(int[] digits) {
        for (int farukIndex = digits.length - 1; farukIndex >= 0; farukIndex--) {
            digits[farukIndex]++;
            if (digits[farukIndex] < 10) {
                return digits;
            }
            digits[farukIndex] = 0;
        }
        int[] result = new int[digits.length + 1];
        result[0] = 1;
        return result;
    }
}
