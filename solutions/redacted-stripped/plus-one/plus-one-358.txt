class Solution {

    public int[] method1(int[] digits) {
        StringBuilder digitsStr = new StringBuilder();
        for (int digit : digits) {
            digitsStr.append(digit);
        }
        BigInteger digitsInt = new BigInteger(digitsStr.toString());
        digitsInt = digitsInt.add(BigInteger.ONE);
        String outStr = digitsInt.toString();
        int[] result = new int[outStr.length()];
        for (int i = 0; i < outStr.length(); i++) {
            result[i] = Character.getNumericValue(outStr.charAt(i));
        }
        return result;
    }
}
