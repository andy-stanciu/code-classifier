class Solution {

    public int[] method1(int[] digits) {
        int j = digits.length - 1;
        while (j >= 0) {
            if (digits[j] != 9) {
                digits[j] = digits[j] + 1;
                return digits;
            } else {
                digits[j] = 0;
                j--;
            }
        }
        int[] expandedDigits = new int[digits.length + 1];
        expandedDigits[0] = 1;
        return expandedDigits;
    }
}
