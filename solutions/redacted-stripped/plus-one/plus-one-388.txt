class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        if (n == 1 && digits[0] == 9)
            return (new int[] { 1, 0 });
        StringBuilder s = new StringBuilder();
        int c = 1;
        int x = 0;
        for (int i = n - 1; i >= 0; i--) {
            if (i == 0 && digits[i] + c > 9) {
                s.append(0);
                s.append(1);
                break;
            }
            if (digits[i] + c > 9) {
                s.append(0);
                c = 1;
            } else {
                x = digits[i] + c;
                s.append(x);
                c = 0;
            }
        }
        s.reverse().toString();
        int y = s.length();
        int[] a = new int[y];
        for (int i = 0; i < y; i++) {
            a[i] = Character.getNumericValue(s.charAt(i));
        }
        return a;
    }
}
