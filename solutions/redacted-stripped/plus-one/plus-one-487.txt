class Solution {

    public int[] method1(int[] digits) {
        int x = digits.length;
        for (int i = x - 1; i >= 0; i--) {
            if (digits[i] != 9) {
                digits[i]++;
                return digits;
            } else {
                digits[i] = 0;
            }
        }
        int[] edgeDigits = new int[x + 1];
        edgeDigits[0] = 1;
        for (int j = 0; j < x; j++) {
            edgeDigits[j + 1] = digits[j];
        }
        return edgeDigits;
    }
}
