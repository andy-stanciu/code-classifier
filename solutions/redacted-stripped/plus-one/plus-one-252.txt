class Solution {

    public int[] method1(int[] digits) {
        StringBuilder str = new StringBuilder();
        int i = 0;
        while (i < digits.length) {
            str.append(digits[i]);
            i++;
        }
        BigInteger bigNum = new BigInteger(str.toString());
        bigNum = bigNum.add(BigInteger.ONE);
        String resultString = bigNum.toString();
        ArrayList<Integer> digitsList = new ArrayList<>();
        for (int j = 0; j < resultString.length(); j++) {
            digitsList.add(Character.getNumericValue(resultString.charAt(j)));
        }
        int[] digitsArray = digitsList.stream().mapToInt(Integer::intValue).toArray();
        return digitsArray;
    }
}
