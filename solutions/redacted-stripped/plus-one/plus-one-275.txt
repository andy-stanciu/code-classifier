class Solution {

    public int[] method1(int[] digits) {
        int carry = 1;
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] == 9 && carry == 1) {
                digits[i] = 0;
            } else {
                digits[i] += 1;
                carry = 0;
                return digits;
            }
        }
        int[] number = new int[digits.length + 1];
        number[0] = 1;
        return number;
    }
}
