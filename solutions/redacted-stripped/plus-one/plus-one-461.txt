class Solution {

    public int[] method1(int[] digits) {
        int lastNum = 0;
        int carryOver = 0;
        int subs = 0;
        int[] outputVal = new int[digits.length];
        for (int j = digits.length - 1, i = 0; j >= 0; j--, i++) {
            if (i == 0 || carryOver == 1) {
                carryOver = 1;
            }
            if (carryOver == 0) {
                lastNum = digits[j];
            }
            if (carryOver == 1) {
                lastNum = digits[j] + 1;
                carryOver = 0;
            }
            if (lastNum == 10) {
                carryOver = 1;
                lastNum = 0;
            }
            outputVal[j] = lastNum;
        }
        if (outputVal[0] == 0) {
            outputVal = new int[digits.length + 1];
            outputVal[0] = 1;
            outputVal[1] = 0;
        }
        return outputVal;
    }
}
