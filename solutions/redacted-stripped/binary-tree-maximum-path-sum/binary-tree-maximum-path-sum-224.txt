class Solution {

    int result = 0;

    int method1(TreeNode root) {
        if (root == null)
            return 0;
        int l = method1(root.left);
        int r = method1(root.right);
        result = Math.max(result, Math.max(root.val, Math.max(root.val + l, Math.max(root.val + r, root.val + l + r))));
        return Math.max(root.val, Math.max(root.val + l, root.val + r));
    }

    public int method2(TreeNode root) {
        result = root.val;
        method1(root);
        return result;
    }
}
