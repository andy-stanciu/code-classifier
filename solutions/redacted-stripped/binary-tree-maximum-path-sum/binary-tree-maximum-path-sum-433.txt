class Solution {

    private int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = method1(root.left);
        int right = method1(root.right);
        int max = Math.max(left, right);
        return max > 0 ? root.val + max : root.val;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return Integer.MIN_VALUE;
        }
        int left = method2(root.left);
        int right = method2(root.right);
        int own = root.val;
        int ownleft = method1(root.left);
        int ownright = method1(root.right);
        if (ownleft > 0)
            own += ownleft;
        if (ownright > 0)
            own += ownright;
        return Math.max(own, Math.max(left, right));
    }
}
