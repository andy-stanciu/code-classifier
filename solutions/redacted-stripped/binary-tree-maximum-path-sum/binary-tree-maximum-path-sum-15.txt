class Solution {

    public int method1(TreeNode root) {
        int[] ans = new int[1];
        ans[0] = Integer.MIN_VALUE;
        method2(root, ans);
        return ans[0];
    }

    int method2(TreeNode node, int[] arr) {
        if (node == null) {
            return 0;
        }
        int leftNodesValues = Math.max(0, method2(node.left, arr));
        int rightNodesValues = Math.max(0, method2(node.right, arr));
        int var1 = leftNodesValues + node.val + rightNodesValues;
        arr[0] = Math.max(arr[0], var1);
        return node.val + Math.max(leftNodesValues, rightNodesValues);
    }
}
