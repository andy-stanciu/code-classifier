class Solution {

    int max = -1000;

    public int method1(TreeNode root) {
        return Math.max(method2(root), max);
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return -1000;
        }
        int l = method2(root.left);
        int r = method2(root.right);
        int t = Math.max(Math.max(l, r), root.val);
        max = Math.max(Math.max(max, l + r + root.val), t);
        return Math.max(Math.max(l, r) + root.val, root.val);
    }
}
