class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        maxGain(root);
        return var1;
    }

    public int maxGain(TreeNode root) {
        if (root == null)
            return 0;
        int var2 = Math.max(maxGain(root.left), 0);
        int var3 = Math.max(maxGain(root.right), 0);
        int var4 = root.val + var2 + var3;
        var1 = Math.max(var1, var4);
        return root.val + Math.max(var2, var3);
    }
}
