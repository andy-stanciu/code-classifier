class Solution {

    public int var1 = -999999;

    public int method1(TreeNode root) {
        recursion(root);
        return var1;
    }

    public int recursion(TreeNode curr) {
        if (curr == null) {
            return 0;
        }
        int var2 = 0;
        int var3 = 0;
        int var4;
        int currentValue = curr.val;
        var1 = Math.max(var1, currentValue);
        var2 += recursion(curr.left);
        var3 += recursion(curr.right);
        int tmp = Math.max(var2 + var3, Math.max(var2, var3));
        var1 = Math.max(var1, (currentValue + tmp));
        var4 = Math.max(currentValue + var2, currentValue + var3);
        var4 = Math.max(var4, currentValue);
        return var4;
    }
}
