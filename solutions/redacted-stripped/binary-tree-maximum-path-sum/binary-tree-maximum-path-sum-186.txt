class Solution {

    public int method1(TreeNode root, int[] maxi) {
        if (root == null)
            return 0;
        int left = Math.max(0, method1(root.left, maxi));
        int right = Math.max(0, method1(root.right, maxi));
        maxi[0] = Math.max(maxi[0], left + right + root.val);
        return root.val + Math.max(left, right);
    }

    public int method2(TreeNode root) {
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        method1(root, maxi);
        return maxi[0];
    }
}
