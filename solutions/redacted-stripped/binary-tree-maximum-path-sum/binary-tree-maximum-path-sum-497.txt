class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        a(root);
        return ans;
    }

    public int a(TreeNode root) {
        if (root == null)
            return 0;
        int l = a(root.left);
        int r = a(root.right);
        if (l < 0)
            l = 0;
        if (r < 0)
            r = 0;
        ans = Math.max(ans, l + r + root.val);
        return root.val + Math.max(l, r);
    }
}
