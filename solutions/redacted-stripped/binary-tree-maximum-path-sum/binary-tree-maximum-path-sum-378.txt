class Solution {

    int max;

    public int method1(TreeNode root) {
        max = Integer.MIN_VALUE;
        postOrder(root);
        return max;
    }

    public int postOrder(TreeNode root) {
        if (root == null)
            return 0;
        int left = postOrder(root.left);
        int right = postOrder(root.right);
        int var1 = Math.max(root.val, Math.max(root.val + left, root.val + right));
        int var2 = Math.max(var1, root.val + left + right);
        max = max > var2 ? max : var2;
        return var1;
    }
}
