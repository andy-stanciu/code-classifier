class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return var1;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftMax = Math.max(0, method2(root.left));
        int rightMax = Math.max(0, method2(root.right));
        var1 = Math.max(var1, leftMax + rightMax + root.val);
        return root.val + Math.max(leftMax, rightMax);
    }
}
