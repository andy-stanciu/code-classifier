class Solution {

    int maxAnswer = 0;

    public int method1(TreeNode root) {
        maxAnswer = root.val;
        dfs(root, 0);
        return maxAnswer;
    }

    private int dfs(TreeNode root, int param1) {
        if (root == null)
            return 0;
        int left = Math.max(0, dfs(root.left, param1 + root.val));
        int right = Math.max(0, dfs(root.right, param1 + root.val));
        maxAnswer = Math.max(maxAnswer, root.val + left + right);
        return Math.max(left, right) + root.val;
    }
}
