class Solution {

    int res = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return root.val;
        helper(root);
        return res;
    }

    int helper(TreeNode root) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            res = Math.max(root.val, res);
            return root.val;
        }
        res = Math.max(root.val, res);
        int l = helper(root.left);
        int r = helper(root.right);
        int tempAns = root.val + Math.max(l, r);
        int currAns = root.val + l + r;
        res = Math.max(res, Math.max(currAns, tempAns));
        return Math.max(root.val, Math.max(tempAns, Math.max(root.val + l, root.val + r)));
    }
}
