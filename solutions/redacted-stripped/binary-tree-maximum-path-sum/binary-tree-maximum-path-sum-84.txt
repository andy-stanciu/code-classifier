class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        int a = dfs(root);
        max = Math.max(a, max);
        return max;
    }

    public int dfs(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftMax = dfs(root.left);
        int rightMax = dfs(root.right);
        max = Math.max(max, root.val + Math.max(0, leftMax) + Math.max(0, rightMax));
        return Math.max(Math.max(0, leftMax), Math.max(0, rightMax)) + root.val;
    }
}
