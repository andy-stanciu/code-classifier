class Solution {

    public int method1(TreeNode root) {
        int[] maxi = { Integer.MIN_VALUE };
        method2(root, maxi);
        return maxi[0];
    }

    private int method2(TreeNode nodes, int[] maxi) {
        if (nodes == null) {
            return 0;
        }
        int left = Math.max(0, method2(nodes.left, maxi));
        int right = Math.max(0, method2(nodes.right, maxi));
        maxi[0] = Math.max(maxi[0], left + right + nodes.val);
        return Math.max(left, right) + nodes.val;
    }
}
