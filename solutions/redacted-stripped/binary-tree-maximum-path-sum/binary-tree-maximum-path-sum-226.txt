class Solution {

    private int maxi = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return maxi;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int left = Math.max(0, method2(root.left));
        int right = Math.max(0, method2(root.right));
        maxi = Math.max(maxi, left + right + root.val);
        return root.val + Math.max(left, right);
    }
}
