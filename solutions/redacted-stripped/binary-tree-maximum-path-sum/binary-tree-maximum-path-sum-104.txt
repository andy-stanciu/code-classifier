class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return max;
    }

    public int method2(TreeNode node) {
        if (node == null)
            return 0;
        int left = method2(node.left);
        int right = method2(node.right);
        if (left < 0)
            left = 0;
        if (right < 0)
            right = 0;
        int newp = left + right + node.val;
        max = Math.max(max, newp);
        return Math.max(left, right) + node.val;
    }
}
