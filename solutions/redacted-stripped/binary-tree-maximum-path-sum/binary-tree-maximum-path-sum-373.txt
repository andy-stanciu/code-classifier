class Solution {

    int var1 = Integer.MIN_VALUE;

    public int var1(TreeNode root) {
        method1(root);
        return var1;
    }

    private int method1(TreeNode node) {
        if (node == null)
            return 0;
        int var2 = Math.max(method1(node.left), 0);
        int var3 = Math.max(method1(node.right), 0);
        var1 = Math.max(var1, var2 + var3 + node.val);
        return Math.max(var2, var3) + node.val;
    }
}
