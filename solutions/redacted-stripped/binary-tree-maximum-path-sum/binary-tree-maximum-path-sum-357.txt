class Solution {

    int res;

    public int method1(TreeNode root) {
        res = root.val;
        helper(root);
        return res;
    }

    public int helper(TreeNode root) {
        if (root == null)
            return 0;
        int lVal = helper(root.left);
        int rVal = helper(root.right);
        int temp = Math.max(root.val, (root.val + Math.max(lVal, rVal)));
        res = Math.max(res, Math.max(temp, (root.val + lVal + rVal)));
        return temp;
    }
}
