class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return var1;
    }

    private int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = method2(root.left);
        int right = method2(root.right);
        int max = Math.max(left + root.val, Math.max(right + root.val, Math.max(root.val, left + right + root.val)));
        var1 = Math.max(var1, max);
        return Math.max(Math.max(left, right) + root.val, root.val);
    }
}
