class Solution {

    int var1 = Integer.MIN_VALUE;

    private int dfs(TreeNode root) {
        if (root == null)
            return 0;
        int l = dfs(root.left);
        int r = dfs(root.right);
        int max = Math.max(l + root.val, Math.max(r + root.val, Math.max(l + root.val + r, root.val)));
        int maxToReturn = Math.max(l + root.val, Math.max(r + root.val, root.val));
        var1 = Math.max(var1, max);
        return maxToReturn;
    }

    public int var1(TreeNode root) {
        dfs(root);
        return var1;
    }
}
