class Solution {

    public int method1(TreeNode root) {
        int[] maxValue = new int[1];
        maxValue[0] = Integer.MIN_VALUE;
        method2(root, maxValue);
        return maxValue[0];
    }

    public static int method2(TreeNode root, int[] maxValue) {
        if (root == null)
            return 0;
        int lh = Math.max(0, method2(root.left, maxValue));
        int rh = Math.max(0, method2(root.right, maxValue));
        maxValue[0] = Math.max(maxValue[0], root.val + lh + rh);
        return root.val + Math.max(lh, rh);
    }
}
