class Solution {

    public int max;

    public int dfs(TreeNode root) {
        if (root == null)
            return -1;
        int var1 = dfs(root.left);
        int var2 = dfs(root.right);
        var1 = Math.max(var1, 0);
        var2 = Math.max(var2, 0);
        int var3 = var1 + var2 + root.val;
        max = Math.max(var3, max);
        return Math.max(var1, var2) + root.val;
    }

    public int method1(TreeNode root) {
        max = Integer.MIN_VALUE;
        dfs(root);
        return max;
    }
}
