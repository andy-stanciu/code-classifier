class Solution {

    private int max;

    public int helper(TreeNode node) {
        if (node == null)
            return 0;
        int var1 = Math.max(0, helper(node.left));
        int var2 = Math.max(0, helper(node.right));
        max = Math.max(max, var1 + var2 + node.val);
        return Math.max(var1, var2) + node.val;
    }

    public int method1(TreeNode root) {
        max = Integer.MIN_VALUE;
        helper(root);
        return max;
    }
}
