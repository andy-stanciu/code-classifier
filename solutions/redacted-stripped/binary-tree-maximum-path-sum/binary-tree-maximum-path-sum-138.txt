class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return max;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = root.val;
        int var2 = method2(root.left);
        int var3 = method2(root.right);
        var1 = Math.max(var1, Math.max(var1 + var2, var1 + var3));
        max = Math.max(max, Math.max(var1, root.val + var2 + var3));
        return var1;
    }
}
