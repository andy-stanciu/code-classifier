class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null) {
            return -1;
        }
        int value = root.val;
        int left = Math.max(method1(root.left), 0);
        int right = Math.max(method1(root.right), 0);
        max = Math.max(max, left + right + value);
        return Math.max(left, right) + value;
    }

    public int method2(TreeNode root) {
        method1(root);
        return max;
    }
}
