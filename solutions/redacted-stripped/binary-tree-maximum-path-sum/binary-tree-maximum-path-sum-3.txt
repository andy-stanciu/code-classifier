class Solution {

    private int var1;

    public int method1(TreeNode root) {
        var1 = Integer.MIN_VALUE;
        method2(root);
        return var1;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int var2 = Math.max(0, method2(root.left));
        int var3 = Math.max(0, method2(root.right));
        int var4 = root.val + var2 + var3;
        var1 = Math.max(var4, var1);
        return root.val + Math.max(var2, var3);
    }
}
