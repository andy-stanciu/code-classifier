class Solution {

    public int method1(TreeNode node, int[] ans) {
        if (node == null)
            return 0;
        int left = Math.max(0, method1(node.left, ans));
        int right = Math.max(0, method1(node.right, ans));
        ans[0] = Math.max(ans[0], (left + right + node.val));
        return Math.max(left, right) + node.val;
    }

    public int method2(TreeNode root) {
        int[] ans = new int[] { Integer.MIN_VALUE };
        method1(root, ans);
        return ans[0];
    }
}
