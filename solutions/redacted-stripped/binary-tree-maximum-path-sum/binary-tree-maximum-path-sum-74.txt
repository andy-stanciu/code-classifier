class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        total(root);
        return max;
    }

    public int total(TreeNode root) {
        if (root == null)
            return 0;
        int left = total(root.left);
        int right = total(root.right);
        int var1 = root.val + left + right;
        if (root.val + left + right > max) {
            max = var1;
        }
        if (root.val + left > max) {
            var1 = root.val + left;
            max = var1;
        }
        if (root.val + right > max) {
            var1 = root.val + right;
            max = var1;
        }
        if (max < root.val) {
            var1 = root.val;
            max = var1;
        }
        if (left > right)
            var1 = root.val + left;
        else
            var1 = root.val + right;
        if (var1 < root.val)
            return root.val;
        return var1;
    }
}
