class Solution {

    public int method1(TreeNode root) {
        int[] var1 = { -1001 };
        func(root, var1);
        return var1[0];
    }

    public int func(TreeNode root, int[] var1) {
        if (root == null) {
            return 0;
        }
        int lH = Math.max(0, func(root.left, var1));
        int rH = Math.max(0, func(root.right, var1));
        var1[0] = Math.max(var1[0], root.val + lH + rH);
        return (root.val) + Math.max(lH, rH);
    }
}
