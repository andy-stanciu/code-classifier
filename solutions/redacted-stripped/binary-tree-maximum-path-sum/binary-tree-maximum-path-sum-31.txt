class Solution {

    int ans;

    public int method1(TreeNode root) {
        ans = root.val;
        method2(root);
        return ans;
    }

    int method2(TreeNode node) {
        if (node == null)
            return 0;
        int left = method2(node.left);
        int right = method2(node.right);
        int max = max(left + node.val, right + node.val, node.val);
        ans = max(ans, max, left + right + node.val);
        return max;
    }

    int max(int a, int b, int c) {
        return Math.max(Math.max(a, b), c);
    }
}
