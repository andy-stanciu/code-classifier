class Solution {

    int result = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return result;
    }

    private int helper(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = helper(root.left) + root.val;
        int var2 = helper(root.right) + root.val;
        int var3 = var1 + var2 - root.val;
        int var4 = Math.max(var1, Math.max(var2, Math.max(var3, root.val)));
        result = Math.max(result, var4);
        return Math.max(var1, Math.max(var2, root.val));
    }
}
