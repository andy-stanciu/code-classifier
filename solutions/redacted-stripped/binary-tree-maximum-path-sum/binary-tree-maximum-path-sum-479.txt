class Solution {

    public int help(TreeNode root, int[] arr) {
        if (root == null)
            return 0;
        int left = help(root.left, arr);
        int right = help(root.right, arr);
        int temp = Math.max(left, right) + root.val;
        int b = root.val;
        int tempAns = Math.max(temp, b);
        int ans = Math.max(tempAns, root.val + left + right);
        arr[0] = Math.max(arr[0], ans);
        return tempAns;
    }

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int[] arr = new int[1];
        arr[0] = Integer.MIN_VALUE;
        help(root, arr);
        return arr[0];
    }
}
