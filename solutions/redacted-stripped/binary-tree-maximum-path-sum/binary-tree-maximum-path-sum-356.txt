class Solution {

    int var1 = (int) -1e8;

    public int method1(TreeNode node) {
        method2(node);
        return var1;
    }

    public int method2(TreeNode node) {
        if (node == null)
            return (int) -1e8;
        int var2 = method2(node.left);
        int var3 = method2(node.right);
        int sideMax = Math.max(var2, var3) + node.val;
        var1 = Math.max(Math.max(var1, sideMax), Math.max(var2 + var3 + node.val, node.val));
        return Math.max(sideMax, node.val);
    }
}
