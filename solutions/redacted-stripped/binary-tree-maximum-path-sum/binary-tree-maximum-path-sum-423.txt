class Solution {

    int maxValue = -1001;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        method2(root);
        return maxValue;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = method2(root.left);
        int right = method2(root.right);
        maxValue = (int) Math.max(maxValue, root.val + ((left > 0) ? left : 0) + ((right > 0) ? right : 0));
        return root.val + (int) Math.max(left, (int) Math.max(right, 0));
    }
}
