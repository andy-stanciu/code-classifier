class Solution {

    public int method1(TreeNode root, int[] max) {
        if (root == null)
            return 0;
        int var1 = Math.max(0, method1(root.left, max));
        int var2 = Math.max(0, method1(root.right, max));
        max[0] = Math.max(max[0], var1 + var2 + root.val);
        return root.val + Math.max(var1, var2);
    }

    public int method2(TreeNode root) {
        int[] max = { Integer.MIN_VALUE };
        method1(root, max);
        return max[0];
    }
}
