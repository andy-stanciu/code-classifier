class Solution {

    public int method1(TreeNode root) {
        int[] value = new int[1];
        value[0] = Integer.MIN_VALUE;
        method2(root, value);
        return value[0];
    }

    private int method2(TreeNode root, int[] value) {
        if (root == null)
            return 0;
        int left = Math.max(0, method2(root.left, value));
        int right = Math.max(0, method2(root.right, value));
        value[0] = Math.max(value[0], left + right + (root.val));
        return (root.val) + Math.max(left, right);
    }
}
