class Solution {

    public int method1(TreeNode root, int[] maxValue) {
        if (root == null)
            return 0;
        int left = Math.max(0, method1(root.left, maxValue));
        int right = Math.max(0, method1(root.right, maxValue));
        maxValue[0] = Math.max(maxValue[0], left + right + root.val);
        return Math.max(left, right) + root.val;
    }

    public int method2(TreeNode root) {
        int[] maxValue = new int[1];
        maxValue[0] = Integer.MIN_VALUE;
        method1(root, maxValue);
        return maxValue[0];
    }
}
