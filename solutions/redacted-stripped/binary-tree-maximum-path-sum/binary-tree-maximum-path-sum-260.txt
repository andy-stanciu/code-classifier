class Solution {

    int var1 = Integer.MIN_VALUE;

    int recurr(TreeNode root) {
        if (root == null)
            return 0;
        int left = Math.max(0, recurr(root.left));
        int right = Math.max(0, recurr(root.right));
        int var2 = root.val + left + right;
        var1 = Math.max(var1, var2);
        return root.val + Math.max(left, right);
    }

    public int method1(TreeNode root) {
        int temp = recurr(root);
        return var1;
    }
}
