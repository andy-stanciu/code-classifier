class Solution {

    public static int findmax(TreeNode root, int[] maxi) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(0, findmax(root.left, maxi));
        int var2 = Math.max(0, findmax(root.right, maxi));
        maxi[0] = Math.max(maxi[0], var1 + var2 + root.val);
        return root.val + Math.max(var1, var2);
    }

    public int method1(TreeNode root) {
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        findmax(root, maxi);
        return maxi[0];
    }
}
