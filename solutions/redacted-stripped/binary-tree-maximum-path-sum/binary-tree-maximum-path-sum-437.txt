class Solution {

    public int dfs(TreeNode curr, int[] maxi) {
        if (curr == null)
            return 0;
        int lh = Math.max(0, dfs(curr.left, maxi));
        int rh = Math.max(0, dfs(curr.right, maxi));
        maxi[0] = Math.max(lh + rh + curr.val, maxi[0]);
        return curr.val + Math.max(lh, rh);
    }

    public int method1(TreeNode root) {
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        dfs(root, maxi);
        return maxi[0];
    }
}
