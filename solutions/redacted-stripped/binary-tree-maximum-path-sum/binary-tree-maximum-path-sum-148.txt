class Solution {

    int var1 = Integer.MIN_VALUE;

    int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = method1(root.left);
        int right = method1(root.right);
        var1 = Math.max(root.val, Math.max(Math.max(var1, root.val + right + left), root.val + Math.max(left, right)));
        return Math.max(root.val, root.val + Math.max(left, right));
    }

    public int method2(TreeNode root) {
        method1(root);
        return var1;
    }
}
