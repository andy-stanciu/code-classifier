class Solution {

    int ans = Integer.MIN_VALUE;

    public int solve(TreeNode root) {
        if (root == null)
            return 0;
        int l = solve(root.left);
        int r = solve(root.right);
        ans = Math.max(Math.max(Math.max(Math.max(r, l), r + l + root.val), ans), Math.max(root.val, Math.max(r, l) + root.val));
        return Math.max(Math.max(l, r) + root.val, root.val);
    }

    public void solve2(TreeNode root) {
        if (root == null)
            return;
        ans = Math.max(ans, root.val);
        solve2(root.left);
        solve2(root.right);
    }

    public int method1(TreeNode root) {
        solve(root);
        if (ans == 0) {
            ans = Integer.MIN_VALUE;
            solve2(root);
        }
        return ans;
    }
}
