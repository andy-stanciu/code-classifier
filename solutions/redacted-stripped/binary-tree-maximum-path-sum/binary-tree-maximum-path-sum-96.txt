class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        var1[0] = Integer.MIN_VALUE;
        method2(root, var1);
        return var1[0];
    }

    public int method2(TreeNode root, int[] var1) {
        if (null == root) {
            return 0;
        }
        int var2 = Math.max(method2(root.left, var1), 0);
        int var3 = Math.max(method2(root.right, var1), 0);
        var1[0] = Math.max(var1[0], var2 + var3 + root.val);
        return root.val + Math.max(var2, var3);
    }
}
