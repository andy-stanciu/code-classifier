class Solution {

    public int maxVal = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        int potentialMaxVal = method2(root);
        return maxVal;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(method2(root.left), 0);
        int right = Math.max(method2(root.right), 0);
        int var1 = left + right + root.val;
        maxVal = Math.max(maxVal, var1);
        return root.val + Math.max(left, right);
    }
}
