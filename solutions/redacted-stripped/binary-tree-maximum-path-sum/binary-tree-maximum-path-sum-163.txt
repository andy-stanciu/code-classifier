class Solution {

    private int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        recursive(root);
        return ans;
    }

    private int recursive(TreeNode node) {
        if (node == null)
            return 0;
        int lv = recursive(node.left);
        int rv = recursive(node.right);
        int ret = node.val;
        ret = Math.max(ret, ret + Math.max(lv, rv));
        ans = Math.max(ans, Math.max(ret, node.val + lv + rv));
        return ret;
    }
}
