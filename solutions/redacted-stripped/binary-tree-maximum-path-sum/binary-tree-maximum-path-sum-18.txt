class Solution {

    static int method1(TreeNode root, int[] maxValue) {
        if (root == null)
            return 0;
        int var1 = Math.max(0, method1(root.left, maxValue));
        int var2 = Math.max(0, method1(root.right, maxValue));
        maxValue[0] = Math.max(maxValue[0], var1 + var2 + root.val);
        return root.val + Math.max(var1, var2);
    }

    public int method2(TreeNode root) {
        int[] maxValue = new int[1];
        maxValue[0] = Integer.MIN_VALUE;
        method1(root, maxValue);
        return maxValue[0];
    }
}
