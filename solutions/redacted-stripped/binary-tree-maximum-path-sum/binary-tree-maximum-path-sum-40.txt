class Solution {

    int var1;

    public int method1(TreeNode root) {
        var1 = root.val;
        var1(root);
        return var1;
    }

    public int var1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = var1(root.left);
        int right = var1(root.right);
        int max = Math.max(left, right);
        var1 = Math.max(root.val, Math.max(max + root.val, Math.max(var1, left + right + root.val)));
        return Math.max(root.val, max + root.val);
    }
}
