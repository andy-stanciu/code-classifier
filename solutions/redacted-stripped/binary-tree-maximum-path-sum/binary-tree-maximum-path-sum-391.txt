class Solution {

    int result = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return result;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = method2(root.left);
        int var2 = method2(root.right);
        result = Math.max(root.val, result);
        result = Math.max(result, var1 + root.val);
        result = Math.max(result, var2 + root.val);
        result = Math.max(result, var1 + var2 + root.val);
        int total = result;
        return Math.max(var1 + root.val, Math.max(var2 + root.val, root.val));
    }
}
