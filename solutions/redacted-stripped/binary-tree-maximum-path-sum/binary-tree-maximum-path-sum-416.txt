class Solution {

    private int max;

    public int method1(TreeNode root) {
        max = Integer.MIN_VALUE;
        method2(root);
        return max;
    }

    private int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int lmax = Math.max(method2(root.left), 0);
        int rmax = Math.max(method2(root.right), 0);
        max = Math.max(max, lmax + rmax + root.val);
        return Math.max(lmax + root.val, rmax + root.val);
    }
}
