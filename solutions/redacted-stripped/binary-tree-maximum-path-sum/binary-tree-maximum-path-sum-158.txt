class Solution {

    int ans = -1001;

    public int method1(TreeNode root) {
        traversal(root);
        return ans;
    }

    private int traversal(TreeNode root) {
        if (root == null)
            return 0;
        int a = Math.max(0, traversal(root.left));
        int b = Math.max(0, traversal(root.right));
        ans = Math.max(ans, a + b + root.val);
        return root.val + Math.max(a, b);
    }
}
