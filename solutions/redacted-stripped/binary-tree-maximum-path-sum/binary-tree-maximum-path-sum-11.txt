class Solution {

    public int method1(TreeNode root) {
        int[] max = new int[1];
        max[0] = Integer.MIN_VALUE;
        method2(root, max);
        return max[0];
    }

    static int method2(TreeNode node, int[] max) {
        if (node == null)
            return 0;
        int l = Math.max(0, method2(node.left, max));
        int r = Math.max(0, method2(node.right, max));
        max[0] = Math.max(max[0], l + r + node.val);
        return Math.max(l, r) + node.val;
    }
}
