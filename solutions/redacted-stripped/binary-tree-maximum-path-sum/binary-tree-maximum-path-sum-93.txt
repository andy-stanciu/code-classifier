class Solution {

    int var1;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            var1 = Math.max(root.val, var1);
            return root.val;
        }
        int left = method1(root.left);
        int right = method1(root.right);
        int oneChild = Math.max(root.val, root.val + Math.max(left, right));
        int bothChild = Math.max(oneChild, root.val + left + right);
        var1 = Math.max(var1, bothChild);
        return oneChild;
    }

    public int method2(TreeNode root) {
        var1 = Integer.MIN_VALUE;
        method1(root);
        return var1;
    }
}
