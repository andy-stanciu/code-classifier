class Solution {

    private int var1;

    public int method1(TreeNode root) {
        var1 = Integer.MIN_VALUE;
        helper(root);
        return var1;
    }

    private int helper(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int var2 = Math.max(0, helper(node.left));
        int var3 = Math.max(0, helper(node.right));
        int var4 = node.val + var2 + var3;
        var1 = Math.max(var1, var4);
        return node.val + Math.max(var2, var3);
    }
}
