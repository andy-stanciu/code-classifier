class Solution {

    private int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return root.val;
        }
        method2(root);
        return this.var1;
    }

    private int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var2 = method2(root.left);
        int var3 = method2(root.right);
        this.var1 = Math.max(var1, var2 + var3 + root.val);
        return Math.max(root.val + Math.max(var2, var3), 0);
    }
}
