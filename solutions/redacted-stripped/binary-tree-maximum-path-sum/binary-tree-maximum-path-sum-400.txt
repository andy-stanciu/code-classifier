class Solution {

    public int method1(TreeNode root) {
        int[] max = new int[1];
        max[0] = Integer.MIN_VALUE;
        solve(root, max);
        return max[0];
    }

    int solve(TreeNode node, int[] max) {
        if (node == null)
            return 0;
        int ls = Math.max(0, solve(node.left, max));
        int rs = Math.max(0, solve(node.right, max));
        max[0] = Math.max(max[0], ls + rs + node.val);
        return node.val + Math.max(ls, rs);
    }
}
