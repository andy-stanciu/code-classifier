class Solution {

    public int solve(TreeNode root, int[] res) {
        if (root == null) {
            return 0;
        }
        int leftMax = Math.max(solve(root.left, res), 0);
        int rightMax = Math.max(solve(root.right, res), 0);
        int var1 = root.val + leftMax + rightMax;
        res[0] = Math.max(res[0], var1);
        return root.val + Math.max(leftMax, rightMax);
    }

    public int method1(TreeNode root) {
        int[] res = new int[1];
        res[0] = Integer.MIN_VALUE;
        solve(root, res);
        return res[0];
    }
}
