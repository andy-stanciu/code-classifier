class Solution {

    public int method1(TreeNode root) {
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        method2(root, maxi);
        return maxi[0];
    }

    public static int method2(TreeNode root, int[] maxi) {
        if (root == null)
            return 0;
        int left = Math.max(method2(root.left, maxi), 0);
        int right = Math.max(method2(root.right, maxi), 0);
        maxi[0] = Math.max(maxi[0], left + right + root.val);
        return root.val + Math.max(left, right);
    }
}
