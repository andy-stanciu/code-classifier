class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return var1;
    }

    public int method2(TreeNode node) {
        if (node == null)
            return 0;
        int LHS = Math.max(0, method2(node.left));
        int RHS = Math.max(0, method2(node.right));
        var1 = Math.max(var1, node.val + LHS + RHS);
        return node.val + Math.max(LHS, RHS);
    }
}
