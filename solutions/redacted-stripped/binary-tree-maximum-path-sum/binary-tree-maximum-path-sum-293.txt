class Solution {

    int maxValue = Integer.MIN_VALUE;

    public int findMax(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = Math.max(0, findMax(root.left));
        int var2 = Math.max(0, findMax(root.right));
        maxValue = Math.max(maxValue, root.val + var1 + var2);
        return root.val + Math.max(var1, var2);
    }

    public int method1(TreeNode root) {
        findMax(root);
        return maxValue;
    }
}
