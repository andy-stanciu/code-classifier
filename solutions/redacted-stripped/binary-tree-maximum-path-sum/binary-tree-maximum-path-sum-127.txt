class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return ans;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int l = Math.max(0, method2(root.left));
        int r = Math.max(0, method2(root.right));
        ans = Math.max(ans, l + r + root.val);
        return Math.max(l, r) + root.val;
    }
}
