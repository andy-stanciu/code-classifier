class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        var1[0] = Integer.MIN_VALUE;
        recursiveTraversalFindDepth(root, var1);
        return var1[0];
    }

    public int recursiveTraversalFindDepth(TreeNode root, int[] var1) {
        if (root == null)
            return 0;
        int left = recursiveTraversalFindDepth(root.left, var1);
        left = left < 0 ? 0 : left;
        int right = recursiveTraversalFindDepth(root.right, var1);
        right = right < 0 ? 0 : right;
        var1[0] = Math.max(var1[0], root.val + left + right);
        return root.val + Math.max(left, right);
    }
}
