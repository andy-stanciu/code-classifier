class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return ans;
    }

    int helper(TreeNode node) {
        if (node == null)
            return 0;
        int ls = Math.max(helper(node.left), 0);
        int rs = Math.max(helper(node.right), 0);
        ans = Math.max(ans, node.val + ls + rs);
        return node.val + Math.max(ls, rs);
    }
}
