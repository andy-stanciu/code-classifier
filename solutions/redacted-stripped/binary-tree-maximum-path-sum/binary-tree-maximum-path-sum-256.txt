class Solution {

    int globalMax = -Integer.MAX_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return globalMax;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int leftMax = method2(root.left);
        int rightMax = method2(root.right);
        leftMax = Math.max(leftMax, 0);
        rightMax = Math.max(rightMax, 0);
        int localmaxima = leftMax + rightMax + root.val;
        globalMax = Math.max(globalMax, localmaxima);
        return Math.max(leftMax, rightMax) + root.val;
    }
}
