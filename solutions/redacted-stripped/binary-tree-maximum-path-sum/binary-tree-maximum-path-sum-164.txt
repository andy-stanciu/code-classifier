class Solution {

    private int globalMax;

    public int method1(TreeNode root) {
        globalMax = Integer.MIN_VALUE;
        method2(root);
        return globalMax;
    }

    public int method2(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int leftMax = Math.max(method2(node.left), 0);
        int rightMax = Math.max(method2(node.right), 0);
        int var1 = node.val + leftMax + rightMax;
        globalMax = Math.max(globalMax, var1);
        return node.val + Math.max(leftMax, rightMax);
    }
}
