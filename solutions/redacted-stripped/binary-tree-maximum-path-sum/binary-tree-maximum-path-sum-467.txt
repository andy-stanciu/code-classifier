class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        dfs(0, root);
        return max;
    }

    int dfs(int param1, TreeNode root) {
        if (root == null) {
            return 0;
        }
        max = Math.max(max, root.val);
        int var1 = param1 + root.val;
        int var2 = dfs(var1, root.left);
        int var3 = dfs(var1, root.right);
        max = Math.max(max, root.val + var2 + var3);
        max = Math.max(max, root.val + var2);
        max = Math.max(max, root.val + var3);
        return Math.max(root.val, root.val + Math.max(var2, var3));
    }
}
