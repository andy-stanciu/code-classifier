class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        method2(root);
        return var1;
    }

    private int method2(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int var2 = Math.max(0, method2(node.left));
        int var3 = Math.max(0, method2(node.right));
        var1 = Math.max(var1, var2 + var3 + node.val);
        return Math.max(var2, var3) + node.val;
    }
}
