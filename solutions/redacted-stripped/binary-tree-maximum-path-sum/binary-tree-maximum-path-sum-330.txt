class Solution {

    private int var1 = Integer.MIN_VALUE;

    public int var1(TreeNode root) {
        method1(root);
        return var1;
    }

    private int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var2 = Math.max(0, method1(root.left));
        int var3 = Math.max(0, method1(root.right));
        int var4 = var2 + var3 + root.val;
        var1 = Math.max(var1, var4);
        return Math.max(var2, var3) + root.val;
    }
}
