class Solution {

    int[] maxVal = new int[1];

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        maxVal[0] = Integer.MIN_VALUE;
        method2(root, maxVal);
        return maxVal[0];
    }

    public int method2(TreeNode root, int[] maxVal) {
        if (root == null)
            return 0;
        int var1 = Math.max(0, method2(root.left, maxVal));
        int var2 = Math.max(0, method2(root.right, maxVal));
        maxVal[0] = Math.max(maxVal[0], (var1 + var2 + root.val));
        return Math.max(var1, var2) + root.val;
    }
}
