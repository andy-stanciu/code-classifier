class Solution {

    public int method1(TreeNode root) {
        var result = new int[] { Integer.MIN_VALUE };
        method1(root, result);
        return result[0];
    }

    int method1(TreeNode root, int[] result) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(method1(root.left, result), 0);
        int right = Math.max(method1(root.right, result), 0);
        result[0] = Math.max(left + right + root.val, result[0]);
        return Math.max(left, right) + root.val;
    }
}
