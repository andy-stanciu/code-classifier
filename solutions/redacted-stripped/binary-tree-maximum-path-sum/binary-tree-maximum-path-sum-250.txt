class Solution {

    public int method1(TreeNode root) {
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        method2(root, maxi);
        return maxi[0];
    }

    public int method2(TreeNode node, int[] maxi) {
        if (node == null)
            return 0;
        int right = Math.max(0, method2(node.right, maxi));
        int left = Math.max(0, method2(node.left, maxi));
        maxi[0] = Math.max(maxi[0], right + left + node.val);
        return Math.max(left, right) + node.val;
    }
}
