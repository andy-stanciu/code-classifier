class Solution {

    private int var1;

    private int helper(TreeNode head) {
        if (head == null)
            return 0;
        int left = helper(head.left);
        int right = helper(head.right);
        int var2 = Math.max(head.val, (Math.max(left + head.val, right + head.val)));
        int var3 = Math.max(var2, head.val + left + right);
        var1 = Math.max(var1, var3);
        return var2;
    }

    public int method1(TreeNode root) {
        var1 = Integer.MIN_VALUE;
        helper(root);
        return var1;
    }
}
