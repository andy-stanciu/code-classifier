class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return max;
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int lh = Math.max(0, method2(root.left));
        int rh = Math.max(0, method2(root.right));
        max = Math.max(max, root.val + lh + rh);
        return root.val + Math.max(lh, rh);
    }
}
