class Solution {

    int MAX = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = method1(root.left);
        int leftWeight = left + root.val;
        int right = method1(root.right);
        int rightWeight = right + root.val;
        int var1 = left + right + root.val;
        int var2 = left + root.val;
        int var3 = right + root.val;
        if (root.left != null && root.right != null) {
            MAX = Math.max(left, MAX);
            MAX = Math.max(right, MAX);
        }
        MAX = Math.max(var1, MAX);
        MAX = Math.max(var3, MAX);
        MAX = Math.max(var2, MAX);
        MAX = Math.max(root.val, MAX);
        int maxVal = Math.max(leftWeight, rightWeight);
        return Math.max(root.val, maxVal);
    }

    public int method2(TreeNode root) {
        if (root.left == null && root.right == null) {
            return root.val;
        }
        MAX = root.val;
        method1(root);
        return MAX;
    }
}
