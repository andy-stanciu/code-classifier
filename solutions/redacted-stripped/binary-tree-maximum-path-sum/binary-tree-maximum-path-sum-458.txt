class Solution {

    public static class Maxi {

        int val;

        Maxi() {
            this.val = Integer.MIN_VALUE;
        }
    }

    public static int method1(TreeNode root, Maxi maxi) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(0, method1(root.left, maxi));
        int var2 = Math.max(0, method1(root.right, maxi));
        maxi.val = Math.max(maxi.val, (var2 + var1 + root.val));
        return Math.max(var1, var2) + root.val;
    }

    public int method2(TreeNode root) {
        Maxi maxi = new Maxi();
        method1(root, maxi);
        return maxi.val;
    }
}
