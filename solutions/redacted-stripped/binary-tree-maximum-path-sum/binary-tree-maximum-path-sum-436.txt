class Solution {

    int max = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = method1(root.left);
        int right = method1(root.right);
        int nodeMax = Math.max(root.val, root.val + Math.max(left, right));
        max = Math.max(max, Math.max(nodeMax, root.val + left + right));
        return nodeMax;
    }

    public int method2(TreeNode root) {
        max = Integer.MIN_VALUE;
        method1(root);
        return max;
    }
}
