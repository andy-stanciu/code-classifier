class Solution {

    public int method1(TreeNode root) {
        int[] maxval = new int[1];
        maxval[0] = Integer.MIN_VALUE;
        method2(root, maxval);
        return maxval[0];
    }

    static int method2(TreeNode node, int[] maxval) {
        if (node == null)
            return 0;
        int left = Math.max(0, method2(node.left, maxval));
        ;
        int right = Math.max(0, method2(node.right, maxval));
        maxval[0] = Math.max(maxval[0], left + right + node.val);
        return node.val + Math.max(left, right);
    }
}
