class Solution {

    public void method1(int[] nums) {
        int len = nums.length;
        int temp;
        int sorted_point = -1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] < nums[i + 1])
                sorted_point = i;
        }
        if (sorted_point == -1) {
            Arrays.sort(nums);
        } else {
            int swapper = Integer.MAX_VALUE;
            int idx = -1;
            for (int i = sorted_point; i < len; i++) {
                if (nums[i] > nums[sorted_point]) {
                    if (nums[i] < swapper) {
                        swapper = nums[i];
                        idx = i;
                    }
                }
            }
            temp = nums[idx];
            nums[idx] = nums[sorted_point];
            nums[sorted_point] = temp;
            Arrays.sort(nums, sorted_point + 1, len);
        }
    }
}
