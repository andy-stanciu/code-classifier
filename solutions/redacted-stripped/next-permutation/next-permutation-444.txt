class Solution {

    public void method1(int[] nums) {
        int index = findIndex(nums);
        if (index == -1) {
            reverse(nums, 0, nums.length - 1);
            return;
        } else {
            int i = nums.length - 1;
            while (i > index) {
                if (nums[i] > nums[index]) {
                    break;
                } else {
                    i -= 1;
                }
            }
            int temp = nums[i];
            nums[i] = nums[index];
            nums[index] = temp;
            reverse(nums, index + 1, nums.length - 1);
            return;
        }
    }

    private int findIndex(int[] arr) {
        int i = arr.length - 2;
        while (i >= 0) {
            if (arr[i] >= arr[i + 1]) {
                i -= 1;
            } else {
                break;
            }
        }
        return i;
    }

    private void reverse(int[] arr, int low, int high) {
        while (low <= high) {
            int temp = arr[low];
            arr[low] = arr[high];
            arr[high] = temp;
            low += 1;
            high -= 1;
        }
    }
}
