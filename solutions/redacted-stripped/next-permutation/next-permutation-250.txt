class Solution {

    public void swap(int idx1, int idx2, int[] arr) {
        int temp = arr[idx1];
        arr[idx1] = arr[idx2];
        arr[idx2] = temp;
    }

    public void reverse(int start, int end, int[] arr) {
        for (int i = start, j = end; i < j; i++, j--) {
            swap(i, j, arr);
        }
    }

    public void method1(int[] nums) {
        int brkPt = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                brkPt = i;
                break;
            }
        }
        if (brkPt == -1) {
            reverse(0, nums.length - 1, nums);
            return;
        }
        for (int i = nums.length - 1; i > brkPt; i--) {
            if (nums[i] > nums[brkPt]) {
                swap(i, brkPt, nums);
                break;
            }
        }
        reverse(brkPt + 1, nums.length - 1, nums);
    }
}
