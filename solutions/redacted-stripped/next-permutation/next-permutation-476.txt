class Solution {

    public void method1(int[] nums) {
        int inx = -1;
        for (int i = nums.length - 1; i >= 1; i--) {
            if (nums[i] > nums[i - 1]) {
                inx = i - 1;
                break;
            }
        }
        if (inx == -1) {
            reverse(nums, 0, nums.length - 1);
            return;
        }
        for (int i = nums.length - 1; i > inx; i--) {
            if (nums[i] > nums[inx]) {
                int temp = nums[i];
                nums[i] = nums[inx];
                nums[inx] = temp;
                break;
            }
        }
        reverse(nums, inx + 1, nums.length - 1);
    }

    public void reverse(int[] nums, int inx, int n) {
        while (inx <= n) {
            int temp = nums[inx];
            nums[inx] = nums[n];
            nums[n] = temp;
            inx++;
            n--;
        }
    }
}
