class Solution {

    public void method1(int[] nums) {
        int ind1 = -1;
        for (int i = nums.length - 1; i >= 1; i--) {
            if (nums[i - 1] < nums[i]) {
                ind1 = i - 1;
                break;
            }
        }
        int ind2 = -1;
        if (ind1 != -1) {
            for (int i = nums.length - 1; i >= ind1; i--) {
                if (nums[i] > nums[ind1]) {
                    ind2 = i;
                    break;
                }
            }
            int temp = nums[ind1];
            nums[ind1] = nums[ind2];
            nums[ind2] = temp;
        }
        for (int i = ind1 + 1, j = nums.length - 1; i < j; i++, j--) {
            int t = nums[i];
            nums[i] = nums[j];
            nums[j] = t;
        }
    }
}
