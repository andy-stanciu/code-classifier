class Solution {

    public void method1(int[] nums) {
        int ind = -1;
        int n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        int swapIndex = -1;
        for (int i = n - 1; i >= 0; i--) {
            if (nums[i] > nums[ind]) {
                swapIndex = i;
                break;
            }
        }
        int temp = nums[swapIndex];
        nums[swapIndex] = nums[ind];
        nums[ind] = temp;
        reverse(nums, ind + 1, n - 1);
    }

    public void reverse(int[] nums, int x, int y) {
        while (x < y) {
            int temp = nums[x];
            nums[x] = nums[y];
            nums[y] = temp;
            x++;
            y--;
        }
    }
}
