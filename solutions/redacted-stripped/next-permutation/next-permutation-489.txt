class Solution {

    public void reverse(int[] arr, int s, int e) {
        while (s <= e) {
            swap(s, e, arr);
            s++;
            e--;
        }
    }

    public void swap(int a, int b, int[] arr) {
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }

    public void method1(int[] nums) {
        int idx = -1;
        int n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                idx = i;
                break;
            }
        }
        if (idx == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        for (int i = n - 1; i >= 0; i--) {
            if (nums[i] > nums[idx]) {
                swap(i, idx, nums);
                break;
            }
        }
        reverse(nums, idx + 1, n - 1);
        return;
    }
}
