class Solution {

    public void method1(int[] nums) {
        int i = nums.length - 1;
        while (i > 0 && nums[i - 1] >= nums[i]) {
            i--;
        }
        if (i - 1 < 0) {
            reverseFromIndex(nums, 0);
        } else {
            int pivot = nums[i - 1];
            int j = nums.length - 1;
            while (nums[j] <= pivot) {
                j--;
            }
            nums[i - 1] = nums[j];
            nums[j] = pivot;
            reverseFromIndex(nums, i);
        }
    }

    public void reverseFromIndex(int[] nums, int index) {
        int i = index;
        int j = nums.length - 1;
        while (i < j) {
            int swap = nums[i];
            nums[i] = nums[j];
            nums[j] = swap;
            i++;
            j--;
        }
    }
}
