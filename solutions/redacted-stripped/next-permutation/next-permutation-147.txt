class Solution {

    public void method1(int[] nums) {
        int s = nums.length;
        int i = s - 2;
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            i--;
        }
        if (i >= 0) {
            int j = s - 1;
            while (j > i && nums[i] >= nums[j]) {
                j--;
            }
            nums[i] = nums[i] ^ nums[j];
            nums[j] = nums[i] ^ nums[j];
            nums[i] = nums[i] ^ nums[j];
        }
        for (int p = i + 1, q = s - 1; p < q; p++, q--) {
            nums[p] = nums[p] ^ nums[q];
            nums[q] = nums[p] ^ nums[q];
            nums[p] = nums[p] ^ nums[q];
        }
    }
}
