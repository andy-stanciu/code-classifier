class Solution {

    public void method1(int[] nums) {
        int pivot_index = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                pivot_index = i;
                break;
            }
        }
        if (pivot_index == -1) {
            nums = reverse_array(nums, 0, nums.length - 1);
        } else {
            int right_most_index = 0;
            for (int i = nums.length - 1; i > pivot_index; i--) {
                if (nums[i] > nums[pivot_index]) {
                    right_most_index = i;
                    break;
                }
            }
            int temp = nums[pivot_index];
            nums[pivot_index] = nums[right_most_index];
            nums[right_most_index] = temp;
            int end = nums.length - 1;
            int start = pivot_index + 1;
            nums = reverse_array(nums, start, end);
        }
        for (int i = 0; i < nums.length; i++) {
            System.out.print(nums[i] + " ");
        }
    }

    public int[] reverse_array(int[] nums, int start, int end) {
        while (start < end) {
            int temp1 = nums[start];
            nums[start] = nums[end];
            nums[end] = temp1;
            start++;
            end--;
        }
        return nums;
    }
}
