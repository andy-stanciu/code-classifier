class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int k = -1;
        if (n <= 1 || nums == null) {
            return;
        }
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                k = i;
                break;
            }
        }
        if (k == -1) {
            reverse(nums, 0, n - 1);
        } else {
            for (int i = n - 1; i > k; i--) {
                if (nums[i] > nums[k]) {
                    int temp = nums[k];
                    nums[k] = nums[i];
                    nums[i] = temp;
                    break;
                }
            }
            reverse(nums, k + 1, n - 1);
        }
    }

    public static void reverse(int[] nums, int start, int end) {
        while (start < end) {
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }
}
