class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int index = -1;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                index = i - 1;
                break;
            }
        }
        if (index != -1) {
            int swap_index = index;
            for (int j = n - 1; j > index; j--) {
                if (nums[j] > nums[index]) {
                    swap_index = j;
                    break;
                }
            }
            int temp = nums[index];
            nums[index] = nums[swap_index];
            nums[swap_index] = temp;
        }
        reverseFromIndex(nums, index + 1);
    }

    public static void reverseFromIndex(int[] arr, int startIndex) {
        int left = startIndex;
        int right = arr.length - 1;
        while (left < right) {
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        }
    }
}
