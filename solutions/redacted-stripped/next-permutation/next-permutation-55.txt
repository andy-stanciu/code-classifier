class Solution {

    private static void rev(int[] arr, int low, int high) {
        int i = low, j = high;
        while (i < j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
    }

    public void method1(int[] A) {
        int n = A.length;
        int ind = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (A[i] < A[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            rev(A, 0, n - 1);
            return;
        }
        for (int i = n - 1; i > ind; i--) {
            if (A[i] > A[ind]) {
                int tt = A[i];
                A[i] = A[ind];
                A[ind] = tt;
                break;
            }
        }
        rev(A, ind + 1, n - 1);
        return;
    }
}
