class Solution {

    public void method1(int[] nums) {
        int l = nums.length - 1, r = nums.length - 1;
        while (l > 0 && nums[l - 1] >= nums[l]) {
            l--;
        }
        if (l > 0) {
            int tmpR = r;
            while (nums[tmpR] <= nums[l - 1]) tmpR--;
            int tmp = nums[tmpR];
            nums[tmpR] = nums[l - 1];
            nums[l - 1] = tmp;
        }
        while (l <= r) {
            int tmp = nums[r];
            nums[r] = nums[l];
            nums[l] = tmp;
            r--;
            l++;
        }
    }
}
