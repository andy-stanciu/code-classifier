class Solution {

    public void method1(int[] nums) {
        int k = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                k = i - 1;
                break;
            }
        }
        if (k == -1) {
            for (int i = 0; i < nums.length / 2; i++) {
                int temp = nums[i];
                nums[i] = nums[nums.length - 1 - i];
                nums[nums.length - 1 - i] = temp;
            }
            return;
        }
        int l = nums.length - 1;
        for (; l > k; l--) {
            if (nums[l] > nums[k]) {
                break;
            }
        }
        int temp = nums[k];
        nums[k] = nums[l];
        nums[l] = temp;
        for (int i = k + 1, j = nums.length - 1; i < j; i++, j--) {
            temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
        }
    }
}
