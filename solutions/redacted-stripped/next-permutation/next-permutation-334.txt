class Solution {

    public void method1(int[] nums) {
        if (nums.length < 2) {
            return;
        }
        int index = -1;
        int moveIndex = -1;
        for (int currIndex = nums.length - 2; currIndex >= 0; currIndex--) {
            var minimalBigger = Integer.MAX_VALUE;
            for (int searchBiggerIndex = currIndex + 1; searchBiggerIndex < nums.length; searchBiggerIndex++) {
                if (nums[searchBiggerIndex] > nums[currIndex] && nums[searchBiggerIndex] < minimalBigger) {
                    minimalBigger = nums[searchBiggerIndex];
                    moveIndex = searchBiggerIndex;
                }
            }
            if (moveIndex != -1) {
                var tmp = nums[currIndex];
                nums[currIndex] = nums[moveIndex];
                nums[moveIndex] = tmp;
                index = currIndex;
                break;
            }
        }
        if (index == -1) {
            Arrays.sort(nums);
        } else {
            Arrays.sort(nums, index + 1, nums.length);
        }
    }
}
