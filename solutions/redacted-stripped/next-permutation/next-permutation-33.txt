class Solution {

    int method1(int[] nums, int i, int j) {
        int k = i, mink = Integer.MAX_VALUE;
        while (i < nums.length) {
            if (j < nums[i]) {
                if (mink > (nums[i] - j)) {
                    mink = nums[i] - j;
                    k = i;
                }
            }
            i++;
        }
        return k;
    }

    public void method2(int[] nums) {
        boolean b = true;
        for (int i = nums.length - 2; i > -1; i--) {
            if (nums[i] < nums[i + 1]) {
                int nG = method1(nums, i, nums[i]);
                int t = nums[i];
                nums[i] = nums[nG];
                nums[nG] = t;
                b = false;
                Arrays.sort(nums, i + 1, nums.length);
                break;
            }
        }
        if (b) {
            Arrays.sort(nums);
        }
    }
}
