class Solution {

    public void reverse(int[] arr, int i, int j) {
        while (i < j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
    }

    public void method1(int[] nums) {
        int n = nums.length;
        int i = n - 1;
        for (; i >= 1; i--) {
            if (nums[i] > nums[i - 1]) {
                break;
            }
        }
        if (i == 0) {
            reverse(nums, 0, n - 1);
        } else {
            i--;
            int var1 = i + 1;
            for (int j = i + 1; j < n; j++) {
                if (nums[j] > nums[i]) {
                    var1 = j;
                } else {
                    break;
                }
            }
            int temp = nums[i];
            nums[i] = nums[var1];
            nums[var1] = temp;
            reverse(nums, i + 1, n - 1);
        }
    }
}
