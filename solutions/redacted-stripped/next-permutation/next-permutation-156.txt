class Solution {

    public void method1(int[] nums) {
        int f1 = -1;
        int f2 = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                f1 = i - 1;
                break;
            }
        }
        if (f1 == -1) {
            reverse(nums, 0);
            return;
        }
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[f1]) {
                f2 = i;
                break;
            }
        }
        swap(nums, f1, f2);
        reverse(nums, f1 + 1);
    }

    public void reverse(int[] nums, int s) {
        int i = s, j = nums.length - 1;
        while (i < j) {
            swap(nums, i, j);
            i++;
            j--;
        }
    }

    public void swap(int[] nums, int i, int j) {
        int t = nums[i];
        nums[i] = nums[j];
        nums[j] = t;
    }
}
