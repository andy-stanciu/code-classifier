class Solution {

    public void method1(int[] nums) {
        int n = nums.length - 1;
        int i = n;
        for (; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                break;
            }
        }
        if (i != 0) {
            int swapIndx = i;
            for (int j = n; j >= i; j--) {
                if (nums[j] > nums[i - 1]) {
                    swapIndx = j;
                    break;
                }
            }
            swap(nums, i - 1, swapIndx);
            reverse(nums, i, n);
        } else {
            reverse(nums, 0, n);
        }
    }

    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }

    public void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }
}
