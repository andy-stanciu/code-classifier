class Solution {

    public void method1(int[] arr) {
        int n = arr.length;
        int k = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (arr[i] < arr[i + 1]) {
                k = i;
                break;
            }
        }
        if (k == -1) {
            reverse(arr, 0, n - 1);
            return;
        }
        int l = n - 1;
        while (arr[l] <= arr[k]) {
            l--;
        }
        int temp = arr[k];
        arr[k] = arr[l];
        arr[l] = temp;
        reverse(arr, k + 1, n - 1);
    }

    public void reverse(int[] arr, int start, int end) {
        while (start < end) {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }
}
