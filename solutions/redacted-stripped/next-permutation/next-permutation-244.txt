class Solution {

    public void method1(int[] nums) {
        int pidx = -1;
        for (int i = nums.length - 1; i >= 1; i--) {
            if (nums[i - 1] < nums[i]) {
                pidx = i - 1;
                break;
            }
        }
        if (pidx == -1) {
            reverse(nums, 0, nums.length - 1);
            return;
        }
        for (int i = nums.length - 1; i > pidx; i--) {
            if (nums[i] > nums[pidx]) {
                int temp = nums[i];
                nums[i] = nums[pidx];
                nums[pidx] = temp;
                break;
            }
        }
        reverse(nums, pidx + 1, nums.length - 1);
        return;
    }

    public void reverse(int[] arr, int s, int e) {
        while (s < e) {
            int temp = arr[s];
            arr[s] = arr[e];
            arr[e] = temp;
            s++;
            e--;
        }
    }
}
