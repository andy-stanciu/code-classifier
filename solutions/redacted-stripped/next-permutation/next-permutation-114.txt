class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int a = 0;
        int b = 0;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                a = i;
                break;
            }
        }
        for (int i = n - 1; i >= 0; i--) {
            if (nums[i] > nums[a]) {
                b = i;
                break;
            }
        }
        if (a == b) {
            Arrays.sort(nums);
            return;
        }
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
        int left = a + 1;
        int right = n - 1;
        while (left < right) {
            int t = nums[left];
            nums[left] = nums[right];
            nums[right] = t;
            left++;
            right--;
        }
    }
}
