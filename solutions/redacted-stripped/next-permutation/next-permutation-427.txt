class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int i = 0, j = 0;
        for (i = n - 2; i >= 0 && nums[i] >= nums[i + 1]; i--) {
        }
        if (i >= 0) {
            for (j = n - 1; nums[j] <= nums[i]; j--) {
            }
            swap(nums, i, j);
        }
        reverse(nums, i + 1, n - 1);
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start++, end--);
        }
    }
}
