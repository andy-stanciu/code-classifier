class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int idx = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                idx = i;
                break;
            }
        }
        if (idx == -1) {
            reverse(nums, 0);
            return;
        }
        for (int i = n - 1; i > idx; i--) {
            if (nums[i] > nums[idx]) {
                swap(i, idx, nums);
                break;
            }
        }
        reverse(nums, idx + 1);
    }

    public void swap(int i, int idx, int[] nums) {
        int temp = nums[i];
        nums[i] = nums[idx];
        nums[idx] = temp;
    }

    public void reverse(int[] nums, int start) {
        int s = start, e = nums.length - 1;
        while (s <= e) {
            swap(s, e, nums);
            s++;
            e--;
        }
    }
}
