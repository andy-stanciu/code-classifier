class Solution {

    void reverse(int[] nums, int i) {
        for (int j = nums.length - 1; i < j; i++, j--) {
            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
        }
    }

    public void method1(int[] nums) {
        int idx = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i - 1] < nums[i]) {
                idx = i - 1;
                break;
            }
        }
        if (idx != -1) {
            int jdx = idx;
            for (int j = nums.length - 1; j >= idx + 1; j--) {
                if (nums[j] > nums[idx]) {
                    jdx = j;
                    break;
                }
            }
            int temp = nums[idx];
            nums[idx] = nums[jdx];
            nums[jdx] = temp;
        }
        reverse(nums, idx + 1);
    }
}
