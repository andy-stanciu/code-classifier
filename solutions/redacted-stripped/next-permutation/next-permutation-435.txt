class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int i = n - 2;
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            i--;
        }
        if (i == -1) {
            reverseArray(nums, 0);
            return;
        }
        for (int j = n - 1; j > i; j--) {
            if (nums[j] > nums[i]) {
                int tmp = nums[i];
                nums[i] = nums[j];
                nums[j] = tmp;
                break;
            }
        }
        reverseArray(nums, i + 1);
    }

    private void reverseArray(int[] nums, int startIndex) {
        int endIndex = nums.length - 1;
        while (startIndex < endIndex) {
            int tmp = nums[startIndex];
            nums[startIndex] = nums[endIndex];
            nums[endIndex] = tmp;
            startIndex++;
            endIndex--;
        }
    }
}
