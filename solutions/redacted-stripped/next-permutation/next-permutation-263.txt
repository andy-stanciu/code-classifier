class Solution {

    public void method1(int[] nums) {
        int[] counts = new int[101];
        int target = -1;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (target == -1) {
                counts[nums[i]]++;
                if (i < nums.length - 1) {
                    if (nums[i] < nums[i + 1]) {
                        target = i;
                        break;
                    }
                }
            }
        }
        if (target != -1) {
            int min = nums[target] + 1;
            while (counts[min] == 0 && min < 101) {
                min++;
            }
            nums[target] = min;
            counts[min]--;
        }
        target++;
        for (int i = 0; i < 101; i++) {
            while (counts[i] != 0) {
                nums[target] = i;
                counts[i]--;
                target++;
            }
        }
        return;
    }
}
