class Solution {

    public void method1(int[] nums) {
        int len = nums.length;
        int peak = len - 1;
        while (peak > 0) {
            if (nums[peak - 1] < nums[peak]) {
                break;
            }
            peak--;
        }
        if (peak == 0) {
            int mid = len / 2;
            for (int i = 0; i < mid; i++) {
                int temp = nums[i];
                nums[i] = nums[(len - 1) - i];
                nums[(len - 1) - i] = temp;
            }
            return;
        }
        int swap = nums[peak - 1];
        int min = 100;
        int minIndex = -1;
        int start = peak;
        while (start <= len - 1) {
            int temp = nums[start];
            if (temp > swap && temp < min) {
                min = temp;
                minIndex = start;
            }
            start++;
        }
        nums[peak - 1] = nums[minIndex];
        nums[minIndex] = swap;
        Arrays.sort(nums, peak, nums.length);
    }
}
