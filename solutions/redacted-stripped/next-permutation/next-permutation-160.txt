class Solution {

    public void method1(int[] nums) {
        int pointer = nums.length - 1;
        while (pointer > 0 && nums[pointer] <= nums[pointer - 1]) {
            pointer--;
        }
        if (pointer == 0 || nums.length == 1) {
            Arrays.sort(nums);
        } else {
            int right = nums.length - 1;
            while (right > pointer && nums[right] <= nums[pointer - 1]) {
                right--;
            }
            int temp = nums[pointer - 1];
            nums[pointer - 1] = nums[right];
            nums[right] = temp;
            int start = pointer;
            int end = nums.length - 1;
            while (start < end) {
                temp = nums[start];
                nums[start] = nums[end];
                nums[end] = temp;
                start++;
                end--;
            }
        }
    }
}
