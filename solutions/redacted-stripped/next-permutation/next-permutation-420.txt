class Solution {

    public void method1(int[] nums) {
        boolean flag = false;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                int min = getRes(nums, i, i + 1);
                swap(nums, i, min);
                reverse(nums, i + 1);
                flag = true;
                break;
            }
        }
        if (!flag) {
            reverse(nums, 0);
        }
    }

    public static int getRes(int[] nums, int i, int min) {
        for (int j = i + 1; j < nums.length; j++) {
            if (nums[i] < nums[j] && nums[j] <= nums[min]) {
                min = j;
            }
        }
        return min;
    }

    public static void reverse(int[] nums, int start) {
        int left = start;
        int right = nums.length - 1;
        while (left < right) {
            swap(nums, right, left);
            left++;
            right--;
        }
    }

    public static void swap(int[] nums, int i, int j) {
        int temp = nums[j];
        nums[j] = nums[i];
        nums[i] = temp;
    }
}
