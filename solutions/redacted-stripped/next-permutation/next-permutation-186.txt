class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int i = n - 2;
        int pos = -1;
        if (n > 1)
            while (i >= 0) {
                if (nums[i] < nums[i + 1]) {
                    pos = i;
                    break;
                }
                i--;
            }
        if (pos != -1)
            for (int j = n - 1; j > pos; j--) {
                if (nums[pos] < nums[j]) {
                    int temp = nums[pos];
                    nums[pos] = nums[j];
                    nums[j] = temp;
                    break;
                }
            }
        int start = pos + 1;
        int end = n - 1;
        while (start <= end) {
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }
}
