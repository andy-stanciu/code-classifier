class Solution {

    public static int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        for (int l = 0, r = 0, var1 = 0; r < nums.length; r++) {
            var1 += nums[r];
            while (var1 - nums[l] >= target) {
                var1 -= nums[l++];
            }
            if (var1 >= target) {
                ans = Math.min(ans, r - l + 1);
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
