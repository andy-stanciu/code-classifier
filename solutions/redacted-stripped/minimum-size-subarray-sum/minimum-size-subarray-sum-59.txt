class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int var1 = 0;
        int res = Integer.MAX_VALUE;
        for (right = 0; right < nums.length; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                res = Math.min(res, right - left + 1);
                var1 -= nums[left++];
            }
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}
