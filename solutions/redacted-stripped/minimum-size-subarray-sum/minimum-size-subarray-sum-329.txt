class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        while (right < nums.length) {
            var1 += nums[right];
            while (left <= right && var1 >= target) {
                min = Math.min(min, right - left + 1);
                var1 -= nums[left++];
            }
            right++;
        }
        return min > nums.length ? 0 : min;
    }
}
