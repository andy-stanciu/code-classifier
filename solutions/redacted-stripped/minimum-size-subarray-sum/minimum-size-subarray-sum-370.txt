class Solution {

    public int method1(int target, int[] nums) {
        int end = 0;
        int start = 0;
        int var1 = 0;
        int ans = Integer.MAX_VALUE;
        while (end < nums.length) {
            var1 += nums[end];
            while (var1 >= target && start <= end) {
                ans = Math.min(ans, end - start + 1);
                var1 -= nums[start++];
            }
            end++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
