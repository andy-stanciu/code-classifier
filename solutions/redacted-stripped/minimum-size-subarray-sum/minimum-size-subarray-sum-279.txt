class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE, i = 0;
        int var1 = 0;
        for (int j = 0; j < nums.length; j++) {
            var1 += nums[j];
            while (var1 >= target) {
                ans = Math.min(ans, j - i + 1);
                var1 -= nums[i++];
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
