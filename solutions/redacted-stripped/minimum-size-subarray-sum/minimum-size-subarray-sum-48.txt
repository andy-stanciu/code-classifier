class Solution {

    public int method1(int target, int[] nums) {
        int ans = nums.length + 1;
        int var1 = 0;
        int st = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target && st <= i) {
                ans = Math.min(ans, i - st + 1);
                var1 -= nums[st++];
            }
        }
        return ans == nums.length + 1 ? 0 : ans;
    }
}
