class Solution {

    public int method1(int target, int[] nums) {
        int j = 0;
        int var1 = 0;
        int var2 = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                var2 = Math.min(var2, i - j + 1);
                var1 -= nums[j++];
            }
        }
        return var2 == Integer.MAX_VALUE ? 0 : var2;
    }
}
