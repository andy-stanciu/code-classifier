class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int min = nums.length + 1;
        int var1 = 0;
        for (j = 0; j < nums.length; j++) {
            var1 += nums[j];
            while (var1 >= target) {
                min = Math.min(min, j - i + 1);
                var1 -= nums[i];
                i++;
            }
        }
        if (min <= nums.length) {
            return min;
        }
        return 0;
    }
}
