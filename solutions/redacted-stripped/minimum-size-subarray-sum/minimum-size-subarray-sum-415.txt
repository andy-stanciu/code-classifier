class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int ps = 0, i = 0, ans = Integer.MAX_VALUE;
        for (int j = 0; j < n; j++) {
            ps += nums[j];
            while (ps >= target) {
                ans = Math.min(j - i + 1, ans);
                ps -= nums[i++];
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
