class Solution {

    public int method1(int target, int[] nums) {
        int front = 0, end = 0;
        int min = Integer.MAX_VALUE;
        int var1 = 0;
        while (end < nums.length) {
            var1 += nums[end++];
            while (front < end && var1 >= target) {
                min = Math.min(end - front, min);
                var1 -= nums[front++];
            }
        }
        if (min == Integer.MAX_VALUE) {
            min = 0;
        }
        return min;
    }
}
