class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = 0, var1 = 0, min = Integer.MAX_VALUE;
        while (i < nums.length && j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                min = Math.min(min, j - i + 1);
                var1 -= nums[i++];
            }
            j++;
        }
        return (min == Integer.MAX_VALUE) ? 0 : min;
    }
}
