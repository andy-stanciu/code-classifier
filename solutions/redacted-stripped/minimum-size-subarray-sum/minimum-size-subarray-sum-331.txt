class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = 0, res = Integer.MAX_VALUE, var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                res = Math.min(res, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        if (res == Integer.MAX_VALUE)
            return 0;
        return res;
    }
}
