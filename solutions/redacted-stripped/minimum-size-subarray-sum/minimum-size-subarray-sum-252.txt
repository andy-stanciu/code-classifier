class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int output = Integer.MAX_VALUE;
        int leftPointer = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                output = Math.min(output, i + 1 - leftPointer);
                var1 -= nums[leftPointer];
                leftPointer++;
            }
        }
        return output < Integer.MAX_VALUE ? output : 0;
    }
}
