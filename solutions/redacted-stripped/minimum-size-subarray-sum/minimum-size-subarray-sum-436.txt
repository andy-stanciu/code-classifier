class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, r = 0, n = nums.length, ans = n + 1;
        long var1 = 0;
        for (; r < n; r++) {
            var1 += nums[r];
            while (l <= r && var1 >= target) {
                ans = Math.min(ans, r - l + 1);
                var1 -= nums[l];
                l++;
            }
        }
        return ans == n + 1 ? 0 : ans;
    }
}
