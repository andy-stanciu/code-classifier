class Solution {

    public int method1(int S, int[] arr) {
        int var1 = 0, minLength = Integer.MAX_VALUE;
        int windowStart = 0;
        for (int windowEnd = 0; windowEnd < arr.length; windowEnd++) {
            var1 += arr[windowEnd];
            while (var1 >= S) {
                minLength = Math.min(minLength, windowEnd - windowStart + 1);
                var1 -= arr[windowStart];
                windowStart++;
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
