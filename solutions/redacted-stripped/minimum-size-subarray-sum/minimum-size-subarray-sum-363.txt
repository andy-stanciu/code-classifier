class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int var1 = 0;
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                ans = Math.min(i - j + 1, ans);
                var1 -= nums[j++];
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
