class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int min = Integer.MAX_VALUE;
        int var1 = 0;
        for (int left = 0, right = 0; right < n; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                min = Math.min(min, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        return (min == Integer.MAX_VALUE) ? 0 : min;
    }
}
