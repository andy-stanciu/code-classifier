class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int var1 = 0;
        int len = Integer.MAX_VALUE;
        for (int j = 0; j < nums.length; j++) {
            var1 += nums[j];
            while (var1 >= target) {
                len = Math.min(len, j - i + 1);
                var1 -= nums[i];
                i++;
            }
        }
        if (len == Integer.MAX_VALUE)
            return 0;
        return len;
    }
}
