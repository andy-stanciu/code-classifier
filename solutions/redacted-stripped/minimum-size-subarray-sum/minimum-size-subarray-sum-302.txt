class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, r = 0, var1 = 0, min = Integer.MAX_VALUE;
        while (r < nums.length) {
            var1 = var1 + nums[r];
            while (var1 >= target) {
                var1 -= nums[l];
                min = Math.min(min, r - l + 1);
                l++;
            }
            r++;
        }
        return (min == Integer.MAX_VALUE) ? 0 : min;
    }
}
