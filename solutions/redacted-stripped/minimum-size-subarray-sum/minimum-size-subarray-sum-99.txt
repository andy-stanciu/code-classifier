class Solution {

    public int method1(int target, int[] nums) {
        int minimal_length = nums.length + 1;
        int L = 0, R = 0, var1 = 0;
        while (R < nums.length) {
            var1 += nums[R];
            while (var1 >= target) {
                minimal_length = Math.min(minimal_length, R - L + 1);
                var1 -= nums[L];
                L++;
            }
            R++;
        }
        return minimal_length == nums.length + 1 ? 0 : minimal_length;
    }
}
