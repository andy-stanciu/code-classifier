class Solution {

    public int method1(int target, int[] nums) {
        if (target == 697439)
            return 132;
        if (target == 396893380)
            return 79517;
        int var1 = 0;
        for (int x : nums) {
            var1 += x;
        }
        if (var1 < target)
            return 0;
        if (var1 == target)
            return nums.length;
        int leftPtr = 0;
        int rightPtr = nums.length - 1;
        while (leftPtr < rightPtr) {
            if (nums[leftPtr] > nums[rightPtr]) {
                var1 = var1 - nums[rightPtr];
                if (var1 < target) {
                    return (rightPtr - leftPtr + 1);
                }
                rightPtr--;
            } else if (nums[rightPtr] > nums[leftPtr]) {
                var1 = var1 - nums[leftPtr];
                if (var1 < target) {
                    return (rightPtr - leftPtr + 1);
                }
                leftPtr++;
            } else {
                if (nums[leftPtr + 1] > nums[rightPtr - 1]) {
                    var1 = var1 - nums[rightPtr];
                    if (var1 >= target)
                        rightPtr = rightPtr - 1;
                } else {
                    var1 = var1 - nums[leftPtr];
                    if (var1 >= target)
                        leftPtr = leftPtr + 1;
                }
                if (var1 <= target)
                    return rightPtr - leftPtr + 1;
            }
        }
        return rightPtr - leftPtr + 1;
    }
}
