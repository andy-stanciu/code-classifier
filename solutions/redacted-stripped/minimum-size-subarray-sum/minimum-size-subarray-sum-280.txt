class Solution {

    public static int method1(int target, int[] nums) {
        int l = 0;
        int minLen = Integer.MAX_VALUE;
        int var1 = 0;
        for (int r = 0; r < nums.length; r++) {
            var1 += nums[r];
            while (var1 >= target) {
                minLen = Math.min(minLen, r - l + 1);
                var1 -= nums[l];
                l++;
            }
        }
        if (minLen == Integer.MAX_VALUE) {
            return 0;
        }
        return minLen;
    }
}
