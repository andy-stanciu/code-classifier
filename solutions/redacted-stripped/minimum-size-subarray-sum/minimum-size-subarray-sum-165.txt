class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = 0, t = 0, res = Integer.MAX_VALUE;
        boolean flag = false;
        while (i < nums.length) {
            t += nums[i];
            if (t >= target) {
                flag = true;
                res = Math.min(i - j + 1, res);
                i = j;
                ++j;
                t = 0;
            }
            ++i;
        }
        if (flag != false)
            return res;
        else
            return 0;
    }
}
