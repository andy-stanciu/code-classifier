class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int minimalLength = Integer.MAX_VALUE;
        int var1 = 0;
        while (j < nums.length) {
            if (var1 >= target) {
                minimalLength = Math.min(minimalLength, j - i);
                var1 -= nums[i++];
            } else {
                var1 += nums[j++];
            }
        }
        while (i < nums.length) {
            if (var1 >= target) {
                minimalLength = Math.min(minimalLength, j - i);
                var1 -= nums[i];
            }
            i++;
        }
        return minimalLength == Integer.MAX_VALUE ? 0 : minimalLength;
    }
}
