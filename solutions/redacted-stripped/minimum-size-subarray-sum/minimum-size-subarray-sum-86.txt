class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int i = 0, j = 0, var1 = 0, result = Integer.MAX_VALUE;
        while (j < n) {
            var1 += nums[j];
            while (var1 >= target) {
                result = Math.min(result, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
