class Solution {

    public int method1(int target, int[] nums) {
        int l = nums.length;
        int var1 = 0;
        Integer min = Integer.MAX_VALUE;
        int i = 0, j = 0;
        while (j < l) {
            var1 += nums[j];
            while (var1 >= target) {
                min = Math.min(min, j - i + 1);
                var1 -= nums[i++];
            }
            j++;
        }
        return min.equals(Integer.MAX_VALUE) ? 0 : min;
    }
}
