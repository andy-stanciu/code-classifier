class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int window = 0, left = 0, ans = n + 1;
        for (int i = 0; i < nums.length; i++) {
            window += nums[i];
            while (window >= target) {
                ans = Math.min(ans, i - left + 1);
                window -= nums[left++];
            }
        }
        return ans == n + 1 ? 0 : ans;
    }
}
