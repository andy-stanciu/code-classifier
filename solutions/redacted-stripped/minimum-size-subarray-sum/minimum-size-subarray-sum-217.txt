class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int var1 = 0;
        int len = Integer.MAX_VALUE;
        for (int right = 0; right < n; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                len = Math.min(len, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        return len == Integer.MAX_VALUE ? 0 : len;
    }
}
