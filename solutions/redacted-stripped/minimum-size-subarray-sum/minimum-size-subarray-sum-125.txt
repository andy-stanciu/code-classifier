class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0, i = 0, j = 0, min = Integer.MAX_VALUE;
        while (i < nums.length) {
            var1 += nums[i];
            while (var1 >= target && j <= i) {
                min = Math.min(min, i - j + 1);
                var1 -= nums[j];
                j++;
            }
            i++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
