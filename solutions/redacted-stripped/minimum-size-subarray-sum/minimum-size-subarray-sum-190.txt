class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int start = 0;
        int end = 0;
        int minLength = Integer.MAX_VALUE;
        for (; end < nums.length; end++) {
            var1 += nums[end];
            while (var1 >= target) {
                minLength = Math.min(minLength, end - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
