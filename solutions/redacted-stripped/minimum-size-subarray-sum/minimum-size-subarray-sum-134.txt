class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int var1 = 0;
        int ans = Integer.MAX_VALUE;
        for (int r = 0; r < nums.length; r++) {
            var1 += nums[r];
            while (var1 >= target) {
                ans = Math.min(ans, r - l + 1);
                var1 -= nums[l];
                l++;
            }
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        } else {
            return ans;
        }
    }
}
