class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int var1 = 0;
        int answer = Integer.MAX_VALUE;
        int len = 0;
        while (i < nums.length) {
            var1 += nums[i];
            len++;
            while (var1 >= target) {
                answer = Math.min(answer, len);
                var1 -= nums[j];
                j++;
                len--;
            }
            i++;
        }
        return answer == Integer.MAX_VALUE ? 0 : answer;
    }
}
