class Solution {

    public int method1(int target, int[] nums) {
        int minWindow = Integer.MAX_VALUE;
        int var1 = 0;
        int low = 0;
        int high = 0;
        while (high < nums.length) {
            var1 += nums[high];
            high++;
            while (var1 >= target) {
                int window = high - low;
                minWindow = Math.min(minWindow, window);
                var1 -= nums[low];
                low++;
            }
        }
        return minWindow == Integer.MAX_VALUE ? 0 : minWindow;
    }
}
