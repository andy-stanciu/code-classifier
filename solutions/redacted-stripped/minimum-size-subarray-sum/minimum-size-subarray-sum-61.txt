class Solution {

    public int method1(int t, int[] num) {
        int i, n = num.length, var1 = 0, lptr = 0, len = Integer.MAX_VALUE;
        for (i = 0; i < n; i++) {
            var1 += num[i];
            while (var1 >= t) {
                len = Math.min(len, i - lptr + 1);
                var1 -= num[lptr];
                lptr++;
            }
        }
        return len == Integer.MAX_VALUE ? 0 : len;
    }
}
