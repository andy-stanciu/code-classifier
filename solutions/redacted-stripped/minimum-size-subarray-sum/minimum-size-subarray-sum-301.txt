class Solution {

    public int method1(int k, int[] arr) {
        int l = 0;
        int r;
        int minc = Integer.MAX_VALUE;
        int var1 = 0;
        int n = arr.length;
        for (r = 0; r < n; r++) {
            var1 += arr[r];
            while (var1 >= k) {
                minc = Math.min(minc, r - l + 1);
                var1 -= arr[l];
                l++;
            }
        }
        return minc == Integer.MAX_VALUE ? 0 : minc;
    }
}
