class Solution {

    public int method1(int target, int[] nums) {
        if (target == 0 || nums.length == 0) {
            return 0;
        }
        int x = 0, y = 0, total = 0, min = Integer.MAX_VALUE;
        while (y < nums.length) {
            total += nums[y++];
            while (total >= target) {
                min = Math.min(min, y - x);
                total -= nums[x++];
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
