class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int ans = Integer.MAX_VALUE;
        int var1 = 0;
        int start = 0;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                ans = Math.min(ans, i - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
