class Solution {

    public int method1(int target, int[] nums) {
        int left = 0, right = 0;
        int result = Integer.MAX_VALUE, var1 = 0;
        while (right < nums.length) {
            var1 += nums[right++];
            while (var1 >= target) {
                result = Math.min(result, right - left);
                var1 = var1 - nums[left++];
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
