class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int res = n + 1, l = 0, var1 = 0;
        for (int r = 0; r < n; r++) {
            var1 += nums[r];
            while (var1 >= target) {
                res = Math.min(res, r - l + 1);
                var1 -= nums[l];
                l++;
            }
        }
        return res == n + 1 ? 0 : res;
    }
}
