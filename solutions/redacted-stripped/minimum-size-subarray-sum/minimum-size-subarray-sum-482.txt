class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        while (j < nums.length) {
            var1 += nums[j];
            j++;
            while (var1 >= target) {
                min = Math.min(j - i, min);
                var1 = var1 - nums[i];
                i++;
            }
        }
        if (min == Integer.MAX_VALUE)
            return 0;
        return min;
    }
}
