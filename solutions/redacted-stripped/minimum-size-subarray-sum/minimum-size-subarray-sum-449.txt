class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int left = 0;
        int right = 0;
        int min = Integer.MAX_VALUE;
        int n = nums.length;
        while (left < n) {
            while (right < n && var1 < target) {
                var1 += nums[right];
                right++;
            }
            if (var1 >= target) {
                min = Math.min(min, right - left);
            }
            var1 -= nums[left];
            left++;
        }
        if (min == Integer.MAX_VALUE) {
            return 0;
        }
        return min;
    }
}
