class Solution {

    public int method1(int target, int[] nums) {
        int minlen = Integer.MAX_VALUE;
        int var1 = 0, left = 0, right = 0;
        while (right < nums.length) {
            var1 += nums[right];
            while (var1 >= target) {
                minlen = Math.min(minlen, right - left + 1);
                var1 -= nums[left];
                left++;
            }
            right++;
        }
        return minlen != Integer.MAX_VALUE ? minlen : 0;
    }
}
