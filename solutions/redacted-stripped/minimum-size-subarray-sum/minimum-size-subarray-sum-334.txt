public class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int answer = n + 1;
        int left = 0;
        int total = 0;
        for (int right = 0; right < n; right++) {
            total += nums[right];
            while (total >= target) {
                answer = Math.min(answer, right - left + 1);
                total -= nums[left];
                left++;
            }
        }
        return answer == n + 1 ? 0 : answer;
    }
}
