class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int start = 0;
        int end = n - 1;
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        int left = 0;
        while (start <= end) {
            var1 += nums[start];
            while (var1 >= target) {
                min = Math.min(min, start - left + 1);
                var1 -= nums[left];
                left++;
            }
            start++;
        }
        return (min == Integer.MAX_VALUE) ? 0 : min;
    }
}
