class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int i = 0;
        int j = 0;
        int minl = n + 1;
        int var1 = 0;
        while (j < n) {
            var1 += nums[j];
            while (var1 >= target) {
                minl = Math.min(minl, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return (minl != n + 1) ? minl : 0;
    }
}
