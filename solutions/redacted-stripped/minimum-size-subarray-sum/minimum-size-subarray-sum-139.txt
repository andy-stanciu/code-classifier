class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int method1 = Integer.MAX_VALUE;
        int var1 = 0;
        int left = 0;
        for (int right = 0; right < n; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                method1 = Math.min(method1, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        return (method1 != Integer.MAX_VALUE) ? method1 : 0;
    }
}
