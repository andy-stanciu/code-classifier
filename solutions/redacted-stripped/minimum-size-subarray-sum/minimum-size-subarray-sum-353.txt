class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int len = nums.length;
        for (int num : nums) var1 = var1 + num;
        if (var1 < target)
            return 0;
        if (var1 == target)
            return len;
        int i = 0, j = 0;
        var1 = 0;
        int minLen = Integer.MAX_VALUE;
        while (i < len && j < len) {
            var1 = var1 + nums[j];
            while (var1 >= target) {
                minLen = Math.min(minLen, j - i + 1);
                var1 = var1 - nums[i];
                i++;
            }
            j++;
        }
        return minLen;
    }
}
