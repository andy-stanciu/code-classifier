class Solution {

    public int method1(int target, int[] nums) {
        int len = nums.length;
        int min_l = Integer.MAX_VALUE;
        int i = 0, j = 0;
        int var1 = 0;
        while (j < len) {
            var1 += nums[j];
            while (var1 >= target) {
                min_l = Math.min(min_l, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return min_l == Integer.MAX_VALUE ? 0 : min_l;
    }
}
