class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int i = 0, j = 0;
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        while (j < n) {
            var2 += nums[j];
            while (var2 >= target) {
                var1 = Math.min(var1, j - i + 1);
                var2 -= nums[i];
                i++;
            }
            j++;
        }
        return var1 == Integer.MAX_VALUE ? 0 : var1;
    }
}
