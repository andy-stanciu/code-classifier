class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int left = 0, var1 = 0, min = n + 1;
        for (int right = 0; right < n; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                min = Math.min(min, right - left + 1);
                var1 -= nums[left++];
            }
        }
        return min == n + 1 ? 0 : min;
    }
}
