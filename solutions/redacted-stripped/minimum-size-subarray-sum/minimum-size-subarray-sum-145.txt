class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, r = 0, var1 = 0, ans = Integer.MAX_VALUE;
        boolean noAns = true;
        while (r < nums.length) {
            var1 += nums[r];
            while (var1 >= target) {
                ans = Math.min(r - l + 1, ans);
                var1 -= nums[l++];
                noAns = false;
            }
            r++;
        }
        if (noAns)
            return 0;
        return ans;
    }
}
