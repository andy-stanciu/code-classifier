class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int length = Integer.MAX_VALUE;
        int i = 0, j = 0;
        boolean found = false;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                found = true;
                var1 = var1 - nums[i];
                length = Math.min(length, j - i + 1);
                i++;
            }
            j++;
        }
        if (found == false) {
            return 0;
        }
        return length;
    }
}
