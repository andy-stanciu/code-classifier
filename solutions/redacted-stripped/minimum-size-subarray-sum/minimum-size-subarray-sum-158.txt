class Solution {

    public int method1(int target, int[] nums) {
        int var1 = Integer.MAX_VALUE;
        int var2 = 0, left = 0;
        for (int i = 0; i < nums.length; i++) {
            var2 += nums[i];
            while (target <= var2) {
                var1 = Math.min((i - left + 1), var1);
                var2 -= nums[left];
                left++;
            }
        }
        return var1 == Integer.MAX_VALUE ? 0 : var1;
    }
}
