class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int j = -1;
        int var1 = 0;
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                j++;
                res = Math.min(res, i - j + 1);
                var1 -= nums[j];
            }
        }
        return (res != Integer.MAX_VALUE) ? res : 0;
    }
}
