class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int ind = Integer.MAX_VALUE;
        int left = 0;
        for (int right = 0; right < nums.length; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                ind = Math.min(ind, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        return ind == Integer.MAX_VALUE ? 0 : ind;
    }
}
