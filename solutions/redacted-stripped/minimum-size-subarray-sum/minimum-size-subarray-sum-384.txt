class Solution {

    public int method1(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int fr = 0;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                min = Math.min(min, i - fr + 1);
                var1 -= nums[fr];
                fr++;
            }
        }
        if (min != Integer.MAX_VALUE) {
            return min;
        }
        return 0;
    }
}
