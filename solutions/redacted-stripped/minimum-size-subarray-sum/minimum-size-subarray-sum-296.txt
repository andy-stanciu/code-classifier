class Solution {

    public int method1(int target, int[] nums) {
        int minLen = Integer.MAX_VALUE;
        int l = 0, r = 0, var1 = 0;
        while (r < nums.length) {
            var1 += nums[r++];
            while (var1 >= target) {
                minLen = Math.min(minLen, r - l);
                var1 -= nums[l++];
            }
        }
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
