class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int left = 0;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                ans = Math.min(ans, i + 1 - left);
                var1 -= nums[left++];
            }
        }
        return ans != Integer.MAX_VALUE ? ans : 0;
    }
}
