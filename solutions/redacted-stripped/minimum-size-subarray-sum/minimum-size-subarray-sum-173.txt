class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int i = 0, j = 0;
        int var1 = 0;
        while (i < nums.length) {
            var1 += nums[i];
            while (var1 >= target) {
                ans = Math.min(ans, i - j + 1);
                var1 -= nums[j];
                j++;
            }
            i++;
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
