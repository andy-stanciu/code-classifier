class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, r = 0, minLen = nums.length, len = 0, var1 = 0;
        boolean found = false;
        while (r < nums.length) {
            var1 += nums[r];
            if (var1 < target) {
                r++;
            } else if (var1 >= target) {
                found = true;
                while (l <= r && var1 >= target) {
                    minLen = Math.min(minLen, r - l + 1);
                    var1 -= nums[l];
                    l++;
                }
                r++;
            }
        }
        if (!found) {
            return 0;
        }
        return minLen;
    }
}
