class Solution {

    public int method1(int target, int[] nums) {
        int start = 0, end = 0, var1 = 0;
        int count = Integer.MAX_VALUE;
        while (end < nums.length) {
            var1 += nums[end];
            while (var1 >= target) {
                count = Math.min(count, end - start + 1);
                var1 -= nums[start++];
            }
            end++;
        }
        return (count == Integer.MAX_VALUE) ? 0 : count;
    }
}
