class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        int len = 0;
        int start = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 = var1 + nums[i];
            if (var1 >= target) {
                min = Math.min(i - start + 1, min);
                var1 = var1 - nums[start];
                start++;
                while (var1 > target) {
                    var1 = var1 - nums[start];
                    start++;
                }
                if (var1 == target) {
                } else {
                    start--;
                    var1 = var1 + nums[start];
                }
                min = Math.min(i - start + 1, min);
            }
        }
        if (min == Integer.MAX_VALUE) {
            return 0;
        }
        return min;
    }
}
