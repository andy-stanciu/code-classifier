class Solution {

    public int method1(int target, int[] nums) {
        int ans = nums.length + 1, j = 0, var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 = var1 + nums[i];
            if (var1 >= target) {
                while (var1 >= target) {
                    var1 -= nums[j++];
                }
                ans = Math.min(ans, i - j + 2);
            }
        }
        return ans == nums.length + 1 ? 0 : ans;
    }
}
