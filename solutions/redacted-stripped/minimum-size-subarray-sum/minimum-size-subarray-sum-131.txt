class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE, l = 0, var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                ans = Math.min(ans, i - l + 1);
                var1 -= nums[l];
                l++;
            }
        }
        if (ans == Integer.MAX_VALUE)
            return 0;
        return ans;
    }
}
