class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length, var1 = 0, l = 0;
        int var2 = n + 1;
        for (int r = 0; r < n; r++) {
            var1 += nums[r];
            while (l < r && (var1 - nums[l]) >= target) {
                var1 -= nums[l];
                l++;
            }
            if (var1 >= target)
                var2 = Math.min(var2, r - l + 1);
        }
        return var2 == n + 1 ? 0 : var2;
    }
}
