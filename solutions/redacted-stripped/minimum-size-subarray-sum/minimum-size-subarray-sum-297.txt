class Solution {

    public int method1(int target, int[] nums) {
        if (nums.length == 1) {
            if (target <= nums[0]) {
                return 1;
            }
            return 0;
        }
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
        }
        if (var1 < target)
            return 0;
        int ans = Integer.MAX_VALUE;
        int i = 0;
        int j = -1;
        int var2 = 0;
        while (i < nums.length && j < nums.length) {
            if (var2 < target) {
                j++;
                if (j < nums.length) {
                    var2 += nums[j];
                }
            } else {
                ans = Math.min(j - i + 1, ans);
                var2 -= nums[i];
                i++;
            }
        }
        return ans;
    }
}
