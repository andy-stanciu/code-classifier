class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0, l = 0, r = 0;
        int minLen = Integer.MAX_VALUE;
        for (r = 0; r < nums.length; r++) {
            var1 = var1 + nums[r];
            while (var1 >= target) {
                minLen = Math.min(minLen, r - l + 1);
                var1 = var1 - nums[l];
                l = l + 1;
            }
        }
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
