class Solution {

    public static int method1(int target, int[] nums) {
        int start = 0, minLength = Integer.MAX_VALUE, currLength = 0, end = 0;
        int var1 = nums[0];
        while (true) {
            currLength = end - start + 1;
            if (var1 >= target) {
                if (start == end) {
                    return 1;
                }
                minLength = minLength > currLength ? currLength : minLength;
                var1 -= nums[start++];
            } else {
                if (end == nums.length - 1) {
                    break;
                }
                var1 += nums[++end];
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
