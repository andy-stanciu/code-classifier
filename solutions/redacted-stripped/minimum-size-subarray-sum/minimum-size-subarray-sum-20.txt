class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0, n = nums.length, ans = n + 1;
        for (int i = 0, j = 0; j < n; j++) {
            var1 += nums[j];
            while (var1 >= target) {
                ans = Math.min(ans, j - i + 1);
                var1 -= nums[i];
                i++;
            }
        }
        return ans == n + 1 ? 0 : ans;
    }
}
