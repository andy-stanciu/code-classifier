class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int var1 = 0, minLen = Integer.MAX_VALUE, l = 0, r = 0;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                minLen = Math.min(minLen, r - l + 1);
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        return (minLen == Integer.MAX_VALUE) ? 0 : minLen;
    }
}
