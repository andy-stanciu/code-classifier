class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int j = 0;
        int i = 0;
        int min = Integer.MAX_VALUE;
        while (j < nums.length) {
            var1 = var1 + nums[j];
            if (var1 < target) {
                j++;
            } else {
                min = Math.min(min, j - i + 1);
                while (var1 > target) {
                    var1 = var1 - nums[i];
                    if (var1 >= target) {
                        min = Math.min(min, j - i);
                    }
                    i++;
                }
                j++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
