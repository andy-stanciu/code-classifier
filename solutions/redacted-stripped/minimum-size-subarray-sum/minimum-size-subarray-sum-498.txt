class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int tw = 0;
        int j = 0;
        int sw = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (i == 0) {
                var1 = 0;
                tw = 0;
            } else {
                var1 -= nums[i - 1];
                tw--;
            }
            while (var1 < target && j < nums.length) {
                var1 += nums[j];
                tw++;
                j++;
            }
            if (var1 >= target) {
                sw = Math.min(sw, tw);
            }
        }
        if (sw == Integer.MAX_VALUE) {
            return 0;
        } else {
            return sw;
        }
    }
}
