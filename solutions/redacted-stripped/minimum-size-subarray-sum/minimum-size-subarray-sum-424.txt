class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int min = Integer.MAX_VALUE;
        int var1 = nums.length;
        int var2 = 0;
        while (j < var1) {
            var2 += nums[j];
            if (var2 < target)
                j++;
            else if (var2 == target) {
                if (j - i + 1 < min)
                    min = j - i + 1;
                j++;
            } else if (var2 > target) {
                if (j - i + 1 < min)
                    min = j - i + 1;
                while (var2 > target) {
                    if (j - i + 1 < min)
                        min = j - i + 1;
                    var2 -= nums[i];
                    i++;
                    if (var2 == target)
                        min = Math.min(j - i + 1, min);
                }
                j++;
            }
        }
        if (min == Integer.MAX_VALUE)
            return 0;
        return min;
    }
}
