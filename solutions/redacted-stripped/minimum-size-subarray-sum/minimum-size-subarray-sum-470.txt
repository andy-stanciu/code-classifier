class Solution {

    public int method1(int target, int[] nums) {
        int minLen = Integer.MAX_VALUE;
        int var1 = 0;
        int start = 0;
        for (int end = 0; end < nums.length; end++) {
            var1 += nums[end];
            while (var1 >= target) {
                minLen = Math.min(minLen, end - start + 1);
                var1 -= nums[start++];
            }
        }
        return minLen != Integer.MAX_VALUE ? minLen : 0;
    }
}
