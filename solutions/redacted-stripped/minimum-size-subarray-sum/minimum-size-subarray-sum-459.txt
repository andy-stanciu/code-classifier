class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int shortest = Integer.MAX_VALUE;
        int var1 = 0;
        int numLen = nums.length;
        while (right < numLen) {
            var1 = var1 + nums[right];
            if (var1 >= target) {
                while (var1 >= target) {
                    var1 = var1 - nums[left];
                    left++;
                }
                shortest = Math.min(shortest, right - left + 2);
            }
            right++;
        }
        return shortest == Integer.MAX_VALUE ? 0 : shortest;
    }
}
