class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int ans = Integer.MAX_VALUE;
        int st = 0;
        for (int ed = 0; ed < nums.length; ed++) {
            var1 = var1 + nums[ed];
            while (st <= ed) {
                if (var1 >= target) {
                    ans = Math.min(ans, ed - st + 1);
                    var1 = var1 - nums[st];
                    st++;
                } else {
                    break;
                }
            }
        }
        if (ans == Integer.MAX_VALUE)
            return 0;
        return ans;
    }
}
