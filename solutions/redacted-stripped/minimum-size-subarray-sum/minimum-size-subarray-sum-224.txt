class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, r = 0, var1 = 0, minlen = Integer.MAX_VALUE;
        while (r < nums.length) {
            var1 += nums[r];
            while (var1 >= target) {
                minlen = Math.min(minlen, r - l + 1);
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        return minlen == Integer.MAX_VALUE ? 0 : minlen;
    }
}
