class Solution {

    public int method1(int target, int[] nums) {
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        int left = 0;
        for (int right = 0; right < nums.length; right++) {
            var2 += nums[right];
            while (var2 >= target) {
                var1 = Math.min(var1, right - left + 1);
                var2 -= nums[left];
                left++;
            }
        }
        return var1 == Integer.MAX_VALUE ? 0 : var1;
    }
}
