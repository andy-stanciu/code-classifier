class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int var1 = 0;
        int miniL = Integer.MAX_VALUE;
        while (r < nums.length) {
            var1 += nums[r];
            while (l <= r && var1 >= target) {
                miniL = Math.min(miniL, r - l + 1);
                var1 -= nums[l++];
            }
            ++r;
        }
        return miniL == Integer.MAX_VALUE ? 0 : miniL;
    }
}
