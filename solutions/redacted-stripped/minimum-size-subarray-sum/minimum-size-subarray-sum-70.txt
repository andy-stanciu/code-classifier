class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int minSpan = Integer.MAX_VALUE;
        int endIndex = 0;
        int startIndex = 0;
        while (endIndex < nums.length) {
            if (var1 >= target) {
                minSpan = Math.min(minSpan, endIndex - startIndex);
                var1 -= nums[startIndex++];
            } else
                var1 += nums[endIndex++];
        }
        while (var1 >= target) {
            minSpan = Math.min(minSpan, endIndex - startIndex);
            var1 -= nums[startIndex++];
        }
        return minSpan % Integer.MAX_VALUE;
    }
}
