class Solution {

    public int method1(int target, int[] nums) {
        int start = 0;
        int end = 0;
        int length = nums.length;
        int minSub = length + 1;
        int var1 = 0;
        while (end < length) {
            var1 += nums[end++];
            while (var1 >= target) {
                minSub = (end - start < minSub) ? end - start : minSub;
                var1 -= nums[start++];
            }
        }
        if (minSub > length)
            return 0;
        return minSub;
    }
}
