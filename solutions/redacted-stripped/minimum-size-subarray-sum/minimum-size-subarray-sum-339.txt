class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int var1 = 0;
        int n = nums.length;
        int start = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (start < i && (var1 - nums[start]) >= target) {
                var1 -= nums[start];
                start++;
            }
            if (var1 >= target)
                ans = Math.min(ans, i - start + 1);
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
