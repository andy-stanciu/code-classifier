class Solution {

    public int method1(int target, int[] nums) {
        int start = 0, result = 0, var1 = 0;
        for (int end = 0; end < nums.length; end++) {
            var1 += nums[end];
            while (var1 >= target) {
                if (result == 0)
                    result = end - start + 1;
                else {
                    if (end - start + 1 < result)
                        result = end - start + 1;
                }
                var1 -= nums[start++];
            }
        }
        return result;
    }
}
