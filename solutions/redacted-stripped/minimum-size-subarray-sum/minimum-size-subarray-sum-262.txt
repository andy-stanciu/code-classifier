class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int result = Integer.MAX_VALUE;
        int var1 = 0;
        for (int j = 0; j < nums.length; j++) {
            var1 += nums[j];
            while (var1 >= target) {
                result = Math.min(result, j - i + 1);
                var1 -= nums[i++];
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
