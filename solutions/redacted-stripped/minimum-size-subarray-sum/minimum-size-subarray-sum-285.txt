class Solution {

    public int method1(int k, int[] nums) {
        int j = 0, i = 0;
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        int n = nums.length;
        while (j < n) {
            var1 += nums[j];
            while (var1 >= k) {
                min = Math.min(min, (j - i + 1));
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
