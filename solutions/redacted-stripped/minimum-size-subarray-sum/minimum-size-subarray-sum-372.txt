class Solution {

    public int method1(int target, int[] nums) {
        int start = 0;
        int end = 0;
        int var1 = 0;
        int len = 0;
        int n = nums.length;
        int minlen = n + 1;
        while (end < n) {
            if (var1 < target) {
                var1 = var1 + nums[end];
                len++;
                end++;
            }
            while (target <= var1) {
                minlen = Math.min(minlen, len);
                var1 = var1 - nums[start];
                start++;
                len--;
            }
        }
        if (minlen == n + 1) {
            return 0;
        } else {
            return minlen;
        }
    }
}
