class Solution {

    public int method1(int target, int[] nums) {
        int r = 0;
        int l = 0;
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        while (l <= r && r < nums.length) {
            var1 += nums[r];
            while (var1 >= target) {
                min = Math.min(min, r - l + 1);
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
