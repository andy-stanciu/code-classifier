class Solution {

    public int method1(int target, int[] nums) {
        int count = Integer.MAX_VALUE;
        int start = 0;
        int var1 = 0;
        for (int end = 0; end < nums.length; end++) {
            var1 += nums[end];
            while (start <= end && var1 >= target) {
                count = Math.min(count, end - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        if (count == Integer.MAX_VALUE) {
            return 0;
        }
        return count;
    }
}
