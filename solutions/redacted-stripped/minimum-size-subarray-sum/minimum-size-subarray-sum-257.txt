class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int var1 = 0;
        int var2 = Integer.MAX_VALUE;
        while (r < nums.length) {
            var1 += nums[r];
            while (var1 >= target) {
                var2 = Math.min(r - l + 1, var2);
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        System.gc();
        return var2 == Integer.MAX_VALUE ? 0 : var2;
    }
}
