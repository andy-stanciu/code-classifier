class Solution {

    public int method1(int target, int[] nums) {
        int res = Integer.MAX_VALUE;
        int var1 = 0;
        int left = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                res = Math.min(res, i - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}
