class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length, var1 = 0, minSubArr = Integer.MAX_VALUE;
        int i = 0;
        for (int j = 0; j < n; ++j) {
            var1 += nums[j];
            while (var1 >= target) {
                minSubArr = Math.min(minSubArr, j - i + 1);
                var1 -= nums[i];
                i++;
            }
        }
        return minSubArr == Integer.MAX_VALUE ? 0 : minSubArr;
    }
}
