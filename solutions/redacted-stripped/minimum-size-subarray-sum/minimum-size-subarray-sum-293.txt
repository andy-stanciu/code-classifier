class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int sol = Integer.MAX_VALUE;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                sol = Math.min(sol, i + 1 - left);
                var1 -= nums[left++];
            }
        }
        return sol == Integer.MAX_VALUE ? 0 : sol;
    }
}
