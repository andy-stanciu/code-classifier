class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = 0;
        int n = nums.length;
        int var1 = 0;
        int minlen = Integer.MAX_VALUE;
        while (j < n) {
            var1 += nums[j];
            while (var1 >= target) {
                minlen = Math.min(minlen, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return minlen == Integer.MAX_VALUE ? 0 : minlen;
    }
}
