class Solution {

    public int method1(int target, int[] nums) {
        int cnt = 0;
        int j = 0;
        int minlen = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            cnt += nums[i];
            while (cnt >= target) {
                minlen = Math.min(minlen, i - j + 1);
                cnt -= nums[j++];
            }
        }
        return minlen == Integer.MAX_VALUE ? 0 : minlen;
    }
}
