class Solution {

    public int method1(int target, int[] nums) {
        int start = 0, n = nums.length, min = Integer.MAX_VALUE, var1 = 0;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                min = Math.min(min, i + 1 - start);
                var1 -= nums[start++];
            }
        }
        if (min > n)
            return 0;
        return min;
    }
}
