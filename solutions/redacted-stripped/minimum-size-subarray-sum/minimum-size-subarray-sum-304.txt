class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int var1 = 0;
        int var2 = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            var1 = var1 + nums[i];
            while (var1 >= target) {
                var2 = Math.min(var2, i - left + 1);
                var1 -= nums[left++];
            }
        }
        return var2 == Integer.MAX_VALUE ? 0 : var2;
    }
}
