class Solution {

    public int method1(int target, int[] nums) {
        int length = Integer.MAX_VALUE;
        int var1 = 0;
        int left = 0;
        for (int right = 0; right < nums.length; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                length = Math.min(length, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        return length == Integer.MAX_VALUE ? 0 : length;
    }
}
