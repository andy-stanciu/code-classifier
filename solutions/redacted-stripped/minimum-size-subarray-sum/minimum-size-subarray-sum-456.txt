class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int minLength = nums.length + 1;
        int var1 = 0;
        for (int r = 0; r < nums.length; r++) {
            var1 += nums[r];
            while (var1 >= target) {
                minLength = Math.min(minLength, r - l + 1);
                var1 -= nums[l];
                l++;
            }
        }
        return (minLength == nums.length + 1) ? 0 : minLength;
    }
}
