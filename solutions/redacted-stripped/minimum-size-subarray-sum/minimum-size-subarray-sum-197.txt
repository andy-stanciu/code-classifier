class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int min_len = nums.length + 1;
        int var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            j++;
            while (var1 >= target) {
                min_len = Math.min(min_len, (j - i));
                var1 -= nums[i];
                i++;
            }
        }
        return ((min_len == nums.length + 1) ? 0 : min_len);
    }
}
