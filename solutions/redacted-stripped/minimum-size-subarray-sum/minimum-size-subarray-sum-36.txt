class Solution {

    public int method1(int target, int[] A) {
        int i = 0;
        int j = 0;
        int var1 = 0;
        int ans = 1000000;
        while (j < A.length) {
            var1 += A[j];
            while (var1 >= target) {
                ans = Math.min(ans, j - i + 1);
                var1 -= A[i];
                i++;
            }
            j++;
        }
        return ans == 1000000 ? 0 : ans;
    }
}
