class Solution {

    public int method1(int target, int[] nums) {
        if (nums == null || nums.length == 0)
            return 0;
        int i = 0, j = 0, var1 = 0, min = Integer.MAX_VALUE;
        while (j < nums.length) {
            var1 += nums[j++];
            while (var1 >= target) {
                min = Math.min(min, j - i);
                var1 -= nums[i++];
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
