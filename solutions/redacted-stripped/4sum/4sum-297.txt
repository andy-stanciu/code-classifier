class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        ArrayList<List<Integer>> list = new ArrayList<>();
        if (nums.length < 4 || (long) nums[0] + nums[1] + nums[2] + nums[3] > target)
            return list;
        for (int i = 0; i < nums.length - 3; i++) {
            if (i > 0 && nums[i] == nums[i - 1])
                continue;
            for (int j = i + 1; j < nums.length - 2; j++) {
                if (j > i + 1 && nums[j] == nums[j - 1])
                    continue;
                int k = j + 1;
                int r = nums.length - 1;
                while (k < r) {
                    long sum = nums[i] + nums[j] + nums[k] + nums[r];
                    if (sum == target) {
                        list.add(Arrays.asList(nums[i], nums[j], nums[k], nums[r]));
                        k++;
                        r--;
                        while (k < r && nums[k] == nums[k - 1]) k++;
                        while (k < r && nums[r] == nums[r + 1]) r--;
                    } else if (sum < target) {
                        k++;
                    } else {
                        r--;
                    }
                }
            }
        }
        return list;
    }
}
