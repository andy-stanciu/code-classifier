class Solution {

    public String method1(String s) {
        Map<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        List<Tuple> arr = new ArrayList<>();
        map.forEach((k, v) -> arr.add(new Tuple(k, v)));
        arr.sort((a, b) -> (b.count - a.count));
        StringBuilder sb = new StringBuilder();
        for (Tuple tuple : arr) {
            char c = tuple.c;
            int t = tuple.count;
            while (t > 0) {
                sb.append(c);
                t--;
            }
        }
        return sb.toString();
    }

    class Tuple {

        char c;

        int count;

        public Tuple(char c, int count) {
            this.c = c;
            this.count = count;
        }
    }
}
