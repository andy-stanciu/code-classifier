class Solution {

    public String method1(String s) {
        HashMap<Character, Integer> x = new HashMap<>();
        StringBuilder sb = new StringBuilder();
        for (char c : s.toCharArray()) {
            x.put(c, x.getOrDefault(c, 0) + 1);
        }
        ArrayList<Character> l = new ArrayList<>(x.keySet());
        Collections.sort(l, (a, b) -> x.get(b) - x.get(a));
        for (Character ch : l) {
            for (int i = 0; i < x.get(ch); i++) {
                sb.append(ch);
            }
        }
        return sb.toString();
    }
}
