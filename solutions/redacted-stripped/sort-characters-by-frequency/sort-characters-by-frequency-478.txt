class Solution {

    public String method1(String s) {
        int n = s.length();
        int[] arr = new int[128];
        for (int i = 0; i < n; i++) {
            char c = s.charAt(i);
            arr[c]++;
        }
        StringBuilder res = new StringBuilder();
        while (true) {
            int maxFreq = 0;
            int maxIndex = -1;
            for (int i = 0; i < 128; i++) {
                if (arr[i] > maxFreq) {
                    maxFreq = arr[i];
                    maxIndex = i;
                }
            }
            if (maxFreq == 0)
                break;
            char c = (char) maxIndex;
            while (maxFreq-- > 0) {
                res.append(c);
            }
            arr[maxIndex] = 0;
        }
        return res.toString();
    }
}
