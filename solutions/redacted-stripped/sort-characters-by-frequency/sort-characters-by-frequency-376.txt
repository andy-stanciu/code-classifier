class Solution {

    public String method1(String s) {
        int[] arr = new int[256];
        for (int i = 0; i < s.length(); i++) {
            arr[(int) s.charAt(i)]++;
        }
        HashMap<Integer, ArrayList<Integer>> map = new HashMap<>();
        String ans = "";
        List<Integer> var1 = new ArrayList<>();
        for (int i = 0; i < 256; i++) {
            if (arr[i] != 0) {
                if (map.containsKey(arr[i])) {
                    ArrayList<Integer> list = map.get(arr[i]);
                    list.add(i);
                    map.put(arr[i], list);
                } else {
                    ArrayList<Integer> list = new ArrayList<>();
                    list.add(i);
                    map.put(arr[i], list);
                    var1.add(arr[i]);
                }
            }
        }
        Collections.sort(var1, Collections.reverseOrder());
        for (int i = 0; i < var1.size(); i++) {
            ArrayList<Integer> list = map.get(var1.get(i));
            for (int j = 0; j < list.size(); j++) {
                while (arr[list.get(j)] != 0) {
                    int num = list.get(j);
                    ans = ans + (char) (num);
                    arr[list.get(j)]--;
                }
            }
        }
        return ans;
    }
}
