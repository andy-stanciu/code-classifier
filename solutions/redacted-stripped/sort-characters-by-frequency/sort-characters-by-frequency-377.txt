class Solution {

    public String method1(String s) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
        }
        List<Map.Entry<Character, Integer>> el = new ArrayList<>(map.entrySet());
        el.sort((entry1, entry2) -> entry2.getValue().compareTo(entry1.getValue()));
        StringBuilder ss = new StringBuilder();
        for (Map.Entry<Character, Integer> entry : el) {
            char c = entry.getKey();
            int n = entry.getValue();
            ss.append(String.valueOf(c).repeat(n));
        }
        String sst = ss.toString();
        return sst;
    }
}
