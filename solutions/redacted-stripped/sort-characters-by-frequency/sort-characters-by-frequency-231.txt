class Solution {

    public String method1(String s) {
        int[] count = new int[256];
        String str = "";
        for (int i = 0; i < s.length(); i++) count[s.charAt(i)]++;
        for (int i = s.length(); i > 0; i--) {
            for (int j = 0; j < 256; j++) {
                if (count[j] == i) {
                    int c = 0;
                    while (c++ != i) str = str + (char) (j);
                }
            }
        }
        return str;
    }
}
