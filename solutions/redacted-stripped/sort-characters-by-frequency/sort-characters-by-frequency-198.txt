class Solution {

    public String method1(String s) {
        Map<Character, Integer> mp = new HashMap<>();
        for (char item : s.toCharArray()) {
            mp.put(item, mp.getOrDefault(item, 0) + 1);
        }
        PriorityQueue<Map.Entry<Character, Integer>> maxHeap = new PriorityQueue<>(Map.Entry.comparingByValue(Comparator.reverseOrder()));
        for (Map.Entry<Character, Integer> entry : mp.entrySet()) {
            maxHeap.offer(entry);
        }
        StringBuilder sb = new StringBuilder();
        while (!maxHeap.isEmpty()) {
            char key = maxHeap.peek().getKey();
            int value = maxHeap.peek().getValue();
            for (int i = 0; i < value; i++) {
                sb.append(key);
            }
            maxHeap.poll();
        }
        return sb.toString();
    }
}
