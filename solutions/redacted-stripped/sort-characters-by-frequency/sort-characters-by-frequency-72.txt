class Solution {

    public String method1(String s) {
        int[] freq = new int[128];
        for (char c : s.toCharArray()) freq[c]++;
        StringBuilder sb = new StringBuilder();
        while (true) {
            int max = 0;
            int ind = -1;
            for (int i = 0; i < 128; i++) {
                if (freq[i] > max) {
                    max = freq[i];
                    ind = i;
                }
            }
            if (ind == -1)
                break;
            freq[ind] = 0;
            char c = (char) ind;
            for (int i = 0; i < max; i++) sb.append(c);
        }
        return sb.toString();
    }
}
