class Solution {

    static class Node {

        int f;

        char i;

        Node(int _f, char _i) {
            f = _f;
            i = _i;
        }
    }

    public String method1(String s) {
        int n = s.length();
        HashMap<Character, Node> map = new HashMap();
        for (char i : s.toCharArray()) {
            Node curr = map.get(i);
            if (curr == null)
                map.put(i, new Node(1, i));
            else
                curr.f++;
        }
        PriorityQueue<Node> pq = new PriorityQueue<>((a, b) -> b.f - a.f);
        pq.addAll(map.values());
        String ans = "";
        while (!pq.isEmpty()) {
            Node curr = pq.remove();
            ans += ("" + curr.i).repeat(curr.f);
        }
        return ans;
    }
}
