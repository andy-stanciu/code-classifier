class Solution {

    public String method1(String s) {
        Map<Character, String> var1 = new HashMap<>();
        for (char ch : s.toCharArray()) {
            String str = String.valueOf(ch);
            var1.put(ch, var1.getOrDefault(ch, "") + str);
        }
        return var1.values().stream().sorted((e1, e2) -> Integer.compare(e2.length(), e1.length())).collect(Collectors.joining());
    }
}
