class Solution {

    public String method1(String s) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (map.containsKey(ch)) {
                map.put(ch, map.get(ch) + 1);
            } else {
                map.put(ch, 1);
            }
        }
        List<Character> var1 = new ArrayList<>(map.keySet());
        Collections.sort(var1, Comparator.comparingInt(map::get).reversed());
        String str = "";
        for (Character ch : var1) {
            int freq = map.get(ch);
            for (int j = 0; j < freq; j++) {
                str += ch;
            }
        }
        return str;
    }
}
