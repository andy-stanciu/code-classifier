class Solution {

    public String method1(String s) {
        return s.chars().parallel().mapToObj(item -> Character.toString((char) item)).collect(Collectors.toMap(item -> item, item -> 1, Integer::sum)).entrySet().parallelStream().sorted((item1, item2) -> item2.getValue() - item1.getValue()).map(item -> item.getKey().repeat(item.getValue())).collect(Collectors.joining());
    }
}
