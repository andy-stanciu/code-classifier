class Solution {

    public String method1(String s) {
        int[] arr = new int[255];
        for (char ch : s.toCharArray()) {
            arr[ch]++;
        }
        StringBuilder ans = new StringBuilder();
        while (true) {
            int maxFreqIndex = max(arr);
            if (maxFreqIndex == -1)
                break;
            int max = arr[maxFreqIndex];
            char val = (char) maxFreqIndex;
            while (max > 0) {
                ans.append(val);
                max--;
            }
            arr[maxFreqIndex] = 0;
        }
        return ans.toString();
    }

    public int max(int[] arr) {
        int maxFreq = 0;
        int maxIndex = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] > maxFreq) {
                maxFreq = arr[i];
                maxIndex = i;
            }
        }
        if (maxFreq == 0)
            return -1;
        return maxIndex;
    }
}
