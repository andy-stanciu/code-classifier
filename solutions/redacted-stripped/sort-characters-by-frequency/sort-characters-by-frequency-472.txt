class Solution {

    public String method1(String s) {
        Map<Character, Integer> freq = new HashMap<>();
        for (char c : s.toCharArray()) {
            freq.put(c, freq.getOrDefault(c, 0) + 1);
        }
        return freq.entrySet().stream().sorted((e1, e2) -> e2.getValue() - e1.getValue()).map(entry -> String.valueOf(entry.getKey()).repeat(entry.getValue())).reduce("", (e1, e2) -> e1 + e2);
    }
}
