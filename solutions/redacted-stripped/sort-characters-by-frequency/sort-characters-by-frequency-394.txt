class Solution {

    public String method1(String s) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (char c : s.toCharArray()) map.put(c, map.getOrDefault(c, 0) + 1);
        TreeMap<Integer, List<Character>> var1 = new TreeMap<>(Collections.reverseOrder());
        for (char key : map.keySet()) {
            int freq = map.get(key);
            if (!var1.containsKey(freq)) {
                var1.put(freq, new ArrayList<>());
            }
            var1.get(freq).add(key);
        }
        StringBuilder res = new StringBuilder();
        for (int freq : var1.keySet()) {
            List<Character> var2 = var1.get(freq);
            for (char c : var2) {
                int count = map.get(c);
                for (int i = 0; i < count; i++) {
                    res.append(c);
                }
            }
        }
        return res.toString();
    }
}
