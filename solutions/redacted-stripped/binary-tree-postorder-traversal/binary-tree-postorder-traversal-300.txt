class Solution {

    public List<Integer> method1(TreeNode node, List<Integer> list) {
        if (node == null) {
            return list;
        }
        method1(node.left, list);
        method1(node.right, list);
        list.add(node.val);
        return list;
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        return method1(root, list);
    }
}
