class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    private void method2(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        method2(node.left, list);
        method2(node.right, list);
        list.add(node.val);
    }
}
