public class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        TreeNode cur = root;
        while (cur != null) {
            if (cur.right == null) {
                var1.add(cur.val);
                cur = cur.left;
            } else {
                TreeNode pred = cur.right;
                while (pred.left != null && pred.left != cur) {
                    pred = pred.left;
                }
                if (pred.left == null) {
                    pred.left = cur;
                    var1.add(cur.val);
                    cur = cur.right;
                } else {
                    pred.left = null;
                    cur = cur.left;
                }
            }
        }
        Collections.reverse(var1);
        return var1;
    }
}
