class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        post(root, result);
        return result;
    }

    public void post(TreeNode root, List<Integer> result) {
        if (root == null) {
            return;
        }
        post(root.left, result);
        post(root.right, result);
        result.add(root.val);
    }
}
