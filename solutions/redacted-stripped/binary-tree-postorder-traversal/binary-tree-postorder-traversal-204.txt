class Solution {

    public static List<Integer> ans;

    public List<Integer> method1(TreeNode root) {
        ans = new ArrayList<>();
        solve(root);
        return ans;
    }

    public static void solve(TreeNode root) {
        if (root == null)
            return;
        solve(root.left);
        solve(root.right);
        ans.add(root.val);
    }
}
