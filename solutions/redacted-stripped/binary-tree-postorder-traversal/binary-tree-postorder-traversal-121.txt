class Solution {

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return list;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        method2(root.right);
        list.add(root.val);
        return;
    }
}
