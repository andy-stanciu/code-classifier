class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        post(root, list);
        return list;
    }

    void post(TreeNode root, List<Integer> list) {
        if (root != null) {
            post(root.left, list);
            post(root.right, list);
            list.add(root.val);
        }
    }
}
