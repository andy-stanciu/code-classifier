class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        return var1;
    }

    private void method2(TreeNode node, List<Integer> var1) {
        if (node == null)
            return;
        method2(node.left, var1);
        method2(node.right, var1);
        var1.add(node.val);
    }
}
