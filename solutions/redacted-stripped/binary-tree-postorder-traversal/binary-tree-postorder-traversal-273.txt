class Solution {

    List<Integer> at = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return at;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        method2(root.right);
        at.add(root.val);
    }
}
