class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        recursive(list, root);
        return list;
    }

    private void recursive(List<Integer> list, TreeNode root) {
        if (root != null) {
            recursive(list, root.left);
            recursive(list, root.right);
            list.add(root.val);
        }
    }
}
