class Solution {

    List<Integer> ans;

    public List<Integer> method1(TreeNode root) {
        ans = new ArrayList<Integer>();
        f(root);
        return ans;
    }

    public void f(TreeNode root) {
        if (root == null) {
            return;
        }
        f(root.left);
        f(root.right);
        ans.add(root.val);
    }
}
