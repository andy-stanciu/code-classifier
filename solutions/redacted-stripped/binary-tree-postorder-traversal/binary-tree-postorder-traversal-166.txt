class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return var1;
        Stack<TreeNode> s1 = new Stack<>();
        TreeNode curr = root;
        while (!s1.isEmpty() || curr != null) {
            if (curr != null) {
                s1.push(curr);
                curr = curr.left;
            } else {
                TreeNode temp = s1.peek().right;
                if (temp == null) {
                    temp = s1.pop();
                    var1.add(temp.val);
                    while (!s1.isEmpty() && s1.peek().right == temp) {
                        temp = s1.peek();
                        s1.pop();
                        var1.add(temp.val);
                    }
                } else {
                    curr = temp;
                }
            }
        }
        return var1;
    }
}
