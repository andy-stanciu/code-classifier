class Solution {

    List<Integer> post = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        addList(root);
        return post;
    }

    void addList(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        method1(root.right);
        post.add(root.val);
    }
}
