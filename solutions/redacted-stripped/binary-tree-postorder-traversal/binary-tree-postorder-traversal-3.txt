class Solution {

    private List<Integer> post = new ArrayList<>();

    private void method1(TreeNode root) {
        if (root != null) {
            method1(root.left);
            method1(root.right);
            post.add(root.val);
        }
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return post;
    }
}
