class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        traverse(root, ans);
        return ans;
    }

    private void traverse(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        traverse(root.left, ans);
        traverse(root.right, ans);
        ans.add(root.val);
    }
}
