class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return var1;
        List<Integer> var2 = method1(root.left);
        var1.addAll(var2);
        List<Integer> var3 = method1(root.right);
        var1.addAll(var3);
        var1.add(root.val);
        return var1;
    }
}
