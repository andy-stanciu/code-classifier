class Solution {

    public void method1(TreeNode root, List<Integer> ls) {
        if (root == null) {
            return;
        }
        method1(root.left, ls);
        method1(root.right, ls);
        ls.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        if (root == null) {
            return ls;
        }
        method1(root, ls);
        return ls;
    }
}
