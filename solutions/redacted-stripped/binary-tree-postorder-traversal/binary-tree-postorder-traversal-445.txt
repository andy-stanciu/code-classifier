class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        prehelper(root, ans);
        return ans;
    }

    private void prehelper(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        prehelper(root.left, ans);
        prehelper(root.right, ans);
        ans.add(root.val);
    }
}
