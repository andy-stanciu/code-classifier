class Solution {

    public List<Integer> method1(TreeNode root) {
        TreeNode cur = root;
        List<Integer> list = new ArrayList<>();
        if (cur == null) {
            return list;
        }
        list.addAll(method1(cur.left));
        list.addAll(method1(cur.right));
        list.add(cur.val);
        return list;
    }
}
