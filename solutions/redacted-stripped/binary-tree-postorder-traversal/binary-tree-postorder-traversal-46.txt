class Solution {

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return ans;
    }

    List<Integer> ans = new ArrayList<>();

    public void method2(TreeNode node) {
        if (node == null) {
            return;
        }
        method2(node.left);
        method2(node.right);
        ans.add(node.val);
    }
}
