class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        return method2(list, root);
    }

    public List<Integer> method2(List<Integer> result, TreeNode root) {
        if (root == null)
            return result;
        method2(result, root.left);
        method2(result, root.right);
        result.add(root.val);
        return result;
    }
}
