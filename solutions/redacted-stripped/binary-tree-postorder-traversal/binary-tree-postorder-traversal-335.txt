class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        dfs(root, var1);
        return var1;
    }

    public void dfs(TreeNode node, List<Integer> var1) {
        if (node == null)
            return;
        dfs(node.left, var1);
        dfs(node.right, var1);
        var1.add(node.val);
    }
}
