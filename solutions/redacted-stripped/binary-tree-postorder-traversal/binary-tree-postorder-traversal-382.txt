class Solution {

    public void method1(TreeNode root, List<Integer> l) {
        if (root == null)
            return;
        method1(root.left, l);
        method1(root.right, l);
        l.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        method1(root, l);
        return l;
    }
}
