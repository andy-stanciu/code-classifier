class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method1(root, list);
        return list;
    }

    void method1(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        method1(root.left, list);
        method1(root.right, list);
        list.add(root.val);
    }
}
