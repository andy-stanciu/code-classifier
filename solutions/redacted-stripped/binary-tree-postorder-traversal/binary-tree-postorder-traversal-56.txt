class Solution {

    List<Integer> ans = new ArrayList<>();

    private void method1(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        method1(root.right);
        ans.add(root.val);
        return;
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return ans;
    }
}
