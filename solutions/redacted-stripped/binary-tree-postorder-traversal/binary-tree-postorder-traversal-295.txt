class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        pot(root, res);
        return res;
    }

    public void pot(TreeNode t, List<Integer> l) {
        if (t == null)
            return;
        pot(t.left, l);
        pot(t.right, l);
        l.add(t.val);
    }
}
