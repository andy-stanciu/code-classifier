class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new LinkedList<>();
        helper(res, root);
        return res;
    }

    public TreeNode helper(List<Integer> res, TreeNode node) {
        if (node == null)
            return null;
        node.left = helper(res, node.left);
        node.right = helper(res, node.right);
        res.add(node.val);
        return node;
    }
}
