class Solution {

    public void method1(List<Integer> ans, TreeNode root) {
        if (root == null) {
            return;
        }
        method1(ans, root.left);
        method1(ans, root.right);
        ans.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        method1(ans, root);
        return ans;
    }
}
