class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        print(root, res);
        return res;
    }

    void print(TreeNode node, List<Integer> res) {
        if (node == null) {
            return;
        }
        print(node.left, res);
        print(node.right, res);
        res.add(node.val);
    }
}
