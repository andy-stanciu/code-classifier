class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        fxn(root, res);
        return res;
    }

    private void fxn(TreeNode root, List<Integer> res) {
        if (root != null) {
            fxn(root.left, res);
            fxn(root.right, res);
            res.add(root.val);
        }
    }
}
