class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        if (root.left != null)
            ans.addAll(method1(root.left));
        if (root.right != null)
            ans.addAll(method1(root.right));
        ans.add(root.val);
        return ans;
    }
}
