class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method2(result, root);
        return result;
    }

    public static void method2(List<Integer> result, TreeNode root) {
        if (root != null) {
            method2(result, root.left);
            method2(result, root.right);
            result.add(root.val);
        }
    }
}
