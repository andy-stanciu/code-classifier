class Solution {

    List<Integer> l = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return l;
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        method2(root.right);
        l.add(root.val);
    }
}
