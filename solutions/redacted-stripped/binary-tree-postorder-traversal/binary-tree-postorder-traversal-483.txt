class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> List = new ArrayList<>();
        method1(root, List);
        return List;
    }

    void method1(TreeNode node, ArrayList<Integer> List) {
        if (node == null) {
            return;
        }
        method1(node.left, List);
        method1(node.right, List);
        List.add(node.val);
    }
}
