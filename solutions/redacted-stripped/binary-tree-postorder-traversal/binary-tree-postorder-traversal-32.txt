class Solution {

    List<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<>();
        TreeNode p = root;
        while (p != null || !st.isEmpty()) {
            if (p != null) {
                ans.add(p.val);
                st.push(p);
                p = p.right;
            } else {
                p = st.pop();
                p = p.left;
            }
        }
        Collections.reverse(ans);
        return ans;
    }
}
