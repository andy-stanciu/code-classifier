class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        helper(list, root);
        return list;
    }

    private void helper(List<Integer> list, TreeNode node) {
        if (node == null) {
            return;
        }
        helper(list, node.left);
        helper(list, node.right);
        list.add(node.val);
    }
}
