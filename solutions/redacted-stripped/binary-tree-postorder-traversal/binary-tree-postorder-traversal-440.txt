class Solution {

    private List<Integer> res = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return res;
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        method2(root.right);
        res.add(root.val);
    }
}
