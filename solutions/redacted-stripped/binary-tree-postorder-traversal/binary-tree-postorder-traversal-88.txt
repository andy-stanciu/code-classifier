class Solution {

    List<Integer> al = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        recursion(root);
        return al;
    }

    public void recursion(TreeNode root) {
        if (root == null)
            return;
        recursion(root.left);
        recursion(root.right);
        al.add(root.val);
    }
}
