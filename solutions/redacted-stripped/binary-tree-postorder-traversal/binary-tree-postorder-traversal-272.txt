class Solution {

    private void travail(List<Integer> res, TreeNode root) {
        if (root == null) {
            return;
        }
        travail(res, root.left);
        travail(res, root.right);
        res.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        travail(res, root);
        return res;
    }
}
