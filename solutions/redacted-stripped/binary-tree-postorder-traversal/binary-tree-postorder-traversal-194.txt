class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method2(root, res);
        return res;
    }

    public static void method2(TreeNode root, List<Integer> arr) {
        if (root == null) {
            return;
        }
        method2(root.left, arr);
        method2(root.right, arr);
        arr.add(root.val);
    }
}
