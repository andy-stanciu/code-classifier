class Solution {

    List<Integer> arr = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        post(root);
        return arr;
    }

    public void post(TreeNode root) {
        if (root == null) {
            return;
        }
        post(root.left);
        post(root.right);
        arr.add(root.val);
    }
}
