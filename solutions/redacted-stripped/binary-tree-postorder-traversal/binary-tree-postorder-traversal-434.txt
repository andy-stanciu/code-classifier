class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        preHelper(root, pre);
        return pre;
    }

    public void preHelper(TreeNode root, List<Integer> pre) {
        if (root == null)
            return;
        preHelper(root.left, pre);
        preHelper(root.right, pre);
        pre.add(root.val);
    }
}
