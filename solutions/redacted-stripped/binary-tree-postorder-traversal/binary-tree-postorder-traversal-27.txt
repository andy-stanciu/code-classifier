class Solution {

    LinkedList<Integer> res = new LinkedList<>();

    public List<Integer> method1(TreeNode root) {
        postOder(root);
        return res;
    }

    public int postOder(TreeNode root) {
        if (root == null) {
            return 0;
        }
        postOder(root.left);
        postOder(root.right);
        res.add(root.val);
        return 0;
    }
}
