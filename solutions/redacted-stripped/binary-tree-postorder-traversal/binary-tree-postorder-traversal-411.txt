class Solution {

    private void method1(List<Integer> numbers, TreeNode root) {
        if (root == null)
            return;
        method1(numbers, root.left);
        method1(numbers, root.right);
        numbers.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> numbers = new ArrayList<>();
        method1(numbers, root);
        return numbers;
    }
}
