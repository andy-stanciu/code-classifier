class Solution {

    List<Integer> list = new ArrayList<>();

    private void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        method1(root.right);
        list.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return list;
    }
}
