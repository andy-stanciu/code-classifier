class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> res = new ArrayList<>();
        method1(root, res);
        return res;
    }

    public static void method1(TreeNode root, ArrayList<Integer> res) {
        if (root == null) {
            return;
        }
        method1(root.left, res);
        method1(root.right, res);
        res.add(root.val);
    }
}
