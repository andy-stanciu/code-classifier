class Solution {

    private void method1(TreeNode root, List<Integer> List) {
        if (root == null)
            return;
        method1(root.left, List);
        method1(root.right, List);
        List.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> List = new ArrayList<>();
        method1(root, List);
        return List;
    }
}
