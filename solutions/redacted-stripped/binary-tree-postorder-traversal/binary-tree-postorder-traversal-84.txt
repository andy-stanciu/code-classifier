class Solution {

    List<Integer> al = new LinkedList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return al;
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        method2(root.right);
        al.add(root.val);
    }
}
