class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        po(root, res);
        return res;
    }

    private void po(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        po(root.left, res);
        po(root.right, res);
        res.add(root.val);
    }
}
