class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        dfs(root, list);
        return list;
    }

    List<Integer> dfs(TreeNode root, List<Integer> list) {
        if (root == null) {
            return list;
        }
        dfs(root.left, list);
        dfs(root.right, list);
        list.add(root.val);
        return list;
    }
}
