class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method1(root, list);
        return list;
    }

    void method1(TreeNode node, List<Integer> list) {
        if (node == null) {
            return;
        }
        method1(node.left, list);
        method1(node.right, list);
        list.add(node.val);
    }
}
