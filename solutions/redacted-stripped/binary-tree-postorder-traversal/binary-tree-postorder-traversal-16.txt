class Solution {

    public List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            return result;
        }
        dfs(root);
        return result;
    }

    public void dfs(TreeNode root) {
        if (root == null) {
            return;
        }
        dfs(root.left);
        dfs(root.right);
        result.add(root.val);
    }
}
