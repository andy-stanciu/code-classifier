class Solution {

    List<Integer> ans;

    public List<Integer> method1(TreeNode root) {
        ans = new ArrayList<>();
        method2(root);
        return ans;
    }

    void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        method2(root.right);
        ans.add(root.val);
    }
}
