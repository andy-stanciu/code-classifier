class Solution {

    public static void method1(TreeNode root, List<Integer> al) {
        if (root == null) {
            return;
        }
        method1(root.left, al);
        method1(root.right, al);
        al.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> al = new ArrayList<>();
        method1(root, al);
        return al;
    }
}
