class Solution {

    private List<Integer> result;

    public List<Integer> method1(TreeNode root) {
        result = new ArrayList<>();
        dfs(root);
        return result;
    }

    private void dfs(TreeNode node) {
        if (node == null) {
            return;
        }
        dfs(node.left);
        dfs(node.right);
        result.add(node.val);
    }
}
