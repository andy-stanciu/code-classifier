class Solution {

    public List<Integer> list = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            return list;
        } else {
            pp(root, list);
        }
        return list;
    }

    public static void pp(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        pp(root.left, list);
        pp(root.right, list);
        list.add(root.val);
    }
}
