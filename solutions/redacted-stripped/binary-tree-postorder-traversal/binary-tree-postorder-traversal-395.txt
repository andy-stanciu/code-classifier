class Solution {

    ArrayList<Integer> al = new ArrayList<>();

    public void method1(TreeNode node) {
        if (node == null) {
            return;
        }
        method1(node.left);
        method1(node.right);
        al.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return al;
    }
}
