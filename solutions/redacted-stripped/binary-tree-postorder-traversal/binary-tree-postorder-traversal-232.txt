class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<>();
        helper(ans, root);
        return ans;
    }

    public static void helper(ArrayList<Integer> al, TreeNode root) {
        if (root == null)
            return;
        helper(al, root.left);
        helper(al, root.right);
        al.add(root.val);
    }
}
