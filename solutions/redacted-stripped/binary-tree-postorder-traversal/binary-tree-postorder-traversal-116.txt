class Solution {

    public List<Integer> method1(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return ans;
        }
        method1(root.left, ans);
        method1(root.right, ans);
        ans.add(root.val);
        return ans;
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        return method1(root, ans);
    }
}
