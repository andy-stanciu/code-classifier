class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        result = doTransform(root, result);
        return result;
    }

    public List<Integer> doTransform(TreeNode root, List<Integer> result) {
        if (root == null) {
            return result;
        }
        result = doTransform(root.left, result);
        result = doTransform(root.right, result);
        result.add(root.val);
        return result;
    }
}
