class Solution {

    List<Integer> x = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return x;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        method2(root.right);
        x.add(root.val);
    }
}
