class Solution {

    List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return result;
    }

    public void method2(TreeNode node) {
        if (node == null)
            return;
        method2(node.left);
        method2(node.right);
        result.add(node.val);
    }
}
