class Solution {

    public List<Integer> method1(TreeNode root) {
        LinkedList<Integer> list = new LinkedList<>();
        Deque<TreeNode> deque = new ArrayDeque<>();
        if (root == null)
            return list;
        deque.push(root);
        while (!deque.isEmpty()) {
            root = deque.pop();
            list.addFirst(root.val);
            if (root.left != null)
                deque.push(root.left);
            if (root.right != null)
                deque.push(root.right);
        }
        return list;
    }
}
