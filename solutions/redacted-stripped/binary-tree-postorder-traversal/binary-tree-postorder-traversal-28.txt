class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        TreeNode curr = root;
        while (curr != null) {
            if (curr.right != null) {
                TreeNode temp = curr.right;
                while (temp.left != null && temp.left != curr) temp = temp.left;
                if (temp.left == null) {
                    temp.left = curr;
                    var1.add(curr.val);
                    curr = curr.right;
                } else {
                    temp.left = null;
                    curr = curr.left;
                }
            } else {
                var1.add(curr.val);
                curr = curr.left;
            }
        }
        Collections.reverse(var1);
        return var1;
    }
}
