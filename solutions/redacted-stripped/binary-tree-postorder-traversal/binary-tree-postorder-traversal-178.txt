class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> arr = new ArrayList<>();
        post(root, arr);
        return arr;
    }

    public void post(TreeNode root, List<Integer> arr) {
        if (root != null) {
            post(root.left, arr);
            post(root.right, arr);
            arr.add(root.val);
        }
    }
}
