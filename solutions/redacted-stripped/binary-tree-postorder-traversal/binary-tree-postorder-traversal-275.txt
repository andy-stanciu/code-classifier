class Solution {

    private List<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        fillPost(root);
        return ans;
    }

    private void fillPost(TreeNode root) {
        if (root == null)
            return;
        fillPost(root.left);
        fillPost(root.right);
        ans.add(root.val);
    }
}
