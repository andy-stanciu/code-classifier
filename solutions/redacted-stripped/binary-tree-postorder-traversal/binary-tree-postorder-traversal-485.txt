class Solution {

    private void method1(TreeNode node, List<Integer> param1) {
        if (node == null) {
            return;
        }
        method1(node.left, param1);
        method1(node.right, param1);
        param1.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> param1 = new ArrayList<>();
        if (root == null) {
            return param1;
        }
        method1(root, param1);
        return param1;
    }
}
