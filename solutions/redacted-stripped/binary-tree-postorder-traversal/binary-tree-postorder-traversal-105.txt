class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        postOt(root, ls);
        return ls;
    }

    public void postOt(TreeNode root, List<Integer> ls) {
        if (root == null)
            return;
        postOt(root.left, ls);
        postOt(root.right, ls);
        ls.add(root.val);
    }
}
