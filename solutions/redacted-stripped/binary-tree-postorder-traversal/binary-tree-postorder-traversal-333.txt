class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        method2(root, li);
        return li;
    }

    public void method2(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        method2(root.left, ans);
        method2(root.right, ans);
        ans.add(root.val);
    }
}
