class Solution {

    List<Integer> a1 = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return a1;
    }

    public void method2(TreeNode root) {
        if (root != null) {
            method1(root.left);
            method1(root.right);
            a1.add(root.val);
        }
    }
}
