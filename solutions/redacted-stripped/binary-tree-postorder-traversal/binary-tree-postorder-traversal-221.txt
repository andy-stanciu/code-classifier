class Solution {

    private void find(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        find(root.left, ans);
        find(root.right, ans);
        ans.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> solution = new ArrayList<>();
        find(root, solution);
        return solution;
    }
}
