class Solution {

    List<Integer> l = new ArrayList<Integer>();

    public void preorder(TreeNode root) {
        if (root == null)
            return;
        preorder(root.left);
        preorder(root.right);
        l.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        preorder(root);
        return l;
    }
}
