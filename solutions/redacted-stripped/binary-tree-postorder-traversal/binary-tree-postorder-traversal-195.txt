class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        method2(root, l);
        return l;
    }

    public void method2(TreeNode node, List<Integer> l) {
        if (node != null) {
            method2(node.left, l);
            method2(node.right, l);
            l.add(node.val);
        }
    }
}
