class Solution {

    List<Integer> li = new ArrayList<>();

    public void dfs(TreeNode root) {
        if (root == null)
            return;
        dfs(root.left);
        dfs(root.right);
        li.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        dfs(root);
        return li;
    }
}
