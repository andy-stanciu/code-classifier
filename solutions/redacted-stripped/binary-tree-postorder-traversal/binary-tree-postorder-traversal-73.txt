class Solution {

    List<Integer> l = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return l;
        method2(root);
        return l;
    }

    void method2(TreeNode r) {
        if (r != null) {
            method2(r.left);
            method2(r.right);
            l.add(r.val);
        }
    }
}
