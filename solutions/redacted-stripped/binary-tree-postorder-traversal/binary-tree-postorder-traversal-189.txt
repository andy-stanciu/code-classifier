class Solution {

    public void solve(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        solve(root.left, res);
        solve(root.right, res);
        res.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        solve(root, res);
        return res;
    }
}
