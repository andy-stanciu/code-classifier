class Solution {

    static int c = 0;

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return list;
    }

    public void method2(TreeNode node) {
        if (node == null)
            return;
        method2(node.left);
        method2(node.right);
        list.add(node.val);
    }
}
