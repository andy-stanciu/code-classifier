class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null)
            return res;
        Deque<TreeNode> st = new ArrayDeque<>();
        st.push(root);
        while (st.size() > 0) {
            root = st.pop();
            res.add(root.val);
            if (root.left != null)
                st.push(root.left);
            if (root.right != null)
                st.push(root.right);
        }
        Collections.reverse(res);
        return res;
    }
}
