class Solution {

    private List<Integer> post(TreeNode root, List<Integer> res) {
        if (root == null) {
            return res;
        }
        TreeNode currNode = root;
        post(currNode.left, res);
        post(currNode.right, res);
        res.add(currNode.val);
        return res;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        return post(root, res);
    }
}
