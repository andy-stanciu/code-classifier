class Solution {

    ArrayList<Integer> arr = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return arr;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        method2(root.right);
        arr.add(root.val);
    }
}
