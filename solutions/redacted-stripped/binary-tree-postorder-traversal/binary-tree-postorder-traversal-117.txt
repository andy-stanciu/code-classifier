class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<Integer>();
        method2(root, result);
        return result;
    }

    public void method2(TreeNode root, List<Integer> result) {
        if (root != null) {
            method2(root.left, result);
            method2(root.right, result);
            result.add(root.val);
        }
    }
}
