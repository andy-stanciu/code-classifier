class Solution {

    public void traverse(TreeNode root, List<Integer> answer) {
        if (root != null) {
            traverse(root.left, answer);
            traverse(root.right, answer);
            answer.add(root.val);
        }
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> answer = new ArrayList<>();
        traverse(root, answer);
        return answer;
    }
}
