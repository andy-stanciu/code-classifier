class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    static void method2(TreeNode root, List list) {
        if (root == null)
            return;
        method2(root.left, list);
        method2(root.right, list);
        list.add(root.val);
    }
}
