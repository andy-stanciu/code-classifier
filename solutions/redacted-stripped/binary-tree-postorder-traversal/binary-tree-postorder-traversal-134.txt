class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ret = new ArrayList<>();
        if (root != null) {
            if (root.left != null) {
                ret.addAll(method1(root.left));
            }
            if (root.right != null) {
                ret.addAll(method1(root.right));
            }
            ret.add(root.val);
        }
        return ret;
    }
}
