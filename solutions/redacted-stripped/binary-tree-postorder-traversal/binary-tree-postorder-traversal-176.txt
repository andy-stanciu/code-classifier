class Solution {

    public void solve(List<Integer> arr, TreeNode root) {
        if (root == null)
            return;
        solve(arr, root.left);
        solve(arr, root.right);
        arr.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        if (root == null)
            return arr;
        solve(arr, root);
        return arr;
    }
}
