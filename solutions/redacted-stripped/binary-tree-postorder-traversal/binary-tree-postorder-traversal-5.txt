class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        solve(root, l);
        return l;
    }

    public void solve(TreeNode n, List<Integer> l) {
        if (n == null)
            return;
        solve(n.left, l);
        solve(n.right, l);
        l.add(n.val);
    }
}
