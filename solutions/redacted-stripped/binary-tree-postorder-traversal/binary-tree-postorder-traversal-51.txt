class Solution {

    void method1(ArrayList<Integer> ans, TreeNode root) {
        if (root == null) {
            return;
        }
        method1(ans, root.left);
        method1(ans, root.right);
        ans.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<>();
        method1(ans, root);
        return ans;
    }
}
