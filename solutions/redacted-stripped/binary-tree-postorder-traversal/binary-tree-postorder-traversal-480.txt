class Solution {

    List<Integer> l1 = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        pre(root);
        return l1;
    }

    public void pre(TreeNode root) {
        if (root == null) {
            return;
        }
        pre(root.left);
        pre(root.right);
        l1.add(root.val);
    }
}
