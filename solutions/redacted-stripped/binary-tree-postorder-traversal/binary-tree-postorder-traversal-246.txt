class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList();
        Stack<TreeNode> s = new Stack();
        while (root != null || !s.isEmpty()) {
            if (root != null) {
                res.add(root.val);
                s.add(root);
                root = root.right;
            } else {
                root = s.pop();
                root = root.left;
            }
        }
        Collections.reverse(res);
        return res;
    }
}
