class Solution {

    public List<Integer> method1(TreeNode r) {
        Stack<TreeNode> s1 = new Stack<>();
        Stack<TreeNode> s2 = new Stack<>();
        List<Integer> ans = new ArrayList<>();
        if (r == null)
            return ans;
        s1.push(r);
        while (!s1.isEmpty()) {
            TreeNode root = s1.peek();
            s2.push(s1.pop());
            if (root.left != null)
                s1.push(root.left);
            if (root.right != null)
                s1.push(root.right);
        }
        while (!s2.isEmpty()) {
            TreeNode root = s2.peek();
            ans.add(root.val);
            s2.pop();
        }
        return ans;
    }
}
