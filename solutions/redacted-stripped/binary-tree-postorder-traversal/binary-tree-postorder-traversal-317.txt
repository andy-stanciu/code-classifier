class Solution {

    private void util(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        util(root.left, res);
        util(root.right, res);
        res.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        util(root, res);
        return res;
    }
}
