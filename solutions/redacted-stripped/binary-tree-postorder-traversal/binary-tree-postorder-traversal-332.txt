class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        ArrayDeque<TreeNode> stack = new ArrayDeque<>();
        if (root != null)
            stack.add(root);
        while (!stack.isEmpty()) {
            TreeNode curr = stack.pollLast();
            var1.add(curr.val);
            if (curr.left != null)
                stack.add(curr.left);
            if (curr.right != null)
                stack.add(curr.right);
        }
        Collections.reverse(var1);
        return var1;
    }
}
