class Solution {

    List<Integer> postTraverse(TreeNode root, List<Integer> list) {
        if (root == null)
            return list;
        postTraverse(root.left, list);
        postTraverse(root.right, list);
        list.add(root.val);
        return list;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        list = postTraverse(root, list);
        return list;
    }
}
