class Solution {

    public List<Integer> method1(TreeNode root) {
        TreeNode node = root;
        List<Integer> result = new ArrayList<>();
        findOrder(node, result);
        return result;
    }

    private void findOrder(TreeNode node, List<Integer> result) {
        if (node == null) {
            return;
        }
        findOrder(node.left, result);
        findOrder(node.right, result);
        result.add(node.val);
    }
}
