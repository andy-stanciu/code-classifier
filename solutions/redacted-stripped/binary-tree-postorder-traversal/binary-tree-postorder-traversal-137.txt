class Solution {

    public List<Integer> calculate(TreeNode root, List<Integer> l) {
        if (root == null) {
            return l;
        }
        l = calculate(root.left, l);
        l = calculate(root.right, l);
        l.add(root.val);
        return l;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        return calculate(root, l);
    }
}
