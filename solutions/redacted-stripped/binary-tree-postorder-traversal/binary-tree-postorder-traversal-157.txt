class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> newList = new ArrayList<Integer>();
        if (root != null) {
            if (root.left != null)
                newList.addAll(method1(root.left));
            if (root.right != null)
                newList.addAll(method1(root.right));
            newList.add(root.val);
            return newList;
        }
        return newList;
    }
}
