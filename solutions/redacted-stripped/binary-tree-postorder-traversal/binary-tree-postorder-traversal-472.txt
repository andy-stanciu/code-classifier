class Solution {

    public void method1(TreeNode node) {
        if (node == null) {
            return;
        }
        method2(node.left);
        method2(node.right);
        li.add(node.val);
    }

    List<Integer> li = new ArrayList<>();

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return li;
    }
}
