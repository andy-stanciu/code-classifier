class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        pre(root, ans);
        return ans;
    }

    public static void pre(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        pre(root.left, ans);
        pre(root.right, ans);
        ans.add(root.val);
    }
}
