class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    public void method2(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        method2(root.left, list);
        method2(root.right, list);
        list.add(root.val);
    }
}
