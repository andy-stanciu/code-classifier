class Solution {

    public void post(List<Integer> output, TreeNode root) {
        if (root == null) {
            return;
        }
        post(output, root.left);
        post(output, root.right);
        output.add(root.val);
        return;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new ArrayList<>();
        post(output, root);
        return output;
    }
}
