class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans;
    }

    private void method2(TreeNode node, List<Integer> ans) {
        if (node == null)
            return;
        method2(node.left, ans);
        method2(node.right, ans);
        ans.add(node.val);
    }
}
