class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        solve(root, ans);
        return ans;
    }

    public void solve(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        solve(root.left, ans);
        solve(root.right, ans);
        ans.add(root.val);
    }
}
