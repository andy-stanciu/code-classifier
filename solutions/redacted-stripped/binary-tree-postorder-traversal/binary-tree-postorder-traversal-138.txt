class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        Stack<TreeNode> s = new Stack<>();
        s.push(root);
        while (!s.isEmpty()) {
            TreeNode n = s.pop();
            ans.add(0, n.val);
            if (n.left != null)
                s.push(n.left);
            if (n.right != null)
                s.push(n.right);
        }
        return ans;
    }
}
