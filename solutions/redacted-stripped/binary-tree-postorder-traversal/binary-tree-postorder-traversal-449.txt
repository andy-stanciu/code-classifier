class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> l = new ArrayList<>();
        method2(root, l);
        return l;
    }

    private void method2(TreeNode root, ArrayList l) {
        if (root == null) {
            return;
        }
        method2(root.left, l);
        method2(root.right, l);
        l.add(root.val);
    }
}
