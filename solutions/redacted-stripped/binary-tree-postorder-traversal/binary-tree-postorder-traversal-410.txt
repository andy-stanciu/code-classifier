class Solution {

    List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        traverse(root);
        return result;
    }

    public void traverse(TreeNode root) {
        if (root == null) {
            return;
        } else {
            traverse(root.left);
            traverse(root.right);
            result.add(root.val);
        }
    }
}
