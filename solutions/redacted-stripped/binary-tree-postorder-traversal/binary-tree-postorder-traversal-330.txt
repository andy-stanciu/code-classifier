class Solution {

    public List<Integer> method1(Class1 node) {
        List<Integer> output = new ArrayList<>();
        traverse(node, output);
        return output;
    }

    private void traverse(Class1 node, List<Integer> output) {
        if (node == null) {
            return;
        }
        traverse(node.left, output);
        traverse(node.right, output);
        output.add(node.val);
    }

    public static void main(String[] args) {
        Class1 root = new Class1(1);
        root.right = new Class1(2);
        root.right.left = new Class1(3);
        Solution sol = new Solution();
        List<Integer> result = sol.method1(root);
        System.out.println(result);
    }
}

class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int x) {
        val = x;
    }
}
