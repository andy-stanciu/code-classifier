class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> L = new ArrayList<>();
        method2(root, L);
        return L;
    }

    public static void method2(TreeNode root, List<Integer> L) {
        if (root == null) {
            return;
        }
        method2(root.left, L);
        method2(root.right, L);
        L.add(root.val);
    }
}
