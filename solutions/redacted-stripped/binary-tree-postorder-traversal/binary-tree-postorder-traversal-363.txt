class Solution {

    public List<Integer> method1(TreeNode root) {
        LinkedList<Integer> list = new LinkedList<>();
        Stack<TreeNode> stack = new Stack<>();
        while (true) {
            if (root != null) {
                list.addFirst(root.val);
                stack.push(root);
                root = root.right;
            } else {
                if (stack.isEmpty())
                    break;
                root = stack.pop();
                root = root.left;
            }
        }
        return list;
    }
}
