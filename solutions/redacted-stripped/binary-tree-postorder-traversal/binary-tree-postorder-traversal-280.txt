class Solution {

    private List<Integer> arr = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root != null) {
            this.addValuesToList(root);
        }
        return this.arr;
    }

    public void addValuesToList(TreeNode root) {
        if (root.left != null) {
            addValuesToList(root.left);
        }
        if (root.right != null) {
            addValuesToList(root.right);
        }
        arr.add(root.val);
    }
}
