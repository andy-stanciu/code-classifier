class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<Integer>();
        method1(root, result);
        return result;
    }

    private void method1(TreeNode root, List<Integer> result) {
        if (root == null)
            return;
        method1(root.left, result);
        method1(root.right, result);
        result.add(root.val);
    }
}
