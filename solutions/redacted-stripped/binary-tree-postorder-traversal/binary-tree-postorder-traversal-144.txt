class Solution {

    private void helper(TreeNode root, List<Integer> output) {
        if (root == null)
            return;
        helper(root.left, output);
        helper(root.right, output);
        output.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new ArrayList<>();
        helper(root, output);
        return output;
    }
}
