class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> lst = new ArrayList<>();
        return traverse(lst, root);
    }

    public List<Integer> traverse(List<Integer> lst, TreeNode root) {
        if (root == null) {
            return lst;
        }
        traverse(lst, root.left);
        traverse(lst, root.right);
        lst.add(root.val);
        return lst;
    }
}
