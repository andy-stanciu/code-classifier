class Solution {

    public void method1(TreeNode root, List<Integer> arr) {
        if (root == null) {
            return;
        }
        method1(root.left, arr);
        method1(root.right, arr);
        arr.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> ms = new ArrayList<>();
        method1(root, ms);
        return ms;
    }
}
