class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ar = new ArrayList<Integer>();
        po(root, ar);
        return ar;
    }

    public static void po(TreeNode root, ArrayList<Integer> ar) {
        if (root == null)
            return;
        po(root.left, ar);
        po(root.right, ar);
        ar.add(root.val);
    }
}
