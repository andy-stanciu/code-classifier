class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> post = new ArrayList<Integer>();
        method2(root, post);
        return post;
    }

    static void method2(TreeNode root, List<Integer> p) {
        if (root == null)
            return;
        method2(root.left, p);
        method2(root.right, p);
        p.add(root.val);
    }
}
