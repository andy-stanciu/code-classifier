class Solution {

    private void method1(TreeNode root, List<Integer> result) {
        if (root != null) {
            method1(root.left, result);
            method1(root.right, result);
            result.add(root.val);
        }
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method1(root, result);
        return result;
    }
}
