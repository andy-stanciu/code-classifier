class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method2(root, res);
        return res;
    }

    public void method2(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        method2(root.left, res);
        method2(root.right, res);
        res.add(root.val);
    }
}
