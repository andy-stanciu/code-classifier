class Solution {

    List<Integer> ansList = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        traverse(root);
        return ansList;
    }

    public void traverse(TreeNode root) {
        if (root == null)
            return;
        traverse(root.left);
        traverse(root.right);
        ansList.add(root.val);
    }
}
