class Solution {

    List<Integer> result = new ArrayList<>();

    public void method1(TreeNode node) {
        if (node == null) {
            return;
        }
        method1(node.left);
        method1(node.right);
        result.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return result;
    }
}
