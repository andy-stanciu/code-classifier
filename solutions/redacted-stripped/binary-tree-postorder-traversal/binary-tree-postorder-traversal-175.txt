class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ar = new ArrayList<>();
        method2(root, ar);
        System.out.println(ar);
        return ar;
    }

    public void method2(TreeNode head, ArrayList<Integer> ar) {
        if (head == null)
            return;
        method2(head.left, ar);
        method2(head.right, ar);
        ar.add(head.val);
    }
}
