class Solution {

    private List<Integer> res = new ArrayList();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return res;
        if (root.left == null && root.right == null) {
            res.add(root.val);
        } else {
            if (root.left != null)
                method1(root.left);
            if (root.right != null)
                method1(root.right);
            res.add(root.val);
        }
        return res;
    }
}
