class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> stack = new Stack<TreeNode>();
        List<Integer> result = new LinkedList();
        if (root == null) {
            return result;
        }
        Stack<TreeNode> queue = new Stack<>();
        queue.add(root);
        TreeNode temp = root;
        while (!queue.isEmpty()) {
            if (temp != null) {
                queue.push(temp);
                stack.push(temp);
                temp = temp.right;
            } else {
                temp = queue.pop();
                temp = temp.left;
            }
        }
        while (!stack.isEmpty()) {
            result.add(stack.pop().val);
        }
        return result;
    }
}
