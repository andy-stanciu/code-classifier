class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        dfs(root, list);
        return list;
    }

    public void dfs(TreeNode root, List<Integer> list) {
        if (null == root) {
            return;
        }
        if (null == root.left && null == root.right) {
            list.add(root.val);
            return;
        }
        if (null != root.left) {
            dfs(root.left, list);
        }
        if (null != root.right) {
            dfs(root.right, list);
        }
        list.add(root.val);
        return;
    }
}
