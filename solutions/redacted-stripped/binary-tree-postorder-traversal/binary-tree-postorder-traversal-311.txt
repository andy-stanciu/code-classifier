class Solution {

    static List<Integer> result;

    private static void method1(TreeNode node) {
        if (node == null)
            return;
        if (node.left != null) {
            method1(node.left);
        }
        if (node.right != null) {
            method1(node.right);
        }
        result.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        result = new ArrayList<>();
        method1(root);
        return result;
    }
}
