class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        post(root, res);
        return res;
    }

    void post(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        post(root.left, res);
        post(root.right, res);
        res.add(root.val);
    }
}
