class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> res = new ArrayList<>();
        List<String> path = new ArrayList<>();
        int index = 0;
        return part(s, res, path, index);
    }

    public static List<List<String>> part(String s, List<List<String>> res, List<String> path, int index) {
        if (index == s.length()) {
            res.add(new ArrayList<>(path));
            return res;
        }
        for (int i = index; i < s.length(); ++i) {
            if (isPalli(s, index, i)) {
                path.add(s.substring(index, i + 1));
                part(s, res, path, i + 1);
                path.remove(path.size() - 1);
            }
        }
        return res;
    }

    public static boolean isPalli(String s, int start, int end) {
        while (start <= end) {
            if (s.charAt(start++) != s.charAt(end--)) {
                return false;
            }
        }
        return true;
    }
}
