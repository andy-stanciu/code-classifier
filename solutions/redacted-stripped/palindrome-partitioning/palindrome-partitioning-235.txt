class Solution {

    List<List<String>> ar = new ArrayList<>();

    public List<List<String>> partition(String s) {
        check(s, new ArrayList<>(), 0);
        return ar;
    }

    public void check(String str, List<String> a, int i) {
        if (i == str.length()) {
            ar.add(new ArrayList<String>(a));
        }
        for (int j = i; j < str.length(); j++) {
            if (pal(str.substring(i, j + 1))) {
                a.add(str.substring(i, j + 1));
                check(str, a, j + 1);
                a.remove(a.size() - 1);
            }
        }
    }

    public boolean pal(String s) {
        for (int i = 0; i < s.length() / 2; i++) {
            if (s.charAt(i) != s.charAt(s.length() - 1 - i)) {
                return false;
            }
        }
        return true;
    }
}
