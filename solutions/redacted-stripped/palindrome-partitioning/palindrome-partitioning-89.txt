class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        List<String> path = new ArrayList<>();
        StringPartition(0, s, path, ans);
        return ans;
    }

    private void StringPartition(int index, String s, List<String> path, List<List<String>> ans) {
        if (index == s.length()) {
            ans.add(new ArrayList<>(path));
            return;
        }
        for (int i = index; i < s.length(); i++) {
            if (isPalindrom(index, i, s)) {
                path.add(s.substring(index, i + 1));
                StringPartition(i + 1, s, path, ans);
                path.remove(path.size() - 1);
            }
        }
    }

    private boolean isPalindrom(int start, int end, String s) {
        while (start <= end) {
            if (s.charAt(start++) != s.charAt(end--)) {
                return false;
            }
        }
        return true;
    }
}
