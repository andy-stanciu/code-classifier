class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        List<String> subset = new ArrayList<>();
        soln(0, s, ans, subset);
        return ans;
    }

    private void soln(int idx, String s, List<List<String>> ans, List<String> subset) {
        if (idx == s.length()) {
            ans.add(new ArrayList<>(subset));
            return;
        }
        for (int i = idx; i < s.length(); i++) {
            if (isPallindrome(s, idx, i)) {
                subset.add(s.substring(idx, i + 1));
                soln(i + 1, s, ans, subset);
                subset.remove(subset.size() - 1);
            }
        }
    }

    private boolean isPallindrome(String s, int start, int end) {
        while (start <= end) {
            if (s.charAt(start) != s.charAt(end))
                return false;
            start++;
            end--;
        }
        return true;
    }
}
