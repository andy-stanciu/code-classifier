class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> s = new Stack();
        for (int i = 0; i < param1.length; i++) {
            if (s.isEmpty() || param1[i] > 0) {
                s.push(param1[i]);
            } else {
                while (true) {
                    int peek = s.peek();
                    if (peek < 0) {
                        s.push(param1[i]);
                        break;
                    } else if (peek > -param1[i]) {
                        break;
                    } else if (peek == -param1[i]) {
                        s.pop();
                        break;
                    } else {
                        s.pop();
                        if (s.isEmpty()) {
                            s.push(param1[i]);
                            break;
                        }
                    }
                }
            }
        }
        int[] output_arr = new int[s.size()];
        for (int i = s.size() - 1; i >= 0; i--) {
            output_arr[i] = s.pop();
        }
        return output_arr;
    }
}
