class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> s = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (s.empty())
                s.push(param1[i]);
            else if (s.peek() > 0 && param1[i] < 0) {
                if (Math.abs(param1[i]) == Math.abs(s.peek()))
                    s.pop();
                else if (Math.abs(param1[i]) > Math.abs(s.peek())) {
                    s.pop();
                    i--;
                }
            } else
                s.push(param1[i]);
        }
        if (s.empty())
            return new int[0];
        int[] res = new int[s.size()];
        for (int i = s.size() - 1; i >= 0; i--) res[i] = s.pop();
        return res;
    }
}
