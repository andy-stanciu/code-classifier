class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int var1 : param1) {
            boolean alive = true;
            while (!st.isEmpty() && var1 < 0 && st.peek() > 0 && alive) {
                if (st.peek() < Math.abs(var1)) {
                    st.pop();
                } else if (st.peek() == Math.abs(var1)) {
                    st.pop();
                    alive = false;
                } else {
                    alive = false;
                }
            }
            if (alive)
                st.push(var1);
        }
        int[] result = new int[st.size()];
        for (int i = st.size() - 1; i >= 0; i--) {
            result[i] = st.pop();
        }
        return result;
    }
}
