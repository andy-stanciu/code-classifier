class Solution {

    public int[] method1(int[] param1) {
        ArrayList<Integer> cList = new ArrayList<Integer>();
        for (int i = 0; i < param1.length; i++) cList.add(param1[i]);
        int left = 0, ptr = 1, skips = 0;
        while (ptr < cList.size()) {
            if (cList.get(left) < 0) {
                left++;
                ptr = Math.max(ptr, left + 1);
            } else if (cList.get(ptr) < 0 && skips == 0 && Math.abs(cList.get(ptr)) > cList.get(left)) {
                cList.remove(left);
            } else if (cList.get(ptr) < 0 && skips > 0 && Math.abs(cList.get(ptr)) > cList.get(left + skips)) {
                cList.remove(left + skips);
                skips--;
                ptr = Math.max(ptr - 1, left + 1);
            } else if (cList.get(ptr) < 0 && skips == 0 && cList.get(left) > Math.abs(cList.get(ptr))) {
                cList.remove(ptr);
            } else if (cList.get(ptr) < 0 && skips > 0 && cList.get(left + skips) > Math.abs(cList.get(ptr))) {
                cList.remove(ptr);
            } else if (cList.get(ptr) < 0 && skips == 0 && Math.abs(cList.get(ptr)) == cList.get(left)) {
                cList.remove(ptr);
                cList.remove(left);
            } else if (cList.get(ptr) < 0 && skips > 0 && cList.get(left + skips) == Math.abs(cList.get(ptr))) {
                cList.remove(ptr);
                cList.remove(left + skips);
                skips--;
                ptr--;
            } else if (cList.get(ptr) > 0) {
                skips++;
                ptr++;
            }
        }
        int[] result = new int[cList.size()];
        for (int i = 0; i < cList.size(); i++) result[i] = cList.get(i);
        return result;
    }
}
