class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<Integer>();
        stack.push(0);
        for (int i = 0; i < param1.length; i++) {
            if (stack.peek() > 0 && stack.peek() == -param1[i])
                stack.pop();
            else if (stack.peek() > 0 && param1[i] < 0) {
                while (!stack.isEmpty() && stack.peek() > 0 && Math.abs(param1[i]) > stack.peek()) stack.pop();
                if (!stack.isEmpty() && Math.abs(param1[i]) == stack.peek() && stack.peek() > 0)
                    stack.pop();
                else if (stack.peek() <= 0)
                    stack.push(param1[i]);
            } else
                stack.push(param1[i]);
        }
        int[] fin = new int[stack.size() - 1];
        int j = 0;
        for (int i : stack) {
            if (i == 0)
                continue;
            fin[j++] = i;
        }
        return fin;
    }
}
