class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        st.push(param1[0]);
        boolean isPart = true;
        for (int i = 1; i < param1.length; i++) {
            if (!st.isEmpty() && param1[i] < 0 && st.peek() > 0) {
                if (Math.abs(param1[i]) == st.peek()) {
                    st.pop();
                    continue;
                }
                while (!st.isEmpty() && st.peek() > 0) {
                    if (Math.abs(param1[i]) == st.peek()) {
                        st.pop();
                        isPart = false;
                        break;
                    } else if (Math.abs(param1[i]) < st.peek()) {
                        isPart = false;
                        break;
                    } else {
                        st.pop();
                    }
                }
                if (isPart)
                    st.push(param1[i]);
                isPart = true;
            } else {
                st.push(param1[i]);
            }
        }
        int[] arr = new int[st.size()];
        int i = 0;
        for (Integer x : st) {
            arr[i] = x;
            i++;
        }
        return arr;
    }
}
