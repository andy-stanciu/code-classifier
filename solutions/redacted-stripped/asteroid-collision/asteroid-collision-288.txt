class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stk = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            boolean flag = true;
            while (!stk.isEmpty() && (param1[i] < 0 && stk.peek() > 0)) {
                if (-param1[i] > stk.peek()) {
                    stk.pop();
                    flag = true;
                } else if (-param1[i] < stk.peek()) {
                    flag = false;
                    break;
                } else if (-param1[i] == stk.peek()) {
                    stk.pop();
                    flag = false;
                    break;
                }
            }
            if (flag) {
                stk.push(param1[i]);
            }
        }
        int[] arr = new int[stk.size()];
        int i = stk.size() - 1;
        while (!stk.isEmpty()) {
            arr[i] = stk.pop();
            i--;
        }
        return arr;
    }
}
