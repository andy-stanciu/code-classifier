class Solution {

    public int[] method1(int[] param1) {
        int len = param1.length;
        int top = -1;
        for (int pos = 0; pos < len; pos++) {
            int var1 = param1[pos];
            if (var1 > 0) {
                param1[++top] = param1[pos];
            } else {
                boolean needAdd = true;
                while (top > -1 && param1[top] > 0) {
                    int sum = param1[top] + var1;
                    if (sum <= 0) {
                        top--;
                    }
                    if (sum >= 0) {
                        needAdd = false;
                        break;
                    }
                }
                if (needAdd)
                    param1[++top] = param1[pos];
            }
        }
        return Arrays.copyOf(param1, top + 1);
    }
}
