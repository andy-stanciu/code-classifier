class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int i : param1) {
            if (st.isEmpty() || i > 0) {
                st.push(i);
            } else {
                while (true) {
                    int peek = st.peek();
                    if (peek < 0) {
                        st.push(i);
                        break;
                    } else if (peek > -i) {
                        break;
                    } else if (peek == -i) {
                        st.pop();
                        break;
                    } else {
                        st.pop();
                        if (st.isEmpty()) {
                            st.push(i);
                            break;
                        }
                    }
                }
            }
        }
        int[] output_arr = new int[st.size()];
        for (int i = st.size() - 1; i >= 0; i--) output_arr[i] = st.pop();
        return output_arr;
    }
}
