class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int element : param1) {
            while (!stack.isEmpty() && element < 0 && stack.peek() > 0) {
                if (Math.abs(element) > stack.peek()) {
                    stack.pop();
                } else if (Math.abs(element) == stack.peek()) {
                    stack.pop();
                    element = 0;
                } else {
                    element = 0;
                }
            }
            if (element != 0) {
                stack.push(element);
            }
        }
        return stack.stream().mapToInt(Integer::intValue).toArray();
    }
}
