class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            int curr = param1[i];
            if (curr > 0) {
                st.push(curr);
            } else {
                while (!st.isEmpty() && st.peek() > 0 && st.peek() < -curr) {
                    st.pop();
                }
                if (st.isEmpty() || st.peek() < 0) {
                    st.push(curr);
                }
                if (st.peek() == -curr) {
                    st.pop();
                }
            }
        }
        int[] ans = new int[st.size()];
        for (int i = st.size() - 1; i >= 0; i--) {
            ans[i] = st.pop();
        }
        return ans;
    }
}
