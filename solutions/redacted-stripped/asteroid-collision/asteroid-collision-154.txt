class Solution {

    public boolean check(int a, int b) {
        if (a > 0 && b < 0)
            return true;
        return false;
    }

    public int[] method1(int[] param1) {
        Stack<Integer> s = new Stack<>();
        s.add(param1[0]);
        for (int i = 1; i < param1.length; i++) {
            boolean flag = true;
            boolean check2 = true;
            while (s.isEmpty() == false && check(s.peek(), param1[i]) && check2) {
                if (Math.abs(s.peek()) < Math.abs(param1[i])) {
                    s.pop();
                    flag = false;
                } else if (Math.abs(s.peek()) == Math.abs(param1[i])) {
                    s.pop();
                    flag = false;
                    check2 = false;
                } else {
                    flag = false;
                    check2 = false;
                }
            }
            if (flag || check2)
                s.add(param1[i]);
        }
        int[] res = new int[s.size()];
        while (s.size() > 0) {
            res[s.size() - 1] = s.pop();
        }
        return res;
    }
}
