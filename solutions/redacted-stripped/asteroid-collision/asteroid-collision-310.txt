class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int a : param1) {
            if (a > 0)
                st.push(a);
            else {
                while (!st.isEmpty() && st.peek() > 0 && st.peek() < Math.abs(a)) st.pop();
                if (st.isEmpty() || st.peek() < 0)
                    st.push(a);
                else if (st.peek() + a == 0)
                    st.pop();
            }
        }
        int[] ans = new int[st.size()];
        for (int i = ans.length - 1; i >= 0; i--) ans[i] = st.pop();
        return ans;
    }
}
