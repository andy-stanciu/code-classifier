class Solution {

    public int[] method1(int[] param1) {
        int i = 0;
        int n = param1.length;
        Stack<Integer> st = new Stack<>();
        while (i < n) {
            if (param1[i] < 0) {
                if (st.isEmpty() || st.peek() < 0) {
                    st.push(param1[i]);
                    i++;
                } else if (Math.abs(param1[i]) == st.peek()) {
                    st.pop();
                    i++;
                } else if (st.peek() > Math.abs(param1[i])) {
                    i++;
                } else if (st.peek() < Math.abs(param1[i])) {
                    st.pop();
                }
            } else if (param1[i] > 0) {
                st.push(param1[i]);
                i++;
            }
        }
        int[] arr = new int[st.size()];
        for (int j = st.size() - 1; j >= 0; j--) {
            arr[j] = st.pop();
        }
        return arr;
    }
}
