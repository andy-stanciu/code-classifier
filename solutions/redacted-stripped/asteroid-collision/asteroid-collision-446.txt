class Solution {

    public int[] method1(int[] param1) {
        int n = param1.length;
        Stack<Integer> s = new Stack<>();
        s.push(param1[0]);
        for (int i = 1; i < n; i++) {
            if (param1[i] > 0) {
                s.push(param1[i]);
            } else {
                if (s.isEmpty() || s.peek() < 0) {
                    s.push(param1[i]);
                    continue;
                }
                if (s.peek() > Math.abs(param1[i]))
                    continue;
                else if (s.peek() < Math.abs(param1[i])) {
                    s.pop();
                    i--;
                } else {
                    s.pop();
                }
            }
        }
        int[] r = new int[s.size()];
        for (int i = s.size() - 1; i >= 0; i--) {
            r[i] = s.pop();
        }
        return r;
    }
}
