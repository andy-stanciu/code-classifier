class Solution {

    public int[] method1(int[] param1) {
        int n = param1.length;
        int top = -1;
        int[] stk = new int[n];
        for (int c : param1) {
            if (c > 0) {
                stk[++top] = c;
            } else {
                while (top >= 0 && stk[top] > 0 && stk[top] < -c) {
                    top--;
                }
                if (top >= 0 && stk[top] == -c) {
                    top--;
                } else if (top == -1 || stk[top] < 0) {
                    stk[++top] = c;
                }
            }
        }
        if (top < 0)
            return new int[0];
        int[] res = new int[top + 1];
        while (top >= 0) {
            res[top] = stk[top];
            top--;
        }
        return res;
    }
}
