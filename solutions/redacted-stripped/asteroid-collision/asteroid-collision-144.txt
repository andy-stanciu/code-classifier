class Solution {

    public int[] method1(int[] param1) {
        List<Integer> list = new ArrayList<>();
        Stack<Integer> stack = new Stack<>();
        for (int ast : param1) {
            if (ast > 0) {
                stack.push(ast);
            } else {
                boolean dest = false;
                while (!stack.isEmpty() && stack.peek() > 0) {
                    if (stack.peek() < Math.abs(ast)) {
                        stack.pop();
                        dest = false;
                    } else if (stack.peek() == Math.abs(ast)) {
                        stack.pop();
                        dest = true;
                        break;
                    } else {
                        dest = true;
                        break;
                    }
                }
                if (dest == false) {
                    stack.push(ast);
                }
            }
        }
        while (!stack.isEmpty()) list.add(stack.pop());
        int n = list.size();
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            ans[i] = list.get(n - i - 1);
        }
        return ans;
    }
}
