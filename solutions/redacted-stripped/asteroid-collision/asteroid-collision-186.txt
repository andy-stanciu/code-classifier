class Solution {

    public int[] method1(int[] param1) {
        List<Integer> listWithAsteriods = new ArrayList<>(Arrays.stream(param1).boxed().toList());
        for (int i = 1; i < listWithAsteriods.size(); ) {
            int var1 = listWithAsteriods.get(i - 1);
            int var2 = listWithAsteriods.get(i);
            if ((var1 > 0 && var2 > 0) || (var1 < 0 && var2 < 0) || var1 < 0 && var2 > 0) {
                i++;
                continue;
            }
            var1 = Math.abs(var1);
            var2 = Math.abs(var2);
            if (var1 == var2) {
                listWithAsteriods.remove(i);
                listWithAsteriods.remove(i - 1);
                if (i != 1)
                    i--;
            } else if (var1 > var2)
                listWithAsteriods.remove(i);
            else {
                listWithAsteriods.remove(i - 1);
                if (i != 1)
                    i--;
            }
        }
        return listWithAsteriods.stream().mapToInt(Integer::intValue).toArray();
    }
}
