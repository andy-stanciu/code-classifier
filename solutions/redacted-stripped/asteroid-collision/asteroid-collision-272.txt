class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (stack.isEmpty() || param1[i] > 0) {
                stack.push(param1[i]);
            } else {
                while (!stack.isEmpty()) {
                    int top = stack.peek();
                    if (top < 0) {
                        stack.push(param1[i]);
                        break;
                    }
                    int num = Math.abs(param1[i]);
                    if (num == top) {
                        stack.pop();
                        break;
                    } else if (num > top) {
                        stack.pop();
                        if (stack.isEmpty()) {
                            stack.push(param1[i]);
                            break;
                        }
                    } else {
                        break;
                    }
                }
            }
        }
        int len = stack.size();
        int[] ans = new int[len];
        for (int i = len - 1; i >= 0; i--) {
            ans[i] = stack.peek();
            stack.pop();
        }
        return ans;
    }
}
