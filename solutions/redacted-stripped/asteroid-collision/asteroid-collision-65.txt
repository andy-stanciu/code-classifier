class Solution {

    public int[] method1(int[] param1) {
        int n = param1.length;
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < n; i++) {
            if (param1[i] > 0) {
                st.push(param1[i]);
            } else {
                while (!st.isEmpty() && st.peek() > 0 && st.peek() < (param1[i] * -1)) {
                    st.pop();
                }
                if (st.isEmpty() || (!st.isEmpty() && st.peek() < 0)) {
                    st.push(param1[i]);
                }
                if (st.peek() == (param1[i] * -1)) {
                    st.pop();
                }
            }
        }
        int[] sol = new int[st.size()];
        for (int i = (st.size() - 1); i >= 0; i--) {
            sol[i] = st.pop();
        }
        return sol;
    }
}
