class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            boolean var1 = true;
            while (param1[i] < 0 && !stack.isEmpty() && stack.peek() > 0) {
                int prev = stack.peek();
                if (prev < -param1[i]) {
                    stack.pop();
                } else if (prev == -param1[i]) {
                    var1 = false;
                    stack.pop();
                    break;
                } else {
                    var1 = false;
                    break;
                }
            }
            if (var1) {
                stack.push(param1[i]);
            }
        }
        int[] res = new int[stack.size()];
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = stack.pop();
        }
        return res;
    }
}
