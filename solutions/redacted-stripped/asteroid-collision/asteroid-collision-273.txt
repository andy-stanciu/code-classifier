class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack();
        for (int i : param1) {
            if (stack.isEmpty() || i > 0)
                stack.push(i);
            else {
                while (true) {
                    int peek = stack.peek();
                    if (peek == -i) {
                        stack.pop();
                        break;
                    } else if (peek < 0) {
                        stack.push(i);
                        break;
                    } else if (peek > -i)
                        break;
                    else {
                        stack.pop();
                        if (stack.isEmpty()) {
                            stack.push(i);
                            break;
                        }
                    }
                }
            }
        }
        int[] ans = new int[stack.size()];
        for (int i = ans.length - 1; i >= 0; i--) ans[i] = stack.pop();
        return ans;
    }
}
