class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] > 0)
                stack.add(param1[i]);
            else {
                while (!stack.isEmpty() && stack.peek() > 0 && stack.peek() < Math.abs(param1[i])) stack.pop();
                if (stack.isEmpty() || stack.peek() < 0)
                    stack.add(param1[i]);
                else if (stack.peek() == Math.abs(param1[i])) {
                    stack.pop();
                }
            }
        }
        int[] ans = new int[stack.size()];
        for (int i = ans.length - 1; i >= 0; i--) ans[i] = stack.pop();
        return ans;
    }
}
