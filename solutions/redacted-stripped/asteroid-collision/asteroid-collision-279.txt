class Solution {

    public int[] method1(int[] param1) {
        if (param1.length < 2) {
            return param1;
        }
        int i = 0;
        int j = 1;
        while (j < param1.length) {
            if (i >= 0 && param1[i] > 0 && param1[j] < 0) {
                if (param1[i] > -1 * param1[j]) {
                    j += 1;
                    continue;
                } else if (param1[i] < -1 * param1[j]) {
                    i -= 1;
                    continue;
                } else {
                    j += 1;
                    i -= 1;
                }
            } else {
                i += 1;
                param1[i] = param1[j];
                j += 1;
            }
        }
        return Arrays.copyOf(param1, i + 1);
    }
}
