class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> ashu = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (!ashu.isEmpty() && ashu.peek() > 0 && param1[i] < 0) {
                if (Math.abs(param1[i]) == Math.abs(ashu.peek())) {
                    ashu.pop();
                } else if (Math.abs(param1[i]) > Math.abs(ashu.peek())) {
                    ashu.pop();
                    i--;
                }
            } else {
                ashu.push(param1[i]);
            }
        }
        int resultSize = ashu.size();
        int[] result = new int[resultSize];
        for (int i = resultSize - 1; i >= 0; i--) {
            result[i] = ashu.pop();
        }
        return result;
    }
}
