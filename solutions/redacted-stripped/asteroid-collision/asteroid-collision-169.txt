class Solution {

    public int[] method1(int[] param1) {
        int[] ans = null;
        Stack<Integer> stac = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] > 0)
                stac.push(param1[i]);
            else {
                while (!stac.isEmpty() && stac.peek() > 0 && stac.peek() < Math.abs(param1[i])) {
                    stac.pop();
                    ;
                }
                if (!stac.isEmpty() && stac.peek() == Math.abs(param1[i])) {
                    stac.pop();
                } else if (stac.isEmpty() || stac.peek() < 0) {
                    stac.add(param1[i]);
                }
            }
        }
        ans = new int[stac.size()];
        for (int i = 0; i < stac.size(); i++) {
            ans[i] = stac.get(i);
        }
        return ans;
    }
}
