class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> s = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            while (!s.isEmpty() && (s.peek() > 0 && param1[i] < 0)) {
                if (Math.abs(s.peek()) < Math.abs(param1[i])) {
                    s.pop();
                } else {
                    break;
                }
            }
            if (!s.isEmpty() && (s.peek() > 0 && param1[i] < 0) && Math.abs(s.peek()) == Math.abs(param1[i])) {
                s.pop();
            } else if (!s.isEmpty() && (s.peek() > 0 && param1[i] < 0) && Math.abs(s.peek()) > Math.abs(param1[i])) {
            } else {
                s.push(param1[i]);
            }
        }
        int[] ans = new int[s.size()];
        int k = s.size() - 1;
        while (!s.isEmpty()) {
            ans[k] = s.pop();
            k--;
        }
        return ans;
    }
}
