class Solution {

    public int method1(int[] nums) {
        if (nums.length < 2)
            return nums.length;
        int j = 0;
        int i = 1;
        while (i < nums.length) {
            if (nums[i] == nums[j]) {
                i++;
            } else {
                j++;
                nums[j] = nums[i];
                i++;
            }
        }
        int[] B = Arrays.copyOf(nums, j + 1);
        return B.length;
    }
}
