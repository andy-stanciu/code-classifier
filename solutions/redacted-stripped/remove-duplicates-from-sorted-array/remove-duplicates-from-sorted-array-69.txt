class Solution {

    public int method1(int[] nums) {
        int i = 1, j = 1;
        while (j < nums.length) {
            while (j < nums.length && nums[j - 1] == nums[j]) j++;
            if (j < nums.length) {
                nums[i] = nums[j];
                i++;
                j++;
            }
        }
        return i;
    }
}
