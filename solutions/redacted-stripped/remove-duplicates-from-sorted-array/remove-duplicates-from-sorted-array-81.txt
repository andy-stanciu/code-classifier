class Solution {

    public int method1(int[] arr) {
        if (arr.length == 0) {
            return 0;
        }
        int k = 1;
        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i] != arr[i + 1]) {
                arr[k] = arr[i + 1];
                k++;
            }
        }
        return k;
    }
}
