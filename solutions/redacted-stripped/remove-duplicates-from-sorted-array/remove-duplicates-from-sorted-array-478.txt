class Solution {

    public int method1(int[] nums) {
        int size = nums.length;
        if (size < 1)
            return 0;
        int slow = 0;
        for (int fast = 0; fast < size; fast++) {
            if (nums[fast] != nums[slow]) {
                slow++;
                nums[slow] = nums[fast];
            }
        }
        return slow + 1;
    }
}
