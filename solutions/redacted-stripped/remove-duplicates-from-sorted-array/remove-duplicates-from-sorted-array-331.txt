class Solution {

    public int method1(int[] nums) {
        int duplicateIndex = 1;
        int max = nums[0];
        for (int i = 1; i < nums.length; i++) {
            int element = nums[i];
            if (element != max) {
                nums[duplicateIndex] = element;
                duplicateIndex++;
                max = element;
            }
        }
        return duplicateIndex;
    }
}
