class Solution {

    public int method1(int[] nums) {
        int slow = 0;
        int fast = 0;
        while (fast < nums.length) {
            int curr = nums[fast];
            while (fast < nums.length && nums[fast] == curr) {
                fast++;
            }
            nums[slow] = curr;
            slow++;
        }
        return slow;
    }
}
