class Solution {

    public int method1(int[] nums) {
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            int val = nums[i];
            while (i < nums.length - 1 && val == nums[i + 1]) {
                i++;
            }
            nums[j++] = val;
        }
        return j;
    }
}
