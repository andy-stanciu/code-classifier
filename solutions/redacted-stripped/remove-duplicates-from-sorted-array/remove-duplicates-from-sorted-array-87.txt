class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int start = 0;
        int end = 1;
        int count = 1;
        while (end < n) {
            if (nums[start] == nums[end]) {
                end++;
            } else {
                start++;
                nums[start] = nums[end];
                count++;
            }
        }
        return count;
    }
}
