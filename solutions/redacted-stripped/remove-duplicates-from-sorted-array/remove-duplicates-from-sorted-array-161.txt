class Solution {

    public int method1(int[] nums) {
        int res = 1;
        for (int i = 0, num = nums[0]; i < nums.length; i++) {
            int num1 = nums[i];
            if (num1 > num) {
                nums[res++] = num1;
                num = num1;
            }
        }
        return res;
    }
}
