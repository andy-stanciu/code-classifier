class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return 1;
        }
        int p = 1;
        int fp = 1;
        int val = nums[0];
        while (fp < nums.length && p < nums.length) {
            if (nums[fp] == val) {
                fp++;
            } else {
                val = nums[fp];
                nums[p] = nums[fp];
                p++;
                fp++;
            }
        }
        return p;
    }
}
