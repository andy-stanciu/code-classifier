class Solution {

    public int method1(int[] nums) {
        int current = nums[0];
        int index = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == current)
                continue;
            nums[index] = nums[i];
            current = nums[i];
            index++;
        }
        return index;
    }
}
