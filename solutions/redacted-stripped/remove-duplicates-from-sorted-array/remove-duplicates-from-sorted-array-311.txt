class Solution {

    public int method1(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }
        int prevVal = nums[0];
        int prevIndex = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != prevVal) {
                prevVal = nums[i];
                nums[++prevIndex] = prevVal;
            }
        }
        return prevIndex + 1;
    }
}
