class Solution {

    public int method1(int[] nums) {
        int cont = -111, j = 0;
        for (int a = 0; a < nums.length; a++) {
            nums[j] = nums[a];
            if (cont != nums[a]) {
                cont = nums[a];
                j++;
            } else if (j != 0) {
                nums[a] = 0;
            }
        }
        return j;
    }
}
