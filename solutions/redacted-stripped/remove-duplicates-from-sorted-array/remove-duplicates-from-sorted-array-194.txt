class Solution {

    public int method1(int[] nums) {
        if (nums.length < 2)
            return 1;
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[k - 1] != nums[i]) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
