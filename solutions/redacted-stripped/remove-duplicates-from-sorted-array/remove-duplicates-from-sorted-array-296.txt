class Solution {

    public int method1(int[] nums) {
        int j = -1, k = 0, z = 0;
        for (int i = 0; i < nums.length; i++) {
            if (j == -1) {
                nums[++j] = nums[i];
                k++;
            }
            if (nums[i] > nums[j]) {
                nums[++j] = nums[i];
                k++;
            }
        }
        return k;
    }
}
