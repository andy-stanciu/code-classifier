class Solution {

    public int method1(int[] nums) {
        ArrayList<Integer> newList = new ArrayList<>();
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] != nums[i + 1]) {
                newList.add(nums[i]);
            }
        }
        newList.add(nums[nums.length - 1]);
        for (int i = 0; i < newList.size(); i++) {
            nums[i] = newList.get(i);
        }
        return newList.size();
        int i = 0;
        int j = i + 1;
        while (j < nums.length) {
            if (nums[i] == nums[j])
                j++;
            else if (nums[i] != nums[j]) {
                i++;
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                j++;
            }
        }
        return i + 1;
    }
}
