class Solution {

    public int method1(int[] nums) {
        int DiffElement = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[DiffElement] = nums[i];
                DiffElement++;
            }
        }
        return DiffElement;
    }
}
