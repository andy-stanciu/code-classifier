class Solution {

    public int[] var1;

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        var1 = new int[nums.length];
        int uniqueCount = 0;
        var1[uniqueCount] = nums[0];
        uniqueCount++;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != var1[uniqueCount - 1]) {
                var1[uniqueCount] = nums[i];
                uniqueCount++;
            }
        }
        for (int i = 0; i < uniqueCount; i++) {
            nums[i] = var1[i];
        }
        return uniqueCount;
    }
}
