class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int a = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[a - 1]) {
                nums[a] = nums[i];
                a++;
            }
        }
        return a;
    }
}
