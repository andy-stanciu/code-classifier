class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int j = 1;
        int count = 1;
        while (j < nums.length) {
            if (nums[i] != nums[j]) {
                count++;
                nums[i + 1] = nums[j];
                i++;
            }
            j++;
        }
        return count;
    }
}
