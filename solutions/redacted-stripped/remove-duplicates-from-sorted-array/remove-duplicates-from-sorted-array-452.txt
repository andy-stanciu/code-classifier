class Solution {

    public int method1(int[] arr) {
        int s = 0, e = 0;
        for (e = 0; e < arr.length; e++) {
            if (arr[s] == arr[e]) {
                continue;
            } else {
                s++;
                arr[s] = arr[e];
            }
        }
        return s + 1;
    }
}
