class Solution {

    public int method1(int[] arr) {
        int j = 0;
        HashSet<Integer> hs = new HashSet<>();
        for (int val : arr) hs.add(val);
        for (int val : arr) if (hs.contains(val)) {
            arr[j++] = val;
            hs.remove(val);
        }
        return j;
    }
}
