class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i < nums.length - 1 && nums[i] == nums[i + 1]) {
                continue;
            }
            nums[j] = nums[i];
            j++;
        }
        return j;
    }
}
