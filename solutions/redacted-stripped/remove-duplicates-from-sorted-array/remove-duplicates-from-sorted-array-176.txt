class Solution {

    public int method1(int[] nums) {
        if (nums.length < 1)
            return 0;
        int result = 1;
        for (int i = 0; (i) < nums.length - 1; i++) {
            if (nums[i] != nums[i + 1]) {
                nums[result] = nums[i + 1];
                result++;
            }
        }
        return result;
    }
}
