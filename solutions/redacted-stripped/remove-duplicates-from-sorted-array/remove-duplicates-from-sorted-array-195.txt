class Solution {

    public int method1(int[] nums) {
        int prev = nums[0];
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != prev) {
                nums[k] = nums[i];
                k++;
                prev = nums[i];
            }
        }
        return k;
    }
}
