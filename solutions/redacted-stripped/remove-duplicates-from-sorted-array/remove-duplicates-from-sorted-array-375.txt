class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int previousInt = -101;
        for (int i = 0; i < nums.length; i++) {
            if (previousInt != nums[i]) {
                previousInt = nums[i];
                nums[count] = nums[i];
                count++;
            }
        }
        return count;
    }
}
