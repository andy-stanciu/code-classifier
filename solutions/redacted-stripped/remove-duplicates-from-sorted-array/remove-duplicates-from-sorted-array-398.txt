class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int count = 1;
        int prev = nums[0];
        for (int num : nums) {
            if (num != prev) {
                nums[count] = num;
                count++;
            }
            prev = num;
        }
        return count;
    }
}
