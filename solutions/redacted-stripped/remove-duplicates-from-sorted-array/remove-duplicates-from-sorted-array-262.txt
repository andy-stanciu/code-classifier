class Solution {

    public int method1(int[] nums) {
        int rd = 0;
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[rd]) {
                rd++;
                count++;
                nums[rd] = nums[i];
            }
        }
        return count;
    }
}
