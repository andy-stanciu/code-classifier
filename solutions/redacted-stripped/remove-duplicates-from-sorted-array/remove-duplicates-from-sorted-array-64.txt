class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (mp.containsKey(nums[i])) {
                mp.put(nums[i], mp.get(nums[i]) + 1);
            } else {
                mp.put(nums[i], 1);
            }
            if (mp.get(nums[i]) == 1) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
