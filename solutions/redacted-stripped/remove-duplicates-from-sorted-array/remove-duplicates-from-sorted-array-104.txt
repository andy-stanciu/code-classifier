class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n < 2)
            return n;
        int ans = 1, j = 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] != nums[j]) {
                j++;
                nums[j] = nums[i];
                ans++;
            }
        }
        return ans;
    }
}
