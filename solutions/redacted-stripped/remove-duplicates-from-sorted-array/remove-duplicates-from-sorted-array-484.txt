class Solution {

    public int method1(int[] nums) {
        int i = 1;
        int val = nums[0];
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != val) {
                nums[i] = nums[j];
                val = nums[j];
                i++;
            }
        }
        return i;
    }
}
