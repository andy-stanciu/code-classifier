class Solution {

    public int method1(int[] nums) {
        int prev = nums[0];
        int uniqueidx = 1;
        for (int i = 1; i < nums.length; i++) {
            if (prev != nums[i]) {
                nums[uniqueidx++] = nums[i];
            }
            prev = nums[i];
        }
        return uniqueidx;
    }
}
