class Solution {

    public int method1(int[] nums) {
        int N = nums.length;
        if (N == 0 || N == 1) {
            return N;
        }
        int i = 0;
        int j = 1;
        while (j < N) {
            if (nums[i] != nums[j]) {
                nums[++i] = nums[j];
            }
            j++;
        }
        return i + 1;
    }
}
