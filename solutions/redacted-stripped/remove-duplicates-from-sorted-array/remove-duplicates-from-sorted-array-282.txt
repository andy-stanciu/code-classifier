class Solution {

    public int method1(int[] nums) {
        int front = 0;
        int back = 0;
        int count = 0;
        int x = 0;
        while (front < nums.length) {
            while (front < nums.length && nums[front] == nums[back]) {
                front++;
            }
            count++;
            nums[x++] = nums[back];
            back = front;
        }
        return count;
    }
}
