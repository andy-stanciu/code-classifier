public class Solution {

    public int method1(int[] nums, int writeIndex, int readIndex) {
        if (readIndex == nums.length) {
            return writeIndex;
        }
        if (readIndex == 0 || nums[readIndex] != nums[readIndex - 1]) {
            nums[writeIndex] = nums[readIndex];
            writeIndex++;
        }
        return method1(nums, writeIndex, readIndex + 1);
    }

    public int method2(int[] nums) {
        return method1(nums, 0, 0);
    }

    public static void main(String[] args) {
        int[] arr = { 1, 2, 3, 4, 4, 5, 6 };
        Solution solution = new Solution();
        int newLength = solution.method2(arr);
        for (int i = 0; i < newLength; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}
