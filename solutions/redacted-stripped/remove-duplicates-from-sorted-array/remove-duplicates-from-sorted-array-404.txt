class Solution {

    public int method1(int[] nums) {
        List<Integer> listInt = new ArrayList(Arrays.stream(nums).boxed().collect(Collectors.toCollection(LinkedHashSet::new)));
        if (listInt.size() == 1)
            return 1;
        int pointer = 0;
        for (Integer el : listInt) {
            nums[pointer++] = el;
        }
        ;
        return listInt.size();
    }
}
