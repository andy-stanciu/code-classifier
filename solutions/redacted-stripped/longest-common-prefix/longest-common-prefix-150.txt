class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        for (String s : strs) {
            if (s.equals(var1)) {
                continue;
            }
            int end = var1.length();
            while (end >= 0) {
                var1 = var1.substring(0, end);
                if (s.startsWith(var1)) {
                    break;
                }
                end--;
            }
        }
        return var1;
    }
}
