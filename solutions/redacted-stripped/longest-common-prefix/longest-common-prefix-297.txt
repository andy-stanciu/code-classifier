class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        int m = strs.length;
        for (int a = 1; a < m; a++) {
            int n = var1.length();
            for (int i = 0; i < n; i++) {
                if (i >= strs[a].length() || var1.charAt(i) != strs[a].charAt(i)) {
                    var1 = var1.substring(0, i);
                    break;
                }
            }
        }
        return var1;
    }
}
