class Solution {

    public static String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        String shortestStr = findShortestStr(strs);
        int min = shortestStr.length();
        for (String word : strs) {
            for (int i = 0; i < min; i++) {
                if (word.charAt(i) != shortestStr.charAt(i)) {
                    min = i;
                    break;
                }
            }
        }
        return shortestStr.substring(0, min);
    }

    public static String findShortestStr(String[] strs) {
        String shortest = strs[0];
        for (String word : strs) {
            if (word.length() < shortest.length()) {
                shortest = word;
            }
        }
        return shortest;
    }
}
