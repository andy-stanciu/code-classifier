class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        String first = strs[0];
        String last = strs[strs.length - 1];
        String method1 = "";
        for (int i = 0; i < first.length(); i++) {
            char currChar = last.charAt(i);
            if (currChar == first.charAt(i)) {
                method1 += currChar;
            } else {
                break;
            }
        }
        return method1;
    }
}
