class Solution {

    public String method1(String[] strs) {
        String firstElement = strs[0];
        StringBuffer var1 = new StringBuffer();
        int loopLegth = strs[0].length();
        for (int i = 1; i < strs.length; i++) {
            loopLegth = Math.min(loopLegth, strs[i].length());
        }
        for (int i = 0; i < loopLegth; i++) {
            char checkChar = firstElement.charAt(i);
            boolean isMatches = true;
            for (int j = 1; j < strs.length; j++) {
                if (strs[j].charAt(i) != checkChar) {
                    isMatches = false;
                    break;
                }
            }
            if (isMatches) {
                var1.append(checkChar);
            } else {
                break;
            }
        }
        return var1.toString();
    }
}
