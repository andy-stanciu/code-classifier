class Solution {

    public String method1(String[] strs) {
        Map<String, Integer> var1 = new HashMap<String, Integer>();
        for (String var2 : strs) {
            StringBuilder building = new StringBuilder();
            for (char c : var2.toCharArray()) {
                building.append(c);
                var1.put(building.toString(), var1.getOrDefault(building.toString(), 0) + 1);
            }
        }
        int desiredSize = strs.length;
        int var3 = 0;
        String toReturn = "";
        for (Map.Entry<String, Integer> var4 : var1.entrySet()) {
            if (var4.getValue().equals(desiredSize)) {
                int stringSize = var4.getKey().length();
                if (stringSize > var3) {
                    var3 = stringSize;
                    toReturn = var4.getKey();
                }
            }
        }
        return toReturn;
    }
}
