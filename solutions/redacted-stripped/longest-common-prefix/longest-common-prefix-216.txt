class Solution {

    public static String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        String minStr = strs[0];
        for (String str : strs) {
            if (str.length() < minStr.length()) {
                minStr = str;
            }
        }
        for (int i = 0; i < minStr.length(); i++) {
            char c = minStr.charAt(i);
            for (String str : strs) {
                if (str.charAt(i) != c) {
                    return minStr.substring(0, i);
                }
            }
        }
        return minStr;
    }
}
