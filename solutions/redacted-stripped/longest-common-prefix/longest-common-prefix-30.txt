class Solution {

    public String method1(String[] s) {
        if (s.length == 0)
            return "";
        String var1 = s[0];
        for (int i = 1; i < s.length; i++) {
            while (!s[i].startsWith(var1)) var1 = var1.substring(0, var1.length() - 1);
            if (var1.isEmpty())
                break;
        }
        return var1;
    }
}
