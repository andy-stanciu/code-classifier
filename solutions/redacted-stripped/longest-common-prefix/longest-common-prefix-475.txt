class Solution {

    public static boolean checkSub(String[] s, String sub) {
        boolean ans = true;
        for (int i = 1; i < s.length; i++) {
            if (!s[i].startsWith(sub)) {
                ans = false;
            }
        }
        return ans;
    }

    public static String method1(String[] strs) {
        int n = strs.length;
        String sub = "";
        boolean ans = false;
        int strLen = strs[0].length();
        for (int i = strLen; i >= 0; i--) {
            sub = new String(strs[0].substring(0, i));
            if (checkSub(strs, sub)) {
                return sub;
            }
        }
        return "";
    }
}
