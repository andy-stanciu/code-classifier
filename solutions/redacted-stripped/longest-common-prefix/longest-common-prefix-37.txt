class Solution {

    public String method1(String[] strs) {
        String r = "";
        int n = strs.length;
        Arrays.sort(strs);
        String s = strs[0];
        String h = strs[n - 1];
        for (//taken smallest length so that run time will be less
        int i = 0; //taken smallest length so that run time will be less
        i < s.length(); i++) {
            if (s.charAt(i) != h.charAt(i))
                break;
            r = r + s.charAt(i);
        }
        return r;
    }
}
