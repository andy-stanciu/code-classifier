class Solution {

    public String method1(String[] strs) {
        String var1 = new String(strs[0]);
        for (int i = 1; i < strs.length && !var1.equals(""); i++) {
            while (!var1.equals("") && !strs[i].startsWith(var1)) {
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
