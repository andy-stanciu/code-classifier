class Solution {

    public String method1(String[] strs) {
        StringBuilder str = new StringBuilder();
        Arrays.sort(strs);
        String first = strs[0];
        String last = strs[strs.length - 1];
        int l = Math.min(first.length(), last.length());
        for (int i = 0; i < l; i++) {
            if (first.charAt(i) != last.charAt(i)) {
                return str.toString();
            }
            str.append(first.charAt(i));
        }
        return str.toString();
    }
}
