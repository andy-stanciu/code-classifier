class Solution {

    public String method1(String[] S) {
        if (S.length == 0)
            return "";
        String var1 = S[0];
        for (int i = 1; i < S.length; i++) while (S[i].indexOf(var1) != 0) {
            var1 = var1.substring(0, var1.length() - 1);
            if (var1.isEmpty())
                return "";
        }
        return var1;
    }
}
