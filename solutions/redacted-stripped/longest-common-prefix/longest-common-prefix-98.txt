class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0)
            return "";
        Arrays.sort(strs);
        String first = strs[0];
        String last = strs[strs.length - 1];
        int var1 = 0;
        while (var1 < first.length()) {
            if (first.charAt(var1) == last.charAt(var1)) {
                var1++;
            } else {
                break;
            }
        }
        return var1 == 0 ? "" : first.substring(0, var1);
    }
}
