class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        int n = strs.length;
        for (int i = 1; i < n; i++) {
            String word = strs[i];
            while (!word.startsWith(var1)) {
                if (var1.length() == 1)
                    return "";
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
