class Solution {

    public String method1(String[] v) {
        String var1 = v[0];
        for (int i = 1; i < v.length; i++) {
            while (v[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
            }
            if (var1.length() == 0)
                return "";
        }
        return var1;
    }
}
