class Solution {

    public String method1(String[] strs) {
        String res = strs[0];
        for (int i = 1; i < strs.length; i++) {
            res = doFind(res, strs[i]);
        }
        return res;
    }

    private String doFind(String main, String temp) {
        while (!temp.startsWith(main)) {
            main = main.substring(0, main.length() - 1);
        }
        return main;
    }
}
