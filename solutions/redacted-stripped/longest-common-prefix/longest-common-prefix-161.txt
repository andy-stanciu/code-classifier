class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        int n = strs.length - 1;
        String s = "";
        for (int i = 0; i < strs[0].length(); i++) if (strs[0].charAt(i) == strs[n].charAt(i))
            s += strs[0].charAt(i);
        else
            return s;
        return s;
    }
}
