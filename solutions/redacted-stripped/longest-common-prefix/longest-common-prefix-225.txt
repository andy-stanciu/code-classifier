class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        String one = strs[0];
        String two = strs[strs.length - 1];
        int i = 0;
        int j = 0;
        while (i < one.length() && j < two.length()) {
            if (one.charAt(i) == two.charAt(j)) {
                i++;
                j++;
            } else {
                break;
            }
        }
        return one.substring(0, j);
    }
}
