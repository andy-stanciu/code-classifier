class Solution {

    public String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        if (strs.length == 1) {
            return strs[0];
        }
        int var1 = 0;
        while (true) {
            for (int i = 0; i < strs.length - 1; i++) {
                if (var1 >= strs[i].length() || var1 >= strs[i + 1].length())
                    return strs[0].substring(0, var1);
                if (strs[i].charAt(var1) != strs[i + 1].charAt(var1))
                    return strs[0].substring(0, var1);
            }
            var1++;
        }
    }
}
