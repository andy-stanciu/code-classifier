class Solution {

    public String method1(String[] strs) {
        String start = strs[0];
        if (strs.length == 1)
            return start;
        StringBuilder sb = new StringBuilder();
        int minLen = 201;
        for (// keep track of the minimum length amongst all the strings since this will be the MAX possible common prefix length we can get
        String str : strs) minLen = Math.min(minLen, str.length());
        for (int i = 0; i < minLen; i++) {
            for (int j = 0; j < strs.length; j++) {
                if (start.charAt(i) != strs[j].charAt(i))
                    return sb.toString();
            }
            sb.append(start.charAt(i));
        }
        return sb.toString();
    }
}
