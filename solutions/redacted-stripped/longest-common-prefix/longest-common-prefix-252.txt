class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        String smallest = strs[0];
        String biggest = strs[strs.length - 1];
        int i = 0;
        while (i < smallest.length() && smallest.charAt(i) == biggest.charAt(i)) {
            i++;
        }
        return smallest.substring(0, i);
    }
}
