class Solution {

    public String method1(String[] strs) {
        String output = strs[0];
        for (int i = 1; i < strs.length & !output.equals(""); i++) {
            int j = 0;
            String curr = strs[i];
            while ((j < curr.length() & j < output.length()) && output.charAt(j) == curr.charAt(j)) {
                j++;
            }
            output = output.substring(0, j);
        }
        return output;
    }
}
