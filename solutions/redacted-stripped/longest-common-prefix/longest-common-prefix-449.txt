class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        int len = strs.length;
        String ans = "";
        String a = strs[0];
        String b = strs[len - 1];
        int i = 0;
        while (i < a.length() && i < b.length() && a.charAt(i) == b.charAt(i)) {
            i++;
        }
        return a.substring(0, i);
    }
}
