class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        for (int index = 1; index < strs.length; index++) {
            while (strs[index].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
