class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        int len = var1.length();
        int i = 0;
        do {
            while (!strs[i].startsWith(var1)) {
                var1 = var1.substring(0, --len);
            }
            i++;
        } while (i < strs.length);
        return var1;
    }
}
