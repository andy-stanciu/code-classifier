class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        int n = strs.length;
        StringBuilder str = new StringBuilder("");
        char[] first = strs[0].toCharArray();
        char[] last = strs[n - 1].toCharArray();
        for (int i = 0; i < first.length; i++) {
            if (first[i] != last[i]) {
                return str.toString();
            } else {
                str.append(first[i]);
            }
        }
        return str.toString();
    }
}
