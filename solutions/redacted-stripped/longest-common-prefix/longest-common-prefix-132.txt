class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0)
            return "";
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            String str = strs[i];
            int j = 0;
            while (j < var1.length()) {
                if (str.length() <= j || str.charAt(j) != var1.charAt(j)) {
                    var1 = var1.substring(0, j);
                    break;
                }
                j++;
            }
        }
        return var1;
    }
}
