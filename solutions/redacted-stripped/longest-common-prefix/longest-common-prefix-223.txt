class Solution {

    public String method1(String[] strs) {
        if (Objects.isNull(strs) || strs.length == 0)
            return "";
        if (strs.length == 1)
            return strs[0];
        Arrays.sort(strs);
        int matchCount = 0;
        while (strs[0].length() > matchCount && strs[0].charAt(matchCount) == strs[strs.length - 1].charAt(matchCount)) matchCount++;
        return strs[strs.length - 1].substring(0, matchCount);
    }
}
