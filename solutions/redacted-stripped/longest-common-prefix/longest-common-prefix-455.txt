class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        int idx = 0;
        String fsas = strs[0];
        String lsas = strs[strs.length - 1];
        while (idx < fsas.length() && idx < lsas.length()) {
            if (fsas.charAt(idx) == lsas.charAt(idx)) {
                idx++;
            } else {
                break;
            }
        }
        return fsas.substring(0, idx);
    }
}
