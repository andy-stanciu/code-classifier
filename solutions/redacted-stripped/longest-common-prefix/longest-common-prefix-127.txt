class Solution {

    public String method1(String[] strs) {
        if (strs.length <= 0)
            return "";
        String var1 = strs[0];
        for (int i = 0; i < strs.length; i++) {
            int index = 0;
            while ((index < var1.length() && index < strs[i].length()) && var1.charAt(index) == strs[i].charAt(index)) {
                index++;
                ;
            }
            var1 = var1.substring(0, index);
        }
        if (var1.length() > 0)
            return var1;
        else
            return "";
    }
}
