class Solution {

    public String method1(String[] strs) {
        String pre = strs[0];
        if (pre.isEmpty()) {
            return "";
        }
        for (int i = 0; i < strs.length; i++) {
            while (strs[i].indexOf(pre) != 0) {
                pre = pre.substring(0, pre.length() - 1);
            }
        }
        return pre;
    }
}
