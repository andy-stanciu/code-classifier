class Solution {

    public String method1(String[] strs) {
        String s = "";
        int t = Integer.MAX_VALUE, k = 0;
        for (int i = 0; i < strs.length; i++) t = Math.min(t, strs[i].length());
        while (k < t) {
            char ch = strs[0].charAt(k);
            for (int i = 1; i < strs.length; i++) if (strs[i].charAt(k) != ch)
                return s;
            s += ch;
            k++;
        }
        return s;
    }
}
