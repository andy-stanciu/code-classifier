class Solution {

    public String method1(String[] strs) {
        int index = 0;
        StringBuilder sb = new StringBuilder();
        outerLoop: while (true) {
            if (index >= strs[0].length())
                break;
            char cur = strs[0].charAt(index);
            for (String s : strs) {
                if (index >= s.length() || s.charAt(index) != cur)
                    break outerLoop;
            }
            sb.append(cur);
            index++;
        }
        return sb.toString();
    }
}
