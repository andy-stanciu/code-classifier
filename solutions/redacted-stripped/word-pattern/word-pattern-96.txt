class Solution {

    public boolean method1(String param1, String s) {
        String[] sArray = s.split(" ");
        if (param1.length() != sArray.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String s1 = sArray[i];
            if (!map.containsKey(c)) {
                if (map.containsValue(s1)) {
                    return false;
                }
                map.put(c, s1);
            } else if (map.containsKey(c) && !map.get(c).equals(s1)) {
                return false;
            }
        }
        return true;
    }
}
