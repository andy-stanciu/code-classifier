class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        char[] p = param1.toCharArray();
        if (var1.length != p.length || p == null || var1 == null) {
            return false;
        }
        HashMap<Character, String> H1 = new HashMap<>();
        HashMap<String, Character> H2 = new HashMap<>();
        for (int i = 0; i < p.length; i++) {
            if (H1.containsKey(p[i])) {
                String value = H1.get(p[i]);
                if (value.equals(var1[i])) {
                    continue;
                } else {
                    return false;
                }
            } else {
                H1.put(p[i], var1[i]);
            }
            if (H2.containsKey(var1[i])) {
                char value1 = H2.get(var1[i]);
                if (value1 == p[i]) {
                    continue;
                } else {
                    return false;
                }
            } else {
                H2.put(var1[i], p[i]);
            }
        }
        return true;
    }
}
