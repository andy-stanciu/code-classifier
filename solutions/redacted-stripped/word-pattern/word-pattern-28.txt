class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (arr.length != param1.length())
            return false;
        Map<Character, String> hs = new HashMap<Character, String>();
        for (int i = 0; i < arr.length; i++) {
            char c = param1.charAt(i);
            String str = arr[i];
            if (hs.containsKey(c)) {
                if (!hs.get(c).equals(str))
                    return false;
            } else {
                if (hs.containsValue(str))
                    return false;
                hs.put(c, str);
            }
        }
        return true;
    }
}
