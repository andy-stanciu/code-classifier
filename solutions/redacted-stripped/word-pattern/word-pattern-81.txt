class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        if (var1.length != param1.length())
            return false;
        Map index = new HashMap();
        for (Integer i = 0; i < var1.length; ++i) if (index.put(param1.charAt(i), i) != index.put(var1[i], i))
            return false;
        return true;
    }
}
