class Solution {

    public boolean method1(String param1, String s) {
        int n = param1.length();
        String[] var1 = s.split(" ");
        if (n != var1.length) {
            return false;
        }
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            char c = param1.charAt(i);
            String var2 = var1[i];
            if (map.containsKey(c)) {
                if (!map.get(c).equals(var2)) {
                    return false;
                }
            } else {
                if (map.containsValue(var2)) {
                    return false;
                }
                map.put(c, var2);
            }
        }
        return true;
    }
}
