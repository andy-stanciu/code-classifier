class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        HashSet<String> set = new HashSet<>();
        for (int i = 0; i < arr.length; i++) {
            if (!set.contains(arr[i])) {
                map.put(param1.charAt(i), arr[i]);
                set.add(arr[i]);
            }
        }
        for (int i = 0; i < arr.length; i++) {
            if (map.get(param1.charAt(i)) != null && map.get(param1.charAt(i)).equals(arr[i])) {
            } else {
                return false;
            }
        }
        return true;
    }
}
