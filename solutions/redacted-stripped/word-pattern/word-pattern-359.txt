class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        HashMap<String, Character> map2 = new HashMap<>();
        boolean isTrue = true;
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(arr[i])) {
                    isTrue = false;
                }
            } else {
                map.put(ch, arr[i]);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            String str = arr[i];
            if (map2.containsKey(str)) {
                if (map2.get(str) != param1.charAt(i)) {
                    isTrue = false;
                    break;
                }
            } else {
                map2.put(arr[i], param1.charAt(i));
            }
        }
        return isTrue;
    }
}
