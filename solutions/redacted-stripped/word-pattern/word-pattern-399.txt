class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<Character, String> hm = new HashMap<Character, String>();
        HashMap<String, Character> reverseMap = new HashMap<String, Character>();
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            String var1 = arr[i];
            if (hm.containsKey(ch)) {
                if (!hm.get(ch).equals(var1)) {
                    return false;
                }
            } else {
                if (reverseMap.containsKey(var1)) {
                    return false;
                }
                hm.put(ch, var1);
                reverseMap.put(var1, ch);
            }
        }
        return true;
    }
}
