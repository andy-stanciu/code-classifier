class Solution {

    public boolean method1(String param1, String str) {
        String[] var1 = str.split(" ");
        if (var1.length != param1.length())
            return false;
        Map<Object, Integer> map = new HashMap();
        for (Integer i = 0; i < param1.length(); i++) {
            if (map.put(param1.charAt(i), i) != map.put(var1[i], i))
                return false;
        }
        return true;
    }
}
