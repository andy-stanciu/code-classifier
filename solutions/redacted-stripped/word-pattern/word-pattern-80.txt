class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map1 = new HashMap<>();
        HashMap<String, Character> map2 = new HashMap<>();
        String[] arr = s.split(" ");
        if (arr.length != param1.length())
            return false;
        int num = 0;
        for (int i = 0; i < param1.length(); i++) {
            if (!map1.containsKey(param1.charAt(i))) {
                map1.put(param1.charAt(i), arr[num]);
                if (!map2.containsKey(arr[num])) {
                    map2.put(arr[num], param1.charAt(i));
                } else {
                    if (param1.charAt(i) != map2.get(arr[num]))
                        return false;
                }
                num++;
            } else {
                if (!arr[num].equals(map1.get(param1.charAt(i))))
                    return false;
                num++;
            }
        }
        return true;
    }
}
