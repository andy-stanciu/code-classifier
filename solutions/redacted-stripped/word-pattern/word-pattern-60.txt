class Solution {

    public boolean method1(String param1, String s) {
        List<String> s1 = Arrays.asList(s.split(" "));
        if (param1.length() != s1.size()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (param1.indexOf(param1.charAt(i)) != s1.indexOf(s1.get(i))) {
                return false;
            }
        }
        return true;
    }
}
