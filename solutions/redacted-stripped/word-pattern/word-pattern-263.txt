class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        int n = param1.length();
        HashMap<Character, String> map = new HashMap<>();
        HashMap<String, Boolean> mappedString = new HashMap<>();
        for (int i = 0; i < n; i++) {
            char ch = param1.charAt(i);
            String val = arr[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(val)) {
                    return false;
                }
            } else {
                if (mappedString.getOrDefault(val, false)) {
                    return false;
                }
                mappedString.put(val, true);
                map.put(ch, val);
            }
        }
        return true;
    }
}
