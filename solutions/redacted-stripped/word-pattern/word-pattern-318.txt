class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        HashMap<Character, String> mp = new HashMap<Character, String>();
        int n = param1.length();
        if (n != var1.length)
            return false;
        for (int i = 0; i < n; i++) {
            if (mp.get(param1.charAt(i)) == null) {
                if (!mp.values().contains(var1[i])) {
                    mp.put(param1.charAt(i), var1[i]);
                } else
                    return false;
            } else {
                if (!mp.get(param1.charAt(i)).equals(var1[i]))
                    return false;
            }
        }
        return true;
    }
}
