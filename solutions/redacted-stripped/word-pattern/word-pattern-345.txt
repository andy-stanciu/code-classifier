class Solution {

    public boolean method1(String param1, String str) {
        String[] var1 = str.split(" ");
        if (var1.length != param1.length())
            return false;
        Map<Character, Integer> charToIndex = new HashMap<>();
        Map<String, Integer> stringToIndex = new HashMap<>();
        for (Integer i = 0; i < param1.length(); ++i) if (charToIndex.put(param1.charAt(i), i) != stringToIndex.put(var1[i], i))
            return false;
        return true;
    }
}
