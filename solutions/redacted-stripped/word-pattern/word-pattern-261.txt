class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> char_map = new HashMap<Character, String>();
        Map<String, Character> var1 = new HashMap<String, Character>();
        String[] var2 = s.split(" ");
        if (param1.length() != var2.length) {
            return false;
        }
        for (int i = 0; i < var2.length; i++) {
            char ch = param1.charAt(i);
            String str = var2[i];
            if (!char_map.containsKey(ch)) {
                if (var1.containsKey(str)) {
                    return false;
                } else {
                    char_map.put(ch, str);
                    var1.put(str, ch);
                }
            } else {
                String maped = char_map.get(ch);
                if (!maped.equals(str)) {
                    return false;
                }
            }
        }
        return true;
    }
}
