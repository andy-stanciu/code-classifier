class Solution {

    public boolean method1(String param1, String s) {
        String[] secondQue = s.split(" ");
        if (param1.length() != secondQue.length)
            return false;
        Map<Character, String> first = new HashMap<>();
        Map<String, Character> second = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (first.containsKey(param1.charAt(i))) {
                if (second.containsKey(secondQue[i])) {
                    if (!first.get(param1.charAt(i)).equals(secondQue[i]))
                        return false;
                } else {
                    return false;
                }
            } else {
                if (second.containsKey(secondQue[i]))
                    return false;
                first.put(param1.charAt(i), secondQue[i]);
                second.put(secondQue[i], param1.charAt(i));
            }
        }
        return true;
    }
}
