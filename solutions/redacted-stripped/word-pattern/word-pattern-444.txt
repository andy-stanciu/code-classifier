class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> var1 = new HashMap<>();
        Map<String, Character> var2 = new HashMap<>();
        String[] var3 = s.split(" ");
        if (var3.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < var3.length; i++) {
            char a = param1.charAt(i);
            String var4 = var3[i];
            if (!var1.containsKey(a)) {
                var1.put(a, var4);
            }
            if (!var2.containsKey(var4)) {
                var2.put(var4, a);
            }
            if (!var1.get(a).equals(var4) || !var2.get(var4).equals(a)) {
                return false;
            }
        }
        return true;
    }
}
