class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split("\\\\s+");
        char[] characters = param1.toCharArray();
        HashMap<Character, String> map = new HashMap<>();
        boolean check = true;
        if (var1.length != characters.length) {
            return false;
        }
        outer: for (int i = 0; i < characters.length; i++) {
            if (map.isEmpty()) {
                map.put(characters[i], var1[i]);
                continue;
            }
            if (!map.containsKey(characters[i])) {
                for (Map.Entry<Character, String> e : map.entrySet()) {
                    if (var1[i].equals(e.getValue())) {
                        check = false;
                        break outer;
                    }
                }
                map.put(characters[i], var1[i]);
            } else {
                if (map.get(characters[i]).equals(var1[i])) {
                    check = true;
                } else {
                    check = false;
                    break outer;
                }
            }
        }
        return check;
    }
}
