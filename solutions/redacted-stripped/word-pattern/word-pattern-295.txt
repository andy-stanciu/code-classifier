class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        char[] chars = param1.toCharArray();
        if (var1.length != chars.length)
            return false;
        Map<Character, String> relations = new HashMap<>();
        for (int i = 0; i < chars.length; i++) {
            if (relations.containsKey(chars[i])) {
                if (!var1[i].equals(relations.get(chars[i])))
                    return false;
            } else if (relations.containsValue(var1[i]))
                return false;
            relations.put(chars[i], var1[i]);
        }
        return true;
    }
}
