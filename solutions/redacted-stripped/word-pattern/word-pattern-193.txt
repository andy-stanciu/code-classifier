class Solution {

    public boolean method1(String p, String s) {
        HashMap<String, Character> mp1 = new HashMap<>();
        HashMap<Character, String> mp2 = new HashMap<>();
        String[] arr = s.split(" ");
        if (arr.length != p.length())
            return false;
        for (int i = 0; i < arr.length; i++) {
            if (mp1.containsKey(arr[i]) && mp2.containsKey(p.charAt(i)) && p.charAt(i) == mp1.get(arr[i]) && mp2.get(p.charAt(i)).equals(arr[i])) {
                continue;
            } else if (mp1.containsKey(arr[i]) || mp2.containsKey(p.charAt(i))) {
                return false;
            } else {
                mp1.put(arr[i], p.charAt(i));
                mp2.put(p.charAt(i), arr[i]);
            }
        }
        return true;
    }
}
