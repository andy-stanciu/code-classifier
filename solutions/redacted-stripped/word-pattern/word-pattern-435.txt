class Solution {

    public boolean method1(String param1, String s) {
        String[] splited = s.split(" ");
        Map<Character, String> map = new HashMap<Character, String>();
        if (param1.length() != splited.length)
            return false;
        for (int i = 0; i < param1.length(); i++) {
            if ((map.containsKey(param1.charAt(i)))) {
                if (!map.get(param1.charAt(i)).equals(splited[i]))
                    return false;
            } else {
                if (map.containsValue(splited[i]))
                    return false;
                map.put(param1.charAt(i), splited[i]);
            }
        }
        return true;
    }
}
