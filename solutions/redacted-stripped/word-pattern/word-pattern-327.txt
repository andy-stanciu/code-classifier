class Solution {

    public boolean method1(String param1, String s) {
        boolean flag = false;
        Map<String, Character> values = new HashMap();
        String[] sValue = s.split(" ");
        if (sValue.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < sValue.length; i++) {
            if (values.containsKey(sValue[i]) && values.containsValue(param1.charAt(i))) {
                if (values.get(sValue[i]) != param1.charAt(i)) {
                    return false;
                }
            } else if (!values.containsKey(sValue[i]) && values.containsValue(param1.charAt(i))) {
                return false;
            } else if (values.containsKey(sValue[i]) && !values.containsValue(param1.charAt(i))) {
                return false;
            } else {
                values.put(sValue[i], param1.charAt(i));
            }
        }
        return true;
    }
}
