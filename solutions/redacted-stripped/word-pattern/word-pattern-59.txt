class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] ss = s.split(" ");
        Set<String> set = new HashSet<>();
        if (ss.length != param1.length())
            return false;
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (!map.get(param1.charAt(i)).equals(ss[i]))
                    return false;
            } else {
                if (set.contains(ss[i]))
                    return false;
                set.add(ss[i]);
                map.put(param1.charAt(i), ss[i]);
            }
        }
        return true;
    }
}
