class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        if (var1.length != param1.length()) {
            return false;
        }
        Map<Character, String> pMap = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char p = param1.charAt(i);
            String w = var1[i];
            if (pMap.containsKey(p)) {
                if (!pMap.get(p).equals(w)) {
                    return false;
                }
            } else {
                if (pMap.containsValue(w)) {
                    return false;
                }
                pMap.put(p, w);
            }
        }
        return true;
    }
}
