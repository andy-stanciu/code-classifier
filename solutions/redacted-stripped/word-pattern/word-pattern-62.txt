class Solution {

    public boolean method1(String s, String t) {
        String[] arr = t.split(" ");
        if (s.length() != arr.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            String d = arr[i];
            if (map.containsKey(c)) {
                if (!map.get(c).equals(d)) {
                    return false;
                }
            } else {
                if (map.containsValue(d)) {
                    return false;
                }
                map.put(c, d);
            }
        }
        return true;
    }
}
