class Solution {

    public boolean method1(String param1, String s) {
        String[] splitstr = s.split(" ");
        HashMap<Character, String> map1 = new HashMap<>();
        if (param1.length() != splitstr.length) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            char ch1 = param1.charAt(i);
            String ch2 = splitstr[i];
            if (map1.containsKey(ch1) == true) {
                if (!map1.get(ch1).equals(ch2)) {
                    return false;
                }
            } else {
                if (map1.containsValue(ch2)) {
                    return false;
                }
                map1.put(ch1, ch2);
            }
        }
        return true;
    }
}
