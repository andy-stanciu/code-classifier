class Solution {

    public boolean method1(String param1, String s) {
        String[] split = s.split(" ");
        if (param1.length() != split.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char currentP = param1.charAt(i);
            if (map.containsKey(currentP)) {
                if (!map.get(currentP).equals(split[i])) {
                    return false;
                }
            } else {
                if (map.containsValue(split[i])) {
                    return false;
                } else {
                    map.put(currentP, split[i]);
                }
            }
        }
        return true;
    }
}
