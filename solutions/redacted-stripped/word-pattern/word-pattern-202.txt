class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> map = new HashMap<>();
        Set<String> set = new HashSet<>();
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length)
            return false;
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (!(map.get(param1.charAt(i)).equals(var1[i])))
                    return false;
            } else
                map.put(param1.charAt(i), var1[i]);
        }
        for (Map.Entry<Character, String> e : map.entrySet()) {
            if (set.contains(e.getValue()))
                return false;
            else
                set.add(e.getValue());
        }
        return true;
    }
}
