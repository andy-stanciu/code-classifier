class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (!map.get(param1.charAt(i)).equals(var1[i])) {
                    return false;
                }
            } else if (map.containsValue(var1[i])) {
                return false;
            }
            map.put(param1.charAt(i), var1[i]);
        }
        return true;
    }
}
