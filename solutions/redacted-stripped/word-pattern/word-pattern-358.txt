class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        if (var1.length != param1.length())
            return false;
        HashMap<Character, String> charMapping = new HashMap<>();
        HashMap<String, Character> var2 = new HashMap<>();
        for (int i = 0; i < var1.length; i++) {
            charMapping.put(param1.charAt(i), var1[i]);
            var2.put(var1[i], param1.charAt(i));
        }
        for (int i = 0; i < var1.length; i++) {
            char ch = var2.get(var1[i]);
            String str = charMapping.get(param1.charAt(i));
            if (ch != param1.charAt(i) || !str.equals(var1[i]))
                return false;
        }
        return true;
    }
}
