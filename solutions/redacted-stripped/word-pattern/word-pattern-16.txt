class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> hashMap = new HashMap<>();
        int i = 0;
        boolean can = true;
        String[] arr = s.split(" ");
        if (arr.length != param1.length())
            return false;
        for (char ch : param1.toCharArray()) {
            if (hashMap.containsKey(ch)) {
                if (!Objects.equals(hashMap.get(ch), arr[i])) {
                    return false;
                }
            } else {
                if (hashMap.containsValue(arr[i])) {
                    return false;
                }
                hashMap.put(ch, arr[i]);
            }
            i++;
        }
        return can;
    }
}
