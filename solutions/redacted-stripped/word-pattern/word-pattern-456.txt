class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> map = new HashMap<>();
        String[] arr = s.split("\\\\s");
        if (param1.length() != arr.length)
            return false;
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            if (map.containsKey(c) && !map.get(c).equalsIgnoreCase(arr[i])) {
                return false;
            }
            if (!map.containsKey(c) && map.containsValue(arr[i])) {
                return false;
            }
            map.put(c, arr[i]);
        }
        return true;
    }
}
