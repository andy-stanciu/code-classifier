class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<Character, String>();
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i)) == false) {
                if (map.containsValue(arr[i]) == false)
                    map.put(param1.charAt(i), arr[i]);
                else
                    return false;
            } else {
                if ((map.get(param1.charAt(i)).equals(arr[i])) == false)
                    return false;
            }
        }
        return true;
    }
}
