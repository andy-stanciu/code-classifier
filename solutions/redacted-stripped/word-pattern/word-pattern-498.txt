class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (arr.length != param1.length())
            return false;
        Hashtable<Character, String> hashtable = new Hashtable<>();
        for (int i = 0; i < param1.length(); i++) {
            char c1 = param1.charAt(i);
            String var1 = arr[i];
            if (hashtable.containsKey(c1)) {
                if (!hashtable.get(c1).equals(var1))
                    return false;
            } else {
                if (hashtable.containsValue(var1))
                    return false;
                hashtable.put(c1, var1);
            }
        }
        return true;
    }
}
