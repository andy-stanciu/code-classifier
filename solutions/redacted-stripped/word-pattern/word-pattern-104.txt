class Solution {

    public boolean method1(String param1, String s) {
        String[] sArray = s.split(" ");
        char[] pArray = param1.toCharArray();
        Map<Character, String> map = new HashMap<>();
        if (sArray.length != pArray.length)
            return false;
        for (int i = 0; i < pArray.length; i++) {
            char ch = pArray[i];
            String string = sArray[i];
            if (!map.containsKey(ch)) {
                if (!map.containsValue(string))
                    map.put(ch, string);
                else
                    return false;
            } else {
                if (!map.get(ch).equals(string))
                    return false;
            }
        }
        return true;
    }
}
