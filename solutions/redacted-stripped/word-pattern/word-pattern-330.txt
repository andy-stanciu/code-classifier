class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        if (str.length != param1.length()) {
            return false;
        }
        Map<Character, String> mp = new HashMap<>();
        Map<String, Boolean> check = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (!mp.containsKey(param1.charAt(i))) {
                if (check.getOrDefault(str[i], false) == false) {
                    mp.put(param1.charAt(i), str[i]);
                    check.put(str[i], true);
                } else {
                    return false;
                }
            } else {
                if (!mp.get(param1.charAt(i)).equals(str[i])) {
                    return false;
                }
            }
        }
        return true;
    }
}
