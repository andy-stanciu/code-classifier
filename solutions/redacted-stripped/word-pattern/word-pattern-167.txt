class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> var1 = new HashMap<>();
        Map<String, Character> sMap = new HashMap<>();
        String[] var2 = s.split("\\\\s");
        if (var2.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String var3 = var2[i];
            if (var1.containsKey(c) && !var1.get(c).equals(var3))
                return false;
            if (sMap.containsKey(var3) && !sMap.get(var3).equals(c))
                return false;
            var1.put(c, var2[i]);
            sMap.put(var3, c);
        }
        return true;
    }
}
