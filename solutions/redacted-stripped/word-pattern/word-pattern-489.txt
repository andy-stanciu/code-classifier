class Solution {

    public boolean method1(String param1, String s) {
        String[] ans = s.split(" ");
        int n = ans.length;
        HashMap<Character, String> map = new HashMap<>();
        if (n != param1.length()) {
            return false;
        }
        for (int i = 0; i < n; i++) {
            char ch = param1.charAt(i);
            String var1 = ans[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(var1)) {
                    return false;
                }
            } else {
                if (map.containsValue(var1)) {
                    return false;
                }
                map.put(ch, var1);
            }
        }
        return true;
    }
}
