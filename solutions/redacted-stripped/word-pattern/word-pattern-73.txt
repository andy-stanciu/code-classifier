class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> charToStr = new HashMap<Character, String>();
        HashMap<String, Character> strToChar = new HashMap<String, Character>();
        int strLen = param1.length();
        String[] sArr = s.split(" ");
        int sArrLen = sArr.length;
        if (strLen != sArrLen) {
            return false;
        }
        for (int i = 0; i < sArrLen; i++) {
            if (charToStr.containsKey(param1.charAt(i))) {
                if (!charToStr.get(param1.charAt(i)).equals(sArr[i])) {
                    return false;
                }
                if (strToChar.get(sArr[i]) != param1.charAt(i)) {
                    return false;
                }
            } else {
                charToStr.put(param1.charAt(i), sArr[i]);
                if (strToChar.containsKey(sArr[i]))
                    return false;
                else
                    strToChar.put(sArr[i], param1.charAt(i));
            }
        }
        return true;
    }
}
