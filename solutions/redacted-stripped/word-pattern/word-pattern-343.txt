class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        char[] pat = param1.toCharArray();
        if (pat.length != str.length) {
            return false;
        }
        HashMap<Character, String> map1 = new HashMap<>();
        HashMap<String, Character> map2 = new HashMap<>();
        for (int k = 0; k < str.length; k++) {
            char var1 = pat[k];
            String var2 = str[k];
            if (!map1.containsKey(var1) && !map2.containsKey(var2)) {
                map1.put(var1, var2);
                map2.put(var2, var1);
            } else if (!var2.equals(map1.get(var1)) || pat[k] != map2.get(var2)) {
                return false;
            }
        }
        return true;
    }
}
