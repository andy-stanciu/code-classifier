class Solution {

    public boolean method1(String param1, String s) {
        String[] parts = s.split(" ");
        if (param1.length() != parts.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            boolean containsKey = map.containsKey(ch);
            if (map.containsValue(parts[i]) && !containsKey) {
                return false;
            }
            if (containsKey && !map.get(ch).equals(parts[i])) {
                return false;
            } else
                map.put(ch, parts[i]);
        }
        return true;
    }
}
