class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        return stringToDiff(List.of(var1)).equals(stringToDiff(param1.chars().mapToObj(c -> (char) c).collect(Collectors.toList())));
    }

    public <T> String stringToDiff(List<T> elements) {
        HashMap<T, Integer> var2 = new HashMap<>();
        StringBuilder sb = new StringBuilder();
        int count = 0;
        for (T var3 : elements) {
            if (!var2.containsKey(var3)) {
                var2.put(var3, count);
                count++;
            }
            sb.append(var2.get(var3));
        }
        return sb.toString();
    }
}
