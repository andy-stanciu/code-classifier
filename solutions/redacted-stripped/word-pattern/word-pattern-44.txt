class Solution {

    public boolean method1(String param1, String s) {
        List<String> var1 = new ArrayList<>(Arrays.asList(s.split(" ")));
        if (var1.size() != param1.length())
            return false;
        for (int i = 0; i < var1.size(); i++) {
            if (var1.indexOf(var1.get(i)) != param1.indexOf(param1.charAt(i))) {
                return false;
            }
        }
        return true;
    }
}
