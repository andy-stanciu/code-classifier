class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<Character, String>();
        for (int i = 0; i < arr.length; i++) {
            if (!map.containsKey(param1.charAt(i))) {
                if (map.containsValue(arr[i])) {
                    return false;
                }
                map.put(param1.charAt(i), arr[i]);
            } else {
                String replace = map.get(param1.charAt(i));
                if (!replace.equals(arr[i]))
                    return false;
            }
        }
        return true;
    }
}
