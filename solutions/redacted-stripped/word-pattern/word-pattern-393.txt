class Solution {

    public boolean method1(String param1, String s) {
        int len1 = param1.length();
        String[] testParts = s.split(" ");
        if (len1 != testParts.length)
            return false;
        String[] var1 = new String[26];
        Set<String> valueSet = new HashSet<>();
        int i = 0, j = 0;
        while (i < len1 && j < testParts.length) {
            int indexOfChar = (int) (param1.charAt(i)) - 97;
            if (var1[indexOfChar] == null && !valueSet.contains(testParts[j])) {
                var1[indexOfChar] = testParts[j];
                valueSet.add(testParts[j]);
            } else if (var1[indexOfChar] == null && valueSet.contains(testParts[j]))
                return false;
            else if (var1[indexOfChar] != null && !testParts[j].equals(var1[indexOfChar]))
                return false;
            i++;
            j++;
        }
        return true;
    }
}
