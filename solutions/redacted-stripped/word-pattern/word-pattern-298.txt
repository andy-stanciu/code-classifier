class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> map = new HashMap<>();
        String[] var1 = s.split(" ");
        if (var1.length != param1.length())
            return false;
        int i = 0;
        for (String var2 : var1) {
            if (!map.containsKey(param1.charAt(i)) && !map.containsValue(var2))
                map.put(param1.charAt(i), var2);
            if (!Objects.equals(map.get(param1.charAt(i)), var2))
                return false;
            i++;
        }
        return true;
    }
}
