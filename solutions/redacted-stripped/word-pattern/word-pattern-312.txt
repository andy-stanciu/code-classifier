class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> var1 = new HashMap<>();
        String[] s2 = s.split(" ");
        char[] charArray = param1.toCharArray();
        if (charArray.length != s2.length) {
            return false;
        }
        for (int i = 0; i < charArray.length; i++) {
            char character = charArray[i];
            String var2 = s2[i];
            if (!var1.containsKey(character)) {
                if (var1.containsValue(var2))
                    return false;
                var1.put(character, var2);
            } else {
                if (!var1.get(character).equals(var2)) {
                    return false;
                }
            }
        }
        return true;
    }
}
