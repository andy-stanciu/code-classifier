class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        int len = param1.length();
        if (len != arr.length) {
            return false;
        }
        Map<Character, String> hm = new HashMap<>();
        boolean res = true;
        for (int i = 0; i < len; i++) {
            char ch = param1.charAt(i);
            if (hm.containsKey(ch)) {
                if (!hm.get(ch).equals(arr[i])) {
                    res = false;
                    break;
                }
            } else {
                if (hm.containsValue(arr[i])) {
                    res = false;
                    break;
                }
                hm.put(ch, arr[i]);
            }
        }
        return res;
    }
}
