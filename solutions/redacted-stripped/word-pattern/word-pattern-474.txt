class Solution {

    public boolean method1(String param1, String s) {
        var sArray = s.split(" ");
        var var1 = param1.split("");
        if (sArray.length != var1.length) {
            return false;
        }
        HashMap<String, Integer> sToNum = new HashMap<String, Integer>();
        HashMap<String, Integer> var2 = new HashMap<String, Integer>();
        for (String str : sArray) {
            if (!sToNum.containsKey(str)) {
                sToNum.put(str, sToNum.size());
            }
        }
        for (String str : var1) {
            if (!var2.containsKey(str)) {
                var2.put(str, var2.size());
            }
        }
        int[] var3 = new int[var1.length];
        for (int i = 0; i < var1.length; i++) {
            var3[i] = var2.get(var1[i]);
        }
        int[] sSequence = new int[sArray.length];
        for (int i = 0; i < sArray.length; i++) {
            sSequence[i] = sToNum.get(sArray[i]);
        }
        return Arrays.equals(sSequence, var3);
    }
}
