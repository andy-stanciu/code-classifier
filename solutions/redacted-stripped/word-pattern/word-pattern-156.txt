class Solution {

    public boolean method1(String param1, String s) {
        String[] aa = s.split(" ");
        HashMap<Character, String> map = new HashMap();
        HashSet<String> set = new HashSet();
        if (param1.length() != aa.length) {
            return false;
        }
        for (int i = 0; i < aa.length; i++) {
            String var1 = aa[i];
            char val2 = param1.charAt(i);
            if (map.containsKey(val2) && !map.get(val2).equals(var1)) {
                return false;
            } else if (!map.containsKey(val2) && set.contains(var1)) {
                return false;
            }
            map.put(val2, var1);
            set.add(var1);
        }
        return true;
    }
}
