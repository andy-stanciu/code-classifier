class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<String, String> map = new HashMap<String, String>();
        for (int arrIndex = 0, var2 = 0; arrIndex < arr.length; arrIndex++, var2++) {
            String var1 = String.valueOf(param1.charAt(var2));
            if (map.containsKey(var1)) {
                if (!map.get(var1).equals(arr[arrIndex])) {
                    return false;
                }
            } else {
                if (map.containsValue(arr[arrIndex])) {
                    return false;
                }
                map.put(var1, arr[arrIndex]);
            }
        }
        return true;
    }
}
