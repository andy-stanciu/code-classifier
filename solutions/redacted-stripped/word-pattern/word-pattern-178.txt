class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> dict = new HashMap<>();
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (dict.containsKey(param1.charAt(i))) {
                if (!dict.get(param1.charAt(i)).equals(var1[i])) {
                    return false;
                }
            } else if (dict.containsValue(var1[i])) {
                return false;
            } else {
                dict.put(param1.charAt(i), var1[i]);
            }
        }
        return true;
    }
}
