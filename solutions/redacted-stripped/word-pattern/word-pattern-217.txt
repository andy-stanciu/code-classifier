class Solution {

    public boolean method1(String param1, String s) {
        char[] cArray = param1.toCharArray();
        HashMap<String, String> map = new HashMap<>();
        HashMap<String, String> reverseMap = new HashMap<>();
        String[] var1 = s.split(" ");
        if (cArray.length != var1.length) {
            return false;
        }
        for (int i = 0; i < cArray.length; i++) {
            String p = cArray[i] + "";
            String var2 = var1[i];
            if (map.containsKey(p)) {
                String sP = map.get(p);
                if (!sP.equals(var2)) {
                    return false;
                }
            } else {
                if (reverseMap.containsKey(var2)) {
                    return false;
                }
                map.put(p, var2);
                reverseMap.put(var2, p);
            }
        }
        return true;
    }
}
