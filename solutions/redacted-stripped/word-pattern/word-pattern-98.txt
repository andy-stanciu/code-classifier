class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        if (param1.length() != str.length)
            return false;
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < str.length; i++) {
            Character ch = param1.charAt(i);
            String var1 = str[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(var1))
                    return false;
            } else {
                if (map.containsValue(var1))
                    return false;
                map.put(ch, var1);
            }
        }
        return true;
    }
}
