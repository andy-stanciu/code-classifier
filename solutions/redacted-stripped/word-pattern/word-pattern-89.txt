class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap();
        String[] a = s.split(" ");
        if (param1.length() != a.length) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (!a[i].equals(map.get(param1.charAt(i)))) {
                    return false;
                }
            } else if (map.containsValue(a[i])) {
                return false;
            } else {
                map.put(param1.charAt(i), a[i]);
            }
        }
        return true;
    }
}
