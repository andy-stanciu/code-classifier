class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> temp = new HashMap<>();
        String[] arr = s.split(" ");
        int i = 0;
        int n = param1.length();
        if (n != arr.length) {
            return false;
        }
        for (char c : param1.toCharArray()) {
            if ((temp.containsKey(c) && !temp.get(c).equals(arr[i])) || (!temp.containsKey(c) && temp.containsValue(arr[i]))) {
                return false;
            } else {
                temp.put(c, arr[i]);
                i++;
            }
        }
        return true;
    }
}
