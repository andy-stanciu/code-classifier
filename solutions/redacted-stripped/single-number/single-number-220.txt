class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        } else {
            for (int i = 0; i < nums.length - 1; i++) {
                for (int j = i + 1; j < nums.length; j++) {
                    if (nums[i] == nums[j]) {
                        nums[i] = Integer.MAX_VALUE;
                        nums[j] = Integer.MAX_VALUE;
                    }
                }
            }
            Arrays.sort(nums);
            return nums[0];
        }
    }
}
