class Solution {

    public int method1(int[] nums) {
        Set<Integer> setobj = new HashSet<Integer>();
        if (nums.length == 1)
            return nums[0];
        int ans = Integer.MIN_VALUE;
        for (int element : nums) {
            if (setobj.contains(element) == true) {
                setobj.remove(element);
            } else
                setobj.add(element);
        }
        for (int element : setobj) ans = element;
        return ans;
    }
}
