class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (i == nums.length - 1)
                return nums[i];
            if (nums[i] != nums[i + 1]) {
                if (i == 0)
                    return nums[i];
                if (nums[i - 1] != nums[i])
                    return nums[i];
            }
        }
        return 0;
    }
}
