class Solution {

    public int method1(int[] nums) {
        Set<Map.Entry<Integer, Long>> entrySett = Arrays.stream(nums).boxed().collect(Collectors.groupingBy(Function.identity(), Collectors.counting())).entrySet();
        for (Entry<Integer, Long> entry : entrySett) {
            if (entry.getValue() == 1)
                return entry.getKey();
        }
        return -1;
    }
}
