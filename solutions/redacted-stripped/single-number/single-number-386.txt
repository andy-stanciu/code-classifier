class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        int v = 0;
        for (int n : nums) {
            map.put(n, map.getOrDefault(n, 0) + 1);
        }
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            if (entry.getValue() < 2) {
                v = entry.getKey();
            }
        }
        return v;
    }
}
