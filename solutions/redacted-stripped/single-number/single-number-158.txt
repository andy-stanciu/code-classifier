class Solution {

    public int method1(int[] nums) {
        int size = nums.length;
        List<Integer> list = new ArrayList<Integer>();
        for (int i = 0; i < size; i++) {
            if (list.contains(nums[i])) {
                list.remove(new Integer(nums[i]));
            } else
                list.add(nums[i]);
        }
        return list.get(0);
    }
}
