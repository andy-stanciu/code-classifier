class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        }
        Arrays.sort(nums);
        int i = 0;
        for (i = 0; i < nums.length - 1; i += 2) {
            System.out.println(nums[i] + "  " + nums[i + 1]);
            if (nums[i] != nums[i + 1]) {
                return nums[i];
            }
        }
        if (i > 1 && nums[i - 1] != nums[i]) {
            return nums[i];
        }
        return -1;
    }
}
