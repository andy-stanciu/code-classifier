class Solution {

    public static int method1(int[] nums) {
        int result = 0;
        for (int num : nums) {
            result ^= num;
        }
        return result;
    }

    public static void main(String[] args) {
        int[] nums = { 2, 2, 1 };
        System.out.println("The Single Num is: " + method1(nums));
    }
}
