class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> mp = new HashMap<>();
        for (int num : nums) {
            mp.put(num, mp.getOrDefault(num, 0) + 1);
        }
        for (int k : mp.keySet()) {
            if (mp.get(k) == 1) {
                return k;
            }
        }
        return 0;
    }
}
