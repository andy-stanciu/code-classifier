class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        boolean n = true;
        int x = 31000;
        for (int i = 0; i < nums.length; i++) {
            if (x == 31000) {
                x = nums[i];
                n = false;
            } else if (x == nums[i] && n == false) {
                n = true;
                x = 31000;
            } else {
                return x;
            }
        }
        return x;
    }
}
