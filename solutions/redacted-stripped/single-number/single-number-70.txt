class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int count = 0;
        for (int i = 0; i < nums.length; ) {
            int j = i;
            while (j < nums.length && nums[i] == nums[j]) {
                j++;
                count++;
            }
            if (count == 1)
                return nums[i];
            i = j;
            count = 0;
        }
        return -1;
    }
}
