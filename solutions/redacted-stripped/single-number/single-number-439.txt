class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int i = 0;
        for (; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (nums[i] == nums[j])
                    count++;
            }
            if (count != 2)
                break;
            else {
                count = 0;
            }
        }
        return nums[i];
    }
}
