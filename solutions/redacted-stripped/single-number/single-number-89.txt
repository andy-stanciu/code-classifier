class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            if (set.contains(nums[i]))
                set.remove(nums[i]);
            else
                set.add(nums[i]);
        }
        int out = 0;
        Iterator<Integer> itr = set.iterator();
        while (itr.hasNext()) out = itr.next();
        return out;
    }
}
