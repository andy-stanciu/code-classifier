class Solution {

    public int method1(int[] nums) {
        int p = 0;
        int ans = 0;
        for (int i = 0; i < nums.length; i++) {
            int a = nums[i];
            p = 0;
            for (int k = 0; k < nums.length; k++) {
                if (nums[k] == a) {
                    p = p + 1;
                }
            }
            if (p == 1) {
                ans = a;
                break;
            }
        }
        return ans;
    }
}
