class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int i = 0;
        while (i < nums.length - 2) {
            if (nums[i] == nums[i + 1]) {
                i += 2;
            } else
                return nums[i];
        }
        return nums[nums.length - 1];
    }
}
