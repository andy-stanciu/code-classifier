class Solution {

    public int mySqrt(int x) {
        long start = 0, end = x, mid = 0;
        if (x < 2) {
            return x;
        }
        while (start <= end) {
            mid = (start + end) / 2;
            if ((mid * mid) == x) {
                return (int) mid;
            } else if ((mid * mid) < x) {
                start = mid + 1;
            } else if ((mid * mid) > x) {
                end = mid - 1;
            }
        }
        return (int) start - 1;
    }
}
