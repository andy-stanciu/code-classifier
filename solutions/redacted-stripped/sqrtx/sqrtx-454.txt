class Solution {

    public int mySqrt(int x) {
        int lo = 1;
        int hi = x / 2;
        if (x == 1)
            return 1;
        int ans = x;
        while (lo <= hi) {
            long mid = (lo + hi) / 2;
            if (mid * mid == x) {
                return (int) mid;
            } else if (mid > x / mid)
                hi = (int) mid - 1;
            else
                lo = (int) mid + 1;
        }
        return hi;
    }
}
