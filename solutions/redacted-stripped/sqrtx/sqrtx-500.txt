class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        long low = 1, high = x;
        long mid;
        long ans = 0;
        while (low <= high) {
            mid = (low + high) / 2;
            if ((long) mid * mid <= x) {
                ans = mid;
                low = mid + 1;
            } else
                high = mid - 1;
        }
        return (int) ans;
    }
}
