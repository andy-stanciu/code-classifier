class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int start = 0, end = x, result = 0;
        while (start <= end) {
            int mid = (start + end) / 2;
            if (mid <= (x / mid)) {
                start = mid + 1;
                result = mid;
            } else {
                end = mid - 1;
            }
        }
        return result;
    }
}
