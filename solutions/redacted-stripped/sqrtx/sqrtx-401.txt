class Solution {

    public int mySqrt(int x) {
        long left = 0;
        long right = x;
        long mid = 0;
        long root = 1;
        while (left <= right) {
            mid = (left + right) / 2;
            if (mid * mid <= x) {
                root = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return (int) root;
    }
}
