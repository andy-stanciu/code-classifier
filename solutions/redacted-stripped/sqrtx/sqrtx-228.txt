class Solution {

    public int mySqrt(int x) {
        int start = 0, end = x;
        while (start <= end) {
            int current = (start + end) / 2;
            long sqrt = (long) current * current;
            if (sqrt == x) {
                return current;
            } else if (sqrt < x) {
                start = current + 1;
            } else {
                end = current - 1;
            }
        }
        return start - 1;
    }
}
