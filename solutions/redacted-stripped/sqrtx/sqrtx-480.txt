class Solution {

    public int mySqrt(int x) {
        int start = 0;
        if (x == 0) {
            return 0;
        }
        if (x == 1 || x == 2) {
            return 1;
        }
        while (start != x) {
            if (start * start == x) {
                return start;
            } else if (start * start > x) {
                return start - 1;
            }
            start++;
        }
        return 46340;
    }
}
