class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int half = x / 2;
        int res = -1;
        for (int i = 1; i <= half; i++) {
            if (i <= x / i) {
                res = i;
            } else {
                break;
            }
        }
        return res;
    }
}
