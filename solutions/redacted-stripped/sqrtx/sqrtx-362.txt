class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        int s = 1, e = x, t = 0;
        while (s <= e) {
            int m = s + (e - s) / 2;
            long square = (long) m * m;
            if (square == x)
                return m;
            else if (square > x)
                e = m - 1;
            else {
                s = m + 1;
                t = m;
            }
        }
        return t;
    }
}
