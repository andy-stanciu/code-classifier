class Solution {

    public int mySqrt(int x) {
        int start = 0;
        int end = x;
        int result = 0;
        if (x == 0 || x == 1) {
            return x;
        }
        while (start <= end) {
            int mid = start + (end - start) / 2;
            long midsqare = (long) mid * mid;
            if (midsqare == x) {
                return mid;
            } else if (midsqare > x) {
                end = mid - 1;
            } else if (midsqare < x) {
                result = mid;
                start = mid + 1;
            }
        }
        return result;
    }
}
