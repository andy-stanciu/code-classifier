class Solution {

    public int mySqrt(int x) {
        int low = 0;
        int high = 46340;
        int ans = 0;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (mid * mid == x)
                return mid;
            else if ((mid * mid) < x) {
                ans = mid;
                low = mid + 1;
            } else
                high = mid - 1;
        }
        return ans;
    }
}
