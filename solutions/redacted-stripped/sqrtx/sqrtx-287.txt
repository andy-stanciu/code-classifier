class Solution {

    public int mySqrt(int x) {
        int mid = 0;
        if (x == 0)
            return 0;
        if (x > 0 && x <= 3)
            return 1;
        else {
            int l = 2;
            int m = 46340;
            mid = (l + m) / 2;
            while (l < m) {
                if ((mid * mid) == x)
                    return mid;
                else if ((mid * mid) < x && ((mid + 1) * (mid + 1) > x))
                    return mid;
                else if ((mid * mid) < x)
                    l = mid + 1;
                else if ((mid * mid) > x)
                    m = mid - 1;
                mid = (l + m) / 2;
            }
        }
        return mid;
    }
}
