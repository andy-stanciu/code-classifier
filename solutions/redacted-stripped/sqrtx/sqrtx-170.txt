class Solution {

    public int mySqrt(int x) {
        int low = 1;
        int mid = -1;
        int high = x;
        if (x == 0 || x == 1) {
            return x;
        }
        while (low <= high) {
            mid = low + (high - low) / 2;
            if ((mid * mid) == x) {
                return mid;
            } else if ((long) mid * mid < (long) x) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return high;
    }
}
