class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1)
            return x;
        int left = 1;
        int right = x;
        int ans = 46340;
        while (left <= right) {
            int mid = (left + right) / 2;
            if ((long) mid * mid <= (long) x) {
                ans = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return ans;
    }
}
