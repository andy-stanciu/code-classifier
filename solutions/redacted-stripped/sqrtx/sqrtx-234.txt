class Solution {

    public int mySqrt(int x) {
        if (x < 2)
            return x;
        long right = x / 2, left = 0, result = 1;
        while (left <= right) {
            long mid = left + (right - left) / 2;
            long val = mid * mid;
            if (val == x) {
                return (int) mid;
            }
            if (val < x) {
                result = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return (int) result;
    }
}
