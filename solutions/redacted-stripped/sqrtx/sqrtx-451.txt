class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int left = 1, right = x / 2;
        while (left <= right) {
            int mid = (left - right) / 2 + right;
            if (mid * mid == x) {
                return mid;
            } else if (mid < x / mid + 1) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return right;
    }
}
