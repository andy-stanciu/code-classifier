class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        if (x == 1)
            return 1;
        int start = 0;
        int end = x;
        int res = -1;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (mid == (x / mid)) {
                return mid;
            } else if (mid > (x / mid)) {
                res = mid;
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return res - 1;
    }
}
