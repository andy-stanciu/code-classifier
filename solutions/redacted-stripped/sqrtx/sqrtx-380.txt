class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int s = 1, e = x, m;
        while (s <= e) {
            m = s + (e - s) / 2;
            if (Math.pow(m, 2) == x) {
                return m;
            } else if (Math.pow(m, 2) < x) {
                s = m + 1;
            } else {
                e = m - 1;
            }
        }
        return e;
    }
}
