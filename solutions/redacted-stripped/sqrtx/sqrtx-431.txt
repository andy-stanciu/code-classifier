class Solution {

    public int mySqrt(int n) {
        double x = n / 2;
        if (n == 1) {
            return n;
        } else {
            for (int i = 0; i <= 100; i++) {
                x = (x + (n / x)) * 0.5;
                System.out.println(x);
            }
        }
        return (int) x;
    }
}
