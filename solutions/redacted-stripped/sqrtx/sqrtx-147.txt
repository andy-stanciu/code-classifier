class Solution {

    public int mySqrt(int x) {
        long low = 1;
        long high = x;
        while (true) {
            long mid = (low + high) / 2;
            long mid2 = mid * mid;
            if (mid2 <= x && (mid + 1) * (mid + 1) > x) {
                return (int) mid;
            } else if (mid2 > x) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
    }
}
