class Solution {

    public int mySqrt(long x) {
        for (long i = 0; i < x; i++) {
            if (i * i == x || i * i < x && (i + 1) * (i + 1) > x) {
                return (int) i;
            }
        }
        if (x == 0) {
            return 0;
        }
        if (x == 1) {
            return 1;
        }
        return -1;
    }
}
