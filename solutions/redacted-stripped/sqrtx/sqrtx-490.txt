class Solution {

    public int mySqrt(int x) {
        long l = 1;
        long h = x;
        long mid = l + (h - l) / 2;
        while (l <= h) {
            long prod = mid * mid;
            if (prod == x)
                return (int) mid;
            if (prod > x) {
                h = mid - 1;
            } else {
                l = mid + 1;
            }
            mid = l + (h - l) / 2;
        }
        return (int) mid - 1;
    }
}
