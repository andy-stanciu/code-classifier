class Solution {

    public int mySqrt(int x) {
        int s = 0, e = x, m = 0;
        while (s <= e) {
            m = s + (e - s) / 2;
            long sq = (long) m * m;
            if (sq == x) {
                return m;
            } else if (sq > x) {
                e = m - 1;
            } else {
                s = m + 1;
            }
        }
        return e;
    }
}
