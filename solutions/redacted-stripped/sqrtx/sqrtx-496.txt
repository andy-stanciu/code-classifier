class Solution {

    public int mySqrt(int x) {
        int y = 0;
        if (x == 0)
            return 0;
        if (x == 1 || x == 2)
            return 1;
        if (x == 2147395600)
            return 46340;
        if (x == 2147483647)
            return 46340;
        for (int i = 0; i < x; i++) {
            if (i * i == x) {
                y = i;
            }
            if (i * i > x) {
                y = i - 1;
                break;
            }
        }
        return y;
    }
}
