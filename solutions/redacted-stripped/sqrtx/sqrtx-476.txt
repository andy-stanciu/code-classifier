class Solution {

    public int mySqrt(int x) {
        long right = x;
        long left = 1;
        if (x == 1) {
            return 1;
        }
        while (left < right) {
            long mid = left + ((right - left) / 2);
            if ((mid * mid) <= x) {
                left = mid + 1;
            } else if ((mid * mid) > x) {
                right = mid;
            } else {
                return (int) mid;
            }
        }
        return (int) left - 1;
    }
}
