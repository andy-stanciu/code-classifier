class Solution {

    public int mySqrt(int x) {
        long st = 0, end = x;
        long ans = -1;
        while (st <= end) {
            long mid = st + (end - st) / 2;
            long val = mid * mid;
            if (val == x) {
                return (int) mid;
            } else if (val < x) {
                ans = mid;
                st = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return (int) ans;
    }
}
