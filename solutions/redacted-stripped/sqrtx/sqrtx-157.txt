class Solution {

    public int mySqrt(int x) {
        long l = 1, r = x;
        int ans = 1;
        if (x == 0) {
            return 0;
        }
        while (l <= r) {
            long mid = l + (r - l) / 2;
            if (mid * mid <= x) {
                ans = (int) mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }
}
