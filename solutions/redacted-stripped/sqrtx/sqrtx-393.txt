class Solution {

    public int mySqrt(int x) {
        if (x == 1 || x == 0) {
            return x;
        }
        int l = 1, h = x / 2, m = 0;
        int res = 0;
        while (l <= h) {
            m = (l + h) / 2;
            if (m * m == x) {
                return m;
            } else if ((long) m * m < (long) x) {
                l = m + 1;
                res = m;
            } else {
                h = m - 1;
            }
        }
        return res;
    }
}
