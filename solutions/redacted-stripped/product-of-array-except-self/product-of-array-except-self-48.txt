class Solution {

    public int[] method1(int[] nums) {
        int zeroCount = 0;
        int var1 = 1;
        for (int n : nums) {
            if (n == 0) {
                zeroCount++;
            }
        }
        if (zeroCount >= 2) {
            for (int i = 0; i < nums.length; i++) {
                nums[i] = 0;
            }
        } else if (zeroCount == 1) {
            int zeroInd = 0;
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] != 0) {
                    var1 = var1 * nums[i];
                    nums[i] = 0;
                } else {
                    zeroInd = i;
                }
            }
            nums[zeroInd] = var1;
        } else {
            for (int i = 0; i < nums.length; i++) {
                var1 = var1 * nums[i];
            }
            for (int i = 0; i < nums.length; i++) {
                nums[i] = var1 / nums[i];
            }
        }
        return nums;
    }
}
