class Solution {

    public int[] method1(int[] nums) {
        var result = new int[nums.length];
        var var1 = 1;
        for (var i = 0; i < nums.length; i++) {
            result[i] = var1;
            var1 = var1 * nums[i];
        }
        var1 = 1;
        for (var j = nums.length - 1; j >= 0; j--) {
            result[j] = var1 * result[j];
            var1 = var1 * nums[j];
        }
        return result;
    }
}
