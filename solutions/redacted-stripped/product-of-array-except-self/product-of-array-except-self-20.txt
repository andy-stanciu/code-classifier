class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        int var1 = 1;
        int zeroCount = 0;
        for (int num : nums) {
            if (num != 0) {
                var1 *= num;
            } else {
                zeroCount++;
            }
        }
        for (int i = 0; i < n; i++) {
            if (zeroCount > 1) {
                result[i] = 0;
            } else if (zeroCount == 1) {
                result[i] = nums[i] == 0 ? var1 : 0;
            } else {
                result[i] = var1 / nums[i];
            }
        }
        return result;
    }
}
