class Solution {

    public int[] method1(int[] nums) {
        int N = nums.length;
        int var1 = 1;
        int[] ans = new int[N];
        for (int i = 0; i < N; i++) {
            ans[i] = var1;
            var1 *= nums[i];
        }
        var1 = 1;
        for (int i = N - 1; i >= 0; i--) {
            ans[i] *= var1;
            var1 *= nums[i];
        }
        return ans;
    }
}
