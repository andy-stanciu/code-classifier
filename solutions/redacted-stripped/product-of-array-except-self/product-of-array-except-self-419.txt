class Solution {

    public int[] method1(int[] nums) {
        int length = nums.length;
        if (length == 1)
            return nums;
        int[] var1 = new int[length];
        int prefix = 1;
        for (int i = 0; i < length; i++) {
            var1[i] = prefix;
            prefix = nums[i] * prefix;
        }
        int suffix = 1;
        for (int j = length - 1; j >= 0; j--) {
            var1[j] = var1[j] * suffix;
            suffix = nums[j] * suffix;
        }
        return var1;
    }
}
