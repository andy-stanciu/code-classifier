class Solution {

    public int[] method1(int[] nums) {
        int[] from_first = new int[nums.length];
        from_first[0] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            from_first[i] = from_first[i - 1] * nums[i];
        }
        int[] from_last = new int[nums.length];
        from_last[nums.length - 1] = nums[nums.length - 1];
        for (int j = nums.length - 2; j >= 0; j--) {
            from_last[j] = nums[j] * from_last[j + 1];
        }
        for (int x = 0; x < nums.length; x++) {
            if (x == 0) {
                nums[x] = from_last[1];
            } else if (x == nums.length - 1) {
                nums[x] = from_first[x - 1];
            } else {
                nums[x] = from_last[x + 1] * from_first[x - 1];
            }
        }
        return nums;
    }
}
