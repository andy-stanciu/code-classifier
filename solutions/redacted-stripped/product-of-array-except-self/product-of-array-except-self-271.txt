class Solution {

    public int[] method1(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        int var1 = 1;
        for (int i = 0; i < len; i++) {
            result[i] = var1;
            var1 *= nums[i];
        }
        int var2 = 1;
        for (int i = len - 1; i >= 0; i--) {
            result[i] *= var2;
            var2 *= nums[i];
        }
        return result;
    }
}
