class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] var1 = new int[n];
        int[] result = new int[n];
        var1[n - 1] = 1;
        for (int i = n - 2; i >= 0; i--) {
            var1[i] = nums[i + 1] * var1[i + 1];
        }
        int left = 1;
        for (int x = 0; x < n; x++) {
            result[x] = left * var1[x];
            left = left * nums[x];
        }
        return result;
    }
}
