class Solution {

    public int[] method1(int[] nums) {
        int len = nums.length;
        int[] res = new int[len];
        int left = 1;
        res[len - 1] = 1;
        for (int j = len - 2; j >= 0; j--) {
            res[j] = nums[j + 1] * res[j + 1];
        }
        for (int i = 1; i < len; i++) {
            left = left * nums[i - 1];
            res[i] = left * res[i];
        }
        return res;
    }
}
