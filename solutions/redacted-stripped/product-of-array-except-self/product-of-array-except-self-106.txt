class Solution {

    public int[] method1(int[] nums) {
        int mulOne = 1;
        int[] result = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            result[i] = mulOne;
            mulOne *= nums[i];
        }
        mulOne = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            result[i] = result[i] * mulOne;
            mulOne *= nums[i];
        }
        return result;
    }
}
