class Solution {

    public int[] method1(int[] nums) {
        int[] fin = new int[nums.length];
        int zeroCount = 0;
        int zeroInd = -1;
        long multi = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zeroCount++;
                zeroInd = i;
            } else {
                multi *= nums[i];
            }
        }
        if (zeroCount == 0) {
            for (int i = 0; i < nums.length; i++) {
                fin[i] = (int) multi / nums[i];
            }
        } else if (zeroCount == 1) {
            fin[zeroInd] = (int) multi;
        }
        return fin;
    }
}
