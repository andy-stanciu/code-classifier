class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        ans[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            ans[i] = ans[i - 1] * nums[i - 1];
        }
        int a = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            a = a * nums[i + 1];
            ans[i] = a * ans[i];
        }
        return ans;
    }
}
