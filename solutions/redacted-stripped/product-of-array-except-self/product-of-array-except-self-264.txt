class Solution {

    public int[] method1(int[] nums) {
        int var1 = nums.length;
        int[] var2 = new int[nums.length];
        var2[0] = 1;
        for (int i = 1; i < var1; i++) {
            var2[i] = var2[i - 1] * nums[i - 1];
        }
        int var3 = 1;
        for (int i = var1 - 1; i >= 0; i--) {
            var2[i] *= var3;
            var3 *= nums[i];
        }
        return var2;
    }
}
