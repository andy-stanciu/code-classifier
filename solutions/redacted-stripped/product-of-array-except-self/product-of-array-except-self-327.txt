class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int zeros = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zeros++;
                continue;
            }
            var1 *= nums[i];
        }
        int[] p = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            if (zeros == 1) {
                if (nums[i] == 0) {
                    p[i] = var1;
                    continue;
                }
                p[i] = 0;
            } else if (zeros > 1)
                p[i] = 0;
            else
                p[i] = var1 / nums[i];
        }
        return p;
    }
}
