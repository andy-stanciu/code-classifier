class Solution {

    public int[] method1(int[] nums) {
        int[] var1 = new int[nums.length];
        var1[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            var1[i] = var1[i - 1] * nums[i - 1];
        }
        int var2 = 1, num = 0;
        for (int j = nums.length - 1; j >= 0; j--) {
            num = nums[j];
            nums[j] = var1[j] * var2;
            var2 = num * var2;
        }
        return nums;
    }
}
