class Solution {

    public int findData(int[] n, int b, int i, int[] p) {
        if (i == n.length - 1) {
            p[i] = b;
            return n[i];
        }
        int v = findData(n, b * n[i], i + 1, p);
        p[i] = b * v;
        return v * n[i];
    }

    public int[] method1(int[] nums) {
        int[] var1 = new int[nums.length];
        int v = findData(nums, 1, 0, var1);
        return var1;
    }
}
