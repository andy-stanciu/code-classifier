class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] res = new int[n];
        int var1 = 1;
        int var2 = 1;
        for (int i = 0; i < n; i++) {
            res[i] = var1;
            var1 *= nums[i];
        }
        for (int i = n - 1; i >= 0; i--) {
            res[i] *= var2;
            var2 *= nums[i];
        }
        return res;
    }
}
