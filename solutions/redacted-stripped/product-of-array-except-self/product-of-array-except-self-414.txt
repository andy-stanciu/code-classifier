class Solution {

    public int[] method1(int[] nums) {
        int len = nums.length;
        int[] ans = new int[len];
        int pro = 1;
        for (int i = 0; i < len; i++) {
            ans[i] = pro;
            pro *= nums[i];
        }
        pro = 1;
        for (int i = len - 1; i >= 0; i--) {
            ans[i] *= pro;
            pro *= nums[i];
        }
        return ans;
    }
}
