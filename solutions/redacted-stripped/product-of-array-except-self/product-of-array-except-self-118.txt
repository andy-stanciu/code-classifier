class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int[] ans = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            ans[i] = var1;
            var1 = var1 * nums[i];
        }
        var1 = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            ans[i] = ans[i] * var1;
            var1 = var1 * nums[i];
        }
        return ans;
    }
}
