class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] before = new int[n];
        int[] after = new int[n];
        before[0] = 1;
        after[n - 1] = 1;
        for (int i = 1; i < n; i++) {
            before[i] = nums[i - 1] * before[i - 1];
            after[n - i - 1] = nums[n - i] * after[n - i];
        }
        int[] res = new int[n];
        for (int i = 0; i < n; i++) res[i] = before[i] * after[i];
        return res;
    }
}
