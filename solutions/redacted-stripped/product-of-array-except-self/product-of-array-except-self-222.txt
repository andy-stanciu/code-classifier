class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int a = 1;
        for (int i = 0; i < nums.length; i++) {
            ans[i] = a;
            a *= nums[i];
        }
        a = 1;
        for (int j = nums.length - 1; j >= 0; j--) {
            ans[j] *= a;
            a *= nums[j];
        }
        return ans;
    }
}
