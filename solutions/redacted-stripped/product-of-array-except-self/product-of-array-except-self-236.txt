class Solution {

    public int[] method1(int[] nums) {
        int[] output = new int[nums.length];
        for (int i = 0; i < nums.length; i++) output[i] = 1;
        for (int i = 0, prefix = 1; i < nums.length - 1; i++) {
            prefix *= nums[i];
            output[i + 1] *= prefix;
        }
        for (int i = nums.length - 1, postfix = 1; i >= 1; i--) {
            postfix *= nums[i];
            output[i - 1] *= postfix;
        }
        return output;
    }
}
