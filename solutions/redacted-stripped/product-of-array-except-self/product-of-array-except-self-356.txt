class Solution {

    public static int[] method1(int[] nums) {
        int len = nums.length;
        int[] ANSWER = new int[len];
        int[] var1 = new int[len];
        int[] var2 = new int[len];
        int prodLEFT = 1, prodRIGHT = 1;
        for (int i = 0; i < len; i++) {
            prodLEFT = prodLEFT * nums[i];
            var1[i] = prodLEFT;
        }
        for (int i = len - 1; i >= 0; i--) {
            prodRIGHT = prodRIGHT * nums[i];
            var2[i] = prodRIGHT;
        }
        for (int i = 0; i < len; i++) {
            if (i == 0) {
                ANSWER[i] = var2[i + 1];
            } else if (i == (len - 1)) {
                ANSWER[i] = var1[i - 1];
            } else {
                ANSWER[i] = var1[i - 1] * var2[i + 1];
            }
        }
        return ANSWER;
    }
}
