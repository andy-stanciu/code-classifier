class Solution {

    public int[] method1(int[] nums) {
        if (nums == null || nums.length == 0)
            return new int[0];
        int[] ans = new int[nums.length];
        int var1 = 1;
        for (int i = 0; i < ans.length; i++) {
            ans[i] = var1;
            var1 *= nums[i];
        }
        var1 = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            ans[i] *= var1;
            var1 *= nums[i];
        }
        return ans;
    }
}
