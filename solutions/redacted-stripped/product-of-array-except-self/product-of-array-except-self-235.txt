class Solution {

    public int[] method1(int[] nums) {
        int len = nums.length;
        int[] var1 = new int[len];
        int[] var2 = new int[len];
        var1[0] = nums[0];
        for (int i = 1; i < len; i++) {
            var1[i] = var1[i - 1] * nums[i];
        }
        var2[len - 1] = nums[len - 1];
        for (int i = len - 2; i >= 0; i--) {
            var2[i] = var2[i + 1] * nums[i];
        }
        for (int i = 0; i < len; i++) {
            if (i == 0) {
                nums[i] = var2[i + 1];
            } else if (i == len - 1) {
                nums[i] = var1[i - 1];
            } else {
                nums[i] = var1[i - 1] * var2[i + 1];
            }
        }
        return nums;
    }
}
