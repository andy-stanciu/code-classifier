class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int pro = 1;
        boolean flag = false;
        int zeroIdx = 0;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0)
                pro *= nums[i];
            else {
                flag = true;
                zeroIdx = i;
                count++;
            }
        }
        if (!flag) {
            for (int i = 0; i < nums.length; i++) {
                ans[i] = pro / nums[i];
            }
        } else if (count > 1) {
            return ans;
        } else {
            ans[zeroIdx] = pro;
        }
        return ans;
    }
}
