class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        int pref = 1;
        for (int i = 0; i < n; i++) {
            ans[i] = pref;
            pref *= nums[i];
        }
        int suff = 1;
        for (int i = n - 1; i >= 0; i--) {
            ans[i] *= suff;
            suff *= nums[i];
        }
        return ans;
    }
}
