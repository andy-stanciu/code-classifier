class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        answer[0] = 1;
        for (int i = 1; i < n; i++) {
            answer[i] = answer[i - 1] * nums[i - 1];
        }
        int var1 = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] *= var1;
            var1 *= nums[i];
        }
        return answer;
    }
}
