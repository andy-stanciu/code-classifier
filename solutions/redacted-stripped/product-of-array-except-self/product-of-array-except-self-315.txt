class Solution {

    public int[] method1(int[] nums) {
        int[] result = new int[nums.length];
        result[0] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            result[i] = nums[i] * result[i - 1];
        }
        int var1 = 1;
        for (int i = nums.length - 1; i > 0; i--) {
            result[i] = result[i - 1] * var1;
            var1 = var1 * nums[i];
        }
        result[0] = var1;
        return result;
    }
}
