class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        int preProd = 1;
        for (int i = 0; i < n; i++) {
            result[i] = preProd;
            preProd *= nums[i];
        }
        int postProd = 1;
        for (int i = n - 1; i >= 0; i--) {
            result[i] = result[i] * postProd;
            postProd *= nums[i];
        }
        return result;
    }
}
