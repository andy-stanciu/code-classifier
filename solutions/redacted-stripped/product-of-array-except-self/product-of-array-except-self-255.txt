class Solution {

    public int[] method1(int[] nums) {
        int[] var1 = new int[nums.length];
        var1[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            var1[i] = var1[i - 1] * nums[i - 1];
        }
        int var2 = 1;
        for (int i = nums.length - 1; i > 0; i--) {
            var2 *= nums[i];
            var1[i - 1] = var1[i - 1] * var2;
        }
        return var1;
    }
}
