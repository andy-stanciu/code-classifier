class Solution {

    public int[] method1(int[] nums) {
        int l = nums.length;
        int[] prod = new int[l];
        prod[0] = nums[0];
        for (int i = 1; i < l; i++) {
            prod[i] = prod[i - 1] * nums[i];
        }
        int suffix = 1;
        for (int i = l - 1; i > 0; i--) {
            prod[i] = prod[i - 1] * suffix;
            suffix = suffix * nums[i];
        }
        prod[0] = suffix;
        return prod;
    }
}
