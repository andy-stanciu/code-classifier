class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int n = nums.length;
        int var1 = 1;
        for (int i = 0; i < n; i++) {
            ans[i] = var1;
            var1 *= nums[i];
        }
        var1 = 1;
        for (int i = n - 1; i >= 0; i--) {
            ans[i] *= var1;
            var1 *= nums[i];
        }
        return ans;
    }
}
