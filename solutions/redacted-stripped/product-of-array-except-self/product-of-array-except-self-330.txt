class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        int var1 = nums[0];
        for (int i = 1; i < n; i++) {
            result[i] = var1;
            var1 *= nums[i];
        }
        var1 = nums[n - 1];
        for (int i = n - 2; i > 0; i--) {
            result[i] *= var1;
            var1 *= nums[i];
        }
        result[0] = var1;
        return result;
    }
}
