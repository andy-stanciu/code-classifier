class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int var2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                var2++;
            } else
                var1 *= nums[i];
        }
        int[] res = new int[nums.length];
        if (var2 == 0) {
            for (int i = 0; i < nums.length; i++) {
                res[i] = var1 / nums[i];
            }
        } else if (var2 == 1) {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] == 0) {
                    res[i] = var1;
                }
            }
        }
        return res;
    }
}
