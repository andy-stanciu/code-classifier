class Solution {

    public int[] method1(int[] nums) {
        int countZero = 0;
        int[] ans = new int[nums.length];
        int var1 = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0)
                var1 *= nums[i];
            else
                countZero++;
        }
        if (countZero >= 2)
            return ans;
        if (countZero == 1) {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] != 0)
                    ans[i] = 0;
                else
                    ans[i] = var1;
            }
            return ans;
        }
        for (int i = 0; i < ans.length; i++) {
            if (nums[i] != 0) {
                ans[i] = var1 / nums[i];
            } else {
                ans[i] = var1;
            }
        }
        return ans;
    }
}
