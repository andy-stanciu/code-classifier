class Solution {

    public int[] method1(int[] nums) {
        int[] a = new int[nums.length];
        int s = 1, p = 1;
        for (int i = 0; i < nums.length; ++i) {
            a[i] = s;
            s *= nums[i];
        }
        for (int j = nums.length - 1; j >= 0; --j) {
            a[j] *= p;
            p *= nums[j];
        }
        System.gc();
        return a;
    }
}
