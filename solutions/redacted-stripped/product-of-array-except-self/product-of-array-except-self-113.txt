class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int[] right = new int[nums.length];
        int count = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            right[i] = count;
            count *= nums[i];
        }
        count = 1;
        for (int i = 0; i < nums.length; i++) {
            ans[i] = count * right[i];
            count *= nums[i];
        }
        return ans;
    }
}
