class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        arr[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            arr[i] = arr[i + 1] * nums[i];
        }
        int var1 = 1;
        for (int i = 0; i < n - 1; i++) {
            arr[i] = var1 * arr[i + 1];
            var1 = var1 * nums[i];
        }
        arr[n - 1] = var1;
        return arr;
    }
}
