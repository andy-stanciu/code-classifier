class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] left = new int[n];
        if (n < 1) {
            return left;
        }
        int var1 = 1;
        for (int i = 0; i < n; i++) {
            var1 *= nums[i];
            left[i] = var1;
        }
        var1 = 1;
        for (int i = n - 1; i > 0; i--) {
            left[i] = left[i - 1] * var1;
            var1 *= nums[i];
        }
        left[0] = var1;
        return left;
    }
}
