class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        int t = 1;
        result[0] = 1;
        for (int i = 1; i < n; ++i) {
            t = t * nums[i - 1];
            result[i] = t;
        }
        t = 1;
        for (int i = n - 2; i >= 0; --i) {
            t = t * nums[i + 1];
            result[i] = result[i] * t;
        }
        return result;
    }
}
