class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] var1 = new int[n];
        var1[0] = 1;
        for (int i = 1; i < n; i++) {
            var1[i] = var1[i - 1] * nums[i - 1];
        }
        int suffix = 1;
        for (int i = n - 1; i >= 0; i--) {
            var1[i] = var1[i] * suffix;
            suffix = suffix * nums[i];
        }
        return var1;
    }
}
