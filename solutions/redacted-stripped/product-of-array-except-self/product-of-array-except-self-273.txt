class Solution {

    public int[] method1(int[] nums) {
        int[] result = new int[nums.length];
        for (int i = 0, var1 = 1; i < nums.length; i++) {
            result[i] = var1;
            var1 *= nums[i];
        }
        for (int i = nums.length - 1, var1 = 1; i >= 0; i--) {
            result[i] *= var1;
            var1 *= nums[i];
        }
        return result;
    }
}
