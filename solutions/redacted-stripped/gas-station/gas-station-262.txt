class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        final int var1 = Arrays.stream(param1).sum();
        final int costSum = Arrays.stream(cost).sum();
        if (var1 - costSum < 0)
            return -1;
        int ans = 0;
        int sum = 0;
        for (int i = 0; i < param1.length; ++i) {
            sum += param1[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                ans = i + 1;
            }
        }
        return ans;
    }
}
