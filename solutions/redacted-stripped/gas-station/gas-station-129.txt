class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0, tank = 0, diff = 0;
        for (int i = 0; i < param1.length; i++) {
            diff += param1[i] - cost[i];
            tank += param1[i] - cost[i];
            if (tank < 0) {
                tank = 0;
                start = i + 1;
            }
        }
        return diff < 0 ? -1 : start;
    }
}
