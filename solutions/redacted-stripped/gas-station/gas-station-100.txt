class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int cand = 0, curr = 0, total = 0;
        for (int i = 0; i < param1.length; i++) {
            curr += param1[i] - cost[i];
            total += param1[i] - cost[i];
            if (curr < 0) {
                cand = i + 1;
                curr = 0;
            }
        }
        return total >= 0 ? cand : -1;
    }
}
