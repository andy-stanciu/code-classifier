class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int idx = 0;
        int var1 = 0;
        int tcost = 0;
        int var2 = 0;
        for (int y = 0; y < param1.length; y++) {
            var2 = var2 + param1[y] - cost[y];
            var1 += param1[y];
            tcost += cost[y];
            if (var2 < 0) {
                idx = y + 1;
                var2 = 0;
            }
        }
        return var1 >= tcost ? idx : -1;
    }
}
