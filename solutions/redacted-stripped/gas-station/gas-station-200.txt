class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int diff = 0, res = 0, ans = 0;
        for (int i = 0; i < param1.length; i++) {
            diff += param1[i] - cost[i];
            res += param1[i] - cost[i];
            if (res < 0) {
                res = 0;
                ans = i + 1;
            }
        }
        return diff < 0 || ans >= param1.length ? -1 : ans;
    }
}
