class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            int tank = 0;
            boolean roundTrip = true;
            for (int j = i; j < n + i + 1; j++) {
                tank = tank + param1[j % n] - cost[j % n];
                if (tank < 0) {
                    roundTrip = false;
                    break;
                }
            }
            if (roundTrip)
                return i;
        }
        return -1;
    }
}
