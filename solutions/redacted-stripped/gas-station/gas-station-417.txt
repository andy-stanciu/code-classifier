class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        for (int g : param1) {
            var1 += g;
        }
        for (int c : cost) {
            totalCost += c;
        }
        if (var1 < totalCost) {
            return -1;
        }
        int total = 0;
        int res = 0;
        for (int i = 0; i < param1.length; i++) {
            total += (param1[i] - cost[i]);
            if (total < 0) {
                total = 0;
                res = i + 1;
            }
        }
        return res;
    }
}
