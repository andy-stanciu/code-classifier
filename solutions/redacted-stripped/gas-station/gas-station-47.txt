class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0;
        int total_tank = 0;
        int cur_tank = 0;
        for (int i = 0; i < param1.length; i++) {
            total_tank += param1[i] - cost[i];
            cur_tank += param1[i] - cost[i];
            if (cur_tank < 0) {
                cur_tank = 0;
                start = i + 1;
            }
        }
        return (total_tank < 0) ? -1 : start;
    }
}
