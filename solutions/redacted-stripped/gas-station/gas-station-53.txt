class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int total = 0;
        int var1 = 0;
        int index = 0;
        for (int i = 0; i < n; i++) {
            total += param1[i] - cost[i];
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                var1 = 0;
                index = i + 1;
            }
        }
        return (total < 0) ? -1 : index;
    }
}
