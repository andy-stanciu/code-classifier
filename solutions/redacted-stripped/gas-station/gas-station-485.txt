class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int fuelSum = 0, n = param1.length;
        int var1 = 0, start = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i] - cost[i];
            fuelSum += param1[i] - cost[i];
            if (var1 < 0) {
                start = i + 1;
                var1 = 0;
            }
        }
        return fuelSum >= 0 ? start : -1;
    }
}
