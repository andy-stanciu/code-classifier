class Solution {

    public int canCompleteCircuit(int[] A, int[] B) {
        int a = 0, i = 0;
        for (int j = 0; j < A.length; j++) {
            a = a + A[j] - B[j];
            if (a < 0) {
                a = 0;
                i = j + 1;
            }
        }
        for (int j = 0; j < i; j++) {
            a = a + A[j] - B[j];
            if (a < 0)
                return -1;
        }
        return i;
    }
}
