class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0;
        int var1 = 0;
        int fuelreq = 0;
        int remaining = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 = var1 + param1[i];
            fuelreq = fuelreq + cost[i];
        }
        if (var1 < fuelreq) {
            return -1;
        }
        for (int i = 0; i < param1.length; i++) {
            remaining = remaining + param1[i] - cost[i];
            if (remaining < 0) {
                start = i + 1;
                remaining = 0;
            }
        }
        return start;
    }
}
