class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int minCumulative = 0;
        int minCumulativeIndex = -1;
        int currentCumulative = 0;
        int nextCumulative;
        for (int i = 0; i < param1.length; i++) {
            nextCumulative = currentCumulative + param1[i] - cost[i];
            if (nextCumulative < minCumulative) {
                minCumulative = nextCumulative;
                minCumulativeIndex = i;
            }
            currentCumulative = nextCumulative;
        }
        if (currentCumulative < 0) {
            return -1;
        }
        return ((minCumulativeIndex + 1) % param1.length);
    }
}
