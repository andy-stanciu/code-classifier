class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int total = 0, position = 0, sum = 0;
        for (int i = 0; i < n; i++) {
            sum += (param1[i] - cost[i]);
            if (sum < 0) {
                total += sum;
                sum = 0;
                position = i + 1;
            }
        }
        total += sum;
        if (total >= 0)
            return position;
        else
            return -1;
    }
}
