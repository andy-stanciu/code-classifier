class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (Arrays.stream(param1).sum() < Arrays.stream(cost).sum())
            return -1;
        int total = 0;
        int ans = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            if (total < 0) {
                total = 0;
                ans = i + 1;
            }
        }
        return ans;
    }
}
