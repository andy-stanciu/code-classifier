class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int n1 = cost.length;
        int var1 = 0;
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < n1; i++) {
            var1 += param1[i] - cost[i];
            var3 += param1[i] - cost[i];
            if (var3 < 0) {
                var2 = i + 1;
                var3 = 0;
            }
        }
        return var1 >= 0 ? var2 : -1;
    }
}
