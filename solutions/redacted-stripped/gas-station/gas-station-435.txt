class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0;
        int curSum = 0;
        int idx = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            curSum += param1[i] - cost[i];
            if (curSum < 0) {
                curSum = 0;
                idx = i + 1;
            }
        }
        return total < 0 ? -1 : idx;
    }
}
