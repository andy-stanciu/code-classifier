class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int totalCost = 0;
        int var1 = 0;
        int tank = 0;
        int start = 0;
        for (int i = 0; i < n; i++) {
            totalCost += cost[i];
            var1 += param1[i];
            tank += (param1[i] - cost[i]);
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        if (var1 < totalCost)
            return -1;
        return start;
    }
}
