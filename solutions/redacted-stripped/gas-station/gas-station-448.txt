class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int res = 0;
        int n = param1.length;
        int curr = 0, prev = 0;
        for (int i = 0; i < n; i++) {
            curr += param1[i] - cost[i];
            if (curr < 0) {
                res = i + 1;
                prev += curr;
                curr = 0;
            }
        }
        return (curr + prev >= 0) ? res : -1;
    }
}
