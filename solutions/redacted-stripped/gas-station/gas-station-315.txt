class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int total = 0;
        for (int i = 0; i < n; i++) {
            total += param1[i] - cost[i];
        }
        if (total < 0)
            return -1;
        int index = 0;
        total = 0;
        for (int i = 0; i < n; i++) {
            total += param1[i] - cost[i];
            if (total < 0) {
                total = 0;
                index = i + 1;
            }
        }
        return index;
    }
}
