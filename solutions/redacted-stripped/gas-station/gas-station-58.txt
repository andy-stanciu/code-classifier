class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int sum = 0, max = 0, ans = -1;
        for (int i = param1.length - 1; i >= 0; i--) {
            sum = sum + (param1[i] - cost[i]);
            if (sum >= max) {
                ans = i;
                max = sum;
            }
        }
        return sum >= 0 ? ans : -1;
    }
}
