class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int totalCost = 0;
        int currentTank = 0;
        int var2 = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            currentTank += param1[i] - cost[i];
            if (currentTank < 0) {
                var2 = i + 1;
                currentTank = 0;
            }
        }
        if (var1 < totalCost) {
            return -1;
        }
        return var2;
    }
}
