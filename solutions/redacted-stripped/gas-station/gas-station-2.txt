class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0;
        int var1 = 0;
        int var2 = 0;
        for (int i = 0; i < param1.length; i++) {
            int fuelGain = param1[i] - cost[i];
            var1 += fuelGain;
            var2 += fuelGain;
            if (var2 < 0) {
                var2 = 0;
                start = i + 1;
            }
        }
        return var1 < 0 ? -1 : start;
    }
}
