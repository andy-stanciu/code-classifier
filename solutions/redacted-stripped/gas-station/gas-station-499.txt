class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0, totalCost = 0;
        int var2 = 0, startingPoint = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                startingPoint = i + 1;
                var2 = 0;
            }
        }
        if (var1 >= totalCost) {
            return startingPoint;
        } else {
            return -1;
        }
    }
}
