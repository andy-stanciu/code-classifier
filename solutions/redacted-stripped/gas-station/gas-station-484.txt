class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int leftOver = 0, maxLeftOver = 0, maxIndex = 0;
        int n = param1.length;
        for (int i = n - 1; i >= 0; i--) {
            leftOver += param1[i] - cost[i];
            if (leftOver > maxLeftOver) {
                maxLeftOver = leftOver;
                maxIndex = i;
            }
        }
        if (leftOver < 0)
            return -1;
        return maxIndex;
    }
}
