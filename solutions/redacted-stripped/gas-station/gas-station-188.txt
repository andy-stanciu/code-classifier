class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int diff = 0, var1 = 0, pos = 0;
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            diff += param1[i] - cost[i];
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                pos = i + 1;
                var1 = 0;
            }
        }
        if (diff < 0) {
            pos = -1;
        }
        return pos;
    }
}
