class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int totalCost = 0;
        for (int a : param1) {
            var1 += a;
        }
        for (int b : cost) {
            totalCost += b;
        }
        if (var1 < totalCost)
            return -1;
        int index = 0;
        int remainingFuel = 0;
        for (int i = 0; i < n; i++) {
            remainingFuel += param1[i] - cost[i];
            if (remainingFuel < 0) {
                index = i + 1;
                remainingFuel = 0;
            }
        }
        return index;
    }
}
