class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int cursum = 0;
        int totalsum = 0;
        int ind = 0;
        for (int i = 0; i < param1.length; i++) {
            cursum += param1[i] - cost[i];
            totalsum += param1[i] - cost[i];
            if (cursum < 0) {
                ind = i + 1;
                cursum = 0;
            }
        }
        if (totalsum < 0) {
            return -1;
        }
        return ind;
    }
}
