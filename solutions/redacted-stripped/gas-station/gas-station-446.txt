class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int bal = 0;
        int start = 0;
        int def = 0;
        for (int i = 0; i < n; i++) {
            bal += param1[i] - cost[i];
            if (bal < 0) {
                def += bal;
                start = i + 1;
                bal = 0;
            }
        }
        if (def + bal >= 0) {
            return start;
        } else {
            return -1;
        }
    }
}
