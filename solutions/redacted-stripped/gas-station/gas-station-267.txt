class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int var2 = 0;
        int AmountPaid = 0;
        int StartPoint = 0;
        for (int i = 0; i < n; i++) {
            var2 += param1[i];
            AmountPaid += cost[i];
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                var1 = 0;
                StartPoint = i + 1;
            }
        }
        return AmountPaid > var2 ? -1 : StartPoint;
    }
}
