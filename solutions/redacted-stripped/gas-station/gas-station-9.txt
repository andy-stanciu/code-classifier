class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int sum = 0, total = 0, ind = 0;
        for (int i = 0; i < param1.length; i++) {
            sum += param1[i] - cost[i];
            total += param1[i] - cost[i];
            if (sum < 0) {
                ind = i + 1;
                sum = 0;
            }
        }
        if (total >= 0)
            return ind;
        else
            return -1;
    }
}
