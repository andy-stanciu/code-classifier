class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total_tank = 0;
        int current_tank = 0;
        int var1 = 0;
        for (int i = 0; i < param1.length; i++) {
            total_tank += param1[i] - cost[i];
            current_tank += param1[i] - cost[i];
            if (current_tank < 0) {
                var1 = i + 1;
                current_tank = 0;
            }
        }
        return total_tank >= 0 ? var1 : -1;
    }
}
