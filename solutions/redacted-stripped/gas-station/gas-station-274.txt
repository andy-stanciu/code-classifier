class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int maxNegative = 0;
        int maxNegativeIndex = 0;
        int totalCost = 0;
        for (int i = 0; i < param1.length; i++) {
            totalCost = totalCost + param1[i] - cost[i];
            if (totalCost < maxNegative) {
                maxNegative = totalCost;
                maxNegativeIndex = i;
            }
        }
        if (totalCost < 0) {
            return -1;
        }
        return maxNegative < 0 ? (maxNegativeIndex + 1) % param1.length : 0;
    }
}
