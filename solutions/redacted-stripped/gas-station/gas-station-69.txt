class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (sum(param1) < sum(cost))
            return -1;
        int[] diff = new int[cost.length];
        for (int i = 0; i < cost.length; i++) {
            diff[i] = (param1[i] - cost[i]);
        }
        int total = 0;
        int start = 0;
        for (int i = 0; i < diff.length; i++) {
            total += diff[i];
            if (total < 0) {
                total = 0;
                start = i + 1;
            }
        }
        return start;
    }

    private int //noting just returns the sum of an array..
    sum(//noting just returns the sum of an array..
    int[] arr) {
        int sum = 0;
        for (int i = 0; i < arr.length; i++) {
            sum += arr[i];
        }
        return sum;
    }
}
