class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int minCost = Integer.MAX_VALUE;
        int src = -1;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] < cost[i] || (param1[i] == cost[i] && param1.length != 1)) {
                continue;
            }
            int var1 = 0;
            int costs = 0;
            int index = i;
            costs = param1[index];
            boolean valid = true;
            while (var1 != param1.length) {
                var1 = var1 + 1;
                costs = costs - cost[index];
                index = index + 1;
                if (index == param1.length)
                    index = 0;
                costs = costs + param1[index];
                if (costs < cost[index]) {
                    valid = false;
                    break;
                }
            }
            if (minCost > costs && valid) {
                minCost = costs;
                src = i;
            }
        }
        return src;
    }
}
