class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int total_tank = 0;
        int curr_tank = 0;
        int var1 = 0;
        for (int i = 0; i < n; ++i) {
            total_tank += param1[i] - cost[i];
            curr_tank += param1[i] - cost[i];
            if (curr_tank < 0) {
                var1 = i + 1;
                curr_tank = 0;
            }
        }
        return total_tank >= 0 ? var1 : -1;
    }
}
