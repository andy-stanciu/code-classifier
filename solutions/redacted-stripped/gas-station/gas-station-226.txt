class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int tank = param1[0];
        int[] first = new int[param1.length];
        int[] second = new int[param1.length];
        for (int i = 1; i < param1.length; i++) {
            tank -= cost[i - 1];
            first[i] = tank;
            tank += param1[i];
        }
        for (int i = 0; i < param1.length; i++) {
            if (i == 0) {
                tank -= cost[cost.length - 1];
            } else
                tank -= cost[i - 1];
            second[i] = tank;
            tank += param1[i];
        }
        int last = 0;
        for (int i = 0; i < first.length; i++) {
            if (first[i] < first[last]) {
                last = i;
            }
        }
        for (int i = 0; i < second.length; i++) {
            if (second[i] < first[last]) {
                return -1;
            }
        }
        return last;
    }
}
