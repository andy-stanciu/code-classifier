class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int startIndex = 0;
        int var1 = 0;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i] - cost[i];
            sum += param1[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                startIndex = i + 1;
            }
        }
        return var1 < 0 ? -1 : startIndex;
    }
}
