class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                var3 = i + 1;
                var2 = 0;
            }
        }
        if (var1 < totalCost) {
            return -1;
        } else {
            return var3;
        }
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        int[] var4 = { 1, 2, 3, 4, 5 };
        int[] cost1 = { 3, 4, 5, 1, 2 };
        System.out.println(sol.canCompleteCircuit(var4, cost1));
        int[] var5 = { 2, 3, 4 };
        int[] cost2 = { 3, 4, 3 };
        System.out.println(sol.canCompleteCircuit(var5, cost2));
    }
}
