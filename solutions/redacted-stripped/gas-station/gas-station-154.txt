class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int var2 = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                start = i + 1;
                var2 = 0;
            }
        }
        return (var1 >= totalCost) ? start : -1;
    }
}
