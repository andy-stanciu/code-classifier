class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int diff = 0, total = 0;
        int idx = 0;
        for (int i = 0; i < param1.length; i++) {
            diff += param1[i] - cost[i];
            total += param1[i] - cost[i];
            if (diff < 0) {
                diff = 0;
                idx = i + 1;
            }
        }
        return total < 0 ? -1 : idx;
    }
}
