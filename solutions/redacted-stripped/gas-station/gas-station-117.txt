class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int sumOfDiff = 0;
        for (int i = 0; i < n; i++) {
            sumOfDiff += (param1[i] - cost[i]);
        }
        if (sumOfDiff < 0)
            return -1;
        int i = 0, tmp = 0, start = 0;
        while (i < n) {
            tmp += (param1[i] - cost[i]);
            if (tmp < 0) {
                start = i + 1;
                tmp = 0;
            }
            i++;
        }
        return start;
    }
}
