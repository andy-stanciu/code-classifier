class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int position = 0, rem = 0, total = 0;
        for (int i = 0; i < param1.length; i++) {
            rem += param1[i] - cost[i];
            if (rem < 0) {
                rem = 0;
                position = i + 1;
            }
            total += param1[i] - cost[i];
        }
        return total >= 0 ? position : -1;
    }
}
