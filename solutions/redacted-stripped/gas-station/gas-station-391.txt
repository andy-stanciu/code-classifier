class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int srt = 0;
        int rem = 0;
        int deficit = 0;
        for (int i = 0; i < param1.length; i++) {
            rem = rem + (param1[i] - cost[i]);
            if (rem < 0) {
                srt = i + 1;
                deficit = deficit + rem;
                rem = 0;
            }
        }
        if (deficit + rem >= 0) {
            return srt;
        }
        return -1;
    }
}
