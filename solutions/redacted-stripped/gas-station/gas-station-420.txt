class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int netEarn = 0;
        int res = 0;
        int maxNetEarn = 0;
        for (int i = param1.length - 1; i >= 0; i--) {
            netEarn += param1[i] - cost[i];
            if (netEarn > maxNetEarn) {
                res = i;
                maxNetEarn = netEarn;
            }
        }
        if (netEarn < 0)
            return -1;
        return res;
    }
}
