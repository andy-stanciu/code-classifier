class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int i = 0;
        int j = param1.length;
        int difference = j - i;
        int sum = param1[i] - cost[i];
        int totalSum = 0;
        for (int k = 0; k < param1.length; k++) {
            totalSum += param1[k] - cost[k];
        }
        if (totalSum < 0) {
            return -1;
        } else {
            while (difference != 1) {
                if (sum < 0) {
                    j--;
                    sum += param1[j] - cost[j];
                } else if (sum >= 0) {
                    i++;
                    sum += param1[i] - cost[i];
                }
                difference = j - i;
            }
            if (j == param1.length) {
                return 0;
            } else {
                return j;
            }
        }
    }
}
