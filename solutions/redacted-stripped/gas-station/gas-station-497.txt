class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int origin = 0;
        int startPnt = 0;
        for (int i = 0; i < param1.length; i++) {
            origin = origin + param1[i] - cost[i];
            if (origin < 0) {
                origin = 0;
                startPnt = i + 1;
            }
        }
        return (Arrays.stream(param1).sum() < Arrays.stream(cost).sum()) ? -1 : startPnt;
    }
}
