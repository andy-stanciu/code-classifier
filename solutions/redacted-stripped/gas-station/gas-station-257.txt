class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int costSum = 0;
        int len = cost.length;
        for (int i = 0; i < len; i++) {
            var1 += param1[i];
            costSum += cost[i];
        }
        if (var1 < costSum)
            return -1;
        int start = 0;
        int tank = 0;
        for (int i = 0; i < len; i++) {
            tank += param1[i] - cost[i];
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        return start;
    }
}
