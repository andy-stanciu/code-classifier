class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int st = -1;
        int sum = 0;
        int asum = 0;
        for (int i = 0; i < param1.length; i++) {
            param1[i] -= cost[i];
            sum += param1[i];
            asum += param1[i];
            if (asum < 0) {
                asum = 0;
                st = i + 1;
            } else if (st == -1) {
                st = i;
            }
        }
        if (sum < 0) {
            return -1;
        }
        return st;
    }
}
