class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int ans = 0, sum = 0, total = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            sum += param1[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                ans = i + 1;
            }
        }
        if (total >= 0) {
            return ans;
        } else {
            return -1;
        }
    }
}
