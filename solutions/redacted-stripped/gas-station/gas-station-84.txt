class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalFuel = 0;
        int totalCost = 0;
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            totalFuel += param1[i];
        }
        for (int i = 0; i < n; i++) {
            totalCost += cost[i];
        }
        if (totalFuel < totalCost) {
            return -1;
        }
        int start = 0;
        int currFuel = 0;
        for (int i = 0; i < n; i++) {
            currFuel += (param1[i] - cost[i]);
            if (currFuel < 0) {
                start = i + 1;
                currFuel = 0;
            }
        }
        return start;
    }
}
