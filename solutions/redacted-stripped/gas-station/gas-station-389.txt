class Solution {

    public int sum(int[] arr) {
        int s = 0;
        for (int num : arr) s += num;
        return s;
    }

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (sum(param1) < sum(cost)) {
            return -1;
        }
        int total = 0;
        int pos = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            if (total < 0) {
                pos = i + 1;
                total = 0;
            }
        }
        return total >= 0 ? pos : -1;
    }
}
