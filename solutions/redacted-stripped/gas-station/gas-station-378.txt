class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int startIndex = 0;
        int var1 = 0;
        int var2 = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i] - cost[i];
            if (var1 < var2) {
                var2 = var1;
                startIndex = (i + 1) % param1.length;
            }
        }
        if (var1 >= 0) {
            return startIndex;
        }
        return -1;
    }
}
