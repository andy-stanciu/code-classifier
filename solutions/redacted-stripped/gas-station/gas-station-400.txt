class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int var2 = 0;
        int startingIndex = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                var2 += var1;
                var1 = 0;
                startingIndex = i + 1;
            }
        }
        return var1 + var2 >= 0 ? startingIndex : -1;
    }
}
