class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int amt = 0;
        int t = 0;
        int idx = 0;
        for (int i = 0; i < param1.length; i++) {
            amt += param1[i] - cost[i];
            t += param1[i] - cost[i];
            if (amt < 0) {
                amt = 0;
                idx = i + 1;
            }
        }
        if (t < 0) {
            return -1;
        }
        return idx;
    }
}
