class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0, available = 0, index = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            available += param1[i] - cost[i];
            if (available < 0) {
                available = 0;
                index = i + 1;
            }
        }
        return total < 0 ? -1 : index;
    }
}
