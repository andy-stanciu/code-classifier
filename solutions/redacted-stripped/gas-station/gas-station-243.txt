class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int potentialPosition = -1;
        int largestStartPossible = -1;
        int var1 = 0;
        int len = param1.length;
        for (int i = len - 1; i >= 0; i--) {
            int diff = param1[i] - cost[i];
            var1 += diff;
            if (var1 > largestStartPossible) {
                potentialPosition = i;
                largestStartPossible = var1;
            }
        }
        if (potentialPosition == -1 || var1 < 0)
            return -1;
        return potentialPosition;
    }
}
