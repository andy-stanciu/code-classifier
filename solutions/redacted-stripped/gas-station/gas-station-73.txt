class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length, total = 0, extra = 0, start = 0;
        for (int i = 0; i < n; i++) {
            total += param1[i] - cost[i];
            extra += param1[i] - cost[i];
            if (extra < 0) {
                extra = 0;
                start = i + 1;
            }
        }
        return total >= 0 ? start : -1;
    }
}
