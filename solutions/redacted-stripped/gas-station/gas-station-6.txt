public class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int var2 = 0;
        int startIndex = 0;
        for (int i = 0; i < param1.length; i++) {
            int balance = param1[i] - cost[i];
            var1 += balance;
            var2 += balance;
            if (var2 < 0) {
                startIndex = i + 1;
                var2 = 0;
            }
        }
        return var1 >= 0 ? startIndex : -1;
    }
}
