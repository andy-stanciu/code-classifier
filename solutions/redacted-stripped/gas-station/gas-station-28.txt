class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int total_cost = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
        }
        for (int i = 0; i < cost.length; i++) {
            total_cost += cost[i];
        }
        if (var1 < total_cost) {
            return -1;
        }
        int start = 0;
        int currentFuel = 0;
        for (int i = 0; i < param1.length; i++) {
            currentFuel += param1[i] - cost[i];
            if (currentFuel < 0) {
                start = i + 1;
                currentFuel = 0;
            }
        }
        return start;
    }
}
