class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int fuel = 0;
        int sum = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            sum += (param1[i] - cost[i]);
            fuel += (param1[i] - cost[i]);
            if (fuel < 0) {
                fuel = 0;
                start = i + 1;
            }
        }
        if (sum >= 0) {
            return start;
        }
        return -1;
    }
}
