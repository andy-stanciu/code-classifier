class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int var2 = 0;
        int startPoint = -1;
        for (int i = 0; i < param1.length; i++) {
            var2 = var2 + param1[i] - cost[i];
            if (var2 < 0) {
                var1 += var2;
                var2 = 0;
                startPoint = -1;
            } else if (startPoint == -1) {
                startPoint = i;
            }
        }
        return ((var2 + var1) >= 0) ? startPoint : -1;
    }
}
