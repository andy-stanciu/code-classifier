class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int dis = 0;
        int prevDis = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            dis += param1[i] - cost[i];
            if (dis < 0) {
                start = i + 1;
                prevDis += dis;
                dis = 0;
            }
        }
        if (dis + prevDis >= 0) {
            return start;
        }
        return -1;
    }
}
