class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int costsum = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            costsum += cost[i];
        }
        if (var1 < costsum)
            return -1;
        else {
            int temp = 0;
            int result = 0;
            for (int i = 0; i < n; i++) {
                temp += param1[i] - cost[i];
                if (temp < 0) {
                    temp = 0;
                    result = i + 1;
                }
            }
            return result;
        }
    }
}
