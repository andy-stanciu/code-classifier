class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        final int len = param1.length;
        int var1 = 0;
        int maxDeficit = 0;
        int currentFuel = 0;
        for (int i = 0; i < len; ++i) {
            currentFuel += param1[i] - cost[i];
            if (currentFuel < maxDeficit) {
                maxDeficit = currentFuel;
                var1 = i + 1;
            }
        }
        if (currentFuel < 0)
            return -1;
        return var1;
    }
}
