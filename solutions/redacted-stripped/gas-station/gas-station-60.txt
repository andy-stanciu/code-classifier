class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = param1.length, total = 0, ans = 0, ind = 0;
        for (int i = 0; i < var1; i++) {
            total += (param1[i] - cost[i]);
            ans += (param1[i] - cost[i]);
            if (ans < 0) {
                ans = 0;
                ind = i + 1;
            }
        }
        return total < 0 ? -1 : ind;
    }
}
