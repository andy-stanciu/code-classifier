class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, starting_point = 0, var2 = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i] - cost[i];
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                starting_point = i + 1;
                var2 = 0;
            }
        }
        if (var1 < 0) {
            return -1;
        }
        return starting_point;
    }
}
