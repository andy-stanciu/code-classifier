class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalExtraFuel = 0;
        int extraFuel = 0;
        int index = 0;
        int l = param1.length;
        for (int i = 0; i < l; i++) {
            int diff = param1[i] - cost[i];
            totalExtraFuel += diff;
            extraFuel += diff;
            if (extraFuel < 0) {
                index = i + 1;
                extraFuel = 0;
            }
        }
        return (totalExtraFuel < 0) ? -1 : index;
    }
}
