class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < n; i++) {
            int var4 = param1[i] - cost[i];
            var1 += var4;
            var2 += var4;
            if (var2 < 0) {
                var3 = i + 1;
                var2 = 0;
            }
        }
        return var1 >= 0 ? var3 : -1;
    }
}
