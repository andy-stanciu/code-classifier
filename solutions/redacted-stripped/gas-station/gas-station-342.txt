class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int currentFuel = 0;
        int start = 0;
        int var1 = Arrays.stream(param1).sum();
        int costSum = Arrays.stream(cost).sum();
        if (var1 < costSum) {
            return -1;
        }
        for (int i = 0; i < param1.length; i++) {
            currentFuel += param1[i] - cost[i];
            if (currentFuel < 0) {
                start = i + 1;
                currentFuel = 0;
            }
        }
        return start;
    }
}
