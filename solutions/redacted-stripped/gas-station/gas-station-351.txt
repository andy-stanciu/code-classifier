class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int s = -1;
        int q = 0;
        int g = 0;
        for (int i = param1.length - 1; i >= 0; --i) {
            q += param1[i] - cost[i];
            if (s == -1 || q > g) {
                s = i;
                g = q;
            }
        }
        if (q < 0) {
            return -1;
        }
        return s;
    }
}
