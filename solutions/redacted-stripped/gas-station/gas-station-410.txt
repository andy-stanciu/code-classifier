class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int index = 0;
        int totalFuel = 0;
        int fuel = 0;
        for (int i = 0; i < n; i++) {
            fuel += param1[i] - cost[i];
            totalFuel += param1[i] - cost[i];
            if (fuel < 0) {
                index = 1 + i;
                fuel = 0;
            }
        }
        return totalFuel >= 0 ? index : -1;
    }
}
