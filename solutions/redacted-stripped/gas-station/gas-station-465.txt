class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0, finds = 0, index = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            finds += param1[i] - cost[i];
            if (finds < 0) {
                finds = 0;
                index = i + 1;
            }
        }
        if (total < 0) {
            return -1;
        }
        return index;
    }
}
