class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        var bor = 0;
        var var1 = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                bor += -var1;
                var1 = 0;
                start = i + 1;
            }
        }
        return (var1 < bor) ? -1 : start;
    }
}
