class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < param1.length; i++) {
            int diff = param1[i] - cost[i];
            var1 += diff;
            var2 += diff;
            if (var2 < 0) {
                var3 = i + 1;
                var2 = 0;
            }
        }
        if (var1 < 0) {
            return -1;
        }
        return var3;
    }
}
