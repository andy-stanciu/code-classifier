class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0, total_cost = 0;
        int var2 = 0, starting_point = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            total_cost += cost[i];
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                starting_point = i + 1;
                var2 = 0;
            }
        }
        return (var1 < total_cost) ? -1 : starting_point;
    }
}
