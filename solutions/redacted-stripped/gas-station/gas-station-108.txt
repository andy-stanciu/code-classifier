public class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalCost = 0;
        for (int i = 0; i < param1.length; i++) {
            totalCost += (param1[i] - cost[i]);
        }
        if (totalCost < 0)
            return -1;
        int curr = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            curr += (param1[i] - cost[i]);
            if (curr < 0) {
                curr = 0;
                start = i + 1;
            }
        }
        return start;
    }
}
