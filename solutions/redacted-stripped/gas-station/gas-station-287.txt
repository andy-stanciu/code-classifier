class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalCost = 0;
        int circuitCheck = 0;
        int index = 0;
        for (int i = 0; i < param1.length; i++) {
            if (totalCost < 0) {
                index = i;
                totalCost = 0;
            }
            totalCost += (param1[i] - cost[i]);
            circuitCheck += (param1[i] - cost[i]);
        }
        return (circuitCheck >= 0) ? index : -1;
    }
}
