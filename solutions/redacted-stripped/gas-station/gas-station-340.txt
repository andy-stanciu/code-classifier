class Solution {

    public int canCompleteCircuit(int[] fuel, int[] expenditure) {
        int fuelSum = 0;
        int expenditureSum = 0;
        for (int i = 0; i < fuel.length; i++) {
            fuelSum += fuel[i];
            expenditureSum += expenditure[i];
        }
        if (fuelSum - expenditureSum < 0) {
            return -1;
        }
        int answer = 0;
        int currentSum = 0;
        for (int i = 0; i < fuel.length; i++) {
            currentSum += fuel[i] - expenditure[i];
            if (currentSum < 0) {
                currentSum = 0;
                answer = i + 1;
            }
        }
        return answer;
    }
}
