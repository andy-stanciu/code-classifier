class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int prev = 0, count = 0, ans = -1;
        for (int i = 0; i < param1.length; i++) {
            count += param1[i] - cost[i];
            if (count >= 0) {
                if (ans == -1) {
                    ans = i;
                }
            } else {
                prev += count;
                count = 0;
                ans = -1;
            }
        }
        if (count + prev < 0) {
            return -1;
        }
        return ans;
    }
}
