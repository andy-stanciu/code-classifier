class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0, var1 = 0, start = 0;
        for (int i = 0; i < param1.length; i++) {
            int cur = param1[i] - cost[i];
            total += cur;
            var1 += cur;
            if (var1 < 0) {
                var1 = 0;
                start = i + 1;
            }
        }
        if (total < 0)
            return -1;
        return start;
    }
}
