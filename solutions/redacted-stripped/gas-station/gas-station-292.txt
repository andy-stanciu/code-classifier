class Solution {

    public int canCompleteCircuit(int[] petrol, int[] distance) {
        int l = petrol.length;
        int n = 0, d = 0, i = 0;
        int prev_req = 0;
        while (i < l) {
            int a = petrol[i];
            int b = distance[i];
            int diff = a - b;
            d += diff;
            if (d < 0) {
                if (i == l - 1) {
                    return -1;
                }
                prev_req += d;
                n = i + 1;
                d = 0;
            }
            i++;
        }
        return (prev_req + d >= 0) ? n : -1;
    }
}
