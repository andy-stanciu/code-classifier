class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        pre(root, list);
        return list;
    }

    public void pre(TreeNode root, ArrayList<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        pre(root.left, list);
        pre(root.right, list);
    }
}
