class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        helper(root, li);
        return li;
    }

    private void helper(TreeNode root, List<Integer> li) {
        if (root == null) {
            return;
        }
        li.add(root.val);
        helper(root.left, li);
        helper(root.right, li);
    }
}
