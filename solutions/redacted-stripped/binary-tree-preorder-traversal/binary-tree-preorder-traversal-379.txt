class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        method2(root, ls);
        return ls;
    }

    public static void method2(TreeNode node, List<Integer> ls) {
        if (node == null)
            return;
        ls.add(node.val);
        method2(node.left, ls);
        method2(node.right, ls);
    }
}
