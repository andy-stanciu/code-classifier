class Solution {

    List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return result;
        addElement(root);
        return result;
    }

    public void addElement(TreeNode root) {
        if (root == null)
            return;
        result.add(root.val);
        addElement(root.left);
        addElement(root.right);
    }
}
