class Solution {

    public static void order(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        order(root.left, list);
        order(root.right, list);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        order(root, list);
        return list;
    }
}
