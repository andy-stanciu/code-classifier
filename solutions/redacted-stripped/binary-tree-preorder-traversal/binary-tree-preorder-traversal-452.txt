class Solution {

    private static List<Integer> rec(TreeNode root, List<Integer> li) {
        if (root == null) {
            return li;
        }
        li.add(root.val);
        rec(root.left, li);
        rec(root.right, li);
        return li;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        return rec(root, li);
    }
}
