class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new LinkedList<>();
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
        while (!stack.isEmpty()) {
            TreeNode temp = stack.pop();
            if (temp != null) {
                output.add(temp.val);
                stack.push(temp.right);
                stack.push(temp.left);
            }
        }
        return output;
    }
}
