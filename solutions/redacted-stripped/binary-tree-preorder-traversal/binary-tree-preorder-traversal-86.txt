class Solution {

    static List<Integer> ar = new ArrayList<Integer>();

    static void PreeOrder(TreeNode root) {
        if (root == null) {
            return;
        }
        ar.add(root.val);
        PreeOrder(root.left);
        PreeOrder(root.right);
    }

    public List<Integer> method1(TreeNode root) {
        ar = new ArrayList<Integer>();
        PreeOrder(root);
        return ar;
    }
}
