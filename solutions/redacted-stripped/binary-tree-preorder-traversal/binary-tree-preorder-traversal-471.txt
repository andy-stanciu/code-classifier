class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        method2(root, result);
        return result;
    }

    private void method2(TreeNode node, List<Integer> result) {
        if (node == null) {
            return;
        }
        result.add(node.val);
        method2(node.left, result);
        method2(node.right, result);
    }
}
