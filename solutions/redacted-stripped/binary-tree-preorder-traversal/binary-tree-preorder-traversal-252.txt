class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        traverse(root, ans);
        return ans;
    }

    public void traverse(TreeNode node, List<Integer> ans) {
        if (node == null)
            return;
        ans.add(node.val);
        traverse(node.left, ans);
        traverse(node.right, ans);
    }
}
