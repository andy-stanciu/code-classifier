class Solution {

    static List<Integer> lst;

    public List<Integer> method1(TreeNode root) {
        lst = new ArrayList<>();
        pretrav(root);
        return lst;
    }

    static void pretrav(TreeNode root) {
        if (root == null)
            return;
        lst.add(root.val);
        pretrav(root.left);
        pretrav(root.right);
    }
}
