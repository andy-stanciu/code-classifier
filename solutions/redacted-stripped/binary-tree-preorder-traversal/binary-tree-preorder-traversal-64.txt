class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<Integer>();
        if (root == null)
            return ans;
        Stack<TreeNode> st = new Stack<TreeNode>();
        st.add(root);
        TreeNode n;
        while (!st.isEmpty()) {
            n = st.pop();
            ans.add(n.val);
            if (n.right != null)
                st.push(n.right);
            if (n.left != null)
                st.push(n.left);
        }
        return ans;
    }
}
