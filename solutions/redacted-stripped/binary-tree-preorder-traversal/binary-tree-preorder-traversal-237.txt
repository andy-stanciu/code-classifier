class Solution {

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return list;
    }

    List<Integer> list = new ArrayList<>();

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        method2(root.left);
        method2(root.right);
    }
}
