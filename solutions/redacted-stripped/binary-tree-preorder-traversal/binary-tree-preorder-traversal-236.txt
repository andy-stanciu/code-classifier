class Solution {

    public static void method1(TreeNode node, ArrayList<Integer> list1) {
        if (node == null) {
            return;
        }
        list1.add(node.val);
        method1(node.left, list1);
        method1(node.right, list1);
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        method1(root, list);
        return list;
    }
}
