class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        return var1;
    }

    public void method2(TreeNode root, List<Integer> var1) {
        if (root == null)
            return;
        var1.add(root.val);
        method2(root.left, var1);
        method2(root.right, var1);
    }
}
