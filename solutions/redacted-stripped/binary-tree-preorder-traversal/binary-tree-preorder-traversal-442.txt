class Solution {

    List<Integer> list = new ArrayList<Integer>();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        method2(root.left);
        method2(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return list;
    }
}
