class Solution {

    public void amrit(TreeNode root, List<Integer> ans) {
        List<Integer> a = new ArrayList<>();
        if (root == null) {
            return;
        }
        ans.add(root.val);
        amrit(root.left, ans);
        amrit(root.right, ans);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        amrit(root, ans);
        return ans;
    }
}
