class Solution {

    ArrayList<Integer> arr = new ArrayList<>();

    public void method1(TreeNode root) {
        if (root != null) {
            arr.add(root.val);
            method1(root.left);
            method1(root.right);
        }
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return arr;
    }
}
