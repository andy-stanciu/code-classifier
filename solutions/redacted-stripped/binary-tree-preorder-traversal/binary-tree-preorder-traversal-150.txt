class Solution {

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<TreeNode>();
        if (root == null)
            return list;
        st.push(root);
        while (!st.isEmpty()) {
            root = st.pop();
            list.add(root.val);
            if (root.right != null) {
                st.push(root.right);
            }
            if (root.left != null) {
                st.push(root.left);
            }
        }
        return list;
    }
}
