class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> temp = new ArrayList<>();
        if (root == null) {
            return temp;
        }
        temp.add(root.val);
        List<Integer> left = method1(root.left);
        List<Integer> right = method1(root.right);
        temp.addAll(left);
        temp.addAll(right);
        return temp;
    }
}
