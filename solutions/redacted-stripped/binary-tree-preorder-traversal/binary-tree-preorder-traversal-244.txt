class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        fun(root, result);
        return result;
    }

    public void fun(TreeNode root, List<Integer> result) {
        if (root != null) {
            result.add(root.val);
            fun(root.left, result);
            fun(root.right, result);
        }
    }
}
