class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<TreeNode>();
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return var1;
        st.push(root);
        while (!st.isEmpty()) {
            root = st.pop();
            var1.add(root.val);
            if (root.right != null) {
                st.push(root.right);
            }
            if (root.left != null) {
                st.push(root.left);
            }
        }
        return var1;
    }
}
