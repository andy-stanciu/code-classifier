class Solution {

    public static void method1(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        ans.add(root.val);
        method1(root.left, ans);
        method1(root.right, ans);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method1(root, result);
        return result;
    }
}
