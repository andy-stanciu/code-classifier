class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        pre(list, root);
        return list;
    }

    public void pre(ArrayList<Integer> list, TreeNode root) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        pre(list, root.left);
        pre(list, root.right);
    }
}
