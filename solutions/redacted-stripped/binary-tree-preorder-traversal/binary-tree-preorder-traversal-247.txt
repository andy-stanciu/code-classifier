class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        pre(root, ls);
        return ls;
    }

    private void pre(TreeNode root, List<Integer> ls) {
        if (root == null)
            return;
        ls.add(root.val);
        pre(root.left, ls);
        pre(root.right, ls);
    }
}
