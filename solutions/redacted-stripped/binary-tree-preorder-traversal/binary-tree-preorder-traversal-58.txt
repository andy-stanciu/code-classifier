class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<Integer>();
        method2(root, ans);
        return ans;
    }

    public void method2(TreeNode root, ArrayList<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        method2(root.left, list);
        method2(root.right, list);
    }
}
