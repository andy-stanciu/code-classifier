class Solution {

    static void method1(TreeNode root, List<Integer> arr) {
        if (root == null)
            return;
        arr.add(root.val);
        method1(root.left, arr);
        method1(root.right, arr);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> arr = new ArrayList<>();
        method1(root, arr);
        return arr;
    }
}
