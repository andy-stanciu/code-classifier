class Solution {

    public List<Integer> method1(TreeNode root) {
        Deque<TreeNode> stack = new LinkedList<>();
        List<Integer> list = new LinkedList<>();
        if (root == null)
            return list;
        stack.add(root);
        while (!stack.isEmpty()) {
            root = stack.pollLast();
            if (root.right != null) {
                stack.add(root.right);
            }
            if (root.left != null) {
                stack.add(root.left);
            }
            list.add(root.val);
        }
        return list;
    }
}
