class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> a1 = new ArrayList<>();
        method2(root, a1);
        return a1;
    }

    public void method2(TreeNode root, List<Integer> a1) {
        if (root != null) {
            a1.add(root.val);
            method2(root.left, a1);
            method2(root.right, a1);
        }
    }
}
