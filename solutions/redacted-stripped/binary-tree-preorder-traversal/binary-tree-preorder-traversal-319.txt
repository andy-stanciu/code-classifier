class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> stack = new Stack<>();
        List<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        stack.push(root);
        while (!stack.isEmpty()) {
            TreeNode temp = stack.pop();
            ans.add(temp.val);
            if (temp.right != null) {
                stack.push(temp.right);
            }
            if (temp.left != null) {
                stack.push(temp.left);
            }
        }
        return ans;
    }
}
