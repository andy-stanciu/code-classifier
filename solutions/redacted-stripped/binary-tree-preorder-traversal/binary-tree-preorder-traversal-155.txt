class Solution {

    static ArrayList<TreeNode> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<>();
        method2(ans, root);
        return ans;
    }

    public void method2(List<Integer> ans, TreeNode root) {
        if (root == null) {
            return;
        }
        ans.add(root.val);
        method2(ans, root.left);
        method2(ans, root.right);
    }
}
