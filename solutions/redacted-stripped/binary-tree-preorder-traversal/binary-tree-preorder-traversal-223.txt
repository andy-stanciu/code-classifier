class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null)
            return result;
        result.add(root.val);
        if (root.left != null)
            result.addAll(method1(root.left));
        if (root.right != null)
            result.addAll(method1(root.right));
        return result;
    }
}
