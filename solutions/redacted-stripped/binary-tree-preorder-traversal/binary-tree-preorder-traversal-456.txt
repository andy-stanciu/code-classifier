class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> n = new ArrayList<>();
        Stack<TreeNode> s = new Stack();
        s.add(root);
        while (!s.isEmpty()) {
            TreeNode curr = s.pop();
            if (curr != null) {
                n.add(curr.val);
                s.add(curr.right);
                s.add(curr.left);
            }
        }
        return n;
    }
}
