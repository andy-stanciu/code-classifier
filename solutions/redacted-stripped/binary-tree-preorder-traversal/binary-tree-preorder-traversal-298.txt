class Solution {

    List<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return ans;
    }

    private void method2(TreeNode node) {
        if (node == null)
            return;
        ans.add(node.val);
        method2(node.left);
        method2(node.right);
    }
}
