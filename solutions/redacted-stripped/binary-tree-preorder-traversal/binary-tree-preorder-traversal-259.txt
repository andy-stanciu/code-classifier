class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList();
        return get(root, list);
    }

    public List<Integer> get(TreeNode root, List<Integer> list) {
        if (root == null)
            return list;
        list.add(root.val);
        get(root.left, list);
        get(root.right, list);
        return list;
    }
}
