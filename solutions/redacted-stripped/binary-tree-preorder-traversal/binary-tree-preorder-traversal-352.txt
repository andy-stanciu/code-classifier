class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        if (root == null) {
            return ls;
        }
        method2(root, ls);
        return ls;
    }

    void method2(TreeNode root, List<Integer> ls) {
        if (root == null) {
            return;
        }
        ls.add(root.val);
        method2(root.left, ls);
        method2(root.right, ls);
    }
}
