class Solution {

    public void method1(List<Integer> res, TreeNode root) {
        if (root == null)
            return;
        res.add(root.val);
        method1(res, root.left);
        method1(res, root.right);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method1(res, root);
        return res;
    }
}
