class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    void method2(TreeNode root, ArrayList<Integer> result) {
        if (root == null) {
            return;
        }
        result.add(root.val);
        method2(root.left, result);
        method2(root.right, result);
        return;
    }
}
