class Solution {

    List<Integer> res = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return res;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        res.add(root.val);
        method2(root.left);
        method2(root.right);
        return;
    }
}
