class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> integerList = new LinkedList<>();
        integerList = method1(root, integerList);
        return integerList;
    }

    private List<Integer> method1(TreeNode root, List<Integer> integerList) {
        if (root == null)
            return integerList;
        integerList.add(root.val);
        method1(root.left, integerList);
        method1(root.right, integerList);
        return integerList;
    }
}
