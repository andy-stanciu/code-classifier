class Solution {

    List<Integer> list;

    public void saveNode(TreeNode node) {
        if (node == null)
            return;
        list.add(node.val);
        saveNode(node.left);
        saveNode(node.right);
    }

    public List<Integer> method1(TreeNode root) {
        list = new ArrayList<>();
        saveNode(root);
        return list;
    }
}
