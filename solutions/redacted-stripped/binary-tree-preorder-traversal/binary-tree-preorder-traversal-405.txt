class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null)
            return list;
        list.add(root.val);
        if (root.left != null)
            list.addAll(method1(root.left));
        if (root.right != null)
            list.addAll(method1(root.right));
        return list;
    }
}
