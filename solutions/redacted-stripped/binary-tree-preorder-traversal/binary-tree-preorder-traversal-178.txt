class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        method2(root, l);
        return l;
    }

    public void method2(TreeNode root, List<Integer> l) {
        if (root != null) {
            l.add(root.val);
            method2(root.left, l);
            method2(root.right, l);
        }
    }
}
