class Solution {

    public List<Integer> method1(TreeNode root) {
        ls = new ArrayList<>();
        preOt(root);
        return ls;
    }

    public List<Integer> ls;

    public void preOt(TreeNode root) {
        if (root == null)
            return;
        ls.add(root.val);
        preOt(root.left);
        preOt(root.right);
    }
}
