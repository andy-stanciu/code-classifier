class Solution {

    public List<Integer> Answer = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return Answer;
        Travers(root);
        return Answer;
    }

    public void Travers(TreeNode root) {
        Answer.add(root.val);
        if (root.left != null)
            Travers(root.left);
        if (root.right != null)
            Travers(root.right);
    }
}
