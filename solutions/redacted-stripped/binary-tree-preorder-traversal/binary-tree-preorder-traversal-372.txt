class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<Integer>();
        return method1(root, list);
    }

    List<Integer> method1(TreeNode root, List<Integer> list) {
        if (root == null)
            return list;
        list.add(root.val);
        method1(root.left, list);
        method1(root.right, list);
        return list;
    }
}
