class Solution {

    List<Integer> pre = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        addList(root);
        return pre;
    }

    void addList(TreeNode root) {
        if (root == null)
            return;
        pre.add(root.val);
        addList(root.left);
        addList(root.right);
    }
}
