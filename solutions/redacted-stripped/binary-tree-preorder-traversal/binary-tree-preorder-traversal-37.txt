class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> L = new ArrayList();
        Stack<TreeNode> S = new Stack();
        while (root != null || !S.isEmpty()) {
            if (root != null) {
                S.push(root);
                L.add(root.val);
                root = root.left;
            } else {
                root = S.pop();
                root = root.right;
            }
        }
        return L;
    }
}
