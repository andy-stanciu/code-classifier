class Solution {

    public void method1(TreeNode root, List<Integer> pre) {
        if (root == null) {
            return;
        }
        pre.add(root.val);
        method1(root.left, pre);
        method1(root.right, pre);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        method1(root, pre);
        return pre;
    }
}
