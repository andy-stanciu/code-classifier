class Solution {

    public List<Integer> ans;

    public List<Integer> method1(TreeNode root) {
        ans = new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        if (root == null) {
            return ans;
        }
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode r = st.pop();
            if (r.right != null) {
                st.push(r.right);
            }
            if (r.left != null) {
                st.push(r.left);
            }
            ans.add(r.val);
        }
        return ans;
    }
}
