class Solution {

    List<Integer> l = new ArrayList<Integer>();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        l.add(root.val);
        method1(root.left);
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return l;
    }
}
