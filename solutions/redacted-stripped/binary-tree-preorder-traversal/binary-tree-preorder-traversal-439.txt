class Solution {

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        List<Integer> ans = new ArrayList<>();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode top = st.pop();
            ans.add(top.val);
            if (top.right != null) {
                st.push(top.right);
            }
            if (top.left != null) {
                st.push(top.left);
            }
        }
        return ans;
    }
}
