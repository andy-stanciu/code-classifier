class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        Stack<TreeNode> st = new Stack<>();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode curntroot = st.pop();
            ans.add(curntroot.val);
            if (curntroot.right != null)
                st.push(curntroot.right);
            if (curntroot.left != null)
                st.push(curntroot.left);
        }
        return ans;
    }
}
