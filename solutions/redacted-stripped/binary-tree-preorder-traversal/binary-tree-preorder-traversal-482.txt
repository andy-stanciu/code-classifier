class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> a = new ArrayList<>();
        method2(root, a);
        return a;
    }

    public static void method2(TreeNode root, List a) {
        if (root == null)
            return;
        a.add(root.val);
        method2(root.left, a);
        method2(root.right, a);
    }
}
