class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list1 = new ArrayList<>();
        method2(root, list1);
        return list1;
    }

    private void method2(TreeNode root, List<Integer> list1) {
        if (root == null) {
            return;
        }
        list1.add(root.val);
        method2(root.left, list1);
        method2(root.right, list1);
    }
}
