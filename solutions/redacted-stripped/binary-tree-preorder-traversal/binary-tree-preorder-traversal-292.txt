class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        pre(root, result);
        return result;
    }

    public void pre(TreeNode root, List<Integer> result) {
        if (root == null) {
            return;
        }
        result.add(root.val);
        pre(root.left, result);
        pre(root.right, result);
    }
}
