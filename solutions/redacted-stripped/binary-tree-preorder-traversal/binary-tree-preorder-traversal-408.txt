class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<Integer>();
        if (root != null) {
            list.add(root.val);
            if (root.left != null)
                list.addAll(method1(root.left));
            if (root.right != null)
                list.addAll(method1(root.right));
        }
        return list;
    }
}
