class Solution {

    static List<Integer> method1(TreeNode root, List<Integer> l) {
        if (root == null)
            return l;
        l.add(root.val);
        method1(root.left, l);
        method1(root.right, l);
        return l;
    }

    public List<Integer> method2(TreeNode root) {
        return method1(root, new ArrayList<>());
    }
}
