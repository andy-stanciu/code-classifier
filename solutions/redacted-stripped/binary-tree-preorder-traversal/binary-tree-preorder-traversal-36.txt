class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> pre = new ArrayList<>();
        if (root == null)
            return pre;
        traverse(root, pre);
        return pre;
    }

    public void traverse(TreeNode root, ArrayList<Integer> pre) {
        pre.add(root.val);
        if (root.left != null)
            traverse(root.left, pre);
        if (root.right != null)
            traverse(root.right, pre);
    }
}
