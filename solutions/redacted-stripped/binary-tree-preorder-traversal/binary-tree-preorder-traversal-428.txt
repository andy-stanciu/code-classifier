class Solution {

    ArrayList<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        preor(root);
        return list;
    }

    public void preor(TreeNode root) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        method1(root.left);
        method1(root.right);
    }
}
