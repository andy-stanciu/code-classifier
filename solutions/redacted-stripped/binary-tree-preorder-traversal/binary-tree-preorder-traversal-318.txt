class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        preHelper(root, list);
        return list;
    }

    public static void preHelper(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        preHelper(root.left, list);
        preHelper(root.right, list);
    }
}
