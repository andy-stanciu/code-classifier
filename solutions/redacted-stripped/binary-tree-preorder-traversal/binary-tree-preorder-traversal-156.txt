class Solution {

    ArrayList<Integer> ans = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        pre(root);
        return ans;
    }

    public void pre(TreeNode root) {
        if (root == null) {
            return;
        }
        ans.add(root.val);
        method1(root.left);
        method1(root.right);
    }
}
