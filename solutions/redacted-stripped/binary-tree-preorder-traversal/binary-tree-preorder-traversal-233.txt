class Solution {

    public List<Integer> method1(TreeNode root) {
        return method2(root);
    }

    public List<Integer> method2(TreeNode node) {
        List<Integer> ans = new ArrayList<>();
        while (node != null) {
            if (node.left == null) {
                ans.add(node.val);
                node = node.right;
            } else {
                TreeNode nodep1 = node.left;
                while (nodep1.right != null && nodep1.right != node) {
                    nodep1 = nodep1.right;
                }
                if (nodep1.right == null) {
                    ans.add(node.val);
                    nodep1.right = node;
                    node = node.left;
                } else {
                    nodep1.right = null;
                    node = node.right;
                }
            }
        }
        return ans;
    }
}
