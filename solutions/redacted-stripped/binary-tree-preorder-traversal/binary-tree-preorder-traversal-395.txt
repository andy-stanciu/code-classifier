class Solution {

    List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            return result;
        }
        Stack<TreeNode> s = new Stack<>();
        TreeNode ptr = root;
        while (ptr != null || !s.isEmpty()) {
            while (ptr != null) {
                result.add(ptr.val);
                s.push(ptr);
                ptr = ptr.left;
            }
            ptr = s.pop();
            ptr = ptr.right;
        }
        return result;
    }
}
