class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        traversepre(root, ans);
        return ans;
    }

    static void traversepre(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        ans.add(root.val);
        traversepre(root.left, ans);
        traversepre(root.right, ans);
    }
}
