class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> pretraverse = new ArrayList<>();
        if (root == null)
            return pretraverse;
        method2(pretraverse, root);
        return pretraverse;
    }

    static void method2(List pretraverse, TreeNode root) {
        if (root == null)
            return;
        pretraverse.add(root.val);
        method2(pretraverse, root.left);
        method2(pretraverse, root.right);
    }
}
