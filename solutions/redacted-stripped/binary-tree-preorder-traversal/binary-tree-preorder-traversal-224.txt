class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null)
            return list;
        traverselHelper(root, list);
        return list;
    }

    public void traverselHelper(TreeNode root, List<Integer> list) {
        list.add(root.val);
        if (root.left != null) {
            traverselHelper(root.left, list);
        }
        if (root.right != null) {
            traverselHelper(root.right, list);
        }
    }
}
