class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> al = new ArrayList<>();
        method2(root, al);
        return al;
    }

    private void method2(TreeNode node, List<Integer> al) {
        if (node == null) {
            return;
        }
        al.add(node.val);
        method2(node.left, al);
        method2(node.right, al);
    }
}
