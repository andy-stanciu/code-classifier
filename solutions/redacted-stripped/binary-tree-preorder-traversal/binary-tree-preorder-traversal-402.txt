class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        return method2(root, list);
    }

    private List<Integer> method2(TreeNode curr, List<Integer> list) {
        if (curr == null) {
            return list;
        }
        list.add(curr.val);
        if (curr.left != null) {
            method2(curr.left, list);
        }
        if (curr.right != null) {
            method2(curr.right, list);
        }
        return list;
    }
}
