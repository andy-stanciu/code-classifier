class Solution {

    public List<Integer> method1(TreeNode root) {
        TreeNode cur = root;
        List<Integer> ans = new ArrayList<>();
        while (cur != null) {
            if (cur.left == null) {
                TreeNode temp = cur;
                ans.add(cur.val);
                cur = cur.right;
            } else {
                TreeNode right_most_node = cur.left;
                while (right_most_node.right != null && right_most_node.right != cur) right_most_node = right_most_node.right;
                if (right_most_node.right == null) {
                    ans.add(cur.val);
                    right_most_node.right = cur;
                    cur = cur.left;
                } else {
                    cur = cur.right;
                    right_most_node.right = null;
                }
            }
        }
        return ans;
    }
}
