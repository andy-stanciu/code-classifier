class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> alist = new ArrayList<>();
        method2(root, alist);
        return alist;
    }

    public void method2(TreeNode root, List<Integer> alist) {
        if (root == null) {
            return;
        }
        alist.add(root.val);
        method2(root.left, alist);
        method2(root.right, alist);
    }
}
