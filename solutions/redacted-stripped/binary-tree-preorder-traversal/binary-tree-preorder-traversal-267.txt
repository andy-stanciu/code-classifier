class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        Stack<TreeNode> s = new Stack<>();
        s.push(root);
        if (!(root == null)) {
            while (!s.isEmpty()) {
                TreeNode curr = s.pop();
                ans.add(curr.val);
                if (curr.right != null) {
                    s.push(curr.right);
                }
                if (curr.left != null) {
                    s.push(curr.left);
                }
            }
        }
        return ans;
    }
}
