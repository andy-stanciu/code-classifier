class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        TreeNode curr = root;
        while (curr != null) {
            if (curr.left == null) {
                var1.add(curr.val);
                curr = curr.right;
            } else {
                TreeNode leftNode = curr.left;
                while (leftNode.right != null && leftNode.right != curr) {
                    leftNode = leftNode.right;
                }
                if (leftNode.right == null) {
                    leftNode.right = curr;
                    var1.add(curr.val);
                    curr = curr.left;
                } else {
                    leftNode.right = null;
                    curr = curr.right;
                }
            }
        }
        return var1;
    }
}
