class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> x = new ArrayList<>();
        if (root == null)
            return x;
        Stack<TreeNode> st = new Stack<>();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode node = st.pop();
            x.add(node.val);
            if (node.right != null)
                st.push(node.right);
            if (node.left != null)
                st.push(node.left);
        }
        return x;
    }
}
