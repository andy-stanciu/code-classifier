class Solution {

    public void method1(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        ans.add(root.val);
        method1(root.left, ans);
        method1(root.right, ans);
        return;
    }

    public List<Integer> method2(TreeNode root) {
        if (root == null)
            return new ArrayList<>(Arrays.asList());
        List<Integer> ans = new ArrayList<>();
        ans.add(root.val);
        method1(root.left, ans);
        method1(root.right, ans);
        return ans;
    }
}
