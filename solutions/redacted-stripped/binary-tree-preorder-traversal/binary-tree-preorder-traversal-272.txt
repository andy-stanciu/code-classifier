class Solution {

    List<Integer> result = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        dfs(root);
        return result;
    }

    public void dfs(TreeNode node) {
        if (node == null) {
            return;
        }
        result.add(node.val);
        dfs(node.left);
        dfs(node.right);
    }
}
