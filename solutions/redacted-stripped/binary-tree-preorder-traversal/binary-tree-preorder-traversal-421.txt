class Solution {

    public static void method1(TreeNode root, ArrayList<Integer> result) {
        if (root == null) {
            return;
        }
        result.add(root.val);
        method1(root.left, result);
        method1(root.right, result);
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> result = new ArrayList<>();
        method1(root, result);
        return result;
    }
}
