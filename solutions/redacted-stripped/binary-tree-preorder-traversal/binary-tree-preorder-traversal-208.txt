class Solution {

    public List<Integer> method1(TreeNode root) {
        final var list = new ArrayList<Integer>();
        preTra(root, list);
        return list;
    }

    void preTra(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        list.add(node.val);
        preTra(node.left, list);
        preTra(node.right, list);
    }
}
