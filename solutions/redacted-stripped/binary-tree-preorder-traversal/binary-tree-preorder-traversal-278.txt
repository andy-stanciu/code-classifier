class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        return order(root, ans);
    }

    public List<Integer> order(TreeNode root, List<Integer> ans) {
        if (root == null)
            return ans;
        ans.add(root.val);
        order(root.left, ans);
        order(root.right, ans);
        return ans;
    }
}
