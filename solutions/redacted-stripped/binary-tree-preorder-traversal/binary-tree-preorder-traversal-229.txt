class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> arr = new ArrayList<>();
        pre(root, arr);
        return arr;
    }

    public void pre(TreeNode root, List<Integer> arr) {
        if (root != null) {
            arr.add(root.val);
            pre(root.left, arr);
            pre(root.right, arr);
        }
    }
}
