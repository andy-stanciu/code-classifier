class Solution {

    public List<Integer> method1(TreeNode root) {
        LinkedList<Integer> l = new LinkedList<Integer>();
        po(root, l);
        return l;
    }

    public static void po(TreeNode root, LinkedList<Integer> l) {
        if (root == null)
            return;
        l.add(root.val);
        po(root.left, l);
        po(root.right, l);
    }
}
