class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> arr = new ArrayList<Integer>();
        method2(root, arr);
        return arr;
    }

    static void method2(TreeNode root, List<Integer> p) {
        if (root == null)
            return;
        p.add(root.val);
        method2(root.left, p);
        method2(root.right, p);
    }
}
