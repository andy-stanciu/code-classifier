class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> al = new LinkedList<>();
        Stack<TreeNode> stk = new Stack<>();
        if (root == null)
            return al;
        stk.push(root);
        while (!stk.isEmpty()) {
            TreeNode temp = stk.pop();
            al.add(temp.val);
            if (temp.right != null) {
                stk.add(temp.right);
            }
            if (temp.left != null) {
                stk.add(temp.left);
            }
        }
        return al;
    }
}
