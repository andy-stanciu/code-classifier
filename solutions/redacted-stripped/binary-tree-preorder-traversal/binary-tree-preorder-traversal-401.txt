class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method1(root, result);
        return result;
    }

    public void method1(TreeNode root, List<Integer> arr) {
        if (root == null) {
            return;
        }
        arr.add(root.val);
        method1(root.left, arr);
        method1(root.right, arr);
    }
}
