class Solution {

    public void method1(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        res.add(root.val);
        method1(root.left, res);
        method1(root.right, res);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method1(root, res);
        return res;
    }
}
