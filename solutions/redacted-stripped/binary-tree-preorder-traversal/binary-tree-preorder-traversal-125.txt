class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> L = new ArrayList<>();
        method2(root, L);
        return L;
    }

    private void method2(TreeNode node, List<Integer> L) {
        if (node == null) {
            return;
        }
        L.add(node.val);
        method2(node.left, L);
        method2(node.right, L);
    }
}
