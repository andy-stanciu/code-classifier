class Solution {

    List<Integer> list;

    public List<Integer> method1(TreeNode root) {
        list = new ArrayList<>();
        method2(root);
        return list;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        list.add(root.val);
        method2(root.left);
        method2(root.right);
    }
}
