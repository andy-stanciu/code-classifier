class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans;
    }

    public static void method2(TreeNode root, List<Integer> ans) {
        if (root == null) {
            return;
        }
        ans.add(root.val);
        method2(root.left, ans);
        method2(root.right, ans);
    }
}
