class Solution {

    private List<Integer> l = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        pretravel(root);
        return l;
    }

    private void pretravel(TreeNode root) {
        if (root == null) {
            return;
        }
        l.add(root.val);
        pretravel(root.left);
        pretravel(root.right);
    }
}
