class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        method2(pre, root);
        return pre;
    }

    static void method2(List<Integer> pre, TreeNode node) {
        if (node == null)
            return;
        TreeNode temp = node;
        pre.add(temp.val);
        method2(pre, temp.left);
        method2(pre, temp.right);
    }
}
