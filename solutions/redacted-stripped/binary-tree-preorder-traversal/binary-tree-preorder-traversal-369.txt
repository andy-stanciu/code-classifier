class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ll = new ArrayList<>();
        method2(root, ll);
        return ll;
    }

    private void method2(TreeNode root, List<Integer> ll) {
        if (root == null) {
            return;
        }
        ll.add(root.val);
        method2(root.left, ll);
        method2(root.right, ll);
    }
}
