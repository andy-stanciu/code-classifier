class Solution {

    List<Integer> ans;

    public void method1(TreeNode root) {
        if (root == null)
            return;
        ans.add(root.val);
        method1(root.left);
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        ans = new ArrayList<>();
        method1(root);
        return ans;
    }
}
