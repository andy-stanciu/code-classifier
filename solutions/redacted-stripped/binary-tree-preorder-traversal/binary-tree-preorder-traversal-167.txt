class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        return method2(root, list);
    }

    public List<Integer> method2(TreeNode root, List<Integer> list) {
        if (root == null) {
            return list;
        }
        list.add(root.val);
        method2(root.left, list);
        method2(root.right, list);
        return list;
    }
}
