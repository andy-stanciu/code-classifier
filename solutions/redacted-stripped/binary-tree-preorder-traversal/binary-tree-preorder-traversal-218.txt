class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
        while (!stack.isEmpty()) {
            TreeNode var1 = stack.pop();
            res.add(var1.val);
            if (var1.right != null) {
                stack.push(var1.right);
            }
            if (var1.left != null) {
                stack.push(var1.left);
            }
        }
        return res;
    }
}
