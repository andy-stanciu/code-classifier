class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        method2(root, arr);
        return arr;
    }

    static void method2(TreeNode root, ArrayList<Integer> arr) {
        if (root == null) {
            return;
        }
        arr.add(root.val);
        method2(root.left, arr);
        method2(root.right, arr);
    }
}
