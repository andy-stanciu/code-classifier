class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> temp = new ArrayList<>();
        method2(root, temp);
        return temp;
    }

    void method2(TreeNode node, List<Integer> temp) {
        if (node == null)
            return;
        temp.add(node.val);
        method2(node.left, temp);
        method2(node.right, temp);
    }
}
