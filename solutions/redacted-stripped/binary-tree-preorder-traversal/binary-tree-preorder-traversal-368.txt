class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<Integer>();
        Stack<TreeNode> st = new Stack<TreeNode>();
        if (root == null) {
            return ans;
        }
        while (true) {
            while (root != null) {
                ans.add(root.val);
                st.push(root);
                root = root.left;
            }
            if (st.empty()) {
                break;
            }
            root = st.pop();
            root = root.right;
        }
        return ans;
    }
}
