class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int x) {
        val = x;
    }
}

public class Solution {

    public List<Integer> method1(Class1 root) {
        List<Integer> result = new ArrayList<>();
        method1(root, result);
        return result;
    }

    private void method1(Class1 root, List<Integer> result) {
        if (root == null) {
            return;
        }
        result.add(root.val);
        method1(root.left, result);
        method1(root.right, result);
    }
}
