class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> finalArr = new ArrayList<>();
        method2(finalArr, root);
        return finalArr;
    }

    public void method2(List finalArr, TreeNode node) {
        if (node == null) {
            return;
        }
        finalArr.add(node.val);
        method2(finalArr, node.left);
        method2(finalArr, node.right);
    }
}
