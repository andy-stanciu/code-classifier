class Solution {

    List<Integer> var1 = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        helper(root);
        return var1;
    }

    void helper(TreeNode root) {
        if (root == null)
            return;
        var1.add(root.val);
        helper(root.left);
        helper(root.right);
        return;
    }
}
