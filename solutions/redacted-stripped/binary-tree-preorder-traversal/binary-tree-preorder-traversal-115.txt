class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList();
        dfs(res, root);
        return res;
    }

    void dfs(List<Integer> res, TreeNode node) {
        if (node != null) {
            res.add(node.val);
            if (node.left != null) {
                dfs(res, node.left);
            }
            if (node.right != null) {
                dfs(res, node.right);
            }
        }
    }
}
