class Solution {

    public void helper(TreeNode root, ArrayList<Integer> ans) {
        if (root == null)
            return;
        ans.add(root.val);
        helper(root.left, ans);
        helper(root.right, ans);
    }

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        helper(root, ans);
        return ans;
    }
}
