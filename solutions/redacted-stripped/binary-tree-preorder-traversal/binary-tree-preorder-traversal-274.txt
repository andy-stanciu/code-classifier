class Solution {

    List<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return ans;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        ans.add(root.val);
        method2(root.left);
        method2(root.right);
    }
}
