class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root != null) {
            method1(root, result);
        }
        return result;
    }

    public void method1(TreeNode root, List<Integer> result) {
        result.add(root.val);
        if (root.left != null) {
            method1(root.left, result);
        }
        if (root.right != null) {
            method1(root.right, result);
        }
    }
}
