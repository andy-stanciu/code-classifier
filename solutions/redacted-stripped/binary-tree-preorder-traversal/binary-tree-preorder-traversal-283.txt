class Solution {

    List<Integer> res = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        pre(root);
        return res;
    }

    public void pre(TreeNode root) {
        if (root == null) {
            return;
        }
        res.add(root.val);
        method1(root.left);
        method1(root.right);
    }
}
