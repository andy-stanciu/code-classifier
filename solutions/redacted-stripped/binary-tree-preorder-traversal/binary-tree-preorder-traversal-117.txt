class Solution {

    private static List<Integer> result;

    public List<Integer> method1(TreeNode root) {
        result = new ArrayList<>();
        if (root == null)
            return result;
        result.add(root.val);
        if (root.left != null) {
            method2(root.left);
        }
        if (root.right != null) {
            method2(root.right);
        }
        return result;
    }

    private void method2(TreeNode node) {
        result.add(node.val);
        if (node.left != null) {
            method2(node.left);
        }
        if (node.right != null) {
            method2(node.right);
        }
    }
}
