class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        pre(root, res);
        return res;
    }

    void pre(TreeNode root, List<Integer> res) {
        if (root == null)
            return;
        res.add(root.val);
        pre(root.left, res);
        pre(root.right, res);
    }
}
