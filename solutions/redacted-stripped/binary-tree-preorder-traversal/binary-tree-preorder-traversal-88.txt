class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<>();
        st.add(root);
        List<Integer> ans = new ArrayList<>();
        if (root == null)
            return ans;
        while (st.size() > 0) {
            TreeNode node = st.pop();
            ans.add(node.val);
            if (node.right != null)
                st.add(node.right);
            if (node.left != null)
                st.add(node.left);
        }
        return ans;
    }
}
