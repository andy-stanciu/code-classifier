class Solution {

    private List<Integer> pre(TreeNode root, List<Integer> li) {
        if (root != null) {
            li.add(root.val);
            pre(root.left, li);
            pre(root.right, li);
        }
        return li;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> li = new ArrayList<>();
        return pre(root, li);
    }
}
