class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        return (method2(root, result));
    }

    public List<Integer> method2(TreeNode root, List<Integer> result) {
        if (root == null) {
            return result;
        }
        result.add(root.val);
        result = method2(root.left, result);
        result = method2(root.right, result);
        return result;
    }
}
