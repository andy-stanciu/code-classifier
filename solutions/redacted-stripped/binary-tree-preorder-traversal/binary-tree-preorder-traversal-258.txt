class Solution {

    List<Integer> preTraverse(TreeNode root, List<Integer> list) {
        if (root == null)
            return list;
        list.add(root.val);
        preTraverse(root.left, list);
        preTraverse(root.right, list);
        return list;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        list = preTraverse(root, list);
        return list;
    }
}
