class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        sol(root, l);
        return l;
    }

    public void sol(TreeNode t, List<Integer> l) {
        if (t == null)
            return;
        l.add(t.val);
        sol(t.left, l);
        sol(t.right, l);
    }
}
