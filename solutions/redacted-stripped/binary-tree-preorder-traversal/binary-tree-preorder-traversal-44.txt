class Solution {

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return list;
        list.add(root.val);
        if (root.left != null)
            method1(root.left);
        if (root.right != null)
            method1(root.right);
        return list;
    }
}
