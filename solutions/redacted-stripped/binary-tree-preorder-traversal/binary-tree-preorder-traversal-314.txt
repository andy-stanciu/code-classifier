class Solution {

    public ArrayList<Integer> list;

    public void order(TreeNode root) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        order(root.left);
        order(root.right);
    }

    public List<Integer> method1(TreeNode root) {
        list = new ArrayList<>();
        order(root);
        return list;
    }
}
