class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> lst = new ArrayList<>();
        TreeNode node;
        if (root == null) {
            return lst;
        }
        Stack<TreeNode> stk = new Stack<>();
        stk.add(root);
        while (!stk.isEmpty()) {
            node = stk.pop();
            lst.add(node.val);
            if (node.right != null) {
                stk.add(node.right);
            }
            if (node.left != null) {
                stk.add(node.left);
            }
        }
        return lst;
    }
}
