class Solution {

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return new LinkedList<>();
        Stack<TreeNode> st = new Stack<>();
        List<Integer> res = new LinkedList<>();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode temp = st.pop();
            res.add(temp.val);
            if (temp.right != null)
                st.push(temp.right);
            if (temp.left != null)
                st.push(temp.left);
        }
        return res;
    }
}
