class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        if (root == null)
            return res;
        Stack<TreeNode> st = new Stack<>();
        TreeNode temp = root;
        while (!st.isEmpty() || temp != null) {
            if (temp != null) {
                res.add(temp.val);
                st.push(temp);
                temp = temp.left;
            } else {
                TreeNode node = st.pop();
                temp = node.right;
            }
        }
        return res;
    }
}
