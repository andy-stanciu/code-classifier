class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> var1 = new ArrayList<>();
        var1(root, var1);
        return var1;
    }

    void var1(TreeNode curr, ArrayList<Integer> var1) {
        if (curr == null) {
            return;
        }
        var1.add(curr.val);
        var1(curr.left, var1);
        var1(curr.right, var1);
    }
}
