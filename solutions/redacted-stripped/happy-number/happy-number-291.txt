class Solution {

    public boolean method1(int n) {
        return method2(n);
    }

    public boolean method2(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = SQR(slow);
            fast = SQR(SQR(fast));
        } while (slow != fast);
        return slow == 1;
    }

    public int SQR(int n) {
        int val = 0;
        while (n > 0) {
            val += (n % 10) * (n % 10);
            n /= 10;
        }
        return val;
    }
}
