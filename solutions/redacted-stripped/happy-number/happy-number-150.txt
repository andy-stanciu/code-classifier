class Solution {

    public boolean method1(int n) {
        int iter = 0;
        while (n != 1 && iter < 10) {
            int num = 0;
            while (n > 0) {
                int digit = n % 10;
                num += digit * digit;
                n /= 10;
            }
            n = num;
            iter += 1;
        }
        if (n == 1)
            return true;
        else
            return false;
    }
}
