class Solution {

    public boolean method1(int n) {
        int x = n;
        HashSet<Integer> visit = new HashSet<>();
        while (!visit.contains(x)) {
            visit.add(x);
            int sum = 0;
            while (x > 0) {
                sum += Math.pow((x % 10), 2);
                x /= 10;
            }
            x = sum;
            if (x == 1) {
                return true;
            }
        }
        return false;
    }
}
