class Solution {

    public boolean method1(int n) {
        int sum = n;
        Set<Integer> sumSet = new HashSet<>();
        while (sum != 1) {
            sum = this.getSumOfSquares(sum);
            if (sumSet.contains(sum))
                return false;
            sumSet.add(sum);
        }
        return true;
    }

    public int getSumOfSquares(int n) {
        int divident = 10;
        int sum = 0;
        while (n >= 1) {
            int d = n % divident;
            sum += d * d;
            n = n / divident;
        }
        return sum;
    }
}
