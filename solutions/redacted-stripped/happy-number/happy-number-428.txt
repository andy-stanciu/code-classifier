class Solution {

    public boolean method1(int n) {
        if (n < 10 && n != 1 && n != 7)
            return false;
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            n = n / 10;
            sum += digit * digit;
        }
        if (sum == 1)
            return true;
        else
            return method1(sum);
    }
}
