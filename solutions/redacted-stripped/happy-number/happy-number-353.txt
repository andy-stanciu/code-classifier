class Solution {

    public boolean method1(int n) {
        int sum = 0;
        do {
            sum = 0;
            while (n > 0) {
                int digit = n % 10;
                sum += digit * digit;
                n /= 10;
            }
            n = sum;
        } while (sum >= 10);
        return sum == 1 || sum == 7 ? true : false;
    }
}
