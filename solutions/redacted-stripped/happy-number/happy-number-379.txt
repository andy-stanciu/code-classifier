class Solution {

    public boolean method1(int n) {
        while (n > 6) {
            int sq = 0;
            while (n > 0) {
                int x = n % 10;
                sq = sq + (x * x);
                n = n / 10;
            }
            if (sq == 1) {
                return true;
            }
            n = sq;
        }
        return n == 1;
    }
}
