class Solution {

    public boolean method1(int n) {
        int slow = method2(n);
        int fast = method2(method2(n));
        while (slow != fast) {
            if (fast == 1)
                return true;
            slow = method2(slow);
            fast = method2(method2(fast));
        }
        return slow == 1;
    }

    private int method2(int n) {
        int output = 0;
        while (n > 0) {
            int digit = n % 10;
            output += digit * digit;
            n = n / 10;
        }
        return output;
    }
}
