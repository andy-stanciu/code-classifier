class Solution {

    public boolean method1(int n) {
        int slowRunner = n;
        int fastRunner = getSum(n);
        while (fastRunner != 1 && slowRunner != fastRunner) {
            fastRunner = getSum(getSum(fastRunner));
            slowRunner = getSum(slowRunner);
        }
        return fastRunner == 1;
    }

    private int getSum(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += (digit * digit);
            n = n / 10;
        }
        return sum;
    }
}
