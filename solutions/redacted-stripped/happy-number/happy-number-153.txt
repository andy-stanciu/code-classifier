class Solution {

    public boolean method1(int n) {
        while (n != 1 && n != 4) {
            int sum = 0;
            while (n > 0) {
                int digit = n % 10;
                sum = sum + (digit * digit);
                n /= 10;
            }
            n = sum;
        }
        return n == 1;
    }
}
