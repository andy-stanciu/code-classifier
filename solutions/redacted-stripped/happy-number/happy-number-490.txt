class Solution {

    static int sum(int x) {
        int res = 0;
        while (x != 0) {
            int a = x % 10;
            res += a * a;
            x /= 10;
        }
        return res;
    }

    public boolean method1(int n) {
        int slow = n;
        int fast = sum(n);
        while (slow != fast) {
            if (fast == 1)
                return true;
            slow = sum(slow);
            fast = sum(sum(fast));
        }
        return slow == 1;
    }
}
