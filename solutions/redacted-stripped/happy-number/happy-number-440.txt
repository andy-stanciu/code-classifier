class Solution {

    public boolean method1(int n) {
        List<Integer> numList = String.valueOf(n).chars().map(Character::getNumericValue).boxed().collect(Collectors.toList());
        int sum = 0;
        for (Integer var1 : numList) {
            sum += var1 * var1;
        }
        if (sum == 1) {
            return true;
        } else if ((sum > 1 && sum <= 9) && (sum % 2 == 0)) {
            return false;
        } else {
            return method1(sum);
        }
    }
}
