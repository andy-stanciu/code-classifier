class Solution {

    public boolean method1(int n) {
        HashSet<Integer> var1 = new HashSet<Integer>();
        int c = 0, m = 0;
        if (n == 1 || n == 7) {
            return true;
        }
        if (var1.contains(n) || n < 10) {
            return false;
        }
        var1.add(n);
        while (n != 0) {
            c = n % 10;
            m += c * c;
            n = n / 10;
        }
        n = m;
        return method1(n);
    }
}
