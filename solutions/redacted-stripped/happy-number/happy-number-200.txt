class Solution {

    public boolean method1(int n) {
        while (n != 1 && n != 4) {
            n = square(n);
        }
        if (n == 1) {
            return true;
        }
        return false;
    }

    static int square(int n) {
        int sum = 0;
        while (n != 0) {
            int y = n % 10;
            sum = sum + (y * y);
            n = n / 10;
        }
        return sum;
    }
}
