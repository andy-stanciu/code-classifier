class Solution {

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        while (n != 1 && !set.contains(n)) {
            set.add(n);
            n = helper(n);
        }
        return n == 1;
    }

    private int helper(int num) {
        int res = 0;
        while (num > 0) {
            int remainder = num % 10;
            res = res + remainder * remainder;
            num = num / 10;
        }
        return res;
    }
}
