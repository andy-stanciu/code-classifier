class Solution {

    public boolean method1(int n) {
        while (n > 4 || n == 1) {
            int s = 0;
            while (n != 0) {
                s += (n % 10) * (n % 10);
                n /= 10;
            }
            n = s;
            if (s == 1)
                return true;
        }
        return false;
    }
}
