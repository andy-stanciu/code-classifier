class Solution {

    public boolean method1(int n) {
        int s = 0;
        while (n > 0) {
            int a = n % 10;
            s += a * a;
            ;
            n /= 10;
        }
        if (s == 1) {
            return true;
        } else if (s == 4) {
            return false;
        }
        return method1(s);
    }
}
