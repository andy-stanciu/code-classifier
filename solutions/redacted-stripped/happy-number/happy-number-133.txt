public class Class1 {

    public int sum(int n) {
        int sum = 0;
        while (n != 0) {
            int dig = n % 10;
            sum = sum + (dig * dig);
            n = n / 10;
        }
        return sum;
    }
}

class Solution {

    public boolean method1(int n) {
        Class1 hpy = new Class1();
        while (n >= 10 || n == 7) {
            int temp = hpy.sum(n);
            n = temp;
        }
        return n == 1;
    }
}
