class Solution {

    public boolean method1(int n) {
        int sum = 0;
        while (n > 9) {
            sum = 0;
            while (n > 0) {
                int d = n % 10;
                sum += d * d;
                n /= 10;
            }
            n = sum;
        }
        return n == 1 || n == 7 ? true : false;
    }
}
