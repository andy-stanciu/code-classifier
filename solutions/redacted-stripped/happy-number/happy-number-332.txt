class Solution {

    public boolean method1(int n) {
        Set<Integer> seenNums = new HashSet<>();
        int c = n;
        System.out.println("c: " + c);
        while (c != 1 && !seenNums.contains(c)) {
            seenNums.add(c);
            c = calculation(c);
        }
        return c == 1;
    }

    int calculation(int n) {
        char[] nums = String.valueOf(n).toCharArray();
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            int t = Character.getNumericValue(nums[i]);
            int temp = t * t;
            count = count + temp;
        }
        return count;
    }
}
