class Solution {

    public boolean method1(int n) {
        while (n != 1 && n != 4) {
            n = findingPower(n);
        }
        return n == 1;
    }

    private int findingPower(int value) {
        int sum = 0;
        while (value > 0) {
            int digit = value % 10;
            sum = sum + (int) Math.pow(digit, 2);
            value = value / 10;
        }
        return sum;
    }
}
