class Solution {

    public boolean method1(int n) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        while (n != 1) {
            int sum = 0;
            int current = n;
            while (current > 0) {
                int digit = current % 10;
                sum += digit * digit;
                current /= 10;
            }
            if (mp.containsKey(sum))
                return false;
            else {
                mp.put(sum, 1);
            }
            n = sum;
        }
        return true;
    }
}
