class Solution {

    public boolean method1(int n) {
        int sum = 0;
        do {
            while (n > 0) {
                sum += Math.pow(n % 10, 2);
                n = n / 10;
            }
            n = sum;
            sum = 0;
        } while (n >= 10);
        if (n == 7)
            return true;
        return (n % 10 == 1);
    }
}
