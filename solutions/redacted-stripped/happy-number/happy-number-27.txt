class Solution {

    public boolean method1(int n) {
        HashSet<Integer> seen = new HashSet<>();
        while (n != 1 && !seen.contains(n)) {
            seen.add(n);
            int totalsum = 0;
            while (n > 0) {
                int digit = n % 10;
                n = n / 10;
                totalsum += digit * digit;
            }
            n = totalsum;
        }
        return n == 1;
    }
}
