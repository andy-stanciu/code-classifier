class Solution {

    public boolean method1(int n) {
        int d, sum = 0;
        int num = n;
        while (num != 1 && num != 4) {
            while (num > 0) {
                d = num % 10;
                sum = sum + (d * d);
                num /= 10;
            }
            num = sum;
            sum = 0;
        }
        if (num == 1)
            return true;
        else
            return false;
    }
}
