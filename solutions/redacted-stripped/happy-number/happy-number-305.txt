class Solution {

    public boolean method1(int n) {
        if (n < 10) {
            if (n == 1 || n == 7) {
                return true;
            }
            return false;
        }
        return method1(sumsquare(n));
    }

    public int sumsquare(int n) {
        int sum = 0;
        while (n > 0) {
            int rem = n % 10;
            sum += rem * rem;
            n /= 10;
        }
        return sum;
    }
}
