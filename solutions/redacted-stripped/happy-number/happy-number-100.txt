class Solution {

    public boolean method1(int n) {
        if (n == 1)
            return true;
        int p, q, np = n, nq = n;
        while (np != 1 || nq != 1) {
            p = sumSqD(sumSqD(np));
            q = sumSqD(nq);
            if (p == 1 || q == 1)
                return true;
            if (p == q)
                break;
            np = p;
            nq = q;
        }
        return false;
    }

    private int sumSqD(int n) {
        int sum = 0;
        while (n != 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }
}
