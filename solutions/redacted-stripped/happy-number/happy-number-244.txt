class Solution {

    public boolean method1(int n) {
        int sum = 0;
        int digit;
        if (n == 1)
            return true;
        while (true) {
            digit = n % 10;
            sum += digit * digit;
            n /= 10;
            if (sum == 1 && n == 0)
                return true;
            if (n == 0) {
                if (sum == 37)
                    break;
                n = sum;
                sum = 0;
            }
        }
        return false;
    }
}
