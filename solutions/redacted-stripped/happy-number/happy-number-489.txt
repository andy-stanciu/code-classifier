class Solution {

    public boolean method1(int inputNum) {
        int fast = inputNum;
        int slow = inputNum;
        do {
            fast = getSumOfAllDigitsSquare(getSumOfAllDigitsSquare(fast));
            slow = getSumOfAllDigitsSquare(slow);
            if (slow == 1 || fast == 1)
                return true;
        } while (fast != slow);
        return false;
    }

    public int getSumOfAllDigitsSquare(int inputNum) {
        int sum = 0;
        while (inputNum > 0) {
            int digit = inputNum % 10;
            sum = sum + (digit * digit);
            inputNum = inputNum / 10;
        }
        return sum;
    }
}
