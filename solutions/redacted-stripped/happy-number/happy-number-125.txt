class Solution {

    public boolean method1(int n) {
        if (n == 1 || n == 7)
            return true;
        int sum = 0;
        if (n < 10 && n != 1)
            return false;
        while (n > 0) {
            sum = sum + (n % 10) * (n % 10);
            n /= 10;
        }
        return method1(sum);
    }
}
