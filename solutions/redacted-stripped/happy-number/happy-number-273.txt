class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = getNext(n);
        while (fast != 1 && slow != fast) {
            slow = getNext(slow);
            fast = getNext(getNext(fast));
        }
        return fast == 1;
    }

    private int getNext(int param1) {
        int totalSum = 0;
        while (param1 > 0) {
            int digit = param1 % 10;
            totalSum += digit * digit;
            param1 /= 10;
        }
        return totalSum;
    }
}
