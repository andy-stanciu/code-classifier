class Solution {

    public boolean method1(int n) {
        if (n == 1 || n == 7) {
            return true;
        }
        if (n > 2 && n < 9) {
            return false;
        }
        int dup = n;
        int sum = 0;
        while (dup != 0) {
            int dig = dup % 10;
            sum += dig * dig;
            dup = dup / 10;
        }
        return method1(sum);
    }
}
