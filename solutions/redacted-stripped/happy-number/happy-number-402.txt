class Solution {

    public int squareNum(int n) {
        int squareSum = 0;
        while (n != 0) {
            squareSum += (n % 10) * (n % 10);
            n = n / 10;
        }
        return squareSum;
    }

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = squareNum(slow);
            fast = squareNum(squareNum(fast));
        } while (slow != fast);
        return (slow == 1);
    }
}
