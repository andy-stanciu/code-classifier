class Solution {

    public boolean method1(int n) {
        int slow = squaredSumOfDigits(n);
        int fast = squaredSumOfDigits(squaredSumOfDigits(n));
        while (fast != 1) {
            if (slow == fast) {
                return false;
            }
            slow = squaredSumOfDigits(slow);
            fast = squaredSumOfDigits(squaredSumOfDigits(fast));
        }
        return true;
    }

    int squaredSumOfDigits(int num) {
        int total = 0;
        while (num > 0) {
            int digit = num % 10;
            total += digit * digit;
            num = num / 10;
        }
        return total;
    }
}
