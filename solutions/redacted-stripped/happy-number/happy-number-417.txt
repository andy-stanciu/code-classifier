class Solution {

    public boolean method1(int n) {
        Set<Integer> s = new HashSet();
        while (true) {
            if (!s.contains(n)) {
                if (n == 1) {
                    return true;
                }
                s.add(n);
                n = square(n);
            } else {
                return false;
            }
        }
    }

    private int square(int n) {
        int ans = 0;
        while (n > 0) {
            int rem = n % 10;
            ans += rem * rem;
            n = n / 10;
        }
        return ans;
    }
}
