class Solution {

    public boolean method1(int n) {
        return method2(n);
    }

    public boolean method2(int r) {
        if (r == 1)
            return true;
        if (r > 1 && r <= 4)
            return false;
        int sum = 0;
        while (r != 0) {
            sum += (r % 10) * (r % 10);
            r /= 10;
        }
        return method2(sum);
    }
}
