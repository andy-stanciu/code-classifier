class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = findSquare(slow);
            fast = findSquare(findSquare(fast));
        } while (slow != fast);
        if (fast == 1) {
            return true;
        }
        return false;
    }

    public int findSquare(int a) {
        int ans = 0;
        while (a > 0) {
            int rem = a % 10;
            ans += rem * rem;
            a = a / 10;
        }
        return ans;
    }
}
