class Solution {

    int Sum(int n) {
        for (int i = n; i > 0; i = i / 10) {
            int digit = i % 10;
            sum += (digit * digit);
        }
        return sum;
    }

    public boolean method1(int n) {
        if (n == 1)
            return true;
        while (n != 1) {
            n = Sum(n);
            if (n < 10 && n > 1 && n != 7)
                return false;
        }
        return true;
    }
}
