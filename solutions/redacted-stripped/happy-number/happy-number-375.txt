class Solution {

    public boolean method1(int n) {
        Map<Integer, Integer> map = new HashMap<>();
        int curr = multDigits(n);
        while (!map.containsKey(curr)) {
            map.put(curr, 1);
            curr = multDigits(curr);
            if (curr == 1) {
                return true;
            }
        }
        return false;
    }

    public static int multDigits(int n) {
        int temp = 0;
        while (n > 0) {
            int rem = n % 10;
            temp += rem * rem;
            n = n / 10;
        }
        return temp;
    }
}
