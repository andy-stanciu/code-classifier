class Solution {

    public boolean method1(int n) {
        Set<Integer> temp = new HashSet<>();
        while (true) {
            int digitSquare = squareOfDigits(n);
            if (digitSquare == 1)
                return true;
            if (temp.contains(digitSquare)) {
                return false;
            } else {
                temp.add(digitSquare);
            }
            n = digitSquare;
        }
    }

    int squareOfDigits(int n) {
        int squares = 0;
        while (n != 0) {
            int num = n % 10;
            squares += num * num;
            n = n / 10;
        }
        return squares;
    }
}
