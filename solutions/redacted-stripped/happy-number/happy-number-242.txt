class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = getSum(slow);
            fast = getSum(getSum(fast));
        } while (fast != slow);
        if (slow == 1) {
            return true;
        }
        return false;
    }

    int getSum(int n) {
        int res = 0;
        while (n > 0) {
            int d = n % 10;
            res += (d * d);
            n = n / 10;
        }
        return res;
    }
}
