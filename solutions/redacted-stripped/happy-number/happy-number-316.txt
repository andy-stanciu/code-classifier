class Solution {

    public boolean method1(int n) {
        int temp = n, rem, sum;
        while (temp != 4) {
            sum = 0;
            while (temp > 0) {
                rem = temp % 10;
                sum += Math.pow(rem, 2);
                temp /= 10;
            }
            if (sum == 1)
                return true;
            temp = sum;
        }
        return false;
    }
}
