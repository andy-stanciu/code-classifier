class Solution {

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        while (true) {
            int sum = 0;
            while (n != 0) {
                int digit = n % 10;
                sum += digit * digit;
                n /= 10;
            }
            if (sum == 1)
                return true;
            n = sum;
            if (set.contains(n))
                return false;
            set.add(n);
        }
    }
}
