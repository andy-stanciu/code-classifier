class Solution {

    public boolean method1(int n) {
        int sum = 0;
        while (n > 9) {
            int lastDigit = n % 10;
            sum += lastDigit * lastDigit;
            n = n / 10;
        }
        sum += n * n;
        if (sum < 10) {
            if (sum == 1 || sum == 7)
                return true;
            else
                return false;
        }
        return method1(sum);
    }
}
