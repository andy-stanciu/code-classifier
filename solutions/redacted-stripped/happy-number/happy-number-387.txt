class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = getnext(n);
        while (fast != 1 && slow != fast) {
            slow = getnext(slow);
            fast = getnext(getnext(fast));
        }
        return fast == 1;
    }

    private int getnext(int n) {
        int sum = 0;
        while (n > 0) {
            sum += ((n % 10) * (n % 10));
            n /= 10;
        }
        return sum;
    }
}
