class Solution {

    public boolean method1(int n) {
        int s = n;
        int f = n;
        do {
            s = square(s);
            f = square(square(f));
        } while (s != f);
        if (s == 1) {
            return true;
        }
        return false;
    }

    public int square(int n) {
        int ans = 0;
        while (n > 0) {
            int r = n % 10;
            ans += r * r;
            n = n / 10;
        }
        return ans;
    }
}
