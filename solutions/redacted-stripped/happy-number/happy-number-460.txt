class Solution {

    public boolean method1(int n) {
        int slow = n, fast = n;
        while (fast != 1 && method2(fast) != 1) {
            slow = method2(slow);
            fast = method2(method2(fast));
            if (fast == slow)
                return false;
        }
        return true;
    }

    public int method2(int num) {
        int temp = num;
        int sum = 0;
        while (temp != 0) {
            sum += Math.pow((temp % 10), 2);
            temp /= 10;
        }
        return sum;
    }
}
