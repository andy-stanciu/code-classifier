class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = sumOfSqureOfDigit(slow);
            fast = sumOfSqureOfDigit(sumOfSqureOfDigit(fast));
        } while (slow != fast);
        return slow == 1 ? true : false;
    }

    private int sumOfSqureOfDigit(int n) {
        int ans = 0;
        while (n > 0) {
            int rem = n % 10;
            ans += rem * rem;
            n = n / 10;
        }
        return ans;
    }
}
