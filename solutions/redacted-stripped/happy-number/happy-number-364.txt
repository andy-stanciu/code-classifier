class Solution {

    public boolean method1(int n) {
        int sum;
        do {
            sum = 0;
            while (n != 0) {
                int rem = n % 10;
                sum += (rem * rem);
                n /= 10;
            }
            n = sum;
        } while (sum > 9);
        if (sum == 1 || sum == 7) {
            return true;
        }
        return false;
    }
}
