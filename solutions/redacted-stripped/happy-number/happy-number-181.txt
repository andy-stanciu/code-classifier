class Solution {

    public boolean method1(int n) {
        n = mul(n);
        do {
            n = mul(n);
        } while (n >= 10);
        return 1 == n;
    }

    public int mul(int m) {
        int sum = 0;
        while (m > 0) {
            sum = sum + ((m % 10) * (m % 10));
            m /= 10;
        }
        return sum;
    }
}
