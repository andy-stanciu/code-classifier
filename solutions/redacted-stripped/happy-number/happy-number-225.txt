class Solution {

    public boolean method1(int n) {
        HashSet<Integer> seen = new HashSet<>();
        int num = 0;
        while (n != 1) {
            while (n > 0) {
                int rem = n % 10;
                num += Math.pow(rem, 2);
                n /= 10;
            }
            n = num;
            num = 0;
            if (seen.contains(n)) {
                return false;
            }
            seen.add(n);
        }
        return true;
    }
}
