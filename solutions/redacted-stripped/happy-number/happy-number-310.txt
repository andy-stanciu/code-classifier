class Solution {

    public boolean method1(int n) {
        int sum = 0;
        while (n > 4) {
            sum = 0;
            while (n > 0) {
                int k = n % 10;
                sum = sum + k * k;
                n = n / 10;
            }
            n = sum;
        }
        return (n == 1);
    }
}
