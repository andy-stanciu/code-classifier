class Solution {

    public boolean method1(int n) {
        if (n == 1)
            return true;
        Set<Integer> visited = new HashSet<>();
        while (n != 1) {
            if (visited.contains(n))
                return false;
            visited.add(n);
            n = sumOfDigits(n);
        }
        return true;
    }

    private int sumOfDigits(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }
}
