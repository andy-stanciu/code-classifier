class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = square(slow);
            fast = square(square(fast));
        } while (slow != fast);
        return slow == 1;
    }

    private int square(int num) {
        int res = 0;
        while (num > 0) {
            int reminder = num % 10;
            res += reminder * reminder;
            num /= 10;
        }
        return res;
    }
}
