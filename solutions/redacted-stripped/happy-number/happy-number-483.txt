class Solution {

    public boolean method1(int n) {
        while (n > 9) {
            int sum = 0;
            while (n > 0) {
                int d = n % 10;
                sum = sum + d * d;
                n = n / 10;
            }
            n = sum;
        }
        if (n == 1 || n == 7) {
            return true;
        } else {
            return false;
        }
    }
}
