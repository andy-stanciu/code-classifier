class Solution {

    public boolean method1(int n) {
        HashSet<Integer> set = new HashSet<>();
        int sum = 0;
        while (n != 1) {
            while (n > 0) {
                sum += (n % 10) * (n % 10);
                n /= 10;
            }
            n = sum;
            sum = 0;
            if (!set.contains(n))
                set.add(n);
            else
                return false;
        }
        return true;
    }
}
