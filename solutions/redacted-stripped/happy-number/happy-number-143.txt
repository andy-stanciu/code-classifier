class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = method2(n);
        while (fast != 1 && slow != fast) {
            slow = method2(slow);
            fast = method2(method2(fast));
        }
        return fast == 1;
    }

    public int method2(int num) {
        int sum = 0;
        while (num != 0) {
            int d = num % 10;
            num = num / 10;
            sum += d * d;
        }
        return sum;
    }
}
