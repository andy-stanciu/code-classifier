class Solution {

    public boolean method1(int n) {
        int haze = n;
        int turtle = n;
        do {
            haze = square(square(haze));
            turtle = square(turtle);
        } while (haze != turtle);
        return haze == 1;
    }

    private int square(int n) {
        int sum = 0;
        while (n != 0) {
            int remainder = n % 10;
            sum = sum + remainder * remainder;
            n = n / 10;
        }
        return sum;
    }
}
