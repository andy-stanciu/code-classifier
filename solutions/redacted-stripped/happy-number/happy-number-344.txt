class Solution {

    public boolean method1(int n) {
        int square = findSquare(n);
        int slow = square;
        int fast = square;
        while (slow != 1) {
            slow = findSquare(slow);
            fast = findSquare(findSquare(fast));
            if (slow == fast && slow != 1) {
                return false;
            }
        }
        return true;
    }

    private int findSquare(int param1) {
        int ans = 0;
        while (param1 > 0) {
            int rem = param1 % 10;
            ans = ans + (rem * rem);
            param1 = param1 / 10;
        }
        return ans;
    }
}
