class Solution {

    public boolean method1(int n) {
        int rem, sum = 0;
        if (n == 2)
            return false;
        while (true) {
            sum = 0;
            while (n != 0) {
                rem = n % 10;
                sum = sum + ((int) Math.pow(rem, 2));
                n = n / 10;
            }
            if (sum == 1 || sum == 7)
                return true;
            else if (sum < 10)
                return false;
            n = sum;
        }
    }
}
