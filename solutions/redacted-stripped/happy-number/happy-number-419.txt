class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = computeNum(slow);
            fast = computeNum(computeNum(fast));
            if (slow == 1)
                return true;
        } while (slow != fast);
        return false;
    }

    public int computeNum(int n) {
        int sum = 0;
        while (n != 0) {
            sum += (n % 10) * (n % 10);
            n = n / 10;
        }
        return sum;
    }
}
