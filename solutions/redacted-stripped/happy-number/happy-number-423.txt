class Solution {

    public boolean method1(int n) {
        if (n == 1)
            return true;
        if (n > 1 && n <= 4)
            return false;
        int sum = 0;
        while (n != 0) {
            sum += Math.pow(n % 10, 2);
            n /= 10;
        }
        return method1(sum);
    }
}
