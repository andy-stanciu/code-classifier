class Solution {

    public boolean method1(int n) {
        int result = sqrdigit(n);
        while (true) {
            if (result == 1) {
                return true;
            }
            result = sqrdigit(result);
            if (result == 4) {
                return false;
            }
        }
    }

    public int sqrdigit(int num) {
        int sum = 0;
        while (num > 0) {
            int digit = num % 10;
            sum = sum + digit * digit;
            num = num / 10;
        }
        return sum;
    }
}
