class Solution {

    public boolean method1(int n) {
        if (n == 1)
            return true;
        while (n > 4) {
            int sum = 0;
            while (n > 0) {
                int r = n % 10;
                sum += r * r;
                n /= 10;
            }
            if (sum == 1)
                return true;
            n = sum;
        }
        return false;
    }
}
