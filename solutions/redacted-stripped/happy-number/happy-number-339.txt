class Solution {

    public boolean method1(int n) {
        int s = n;
        int f = n;
        do {
            s = check(s);
            f = check(check(f));
        } while (f != s);
        if (s == 1)
            return true;
        return false;
    }

    public int check(int n) {
        int ans = 0;
        while (n > 0) {
            int rem = n % 10;
            ans += rem * rem;
            n /= 10;
        }
        return ans;
    }
}
