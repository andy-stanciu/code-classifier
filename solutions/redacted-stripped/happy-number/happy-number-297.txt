class Solution {

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        while (!set.contains(n) && !(n == 1)) {
            set.add(n);
            n = calculate(n);
        }
        return n == 1;
    }

    public int calculate(int n) {
        int i = 0;
        int sum = 0;
        while (n > 0) {
            i = n % 10;
            sum += i * i;
            n = n / 10;
        }
        return sum;
    }
}
