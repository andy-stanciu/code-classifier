class Solution {

    public boolean method1(int n) {
        int sum = n;
        while (sum != 1) {
            if (sum > 1 && sum < 7) {
                return false;
            }
            sum = square(sum);
        }
        return true;
    }

    public static int square(int n) {
        int result = 0;
        while (n != 0) {
            int temp = n % 10;
            result = result + temp * temp;
            n = n / 10;
        }
        return result;
    }
}
