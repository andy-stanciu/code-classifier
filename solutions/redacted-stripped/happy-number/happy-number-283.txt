class Solution {

    public static int sq(int n) {
        int sum = 0;
        while (n > 0) {
            int r = n % 10;
            sum += r * r;
            n = n / 10;
        }
        return sum;
    }

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        if (n < 10 && n != 1 && n % 2 == 0) {
            return false;
        }
        while (n >= 1) {
            int sqq = sq(n);
            n = sq(sqq);
            if (set.contains(n)) {
                break;
            }
            set.add(n);
            if (n == 1) {
                break;
            }
        }
        if (n == 1) {
            return true;
        } else {
            return false;
        }
    }
}
