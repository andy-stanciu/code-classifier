public class Solution {

    public boolean method1(int n) {
        var list = new HashSet<Integer>();
        while (true) {
            var square = square(n);
            if (square == 1)
                return true;
            if (list.contains(square))
                return false;
            n = square;
            list.add(square);
        }
    }

    public int square(int num) {
        var ans = 0;
        while (num > 0) {
            var remainder = num % 10;
            ans += remainder * remainder;
            num /= 10;
        }
        return ans;
    }
}
