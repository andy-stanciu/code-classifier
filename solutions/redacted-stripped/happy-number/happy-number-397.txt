class Solution {

    public boolean method1(int n) {
        if (n <= 9) {
            if (n == 1 || n == 7)
                return true;
            else
                return false;
        }
        int ans = 0;
        while (n >= 1) {
            ans += (n % 10) * (n % 10);
            n /= 10;
        }
        return method1(ans);
    }
}
