class Solution {

    public boolean method1(int n) {
        while (n > 9) {
            n = check(n);
        }
        if (n == 1 || n == 7) {
            return true;
        } else {
            return false;
        }
    }

    private int check(int n) {
        int sum = 0;
        while (n != 0) {
            int rem = n % 10;
            sum += rem * rem;
            n = n / 10;
        }
        return sum;
    }
}
