class Solution {

    static int sumOfSquares(int num) {
        int sum = 0;
        while (num > 0) {
            double last_diget = num % 10;
            sum += Math.pow(last_diget, 2);
            num = Math.floorDiv(num, 10);
        }
        return sum;
    }

    public boolean method1(int n) {
        if (n == 1)
            return true;
        int slow = n;
        int fast = n;
        while (true) {
            slow = sumOfSquares(slow);
            fast = sumOfSquares(sumOfSquares(fast));
            if (slow == fast)
                break;
        }
        return slow == 1;
    }
}
