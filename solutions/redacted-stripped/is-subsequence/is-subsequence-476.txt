class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int n = s.length();
        if (n == 0)
            return true;
        for (char rightChar : t.toCharArray()) {
            char leftChar = s.charAt(i);
            if (leftChar == rightChar)
                i++;
            if (i == n)
                return true;
        }
        return false;
    }
}
