class Solution {

    public boolean method1(String s, String t) {
        char[] charSubStringArray = s.toCharArray();
        char[] charArray = t.toCharArray();
        int j = 0;
        for (int i = 0; i < charArray.length; i++) {
            if (j == charSubStringArray.length)
                return true;
            if (charSubStringArray[j] == charArray[i])
                j++;
        }
        return j == charSubStringArray.length;
    }
}
