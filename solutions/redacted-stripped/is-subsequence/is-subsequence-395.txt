class Solution {

    public boolean method1(String s, String t) {
        int n = s.trim().length();
        int m = t.length();
        int q = 0;
        if (n == 0) {
            return true;
        } else if (m == 0) {
            return false;
        }
        for (int i = 0; i < t.length(); i++) {
            if (s.charAt(q) == t.charAt(i)) {
                q++;
                if (q == n)
                    return true;
            }
        }
        return false;
    }
}
