class Solution {

    public boolean method1(String s, String t) {
        int start = 0;
        int end = t.length() - 1;
        int j = 0;
        while (start <= end && j < s.length()) {
            if (t.charAt(start) == s.charAt(j)) {
                j++;
                start++;
            } else {
                start++;
            }
        }
        if (j == s.length())
            return true;
        return false;
    }
}
