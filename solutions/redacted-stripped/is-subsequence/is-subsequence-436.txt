class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        char[] chars = s.toCharArray();
        char[] chart = t.toCharArray();
        for (int i = 0; i < t.length(); i++) {
            if (j < chars.length && chars[j] == chart[i]) {
                j++;
            }
        }
        return j == chars.length;
    }
}
