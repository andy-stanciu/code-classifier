class Solution {

    public boolean method1(String s, String t) {
        if (s.length() <= 0) {
            return true;
        }
        int b = t.length() - 1;
        int count = s.length();
        int a = s.length() - 1;
        while (b >= 0 && a >= 0) {
            if (s.charAt(a) == t.charAt(b)) {
                a--;
                count--;
            }
            b--;
        }
        if (count == 0) {
            return true;
        }
        return false;
    }
}
