class Solution {

    public boolean method1(String s, String t) {
        int n = s.length();
        int m = t.length();
        if (n > m) {
            return false;
        }
        int i = 0;
        for (int j = 0; j < m && i < n; j++) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
            }
        }
        return i == n;
    }
}
