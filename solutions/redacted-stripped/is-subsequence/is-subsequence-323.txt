class Solution {

    public boolean method1(String s, String t) {
        int pointerS = 0;
        int pointerT = 0;
        for (int i = 0; i < s.length(); i++) {
            for (int j = pointerT; j < t.length(); j++) {
                if (s.charAt(pointerS) == t.charAt(j)) {
                    pointerS++;
                    pointerT = j + 1;
                    break;
                }
            }
        }
        return pointerS == s.length();
    }
}
