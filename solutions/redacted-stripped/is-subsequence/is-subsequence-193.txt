class Solution {

    public boolean method1(String param1, String original) {
        int originalIndex = 0;
        int var1 = 0;
        while (var1 < param1.length()) {
            boolean var2 = false;
            while (originalIndex < original.length()) {
                if (original.charAt(originalIndex) == param1.charAt(var1)) {
                    var2 = true;
                    originalIndex++;
                    break;
                }
                originalIndex++;
            }
            if (!var2) {
                return false;
            }
            var1++;
        }
        return true;
    }
}
