class Solution {

    public boolean method1(String s, String t) {
        int sIndex = 0;
        for (int i = 0; i < t.length(); i++) {
            if (sIndex < s.length() && s.charAt(sIndex) == t.charAt(i)) {
                sIndex++;
            }
        }
        return sIndex == s.length();
    }
}
