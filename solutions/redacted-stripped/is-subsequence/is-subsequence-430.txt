class Solution {

    public boolean method1(String s, String t) {
        int sCounter = 0;
        int tCounter = 0;
        int sSize = s.length();
        int tSize = t.length();
        while (sCounter < sSize && tCounter < tSize) {
            if (s.charAt(sCounter) == t.charAt(tCounter)) {
                sCounter++;
            }
            tCounter++;
        }
        return sCounter == sSize;
    }
}
