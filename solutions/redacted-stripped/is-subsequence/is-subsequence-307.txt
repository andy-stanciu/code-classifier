class Solution {

    public boolean method1(String s, String t) {
        int it = 0;
        int var1 = 0;
        while (var1 < s.length() && it < t.length()) {
            if (s.charAt(var1) == t.charAt(it)) {
                var1++;
            }
            it++;
        }
        return var1 == s.length();
    }
}
