class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == t.length()) {
            if (s.equals(t)) {
                return true;
            }
            return false;
        }
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            int trueValue = 0;
            for (int j = count; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    trueValue = 1;
                    count = j + 1;
                    break;
                }
            }
            if (trueValue == 0) {
                return false;
            }
        }
        return true;
    }
}
