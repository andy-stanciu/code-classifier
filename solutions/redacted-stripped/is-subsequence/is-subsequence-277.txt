class Solution {

    public boolean method1(String s, String t) {
        int p = 0, q = 0;
        if (s.length() == 0) {
            return true;
        }
        while (p < t.length()) {
            if (s.charAt(q) == t.charAt(p)) {
                q++;
            }
            if (q == s.length()) {
                return true;
            }
            p++;
        }
        return false;
    }
}
