class Solution {

    public boolean method1(String s, String t) {
        int a = s.length();
        int b = t.length();
        if (a == 0) {
            return true;
        }
        int ps = 0;
        int pt = 0;
        while (ps < a && pt < b) {
            if (s.charAt(ps) == t.charAt(pt)) {
                ps++;
            }
            pt++;
        }
        if (a == ps) {
            return true;
        }
        return false;
    }
}
