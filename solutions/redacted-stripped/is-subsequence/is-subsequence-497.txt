class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        int n = 0;
        if (s.length() == 0) {
            return true;
        }
        while (i < t.length()) {
            if (t.charAt(i) == s.charAt(j)) {
                j++;
                n++;
                if (n == s.length()) {
                    return true;
                }
            }
            i++;
        }
        return false;
    }
}
