class Solution {

    public boolean method1(String s, String t) {
        int sIndex = 0;
        int tIndex = 0;
        while (tIndex < t.length() && sIndex < s.length()) {
            if (t.charAt(tIndex) == s.charAt(sIndex)) {
                tIndex++;
                sIndex++;
            } else {
                tIndex++;
            }
        }
        return sIndex == s.length();
    }
}
