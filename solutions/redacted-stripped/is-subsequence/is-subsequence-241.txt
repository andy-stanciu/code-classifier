class Solution {

    public boolean subseq(String t, String s, int index, StringBuilder str) {
        if (index >= t.length()) {
            if (s.equalsIgnoreCase(str.toString()))
                return true;
            return false;
        }
        str.append(t.charAt(index));
        if (subseq(t, s, index + 1, str))
            return true;
        str.deleteCharAt(str.length() - 1);
        if (subseq(t, s, index + 1, str))
            return true;
        return false;
    }

    public boolean method1(String s, String t) {
        int j = 0;
        for (int i = 0; i < t.length(); i++) {
            if (j >= s.length())
                return true;
            if (t.charAt(i) == s.charAt(j)) {
                j++;
            }
        }
        if (j == s.length())
            return true;
        return false;
    }
}
