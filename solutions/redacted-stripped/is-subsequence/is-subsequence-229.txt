class Solution {

    public boolean method1(String s, String t) {
        boolean var1 = false;
        int i = 0;
        int j = 0;
        int slenght = s.length();
        int tLength = t.length();
        if (slenght > 0 && tLength > 0) {
            while (j < tLength) {
                if (s.charAt(i) == t.charAt(j)) {
                    i++;
                }
                j++;
            }
            if (i == slenght) {
                var1 = true;
            }
        }
        return var1;
    }
}
