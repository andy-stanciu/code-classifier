class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        while (s.length() > i && t.length() > j) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        if (s.length() == i) {
            return true;
        }
        return false;
    }
}
