class Solution {

    public boolean method1(String s, String t) {
        int m = s.length();
        int n = t.length();
        if (m == 0)
            return true;
        if (n == 0)
            return false;
        int i = 0;
        int j = 0;
        while (j < n) {
            if (i < m && s.charAt(i) == t.charAt(j)) {
                i++;
            }
            j++;
        }
        return i == m;
    }
}
