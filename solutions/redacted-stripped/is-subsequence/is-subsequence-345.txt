class Solution {

    public boolean method1(String s, String t) {
        int m = s.length(), n = t.length();
        if (m == n)
            return t.equals(s);
        if (m == 0)
            return true;
        int i = 0, j = 0;
        while (i < m && j < n) {
            if (s.charAt(i) == t.charAt(j)) {
                if (i == m - 1) {
                    return true;
                }
                i++;
            }
            j++;
        }
        return false;
    }
}
