class Solution {

    public boolean method1(String s, String t) {
        int k = 0;
        for (char i : t.toCharArray()) {
            if (k < s.length() && s.charAt(k) == i) {
                k++;
            }
            if (k >= s.length()) {
                break;
            }
        }
        return true ? k == s.length() : false;
    }
}
