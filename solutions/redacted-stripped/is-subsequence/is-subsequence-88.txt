class Solution {

    public boolean method1(String s, String t) {
        char[] new1 = s.toCharArray();
        char[] new2 = t.toCharArray();
        int n1 = new1.length;
        int n2 = new2.length;
        int j = 0;
        if (n1 == 0) {
            return true;
        }
        if (n2 == 0) {
            return false;
        }
        for (int i = 0; i < n2; i++) {
            if (j == n1 - 1) {
                if (new1[j] == new2[i]) {
                    return true;
                }
            } else {
                if (new1[j] == new2[i]) {
                    j++;
                }
            }
        }
        return false;
    }
}
