class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        while (i < s.length() && j < t.length()) {
            System.out.println("i:" + s.charAt(i) + "j:" + t.charAt(j));
            if (s.charAt(i) == t.charAt(j)) {
                i++;
            }
            j++;
        }
        System.out.println(j);
        if (i == s.length()) {
            return true;
        }
        return false;
    }
}
