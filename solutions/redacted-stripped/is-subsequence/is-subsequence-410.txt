class Solution {

    public boolean method1(String s, String t) {
        int p1 = 0;
        int p2 = 0;
        int n1 = s.length();
        int n2 = t.length();
        while (p1 < n1 && p2 < n2) {
            if (s.charAt(p1) == t.charAt(p2)) {
                p2++;
                p1++;
            } else {
                p2++;
            }
        }
        return p1 == n1;
    }
}
