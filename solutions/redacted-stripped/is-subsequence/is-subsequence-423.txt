class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0)
            return true;
        int i = 0;
        char search;
        for (char c : t.toCharArray()) {
            search = s.charAt(i);
            if (c == search)
                i++;
            if (i == s.length())
                return true;
        }
        return false;
    }
}
