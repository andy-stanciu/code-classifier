class Solution {

    public boolean method1(String s, String t) {
        int i = 0, j = 0;
        char[] mains = t.toCharArray();
        char[] sub = s.toCharArray();
        while (i < mains.length && j < sub.length) {
            if (mains[i] == sub[j]) {
                j++;
            }
            i++;
        }
        if (j == sub.length)
            return true;
        else
            return false;
    }
}
