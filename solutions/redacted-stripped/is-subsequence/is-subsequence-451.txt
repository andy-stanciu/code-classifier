class Solution {

    public boolean method1(String s, String t) {
        int i = 0, j = 0;
        int n2 = t.length(), n1 = s.length();
        if (n1 == 0)
            return true;
        while (j < n2) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                j++;
                if (i == n1)
                    break;
            } else {
                j++;
            }
        }
        return i == (n1);
    }
}
