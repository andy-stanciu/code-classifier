class Solution {

    public boolean method1(String s, String t) {
        if (s == null || s.equals("")) {
            return true;
        }
        char[] sChars = s.toCharArray();
        char[] tChars = t.toCharArray();
        int found = 0;
        for (char tChar : tChars) {
            if (found >= sChars.length) {
                break;
            }
            if (tChar == sChars[found]) {
                found++;
            }
        }
        if (found == sChars.length) {
            return true;
        }
        return false;
    }
}
