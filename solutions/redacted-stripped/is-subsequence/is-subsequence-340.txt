class Solution {

    public boolean method1(String s, String t) {
        boolean an = true;
        for (int i = 0; i < s.length(); i++) {
            if (t.indexOf(s.charAt(i)) == -1) {
                an = false;
                break;
            }
            t = t.substring(t.indexOf(s.charAt(i)) + 1);
        }
        return an;
    }
}
