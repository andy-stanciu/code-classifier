class Solution {

    public boolean method1(String s, String t) {
        boolean k = false;
        int i = 0;
        int j = 0;
        while (i < s.length() && j < t.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        if (i == s.length()) {
            k = true;
        }
        return k;
    }
}
