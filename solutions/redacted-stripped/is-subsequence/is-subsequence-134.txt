class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        int i = 0;
        while (i < t.length() && j < s.length()) {
            if (s.charAt(j) == t.charAt(i)) {
                j++;
            }
            i++;
        }
        return s.length() == j;
    }
}
