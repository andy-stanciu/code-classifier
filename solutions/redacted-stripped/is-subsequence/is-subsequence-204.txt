class Solution {

    public boolean method1(String s, String t) {
        char[] tCharArray = t.toCharArray();
        char[] sCharArray = s.toCharArray();
        int tLength = t.length();
        int sLength = s.length();
        if (tLength == 0 && sLength == 0)
            return true;
        else if (tLength < sLength || tLength == 0)
            return false;
        else if (sLength == 0)
            return true;
        int j = 0;
        for (int i = 0; i < tLength; i++) {
            if (tCharArray[i] == sCharArray[j]) {
                j++;
            }
            if (j == sLength) {
                return true;
            }
        }
        return false;
    }
}
