class Solution {

    public boolean method1(String s, String t) {
        char[] str = s.toCharArray();
        char[] str2 = t.toCharArray();
        int i = 0;
        int n = s.length();
        int m = t.length();
        if (n <= 0) {
            return true;
        }
        for (int j = 0; j < m; j++) {
            if (str[i] == str2[j]) {
                i++;
            }
            if (i == n) {
                return true;
            }
        }
        return false;
    }
}
