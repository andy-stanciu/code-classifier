class Solution {

    public boolean method1(String s, String t) {
        if (s.length() < 1) {
            return true;
        }
        int i = 0;
        int j = 0;
        while (i < t.length()) {
            if (s.charAt(j) == t.charAt(i)) {
                j++;
            }
            i++;
            if (j == s.length()) {
                return true;
            }
        }
        return false;
    }
}
