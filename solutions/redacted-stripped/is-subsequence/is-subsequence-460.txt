class Solution {

    public boolean method1(String s, String t) {
        if (s == null && t == null) {
            return true;
        }
        if (s == null || t == null) {
            return false;
        }
        int c1 = 0, c2 = 0;
        while (c1 < s.length() && c2 < t.length()) {
            if (s.charAt(c1) == t.charAt(c2)) {
                c1++;
            }
            c2++;
        }
        return c1 == s.length();
    }
}
