class Solution {

    public boolean method1(String s, String t) {
        int currPos = 0;
        for (int i = 0; i < s.length(); i++) {
            String temp = "" + s.charAt(i);
            if (t.indexOf(temp, currPos) != -1) {
                currPos = t.indexOf(temp, currPos) + 1;
            } else {
                return false;
            }
        }
        return true;
    }
}
