class Solution {

    public boolean[][] memo;

    public boolean method1(String s, String t) {
        memo = new boolean[s.length()][t.length()];
        int i = 0;
        int j = 0;
        while (i < s.length() && j < t.length()) {
            while (j < t.length() && s.charAt(i) != t.charAt(j)) {
                j++;
            }
            if (j == t.length())
                return false;
            memo[i][j] = true;
            i++;
            j++;
        }
        for (boolean[] booleans : memo) {
            int count = 0;
            for (int k = 0; k < memo[0].length; k++) {
                if (booleans[k])
                    count++;
            }
            if (count == 0)
                return false;
        }
        return true;
    }
}
