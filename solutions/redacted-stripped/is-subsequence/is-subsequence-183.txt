class Solution {

    public boolean method1(String s, String t) {
        int m = s.length();
        int n = t.length();
        if (m > n)
            return false;
        int i = 0;
        int j = 0;
        while (i < m && j < n) {
            if (s.charAt(i) == t.charAt(j)) {
                j++;
                i++;
            } else {
                j++;
            }
        }
        return i == m;
    }
}
