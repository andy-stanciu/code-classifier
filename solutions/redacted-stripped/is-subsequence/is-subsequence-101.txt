class Solution {

    public boolean method1(String s, String t) {
        String var1 = "";
        int j = 0;
        for (int i = 0; i < t.length(); i++) {
            if (j < s.length() && (t.charAt(i) == s.charAt(j))) {
                var1 += t.charAt(i);
                j++;
            }
        }
        return var1.equals(s);
    }
}
