class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0)
            return true;
        if (t.length() == 0)
            return false;
        char a = s.charAt(0);
        char b = t.charAt(0);
        if (a == b) {
            return method1(s.substring(1), t.substring(1));
        } else {
            return method1(s, t.substring(1));
        }
    }
}
