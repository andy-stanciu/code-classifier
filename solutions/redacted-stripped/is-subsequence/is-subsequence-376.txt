class Solution {

    public boolean method1(String s, String t) {
        int n = s.length();
        int k = 0;
        for (int i = 0; i < n; i++) {
            k = t.indexOf(s.charAt(i), k);
            if (k < 0) {
                return false;
            }
            k++;
        }
        return true;
    }
}
