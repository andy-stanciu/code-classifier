class Solution {

    public boolean method1(String s, String t) {
        int a = 0, b = 0;
        for (int i = 0; i < s.length(); i++) {
            for (int j = 0; j < t.length(); j++) {
                if (j >= a) {
                    if (s.charAt(i) == t.charAt(j)) {
                        a = j + 1;
                        b++;
                        break;
                    }
                }
            }
        }
        if (b == s.length()) {
            return true;
        } else {
            return false;
        }
    }
}
