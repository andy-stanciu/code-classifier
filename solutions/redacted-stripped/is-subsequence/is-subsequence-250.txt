class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            while (j < t.length()) {
                if (s.charAt(i) == t.charAt(j)) {
                    ans.append(t.charAt(j));
                    j++;
                    break;
                }
                j++;
            }
        }
        return ans.toString().equals(s);
    }
}
