class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        int sp = 0;
        for (int i = 0; i < t.length(); i++) {
            if (t.charAt(i) == s.charAt(sp)) {
                sp++;
                if (sp == s.length()) {
                    return true;
                }
            }
        }
        return false;
    }
}
