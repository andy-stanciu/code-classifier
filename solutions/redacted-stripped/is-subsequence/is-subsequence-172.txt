class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0)
            return true;
        Deque<Character> deque = new ArrayDeque<>();
        for (char c : s.toCharArray()) deque.offer(c);
        char temp = deque.removeFirst();
        for (char c : t.toCharArray()) {
            if (c == temp) {
                if (deque.isEmpty())
                    return true;
                else
                    temp = deque.removeFirst();
            }
        }
        return false;
    }
}
