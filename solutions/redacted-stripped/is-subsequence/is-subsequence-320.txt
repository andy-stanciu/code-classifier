class Solution {

    public boolean method1(String s, String t) {
        char[] tChar = t.toCharArray();
        char[] sChar = s.toCharArray();
        int j = 0;
        for (int i = 0; i < t.length(); i++) {
            if (j == sChar.length) {
                return true;
            }
            if (sChar[j] == tChar[i]) {
                j += 1;
            }
        }
        return j == sChar.length;
    }
}
