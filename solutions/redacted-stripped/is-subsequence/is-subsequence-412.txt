class Solution {

    public boolean method1(String s, String t) {
        int n = s.length();
        char[] sn = s.toCharArray();
        int i = 0;
        for (char c : t.toCharArray()) {
            if (i < n && c == sn[i]) {
                i++;
            }
        }
        return i == s.length();
    }
}
