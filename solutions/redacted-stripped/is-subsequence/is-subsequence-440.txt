class Solution {

    public boolean method1(String s, String t) {
        int sLen = s.length();
        int tLen = t.length();
        int sIndex = 0;
        int tIndex = 0;
        while (sIndex < sLen && tIndex < tLen) {
            if (s.charAt(sIndex) == t.charAt(tIndex)) {
                sIndex++;
            }
            tIndex++;
        }
        return sIndex == sLen;
    }
}
