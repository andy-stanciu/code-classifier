class Solution {

    public boolean method1(String s, String t) {
        int count = 0;
        int lenS = s.length();
        int lenT = t.length();
        if (lenS == 0) {
            return true;
        }
        for (int i = 0; i < lenT; i++) {
            if (s.charAt(count) == t.charAt(i)) {
                count++;
            }
            if (count == lenS) {
                return true;
            }
        }
        return false;
    }
}
