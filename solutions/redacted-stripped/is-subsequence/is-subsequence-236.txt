class Solution {

    public boolean method1(String s, String t) {
        int dimS = s.length();
        int dimT = t.length();
        char[] sChars = s.toCharArray();
        char[] tChars = t.toCharArray();
        int var1 = 0;
        int iT = 0;
        while (iT < dimT && var1 < dimS) {
            var1 += (tChars[iT] == sChars[var1]) ? 1 : 0;
            iT++;
        }
        return (var1 == dimS);
    }
}
