class Solution {

    public boolean method1(String s, String t) {
        if (s.isBlank()) {
            return true;
        }
        char[] Schar = s.toCharArray();
        char[] Tchar = t.toCharArray();
        int j = 0;
        for (int i = 0; i < Tchar.length; i++) {
            if (Tchar[i] == Schar[j]) {
                if (j == Schar.length - 1) {
                    return true;
                }
                j++;
            }
        }
        return false;
    }
}
