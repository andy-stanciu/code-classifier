class Solution {

    public boolean method1(String s, String t) {
        int j = s.length() - 1;
        for (int i = t.length() - 1; i >= 0; i--) {
            if (j < 0)
                break;
            if (s.charAt(j) == t.charAt(i))
                j--;
        }
        return j < 0;
    }
}
