class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0)
            return true;
        boolean[] chk = new boolean[s.length()];
        int l = 0;
        for (int i = 0; i < t.length(); i++) {
            if (t.charAt(i) == s.charAt(l)) {
                chk[l] = true;
                l++;
                if (l == s.length()) {
                    break;
                }
            }
        }
        for (int i = 0; i < s.length(); i++) {
            if (chk[i] == false) {
                return false;
            }
        }
        return true;
    }
}
