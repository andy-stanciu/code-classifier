class Solution {

    public boolean method1(String s, String t) {
        int sPointer = 0;
        int tPointer = 0;
        StringBuilder sb = new StringBuilder("");
        if (t.length() < s.length())
            return false;
        while (sPointer < s.length() && tPointer < t.length()) {
            char chS = s.charAt(sPointer);
            char chT = t.charAt(tPointer);
            if (chS == chT) {
                sb.append(chS);
                sPointer++;
                tPointer++;
            } else {
                tPointer++;
            }
        }
        String str = sb.toString();
        if (str.equals(s)) {
            return true;
        } else {
            return false;
        }
    }
}
