class Solution {

    public boolean method1(String s, String t) {
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < s.length(); i++) {
            stack.push(s.charAt(i));
        }
        for (int i = t.length() - 1; i >= 0 && stack.size() > 0; i--) {
            if (t.charAt(i) == stack.peek()) {
                stack.pop();
            }
        }
        return stack.isEmpty();
    }
}
