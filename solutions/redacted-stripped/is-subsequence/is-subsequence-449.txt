class Solution {

    public boolean method1(String s, String t) {
        if (s.length() > t.length()) {
            return false;
        }
        int prev = -1;
        for (int i = 0; i < s.length(); i++) {
            int tIndex = t.indexOf(s.charAt(i), prev + 1);
            if (tIndex == -1) {
                return false;
            }
            prev = tIndex;
        }
        return true;
    }
}
