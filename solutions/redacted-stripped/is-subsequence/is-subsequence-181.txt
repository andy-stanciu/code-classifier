class Solution {

    public boolean method1(String s, String t) {
        int sl = s.length();
        int tl = t.length();
        char[] sa = s.toCharArray();
        char[] ta = t.toCharArray();
        int i = 0, j = 0;
        while (i < sl && j < tl) {
            char sc = sa[i];
            char tc = ta[j];
            if (sc == tc) {
                i++;
            }
            j++;
        }
        return i == sl;
    }
}
