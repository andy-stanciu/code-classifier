class Solution {

    public boolean method1(String s, String t) {
        if (s.isEmpty()) {
            return true;
        }
        char[] arrayS = s.toCharArray();
        int index = t.indexOf(arrayS[0]);
        if (index < 0) {
            return false;
        }
        int j = 1;
        while (j < arrayS.length) {
            index = t.indexOf(arrayS[j], index + 1);
            if (index < 0) {
                return false;
            }
            j++;
        }
        return true;
    }
}
