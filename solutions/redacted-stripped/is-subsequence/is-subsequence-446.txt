class Solution {

    public boolean method1(String s, String t) {
        if (s.isEmpty())
            return true;
        int i = 0;
        for (int j = 0; j < t.length(); j++) {
            if (i < s.length() && t.charAt(j) == s.charAt(i)) {
                i++;
            }
        }
        if (i == s.length()) {
            return true;
        }
        return false;
    }
}
