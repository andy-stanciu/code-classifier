class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int m = s.length();
        int n = t.length();
        for (int j = 0; i < m && j < n; j++) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
            }
        }
        if (i == m)
            return true;
        return false;
    }
}
