class Solution {

    public boolean method1(String s, String t) {
        if (t.isEmpty() & s.isEmpty()) {
            return true;
        } else if (t.isEmpty()) {
            return false;
        }
        if (s.isEmpty()) {
            return true;
        }
        char[] ch1 = t.toCharArray();
        char[] ch2 = s.toCharArray();
        char ssc = ch2[0];
        int l = 1;
        for (int i = 0; i < ch1.length; i++) {
            if (ch1[i] == ssc) {
                if (ch2.length == l) {
                    return true;
                }
                ssc = ch2[l];
                l++;
            }
        }
        return false;
    }
}
