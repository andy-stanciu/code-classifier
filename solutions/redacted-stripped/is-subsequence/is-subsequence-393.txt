class Solution {

    public boolean method1(String s, String t) {
        int sCounter = 0;
        int tCounter = 0;
        while (sCounter < s.length() && tCounter < t.length()) {
            if (s.charAt(sCounter) == t.charAt(tCounter)) {
                sCounter += 1;
            }
            tCounter += 1;
        }
        return sCounter == s.length();
    }
}
