class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        for (int i = 0; i < s.length(); i++) {
            int x = t.indexOf(s.charAt(i), j);
            if (x == -1) {
                return false;
            } else {
                j = x + 1;
            }
        }
        return true;
    }
}
