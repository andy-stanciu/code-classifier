class Solution {

    public boolean method1(String s, String t) {
        if (s.equals("")) {
            return true;
        }
        int sIndex = 0, sLength = s.length();
        for (int i = 0; i < t.length(); i++) {
            if (t.charAt(i) == s.charAt(sIndex))
                sIndex++;
            if (sIndex == sLength)
                return true;
        }
        return false;
    }
}
