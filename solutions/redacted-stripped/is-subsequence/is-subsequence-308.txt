class Solution {

    public boolean method1(String s, String t) {
        int lens = s.length();
        int lent = t.length();
        int j = 0;
        for (int i = 0; i < lent && j < lens; i++) {
            if (s.charAt(j) == t.charAt(i)) {
                j++;
            }
        }
        if (j == lens) {
            return true;
        }
        return false;
    }
}
