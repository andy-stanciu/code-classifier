class Solution {

    public int method1(int[] nums) {
        int r_sum = 0;
        int l_sum = 0;
        int t_sum = 0;
        for (int num : nums) {
            t_sum += num;
        }
        for (int i = 0; i < nums.length; i++) {
            if (i != 0) {
                l_sum += nums[i - 1];
            }
            r_sum = t_sum - (l_sum + nums[i]);
            if (r_sum == l_sum) {
                return i;
            }
        }
        return -1;
    }
}
