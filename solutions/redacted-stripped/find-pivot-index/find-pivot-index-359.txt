class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int i, j;
        for (i = 0; i <= n - 1; i++) {
            int leftsum = 0, rightsum = 0;
            for (j = 0; j < i; j++) {
                leftsum = leftsum + nums[j];
            }
            for (j = i + 1; j <= n - 1; j++) {
                rightsum = rightsum + nums[j];
            }
            if (rightsum == leftsum)
                return i;
        }
        return -1;
    }
}
