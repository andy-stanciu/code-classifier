class Solution {

    public int method1(int[] nums) {
        int ls = 0;
        int ts = 0;
        for (int i = 0; i < nums.length; i++) {
            ts += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            int rs = ts - nums[i] - ls;
            if (ls == rs) {
                return i;
            }
            ls += nums[i];
        }
        return -1;
    }
}
