class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int leftsum = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            sum = sum + nums[i];
        }
        for (int i = 0; i < n; i++) {
            sum = sum - nums[i];
            if (sum == leftsum) {
                return i;
            }
            leftsum = leftsum + nums[i];
        }
        return -1;
    }
}
