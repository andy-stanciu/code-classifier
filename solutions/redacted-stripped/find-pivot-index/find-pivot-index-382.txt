class Solution {

    public int method1(int[] nums) {
        int left = 0, right = 0;
        for (int num : nums) {
            right += num;
        }
        int i = 0;
        while (i < nums.length) {
            right -= nums[i];
            if (left == right) {
                return i;
            }
            left += nums[i];
            i++;
        }
        return -1;
    }
}
