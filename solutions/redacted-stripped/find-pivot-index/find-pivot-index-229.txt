class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int left = 0;
        int var1 = -1;
        for (int i = nums.length - 1; i >= 0; i--) {
            sum -= nums[i];
            if (sum == left) {
                var1 = i;
            }
            left += nums[i];
        }
        return var1;
    }
}
