class Solution {

    public int method1(int[] nums) {
        if (nums.length < 1)
            return -1;
        int rsum = 0, lsum = 0;
        for (int num : nums) rsum = rsum + num;
        for (int i = 0; i < nums.length; i++) {
            rsum = rsum - nums[i];
            if (rsum == lsum)
                return i;
            lsum = lsum + nums[i];
        }
        return -1;
    }
}
