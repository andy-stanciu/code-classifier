class Solution {

    public int method1(int[] nums) {
        int sumLeft = 0;
        int totalSum = 0;
        int var1 = -1;
        for (int num : nums) {
            totalSum += num;
        }
        for (int i = 0; i < nums.length; i++) {
            if (2 * sumLeft == totalSum - nums[i]) {
                return i;
            }
            sumLeft += nums[i];
        }
        return var1;
    }
}
