class Solution {

    public int method1(int[] nums) {
        int sumleft = 0;
        int sumright = 0;
        for (int i = 1; i < nums.length; i++) {
            sumright += nums[i];
        }
        if (sumleft == sumright) {
            return 0;
        }
        for (int i = 1; i < nums.length; i++) {
            sumleft = sumleft + nums[i - 1];
            sumright = sumright - nums[i];
            if (sumleft == sumright) {
                return i;
            }
        }
        return -1;
    }
}
