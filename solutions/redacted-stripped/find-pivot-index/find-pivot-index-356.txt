class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int left = 0;
        int total = 0;
        for (int i = 0; i < n; i++) {
            total += nums[i];
        }
        for (int i = 0; i < n; i++) {
            total -= nums[i];
            if (left == total) {
                return i;
            }
            left += nums[i];
        }
        return -1;
    }
}
