class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int totalsum = 0;
        for (int i = 0; i < n; i++) {
            totalsum = totalsum + nums[i];
        }
        int currsum = 0;
        int idx = -1;
        for (int i = 0; i < n; i++) {
            if (currsum == totalsum - currsum - nums[i]) {
                idx = i;
                break;
            }
            currsum = currsum + nums[i];
        }
        return idx;
    }
}
