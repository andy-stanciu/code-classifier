class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return 0;
        int sum = 0;
        int[] leftArr = new int[nums.length];
        int[] rightArr = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            leftArr[i] = sum;
        }
        sum = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            sum += nums[i];
            rightArr[i] = sum;
        }
        for (int i = 0; i < nums.length; i++) {
            if (i == 0 && rightArr[i + 1] == 0)
                return i;
            if (i == (nums.length - 1) && leftArr[i - 1] == 0)
                return i;
            if (i != 0 && i != (nums.length - 1) && leftArr[i - 1] == rightArr[i + 1])
                return i;
        }
        return -1;
    }
}
