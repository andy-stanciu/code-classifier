class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        int leftSum = 0;
        for (int val : nums) {
            totalSum += val;
        }
        for (int i = 0; i < nums.length; ) {
            if (leftSum * 2 == totalSum - nums[i]) {
                return i;
            }
            leftSum += nums[i++];
        }
        return -1;
    }
}
