class Solution {

    public int method1(int[] nums) {
        int left = 0, sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            sum = sum - nums[i];
            if (sum == left) {
                return i;
            }
            left = left + nums[i];
        }
        return -1;
    }
}
