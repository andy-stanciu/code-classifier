class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        int totalSum = 0;
        for (int i = 0; i < len; i++) {
            totalSum += nums[i];
        }
        int prefixSum = 0;
        for (int i = 0; i < len; i++) {
            if (prefixSum == (totalSum - prefixSum - nums[i])) {
                return i;
            } else {
                prefixSum += nums[i];
            }
        }
        return -1;
    }
}
