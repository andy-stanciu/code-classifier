class Solution {

    public int method1(int[] nums) {
        int l = nums.length, s = 0, i = 0, s1 = 0, p = -1;
        for (i = 0; i < l; i++) {
            s = s + nums[i];
        }
        for (i = 0; i < l; i++) {
            s1 += nums[i];
            if (s1 - nums[i] == (s - s1)) {
                p = i;
                break;
            }
        }
        return p;
    }
}
