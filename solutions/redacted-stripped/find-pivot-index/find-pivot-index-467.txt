class Solution {

    public int method1(int[] nums) {
        int sum = Arrays.stream(nums).sum();
        System.out.println(sum);
        int[] prefix = new int[nums.length + 1];
        prefix[0] = 0;
        for (int i = 0; i < nums.length; i++) {
            prefix[i + 1] = prefix[i] + nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (prefix[i] == sum - prefix[i + 1]) {
                return i;
            }
        }
        return -1;
    }
}
