class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalSum += nums[i];
        }
        int leftTotal = 0;
        for (int j = 0; j < nums.length; j++) {
            int rightTotal = totalSum - leftTotal - nums[j];
            if (rightTotal == leftTotal) {
                return j;
            }
            leftTotal += nums[j];
        }
        return -1;
    }
}
