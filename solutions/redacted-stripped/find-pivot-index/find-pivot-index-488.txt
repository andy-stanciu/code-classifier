class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        System.out.println(sum);
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            int rightSum = sum - nums[i];
            if (leftSum == rightSum) {
                return i;
            }
            leftSum += nums[i];
            sum -= nums[i];
        }
        return -1;
    }
}
