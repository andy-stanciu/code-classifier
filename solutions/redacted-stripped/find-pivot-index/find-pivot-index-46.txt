class Solution {

    public int method1(int[] nums) {
        int[] left = new int[nums.length];
        int[] right = new int[nums.length];
        int leftsum = 0;
        for (int i = 0; i < nums.length; i++) {
            leftsum += nums[i];
            left[i] = leftsum;
        }
        int rightsum = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            rightsum += nums[i];
            right[i] = rightsum;
        }
        for (int i = 0; i < nums.length; i++) {
            if (left[i] == right[i])
                return i;
        }
        return -1;
    }
}
