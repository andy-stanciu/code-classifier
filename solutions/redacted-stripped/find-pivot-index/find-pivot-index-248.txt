class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return -1;
        int leftSum = 0;
        int rightSum = 0;
        for (int x : nums) {
            rightSum += x;
        }
        for (int i = 0; i < nums.length; i++) {
            rightSum -= nums[i];
            if (rightSum == leftSum)
                return i;
            leftSum += nums[i];
        }
        return -1;
    }
}
