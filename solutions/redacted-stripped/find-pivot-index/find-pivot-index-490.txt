class Solution {

    public int method1(int[] nums) {
        int cs = 0;
        int totalSum = 0;
        for (int elements : nums) {
            totalSum += elements;
        }
        for (int i = 0; i < nums.length; i++) {
            if (cs == totalSum - cs - nums[i])
                return i;
            cs = cs + nums[i];
        }
        return -1;
    }
}
