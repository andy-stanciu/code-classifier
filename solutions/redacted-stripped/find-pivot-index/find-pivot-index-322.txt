class Solution {

    public int method1(int[] nums) {
        int sr = 0, sl = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int k = 0; k <= i - 1; k++) {
                sl += nums[k];
            }
            for (int j = i + 1; j < nums.length; j++) {
                sr += nums[j];
            }
            if (sr == sl) {
                return i;
            } else {
                sr = 0;
                sl = 0;
            }
        }
        return -1;
    }
}
