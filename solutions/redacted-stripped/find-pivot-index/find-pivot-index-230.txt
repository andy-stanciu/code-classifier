class Solution {

    public int method1(int[] nums) {
        int sum = 0, lsum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (lsum == sum - lsum - nums[i]) {
                return i;
            }
            lsum += nums[i];
        }
        return -1;
    }
}
