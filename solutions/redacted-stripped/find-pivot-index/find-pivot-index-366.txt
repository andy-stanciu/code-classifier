class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i : nums) sum += i;
        int left = 0;
        int right = sum;
        for (int i = 0; i < nums.length; i++) {
            if (left == right - nums[i])
                return i;
            left += nums[i];
            right -= nums[i];
        }
        return -1;
    }
}
