class Solution {

    public int method1(int[] nums) {
        int rs = 0;
        int var1;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            rs += nums[i];
        }
        int ls = 0;
        for (int i = 0; i < n; i++) {
            rs -= nums[i];
            if (ls == rs)
                return i;
            ls += nums[i];
        }
        return -1;
    }
}
