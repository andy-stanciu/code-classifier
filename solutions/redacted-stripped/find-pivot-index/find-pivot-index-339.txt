class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int rightSum = 0;
        int var1 = 0;
        for (int i = 1; i < nums.length; i++) {
            rightSum = rightSum + nums[i];
        }
        if (leftSum == rightSum)
            return var1;
        while (leftSum != rightSum && var1 < nums.length - 1) {
            leftSum = leftSum + nums[var1];
            rightSum = rightSum - nums[var1 + 1];
            var1++;
        }
        if (var1 == nums.length - 1 && leftSum != rightSum)
            return -1;
        return var1;
    }
}
