class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int sum = Arrays.stream(nums).sum();
        int rightSum = sum;
        for (int i = 0; i < nums.length; i++) {
            leftSum = nums[i] + leftSum;
            rightSum = rightSum - nums[i];
            if (leftSum - nums[i] == rightSum) {
                return i;
            }
        }
        return -1;
    }
}
