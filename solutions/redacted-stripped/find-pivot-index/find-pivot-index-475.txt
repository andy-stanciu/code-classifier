class Solution {

    public int method1(int[] nums) {
        int sumOnLeft = 0;
        int totalSum = 0;
        for (int x : nums) {
            totalSum = totalSum + x;
        }
        for (int i = 0; i < nums.length; i++) {
            int sumOnRight = totalSum - nums[i] - sumOnLeft;
            if (sumOnRight == sumOnLeft)
                return i;
            sumOnLeft = sumOnLeft + nums[i];
        }
        return -1;
    }
}
