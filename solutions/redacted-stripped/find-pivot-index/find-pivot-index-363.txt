class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 1)
            return 0;
        int[] l = new int[n];
        l[0] = nums[0];
        for (int i = 1; i < n; i++) {
            l[i] = l[i - 1] + nums[i];
        }
        int[] r = new int[n];
        r[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            r[i] = r[i + 1] + nums[i];
        }
        for (int i = 0; i < n; i++) {
            if (l[i] == r[i]) {
                return i;
            }
        }
        return -1;
    }
}
