class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        for (int num : nums) {
            totalSum += num;
        }
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalSum = totalSum - nums[i];
            if (totalSum == sum) {
                return i;
            }
            sum += nums[i];
        }
        return -1;
    }
}
