class Solution {

    public int method1(int[] nums) {
        if (nums.length < 1) {
            return 1;
        }
        int totalsum = 0;
        for (int i : nums) {
            totalsum += i;
        }
        int leftsum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalsum = totalsum - nums[i];
            if (leftsum == totalsum) {
                return i;
            }
            leftsum = leftsum + nums[i];
        }
        return -1;
    }
}
