class Solution {

    public int method1(int[] nums) {
        int left_sum = 0;
        int total_sum = 0;
        for (int num : nums) {
            total_sum += num;
        }
        for (int j = 0; j < nums.length; j++) {
            int right_sum = total_sum - left_sum - nums[j];
            if (left_sum == right_sum) {
                return j;
            }
            left_sum += nums[j];
        }
        return -1;
    }
}
