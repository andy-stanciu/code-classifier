class Solution {

    public int method1(int[] arr) {
        int leftSum = 0, rightSum = 0, sum = 0;
        for (int i = 0; i < arr.length; i++) {
            sum += arr[i];
        }
        for (int i = 0; i < arr.length; i++) {
            rightSum = sum - arr[i] - leftSum;
            if (leftSum == rightSum)
                return i;
            leftSum += arr[i];
        }
        return -1;
    }
}
