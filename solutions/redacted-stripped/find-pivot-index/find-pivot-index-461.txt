class Solution {

    public int method1(int[] nums) {
        int sumLeft = 0;
        int sumRight = 0;
        int var1 = -1;
        for (int i = 0; i < nums.length; i++) {
            sumLeft += nums[i];
            for (int k = i; k < nums.length; k++) {
                sumRight += nums[k];
            }
            if (sumLeft == sumRight) {
                return i;
            } else {
                sumRight = 0;
            }
        }
        return var1;
    }
}
