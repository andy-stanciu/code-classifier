class Solution {

    public int method1(int[] nums) {
        int pi = -1;
        int ts = 0;
        for (int i = 0; i < nums.length; i++) {
            ts += nums[i];
        }
        int sl = 0;
        int sr = 0;
        for (int j = 0; j < nums.length; j++) {
            sr = ts - sl - nums[j];
            if (sl == sr) {
                pi = j;
                break;
            }
            sl += nums[j];
        }
        return pi;
    }
}
