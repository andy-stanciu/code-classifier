class Solution {

    public int method1(int[] arr) {
        int sum = 0;
        for (int i = 0; i < arr.length; i++) {
            sum += arr[i];
        }
        int temp = 0;
        for (int i = 0; i < arr.length; i++) {
            sum -= arr[i];
            if (sum == temp)
                return i;
            temp += arr[i];
        }
        return -1;
    }
}
