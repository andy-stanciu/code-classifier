class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        int leftsum = 0;
        int rightsum = sum;
        for (int i = 0; i < n; i++) {
            rightsum = rightsum - nums[i];
            if (leftsum == rightsum) {
                return i;
            }
            leftsum += nums[i];
        }
        return -1;
    }
}
