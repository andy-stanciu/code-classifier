class Solution {

    public int method1(int[] nums) {
        int rightsum = 0;
        for (int i = 0; i < nums.length; i++) {
            rightsum += nums[i];
        }
        int leftsum = 0;
        for (int p = 0; p < nums.length; p++) {
            rightsum -= nums[p];
            leftsum += nums[p];
            if (leftsum - nums[p] == rightsum) {
                return p;
            }
        }
        return -1;
    }
}
