class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i : nums) sum += i;
        int right_sum = sum;
        int left_sum = 0;
        for (int i = 0; i < nums.length; i++) {
            right_sum -= nums[i];
            if (left_sum == right_sum)
                return i;
            left_sum += nums[i];
        }
        return -1;
    }
}
