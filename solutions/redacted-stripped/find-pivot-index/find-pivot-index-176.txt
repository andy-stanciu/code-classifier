class Solution {

    public int method1(int[] nums) {
        if (nums == null)
            return -1;
        long sumRight = 0;
        long sumLeft = 0;
        for (int i : nums) sumRight += i;
        for (int i = 0; i < nums.length; i++) {
            sumRight -= nums[i];
            if (sumRight == sumLeft)
                return i;
            sumLeft += nums[i];
        }
        return -1;
    }
}
