class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        int[] sum = new int[len];
        sum[0] = nums[0];
        for (int i = 1; i < len; i++) {
            sum[i] = sum[i - 1] + nums[i];
        }
        for (int j = 0; j < len; j++) {
            if (sum[len - 1] - sum[j] == sum[j] - nums[j]) {
                return j;
            }
        }
        return -1;
    }
}
