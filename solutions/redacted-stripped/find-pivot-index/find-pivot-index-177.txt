class Solution {

    public int method1(int[] A) {
        int n = A.length;
        int[] L = new int[n];
        int[] R = new int[n];
        for (int i = 1; i < n; i++) L[i] = L[i - 1] + A[i - 1];
        for (int i = n - 2; i >= 0; i--) R[i] = R[i + 1] + A[i + 1];
        for (int i = 0; i < n; i++) if (L[i] == R[i])
            return i;
        return -1;
    }
}
