class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        int ltsum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (ltsum == sum - ltsum - nums[i]) {
                return i;
            }
            ltsum += nums[i];
        }
        return -1;
    }
}
