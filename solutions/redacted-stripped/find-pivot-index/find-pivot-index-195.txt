class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        for (int i = 1; i < nums.length; i++) {
            nums[i] += nums[i - 1];
        }
        if (nums[n - 1] - nums[0] == 0)
            return 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[n - 1] - nums[i] == nums[i - 1])
                return i;
        }
        return -1;
    }
}
