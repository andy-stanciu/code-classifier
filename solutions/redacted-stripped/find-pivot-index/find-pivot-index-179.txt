class Solution {

    public int method1(int[] nums) {
        int ans = -1;
        int n = nums.length;
        int[] prefixSum = new int[n + 2];
        for (int i = 0; i < n; i++) {
            prefixSum[i + 1] = prefixSum[i] + nums[i];
        }
        for (int i = 1; i <= n; i++) {
            if (prefixSum[i - 1] == prefixSum[n] - prefixSum[i]) {
                return i - 1;
            }
        }
        return ans;
    }
}
