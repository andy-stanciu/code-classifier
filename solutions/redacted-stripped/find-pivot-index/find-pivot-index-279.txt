class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int sumleft = 0;
        int sumright = 0;
        for (int i = 0; i < nums.length; i++) {
            sumright += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            sumright -= nums[i];
            if (sumleft == sumright) {
                return i;
            }
            sumleft += nums[i];
        }
        return -1;
    }
}
