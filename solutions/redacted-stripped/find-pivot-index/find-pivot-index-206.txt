class Solution {

    public int method1(int[] nums) {
        int numsSize = nums.length;
        int leftsum = 0;
        int rightsum = 0;
        int totalsum = 0;
        for (int i = 0; i < numsSize; i++) {
            totalsum += nums[i];
        }
        for (int i = 0; i < numsSize; i++) {
            rightsum = totalsum - leftsum - nums[i];
            if (leftsum == rightsum)
                return i;
            else
                leftsum += nums[i];
        }
        return -1;
    }
}
