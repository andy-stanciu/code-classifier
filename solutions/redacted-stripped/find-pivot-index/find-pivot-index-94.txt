class Solution {

    public int method1(int[] nums) {
        int ts = 0;
        int ls = 0;
        for (int i : nums) {
            ts += i;
        }
        for (int i = 0; i < nums.length; i++) {
            if (ts - ls - nums[i] == ls) {
                return i;
            }
            ls += nums[i];
        }
        return -1;
    }
}
