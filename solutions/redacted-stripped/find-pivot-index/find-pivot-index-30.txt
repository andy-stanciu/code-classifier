class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int prefix = 0;
        for (int num : nums) sum += num;
        for (int i = 0; i < nums.length; i++) {
            if (prefix == sum - prefix - nums[i])
                return i;
            prefix += nums[i];
        }
        return -1;
    }
}
