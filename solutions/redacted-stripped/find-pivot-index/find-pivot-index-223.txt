class Solution {

    public int method1(int[] nums) {
        int leftsum = 0;
        int rightsum = 0;
        for (int num : nums) {
            rightsum += num;
        }
        for (int i = 0; i < nums.length; i++) {
            if (leftsum == rightsum - nums[i]) {
                return i;
            }
            leftsum += nums[i];
            rightsum -= nums[i];
        }
        return -1;
    }
}
