class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int tsum = 0;
        for (int num : nums) {
            tsum += num;
        }
        int lsum = 0;
        for (int i = 0; i < n; i++) {
            if (lsum == tsum - lsum - nums[i]) {
                return i;
            }
            lsum += nums[i];
        }
        return -1;
    }
}
