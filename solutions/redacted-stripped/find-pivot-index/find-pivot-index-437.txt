class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        int sum1 = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        if (sum - nums[0] == 0) {
            return 0;
        }
        for (int i = 0; i < n - 1; i++) {
            sum1 += nums[i];
            if (sum1 == sum - sum1 - nums[i + 1]) {
                return i + 1;
            }
        }
        return -1;
    }
}
