class Solution {

    public int method1(int[] nums) {
        int rightSum = 0;
        for (int value : nums) {
            rightSum += value;
        }
        int var1 = -1;
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            rightSum -= nums[i];
            if (leftSum == rightSum) {
                var1 = i;
                break;
            }
            leftSum += nums[i];
        }
        return var1;
    }
}
