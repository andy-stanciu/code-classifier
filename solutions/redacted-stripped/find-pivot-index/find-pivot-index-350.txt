class Solution {

    public int method1(int[] nums) {
        int tSum = 0;
        for (int i = 0; i < nums.length; i++) {
            tSum += nums[i];
        }
        int lSum = 0;
        for (int i = 0; i < nums.length; i++) {
            int rSum = tSum - lSum - nums[i];
            if (lSum == rSum) {
                return i;
            }
            lSum += nums[i];
        }
        return -1;
    }
}
