class Solution {

    public int method1(int[] nums) {
        int sum = 0, ls = 0;
        for (int i = 0; i < nums.length; i++) sum += nums[i];
        for (int i = 0; i < nums.length; i++) {
            sum -= nums[i];
            if (sum == ls)
                return i;
            ls += nums[i];
        }
        return -1;
    }
}
