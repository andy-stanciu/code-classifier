class Solution {

    public int method1(int[] nums) {
        int method1 = 0;
        while (method1 < nums.length) {
            int leftSum = 0;
            int rightSum = 0;
            for (int i = 0; i < method1; i++) {
                leftSum += nums[i];
            }
            if (method1 != nums.length - 1) {
                for (int j = method1 + 1; j < nums.length; j++) {
                    rightSum += nums[j];
                }
            }
            if (leftSum == rightSum) {
                return method1;
            } else {
                method1++;
            }
        }
        return -1;
    }
}
