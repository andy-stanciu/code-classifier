class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int[] ps = new int[nums.length];
        ps[0] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            ps[i] = ps[i - 1] + nums[i];
        }
        if (ps[n - 1] - ps[0] == 0) {
            return 0;
        }
        for (int i = 1; i < nums.length; i++) {
            if (ps[i - 1] == (ps[n - 1] - ps[i])) {
                return i;
            }
        }
        return -1;
    }
}
