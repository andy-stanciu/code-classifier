class Solution {

    public int method1(int[] arr) {
        int n = arr.length;
        int totalSum = 0;
        for (int i = 0; i <= n - 1; i++) {
            totalSum = totalSum + arr[i];
        }
        int sumLeft = 0;
        for (int i = 0; i <= n - 1; i++) {
            int sumRight = totalSum - sumLeft - arr[i];
            if (sumLeft == sumRight) {
                return i;
            }
            sumLeft = sumLeft + arr[i];
        }
        return -1;
    }
}
