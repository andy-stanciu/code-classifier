class Solution {

    public int method1(int[] nums) {
        int lSum = 0, rSum = 0;
        for (int x : nums) rSum += x;
        for (int i = 0; i < nums.length; i++) {
            rSum -= nums[i];
            if (lSum == rSum)
                return i;
            lSum += nums[i];
        }
        return -1;
    }
}
