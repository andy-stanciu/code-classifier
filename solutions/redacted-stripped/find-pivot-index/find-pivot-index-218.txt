class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int val : nums) {
            sum += val;
        }
        int left = 0;
        for (int i = 0; i < nums.length; i++) {
            int r = sum - left - nums[i];
            if (r == left) {
                return i;
            }
            left += nums[i];
        }
        return -1;
    }
}
