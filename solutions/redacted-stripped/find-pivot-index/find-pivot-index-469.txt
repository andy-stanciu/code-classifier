class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int left_sum = 0;
        for (int num : nums) {
            sum += num;
        }
        for (int i = 0; i < nums.length; i++) {
            sum -= nums[i];
            if (left_sum == sum) {
                return i;
            }
            left_sum += nums[i];
            System.out.println(left_sum);
        }
        return -1;
    }
}
