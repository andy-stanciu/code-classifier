class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n + 2];
        dp[0] = 0;
        for (int i = 0; i < n; i++) {
            dp[i + 1] = dp[i] + nums[i];
        }
        dp[n + 1] = dp[n];
        for (int i = 0; i < n; i++) {
            if (dp[i] == dp[n + 1] - dp[i + 1])
                return i;
        }
        return -1;
    }
}
