class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        int prefix = 0;
        for (int i = 0; i < nums.length; i++) {
            if (sum - nums[i] - prefix == prefix) {
                return i;
            }
            prefix += nums[i];
        }
        return -1;
    }
}
