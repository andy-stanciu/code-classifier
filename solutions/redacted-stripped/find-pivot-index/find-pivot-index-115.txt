class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int tSum = 0;
        for (int i = 0; i < n; i++) tSum += nums[i];
        int prefixSum = 0;
        int suffixSum = tSum;
        for (int i = 0; i < n; i++) {
            if (i == 0) {
                prefixSum = nums[i];
                suffixSum = tSum;
            } else {
                nums[i] += nums[i - 1];
                prefixSum = nums[i];
                suffixSum = tSum - nums[i - 1];
            }
            if (prefixSum == suffixSum)
                return i;
        }
        return -1;
    }
}
