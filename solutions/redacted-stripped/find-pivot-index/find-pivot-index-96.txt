class Solution {

    public int method1(int[] nums) {
        int current = 0;
        int sumLeft = 0, sumRight = 0;
        for (int i = 1; i < nums.length; i++) sumRight += nums[i];
        for (current = 1; current < nums.length && sumLeft != sumRight; current++) {
            sumLeft += nums[current - 1];
            sumRight -= nums[current];
        }
        if (sumLeft == sumRight)
            return current - 1;
        return -1;
    }
}
