class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return -1;
        }
        for (int i = 0; i < n; i++) {
            int sum1 = 0;
            int sum2 = 0;
            for (int k = 0; k <= i; k++) {
                sum1 += nums[k];
            }
            for (int j = n - 1; j >= i; j--) {
                sum2 += nums[j];
            }
            if (sum1 == sum2) {
                return i;
            }
        }
        return -1;
    }
}
