class Solution {

    public int method1(int[] nums) {
        int var1 = -1;
        for (int x = 0; x < nums.length; x++) {
            int var2 = leftSum(nums, x);
            int var3 = rightSum(nums, x);
            if (var2 == var3) {
                return x;
            }
        }
        return var1;
    }

    private int leftSum(int[] nums, int n) {
        int var2 = 0;
        for (int x = n - 1; x >= 0; x--) {
            var2 += nums[x];
        }
        return var2;
    }

    private int rightSum(int[] nums, int n) {
        int var3 = 0;
        for (int x = n + 1; x < nums.length; x++) {
            var3 += nums[x];
        }
        return var3;
    }
}
