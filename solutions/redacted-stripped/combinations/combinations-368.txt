class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> list = new ArrayList<>();
        for (int i = 1; n - i + 1 >= k; i++) {
            addCombination(n, k, new ArrayList<>(), list, i);
        }
        return list;
    }

    public void addCombination(int n, int k, List<Integer> combination, List<List<Integer>> list, int i) {
        combination.add(i);
        int size = combination.size();
        if (size == k) {
            list.add(combination);
            return;
        }
        for (int j = i + 1; size + n - j + 1 >= k; j++) {
            addCombination(n, k, new ArrayList<>(combination), list, j);
        }
    }
}
