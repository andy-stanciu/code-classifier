class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<Integer> smallList = new ArrayList<>();
        List<List<Integer>> finalList = new ArrayList<>();
        combineHelper(1, n, k, smallList, finalList);
        return finalList;
    }

    static void combineHelper(int start, int n, int k, List<Integer> smallList, List<List<Integer>> finalList) {
        if (smallList.size() == k) {
            finalList.add(new ArrayList(smallList));
            return;
        }
        for (int i = start; i <= n; i++) {
            if (!smallList.contains(i)) {
                smallList.add(i);
                combineHelper(i, n, k, smallList, finalList);
                smallList.remove(smallList.size() - 1);
            }
        }
    }
}
