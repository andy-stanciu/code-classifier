class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ll = new ArrayList<>();
        List<Integer> al = new ArrayList<>();
        backtrack(1, ll, al, n, k);
        return ll;
    }

    public void backtrack(int s, List<List<Integer>> ll, List<Integer> al, int n, int k) {
        if (al.size() == k) {
            ll.add(new ArrayList<>(al));
            return;
        }
        for (int i = s; i <= n; i++) {
            al.add(i);
            backtrack(i + 1, ll, al, n, k);
            al.remove(al.size() - 1);
        }
    }
}
