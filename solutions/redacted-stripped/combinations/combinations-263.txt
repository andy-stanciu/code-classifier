class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        LinkedList<Integer> list = new LinkedList<>();
        helper(1, n, k, res, list);
        return res;
    }

    private void helper(int num, int n, int remaining, List<List<Integer>> res, List<Integer> list) {
        if (remaining == 0) {
            res.add(new ArrayList<>(list));
            return;
        }
        for (int i = num; i <= n; i++) {
            list.add(i);
            helper(i + 1, n, remaining - 1, res, list);
            list.remove(list.size() - 1);
        }
    }
}
