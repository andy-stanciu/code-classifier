class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        combination(n, k, new ArrayList<>(), 1, result);
        return result;
    }

    static void combination(int n, int k, List<Integer> inner, int idx, List<List<Integer>> outer) {
        if (inner.size() == k) {
            outer.add(new ArrayList<>(inner));
            return;
        }
        for (int i = idx; i <= n; i++) {
            inner.add(i);
            combination(n, k, inner, i + 1, outer);
            inner.remove(inner.size() - 1);
        }
        return;
    }
}
