class Solution {

    static void combineEngine(List<List<Integer>> result, List<Integer> array, int index, int k, int n) {
        if (k == 0) {
            result.add(new ArrayList<>(array));
            return;
        }
        for (int i = index; i <= n; i++) {
            array.add(i);
            combineEngine(result, array, i + 1, k - 1, n);
            if (!array.isEmpty())
                array.remove(array.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> array = new ArrayList<>();
        combineEngine(res, array, 1, k, n);
        return res;
    }
}
