class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        getCombination(1, n, k, new ArrayList<>(), result);
        return result;
    }

    public void getCombination(int c, int n, int k, List<Integer> temp, List<List<Integer>> result) {
        if (temp.size() == k) {
            result.add(new ArrayList<>(temp));
            return;
        }
        for (int i = c; i <= n; i++) {
            temp.add(i);
            getCombination(i + 1, n, k, temp, result);
            temp.remove(temp.size() - 1);
        }
        return;
    }
}
