class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> answer = new ArrayList<List<Integer>>();
        combine_helper(1, n, k, new ArrayList<Integer>(), answer);
        return answer;
    }

    public void combine_helper(int ind, int n, int k, List<Integer> subset, List<List<Integer>> answer) {
        if (k == 0) {
            answer.add(new ArrayList<Integer>(subset));
            return;
        }
        for (int i = ind; i <= n - k + 1; i++) {
            subset.add(i);
            combine_helper(i + 1, n, k - 1, subset, answer);
            subset.remove(subset.size() - 1);
        }
    }
}
