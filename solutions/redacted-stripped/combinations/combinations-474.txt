class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList();
        method1(1, n, new ArrayList(), res, k);
        return res;
    }

    void method1(int i, int n, List<Integer> curr, List<List<Integer>> res, int k) {
        if (curr.size() == k) {
            res.add(new ArrayList(curr));
            return;
        }
        for (int j = i; j <= n; j++) {
            curr.add(j);
            method1(j + 1, n, curr, res, k);
            curr.remove(curr.size() - 1);
        }
    }
}
