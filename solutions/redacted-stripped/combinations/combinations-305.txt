class Solution {

    public List<List<Integer>> combine(int n, int k, int i) {
        ArrayList<List<Integer>> ret = new ArrayList<List<Integer>>();
        for (; i <= n; ++i) {
            final int i_t = i;
            if (k == 1) {
                ret.add(new ArrayList<Integer>() {

                    {
                        add(i_t);
                    }
                });
            } else {
                for (List<Integer> j : combine(n, k - 1, i + 1)) {
                    ret.add(new ArrayList<Integer>() {

                        {
                            add(i_t);
                            addAll(j);
                        }
                    });
                }
            }
        }
        return ret;
    }

    public List<List<Integer>> combine(int n, int k) {
        return combine(n, k, 1);
    }
}
