class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        combine_recusion(n, k, 1, new ArrayList<>(), result);
        return result;
    }

    private void combine_recusion(int n, int k, int start, List<Integer> combination, List<List<Integer>> result) {
        if (combination.size() == k) {
            result.add(new ArrayList<>(combination));
            return;
        }
        for (int i = start; i <= n; i++) {
            combination.add(i);
            combine_recusion(n, k, i + 1, combination, result);
            combination.remove(combination.size() - 1);
        }
    }
}
