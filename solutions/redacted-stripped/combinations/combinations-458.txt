class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        combination(1, n, k, new ArrayList<>(), result);
        return result;
    }

    private void combination(int start, int n, int k, List<Integer> combination, List<List<Integer>> result) {
        if (combination.size() == k) {
            result.add(new ArrayList<>(combination));
            return;
        }
        int maxNum = n - (k - combination.size()) + 1;
        for (int i = start; i <= maxNum; i++) {
            combination.add(i);
            combination(i + 1, n, k, combination, result);
            combination.remove(combination.size() - 1);
        }
    }
}
