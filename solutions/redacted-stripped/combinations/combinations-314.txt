class Solution {

    private List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> combine(int n, int k) {
        int[] arr = new int[n];
        for (int i = 1; i <= n; i++) arr[i - 1] = i;
        perform(arr, k, 0, new ArrayList<>());
        return res;
    }

    private void perform(int[] arr, int k, int curr, ArrayList<Integer> currList) {
        if (curr > arr.length) {
            return;
        }
        if (currList.size() == k) {
            res.add(new ArrayList<Integer>(currList));
            return;
        }
        for (int i = curr; i < arr.length; i++) {
            currList.add(arr[i]);
            perform(arr, k, i + 1, currList);
            currList.remove(currList.size() - 1);
        }
    }
}
