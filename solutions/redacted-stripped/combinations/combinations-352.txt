class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> values = new ArrayList<>();
        result = check(n, k, 1, result, values);
        return result;
    }

    private static List<List<Integer>> check(int n, int k, int start, List<List<Integer>> result, List<Integer> values) {
        if (values.size() >= k) {
            result.add(new ArrayList<>(values));
            return result;
        }
        for (int i = start; i <= n; i++) {
            values.add(i);
            check(n, k, i + 1, result, values);
            values.remove(values.size() - 1);
        }
        return result;
    }
}
