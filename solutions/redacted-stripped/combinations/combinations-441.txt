class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList();
        List<Integer> list = new ArrayList();
        generateSubsets(1, n, list, res, k);
        return res;
    }

    public void generateSubsets(int start, int n, List<Integer> cur, List<List<Integer>> res, int k) {
        if (cur.size() == k) {
            res.add(new ArrayList(cur));
        }
        for (int i = start; i <= n; i++) {
            cur.add(i);
            generateSubsets(i + 1, n, cur, res, k);
            cur.remove(cur.size() - 1);
        }
    }
}
