class Solution {

    public List<List<Integer>> combine(int n, int k) {
        final var result = new ArrayList<List<Integer>>();
        final var allComb = new LinkedList<Integer>();
        combine(n, k, 1, result, allComb);
        return result;
    }

    public void combine(final int n, final int k, final int left, final List<List<Integer>> result, final LinkedList<Integer> allComb) {
        if (k == 0) {
            result.add(new ArrayList<>(allComb));
        } else {
            for (int i = left; i <= n; i++) {
                allComb.offer(i);
                combine(n, k - 1, i + 1, result, allComb);
                allComb.pollLast();
            }
        }
    }
}
