class Solution {

    void combSum(int i, int k, int N, List<Integer> result, List<List<Integer>> results) {
        if (result.size() == k) {
            results.add(new ArrayList<>(result));
            return;
        }
        for (int j = i; j <= N; j++) {
            result.add(j);
            combSum(j + 1, k, N, result, results);
            result.remove(result.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> results = new ArrayList<>();
        List<Integer> result = new ArrayList<>();
        combSum(1, k, n, result, results);
        return results;
    }
}
