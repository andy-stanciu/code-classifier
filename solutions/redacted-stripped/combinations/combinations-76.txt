class Solution {

    public List<List<Integer>> combine(int n, int k) {
        return helper(n, k, new ArrayList<>(), new ArrayList<>(), 1);
    }

    private static List<List<Integer>> helper(int n, int k, List<List<Integer>> result, List<Integer> current, int index) {
        if (current.size() == k) {
            result.add(new ArrayList<>(current));
            return result;
        }
        for (int i = index; i <= n; i++) {
            current.add(i);
            helper(n, k, result, current, i + 1);
            current.remove(current.size() - 1);
        }
        return result;
    }
}
