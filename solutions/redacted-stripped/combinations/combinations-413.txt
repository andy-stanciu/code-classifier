class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        method1(1, n, k, new ArrayList<>(), ans);
        return ans;
    }

    private void method1(int i, int n, int k, List<Integer> curr, List<List<Integer>> ans) {
        if (i > n) {
            if (curr.size() == k) {
                ans.add(new ArrayList<>(curr));
            }
            return;
        }
        if (curr.size() < k) {
            curr.add(i);
            method1(i + 1, n, k, curr, ans);
            curr.remove(curr.size() - 1);
        }
        method1(i + 1, n, k, curr, ans);
    }
}
