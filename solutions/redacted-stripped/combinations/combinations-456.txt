class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> integerList = new ArrayList<>();
        combinebyRecur(list, integerList, n, k, 0);
        return list;
    }

    public void combinebyRecur(List<List<Integer>> list, List<Integer> integerList, int n, int k, int index) {
        if (integerList.size() == k) {
            list.add(new ArrayList<>(integerList));
            return;
        }
        for (int i = index + 1; i < n + 1; i++) {
            if (integerList.contains(i))
                continue;
            integerList.add(i);
            combinebyRecur(list, integerList, n, k, i);
            integerList.remove(integerList.size() - 1);
        }
    }
}
