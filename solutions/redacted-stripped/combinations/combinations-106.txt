class Solution {

    static void generate(int start, int n, int k, List<List<Integer>> res, List<Integer> pair) {
        if (k == 0) {
            res.add(new ArrayList<>(pair));
            return;
        }
        for (int i = start; i <= n - k + 1; i++) {
            pair.add(i);
            generate(i + 1, n, k - 1, res, pair);
            pair.remove(pair.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        generate(1, n, k, res, new ArrayList<>());
        return res;
    }
}
