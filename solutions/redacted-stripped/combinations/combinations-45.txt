class Solution {

    List<List<Integer>> res = new ArrayList<List<Integer>>();

    public List<List<Integer>> combine(int n, int k) {
        go(new ArrayList<Integer>(), n, k);
        return (res);
    }

    private void go(ArrayList<Integer> l, int n, int k) {
        if (k == 0)
            res.add(l);
        for (int i = n; i >= k && k > 0; --i) {
            ArrayList<Integer> cl = new ArrayList<>(l);
            cl.add(i);
            go(cl, i - 1, k - 1);
        }
    }
}
