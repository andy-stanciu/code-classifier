class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> lst = new ArrayList<>();
        finder(n, k, 1, lst, new ArrayList<>());
        return lst;
    }

    public static void finder(int n, int k, int curr, List<List<Integer>> lst, ArrayList<Integer> sub) {
        if (sub.size() == k) {
            lst.add(new ArrayList<>(sub));
            return;
        }
        if (curr > n) {
            return;
        }
        if (sub.size() < k) {
            sub.add(curr);
            finder(n, k, curr + 1, lst, sub);
            sub.remove(sub.size() - 1);
        }
        finder(n, k, curr + 1, lst, sub);
    }
}
