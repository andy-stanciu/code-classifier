class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> combine(int n, int k) {
        ncr(new ArrayList<>(), 1, k, n);
        return res;
    }

    private void ncr(List<Integer> ls, int currIndex, int k, int n) {
        if (ls.size() == k) {
            res.add(new ArrayList<>(ls));
            return;
        }
        ;
        if (currIndex > n)
            return;
        ls.add(currIndex);
        ncr(ls, currIndex + 1, k, n);
        ls.remove(ls.size() - 1);
        ncr(ls, currIndex + 1, k, n);
    }
}
