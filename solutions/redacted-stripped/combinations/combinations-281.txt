class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        method1(1, n, 0, k, new ArrayList<>(), result);
        return result;
    }

    public static void method1(int cb, int tb, int ssf, int ts, List<Integer> combination, List<List<Integer>> result) {
        if (cb > tb) {
            if (ssf == ts) {
                result.add(new ArrayList<>(combination));
            }
            return;
        }
        combination.add(cb);
        method1(cb + 1, tb, ssf + 1, ts, combination, result);
        combination.remove(combination.size() - 1);
        method1(cb + 1, tb, ssf, ts, combination, result);
    }
}
