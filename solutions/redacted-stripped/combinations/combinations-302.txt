class Solution {

    List<List<Integer>> Mlist = new ArrayList<>();

    List<Integer> list = new ArrayList<>();

    public List<List<Integer>> combine(int n, int k) {
        if (k == 0) {
            Mlist.add(new ArrayList(list));
        } else if (n > 0) {
            list.add(n);
            combine(n - 1, k - 1);
            list.remove(list.size() - 1);
            if (k < n) {
                combine(n - 1, k);
            }
        }
        return Mlist;
    }
}
