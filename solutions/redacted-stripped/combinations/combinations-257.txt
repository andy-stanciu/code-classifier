class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        comb(n, k, new ArrayList<Integer>(), 1, ans);
        return ans;
    }

    public static void comb(int a, int k, ArrayList<Integer> form, int st, List<List<Integer>> f) {
        if (k == 0) {
            f.add(new ArrayList<>(form));
            return;
        }
        for (int i = st; i <= a; i++) {
            form.add(i);
            k = k - 1;
            comb(a, k, form, i + 1, f);
            k = k + 1;
            form.remove(form.size() - 1);
        }
    }
}
