class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> uniqu = new HashSet<Integer>();
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 0; j < nums2.length; j++) {
                if (nums1[i] == nums2[j]) {
                    uniqu.add(nums1[i]);
                }
            }
        }
        int[] result = new int[uniqu.size()];
        int i = 0;
        for (int num : uniqu) {
            result[i++] = num;
        }
        return result;
    }
}
