class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> s1 = new HashSet<>();
        Set<Integer> s2 = new HashSet<>();
        for (int i : nums1) s1.add(i);
        for (int k : nums2) s2.add(k);
        s1.retainAll(s2);
        int i = 0;
        int[] a = new int[s1.size()];
        for (int l : s1) a[i++] = l;
        return a;
    }
}
