class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> nums1_set = new HashSet<>();
        Set<Integer> nums2_set = new HashSet<>();
        for (int num : nums1) {
            nums1_set.add(num);
        }
        for (int num : nums2) {
            nums2_set.add(num);
        }
        nums1_set.retainAll(nums2_set);
        int[] result = new int[nums1_set.size()];
        int index = 0;
        for (int num : nums1_set) {
            result[index++] = num;
        }
        return result;
    }
}
