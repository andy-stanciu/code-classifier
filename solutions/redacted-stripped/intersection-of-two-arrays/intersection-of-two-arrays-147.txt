class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> resultset = new HashSet<>();
        int n = nums1.length;
        int m = nums2.length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (nums1[i] == nums2[j]) {
                    resultset.add(nums1[i]);
                }
            }
        }
        int[] arr = resultset.stream().mapToInt(Integer::intValue).toArray();
        return arr;
    }
}
