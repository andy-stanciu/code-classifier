class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> result = new HashSet<>();
        for (int i = 0; i < nums1.length; i++) for (int j = 0; j < nums2.length; j++) if (nums1[i] == nums2[j])
            result.add(nums1[i]);
        int[] arr = new int[result.size()];
        int k = 0;
        for (int i : result) arr[k++] = i;
        return arr;
    }
}
