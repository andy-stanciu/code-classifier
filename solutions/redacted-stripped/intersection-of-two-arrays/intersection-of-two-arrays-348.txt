class Solution {

    public int[] method1(int[] a, int[] b) {
        boolean[] exist = new boolean[1001];
        int[] inter = new int[Math.max(a.length, b.length)];
        int k = 0;
        for (int A : a) {
            exist[A] = true;
        }
        for (int B : b) {
            if (exist[B]) {
                inter[k] = B;
                k++;
                exist[B] = false;
            }
        }
        return Arrays.copyOfRange(inter, 0, k);
    }
}
