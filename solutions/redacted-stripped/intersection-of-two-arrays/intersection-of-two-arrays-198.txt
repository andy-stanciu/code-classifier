class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> ans = new HashSet<>();
        for (int value : nums1) {
            set.add(value);
        }
        for (int value : nums2) {
            if (set.contains(value)) {
                ans.add(value);
            }
        }
        int[] array = new int[ans.size()];
        int index = 0;
        for (int value : ans) {
            array[index] = value;
            index++;
        }
        return array;
    }
}
