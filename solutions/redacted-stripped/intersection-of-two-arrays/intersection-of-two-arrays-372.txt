class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        HashSet<Integer> set1 = new HashSet<Integer>();
        HashSet<Integer> set2 = new HashSet<Integer>();
        for (int i : nums1) set1.add(i);
        for (int j : nums2) {
            if (set1.contains(j))
                set2.add(j);
        }
        int[] arr = new int[set2.size()];
        int index = 0;
        for (int num : set2) {
            arr[index] = num;
            index++;
        }
        return arr;
    }
}
