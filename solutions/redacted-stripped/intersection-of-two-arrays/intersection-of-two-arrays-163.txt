class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        List<Integer> arr = new ArrayList<>();
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 0; j < nums2.length; j++) {
                if (nums1[i] == nums2[j]) {
                    if (!isPresent(arr, nums1[i]))
                        arr.add(nums1[i]);
                }
            }
        }
        int[] res = new int[arr.size()];
        for (int i = 0; i < arr.size(); i++) {
            res[i] = arr.get(i);
        }
        return res;
    }

    private boolean isPresent(List<Integer> arr, int val) {
        for (int i = 0; i < arr.size(); i++) {
            int num = arr.get(i);
            if (num == val)
                return true;
        }
        return false;
    }
}
