class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] temp = new int[1001];
        for (int i = 0; i < nums1.length; i++) {
            if (temp[nums1[i]] > 0)
                continue;
            temp[nums1[i]]++;
        }
        List<Integer> l = new ArrayList<>();
        for (int i = 0; i < nums2.length; i++) {
            if (temp[nums2[i]] > 0) {
                l.add(nums2[i]);
                temp[nums2[i]] = 0;
            }
        }
        Iterator<Integer> it = l.iterator();
        int[] ans = new int[l.size()];
        int x = 0;
        while (it.hasNext()) {
            ans[x++] = (int) it.next();
        }
        return ans;
    }
}
