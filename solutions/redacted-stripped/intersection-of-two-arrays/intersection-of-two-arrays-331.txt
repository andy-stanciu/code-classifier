class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        boolean[] count = new boolean[1001];
        for (int i : nums1) {
            count[i] = true;
        }
        int index = 0;
        int[] ans = new int[nums2.length];
        for (int i : nums2) {
            if (count[i]) {
                ans[index++] = i;
                count[i] = false;
            }
        }
        return Arrays.copyOf(ans, index);
    }
}
