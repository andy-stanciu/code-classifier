class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> method1 = new HashSet<>();
        for (int i : nums1) {
            set.add(i);
        }
        for (int i : nums2) {
            if (set.contains(i)) {
                method1.add(i);
            }
        }
        int[] result = new int[method1.size()];
        int count = 0;
        for (Integer value : method1) {
            result[count] = value;
            count++;
        }
        return result;
    }
}
