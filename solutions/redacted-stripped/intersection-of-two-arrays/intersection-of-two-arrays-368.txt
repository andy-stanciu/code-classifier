class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        ArrayList<Integer> l = new ArrayList<Integer>();
        int p = 0, q = 0;
        int ind = 0;
        while (p < nums1.length && q < nums2.length) {
            if (nums1[p] == nums2[q]) {
                if (l.size() != 0) {
                    if (l.get(l.size() - 1) == nums1[p]) {
                    } else {
                        l.add(nums1[p]);
                    }
                } else {
                    l.add(nums1[p]);
                }
                p++;
                q++;
            } else if (nums1[p] < nums2[q]) {
                p++;
            } else {
                q++;
            }
        }
        int[] method1 = new int[l.size()];
        for (int i = 0; i < l.size(); i++) {
            method1[i] = l.get(i);
        }
        return method1;
    }
}
