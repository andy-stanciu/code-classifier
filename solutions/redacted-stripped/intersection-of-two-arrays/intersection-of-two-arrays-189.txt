class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        ArrayList<Integer> al = new ArrayList<>();
        HashSet<Integer> hs = new HashSet<>();
        for (Integer n : nums1) {
            hs.add(n);
        }
        for (Integer n : nums2) {
            if (hs.contains(n) && !al.contains(n)) {
                al.add(n);
            }
        }
        int[] a = new int[al.size()];
        for (int i = 0; i < al.size(); i++) {
            a[i] = al.get(i);
        }
        return a;
    }
}
