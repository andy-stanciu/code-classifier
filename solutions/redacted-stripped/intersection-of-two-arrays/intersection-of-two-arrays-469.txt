class Solution {

    public int[] method1(int[] a, int[] b) {
        int n = a.length;
        int m = b.length;
        ArrayList<Integer> list = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int ele = a[i];
                int ele2 = b[j];
                if (a[i] == b[j] && (!(list.contains(ele))) && (!(list.contains(ele2)))) {
                    list.add(a[i]);
                }
            }
        }
        int[] arr = new int[list.size()];
        for (int k = 0; k < list.size(); k++) {
            arr[k] = list.get(k);
        }
        return arr;
    }
}
