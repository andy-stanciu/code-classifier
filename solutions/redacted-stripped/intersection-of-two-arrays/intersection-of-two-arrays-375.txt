class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] count = new int[1001];
        for (int num : nums1) {
            count[num]++;
        }
        int[] returningArray = new int[1001];
        int counter = 0;
        for (int num : nums2) {
            if (count[num] > 0) {
                returningArray[counter++] = num;
                count[num] = 0;
            }
        }
        return Arrays.copyOfRange(returningArray, 0, counter);
    }
}
