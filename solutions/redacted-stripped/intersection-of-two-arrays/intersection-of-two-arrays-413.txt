class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> s1 = new HashSet<Integer>();
        for (int v : nums1) {
            s1.add(v);
        }
        Set<Integer> s2 = new HashSet<Integer>();
        for (int val : nums2) {
            if (s1.contains(val)) {
                s2.add(val);
            }
        }
        int[] ans = new int[s2.size()];
        int k = 0;
        for (Integer i : s2) {
            ans[k] = i;
            k++;
        }
        return ans;
    }
}
