class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        boolean[] histogram = new boolean[1001];
        for (int n : nums1) {
            histogram[n] = true;
        }
        int[] method1 = new int[nums1.length];
        int j = 0;
        for (int n : nums2) {
            if (histogram[n])
                method1[j++] = n;
            histogram[n] = false;
        }
        return Arrays.copyOf(method1, j);
    }
}
