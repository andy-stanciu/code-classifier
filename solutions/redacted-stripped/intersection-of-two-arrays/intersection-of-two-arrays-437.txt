class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] seem = new int[1000];
        for (int i : nums1) seem[i]++;
        int[] result = new int[Math.min(nums1.length, nums2.length)];
        int j = 0;
        for (int i : nums2) {
            if (seem[i] > 0) {
                result[j++] = i;
                seem[i] = 0;
            }
        }
        int[] res = Arrays.copyOf(result, j);
        return res;
    }
}
