class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> map = new HashSet<>();
        for (int k : nums1) {
            map.add(k);
        }
        HashSet<Integer> res = new HashSet<>();
        for (int i : nums2) {
            if (map.contains(i)) {
                res.add(i);
            }
        }
        int[] resultArray = new int[res.size()];
        int index = 0;
        for (int k : res) {
            resultArray[index++] = k;
        }
        return resultArray;
    }
}
