public class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> method1 = new HashSet<>();
        for (int num : nums1) {
            set.add(num);
        }
        for (int num : nums2) {
            if (set.contains(num)) {
                method1.add(num);
            }
        }
        int[] result = new int[method1.size()];
        int i = 0;
        for (Integer num : method1) {
            result[i++] = num;
        }
        return result;
    }
}
