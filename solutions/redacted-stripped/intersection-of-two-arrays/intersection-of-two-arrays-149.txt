class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] num = new int[1001];
        int i = 0;
        while (i < nums1.length) {
            if (num[nums1[i]] == 0) {
                num[nums1[i]] = 1;
            }
            i++;
        }
        i = 0;
        int count = 0;
        while (i < nums2.length) {
            if (num[nums2[i]] == 1) {
                count++;
                num[nums2[i]] += 1;
            }
            i++;
        }
        i = 0;
        int j = 0;
        int[] result = new int[count];
        while (i < num.length) {
            if (num[i] > 1 && j < count) {
                result[j] = i;
                j++;
            }
            i++;
        }
        return result;
    }
}
