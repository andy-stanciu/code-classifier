class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> var1 = new HashSet<>();
        Set<Integer> nums2set = new HashSet<>();
        for (int num2 : nums2) {
            nums2set.add(num2);
        }
        for (int num1 : nums1) {
            if (nums2set.contains(num1))
                var1.add(num1);
        }
        return var1.stream().mapToInt(Integer::intValue).toArray();
    }
}
