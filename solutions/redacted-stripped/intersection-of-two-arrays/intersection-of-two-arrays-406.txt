class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> sets1 = new HashSet<>();
        ArrayList<Integer> var1 = new ArrayList<>();
        for (int i = 0; i < nums1.length; i++) {
            sets1.add(nums1[i]);
        }
        for (int j = 0; j < nums2.length; j++) {
            if (sets1.contains(nums2[j])) {
                sets1.remove(nums2[j]);
                var1.add(nums2[j]);
            }
        }
        int[] var2 = new int[var1.size()];
        for (int k = 0; k < var1.size(); k++) {
            var2[k] = var1.get(k);
        }
        return var2;
    }
}
