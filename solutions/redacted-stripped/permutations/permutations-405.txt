class Solution {

    private List<Integer> makeIntegerList(String[] a) {
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < a.length; i++) {
            ans.add(Integer.parseInt(a[i]));
        }
        return ans;
    }

    private void permutation(int[] a, int n, List<List<Integer>> ans, int mask, String psf) {
        if (mask == Math.pow(2, n) - 1) {
            ans.add(makeIntegerList(psf.split(" ")));
            return;
        }
        for (int i = 0; i < n; i++) {
            if ((mask & (1 << i)) > 0)
                continue;
            permutation(a, n, ans, mask + (1 << i), psf + a[i] + " ");
        }
    }

    public List<List<Integer>> permute(int[] a) {
        List<List<Integer>> ans = new ArrayList<>();
        int mask = 0;
        permutation(a, a.length, ans, mask, "");
        return ans;
    }
}
