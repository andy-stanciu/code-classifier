class Solution {

    public void printp(int[] nums, int indx, List<List<Integer>> ans) {
        if (indx == nums.length - 1) {
            List<Integer> als = new ArrayList<>();
            for (int i = 0; i < nums.length; i++) {
                als.add(nums[i]);
            }
            ans.add(als);
            return;
        }
        for (int i = indx; i < nums.length; i++) {
            swap(indx, i, nums);
            printp(nums, indx + 1, ans);
            swap(indx, i, nums);
        }
    }

    public void swap(int a1, int a2, int[] nums) {
        int temp = nums[a1];
        nums[a1] = nums[a2];
        nums[a2] = temp;
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        printp(nums, 0, ans);
        return ans;
    }
}
