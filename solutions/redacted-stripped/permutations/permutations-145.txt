class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        method1(result, new ArrayList<>(), nums, 0);
        return result;
    }

    public static List<List<Integer>> method1(List<List<Integer>> result, List<Integer> processed, int[] unprocessed, int n) {
        if (processed.size() == unprocessed.length) {
            result.add(new ArrayList<>(processed));
            return result;
        }
        for (int i = 0; i <= processed.size(); i++) {
            processed.add(i, unprocessed[n]);
            method1(result, processed, unprocessed, n + 1);
            processed.remove(i);
        }
        return result;
    }
}
