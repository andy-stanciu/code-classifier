class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> a = new ArrayList<>();
        generate(nums, a, new ArrayList<>(), new boolean[21]);
        return a;
    }

    private void generate(int[] nums, List<List<Integer>> list, ArrayList<Integer> temp, boolean[] used) {
        if (temp.size() == nums.length) {
            list.add(new ArrayList<>(temp));
            return;
        }
        for (int x : nums) {
            if (!used[x + 10]) {
                used[x + 10] = true;
                temp.add(x);
                generate(nums, list, temp, used);
                used[temp.remove(temp.size() - 1) + 10] = false;
            }
        }
    }
}
