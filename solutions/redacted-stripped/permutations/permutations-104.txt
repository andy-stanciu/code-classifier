class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> list = new ArrayList<>();
        method1(list, nums, 0);
        return list;
    }

    public void method1(List<List<Integer>> list, int[] nums, int start) {
        if (start == nums.length) {
            List<Integer> temp = new ArrayList<>();
            for (int i = 0; i < nums.length; i++) {
                temp.add(nums[i]);
            }
            list.add(temp);
            return;
        }
        for (int i = start; i < nums.length; i++) {
            swap(nums, start, i);
            method1(list, nums, start + 1);
            swap(nums, start, i);
        }
    }

    public void swap(int[] nums, int start, int i) {
        int temp = nums[i];
        nums[i] = nums[start];
        nums[start] = temp;
    }
}
