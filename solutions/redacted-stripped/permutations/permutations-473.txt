class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> data = new ArrayList<>();
        helper(0, nums.length, nums, data);
        return data;
    }

    public void helper(int pos, int size, int[] arr, List<List<Integer>> data) {
        if (pos >= size) {
            List<Integer> subAns = new ArrayList<>();
            for (int i : arr) {
                subAns.add(i);
            }
            data.add(subAns);
            return;
        }
        helper(pos + 1, size, arr, data);
        for (int i = pos + 1; i < size; i++) {
            int temp = arr[pos];
            arr[pos] = arr[i];
            arr[i] = temp;
            helper(pos + 1, size, arr, data);
            temp = arr[pos];
            arr[pos] = arr[i];
            arr[i] = temp;
        }
    }
}
