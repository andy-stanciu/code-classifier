class Solution {

    public List<List<Integer>> solve(int[] nums, int[] v, List<List<Integer>> ans, ArrayList<Integer> arr) {
        if (arr.size() == nums.length) {
            ans.add(new ArrayList<>(arr));
            return ans;
        }
        for (int i = 0; i < nums.length; i++) {
            if (v[i] == 0) {
                v[i] = 1;
                arr.add(nums[i]);
                solve(nums, v, ans, arr);
                arr.remove(arr.size() - 1);
                v[i] = 0;
            }
        }
        return ans;
    }

    public List<List<Integer>> permute(int[] nums) {
        return solve(nums, new int[nums.length], new ArrayList<>(), new ArrayList<>());
    }
}
