class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public void dfs(int[] nums, List<Integer> sol, int len) {
        if (sol.size() == len) {
            List<Integer> in = new ArrayList<>();
            for (int val : sol) in.add(val);
            res.add(in);
            return;
        }
        for (int num : nums) {
            if (!sol.contains(num)) {
                sol.add(num);
                dfs(nums, sol, len);
                sol.remove(sol.size() - 1);
            }
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        dfs(nums, new ArrayList<>(), nums.length);
        return res;
    }
}
