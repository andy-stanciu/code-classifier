class Solution {

    List<List<Integer>> ml;

    public List<List<Integer>> permute(int[] nums) {
        ml = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        method1(nums, new ArrayList<>(), visited);
        return ml;
    }

    public void method1(int[] nums, List<Integer> cl, boolean[] visited) {
        if (cl.size() == nums.length) {
            ml.add(new ArrayList<>(cl));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!visited[i]) {
                visited[i] = true;
                cl.add(nums[i]);
                method1(nums, cl, visited);
                cl.remove(cl.size() - 1);
                visited[i] = false;
            }
        }
    }
}
