class Solution {

    void cal(int[] a, int[] pick, List<Integer> ds, List<List<Integer>> ans) {
        if (ds.size() == a.length) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = 0; i < a.length; i++) {
            if (pick[i] == 0) {
                pick[i] = 1;
                ds.add(a[i]);
                cal(a, pick, ds, ans);
                pick[i] = 0;
                ds.remove(ds.size() - 1);
            }
        }
    }

    public List<List<Integer>> permute(int[] a) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        int[] pick = new int[a.length];
        cal(a, pick, ds, ans);
        return ans;
    }
}
