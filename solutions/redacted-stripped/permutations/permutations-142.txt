class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        backtrack(nums, new ArrayList<>());
        return result;
    }

    private void backtrack(int[] nums, List<Integer> curr_list) {
        if (curr_list.size() == nums.length) {
            result.add(new ArrayList<>(curr_list));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!curr_list.contains(nums[i])) {
                curr_list.add(nums[i]);
                backtrack(nums, curr_list);
                curr_list.remove(curr_list.size() - 1);
            }
        }
    }
}
