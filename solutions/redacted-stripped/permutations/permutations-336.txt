class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> var1 = new ArrayList<>();
        backtrack(nums, 0, var1);
        return var1;
    }

    private void backtrack(int[] nums, int start, List<List<Integer>> var1) {
        if (start == nums.length) {
            List<Integer> temp = new ArrayList<>();
            for (int num : nums) {
                temp.add(num);
            }
            var1.add(temp);
        } else {
            for (int i = start; i < nums.length; i++) {
                swap(nums, start, i);
                backtrack(nums, start + 1, var1);
                swap(nums, start, i);
            }
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
