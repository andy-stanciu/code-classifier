class Solution {

    public static void generate(int[] arr, ArrayList<Integer> list, ArrayList<List<Integer>> result, boolean[] digit) {
        if (list.size() == arr.length) {
            result.add(new ArrayList<>(list));
            return;
        }
        for (int i = 0; i < arr.length; i++) {
            if (!digit[i]) {
                list.add(arr[i]);
                digit[i] = true;
                generate(arr, list, result, digit);
                digit[i] = false;
                list.remove(list.size() - 1);
            }
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        ArrayList<Integer> list = new ArrayList<Integer>();
        boolean[] digit = new boolean[nums.length];
        ArrayList<List<Integer>> result = new ArrayList<>();
        generate(nums, list, result, digit);
        return result;
    }
}
