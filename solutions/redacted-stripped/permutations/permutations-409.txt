class Solution {

    List<List<Integer>> li = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        myPermutation(nums, 0, nums.length - 1);
        return li;
    }

    private void myPermutation(int[] arr, int start, int end) {
        if (start == end) {
            List<Integer> list = new ArrayList<>();
            for (int x : arr) {
                list.add(x);
            }
            li.add(list);
            return;
        }
        for (int i = start; i <= end; i++) {
            int[] temp = swap(arr, start, i);
            myPermutation(temp, start + 1, end);
        }
    }

    private int[] swap(int[] arr, int i, int j) {
        int[] copy = arr.clone();
        int k = copy[i];
        copy[i] = copy[j];
        copy[j] = k;
        return copy;
    }
}
