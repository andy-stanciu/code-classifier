class Solution {

    public void getanswer(int[] nums, List<List<Integer>> result, List<Integer> pas) {
        if (nums.length == pas.size()) {
            result.add(new ArrayList<>(pas));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (pas.contains(nums[i])) {
                continue;
            }
            pas.add(nums[i]);
            getanswer(nums, result, pas);
            pas.remove((Integer) nums[i]);
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> pas = new ArrayList<>();
        getanswer(nums, result, pas);
        return result;
    }
}
