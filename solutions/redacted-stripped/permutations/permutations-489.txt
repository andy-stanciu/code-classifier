class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> answer = new ArrayList<>();
        shift(nums, 0, answer);
        return answer;
    }

    public void shift(int[] nums, int index, List<List<Integer>> l) {
        if (index == nums.length - 1) {
            List<Integer> list = new ArrayList<Integer>();
            for (int i : nums) list.add(i);
            l.add(list);
            return;
        }
        for (int i = index; i < nums.length; i++) {
            int temp = nums[i];
            nums[i] = nums[index];
            nums[index] = temp;
            shift(nums, index + 1, l);
            int temp1 = nums[i];
            nums[i] = nums[index];
            nums[index] = temp1;
        }
    }
}
