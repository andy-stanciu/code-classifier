class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<Integer> tempList = new ArrayList<>();
        List<List<Integer>> mainList = new ArrayList<>();
        backTracking(tempList, mainList, nums);
        return mainList;
    }

    static void backTracking(List<Integer> tempList, List<List<Integer>> mainList, int[] nums) {
        if (tempList.size() == nums.length) {
            mainList.add(new ArrayList<>(tempList));
            return;
        }
        for (int data : nums) {
            if (tempList.contains(data)) {
                continue;
            }
            tempList.add(data);
            backTracking(tempList, mainList, nums);
            tempList.remove(tempList.size() - 1);
        }
    }
}
