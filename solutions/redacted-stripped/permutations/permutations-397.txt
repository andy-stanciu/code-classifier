class Solution {

    List<List<Integer>> out = null;

    public List<List<Integer>> permute(int[] nums) {
        out = new ArrayList<>();
        helper(nums, new ArrayList<>(), 0, nums.length);
        return out;
    }

    public void helper(int[] nums, List<Integer> curr, int pos, int size) {
        if (curr.size() == size) {
            out.add(new ArrayList<>(curr));
            return;
        }
        for (int i = 0; i < size; i++) {
            if (!curr.contains(nums[i])) {
                curr.add(nums[i]);
                helper(nums, curr, pos + 1, nums.length);
                curr.remove(curr.size() - 1);
            }
        }
    }
}
