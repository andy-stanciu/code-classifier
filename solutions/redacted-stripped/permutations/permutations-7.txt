class Solution {

    public void fun(List<List<Integer>> list, List<Integer> dummy, boolean[] vis, int[] nums) {
        if (dummy.size() == nums.length) {
            list.add(new ArrayList<>(dummy));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!vis[i]) {
                vis[i] = true;
                dummy.add(nums[i]);
                fun(list, dummy, vis, nums);
                dummy.remove(dummy.size() - 1);
                vis[i] = false;
            }
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        boolean[] vis = new boolean[nums.length + 1];
        List<List<Integer>> list = new ArrayList<>();
        fun(list, new ArrayList<>(), vis, nums);
        return list;
    }
}
