class Solution {

    public int[] findOrder(int param1, int[][] prerequisites) {
        int[] visited = new int[param1];
        Stack<Integer> st = new Stack<>();
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < param1; i++) {
            adj.add(new ArrayList<>());
        }
        for (int[] val : prerequisites) {
            adj.get(val[1]).add(val[0]);
        }
        for (int i = 0; i < param1; i++) {
            if (visited[i] == 0) {
                if (!dfs(i, visited, adj, st))
                    return new int[0];
            }
        }
        int[] ans = new int[param1];
        int i = 0;
        while (!st.isEmpty()) {
            ans[i++] = st.pop();
        }
        return ans;
    }

    private boolean dfs(int node, int[] visited, List<List<Integer>> adj, Stack<Integer> st) {
        visited[node] = 1;
        for (int neighbor : adj.get(node)) {
            if (visited[neighbor] == 0) {
                if (!dfs(neighbor, visited, adj, st))
                    return false;
            } else if (visited[neighbor] == 1)
                return false;
        }
        visited[node] = 2;
        st.push(node);
        return true;
    }
}
