class Solution {

    public int[] findOrder(int param1, int[][] prerequisites) {
        int[] indegree = new int[param1];
        List<List<Integer>> adj = new ArrayList<>(param1);
        for (int i = 0; i < param1; i++) adj.add(new ArrayList<>());
        for (int[] p : prerequisites) {
            adj.get(p[1]).add(p[0]);
            indegree[p[0]]++;
        }
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < param1; i++) {
            if (indegree[i] == 0)
                queue.add(i);
        }
        int[] order = new int[param1];
        int index = 0;
        while (!queue.isEmpty()) {
            int current = queue.poll();
            order[index++] = current;
            for (int next : adj.get(current)) {
                if (--indegree[next] == 0)
                    queue.add(next);
            }
        }
        if (index == param1)
            return order;
        else
            return new int[0];
    }
}
