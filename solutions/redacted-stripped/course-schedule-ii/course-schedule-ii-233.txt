class Solution {

    public int[] findOrder(int num, int[][] prerequisites) {
        int[] indegree = new int[num];
        for (int[] arr : prerequisites) {
            indegree[arr[0]]++;
        }
        Queue<Integer> queue = new LinkedList();
        for (int i = 0; i < num; i++) {
            if (indegree[i] == 0) {
                queue.add(i);
            }
        }
        int[] res = new int[num];
        int count = 0;
        while (!queue.isEmpty()) {
            int curr = queue.poll();
            res[count++] = curr;
            for (int[] arr : prerequisites) {
                if (arr[1] == curr) {
                    indegree[arr[0]]--;
                    if (indegree[arr[0]] == 0) {
                        queue.add(arr[0]);
                    }
                }
            }
        }
        return count == num ? res : new int[] {};
    }
}
