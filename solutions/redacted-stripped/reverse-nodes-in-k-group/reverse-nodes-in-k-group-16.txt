class Solution {

    public ListNode method1(ListNode head, int k) {
        int var1 = getLen(head);
        if (var1 < k) {
            return head;
        }
        int count = 0;
        ListNode curr = head;
        ListNode prev = null;
        ListNode next = null;
        while (count < k && curr != null) {
            count++;
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        head.next = method1(curr, k);
        return prev;
    }

    private int getLen(ListNode head) {
        int i = 0;
        while (head != null) {
            i++;
            head = head.next;
        }
        return i;
    }
}
