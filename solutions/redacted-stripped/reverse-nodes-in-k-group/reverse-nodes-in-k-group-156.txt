class Solution {

    public ListNode method1(ListNode head, ListNode stop) {
        ListNode start = head.next;
        ListNode prev = head;
        prev.next = null;
        while (start != stop) {
            ListNode n = start.next;
            start.next = prev;
            prev = start;
            start = n;
        }
        return prev;
    }

    public ListNode method2(ListNode head, int k) {
        ListNode var1 = null;
        ListNode start = null;
        ListNode end = null;
        ListNode prev = null;
        while (head != null) {
            start = head;
            int i = 0;
            for (i = 0; i < k && head != null; i++) {
                head = head.next;
            }
            end = head;
            ListNode rev = null;
            if (i == k) {
                rev = method1(start, end);
            } else {
                rev = start;
            }
            if (var1 == null) {
                var1 = rev;
                prev = start;
            } else {
                prev.next = rev;
                prev = start;
            }
        }
        return var1;
    }
}
