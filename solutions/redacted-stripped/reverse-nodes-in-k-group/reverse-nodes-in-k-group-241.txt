class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head;
        int count = 0;
        while (head != null) {
            head = head.next;
            count++;
        }
        head = temp;
        if (count < k)
            return head;
        ListNode prev = null;
        count = 0;
        while (count < k) {
            count++;
            ListNode next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        temp.next = method1(head, k);
        return prev;
    }
}
