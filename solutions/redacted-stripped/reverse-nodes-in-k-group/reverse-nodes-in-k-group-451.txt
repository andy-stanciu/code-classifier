class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dummy = new ListNode();
        dummy.next = recur(head, k);
        return dummy.next;
    }

    public ListNode recur(ListNode head, int k) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode curr = head;
        ListNode prev = head;
        for (int i = 0; i < k - 1; i++) {
            curr = curr.next;
            if (curr == null) {
                return head;
            }
        }
        prev = curr;
        curr = curr.next;
        prev.next = null;
        ListNode var1 = method2(head);
        head.next = recur(curr, k);
        return var1;
    }

    private ListNode method2(ListNode head) {
        ListNode curr = head;
        if (curr == null || curr.next == null) {
            return curr;
        }
        ListNode prev = method2(curr.next);
        curr.next.next = curr;
        curr.next = null;
        return prev;
    }
}
