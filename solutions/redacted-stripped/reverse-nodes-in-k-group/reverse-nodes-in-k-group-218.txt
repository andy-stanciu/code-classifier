class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode prev = null;
        ListNode pres = head;
        ListNode next = null;
        while (k > 0 && pres != null) {
            next = pres.next;
            pres.next = prev;
            prev = pres;
            pres = next;
            k--;
        }
        return prev;
    }

    public ListNode method2(ListNode head, int k) {
        if (k == 0 || k == 1 || head == null)
            return head;
        ListNode temp = head;
        int sum = 0;
        while (sum < k && temp != null) {
            sum++;
            temp = temp.next;
        }
        if (sum < k)
            return head;
        ListNode j = method1(head, k);
        head.next = method2(temp, k);
        return j;
    }
}
