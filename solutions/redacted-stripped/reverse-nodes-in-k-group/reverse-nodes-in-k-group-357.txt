class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode cur = head;
        ListNode prev = null;
        ListNode next = null;
        if (head == null) {
            return null;
        }
        ListNode cursor = head;
        for (int i = 0; i < k; i++) {
            if (cursor == null) {
                return head;
            }
            cursor = cursor.next;
        }
        int count = 0;
        while (cur != null && count < k) {
            next = cur.next;
            cur.next = prev;
            prev = cur;
            cur = next;
            count++;
        }
        if (cur != null) {
            head.next = method1(next, k);
        }
        return prev;
    }
}
