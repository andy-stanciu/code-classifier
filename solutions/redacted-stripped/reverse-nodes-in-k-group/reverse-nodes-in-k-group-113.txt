public class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k == 1)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode var1 = dummy;
        while (true) {
            ListNode kthNode = var1;
            for (int i = 0; i < k; i++) {
                kthNode = kthNode.next;
                if (kthNode == null) {
                    return dummy.next;
                }
            }
            ListNode var2 = var1.next;
            ListNode var3 = kthNode.next;
            ListNode prev = var3;
            ListNode curr = var2;
            while (curr != var3) {
                ListNode next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            var1.next = kthNode;
            var1 = var2;
        }
    }
}
