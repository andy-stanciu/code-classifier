class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head;
        ListNode nxt = head;
        int count = 0;
        ListNode prev = null;
        while (curr != null && count < k) {
            nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
            count++;
        }
        if (count < k) {
            curr = prev;
            prev = null;
            while (curr != null) {
                nxt = curr.next;
                curr.next = prev;
                prev = curr;
                curr = nxt;
            }
            return head;
        }
        head.next = method1(curr, k);
        return prev;
    }
}
