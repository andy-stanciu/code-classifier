class Solution {

    public ListNode method1(ListNode head, int k) {
        int n = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            n++;
        }
        ListNode dummyNode = new ListNode(0), dummyNodeTail = dummyNode;
        int var1 = n / k;
        while (var1 > 0) {
            ListNode tail = head;
            ListNode prev = null;
            for (int i = 0; i < k; i++) {
                ListNode next = head.next;
                head.next = prev;
                prev = head;
                head = next;
            }
            dummyNodeTail.next = prev;
            dummyNodeTail = tail;
            var1--;
        }
        dummyNodeTail.next = head;
        return dummyNode.next;
    }
}
