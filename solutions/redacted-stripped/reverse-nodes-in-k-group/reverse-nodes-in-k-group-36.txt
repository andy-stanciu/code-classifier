class Solution {

    ListNode method1(ListNode head) {
        if (head.next == null)
            return head;
        ListNode rec = method1(head.next);
        head.next.next = head;
        head.next = null;
        return rec;
    }

    int length(ListNode head) {
        return (head == null) ? 0 : length(head.next) + 1;
    }

    public ListNode method2(ListNode head, int k) {
        int l = length(head);
        if (k == l)
            return method1(head);
        if (k > l)
            return head;
        int i = 1;
        ListNode temp = head;
        while (temp != null) {
            if (i == k) {
                ListNode nexthead = temp.next;
                temp.next = null;
                ListNode rec = method1(head);
                ListNode rec2 = method2(nexthead, k);
                head.next = rec2;
                return rec;
            }
            temp = temp.next;
            i++;
        }
        return null;
    }
}
