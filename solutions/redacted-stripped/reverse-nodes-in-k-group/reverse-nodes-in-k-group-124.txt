class Solution {

    public ListNode method1(ListNode head, int k) {
        int size = 0;
        ListNode si = head;
        while (si != null) {
            size++;
            si = si.next;
        }
        if (k > size) {
            return head;
        }
        if (head == null) {
            return null;
        }
        int count = 0;
        ListNode prev = null;
        ListNode cur = head;
        ListNode next = null;
        while (cur != null && count < k) {
            next = cur.next;
            cur.next = prev;
            prev = cur;
            cur = next;
            count++;
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }
}
