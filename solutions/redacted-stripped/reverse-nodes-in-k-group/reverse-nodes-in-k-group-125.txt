class Solution {

    public ListNode method1(ListNode head, int k) {
        if (k == 1 || head.next == null)
            return head;
        ListNode curr = head;
        int len = 0;
        while (curr != null) {
            len++;
            curr = curr.next;
        }
        ListNode prev = null;
        curr = head;
        int j = 0;
        while ((len - j) >= k) {
            ListNode prevEnd = prev;
            ListNode subListEnd = curr;
            ListNode next = null;
            for (int i = 0; i < k && curr != null; i++) {
                next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            j += k;
            if (prevEnd != null) {
                prevEnd.next = prev;
            } else {
                head = prev;
            }
            subListEnd.next = curr;
            if (curr == null)
                break;
            prev = subListEnd;
        }
        return head;
    }
}
