class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dummy = new ListNode(-1, head);
        ListNode slow = dummy;
        while (head != null) {
            ListNode temp = head;
            int count = k;
            while (count > 0 && head != null) {
                head = head.next;
                count--;
            }
            if (count > 0) {
                slow.next = temp;
            } else {
                slow.next = method2(temp, k);
                slow = temp;
            }
        }
        return dummy.next;
    }

    public ListNode method2(ListNode head, int k) {
        ListNode prev = null;
        ListNode pres = head;
        ListNode temp = null;
        while (k > 0) {
            temp = pres.next;
            pres.next = prev;
            prev = pres;
            pres = temp;
            k--;
        }
        return prev;
    }
}
