class Solution {

    public ListNode method1(ListNode head, int k) {
        int length = 0;
        ListNode current = head;
        while (current != null) {
            length++;
            current = current.next;
        }
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode var1 = dummy;
        while (length >= k) {
            ListNode var2 = var1.next;
            ListNode var3 = var2;
            for (int i = 1; i < k; i++) {
                var3 = var3.next;
            }
            ListNode var4 = var3.next;
            var3.next = null;
            var1.next = method2(var2);
            var2.next = var4;
            var1 = var2;
            length -= k;
        }
        return dummy.next;
    }

    private ListNode method2(ListNode head) {
        ListNode prev = null;
        while (head != null) {
            ListNode next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        return prev;
    }
}
