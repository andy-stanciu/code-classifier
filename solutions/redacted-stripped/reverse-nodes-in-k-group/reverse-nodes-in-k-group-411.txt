class Solution {

    public ListNode method1(ListNode head, int k) {
        int cnt = 0;
        ListNode p = head;
        while (p != null) {
            cnt++;
            p = p.next;
        }
        if (cnt < k) {
            return head;
        }
        ListNode curr = head;
        ListNode prev = null;
        ListNode next = null;
        cnt = 0;
        while (curr != null && cnt < k) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            cnt++;
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }
}
