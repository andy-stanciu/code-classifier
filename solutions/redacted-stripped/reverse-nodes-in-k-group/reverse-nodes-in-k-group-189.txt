class Solution {

    public ListNode method1(ListNode head, int k) {
        int size = 0;
        ListNode dummy = new ListNode();
        dummy.next = head;
        ListNode left = dummy;
        ListNode right = head;
        while (right != null) {
            size++;
            if (size % k == 0) {
                ListNode next = right.next;
                right.next = null;
                ListNode var1 = method2(left.next);
                ListNode var2 = left.next;
                left.next = var1;
                var2.next = next;
                left = var2;
                right = next;
            } else {
                right = right.next;
            }
        }
        return dummy.next;
    }

    private ListNode method2(ListNode head) {
        ListNode current = head;
        ListNode prev = null;
        while (current != null) {
            ListNode next = current.next;
            current.next = prev;
            prev = current;
            current = next;
        }
        return prev;
    }
}
