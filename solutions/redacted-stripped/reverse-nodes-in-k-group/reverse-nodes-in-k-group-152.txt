class Solution {

    public boolean method1(ListNode head, int k) {
        ListNode temp = head;
        for (int i = 0; i < k; i++) {
            if (temp == null)
                return false;
            temp = temp.next;
        }
        return true;
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null)
            return null;
        if (!method1(head, k))
            return head;
        ListNode prev = null, curr = head;
        for (int i = 0; i < k; i++) {
            ListNode temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
        }
        head.next = method2(curr, k);
        return prev;
    }
}
