class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k == 1)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode var1 = dummy;
        while (true) {
            ListNode kth = getKthNode(var1, k);
            if (kth == null) {
                break;
            }
            ListNode var2 = kth.next;
            ListNode prev = var2;
            ListNode curr = var1.next;
            method2(curr, var2);
            ListNode temp = var1.next;
            var1.next = kth;
            var1 = temp;
        }
        return dummy.next;
    }

    private ListNode getKthNode(ListNode start, int k) {
        while (start != null && k > 0) {
            start = start.next;
            k--;
        }
        return start;
    }

    private void method2(ListNode start, ListNode end) {
        ListNode prev = end;
        ListNode curr = start;
        while (curr != end) {
            ListNode temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
        }
    }
}
