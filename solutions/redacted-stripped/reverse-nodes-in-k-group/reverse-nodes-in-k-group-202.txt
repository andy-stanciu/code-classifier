class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null || k == 1)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode curr = dummy, next = dummy, pre = dummy;
        int cnt = 0;
        while (curr.next != null) {
            curr = curr.next;
            cnt++;
        }
        while (cnt >= k) {
            curr = pre.next;
            next = curr.next;
            for (int i = 1; i < k; i++) {
                curr.next = next.next;
                next.next = pre.next;
                pre.next = next;
                next = curr.next;
            }
            pre = curr;
            cnt -= k;
        }
        return dummy.next;
    }
}
