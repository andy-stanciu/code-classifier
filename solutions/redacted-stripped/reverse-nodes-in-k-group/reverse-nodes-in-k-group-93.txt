class Solution {

    int lengthLL(ListNode head) {
        ListNode temp = head;
        int len = 0;
        while (temp != null) {
            temp = temp.next;
            len++;
        }
        return len;
    }

    public ListNode method1(ListNode head, int k) {
        ListNode prev = null, curr = head, nxt = null;
        int t = k;
        if (k > lengthLL(head)) {
            return head;
        }
        while (curr != null && k > 0) {
            nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
            k--;
        }
        head.next = method1(curr, t);
        return prev;
    }
}
