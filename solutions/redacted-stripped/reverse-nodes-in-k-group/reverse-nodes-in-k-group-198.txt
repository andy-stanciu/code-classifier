class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode curr = head;
        for (int i = 0; i < k; i++) {
            if (curr == null) {
                return head;
            }
            curr = curr.next;
        }
        ListNode prev = null, next = null;
        curr = head;
        int c = 0;
        while (c < k && curr != null) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            c++;
        }
        head.next = method1(curr, k);
        return prev;
    }
}
