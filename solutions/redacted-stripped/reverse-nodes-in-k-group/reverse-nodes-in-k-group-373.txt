class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null)
            return null;
        int count = 0;
        ListNode t = head;
        while (t != null) {
            count++;
            t = t.next;
        }
        if (count < k)
            return head;
        int cnt = 0;
        ListNode prev = null;
        ListNode curr = head;
        ListNode temp = null;
        while (cnt < k && curr != null) {
            temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
            cnt++;
        }
        if (curr != null)
            head.next = method1(curr, k);
        return prev;
    }
}
