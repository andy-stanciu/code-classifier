class Solution {

    public void method1(ListNode start) {
        ListNode tmp = start;
        while (tmp != null) {
            System.out.print(tmp.val + " ");
            tmp = tmp.next;
        }
    }

    public ListNode method2(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode rev = method2(head.next);
        head.next.next = head;
        head.next = null;
        return rev;
    }

    public ListNode method3(ListNode head, int k) {
        ListNode iterate = head;
        ListNode var1 = null;
        ListNode getend = null;
        ListNode start = head;
        int klength = 1;
        while (iterate != null) {
            if (klength == k) {
                ListNode end = iterate;
                iterate = iterate.next;
                end.next = null;
                klength = 1;
                ListNode rev = method2(start);
                if (var1 == null) {
                    var1 = rev;
                    getend = start;
                }
                getend.next = end;
                getend = start;
                start = iterate;
                continue;
            }
            iterate = iterate.next;
            klength++;
        }
        getend.next = start;
        return var1;
    }
}
