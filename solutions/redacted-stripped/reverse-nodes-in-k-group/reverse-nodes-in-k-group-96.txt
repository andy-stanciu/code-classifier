class Solution {

    public ListNode method1(ListNode head, int k) {
        if (k == 1)
            return head;
        ListNode curr = head;
        for (int i = 0; i < k; i++) {
            if (head != null) {
                head = head.next;
            } else {
                return curr;
            }
        }
        ListNode var1 = method1(head, k);
        ListNode var2 = curr;
        ListNode prev = curr;
        ListNode tempNext = null;
        for (int i = 0; i < k; i++) {
            tempNext = curr.next;
            curr.next = prev;
            prev = curr;
            curr = tempNext;
        }
        var2.next = var1;
        return prev;
    }
}
