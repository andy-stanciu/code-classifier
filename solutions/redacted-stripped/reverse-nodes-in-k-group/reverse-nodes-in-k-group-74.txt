class Solution {

    public ListNode method1(ListNode head, int k) {
        if (!checkvalid(head, k)) {
            return head;
        }
        ListNode curr = head;
        ListNode prev = null;
        ListNode next = null;
        int count = 0;
        while (count < k && curr != null) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count += 1;
        }
        head.next = method1(next, k);
        return prev;
    }

    public Boolean checkvalid(ListNode head, int k) {
        if (head == null) {
            return false;
        }
        int count = 0;
        ListNode temp = head;
        while (count < k && temp != null) {
            temp = temp.next;
            count += 1;
        }
        return count >= k;
    }
}
