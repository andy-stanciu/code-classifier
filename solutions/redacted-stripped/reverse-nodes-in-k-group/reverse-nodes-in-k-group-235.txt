class Solution {

    private ListNode method1(ListNode head, ListNode tail) {
        if (head == null || head == tail)
            return head;
        ListNode next = head.next;
        ListNode newHead = method1(next, tail);
        next.next = head;
        return newHead;
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null)
            return head;
        ListNode temp = head;
        for (int i = 0; i < k - 1; i++) {
            temp = temp.next;
            if (temp == null)
                return head;
        }
        ListNode next = temp.next;
        ListNode newHead = method1(head, temp);
        head.next = method2(next, k);
        return newHead;
    }
}
