class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dummyNode = new ListNode();
        ListNode dummy = dummyNode;
        ListNode curr = head;
        ListNode st = curr;
        int count = 1;
        while (curr != null) {
            if (count < k) {
                curr = curr.next;
                count++;
            } else {
                ListNode next = curr.next;
                ListNode chead = st;
                curr.next = null;
                ListNode newHead = method2(st);
                dummy.next = newHead;
                dummy = chead;
                st = next;
                count = 1;
                curr = next;
            }
        }
        dummy.next = st;
        return dummyNode.next;
    }

    private ListNode method2(ListNode head) {
        ListNode curr = head;
        ListNode prev = null;
        while (curr != null) {
            ListNode temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
        }
        return prev;
    }
}
