class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            temp = temp.next;
            count++;
            if (count >= k)
                break;
        }
        if (count < k)
            return head;
        ListNode prev = null;
        ListNode curr = head;
        ListNode next = null;
        count = 0;
        while (count < k && curr != null) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        head.next = method1(next, k);
        return prev;
    }
}
