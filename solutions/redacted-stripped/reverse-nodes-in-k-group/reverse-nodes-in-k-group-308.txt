class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null) {
            return null;
        }
        int c = k;
        ListNode cur = head, prev = null, next = null;
        while (cur != null && c > 0) {
            next = cur.next;
            cur.next = prev;
            prev = cur;
            cur = next;
            c--;
        }
        if (cur == null && c > 0) {
            cur = prev;
            prev = null;
            next = null;
            while (cur != null) {
                next = cur.next;
                cur.next = prev;
                prev = cur;
                cur = next;
            }
            if (c > 0) {
                return prev;
            }
        }
        head.next = method1(cur, k);
        return prev;
    }
}
