class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dum = new ListNode(0);
        dum.next = head;
        ListNode var1 = dum;
        while (var1 != null) {
            ListNode node = var1;
            for (int i = 0; i < k && node != null; i++) {
                node = node.next;
            }
            if (node == null)
                break;
            ListNode prev = null, curr = var1.next, next = null;
            for (int i = 0; i < k; i++) {
                next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            ListNode tail = var1.next;
            tail.next = curr;
            var1.next = prev;
            var1 = tail;
        }
        return dum.next;
    }
}
