class Solution {

    public ListNode method1(ListNode head, int k) {
        List<Integer> ls = new ArrayList<>();
        ListNode ph = head;
        while (ph != null) {
            ls.add(ph.val);
            ph = ph.next;
        }
        int i = 0;
        while (i + k <= ls.size()) {
            Collections.reverse(ls.subList(i, i + k));
            i = i + k;
        }
        ListNode res = new ListNode(0), pr = res;
        for (int j : ls) {
            pr.next = new ListNode(j);
            pr = pr.next;
        }
        return res.next;
    }
}
