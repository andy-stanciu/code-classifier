class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode prev = null;
        ListNode current = head;
        int count = 0;
        ListNode next = null;
        while (current != null && count < k) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
            count++;
        }
        if (count < k) {
            current = prev;
            prev = null;
            next = null;
            while (count != 0) {
                count--;
                next = current.next;
                current.next = prev;
                prev = current;
                current = next;
            }
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }
}
