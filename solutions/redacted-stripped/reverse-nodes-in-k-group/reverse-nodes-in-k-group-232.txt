class Solution {

    public ListNode method1(ListNode head, int k) {
        int count = 0;
        int check = 0;
        ListNode temp = head, curr = head, next = null, prev = null;
        while (temp != null && check != k) {
            temp = temp.next;
            check++;
        }
        if (check != k) {
            return head;
        }
        while (curr != null && count < k) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }
}
