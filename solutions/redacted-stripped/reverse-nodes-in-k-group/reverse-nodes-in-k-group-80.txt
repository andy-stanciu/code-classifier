class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode pre = null, curr = head, nexty = head.next;
        while (nexty != null) {
            curr.next = pre;
            pre = curr;
            curr = nexty;
            nexty = nexty.next;
        }
        curr.next = pre;
        return curr;
    }

    public ListNode method2(ListNode head, int k) {
        ListNode dummy = new ListNode(-1);
        dummy.next = head;
        ListNode cur = dummy;
        while (cur != null) {
            ListNode temp = cur;
            for (int i = 0; i < k; i++) {
                temp = temp.next;
                if (temp == null)
                    return dummy.next;
            }
            ListNode nexty = temp.next;
            ListNode last = cur.next;
            temp.next = null;
            cur.next = method1(cur);
            last.next = nexty;
            cur = last;
        }
        return dummy.next;
    }
}
