class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head, newhead = head, prev = head;
        ListNode ans = null;
        int n = k;
        if (n <= 1)
            return head;
        while (temp != null) {
            n = k;
            while (n > 1) {
                temp = temp.next;
                if (temp == null)
                    return head;
                n--;
            }
            ans = method2(newhead, temp);
            if (prev == head) {
                head = ans;
            } else {
                prev.next = ans;
            }
            prev = newhead;
            temp = prev.next;
            newhead = temp;
        }
        return head;
    }

    public ListNode method2(ListNode head, ListNode end) {
        ListNode temp = head;
        ListNode cur = null, nex;
        while (temp != end) {
            nex = temp.next;
            temp.next = cur;
            cur = temp;
            temp = nex;
        }
        head.next = end.next;
        temp.next = cur;
        head = temp;
        return head;
    }
}
