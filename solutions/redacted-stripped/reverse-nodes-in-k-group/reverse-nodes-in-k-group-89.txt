class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode pivNode = null;
        ListNode next = null;
        ListNode currNode = head;
        int count = 0;
        int i = 1;
        ListNode temp = head;
        while (i < k && temp.next != null) {
            temp = temp.next;
            i++;
        }
        if (temp.next == null && i < k)
            return head;
        while (currNode != null && count < k) {
            next = currNode.next;
            currNode.next = pivNode;
            pivNode = currNode;
            currNode = next;
            count++;
        }
        if (next != null)
            head.next = method1(next, k);
        return pivNode;
    }
}
