class Solution {

    public ListNode method1(ListNode head, int k) {
        int var1 = 0;
        ListNode current = head;
        ListNode next = null;
        ListNode prev = null;
        while (current != null && var1 < k) {
            current = current.next;
            var1++;
        }
        current = head;
        if (var1 == k) {
            var1 = 0;
            while (current != null && var1 < k) {
                next = current.next;
                current.next = prev;
                prev = current;
                current = next;
                var1++;
            }
        } else {
            prev = head;
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }
}
