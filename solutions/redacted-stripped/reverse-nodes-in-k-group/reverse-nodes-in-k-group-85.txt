class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dummy = new ListNode(-1, head);
        ListNode var1 = dummy;
        while (true) {
            int kCopy = k;
            ListNode kth = var1;
            while (kth != null && kCopy > 0) {
                kth = kth.next;
                kCopy--;
            }
            if (kth == null)
                break;
            ListNode var2 = kth.next;
            ListNode prev = kth.next;
            ListNode curr = var1.next;
            while (curr != var2) {
                ListNode currNext = curr.next;
                curr.next = prev;
                prev = curr;
                curr = currNext;
            }
            ListNode var3 = var1.next;
            var1.next = kth;
            var1 = var3;
        }
        return dummy.next;
    }
}
