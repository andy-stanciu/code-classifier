class Solution {

    public ListNode method1(ListNode head, int k) {
        int i = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            i++;
        }
        return method2(head, k, i);
    }

    private ListNode method2(ListNode head, int k, int len) {
        if (head == null || head.next == null || len < k) {
            return head;
        }
        ListNode prev = null;
        ListNode var1 = head;
        ListNode cur = var1;
        int i = 0;
        while (i != k) {
            i++;
            ListNode nxt = cur.next;
            cur.next = prev;
            prev = cur;
            cur = nxt;
            if (nxt != null) {
                nxt = nxt.next;
            }
        }
        var1.next = method2(cur, k, len - i);
        return prev;
    }
}
