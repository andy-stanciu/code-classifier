class Solution {

    public ListNode method1(ListNode head, int k) {
        int len = 0;
        ListNode curr = head;
        while (curr != null) {
            curr = curr.next;
            len = len + 1;
        }
        if (len < k)
            return head;
        ListNode prev = null;
        curr = head;
        int i = 0;
        while (i < k) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            i++;
        }
        head.next = method1(curr, k);
        return prev;
    }
}
