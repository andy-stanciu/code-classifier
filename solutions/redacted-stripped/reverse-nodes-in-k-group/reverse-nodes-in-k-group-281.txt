class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k == 1)
            return head;
        ListNode curr = head;
        int count = 0;
        while (curr != null) {
            curr = curr.next;
            count++;
        }
        return method2(head, k, count);
    }

    private static ListNode method2(ListNode head, int k, int count) {
        if (count < k)
            return head;
        ListNode curr = head;
        ListNode prev = null;
        int c = k;
        while (c > 0 && curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            c--;
            count--;
        }
        head.next = method2(curr, k, count);
        return prev;
    }
}
