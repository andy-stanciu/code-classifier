class Solution {

    public ListNode method1(ListNode head, int k) {
        int c = 0;
        ListNode temp = head;
        while (c < k && temp != null) {
            temp = temp.next;
            c++;
        }
        if (c == k) {
            ListNode prev = null;
            ListNode curr = head;
            for (int i = 0; i < k; i++) {
                ListNode nxt = curr.next;
                curr.next = prev;
                prev = curr;
                curr = nxt;
            }
            head.next = method1(curr, k);
            return prev;
        }
        return head;
    }
}
