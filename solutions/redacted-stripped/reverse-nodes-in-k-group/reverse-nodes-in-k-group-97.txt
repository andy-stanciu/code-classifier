class Solution {

    public ListNode method1(ListNode head, int k) {
        if (k == 1 || head == null || head.next == null)
            return head;
        int n = length(head) / k;
        ListNode temp = head;
        ListNode senital = new ListNode(0, head);
        ListNode prev = senital;
        ListNode nxt = head;
        while (n != 0) {
            for (int i = 0; i < k; i++) {
                nxt = nxt.next;
            }
            prev.next = method2(temp, k);
            while (prev.next != null) {
                prev = prev.next;
            }
            prev.next = nxt;
            temp = nxt;
            n--;
        }
        return senital.next;
    }

    public int length(ListNode node) {
        int l = 0;
        while (node != null) {
            l++;
            node = node.next;
        }
        return l;
    }

    public ListNode method2(ListNode node, int k) {
        ListNode prev = null, curr = node;
        int i = 0;
        while (i < k) {
            ListNode temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
            i++;
        }
        return prev;
    }
}
