class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null || k == 1)
            return head;
        ListNode curr = head;
        int count = 0;
        while (curr != null) {
            curr = curr.next;
            count++;
        }
        if (count < k) {
            return head;
        }
        ListNode prev = head;
        curr = prev.next;
        prev.next = null;
        ListNode last = prev;
        count = 1;
        while (curr != null && count < k) {
            ListNode temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
            count++;
        }
        if (curr != null) {
            last.next = method1(curr, k);
        }
        return prev;
    }
}
