class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        int times = count / k;
        return method2(head, k, times);
    }

    public ListNode method2(ListNode head, int k, int t) {
        if (t == 0)
            return head;
        ListNode prev = null;
        ListNode curr = head;
        ListNode nextNode = null;
        for (int i = 0; i < k; i++) {
            nextNode = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nextNode;
        }
        head.next = method2(curr, k, t - 1);
        return prev;
    }
}
