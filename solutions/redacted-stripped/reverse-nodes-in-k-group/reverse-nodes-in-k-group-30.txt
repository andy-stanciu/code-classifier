class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dummy = new ListNode(0);
        ListNode predecessor = dummy;
        ListNode tail = head;
        ListNode successor = null;
        ListNode curr = head;
        for (int i = 1; curr != null; i++) {
            if (i % k == 0) {
                successor = curr.next;
                ListNode var1 = method2(tail, k);
                predecessor.next = var1;
                tail.next = successor;
                predecessor = tail;
                curr = tail = successor;
                continue;
            }
            curr = curr.next;
        }
        return dummy.next;
    }

    private ListNode method2(ListNode head, int k) {
        ListNode prevHead = null;
        ListNode curr = head;
        for (int i = 1; i <= k; i++) {
            ListNode recordNext = curr.next;
            curr.next = prevHead;
            prevHead = curr;
            curr = recordNext;
        }
        return prevHead;
    }
}
