class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head;
        int len = 0;
        while (temp != null) {
            len++;
            temp = temp.next;
        }
        ListNode dummy = new ListNode(-1, head);
        ListNode var1 = dummy;
        while (len >= k) {
            ListNode var2 = var1.next;
            ListNode curr = var2;
            ListNode prev = null;
            for (int i = 0; i < k; i++) {
                ListNode next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            var1.next = prev;
            var2.next = curr;
            var1 = var2;
            len -= k;
        }
        return dummy.next;
    }
}
