class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k <= 1) {
            return head;
        }
        ListNode prev = null;
        ListNode current = head;
        ListNode next = null;
        while (method2(current, k)) {
            ListNode last = prev;
            ListNode newEnd = current;
            for (int i = 0; i < k; i++) {
                next = current.next;
                current.next = prev;
                prev = current;
                current = next;
            }
            if (last != null) {
                last.next = prev;
            } else {
                head = prev;
            }
            newEnd.next = current;
            prev = newEnd;
        }
        return head;
    }

    private boolean method2(ListNode node, int k) {
        while (node != null && k > 0) {
            node = node.next;
            k--;
        }
        return k == 0;
    }
}
