class Solution {

    public ListNode method1(ListNode head, int k) {
        int length = getLength(head);
        if (length < k) {
            return head;
        }
        ListNode current = head;
        ListNode prev = null;
        ListNode next = null;
        int len = 0;
        while (current != null && len < k) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
            len++;
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }

    private static int getLength(ListNode head) {
        ListNode temp = head;
        int n = 0;
        while (temp != null) {
            n++;
            temp = temp.next;
        }
        return n;
    }
}
