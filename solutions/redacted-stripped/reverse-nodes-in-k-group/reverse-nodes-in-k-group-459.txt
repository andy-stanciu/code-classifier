class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode var1 = new ListNode(-1);
        ListNode dummy = var1;
        var1.next = head;
        while (var1.next != null) {
            ListNode var2 = var1.next;
            int count = k;
            ListNode var3 = var2;
            while (count > 1) {
                if (var3.next == null)
                    break;
                var3 = var3.next;
                count--;
            }
            if (count > 1)
                break;
            ListNode var4 = var3.next;
            var3.next = null;
            method2(var2);
            var1.next = var3;
            var2.next = var4;
            var1 = var2;
        }
        return dummy.next;
    }

    public ListNode method2(ListNode head) {
        ListNode prev = null;
        ListNode curr = head;
        while (curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
}
