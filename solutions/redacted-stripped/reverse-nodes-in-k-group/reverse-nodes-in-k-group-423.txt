class Solution {

    public ListNode method1(ListNode start, ListNode end) {
        ListNode prev = null;
        ListNode temp = start;
        while (temp != end) {
            ListNode ret = temp.next;
            temp.next = prev;
            prev = temp;
            temp = ret;
        }
        ListNode nxtnode = end.next;
        end.next = prev;
        return nxtnode;
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null)
            return null;
        ListNode temp = head;
        ListNode start = null, end = null;
        int i = 1;
        start = temp;
        while (temp != null && i < k) {
            i++;
            temp = temp.next;
        }
        if (i == k && temp != null) {
            end = temp;
            temp = method1(start, end);
            start.next = method2(temp, k);
            return end;
        }
        return start;
    }
}
