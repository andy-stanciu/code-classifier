class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null)
            return head;
        int count = 0;
        ListNode temp = null;
        ListNode current = head;
        ListNode prev = null;
        while (current != null && count < k) {
            current = current.next;
            count++;
        }
        current = head;
        if (count == k) {
            count = 0;
            while (current != null && count < k) {
                temp = current.next;
                current.next = prev;
                prev = current;
                current = temp;
                count++;
            }
        } else {
            prev = head;
        }
        if (temp != null) {
            head.next = method1(temp, k);
        }
        return prev;
    }
}
