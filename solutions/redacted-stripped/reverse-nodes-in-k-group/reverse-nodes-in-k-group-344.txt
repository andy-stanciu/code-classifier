class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode var1 = head;
        ListNode iterator = head;
        ListNode dummy = new ListNode(100);
        ListNode dummyIterator = dummy;
        while (iterator != null) {
            var1 = iterator;
            int count = 0;
            ListNode previous = null;
            while (iterator != null && count < k) {
                previous = iterator;
                iterator = iterator.next;
                count++;
            }
            if (count == k) {
                previous.next = null;
                ListNode var2 = method2(var1);
                dummyIterator.next = var2;
                dummyIterator = var1;
            } else {
                dummyIterator.next = var1;
            }
        }
        return dummy.next;
    }

    public ListNode method2(ListNode head) {
        ListNode iterator = head;
        ListNode prev = null;
        while (iterator != null) {
            ListNode next = iterator.next;
            iterator.next = prev;
            prev = iterator;
            iterator = next;
        }
        return prev;
    }
}
