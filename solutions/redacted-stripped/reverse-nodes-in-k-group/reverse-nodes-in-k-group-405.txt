class Solution {

    public ListNode method1(ListNode head, int k) {
        int CNT = 0;
        ListNode temp = head;
        while (temp != null) {
            CNT++;
            temp = temp.next;
        }
        ListNode newNode = null;
        ListNode prevhead = null;
        ListNode currhead = head;
        ListNode prev = null;
        ListNode curr = head;
        while (CNT >= k) {
            int count = 0;
            while (count < k) {
                count++;
                CNT--;
                ListNode next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            if (newNode == null) {
                newNode = prev;
                prevhead = currhead;
                currhead = curr;
            } else {
                prevhead.next = prev;
                prevhead = currhead;
                currhead = curr;
            }
        }
        if (CNT == 0)
            prevhead.next = null;
        else
            prevhead.next = curr;
        return newNode;
    }
}
