class Solution {

    public ListNode method1(ListNode head, int k) {
        int n = 0;
        ListNode c = head;
        while (c != null) {
            c = c.next;
            n++;
        }
        ListNode dummy = new ListNode(0, head), preHead = dummy, curr = head, prev = dummy;
        while (n >= k) {
            for (int i = 0; i < k; i++) {
                ListNode next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            ListNode tail = preHead.next;
            tail.next = curr;
            preHead.next = prev;
            preHead = tail;
            n -= k;
        }
        return dummy.next;
    }
}
