class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode dummy = new ListNode(0, head);
        ListNode p0 = dummy;
        int len = 0;
        while (p0.next != null) {
            len++;
            p0 = p0.next;
        }
        p0 = dummy;
        for (; len >= k; len -= k) {
            ListNode pre = p0;
            ListNode cur = p0.next;
            for (int i = 0; i < k; i++) {
                ListNode next = cur.next;
                cur.next = pre;
                pre = cur;
                cur = next;
            }
            ListNode nxt = p0.next;
            p0.next.next = cur;
            p0.next = pre;
            p0 = nxt;
        }
        return dummy.next;
    }
}
