class Solution {

    public ListNode method1(ListNode head) {
        ListNode prev = null;
        ListNode nxt = null;
        ListNode curr = head;
        while (curr != null) {
            nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
        }
        return prev;
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null)
            return head;
        ListNode temp1 = head;
        ListNode temp2 = head;
        int ctr = k;
        while (ctr != 1 && temp2 != null) {
            temp2 = temp2.next;
            ctr--;
        }
        if (temp2 == null)
            return temp1;
        ListNode connect = temp2.next;
        temp2.next = null;
        ListNode newHead = method1(temp1);
        ListNode node = method2(connect, k);
        temp1.next = node;
        return newHead;
    }
}
