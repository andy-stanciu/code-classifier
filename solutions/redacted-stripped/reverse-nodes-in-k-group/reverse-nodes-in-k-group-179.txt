class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode node = head;
        int count = 0;
        while (node != null && count < k) {
            node = node.next;
            count++;
        }
        if (count < k) {
            return head;
        }
        ListNode prev = null;
        ListNode curr = head;
        ListNode next = null;
        count = 0;
        while (count < k) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        head.next = method1(next, k);
        return prev;
    }
}
