class Solution {

    public ListNode method1(ListNode head, int k) {
        if (k <= 1 || head == null)
            return head;
        ListNode current = head;
        int count = 0;
        while (current != null && count < k) {
            current = current.next;
            count++;
        }
        if (count < k)
            return head;
        ListNode prev = null;
        ListNode next = null;
        current = head;
        for (int i = 0; i < k; i++) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
        }
        if (head != null) {
            head.next = method1(current, k);
        }
        return prev;
    }
}
