class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k <= 1)
            return head;
        ListNode temp = head;
        int size = 0;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        if (size < k)
            return head;
        ListNode future = head;
        ListNode prev = null;
        ListNode curr = head;
        int length = k;
        while (future != null && length > 0) {
            future = future.next;
            curr.next = prev;
            prev = curr;
            curr = future;
            length--;
        }
        if (future != null) {
            ListNode ref = method1(future, k);
            head.next = ref;
        }
        return prev;
    }
}
