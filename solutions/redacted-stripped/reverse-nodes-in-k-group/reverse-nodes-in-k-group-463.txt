class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k <= 1) {
            return head;
        }
        ListNode temp = head;
        int size = 0;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        if (size < k) {
            return head;
        }
        int c = k;
        ListNode curr = head;
        ListNode prev = null;
        ListNode next;
        while (curr != null && c > 0) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            c--;
        }
        if (curr != null) {
            head.next = method1(curr, k);
        }
        return prev;
    }
}
