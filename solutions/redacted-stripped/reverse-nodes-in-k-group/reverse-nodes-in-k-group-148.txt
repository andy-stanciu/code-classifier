class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        if (count < k)
            return head;
        ListNode curr = head, prev = null, nextEle = null;
        int c = 0;
        while (curr != null && c < k) {
            nextEle = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nextEle;
            c++;
        }
        if (curr != null)
            head.next = method1(curr, k);
        return prev;
    }
}
