class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null) {
            return head;
        }
        if (check(head, k)) {
            return head;
        }
        ListNode th = head;
        ListNode dummy = null;
        ListNode t = head;
        int count = k;
        while (count > 0 && head != null) {
            t = head;
            head = head.next;
            t.next = dummy;
            dummy = t;
            count--;
        }
        th.next = head;
        ListNode x = method1(head, k);
        th.next = x;
        return dummy;
    }

    public boolean check(ListNode head, int k) {
        while (k > 0 && head != null) {
            head = head.next;
            k--;
        }
        if (k == 0) {
            return false;
        }
        return true;
    }
}
