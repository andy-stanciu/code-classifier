class Solution {

    public static ListNode method1(ListNode headN, ListNode tailN) {
        ListNode prev = tailN.next;
        ListNode curr = headN;
        while (curr != tailN) {
            ListNode nex = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nex;
        }
        if (curr != null) {
            curr.next = prev;
        }
        return tailN;
    }

    public ListNode method2(ListNode head, int k) {
        ListNode start = head;
        boolean first = true;
        ListNode prevHead = null;
        while (start != null) {
            ListNode headN = start;
            for (int i = 0; start != null && i < k - 1; i++) {
                start = start.next;
            }
            if (start == null)
                break;
            ListNode tailN = start;
            ListNode nex = start.next;
            if (prevHead != null) {
                prevHead.next = tailN;
            }
            if (first == true) {
                head = method1(headN, tailN);
                first = false;
            } else {
                method1(headN, tailN);
            }
            start = nex;
            prevHead = headN;
        }
        return head;
    }
}
