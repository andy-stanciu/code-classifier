class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode ptr = head;
        ListNode temp = head;
        int cnt = k;
        while (temp != null) {
            if (cnt == 0)
                break;
            temp = temp.next;
            cnt--;
        }
        if (cnt > 0)
            return head;
        ptr = temp;
        ListNode prev = null;
        ListNode cur = head;
        temp = head;
        ListNode nextNode = head;
        while (temp != ptr) {
            nextNode = temp.next;
            temp.next = prev;
            prev = temp;
            temp = nextNode;
        }
        ListNode var1 = prev;
        while (ptr != null) {
            ListNode curPtr = ptr;
            cnt = k;
            while (temp != null) {
                if (cnt == 0)
                    break;
                temp = temp.next;
                cnt--;
            }
            if (cnt > 0) {
                cur.next = ptr;
                return var1;
            }
            ptr = temp;
            prev = null;
            ListNode cur1 = curPtr;
            temp = curPtr;
            nextNode = curPtr;
            while (temp != ptr) {
                nextNode = temp.next;
                temp.next = prev;
                prev = temp;
                temp = nextNode;
            }
            cur.next = prev;
            cur = cur1;
        }
        return var1;
    }
}
