class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode check = head;
        for (int i = 0; i < k; i++) {
            if (check == null) {
                return head;
            }
            check = check.next;
        }
        ListNode curr = head, next = null, prev = null;
        int count = 0;
        while (curr != null && count < k) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        if (next != null) {
            ListNode restHead = method1(next, k);
            head.next = restHead;
        }
        return prev;
    }
}
