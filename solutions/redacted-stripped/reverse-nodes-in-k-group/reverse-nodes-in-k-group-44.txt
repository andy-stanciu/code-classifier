class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null)
            return null;
        int Listlength = 1;
        ListNode temp = head;
        while (temp.next != null) {
            Listlength++;
            temp = temp.next;
        }
        if (Listlength < k)
            return head;
        int var1 = k;
        ListNode nextHead = head;
        ListNode var2 = null;
        while (nextHead != null && var1-- > 0) {
            var2 = nextHead;
            nextHead = nextHead.next;
        }
        if (var2 != null)
            var2.next = null;
        method2(head);
        ListNode resolvedList = method1(nextHead, k);
        head.next = resolvedList;
        return var2;
    }

    public void method2(ListNode head) {
        ListNode temp = head;
        ListNode prev = null;
        while (temp != null) {
            head = temp.next;
            temp.next = prev;
            prev = temp;
            temp = head;
        }
    }
}
