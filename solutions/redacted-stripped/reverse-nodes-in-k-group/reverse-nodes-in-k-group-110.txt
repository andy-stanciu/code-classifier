class Solution {

    public int method1(ListNode head) {
        if (head == null) {
            return 0;
        }
        return 1 + method1(head.next);
    }

    public void method2(ListNode prev, int k) {
        ListNode curr = prev.next;
        ListNode forward = curr.next;
        for (int i = 0; i < k - 1; i++) {
            curr.next = forward.next;
            forward.next = prev.next;
            prev.next = forward;
            forward = curr.next;
        }
    }

    public ListNode method3(ListNode head, int k) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy;
        int num = method1(head);
        int loop = num / k;
        while (loop > 0) {
            method2(prev, k);
            for (int i = 0; i < k; i++) {
                prev = prev.next;
            }
            loop--;
        }
        return dummy.next;
    }
}
