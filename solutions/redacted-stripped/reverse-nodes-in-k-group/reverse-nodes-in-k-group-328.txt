class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null)
            return head;
        ListNode current = head;
        ListNode prev = null;
        int count = 0;
        while (current != null && count < k) {
            current = current.next;
            count++;
        }
        if (count == k) {
            current = head;
            ListNode next = null;
            int i = 0;
            while (i < k && current != null) {
                next = current.next;
                current.next = prev;
                prev = current;
                current = next;
                i++;
            }
            ListNode restNode = method1(current, k);
            head.next = restNode;
            return prev;
        } else {
            return head;
        }
    }
}
