class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode prev = null;
        ListNode current = head;
        ListNode next = null;
        while (k > 0) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
            k--;
        }
        return prev;
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null) {
            return null;
        }
        ListNode temp = head;
        int count = 0;
        while (temp != null && count < k) {
            temp = temp.next;
            count++;
        }
        if (count == k) {
            ListNode newHead = method1(head, k);
            head.next = method2(temp, k);
            return newHead;
        }
        return head;
    }
}
