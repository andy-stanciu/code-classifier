class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null)
            return head;
        if (k == 1)
            return head;
        ListNode root = head;
        int count = 0;
        while (count < k && root != null) {
            count++;
            root = root.next;
        }
        if (count < k)
            return head;
        root = head;
        count = 0;
        ListNode p = null;
        ListNode c = root;
        ListNode n = root.next;
        while (count < k) {
            c.next = p;
            count++;
            if (count < k) {
                p = c;
                c = n;
                n = c.next;
            }
        }
        root.next = method1(n, k);
        return c;
    }
}
