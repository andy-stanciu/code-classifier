class Solution {

    public int len(ListNode head) {
        int counter = 0;
        while (head != null) {
            counter++;
            head = head.next;
        }
        return counter;
    }

    public ListNode method1(ListNode head, int k) {
        int N = len(head);
        int var1 = N / k;
        ListNode prevHead = null;
        ListNode currHead = head;
        ListNode ansNode = null;
        for (int i = 0; i < var1; i++) {
            ListNode prev = null;
            ListNode curr = currHead;
            ListNode temp = null;
            for (int j = 0; j < k; j++) {
                temp = curr.next;
                curr.next = prev;
                prev = curr;
                curr = temp;
            }
            if (prevHead == null) {
                ansNode = prev;
            } else {
                prevHead.next = prev;
            }
            prevHead = currHead;
            currHead = curr;
        }
        prevHead.next = currHead;
        return ansNode;
    }
}
