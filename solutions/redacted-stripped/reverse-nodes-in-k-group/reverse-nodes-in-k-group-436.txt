class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode curr = head;
        int n = 1;
        while (curr.next != null) {
            curr = curr.next;
            n++;
        }
        if (head == null || head.next == null || n < k) {
            return head;
        }
        curr = head;
        ListNode prev = null;
        ListNode nextNode = null;
        int count = 0;
        while (count < k) {
            nextNode = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nextNode;
            count++;
        }
        if (curr != null) {
            head.next = method1(curr, k);
        }
        return prev;
    }
}
