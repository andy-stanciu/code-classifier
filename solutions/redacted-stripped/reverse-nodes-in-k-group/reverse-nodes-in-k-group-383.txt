class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode start = head, end = head;
        int count = 1;
        while (end != null && count != k) {
            count++;
            end = end.next;
        }
        if (end == null) {
            return head;
        } else {
            ListNode ans = method1(end.next, k);
            return method2(start, end, ans, k);
        }
    }

    public ListNode method2(ListNode start, ListNode end, ListNode prev, int k) {
        ListNode temp = start, next = start;
        int count = 1;
        while (count <= k) {
            next = temp.next;
            temp.next = prev;
            prev = temp;
            temp = next;
            count++;
        }
        return prev;
    }
}
