class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || k == 1)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode ptr = dummy;
        while (ptr != null) {
            ListNode tracker = ptr;
            for (int i = 0; i < k && tracker != null; i++) {
                tracker = tracker.next;
            }
            if (tracker == null) {
                break;
            }
            ListNode[] var1 = method2(ptr.next, k);
            ListNode previous = var1[0];
            ListNode current = var1[1];
            ListNode var2 = ptr.next;
            var2.next = current;
            ptr.next = previous;
            ptr = var2;
        }
        return dummy.next;
    }

    private ListNode[] method2(ListNode head, int k) {
        ListNode previous = null;
        ListNode current = head;
        ListNode next = null;
        for (int i = 0; i < k; i++) {
            next = current.next;
            current.next = previous;
            previous = current;
            current = next;
        }
        return new ListNode[] { previous, current };
    }
}
