class Solution {

    public ListNode method1(ListNode head, int k) {
        if (k == 1 || head == null)
            return head;
        ListNode temp = head;
        int i = 0;
        while (i < k && temp != null) {
            i++;
            temp = temp.next;
        }
        if (i != k)
            return head;
        Stack<ListNode> st = new Stack<ListNode>();
        i = 0;
        ListNode it = head;
        while (i < k) {
            i++;
            st.push(it);
            it = it.next;
        }
        head = st.pop();
        it = head;
        while (!st.isEmpty()) {
            it.next = st.pop();
            it = it.next;
        }
        it.next = method1(temp, k);
        return head;
    }
}
