class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null && head.next == null)
            return head;
        ListNode temp = new ListNode(0);
        temp.next = head;
        ListNode prev = temp, curr = head, front;
        int size = 0;
        while (curr != null) {
            curr = curr.next;
            size++;
        }
        while (size >= k) {
            curr = prev.next;
            front = curr.next;
            for (int i = 1; curr.next != null && i < k; i++) {
                curr.next = front.next;
                front.next = prev.next;
                prev.next = front;
                front = curr.next;
            }
            prev = curr;
            size -= k;
        }
        return temp.next;
    }
}
