class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode fast = head;
        int current = k;
        while (fast != null && current > 1) {
            fast = fast.next;
            current--;
        }
        if (current == 1 && fast != null) {
            ListNode var1 = method1(fast.next, k);
            method2(head, k);
            head.next = var1;
            head = fast;
        }
        return head;
    }

    void method2(ListNode head, int k) {
        if (k == 1)
            return;
        method2(head.next, k - 1);
        head.next.next = head;
        head.next = null;
    }
}
