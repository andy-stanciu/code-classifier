class Solution {

    public static int length(ListNode head) {
        int count = 0;
        while (head != null) {
            count++;
            head = head.next;
        }
        return count;
    }

    public ListNode method1(ListNode head, int k) {
        int n = length(head);
        int var1 = n / k;
        ListNode preHead = null;
        ListNode curHead = head;
        ListNode ansNode = null;
        for (int i = 0; i < var1; i++) {
            ListNode pre = null;
            ListNode cur = curHead;
            ListNode nextnode = null;
            for (int j = 0; j < k; j++) {
                nextnode = cur.next;
                cur.next = pre;
                pre = cur;
                cur = nextnode;
            }
            if (preHead == null) {
                ansNode = pre;
            } else
                preHead.next = pre;
            preHead = curHead;
            curHead = cur;
        }
        preHead.next = curHead;
        return ansNode;
    }
}
