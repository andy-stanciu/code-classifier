class Solution {

    public ListNode method1(ListNode head, int k) {
        int length = length(head);
        ListNode dummy = new ListNode(0, head);
        int count = 0;
        ListNode currTailAfterReversal = head;
        ListNode var1 = dummy;
        ListNode curr = head, prev = null, next = null;
        while (curr != null) {
            currTailAfterReversal = curr;
            prev = null;
            next = null;
            for (int i = 0; i < k; i++) {
                next = curr.next;
                curr.next = prev;
                prev = curr;
                curr = next;
            }
            count += k;
            var1.next = prev;
            var1 = currTailAfterReversal;
            if (length - count < k) {
                var1.next = curr;
                break;
            }
        }
        return dummy.next;
    }

    private int length(ListNode head) {
        ListNode ptr = head;
        int count = 0;
        while (ptr != null) {
            count++;
            ptr = ptr.next;
        }
        return count;
    }
}
