class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null)
            return null;
        ListNode curr = head, prev = null, temp = null;
        int c = k;
        while (curr != null && c > 0) {
            temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
            c--;
        }
        if (curr == null && c > 0) {
            curr = prev;
            prev = null;
            temp = null;
            while (curr != null) {
                temp = curr.next;
                curr.next = prev;
                prev = curr;
                curr = temp;
            }
        }
        if (c > 0) {
            return prev;
        } else {
            head.next = method1(curr, k);
        }
        return prev;
    }
}
