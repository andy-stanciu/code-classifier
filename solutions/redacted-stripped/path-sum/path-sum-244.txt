class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null && param1 == 0)
            return false;
        return hf(root, param1);
    }

    public boolean hf(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            return root.val == param1;
        }
        return hf(root.left, param1 - root.val) || hf(root.right, param1 - root.val);
    }
}
