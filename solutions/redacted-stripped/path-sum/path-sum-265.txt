class Solution {

    public boolean method1(TreeNode root, int target) {
        if (root == null)
            return false;
        target -= root.val;
        if (target == 0 && root.left == null && root.right == null)
            return true;
        boolean left = method1(root.left, target);
        boolean right = method1(root.right, target);
        return (left || right);
    }
}
