class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.val == param1 && root.right == null && root.left == null)
            return true;
        if (root.left != null)
            root.left.val += root.val;
        if (root.right != null)
            root.right.val += root.val;
        return method1(root.left, param1) || method1(root.right, param1);
    }
}
