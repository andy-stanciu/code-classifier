class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method1(root, param1, 0);
    }

    private boolean method1(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        param2 += root.val;
        if (param2 == param1 && root.left == null && root.right == null)
            return true;
        return method1(root.left, param1, param2) || method1(root.right, param1, param2);
    }
}
