class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, param1, root.val);
    }

    public boolean method2(TreeNode root, int param1, int param2) {
        boolean flag1 = false, flag2 = false;
        if (root.right == null && root.left == null && param2 == param1)
            return true;
        if (root.right != null)
            flag1 = method2(root.right, param1, param2 + root.right.val);
        if (root.left != null)
            flag2 = method2(root.left, param1, param2 + root.left.val);
        return flag1 || flag2;
    }
}
