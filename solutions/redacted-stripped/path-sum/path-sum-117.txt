class Solution {

    public boolean method1(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == param2) {
                return true;
            }
        }
        boolean a = method1(root.left, param1, param2);
        boolean b = method1(root.right, param1, param2);
        return a || b;
    }

    public boolean method2(TreeNode root, int param3) {
        return method1(root, 0, param3);
    }
}
