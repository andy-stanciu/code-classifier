class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        int rem = param1 - root.val;
        if (root.right == null && root.left == null) {
            return rem == 0;
        }
        return method1(root.left, rem) || method1(root.right, rem);
    }
}
