class Solution {

    public boolean method1(TreeNode root, int param1) {
        return traverse(root, param1);
    }

    public boolean traverse(TreeNode ptr, int target) {
        if (ptr == null) {
            return false;
        }
        if (ptr.left == null && ptr.right == null && ptr.val == target) {
            return true;
        }
        int newTarget = target - ptr.val;
        return traverse(ptr.left, newTarget) || traverse(ptr.right, newTarget);
    }
}
