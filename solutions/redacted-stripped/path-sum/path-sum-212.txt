class Solution {

    public boolean method1(TreeNode root, int param1) {
        int check = 0;
        Stack<TreeNode> stack = new Stack<>();
        TreeNode last = null;
        while (!stack.isEmpty() || root != null) {
            if (root != null) {
                stack.add(root);
                check += root.val;
                root = root.left;
            } else {
                TreeNode peeked = stack.peek();
                if (peeked.right != null && last != peeked.right) {
                    root = peeked.right;
                } else {
                    if (param1 == check) {
                        if (peeked.right == null && peeked.left == null) {
                            return true;
                        }
                    }
                    check -= peeked.val;
                    last = stack.pop();
                }
            }
        }
        return false;
    }
}
