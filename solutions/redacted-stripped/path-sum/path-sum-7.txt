class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1);
    }

    public boolean method2(TreeNode root, int currentBal) {
        if (root == null) {
            return false;
        }
        currentBal -= root.val;
        if (root.left == null && root.right == null && currentBal == 0) {
            return true;
        }
        return method2(root.left, currentBal) || method2(root.right, currentBal);
    }
}
