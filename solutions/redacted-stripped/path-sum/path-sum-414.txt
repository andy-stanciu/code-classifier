class Solution {

    boolean result = false;

    int target;

    public void traverse(TreeNode root, int param1) {
        if (root == null) {
            return;
        }
        param1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == target) {
                result = true;
            }
        }
        traverse(root.left, param1);
        traverse(root.right, param1);
    }

    public boolean method1(TreeNode root, int param2) {
        target = param2;
        traverse(root, 0);
        return result;
    }
}
