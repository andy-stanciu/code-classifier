class Solution {

    public boolean method1(TreeNode root, int param1) {
        return dfs(root, 0, param1);
    }

    private boolean dfs(TreeNode root, int param2, int param1) {
        if (root == null)
            return false;
        param2 += root.val;
        if (root.left == null && root.right == null)
            if (param2 == param1)
                return true;
        return dfs(root.left, param2, param1) || dfs(root.right, param2, param1);
    }
}
