class Solution {

    public boolean method1(TreeNode root, int param1, int count) {
        if (root == null) {
            return false;
        }
        count += root.val;
        if (root.left == null && root.right == null) {
            return count == param1;
        }
        return method1(root.left, param1, count) || method1(root.right, param1, count);
    }

    public boolean method2(TreeNode root, int param1) {
        if (root == null && param1 == 0) {
            return false;
        }
        int count = 0;
        return method1(root, param1, count);
    }
}
