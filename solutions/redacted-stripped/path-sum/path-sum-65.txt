class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        int temp = param1 - root.val;
        if (temp == 0 && root.left == null && root.right == null) {
            return true;
        }
        return method1(root.left, temp) || method1(root.right, temp);
    }
}
