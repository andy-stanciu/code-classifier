class Solution {

    public boolean method1(TreeNode root, int param1) {
        return targetMet(root, param1);
    }

    boolean targetMet(TreeNode root, int param1) {
        if (root == null)
            return false;
        param1 -= root.val;
        if (root.left == null && root.right == null)
            return param1 == 0;
        return targetMet(root.left, param1) || targetMet(root.right, param1);
    }
}
