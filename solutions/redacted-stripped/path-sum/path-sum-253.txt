class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        if (root == null) {
            return false;
        }
        var1 += root.val;
        if (var1 == param1 && root.left == null && root.right == null) {
            return true;
        }
        int var2 = param1 - root.val;
        return method1(root.left, var2) || method1(root.right, var2);
    }
}
