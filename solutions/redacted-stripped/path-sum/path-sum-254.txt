class Solution {

    public boolean method1(TreeNode root, int param1) {
        return recursive(root, param1);
    }

    boolean recursive(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if ((param1 - root.val) == 0) {
                return true;
            }
            return false;
        }
        return recursive(root.left, param1 - root.val) || recursive(root.right, param1 - root.val);
    }
}
