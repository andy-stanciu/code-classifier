class Solution {

    private boolean res(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && param1 == param2 + root.val) {
            return true;
        }
        return res(root.left, param1, param2 + root.val) || res(root.right, param1, param2 + root.val);
    }

    public boolean method1(TreeNode root, int param1) {
        return res(root, param1, 0);
    }
}
