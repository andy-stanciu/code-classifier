class Solution {

    public boolean method1(TreeNode root, int param1) {
        boolean flag = false;
        int target = param1;
        if (root == null)
            return false;
        flag = answer(root, 0, target, false);
        if (flag == true)
            return true;
        return false;
    }

    public boolean answer(TreeNode root, int param2, int target, boolean flag) {
        if (root == null) {
            return false;
        }
        if (root.right == null && root.left == null) {
            param2 = param2 + root.val;
            if (param2 == target) {
                System.out.println("true");
                return true;
            }
            return false;
        } else {
            param2 = param2 + root.val;
            return answer(root.left, param2, target, flag) | answer(root.right, param2, target, flag);
        }
    }
}
