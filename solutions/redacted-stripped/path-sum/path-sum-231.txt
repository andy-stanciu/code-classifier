class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, param1);
    }

    static boolean helper(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (param1 == root.val) {
                return true;
            }
        }
        int var1 = param1 - root.val;
        return helper(root.left, var1) || helper(root.right, var1);
    }
}
