class Solution {

    public boolean solve(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            return param1 + root.val == param2;
        }
        boolean var1 = solve(root.left, param1 + root.val, param2);
        boolean var2 = solve(root.right, param1 + root.val, param2);
        return var1 || var2;
    }

    public boolean method1(TreeNode root, int param2) {
        return solve(root, 0, param2);
    }
}
