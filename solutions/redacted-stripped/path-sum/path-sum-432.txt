class Solution {

    public boolean method1(TreeNode root, int param1) {
        List<Boolean> ds = new ArrayList<>();
        traverse(root, param1, 0, ds);
        if (ds.isEmpty()) {
            return false;
        }
        return ds.get(0);
    }

    public static void traverse(TreeNode root, int param1, int val, List<Boolean> ds) {
        if (root == null) {
            return;
        }
        val = val + root.val;
        traverse(root.left, param1, val, ds);
        if (root.left == null && root.right == null) {
            if (val == param1) {
                ds.add(true);
                return;
            }
        }
        traverse(root.right, param1, val, ds);
        val -= root.val;
    }
}
