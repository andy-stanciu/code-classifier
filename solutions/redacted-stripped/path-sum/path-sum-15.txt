class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return check(root, 0, param1);
    }

    private boolean check(TreeNode root, int param2, int param1) {
        if (root == null)
            return false;
        param2 = param2 + root.val;
        if (param2 == param1 && root.left == null && root.right == null)
            return true;
        return check(root.left, param2, param1) || check(root.right, param2, param1);
    }
}
