class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return checkThis(var1, param1, root);
    }

    private boolean checkThis(int var1, int param1, TreeNode root) {
        if (root == null) {
            return false;
        }
        var1 += root.val;
        if (root.left == null && root.right == null) {
            return var1 == param1;
        }
        return checkThis(var1, param1, root.left) || checkThis(var1, param1, root.right);
    }
}
