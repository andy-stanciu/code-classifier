class Solution {

    boolean bMatchFound = false;

    public boolean method1(TreeNode root, int param1) {
        traverse(root, param1, 0);
        return bMatchFound;
    }

    public void traverse(TreeNode root, int param1, int param2) {
        if (root == null)
            return;
        if (!bMatchFound && root.left == null && root.right == null && root.val + param2 == param1) {
            bMatchFound = true;
            return;
        } else {
            traverse(root.left, param1, root.val + param2);
            traverse(root.right, param1, root.val + param2);
        }
    }
}
