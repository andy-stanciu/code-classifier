class Solution {

    public boolean method1(int param1, TreeNode root, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && param1 + root.val == param2) {
            return true;
        }
        return method1(param1 + root.val, root.left, param2) || method1(param1 + root.val, root.right, param2);
    }

    public boolean method2(TreeNode root, int param2) {
        return method1(0, root, param2);
    }
}
