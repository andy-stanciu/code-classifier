class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (param1 == root.val && root.left == null && root.right == null)
            return true;
        boolean l1 = method1(root.left, param1 - root.val);
        boolean l2 = method1(root.right, param1 - root.val);
        return (l1 || l2);
    }
}
