class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, 0, param1);
    }

    private boolean method2(TreeNode root, int param2, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null && param1 == param2 + root.val)
            return true;
        return method2(root.left, param2 + root.val, param1) || method2(root.right, param2 + root.val, param1);
    }
}
