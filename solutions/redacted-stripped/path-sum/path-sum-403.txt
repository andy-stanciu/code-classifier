class Solution {

    public boolean ans = false;

    public void solve(TreeNode root, int param1, int target) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            param1 = param1 + root.val;
            if (param1 == target) {
                ans = true;
            }
            return;
        }
        solve(root.left, param1 + root.val, target);
        solve(root.right, param1 + root.val, target);
    }

    public boolean method1(TreeNode root, int param2) {
        solve(root, 0, param2);
        return ans;
    }
}
