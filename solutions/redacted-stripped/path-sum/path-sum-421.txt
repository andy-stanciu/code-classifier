class Solution {

    public boolean method1(TreeNode node, int param1) {
        if (node == null)
            return false;
        boolean ans = false;
        int var1 = param1 - node.val;
        if (var1 == 0 && node.left == null && node.right == null)
            return (ans = true);
        if (node.left != null)
            ans = ans || method1(node.left, var1);
        if (node.right != null)
            ans = ans || method1(node.right, var1);
        return (ans);
    }
}
