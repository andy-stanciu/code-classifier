class Solution {

    public boolean helper(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param1 = (param1 + root.val);
        if (root.left == null && root.right == null) {
            if (param1 == param2) {
                return true;
            }
            return false;
        }
        boolean left = helper(root.left, param1, param2);
        boolean right = helper(root.right, param1, param2);
        return (left || right);
    }

    public boolean method1(TreeNode root, int param2) {
        return helper(root, 0, param2);
    }
}
