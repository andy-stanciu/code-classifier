class Solution {

    public boolean method1(TreeNode node, int target, int param1) {
        if (node == null) {
            return false;
        }
        param1 += node.val;
        if (node.left == null && node.right == null && param1 == target) {
            return true;
        }
        boolean b = method1(node.left, target, param1);
        boolean a = method1(node.right, target, param1);
        return a || b;
    }

    public boolean method2(TreeNode root, int param2) {
        return method1(root, param2, 0);
    }
}
