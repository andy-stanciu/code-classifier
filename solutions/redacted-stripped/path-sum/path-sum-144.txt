class Solution {

    boolean help(TreeNode node, int target, int param1) {
        if (node == null)
            return false;
        if (target == param1 + node.val && node.left == null && node.right == null) {
            return true;
        }
        return help(node.left, target, param1 + node.val) || help(node.right, target, param1 + node.val);
    }

    public boolean method1(TreeNode root, int param2) {
        return help(root, param2, 0);
    }
}
