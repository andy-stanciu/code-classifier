class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return PreOrder(root, 0, param1);
    }

    public boolean PreOrder(TreeNode root, int cal, int param1) {
        if (root == null) {
            return false;
        }
        cal += root.val;
        if (root.left == null && root.right == null && cal == param1) {
            return true;
        }
        return PreOrder(root.left, cal, param1) || PreOrder(root.right, cal, param1);
    }
}
