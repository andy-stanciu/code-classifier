class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return traverse(root, 0, param1);
    }

    private boolean traverse(TreeNode node, int param2, int target) {
        if (node == null) {
            return false;
        }
        param2 += node.val;
        if (node.left == null && node.right == null) {
            if (param2 == target)
                return true;
            else
                return false;
        }
        return (traverse(node.left, param2, target) || traverse(node.right, param2, target));
    }
}
