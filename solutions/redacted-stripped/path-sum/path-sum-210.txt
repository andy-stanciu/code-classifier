class Solution {

    private boolean helper(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param1 += root.val;
        if (root.left == null && root.right == null) {
            return param1 == param2;
        }
        return helper(root.left, param1, param2) || helper(root.right, param1, param2);
    }

    public boolean method1(TreeNode root, int param2) {
        if (root == null)
            return false;
        return helper(root, 0, param2);
    }
}
