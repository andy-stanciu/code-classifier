class Solution {

    public boolean method1(TreeNode root, int param1) {
        return solve(root, param1, 0);
    }

    private boolean solve(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param2 = param2 + root.val;
        if (param2 == param1 && (root.left == null && root.right == null)) {
            return true;
        }
        boolean left = solve(root.left, param1, param2);
        boolean right = solve(root.right, param1, param2);
        if (left == true || right == true) {
            return true;
        }
        return false;
    }
}
