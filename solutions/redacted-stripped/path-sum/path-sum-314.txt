class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return find(root, param1);
    }

    boolean find(TreeNode node, int f) {
        if (node == null) {
            return false;
        }
        if (node.val == f && node.left == null && node.right == null) {
            return true;
        }
        return find(node.right, f - node.val) || find(node.left, f - node.val);
    }
}
