class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1);
    }

    public boolean method2(TreeNode root, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            return root.val == param2;
        }
        int var1 = param2 - root.val;
        return method2(root.left, var1) || method2(root.right, var1);
    }
}
