class Solution {

    public boolean method1(TreeNode root, int ts, int s) {
        if (root == null) {
            return false;
        }
        if (s + root.val == ts) {
            if (root.left == null && root.right == null) {
                return true;
            }
        }
        boolean lt = method1(root.left, ts, s + root.val);
        boolean rt = method1(root.right, ts, s + root.val);
        if (lt || rt) {
            return true;
        }
        return false;
    }

    public boolean method2(TreeNode root, int param1) {
        return method1(root, param1, 0);
    }
}
