class Solution {

    int target;

    public boolean method1(TreeNode root, int param1) {
        target = param1;
        return dfs(root, 0);
    }

    public boolean dfs(TreeNode node, int param2) {
        if (node == null) {
            return false;
        }
        if (node.left == null && node.right == null) {
            return target == param2 + node.val;
        }
        boolean left = dfs(node.left, param2 + node.val);
        boolean right = dfs(node.right, param2 + node.val);
        return left || right;
    }
}
