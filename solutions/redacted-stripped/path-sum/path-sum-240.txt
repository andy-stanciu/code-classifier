class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.right == null && root.left == null) {
            if (root.val == param1)
                return true;
        }
        int value = root.val;
        boolean left = method1(root.left, param1 - value);
        boolean right = method1(root.right, param1 - value);
        if (left == true || right == true)
            return true;
        else
            return false;
    }
}
