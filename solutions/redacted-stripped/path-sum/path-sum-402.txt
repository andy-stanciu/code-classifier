class Solution {

    boolean var1 = false;

    public void method1(TreeNode root, int param1) {
        if (root == null)
            return;
        if (param1 - root.val == 0 && root.left == null && root.right == null)
            var1 = true;
        method1(root.left, param1 - root.val);
        method1(root.right, param1 - root.val);
    }

    public boolean method2(TreeNode root, int method1) {
        method1(root, method1);
        return var1;
    }
}
