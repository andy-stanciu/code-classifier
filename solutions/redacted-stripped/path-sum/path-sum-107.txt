class Solution {

    boolean f = false;

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        s(root, param1);
        return f;
    }

    public void s(TreeNode root, int t) {
        if (root == null)
            return;
        t -= root.val;
        if (root.left == null && root.right == null) {
            if (t == 0)
                f = true;
            return;
        }
        s(root.left, t);
        s(root.right, t);
    }
}
