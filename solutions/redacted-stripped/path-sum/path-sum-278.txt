class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null)
            return root.val == param1;
        boolean lhs = method1(root.left, param1 - root.val);
        boolean rhs = method1(root.right, param1 - root.val);
        if (lhs || rhs) {
            return true;
        }
        return false;
    }
}
