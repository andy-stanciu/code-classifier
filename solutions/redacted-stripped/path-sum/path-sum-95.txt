class Solution {

    public boolean method1(TreeNode root, int param1) {
        return solve(root, param1, 0);
    }

    private boolean solve(TreeNode root, int target, int curr) {
        if (root == null) {
            return false;
        }
        curr += root.val;
        if (root.left == null && root.right == null && curr == target) {
            return true;
        }
        return solve(root.left, target, curr) || solve(root.right, target, curr);
    }
}
