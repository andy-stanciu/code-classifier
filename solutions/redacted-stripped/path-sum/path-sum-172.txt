class Solution {

    public boolean method1(TreeNode root, int param1) {
        return test(root, 0, param1);
    }

    boolean test(TreeNode node, int param2, int target) {
        if (null == node)
            return false;
        var current = param2 + node.val;
        if (target == current && node.left == null && node.right == null) {
            return true;
        }
        return test(node.left, current, target) || test(node.right, current, target);
    }
}
