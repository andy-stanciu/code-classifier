class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return method2(root, 0, param1);
    }

    public boolean method2(TreeNode node, int param2, int param1) {
        if (node.left == null && node.right == null) {
            if (param2 + node.val == param1) {
                return true;
            }
        }
        if (node.left != null) {
            boolean possible = method2(node.left, param2 + node.val, param1);
            if (possible) {
                return true;
            }
        }
        if (node.right != null) {
            boolean possible = method2(node.right, param2 + node.val, param1);
            if (possible) {
                return true;
            }
        }
        return false;
    }
}
