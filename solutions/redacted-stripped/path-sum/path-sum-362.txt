class Solution {

    int count = 0;

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        count += root.val;
        if (root.left == null && root.right == null && param1 == count) {
            return true;
        }
        boolean leftFlag = method1(root.left, param1);
        boolean rightFlag = method1(root.right, param1);
        if (leftFlag || rightFlag) {
            return true;
        } else {
            count = count - root.val;
        }
        return false;
    }
}
