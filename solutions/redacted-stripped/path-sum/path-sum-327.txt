class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        param1 -= root.val;
        if (root.left == null && root.right == null) {
            if (param1 == 0) {
                return true;
            } else {
                return false;
            }
        }
        boolean left = method1(root.left, param1);
        boolean right = method1(root.right, param1);
        return left || right;
    }
}
