class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        boolean left = false, right = false;
        if (root.left != null) {
            left = method1(root.left, param1 - root.val);
        }
        if (root.right != null) {
            right = method1(root.right, param1 - root.val);
        }
        if (root.left == null && root.right == null && root.val == param1)
            return true;
        if (left || right)
            return true;
        return false;
    }
}
