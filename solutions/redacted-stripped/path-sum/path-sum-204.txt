class Solution {

    public boolean ans = false;

    public boolean method1(TreeNode root, int param1) {
        if (root != null) {
            dfs(root, param1, root.val);
        }
        return ans;
    }

    public void dfs(TreeNode root, int target, int count) {
        if (root.left == null && root.right == null) {
            if (count == target)
                ans = true;
        }
        if (root.left != null) {
            dfs(root.left, target, count + root.left.val);
        }
        if (root.right != null) {
            dfs(root.right, target, count + root.right.val);
        }
    }
}
