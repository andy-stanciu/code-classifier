class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && root.val == param1) {
            return true;
        }
        boolean ans1 = method1(root.left, param1 - root.val);
        boolean ans2 = method1(root.right, param1 - root.val);
        return ans1 || ans2;
    }
}
