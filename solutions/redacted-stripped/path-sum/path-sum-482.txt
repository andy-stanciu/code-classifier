class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, param1, 0);
    }

    public boolean method2(TreeNode node, int param1, int s) {
        if (node == null)
            return param1 == s;
        if (node.left == null && node.right == null)
            return param1 == (s + node.val);
        if (node.left == null)
            return method2(node.right, param1, s + node.val);
        if (node.right == null)
            return method2(node.left, param1, s + node.val);
        if (method2(node.left, param1, s + node.val) || method2(node.right, param1, s + node.val))
            return true;
        return false;
    }
}
