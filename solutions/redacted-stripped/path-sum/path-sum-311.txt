class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return helper(root, param1, var1);
    }

    public boolean helper(TreeNode root, int param1, int var1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            var1 += root.val;
            if (var1 == param1) {
                return true;
            }
        }
        return helper(root.left, param1, var1 + root.val) || helper(root.right, param1, var1 + root.val);
    }
}
