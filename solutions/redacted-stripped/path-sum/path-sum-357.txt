class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        int var1 = 0;
        var1 += root.val;
        if (var1 == param1 && root.left == null && root.right == null) {
            return true;
        }
        int newtarget = param1 - root.val;
        return (method1(root.left, newtarget) || method1(root.right, newtarget));
    }
}
