class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method1(root, 0, param1);
    }

    private boolean method1(TreeNode root, int param2, int param1) {
        if (root == null) {
            return false;
        }
        if (param2 + root.val == param1 && root.left == null && root.right == null) {
            return true;
        }
        return method1(root.left, param2 + root.val, param1) || method1(root.right, param2 + root.val, param1);
    }
}
