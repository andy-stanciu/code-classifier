class Solution {

    public boolean method1(TreeNode root, int param1) {
        List<Integer> list = new ArrayList<Integer>();
        int var1 = 0;
        if (root == null)
            return false;
        method2(root, var1, list);
        {
            for (int i : list) if (i == param1)
                return true;
        }
        return false;
    }

    public static void method2(TreeNode root, int var1, List<Integer> list) {
        if (root.left == null && root.right == null) {
            var1 = var1 + root.val;
            list.add(var1);
        }
        if (root.left != null) {
            method2(root.left, var1 + root.val, list);
        }
        if (root.right != null) {
            method2(root.right, var1 + root.val, list);
        }
    }
}
