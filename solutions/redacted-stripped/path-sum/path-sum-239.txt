class Solution {

    public boolean dfs(TreeNode root, int t, int param1) {
        if (root == null) {
            return false;
        }
        param1 += root.val;
        if (root.left == null && root.right == null) {
            return t == param1;
        }
        return dfs(root.left, t, param1) || dfs(root.right, t, param1);
    }

    public boolean method1(TreeNode root, int param2) {
        return dfs(root, param2, 0);
    }
}
