class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, 0, param1);
    }

    private boolean helper(TreeNode root, int param2, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return param2 + root.val == param1;
        return helper(root.left, param2 + root.val, param1) || helper(root.right, param2 + root.val, param1);
    }
}
