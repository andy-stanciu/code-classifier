class Solution {

    public boolean method1(TreeNode root, int tar) {
        if (root == null) {
            return false;
        }
        return method2(root, tar, 0);
    }

    public boolean method2(TreeNode root, int tar, int method2) {
        if (root.left == null && root.right == null) {
            method2 += root.val;
            if (method2 == tar) {
                return true;
            }
            return false;
        }
        method2 += root.val;
        boolean left = false, right = false;
        if (root.left != null) {
            left = method2(root.left, tar, method2);
        }
        if (root.right != null) {
            right = method2(root.right, tar, method2);
        }
        return left || right;
    }
}
