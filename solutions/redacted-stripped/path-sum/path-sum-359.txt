class Solution {

    private boolean found = false;

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        dfs(root, param1, root.val);
        return found;
    }

    private void dfs(TreeNode node, int param1, int cur) {
        if (node.left == null && node.right == null) {
            if (cur == param1) {
                found = true;
            }
        }
        if (node.left != null) {
            dfs(node.left, param1, cur + node.left.val);
        }
        if (node.right != null) {
            dfs(node.right, param1, cur + node.right.val);
        }
    }
}
