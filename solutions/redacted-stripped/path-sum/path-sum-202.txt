class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return param1 == root.val;
        int var1 = param1 - root.val;
        boolean var2 = method1(root.left, var1);
        boolean var3 = method1(root.right, var1);
        return var2 || var3;
    }
}
