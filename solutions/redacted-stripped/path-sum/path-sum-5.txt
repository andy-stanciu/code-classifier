class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        Stack<TreeNode> var1 = new Stack<>();
        Stack<Integer> var2 = new Stack<>();
        var1.push(root);
        var2.push(root.val);
        while (!var1.isEmpty()) {
            TreeNode temp = var1.pop();
            int tempval = var2.pop();
            if (temp.left == null && temp.right == null && tempval == param1) {
                return true;
            }
            if (temp.right != null) {
                var1.push(temp.right);
                var2.push(temp.right.val + tempval);
            }
            if (temp.left != null) {
                var1.push(temp.left);
                var2.push(temp.left.val + tempval);
            }
        }
        return false;
    }
}
