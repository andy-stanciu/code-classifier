class Solution {

    public boolean method1(TreeNode root, int param1) {
        return dfs(root, param1);
    }

    private boolean dfs(TreeNode root, int s) {
        if (root == null)
            return false;
        s -= root.val;
        if (root.left == null && root.right == null && s == 0) {
            return true;
        }
        return dfs(root.left, s) || dfs(root.right, s);
    }
}
