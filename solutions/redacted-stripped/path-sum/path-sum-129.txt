class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        boolean ans = check(root, param1, var1);
        return ans;
    }

    boolean check(TreeNode root, int param1, int var1) {
        if (root == null)
            return false;
        var1 += root.val;
        if (root.left == null && root.right == null) {
            if (var1 == param1)
                return true;
            return false;
        }
        boolean left = check(root.left, param1, var1);
        boolean right = check(root.right, param1, var1);
        return left || right;
    }
}
