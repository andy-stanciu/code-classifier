class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method1(root, 0, param1);
    }

    public boolean method1(TreeNode root, int param2, int param1) {
        TreeNode t = root;
        if (t == null) {
            return false;
        }
        if (t.left == null && t.right == null) {
            param2 = param2 + t.val;
            if (param2 == param1) {
                return true;
            } else {
                return false;
            }
        } else {
            boolean leftBranch = method1(t.left, param2 + t.val, param1);
            boolean rightBranch = method1(t.right, param2 + t.val, param1);
            return leftBranch || rightBranch;
        }
    }
}
