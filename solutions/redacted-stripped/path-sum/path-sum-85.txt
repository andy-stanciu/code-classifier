class Solution {

    public boolean method1(TreeNode root, int param1) {
        param1 = param1;
        if (root != null) {
            param1 -= root.val;
            boolean l = false;
            boolean r = false;
            if (param1 == 0 && root.left == null && root.right == null)
                return true;
            if (root.left != null) {
                l = method1(root.left, param1);
            }
            if (root.right != null) {
                r = method1(root.right, param1);
            }
            return l || r;
        }
        return false;
    }
}
