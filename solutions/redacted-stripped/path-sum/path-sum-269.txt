class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return method2(root, param1, 0);
    }

    private boolean method2(TreeNode root, int param1, int param2) {
        param2 += root.val;
        if (root.left == null && root.right == null) {
            return param1 == param2;
        }
        boolean left = false;
        boolean right = false;
        if (root.left != null) {
            left = method2(root.left, param1, param2);
        }
        if (root.right != null) {
            right = method2(root.right, param1, param2);
        }
        return left || right;
    }
}
