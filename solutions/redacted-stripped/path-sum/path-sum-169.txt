class Solution {

    List<Integer> l = new ArrayList<>();

    public boolean method1(TreeNode root, int param1) {
        method2(root, 0);
        return l.contains(param1);
    }

    void method2(TreeNode root, int res) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            l.add(res + root.val);
            return;
        }
        method2(root.left, res + root.val);
        method2(root.right, res + root.val);
    }
}
