class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            if (root.val == param1)
                return true;
            else
                return false;
        }
        if (root.left == null)
            return method1(root.right, param1 - root.val);
        if (root.right == null)
            return method1(root.left, param1 - root.val);
        return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
    }
}
