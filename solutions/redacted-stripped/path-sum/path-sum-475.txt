class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        boolean leftResult = false;
        boolean rightResult = false;
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            return true;
        }
        if (root.left != null) {
            leftResult = method1(root.left, param1 - root.val);
        }
        if (root.right != null) {
            rightResult = method1(root.right, param1 - root.val);
        }
        if (leftResult || rightResult) {
            return true;
        }
        return false;
    }
}
