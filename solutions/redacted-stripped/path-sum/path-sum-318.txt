class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, param1, root.val);
    }

    private boolean method2(TreeNode root, int param1, int param2) {
        if (root.left == null && root.right == null) {
            return param2 == param1;
        }
        boolean left = root.left != null ? method2(root.left, param1, param2 + root.left.val) : false;
        boolean right = root.right != null ? method2(root.right, param1, param2 + root.right.val) : false;
        return left || right;
    }
}
