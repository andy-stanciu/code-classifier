class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, 0, param1);
    }

    public boolean helper(TreeNode root, int curr, int target) {
        if (root == null) {
            return false;
        }
        curr = curr + root.val;
        if (root.left == null && root.right == null) {
            if (target == curr)
                return true;
            else
                return false;
        }
        boolean left = helper(root.left, curr, target);
        boolean right = helper(root.right, curr, target);
        return left || right;
    }
}
