class Solution {

    private boolean helper(TreeNode root, int target, int param1) {
        if (root == null) {
            return false;
        }
        param1 += root.val;
        if (root.left == null && root.right == null && param1 == target) {
            return true;
        }
        if (root.left == null && root.right == null && param1 != target) {
            param1 -= root.val;
        }
        return helper(root.left, target, param1) || helper(root.right, target, param1);
    }

    public boolean method1(TreeNode root, int param2) {
        return helper(root, param2, 0);
    }
}
