class Solution {

    public boolean method1(TreeNode root, int param1) {
        method2(root);
        return has(root, param1);
    }

    public boolean has(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            if (root.val == param1)
                return true;
        }
        return has(root.left, param1) || has(root.right, param1);
    }

    public static void method2(TreeNode root) {
        if (root == null)
            return;
        int data = root.val;
        if (root.left != null)
            root.left.val += data;
        if (root.right != null)
            root.right.val += data;
        method2(root.left);
        method2(root.right);
    }
}
