class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null || root.val < -1000)
            return false;
        return method1(root, param1, 0);
    }

    public boolean method1(TreeNode root, int param1, int param2) {
        int var1 = param2;
        var1 += root.val;
        if (root.left == null && root.right == null) {
            System.out.println("sum : " + param2);
            if (var1 == param1)
                return true;
            else
                return false;
        }
        boolean flag = false;
        if (root.left != null)
            flag = method1(root.left, param1, var1);
        if (!flag)
            if (root.right != null)
                flag = method1(root.right, param1, var1);
        return flag;
    }
}
