class Solution {

    int var1 = 0;

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        var1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == var1) {
                var1 -= root.val;
                return true;
            } else {
                var1 -= root.val;
                return false;
            }
        }
        boolean left = method1(root.left, param1);
        boolean right = method1(root.right, param1);
        var1 -= root.val;
        return (left || right);
    }
}
