class Solution {

    int var1;

    public boolean method1(TreeNode root, int var1) {
        this.var1 = var1;
        return method2(root, 0);
    }

    boolean method2(TreeNode node, int param1) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            return param1 + node.val == var1;
        return method2(node.left, param1 + node.val) || method2(node.right, param1 + node.val);
    }
}
