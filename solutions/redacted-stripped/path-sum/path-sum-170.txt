class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        boolean ans = method2(root, param1, 0);
        return ans;
    }

    private boolean method2(TreeNode node, int param1, int param2) {
        int var1 = param2 + node.val;
        if (node.left == null && node.right == null) {
            if (var1 == param1) {
                return true;
            } else {
                return false;
            }
        }
        if (node.left == null) {
            return method2(node.right, param1, var1);
        }
        if (node.right == null) {
            return method2(node.left, param1, var1);
        }
        boolean a1 = method2(node.left, param1, var1);
        boolean a2 = method2(node.right, param1, var1);
        return a1 || a2;
    }
}
