class Solution {

    public boolean ps(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            param2 = param2 + root.val;
            if (param2 == param1)
                return true;
        }
        return ps(root.left, param1, param2 + root.val) || ps(root.right, param1, param2 + root.val);
    }

    public boolean method1(TreeNode root, int param1) {
        int param2 = 0;
        return ps(root, param1, param2);
    }
}
