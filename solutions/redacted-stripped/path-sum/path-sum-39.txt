class Solution {

    public int count = 0;

    public int var1 = 0;

    public boolean method1(TreeNode root, int param1) {
        dfs(root, param1);
        return count > 0 ? true : false;
    }

    public void dfs(TreeNode root, int param1) {
        if (root == null)
            return;
        var1 += root.val;
        if (root.left == null && root.right == null) {
            if (var1 == param1)
                count++;
        }
        dfs(root.left, param1);
        dfs(root.right, param1);
        var1 -= root.val;
    }
}
