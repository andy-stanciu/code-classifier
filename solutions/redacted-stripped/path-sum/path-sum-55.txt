class Solution {

    public boolean method1(TreeNode root, int param1) {
        traversal(root, param1);
        return res;
    }

    boolean res = false;

    public void traversal(TreeNode root, int param1) {
        if (root != null) {
            root.val = param1 - root.val;
            if (root.left == null && root.right == null && root.val == 0) {
                res = true;
                return;
            }
            traversal(root.left, root.val);
            traversal(root.right, root.val);
        }
    }
}
