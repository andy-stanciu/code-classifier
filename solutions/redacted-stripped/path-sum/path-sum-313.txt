class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return check(root, param1, 0);
    }

    private boolean check(TreeNode root, int param1, int param2) {
        if (root.left == null && root.right == null)
            return param2 + root.val == param1;
        boolean lStatus = false;
        boolean rStatus = false;
        if (root.left != null)
            lStatus = check(root.left, param1, param2 + root.val);
        if (root.right != null)
            rStatus = check(root.right, param1, param2 + root.val);
        return lStatus || rStatus;
    }
}
