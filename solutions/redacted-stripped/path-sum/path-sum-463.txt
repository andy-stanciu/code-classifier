class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (root.val == param1) {
                return true;
            } else {
                return false;
            }
        }
        int var1 = param1 - root.val;
        return method1(root.left, var1) || method1(root.right, var1);
    }
}
