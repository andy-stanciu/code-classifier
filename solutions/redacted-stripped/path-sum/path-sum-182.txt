class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return preOrder(root, root.val, param1);
    }

    boolean preOrder(TreeNode root, int param2, int target) {
        if (root.left == null && root.right == null) {
            if (param2 == target) {
                return true;
            }
            return false;
        }
        boolean ans = false;
        if (root.left != null)
            ans = preOrder(root.left, param2 + root.left.val, target);
        if (ans == true) {
            return true;
        }
        if (root.right != null)
            return preOrder(root.right, param2 + root.right.val, target);
        return false;
    }
}
