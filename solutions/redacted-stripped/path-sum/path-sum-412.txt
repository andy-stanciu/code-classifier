class Solution {

    public boolean method1(TreeNode node, int param1, int param2) {
        if (node != null) {
            if (node.left == null && node.right == null) {
                return node.val + param2 == param1;
            } else {
                return method1(node.left, param1, param2 + node.val) || method1(node.right, param1, param2 + node.val);
            }
        }
        return false;
    }

    public boolean method2(TreeNode root, int param1) {
        if (root != null) {
            return method1(root, param1, 0);
        }
        return false;
    }
}
