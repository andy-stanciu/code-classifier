class Solution {

    public boolean method1(TreeNode root, int param1) {
        return dfs(root, param1);
    }

    public boolean dfs(TreeNode node, int remain) {
        if (node == null) {
            return false;
        }
        remain -= node.val;
        if (node.left == null && node.right == null && remain == 0) {
            return true;
        }
        return dfs(node.left, remain) || dfs(node.right, remain);
    }
}
