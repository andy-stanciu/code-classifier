class Solution {

    public boolean method1(TreeNode root, int target, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            param1 = param1 + root.val;
            if (param1 == target) {
                return true;
            }
        }
        return method1(root.left, target, param1 + root.val) || method1(root.right, target, param1 + root.val);
    }

    public boolean method2(TreeNode root, int param2) {
        int param1 = 0;
        return method1(root, param2, param1);
    }
}
