class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return find(root, 0, param1);
    }

    boolean find(TreeNode root, int param2, int t) {
        if (root == null)
            return false;
        param2 += root.val;
        if (root.left == null && root.right == null) {
            return param2 == t;
        }
        return (find(root.left, param2, t) || find(root.right, param2, t));
    }
}
