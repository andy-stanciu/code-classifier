class Solution {

    public boolean method1(TreeNode root, int param1, int param2) {
        if (root.left == null && root.right == null) {
            if (param2 + root.val == param1)
                return true;
            return false;
        }
        if (root.left != null && method1(root.left, param1, param2 + root.val))
            return true;
        if (root.right != null && method1(root.right, param1, param2 + root.val))
            return true;
        return false;
    }

    public boolean method2(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method1(root, param1, 0);
    }
}
