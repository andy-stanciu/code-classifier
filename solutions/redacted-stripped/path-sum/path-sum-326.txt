class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, param1, 0);
    }

    public boolean method2(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null && param2 + root.val == param1)
            return true;
        if (root.left == null && root.right == null && param2 + root.val != param1)
            return false;
        boolean a = method2(root.left, param1, param2 + root.val);
        boolean b = method2(root.right, param1, param2 + root.val);
        return a || b || (a && b);
    }
}
