class Solution {

    private boolean method1(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param2 = param2 + root.val;
        if (root.left == null && root.right == null && param2 == param1) {
            return true;
        }
        boolean x = method1(root.left, param1, param2);
        boolean y = method1(root.right, param1, param2);
        return (x || y);
    }

    public boolean method2(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return method1(root, param1, 0);
    }
}
