class Solution {

    boolean hasSol = false;

    public boolean method1(TreeNode root, int param1) {
        LinkedList<Integer> temp = new LinkedList<>();
        return method1(root, param1, 0, temp);
    }

    private boolean method1(TreeNode root, int param1, int param2, LinkedList<Integer> temp) {
        if (root == null)
            return hasSol;
        temp.add(root.val);
        param2 += root.val;
        if (root.left == null && root.right == null && param2 == param1) {
            hasSol = true;
            return hasSol;
        }
        if (!hasSol) {
            hasSol = method1(root.right, param1, param2, temp);
            hasSol = method1(root.left, param1, param2, temp);
        }
        param2 -= root.val;
        temp.removeLast();
        return hasSol;
    }
}
