class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        param1 -= root.val;
        if (root.left == null && root.right == null) {
            return param1 == 0;
        }
        boolean left = this.method1(root.left, param1);
        boolean right = this.method1(root.right, param1);
        return left || right;
    }
}
