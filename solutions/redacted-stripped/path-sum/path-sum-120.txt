class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return method2(root, param1, 0, root.right == null && root.left == null);
    }

    public static boolean method2(TreeNode root, int param1, int param2, boolean leaf) {
        if (root == null) {
            if (param2 == param1 && leaf) {
                return true;
            }
            return false;
        }
        return method2(root.left, param1, param2 + root.val, root.right == null && root.left == null) || method2(root.right, param1, param2 + root.val, root.right == null && root.left == null);
    }
}
