class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, 0, param1);
    }

    boolean helper(TreeNode root, int param2, int target) {
        if (root == null)
            return false;
        param2 = param2 + root.val;
        if (root.left == null && root.right == null) {
            if (param2 == target) {
                return true;
            } else
                return false;
        }
        boolean left = helper(root.left, param2, target);
        boolean right = helper(root.right, param2, target);
        if (right == true || left == true)
            return true;
        else
            return false;
    }
}
