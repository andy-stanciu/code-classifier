class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1, 0);
    }

    public boolean method2(TreeNode root, int target, int param2) {
        if (root == null)
            return false;
        param2 += root.val;
        if (root.left == null && root.right == null && param2 == target)
            return true;
        Boolean left = method2(root.left, target, param2);
        Boolean right = method2(root.right, target, param2);
        return left || right;
    }
}
