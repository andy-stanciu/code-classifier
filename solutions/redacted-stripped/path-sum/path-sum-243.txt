class Solution {

    public boolean method1(TreeNode root, int param1) {
        return dfs(root, 0, param1);
    }

    public boolean dfs(TreeNode root, int param2, int tar) {
        if (root == null)
            return false;
        param2 += root.val;
        if (param2 == tar && root.left == null && root.right == null) {
            return true;
        }
        return dfs(root.left, param2, tar) || dfs(root.right, param2, tar);
    }
}
