class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1, 0);
    }

    private boolean method2(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (param2 + root.val == param1) {
                return true;
            }
            return false;
        }
        return method2(root.left, param1, param2 + root.val) || method2(root.right, param1, param2 + root.val);
    }
}
