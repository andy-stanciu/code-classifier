class Solution {

    private boolean t(TreeNode root, int param1, int target) {
        if (root == null)
            return false;
        param1 += root.val;
        if (param1 == target && root.left == null && root.right == null)
            return true;
        return t(root.right, param1, target) || t(root.left, param1, target);
    }

    public boolean method1(TreeNode root, int param2) {
        if (root == null)
            return false;
        boolean flag = t(root, 0, param2);
        return flag;
    }
}
