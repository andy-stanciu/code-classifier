class Solution {

    public static boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return helper(root, param1, 0);
    }

    public static boolean helper(TreeNode root, int param1, int param2) {
        if (root != null) {
            if (root.left == null && root.right == null) {
                param2 += root.val;
                return param1 == param2;
            }
        }
        if (root == null) {
            return false;
        }
        boolean left = helper(root.left, param1, param2 + root.val);
        boolean right = helper(root.right, param1, param2 + root.val);
        return left || right;
    }
}
