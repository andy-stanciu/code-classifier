class Solution {

    boolean flag = false;

    public boolean method1(TreeNode root, int param1) {
        solve(root, param1, 0);
        return flag;
    }

    public void solve(TreeNode root, int target, int param2) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            param2 += root.val;
            if (param2 == target)
                flag = true;
        }
        param2 += root.val;
        solve(root.left, target, param2);
        solve(root.right, target, param2);
    }
}
