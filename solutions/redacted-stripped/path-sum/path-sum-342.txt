class Solution {

    public boolean isLeaf(TreeNode root) {
        return root.left == null && root.right == null;
    }

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.val == param1 && isLeaf(root))
            return true;
        boolean leftResult = method1(root.left, param1 - root.val);
        boolean rightResult = method1(root.right, param1 - root.val);
        return leftResult || rightResult;
    }
}
