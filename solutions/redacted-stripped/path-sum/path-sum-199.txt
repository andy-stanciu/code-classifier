class Solution {

    int var1 = 0;

    public boolean method1(TreeNode root, int var1) {
        if (root == null)
            return false;
        if (root.val == var1 && root.left == null && root.right == null)
            return true;
        return method1(root.left, var1 - root.val) || method1(root.right, var1 - root.val);
    }
}
