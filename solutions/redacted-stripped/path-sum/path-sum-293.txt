class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root != null && param1 - root.val == 0 && root.left == null && root.right == null) {
            return true;
        }
        if (root == null) {
            return false;
        }
        boolean l = false;
        boolean r = false;
        if (root.left != null) {
            l = method1(root.left, param1 - root.val);
        }
        if (root.right != null) {
            r = method1(root.right, param1 - root.val);
        }
        return l || r;
    }
}
