class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            return param1 == root.val;
        }
        int remaining = param1 - root.val;
        return method1(root.left, remaining) || method1(root.right, remaining);
    }
}
