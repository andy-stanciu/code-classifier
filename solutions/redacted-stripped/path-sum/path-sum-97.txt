class Solution {

    public boolean method1(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return param2 + root.val == param1;
        return method1(root.left, param1, param2 + root.val) || method1(root.right, param1, param2 + root.val);
    }

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method1(root, param1, 0);
    }
}
