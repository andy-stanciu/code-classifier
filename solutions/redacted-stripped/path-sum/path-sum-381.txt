class Solution {

    static int t;

    public static boolean fn(int s, TreeNode r) {
        if (r.left == null && r.right == null)
            return s + r.val == t;
        s += r.val;
        if (r.left != null)
            if (fn(s, r.left))
                return true;
        if (r.right != null)
            if (fn(s, r.right))
                return true;
        return false;
    }

    public boolean method1(TreeNode r, int t) {
        if (r == null)
            return 1 < 0;
        this.t = t;
        return fn(0, r);
    }
}
