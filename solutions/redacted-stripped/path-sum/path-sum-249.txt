class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return helper(root, param1, 0);
    }

    public boolean helper(TreeNode root, int target, int param2) {
        if (root == null) {
            return false;
        }
        param2 += root.val;
        if (root.left == null && root.right == null) {
            if (param2 == target) {
                return true;
            }
        }
        if (helper(root.left, target, param2)) {
            return true;
        }
        if (helper(root.right, target, param2)) {
            return true;
        }
        return false;
    }
}
