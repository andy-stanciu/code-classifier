class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        param1 = param1 - root.val;
        if (param1 == 0 && root.left == null && root.right == null)
            return true;
        boolean checkleft = method1(root.left, param1);
        boolean checkright = method1(root.right, param1);
        return checkleft || checkright;
    }
}
