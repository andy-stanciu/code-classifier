class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        Stack<TreeNode> var1 = new Stack<>();
        Stack<Integer> var2 = new Stack<>();
        var1.add(root);
        var2.add(root.val);
        while (!var1.isEmpty()) {
            TreeNode curr = var1.pop();
            int currVal = var2.pop();
            if (curr.left == null && curr.right == null && currVal == param1)
                return true;
            if (curr.left != null) {
                var1.add(curr.left);
                var2.add(currVal + curr.left.val);
            }
            if (curr.right != null) {
                var1.add(curr.right);
                var2.add(currVal + curr.right.val);
            }
        }
        return false;
    }
}
