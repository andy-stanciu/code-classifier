class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1, 0);
    }

    public boolean method2(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param2 += root.val;
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                return true;
            }
        }
        return method2(root.left, param1, param2) || method2(root.right, param1, param2);
    }
}
