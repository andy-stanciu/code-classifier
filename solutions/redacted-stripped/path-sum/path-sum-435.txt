class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        param1 -= root.val;
        if (root.left == null && root.right == null && param1 == 0) {
            return true;
        }
        return method1(root.left, param1) || method1(root.right, param1);
    }

    public boolean method2(TreeNode root, int param1) {
        return method1(root, param1);
    }
}
