class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return param1 == root.val;
        boolean option1 = method1(root.right, param1 - root.val);
        boolean option2 = method1(root.left, param1 - root.val);
        return option1 || option2;
    }
}
