class Solution {

    public int method1(int[] arr) {
        int size = arr.length;
        if (size < 2)
            return size;
        Set<Integer> set = new HashSet<>();
        for (int e : arr) set.add(e);
        int count = 1, max = 1;
        for (int e : set) {
            if (!set.contains(e - 1)) {
                int temp = e + 1;
                while (set.contains(temp)) {
                    count++;
                    temp++;
                }
                max = Math.max(max, count);
                count = 1;
            }
        }
        return Math.max(max, count);
    }
}
