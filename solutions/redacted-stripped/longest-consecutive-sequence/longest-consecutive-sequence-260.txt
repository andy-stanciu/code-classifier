class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int count = 1;
        int maxCount = 1;
        for (int num : set) {
            if (!set.contains(num - 1)) {
                while (set.contains(num + 1)) {
                    count++;
                    num++;
                }
                maxCount = Math.max(count, maxCount);
                count = 1;
            }
        }
        return maxCount;
    }
}
