class Solution {

    public int method1(int[] nums) {
        int result = 0;
        if (nums == null) {
            return result;
        }
        Set<Integer> numbersSet = new HashSet<>();
        for (int num : nums) {
            numbersSet.add(num);
        }
        for (int set : numbersSet) {
            if (!numbersSet.contains(set - 1)) {
                int currentNum = set;
                int currentSeqCount = 1;
                while (numbersSet.contains(currentNum + 1)) {
                    currentNum++;
                    currentSeqCount++;
                }
                result = Math.max(result, currentSeqCount);
            }
        }
        return result;
    }
}
