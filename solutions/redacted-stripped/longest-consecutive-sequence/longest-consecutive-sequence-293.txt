class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        int var1 = 1;
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < n; i++) {
            set.add(nums[i]);
        }
        for (int start : set) {
            if (!set.contains(start - 1)) {
                int cnt = 1;
                int x = start;
                while (set.contains(x + 1)) {
                    x = x + 1;
                    cnt = cnt + 1;
                }
                var1 = Math.max(var1, cnt);
            }
        }
        return var1;
    }
}
