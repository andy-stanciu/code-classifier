class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int count = 0;
        if (nums.length == 0) {
            return 0;
        }
        if (nums.length == 1) {
            return 1;
        }
        int maxLength = 1;
        int currLength = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] != nums[i - 1]) {
                if (nums[i] == nums[i - 1] + 1) {
                    currLength++;
                } else {
                    maxLength = Math.max(maxLength, currLength);
                    currLength = 1;
                }
            }
        }
        maxLength = Math.max(maxLength, currLength);
        return maxLength;
    }
}
