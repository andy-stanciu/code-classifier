class Solution {

    public int method1(int[] nums) {
        Set<Integer> Set = new HashSet<>();
        for (int num : nums) {
            Set.add(num);
        }
        int seq = 0;
        for (int num : Set) {
            if (!Set.contains(num - 1)) {
                int curSeq = 1, curVal = num + 1;
                while (Set.contains(curVal)) {
                    curSeq++;
                    curVal++;
                }
                seq = Math.max(seq, curSeq);
            }
        }
        return seq;
    }
}
