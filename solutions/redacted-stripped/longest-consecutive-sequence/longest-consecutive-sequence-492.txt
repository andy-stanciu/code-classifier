class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int start = 0, len = 1, maxlen = 1;
        if (nums.length == 0)
            return 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1] + 1) {
                len++;
            }
            if (nums[i] != nums[i - 1] + 1 && nums[i] != nums[i - 1]) {
                maxlen = Math.max(len, maxlen);
                len = 1;
            }
        }
        maxlen = Math.max(len, maxlen);
        return maxlen;
    }
}
