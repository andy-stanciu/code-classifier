class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        Set<Integer> s = new HashSet<>();
        for (int num : nums) {
            s.add(num);
        }
        int var1 = 1;
        for (int num : nums) {
            if (!s.contains(num - 1)) {
                int next = num;
                int count = 1;
                while (s.contains(next + 1)) {
                    count++;
                    next++;
                }
                var1 = Math.max(var1, count);
            }
        }
        return var1;
    }
}
