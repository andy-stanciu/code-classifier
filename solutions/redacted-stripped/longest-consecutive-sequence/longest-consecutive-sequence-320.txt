class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int var1 = 0;
        for (int num : nums) {
            if (!set.contains(num - 1)) {
                int currentNum = num;
                int currentSeq = 1;
                while (set.contains(currentNum + 1)) {
                    currentNum += 1;
                    currentSeq += 1;
                }
                var1 = Math.max(var1, currentSeq);
            }
        }
        return var1;
    }
}
