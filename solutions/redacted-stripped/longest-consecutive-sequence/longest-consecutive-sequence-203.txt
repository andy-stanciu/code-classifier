class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int max = 0, cnt = 0;
        Arrays.sort(nums);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] == 1) {
                cnt++;
                max = Math.max(cnt, max);
            } else if (nums[i] != nums[i - 1]) {
                cnt = 0;
            }
        }
        return max + 1;
    }
}
