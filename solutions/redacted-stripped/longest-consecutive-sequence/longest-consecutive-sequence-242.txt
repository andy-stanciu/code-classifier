class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return 1;
        Arrays.sort(nums);
        int count = 1;
        int num = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1] - 1) {
                count++;
            } else if (nums[i] == nums[i + 1]) {
                count = count;
            } else {
                count = 1;
            }
            if (count > num) {
                num = count;
            }
        }
        return num;
    }
}
