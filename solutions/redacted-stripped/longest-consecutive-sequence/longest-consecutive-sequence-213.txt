class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int a : nums) set.add(a);
        int count = 0, ans = 0;
        for (Integer a : set) {
            if (!set.contains(a - 1)) {
                count = 0;
                while (set.contains(a)) {
                    count++;
                    a++;
                }
                ans = Math.max(ans, count);
            }
        }
        return ans;
    }
}
