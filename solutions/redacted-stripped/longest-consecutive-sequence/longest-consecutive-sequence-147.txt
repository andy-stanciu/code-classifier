class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        if (nums.length == 1) {
            return 1;
        }
        Arrays.sort(nums);
        int runLength = 1;
        int maxRunLength = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1] + 1) {
                runLength++;
            } else if (nums[i] == nums[i - 1]) {
            } else if (runLength > maxRunLength) {
                maxRunLength = runLength;
                runLength = 1;
            } else {
                runLength = 1;
            }
        }
        if (runLength > maxRunLength) {
            return runLength;
        }
        return maxRunLength;
    }
}
