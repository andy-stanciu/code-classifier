class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i : nums) {
            set.add(i);
        }
        int longstreak = 0;
        for (int i : nums) {
            if (!set.contains(i - 1)) {
                int currentNum = i;
                int currentStreak = 1;
                while (set.contains(currentNum + 1)) {
                    currentNum++;
                    currentStreak++;
                }
                longstreak = Math.max(longstreak, currentStreak);
            }
        }
        return longstreak;
    }
}
