class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int curr = 1;
        int res = 1;
        int n = nums.length;
        if (n == 0)
            return 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] - 1 == nums[i - 1])
                curr++;
            else if (nums[i] != nums[i - 1])
                curr = 1;
            res = Math.max(res, curr);
        }
        return res;
    }
}
