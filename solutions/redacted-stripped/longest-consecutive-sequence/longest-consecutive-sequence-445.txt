class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return 1;
        } else if (nums.length == 0) {
            return 0;
        } else {
            Arrays.sort(nums);
            int count = 1;
            int max = 1;
            for (int i = 1; i < nums.length; i++) {
                if (nums[i - 1] == nums[i]) {
                    continue;
                }
                if (nums[i - 1] + 1 == nums[i]) {
                    count++;
                } else {
                    count = 1;
                }
                if (count > max) {
                    max = count;
                }
            }
            return max;
        }
    }
}
