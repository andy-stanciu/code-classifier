class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int total = 0;
        int curr = 0;
        for (int i = 0; i < nums.length; i++) {
            if (!set.contains(nums[i])) {
                curr = 0;
                continue;
            }
            if (set.contains(nums[i] - 1)) {
                curr = 0;
                continue;
            }
            int left = nums[i];
            while (set.contains(left)) {
                curr += 1;
                set.remove(left);
                left++;
            }
            total = Math.max(total, curr);
            curr = 0;
        }
        return total;
    }
}
