class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0 || n == 1)
            return n;
        Arrays.sort(nums);
        int maxLength = 0;
        int currMax = 0;
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1])
                continue;
            if (nums[i] == nums[i + 1] - 1)
                currMax++;
            else {
                maxLength = Math.max(maxLength, (currMax + 1));
                currMax = 0;
            }
        }
        maxLength = Math.max(maxLength, (currMax + 1));
        return maxLength;
    }
}
