class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int max = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            max = Math.max(max, traverse(nums[i], map, set));
        }
        return max;
    }

    public int traverse(int num, Map<Integer, Integer> map, HashSet<Integer> set) {
        if (map.containsKey(num)) {
            return map.get(num);
        }
        int count = 1;
        if (set.contains(num + 1)) {
            count += traverse(num + 1, map, set);
        }
        map.put(num, count);
        return count;
    }
}
