class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int count = 1;
        int max = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i + 1 < nums.length)
                if (nums[i] + 1 == nums[i + 1]) {
                    count++;
                } else if (nums[i] == nums[i + 1]) {
                    continue;
                } else {
                    count = 1;
                }
            max = Math.max(max, count);
        }
        return max;
    }
}
