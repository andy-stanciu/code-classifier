class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int var1 = 0;
        for (int num : nums) {
            if (!set.contains(num - 1)) {
                int count = 1;
                while (set.contains(num + count)) count++;
                var1 = Math.max(count, var1);
            }
        }
        return var1;
    }
}
