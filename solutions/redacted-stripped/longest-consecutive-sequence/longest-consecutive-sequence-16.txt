class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        int max = 0;
        for (int i = 0; i < nums.length; i++) {
            int count = 0;
            int x = nums[i];
            while (set.contains(x--)) {
                count++;
            }
            max = Math.max(count, max);
        }
        return max;
    }
}
