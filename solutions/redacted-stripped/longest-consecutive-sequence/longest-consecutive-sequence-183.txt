class Solution {

    public int method1(int[] numsarr) {
        Set<Integer> set = new HashSet<>();
        int var1 = 0;
        for (int num : numsarr) {
            set.add(num);
        }
        for (int num : set) {
            if (!set.contains(num - 1)) {
                int length = 1;
                while (set.contains(num + length)) {
                    length++;
                }
                var1 = Math.max(var1, length);
            }
        }
        return var1;
    }
}
