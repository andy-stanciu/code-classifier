class Solution {

    public int method1(int[] arr) {
        int size = arr.length;
        if (size < 2)
            return size;
        Arrays.sort(arr);
        int count = 1;
        int max = 1;
        for (int i = 1; i < size; i++) {
            if (arr[i] - arr[i - 1] == 1)
                count++;
            else if (arr[i] - arr[i - 1] == 0)
                continue;
            else {
                max = Math.max(max, count);
                count = 1;
            }
        }
        return Math.max(max, count);
    }
}
