class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int var1 = 0;
        for (int num : nums) {
            if (!(set.contains(num - 1))) {
                int currentNum = num;
                int currentSubSeqCount = 0;
                while (set.contains(currentNum)) {
                    currentNum++;
                    currentSubSeqCount++;
                }
                var1 = Math.max(var1, currentSubSeqCount);
            }
        }
        return var1;
    }
}
