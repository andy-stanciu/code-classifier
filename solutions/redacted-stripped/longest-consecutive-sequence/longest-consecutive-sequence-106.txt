class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        Set<Integer> numsSet = new HashSet<Integer>();
        for (int i : nums) {
            numsSet.add(i);
        }
        int var1 = 1;
        for (int num : nums) {
            if (!numsSet.contains(num - 1)) {
                int count = 1;
                while (numsSet.contains(num + 1)) {
                    num++;
                    count++;
                }
                var1 = Math.max(var1, count);
            }
        }
        return var1;
    }
}
