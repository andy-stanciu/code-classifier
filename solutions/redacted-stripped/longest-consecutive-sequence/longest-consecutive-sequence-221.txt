class Solution {

    public int method1(int[] nums) {
        Set<Integer> setNums = new HashSet<>();
        for (int num : nums) {
            setNums.add(num);
        }
        int method1 = 0;
        for (int num : setNums) {
            if (!setNums.contains(num - 1)) {
                int currentNum = num;
                int var1 = 1;
                while (setNums.contains(currentNum + 1)) {
                    var1++;
                    currentNum++;
                }
                method1 = Math.max(method1, var1);
            }
        }
        return method1;
    }
}
