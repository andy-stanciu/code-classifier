class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<Integer>();
        int var1 = 0;
        for (int num : nums) {
            set.add(num);
        }
        for (int i = 0; i < nums.length; i++) {
            int var2 = 1;
            int current = nums[i];
            while (set.contains(--current)) {
                var2++;
                set.remove(current);
            }
            current = nums[i];
            while (set.contains(++current)) {
                var2++;
                set.remove(current);
            }
            var1 = Math.max(var1, var2);
        }
        return var1;
    }
}
