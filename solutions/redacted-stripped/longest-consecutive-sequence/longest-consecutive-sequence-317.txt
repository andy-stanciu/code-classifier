class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int n : nums) set.add(n);
        Set<Integer> starts = new HashSet<>();
        for (int n : nums) {
            if (!set.contains(n - 1)) {
                starts.add(n);
            }
        }
        int max = 0;
        for (int n : starts) {
            int count = 0;
            for (int i = n; set.contains(i); i++) {
                count++;
            }
            max = Math.max(count, max);
        }
        return max;
    }
}
