class Solution {

    public int method1(int[] nums) {
        Set<Integer> numsSet = new HashSet<>();
        for (int num : nums) {
            numsSet.add(num);
        }
        int var1 = 0;
        for (int num : nums) {
            if (!numsSet.contains(num - 1)) {
                int currentLength = 1;
                while (numsSet.contains(num + currentLength)) {
                    currentLength++;
                }
                var1 = Math.max(var1, currentLength);
            }
        }
        return var1;
    }
}
