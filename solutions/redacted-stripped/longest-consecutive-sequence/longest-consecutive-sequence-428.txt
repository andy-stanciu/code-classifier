class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return 0;
        Arrays.sort(nums);
        int var1 = 1;
        int cnt = 1;
        int lastsmaller = nums[0];
        for (int i = 1; i < n; i++) {
            if (nums[i] == lastsmaller + 1) {
                cnt++;
                lastsmaller = nums[i];
            } else if (nums[i] != lastsmaller) {
                cnt = 1;
                lastsmaller = nums[i];
            }
            var1 = Math.max(var1, cnt);
        }
        return var1;
    }
}
