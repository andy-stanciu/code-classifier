class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1 || nums.length == 0)
            return nums.length;
        Arrays.sort(nums);
        int r = 1;
        int m = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i + 1] - nums[i] == 1)
                m++;
            else if (nums[i] != nums[i + 1])
                m = 0;
            r = Math.max(r, m + 1);
        }
        return r;
    }
}
