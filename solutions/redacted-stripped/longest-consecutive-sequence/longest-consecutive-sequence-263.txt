class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        HashSet<Integer> data = new HashSet<>();
        for (int num : nums) {
            data.add(num);
        }
        int maxCount = 0;
        for (int value : data) {
            if (!data.contains(value - 1)) {
                int currentValue = value;
                int count = 1;
                while (data.contains(currentValue + 1)) {
                    currentValue++;
                    count++;
                }
                maxCount = Math.max(maxCount, count);
            }
        }
        return maxCount;
    }
}
