class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        if (nums.length == 0) {
            return 0;
        }
        int count = 1;
        int var1 = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i + 1] == nums[i]) {
                continue;
            } else if (nums[i + 1] == nums[i] + 1) {
                count++;
            } else {
                var1 = Math.max(count, var1);
                count = 1;
            }
        }
        return var1 = Math.max(count, var1);
    }
}
