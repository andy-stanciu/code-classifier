class Solution {

    public int method1(int[] nums) {
        Set<Integer> st = new HashSet<>();
        Arrays.stream(nums).forEach(st::add);
        int count = 0;
        for (int x : st) {
            if (!st.contains(x - 1)) {
                int length = 1;
                while (st.contains(x + length)) length++;
                count = Math.max(length, count);
            }
        }
        return count;
    }
}
