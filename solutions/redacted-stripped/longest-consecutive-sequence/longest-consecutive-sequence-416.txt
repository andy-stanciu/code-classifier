class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int maxi = 0;
        for (int num : nums) {
            if (set.contains(num)) {
                set.remove(num);
                int cnt = 1;
                int currentNum = num;
                while (set.contains(currentNum - 1)) {
                    set.remove(currentNum - 1);
                    currentNum--;
                    cnt++;
                }
                currentNum = num;
                while (set.contains(currentNum + 1)) {
                    set.remove(currentNum + 1);
                    currentNum++;
                    cnt++;
                }
                maxi = Math.max(maxi, cnt);
            }
        }
        return maxi;
    }
}
