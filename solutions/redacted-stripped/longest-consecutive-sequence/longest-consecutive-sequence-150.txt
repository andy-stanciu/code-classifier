class Solution {

    public int method1(int[] nums) {
        int small = Integer.MIN_VALUE;
        int var1 = 1;
        Arrays.sort(nums);
        int count = 0;
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == small) {
                count++;
                small = nums[i];
            } else if (nums[i] == small) {
                continue;
            } else {
                count = 1;
                small = nums[i];
            }
            var1 = Math.max(var1, count);
        }
        return var1;
    }
}
