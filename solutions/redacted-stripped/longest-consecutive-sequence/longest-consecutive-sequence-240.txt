class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int var1 = 1;
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        for (int num : set) {
            if (!set.contains(num - 1)) {
                int count = 1;
                int x = num;
                while (set.contains(x + 1)) {
                    x++;
                    count++;
                }
                var1 = Math.max(var1, count);
            }
        }
        return var1;
    }
}
