class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) set.add(num);
        int lcsLength = 0;
        for (int i = 0; i < nums.length; i++) {
            if (!set.contains(nums[i]))
                continue;
            else {
                set.remove(nums[i]);
                int elementsOnLeft = 0;
                int j = nums[i] - 1;
                while (set.contains(j)) {
                    set.remove(j--);
                    elementsOnLeft++;
                }
                int elementsOnRight = 0;
                j = nums[i] + 1;
                while (set.contains(j)) {
                    set.remove(j++);
                    elementsOnRight++;
                }
                int totalElements = 1 + elementsOnRight + elementsOnLeft;
                lcsLength = Math.max(lcsLength, totalElements);
            }
        }
        return lcsLength;
    }
}
