class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        } else if (n == 1) {
            return 1;
        }
        Arrays.sort(nums);
        int count = 1;
        int max = 0;
        for (int i = 0; i < n - 1; i++) {
            if (nums[i + 1] - nums[i] == 1) {
                count++;
            } else if (nums[i + 1] - nums[i] == 0) {
                continue;
            } else {
                count = 1;
            }
            if (count > max) {
                max = count;
            }
        }
        if (count > max) {
            max = count;
        }
        return max;
    }
}
