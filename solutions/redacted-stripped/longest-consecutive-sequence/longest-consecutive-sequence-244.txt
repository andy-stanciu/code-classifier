class Solution {

    public int method1(int[] nums) {
        int max = 0;
        int n = nums.length;
        Arrays.sort(nums);
        int count = 1;
        int lastSmaller = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastSmaller) {
                count++;
                lastSmaller = nums[i];
            } else if (nums[i] != lastSmaller) {
                count = 1;
                lastSmaller = nums[i];
            }
            max = Math.max(max, count);
        }
        return max;
    }
}
