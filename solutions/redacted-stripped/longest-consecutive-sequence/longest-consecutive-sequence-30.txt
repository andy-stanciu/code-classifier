class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> s = new HashSet<>();
        for (int num : nums) {
            s.add(num);
        }
        int var1 = 0;
        for (int n : nums) {
            if (!s.contains(n - 1)) {
                int length = 0;
                while (s.contains(n + length)) {
                    length += 1;
                }
                var1 = Math.max(var1, length);
            }
        }
        return var1;
    }
}
