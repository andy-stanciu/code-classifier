class Solution {

    public int method1(int[] nums) {
        int res = 0;
        Set<Integer> set = new HashSet();
        for (int i : nums) {
            set.add(i);
        }
        for (int i : set) {
            int c = 0;
            int x = i;
            if (!set.contains(x - 1)) {
                while (set.contains(x)) {
                    c++;
                    x++;
                }
                res = Math.max(res, c);
            }
        }
        return res;
    }
}
