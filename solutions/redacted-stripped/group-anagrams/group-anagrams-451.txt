class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> method1 = new HashMap<>();
        for (String str : strs) {
            char[] charArr = str.toCharArray();
            Arrays.sort(charArr);
            String sortStr = String.valueOf(charArr);
            if (!method1.containsKey(sortStr)) {
                method1.put(sortStr, new ArrayList<>());
            }
            method1.get(sortStr).add(str);
        }
        return new ArrayList<>(method1.values());
    }
}
