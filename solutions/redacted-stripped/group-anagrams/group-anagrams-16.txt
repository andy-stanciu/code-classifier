class Solution {

    public List<List<String>> method1(String[] strs) {
        List<List<String>> var1 = new ArrayList<>();
        Map<String, List<String>> map = new HashMap();
        for (int i = 0; i < strs.length; i++) {
            char[] charsOfWord = strs[i].toCharArray();
            Arrays.sort(charsOfWord);
            String sortedWord = String.valueOf(charsOfWord);
            List<String> var2 = map.get(sortedWord);
            if (var2 == null) {
                var2 = new ArrayList<>();
                map.put(sortedWord, var2);
                var1.add(var2);
            }
            var2.add(strs[i]);
        }
        return var1;
    }
}
