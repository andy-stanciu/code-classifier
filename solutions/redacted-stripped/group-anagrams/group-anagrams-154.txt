class Solution {

    public List<List<String>> method1(String[] strs) {
        List<List<String>> var1 = new ArrayList<>();
        Map<String, List<String>> var2 = new HashMap<>();
        String sortedStr;
        List<String> values;
        char[] chars;
        for (String str : strs) {
            values = new ArrayList<>();
            chars = str.toCharArray();
            Arrays.sort(chars);
            sortedStr = new String(chars);
            if (var2.containsKey(sortedStr))
                values = var2.get(sortedStr);
            values.add(str);
            var2.put(sortedStr, values);
        }
        for (List<String> list : var2.values()) {
            var1.add(list);
        }
        return var1;
    }
}
