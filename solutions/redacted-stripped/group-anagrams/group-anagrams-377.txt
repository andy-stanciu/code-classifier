class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, Integer> map = new HashMap<>();
        List<List<String>> result = new ArrayList<>();
        for (int i = 0; i < strs.length; i++) {
            String sortedString = sorted(strs[i]);
            if (map.containsKey(sortedString)) {
                result.get(map.get(sortedString)).add(strs[i]);
            } else {
                result.add(new ArrayList<>());
                result.get(result.size() - 1).add(strs[i]);
                map.put(sortedString, result.size() - 1);
            }
        }
        return result;
    }

    public String sorted(String str) {
        char[] array = str.toCharArray();
        Arrays.sort(array);
        return new String(array);
    }
}
