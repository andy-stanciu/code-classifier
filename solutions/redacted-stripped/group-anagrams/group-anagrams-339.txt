class Solution {

    public List<List<String>> method1(String[] strs) {
        if (strs == null || strs.length == 0)
            return new ArrayList();
        List<List<String>> ans = new ArrayList();
        Map<String, List<String>> m = new HashMap<>();
        for (String s : strs) {
            char[] c = s.toCharArray();
            Arrays.sort(c);
            String sortedString = String.valueOf(c);
            if (!m.containsKey(sortedString))
                m.put(sortedString, new ArrayList<>());
            m.get(sortedString).add(s);
            System.out.println(m.get(sortedString));
        }
        return new ArrayList<>(m.values());
    }
}
