class Solution {

    public List<List<String>> method1(String[] strs) {
        String[] sortStrings = new String[strs.length];
        for (int i = 0; i < strs.length; ++i) {
            sortStrings[i] = convertToSortString(strs[i]);
        }
        Map<String, List<String>> map = new HashMap<>();
        for (int i = 0; i < strs.length; i++) {
            if (map.containsKey(sortStrings[i])) {
                map.get(sortStrings[i]).add(strs[i]);
            } else {
                map.put(sortStrings[i], new ArrayList<>());
                map.get(sortStrings[i]).add(strs[i]);
            }
        }
        List<List<String>> res = new ArrayList<>();
        for (String s : map.keySet()) {
            res.add(map.get(s));
        }
        return res;
    }

    private String convertToSortString(String s) {
        char[] chars = s.toCharArray();
        Arrays.sort(chars);
        return new String(chars);
    }
}
