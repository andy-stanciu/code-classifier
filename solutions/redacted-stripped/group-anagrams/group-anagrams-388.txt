class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, ArrayList<String>> var1 = new HashMap<>();
        for (String word : strs) {
            char[] wordChars = word.toCharArray();
            Arrays.sort(wordChars);
            String sortedWord = new String(wordChars);
            if (!var1.containsKey(sortedWord)) {
                var1.put(sortedWord, new ArrayList<>());
            }
            var1.get(sortedWord).add(word);
        }
        List<List<String>> var2 = new ArrayList<>();
        for (String key : var1.keySet()) {
            var2.add(var1.get(key));
        }
        return var2;
    }
}
