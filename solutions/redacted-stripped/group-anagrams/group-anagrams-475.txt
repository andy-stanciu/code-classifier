class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> newMap = new HashMap<>();
        for (String str : strs) {
            char[] ch = str.toCharArray();
            Arrays.sort(ch);
            String sorted = new String(ch);
            newMap.putIfAbsent(sorted, new ArrayList<>());
            newMap.get(sorted).add(str);
        }
        return new ArrayList<>(newMap.values());
    }
}
