class Solution {

    public List<List<String>> method1(String[] strs) {
        List<List<String>> answer = new ArrayList<>();
        Map<String, Integer> var1 = new HashMap<>();
        for (String str : strs) {
            char[] charString = str.toCharArray();
            Arrays.sort(charString);
            String sortedString = String.valueOf(charString);
            if (!var1.containsKey(sortedString)) {
                var1.put(sortedString, answer.size());
                answer.add(new ArrayList<>());
            }
            answer.get(var1.get(sortedString)).add(str);
        }
        return answer;
    }
}
