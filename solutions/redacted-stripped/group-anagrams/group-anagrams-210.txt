class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String word : strs) {
            char[] arr = word.toCharArray();
            Arrays.sort(arr);
            String sortedWord = new String(arr);
            if (!var1.containsKey(sortedWord)) {
                var1.put(sortedWord, new ArrayList<>());
            }
            var1.get(sortedWord).add(word);
        }
        List<List<String>> result = new ArrayList<>(var1.values());
        return result;
    }
}
