class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String s : strs) {
            char[] letters = s.toCharArray();
            Arrays.sort(letters);
            String sortedString = String.valueOf(letters);
            if (var1.containsKey(sortedString)) {
                var1.get(sortedString).add(s);
            } else {
                List<String> newList = new LinkedList<>();
                newList.add(s);
                var1.put(sortedString, newList);
            }
        }
        return new LinkedList<>(var1.values());
    }
}
