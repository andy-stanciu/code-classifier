class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> grpanagrm = new HashMap<>();
        for (String s : strs) {
            char[] chr = s.toCharArray();
            Arrays.sort(chr);
            String sortedstr = String.valueOf(chr);
            if (!grpanagrm.containsKey(sortedstr)) {
                grpanagrm.put(sortedstr, new ArrayList<>());
            }
            grpanagrm.get(sortedstr).add(s);
        }
        return new ArrayList<>(grpanagrm.values());
    }
}
