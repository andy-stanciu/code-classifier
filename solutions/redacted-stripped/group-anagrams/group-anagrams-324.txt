class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> m = new HashMap<>();
        for (String s : strs) {
            char[] words = s.toCharArray();
            Arrays.sort(words);
            String sorted = new String(words);
            if (!(m.containsKey(sorted))) {
                m.put(sorted, new ArrayList<>());
            }
            m.get(sorted).add(s);
        }
        return new ArrayList<>(m.values());
    }
}
