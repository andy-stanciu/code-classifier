class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> gr_ana = new HashMap<>();
        for (String s : strs) {
            char[] ca = s.toCharArray();
            Arrays.sort(ca);
            String key = String.valueOf(ca);
            if (!gr_ana.containsKey(key)) {
                gr_ana.put(key, new ArrayList());
            }
            gr_ana.get(key).add(s);
        }
        return new ArrayList(gr_ana.values());
    }
}
