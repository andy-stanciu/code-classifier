class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String str : strs) {
            String sorted = sortChar(str);
            var1.putIfAbsent(sorted, new ArrayList<>());
            var1.get(sorted).add(str);
        }
        return new ArrayList<>(var1.values());
    }

    public static String sortChar(String str) {
        char[] chars = str.toCharArray();
        Arrays.sort(chars);
        return new String(chars);
    }
}
