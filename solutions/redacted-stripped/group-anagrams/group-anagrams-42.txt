class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap();
        for (String word : strs) {
            char[] chars = word.toCharArray();
            Arrays.sort(chars);
            String key = new String(chars);
            if (var1.containsKey(key)) {
                var1.get(key).add(word);
            } else {
                List<String> list = new ArrayList<String>();
                list.add(word);
                var1.put(key, list);
            }
        }
        return new ArrayList<List<String>>(var1.values());
    }
}
