class Solution {

    public List<List<String>> method1(String[] strs) {
        List<List<String>> res = new ArrayList<>();
        Map<String, List<String>> map = new HashMap<>();
        for (String s : strs) {
            char[] arr = s.toCharArray();
            Arrays.sort(arr);
            String sorted_s = new String(arr);
            map.putIfAbsent(sorted_s, new ArrayList<>());
            map.get(sorted_s).add(s);
        }
        for (List<String> l : map.values()) res.add(l);
        return res;
    }
}
