class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List> map = new HashMap();
        for (String s : strs) {
            char[] charArray = s.toCharArray();
            Arrays.sort(charArray);
            String sortedString = new String(charArray);
            List<String> currentList = map.getOrDefault(sortedString, new ArrayList<>());
            currentList.add(s);
            map.put(sortedString, currentList);
        }
        List<List<String>> list = new ArrayList(map.values());
        return list;
    }
}
