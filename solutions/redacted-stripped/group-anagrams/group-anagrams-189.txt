class Solution {

    public List<List<String>> method1(String[] strs) {
        List<List<String>> result = new ArrayList<>();
        if (strs.length == 0)
            return result;
        Map<String, List<String>> map = new HashMap<>();
        for (String s : strs) {
            char[] sArr = s.toCharArray();
            Arrays.sort(sArr);
            String sortS = new String(sArr);
            List<String> val;
            if (map.containsKey(sortS)) {
                val = map.get(sortS);
            } else {
                val = new ArrayList<String>();
            }
            val.add(s);
            map.put(sortS, val);
        }
        for (List<String> list : map.values()) {
            result.add(list);
        }
        return result;
    }
}
