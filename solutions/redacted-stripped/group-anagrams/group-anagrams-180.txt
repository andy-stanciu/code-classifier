class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String string : strs) {
            char[] chars = string.toCharArray();
            Arrays.sort(chars);
            String canonical = new String(chars);
            if (var1.containsKey(canonical)) {
                var1.get(canonical).add(string);
            } else {
                List<String> var2 = new ArrayList<>();
                var2.add(string);
                var1.put(canonical, var2);
            }
        }
        return new ArrayList<>(var1.values());
    }
}
