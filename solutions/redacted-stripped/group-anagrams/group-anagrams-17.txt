class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> map = new HashMap<>();
        for (String s : strs) {
            char[] b = s.toCharArray();
            Arrays.sort(b);
            String arrS = new String(b);
            if (!map.containsKey(arrS)) {
                map.put(arrS, new ArrayList<>());
            }
            map.get(arrS).add(s);
        }
        return new ArrayList<>(map.values());
    }
}
