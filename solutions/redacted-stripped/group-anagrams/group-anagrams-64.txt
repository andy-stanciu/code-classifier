class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            char[] c = str.toCharArray();
            Arrays.sort(c);
            String sortedStr = new String(c);
            List<String> var1 = map.getOrDefault(sortedStr, new ArrayList<>());
            var1.add(str);
            map.put(sortedStr, var1);
        }
        return new ArrayList<>(map.values());
    }
}
