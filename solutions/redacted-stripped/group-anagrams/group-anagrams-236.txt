class Solution {

    public List<List<String>> method1(String[] strs) {
        List<List<String>> ls = new ArrayList<>();
        Map<String, List<String>> map = new HashMap<>();
        for (String s : strs) {
            char[] arr = s.toCharArray();
            Arrays.sort(arr);
            String stroka = new String(arr);
            if (!map.containsKey(stroka)) {
                map.put(stroka, new ArrayList<>());
            }
            map.get(stroka).add(s);
        }
        ls.addAll(map.values());
        return ls;
    }
}
