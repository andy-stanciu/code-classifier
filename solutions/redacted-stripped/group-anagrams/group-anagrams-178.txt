class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> map = new HashMap<>();
        for (String s : strs) {
            char[] sArray = s.toCharArray();
            Arrays.sort(sArray);
            String sortedS = Arrays.toString(sArray);
            List<String> amagrasList = map.getOrDefault(sortedS, new ArrayList<>());
            amagrasList.add(s);
            map.put(sortedS, amagrasList);
        }
        return new ArrayList<>(map.values());
    }
}
