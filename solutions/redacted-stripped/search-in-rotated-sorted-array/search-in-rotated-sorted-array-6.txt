class Solution {

    public int method1(int[] nums, int target) {
        int l = 0;
        int r = nums.length - 1;
        int maxE = nums[0];
        int max = 0;
        for (int i = 1; i < nums.length; i++) {
            if (maxE < nums[i]) {
                maxE = nums[i];
                max = i;
            }
        }
        if (nums[max] == target)
            return max;
        l = 0;
        r = max;
        while (l <= r) {
            int mid = l + ((r - l) / 2);
            if (nums[mid] == target)
                return mid;
            if (nums[mid] > target)
                r = mid - 1;
            else
                l = mid + 1;
        }
        l = max + 1;
        r = nums.length - 1;
        while (l <= r) {
            int mid = l + ((r - l) / 2);
            if (nums[mid] == target)
                return mid;
            if (nums[mid] > target)
                r = mid - 1;
            else
                l = mid + 1;
        }
        return -1;
    }
}
