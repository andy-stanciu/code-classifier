class Solution {

    public int method1(int[] nums, int target) {
        int i = 0, j = nums.length - 1;
        while (i < j) {
            int mid = (i + j) / 2;
            if (target <= nums[j] && nums[mid] > nums[j]) {
                i = mid + 1;
            } else if (target > nums[j] && nums[mid] < nums[j]) {
                j = mid - 1;
            } else {
                if (nums[mid] == target) {
                    return mid;
                } else if (target < nums[mid]) {
                    j = mid - 1;
                } else {
                    i = mid + 1;
                }
            }
        }
        return -1;
    }
}
