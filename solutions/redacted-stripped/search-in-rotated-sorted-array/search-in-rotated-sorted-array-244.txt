class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        if (n == 1 && nums[0] == target) {
            return 0;
        }
        for (int i = 0; i < n; i++) {
            if (target == nums[i]) {
                return i;
            }
        }
        return -1;
    }
}
