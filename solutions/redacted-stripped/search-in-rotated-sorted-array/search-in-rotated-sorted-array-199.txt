class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        int s = 0, e = n - 1, m = -1;
        while (s <= e) {
            m = (s + e) / 2;
            if (nums[m] == target)
                return m;
            if (nums[s] <= nums[m]) {
                if (nums[s] <= target && target < nums[m])
                    e = m - 1;
                else
                    s = m + 1;
            } else {
                if (nums[m] < target && target <= nums[e])
                    s = m + 1;
                else
                    e = m - 1;
            }
        }
        return -1;
    }
}
