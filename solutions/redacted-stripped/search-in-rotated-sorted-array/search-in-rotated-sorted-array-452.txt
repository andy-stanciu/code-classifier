class Solution {

    public int method1(int[] nums, int target) {
        int si = 0;
        int ei = nums.length - 1;
        int taridx = method2(nums, target, si, ei);
        return taridx;
    }

    public int method2(int[] arr, int tar, int si, int ei) {
        if (si > ei) {
            return -1;
        }
        int mid = si + (ei - si) / 2;
        if (arr[mid] == tar) {
            return mid;
        }
        if (arr[si] <= arr[mid]) {
            if (arr[si] <= tar && tar <= arr[mid]) {
                return method2(arr, tar, si, mid - 1);
            } else {
                return method2(arr, tar, mid + 1, ei);
            }
        } else {
            if (arr[mid] <= tar && tar <= arr[ei]) {
                return method2(arr, tar, mid + 1, ei);
            } else {
                return method2(arr, tar, si, mid - 1);
            }
        }
    }
}
