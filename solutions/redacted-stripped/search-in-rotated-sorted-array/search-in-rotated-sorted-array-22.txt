class Solution {

    public int method1(int[] nums, int target) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == target) {
                var1 = i;
                break;
            } else {
                var1 = -1;
            }
        }
        return var1;
    }
}
