class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        int var1 = nums[0];
        int var2 = 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] < var1) {
                var1 = nums[i];
                var2 = i;
            }
        }
        int li = 0;
        int ri = var2 - 1;
        while (li <= ri) {
            int mid = (ri + li) / 2;
            if (nums[mid] == target) {
                return mid;
            } else if (nums[mid] > target) {
                ri = mid - 1;
            }
            if (nums[mid] < target) {
                li = mid + 1;
            }
        }
        li = var2;
        ri = n - 1;
        while (li <= ri) {
            int mid = (ri + li) / 2;
            if (nums[mid] == target) {
                return mid;
            } else if (nums[mid] > target) {
                ri = mid - 1;
            }
            if (nums[mid] < target) {
                li = mid + 1;
            }
        }
        return -1;
    }
}
