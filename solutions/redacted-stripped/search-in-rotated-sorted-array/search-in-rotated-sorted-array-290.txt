class Solution {

    public int method1(int[] nums, int target) {
        int count = 0;
        boolean flag = false;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == target) {
                flag = true;
                return count;
            } else
                ++count;
        }
        return flag ? count : -1;
    }
}
