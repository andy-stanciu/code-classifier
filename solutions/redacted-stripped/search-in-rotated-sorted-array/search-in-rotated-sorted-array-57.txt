public class Solution {

    private int method1(int[] nums, int target, int start, int end) {
        if (start > end || (start == end && nums[start] != target))
            return -1;
        if (start == end)
            return start;
        int mid = (start + end) / 2;
        if (nums[mid] < nums[start]) {
            if (target >= nums[start] || target <= nums[mid])
                return method1(nums, target, start, mid);
            else
                return method1(nums, target, mid + 1, end);
        } else {
            if (target >= nums[start] && target <= nums[mid])
                return method1(nums, target, start, mid);
            else
                return method1(nums, target, mid + 1, end);
        }
    }

    public int method2(int[] nums, int target) {
        return method1(nums, target, 0, nums.length - 1);
    }
}
