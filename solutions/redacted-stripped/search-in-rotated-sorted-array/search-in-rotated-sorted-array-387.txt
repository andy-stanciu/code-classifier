class Solution {

    public int method1(int[] nums, int target) {
        if (nums[0] <= nums[nums.length - 1])
            return BS(nums, 0, nums.length - 1, target);
        else {
            int var1 = method2(nums);
            int res1 = BS(nums, 0, var1 - 1, target);
            if (res1 != -1)
                return res1;
            int res2 = BS(nums, var1, nums.length - 1, target);
            if (res2 != -1)
                return res2;
        }
        return -1;
    }

    public int method2(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        while (start <= end) {
            int mid = start + ((end - start) / 2);
            int next = (mid + 1) % nums.length;
            int prev = (mid + nums.length - 1) % nums.length;
            if (nums[mid] < nums[prev])
                return mid;
            else if (nums[mid] > nums[next])
                return next;
            else if (nums[mid] <= nums[end])
                end = mid - 1;
            else if (nums[mid] >= nums[start])
                start = mid + 1;
        }
        return 0;
    }

    public int BS(int[] nums, int start, int end, int target) {
        while (start <= end) {
            int mid = start + ((end - start) / 2);
            if (nums[mid] == target)
                return mid;
            else if (nums[mid] > target)
                end = mid - 1;
            else if (nums[mid] < target)
                start = mid + 1;
        }
        return -1;
    }
}
