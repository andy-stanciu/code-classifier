class Solution {

    public int method1(int[] nums, int target) {
        int x = 0;
        int n = nums.length - 1;
        if (nums[0] == target) {
            return 0;
        } else if (nums[n] == target) {
            return n;
        } else {
            for (int i = 0; i < nums.length; i++) if (nums[i] == target) {
                x = i;
                return x;
            }
        }
        return -1;
    }
}
