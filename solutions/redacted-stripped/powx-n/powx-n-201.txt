class Solution {

    public double myPow(double x, int n) {
        long m = n;
        if (m < 0) {
            m = -1 * m;
        }
        double res = 1;
        while (m > 0) {
            if (m % 2 != 0) {
                m--;
                res = res * x;
            } else {
                x = x * x;
                m = m / 2;
            }
        }
        if (n < 0) {
            res = 1 / res;
        }
        return res;
    }
}
