class Solution {

    public static double cal(double x, int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return x;
        double t = cal(x, n / 2);
        if (n % 2 == 0) {
            return t * t;
        } else {
            return x * t * t;
        }
    }

    public double myPow(double x, int n) {
        double ans = cal(x, Math.abs(n));
        if (n >= 0) {
            return ans;
        } else {
            return 1 / ans;
        }
    }
}
