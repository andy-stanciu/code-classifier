class Solution {

    public double myPow(double x, int n) {
        if (n < 0)
            x = 1.0 / x;
        return calculatePow(x, n);
    }

    private double calculatePow(double x, int n) {
        if (n == 0)
            return 1;
        if ((n % 2 + 2) % 2 == 1) {
            return x * calculatePow(x * x, n / 2);
        } else
            return calculatePow(x * x, n / 2);
    }
}
