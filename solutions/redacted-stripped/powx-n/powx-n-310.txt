class Solution {

    public double myPow(double x, long n) {
        if (n == 0)
            return 1;
        else if (n == 1)
            return x;
        else if (n >= 2) {
            long q = n / 2;
            long r = n % 2;
            return (myPow((x * x), q) * myPow(x, r));
        } else {
            n = -n;
            return (1 / myPow(x, n));
        }
    }
}
