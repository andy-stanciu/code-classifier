class Solution {

    public double myPow(double x, int n) {
        long power = n;
        if (power < 0) {
            x = 1 / x;
            power = -power;
        }
        return solve(x, power);
    }

    public static double solve(double x, long pow) {
        if (pow == 0)
            return 1;
        if (pow == 1)
            return x;
        double ans = solve(x, pow / 2);
        if (pow % 2 == 0) {
            ans = ans * ans;
        } else {
            ans = ans * ans * x;
        }
        return ans;
    }
}
