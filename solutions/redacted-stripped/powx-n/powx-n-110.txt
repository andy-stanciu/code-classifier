class Solution {

    double result;

    public double myPow(double x, int n) {
        double res = helper(x, Math.abs(n));
        if (n >= 0) {
            return res;
        } else {
            return 1 / res;
        }
    }

    public double helper(double x, int n) {
        if (x == 0) {
            return 0;
        }
        if (n == 0) {
            return 1;
        }
        result = helper(x, n / 2);
        result = result * result;
        if (n % 2 == 0) {
            return result;
        } else {
            return x * result;
        }
    }
}
