class Solution {

    public double myPow(double x, int n) {
        double fac = 1;
        if (n < 0) {
            n = (n + 1) * -1;
            x = (1 / x);
            fac = x;
            if (n == 0)
                return x;
        }
        if (n == 0)
            return 1;
        if (n % 2 == 0)
            return fac * myPow(x * x, n / 2);
        return fac * x * myPow(x, n - 1);
    }
}
