class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1.0;
        }
        if (n < 0) {
            if (n == Integer.MIN_VALUE) {
                return 1 / (myPow(x, -(n + 1)) * x);
            } else {
                return 1 / myPow(x, -n);
            }
        }
        double halfPow = myPow(x, n / 2);
        double result = halfPow * halfPow;
        if (n % 2 != 0) {
            result *= x;
        }
        return result;
    }
}
