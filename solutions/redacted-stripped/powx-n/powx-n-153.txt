class Solution {

    public double myPow(double x, int n) {
        long m = n;
        return myRecPow(x, m);
    }

    private static double myRecPow(double x, long n) {
        if (n == 0)
            return 1;
        if (n < 0)
            return 1 / myRecPow(x, -n);
        double half = myRecPow(x, n / 2);
        if (n % 2 == 0)
            return half * half;
        else
            return x * half * half;
    }
}
