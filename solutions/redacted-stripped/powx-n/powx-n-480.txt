class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (n == 1) {
            return x;
        }
        long N = n;
        if (N < 0) {
            x = 1 / x;
            N = -N;
        }
        double powerHalf = myPow(x, (int) (N / 2));
        double powerSqr = powerHalf * powerHalf;
        if (N % 2 != 0) {
            powerSqr *= x;
        }
        return powerSqr;
    }
}
