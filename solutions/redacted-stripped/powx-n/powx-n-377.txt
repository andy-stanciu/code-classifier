class Solution {

    private double helper(double x, int n) {
        if (n == 0)
            return 1.0;
        double half = helper(x, n / 2);
        return (n % 2 == 0 ? half * half : half * half * x);
    }

    public double myPow(double x, int n) {
        if (x == 0)
            return 0;
        double ans = helper(x, Math.abs(n));
        return (n < 0 ? (1 / ans) : ans);
    }
}
