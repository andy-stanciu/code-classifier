class Solution {

    public double myPow(double n, int p) {
        if (p < 0)
            return Math.pow(n, p);
        double ans = 1;
        while (p > 0) {
            if ((p & 1) == 1) {
                ans *= n;
            }
            n *= n;
            p = p >> 1;
        }
        return ans;
    }
}
