class Solution {

    public double myPow(double x, int n) {
        return calc(x, n);
    }

    public double calc(double x, long n) {
        if (n < 0) {
            x = 1 / x;
            n = -1 * n;
        }
        if (n == 1)
            return x;
        else if (n == 0)
            return 1;
        else {
            double t = calc(x, n / 2);
            if (n % 2 == 0)
                return t * t;
            else
                return t * t * x;
        }
    }
}
