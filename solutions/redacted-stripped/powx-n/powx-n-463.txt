class Solution {

    public double myPow(double x, int n) {
        if (n == 1)
            return x;
        if (n == 0)
            return 1;
        if (x == 0 || x == 1)
            return x;
        if (x == -1) {
            return (n % 2 == 0) ? 1 : -1;
        }
        if (n < 0) {
            if (n == Integer.MIN_VALUE)
                n = -(n + 1);
            else
                n = -n;
            x = 1 / x;
        }
        double half = myPow(x, n / 2);
        if (n % 2 == 0)
            return half * half;
        else
            return half * half * x;
    }
}
