class Solution {

    double residule = 1;

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return x * residule;
        if (n == -1)
            return 1 / (x * residule);
        if (n % 2 == 0) {
            return myPow(x * x, n / 2);
        } else {
            residule = residule * x;
            return myPow(x * x, n / 2);
        }
    }
}
