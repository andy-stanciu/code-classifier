class Solution {

    public double myPow(double x, int n) {
        long dividing = n;
        double ans = 1;
        if (n < 0)
            dividing *= -1;
        while (dividing > 0) {
            if (dividing % 2 == 0) {
                x = x * x;
                dividing = dividing / 2;
            } else {
                ans = ans * x;
                dividing = dividing - 1;
            }
        }
        if (n < 0)
            ans = 1 / ans;
        return ans;
    }
}
