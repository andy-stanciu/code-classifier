class Solution {

    double pow(double a, long n) {
        if (n == 0)
            return 1;
        double ans = pow(a, n / 2);
        ans = ans * ans;
        if (n % 2 == 1)
            ans = ans * a;
        return ans;
    }

    public double myPow(double x, int n) {
        long nn = n;
        if (n >= 0)
            return pow(x, nn);
        else
            return 1.0 / pow(x, -nn);
    }
}
