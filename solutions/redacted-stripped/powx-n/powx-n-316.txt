class Solution {

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return x;
        if (n == -1)
            return 1 / x;
        int half = n / 2;
        double halfPow = myPow(x, half);
        if (n % 2 == 0) {
            return halfPow * halfPow;
        } else {
            if (n > 0)
                return halfPow * halfPow * myPow(x, 1);
            else
                return halfPow * halfPow * myPow(x, -1);
        }
    }
}
