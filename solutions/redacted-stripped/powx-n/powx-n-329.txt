class Solution {

    public double myPow(double x, int n) {
        if (x == -1 && n < 0) {
            return 1;
        }
        if (n == 0) {
            return 1;
        }
        if (n == Integer.MIN_VALUE) {
            x = 1 / x;
            n = Integer.MAX_VALUE;
        } else if (n < 0) {
            x = 1 / x;
            n = -n;
        }
        if (n % 2 == 0) {
            double half = myPow(x, n / 2);
            return half * half;
        }
        return x * myPow(x, n - 1);
    }
}
