class Solution {

    public double myPow(double x, int n) {
        if (n < 0)
            return (1 / x) * myPow(1 / x, -(n + 1));
        if (n == 0)
            return 1.0;
        if (n == 1)
            return x;
        double res = myPow(x * x, n / 2);
        if (n % 2 == 1)
            res *= x;
        return res;
    }
}
