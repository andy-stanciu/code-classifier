class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        long nl = n;
        if (nl > 0)
            return rec(x, nl);
        else
            return 1 / rec(x, -nl);
    }

    double rec(double x, long n) {
        if (n == 1) {
            return x;
        }
        double v = rec(x, n / 2);
        if (n % 2 == 0) {
            return v * v;
        } else {
            return v * v * x;
        }
    }
}
