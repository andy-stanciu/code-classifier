class Solution {

    public double myPow(double x, int n) {
        if (x == 0.0) {
            return 0.0;
        }
        long power = Math.abs((long) n);
        double result = 1.0;
        while (power > 0) {
            if ((power % 2) == 1) {
                result *= x;
            }
            x *= x;
            power /= 2;
        }
        return (n < 0) ? 1.0 / result : result;
    }
}
