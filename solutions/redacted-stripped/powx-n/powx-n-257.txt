class Solution {

    public double myPow(double x, int n) {
        if (x == 0)
            return 0;
        if (n == 0)
            return 1;
        if (n == Integer.MIN_VALUE)
            return 1 / x * myPow(x, n + 1);
        if (n < 0) {
            n = -n;
            x = 1 / x;
        }
        double half = myPow(x, n / 2);
        double result = half * half;
        return n % 2 == 0 ? result : result * x;
    }
}
