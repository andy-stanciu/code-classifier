class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        double sub = myPow(x, n / 2);
        if (n % 2 == 1) {
            return sub * sub * x;
        } else if (n % 2 == -1) {
            return sub * sub * 1 / x;
        } else {
            return sub * sub;
        }
    }
}
