class Solution {

    public double myPow(double x, int n) {
        if (n < 0)
            x = 1 / x;
        if (x == 0)
            return 0;
        return myrec(x, n);
    }

    public double myrec(double x, int n) {
        if (n == 0) {
            return 1;
        }
        double sol = myrec(x, n / 2);
        sol *= sol;
        if ((n & 1) == 1) {
            sol *= x;
        }
        return sol;
    }
}
