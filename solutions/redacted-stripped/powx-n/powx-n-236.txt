class Solution {

    public double myPow(double x, int n) {
        double ans = 1;
        if (n < 0) {
            if (n == Integer.MIN_VALUE) {
                n = Integer.MAX_VALUE;
                x = 1 / x;
                ans = x;
            } else {
                x = 1 / x;
                n = -n;
            }
        }
        while (n > 0) {
            if (n % 2 == 1) {
                ans = ans * x;
            }
            n = n / 2;
            x = x * x;
        }
        return ans;
    }
}
