class Solution {

    public double myPow(double x, int n) {
        double ans = 1.0;
        if (x == 1) {
            return 1.0;
        }
        if (x == -1) {
            return (n & 1) == 1 ? -1.0 : 1.0;
        }
        long p = n;
        if (p < 0) {
            x = 1 / x;
            p = -p;
        }
        while (p > 0) {
            if ((p & 1) == 1) {
                ans *= x;
            }
            x *= x;
            p = p >> 1;
        }
        return ans;
    }
}
