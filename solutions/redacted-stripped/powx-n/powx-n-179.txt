class Solution {

    public double myPow(double x, int n) {
        double res = 1.0d;
        if (n == Integer.MIN_VALUE) {
            if (x == 1 || x == -1)
                return 1.0;
            return 0.0;
        }
        if (n >= Integer.MAX_VALUE) {
            if (x == 1 || x == -1)
                return x;
            return 0.0;
        }
        if (n == 0) {
            return res;
        }
        boolean isValid = n < Integer.MAX_VALUE - 1 && n > Integer.MIN_VALUE;
        if (isValid) {
            if (n < 0) {
                x = 1.0d / x;
                n *= -1;
            }
            for (int i = 0; i < n; i++) {
                res *= x;
            }
        } else {
            res = 0.0d;
        }
        return res;
    }
}
