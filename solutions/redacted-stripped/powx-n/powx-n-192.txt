class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (n < 0) {
            x = 1 / x;
            if (n == Integer.MIN_VALUE) {
                n += 1;
                x *= x;
            }
            n = -n;
        }
        double halfPower = myPow(x, n / 2);
        double hPowerSq = halfPower * halfPower;
        if (n % 2 != 0) {
            hPowerSq *= x;
        }
        return hPowerSq;
    }
}
