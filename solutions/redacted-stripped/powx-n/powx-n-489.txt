class Solution {

    public double myPow(double x, int n) {
        double result = 1.00000;
        if (x == 1.00000) {
            return x;
        }
        if (x == -1.00000) {
            if (n % 2 == 0) {
                return 1.00000;
            } else {
                return -1.00000;
            }
        }
        if (n > 0) {
            for (int i = 0; i < n; i++) {
                result = result * x;
            }
        } else if (n == 0) {
            result = 1;
        } else {
            for (int i = (-n); i > 0; i--) {
                result = result / x;
            }
        }
        if (n <= Integer.MIN_VALUE) {
            return 0.00000;
        }
        return result;
    }
}
