class Solution {

    public double myPow(double x, int n) {
        if (n < 0) {
            return 1 / pow(x, (n * (-1)));
        } else {
            return pow(x, n);
        }
    }

    public double pow(double a, int n) {
        if (n == 0) {
            return 1;
        }
        if (n == 1) {
            return a;
        }
        double temp = pow(a, n / 2);
        if (n % 2 == 0) {
            return temp * temp;
        } else {
            return temp * temp * a;
        }
    }
}
