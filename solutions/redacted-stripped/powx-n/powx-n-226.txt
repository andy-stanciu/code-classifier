class Solution {

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return x;
        int exp = n / 2;
        if (n < 0) {
            exp = Math.abs(n / 2);
        }
        double halfResult = n > 0 ? myPow(x, exp) : 1.0 / myPow(x, exp);
        if (n % 2 == 0) {
            return halfResult * halfResult;
        }
        return n > 0 ? halfResult * halfResult * x : halfResult * halfResult * 1.0 / x;
    }
}
