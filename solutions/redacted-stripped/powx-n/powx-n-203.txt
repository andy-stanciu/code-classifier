class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1;
        }
        if (x == 0) {
            return 0;
        }
        double pow = 1d;
        pow = myPow(x, n / 2);
        if (n % 2 == 0) {
            return pow * pow;
        } else if (n > 0) {
            return x * pow * pow;
        } else {
            return (1 / x) * pow * pow;
        }
    }
}
