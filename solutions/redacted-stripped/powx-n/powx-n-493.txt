class Solution {

    public double answer(double x, long n) {
        if (n == 0) {
            return 1.0;
        }
        if (n < 0) {
            return 1 / answer(x, -n);
        }
        double half = answer(x, n / 2);
        if (n % 2 == 0) {
            return half * half;
        } else {
            return half * half * x;
        }
    }

    public double myPow(double x, int n) {
        return answer(x, (long) n);
    }
}
