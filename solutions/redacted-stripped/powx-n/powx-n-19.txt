class Solution {

    public double myPow(double x, int n) {
        if (n > 0) {
            return (myPow2(x, n));
        } else {
            return (1 / myPow2(x, n));
        }
    }

    public static double myPow2(double x, int n) {
        if (n == 0) {
            return 1.00000;
        }
        double temp = myPow2(x, n / 2);
        if (n % 2 == 0) {
            return temp * temp;
        } else
            return (x * temp * temp);
    }
}
