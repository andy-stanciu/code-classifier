class Solution {

    public double myPow(double x, int n) {
        return pow(x, (long) n);
    }

    double pow(double x, long p) {
        if (p == 0)
            return 1.0;
        if (p < 0)
            return pow(1 / x, -p);
        if (p % 2 == 0)
            return pow(x * x, p / 2);
        else
            return x * pow(x * x, (p - 1) / 2);
    }
}
