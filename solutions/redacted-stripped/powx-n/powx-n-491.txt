class Solution {

    public double myPow(double x, int n) {
        if (x == 1)
            return 1;
        if (n == 0)
            return 1;
        if (n == Integer.MIN_VALUE) {
            x = 1 / x;
            n = Integer.MAX_VALUE;
            return myPow(x, n) * x;
        }
        if (n < 0) {
            x = 1 / x;
            n = -n;
        }
        double result = 1;
        double current_product = x;
        while (n > 0) {
            if (n % 2 == 1) {
                result *= current_product;
            }
            current_product *= current_product;
            n /= 2;
        }
        return result;
    }
}
