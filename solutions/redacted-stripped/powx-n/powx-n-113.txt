class Solution {

    public static double myPow(double x, int n) {
        double ans = pow(x, n, n);
        return ans;
    }

    public static double pow(double x, int n, int k) {
        if (n == 0)
            return 1;
        double help = pow(x, n / 2, k), ans = 0;
        ans = help * help;
        if (n % 2 != 0) {
            ans = x * ans;
        }
        if (k < 0 && n == k)
            return 1 / ans;
        return ans;
    }
}
