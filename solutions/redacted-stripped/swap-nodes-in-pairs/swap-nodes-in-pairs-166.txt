class Solution {

    public ListNode method1(ListNode head) {
        ListNode d = new ListNode(0);
        d.next = head;
        ListNode prev = d;
        ListNode curr = d.next;
        while (curr != null) {
            ListNode curr_next = curr.next;
            if (curr_next == null) {
                break;
            }
            curr.next = curr_next.next;
            curr_next.next = prev.next;
            prev.next = curr_next;
            prev = curr;
            curr = curr.next;
        }
        return d.next;
    }
}
