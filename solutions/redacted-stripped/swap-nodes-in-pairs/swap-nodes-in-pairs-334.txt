class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        return method2(head);
    }

    public ListNode method2(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode left = head;
        ListNode right = head.next;
        ListNode temp = right.next;
        right.next = left;
        left.next = temp;
        head = right;
        left.next = method2(left.next);
        return head;
    }
}
