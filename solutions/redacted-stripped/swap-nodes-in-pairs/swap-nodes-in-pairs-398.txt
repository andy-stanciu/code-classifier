class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode l = head, r = head.next;
        ListNode dummy = new ListNode(-1);
        ListNode first = dummy;
        while (l != null && r != null) {
            ListNode temp = r.next;
            r.next = null;
            dummy.next = r;
            r.next = l;
            l.next = temp;
            dummy = l;
            l = temp;
            if (temp != null)
                r = temp.next;
        }
        return first.next;
    }
}
