class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode dummy = new ListNode();
        ListNode pre = dummy;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            pre.next = curr.next;
            curr.next = pre.next.next;
            pre.next.next = curr;
            pre = curr;
            curr = curr.next;
        }
        head = dummy.next;
        return head;
    }
}
