class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode a = head;
        ListNode b = head.next;
        ListNode c = head.next.next;
        head = b;
        if (c == null) {
            a.next = c;
            b.next = a;
            return head;
        }
        while (a != null) {
            b.next = a;
            a.next = c;
            if (c == null) {
                return head;
            }
            if (c.next == null) {
                return head;
            }
            a.next = c.next;
            a = c;
            b = c.next;
            c = c.next.next;
        }
        return head;
    }
}
