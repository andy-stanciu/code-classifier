class Solution {

    public ListNode method1(ListNode head) {
        ListNode node = head;
        if (node == null || node.next == null)
            return node;
        ListNode nodeNextNext = method1(node.next.next);
        ListNode tmp = node.next;
        node.next = nodeNextNext;
        tmp.next = node;
        return tmp;
    }
}
