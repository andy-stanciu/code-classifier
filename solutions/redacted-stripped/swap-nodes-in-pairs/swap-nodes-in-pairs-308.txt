class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        int i = 1;
        ListNode temp = head.next;
        ListNode pre = head;
        while (temp != null) {
            if (i % 2 != 0) {
                int d = temp.val;
                temp.val = pre.val;
                pre.val = d;
            }
            pre = pre.next;
            temp = temp.next;
            i++;
        }
        return head;
    }
}
