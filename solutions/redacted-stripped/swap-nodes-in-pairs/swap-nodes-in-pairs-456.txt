class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head;
        ListNode tempNext = head.next;
        ListNode ans = tempNext;
        while (temp != null && temp.next != null) {
            ListNode t = tempNext.next;
            tempNext.next = temp;
            if (t == null || t.next == null) {
                temp.next = t;
                break;
            }
            temp.next = t.next;
            temp = t;
            tempNext = temp.next;
        }
        return ans;
    }
}
