class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode ptr1 = head;
        ListNode ptr2 = head.next;
        ListNode ptr3 = head.next.next;
        ptr2.next = ptr1;
        ptr1.next = method1(ptr3);
        return ptr2;
    }
}
