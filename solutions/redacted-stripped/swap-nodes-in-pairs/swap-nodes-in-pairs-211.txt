class Solution {

    public ListNode method1(ListNode head) {
        int temp = 0;
        ListNode dummy = head;
        while (head != null && head.next != null) {
            temp = head.val;
            head.val = head.next.val;
            head.next.val = temp;
            head = head.next.next;
        }
        return dummy;
    }
}
