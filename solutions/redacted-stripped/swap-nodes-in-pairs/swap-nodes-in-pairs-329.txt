class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode var1 = dummy;
        while (var1.next != null && var1.next.next != null) {
            ListNode var2 = var1.next;
            ListNode var3 = var1.next.next;
            var2.next = var3.next;
            var3.next = var2;
            var1.next = var3;
            var1.next.next = var2;
            var1 = var1.next.next;
        }
        return dummy.next;
    }
}
