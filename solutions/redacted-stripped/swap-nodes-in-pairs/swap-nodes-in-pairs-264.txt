class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode node = new ListNode(0), b;
        node.next = head;
        b = node;
        while (head != null && head.next != null) {
            b.next = head.next;
            head.next = head.next.next;
            b.next.next = head;
            b = b.next.next;
            head = head.next;
        }
        return node.next;
    }
}
