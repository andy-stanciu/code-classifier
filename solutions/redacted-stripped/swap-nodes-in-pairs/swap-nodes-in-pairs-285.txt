class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head;
        head = temp.next;
        while (temp != null && temp.next != null) {
            ListNode node = temp.next.next;
            temp.next.next = temp;
            if (node == null || node.next == null) {
                temp.next = node;
            } else {
                temp.next = node.next;
            }
            temp = node;
        }
        return head;
    }
}
