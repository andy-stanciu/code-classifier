class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode current = head, prev = head, forward = current.next;
        while (forward != null) {
            if (current == head) {
                head = head.next;
                current.next = forward.next;
                forward.next = current;
            } else {
                prev.next = current.next;
                current.next = forward.next;
                forward.next = current;
            }
            prev = current;
            current = current.next;
            if (current != null) {
                forward = current.next;
            } else
                break;
        }
        return head;
    }
}
