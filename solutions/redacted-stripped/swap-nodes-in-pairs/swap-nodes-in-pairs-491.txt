class Solution {

    public ListNode method1(ListNode head) {
        int k = 2;
        return method2(head, k);
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null) {
            return head;
        }
        ListNode curr = head;
        ListNode prev = null, next = null;
        int count = 0;
        while (curr != null && count < k) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        if (curr != null) {
            head.next = method2(curr, k);
        }
        return prev;
    }
}
