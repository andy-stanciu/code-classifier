class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        head.next.next = method1(head.next.next);
        ListNode temp = head.next;
        head.next = temp.next;
        temp.next = head;
        head = temp;
        return head;
    }
}
