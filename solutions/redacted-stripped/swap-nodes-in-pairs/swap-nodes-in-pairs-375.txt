class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        ListNode prev = null;
        while (curr != null && curr.next != null) {
            ListNode forward = curr.next;
            curr.next = forward.next;
            forward.next = curr;
            if (prev != null)
                prev.next = forward;
            else
                head = forward;
            System.out.println(curr.val);
            prev = curr;
            curr = curr.next;
        }
        return head;
    }
}
