class Solution {

    public ListNode method1(ListNode head) {
        ListNode nh = new ListNode();
        ListNode pre = nh;
        while (head != null) {
            int temp = head.val;
            head = head.next;
            if (head != null) {
                pre.next = new ListNode(head.val);
                pre = pre.next;
                head = head.next;
            }
            pre.next = new ListNode(temp);
            pre = pre.next;
        }
        return nh.next;
    }
}
