class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode first = head;
        ListNode second = head.next;
        while (first != null && first.next != null) {
            int temp = first.val;
            first.val = second.val;
            second.val = temp;
            first = second.next;
            if (first != null)
                second = first.next;
        }
        return head;
    }
}
