class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode res = new ListNode();
        res.next = head;
        ListNode var1 = res;
        while (var1 != null && var1.next != null && var1.next.next != null) {
            ListNode futureNode = var1.next.next.next;
            ListNode temp = var1.next;
            var1.next = var1.next.next;
            var1.next.next = temp;
            temp.next = futureNode;
            var1 = var1.next.next;
        }
        return res.next;
    }
}
