class Solution {

    public ListNode method1(ListNode head) {
        ListNode dumy = new ListNode(-1, head);
        ListNode prev = dumy;
        ListNode cur = head;
        while (cur != null && cur.next != null) {
            ListNode aux = cur.next.next;
            ListNode second = cur.next;
            second.next = cur;
            cur.next = aux;
            prev.next = second;
            prev = cur;
            cur = aux;
        }
        return dumy.next;
    }
}
