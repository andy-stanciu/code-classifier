class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode ptr = head;
        ListNode n = ptr.next;
        head = head.next;
        ListNode temp;
        while (true) {
            ptr.next = n.next;
            n.next = ptr;
            temp = ptr;
            ptr = ptr.next;
            if (ptr == null || ptr.next == null)
                break;
            n = ptr.next;
            temp.next = n;
        }
        return head;
    }
}
