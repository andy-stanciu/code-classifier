class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head;
        ListNode nextHead = head;
        int count = 0;
        while (curr != null && count < 2) {
            curr = curr.next;
            nextHead = curr;
            count++;
        }
        if (count < 2)
            nextHead = null;
        ListNode prev = head;
        ListNode next = head.next;
        next.next = prev;
        prev.next = method1(nextHead);
        return next;
    }
}
