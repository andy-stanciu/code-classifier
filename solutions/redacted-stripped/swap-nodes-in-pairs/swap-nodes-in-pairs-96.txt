class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode n1 = head;
        ListNode n2 = head.next;
        ListNode next = n2.next;
        n2.next = n1;
        n1.next = next;
        if (next != null)
            n1.next = method1(next);
        return n2;
    }
}
