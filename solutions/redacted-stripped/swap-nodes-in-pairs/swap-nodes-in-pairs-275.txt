class Solution {

    public ListNode method1(ListNode head) {
        ListNode current = head;
        if (head == null)
            return null;
        int length = 0;
        while (current.next != null) {
            length++;
            current = current.next;
        }
        current = head;
        int i = 0, temp = 0;
        while (i < length && current.next != null) {
            if (i % 2 == 0) {
                temp = current.val;
                current.val = current.next.val;
                current.next.val = temp;
            }
            current = current.next;
            i++;
        }
        return head;
    }
}
