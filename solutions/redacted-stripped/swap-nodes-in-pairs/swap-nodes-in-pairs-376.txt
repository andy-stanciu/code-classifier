class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode dummy = new ListNode();
        ListNode prev = dummy;
        prev.next = head;
        ListNode first = head;
        while (first != null && first.next != null) {
            ListNode second = first.next;
            ListNode nextFirst = second.next;
            prev.next = second;
            second.next = first;
            first.next = nextFirst;
            prev = first;
            first = nextFirst;
        }
        return dummy.next;
    }
}
