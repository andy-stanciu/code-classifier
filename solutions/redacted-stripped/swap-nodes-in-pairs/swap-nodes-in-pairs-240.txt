class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = new ListNode(-1);
        temp.next = head;
        ListNode slow = temp;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            ListNode a = fast;
            ListNode b = fast.next;
            a.next = b.next;
            b.next = a;
            slow.next = b;
            slow = a;
            fast = a.next;
        }
        return temp.next;
    }
}
