class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode prev = null;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            ListNode h1 = curr;
            ListNode h2 = curr.next;
            h1.next = h2.next;
            h2.next = h1;
            if (prev != null) {
                prev.next = h2;
            } else {
                head = h2;
            }
            prev = h1;
            curr = h1.next;
        }
        return head;
    }
}
