class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode smallAns = method1(head.next.next);
        ListNode curr = head, next = head.next;
        curr.next = smallAns;
        next.next = curr;
        return next;
    }
}
