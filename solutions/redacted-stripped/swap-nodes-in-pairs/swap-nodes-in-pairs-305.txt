class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode newHead = head.next;
        ListNode curr = head;
        ListNode prev = null;
        while (curr != null && curr.next != null) {
            ListNode adj = curr.next;
            curr.next = adj.next;
            adj.next = curr;
            if (prev != null) {
                prev.next = adj;
            }
            prev = curr;
            curr = curr.next;
        }
        return newHead;
    }
}
