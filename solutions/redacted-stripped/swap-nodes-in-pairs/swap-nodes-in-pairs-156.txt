class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0, head);
        ListNode cur = dummy;
        while (cur.next != null && cur.next.next != null) {
            int temp = cur.next.val;
            cur.next.val = cur.next.next.val;
            cur.next.next.val = temp;
            cur = cur.next.next;
        }
        return dummy.next;
    }
}
