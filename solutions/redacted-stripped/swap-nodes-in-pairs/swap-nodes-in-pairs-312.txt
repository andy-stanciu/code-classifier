class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        ListNode rh = head;
        ListNode prev = null;
        while (curr != null) {
            if (curr.next != null) {
                ListNode nn = curr.next.next;
                ListNode nx = curr.next;
                ListNode c = curr;
                nx.next = c;
                if (prev != null)
                    prev.next = nx;
                else
                    rh = nx;
                c.next = nn;
                prev = c;
                curr = nn;
            } else
                break;
        }
        return rh;
    }
}
