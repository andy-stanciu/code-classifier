class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp1 = head;
        ListNode temp2 = head.next;
        ListNode temp3 = null;
        ListNode prev = null;
        head = head.next;
        while (temp2 != null) {
            temp3 = temp2.next;
            temp2.next = temp1;
            temp1.next = temp3;
            if (prev != null) {
                prev.next = temp2;
                prev = null;
            }
            prev = temp1;
            temp1 = temp3;
            temp2 = temp3 == null ? null : temp3.next;
        }
        return head;
    }
}
