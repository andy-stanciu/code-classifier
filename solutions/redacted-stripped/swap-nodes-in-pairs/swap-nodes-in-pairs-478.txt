class Solution {

    public ListNode method1(ListNode head) {
        ListNode result = head;
        if (result == null || (result != null && result.next == null)) {
            return result;
        } else {
            int temp = result.next.val;
            result.next.val = result.val;
            result.val = temp;
            result.next.next = method1(result.next.next);
            return result;
        }
    }
}
