class Solution {

    public ListNode method1(ListNode head) {
        ListNode dum = new ListNode(0, head);
        ListNode prev = dum;
        ListNode curr = head;
        if (head == null || head.next == null)
            return head;
        while (curr != null && curr.next != null) {
            ListNode third = curr.next.next;
            ListNode next = curr.next;
            next.next = curr;
            curr.next = third;
            prev.next = next;
            prev = curr;
            curr = third;
        }
        return dum.next;
    }
}
