class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode n2 = head.next.next;
        ListNode n1 = head.next;
        head.next = n2;
        n1.next = head;
        n1.next.next = method1(n1.next.next);
        return n1;
    }
}
