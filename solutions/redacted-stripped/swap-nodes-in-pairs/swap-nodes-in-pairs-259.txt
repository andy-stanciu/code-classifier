class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        var temp = new ListNode(head.val, head.next);
        head.val = head.next.val;
        if (head.next.next != null) {
            head.next = new ListNode(temp.val, method1(head.next.next));
        } else {
            head.next = new ListNode(temp.val);
        }
        return head;
    }
}
