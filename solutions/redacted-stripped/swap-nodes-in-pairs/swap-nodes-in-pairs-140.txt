class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode beg = head.next;
        ListNode cur = head;
        ListNode sec = beg;
        ListNode prev = null;
        while (cur != null && sec != null) {
            cur.next = sec.next;
            sec.next = cur;
            if (prev != null) {
                prev.next = sec;
            }
            prev = cur;
            cur = cur.next;
            if (cur == null)
                return beg;
            sec = cur.next;
        }
        return beg;
    }
}
