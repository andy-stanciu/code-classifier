class Solution {

    public ListNode method1(ListNode head) {
        ListNode tHead = null;
        ListNode trailer = null;
        ListNode sTrailer = null;
        ListNode first = head;
        int pos = 1;
        while (first != null) {
            if (pos % 2 == 0) {
                ListNode temp = first.next;
                first.next = trailer;
                trailer.next = temp;
                temp = trailer;
                trailer = first;
                first = temp;
                if (sTrailer != null) {
                    sTrailer.next = trailer;
                }
                if (tHead == null) {
                    tHead = trailer;
                }
            }
            if (pos == 3) {
                sTrailer = head;
            }
            if (pos > 3) {
                sTrailer = sTrailer.next;
            }
            trailer = first;
            first = first.next;
            pos++;
        }
        if (tHead == null) {
            return head;
        }
        return tHead;
    }
}
