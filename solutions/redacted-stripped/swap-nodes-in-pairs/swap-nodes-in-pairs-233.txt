class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode curr = head;
        ListNode next = curr.next;
        while (curr != null && next != null) {
            int temp = curr.val;
            curr.val = next.val;
            next.val = temp;
            curr = curr.next.next;
            if (curr != null) {
                next = next.next.next;
            }
        }
        return head;
    }
}
