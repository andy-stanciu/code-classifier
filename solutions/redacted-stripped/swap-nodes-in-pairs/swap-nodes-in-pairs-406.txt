class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode f = head;
        ListNode s = f.next;
        ListNode newHead = null;
        ListNode prev = null;
        while (s != null) {
            if (newHead == null) {
                ListNode temp = s.next;
                s.next = f;
                f.next = temp;
                newHead = s;
                if (temp != null) {
                    prev = f;
                    f = temp;
                    s = f.next;
                } else {
                    f = temp;
                    s = temp;
                }
            } else {
                ListNode temp = s.next;
                s.next = f;
                f.next = temp;
                prev.next = s;
                if (temp != null) {
                    prev = f;
                    f = temp;
                    s = f.next;
                } else {
                    f = temp;
                    s = temp;
                }
            }
        }
        return newHead;
    }
}
