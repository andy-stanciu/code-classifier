class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0, head);
        ListNode prev = dummy, curr = head;
        while (curr != null && curr.next != null) {
            ListNode first = curr.next.next;
            ListNode second = curr.next;
            second.next = curr;
            curr.next = first;
            prev.next = second;
            prev = curr;
            curr = first;
        }
        return dummy.next;
    }
}
