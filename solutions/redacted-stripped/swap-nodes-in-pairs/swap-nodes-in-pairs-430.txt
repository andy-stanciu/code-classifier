class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(0);
        var1.next = head;
        head = var1;
        while (var1.next != null && var1.next.next != null) {
            ListNode s1 = var1.next;
            ListNode s2 = var1.next.next;
            s1.next = s2.next;
            s2.next = s1;
            var1.next = s2;
            var1 = s1;
        }
        return head.next;
    }
}
