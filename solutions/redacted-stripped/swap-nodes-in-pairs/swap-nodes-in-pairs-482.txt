class Solution {

    public ListNode method1(ListNode head) {
        ListNode thead = head;
        while (thead != null && thead.next != null) {
            int val1 = thead.val;
            int val2 = thead.next.val;
            thead.val = val2;
            thead.next.val = val1;
            thead = thead.next.next;
        }
        return head;
    }
}
