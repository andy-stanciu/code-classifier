class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head.next.next;
        ListNode prev = head;
        head = head.next;
        head.next = prev;
        while (curr != null && curr.next != null) {
            prev.next = curr.next;
            prev = curr;
            ListNode nxt = curr.next.next;
            curr.next.next = curr;
            curr = nxt;
        }
        prev.next = curr;
        return head;
    }
}
