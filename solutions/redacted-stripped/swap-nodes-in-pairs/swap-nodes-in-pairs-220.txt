class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode even = head;
        ListNode odd = head.next;
        ListNode temp = null;
        head = odd;
        while (even != null && odd != null) {
            if (temp != null) {
                temp.next = odd;
            }
            even.next = odd.next;
            odd.next = even;
            temp = even;
            if (even.next == null || even.next.next == null)
                break;
            even = even.next;
            odd = even.next;
        }
        return head;
    }
}
