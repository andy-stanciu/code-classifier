class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head;
        while (temp.next != null) {
            int c = 0;
            c = temp.val;
            temp.val = temp.next.val;
            temp.next.val = c;
            if (temp.next.next != null)
                temp = temp.next.next;
            else
                temp = temp.next;
        }
        return head;
    }
}
