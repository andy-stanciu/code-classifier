class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head;
        head = curr.next;
        ListNode prev = null;
        while (curr != null && curr.next != null) {
            ListNode first = curr;
            ListNode second = curr.next;
            first.next = second != null ? second.next : null;
            second.next = first;
            if (prev != null) {
                prev.next = second;
            }
            prev = first;
            curr = curr.next;
        }
        return head;
    }
}
