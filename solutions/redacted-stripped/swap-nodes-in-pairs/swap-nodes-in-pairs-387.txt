class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode first = head;
        ListNode second = head.next;
        ListNode tail;
        ListNode prev = null;
        ListNode result = second;
        while (first != null && second != null) {
            tail = second.next;
            first.next = tail;
            second.next = first;
            if (prev != null) {
                prev.next = second;
            }
            if (tail == null) {
                break;
            }
            prev = first;
            first = tail;
            second = tail.next;
        }
        return result;
    }
}
