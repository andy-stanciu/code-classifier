class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode curr = dummy;
        ListNode temp1, temp2;
        while (curr.next != null && curr.next.next != null) {
            temp1 = curr.next;
            temp2 = curr.next.next;
            temp1.next = temp2.next;
            temp2.next = curr.next;
            curr.next = temp2;
            curr = curr.next.next;
        }
        return dummy.next;
    }
}
