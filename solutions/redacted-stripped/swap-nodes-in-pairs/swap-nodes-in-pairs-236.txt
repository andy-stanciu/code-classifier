class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        ListNode current = head;
        ListNode temp = null;
        ListNode t = null;
        ListNode tempHead = new ListNode(-1);
        ListNode cc = tempHead;
        while (current != null) {
            if (current.next != null) {
                temp = current.next.next;
                ListNode dummy2 = current.next;
                dummy2.next = null;
                ListNode dummy1 = current;
                dummy1.next = null;
                t = dummy1;
                dummy1 = dummy2;
                dummy2 = t;
                dummy1.next = dummy2;
                cc.next = dummy1;
                cc = cc.next.next;
                current = temp;
            } else {
                cc.next = current;
                current = current.next;
            }
        }
        return tempHead.next;
    }
}
