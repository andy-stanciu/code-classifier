class Solution {

    public ListNode method1(ListNode head) {
        ListNode tempHead1 = new ListNode(0);
        tempHead1.next = head;
        ListNode prev = tempHead1;
        ListNode temp = head;
        while (temp != null && temp.next != null) {
            ListNode temp1 = temp.next;
            temp.next = temp.next.next;
            temp1.next = temp;
            prev.next = temp1;
            prev = temp;
            temp = temp.next;
        }
        return tempHead1.next;
    }
}
