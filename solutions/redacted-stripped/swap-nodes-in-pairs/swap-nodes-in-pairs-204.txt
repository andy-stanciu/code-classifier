class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        int count = 1;
        ListNode curr = head;
        ListNode prev = null;
        while (curr != null) {
            if (count == 2) {
                int temp = prev.val;
                prev.val = curr.val;
                curr.val = temp;
                count = 1;
            } else {
                count++;
            }
            prev = curr;
            curr = curr.next;
        }
        return head;
    }
}
