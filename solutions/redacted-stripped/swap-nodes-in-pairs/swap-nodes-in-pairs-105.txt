class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode prv = head;
        ListNode curr = head.next;
        ListNode next = head.next.next;
        head = curr;
        head.next = prv;
        head.next.next = method1(next);
        return head;
    }
}
