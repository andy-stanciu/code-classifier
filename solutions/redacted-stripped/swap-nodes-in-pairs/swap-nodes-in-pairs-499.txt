class Solution {

    public ListNode method1(ListNode head) {
        ListNode current = head;
        ListNode prev = null;
        ListNode next = null;
        int count = 0, k = 2;
        while (count < k && current != null) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
            count++;
        }
        if (current != null) {
            head.next = method1(next);
        }
        head = prev;
        return head;
    }
}
