class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode var1 = head;
        ListNode proxy = head;
        ListNode prev = null;
        boolean wasHeadChanged = false;
        while (var1 != null && var1.next != null) {
            proxy = var1.next;
            var1.next = proxy.next;
            proxy.next = var1;
            if (prev != null)
                prev.next = proxy;
            if (!wasHeadChanged) {
                wasHeadChanged = true;
                head = proxy;
            }
            prev = var1;
            var1 = var1.next;
        }
        return head;
    }
}
