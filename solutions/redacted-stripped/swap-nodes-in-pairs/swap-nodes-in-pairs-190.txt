class Solution {

    public ListNode method1(ListNode head) {
        ListNode d = new ListNode(0);
        d.next = head;
        ListNode var1 = d;
        while (var1.next != null && var1.next.next != null) {
            ListNode s1 = var1.next;
            ListNode s2 = var1.next.next;
            s1.next = s2.next;
            s2.next = s1;
            var1.next = s2;
            var1 = s1;
        }
        return d.next;
    }
}
