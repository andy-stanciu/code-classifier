class Solution {

    public ListNode method1(ListNode head) {
        boolean flag = true;
        if (head == null || head.next == null) {
            return head;
        } else {
            ListNode curr = head.next.next;
            ListNode perv = head;
            ListNode temp;
            head = head.next;
            head.next = perv;
            while (curr != null && curr.next != null) {
                perv.next = curr.next;
                perv = curr;
                temp = curr.next.next;
                curr.next.next = curr;
                curr = temp;
            }
            perv.next = curr;
        }
        return head;
    }
}
