class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode curr = head.next.next, prev = head;
        head = head.next;
        prev.next = curr;
        head.next = prev;
        while (curr != null && curr.next != null) {
            prev.next = curr.next;
            prev = curr;
            ListNode temp = curr.next;
            curr.next = curr.next.next;
            temp.next = curr;
            curr = curr.next;
        }
        return head;
    }
}
