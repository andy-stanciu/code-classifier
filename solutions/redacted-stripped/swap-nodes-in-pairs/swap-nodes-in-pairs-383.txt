class Solution {

    public static ListNode method1(ListNode head) {
        ListNode grand = null;
        ListNode parent = head;
        ListNode child = head != null ? head.next : null;
        while (parent != null && child != null) {
            if (grand == null) {
                parent.next = child.next;
                child.next = parent;
                head = child;
            } else {
                parent.next = (grand.next = child).next;
                child.next = parent;
            }
            parent = child;
            child = child.next;
            parent = parent.next;
            child = child.next;
            grand = parent;
            if (child != null) {
                parent = parent.next;
                child = child.next;
            }
        }
        return head;
    }
}
