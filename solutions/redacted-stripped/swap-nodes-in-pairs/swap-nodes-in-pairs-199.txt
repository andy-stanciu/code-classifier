class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        ListNode result = null;
        if (head == null) {
            return head;
        } else if (head.next == null) {
            return head;
        }
        while (temp != null && temp.next != null) {
            int k = temp.val;
            temp.val = temp.next.val;
            temp.next.val = k;
            temp = temp.next.next;
        }
        return head;
    }
}
