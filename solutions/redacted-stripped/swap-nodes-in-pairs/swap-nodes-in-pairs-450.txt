class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode orange = null;
        ListNode purple = head;
        ListNode red = head.next;
        purple.next = red.next;
        red.next = purple;
        head = red;
        if (purple.next == null)
            return head;
        while (purple.next != null) {
            orange = purple;
            purple = purple.next;
            red = purple.next;
            if (red == null)
                break;
            purple.next = red.next;
            red.next = purple;
            orange.next = red;
        }
        return head;
    }
}
