class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode slow = head;
        ListNode fast = head.next;
        slow.next = fast.next;
        fast.next = slow;
        head = fast;
        fast = slow.next;
        while (fast != null && fast.next != null) {
            slow.next = fast.next;
            slow = slow.next;
            fast.next = slow.next;
            slow.next = fast;
            slow = slow.next;
            fast = fast.next;
        }
        return head;
    }
}
