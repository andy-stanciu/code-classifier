class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        if (head.next != null) {
            ListNode temp = head.next;
            head.next = temp.next;
            temp.next = head;
            head.next = method1(head.next);
            return temp;
        } else
            return head;
    }
}
