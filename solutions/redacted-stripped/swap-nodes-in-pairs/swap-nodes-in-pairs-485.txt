class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode i = new ListNode();
        ListNode j = new ListNode();
        ListNode temp = new ListNode();
        i = head;
        j = head.next;
        head = j;
        while (i != null && j != null) {
            if (temp == null) {
                temp = i;
            }
            i.next = j.next;
            j.next = i;
            temp.next = j;
            temp = i;
            i = i.next;
            if (i != null) {
                j = i.next;
            }
        }
        return head;
    }
}
