class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode tempHead = head;
        ListNode prev = null;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            ListNode firstPair = curr.next;
            ListNode secondPair = curr.next.next;
            if (prev == null) {
                head = firstPair;
            }
            firstPair.next = curr;
            curr.next = secondPair;
            if (prev != null) {
                prev.next = firstPair;
            }
            prev = curr;
            curr = secondPair;
        }
        return head;
    }
}
