class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        ListNode root = new ListNode(-1, head);
        ListNode prev = root;
        while (curr != null) {
            if (curr.next != null) {
                prev.next = curr.next;
                curr.next = curr.next.next;
                prev.next.next = curr;
            }
            prev = prev.next.next;
            curr = curr.next;
        }
        return root.next;
    }
}
