class Solution {

    public ListNode method1(ListNode head, ListNode currentNode) {
        if (currentNode == null || currentNode.next == null || head == null || head.next == null) {
            return head;
        }
        int temp = currentNode.val;
        currentNode.val = currentNode.next.val;
        currentNode.next.val = temp;
        return method1(head, currentNode.next.next);
    }

    public ListNode method2(ListNode head) {
        return method1(head, head);
    }
}
