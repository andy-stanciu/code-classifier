class Solution {

    public ListNode method1(ListNode head) {
        ListNode dup = new ListNode(0);
        dup.next = head;
        ListNode x = dup;
        while (x.next != null && x.next.next != null) {
            ListNode var1 = x.next;
            ListNode var2 = x.next.next;
            var1.next = var2.next;
            var2.next = var1;
            x.next = var2;
            x = var1;
        }
        return dup.next;
    }
}
