class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode next = head.next.next;
        ListNode temp = head;
        head = head.next;
        head.next = temp;
        head.next.next = method1(next);
        return head;
    }
}
