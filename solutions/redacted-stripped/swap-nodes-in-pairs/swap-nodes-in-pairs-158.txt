class Solution {

    private void method1(ListNode X) {
        ListNode Z = new ListNode();
        Z.next = X.next.next.next;
        X.next.next.next = X.next;
        X.next = X.next.next;
        X.next.next.next = Z.next;
        return;
    }

    public ListNode method2(ListNode head) {
        ListNode dummy = new ListNode();
        dummy.next = head;
        ListNode curNode = dummy;
        while (true) {
            if (curNode == null || curNode.next == null || curNode.next.next == null) {
                return dummy.next;
            } else {
                method1(curNode);
                curNode = curNode.next.next;
            }
        }
    }
}
