class Solution {

    public ListNode method1(ListNode head) {
        ListNode a = new ListNode(0);
        ListNode dummy = a;
        ListNode temp = head;
        while (temp != null && temp.next != null) {
            ListNode next = temp.next.next;
            temp.next.next = null;
            a.next = temp.next;
            a.next.next = temp;
            a.next.next.next = null;
            a = a.next.next;
            temp = next;
        }
        if (temp != null) {
            a.next = temp;
        }
        return dummy.next;
    }
}
