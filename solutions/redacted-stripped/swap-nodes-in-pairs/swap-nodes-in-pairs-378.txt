class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        else if (head.next == null)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy;
        ListNode curr = head;
        ListNode after = curr.next;
        ListNode temp = after.next;
        while (true) {
            prev.next = after;
            after.next = curr;
            curr.next = temp;
            prev.next = after;
            prev = curr;
            curr = temp;
            if (curr == null)
                break;
            after = curr.next;
            if (after == null)
                break;
            temp = after.next;
        }
        return dummy.next;
    }
}
