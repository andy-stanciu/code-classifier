class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        if (head == null || head.next == null) {
            return head;
        }
        while (temp != null && temp.next != null) {
            int val = temp.val;
            temp.val = temp.next.val;
            temp.next.val = val;
            temp = temp.next.next;
        }
        return head;
    }
}
