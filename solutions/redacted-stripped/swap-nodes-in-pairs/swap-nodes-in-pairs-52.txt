class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode temp = head.next;
        method2(head);
        return temp;
    }

    void method2(ListNode head) {
        if (head == null || head.next == null)
            return;
        ListNode temp = head.next;
        head.next = head.next.next;
        temp.next = head;
        ListNode temp2 = null;
        if (head.next != null)
            temp2 = head.next.next;
        method2(head.next);
        if (temp2 != null)
            head.next = temp2;
    }
}
