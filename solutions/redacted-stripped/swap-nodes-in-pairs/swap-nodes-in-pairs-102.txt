class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode var1 = new ListNode();
        var1.next = head;
        ListNode res = head.next;
        while (var1.next.next != null) {
            ListNode first = var1.next;
            ListNode second = var1.next.next;
            ListNode tail = second.next;
            var1.next = second;
            second.next = first;
            first.next = tail;
            var1 = var1.next.next;
            if (var1 == null || var1.next == null)
                break;
        }
        return res;
    }
}
