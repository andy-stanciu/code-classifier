class Solution {

    public ListNode method1(ListNode head) {
        ListNode adjacent_node = null;
        ListNode var1 = null;
        if (head == null || head.next == null)
            return head;
        var1 = head.next.next;
        adjacent_node = head.next;
        adjacent_node.next = head;
        head.next = method1(var1);
        return adjacent_node;
    }
}
