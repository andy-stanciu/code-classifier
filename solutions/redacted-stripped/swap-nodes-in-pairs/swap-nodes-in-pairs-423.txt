class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        if (head.next == null) {
            return head;
        }
        ListNode prv = null;
        ListNode cur = head;
        while (cur != null && cur.next != null) {
            prv = cur;
            cur = cur.next;
            int temp = cur.val;
            cur.val = prv.val;
            prv.val = temp;
            cur = cur.next;
        }
        return head;
    }
}
