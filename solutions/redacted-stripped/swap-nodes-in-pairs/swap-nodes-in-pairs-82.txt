class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(-1);
        ListNode prev = dummy;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            prev.next = curr.next;
            ListNode next = curr.next.next;
            curr.next.next = curr;
            prev = curr;
            curr = next;
        }
        prev.next = curr;
        return dummy.next;
    }
}

class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode newHead = head.next;
        ListNode var1 = method1(head.next.next);
        head.next = var1;
        newHead.next = head;
        return newHead;
    }
}
