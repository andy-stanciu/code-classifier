class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode temp;
        ListNode h2 = null;
        ListNode h3 = head;
        int B = 2;
        while (B > 0 && head != null) {
            temp = head;
            head = head.next;
            temp.next = h2;
            h2 = temp;
            B--;
        }
        h3.next = method1(head);
        return h2;
    }
}
