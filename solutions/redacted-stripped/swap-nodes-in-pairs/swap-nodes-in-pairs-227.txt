class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head;
        ListNode res = head.next;
        ListNode prev = null;
        while (curr != null && curr.next != null) {
            if (prev != null) {
                prev.next = curr.next;
                prev = curr;
            } else
                prev = curr;
            ListNode temp = curr.next.next;
            curr.next.next = curr;
            curr.next = temp;
            curr = curr.next;
        }
        return res;
    }
}
