class Solution {

    public ListNode method1(ListNode head) {
        ListNode d = new ListNode(0);
        d.next = head;
        ListNode t = d;
        while (t.next != null && t.next.next != null) {
            ListNode f = t.next;
            ListNode s = t.next.next;
            f.next = s.next;
            s.next = f;
            t.next = s;
            t = f;
        }
        return d.next;
    }
}
