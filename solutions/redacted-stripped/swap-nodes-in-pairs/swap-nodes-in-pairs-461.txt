class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode fast = null;
        ListNode cur = head;
        ListNode slow = null;
        while (cur != null && cur.next != null) {
            fast = cur.next;
            cur.next = fast.next;
            fast.next = cur;
            if (slow == null) {
                head = fast;
            } else {
                slow.next = fast;
            }
            slow = cur;
            cur = cur.next;
        }
        return head;
    }
}
