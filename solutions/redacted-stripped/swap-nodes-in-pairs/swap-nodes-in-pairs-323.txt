class Solution {

    public ListNode method1(ListNode head) {
        return helper(head);
    }

    private ListNode helper(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head.next;
        head.next = temp.next;
        temp.next = head;
        head = temp;
        head.next.next = helper(head.next.next);
        return head;
    }
}
