class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode pre = dummy;
        while (head != null && head.next != null) {
            ListNode temp = head;
            head = head.next;
            temp.next = head.next;
            head.next = temp;
            pre.next = head;
            pre = temp;
            head = temp.next;
        }
        return dummy.next;
    }
}
