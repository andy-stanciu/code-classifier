class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode dummy = head.next;
        ListNode prev = null;
        while (head != null && head.next != null) {
            if (prev != null) {
                prev.next = head.next;
            }
            prev = head;
            ListNode temp = head.next.next;
            head.next.next = head;
            head.next = temp;
            head = temp;
        }
        return dummy;
    }
}
