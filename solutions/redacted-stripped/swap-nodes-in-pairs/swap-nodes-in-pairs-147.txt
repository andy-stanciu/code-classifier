class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode previous = null;
        ListNode slow = head;
        ListNode fast = head.next;
        while (slow != null && fast != null) {
            slow.next = fast.next;
            fast.next = slow;
            if (previous != null) {
                previous.next = fast;
            } else {
                head = fast;
            }
            previous = slow;
            slow = slow.next;
            if (slow != null) {
                fast = slow.next;
            } else {
                break;
            }
        }
        return head;
    }
}
