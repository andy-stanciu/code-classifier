class Solution {

    public ListNode method1(ListNode head) {
        ListNode p = head;
        ListNode next;
        if (head != null && head.next != null) {
            next = head.next;
        } else {
            next = head;
        }
        ListNode q = null;
        while (p != null) {
            if (p.next != null) {
                if (q != null) {
                    q.next = p.next;
                }
                q = p;
                ListNode list = p.next.next;
                p.next.next = p;
                p.next = list;
                p = list;
            } else {
                p = null;
            }
        }
        return next;
    }
}
