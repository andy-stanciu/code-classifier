class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode temp = head;
        while (temp != null && temp.next != null) {
            int value = temp.val;
            temp.val = temp.next.val;
            temp.next.val = value;
            if (temp.next.next != null) {
                temp = temp.next.next;
            } else {
                break;
            }
        }
        return head;
    }
}
