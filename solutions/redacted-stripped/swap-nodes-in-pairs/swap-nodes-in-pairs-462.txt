class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode dummy = head;
        while (dummy != null && dummy.next != null) {
            int temp = dummy.val;
            dummy.val = dummy.next.val;
            dummy.next.val = temp;
            dummy = dummy.next.next;
        }
        return head;
    }
}
