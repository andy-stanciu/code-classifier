class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode start = head;
        ListNode end = head.next;
        ListNode prev = start;
        if (start.next != null && end.next == null) {
            end.next = head;
            head.next = null;
            head = end;
            return head;
        }
        head = head.next;
        while (start.next != null && end.next != null) {
            prev.next = end;
            prev = start;
            start.next = end.next;
            start = end.next;
            end.next = prev;
            end = start.next;
        }
        if (start.next != null) {
            end.next = prev.next;
            prev.next = start.next;
            start.next = null;
            return head;
        }
        return head;
    }
}
