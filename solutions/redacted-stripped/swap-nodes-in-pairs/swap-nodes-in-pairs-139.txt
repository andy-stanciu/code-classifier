class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode next = head.next;
        method2(head);
        return next == null ? head : next;
    }

    private ListNode method2(ListNode a) {
        if (a == null) {
            return a;
        }
        ListNode b = a.next;
        if (b != null) {
            ListNode c = b.next;
            b.next = a;
            a.next = method2(c);
        } else {
            return a;
        }
        return b;
    }
}
