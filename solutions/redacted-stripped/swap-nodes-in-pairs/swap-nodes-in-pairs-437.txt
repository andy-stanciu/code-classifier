class Solution {

    public ListNode method1(ListNode head) {
        int i = 0;
        int j = 1;
        int length = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            length++;
        }
        while (j < length) {
            head = reverse(head, i, j);
            i += 2;
            j += 2;
        }
        return head;
    }

    public ListNode reverse(ListNode head, int i, int j) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode previous = dummy;
        for (int k = 0; k < i; k++) {
            previous = previous.next;
        }
        ListNode present = previous.next;
        for (int k = i; k < j; k++) {
            ListNode future = present.next;
            present.next = future.next;
            future.next = previous.next;
            previous.next = future;
        }
        return dummy.next;
    }
}
