class Solution {

    public ListNode method1(ListNode head) {
        ListNode resp = null;
        int count = 1;
        if (head == null || head.next == null) {
            return head;
        }
        ListNode prev = null;
        ListNode prevPrev = prev;
        while (head != null) {
            if (count == 1) {
                resp = head.next;
            } else if (count % 2 == 0) {
                ListNode tmp = head.next;
                head.next = prev;
                prev.next = tmp;
                if (prevPrev != null)
                    prevPrev.next = head;
                head = head.next;
            }
            prevPrev = prev;
            prev = head;
            head = head.next;
            count++;
        }
        return resp;
    }
}
