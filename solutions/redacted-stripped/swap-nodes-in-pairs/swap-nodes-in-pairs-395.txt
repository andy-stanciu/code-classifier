class Solution {

    public ListNode method1(ListNode head) {
        ListNode l1 = head;
        ListNode l2 = head;
        ListNode l4 = head;
        if (l1 != null && l1.next != null) {
            head = l1.next;
            l1.next = l1.next.next;
            head.next = l1;
        }
        while (l1 != null && l1.next != null && l1.next.next != null) {
            l2 = l1.next;
            l4 = l1.next.next;
            l1.next = l4;
            if (l4.next != null)
                l2.next = l4.next;
            else {
                l2.next = null;
            }
            l4.next = l2;
            l1 = l2;
        }
        return head;
    }
}
