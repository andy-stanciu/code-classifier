class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        if (head.next.next == null) {
            ListNode newHead = head.next;
            head.next.next = head;
            head.next = null;
            return newHead;
        }
        ListNode node = method1(head.next.next);
        ListNode newHead = head.next;
        head.next.next = head;
        head.next = node;
        return newHead;
    }
}
