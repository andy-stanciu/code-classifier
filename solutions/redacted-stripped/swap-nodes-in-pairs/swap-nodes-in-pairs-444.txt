class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode curr = head;
        ListNode dummy = new ListNode();
        ListNode prev = dummy;
        while (curr != null && curr.next != null) {
            ListNode temp = curr.next;
            curr.next = curr.next.next;
            temp.next = curr;
            prev.next = temp;
            prev = curr;
            curr = curr.next;
        }
        return dummy.next;
    }
}
