class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp1 = head;
        ListNode temp2 = null;
        ListNode temp3 = null;
        if (temp1 == null)
            return head;
        if (temp1 != null && temp1.next != null) {
            temp2 = temp1.next;
            head = temp2;
        }
        while (temp1 != null && temp1.next != null) {
            temp2 = temp1.next;
            temp1.next = temp2.next;
            temp3 = temp2.next;
            temp2.next = temp1;
            if (temp1.next != null && temp1.next.next != null)
                temp1.next = temp1.next.next;
            temp1 = temp3;
        }
        return head;
    }
}
