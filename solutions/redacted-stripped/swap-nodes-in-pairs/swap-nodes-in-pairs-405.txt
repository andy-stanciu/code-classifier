class Solution {

    public ListNode method1(ListNode head) {
        int flag = 0;
        ListNode temp = head;
        ListNode prev = new ListNode(-1);
        int b = 0;
        while (temp != null) {
            prev.next = temp.next;
            if (temp.next == null) {
                prev.next = temp;
                break;
            }
            ListNode temp2 = temp.next.next;
            temp.next.next = temp;
            temp.next = temp2;
            if (flag == 0) {
                flag = 1;
                head = prev.next;
            }
            prev = temp;
            temp = temp2;
        }
        return head;
    }
}
