class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode curr = head;
        ListNode retnode = head.next;
        ListNode prev = null;
        while (curr != null && curr.next != null) {
            ListNode temp = curr.next;
            curr.next = temp.next;
            temp.next = curr;
            if (prev != null) {
                prev.next = temp;
            }
            prev = curr;
            curr = curr.next;
        }
        return retnode;
    }
}
