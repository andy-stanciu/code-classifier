class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head;
        int help;
        while (temp != null && temp.next != null) {
            help = temp.val;
            temp.val = temp.next.val;
            temp.next.val = help;
            temp = temp.next.next;
        }
        return head;
    }
}
