class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode prev = null;
        ListNode curr = head;
        ListNode future = head.next;
        while (curr != null && curr.next != null) {
            ListNode temp = future.next;
            if (curr == head) {
                future.next = curr;
                head = future;
            } else {
                prev.next = future;
                future.next = curr;
            }
            prev = curr;
            curr = temp;
            if (curr != null)
                future = curr.next;
        }
        prev.next = curr;
        return head;
    }
}
