class Solution {

    public ListNode method1(ListNode head) {
        ListNode trav = head, prev = null, res = null;
        while (trav != null && trav.next != null) {
            ListNode second = trav.next;
            if (prev == null) {
                trav.next = second.next;
                second.next = null;
            } else {
                prev.next = second;
                trav.next = second.next;
            }
            second.next = trav;
            prev = trav;
            trav = second;
            if (res == null) {
                res = trav;
            }
            trav = trav.next.next;
        }
        return (res == null) ? head : res;
    }
}
