class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        ListNode x = null;
        ListNode prev = null;
        ListNode nxt = null;
        if (temp != null) {
            nxt = temp.next;
        }
        while (temp != null && nxt != null) {
            if (temp == head) {
                head = temp.next;
                x = temp.next.next;
                head.next = temp;
                prev = temp;
                temp.next = x;
                temp = temp.next;
                if (temp != null) {
                    nxt = temp.next;
                }
            } else {
                prev.next = nxt;
                x = nxt.next;
                nxt.next = temp;
                prev = temp;
                temp.next = x;
                temp = temp.next;
                if (temp != null) {
                    nxt = temp.next;
                }
            }
        }
        return head;
    }
}
