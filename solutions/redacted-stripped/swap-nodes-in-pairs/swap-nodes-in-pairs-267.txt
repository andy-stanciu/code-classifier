class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        return help(head, head.next);
    }

    public static ListNode help(ListNode prev, ListNode curr) {
        if (curr == null)
            return prev;
        ListNode next = curr.next;
        curr.next = prev;
        if (next != null)
            prev.next = help(next, next.next);
        else {
            prev.next = null;
        }
        return curr;
    }
}
