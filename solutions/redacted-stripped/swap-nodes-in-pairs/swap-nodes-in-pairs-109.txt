class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode ptr = head;
        ListNode preptr = head.next;
        while (preptr != null) {
            int temp = ptr.val;
            ptr.val = preptr.val;
            preptr.val = temp;
            if (preptr.next == null)
                break;
            ptr = ptr.next.next;
            preptr = preptr.next.next;
        }
        return head;
    }
}
