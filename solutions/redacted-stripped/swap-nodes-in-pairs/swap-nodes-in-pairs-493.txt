class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode curr = head;
        ListNode nexti = curr.next;
        ListNode nextOfNext = curr.next.next;
        curr.next = method1(nextOfNext);
        nexti.next = curr;
        return nexti;
    }
}
