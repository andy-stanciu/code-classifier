class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode node = head.next;
        method2(head, null);
        return node;
    }

    void method2(ListNode node, ListNode prev) {
        if (node == null || node.next == null)
            return;
        ListNode temp = node.next.next;
        node.next.next = node;
        if (prev != null) {
            prev.next = node.next;
        }
        node.next = temp;
        prev = node;
        method2(node.next, prev);
    }
}
