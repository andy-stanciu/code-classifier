class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        else if (head.next == null)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode cur = head;
        ListNode prev = dummy;
        while (cur != null && cur.next != null) {
            ListNode first = cur;
            ListNode second = cur.next;
            prev.next = second;
            first.next = second.next;
            second.next = first;
            prev = first;
            cur = first.next;
        }
        return dummy.next;
    }
}
