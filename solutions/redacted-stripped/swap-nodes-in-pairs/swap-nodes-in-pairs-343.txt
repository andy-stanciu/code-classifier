class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode current = head;
        ListNode prev = null;
        while (current != null && current.next != null) {
            ListNode first = current;
            ListNode second = current.next;
            first.next = second.next;
            second.next = first;
            if (prev != null) {
                prev.next = second;
            } else {
                head = second;
            }
            prev = first;
            current = first.next;
        }
        return head;
    }
}
