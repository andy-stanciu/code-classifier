class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode pre = dummy;
        ListNode curr = pre.next;
        while (curr != null && curr.next != null) {
            pre.next = curr.next;
            curr.next = pre.next.next;
            pre.next.next = curr;
            pre = curr;
            curr = pre.next;
        }
        return dummy.next;
    }
}
