class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode first = head;
        ListNode second = first.next;
        ListNode prev;
        head = second;
        method2(first);
        while (first.next != null && first.next.next != null) {
            prev = first;
            first = first.next;
            second = first.next;
            prev.next = second;
            method2(first);
        }
        return head;
    }

    private void method2(ListNode first) {
        ListNode second = first.next;
        ListNode temp = second.next;
        second.next = first;
        first.next = temp;
    }
}
