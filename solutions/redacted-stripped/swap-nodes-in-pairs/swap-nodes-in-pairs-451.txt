class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        if (head == null || head.next == null)
            return head;
        while (temp != null) {
            if (temp.next == null)
                return head;
            int x = temp.val;
            temp.val = temp.next.val;
            temp.next.val = x;
            temp = temp.next.next;
        }
        return head;
    }
}
