class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        return method2(head);
    }

    public ListNode method2(ListNode hd) {
        if (hd == null || hd.next == null)
            return hd;
        ListNode temp = hd.next;
        ListNode l = temp.next;
        hd.next = l;
        temp.next = hd;
        hd = temp;
        temp = temp.next;
        temp.next = method2(l);
        return hd;
    }
}
