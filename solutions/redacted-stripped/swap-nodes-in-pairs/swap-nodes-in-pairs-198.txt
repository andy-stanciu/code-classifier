class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode prev = head;
        ListNode curr = head.next;
        ListNode ans = curr;
        while (true) {
            ListNode newNode = curr.next;
            if (newNode == null || newNode.next == null) {
                prev.next = newNode;
                curr.next = prev;
                break;
            }
            prev.next = newNode.next;
            curr.next = prev;
            prev = newNode;
            curr = newNode.next;
        }
        return ans;
    }
}
