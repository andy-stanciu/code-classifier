class Solution {

    public static void method1(ListNode prev, ListNode after) {
        int temp = prev.val;
        prev.val = after.val;
        after.val = temp;
    }

    public ListNode method2(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode prev = head;
        ListNode after = head.next;
        if (after.next == null) {
            method1(prev, after);
            return head;
        }
        while (after != null && after.next != null) {
            method1(prev, after);
            prev = after.next;
            after = prev.next;
            if (after != null && after.next == null) {
                method1(prev, after);
            }
        }
        return head;
    }
}
