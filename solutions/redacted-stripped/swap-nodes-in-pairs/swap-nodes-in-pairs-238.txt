class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummyNode = new ListNode(-1);
        dummyNode.next = head;
        ListNode curr = head;
        ListNode prev = dummyNode;
        while (curr != null && curr.next != null) {
            ListNode temp = curr.next.next;
            ListNode ele2 = curr.next;
            curr.next.next = curr;
            curr.next = temp;
            prev.next = ele2;
            prev = curr;
            if (curr != null)
                curr = curr.next;
        }
        return dummyNode.next;
    }
}
