class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head;
        while (temp != null) {
            int t = temp.val;
            if (temp.next != null) {
                temp.val = temp.next.val;
                temp.next.val = t;
                temp = temp.next;
            }
            temp = temp.next;
        }
        return head;
    }
}
