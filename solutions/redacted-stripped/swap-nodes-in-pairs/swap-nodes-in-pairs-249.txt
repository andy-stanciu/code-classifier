class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        head = helper(head);
        return head;
    }

    private ListNode helper(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode node = helper(head.next.next);
        ListNode temp = head.next;
        temp.next = head;
        head.next = node;
        return temp;
    }
}
