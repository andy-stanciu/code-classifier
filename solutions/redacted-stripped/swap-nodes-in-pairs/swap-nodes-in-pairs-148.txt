class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp;
        temp = head.next;
        head.next = head.next.next;
        temp.next = head;
        head = temp;
        ListNode prev = head.next, odd = prev.next;
        while (odd != null && odd.next != null) {
            temp = odd.next;
            odd.next = odd.next.next;
            temp.next = odd;
            prev.next = temp;
            prev = odd;
            odd = odd.next;
        }
        return head;
    }
}
