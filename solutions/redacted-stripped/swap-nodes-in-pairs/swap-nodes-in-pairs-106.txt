class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp, thead = new ListNode(0, head), p = thead, tem;
        while (p.next != null && p.next.next != null) {
            temp = p.next;
            p.next = (tem = p.next.next);
            temp.next = tem.next;
            tem.next = temp;
            p = p.next.next;
        }
        return thead.next;
    }
}
