class Solution {

    public ListNode method1(ListNode head) {
        ListNode it = head;
        ListNode temp = null;
        ListNode prev = null;
        if (head == null) {
            return head;
        }
        while (it != null) {
            if (it.next == null) {
                it = it.next;
                continue;
            }
            temp = it.next;
            it.next = it.next.next;
            temp.next = it;
            if (prev == null) {
                head = temp;
            } else {
                prev.next = temp;
            }
            prev = it;
            it = it.next;
        }
        return head;
    }
}
