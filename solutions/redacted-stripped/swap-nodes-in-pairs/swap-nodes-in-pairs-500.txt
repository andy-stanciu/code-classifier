class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode temp1 = head;
        ListNode temp2 = null;
        while (temp1 != null && temp1.next != null) {
            temp2 = temp1;
            temp1 = temp1.next;
            int num = temp1.val;
            temp1.val = temp2.val;
            temp2.val = num;
            temp2 = temp1;
            temp1 = temp1.next;
        }
        return head;
    }
}
