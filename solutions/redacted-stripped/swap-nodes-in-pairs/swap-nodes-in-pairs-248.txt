class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            int tempCurr = curr.val;
            int tempCurrNext = curr.next.val;
            curr.val = tempCurrNext;
            curr = curr.next;
            curr.val = tempCurr;
            curr = curr.next;
        }
        return head;
    }
}
