class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head;
        if (curr == null || curr.next == null) {
            return head;
        }
        ListNode next = head.next;
        while (next != null) {
            method2(curr, next);
            System.out.println("curr: " + curr.val + " next: " + next.val);
            curr = next.next;
            if (curr != null) {
                next = curr.next;
            } else {
                break;
            }
        }
        return head;
    }

    public void method2(ListNode a, ListNode b) {
        int temp = a.val;
        a.val = b.val;
        b.val = temp;
    }
}
