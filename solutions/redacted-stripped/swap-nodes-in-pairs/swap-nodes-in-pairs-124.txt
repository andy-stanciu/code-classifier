class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode p = head, q = head.next;
        while (q != null && q.next != null) {
            method2(p, q);
            p = p.next.next;
            q = q.next.next;
        }
        if (q == null)
            return head;
        method2(p, q);
        return head;
    }

    public static void method2(ListNode a, ListNode b) {
        if (a.val != b.val) {
            a.val = a.val + b.val;
            b.val = a.val - b.val;
            a.val = a.val - b.val;
        }
    }
}
