class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode dummy = new ListNode(0, head);
        ListNode prev = dummy;
        ListNode first = head;
        ListNode second = head.next;
        while (true) {
            first.next = second.next;
            second.next = first;
            prev.next = second;
            prev = first;
            if (first.next == null)
                break;
            first = first.next;
            if (first.next == null)
                break;
            second = first.next;
        }
        return dummy.next;
    }
}
