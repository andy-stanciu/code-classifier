class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode prev = null;
        ListNode first = head;
        ListNode second = null;
        ListNode next = null;
        head = first.next;
        while (first != null && first.next != null) {
            second = first.next;
            next = second.next;
            second.next = first;
            first.next = next;
            if (prev != null)
                prev.next = second;
            prev = second.next;
            first = next;
        }
        return head;
    }
}
