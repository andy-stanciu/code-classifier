class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            int value = curr.val;
            curr.val = curr.next.val;
            curr = curr.next;
            curr.val = value;
            curr = curr.next;
        }
        return head;
    }
}
