class Solution {

    public ListNode method1(ListNode head) {
        ListNode root = new ListNode();
        root.next = head;
        ListNode prev = root;
        while (head != null) {
            ListNode thz = head;
            ListNode next = head.next;
            if (next == null)
                break;
            ListNode nextNext = next.next;
            prev.next = next;
            next.next = thz;
            thz.next = nextNext;
            head = nextNext;
            prev = thz;
        }
        return root.next;
    }
}
