class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode root = head.next;
        ListNode curr = head.next;
        ListNode prev = head;
        ListNode prevTail = null;
        while (curr != null) {
            prev.next = curr.next;
            curr.next = prev;
            if (prevTail != null) {
                prevTail.next = curr;
            }
            prevTail = prev;
            if (prev.next == null) {
                break;
            }
            prev = prev.next;
            curr = prev.next;
        }
        return root;
    }
}
