class Solution {

    public ListNode method1(ListNode head) {
        ListNode res = new ListNode(0, head);
        ListNode curr = res;
        while (curr != null && curr.next != null && curr.next.next != null) {
            ListNode first = curr.next;
            ListNode second = curr.next.next;
            curr.next = second;
            first.next = second.next;
            second.next = first;
            curr = curr.next.next;
        }
        return res.next;
    }
}
