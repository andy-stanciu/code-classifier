class Solution {

    public ListNode method1(ListNode head) {
        ListNode newNode = head;
        while (newNode != null && newNode.next != null) {
            ListNode curr = newNode;
            newNode = newNode.next;
            int temp = curr.val;
            curr.val = newNode.val;
            newNode.val = temp;
            newNode = newNode.next;
        }
        return head;
    }
}
