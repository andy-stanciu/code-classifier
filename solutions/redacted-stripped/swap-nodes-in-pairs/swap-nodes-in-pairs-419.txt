class Solution {

    public ListNode method1(ListNode head) {
        ListNode first = head;
        ListNode ret = first;
        ListNode prev = null;
        while (first != null) {
            ListNode second = first.next;
            ListNode temp = null;
            if (second != null) {
                temp = second.next;
                second.next = first;
                first.next = temp;
                if (ret == first)
                    ret = second;
                if (prev == null) {
                    prev = first;
                } else {
                    prev.next = second;
                    prev = first;
                }
            }
            first = temp;
            second = null;
        }
        return ret;
    }
}
