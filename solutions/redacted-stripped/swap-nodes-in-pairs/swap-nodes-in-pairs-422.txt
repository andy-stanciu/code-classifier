class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        ListNode temp = head;
        while (temp != null && temp.next != null) {
            int x = temp.val;
            temp.val = temp.next.val;
            temp.next.val = x;
            temp = temp.next.next;
        }
        return head;
    }
}
