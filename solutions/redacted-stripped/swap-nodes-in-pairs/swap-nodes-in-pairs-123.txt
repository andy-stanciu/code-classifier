class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        return method2(head, null);
    }

    public ListNode method2(ListNode head, ListNode prev) {
        ListNode node = null;
        if (head.next != null) {
            node = head.next;
            head.next = node.next;
            node.next = head;
            if (prev != null) {
                prev.next = node;
            }
            if (head.next != null) {
                method2(head.next, head);
            }
        }
        return node;
    }
}
