class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode p = head;
        ListNode q = head.next;
        while (q != null) {
            int key = q.val;
            q.val = p.val;
            p.val = key;
            if (q.next == null)
                break;
            p = q.next;
            q = p.next;
        }
        return head;
    }
}
