class Solution {

    ListNode method1(ListNode head, int k) {
        if (head == null) {
            return null;
        }
        ListNode prev = null;
        ListNode curr = head;
        ListNode next = null;
        int count = 0;
        while (curr != null && count < k) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        if (next != null) {
            head.next = method1(next, k);
        }
        return prev;
    }

    public ListNode method2(ListNode head) {
        ListNode ans = method1(head, 2);
        return ans;
    }
}
