class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode first = head;
        ListNode second = head.next;
        ListNode head1 = new ListNode(-1);
        head1.next = head;
        ListNode shubh = head1;
        while (second != null) {
            shubh.next = second;
            first.next = second.next;
            second.next = first;
            shubh = first;
            first = first.next;
            if (first == null)
                break;
            second = first.next;
        }
        return head1.next;
    }
}
