class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode temp1 = head;
        ListNode temp2;
        if (temp1.next == null) {
            return head;
        } else {
            temp2 = head.next;
        }
        while (temp1.next != null) {
            int val = temp2.val;
            temp2.val = temp1.val;
            temp1.val = val;
            if (temp1.next.next != null) {
                temp1 = temp1.next.next;
                temp2 = temp1.next;
            } else {
                temp1 = temp1.next;
            }
        }
        return head;
    }
}
