class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode var1 = dummy;
        while (var1.next != null && var1.next.next != null) {
            ListNode one = var1.next;
            ListNode two = var1.next.next;
            one.next = two.next;
            two.next = one;
            var1.next = two;
            var1 = one;
        }
        return dummy.next;
    }
}
