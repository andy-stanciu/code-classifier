class Solution {

    public void method1(ListNode a, ListNode b) {
        if (a == null || b == null)
            return;
        int temp = a.val;
        a.val = b.val;
        b.val = temp;
    }

    public ListNode method2(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode odd = head;
        ListNode even = head.next;
        while (even != null && (even.next != null && odd.next != null)) {
            method1(even, odd);
            even = even.next.next;
            odd = odd.next.next;
        }
        method1(even, odd);
        return head;
    }
}
