class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode slow = dummy;
        ListNode fast = head;
        if (head == null || head.next == null)
            return head;
        while (fast != null && fast.next != null) {
            ListNode temp = fast.next;
            fast.next = temp.next;
            temp.next = fast;
            slow.next = temp;
            slow = fast;
            fast = fast.next;
        }
        return dummy.next;
    }
}
