class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode ptr = head;
        ListNode ptr2 = head.next;
        int temp = 0;
        while (ptr != null || ptr2 != null) {
            temp = ptr.val;
            ptr.val = ptr2.val;
            ptr2.val = temp;
            if (ptr2.next == null || ptr2.next.next == null) {
                return head;
            }
            ptr = ptr2.next;
            ptr2 = ptr.next;
        }
        return head;
    }
}
