class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        int var1 = 0;
        while (temp != null) {
            if (var1 % 2 == 0 && temp.next != null) {
                int tempValue = temp.next.val;
                temp.next.val = temp.val;
                temp.val = tempValue;
            }
            temp = temp.next;
            var1++;
        }
        return head;
    }
}
