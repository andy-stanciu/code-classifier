class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode temp = head;
        while (temp.next.next != null) {
            int t = temp.next.val;
            temp.next.val = temp.val;
            temp.val = t;
            temp = temp.next.next;
            if (temp == null || temp.next == null)
                break;
        }
        if (temp.next != null) {
            int t = temp.next.val;
            temp.next.val = temp.val;
            temp.val = t;
        }
        return head;
    }
}
