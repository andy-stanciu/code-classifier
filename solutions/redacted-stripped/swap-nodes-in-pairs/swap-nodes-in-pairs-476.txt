class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return null;
        }
        if (head.next == null) {
            return head;
        }
        ListNode previous = null;
        ListNode current = head;
        ListNode nextR = current.next;
        ListNode first = nextR;
        while (nextR != null) {
            current.next = nextR.next;
            nextR.next = current;
            previous = current;
            current = current.next;
            try {
                if (current.next == null) {
                    break;
                }
                previous.next = current.next;
                nextR = current.next;
            } catch (Exception param1) {
                break;
            }
        }
        return first;
    }
}
