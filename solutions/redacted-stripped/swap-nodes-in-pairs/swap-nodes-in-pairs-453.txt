class Solution {

    public ListNode method1(ListNode head) {
        ListNode pre = null;
        ListNode second = head;
        ListNode first = null;
        while (second != null && second.next != null) {
            first = second.next;
            second.next = first.next;
            first.next = second;
            if (pre == null) {
                head = first;
            } else {
                pre.next = first;
            }
            pre = second;
            second = second.next;
        }
        return head;
    }
}
