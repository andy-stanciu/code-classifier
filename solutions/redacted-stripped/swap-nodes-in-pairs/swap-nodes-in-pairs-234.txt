class Solution {

    public ListNode method1(ListNode current) {
        if (current == null)
            return null;
        ListNode result = current.next;
        if (result == null)
            return current;
        ListNode prevPrev = null;
        ListNode prev = current;
        current = current.next;
        ListNode next = null;
        while (current != null) {
            next = current.next;
            if (prevPrev != null)
                prevPrev.next = current;
            current.next = prev;
            prev.next = next;
            prevPrev = prev;
            current = next;
            prev = current;
            if (current != null)
                current = current.next;
        }
        return result;
    }
}
