class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy;
        ListNode current = head;
        while (current != null && current.next != null) {
            ListNode first = current;
            ListNode second = current.next;
            first.next = second.next;
            second.next = first;
            prev.next = second;
            prev = first;
            current = first.next;
        }
        return dummy.next;
    }
}
