class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode p1 = head, p2 = head.next, p3 = null;
        p3 = method1(p2.next);
        p1.next = p3;
        p2.next = p1;
        return p2;
    }
}
