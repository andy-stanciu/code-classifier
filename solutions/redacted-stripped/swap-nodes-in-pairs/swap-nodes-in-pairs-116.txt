class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode previous = dummy;
        ListNode first = head;
        ListNode second = head.next;
        while (first != null && second != null) {
            ListNode temp = null;
            temp = second.next;
            first.next = temp;
            second.next = first;
            previous.next = second;
            temp = first;
            first = second;
            second = temp;
            previous = second;
            first = first.next;
            if (first != null)
                first = first.next;
            else
                continue;
            second = second.next;
            if (second != null)
                second = second.next;
            else
                continue;
        }
        return dummy.next;
    }
}
