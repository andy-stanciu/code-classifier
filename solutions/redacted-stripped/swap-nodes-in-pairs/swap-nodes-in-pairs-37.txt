class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode dummy = new ListNode(0);
        dummy.next = head.next;
        ListNode node = head;
        ListNode pre = null;
        while (node != null && node.next != null) {
            ListNode nx = node.next;
            node.next = nx.next;
            nx.next = node;
            if (pre != null) {
                pre.next = nx;
            }
            pre = node;
            node = node.next;
        }
        return dummy.next;
    }
}
