class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode root = head.next;
        ListNode temp = head.next;
        ListNode prev = head;
        while (temp != null) {
            ListNode nextNode = temp.next;
            temp.next = prev;
            if (nextNode == null || nextNode.next == null) {
                prev.next = nextNode;
                break;
            } else {
                prev.next = nextNode.next;
            }
            prev = nextNode;
            temp = prev.next;
        }
        return root;
    }
}
