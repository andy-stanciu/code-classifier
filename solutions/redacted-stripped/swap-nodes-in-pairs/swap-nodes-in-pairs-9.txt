class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode prev = head;
        ListNode curr = head.next;
        ListNode ans = curr;
        while (true) {
            ListNode save = curr.next;
            if (save == null || save.next == null) {
                prev.next = save;
                curr.next = prev;
                break;
            }
            prev.next = save.next;
            curr.next = prev;
            prev = save;
            curr = save.next;
        }
        return ans;
    }
}
