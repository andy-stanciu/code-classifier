class Solution {

    public ListNode method1(ListNode head) {
        List<ListNode> lst = new ArrayList<ListNode>();
        ListNode res = new ListNode();
        ListNode var1 = new ListNode();
        ListNode temp = new ListNode();
        int i;
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        while (head != null) {
            if (head.next != null) {
                lst.add(head.next);
                lst.add(head);
                head = head.next.next;
            } else {
                lst.add(head);
                head = head.next;
            }
        }
        var1 = res;
        for (i = 0; i < lst.size() - 1; i++) {
            res.next = lst.get(i);
            res = res.next;
        }
        temp = lst.get(i);
        temp.next = null;
        res.next = temp;
        return var1.next;
    }
}
