class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode dummy = new ListNode(0, head);
        ListNode previous = dummy;
        while (head != null) {
            ListNode first = head;
            ListNode second = head.next;
            if (second == null) {
                break;
            }
            previous.next = second;
            first.next = second.next;
            second.next = first;
            previous = head;
            head = head.next;
        }
        return dummy.next;
    }
}
