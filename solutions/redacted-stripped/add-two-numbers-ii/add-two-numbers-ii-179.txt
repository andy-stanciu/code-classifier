class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        StringBuilder sb1 = new StringBuilder();
        while (l1 != null) {
            sb1.append(l1.val);
            l1 = l1.next;
        }
        StringBuilder sb2 = new StringBuilder();
        while (l2 != null) {
            sb2.append(l2.val);
            l2 = l2.next;
        }
        BigInteger num1 = BigInteger.ZERO;
        if (!sb1.toString().isEmpty()) {
            num1 = new BigInteger(sb1.toString());
        }
        BigInteger num2 = BigInteger.ZERO;
        if (!sb2.toString().isEmpty()) {
            num2 = new BigInteger(sb2.toString());
        }
        String val = String.valueOf(num1.add(num2));
        ListNode res = new ListNode(0);
        ListNode temp = res;
        for (int i = 0; i < val.length(); i++) {
            temp.next = new ListNode(Character.getNumericValue(val.charAt(i)));
            temp = temp.next;
        }
        return res.next;
    }
}
