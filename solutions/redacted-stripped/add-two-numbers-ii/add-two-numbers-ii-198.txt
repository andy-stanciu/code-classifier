class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        StringBuilder sb = new StringBuilder();
        while (l1 != null) {
            sb.append(l1.val);
            l1 = l1.next;
        }
        StringBuilder sb1 = new StringBuilder();
        while (l2 != null) {
            sb1.append(l2.val);
            l2 = l2.next;
        }
        String firstnum = sb.toString();
        String secondnum = sb1.toString();
        BigInteger i = new BigInteger(firstnum);
        BigInteger j = new BigInteger(secondnum);
        BigInteger result = i.add(j);
        String re = String.valueOf(result);
        int p = re.length();
        ListNode dummyHead = new ListNode(-1);
        ListNode current = dummyHead;
        for (int z = 0; z < p; z++) {
            char c = re.charAt(z);
            int b = Character.getNumericValue(c);
            current.next = new ListNode(b);
            current = current.next;
        }
        return dummyHead.next;
    }
}
