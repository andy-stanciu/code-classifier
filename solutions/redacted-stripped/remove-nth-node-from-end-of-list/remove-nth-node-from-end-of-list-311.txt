class Solution {

    private int length(ListNode head) {
        int cnt = 0;
        while (head != null) {
            cnt++;
            head = head.next;
        }
        return cnt;
    }

    public ListNode method1(ListNode head, int n) {
        int len = length(head);
        int res = len - n;
        ListNode curr = head;
        int i = 1;
        if (head.next == null) {
            return null;
        }
        if (res == 0) {
            head = head.next;
            return head;
        }
        while (curr != null && i < len && curr.next != null) {
            if (i == res) {
                curr.next = curr.next.next;
            }
            i++;
            curr = curr.next;
        }
        return head;
    }
}
