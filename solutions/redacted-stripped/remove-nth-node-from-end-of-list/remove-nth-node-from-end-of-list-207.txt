class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fastPointer = head;
        ListNode basePointer = head;
        if (n == 0)
            return head;
        while (n > 0) {
            n--;
            fastPointer = fastPointer.next;
        }
        if (fastPointer == null) {
            head = head.next;
            return head;
        } else {
            while (fastPointer.next != null) {
                fastPointer = fastPointer.next;
                basePointer = basePointer.next;
            }
            basePointer.next = basePointer.next.next;
        }
        return head;
    }
}
