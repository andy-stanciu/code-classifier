class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        int size = 0;
        ListNode temp = head;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        if (n == size) {
            head = head.next;
            return head;
        }
        ListNode var1 = head;
        for (int i = 1; i < size - n; i++) {
            var1 = var1.next;
        }
        ListNode var2 = var1.next;
        if (var2.next == null)
            var1.next = null;
        else
            var1.next = var2.next;
        return head;
    }
}
