class Solution {

    public ListNode method1(ListNode head, int n) {
        int size = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            size++;
        }
        ListNode var1 = head;
        int var2 = (size - n) - 1;
        if (var2 < 0) {
            return head.next;
        }
        for (int i = 0; i < var2; i++) {
            var1 = var1.next;
        }
        var1.next = var1.next.next;
        return head;
    }
}
