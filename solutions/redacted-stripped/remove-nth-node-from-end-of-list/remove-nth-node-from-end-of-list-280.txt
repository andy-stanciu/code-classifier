class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fast = head;
        for (int i = 1; i <= n; i++) {
            fast = fast.next;
        }
        ListNode slow = head;
        if (fast == null) {
            return head.next;
        }
        while (fast.next != null) {
            fast = fast.next;
            slow = slow.next;
        }
        ListNode del = slow.next;
        slow.next = slow.next.next;
        return head;
    }
}
