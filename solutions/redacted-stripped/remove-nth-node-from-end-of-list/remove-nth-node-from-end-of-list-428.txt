class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode left = dummy;
        while (n > 0 && head != null) {
            head = head.next;
            n--;
        }
        if (n > 0)
            return null;
        while (head != null) {
            head = head.next;
            left = left.next;
        }
        left.next = left.next.next;
        return dummy.next;
    }
}
