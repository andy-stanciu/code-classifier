class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode tail = new ListNode(0, head);
        ListNode temp = tail;
        for (int i = 0; i < n; i++) {
            head = head.next;
        }
        while (head != null) {
            head = head.next;
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return tail.next;
    }
}
