class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        ListNode ptr = head;
        ListNode ptr2 = head;
        int len = 0;
        while (ptr != null) {
            len++;
            ptr = ptr.next;
        }
        int index = len - n;
        ptr = head;
        boolean start = false;
        while (true) {
            if (index == 0) {
                if (ptr == head) {
                    ptr2 = ptr2.next;
                    ptr.next = null;
                    head = ptr2;
                    break;
                } else {
                    ptr2.next = ptr.next;
                    ptr.next = null;
                    break;
                }
            }
            index--;
            if (start)
                ptr2 = ptr2.next;
            start = true;
            ptr = ptr.next;
        }
        return head;
    }
}
