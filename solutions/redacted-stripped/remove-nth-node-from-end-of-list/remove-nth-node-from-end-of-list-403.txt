class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        if (count == n) {
            return head.next;
        }
        temp = head;
        int length = count - n;
        int count1 = 0;
        ListNode prev = null;
        while (temp != null) {
            if (count1 == length) {
                if (prev == null) {
                    head = temp.next;
                } else {
                    prev.next = temp.next;
                }
                break;
            }
            prev = temp;
            temp = temp.next;
            count1++;
        }
        return head;
    }
}
