class Solution {

    static int method1(ListNode head) {
        int count = 0;
        while (head != null) {
            count++;
            head = head.next;
        }
        return count;
    }

    public ListNode method2(ListNode head, int x) {
        int n = method1(head);
        n = n - x;
        ListNode current, prev = null;
        current = head;
        if (n < 0) {
            return prev;
        } else if (n == 0) {
            head = head.next;
            return head;
        }
        while (n > 0) {
            prev = current;
            current = current.next;
            n--;
        }
        prev.next = current.next;
        current.next = null;
        return head;
    }
}
