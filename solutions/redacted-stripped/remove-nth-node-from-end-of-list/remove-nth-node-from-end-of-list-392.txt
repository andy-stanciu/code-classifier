class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode head1 = method2(head);
        ListNode curr = head1;
        ListNode prev = null;
        int i = 1;
        while (curr != null) {
            if (i == n) {
                if (prev == null) {
                    head1 = curr.next;
                } else {
                    prev.next = curr.next;
                }
                break;
            }
            prev = curr;
            curr = curr.next;
            i++;
        }
        return method2(head1);
    }

    private ListNode method2(ListNode head) {
        ListNode prev = null;
        ListNode curr = head;
        while (curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
}
