class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null && n == 1)
            return null;
        ListNode temp = head;
        while (n >= 0 && temp != null) {
            temp = temp.next;
            n--;
        }
        if (n > 0) {
            return head;
        } else if (n == 0 && temp == null) {
            return head.next;
        }
        ListNode prev = head;
        while (temp != null) {
            temp = temp.next;
            prev = prev.next;
        }
        prev.next = prev.next.next;
        return head;
    }
}
