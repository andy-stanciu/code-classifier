class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode left = head, right = head;
        while (n-- > 0) right = right.next;
        if (right == null)
            return head.next;
        while (right.next != null) {
            left = left.next;
            right = right.next;
        }
        left.next = left.next.next;
        return head;
    }
}
