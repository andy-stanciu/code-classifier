class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode Fast = head;
        ListNode slow = head;
        for (int i = 0; i < n; i++) {
            Fast = Fast.next;
        }
        if (Fast == null) {
            return head.next;
        }
        while (Fast.next != null) {
            Fast = Fast.next;
            slow = slow.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
