class Solution {

    public ListNode method1(ListNode head, int n) {
        int count = 0;
        ListNode var1 = head;
        while (count < n) {
            var1 = var1.next;
            count++;
        }
        if (var1 == null)
            return head.next;
        ListNode prev = head;
        while (var1.next != null) {
            prev = prev.next;
            var1 = var1.next;
        }
        if (prev.next != null) {
            prev.next = prev.next.next;
        }
        return head;
    }
}
