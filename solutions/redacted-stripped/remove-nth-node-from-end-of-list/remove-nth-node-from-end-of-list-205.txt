class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        ListNode prev = null;
        int c = 0;
        while (temp != null) {
            c++;
            temp = temp.next;
        }
        if (c == 1) {
            return null;
        }
        temp = head;
        int k = c - n;
        while (k > 0) {
            prev = temp;
            temp = temp.next;
            k--;
        }
        if (n == 1) {
            prev.next = null;
            return head;
        }
        if (n == c) {
            head = head.next;
            return head;
        }
        prev.next = temp.next;
        return head;
    }
}
