class Solution {

    public int size(ListNode head) {
        int count = 0;
        ListNode ptr = head;
        while (ptr != null) {
            count++;
            ptr = ptr.next;
        }
        return count;
    }

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        int totalCount = size(head) - n - 1;
        ListNode ptr = head;
        while (totalCount > 0) {
            ptr = ptr.next;
            totalCount--;
        }
        if (totalCount < 0) {
            return head.next;
        }
        ptr.next = ptr.next.next;
        return head;
    }
}
