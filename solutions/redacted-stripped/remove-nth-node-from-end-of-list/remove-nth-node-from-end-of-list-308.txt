class Solution {

    int n;

    public ListNode method1(ListNode head, int n) {
        this.n = n;
        return helper(head);
    }

    ListNode helper(ListNode param1) {
        if (param1 == null) {
            return null;
        }
        param1.next = helper(param1.next);
        if (--n == 0) {
            return param1.next;
        }
        return param1;
    }
}
