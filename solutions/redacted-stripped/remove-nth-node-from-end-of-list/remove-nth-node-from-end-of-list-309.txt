class Class1 {

    int val;

    Class1 next;

    Class1() {
    }

    Class1(int val) {
        this.val = val;
    }

    Class1(int val, Class1 next) {
        this.val = val;
        this.next = next;
    }
}

public class Solution {

    public Class1 method1(Class1 head, int n) {
        Class1 res = new Class1(0, head);
        Class1 dummy = res;
        for (int i = 0; i < n; i++) {
            head = head.next;
        }
        while (head != null) {
            head = head.next;
            dummy = dummy.next;
        }
        dummy.next = dummy.next.next;
        return res.next;
    }
}
