class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return null;
        }
        int size = 0;
        ListNode ptr = head;
        while (ptr != null) {
            size++;
            ptr = ptr.next;
        }
        if (n == size) {
            return head.next;
        }
        ptr = head;
        for (int i = 0; i < size - n - 1; i++) {
            ptr = ptr.next;
        }
        ptr.next = ptr.next.next;
        return head;
    }
}
