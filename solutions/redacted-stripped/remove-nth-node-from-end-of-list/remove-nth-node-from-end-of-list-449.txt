class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null) {
            ListNode var1 = null;
            return var1;
        }
        ListNode temp = head;
        int size = 0;
        while (temp != null) {
            temp = temp.next;
            size++;
        }
        if (n == size) {
            head = head.next;
            return head;
        }
        ListNode prev = head;
        int idx = size - n + 1;
        int i = 1;
        while (i < idx - 1) {
            prev = prev.next;
            i++;
        }
        prev.next = prev.next.next;
        return head;
    }
}
