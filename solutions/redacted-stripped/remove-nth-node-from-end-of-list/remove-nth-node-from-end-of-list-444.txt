class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode current = head;
        ListNode prev = head;
        for (int i = 0; i < n; i++) {
            if (current == null) {
                return head;
            }
            current = current.next;
        }
        if (current == null) {
            return head.next;
        }
        while (current.next != null) {
            current = current.next;
            prev = prev.next;
        }
        prev.next = prev.next.next;
        return head;
    }
}
