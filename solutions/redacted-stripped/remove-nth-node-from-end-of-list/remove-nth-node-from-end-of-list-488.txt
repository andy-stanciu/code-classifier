class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return null;
        int size = 0;
        ListNode current = head;
        while (current != null) {
            size++;
            current = current.next;
        }
        if (n == size) {
            return head.next;
        }
        int k = size - n;
        current = head;
        for (int i = 0; i < k - 1; i++) {
            current = current.next;
        }
        current.next = current.next.next;
        return head;
    }
}
