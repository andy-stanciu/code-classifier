class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int k = 0;
        while (k < n && curr != null) {
            curr = curr.next;
            k++;
        }
        if (curr == null) {
            return head.next;
        }
        ListNode temp = head;
        while (curr != null && curr.next != null) {
            temp = temp.next;
            curr = curr.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
