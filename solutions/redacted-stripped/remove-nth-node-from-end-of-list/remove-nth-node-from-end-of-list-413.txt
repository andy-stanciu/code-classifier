class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        int len = 1;
        ListNode curr = head;
        while (curr.next != null) {
            curr = curr.next;
            len++;
        }
        System.out.println("Length is " + len);
        len = len - n;
        System.out.println("Remove ngth " + len);
        if (len == 0) {
            curr.next = null;
            return head.next;
        }
        curr = head;
        while (len > 1) {
            curr = curr.next;
            len--;
        }
        curr.next = curr.next.next;
        return head;
    }
}
