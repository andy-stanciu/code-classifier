class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return head;
        }
        ListNode dummy = new ListNode();
        dummy.next = head;
        method2(head, dummy, n);
        return dummy.next;
    }

    public int method2(ListNode param1, ListNode prev, int n) {
        if (param1 == null) {
            return 0;
        }
        int after = method2(param1.next, param1, n);
        if (after == n - 1) {
            prev.next = param1.next;
            param1.next = null;
        }
        return 1 + after;
    }
}
