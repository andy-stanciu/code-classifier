class Solution {

    public ListNode method1(ListNode head, int n) {
        int len = 0;
        ListNode temp = head;
        while (temp != null) {
            len++;
            temp = temp.next;
        }
        System.out.println(len);
        if (len == 1)
            return null;
        if (len - n == 0)
            return head.next;
        int count = 1;
        temp = head;
        while (count != len - n) {
            temp = temp.next;
            count++;
        }
        System.out.println(temp.val);
        temp.next = temp.next.next;
        return head;
    }
}
