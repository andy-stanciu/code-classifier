class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode tmp = head;
        int count = 0;
        while (tmp != null) {
            count++;
            tmp = tmp.next;
        }
        if (count == n) {
            head = head.next;
            return head;
        }
        tmp = head;
        for (int i = 0; i < count - n - 1; i++) {
            tmp = tmp.next;
        }
        if (tmp.next != null && tmp.next.next != null)
            tmp.next = tmp.next.next;
        else
            tmp.next = null;
        return head;
    }
}
