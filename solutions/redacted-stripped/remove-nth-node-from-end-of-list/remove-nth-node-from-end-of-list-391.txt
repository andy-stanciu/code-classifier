class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return null;
        ListNode var1 = head;
        int length = 0;
        while (var1 != null) {
            var1 = var1.next;
            length++;
        }
        if (length == 1)
            return null;
        if (length == n)
            return head.next;
        int counter = 0;
        var1 = head;
        while (counter < length - n - 1) {
            var1 = var1.next;
            counter++;
        }
        var1.next = var1.next.next;
        return head;
    }
}
