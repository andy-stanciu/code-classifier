class Solution {

    private int method1(ListNode head) {
        int size = 0;
        ListNode curr = head;
        while (curr != null) {
            size++;
            curr = curr.next;
        }
        return size;
    }

    public ListNode method2(ListNode head, int n) {
        if (head == null)
            return head;
        int k = method1(head) - n;
        if (k == 0)
            return head.next;
        ListNode var1 = null;
        ListNode var2 = head;
        for (int i = 0; i < k; i++) {
            var1 = var2;
            var2 = var2.next;
        }
        var1.next = var2.next;
        var2.next = null;
        return head;
    }
}
