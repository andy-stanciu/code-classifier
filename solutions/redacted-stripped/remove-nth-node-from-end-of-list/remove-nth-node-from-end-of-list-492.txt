class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode tmp = head;
        int length;
        for (length = 0; tmp != null; length++) {
            tmp = tmp.next;
        }
        int var1 = length - n;
        if (var1 == 0) {
            return head.next;
        }
        ListNode curr = head;
        ListNode prev = null;
        for (int i = 0; i < var1; i++) {
            prev = curr;
            curr = curr.next;
        }
        if (var1 == length - 1) {
            prev.next = null;
        } else {
            prev.next = curr.next;
            curr.next = null;
        }
        return head;
    }
}
