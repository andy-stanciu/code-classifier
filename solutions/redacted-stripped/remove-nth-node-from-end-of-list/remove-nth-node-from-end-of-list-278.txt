class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        n = size(head) - n;
        while (temp != null && n >= 0) {
            if (n == 0)
                return head.next;
            if (n - 1 == 0) {
                System.out.println(temp.val);
                temp.next = temp.next.next;
                break;
            }
            temp = temp.next;
            n--;
        }
        return head;
    }

    public int size(ListNode nd) {
        int size = 0;
        while (nd != null) {
            size++;
            nd = nd.next;
        }
        return size;
    }
}
