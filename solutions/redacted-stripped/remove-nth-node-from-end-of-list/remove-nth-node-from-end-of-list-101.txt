class Solution {

    public int count(ListNode head) {
        int c = 0;
        while (head != null) {
            c++;
            head = head.next;
        }
        return c;
    }

    public ListNode method1(ListNode head, int n) {
        if (n == 0) {
            return head;
        }
        int a = count(head);
        int ans = a - n;
        if (ans == 0) {
            head = head.next;
            return head;
        }
        ListNode temp = head;
        int i = 0;
        while (i < ans - 1) {
            i++;
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
