class Solution {

    public ListNode method1(ListNode head, int n) {
        int size = 0;
        ListNode var1 = head;
        ListNode var2 = head;
        while (var1 != null) {
            size++;
            var1 = var1.next;
        }
        size = size - n;
        if (size == 1) {
            head.next = head.next.next;
            return head;
        } else if (size == 0) {
            return head.next;
        }
        for (int i = 1; i < size; i++) {
            var2 = var2.next;
        }
        var2.next = var2.next.next;
        return head;
    }
}
