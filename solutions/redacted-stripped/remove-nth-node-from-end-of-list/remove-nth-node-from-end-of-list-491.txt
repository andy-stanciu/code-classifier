class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int idx = n;
        int size = 0;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        temp = head;
        if (head.next == null && idx == 1) {
            head = null;
            return head;
        }
        if (idx == size) {
            head = head.next;
            return head;
        }
        for (int i = 1; i < size - idx; i++) {
            temp = temp.next;
        }
        ListNode next = temp.next;
        temp.next = temp.next.next;
        return head;
    }
}
