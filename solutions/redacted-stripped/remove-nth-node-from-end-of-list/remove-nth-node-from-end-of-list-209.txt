class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return head;
        ListNode slow = head, fast = head;
        while (n-- != 0) {
            if (fast.next == null)
                return head.next;
            fast = fast.next;
        }
        while (fast.next != null) {
            slow = slow.next;
            fast = fast.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
