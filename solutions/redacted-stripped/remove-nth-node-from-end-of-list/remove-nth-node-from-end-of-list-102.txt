class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode start = new ListNode(0);
        start.next = head;
        ListNode temp = start;
        int var1 = method2(head);
        int k = var1 - n;
        int count = 1;
        while (count < k && temp != null) {
            temp = temp.next;
            count++;
        }
        if (temp != null && temp.next != null) {
            temp.next = temp.next.next;
        }
        return start.next;
    }

    public int method2(ListNode head) {
        ListNode temp = head;
        int count = 1;
        while (temp != null) {
            temp = temp.next;
            count++;
        }
        return count;
    }
}
