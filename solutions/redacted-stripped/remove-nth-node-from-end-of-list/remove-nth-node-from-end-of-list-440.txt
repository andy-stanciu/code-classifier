class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int count = 0;
        if (head == null || head.next == null) {
            return null;
        }
        while (curr != null) {
            curr = curr.next;
            count++;
        }
        count = count - n;
        curr = head;
        if (count == 0) {
            head = curr.next;
        } else {
            ListNode prev = head;
            while (count > 0 && curr != null) {
                prev = curr;
                curr = curr.next;
                count--;
            }
            prev.next = prev.next.next;
        }
        return head;
    }
}
