class Solution {

    public int getLength(ListNode head) {
        int length = 0;
        while (head != null) {
            length++;
            head = head.next;
        }
        return length;
    }

    public ListNode method1(ListNode head, int n) {
        int var1 = this.getLength(head);
        int element = var1 - n - 1;
        if (element < 0) {
            return head.next;
        }
        int var2 = 0;
        ListNode curr = head;
        while (curr != null) {
            if (var2 == element) {
                curr.next = curr.next.next;
            }
            var2++;
            curr = curr.next;
        }
        return head;
    }
}
