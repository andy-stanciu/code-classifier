class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode var1 = new ListNode(0, head);
        ListNode slow = var1;
        ListNode fast = var1;
        for (int i = 0; i <= n; i++) {
            if (fast != null) {
                fast = fast.next;
            }
        }
        while (fast != null) {
            fast = fast.next;
            slow = slow.next;
        }
        slow.next = slow.next.next;
        return var1.next;
    }
}
