class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        int length = getLen(head);
        int h = length - n;
        ListNode temp = head;
        if (length == n) {
            head = head.next;
            return head;
        }
        for (int i = 0; i < h - 1 && temp.next != null; i++) {
            temp = temp.next;
        }
        if (temp.next != null) {
            temp.next = temp.next.next;
        }
        return head;
    }

    public int getLen(ListNode head) {
        ListNode temp = head;
        int x = 0;
        while (temp != null) {
            temp = temp.next;
            x++;
        }
        return x;
    }
}
