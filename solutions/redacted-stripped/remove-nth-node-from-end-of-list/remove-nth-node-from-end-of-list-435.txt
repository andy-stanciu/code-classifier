class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode var1 = new ListNode();
        ListNode prev = var1;
        var1.next = head;
        ListNode fast = head;
        while (n > 0) {
            fast = fast.next;
            n--;
        }
        while (fast != null) {
            head = head.next;
            prev = prev.next;
            fast = fast.next;
        }
        prev.next = head.next;
        head.next = null;
        return var1.next;
    }
}
