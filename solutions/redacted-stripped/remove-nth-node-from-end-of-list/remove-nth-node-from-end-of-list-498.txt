class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode t = new ListNode(0);
        ListNode d = t;
        int a = 0;
        d.next = head;
        while (n != 0) {
            head = head.next;
            n--;
        }
        while (head != null) {
            head = head.next;
            d = d.next;
        }
        d.next = d.next.next;
        return t.next;
    }
}
