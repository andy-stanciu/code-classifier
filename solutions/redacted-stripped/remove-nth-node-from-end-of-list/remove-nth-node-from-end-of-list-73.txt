class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode j = new ListNode();
        j.next = head;
        ListNode l = j;
        ListNode r = j;
        for (int i = 0; i < n; i++) {
            r = r.next;
        }
        while (r.next != null) {
            l = l.next;
            r = r.next;
        }
        l.next = l.next.next;
        return j.next;
    }
}
