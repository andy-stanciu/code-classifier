class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode slow = head;
        ListNode fast = head;
        if (fast.next == null || fast == null) {
            if (n == 1) {
                return null;
            }
            return head;
        }
        for (int i = 1; i <= n; i++) {
            fast = fast.next;
        }
        if (fast == null) {
            return head.next;
        }
        while (fast.next != null) {
            slow = slow.next;
            fast = fast.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
