class Solution {

    public ListNode method1(ListNode head, int n) {
        int length = 0;
        ListNode curr = head;
        while (curr != null) {
            length++;
            curr = curr.next;
        }
        int jump = length - n;
        if (jump == 0) {
            return head.next;
        }
        curr = head;
        for (int i = 0; i < jump - 1; i++) {
            curr = curr.next;
        }
        curr.next = curr.next.next;
        return head;
    }
}
