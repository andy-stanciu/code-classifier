class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null) {
            return null;
        }
        ListNode curr = head;
        int size = 0;
        while (curr != null) {
            curr = curr.next;
            size++;
        }
        int SearchtoDelete = size - n;
        if (SearchtoDelete == 0) {
            return head.next;
        }
        ListNode prev = head;
        int i = 1;
        while (i < SearchtoDelete) {
            prev = prev.next;
            i++;
        }
        prev.next = prev.next.next;
        return head;
    }
}
