class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null)
            return null;
        int count = 0;
        ListNode temp = head;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        if (count == n) {
            return head.next;
        }
        count = count - n;
        temp = head;
        while (--count != 0) {
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
