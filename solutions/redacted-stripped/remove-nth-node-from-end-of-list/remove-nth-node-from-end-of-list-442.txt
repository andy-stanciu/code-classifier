class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head, curr = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        if (count == n)
            return head.next;
        temp = head;
        for (int i = 1; i < count - n; i++) {
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
