class Solution {

    public ListNode method1(ListNode head, int n) {
        int size = 0;
        ListNode start = head;
        while (start.next != null) {
            size += 1;
            start = start.next;
        }
        size += 1;
        if (size - n == 0) {
            head = head.next;
            return head;
        }
        int var1 = size - n;
        int i = 0;
        ListNode take = head;
        while (i < var1 - 1) {
            take = take.next;
            i += 1;
        }
        take.next = take.next.next;
        return head;
    }
}
