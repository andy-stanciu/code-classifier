class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode current = head, pos = head;
        for (int i = 0; i < n; i++) current = current.next;
        while (current != null && current.next != null) {
            pos = pos.next;
            current = current.next;
        }
        if (current == null)
            return head.next;
        pos.next = pos.next.next;
        return head;
    }
}
