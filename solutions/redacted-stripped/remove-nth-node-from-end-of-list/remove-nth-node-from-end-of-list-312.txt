class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int count = 0;
        while (curr != null) {
            curr = curr.next;
            count++;
        }
        if (count == n) {
            return head.next;
        }
        curr = head;
        for (int idx = 0; idx < count - n - 1; idx++) {
            curr = curr.next;
        }
        curr.next = curr.next.next;
        return head;
    }
}
