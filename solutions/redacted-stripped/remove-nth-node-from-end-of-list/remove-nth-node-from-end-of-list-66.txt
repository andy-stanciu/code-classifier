class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return null;
        }
        int size = 0;
        ListNode var1 = head;
        while (var1 != null) {
            var1 = var1.next;
            size++;
        }
        if (n == size) {
            return head.next;
        }
        int indexToSearch = size - n;
        int i = 1;
        ListNode var2 = head;
        while (i < indexToSearch) {
            var2 = var2.next;
            i++;
        }
        var2.next = var2.next.next;
        return head;
    }
}
