class Solution {

    private static ListNode method1(ListNode head) {
        ListNode prev = null, curr = head, temp = null;
        while (curr != null) {
            temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
        }
        return prev;
    }

    public ListNode method2(ListNode head, int n) {
        if (head.next == null)
            return null;
        ListNode newHead = method1(head), prev = null, curr = newHead;
        while (--n > 0) {
            prev = curr;
            curr = curr.next;
        }
        if (prev == null) {
            prev = curr;
            curr = curr.next;
            prev.next = null;
            head = method1(curr);
        } else {
            prev.next = curr.next;
            curr.next = null;
            head = method1(newHead);
        }
        return head;
    }
}
