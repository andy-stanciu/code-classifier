class Solution {

    public ListNode method1(ListNode head, int n) {
        int c = 1;
        ListNode p = head;
        ListNode d = head;
        while (p.next != null) {
            if (c++ > n)
                d = d.next;
            p = p.next;
        }
        if (c == n)
            return head.next;
        d.next = d.next.next;
        return head;
    }
}
