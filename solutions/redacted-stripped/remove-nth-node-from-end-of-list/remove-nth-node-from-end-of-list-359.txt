class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return null;
        int len = 0;
        ListNode temp = head;
        while (temp != null) {
            len++;
            temp = temp.next;
        }
        int i = 0;
        temp = head;
        if (n == len)
            return head.next;
        while (i < (len - n - 1)) {
            i++;
            temp = temp.next;
        }
        if (temp.next != null)
            temp.next = temp.next.next;
        return head;
    }
}
