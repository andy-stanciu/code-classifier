class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return null;
        int count = 0;
        ListNode refPtr = head;
        ListNode mainPtr = head;
        while (count < n) {
            refPtr = refPtr.next;
            count++;
        }
        if (refPtr == null)
            return head.next;
        while (refPtr.next != null) {
            refPtr = refPtr.next;
            mainPtr = mainPtr.next;
        }
        mainPtr.next = mainPtr.next.next;
        return head;
    }
}
