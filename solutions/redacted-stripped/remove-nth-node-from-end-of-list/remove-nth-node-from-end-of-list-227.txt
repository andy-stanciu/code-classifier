class Solution {

    public ListNode method1(ListNode head, int n) {
        if (n < 0)
            return head;
        ListNode var1 = head;
        int iterator = 1;
        int var2 = 0;
        while (var1 != null) {
            var2++;
            var1 = var1.next;
        }
        if (var2 == n) {
            return head.next;
        }
        ListNode originalHead = head;
        while (var2 - n != iterator) {
            head = head.next;
            iterator++;
        }
        head.next = head.next.next;
        return originalHead;
    }
}
