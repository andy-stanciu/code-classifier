class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        ListNode curr = head;
        ListNode fast = head;
        while (n > 0) {
            fast = fast.next;
            --n;
        }
        if (fast == null)
            return curr.next;
        while (fast.next != null) {
            curr = curr.next;
            fast = fast.next;
        }
        curr.next = curr.next.next;
        return head;
    }
}
