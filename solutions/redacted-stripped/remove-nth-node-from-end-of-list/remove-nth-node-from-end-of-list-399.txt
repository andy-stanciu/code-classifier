class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            if (n == 0) {
                return head;
            } else {
                return null;
            }
        }
        ListNode fast = head;
        for (int i = 1; i <= n; i++) {
            fast = fast.next;
        }
        if (fast == null) {
            return head.next;
        }
        ListNode slow = head;
        while (fast.next != null) {
            slow = slow.next;
            fast = fast.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
