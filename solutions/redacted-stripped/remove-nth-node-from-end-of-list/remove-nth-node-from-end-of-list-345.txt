class Solution {

    ListNode var1;

    int method1(ListNode head, int n) {
        if (head == null)
            return 0;
        int nextDepth = method1(head.next, n);
        if (nextDepth == n) {
            var1 = head.next;
            head.next = head.next.next;
        }
        return nextDepth + 1;
    }

    public ListNode method2(ListNode head, int n) {
        if (head == null)
            return null;
        int var2 = method1(head, n);
        if (var2 == n) {
            return head.next;
        }
        return head;
    }
}
