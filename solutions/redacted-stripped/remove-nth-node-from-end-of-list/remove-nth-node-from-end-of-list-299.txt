class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int var1 = 0;
        while (curr != null) {
            var1++;
            curr = curr.next;
        }
        int var2 = var1 - n;
        if (var2 == 0) {
            return head.next;
        }
        ListNode prev = null;
        curr = head;
        int track = 0;
        while (curr != null) {
            if (track == var2) {
                prev.next = curr.next;
                break;
            }
            prev = curr;
            curr = curr.next;
            track++;
        }
        return head;
    }
}
