class Solution {

    public ListNode method1(ListNode head, int n) {
        int size = 0;
        ListNode curr = head;
        for (ListNode temp = head; temp != null; temp = temp.next) size++;
        size -= n;
        if (size == 0)
            return head.next;
        for (int i = 0; i < size - 1; i++) curr = curr.next;
        curr.next = curr.next != null ? curr.next.next : null;
        return head;
    }
}
