class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode first = dummy, sec = dummy;
        for (int i = 0; i <= n; i++) {
            first = first.next;
        }
        while (first != null) {
            first = first.next;
            sec = sec.next;
        }
        sec.next = sec.next.next;
        return dummy.next;
    }
}
