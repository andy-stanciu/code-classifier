class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode var1 = head;
        int size = 0;
        while (var1 != null) {
            size++;
            var1 = var1.next;
        }
        int target = size - n + 1;
        int count = 1;
        if (size - n == 0) {
            return head.next;
        }
        var1 = head;
        ListNode secondLast = head;
        ListNode var2 = head.next;
        while (var1 != null) {
            if (count == target - 1) {
                secondLast.next = var2.next;
                var2.next = null;
                break;
            }
            secondLast = var2;
            var2 = var2.next;
            count++;
        }
        return head;
    }
}
