class Solution {

    public ListNode method1(ListNode head, int n) {
        head = new ListNode(-1, head);
        method2(head, n);
        return head.next;
    }

    public int method2(ListNode head, int n) {
        if (head.next == null)
            return 0;
        int var1 = method2(head.next, n);
        if (var1 == n - 1) {
            head.next = head.next.next;
        }
        return var1 + 1;
    }
}
