class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        if (head.next == null) {
            return null;
        }
        int count = 0;
        while (temp != null) {
            temp = temp.next;
            count += 1;
        }
        temp = head;
        int a = count - n;
        if (count == n)
            return head.next;
        for (int i = 1; i < a; i++) {
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
