class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        int var1 = count - n;
        if (var1 == 0) {
            return head.next;
        }
        ListNode current = head;
        for (int i = 1; i < var1; i++) {
            current = current.next;
        }
        current.next = current.next.next;
        return head;
    }
}
