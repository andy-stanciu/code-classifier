class Solution {

    private static int var1;

    public ListNode method1(ListNode head, int n) {
        var1 = n + 1;
        ListNode stub = new ListNode(0);
        stub.next = head;
        var var2 = method2(stub);
        if (var2 != null) {
            var2.next = var2.next.next;
        }
        return stub.next;
    }

    private ListNode method2(ListNode param1) {
        if (param1 == null)
            return null;
        var var2 = method2(param1.next);
        if (var1 > 0 && --var1 == 0) {
            var2 = param1;
        }
        return var2;
    }
}
