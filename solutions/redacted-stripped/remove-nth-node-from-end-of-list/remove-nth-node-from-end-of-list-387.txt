class Solution {

    public ListNode method1(ListNode head, int n) {
        int index = LLlength(head) - n;
        ListNode temp = head;
        int i = 0;
        if (LLlength(head) == 1 && index == 0) {
            head = null;
            return head;
        }
        if (LLlength(head) > 1 && index == 0) {
            head = head.next;
            return head;
        }
        if (index == LLlength(head) - 1) {
            while (i < index - 1) {
                temp = temp.next;
                i++;
            }
            temp.next = null;
            return head;
        }
        while (i < index - 1) {
            i++;
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }

    public int LLlength(ListNode head) {
        int len = 0;
        ListNode temp = head;
        while (temp != null) {
            len++;
            temp = temp.next;
        }
        return len;
    }
}
