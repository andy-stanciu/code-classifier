class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode prev = null;
        while (head != null) {
            ListNode next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        ListNode var1 = new ListNode(0);
        ListNode Store = var1;
        while (n > 1) {
            Store.next = new ListNode(prev.val);
            Store = Store.next;
            prev = prev.next;
            n--;
        }
        prev = prev.next;
        Store.next = prev;
        head = var1.next;
        prev = null;
        while (head != null) {
            ListNode next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        return prev;
    }
}
