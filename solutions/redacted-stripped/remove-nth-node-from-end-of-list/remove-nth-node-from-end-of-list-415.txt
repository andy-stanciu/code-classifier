class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null) {
            return null;
        }
        ListNode fast = head;
        ListNode slow = head;
        ListNode prev = null;
        while (fast != null && n > 0) {
            fast = fast.next;
            --n;
        }
        while (fast != null) {
            fast = fast.next;
            prev = slow;
            slow = slow.next;
        }
        if (prev == null) {
            head = head.next;
        } else if (slow != null) {
            prev.next = slow.next;
        }
        return head;
    }
}
