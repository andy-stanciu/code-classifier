class Solution {

    public ListNode method1(ListNode head, int n) {
        if (n == 1) {
            if (head.next == null) {
                return null;
            }
        }
        ListNode current = head;
        ListNode prev = null;
        while (true) {
            n--;
            if (n < 0) {
                if (prev == null) {
                    prev = head;
                } else {
                    prev = prev.next;
                }
            }
            if (current.next != null) {
                current = current.next;
            } else {
                if (prev == null)
                    return head.next;
                else if (prev.next.next == null) {
                    prev.next = null;
                    return head;
                } else {
                    prev.next = prev.next.next;
                    return head;
                }
            }
        }
    }
}
