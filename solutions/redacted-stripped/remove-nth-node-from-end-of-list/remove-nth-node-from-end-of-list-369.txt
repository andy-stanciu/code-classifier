class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return null;
        }
        int size = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            size++;
        }
        if (n == size) {
            return head.next;
        }
        int ptf = size - n;
        temp = head;
        int pN = 1;
        while (pN < ptf) {
            temp = temp.next;
            pN++;
        }
        temp.next = temp.next.next;
        return head;
    }
}
