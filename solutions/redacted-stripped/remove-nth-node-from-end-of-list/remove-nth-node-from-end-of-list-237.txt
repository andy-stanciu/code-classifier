class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode slow = head;
        ListNode fast = head;
        if (head == null || head.next == null)
            return null;
        while (n > 0) {
            fast = fast.next;
            n--;
        }
        while (fast != null && fast.next != null) {
            fast = fast.next;
            slow = slow.next;
        }
        if (fast != null && slow.next != null)
            slow.next = slow.next.next;
        if (fast == null)
            head = head.next;
        return head;
    }
}
