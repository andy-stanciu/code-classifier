class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null) {
            return null;
        }
        ListNode pre = new ListNode(0);
        ListNode ref = pre;
        ListNode main = pre;
        main.next = head;
        int count = 0;
        while (count <= n) {
            ref = ref.next;
            count++;
        }
        while (ref != null) {
            ref = ref.next;
            main = main.next;
        }
        main.next = main.next.next;
        return pre.next;
    }
}
