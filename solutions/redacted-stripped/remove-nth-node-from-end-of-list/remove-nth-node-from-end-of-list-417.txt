class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head, var1 = new ListNode();
        if (method2(head) == n)
            return head.next;
        for (int i = 0; i < (method2(head) - n - 1); i++) {
            temp = temp.next;
        }
        var1 = temp.next;
        temp.next = var1.next;
        var1.next = null;
        return head;
    }

    public int method2(ListNode head) {
        ListNode temp = head;
        int count = 1;
        while (temp.next != null) {
            temp = temp.next;
            count++;
        }
        return count;
    }
}
