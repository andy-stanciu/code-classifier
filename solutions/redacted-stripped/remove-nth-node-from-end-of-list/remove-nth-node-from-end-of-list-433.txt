class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return head;
        }
        ListNode var1;
        int length = 0;
        ListNode temp = head;
        while (temp != null) {
            length++;
            temp = temp.next;
        }
        if (n > length) {
            return head;
        }
        int var2 = length - (n);
        ListNode prev = head;
        temp = head;
        while (temp != null) {
            if (var2 == 0) {
                if (prev == temp) {
                    return temp.next;
                }
                prev.next = temp.next;
                return head;
            }
            prev = temp;
            temp = temp.next;
            var2--;
        }
        return head;
    }
}
