class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode res = head;
        ListNode first = head;
        int i = 0;
        while (first != null && first.next != null && i < n) {
            first = first.next;
            i++;
        }
        if (i < n) {
            res = res.next;
            return res;
        }
        ListNode second = head;
        while (first != null && first.next != null) {
            first = first.next;
            second = second.next;
        }
        second.next = second.next.next;
        return res;
    }
}
