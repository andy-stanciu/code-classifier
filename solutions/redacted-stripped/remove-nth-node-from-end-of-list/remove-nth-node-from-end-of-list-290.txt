class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode var1 = head;
        int len = 0;
        while (var1 != null) {
            len++;
            var1 = var1.next;
        }
        int index = len - n;
        ListNode prev = null;
        ListNode current = head;
        if (index > 0) {
            for (int i = 0; i < index; i++) {
                prev = current;
                current = current.next;
            }
        }
        if (prev == null) {
            if (current.next != null)
                head = current.next;
            else
                return null;
        } else {
            prev.next = current.next;
        }
        return head;
    }
}
