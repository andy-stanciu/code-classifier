class Solution {

    public ListNode method1(ListNode head, int n) {
        int length = 0;
        if (head.next == null || head == null) {
            head = null;
            return head;
        }
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            length++;
        }
        temp = head;
        int num = length - n;
        if (num <= 0) {
            head = head.next;
            return head;
        }
        int i = 0;
        while (i < num - 1) {
            temp = temp.next;
            i++;
        }
        temp.next = temp.next.next;
        return head;
    }
}
