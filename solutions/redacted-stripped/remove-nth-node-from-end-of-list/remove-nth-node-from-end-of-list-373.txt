class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        int count = 0, len = 0;
        ListNode newHead = head, del = head;
        while (head.next != null) {
            if (count != n)
                count++;
            else
                del = del.next;
            head = head.next;
            len++;
        }
        len++;
        if (len == n)
            return newHead.next;
        if (del.next.next == null)
            del.next = null;
        else
            del.next = del.next.next;
        return newHead;
    }
}
