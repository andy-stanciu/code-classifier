class Solution {

    public ListNode method1(ListNode head, int n) {
        List<Integer> var1 = new ArrayList<>();
        ListNode dummy = new ListNode(0);
        ListNode current = dummy;
        while (head != null) {
            var1.add(head.val);
            head = head.next;
        }
        var1.remove(var1.size() - n);
        for (int i : var1) {
            current.next = new ListNode(i);
            current = current.next;
        }
        return dummy.next;
    }
}
