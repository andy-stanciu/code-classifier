class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode dummy = head;
        int idx = 0;
        while (dummy != null) {
            dummy = dummy.next;
            idx++;
        }
        if (idx == 1)
            return null;
        dummy = head;
        ListNode swap = null;
        boolean swapComplete = false;
        for (int i = 1; i <= idx; i++) {
            if (i == idx - n) {
                swap = dummy;
            } else if (i == idx - n + 2 && swap != null) {
                swap.next = dummy;
                swapComplete = true;
                break;
            }
            dummy = dummy.next;
        }
        if (!swapComplete && swap != null)
            swap.next = null;
        else if (swap == null) {
            head = head.next;
        }
        return head;
    }
}
