class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode var1 = head;
        ListNode current = head;
        for (int i = 0; i < n; i++) {
            current = current.next;
        }
        if (current == null) {
            return head.next;
        }
        while (current.next != null) {
            current = current.next;
            var1 = var1.next;
        }
        var1.next = var1.next.next;
        return head;
    }
}
