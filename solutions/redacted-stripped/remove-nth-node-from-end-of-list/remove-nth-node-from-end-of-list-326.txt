class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode var1 = new ListNode();
        var1.next = head;
        ListNode tempHead = var1, var2 = var1;
        while (tempHead.next != null) {
            if (n == 0) {
                var2 = var2.next;
            } else {
                n--;
            }
            tempHead = tempHead.next;
        }
        var2.next = var2.next == null ? null : var2.next.next;
        return var1.next;
    }
}
