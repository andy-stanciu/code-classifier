class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode current = head;
        int i = 0;
        while (current.next != null) {
            i++;
            current = current.next;
        }
        if (n == i + 1) {
            return head.next;
        }
        int m = i - n + 1;
        current = head;
        for (int k = 0; k < m - 1; k++) {
            current = current.next;
        }
        current.next = current.next.next;
        return head;
    }
}
