class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        if (n == 0)
            return head;
        ListNode start = new ListNode();
        start.next = head;
        ListNode fast = start;
        ListNode slow = start;
        for (int i = 1; i <= n; ++i) {
            fast = fast.next;
        }
        while (fast.next != null) {
            fast = fast.next;
            slow = slow.next;
        }
        if (slow.next == head)
            return slow.next.next;
        slow.next = slow.next.next;
        return head;
    }
}
