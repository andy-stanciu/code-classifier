class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int count = 0;
        while (curr != null) {
            count++;
            curr = curr.next;
        }
        int del = count - n;
        if (del == 0) {
            return head.next;
        }
        ListNode prev = null;
        curr = head;
        int count1 = 0;
        while (count1 != del) {
            count1++;
            prev = curr;
            curr = curr.next;
        }
        prev.next = curr.next;
        return head;
    }
}
