class Solution {

    public ListNode method1(ListNode head, int n) {
        int length = 0;
        ListNode temp = head;
        while (temp != null) {
            length++;
            temp = temp.next;
        }
        if (n == length) {
            return head.next;
        }
        int index = length - n;
        ListNode temp2 = head;
        for (int i = 0; i < index - 1; i++) {
            temp2 = temp2.next;
        }
        temp2.next = temp2.next.next;
        return head;
    }
}
