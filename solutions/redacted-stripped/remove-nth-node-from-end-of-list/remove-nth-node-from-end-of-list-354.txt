class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        if (count == n) {
            head = head.next;
            return head;
        }
        temp = head;
        int res = count - n;
        while (temp != null) {
            res--;
            if (res == 0)
                break;
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
