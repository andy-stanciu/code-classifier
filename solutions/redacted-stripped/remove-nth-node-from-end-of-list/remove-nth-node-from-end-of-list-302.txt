class Solution {

    public ListNode method1(ListNode head, int n) {
        int count = 0;
        ListNode temp = head;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        int prevPos = count - n - 1;
        if (prevPos == -1) {
            head = head.next;
            return head;
        }
        count = 0;
        temp = head;
        while (count != prevPos) {
            count++;
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
