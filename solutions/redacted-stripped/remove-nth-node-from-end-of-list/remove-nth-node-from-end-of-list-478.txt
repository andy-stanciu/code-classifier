class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null && n == 1) {
            return null;
        }
        ListNode fast = head;
        int count = 1;
        while (fast.next != null) {
            if (fast.next.next != null) {
                count += 2;
                fast = fast.next.next;
            } else {
                count += 1;
                fast = fast.next;
            }
        }
        int var1 = count - n - 1;
        if (var1 == -1) {
            return head.next;
        }
        ListNode prev = head;
        int itr = 0;
        while (itr != var1) {
            prev = prev.next;
            itr++;
        }
        prev.next = prev.next.next;
        return head;
    }
}
