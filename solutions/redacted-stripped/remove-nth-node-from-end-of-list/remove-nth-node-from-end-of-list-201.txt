class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode p1 = dummy, p2 = dummy;
        while (n-- >= 0) {
            p1 = p1.next;
        }
        while (p1 != null) {
            p2 = p2.next;
            p1 = p1.next;
        }
        p2.next = p2.next.next;
        return dummy.next;
    }
}
