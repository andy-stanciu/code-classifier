class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int sz = 0;
        while (temp != null) {
            temp = temp.next;
            sz++;
        }
        if (sz == n) {
            return head.next;
        }
        int i = 0;
        int toSize = sz - n - 1;
        ListNode prev = head;
        while (i < toSize) {
            prev = prev.next;
            i++;
        }
        prev.next = prev.next.next;
        return head;
    }
}
