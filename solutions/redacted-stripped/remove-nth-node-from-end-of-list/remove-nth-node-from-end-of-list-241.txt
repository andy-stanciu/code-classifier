class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return null;
        ListNode var1 = method2(head, n);
        if (var1 == head)
            return head.next;
        ListNode var2 = method2(head, n + 1);
        var2.next = var1.next;
        return head;
    }

    private ListNode method2(ListNode head, int index) {
        int length = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            length++;
        }
        int var3 = length - index;
        temp = head;
        for (int i = 0; i < var3; i++) {
            temp = temp.next;
        }
        return temp;
    }
}
