class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int len = 0;
        while (curr != null) {
            len++;
            curr = curr.next;
        }
        int var1 = len - n + 1;
        if (var1 == 1)
            return head.next;
        curr = head;
        while (--var1 > 1) {
            curr = curr.next;
        }
        curr.next = curr.next.next;
        return head;
    }
}
