class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode p1 = head;
        ListNode p2 = head;
        ListNode prev = head;
        for (int i = 0; i < n; i++) {
            if (p1 == null) {
                return null;
            }
            p1 = p1.next;
        }
        while (p1 != null) {
            p1 = p1.next;
            prev = p2;
            p2 = p2.next;
        }
        if (p2 == head) {
            head = head.next;
            return head;
        }
        prev.next = p2.next;
        return head;
    }
}
