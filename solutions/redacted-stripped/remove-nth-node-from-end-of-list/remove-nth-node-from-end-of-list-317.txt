class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int cnt = 0;
        while (temp != null) {
            cnt++;
            temp = temp.next;
        }
        temp = head;
        if (cnt == n) {
            head = head.next;
        }
        cnt = cnt - n;
        while (temp != null) {
            cnt--;
            if (cnt == 0) {
                temp.next = temp.next.next;
                break;
            }
            temp = temp.next;
        }
        return head;
    }
}
