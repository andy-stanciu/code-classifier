class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return null;
        }
        int len = 0;
        ListNode curr = head;
        while (curr != null) {
            curr = curr.next;
            len++;
        }
        if (len == n)
            return head.next;
        int count = 1;
        curr = head;
        while (curr != null) {
            if (count == len - n) {
                curr.next = curr.next.next;
            }
            curr = curr.next;
            count++;
        }
        return head;
    }
}
