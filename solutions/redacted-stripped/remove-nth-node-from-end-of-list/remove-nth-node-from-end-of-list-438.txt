class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int size = 0;
        while (temp != null) {
            temp = temp.next;
            size++;
        }
        if (size == n) {
            return head.next;
        }
        temp = head;
        int idx = size - n;
        int k = 1;
        while (k < idx) {
            temp = temp.next;
            k++;
        }
        temp.next = temp.next.next;
        return head;
    }
}
