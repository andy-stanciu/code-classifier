class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return null;
        if (head.next == null && n == 1) {
            head = null;
            return head;
        }
        ListNode last = head;
        int length = 1;
        while (last.next != null) {
            last = last.next;
            length++;
        }
        if (length == n) {
            ListNode temp = head.next;
            head = null;
            return temp;
        }
        ListNode prev = null;
        ListNode var1 = head;
        int var2 = length - n;
        for (int i = 0; i < var2 && var1 != null; i++) {
            prev = var1;
            var1 = var1.next;
        }
        if (var1.next == null) {
            prev.next = null;
        } else {
            prev.next = var1.next;
        }
        return head;
    }
}
