class Solution {

    public ListNode method1(ListNode head, int n) {
        HashMap<Integer, ListNode> hm = new HashMap<>();
        int length = getLength(head, hm);
        if (n == length) {
            return head.next;
        } else if (n == 1) {
            hm.get(2).next = null;
            return head;
        } else {
            hm.get(n + 1).next = hm.get(n).next;
            hm.get(n).next = null;
            return head;
        }
    }

    public int getLength(ListNode head, HashMap<Integer, ListNode> hm) {
        if (head == null) {
            return 0;
        }
        int var1 = getLength(head.next, hm) + 1;
        hm.put(var1, head);
        return var1;
    }
}
