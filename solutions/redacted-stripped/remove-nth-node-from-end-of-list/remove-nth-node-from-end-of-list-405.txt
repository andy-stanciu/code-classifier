class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int length = 0;
        while (temp != null) {
            length++;
            temp = temp.next;
        }
        int skip = length - n;
        if (skip == 0) {
            return head.next;
        }
        temp = head;
        for (int i = 0; i < skip - 1; i++) {
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
