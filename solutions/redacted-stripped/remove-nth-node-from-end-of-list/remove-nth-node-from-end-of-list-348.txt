class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode ptr = head;
        int count = 0;
        while (ptr != null) {
            ptr = ptr.next;
            count++;
        }
        ptr = head;
        int len = count - n;
        if (count == n) {
            return head.next;
        }
        for (int i = 1; i < len; i++) {
            ptr = ptr.next;
        }
        ptr.next = ptr.next.next;
        return head;
    }
}
