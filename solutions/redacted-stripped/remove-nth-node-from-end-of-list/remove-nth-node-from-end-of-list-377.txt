class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode s = head;
        ListNode f = head;
        while (n > 0) {
            f = f.next;
            n--;
        }
        if (f == null) {
            return head.next;
        }
        while (f.next != null) {
            s = s.next;
            f = f.next;
        }
        s.next = s.next.next;
        return head;
    }
}
