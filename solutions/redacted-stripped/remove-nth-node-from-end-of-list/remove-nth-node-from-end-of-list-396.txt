class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head, prev = head;
        int count = 1;
        while (temp.next != null) {
            if (count > n) {
                prev = prev.next;
            }
            temp = temp.next;
            count++;
        }
        if (count == n) {
            head = head.next;
        } else {
            prev.next = prev.next.next;
        }
        return head;
    }
}
