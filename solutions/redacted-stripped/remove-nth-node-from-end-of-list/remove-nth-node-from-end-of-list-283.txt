class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return head;
        }
        ListNode temp = head;
        int i = 0;
        while (temp != null) {
            temp = temp.next;
            i++;
        }
        temp = head;
        while (i - n > 1) {
            temp = temp.next;
            i--;
        }
        if (i - n > 0) {
            temp.next = temp.next.next;
        } else {
            temp = temp.next;
            head = temp;
        }
        return head;
    }
}
