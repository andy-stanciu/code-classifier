class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null)
            return null;
        ListNode fast = head;
        for (int i = 0; i < n; i++) fast = fast.next;
        ListNode slow = head;
        if (fast == null)
            return head.next;
        while (fast.next != null) {
            slow = slow.next;
            fast = fast.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
