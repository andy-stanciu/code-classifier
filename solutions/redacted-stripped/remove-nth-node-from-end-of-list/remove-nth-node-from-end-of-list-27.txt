class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int var1 = 0;
        while (temp != null) {
            temp = temp.next;
            var1++;
        }
        if (n == var1) {
            head = head.next;
            return head;
        }
        if (var1 == 1) {
            return null;
        }
        temp = head;
        int present = 1;
        while (present < var1 - n) {
            temp = temp.next;
            present++;
        }
        temp.next = temp.next.next;
        return head;
    }
}
