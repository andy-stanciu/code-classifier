public class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode slow = dummy;
        ListNode fast = dummy;
        for (int i = 0; i <= n; i++) {
            fast = fast.next;
        }
        if (fast == null) {
            ListNode temp = head;
            head = head.next;
            temp = null;
            return head;
        }
        while (fast != null) {
            slow = slow.next;
            fast = fast.next;
        }
        ListNode toBeDeleted = slow.next;
        slow.next = slow.next.next;
        toBeDeleted = null;
        return dummy.next;
    }
}
