class Solution {

    public ListNode method1(ListNode head, int n) {
        int length = 0;
        ListNode curr = head;
        while (curr != null) {
            length++;
            curr = curr.next;
        }
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        curr = dummy;
        for (int i = 0; i < length - n; i++) {
            curr = curr.next;
        }
        if (curr.next != null) {
            curr.next = curr.next.next;
        }
        return dummy.next;
    }
}
