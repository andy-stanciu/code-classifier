class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null) {
            return null;
        }
        int size = 0;
        ListNode curr = head;
        while (curr != null) {
            curr = curr.next;
            size++;
        }
        if (size == n) {
            return head.next;
        }
        int indexToSearch = size - n;
        int i = 1;
        ListNode prev = head;
        while (i < indexToSearch) {
            prev = prev.next;
            i++;
        }
        prev.next = prev.next.next;
        return head;
    }
}
