class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fast = head;
        ListNode slow = head;
        ListNode prev = null;
        while (n > 0) {
            fast = fast.next;
            n--;
        }
        while (fast != null) {
            prev = slow;
            slow = slow.next;
            fast = fast.next;
        }
        if (slow == head) {
            return head.next;
        }
        prev.next = slow.next;
        return head;
    }
}
