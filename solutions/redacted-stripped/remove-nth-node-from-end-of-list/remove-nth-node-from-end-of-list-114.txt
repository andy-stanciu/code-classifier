class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        int len = 0;
        ListNode curr = head;
        while (curr != null) {
            curr = curr.next;
            len++;
        }
        curr = head;
        len = len - n;
        if (len == 0) {
            head = head.next;
        }
        while (len > 1) {
            curr = curr.next;
            len--;
        }
        curr.next = curr.next.next;
        return head;
    }
}
