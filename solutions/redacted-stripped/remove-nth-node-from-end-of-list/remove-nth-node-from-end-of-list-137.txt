class Solution {

    int length(ListNode head) {
        int len = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            len++;
        }
        return len;
    }

    public ListNode method1(ListNode head, int n) {
        int len = length(head);
        ListNode var1 = head;
        int pos = 1;
        while (pos < len - n) {
            var1 = var1.next;
            pos++;
        }
        if (len == n)
            head = head.next;
        else if (var1.next == null)
            return null;
        else
            var1.next = var1.next.next;
        return head;
    }
}
