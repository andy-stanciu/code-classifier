class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode x = head;
        int size = 0;
        while (x != null) {
            x = x.next;
            size++;
        }
        if (size == 1)
            return null;
        ListNode temp = head;
        int m = size - n;
        if (m == 0) {
            return head.next;
        }
        int count = 0;
        while (count < m - 1) {
            temp = temp.next;
            count++;
        }
        if (temp.next == null)
            temp.next = null;
        else
            temp.next = temp.next.next;
        temp = temp.next;
        return head;
    }
}
