class Solution {

    public ListNode method1(ListNode head, int n) {
        int size = 0;
        ListNode temp = head;
        ListNode curr = head;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        if (size == n) {
            return curr.next;
        }
        int k = size - n;
        for (int i = 1; i < k; i++) {
            curr = curr.next;
        }
        ListNode store = curr.next.next;
        curr.next.next = null;
        curr.next = null;
        curr.next = store;
        return head;
    }
}
