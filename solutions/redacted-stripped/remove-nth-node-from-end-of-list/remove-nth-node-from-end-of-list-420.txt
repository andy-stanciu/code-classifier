class Solution {

    public ListNode method1(ListNode head, int n) {
        int count = 0;
        ListNode cur = head;
        while (cur != null) {
            cur = cur.next;
            count++;
        }
        cur = head;
        if (count == n) {
            return head.next;
        }
        for (int i = 1; i < count - n; i++) {
            cur = cur.next;
        }
        if (cur.next != null) {
            cur.next = cur.next.next;
        }
        return head;
    }
}
