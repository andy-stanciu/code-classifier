class Solution {

    public ListNode method1(ListNode head, int n) {
        int len = 0;
        ListNode temp = head;
        while (head != null) {
            len++;
            head = head.next;
        }
        head = temp;
        int index = len - n;
        int i = 0;
        ListNode parent = null;
        while (head != null) {
            if (i == index)
                break;
            parent = head;
            head = head.next;
            i++;
        }
        if (parent != null) {
            parent.next = head.next;
        } else {
            temp = head.next;
        }
        return temp;
    }
}
