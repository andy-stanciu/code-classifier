class Solution {

    public ListNode method1(ListNode head, int n) {
        int size = 0;
        ListNode curr = head;
        while (curr != null) {
            size++;
            curr = curr.next;
        }
        int pos = size - n + 1;
        if (pos == 1) {
            head = head.next;
            return head;
        }
        curr = head;
        int count = 0;
        while (curr != null) {
            count++;
            if (count == pos - 1) {
                curr.next = curr.next.next;
                break;
            }
            curr = curr.next;
        }
        return head;
    }
}
