class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null) {
            return null;
        }
        ListNode temp = head;
        int size = 0;
        while (temp != null) {
            size += 1;
            temp = temp.next;
        }
        temp = head;
        size = size - n;
        if (size == 0) {
            head = head.next;
        }
        while (size > 1) {
            temp = temp.next;
            size--;
        }
        temp.next = temp.next.next;
        return head;
    }
}
