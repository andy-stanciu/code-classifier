class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode c = head;
        int l = 0;
        while (c != null) {
            l++;
            c = c.next;
        }
        if (l == n) {
            return head.next;
        }
        l = l - n - 1;
        c = head;
        for (int i = 0; i != l; i++) {
            c = c.next;
        }
        c.next = c.next.next;
        return head;
    }
}
