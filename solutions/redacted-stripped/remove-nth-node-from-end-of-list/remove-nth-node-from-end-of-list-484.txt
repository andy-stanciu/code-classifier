class Solution {

    private static int length(ListNode head) {
        int count = 0;
        ListNode curr = head;
        while (curr != null) {
            curr = curr.next;
            count++;
        }
        return count;
    }

    public ListNode method1(ListNode head, int n) {
        ListNode curr = head;
        int len = length(head);
        if (n == len) {
            return head.next;
        }
        for (int i = 0; i < len - n - 1; i++) {
            curr = curr.next;
        }
        if (curr.next != null) {
            curr.next = curr.next.next;
        }
        return head;
    }
}
