class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fast = head, slow = head;
        while (n-- > 0) {
            if (fast == null) {
                return null;
            }
            fast = fast.next;
        }
        if (fast == null && n == 0) {
            return null;
        }
        if (fast == null) {
            slow = slow.next;
            return slow;
        }
        fast = fast.next;
        while (fast != null && slow != null) {
            slow = slow.next;
            fast = fast.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
