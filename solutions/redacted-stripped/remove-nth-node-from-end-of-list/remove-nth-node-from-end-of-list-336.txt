class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return head;
        }
        ListNode current = head;
        int var1 = 0;
        while (current != null) {
            var1++;
            current = current.next;
        }
        if (var1 == n)
            return head.next;
        int var2 = var1 - n;
        int count = 1;
        ListNode newCurrent = head;
        while (count < var2) {
            newCurrent = newCurrent.next;
            count++;
        }
        newCurrent.next = newCurrent.next.next;
        return head;
    }
}
