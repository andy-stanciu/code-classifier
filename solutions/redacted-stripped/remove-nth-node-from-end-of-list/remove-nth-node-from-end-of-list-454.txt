class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode cur = head;
        int i = 0;
        while (cur != null) {
            i++;
            cur = cur.next;
        }
        if (i == n) {
            return head.next;
        }
        int k = Math.abs(i - n);
        int j = 0;
        ListNode temp = head;
        ListNode prev = new ListNode(0);
        while (j < k && temp.next != null) {
            j++;
            prev = temp;
            temp = temp.next;
        }
        prev.next = temp.next;
        return head;
    }
}
