class Solution {

    public ListNode method1(ListNode head, int n) {
        int length = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            length++;
        }
        if (length == n)
            temp = head.next;
        else
            temp = head;
        if (length == n)
            return temp;
        while (length - n > 1) {
            temp = temp.next;
            length--;
        }
        temp.next = temp.next.next;
        return head;
    }
}
