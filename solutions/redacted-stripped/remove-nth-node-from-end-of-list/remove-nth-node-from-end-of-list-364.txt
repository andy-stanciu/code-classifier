class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int sz = 0;
        while (temp != null) {
            temp = temp.next;
            sz++;
        }
        if (n == sz) {
            head = head.next;
            return head;
        }
        int i = 1;
        int var1 = sz - n;
        ListNode prev = head;
        while (i < var1) {
            prev = prev.next;
            i++;
        }
        prev.next = prev.next.next;
        return head;
    }
}
