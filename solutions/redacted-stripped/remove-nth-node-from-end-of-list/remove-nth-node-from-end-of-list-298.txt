class Solution {

    public ListNode method1(ListNode head, int n) {
        int l = 0;
        ListNode tmp = head;
        while (tmp != null) {
            l++;
            tmp = tmp.next;
        }
        if (l == n) {
            return head.next;
        }
        int pos = l - n + 1;
        l = 1;
        tmp = head;
        ListNode prev = null;
        while (l != pos) {
            prev = tmp;
            tmp = tmp.next;
            l++;
        }
        if (tmp != null)
            prev.next = tmp.next;
        else
            prev.next = null;
        return head;
    }
}
