class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode start = head;
        int count = 0;
        while (start != null) {
            count++;
            start = start.next;
        }
        start = head;
        if (n == count)
            return head.next;
        for (int i = 0; i < count - n - 1; i++) {
            start = start.next;
        }
        start.next = start.next.next;
        return head;
    }
}
