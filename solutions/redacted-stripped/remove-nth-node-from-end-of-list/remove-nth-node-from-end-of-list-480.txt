class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fast = head;
        ListNode slow = head;
        int p = 0;
        while (p < n) {
            fast = fast.next;
            p++;
        }
        if (fast == null) {
            return head.next;
        }
        while (fast.next != null) {
            fast = fast.next;
            slow = slow.next;
        }
        slow.next = slow.next.next;
        return head;
    }
}
