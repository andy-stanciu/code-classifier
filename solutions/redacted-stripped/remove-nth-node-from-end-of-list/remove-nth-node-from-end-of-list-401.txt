class Solution {

    public ListNode method1(ListNode head, int param1) {
        ListNode current = head;
        int length = 0;
        while (current != null) {
            length += 1;
            current = current.next;
        }
        length = length - param1;
        current = head;
        if (length == 0)
            return current.next;
        while (true) {
            if (length == 1) {
                if (param1 > 1) {
                    current.next = current.next.next;
                } else {
                    current.next = null;
                }
                break;
            }
            length--;
            current = current.next;
        }
        return head;
    }
}
