class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || head.next == null)
            return null;
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        ListNode temp2 = head;
        if (count - n == 0)
            return head.next;
        for (int i = 1; i < count - n; i++) {
            temp2 = temp2.next;
        }
        temp2.next = temp2.next.next;
        return head;
    }
}
