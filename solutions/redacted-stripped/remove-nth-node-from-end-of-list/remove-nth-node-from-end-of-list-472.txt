class Solution {

    public ListNode method1(ListNode head, int n) {
        var current = head;
        var sec = head;
        var previous = head;
        if (head.next == null)
            return head = head.next;
        for (int i = 0; i < n - 1; i++) {
            sec = sec.next;
        }
        while (sec.next != null) {
            previous = current;
            current = current.next;
            sec = sec.next;
        }
        if (current != head) {
            previous.next = current.next;
            current.next = null;
        } else
            head = current.next;
        return head;
    }
}
