class Solution {

    public ListNode method1(ListNode head, int n) {
        int l = length(head);
        int m = l - n;
        ListNode cur = head;
        if (n == l) {
            return head.next;
        } else {
            while (m - 1 > 0) {
                cur = cur.next;
                m -= 1;
            }
            cur.next = cur.next.next;
            return head;
        }
    }

    public int length(ListNode head) {
        int c = 0;
        ListNode cur = head;
        while (cur != null) {
            cur = cur.next;
            c += 1;
        }
        return c;
    }
}
