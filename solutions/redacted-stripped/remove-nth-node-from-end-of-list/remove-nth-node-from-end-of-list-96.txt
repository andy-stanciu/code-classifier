class Solution {

    private int count = 0;

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return null;
        }
        ListNode var1 = method1(head.next, n);
        count++;
        if (count == n) {
            return var1;
        }
        head.next = var1;
        return head;
    }
}
