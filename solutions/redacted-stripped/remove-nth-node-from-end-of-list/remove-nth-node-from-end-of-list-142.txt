class Solution {

    public ListNode method1(ListNode head, int n) {
        int length = getLength(head);
        ListNode prev = null, temp = head;
        int count = 0;
        while (count < length - n) {
            prev = temp;
            temp = temp.next;
            count++;
        }
        ListNode toBeDeleted = temp;
        ListNode var1 = toBeDeleted.next;
        if (prev != null) {
            prev.next = var1;
        } else {
            return head.next;
        }
        return head;
    }

    private int getLength(ListNode head) {
        int length = 0;
        while (head != null) {
            head = head.next;
            length++;
        }
        return length;
    }
}
