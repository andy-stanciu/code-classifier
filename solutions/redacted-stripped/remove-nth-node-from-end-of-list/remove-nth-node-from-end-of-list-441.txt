class Solution {

    int method1(ListNode head) {
        ListNode temp = head;
        int c = 0;
        while (temp != null) {
            c++;
            temp = temp.next;
        }
        return c;
    }

    public ListNode method2(ListNode head, int n) {
        if (head == null || head.next == null)
            return null;
        int count = method1(head);
        int m = count - n;
        if (m == 0)
            return head.next;
        ListNode temp = head;
        for (int i = 0; i < m - 1; i++) {
            temp = temp.next;
        }
        temp.next = temp.next.next;
        return head;
    }
}
