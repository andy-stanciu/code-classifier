class Solution {

    public String method1(String s) {
        String res = "";
        int n = s.length();
        char[] sc = s.toCharArray();
        for (int i = 0; i < n; i++) {
            int l = i - 1, r = i + 1;
            while (l >= 0 && r < n) {
                if (sc[l] != sc[r])
                    break;
                l--;
                r++;
            }
            if (r - l - 1 > res.length()) {
                res = s.substring(l + 1, r);
            }
            if (r == n)
                break;
            if (i != n - 1 && sc[i] == sc[i + 1]) {
                l = i - 1;
                r = i + 2;
                while (l >= 0 && r < n) {
                    if (sc[l] != sc[r])
                        break;
                    l--;
                    r++;
                }
                if (r - l - 1 > res.length()) {
                    res = s.substring(l + 1, r);
                }
            }
            if (r == n)
                break;
        }
        return res;
    }
}
