class Solution {

    public String method1(String s) {
        boolean[][] dp = new boolean[s.length()][s.length()];
        int maxl = 1;
        int st = 0, e = 0;
        for (int i = 0; i < s.length(); ++i) {
            dp[i][i] = true;
            for (int j = 0; j < i; ++j) {
                if (s.charAt(i) == s.charAt(j) && (i - j <= 2 || dp[j + 1][i - 1])) {
                    dp[j][i] = true;
                    if (i - j + 1 > maxl) {
                        maxl = i - j + 1;
                        st = j;
                        e = i;
                    }
                }
            }
        }
        return s.substring(st, e + 1);
    }
}
