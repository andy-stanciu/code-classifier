class Solution {

    String res = "";

    int max = 0;

    public String method1(String s) {
        if (s.length() == 1) {
            return s;
        }
        String[][] dp = new String[s.length() + 1][s.length() + 1];
        return method2(0, s.length(), s, dp);
    }

    String method2(int i, int j, String s, String[][] dp) {
        if (i >= s.length() || j <= -1) {
            return "";
        }
        if (dp[i][j] != null) {
            return dp[i][j];
        }
        if (i < j) {
            String r = s.substring(i, j);
            if (r.charAt(0) == r.charAt(r.length() - 1)) {
                String var1 = new StringBuilder(r).reverse().toString();
                if (r.equals(var1)) {
                    return r;
                }
            }
        }
        String right = method2(i, j - 1, s, dp);
        String left = method2(i + 1, j, s, dp);
        dp[i][j] = right.length() > left.length() ? right : left;
        return dp[i][j];
    }
}
