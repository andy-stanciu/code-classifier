class Solution {

    public String method1(String s) {
        String largestPalindrome = s.param1(0, 1);
        for (int i = 0; i < s.length(); i++) {
            char currentChar = s.charAt(i);
            int foundMatchAtIndex = i;
            while (foundMatchAtIndex < s.length()) {
                foundMatchAtIndex++;
                if (foundMatchAtIndex < s.length() && currentChar == s.charAt(foundMatchAtIndex)) {
                    String var1 = s.param1(i, foundMatchAtIndex + 1);
                    if (isPalindrome(var1)) {
                        if (var1.length() > largestPalindrome.length()) {
                            largestPalindrome = var1;
                        }
                    }
                }
            }
        }
        return largestPalindrome;
    }

    private boolean isPalindrome(String param1) {
        return param1.equals(new StringBuilder(param1).reverse().toString());
    }
}
