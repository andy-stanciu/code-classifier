class Solution {

    public String method1(String s) {
        int n = s.length();
        if (n == 1)
            return s;
        boolean[][] dp = new boolean[n][n];
        for (int i = 0; i < n; i++) dp[i][i] = true;
        int start = 0, end = 0;
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (s.charAt(i) == s.charAt(j)) {
                    if (i - j == 1 || dp[j + 1][i - 1]) {
                        dp[j][i] = true;
                        if (end - start < i - j) {
                            start = j;
                            end = i;
                        }
                    }
                } else
                    dp[j][i] = false;
            }
        }
        return s.substring(start, end + 1);
    }
}
