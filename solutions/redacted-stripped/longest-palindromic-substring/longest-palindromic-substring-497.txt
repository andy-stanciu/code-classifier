class Solution {

    public String method1(String s) {
        String var1 = "";
        int var2 = 0;
        if (s.length() == 1) {
            return s;
        }
        for (int i = 0; i < s.length(); i++) {
            for (int j = i; j < s.length(); j++) {
                if (s.charAt(i) == s.charAt(j) && s.substring(i, j + 1).length() > var2) {
                    if (isPalindrome(s.substring(i, j + 1))) {
                        var1 = s.substring(i, j + 1);
                        var2 = var1.length();
                    }
                }
            }
        }
        return var1;
    }

    private static boolean isPalindrome(String s) {
        int i = 0;
        int j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i) == s.charAt(j)) {
                i++;
                j--;
            } else {
                return false;
            }
        }
        return true;
    }
}
