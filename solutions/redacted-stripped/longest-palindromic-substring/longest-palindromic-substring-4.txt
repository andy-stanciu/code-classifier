class Solution {

    public String method1(String s) {
        if (s.length() <= 1)
            return s;
        int max = 0;
        int idx = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            int j = max + i - 1;
            int len = 0;
            while (j < s.length()) {
                if (isPalindrome(s, i, j)) {
                    len = j - i;
                }
                if (max < len) {
                    max = len;
                    idx = i;
                }
                j++;
            }
        }
        String str = "";
        System.out.println(max);
        System.out.println(idx);
        for (int i = idx; i <= max + idx; i++) {
            str = str + s.charAt(i);
        }
        return str;
    }

    public boolean isPalindrome(String s, int si, int ei) {
        while (si < ei) {
            if (s.charAt(si) != s.charAt(ei))
                return false;
            si++;
            ei--;
        }
        return true;
    }
}
