class Solution {

    public String method1(String s) {
        String answer = "";
        for (int i = 0; i < s.length(); i++) {
            String pal_odd = expand(s, i, i);
            String pal_even = expand(s, i, i + 1);
            if (pal_odd.length() > answer.length()) {
                answer = pal_odd;
            }
            if (pal_even.length() > answer.length()) {
                answer = pal_even;
            }
        }
        return answer;
    }

    public String expand(String s, int i, int j) {
        int n = s.length();
        while (i >= 0 && j < n && s.charAt(i) == s.charAt(j)) {
            i--;
            j++;
        }
        return s.substring(i + 1, j);
    }
}
