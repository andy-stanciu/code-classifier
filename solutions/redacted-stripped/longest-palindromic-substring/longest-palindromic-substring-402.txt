class Solution {

    public boolean isPalindrome(String str, int i, int j) {
        while (i < j) {
            if (str.charAt(i) != str.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }

    public String method1(String s) {
        if (s.length() <= 1) {
            return s;
        }
        int l = 0;
        int r = s.length() - 1;
        String max = "";
        while (l <= r) {
            if (isPalindrome(s, l, r)) {
                if (max.length() < r - l + 1) {
                    max = s.substring(l, r + 1);
                }
                l++;
                r = s.length() - 1;
            } else {
                if (r == l) {
                    l++;
                    r = s.length() - 1;
                } else {
                    r--;
                }
            }
        }
        return max;
    }
}
