class Solution {

    public String method1(String s) {
        String var1 = "";
        for (int i = 0; i < s.length(); i++) {
            int offset = var1.length() / 2;
            String rslt = expandOut(s, i - offset, i + offset);
            String rslt2 = expandOut(s, i - offset, i + 1 + offset);
            if (rslt.length() > var1.length()) {
                var1 = rslt;
            }
            if (rslt2.length() > var1.length()) {
                var1 = rslt2;
            }
        }
        return var1;
    }

    public String expandOut(String s, int start, int end) {
        int lPtr = start;
        int rPtr = end;
        String var2 = "";
        while (lPtr >= 0 && rPtr < s.length()) {
            if (isPalindrone(s.substring(lPtr, rPtr + 1))) {
                var2 = s.substring(lPtr, rPtr + 1);
                lPtr--;
                rPtr++;
            } else {
                return var2;
            }
        }
        return var2;
    }

    public boolean isPalindrone(String s) {
        for (int i = 0; i < s.length() / 2; i++) {
            if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
