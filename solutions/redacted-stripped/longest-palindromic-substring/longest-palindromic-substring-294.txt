class Solution {

    public String method1(String s) {
        int n = s.length();
        if (n == 0)
            return "";
        if (n == 1)
            return s;
        int start = 0, maxLength = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= 1; j++) {
                int low = i;
                int high = i + j;
                while (low >= 0 && high < n && s.charAt(low) == s.charAt(high)) {
                    int currLen = high - low + 1;
                    if (currLen > maxLength) {
                        maxLength = currLen;
                        start = low;
                    }
                    low--;
                    high++;
                }
            }
        }
        return s.substring(start, start + maxLength);
    }
}
