class Solution {

    private boolean palindrome(String s) {
        int j = s.length() - 1;
        int i = 0;
        while (i < j) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }

    public String method1(String s) {
        if (s.length() <= 1) {
            return s;
        }
        String pali = s.substring(0, 1);
        int max = 1;
        for (int i = 0; i < s.length(); i++) {
            for (int j = i + max; j <= s.length(); j++) {
                if (j - i > max && palindrome(s.substring(i, j))) {
                    max = j - i;
                    pali = s.substring(i, j);
                }
            }
        }
        return pali;
    }
}
