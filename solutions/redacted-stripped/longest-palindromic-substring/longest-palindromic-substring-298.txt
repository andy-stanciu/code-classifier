class Solution {

    public static String method1(String s) {
        int maxLength = 0;
        String strLeString = null;
        int lengths = s.length();
        for (int i = 0; i < lengths; i++) {
            for (int j = lengths - 1; j >= i; j--) {
                if (s.charAt(i) == s.charAt(j)) {
                    String substr = s.substring(i, j + 1);
                    StringBuilder temp = new StringBuilder(substr).reverse();
                    if (substr.equals(temp.toString())) {
                        if (maxLength <= temp.length()) {
                            strLeString = temp.toString();
                            maxLength = temp.length();
                            break;
                        }
                    }
                }
            }
        }
        return strLeString;
    }
}
