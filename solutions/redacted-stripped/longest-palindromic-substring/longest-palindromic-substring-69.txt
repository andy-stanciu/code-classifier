class Solution {

    public String method1(String s) {
        String m = "";
        int var1 = 0;
        for (int i = 0; i < s.length(); i++) {
            int left = i;
            int right = i;
            while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
                if (right - left + 1 > var1) {
                    var1 = right - left + 1;
                    m = s.substring(left, right + 1);
                }
                left--;
                right++;
            }
            left = i;
            right = i + 1;
            while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
                if (right - left + 1 > var1) {
                    var1 = right - left + 1;
                    m = s.substring(left, right + 1);
                }
                left--;
                right++;
            }
        }
        return m;
    }
}
