class Solution {

    int var1;

    String var2;

    public String var2(String s) {
        var1 = 0;
        var2 = "";
        for (int i = 0; i < s.length(); ++i) {
            method1(s, i);
        }
        return var2;
    }

    private int method1(String s, int mid) {
        return Math.max(method2(s, mid - 1, mid + 1), method2(s, mid, mid + 1));
    }

    private int method2(String s, int l, int r) {
        while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
            --l;
            ++r;
        }
        int curr = (r - l) - 1;
        if (curr > var1) {
            var1 = curr;
            var2 = s.substring(l + 1, r);
        }
        return curr;
    }
}
