class Solution {

    public String method1(String S) {
        char[] s = S.toCharArray();
        int n = s.length;
        int left = 0, right = 0;
        for (int i = 0; i < n; i++) {
            int r = i, l = i;
            while (l - 1 >= 0 && r + 1 < n && s[l - 1] == s[r + 1]) {
                l--;
                r++;
            }
            if (r - l > right - left) {
                left = l;
                right = r;
            }
            if (i + 1 < n && s[i] == s[i + 1]) {
                l = i;
                r = i + 1;
                while (l - 1 >= 0 && r + 1 < n && s[l - 1] == s[r + 1]) {
                    l--;
                    r++;
                }
                if (r - l > right - left) {
                    left = l;
                    right = r;
                }
            }
        }
        return S.substring(left, right + 1);
    }
}
