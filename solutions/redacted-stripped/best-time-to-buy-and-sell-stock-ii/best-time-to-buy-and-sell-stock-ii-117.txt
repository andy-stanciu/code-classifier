class Solution {

    public int maxProfit(int[] prices) {
        int len = prices.length;
        int var1 = Integer.MAX_VALUE;
        int profit = 0;
        for (int i = 0; i < len; ++i) {
            int var2 = prices[i];
            int var3 = var2 - var1;
            if (var3 > 0) {
                profit += var3;
            }
            var1 = var2;
        }
        return profit;
    }
}
