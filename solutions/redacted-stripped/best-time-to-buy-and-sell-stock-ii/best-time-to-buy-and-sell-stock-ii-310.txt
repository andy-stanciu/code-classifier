class Solution {

    public int maxProfit(int[] prices) {
        int hold = -Integer.MAX_VALUE, notHold = 0;
        for (int var1 : prices) {
            int prevHold = hold, prevNotHold = notHold;
            hold = Math.max(prevHold, prevNotHold - var1);
            notHold = Math.max(prevNotHold, prevHold + var1);
        }
        return notHold;
    }
}
