class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int[] prev = new int[2];
        prev[0] = 0;
        prev[1] = 0;
        int profit = 0;
        for (int i = n - 1; i >= 0; i--) {
            int[] cur = new int[2];
            for (int j = 1; j >= 0; j--) {
                if (j == 0) {
                    profit = Math.max(0 + prev[0], -prices[i] + prev[1]);
                }
                if (j == 1) {
                    profit = Math.max(0 + prev[1], prices[i] + prev[0]);
                }
                cur[j] = profit;
            }
            prev = cur;
        }
        return prev[0];
    }
}
