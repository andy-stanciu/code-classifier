class Solution {

    public int maxProfit(int[] prices) {
        int bp = prices[0];
        int var1 = 0;
        int profit;
        for (int i = 1; i < prices.length; i++) {
            profit = prices[i] - bp;
            if (profit > 0) {
                var1 += profit;
                bp = prices[i];
            } else {
                bp = Math.min(bp, prices[i]);
            }
        }
        return var1;
    }
}
