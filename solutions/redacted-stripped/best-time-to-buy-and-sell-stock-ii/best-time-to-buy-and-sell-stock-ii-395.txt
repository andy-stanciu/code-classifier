class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int profit = 0;
        for (int a = 1; a < n; a++) {
            if (prices[a - 1] < prices[a]) {
                profit = profit + (prices[a] - prices[a - 1]);
            }
        }
        return profit;
    }
}
