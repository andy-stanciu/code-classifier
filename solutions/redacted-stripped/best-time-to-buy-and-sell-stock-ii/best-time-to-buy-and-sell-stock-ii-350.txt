class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            int var2 = prices[i];
            if (var2 >= var1) {
                maxProfit += var2 - var1;
                var1 = var2;
            } else {
                var1 = var2;
            }
        }
        return maxProfit;
    }
}
