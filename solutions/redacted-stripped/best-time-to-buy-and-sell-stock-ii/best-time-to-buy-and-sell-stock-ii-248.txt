class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        for (int price : prices) {
            if (var1 >= price) {
                var1 = price;
            } else {
                profit += (price - var1);
                var1 = price;
            }
        }
        return profit;
    }
}
