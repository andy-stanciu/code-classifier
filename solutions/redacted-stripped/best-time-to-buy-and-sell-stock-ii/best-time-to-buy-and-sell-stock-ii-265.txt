class Solution {

    public int maxProfit(int[] prices) {
        int maxprice = 0;
        int profit = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (profit < prices[i])
                maxprice += prices[i] - profit;
            profit = prices[i];
        }
        return maxprice;
    }
}
