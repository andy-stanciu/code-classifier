class Solution {

    public int maxProfit(int[] prices) {
        boolean flag = false;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i - 1] < prices[i]) {
                flag = true;
            }
        }
        if (!flag) {
            return 0;
        }
        flag = false;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i - 1] > prices[i]) {
                flag = true;
            }
        }
        if (!flag) {
            return prices[prices.length - 1] - prices[0];
        }
        int sum = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] - prices[i - 1] > 0) {
                sum += prices[i] - prices[i - 1];
            }
        }
        return sum;
    }
}
