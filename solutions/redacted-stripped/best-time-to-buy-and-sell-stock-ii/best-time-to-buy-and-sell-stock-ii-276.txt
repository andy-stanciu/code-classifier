class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < prices[i - 1]) {
                min = prices[i];
            } else if (prices[i] >= prices[i - 1]) {
                profit += prices[i] - min;
                min = prices[i];
            }
        }
        return profit;
    }
}
