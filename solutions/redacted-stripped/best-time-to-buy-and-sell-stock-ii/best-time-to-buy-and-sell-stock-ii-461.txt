class Solution {

    public int maxProfit(int[] prices) {
        int minprice = prices[0];
        int minprofit = 0;
        for (int price : prices) {
            if (price > minprice) {
                minprofit += price - minprice;
                minprice = price;
            } else if (price < minprice) {
                minprice = price;
            }
        }
        return minprofit;
    }
}
