class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length <= 1)
            return 0;
        int profit = 0, var1 = prices[0];
        for (int day = 1; day < prices.length; ++day) {
            int var2 = prices[day];
            if (var2 > var1) {
                profit += var2 - var1;
                var1 = var2;
            } else
                var1 = var2;
        }
        return profit;
    }
}
