class Solution {

    public int maxProfit(int[] prices) {
        int j, result = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            j = i + 1;
            if (prices[i] < prices[j]) {
                int prevResult = prices[j] - prices[i];
                result = result + prevResult;
            }
        }
        return result;
    }
}
