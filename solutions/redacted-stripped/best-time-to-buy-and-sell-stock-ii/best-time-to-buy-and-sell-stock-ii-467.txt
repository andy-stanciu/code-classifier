class Solution {

    public int maxProfit(int[] prices) {
        int var1 = (int) 1e9, profit = 0;
        for (int i : prices) {
            if (var1 >= i) {
                var1 = i;
            } else {
                profit += i - var1;
                var1 = i;
            }
        }
        return profit;
    }
}
