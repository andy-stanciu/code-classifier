class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        if (prices == null || n < 2)
            return 0;
        int var1 = 0;
        int minNo = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            if (prices[i] < minNo) {
                minNo = prices[i];
            } else {
                var1 += prices[i] - minNo;
                minNo = prices[i];
            }
        }
        return var1;
    }
}
