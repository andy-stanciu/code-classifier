class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        for (int i = 1; i < prices.length; i++) {
            int curr = prices[i];
            if (prices[i - 1] < curr) {
                var1 += curr - prices[i - 1];
            }
        }
        return var1;
    }
}
