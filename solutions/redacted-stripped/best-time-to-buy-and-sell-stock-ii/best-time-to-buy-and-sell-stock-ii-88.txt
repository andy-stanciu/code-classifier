class Solution {

    public int maxProfit(int[] arr) {
        int n = arr.length;
        int flag = 0;
        int var1 = 0;
        int profit = 0;
        for (int i = 0; i < n - 1; i++) {
            if (flag == 0) {
                var1 = arr[i];
                flag = 1;
            }
            if (arr[i] > arr[i + 1]) {
                profit = profit + (arr[i] - var1);
                flag = 0;
            }
        }
        if (flag == 1) {
            profit = profit + (arr[n - 1] - var1);
        }
        return profit;
    }
}
