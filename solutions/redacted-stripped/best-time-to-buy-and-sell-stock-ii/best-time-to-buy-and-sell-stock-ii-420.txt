class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int maxprf = 0;
        int current = prices[0];
        int len = prices.length;
        for (int i = 1; i < len; i++) {
            if (prices[i] > current) {
                maxprf = prices[i] - current;
                var1 += maxprf;
            }
            current = prices[i];
        }
        return var1;
    }
}
