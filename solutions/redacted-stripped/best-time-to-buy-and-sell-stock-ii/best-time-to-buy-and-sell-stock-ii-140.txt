class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int ans = 0;
        int l = 0;
        int temp = 0;
        for (int r = 1; r < n; r++) {
            int pc = prices[r];
            if (prices[l] < pc) {
                temp = prices[r] - prices[l];
                ans = ans + temp;
                temp = 0;
                l = r;
            } else {
                l++;
            }
        }
        return ans;
    }
}
