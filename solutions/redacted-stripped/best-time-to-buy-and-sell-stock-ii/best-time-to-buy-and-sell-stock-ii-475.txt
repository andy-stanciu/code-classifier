class Solution {

    public int maxProfit(int[] prices) {
        int x = 0;
        int sum = 0;
        while (x < prices.length - 1) {
            if (prices[x + 1] > prices[x]) {
                sum = sum + (prices[x + 1] - prices[x]);
            }
            x++;
        }
        return sum;
    }
}
