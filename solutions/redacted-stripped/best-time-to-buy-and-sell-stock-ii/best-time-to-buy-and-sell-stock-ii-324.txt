class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int minIdx = 0;
        int maxLocalProfit = 0;
        for (int i = 1; i < prices.length; i++) {
            int profit = prices[i] - prices[minIdx];
            if (profit < maxLocalProfit) {
                var1 += maxLocalProfit;
                minIdx = i;
                maxLocalProfit = 0;
            } else {
                maxLocalProfit = profit;
            }
        }
        return var1 + maxLocalProfit;
    }
}
