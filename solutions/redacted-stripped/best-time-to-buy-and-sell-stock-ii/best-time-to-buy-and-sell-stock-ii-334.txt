class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int var1 = 0;
        int var2 = 0;
        for (int i = n - 1; i >= 0; i--) {
            int var3 = 0;
            int var4 = 0;
            var3 = Math.max(-prices[i] + var2, var1);
            var4 = Math.max(prices[i] + var1, var2);
            var1 = var3;
            var2 = var4;
        }
        return var1;
    }
}
