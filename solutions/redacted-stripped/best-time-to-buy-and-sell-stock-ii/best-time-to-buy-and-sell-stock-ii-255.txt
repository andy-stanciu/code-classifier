class Solution {

    public int maxProfit(int[] prices) {
        int result = 0;
        int leftPointer = 0;
        int rightPointer = 1;
        while (rightPointer < prices.length) {
            int currentProfit = prices[rightPointer] - prices[leftPointer];
            if (currentProfit == 0) {
                rightPointer++;
            } else if (currentProfit > 0) {
                result = result + currentProfit;
                currentProfit = 0;
                leftPointer = rightPointer;
            } else {
                leftPointer = rightPointer;
                rightPointer++;
            }
        }
        return result;
    }
}
