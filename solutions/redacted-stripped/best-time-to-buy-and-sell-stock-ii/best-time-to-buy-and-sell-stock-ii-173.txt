class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        for (int var1 = 1; var1 < prices.length; var1++) {
            int previousDay = var1 - 1;
            if (prices[var1] > prices[previousDay]) {
                maxProfit += prices[var1] - prices[previousDay];
            }
        }
        return maxProfit;
    }
}
