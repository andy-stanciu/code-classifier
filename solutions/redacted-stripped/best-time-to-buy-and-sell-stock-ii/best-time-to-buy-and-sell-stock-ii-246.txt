class Solution {

    public int maxProfit(int[] prices) {
        int number = prices[0];
        int profit = 0;
        int x = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < number) {
                profit += x;
                number = prices[i];
                x = 0;
            } else if (prices[i] - number > x) {
                x = prices[i] - number;
            }
            if (prices[i] - number < x) {
                profit += x;
                x = 0;
                number = prices[i];
            }
        }
        profit += x;
        return profit;
    }
}
