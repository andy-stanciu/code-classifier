class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int maxprofit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            } else {
                int profit = prices[i] - var1;
                maxprofit = Math.max(profit + maxprofit, maxprofit);
                var1 = prices[i];
            }
        }
        return maxprofit;
    }
}
