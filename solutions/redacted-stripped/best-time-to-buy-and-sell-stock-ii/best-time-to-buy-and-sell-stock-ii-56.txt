class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        int n = prices.length;
        for (int i = 1; i < n; i++) {
            if (0 < (prices[i] - prices[i - 1])) {
                profit += (prices[i] - prices[i - 1]);
            }
        }
        return profit;
    }
}
