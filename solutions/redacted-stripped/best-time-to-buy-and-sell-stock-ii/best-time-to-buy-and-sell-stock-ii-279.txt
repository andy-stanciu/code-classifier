class Solution {

    public int maxProfit(int[] prices) {
        int len = prices.length;
        int min = prices[0];
        int var1 = 0;
        for (int i = 0; i < len; i++) {
            min = Math.min(min, prices[i]);
            int profit = prices[i] - min;
            if (profit > 0) {
                var1 += profit;
                min = prices[i];
            }
        }
        return var1;
    }
}
