class Solution {

    public int maxProfit(int[] prices) {
        int leastValue = Integer.MAX_VALUE;
        int maxVal = 0;
        int temp = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < leastValue) {
                leastValue = prices[i];
            }
            temp = prices[i] - leastValue;
            if (temp > 0) {
                maxVal += temp;
                leastValue = prices[i];
            }
        }
        return maxVal;
    }
}
