class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = 0;
        int diff = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            var1 = prices[i];
            if (prices[i] < prices[i + 1]) {
                diff = prices[i + 1] - var1;
                profit += diff;
            }
        }
        return profit;
    }
}
