class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int ans = 0;
        int i = 0;
        while (i < n) {
            int j = i + 1;
            while (j < n && prices[j] > prices[i]) {
                ans += prices[j] - prices[i];
                i = j;
                j++;
            }
            i = j;
        }
        return ans;
    }
}
