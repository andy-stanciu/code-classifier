class Solution {

    public int maxProfit(int[] prices) {
        int currPrice = prices[0];
        int var1 = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < currPrice) {
                currPrice = prices[i];
            } else if (prices[i] > currPrice) {
                var1 += (prices[i] - currPrice);
                currPrice = prices[i];
            }
        }
        return var1;
    }
}
