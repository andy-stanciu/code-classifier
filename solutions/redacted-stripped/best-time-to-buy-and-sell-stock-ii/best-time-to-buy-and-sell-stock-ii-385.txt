class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int answer = 0;
        int[] profits = new int[prices.length];
        for (int i = 0; i < prices.length - 1; i++) {
            if (prices[i + 1] > prices[i]) {
                profit = prices[i + 1] - prices[i];
                profits[i] = profit;
            }
        }
        for (int j = 0; j < profits.length; j++) {
            answer += profits[j];
        }
        return answer;
    }
}
