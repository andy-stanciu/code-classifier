class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        boolean var1 = false;
        int cp = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            if (!var1 && prices[i + 1] > prices[i]) {
                cp = prices[i];
                var1 = true;
            }
            if (var1 && prices[i] > prices[i + 1]) {
                profit += prices[i] - cp;
                var1 = false;
            }
        }
        if (var1)
            profit += prices[prices.length - 1] - cp;
        return profit;
    }
}
