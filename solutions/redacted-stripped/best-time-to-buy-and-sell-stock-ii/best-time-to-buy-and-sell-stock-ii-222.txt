class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < prices[i - 1]) {
                var1 = prices[i];
            }
            if (prices[i] > var1) {
                profit += prices[i] - var1;
                var1 = prices[i];
            }
        }
        return profit;
    }
}
