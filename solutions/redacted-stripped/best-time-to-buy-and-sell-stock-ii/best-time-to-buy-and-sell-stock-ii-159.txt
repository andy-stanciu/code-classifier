class Solution {

    public int maxProfit(int[] prices) {
        int left = 0;
        int profitL = 0;
        int profitR = 0;
        int max = 0;
        for (int right = 1; right < prices.length; right++) {
            if (prices[left] < prices[right]) {
                profitL += prices[right] - prices[left];
                System.out.println(profitL);
            }
            left++;
        }
        return max = Math.max(max, profitL);
    }
}
