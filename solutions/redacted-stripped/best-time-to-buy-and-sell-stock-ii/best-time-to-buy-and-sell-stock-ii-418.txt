class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int var2 = Integer.MIN_VALUE;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < var1 && i != prices.length - 1) {
                if (var2 > var1 && var2 != Integer.MIN_VALUE) {
                    profit += var2 - var1;
                }
                var1 = prices[i];
                var2 = Integer.MIN_VALUE;
            } else {
                if (var2 < prices[i]) {
                    var2 = prices[i];
                } else {
                    profit += var2 - var1;
                    var1 = prices[i];
                    var2 = Integer.MIN_VALUE;
                }
            }
        }
        if (var2 > var1) {
            profit += var2 - var1;
        }
        return Math.max(profit, 0);
    }
}
