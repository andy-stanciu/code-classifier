class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else if (var1 < prices[i]) {
                profit = profit + (prices[i] - var1);
                var1 = prices[i];
            }
        }
        return profit;
    }
}
