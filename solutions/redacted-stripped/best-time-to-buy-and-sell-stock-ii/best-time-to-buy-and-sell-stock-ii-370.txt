class Solution {

    public int maxProfit(int[] prices) {
        int L = 0;
        int profit = 0;
        for (int R = 1; R < prices.length; R++) {
            if (prices[L] > prices[R]) {
                L++;
                continue;
            } else {
                profit += prices[R] - prices[L];
                L++;
            }
        }
        return profit;
    }
}
