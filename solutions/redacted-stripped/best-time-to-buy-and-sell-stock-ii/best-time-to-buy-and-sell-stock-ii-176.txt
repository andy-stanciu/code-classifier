class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int max = 0;
        int i = 0;
        while (i < n) {
            int min = prices[i];
            while (i + 1 < n && prices[i + 1] >= prices[i]) {
                i++;
            }
            max += (prices[i] - min);
            i++;
        }
        return max;
    }
}
