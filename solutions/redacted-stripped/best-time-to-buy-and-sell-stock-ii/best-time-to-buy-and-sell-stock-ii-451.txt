class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length < 2)
            return 0;
        int n = prices.length;
        int[] dp = new int[n];
        dp[n - 1] = 0;
        for (int i = n - 2; i >= 0; i--) {
            int profit = prices[i + 1] - prices[i];
            dp[i] = profit > 0 ? profit + dp[i + 1] : dp[i + 1];
        }
        return dp[0];
    }
}
