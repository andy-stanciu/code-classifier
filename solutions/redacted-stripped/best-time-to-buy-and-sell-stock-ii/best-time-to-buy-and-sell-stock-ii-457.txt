class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length, var1 = prices[0], profit = 0;
        for (int i = 1; i < n; i++) {
            if (prices[i] > var1) {
                profit = profit + prices[i] - var1;
                var1 = prices[i];
            } else {
                var1 = prices[i];
            }
        }
        return profit;
    }
}
