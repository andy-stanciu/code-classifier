class Solution {

    public int maxProfit(int[] nums) {
        int min = nums[0], var1 = 0;
        int pos = 1, max = -1;
        while (pos < nums.length) {
            if (nums[pos] < min)
                min = nums[pos];
            while (pos < nums.length && max < nums[pos]) max = nums[pos++];
            var1 += max - min;
            max = -1;
            if (pos < nums.length)
                min = nums[pos];
            pos++;
        }
        return var1;
    }
}
