class Solution {

    public int maxProfit(int[] prices) {
        int left = 0, right = 1;
        int maxProfit = Integer.MIN_VALUE;
        int outVal = 0;
        while (left < right && right < prices.length) {
            if (prices[left] > prices[right]) {
                left = right;
                right++;
            } else {
                int tempProfit = prices[right] - prices[left];
                outVal = outVal + tempProfit;
                left++;
                right++;
            }
        }
        return outVal;
    }
}
