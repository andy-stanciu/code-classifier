class Solution {

    public int maxProfit(int[] prices) {
        int max = 0;
        int var1 = 0, var2 = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < prices[var1]) {
                var1 = i;
                var2 = i;
            } else if (prices[i] > prices[var1]) {
                max = max + prices[i] - prices[var1];
                var1 = i;
                var2 = i;
            }
        }
        return max;
    }
}
