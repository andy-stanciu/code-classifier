class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0, var2 = 1;
        int var3 = 0;
        while (var2 < prices.length) {
            if (prices[var1] < prices[var2]) {
                var3 += prices[var2] - prices[var1];
                var1++;
            } else {
                var1 = var2;
            }
            var2++;
        }
        return var3;
    }
}
