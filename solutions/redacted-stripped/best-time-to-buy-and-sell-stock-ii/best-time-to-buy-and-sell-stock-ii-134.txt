class Solution {

    public int maxProfit(int[] prices) {
        int currentProfit = 0;
        int maxProfit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            currentProfit = Math.max(0, prices[i + 1] - prices[i]);
            maxProfit += currentProfit;
        }
        return maxProfit;
    }
}
