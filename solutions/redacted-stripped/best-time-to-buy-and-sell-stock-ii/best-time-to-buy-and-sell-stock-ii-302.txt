class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int i = 1;
        int var1 = prices[0];
        while (i < prices.length) {
            while (i < prices.length && prices[i] <= prices[i - 1]) {
                var1 = prices[i];
                i++;
            }
            int var2 = 0;
            while (i < prices.length && prices[i] > prices[i - 1]) {
                var2 = prices[i];
                i++;
            }
            if (var2 > 0)
                profit += var2 - var1;
        }
        return profit;
    }
}
