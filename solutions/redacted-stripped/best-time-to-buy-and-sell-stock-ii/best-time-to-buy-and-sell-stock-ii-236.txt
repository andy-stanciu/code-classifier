class Solution {

    public int maxProfit(int[] prices) {
        int[] dp = new int[prices.length];
        int var1 = prices[0], var2 = prices[0];
        dp[0] = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < var2) {
                int profit = var2 - var1;
                dp[i] = profit + dp[i - 1];
                var1 = prices[i];
                var2 = prices[i];
            } else if (prices[i] >= var2) {
                var2 = prices[i];
                dp[i] = dp[i - 1];
            }
        }
        return dp[dp.length - 1] + (var2 - var1);
    }
}
