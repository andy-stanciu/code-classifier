class Solution {

    public int maxProfit(int[] prices) {
        int j = 0, profit = 0;
        for (int i = 1; i < prices.length; i++) {
            prices[j++] = prices[i] - prices[i - 1];
        }
        for (int i = 0; i < j; i++) {
            if (prices[i] > 0) {
                profit += prices[i];
            }
        }
        return profit;
    }
}
