class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int var1 = 0;
        int var2 = -prices[0];
        for (int i = 1; i < n; i++) {
            var1 = Math.max(var1, var2 + prices[i]);
            var2 = Math.max(var2, var1 - prices[i]);
        }
        return var1;
    }
}
