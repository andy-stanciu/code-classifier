class Solution {

    public int maxProfit(int[] arr) {
        int n = arr.length;
        int min = arr[0];
        int profit = 0;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (arr[i] < min) {
                min = arr[i];
            }
            if (arr[i] - min > profit) {
                profit = arr[i] - min;
                ans = ans + profit;
                profit = 0;
                min = arr[i];
            }
        }
        return ans;
    }
}
