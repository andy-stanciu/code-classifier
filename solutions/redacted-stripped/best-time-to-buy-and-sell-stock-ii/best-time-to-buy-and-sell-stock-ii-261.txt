class Solution {

    public int maxProfit(int[] prices) {
        int pro = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            int curr_pro = prices[i + 1] - prices[i];
            if (curr_pro > 0) {
                pro = curr_pro + pro;
            }
        }
        return pro;
    }
}
