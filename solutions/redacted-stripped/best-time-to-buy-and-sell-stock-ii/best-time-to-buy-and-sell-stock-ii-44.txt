class Solution {

    public int maxProfit(int[] prices) {
        int bought = prices[0], var1 = prices[0];
        int n = prices.length;
        int profit = 0;
        for (int i = 0; i < n; i++) {
            if (prices[i] > var1) {
                var1 = prices[i];
                if (i == n - 1)
                    profit += var1 - bought;
            } else if (prices[i] < var1 || i == n - 1) {
                profit += var1 - bought;
                bought = prices[i];
                var1 = prices[i];
            }
            ;
        }
        return profit;
    }
}
