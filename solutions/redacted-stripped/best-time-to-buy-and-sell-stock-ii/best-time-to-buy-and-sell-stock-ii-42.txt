class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int j = 0, i = 0;
        int sum = 0;
        for (i = 0; i < n - 1; i++) {
            if (prices[i + 1] < prices[i]) {
                sum += prices[i] - prices[j];
                j = i + 1;
            }
        }
        if (prices[i] > prices[j])
            sum += prices[i] - prices[j];
        return sum;
    }
}
