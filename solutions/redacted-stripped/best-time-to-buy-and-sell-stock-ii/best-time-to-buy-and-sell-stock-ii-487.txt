class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int var2 = 1;
        int sum = 0;
        while (var2 < prices.length) {
            if (prices[var2] > prices[var1])
                sum += prices[var2++] - prices[var1++];
            else
                var1 = var2++;
        }
        return sum;
    }
}
