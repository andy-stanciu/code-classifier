class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int profit = 0;
        for (int j = 0; j < n - 1; j++) {
            int BP = prices[j];
            if (prices[j + 1] > BP) {
                profit += (prices[j + 1] - BP);
            }
        }
        return profit;
    }
}
