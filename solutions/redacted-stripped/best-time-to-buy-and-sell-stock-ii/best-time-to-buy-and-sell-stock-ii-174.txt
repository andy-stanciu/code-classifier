class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = prices[0];
        for (int i = 0; i < prices.length - 1; i++) {
            if (prices[i + 1] > prices[i] && i < prices.length - 1) {
                maxProfit += prices[i + 1] - prices[i];
            } else {
                continue;
            }
        }
        return maxProfit;
    }
}
