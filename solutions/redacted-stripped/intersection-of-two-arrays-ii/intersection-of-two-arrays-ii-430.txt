class Solution {

    public void feed(List<Integer> ans, int[] nums1, int[] nums2, int n, int m) {
        int[] st = new int[1001];
        for (int i = 0; i < n; i++) st[nums1[i]]++;
        for (int i = 0; i < m; i++) if (st[nums2[i]] > 0) {
            ans.add(nums2[i]);
            st[nums2[i]]--;
        }
    }

    public int[] intersect(int[] nums1, int[] nums2) {
        int n = nums1.length;
        int m = nums2.length;
        List<Integer> ans = new ArrayList<>();
        if (n > m)
            feed(ans, nums1, nums2, n, m);
        else
            feed(ans, nums2, nums1, m, n);
        int[] intArray = new int[ans.size()];
        for (int i = 0; i < ans.size(); i++) {
            intArray[i] = ans.get(i);
        }
        return intArray;
    }
}
