class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        ArrayList<Integer> sol = new ArrayList<>();
        int i = 0, j = 0;
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] < nums2[j]) {
                i++;
            } else if (nums1[i] > nums2[j]) {
                j++;
            } else {
                sol.add(nums1[i]);
                i++;
                j++;
            }
        }
        int[] result = new int[sol.size()];
        for (i = 0; i < sol.size(); i++) {
            result[i] = sol.get(i);
        }
        return result;
    }
}
