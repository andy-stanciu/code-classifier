class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] freq = new int[1001];
        for (int n : nums1) {
            freq[n]++;
        }
        ArrayList<Integer> res = new ArrayList<>();
        for (int n : nums2) {
            if (freq[n] > 0) {
                res.add(n);
                freq[n]--;
            }
        }
        int[] ans = new int[res.size()];
        for (int i = 0; i < res.size(); i++) {
            ans[i] = res.get(i);
        }
        return ans;
    }
}
