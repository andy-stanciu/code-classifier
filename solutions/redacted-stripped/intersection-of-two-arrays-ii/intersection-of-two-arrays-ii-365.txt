class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        ArrayList<Integer> ai = new ArrayList<>();
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = 0, j = 0;
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] < nums2[j]) {
                i++;
            } else if (nums1[i] > nums2[j]) {
                j++;
            } else {
                ai.add(nums1[i]);
                i++;
                j++;
            }
        }
        int[] arr = new int[ai.size()];
        for (int k = 0; k < ai.size(); k++) {
            arr[k] = ai.get(k);
        }
        return arr;
    }
}
