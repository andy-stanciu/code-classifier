class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        List<Integer> map = new ArrayList<>();
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = 0, j = 0;
        while (i < nums1.length && j < nums2.length) {
            if (nums1[i] == nums2[j]) {
                map.add(nums1[i]);
                i++;
                j++;
            } else if (nums1[i] < nums2[j]) {
                i++;
            } else {
                j++;
            }
        }
        int[] output = new int[map.size()];
        for (int k = 0; k < map.size(); k++) {
            output[k] = map.get(k);
        }
        return output;
    }
}
