public class Solution {

    public int[] intersect(int[] a, int[] b) {
        Arrays.sort(a);
        Arrays.sort(b);
        int l = 0;
        int k = 0;
        List<Integer> temp = new ArrayList<>();
        while (l < a.length && k < b.length) {
            if (a[l] == b[k]) {
                temp.add(a[l]);
                l++;
                k++;
            } else if (a[l] < b[k]) {
                l++;
            } else {
                k++;
            }
        }
        int[] tempArray = new int[temp.size()];
        for (int c = 0; c < temp.size(); c++) {
            tempArray[c] = temp.get(c);
        }
        return tempArray;
    }
}
