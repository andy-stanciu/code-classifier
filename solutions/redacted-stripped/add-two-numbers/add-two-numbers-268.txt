class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode t1 = l1;
        ListNode t2 = l2;
        ListNode dummy = new ListNode(-1);
        ListNode curr = dummy;
        int carry = 0;
        int sum = 0;
        while (t1 != null || t2 != null) {
            sum = carry;
            if (t1 != null) {
                sum = sum + t1.val;
            }
            if (t2 != null) {
                sum = sum + t2.val;
            }
            ListNode sum1 = new ListNode(sum % 10);
            carry = sum / 10;
            curr.next = sum1;
            curr = curr.next;
            if (t1 != null)
                t1 = t1.next;
            if (t2 != null)
                t2 = t2.next;
        }
        if (carry > 0) {
            ListNode carry1 = new ListNode(carry);
            curr.next = carry1;
        }
        return dummy.next;
    }
}
