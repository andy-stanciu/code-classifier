class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode dHead = new ListNode(0);
        ListNode current = dHead;
        int count = 0;
        int x = 0;
        int y = 0;
        while (l1 != null || l2 != null || count != 0) {
            if (l1 != null) {
                x = l1.val;
            } else {
                x = 0;
            }
            if (l2 != null) {
                y = l2.val;
            } else {
                y = 0;
            }
            int sum = count + x + y;
            count = sum / 10;
            current.next = new ListNode(sum % 10);
            current = current.next;
            if (l1 != null)
                l1 = l1.next;
            if (l2 != null)
                l2 = l2.next;
        }
        return dHead.next;
    }
}
