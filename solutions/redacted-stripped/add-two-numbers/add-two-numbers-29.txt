class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        int carry = 0, total = 0;
        ListNode dummy = new ListNode();
        ListNode res = dummy;
        while (l1 != null || l2 != null || carry != 0) {
            total = carry;
            if (l1 != null) {
                total += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                total += l2.val;
                l2 = l2.next;
            }
            int un = total % 10;
            carry = total / 10;
            dummy.next = new ListNode(un);
            dummy = dummy.next;
        }
        return res.next;
    }
}
