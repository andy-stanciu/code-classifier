class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        if (l1 == null)
            return l2;
        if (l2 == null)
            return l1;
        ListNode dummyNode = new ListNode(-1);
        ListNode currentNode = dummyNode;
        ListNode temp1 = l1;
        ListNode temp2 = l2;
        int carry = 0;
        while (temp1 != null || temp2 != null) {
            int sum = carry;
            if (temp1 != null)
                sum += temp1.val;
            if (temp2 != null)
                sum += temp2.val;
            ListNode ansNode = new ListNode(sum % 10);
            carry = sum / 10;
            currentNode.next = ansNode;
            currentNode = currentNode.next;
            if (temp1 != null)
                temp1 = temp1.next;
            if (temp2 != null)
                temp2 = temp2.next;
        }
        if (carry > 0) {
            currentNode.next = new ListNode(carry);
        }
        return dummyNode.next;
    }
}
