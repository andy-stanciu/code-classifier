class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode res = new ListNode();
        ListNode temp = res;
        int remainder = 0;
        while (l1 != null && l2 != null) {
            int num = l1.val + l2.val + remainder;
            temp.next = new ListNode(num % 10);
            remainder = num / 10;
            temp = temp.next;
            l1 = l1.next;
            l2 = l2.next;
        }
        while (l1 != null) {
            int num = l1.val + remainder;
            temp.next = new ListNode(num % 10);
            remainder = num / 10;
            temp = temp.next;
            l1 = l1.next;
        }
        while (l2 != null) {
            int num = l2.val + remainder;
            temp.next = new ListNode(num % 10);
            remainder = num / 10;
            temp = temp.next;
            l2 = l2.next;
        }
        if (remainder > 0) {
            temp.next = new ListNode(remainder);
        }
        return res.next;
    }
}
