class Solution {

    public ListNode method1(ListNode head1, ListNode head2) {
        int carry = 0;
        ListNode ans = new ListNode();
        ListNode current = ans;
        while (head1 != null || head2 != null || carry != 0) {
            int sum = carry;
            if (head1 != null) {
                sum += head1.val;
                head1 = head1.next;
            }
            if (head2 != null) {
                sum += head2.val;
                head2 = head2.next;
            }
            int digit = sum % 10;
            carry = sum / 10;
            current.next = new ListNode(digit);
            current = current.next;
        }
        return ans.next;
    }
}
