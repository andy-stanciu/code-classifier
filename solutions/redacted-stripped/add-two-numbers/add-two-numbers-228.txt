class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode head = new ListNode(-1);
        ListNode trav = head;
        int remain = 0;
        while (l1 != null || l2 != null) {
            int l1Val = 0;
            if (l1 != null) {
                l1Val = l1.val;
                l1 = l1.next;
            }
            int l2Val = 0;
            if (l2 != null) {
                l2Val = l2.val;
                l2 = l2.next;
            }
            int sum = l1Val + l2Val + remain;
            if (sum > 9) {
                remain = 1;
                int rem = sum % 10;
                trav.next = new ListNode(rem);
            } else {
                remain = 0;
                trav.next = new ListNode(sum);
            }
            trav = trav.next;
        }
        if (remain > 0) {
            trav.next = new ListNode(remain);
        }
        return head.next;
    }
}
