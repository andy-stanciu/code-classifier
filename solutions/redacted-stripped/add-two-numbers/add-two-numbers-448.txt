class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        BigDecimal number1 = toNumFromNode(l1);
        BigDecimal number2 = toNumFromNode(l2);
        return numToNode(number1.add(number2));
    }

    public BigDecimal toNumFromNode(ListNode node) {
        StringBuilder sb = new StringBuilder();
        while (node != null) {
            sb.append(node.val);
            node = node.next;
        }
        return new BigDecimal(sb.reverse().toString());
    }

    public ListNode numToNode(BigDecimal num) {
        ListNode ln = null;
        String[] splittedNum = String.valueOf(num).split("");
        for (int i = 0; i < splittedNum.length; i++) {
            ln = new ListNode(Integer.parseInt(splittedNum[i]), ln);
        }
        return ln;
    }
}
