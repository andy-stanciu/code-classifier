class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        int sum = 0;
        int var1 = 0;
        ListNode first = l1;
        ListNode second = l2;
        ListNode sumNode = new ListNode(0);
        ListNode finalNode = sumNode;
        while (first != null || second != null || var1 != 0) {
            int firstValue = first != null ? first.val : 0;
            int secondValue = second != null ? second.val : 0;
            sum = firstValue + secondValue + var1;
            var1 = 0;
            if (sum >= 10) {
                sum = sum % 10;
                var1 = 1;
            }
            ListNode newNode = new ListNode(sum);
            finalNode.next = newNode;
            finalNode = finalNode.next;
            first = first != null ? first.next : null;
            second = second != null ? second.next : null;
        }
        return sumNode.next;
    }
}
