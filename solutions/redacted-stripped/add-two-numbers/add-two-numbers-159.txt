class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        return method2(l1, l2, 0);
    }

    public ListNode method2(ListNode i, ListNode j, int carry) {
        if (i == null && j == null) {
            if (carry == 0)
                return null;
            else
                return new ListNode(carry);
        }
        int iValue = i != null ? i.val : 0;
        int jValue = j != null ? j.val : 0;
        int sum = iValue + jValue + carry;
        ListNode res = new ListNode(sum % 10);
        ListNode iNext = i != null ? i.next : null;
        ListNode jNext = j != null ? j.next : null;
        res.next = method2(iNext, jNext, sum / 10);
        return res;
    }
}
