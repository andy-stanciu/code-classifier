class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int[] var1 = new int[1];
        helper(root, var1);
        return var1[0];
    }

    private static void helper(TreeNode root, int[] var1) {
        if (root == null)
            return;
        var1[0] += 1;
        helper(root.left, var1);
        helper(root.right, var1);
    }
}
