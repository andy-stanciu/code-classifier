/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int getleft(TreeNode root) {
        //if(root==null) return 0;
        int var1 = 0;
        while (root != null) {
            var1++;
            root = root.left;
        }
        return var1;
    }

    public int getright(TreeNode root) {
        //  if(root==null) return 0;
        int var1 = 0;
        while (root != null) {
            var1++;
            root = root.right;
        }
        return var1;
    }

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        // int total=0;
        int lh = getleft(root.left);
        int rh = getright(root.right);
        System.out.println(lh + "" + rh);
        if (lh == rh)
            return (2 << lh) - 1;
        return method1(root.left) + method1(root.right) + 1;
    }
}
