/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int cnt = 0;

    public int method1(TreeNode root) {
        if (root == null)
            //here whatever you may return it works
            return 0;
        cnt += 1;
        method1(root.left);
        method1(root.right);
        return cnt;
    }
}
