/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int cnt = 0;

    public int method1(TreeNode root) {
        //1.base conditon - root equal numm return 0
        if (root == null) {
            return 0;
        }
        //2.apprach -find left and right value
        int left = method1(root.left);
        int right = method1(root.right);
        //3.return - node value with +1 because we need to add current node
        return left + right + 1;
    }
}
