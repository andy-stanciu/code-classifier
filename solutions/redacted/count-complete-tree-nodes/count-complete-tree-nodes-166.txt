/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = ht_l(root);
        int right = ht_r(root);
        if (left == right)
            return (int) Math.pow(2, left) - 1;
        else
            return 1 + method1(root.left) + method1(root.right);
    }

    public int ht_l(TreeNode root) {
        int c = 0;
        while (root != null) {
            root = root.left;
            c++;
        }
        return c;
    }

    public int ht_r(TreeNode root) {
        int c = 0;
        while (root != null) {
            root = root.right;
            c++;
        }
        return c;
    }
}
