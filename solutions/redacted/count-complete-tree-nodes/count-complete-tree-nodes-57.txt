class Solution {

    public int method1(TreeNode root) {
        int lh = 0, rh = 0;
        TreeNode curr = root;
        while (curr != null) {
            lh++;
            curr = curr.left;
        }
        curr = root;
        while (curr != null) {
            rh++;
            curr = curr.right;
        }
        if (lh == rh)
            return (int) (Math.pow(2, lh) - 1);
        return 1 + method1(root.left) + method1(root.right);
    }
}
