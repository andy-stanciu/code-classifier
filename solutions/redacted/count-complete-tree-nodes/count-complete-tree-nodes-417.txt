/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left_cnt = 0, right_cnt = 0;
        TreeNode node = root;
        while (node != null) {
            left_cnt++;
            node = node.left;
        }
        node = root;
        while (node != null) {
            right_cnt++;
            node = node.right;
        }
        if (left_cnt == right_cnt) {
            return (int) Math.pow(2, left_cnt) - 1;
        }
        return 1 + method1(root.left) + method1(root.right);
    }
}
