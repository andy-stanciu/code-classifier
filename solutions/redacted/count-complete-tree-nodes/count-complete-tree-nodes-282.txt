/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static int findHeightL(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + findHeightL(root.left);
    }

    public static int findHeightR(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + findHeightR(root.right);
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int l = findHeightL(root);
        int r = findHeightR(root);
        if (l == r) {
            return (int) (Math.pow(2, l) - 1);
        }
        return 1 + method1(root.left) + method1(root.right);
    }
}
