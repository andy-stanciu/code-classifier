/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        TreeNode var1 = root.left;
        TreeNode var2 = root.right;
        int leftHeight = getHeight(var1);
        int rightHeight = getHeight(var2);
        if (leftHeight == rightHeight) {
            // left sub tree is complete
            return (int) Math.pow(2, leftHeight) + method1(root.right);
        } else {
            // leftHeight > rightHeight => left sub tree is not complete
            return (int) Math.pow(2, rightHeight) + method1(root.left);
        }
    }

    public int getHeight(TreeNode root) {
        if (root == null)
            return 0;
        return getHeight(root.left) + 1;
    }
}
