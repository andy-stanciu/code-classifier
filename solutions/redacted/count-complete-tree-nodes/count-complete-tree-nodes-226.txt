class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = 1;
        if (root.left != null) {
            var1 += method1(root.left);
        }
        if (root.right != null) {
            var1 += method1(root.right);
        }
        return var1;
    }
}
