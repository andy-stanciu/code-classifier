/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        PreOrder(root, var1);
        return var1[0];
    }

    public static void PreOrder(TreeNode root, int[] var1) {
        if (root == null) {
            return;
        }
        var1[0] += 1;
        PreOrder(root.left, var1);
        PreOrder(root.right, var1);
    }
}
