/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        if (root == null) {
            return 0;
        } else {
            var1++;
        }
        int var2 = method1(root.left);
        int var3 = method1(root.right);
        var1 = var1 + var2 + var3;
        return var1;
    }
}
