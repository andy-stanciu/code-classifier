/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = 0;

    void preorder(TreeNode root) {
        if (root == null)
            return;
        ans++;
        preorder(root.left);
        preorder(root.right);
    }

    public int method1(TreeNode root) {
        preorder(root);
        return ans;
    }
}
