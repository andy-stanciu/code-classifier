/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        preorder(root);
        return var1;
    }

    void preorder(TreeNode root) {
        if (root == null) {
            return;
        }
        var1++;
        preorder(root.left);
        preorder(root.right);
    }
}
