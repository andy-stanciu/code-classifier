/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int leftHeight = method2(root.left);
        int rightHeight = method3(root.right);
        if (leftHeight == rightHeight)
            return ((2 << (leftHeight)) - 1);
        return 1 + method1(root.left) + method1(root.right);
    }

    int method2(TreeNode leftHead) {
        int var1 = 0;
        while (leftHead != null) {
            var1++;
            leftHead = leftHead.left;
        }
        return var1;
    }

    int method3(TreeNode rightHead) {
        int var1 = 0;
        while (rightHead != null) {
            var1++;
            rightHead = rightHead.right;
        }
        return var1;
    }
}
