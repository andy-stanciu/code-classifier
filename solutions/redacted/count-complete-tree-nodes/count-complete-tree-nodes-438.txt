/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void inOrder(TreeNode root, int[] a) {
        if (root == null) {
            return;
        }
        a[0]++;
        inOrder(root.left, a);
        inOrder(root.right, a);
    }

    public int method1(TreeNode root) {
        int[] a = new int[1];
        inOrder(root, a);
        return a[0];
    }
}
