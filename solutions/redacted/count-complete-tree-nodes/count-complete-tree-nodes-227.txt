/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int var1 = 0;

    public int method1(TreeNode root) {
        //check if root is null
        if (root == null)
            return 0;
        //get the height of all left nodes
        int lh = method2(root);
        //get height of all right nodes
        int rh = method3(root);
        //if leftheight ==
        return lh == rh ? (1 << rh) - 1 : 1 + method1(root.left) + method1(root.right);
        //rightheight then return 2^lh-1 else return no. of nodes on left + no. of nodes on right + 1(root node)
    }

    public int method2(TreeNode root) {
        int lh = 0;
        while (root != null) {
            root = root.left;
            lh++;
        }
        return lh;
    }

    public int method3(TreeNode root) {
        int rh = 0;
        while (root != null) {
            root = root.right;
            rh++;
        }
        return rh;
    }
}
