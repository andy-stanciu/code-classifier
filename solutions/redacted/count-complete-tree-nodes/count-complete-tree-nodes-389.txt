class Solution {

    public int method1(TreeNode root) {
        // If the root is null, the tree is empty, so return 0.
        if (root == null) {
            return 0;
        }
        // If the root is not null, recursively count the number of nodes
        // in the left subtree and the right subtree, and add 1 for the root.
        // Then return the total count.
        return 1 + method1(root.left) + method1(root.right);
    }
}
