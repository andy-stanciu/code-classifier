/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public void dfs(TreeNode root) {
        if (root == null)
            return;
        var1++;
        dfs(root.left);
        dfs(root.right);
    }

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        dfs(root);
        return var1;
    }
}
