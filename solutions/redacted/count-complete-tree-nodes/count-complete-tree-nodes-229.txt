/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        return method2(root, var1);
    }

    public int method2(TreeNode root, int var1) {
        if (root == null)
            return var1;
        return 1 + method2(root.left, var1) + method2(root.right, var1);
    }
}
