/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        inorder(root);
        return var1;
    }

    int var1 = 0;

    public void inorder(TreeNode root) {
        if (root == null)
            return;
        inorder(root.left);
        var1++;
        inorder(root.right);
    }
}
