/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode curNode, int param1) {
        if (curNode == null) {
            return param1;
        }
        param1++;
        param1 = method1(curNode.left, param1);
        param1 = method1(curNode.right, param1);
        return param1;
    }

    public int method1(TreeNode root) {
        return method1(root, 0);
    }
}
