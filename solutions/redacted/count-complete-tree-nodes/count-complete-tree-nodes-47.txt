/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //global declaration for counting the nodes
    public int var1 = 0;

    public int method1(TreeNode root) {
        //if root is null /there are no nodes
        if (root == null)
            return 0;
        //used inorder for counting
        method1(root.right);
        //incrementing the value
        var1++;
        method1(root.left);
        return var1;
    }
}
