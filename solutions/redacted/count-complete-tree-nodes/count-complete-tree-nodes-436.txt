/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        //base case if root is null so return 0
        if (root == null)
            return 0;
        //call left tree and count nodes
        int var1 = method1(root.left);
        //call right tree and count nodes
        int var2 = method1(root.right);
        //final left and right nodes count and include root node by 1
        int var3 = var1 + var2 + 1;
        //return final result or totals nodes
        return var3;
    }
}
