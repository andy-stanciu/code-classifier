/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    /**
     * 20240324
     * try -2
     * after read solution
     *
     * @param root
     * @return
     */
    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftHeight = method2(root);
        if (leftHeight == method3(root)) {
            // Complete binary tree. A k level complete binary tree has (2^k - 1) nodes.
            // (1 << n) == 2^n
            int ret = (1 << leftHeight) - 1;
            return ret;
        }
        // not complete binary tree
        return 1 + method1(root.left) + method1(root.right);
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + method2(root.left);
    }

    public int method3(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + method3(root.right);
    }
}
