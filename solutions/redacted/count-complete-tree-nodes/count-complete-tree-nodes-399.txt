class Solution {

    // Method to count the total number of nodes in a binary tree
    public int method1(TreeNode root) {
        // If the tree is empty, return 0
        if (root == null) {
            return 0;
        }
        // If the node is a leaf node (no children), return 1
        if (root.left == null && root.right == null) {
            return 1;
        }
        // Recursively count the nodes in the left and right subtrees
        int l = method1(root.left);
        int r = method1(root.right);
        // Return the total count of nodes in the tree
        return l + r + 1;
    }
}
