/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return helper(root, 0, 0) / 2;
    }

    public int helper(TreeNode root, int sum1, int sum2) {
        if (root == null) {
            return sum1 + sum2;
        }
        return helper(root.left, sum1 += 1, 0) + helper(root.right, 0, sum2 += 1);
    }
}
