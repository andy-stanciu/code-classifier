/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return helper(root, 0);
    }

    private int helper(TreeNode current, int param1) {
        if (current == null) {
            return param1;
        }
        param1++;
        if (current.left != null) {
            param1 = helper(current.left, param1);
        }
        if (current.right != null) {
            param1 = helper(current.right, param1);
        }
        return param1;
    }
}
