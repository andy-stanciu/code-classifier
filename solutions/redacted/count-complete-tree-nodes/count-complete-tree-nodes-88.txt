/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int c = 0;

    public int method1(TreeNode root) {
        method2(root);
        return c;
    }

    public void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        c = c + 1;
        method2(root.left);
        method2(root.right);
    }
}
