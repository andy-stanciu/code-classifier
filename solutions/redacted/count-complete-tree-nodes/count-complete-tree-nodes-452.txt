/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return noofnode(root, 0);
    }

    int c = 0;

    public int noofnode(TreeNode root, int param1) {
        if (root == null)
            return 0;
        noofnode(root.left, param1);
        noofnode(root.right, param1);
        c++;
        return c;
    }
}
