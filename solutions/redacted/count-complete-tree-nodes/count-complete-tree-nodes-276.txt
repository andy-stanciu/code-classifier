class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = method2(root);
        int right = method3(root);
        if (left == right)
            return (int) Math.pow(2, left) - 1;
        return 1 + method1(root.left) + method1(root.right);
    }

    public int method2(TreeNode root) {
        int h = 1;
        while (root.left != null) {
            root = root.left;
            h++;
        }
        return h;
    }

    public int method3(TreeNode root) {
        int h = 1;
        while (root.right != null) {
            root = root.right;
            h++;
        }
        return h;
    }
}
