class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int left = height(root, true);
        int right = height(root, false);
        if (left == right)
            return (1 << left) - 1;
        else
            return 1 + method1(root.left) + method1(root.right);
    }

    int height(TreeNode root, boolean isLeft) {
        int var1 = 0;
        while (root != null) {
            var1++;
            root = isLeft ? root.left : root.right;
        }
        return var1;
    }
}
