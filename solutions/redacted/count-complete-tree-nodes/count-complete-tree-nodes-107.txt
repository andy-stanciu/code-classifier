/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] result = new int[1];
        if (root == null)
            return 0;
        result[0] = 1;
        method2(root, result);
        return result[0];
    }

    public static void method2(TreeNode root, int[] result) {
        if (root == null)
            return;
        if (root.left != null) {
            result[0] = result[0] + 1;
            method2(root.left, result);
        }
        if (root.right != null) {
            result[0] = result[0] + 1;
            method2(root.right, result);
        }
    }
}
