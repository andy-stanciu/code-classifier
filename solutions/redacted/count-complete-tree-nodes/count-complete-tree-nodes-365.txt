class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int[] var1 = new int[1];
        method2(root, var1);
        return var1[0];
    }

    private void method2(TreeNode root, int[] var1) {
        if (root == null)
            return;
        method2(root.left, var1);
        var1[0] = var1[0] + 1;
        method2(root.right, var1);
    }
}
