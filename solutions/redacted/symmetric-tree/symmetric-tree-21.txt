/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean Function(TreeNode root, TreeNode Root) {
        if (root == null && Root == null) {
            return true;
        } else if (root == null && Root != null) {
            return false;
        } else if (root != null && Root == null) {
            return false;
        } else if (root.val != Root.val) {
            return false;
        } else if (root.val == Root.val) {
            return Function(root.left, Root.right) && Function(root.right, Root.left);
        }
        return true;
    }

    public boolean method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return true;
        }
        return Function(root.left, root.right);
    }
}
