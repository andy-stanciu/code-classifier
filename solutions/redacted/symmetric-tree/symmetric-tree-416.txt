class Solution {

    public boolean method1(TreeNode root) {
        return method2(root.left, root.right);
    }

    public boolean method2(TreeNode p1, TreeNode p2) {
        if (p1 == null && p2 == null)
            return true;
        if (p1 != null && p2 != null && p1.val == p2.val) {
            return method2(p1.left, p2.right) && method2(p1.right, p2.left);
        }
        return false;
    }
}
