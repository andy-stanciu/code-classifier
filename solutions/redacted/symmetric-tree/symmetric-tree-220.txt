class Solution {

    public boolean method1(TreeNode root) {
        return root == null || method2(root.left, root.right);
    }

    private boolean method2(TreeNode left, TreeNode right) {
        return (left == null && right == null) || (left != null && right != null && left.val == right.val && method2(left.left, right.right) && method2(left.right, right.left));
    }
}
