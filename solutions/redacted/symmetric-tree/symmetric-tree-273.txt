/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        if (root.left == null || root.right == null) {
            return root.left == root.right;
        }
        return isMinorCheck(root.left, root.right);
    }

    public boolean isMinorCheck(TreeNode a, TreeNode b) {
        if (a == null || b == null) {
            return a == b;
        }
        return (a.val == b.val && isMinorCheck(a.left, b.right) && isMinorCheck(a.right, b.left));
    }
}
