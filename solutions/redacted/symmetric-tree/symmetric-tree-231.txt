/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //this was the same code which was used leecode 100\'s question
    //they had just a small difference
    //      boolean l= isSameTree(p.left,q.left);
    //      boolean r= isSameTree(p.right,q.right);
    public boolean method1(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        if (p.val != q.val)
            return false;
        boolean l = method1(p.left, q.right);
        boolean r = method1(p.right, q.left);
        return l && r;
    }

    public boolean method2(TreeNode root) {
        if (root == null)
            return true;
        return method1(root.left, root.right);
    }
}
