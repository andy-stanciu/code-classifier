/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        // Compare entire tree to itself
        return method2(root, root);
    }

    public boolean method2(TreeNode left, TreeNode right) {
        if (left == null && right == null) {
            // Both are null (symmetric)
            return true;
        }
        if (left == null || right == null) {
            // One is null, the other isn\'t (not symmetric)
            return false;
        }
        return left.val == right.val && method2(left.left, right.right) && method2(left.right, right.left);
    }
}
