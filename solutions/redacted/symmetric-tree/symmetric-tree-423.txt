/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean method1(TreeNode p, TreeNode q) {
        if ((p == null && q != null) || (p != null && q == null))
            return false;
        if (p == null && q == null)
            return true;
        if (p.val != q.val)
            return false;
        return method1(p.left, q.left) && method1(p.right, q.right);
    }

    TreeNode method2(TreeNode root) {
        if (root == null || (root.left == null && root.right == null))
            return root;
        TreeNode temp = root.left;
        root.left = root.right;
        root.right = temp;
        method2(root.left);
        method2(root.right);
        return root;
    }

    public boolean method3(TreeNode root) {
        if (root.left == null && root.right == null)
            return true;
        if (root.left == null || root.right == null)
            return false;
        TreeNode invertLeft = method2(root.left);
        if (method1(invertLeft, root.right))
            return true;
        return false;
    }
}
