class Solution {

    public boolean method1(TreeNode root) {
        return isSymetric(root.left, root.right);
    }

    public boolean isSymetric(TreeNode lnode, TreeNode rnode) {
        if (lnode == null || rnode == null) {
            return lnode == rnode;
        } else if (lnode.val != rnode.val) {
            return false;
        }
        return isSymetric(lnode.left, rnode.right) && isSymetric(lnode.right, rnode.left);
    }
}
