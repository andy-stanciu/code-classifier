/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return helper(root.left, root.right, true);
    }

    boolean helper(TreeNode left, TreeNode right, boolean result) {
        if (left == null && right == null)
            return result;
        if (left == null || right == null)
            return false;
        if (left.val == right.val) {
            result = helper(left.left, right.right, result);
            result = helper(left.right, right.left, result);
        } else {
            return false;
        }
        return result;
    }
}
