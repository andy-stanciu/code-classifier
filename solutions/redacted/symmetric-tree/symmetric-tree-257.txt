class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method2(root.left, root.right);
    }

    private boolean method2(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return true;
        }
        if (p == null || q == null) {
            return false;
        }
        return p.val == q.val && method2(p.right, q.left) && method2(p.left, q.right);
    }
}
