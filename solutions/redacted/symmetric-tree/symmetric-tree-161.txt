/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        return root == null || checkSymmentric(root.left, root.right);
    }

    private boolean checkSymmentric(TreeNode l, TreeNode r) {
        if (l == null || r == null) {
            return l == r;
        }
        if (l.val != r.val)
            return false;
        return checkSymmentric(l.left, r.right) && checkSymmentric(l.right, r.left);
    }
}
