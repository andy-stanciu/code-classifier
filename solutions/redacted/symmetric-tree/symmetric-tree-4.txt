class Solution {

    public static boolean method1(TreeNode L_root, TreeNode R_root) {
        if (R_root == null && L_root == null)
            return true;
        if (R_root == null || L_root == null)
            return false;
        return L_root.val == R_root.val && method1(L_root.left, R_root.right) && method1(L_root.right, R_root.left);
    }

    public boolean method2(TreeNode root) {
        if (root == null)
            return true;
        return method1(root.left, root.right);
    }
}
