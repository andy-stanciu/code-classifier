/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    private boolean var1 = true;

    public boolean method1(TreeNode root) {
        check(root.left, root.right);
        return var1;
    }

    private void check(TreeNode node1, TreeNode node2) {
        if ((node1 == null && node2 == null) || !var1)
            return;
        if (node1 == null && node2 != null) {
            var1 = false;
            return;
        }
        if (node1 != null && node2 == null) {
            var1 = false;
            return;
        }
        if (node1.val != node2.val)
            var1 = false;
        check(node1.left, node2.right);
        check(node1.right, node2.left);
    }
}
