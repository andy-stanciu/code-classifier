/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private boolean method1(TreeNode left, TreeNode right) {
        //if any one is null then other should be null for symmetry
        if (left == null || right == null) {
            return left == right;
        }
        //base codn
        if (left.val != right.val)
            return false;
        //recursive call
        return method1(left.left, right.right) && method1(left.right, right.left);
    }

    public boolean method2(TreeNode root) {
        return root == null || method1(root.left, root.right);
    }
}
