/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        // If the root is null, the tree is symmetric
        if (root == null) {
            return true;
        }
        // Call the helper function with the left and right subtrees
        return isMirror(root.left, root.right);
    }

    private boolean isMirror(TreeNode left, TreeNode right) {
        // Base case: If both nodes are null, they are symmetric
        if (left == null && right == null) {
            return true;
        }
        // If one is null and the other is not, they are not symmetric
        if (left == null || right == null) {
            return false;
        }
        // Check values and recursively check subtrees
        return (left.val == right.val) && isMirror(left.left, right.right) && isMirror(left.right, right.left);
    }
}
