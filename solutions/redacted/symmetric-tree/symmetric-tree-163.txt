/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        if (root == null) {
            return false;
        }
        return traverse(root.left, root.right);
    }

    public boolean traverse(TreeNode param1, TreeNode param2) {
        boolean result = false;
        if (param1 == null && param2 == null) {
            return true;
        }
        if ((param1 != null && param2 == null) || (param1 == null && param2 != null)) {
            return result;
        }
        if (param1.val != param2.val) {
            return false;
        }
        result = traverse(param1.left, param2.right);
        if (!result) {
            return result;
        }
        result = traverse(param1.right, param2.left);
        if (!result) {
            return result;
        }
        return true;
    }
}
