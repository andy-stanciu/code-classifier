class Solution {

    public boolean method1(TreeNode root) {
        return method1(root.left, root.right);
    }

    public static boolean method1(TreeNode p, TreeNode q) {
        if (p == null && q == null)
            return true;
        if (p == null || q == null)
            return false;
        if (p.val != q.val)
            return false;
        return method1(p.left, q.right) && method1(p.right, q.left);
    }
}
