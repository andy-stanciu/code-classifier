class Solution {

    public boolean solve(TreeNode r1, TreeNode r2) {
        if (r1 == null || r2 == null)
            return r1 == r2;
        return (r1.val == r2.val) && solve(r1.left, r2.right) && solve(r1.right, r2.left);
    }

    public boolean method1(TreeNode root) {
        return solve(root.left, root.right);
    }
}
