class Solution {

    boolean x = true;

    public boolean method1(TreeNode root) {
        return method1(root.left, root.right);
    }

    private boolean method1(TreeNode n, TreeNode m) {
        if (m == null || n == null)
            return m == n;
        if (n.val != m.val)
            return false;
        return method1(n.left, m.right) && method1(m.left, n.right);
    }
}
