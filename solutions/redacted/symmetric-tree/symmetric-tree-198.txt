/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean check(TreeNode param1, TreeNode param2) {
        if (param1 == null && param2 == null)
            return true;
        if (param1 == null || param2 == null)
            return false;
        if (param1.val == param2.val && check(param1.right, param2.left) && check(param1.left, param2.right))
            return true;
        return false;
    }

    public boolean method1(TreeNode root) {
        return check(root.left, root.right);
    }
}
