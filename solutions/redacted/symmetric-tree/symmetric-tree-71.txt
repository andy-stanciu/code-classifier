class Solution {

    public boolean method1(TreeNode root) {
        TreeNode leftNode = root.left;
        TreeNode rightNode = root.right;
        boolean ans = method2(leftNode, rightNode);
        return ans;
    }

    public boolean method2(TreeNode p, TreeNode q) {
        if (p == null || q == null) {
            return p == q;
        }
        if (p.val != q.val) {
            return false;
        }
        return method2(p.left, q.right) && method2(p.right, q.left);
    }
}
