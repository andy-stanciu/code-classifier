class Solution {

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        return isMirroring(root.left, root.right);
    }

    public static boolean isMirroring(TreeNode param1, TreeNode param2) {
        if (param1 == null || param2 == null)
            return param1 == param2;
        return (param1.val == param2.val) && isMirroring(param1.left, param2.right) && isMirroring(param1.right, param2.left);
    }
}
