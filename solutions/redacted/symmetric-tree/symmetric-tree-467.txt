/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        return areEquals(root.left, root.right);
    }

    public boolean areEquals(TreeNode a, TreeNode b) {
        if ((a != null && b == null) || (a == null && b != null))
            return false;
        if (a == null && b == null) {
            return true;
        }
        if (a.val != b.val)
            return false;
        boolean resultA = areEquals(a.left, b.right);
        boolean resultB = areEquals(b.left, a.right);
        return resultA && resultB;
    }
}
