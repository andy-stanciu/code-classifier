/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        return root == null || method2(root.left, root.right);
    }

    boolean method2(TreeNode leftNode, TreeNode rightNode) {
        // Check if leftN or rightN is null
        if (leftNode == null || rightNode == null) {
            return leftNode == rightNode;
        }
        // Check if leftN.val & rightN.val are same?
        if (leftNode.val != rightNode.val)
            return false;
        return method2(leftNode.left, rightNode.right) && method2(leftNode.right, rightNode.left);
    }
}
