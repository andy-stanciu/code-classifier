/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root) {
        return root == null || method1(root.left, root.right);
    }

    private boolean method1(TreeNode left, TreeNode right) {
        if (left == null || right == null) {
            return left == right;
        }
        if (right.val != left.val) {
            return false;
        }
        return method1(left.left, right.right) && method1(right.left, left.right);
    }
}
