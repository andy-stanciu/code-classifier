/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private boolean method1(TreeNode r1, TreeNode r2) {
        if (r1 == null && r2 == null) {
            return true;
        }
        if (r1 == null && r2 != null) {
            return false;
        }
        if (r1 != null && r2 == null) {
            return false;
        }
        boolean left = method1(r1.left, r2.right);
        boolean right = method1(r1.right, r2.left);
        boolean isSame = r1.val == r2.val;
        if (left && right && isSame) {
            return true;
        }
        return false;
    }

    public boolean method2(TreeNode root) {
        if (root == null) {
            return true;
        }
        return method1(root.left, root.right);
    }
}
