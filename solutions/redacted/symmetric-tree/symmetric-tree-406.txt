/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean isSame = true;

    public boolean method1(TreeNode root) {
        if (root == null)
            return true;
        inOrderTraversal(root.left, root.right);
        return isSame;
    }

    private void inOrderTraversal(TreeNode p, TreeNode q) {
        if (!isSame) {
            return;
        }
        //same structure
        if (p == null && q == null) {
            return;
        }
        //not same structure
        if (p == null || q == null || p.val != q.val) {
            isSame = false;
            return;
        }
        inOrderTraversal(p.left, q.right);
        inOrderTraversal(p.right, q.left);
    }
}
