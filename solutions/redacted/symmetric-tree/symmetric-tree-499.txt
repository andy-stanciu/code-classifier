/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean check = true;

    public void double_traversal(TreeNode p, TreeNode q) {
        if (p == null && q == null) {
            return;
        }
        if ((p == null && q != null) || (p != null && q == null)) {
            check = false;
            return;
        } else if (p.val != q.val) {
            check = false;
            return;
        } else {
            double_traversal(p.left, q.right);
            double_traversal(p.right, q.left);
        }
    }

    public boolean method1(TreeNode root) {
        double_traversal(root.left, root.right);
        return check;
    }
}
