class Solution {

    public boolean method1(int n) {
        int s = n;
        int f = n;
        do {
            s = Square(s);
            f = Square(Square(f));
            if (s == 1) {
                return true;
            }
        } while (s != f);
        return false;
    }

    public int Square(int param1) {
        int ans = 0;
        while (param1 > 0) {
            ans += (param1 % 10) * (param1 % 10);
            param1 /= 10;
        }
        return ans;
    }
}
