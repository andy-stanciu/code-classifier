class Solution {

    public boolean method1(int n) {
        Set<Integer> visited = new HashSet<>();
        while (true) {
            int sum = 0;
            while (n != 0) {
                sum += Math.pow(n % 10, 2);
                n = n / 10;
            }
            if (sum == 1) {
                return true;
            }
            n = sum;
            if (visited.contains(n)) {
                return false;
            }
            visited.add(n);
        }
    }
}
