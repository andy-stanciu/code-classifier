class Solution {

    public boolean method1(int n) {
        int temp, square;
        int count;
        int[] nos = new int[10];
        while (n != 1 && n != 4) {
            temp = n;
            square = 0;
            count = 0;
            while (temp != 0) {
                nos[count] = temp % 10;
                temp = temp / 10;
                count++;
            }
            for (int i = 0; i < count; i++) {
                square = square + (nos[i] * nos[i]);
            }
            n = square;
        }
        if (n == 1) {
            return true;
        } else {
            return false;
        }
    }
}
