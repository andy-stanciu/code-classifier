class Solution {

    public boolean method1(int n) {
        int cnt = 0;
        while (true) {
            int sum2 = 0;
            while (n > 0) {
                sum2 += Math.pow((n % 10), 2);
                n /= 10;
            }
            cnt++;
            if (cnt == 8) {
                return false;
            }
            if (sum2 == 1) {
                return true;
            }
            n = sum2;
        }
    }
}
