class Solution {

    public boolean method1(int n) {
        int s = 0, d = 0;
        while (n != 1 && n != 4) {
            s = 0;
            while (n > 0) {
                d = n % 10;
                s = s + (d * d);
                n = n / 10;
            }
            n = s;
        }
        if (n == 1) {
            return true;
        }
        return false;
    }
}
