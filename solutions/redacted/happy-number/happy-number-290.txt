class Solution {

    public int helper(int a) {
        int ans = 0;
        while (a != 0) {
            int temp = a % 10;
            ans = ans + (temp * temp);
            a = a / 10;
        }
        return ans;
    }

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = helper(slow);
            fast = helper(helper(fast));
        } while (fast != slow);
        return slow == 1;
    }
}
