class Solution {

    public boolean method1(int n) {
        if (n < 0)
            return false;
        while (n > 6) {
            int sum = 0;
            while (n != 0) {
                int a = n % 10;
                n = n / 10;
                sum = sum + a * a;
            }
            n = sum;
        }
        if (n == 1)
            return true;
        return false;
    }
}
