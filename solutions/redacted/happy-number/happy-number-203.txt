class Solution {

    public boolean method1(int n) {
        if (n == 1)
            return true;
        int slow = n, fast = n;
        do {
            slow = sumOfDigits(slow);
            fast = sumOfDigits(sumOfDigits(fast));
        } while (slow != fast);
        return slow == 1;
    }

    private int sumOfDigits(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        return sum;
    }
}
