class Solution {

    public boolean method1(int n) {
        Set<Integer> var1 = new HashSet<>();
        while (n != 1 && !var1.contains(n)) {
            var1.add(n);
            n = method2(n);
        }
        return n == 1;
    }

    public int method2(int n) {
        int sum = 0;
        while (n > 0) {
            int rem = n % 10;
            sum += rem * rem;
            n /= 10;
        }
        return sum;
    }
}
