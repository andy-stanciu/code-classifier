class Solution {

    public boolean method1(int n) {
        int result = n;
        while (result != 1 && result != 4) {
            result = method2(result);
        }
        if (result == 1) {
            return true;
        } else {
            return false;
        }
    }

    int method2(int n) {
        int rem = 0, sum = 0;
        while (n != 0) {
            rem = n % 10;
            sum = sum + (rem * rem);
            n = n / 10;
        }
        return sum;
    }
}
