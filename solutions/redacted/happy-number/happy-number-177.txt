class Solution {

    public boolean method1(int n) {
        int sum = 0;
        int temp;
        while (n > 0) {
            temp = n;
            while (temp > 0) {
                sum = sum + (temp % 10) * (temp % 10);
                temp = temp / 10;
            }
            if (sum == 1)
                return true;
            else if (sum == 4) {
                return false;
            }
            n = sum;
            sum = 0;
        }
        return false;
    }
}
