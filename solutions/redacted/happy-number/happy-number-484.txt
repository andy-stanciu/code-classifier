class Solution {

    public boolean method1(int n) {
        if (n == 1 || n == 7)
            return true;
        if (n < 10)
            return false;
        return method1(sumOfSquares(n));
    }

    public int sumOfSquares(int n) {
        int sum = 0;
        while (n > 0) {
            sum = sum + (n % 10) * (n % 10);
            n = n / 10;
        }
        return sum;
    }
}
