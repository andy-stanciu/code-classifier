class Solution {

    public boolean method1(int n) {
        int countHandler = 0;
        while (countHandler < 7 && method2(n) != 1) {
            n = method2(n);
            countHandler++;
        }
        return !(countHandler == 7);
    }

    public int method2(int n) {
        if (n % 10 == n) {
            return n * n;
        }
        return (int) Math.pow((n % 10), 2) + method2(n / 10);
    }
}
