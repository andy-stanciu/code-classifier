class Solution {

    public boolean method1(int n) {
        Set<Integer> var1 = new HashSet<>();
        while (n != 1 && !var1.contains(n)) {
            var1.add(n);
            int sum = 0;
            while (n > 0) {
                int rem = n % 10;
                sum += rem * rem;
                n /= 10;
            }
            n = sum;
        }
        return n == 1;
    }
}
