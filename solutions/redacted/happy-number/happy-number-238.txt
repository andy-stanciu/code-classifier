class Solution {

    public boolean method1(int n) {
        int slow = n, fast = n;
        do {
            slow = check(slow);
            fast = check(check(fast));
            if (slow == 1)
                return true;
        } while (slow != fast);
        return false;
    }

    public int check(int n) {
        int sum = 0;
        while (n != 0) {
            sum += (n % 10) * (n % 10);
            n /= 10;
        }
        return sum;
    }
}
