class Solution {

    public boolean method1(int n) {
        int f = n;
        int s = n;
        do {
            s = square(s);
            f = square(square(f));
        } while (f != s);
        if (f == 1) {
            return true;
        }
        return false;
    }

    public int square(int a) {
        int sum = 0;
        while (a > 0) {
            sum += (a % 10) * (a % 10);
            a = a / 10;
        }
        return sum;
    }
}
