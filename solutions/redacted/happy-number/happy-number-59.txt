class Solution {

    public boolean method1(int n) {
        int sum = 0;
        while (n >= 7) {
            // less than 7 except 1, all are unhappy number
            while (n > 0) {
                // find last digit
                int temp = n % 10;
                // sum the square
                sum = sum + (temp * temp);
                // drop the last digit or remainder
                n = n / 10;
            }
            // for next cycle calculation hand over to n
            n = sum;
            // for everytime calculation 0ed it
            sum = 0;
        }
        if (n == 1)
            return true;
        return false;
    }
}
