class Solution {

    public boolean method1(int n) {
        int maxLoop = 0;
        while (maxLoop <= 30) {
            maxLoop++;
            int sum = digitsSquareSum(n);
            if (sum == 1)
                return true;
            else
                n = sum;
        }
        return false;
    }

    public int digitsSquareSum(int n) {
        int sum = 0;
        // 123
        do {
            sum += ((n % 10) * (n % 10));
            n /= 10;
        } while (n != 0);
        return sum;
    }
}
