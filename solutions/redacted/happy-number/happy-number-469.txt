class Solution {

    public boolean method1(int n) {
        int f = n;
        int s = n;
        do {
            s = square(s);
            f = square(square(f));
        } while (s != f);
        if (s == 1) {
            return true;
        }
        return false;
    }

    public int square(int param1) {
        int sum = 0;
        while (param1 > 0) {
            int r = param1 % 10;
            sum += r * r;
            param1 = param1 / 10;
        }
        return sum;
    }
}
