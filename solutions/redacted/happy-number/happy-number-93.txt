class Solution {

    public boolean method1(int n) {
        int xx = 0;
        while (n != 1) {
            int k = n, x = 0;
            while (k > 0) {
                x += (k % 10) * (k % 10);
                k /= 10;
            }
            n = x;
            xx++;
            if (xx == 8) {
                return false;
            }
        }
        return true;
    }
}
