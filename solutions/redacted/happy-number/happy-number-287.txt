class Solution {

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        while (n != 1) {
            n = square(n);
            if (set.contains(n))
                return false;
            set.add(n);
        }
        return true;
    }

    public int square(int num) {
        int sum = 0;
        while (num > 0) {
            int rem = num % 10;
            sum += rem * rem;
            num /= 10;
        }
        return sum;
    }
}
