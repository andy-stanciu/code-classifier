class Solution {

    public boolean method1(int n) {
        return method2(n, new HashSet<>());
    }

    private boolean method2(int n, HashSet<Integer> seen) {
        // Base case: If n becomes 1, it\'s a happy number
        if (n == 1) {
            return true;
        }
        // If n is already seen (detects cycle), return false
        if (seen.contains(n)) {
            return false;
        }
        // Add n to seen numbers
        seen.add(n);
        // Calculate the sum of squares of digits
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += digit * digit;
            n /= 10;
        }
        // Recursively check the next number
        return method2(sum, seen);
    }
}
