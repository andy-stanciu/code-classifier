class Solution {

    public boolean method1(int n) {
        // In single digit number, only 1 and 7 are happy numbers
        if (n == 1 || n == 7) {
            return true;
        }
        if (n < 10 && n != 1) {
            return false;
        }
        int s = 0;
        while (n != 0) {
            int l = n % 10;
            s += l * l;
            n = n / 10;
        }
        return method1(s);
    }
}
