class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = helperSum(n);
        while (slow != fast && fast != 1) {
            slow = helperSum(slow);
            fast = helperSum(helperSum(fast));
        }
        return fast == 1;
    }

    static int helperSum(int num) {
        int sum = 0;
        while (num > 0) {
            sum = sum + (num % 10) * (num % 10);
            num = num / 10;
        }
        return sum;
    }
}
