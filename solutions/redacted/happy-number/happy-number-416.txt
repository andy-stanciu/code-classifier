class Solution {

    public boolean method1(int n) {
        int sum = 0;
        while (n > 0) {
            int a = n % 10;
            sum = sum + a * a;
            n = n / 10;
        }
        if (sum == 1) {
            return true;
        } else if (sum == 4) {
            return false;
        }
        return method1(sum);
    }
}
