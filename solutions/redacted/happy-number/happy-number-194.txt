class Solution {

    public boolean method1(int n) {
        ArrayList<Integer> arr = new ArrayList<>();
        while (!arr.contains(n)) {
            arr.add(n);
            int sum = 0;
            while (n > 0) {
                int num = n % 10;
                n = n / 10;
                sum += (num * num);
            }
            n = sum;
            if (n == 1) {
                return true;
            }
        }
        return false;
    }
}
