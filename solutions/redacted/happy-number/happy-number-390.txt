class Solution {

    public boolean method1(int n) {
        int slow = n, fast = n;
        do {
            slow = sumSquareDigit(slow);
            fast = sumSquareDigit(sumSquareDigit(fast));
        } while (slow != fast);
        return slow == 1;
    }

    private int sumSquareDigit(int n) {
        int sum = 0;
        while (n != 0) {
            sum = sum + ((n % 10) * (n % 10));
            n = n / 10;
        }
        return sum;
    }
}
