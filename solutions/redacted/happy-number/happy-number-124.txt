class Solution {

    //Explanation is provided above for using Set.
    Set<Integer> generatedNum = new HashSet<>();

    public boolean method1(int n) {
        // if n becomes 1, it means the number is a happy number.
        if (n == 1)
            return true;
        // used to hold the temp generated number
        int temp = 0;
        while (n != 0) {
            temp += (n % 10) * (n % 10);
            n = n / 10;
        }
        //this if condition is to check the repeating generated temp number.
        if (generatedNum.contains(temp))
            return false;
        generatedNum.add(temp);
        return method1(temp);
    }
}
