class Solution {

    public boolean method1(int n) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        // We will run this loop till n becomes 1
        while (n != 1) {
            int sum = 0;
            int current = n;
            //  We will run this loop till current > 0
            while (current > 0) {
                // Taking every digit and square it and adds in sum variable
                int digit = current % 10;
                sum += digit * digit;
                current /= 10;
            }
            if (mp.containsKey(sum))
                return false;
            else {
                mp.put(sum, 1);
            }
            // Update n to the new sum of squares
            n = sum;
        }
        // If we reach 1, the number is happy
        return true;
    }
}
