class Solution {

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        while (n != 1) {
            n = getSum(n);
            if (set.contains(n)) {
                return false;
            } else {
                set.add(n);
            }
        }
        return true;
    }

    int getSum(int n) {
        int res = 0;
        while (n > 0) {
            int d = n % 10;
            res += (d * d);
            n = n / 10;
        }
        return res;
    }
}
