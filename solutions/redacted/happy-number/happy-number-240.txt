class Solution {

    public boolean method1(int n) {
        HashSet<Integer> used = new HashSet<>();
        while (true) {
            //finding sum of the square
            int sum = 0;
            while (n != 0) {
                sum += Math.pow(n % 10, 2);
                n /= 10;
            }
            //if sum=1, return true;
            if (sum == 1)
                return true;
            // else new number is current sum
            n = sum;
            // checking if we have already encountered that number
            if (used.contains(n))
                return false;
            used.add(n);
        }
    }
}
