class Solution {

    public boolean method1(int n) {
        // directly return true no further computation
        if (n == 1)
            return true;
        HashMap<Integer, Integer> map = new HashMap<>();
        do {
            int sum = 0;
            // find sum of squares
            while (n != 0) {
                sum += (int) Math.pow(n % 10, 2);
                n /= 10;
            }
            // check if sum of squares generated 1
            if (sum == 1)
                return true;
            if (// check if the number generated is already in HashMap then break the loop and return false
            map.containsKey(sum))
                // check if the number generated is already in HashMap then break the loop and return false
                break;
            else
                // otherwise put that number in hashmap
                map.put(sum, 0);
            n = sum;
        } while (n != 1);
        return false;
    }
}
