class Solution {

    public boolean method1(int n) {
        if (n == 1 || n == 7) {
            return true;
        }
        if (n < 10) {
            return false;
        }
        int newNum = 0;
        while (n > 0) {
            int rem = n % 10;
            newNum = newNum + rem * rem;
            n = n / 10;
        }
        return method1(newNum);
    }
}
