class Solution {

    public boolean method1(int n) {
        Set<Integer> seen = new HashSet<>();
        while (n != 1 && !seen.contains(n)) {
            seen.add(n);
            n = getSquareSum(n);
        }
        return n == 1;
    }

    private int getSquareSum(int n) {
        int squareSum = 0;
        while (n > 0) {
            int digit = n % 10;
            squareSum += digit * digit;
            n /= 10;
        }
        return squareSum;
    }
}
