class Solution {

    public boolean method1(int n) {
        int temp = sumDig(n);
        int res = temp;
        int count = 0;
        do {
            if (res == 1)
                return true;
            if (count > 10)
                break;
            res = sumDig(res);
            count++;
        } while (res != temp);
        return false;
    }

    static int sumDig(int n) {
        int sum = 0;
        int lastDig;
        while (n > 0) {
            lastDig = n % 10;
            n /= 10;
            sum += lastDig * lastDig;
        }
        return sum;
    }
}
