class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = square(n);
        while (fast != 1 && slow != fast) {
            slow = square(slow);
            fast = square(square(fast));
        }
        return fast == 1;
    }

    public int square(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            sum += digit * digit;
            n = n / 10;
        }
        return sum;
    }
}
