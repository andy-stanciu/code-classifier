class Solution {

    public boolean method1(int n) {
        int sum = 0;
        do {
            while (n != 0) {
                int r = n % 10;
                int sq = r * r;
                sum += sq;
                n /= 10;
            }
            n = sum;
            sum = 0;
        } while (n > 5);
        if (n == 1)
            return true;
        return false;
    }
}
