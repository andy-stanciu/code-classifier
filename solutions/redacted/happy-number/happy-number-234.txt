class Solution {

    public boolean method1(int n) {
        int fast = n;
        int slow = n;
        do {
            slow = FindSquare(slow);
            fast = FindSquare(FindSquare(fast));
        } while (fast != slow);
        if (slow == 1) {
            return true;
        }
        return false;
    }

    public int FindSquare(int param1) {
        int ans = 0;
        while (param1 > 0) {
            int rem = param1 % 10;
            ans += rem * rem;
            param1 = param1 / 10;
        }
        return ans;
    }
}
