class Solution {

    public int findSum(int n) {
        int sum = 0;
        while (n != 0) {
            sum += (n % 10) * (n % 10);
            n /= 10;
        }
        return sum;
    }

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = findSum(slow);
            fast = findSum(findSum(fast));
        } while (slow != fast);
        if (slow == 1) {
            return true;
        }
        return false;
    }
}
