class Solution {

    public boolean method1(int n) {
        Set<Integer> var1 = new HashSet<>();
        while (n != 1) {
            int current = n;
            int sum = 0;
            while (current != 0) {
                int rem = current % 10;
                sum = sum + (rem * rem);
                current = current / 10;
            }
            if (var1.contains(sum))
                return false;
            n = sum;
            var1.add(n);
        }
        return true;
    }
}
