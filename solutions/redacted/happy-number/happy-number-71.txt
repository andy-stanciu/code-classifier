class Solution {

    public boolean method1(int n) {
        //GIVE VALUE TO SUM
        int sum = n;
        while (sum != 1 && sum != 4) {
            //THEN TEMP
            int temp = sum;
            sum = 0;
            while (temp != 0) {
                int digit = temp % 10;
                //ADDING VALUES
                sum += digit * digit;
                temp = temp / 10;
            }
        }
        return sum == 1;
    }
}
