class Solution {

    public boolean method1(int n) {
        if (n == 1 || n == 7)
            return true;
        else if (n < 10)
            return false;
        int sqr = 0;
        while (n > 0) {
            sqr += (n % 10) * (n % 10);
            n /= 10;
        }
        return method1(sqr);
    }
}
