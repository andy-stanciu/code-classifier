class Solution {

    public boolean method1(int n) {
        if (n < 10) {
            if (n == 1 || n == 7)
                return true;
            else
                return false;
        }
        return method1(findsq(n));
    }

    public static int findsq(int n) {
        // Base case: if n is reduced to 0, return 0
        if (n == 0)
            return 0;
        // Find the remainder (last digit)
        int rem = n % 10;
        // Recursive call with the remaining digits
        return rem * rem + findsq(n / 10);
    }
}
