class Solution {

    public boolean method1(int n) {
        Set<Integer> s = new HashSet<>();
        while (n != 1 && !s.contains(n)) {
            s.add(n);
            int ans = 0;
            while (n != 0) {
                int rem = n % 10;
                ans += (rem * rem);
                n /= 10;
            }
            n = ans;
        }
        return n == 1;
    }
}
