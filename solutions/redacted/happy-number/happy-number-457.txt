class Solution {

    public boolean method1(int n) {
        List<Integer> previousSums = new ArrayList();
        int d = 0;
        int sum = 0;
        boolean firstTimeFlag = false;
        while (!previousSums.contains(n)) {
            //if(firstTimeFlag)
            previousSums.add(sum);
            firstTimeFlag = true;
            sum = 0;
            while (n != 0) {
                d = n % 10;
                sum += (d * d);
                n = n / 10;
            }
            if (sum == 1)
                return true;
            else {
                n = sum;
            }
        }
        return false;
    }
}
