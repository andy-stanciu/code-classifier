class Solution {

    public boolean method1(int n) {
        int t = n;
        while (true) {
            int sum = sum(n);
            if (n == 1) {
                return true;
            }
            if (n == 4) {
                return false;
            }
            n = sum;
        }
    }

    static int sum(int n) {
        int s = 0;
        while (n != 0) {
            s = s + (int) (Math.pow(n % 10, 2));
            n = n / 10;
        }
        return s;
    }
}
