class Solution {

    public boolean method1(int n) {
        int c = 0;
        while (n != 1) {
            c++;
            int sum = 0;
            while (n != 0) {
                int a = n % 10;
                sum += Math.pow(a, 2);
                n /= 10;
            }
            n = sum;
            if (c > 20) {
                break;
            }
        }
        if (n == 1) {
            return true;
        } else {
            return false;
        }
    }
}
