class Solution {

    public static boolean method1(int n) {
        Set<Integer> seen = new HashSet<>();
        while (n != 1 && !seen.contains(n)) {
            seen.add(n);
            n = square(n);
        }
        // if final answer is 1 true else it will return false
        return n == 1;
    }

    public static int square(int param1) {
        int totalSum = 0;
        while (param1 > 0) {
            int digit = param1 % 10;
            totalSum += digit * digit;
            param1 /= 10;
        }
        return totalSum;
    }
}
