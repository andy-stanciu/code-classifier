class Solution {

    public boolean method1(int n) {
        HashMap<Integer, Boolean> h = new HashMap<>();
        while (n != 1 && !h.containsKey(n)) {
            h.put(n, true);
            n = sumofdigits(n);
        }
        return n == 1;
    }

    public int sumofdigits(int n) {
        int sum = 0;
        while (n > 0) {
            int rem = n % 10;
            sum = rem * rem + sum;
            n /= 10;
        }
        return sum;
    }
}
