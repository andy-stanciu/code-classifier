class Solution {

    HashMap<Integer, Integer> map = new HashMap<>();

    public boolean method1(int n) {
        if (getSumOfSquares(n) == 1) {
            return true;
        } else if (map.containsKey(n)) {
            return false;
        } else {
            map.put(n, 1);
            return method1(getSumOfSquares(n));
        }
    }

    public List<Integer> getDigits(int n) {
        List<Integer> digits = new ArrayList<>();
        while (n > 0) {
            digits.add(n % 10);
            n /= 10;
        }
        return digits;
    }

    public int getSumOfSquares(int n) {
        List<Integer> digits = getDigits(n);
        int sum = 0;
        for (int digit : digits) {
            sum += digit * digit;
        }
        return sum;
    }
}
