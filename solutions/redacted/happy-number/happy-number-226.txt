class Solution {

    public boolean method1(int n) {
        while (n != 1 && n != 4) {
            int mul = 0;
            while (n > 0) {
                int t = n % 10;
                mul += t * t;
                n /= 10;
            }
            n = mul;
        }
        return n == 1;
    }
}
