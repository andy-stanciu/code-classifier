class Solution {

    public boolean method1(int n) {
        int res = n;
        List<Integer> list = new ArrayList<>();
        int ans;
        //boolean check=true;
        while (true) {
            ans = square(res);
            if (ans == 1) {
                return true;
            }
            if (list.contains(ans)) {
                return false;
            } else {
                list.add(ans);
            }
            res = ans;
        }
    }

    public static int square(int res) {
        int sum = 0;
        while (res > 0) {
            int digit = res % 10;
            sum = sum + digit * digit;
            res = res / 10;
        }
        return sum;
    }
}
