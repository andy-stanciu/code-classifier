class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = sqrt(slow);
            fast = sqrt(sqrt(fast));
        } while (slow != fast);
        return slow == 1;
    }

    public int sqrt(int param1) {
        int summa = 0;
        while (param1 > 0) {
            int m = param1 % 10;
            summa += m * m;
            param1 /= 10;
        }
        return summa;
    }
}
