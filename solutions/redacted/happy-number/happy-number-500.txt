class Solution {

    public boolean method1(int n) {
        HashSet<Integer> hs = new HashSet<Integer>();
        while (n != 0 && !hs.contains(n)) {
            hs.add(n);
            int sum = 0;
            while (n > 0) {
                int digit = n % 10;
                sum += digit * digit;
                n = n / 10;
            }
            if (sum == 1) {
                return true;
            }
            n = sum;
        }
        return false;
    }
}
