class Solution {

    public boolean method1(int n) {
        int v1 = n, v2 = n;
        do {
            v1 = findSqrs(v1);
            v2 = findSqrs(findSqrs(v2));
        } while (v1 != v2);
        return v1 == 1;
    }

    int findSqrs(int n) {
        int sq = 0;
        while (n > 0) {
            int d = n % 10;
            n = n / 10;
            sq += d * d;
        }
        return sq;
    }
}
