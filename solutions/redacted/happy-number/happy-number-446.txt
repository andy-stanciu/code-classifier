class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        while (true) {
            slow = findSum(slow);
            fast = findSum(findSum(fast));
            if (slow == fast) {
                break;
            }
        }
        return slow == 1;
    }

    public int findSum(int num) {
        int sqsum = 0;
        while (num > 0) {
            int x = num % 10;
            sqsum += x * x;
            num = num / 10;
        }
        return sqsum;
    }
}
