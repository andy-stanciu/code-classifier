class Solution {

    public boolean method1(int n) {
        int counterPreventLoop = 0;
        int maxiterations = 100;
        while (n != 1 && counterPreventLoop < maxiterations) {
            int quadSum = 0;
            int digit;
            while (n > 9) {
                digit = n % 10;
                quadSum += digit * digit;
                n = n / 10;
            }
            quadSum += n * n;
            n = quadSum;
            counterPreventLoop += 1;
        }
        if (counterPreventLoop < 100) {
            return true;
        } else {
            return false;
        }
    }
}
