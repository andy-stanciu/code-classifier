class Solution {

    public boolean method1(int n) {
        while (n > 9) {
            int sum = 0;
            //loops till the number in n is not equal to zero
            while (n != 0) {
                //gets the last digit
                int digit = n % 10;
                //squares the last digit and adds it to the sum
                sum += digit * digit;
                //removes the last digit
                n /= 10;
            }
            //Updates the value of n , to new number derived from the sum of the squares
            n = sum;
        }
        return n == 1 || n == 7;
    }
}
