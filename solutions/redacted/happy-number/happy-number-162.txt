class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = square(slow);
            fast = square(square(fast));
        } while (slow != fast);
        return slow == 1;
    }

    private int square(int n) {
        int ans = 0;
        while (n != 0) {
            int a = n % 10;
            ans += a * a;
            n /= 10;
        }
        return ans;
    }
}
