class Solution {

    public boolean method1(int n) {
        while (n > 9) {
            int sum = 0;
            do {
                int d = n % 10;
                sum = sum + d * d;
                n = n / 10;
            } while (n != 0);
            n = sum;
        }
        return n == 1 || n == 7;
    }
}
