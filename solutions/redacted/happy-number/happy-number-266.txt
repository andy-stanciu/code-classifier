class Solution {

    public boolean method1(int n) {
        Set<Integer> set = new HashSet<>();
        while (n != 1) {
            if (set.contains(n)) {
                return false;
            }
            set.add(n);
            int temp = 0;
            while (n != 0) {
                int digit = n % 10;
                n /= 10;
                temp += (digit * digit);
            }
            n = temp;
        }
        return true;
    }
}
