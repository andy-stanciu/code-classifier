class Solution {

    public boolean method1(int n) {
        // not my code copied it from sols
        Set<Long> var1 = new HashSet<>();
        long m = n;
        while (m != 1 && !var1.contains(m)) {
            var1.add(m);
            long sum = 0;
            while (m > 0) {
                long digit = m % 10;
                sum += digit * digit;
                m /= 10;
            }
            m = sum;
        }
        return m == 1;
    }
}
