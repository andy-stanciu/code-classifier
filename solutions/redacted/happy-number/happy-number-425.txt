class Solution {

    public boolean method1(int n) {
        if (n < 10) {
            if (n == 1 || n == 7) {
                return true;
            } else {
                return false;
            }
        }
        int sum = 0;
        while (n > 0) {
            int num = n % 10;
            sum += num * num;
            n /= 10;
        }
        return method1(sum);
    }
}
