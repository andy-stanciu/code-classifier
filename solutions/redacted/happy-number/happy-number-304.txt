class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = sumOfSqure(n);
        while (fast != 1 && slow != fast) {
            slow = sumOfSqure(slow);
            fast = sumOfSqure(sumOfSqure(fast));
        }
        return fast == 1;
    }

    private int sumOfSqure(int n) {
        int sum = 0;
        while (n > 0) {
            int digit = n % 10;
            n = n / 10;
            sum += Math.pow(digit, 2);
        }
        return sum;
    }
}
