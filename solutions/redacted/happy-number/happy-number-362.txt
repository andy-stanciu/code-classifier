class Solution {

    public boolean method1(int n) {
        while (n > 9) {
            n = sumOfSquaredDigits(n);
        }
        if (n == 1 || n == 7)
            return true;
        return false;
    }

    public int sumOfSquaredDigits(int n) {
        int num = 0;
        while (n >= 1) {
            int temp = n % 10;
            num += temp * temp;
            n = n / 10;
        }
        return num;
    }
}
