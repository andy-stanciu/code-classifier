class Solution {

    public boolean method1(int n) {
        int ans = n;
        if (n == 1) {
            return true;
        }
        while (ans > 4) {
            int sum = 0;
            while (ans > 0) {
                sum += (ans % 10) * (ans % 10);
                ans = ans / 10;
            }
            if (sum == 1) {
                return true;
            }
            ans = sum;
        }
        return false;
    }
}
