class Solution {

    public boolean method1(int n) {
        int slow = n;
        int fast = n;
        do {
            slow = method2(slow);
            fast = method2(method2(fast));
        } while (slow != fast);
        return slow == 1;
    }

    static int method2(int n) {
        int sum = 0;
        while (n != 0) {
            int x = n % 10;
            sum += x * x;
            n /= 10;
        }
        return sum;
    }
}
