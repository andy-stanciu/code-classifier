class Solution {

    public boolean method1(int n) {
        do {
            n = sumOfSquares(n);
        } while (n > 6);
        if (n == 1)
            return true;
        else
            return false;
    }

    public int sumOfSquares(int n) {
        int sum = 0;
        while (n != 0) {
            sum += (int) Math.pow(n % 10, 2);
            n /= 10;
        }
        return sum;
    }
}
