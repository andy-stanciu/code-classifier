class Solution {

    public boolean method1(int n) {
        if (n == 1 || n == 7) {
            return true;
        } else if (n < 10) {
            return false;
        }
        int sum = 0;
        while (n > 0) {
            int k = n % 10;
            sum = sum + (k * k);
            n = n / 10;
        }
        return method1(sum);
    }
}
