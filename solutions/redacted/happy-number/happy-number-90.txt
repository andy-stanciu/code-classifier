class Solution {

    public int method1(int n) {
        int var1 = 0;
        while (n != 0) {
            int num = n % 10;
            var1 += num * num;
            n = n / 10;
        }
        return var1;
    }

    public boolean method2(int n) {
        HashSet<Integer> set = new HashSet<>();
        while (n != 1 && !set.contains(n)) {
            set.add(n);
            n = method1(n);
        }
        return n == 1;
    }
}
