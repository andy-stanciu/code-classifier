class Solution {

    public boolean method1(int n) {
        while (n > 9) {
            n = get(n);
        }
        return (n == 1 || n == 7);
    }

    public int get(int n) {
        int sum = 0;
        // int temp = n;
        while (n != 0) {
            int r = n % 10;
            sum = sum + r * r;
            n /= 10;
        }
        return sum;
    }
}
