class Solution {

    // Function to remove the nth node from end
    public ListNode method1(ListNode head, int n) {
        // Creating pointers
        ListNode fastp = head;
        ListNode slowp = head;
        /* Move the fastp pointer 
        N nodes ahead */
        for (int i = 0; i < n; i++) {
            fastp = fastp.next;
        }
        /* If fastp becomes NULL
        the Nth node from the 
        end is the head */
        if (fastp == null) {
            return head.next;
        }
        /* Move both pointers 
        Until fastp reaches the end */
        while (fastp.next != null) {
            fastp = fastp.next;
            slowp = slowp.next;
        }
        // Delete the Nth node from the end
        ListNode var1 = slowp.next;
        slowp.next = slowp.next.next;
        return head;
    }
}
