class Solution {

    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> var1 = new ArrayList<>();
        zigzagLevelOrder(var1, root, 0);
        return var1;
    }

    public void zigzagLevelOrder(List<List<Integer>> var1, TreeNode root, int height) {
        if (root == null)
            return;
        if (height >= var1.size()) {
            var1.add(new ArrayList<>());
        }
        if (height % 2 == 0) {
            var1.get(height).add(root.val);
        } else {
            var1.get(height).add(0, root.val);
        }
        zigzagLevelOrder(var1, root.left, height + 1);
        zigzagLevelOrder(var1, root.right, height + 1);
    }
}
