class Solution {

    public ListNode method1(ListNode head, int n) {
        int count = 0;
        ListNode temp = head;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        int pos = count - n;
        if (pos == 0) {
            return head.next;
        }
        count = 0;
        ListNode prev = head;
        while (count < pos - 1) {
            count++;
            prev = prev.next;
        }
        prev.next = prev.next.next;
        return head;
    }
}
