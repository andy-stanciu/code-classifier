class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        int var1 = count - n;
        temp = head;
        count = 0;
        ListNode prev = temp;
        if (var1 == 0) {
            return temp.next;
        }
        while (temp != null) {
            if (count == var1) {
                prev.next = temp.next;
            }
            count++;
            prev = temp;
            temp = temp.next;
        }
        return head;
    }
}
