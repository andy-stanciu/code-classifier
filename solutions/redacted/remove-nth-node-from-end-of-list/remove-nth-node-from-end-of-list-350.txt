class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null || (head.next == null && n == 1))
            return null;
        int var1 = 1;
        ListNode curr = head;
        while (curr.next != null) {
            curr = curr.next;
            var1++;
        }
        var1 -= n;
        curr = head;
        ListNode prev = null;
        while (var1 != 0) {
            prev = curr;
            curr = curr.next;
            var1--;
        }
        if (prev != null) {
            prev.next = curr.next;
        } else {
            return head.next;
        }
        return head;
    }
}
