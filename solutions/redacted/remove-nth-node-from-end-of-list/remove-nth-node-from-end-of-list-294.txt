class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode first = head, second = head;
        for (int i = 0; i < n; i++) second = second.next;
        if (second == null)
            return head.next;
        while (second.next != null) {
            first = first.next;
            second = second.next;
        }
        first.next = first.next.next;
        return head;
    }
}
