class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null) {
            return null;
        }
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode cur = dummy, var1 = dummy;
        for (int i = 1; i <= n + 1; i++) {
            cur = cur.next;
        }
        while (cur != null) {
            cur = cur.next;
            var1 = var1.next;
        }
        var1.next = var1.next.next;
        return dummy.next;
    }
}
