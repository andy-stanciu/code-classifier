class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fh = new ListNode(-1, head);
        int len = 0;
        for (; head != null; len++, head = head.next) ;
        for (len -= n, head = fh; len-- != 0; head = head.next) ;
        head.next = head.next.next;
        return fh.next;
    }
}
