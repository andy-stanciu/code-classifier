class Solution {

    public ListNode method1(ListNode head, int n) {
        //find length of tge linked list
        int length = 1;
        ListNode curr = head;
        while (curr.next != null) {
            curr = curr.next;
            length++;
        }
        ListNode temp = head;
        ListNode prev = null;
        for (int i = 0; i < length - n; i++) {
            prev = temp;
            temp = temp.next;
        }
        if (length - n == 0) {
            temp = temp.next;
            return temp;
        } else {
            prev.next = temp.next;
        }
        return head;
    }
}
