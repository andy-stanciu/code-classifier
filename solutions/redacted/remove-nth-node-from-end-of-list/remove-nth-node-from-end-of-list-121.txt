class Solution {

    int count = 0;

    int reverse = 0;

    public ListNode method1(ListNode head, int n) {
        // moving forward
        if (head == null)
            return null;
        count++;
        method1(head.next, n);
        // moving backward
        reverse++;
        if (reverse == n + 1) {
            head.next = head.next.next;
            reverse -= count;
            return head;
        }
        if (reverse == count)
            return head.next;
        return head;
    }
}
