class Solution {

    public ListNode method1(ListNode head, int n) {
        //IF THE LINKEDLIST IS EMPTY RETURN \'NULL\'
        if (head == null)
            return null;
        int count = 0;
        ListNode refPtr = head;
        ListNode mainPtr = head;
        //MOVING A REFERENCE n STEPS AHEAD.
        while (count < n) {
            refPtr = refPtr.next;
            count++;
        }
        //IF LIST HAS ONLY ONE ELEMENT
        if (refPtr == null)
            return head.next;
        while (refPtr.next != null) {
            refPtr = refPtr.next;
            mainPtr = mainPtr.next;
        }
        //BREAKING THE LINK.
        mainPtr.next = mainPtr.next.next;
        return head;
    }
}
