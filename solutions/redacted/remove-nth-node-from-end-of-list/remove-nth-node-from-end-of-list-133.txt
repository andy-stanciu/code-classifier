class Solution {

    public ListNode method1(ListNode head, int n) {
        ArrayList<Integer> arr = new ArrayList<>();
        ListNode temp = head;
        while (temp != null) {
            arr.add(temp.val);
            temp = temp.next;
        }
        arr.remove(arr.size() - n);
        ListNode temp1 = new ListNode(0);
        ListNode temp2 = temp1;
        for (int k : arr) {
            temp2.next = new ListNode(k);
            temp2 = temp2.next;
        }
        temp2.next = null;
        return temp1.next;
    }
}
