/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public static int GetLen(ListNode head) {
        int lens = 0;
        ListNode temp = head;
        while (temp != null) {
            temp = temp.next;
            lens += 1;
        }
        return lens;
    }

    public ListNode method1(ListNode head, int n) {
        int leng = GetLen(head);
        ListNode temp = head;
        int i = 1;
        leng = leng - n + 1;
        if (leng == 1) {
            head = head.next;
            return head;
        }
        while (i < leng - 1) {
            temp = temp.next;
            i++;
        }
        ListNode dels = temp.next;
        if (temp.next != null)
            temp.next = temp.next.next;
        return head;
    }
}
