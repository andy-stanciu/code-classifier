class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head == null)
            return head;
        int k = 0;
        ListNode curr = head;
        for (; curr != null; curr = curr.next) k++;
        //to get position of given element from first node
        k -= n;
        if (k == 0)
            return head.next;
        for (curr = head; k > 1; k--, curr = curr.next) ;
        // to remove that node
        curr.next = curr.next.next;
        return head;
    }
}
