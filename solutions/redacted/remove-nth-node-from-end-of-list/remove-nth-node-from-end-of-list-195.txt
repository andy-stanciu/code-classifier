class Solution {

    public ListNode method1(ListNode head, int n) {
        int count = 0;
        ListNode start = new ListNode(0);
        start.next = head;
        ListNode pointer = start;
        while (pointer != null) {
            pointer = pointer.next;
            //We\'ll get the size of the given LinkedList
            count++;
        }
        ListNode pointer2 = start;
        for (int i = 1; i < count - n; i++) {
            // n is position of the node that is to be deleted from the end so, count - n will give position of the same from the beginning
            pointer2 = pointer2.next;
            // Starting from node 1 , it\'ll go till the node just before the node we want to delete
            // i.e. 0 -> 1 -> 2 -> 3 -> 4 -> 5
            //                     V
            //                  pointer2
        }
        // 0 -> 1 -> 2 -> 3 -> 5
        pointer2.next = pointer2.next.next;
        // 1 -> 2 -> 3 -> 5
        return start.next;
    }
}
