class Solution {

    public ListNode method1(ListNode head, int n) {
        if (head.next == null)
            return null;
        // Step-1 : get the size of the linked list
        int size = 0;
        ListNode temp = head;
        while (temp != null) {
            size++;
            temp = temp.next;
        }
        // if we need to delete the head node
        if (n == size) {
            head = head.next;
            return head;
        }
        // Step-2 : get previous node of the node we want to delete
        ListNode var1 = head;
        for (int i = 1; i < size - n; i++) {
            var1 = var1.next;
        }
        // Step-3 : delete the node (check if we are deleting the tail node)
        ListNode var2 = var1.next;
        if (var2.next == null)
            var1.next = null;
        else
            var1.next = var2.next;
        return head;
    }
}
