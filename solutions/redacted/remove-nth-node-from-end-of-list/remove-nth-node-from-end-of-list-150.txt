class Solution {

    public ListNode method1(ListNode head, int n) {
        head = method2(head);
        ListNode tempHead = head;
        if (n == 1)
            head = head.next;
        else {
            n = n - 2;
            while (n-- > 0) tempHead = tempHead.next;
            tempHead.next = tempHead.next.next;
        }
        head = method2(head);
        return head;
    }

    public ListNode method2(ListNode head) {
        ListNode newHead = null, next = null;
        while (head != null) {
            next = head.next;
            head.next = newHead;
            newHead = head;
            head = next;
        }
        return newHead;
    }
}
