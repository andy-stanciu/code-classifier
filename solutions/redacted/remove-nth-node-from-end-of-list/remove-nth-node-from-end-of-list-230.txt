class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode fast = head, slow = head;
        while (n-- != 0) fast = fast.next;
        if (fast != null) {
            while (fast.next != null) {
                fast = fast.next;
                slow = slow.next;
            }
            slow.next = slow.next.next;
        } else
            head = head.next;
        return head;
    }
}
