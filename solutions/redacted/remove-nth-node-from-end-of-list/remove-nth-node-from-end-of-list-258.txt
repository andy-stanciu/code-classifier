class Solution {

    public ListNode method1(ListNode head, int n) {
        ListNode res = new ListNode(-1, head);
        ListNode first = res, second = res;
        while (n-- > 0) second = second.next;
        while (second.next != null) {
            second = second.next;
            first = first.next;
        }
        first.next = first.next.next;
        return res.next;
    }
}
