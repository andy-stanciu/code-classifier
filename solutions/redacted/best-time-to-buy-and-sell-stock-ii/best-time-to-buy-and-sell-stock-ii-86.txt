class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int var2 = 0;
        int var3, var4;
        for (int i = prices.length - 1; i >= 0; i--) {
            var4 = Math.max(prices[i] + var2, 0 + var1);
            var3 = Math.max(-prices[i] + var1, 0 + var2);
            var1 = var4;
            var2 = var3;
        }
        return var2;
    }
}
