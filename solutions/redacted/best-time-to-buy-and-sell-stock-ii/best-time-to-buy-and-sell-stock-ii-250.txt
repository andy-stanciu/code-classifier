class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        int n = prices.length;
        for (int i = 1; i < n; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            } else if (0 < (prices[i] - var1)) {
                profit += (prices[i] - var1);
                var1 = prices[i];
            }
        }
        return profit;
    }
}
