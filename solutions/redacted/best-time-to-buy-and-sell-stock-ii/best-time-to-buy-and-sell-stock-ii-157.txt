/**
 * Idea: Sell a stock when prices[i]>buy and then buy it again that day to handle continous upward trend
 *       Update the buy prices if we see a lower price
 */
class Solution {

    public int maxProfit(int[] prices) {
        int mp = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else {
                mp += prices[i] - var1;
                var1 = prices[i];
            }
        }
        return mp;
    }
}
