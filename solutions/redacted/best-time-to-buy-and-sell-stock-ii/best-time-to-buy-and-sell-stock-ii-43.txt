class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int maxProfit = 0;
        int var1 = 0;
        int var2 = 0;
        for (int i = 1; i < n; i++) {
            if (prices[i] < prices[i - 1]) {
                maxProfit += prices[var2] - prices[var1];
                var1 = i;
            }
            var2 = i;
        }
        maxProfit += prices[var2] - prices[var1];
        return maxProfit;
    }
}
