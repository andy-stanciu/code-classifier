class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else if (prices[i] > var1) {
                profit += prices[i] - var1;
                var1 = prices[i];
            }
        }
        return profit;
    }
}
