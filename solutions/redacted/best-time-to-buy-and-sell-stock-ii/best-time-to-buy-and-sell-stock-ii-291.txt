class Solution {

    public int maxProfit(int[] prices) {
        int max = 0, min = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (min > prices[i]) {
                min = prices[i];
            } else {
                max += prices[i] - min;
                min = prices[i];
            }
        }
        return max;
    }
}
