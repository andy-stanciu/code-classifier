class Solution {

    public int maxProfit(int[] prices) {
        // Initialize the profit to 0
        int profit = 0;
        // Get the length of the prices array
        int len = prices.length;
        // Iterate over the prices array
        for (int i = 1; i < len; i++) {
            // If the previous price is less than the current price, calculate the profit that would be made
            // by buying at the previous price and selling at the current price, and add it to the total profit
            if (prices[i - 1] < prices[i])
                profit += prices[i] - prices[i - 1];
        }
        // Return the total profit
        return profit;
    }
}
