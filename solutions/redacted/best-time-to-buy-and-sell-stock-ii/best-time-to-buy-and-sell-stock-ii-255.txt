class Solution {

    public int maxProfit(int[] prices) {
        // [7,1,5,3,6,4]
        int result = 0;
        // currentBuy
        int leftPointer = 0;
        // possibleSell
        int rightPointer = 1;
        while (rightPointer < prices.length) {
            int currentProfit = prices[rightPointer] - prices[leftPointer];
            if (currentProfit == 0) {
                rightPointer++;
            } else if (currentProfit > 0) {
                result = result + currentProfit;
                currentProfit = 0;
                leftPointer = rightPointer;
            } else {
                leftPointer = rightPointer;
                rightPointer++;
            }
            // calculate current profit (right val - left val)
            // if currentprofit is 0
            // increment right
            // else if the currentProfit is over zero
            // add it to the result;
            // set left to right
            // else, we have a negative result (means were on an element lower that our current buy)
            // set left pointer to right pointer
            // increment right pointer
        }
        return result;
    }
}
