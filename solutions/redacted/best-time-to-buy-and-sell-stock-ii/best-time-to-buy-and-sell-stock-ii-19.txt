class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = prices[0];
        for (int i : prices) {
            if (i < var1) {
                var1 = i;
            } else if (i > var1) {
                profit = profit + i - var1;
                var1 = i;
            }
        }
        return profit;
    }
}
