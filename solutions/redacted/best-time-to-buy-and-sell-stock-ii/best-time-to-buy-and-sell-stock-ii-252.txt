class Solution {

    public int maxProfit(int[] prices) {
        int p1 = 0;
        if (prices.length == 1) {
            p1 = 0;
        } else {
            for (int i = 0; i < prices.length - 1; i++) {
                int temp = prices[i + 1] - prices[i];
                p1 = Math.max(temp + p1, p1);
            }
        }
        return p1;
    }
}
