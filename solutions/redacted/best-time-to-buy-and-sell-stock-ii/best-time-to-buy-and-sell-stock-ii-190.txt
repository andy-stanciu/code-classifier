class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        for (int i = 1; i < prices.length; i++) {
            // If todays profit is greater than yesterday we make profit
            if (prices[i] > prices[i - 1]) {
                var1 += prices[i] - prices[i - 1];
            }
        }
        return var1;
    }
}
