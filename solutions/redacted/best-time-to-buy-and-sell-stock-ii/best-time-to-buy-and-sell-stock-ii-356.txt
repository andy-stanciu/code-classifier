class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int netprofit = 0;
        for (int i = 1; i < n; i++) {
            int profit = 0;
            int var1 = prices[i - 1];
            profit = Math.max(profit, prices[i] - var1);
            netprofit += profit;
        }
        return netprofit;
    }
}
