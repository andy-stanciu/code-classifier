class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length, minSoFar = prices[0], res = 0;
        for (int i = 1; i < n; i++) {
            int profit = prices[i] - minSoFar;
            minSoFar = Math.min(minSoFar, prices[i]);
            if (profit > 0) {
                res += profit;
                minSoFar = prices[i];
            }
        }
        return res;
    }
}
