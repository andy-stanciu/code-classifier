class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int i = 1;
        while (i < prices.length) {
            int currPrice = prices[i];
            if (currPrice > prices[i - 1]) {
                int currProfit = currPrice - prices[i - 1];
                maxProfit = maxProfit + currProfit;
            }
            i++;
        }
        return maxProfit;
    }
}
