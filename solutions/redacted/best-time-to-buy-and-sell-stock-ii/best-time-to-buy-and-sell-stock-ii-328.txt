class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        //sell in same day if no profit happens in next day
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (var1 < prices[i]) {
                //profit
                int profit = prices[i] - var1;
                maxProfit += profit;
                //as we can hold max one stock only
                var1 = prices[i];
            } else {
                //if no profit sell in same day or not buy stock
                var1 = prices[i];
            }
            //means if we update max profit then go ahead if not then also go ahead
        }
        return maxProfit;
    }
}
