class Solution {

    public int maxProfit(int[] prices) {
        int ans = 0;
        int var1 = prices[0];
        int maxProfit = Integer.MIN_VALUE;
        for (int i = 0; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            maxProfit = Math.max(maxProfit, prices[i] - var1);
            if (maxProfit > 0) {
                ans = ans + maxProfit;
                var1 = prices[i];
                maxProfit = 0;
            }
        }
        return ans;
    }
}
