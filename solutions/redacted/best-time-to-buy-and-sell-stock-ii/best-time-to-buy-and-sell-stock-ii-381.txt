class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int bought = 0;
        for (int i = 0; i < prices.length; i++) {
            if (i < prices.length - 1 && prices[i] < prices[i + 1] && bought == 0) {
                bought = 1;
                profit -= prices[i];
                continue;
            }
            if (bought == 1 && i < prices.length - 1 && prices[i] > prices[i + 1]) {
                bought = 0;
                profit += prices[i];
                continue;
            }
            if (bought == 1 && i == prices.length - 1) {
                profit += prices[i];
            }
        }
        return profit;
    }
}
