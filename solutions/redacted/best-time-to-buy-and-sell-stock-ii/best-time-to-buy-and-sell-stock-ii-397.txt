class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = 0;
        int var2 = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            var1 = prices[i];
            var2 = prices[i + 1];
            if (var1 <= var2) {
                maxProfit += (var2 - var1);
            }
        }
        return maxProfit;
    }
}
