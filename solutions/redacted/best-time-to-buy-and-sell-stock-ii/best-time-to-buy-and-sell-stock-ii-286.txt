class Solution {

    public int maxProfit(int[] prices) {
        int max = 0;
        int var1 = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] > var1) {
                max += prices[i] - var1;
            }
            var1 = prices[i];
        }
        return max;
    }
}
