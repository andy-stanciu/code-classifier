class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        //just count profit we can make after buyinh and selling at one step
        //if we making profit at one step sell it and do same for later
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] > prices[i - 1]) {
                maxProfit += prices[i] - prices[i - 1];
            }
        }
        return maxProfit;
    }
}
