class Solution {

    public int maxProfit(int[] prices) {
        int lp = 0, rp = 1, sum = 0;
        while (rp < prices.length) {
            if (prices[rp] - prices[lp] < 0) {
                rp++;
                lp++;
            } else {
                sum += (prices[rp] - prices[lp]);
                lp++;
                rp++;
            }
        }
        return sum;
    }
}
