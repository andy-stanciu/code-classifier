class Solution {

    public static int maxProfit(int[] prices) {
        int ans = Profit(prices, 0);
        return ans;
    }

    public static int Profit(int[] prices, int index) {
        int maxProfit = 0;
        if (index + 1 < prices.length) {
            int current = prices[index];
            int next = prices[index + 1];
            if (current > next) {
                return Profit(prices, index + 1);
            } else {
                int diff = next - current;
                maxProfit += diff;
                return maxProfit + Profit(prices, index + 1);
            }
        }
        return maxProfit;
    }
}
