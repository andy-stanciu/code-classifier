// Buy and Sell Stock Pattern
// Approach - 2 Tabulation
class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int[][] dp = new int[n + 1][2];
        // defaulted but just note the conversion from memoization
        dp[n][0] = 0;
        // defaulted but just note the conversion from memoization
        dp[n][1] = 0;
        int profit = 0;
        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j <= 1; j++) {
                //NOTE case for but and sell    j==0
                if (j == 0) {
                    profit = Math.max(0 + dp[i + 1][0], -prices[i] + dp[i + 1][1]);
                }
                if (j == 1) {
                    profit = Math.max(0 + dp[i + 1][1], prices[i] + dp[i + 1][0]);
                }
                dp[i][j] = profit;
            }
        }
        return dp[0][0];
    }
}
