class Solution {

    public int maxProfit(int[] prices) {
        int minPrice = prices[0], maxProfit = 0, var1 = 0, cost = 0;
        for (int i = 0; i < prices.length; i++) {
            minPrice = Math.min(minPrice, prices[i]);
            var1 = prices[i];
            cost = var1 - minPrice;
            if (cost > 0) {
                maxProfit += cost;
                minPrice = var1;
            }
        }
        return maxProfit;
    }
}
