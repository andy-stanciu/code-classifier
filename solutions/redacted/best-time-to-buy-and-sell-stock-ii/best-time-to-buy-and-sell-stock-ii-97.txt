class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 0) {
            return 0;
        }
        int min = prices[0];
        int var1 = 0;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < min) {
                //min check
                min = prices[i];
            }
            if (profit < prices[i] - min) {
                //profit check
                profit = prices[i] - min;
            } else {
                //as soon as when profit is less then the previous price we will start from new profit
                min = prices[i];
                var1 += profit;
                profit = 0;
            }
            if (i == prices.length - 1) {
                var1 += profit;
            }
        }
        return var1;
    }
}
