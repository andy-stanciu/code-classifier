class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length, min = Integer.MAX_VALUE, max = -1, maxRes = 0, sum = 0;
        for (int i = 0; i < n; i++) {
            if (prices[i] < min) {
                max = prices[i];
                min = max;
            } else if (prices[i] > max) {
                sum += prices[i] - min;
                min = prices[i];
                max = min;
            }
        }
        return sum;
    }
}
