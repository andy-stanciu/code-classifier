class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int min = prices[0];
        int max = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] > max) {
                max = prices[i];
            } else {
                if (max - min > 0) {
                    profit = profit + max - min;
                }
                min = prices[i];
                max = prices[i];
            }
            if (prices[i] < min) {
                min = prices[i];
                max = prices[i];
            }
        }
        profit = profit + max - min;
        return profit;
    }
}
