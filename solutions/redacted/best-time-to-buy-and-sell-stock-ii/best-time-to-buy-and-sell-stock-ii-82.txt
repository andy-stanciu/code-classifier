class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0, profit = 0, i = 0;
        while (i < prices.length) {
            var1 = prices[i];
            i++;
            while (i < prices.length && prices[i - 1] <= prices[i]) i++;
            profit += prices[i - 1] - var1;
        }
        return profit;
    }
}
