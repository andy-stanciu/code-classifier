class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int var2 = 0;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            while (i < prices.length - 1 && prices[i + 1] <= prices[i]) i++;
            var1 = prices[i];
            while (i < prices.length - 1 && prices[i + 1] > prices[i]) i++;
            var2 = prices[i];
            profit += var2 - var1;
        }
        return profit;
    }
}
