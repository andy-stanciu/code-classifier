class Solution {

    public int maxProfit(int[] prices) {
        int cost = 0;
        int var1 = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] - prices[var1] > 0) {
                cost += prices[i] - prices[var1];
                var1 = i;
            }
            if (prices[i] < prices[var1]) {
                var1 = i;
            }
        }
        return cost;
    }
}
