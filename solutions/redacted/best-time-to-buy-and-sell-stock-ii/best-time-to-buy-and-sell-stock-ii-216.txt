class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        // base case for day 1
        // do nothing
        int var1 = 0;
        // buy
        int var2 = -prices[0];
        for (int i = 1; i < n; i++) {
            // do nothing or sell
            var1 = Math.max(var1, var2 + prices[i]);
            // do nothing or buy
            var2 = Math.max(var2, var1 - prices[i]);
        }
        return var1;
    }
}
