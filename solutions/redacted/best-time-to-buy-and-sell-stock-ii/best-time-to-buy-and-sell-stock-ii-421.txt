class Solution {

    public int maxProfit(int[] prices) {
        int sum = 0;
        int cur = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (cur < prices[i]) {
                sum += (prices[i] - cur);
            }
            cur = prices[i];
        }
        return sum;
    }
}
