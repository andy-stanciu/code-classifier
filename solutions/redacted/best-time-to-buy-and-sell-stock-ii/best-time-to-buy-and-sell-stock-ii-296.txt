class Solution {

    public int maxProfit(int[] prices) {
        int left = 0;
        int count = 0;
        int var1 = prices[0];
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] > var1) {
                count += prices[i] - var1;
            }
            var1 = prices[i];
        }
        return count;
    }
}
