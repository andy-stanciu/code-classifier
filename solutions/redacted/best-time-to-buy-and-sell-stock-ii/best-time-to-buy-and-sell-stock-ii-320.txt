class Solution {

    public int maxProfit(int[] prices) {
        int var1 = -prices[0];
        int var2 = 0;
        for (int i = 1; i < prices.length; i++) {
            int var3 = Math.max(var1, var2 - prices[i]);
            int var4 = Math.max(var2, var1 + prices[i]);
            var1 = var3;
            var2 = var4;
        }
        return var2;
    }
}
