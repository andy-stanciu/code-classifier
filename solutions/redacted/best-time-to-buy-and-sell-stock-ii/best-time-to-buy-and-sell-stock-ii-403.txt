class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0], profit = 0, var2 = 0;
        int n = prices.length;
        for (int i = 1; i < n; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else {
                profit = prices[i] - var1;
                var1 = prices[i];
                var2 += profit;
            }
        }
        return var2;
    }
}
