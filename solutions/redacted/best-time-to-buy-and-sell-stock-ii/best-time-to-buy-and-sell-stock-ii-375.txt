class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int sum = 0;
        for (int i = 0; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            if (prices[i] > var1) {
                sum = sum + prices[i] - var1;
                var1 = prices[i];
            }
        }
        return sum;
    }
}
