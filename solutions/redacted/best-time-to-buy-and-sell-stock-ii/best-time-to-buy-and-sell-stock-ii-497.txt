class Solution {

    public int maxProfit(int[] prices) {
        int least = 0;
        int sum = 0;
        for (int i = 1; i < prices.length; i++) {
            least = Math.min(prices[i], prices[i - 1]);
            int profit = 0;
            profit = Math.max(profit, prices[i] - least);
            sum += profit;
        }
        return sum;
    }
}
