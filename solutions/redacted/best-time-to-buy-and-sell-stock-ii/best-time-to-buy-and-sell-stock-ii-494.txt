class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0], var2 = prices[0], profit = 0;
        for (int i = 0; i < prices.length - 1; i++) {
            if (prices[i] > prices[i + 1]) {
                var1 = prices[i + 1];
            } else {
                var1 = var2;
            }
            if (prices[i + 1] > prices[i]) {
                var2 = prices[i + 1];
            } else {
                var2 = var1;
            }
            profit = profit + var2 - var1;
        }
        return profit;
    }
}
