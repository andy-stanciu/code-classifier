class Solution {

    public int maxProfit(int[] prices) {
        // Variable to store the total accumulated profit.
        int var1 = 0;
        // Start iterating from the second day (i=1) to compare with the previous day.
        for (int i = 1; i < prices.length; i++) {
            // If today\'s price is greater than yesterday\'s price, add the difference to the profit.
            if (prices[i] > prices[i - 1]) {
                // The profit made by buying yesterday and selling today.
                var1 += (prices[i] - prices[i - 1]);
            }
        }
        // Return the total accumulated profit.
        return var1;
    }
}
