class Solution {

    public int maxProfit(int[] prices) {
        int length = prices.length;
        if (prices.length == 0 || prices.length == 1) {
            return 0;
        }
        int count = 0;
        for (int i = 1; i < length; i++) {
            if (prices[i] > prices[i - 1]) {
                count = count + (prices[i] - prices[i - 1]);
            }
        }
        return count;
    }
}
