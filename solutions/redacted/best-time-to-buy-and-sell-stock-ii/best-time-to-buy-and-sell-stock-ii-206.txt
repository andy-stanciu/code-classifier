class Solution {

    public int maxProfit(int[] prices) {
        // Accumulate each arise price everyday
        int profit = 0;
        for (int var2 = 1; var2 < prices.length; var2++) {
            int var1 = prices[var2] - prices[var2 - 1];
            if (var1 > 0) {
                profit += var1;
            }
        }
        return profit;
    }
}
