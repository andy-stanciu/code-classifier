class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int var1 = prices[0];
        int days = prices.length;
        for (int i = 1; i < days; i++) {
            if (var1 < prices[i])
                profit += prices[i] - var1;
            var1 = prices[i];
        }
        return profit;
    }
}
