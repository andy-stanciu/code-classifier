class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int maxprofit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (var1 < prices[i]) {
                int profit = prices[i] - var1;
                maxprofit = maxprofit + profit;
                var1 = prices[i];
            } else {
                var1 = prices[i];
            }
        }
        return maxprofit;
    }
}
