class Solution {

    public int maxProfit(int[] arr) {
        int length = arr.length;
        int var1 = 0;
        int var2 = 0;
        for (int i = 0; i < length - 1; i++) {
            // If the next day\'s price is higher, we have an opportunity to buy
            if (arr[i] < arr[i + 1]) {
                var1 = arr[i];
                // Find the local maximum (highest selling point)
                while ((i < length - 1) && (arr[i] <= arr[i + 1])) {
                    i++;
                }
                // Calculate profit for this transaction
                var2 += arr[i] - var1;
            }
        }
        // Return the total accumulated profit
        return var2;
    }
}
