class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int var1 = prices[0];
        int profit = 0;
        for (int i = 1; i < n; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            }
            if (prices[i] - var1 > 0) {
                profit += prices[i] - var1;
                var1 = prices[i];
            }
        }
        return profit;
    }
}
