class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        // here i represents the day of buying the stock
        int i = 0;
        // j represents the day of selling the stock
        int j = i;
        while (i < prices.length && j < prices.length) {
            // keep shifting selling day to future as price keeps increasing
            while (j + 1 < prices.length && prices[j + 1] > prices[j]) {
                j++;
            }
            // sell the stock and calculate profit
            profit += prices[j] - prices[i];
            // now buy stock on the next day again
            i = j + 1;
            j = i;
        }
        return profit;
    }
}
