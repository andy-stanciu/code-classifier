class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int[] prev = new int[2];
        prev[0] = prev[1] = 0;
        for (int idx = n - 1; idx >= 0; idx--) {
            for (int var1 = 0; var1 <= 1; var1++) {
                int profit = 0;
                if (var1 == 1) {
                    profit = Math.max(-prices[idx] + prev[0], prev[1]);
                } else {
                    profit = Math.max(prices[idx] + prev[1], prev[0]);
                }
                prev[var1] = profit;
            }
        }
        return prev[1];
    }
}
