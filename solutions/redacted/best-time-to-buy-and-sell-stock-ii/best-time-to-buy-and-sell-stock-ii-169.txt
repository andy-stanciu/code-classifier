class Solution {

    public int maxProfit(int[] prices) {
        // greedy approach to sell whenever profit in array
        int maxProfit = 0;
        int l = prices.length;
        if (l < 2)
            return 0;
        int var1 = prices[0];
        int profit = 0;
        for (int i = 0; i < l; i++) {
            int x = prices[i];
            if (x < var1) {
                var1 = x;
            } else if (x - var1 > profit) {
                profit = x - var1;
                maxProfit += profit;
                var1 = Integer.MAX_VALUE;
                profit = 0;
                i--;
            }
        }
        return maxProfit;
    }
}
