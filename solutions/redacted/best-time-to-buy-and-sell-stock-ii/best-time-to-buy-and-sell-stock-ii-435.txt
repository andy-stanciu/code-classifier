class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int max = prices[0];
        int var1 = 0;
        for (int i : prices) {
            if (i > max) {
                max = i;
            }
            if (i < max) {
                var1 += max - min;
                min = i;
                max = i;
            }
        }
        var1 += max - min;
        return var1;
    }
}
