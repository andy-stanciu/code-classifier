class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        int counter = 0;
        for (int x : nums) {
            if (counter >= k) {
                if (x > var1.peek()) {
                    var1.poll();
                    var1.add(x);
                }
            } else
                var1.add(x);
            counter++;
        }
        return var1.peek();
    }
}
