class Solution {

    public int method1(int[] nums, int k) {
        // Arrays.sort(nums);
        // return nums[nums.length-k];
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int val : nums) {
            pq.add(val);
        }
        int x = nums.length - k;
        while (x > 0) {
            pq.remove();
            x--;
        }
        return pq.peek();
    }
}
