class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> queue = new PriorityQueue<>(nums.length);
        for (int i = 0; i < k; i++) {
            queue.offer(nums[i]);
        }
        for (int i = k; i < nums.length; i++) {
            if (queue.peek() <= nums[i]) {
                queue.remove();
                queue.offer(nums[i]);
            }
        }
        return queue.peek();
    }
}
