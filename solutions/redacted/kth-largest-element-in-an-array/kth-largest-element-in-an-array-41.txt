class Solution {

    public int method1(int[] nums, int k) {
        Queue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            if (var1.size() < k || var1.peek() < nums[i]) {
                var1.add(nums[i]);
            }
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.peek();
    }
}
