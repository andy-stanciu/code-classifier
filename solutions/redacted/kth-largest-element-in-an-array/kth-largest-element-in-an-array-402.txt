class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            pq.add(nums[i]);
        }
        k = nums.length - k;
        int count = 0;
        int el = 0;
        while (!pq.isEmpty()) {
            if (count == k) {
                el = pq.poll();
            }
            count++;
            pq.poll();
        }
        return el;
    }
}
