class Solution {

    public int method1(int[] nums, int k) {
        var heap = new PriorityQueue<Integer>();
        if (nums.length == 0 || k <= 0) {
            return 0;
        }
        for (int n : nums) {
            if (heap.size() < k) {
                heap.offer(n);
            } else if (n > heap.peek()) {
                heap.poll();
                heap.offer(n);
            }
        }
        return heap.peek();
    }
}
