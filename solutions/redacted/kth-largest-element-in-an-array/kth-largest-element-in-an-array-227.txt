class Solution {

    public int method1(int[] nums, int k) {
        // Sort the array in descending order
        // Sorts in ascending order by default
        Arrays.sort(nums);
        // To sort in descending order, reverse the array after sorting
        for (int i = 0; i < nums.length / 2; i++) {
            int temp = nums[i];
            nums[i] = nums[nums.length - i - 1];
            nums[nums.length - i - 1] = temp;
        }
        // Return the k-th largest element (k-th index is k-1 in zero-indexed arrays)
        return nums[k - 1];
    }
}
