class Solution {

    public static boolean method1(int mid, int number, int[] nums) {
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (mid > nums[i]) {
                count++;
            }
        }
        if (count <= number)
            return true;
        else
            return false;
    }

    public int method2(int[] nums, int k) {
        int n = nums.length;
        int number = n - k;
        //min element of array
        int start = Integer.MAX_VALUE;
        //max element of array
        int end = Integer.MIN_VALUE;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < start)
                start = nums[i];
            if (nums[i] > end)
                end = nums[i];
        }
        int mid = 0;
        while (start <= end) {
            mid = start + ((end - start) / 2);
            if (method1(mid, number, nums)) {
                var1 = mid;
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return var1;
    }
}
