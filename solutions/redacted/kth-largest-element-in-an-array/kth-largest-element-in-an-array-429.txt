class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            var1.offer(nums[i]);
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.peek();
    }
}
