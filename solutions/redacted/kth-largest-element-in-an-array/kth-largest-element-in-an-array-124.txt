class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int i = 0;
        while (i < k) {
            pq.add(nums[i++]);
        }
        while (i < nums.length) {
            int top = pq.peek();
            if (top < nums[i]) {
                pq.poll();
                pq.add(nums[i]);
            }
            i++;
        }
        return pq.peek();
    }
}
