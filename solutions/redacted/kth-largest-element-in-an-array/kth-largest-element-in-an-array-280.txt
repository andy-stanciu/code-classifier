class Solution {

    //    private PriorityQueue<Integer> minHeap = new PriorityQueue<>();
    //    private int k;
    public int method1(int[] nums, int k) {
        List<Integer> li = new ArrayList<>();
        for (int i : nums) {
            li.add(i);
        }
        //  minHeap.add(0)
        PriorityQueue<Integer> pq = new PriorityQueue<>(li);
        int el = 0;
        while (pq.size() > k) {
            pq.poll();
        }
        return pq.peek();
    }
}
