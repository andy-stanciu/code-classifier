class Solution {

    public int method1(int[] nums, int k) {
        // Create a min heap and if size increases than k then remove and return the top element ,the top element is kth largest element
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            var1.add(nums[i]);
            if (var1.size() > k)
                var1.remove();
        }
        /* while(k!=1){
        maxheap.remove();
        k--;
      }*/
        return var1.peek();
    }
}
