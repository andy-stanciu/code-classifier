class Solution {

    public int method1(int[] nums, int k) {
        int length = nums.length - k + 1;
        PriorityQueue<Integer> priority = new PriorityQueue<>();
        for (int num : nums) {
            priority.add(num);
        }
        int result = 0;
        for (int i = 0; i < length; i++) {
            result = priority.poll();
        }
        return result;
    }
}
