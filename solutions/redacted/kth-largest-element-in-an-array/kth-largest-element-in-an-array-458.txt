class Solution {

    public int method1(int[] nums, int k) {
        int var1 = 0;
        PriorityQueue<Integer> var2 = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            var2.add(nums[i]);
        }
        for (int i = k; i < nums.length; i++) {
            if (var2.peek() < nums[i]) {
                var2.poll();
                var2.add(nums[i]);
            }
        }
        return var2.peek();
    }
}
