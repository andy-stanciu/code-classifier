class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            var1.add(nums[i]);
        }
        int var2 = k;
        while (var2 < nums.length) {
            if (var1.peek() < nums[var2]) {
                var1.poll();
                var1.add(nums[var2]);
            }
            var2++;
        }
        return var1.peek();
    }
}
