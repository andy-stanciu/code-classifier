class Solution {

    public int method1(int[] nums, int k) {
        // PQ that returns largest element
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            pq.add(nums[i]);
            // keep removing smaller elemnts if size of pq increases by k
            if (pq.size() > k) {
                pq.poll();
            }
        }
        // return the kth largest element
        return pq.peek();
    }
}
