class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            // add first K element
            pq.add(nums[i]);
        }
        for (int i = k; i < nums.length; i++) {
            if (pq.peek() < nums[i]) {
                // check the min element of heap is less or not then next element of array
                // if yes then remove the min
                pq.remove();
                // and add the elment in heap and repeat until heap has all k  maximum elemnet of array
                pq.add(nums[i]);
            }
        }
        // return the min of heap which is the kth largest element from last
        return pq.peek();
    }
}
