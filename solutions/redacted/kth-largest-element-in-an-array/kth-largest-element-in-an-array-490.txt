// Brute Force: Sorting the arr, TC: O NLogN
// Optimized: Using Heap, TC: O NLogK
class Solution {

    public int method1(int[] nums, int k) {
        int n = nums.length;
        // Mind Heap
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < n; i++) {
            var1.add(nums[i]);
            if (var1.size() > k) {
                // Keeping the size K , so at the top kth largest element will be there
                var1.poll();
            }
        }
        return var1.peek();
    }
}
