class Solution {

    public int method1(int[] nums, int k) {
        for (int i = 0; i < k; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[j] > nums[i]) {
                    //System.out.println(nums[i]+" "+nums[j]);
                    int temp = nums[i];
                    nums[i] = nums[j];
                    nums[j] = temp;
                }
            }
            System.out.println(nums[i]);
        }
        return nums[k - 1];
    }
}
