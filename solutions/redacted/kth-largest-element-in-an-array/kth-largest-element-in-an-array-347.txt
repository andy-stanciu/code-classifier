class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>((a, b) -> b - a);
        for (int num : nums) {
            q.offer(num);
        }
        int var1 = 0;
        while (!q.isEmpty() && k-- > 0) {
            var1 = q.poll();
        }
        return var1;
    }
}
