public class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            var1.offer(nums[i]);
        }
        for (int i = k; i < nums.length; i++) {
            if (nums[i] > var1.peek()) {
                var1.poll();
                var1.offer(nums[i]);
            }
        }
        return var1.peek();
    }
}
