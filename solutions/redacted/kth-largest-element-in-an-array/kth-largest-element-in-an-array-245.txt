class Solution {

    public int method1(int[] nums, int k) {
        int var1 = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(Comparator.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            pq.add(nums[i]);
        }
        int j = 0;
        while (!pq.isEmpty() && j < k) {
            var1 = pq.remove();
            j++;
        }
        return var1;
    }
}
