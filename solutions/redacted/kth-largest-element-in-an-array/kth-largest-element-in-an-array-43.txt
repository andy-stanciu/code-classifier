class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        for (int num : nums) {
            pq.add(num);
        }
        //pop kth time and return
        int p = 0;
        for (int i = 1; i <= k; i++) {
            p = pq.poll();
            if (i == k)
                return p;
        }
        return p;
    }
}
