class Solution {

    static int count(int[] nums, int mid) {
        int count = 0;
        for (int i : nums) {
            if (i >= mid)
                count++;
        }
        return count;
    }

    public int method1(int[] nums, int k) {
        int var1 = Integer.MAX_VALUE, max = Integer.MIN_VALUE;
        int var2 = -1;
        for (int i : nums) {
            var1 = Math.var1(var1, i);
            max = Math.max(max, i);
        }
        while (var1 <= max) {
            int mid = var1 + (max - var1) / 2;
            if (count(nums, mid) < k) {
                max = mid - 1;
            } else {
                var2 = mid;
                var1 = mid + 1;
            }
        }
        return var2;
    }
}
