class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> p = new PriorityQueue<>(Comparator.reverseOrder());
        for (int n : nums) {
            p.add(n);
        }
        for (int i = 1; i < k; i++) {
            p.poll();
        }
        return p.poll();
    }
}
