class Solution {

    public int method1(int[] nums, int k) {
        // Min Heap
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < nums.length; i++) {
            if (pq.size() < k) {
                pq.add(nums[i]);
            } else if (pq.peek() < nums[i]) {
                // Adding Only when it will effect our result
                // removing the smallest and adding the greater element
                pq.poll();
                pq.add(nums[i]);
            }
        }
        return pq.peek();
    }
}
