class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        for (int i : nums) {
            q.add(i);
        }
        while (k > 1) {
            q.poll();
            k--;
        }
        return q.poll();
    }
}
