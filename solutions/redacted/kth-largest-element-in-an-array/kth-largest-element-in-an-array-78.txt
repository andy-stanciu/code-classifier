class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int num = 0; num < nums.length; num++) {
            var1.offer(nums[num]);
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.peek();
    }
}
