class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int num : nums) {
            var1.add(num);
            if (var1.size() > k) {
                var1.poll();
            }
        }
        return var1.peek();
    }

    public static void method2(String[] args) {
        Solution solution = new Solution();
        Scanner sc = new Scanner(System.in);
        int[] arr = new int[6];
        for (int i = 0; i < 6; i++) {
            arr[i] = sc.nextInt();
        }
        System.out.println("Enter k : ");
        int k = sc.nextInt();
        System.out.println("The " + k + " largest element is " + solution.method1(arr, k));
    }
}
