class Solution {

    public int method1(int[] nums, int k) {
        Queue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        for (int i = 0; i < nums.length; i++) pq.offer(nums[i]);
        int i = 1;
        while (i < k) {
            pq.poll();
            i++;
        }
        return pq.poll();
    }
}
