class Solution {

    public int method1(int[] nums, int k) {
        //Normal method- O(n log n)
        // Arrays.sort(nums);
        // int n=nums.length-k;
        // return nums[n];
        //MINHEAP method - O(k log n)
        PriorityQueue<Integer> var1 = new PriorityQueue<>();
        for (int num : nums) {
            var1.add(num);
            if (var1.size() > k) {
                // Remove the smallest element to maintain heap size k
                var1.poll();
            }
        }
        // The root of the heap is the kth largest element
        return var1.peek();
    }
}
