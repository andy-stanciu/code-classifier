class Solution {

    public int method1(int[] nums, int k) {
        //O(nlogk)- time and //O(k)-space
        PriorityQueue<Integer> heap = new PriorityQueue<Integer>();
        for (int num : nums) {
            heap.add(num);
        }
        while (heap.size() > k) {
            heap.poll();
        }
        return heap.peek();
    }
}
