class Solution {

    public int method1(int[] arr, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < arr.length; i++) {
            if (pq.size() == k) {
                if (pq.peek() < arr[i]) {
                    pq.remove();
                    pq.add(arr[i]);
                }
                continue;
            }
            pq.add(arr[i]);
        }
        return pq.peek();
    }
}
