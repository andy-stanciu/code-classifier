class Solution {

    public int method1(int[] nums, int k) {
        // Put K in priorityQ.
        // For loop through rest, add one, pop one.
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < k; i++) {
            pq.offer(nums[i]);
        }
        for (int i = k; i < nums.length; i++) {
            pq.offer(nums[i]);
            pq.poll();
        }
        return pq.poll();
        // If K > N - K, use max heap.
    }
}
