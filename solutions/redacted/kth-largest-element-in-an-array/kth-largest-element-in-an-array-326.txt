class Solution {

    public int method1(int[] nums, int k) {
        int[] arr = new int[22000];
        for (int i = 0; i < nums.length; i++) {
            arr[nums[i] + 11000]++;
        }
        for (int i = arr.length - 1; i >= 0; i--) {
            if (arr[i] != 0) {
                k = k - arr[i];
                if (k <= 0) {
                    return i - 11000;
                }
            }
        }
        return -1;
    }
}
