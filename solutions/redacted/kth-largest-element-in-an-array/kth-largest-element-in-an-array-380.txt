class Solution {

    public int method1(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        for (int i = 0; i < nums.length; i++) {
            q.offer(nums[i]);
        }
        int var1 = 1;
        while (var1 < k) {
            q.poll();
            var1++;
        }
        return q.poll();
    }
}
