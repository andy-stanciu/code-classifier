class Solution {

    public int method1(int[] nums, int k) {
        int t = nums.length - k;
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int l : nums) {
            pq.offer(l);
        }
        while (t > 0) {
            pq.poll();
            t--;
        }
        return pq.peek();
    }
}
