class Solution {

    public int method1(int[] nums) {
        int expectedSum = 0;
        int sum = 0;
        int len = nums.length;
        for (int i = 0; i <= len; i++) {
            expectedSum += i;
        }
        for (int num : nums) {
            sum += num;
        }
        int var1 = expectedSum - sum;
        return var1;
    }
}
