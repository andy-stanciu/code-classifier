class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        if (len == 0)
            return 0;
        int sum = 0, reqSum = ((len) * (len + 1)) / 2;
        for (int i = 0; i < len; i++) {
            sum += nums[i];
        }
        return reqSum - sum;
    }
}
