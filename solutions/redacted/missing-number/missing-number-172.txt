class Solution {

    public int method1(int[] nums) {
        //10
        int size = nums.length + 1;
        Set<Integer> hs = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            //[0,1,2,3,4,5,6,7,9]
            hs.add(nums[i]);
        }
        for (int i = 0; i < size; i++) {
            if (!hs.contains(i)) {
                //return 8
                return i;
            }
        }
        return -1;
    }
}
