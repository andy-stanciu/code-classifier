class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        if (nums[n - 1] != n) {
            return n;
        }
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] != (nums[i + 1] - 1)) {
                return (nums[i + 1] - 1);
            }
        }
        return 0;
    }
}
