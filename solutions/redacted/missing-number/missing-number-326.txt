class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        int Tsum = 0;
        for (int i = 1; i <= nums.length; i++) {
            Tsum = Tsum + i;
        }
        return (Tsum - sum);
    }
}
