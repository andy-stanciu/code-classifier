class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        // Calculates the sum of all the numbers present in the array
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        // Sn = n(n+1)/2 .This is the formula to calculate sum of \'n\' natural numbers.
        // After Calculating the Sn just subtract your sum from Sn you  will get the missing number.
        return (n * (n + 1)) / 2 - sum;
    }
}
