class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        // int expectedSum = n * (n + 1) / 2;
        // int actualSum = 0;
        // for (int num : nums) {
        //     actualSum += num;
        // }
        // return expectedSum - actualSum;
        int ans = 0;
        Arrays.sort(nums);
        for (int i = 0; i < n; i++) {
            if (nums[i] != i) {
                return i;
            }
        }
        return n;
    }
}
