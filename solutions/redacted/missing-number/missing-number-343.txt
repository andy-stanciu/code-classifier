class Solution {

    public int method1(int[] nums) {
        List<Integer> al = new ArrayList<>();
        for (int i : nums) {
            al.add(i);
        }
        // Collections.sort();
        int m = al.size();
        List<Integer> all = new ArrayList<>();
        for (int i = 1; i <= m; i++) {
            all.add(i);
        }
        all.removeAll(al);
        if (all.isEmpty())
            return 0;
        return all.get(0);
    }
}
