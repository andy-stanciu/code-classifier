class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int[] count = new int[n + 1];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                continue;
            }
            count[nums[i]]++;
        }
        for (int i = 1; i < count.length; i++) {
            if (count[i] == 0) {
                return i;
            }
        }
        return 0;
    }
}
