class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        Arrays.sort(nums);
        int k = 0;
        if (//this condition is only for the edge case
        nums[n - 1] < nums.length)
            return nums[n - 1] + 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1] + 1)
                k = nums[i - 1] + 1;
        }
        return k;
    }
}
