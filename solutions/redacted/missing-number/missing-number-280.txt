class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int actual = 0, sum = 0;
        for (int i = 0; i < nums.length; i++) {
            // find the sum
            actual += nums[i];
        }
        for (int j = 0; j <= nums.length; j++) {
            // find the sum for a given range
            sum += j;
        }
        // difference b/w
        return sum - actual;
    }
}
