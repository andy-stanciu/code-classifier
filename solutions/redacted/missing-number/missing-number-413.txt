class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int len = nums.length;
        int ans = 0;
        int sum = 0, sum1 = 0;
        for (int i = 0; i < len; i++) {
            sum += nums[i];
        }
        for (int i = 0; i < len + 1; i++) {
            sum1 += i;
        }
        ans = sum1 - sum;
        return ans;
    }
}
