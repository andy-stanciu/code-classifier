class Solution {

    public int method1(int[] nums) {
        int expect = 0;
        int real = nums[0];
        for (int i = 1; i <= nums.length; i++) {
            expect = expect ^ i;
            if (i != nums.length) {
                real = real ^ nums[i];
            }
        }
        int res = real ^ expect;
        return res;
    }
}
