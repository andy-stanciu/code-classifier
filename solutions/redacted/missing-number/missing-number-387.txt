class Solution {

    public int method1(int[] nums) {
        // The array contains numbers in the range [0, n]
        int n = nums.length;
        // Sum of numbers from 0 to n
        int totalSum = n * (n + 1) / 2;
        // Sum of the elements in the array
        int arrSum = 0;
        for (int num : nums) {
            arrSum += num;
        }
        // The missing number
        return totalSum - arrSum;
    }
}
