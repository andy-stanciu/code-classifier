class Solution {

    public int method1(int[] nums) {
        int ans = nums[0];
        // xor all numbers in array
        for (int i = 1; i < nums.length; i++) {
            ans ^= nums[i];
        }
        // xor all nums from 0 to n
        // since there is only a single element missing by doing this
        // all numberes would be xor twice except for one number
        // as property of xor suggests (x ^ x = 0)
        // the only element that will remain after this would be the missing number
        for (int i = 0; i <= nums.length; i++) {
            ans ^= i;
        }
        return ans;
    }
}
