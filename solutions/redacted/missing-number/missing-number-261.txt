class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        // Calculate expected sum for numbers from 0 to n
        int sum1 = n * (n + 1) / 2;
        // Calculate the sum of elements in nums
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        // The difference will be the missing number
        return sum1 - sum;
    }
}
