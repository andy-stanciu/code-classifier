class Solution {

    public int method1(int[] nums) {
        // To hold the XOR result
        int xor = 0;
        // Length of the array
        int n = nums.length;
        // XOR all numbers from 0 to n and the elements in nums
        for (int i = 0; i < n; i++) {
            // XOR for all indices from 0 to n-1
            xor ^= i;
            // XOR for each element in the array
            xor ^= nums[i];
        }
        // XOR with n to account for the missing number
        xor ^= n;
        // The result is the missing number
        return xor;
    }
}
