class Solution {

    public int method1(int[] array) {
        int least = 0;
        List<Integer> numb = new ArrayList<>();
        for (int num : array) {
            numb.add(num);
            least = Math.min(least, num);
        }
        for (int i = 0; i < array.length + 1; i++) {
            if (numb.contains(least) == false) {
                break;
            }
            least++;
        }
        return least;
    }
}
