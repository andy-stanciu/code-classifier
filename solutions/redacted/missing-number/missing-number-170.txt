class Solution {

    public int method1(int[] nums) {
        // the sum N terms in A.P is n/2(2a+(n-1)*d), here a=0 and d=1 always so formula becomes n/2((n-1))
        int Sn = ((nums.length + 1) * (nums.length)) / 2;
        int sumNum = 0;
        for (int i = 0; i < nums.length; i++) {
            sumNum += nums[i];
        }
        return Sn - sumNum;
    }
}
