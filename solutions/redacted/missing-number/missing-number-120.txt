class Solution {

    public int method1(int[] nums) {
        int numsLength = nums.length;
        int method1 = 0;
        for (int i = 0; i < numsLength; i++) {
            // A^0 = A and A^A = 0 (With this XOR properties we can find the missing number )
            method1 ^= (nums[i] ^ (i + 1));
        }
        return method1;
    }
}
