class Solution {

    public int method1(int[] nums) {
        // This is formula for sum of first n integers
        int sum = (nums.length) * (nums.length + 1) / 2;
        // Subtracting all the integers from array so that only one integer which was not present in the array will be left in the sum
        for (int value : nums) {
            sum -= value;
        }
        return sum;
    }
}
