class Solution {

    // We want to find first index where nums[i] doesn\'t correspond to i.
    public int method1(int[] nums) {
        Arrays.sort(nums);
        int lo = 0;
        int hi = nums.length - 1;
        while (lo < hi) {
            int mid = (hi + lo) / 2;
            if (nums[mid] != mid) {
                if (nums[mid] > mid) {
                    hi = mid;
                } else {
                    lo = mid + 1;
                }
            } else {
                lo = mid + 1;
            }
        }
        // Edge case when nums has all numbers from 0 to n-1, which means we miss n.
        if (lo == nums[lo]) {
            return lo + 1;
        }
        return lo;
    }
}
