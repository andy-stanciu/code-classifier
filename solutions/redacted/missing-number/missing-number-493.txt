class Solution {

    public int method1(int[] nums) {
        int currentTotal = 0;
        int shouldBeTotal = 0;
        // find what the numbers in nums add up to.
        for (int i = 0; i < nums.length; i++) {
            currentTotal += nums[i];
        }
        // find what our numbers should add up to.
        for (int n = 1; n <= nums.length; n++) {
            shouldBeTotal += n;
        }
        return shouldBeTotal - currentTotal;
    }
}
