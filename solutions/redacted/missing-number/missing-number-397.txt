class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int x1 = 0;
        int x2 = 0;
        for (int i = 0; i < n; i++) {
            x1 = x1 ^ nums[i];
            x2 = x2 ^ i;
        }
        // XOR of the last element
        x1 = x1 ^ n;
        return x2 ^ x1;
    }
}
