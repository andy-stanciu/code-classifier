class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        Arrays.sort(nums);
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = i;
        }
        Arrays.sort(arr);
        for (int i = 0; i < n; i++) {
            for (int j = 0; i < n; i++) {
                if (nums[i] != arr[i]) {
                    return i;
                }
            }
        }
        return n;
    }
}
