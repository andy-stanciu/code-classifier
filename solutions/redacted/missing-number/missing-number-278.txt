class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int len = nums.length;
        int estimatedTotal = (len * (len + 1)) / 2;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int n = estimatedTotal - sum;
        return n;
    }
}
