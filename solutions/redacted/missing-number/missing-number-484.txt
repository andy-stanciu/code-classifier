class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        int tsum = 0;
        for (int i = 0; i <= nums.length; i++) {
            sum = sum + i;
        }
        for (int i = 0; i < nums.length; i++) {
            tsum = tsum + nums[i];
        }
        return sum - tsum;
    }
}
