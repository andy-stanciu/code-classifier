class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum_of_given = 0;
        //find sum from 0 to n;
        int sum_of_full = n * (n + 1) / 2;
        //find sum of given array
        for (int i = 0; i < nums.length; i++) {
            sum_of_given += nums[i];
        }
        //ans is sum_of_full-sum_of_given
        return sum_of_full - sum_of_given;
    }
}
