class Solution {

    public int method1(int[] nums) {
        // Approach-1
        // Arrays.sort(nums);
        // for (int i = 0; i < nums.length; i++) {
        //     if (nums[i] != i) {
        //         return i;
        //     }
        // }
        // return nums[nums.length - 1] + 1;
        // Approach-2
        int len = nums.length;
        int total = (len * (len + 1)) / 2;
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        return total - sum;
    }
}
