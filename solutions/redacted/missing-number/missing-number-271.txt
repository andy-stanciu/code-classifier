class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int xor = 0;
        // XOR all the elements from 0 to 9
        for (int i = 0; i <= n; i++) {
            xor ^= i;
        }
        // XOR all the elements in the array
        for (int num : nums) {
            xor ^= num;
        }
        // The missing element is the result of XORing all the numbers
        return xor;
    }
}
