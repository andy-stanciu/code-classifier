class Solution {

    public int method1(int[] nums) {
        //temporary integer
        int tmp = 0;
        for (int i = 0; i < nums.length; i++) {
            //make tmp XOR nums[i]
            tmp = tmp ^ nums[i];
            //make tmp XOR (i + 1)
            tmp = tmp ^ (i + 1);
        }
        return tmp;
    }
}
