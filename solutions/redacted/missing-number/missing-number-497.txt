class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int a = nums.length;
        if (nums[0] != 0) {
            return (0);
        }
        if (nums[a - 1] != a) {
            return (a);
        }
        for (int i = 1; i < a; i++) {
            if (nums[i] != i) {
                return (i);
            }
        }
        return 0;
    }
}
