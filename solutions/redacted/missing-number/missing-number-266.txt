class Solution {

    public int method1(int[] nums) {
        //my approach is pretty simple calculate the
        //totalSum=sum of 0 to n
        //numsSum=nums sum
        //result= totalSum-numsSum
        int end = nums.length;
        int totalSum = getSum(0, end);
        int numsSum = 0;
        for (int i = 0; i < nums.length; i++) {
            numsSum += nums[i];
        }
        //in last substract totalSum-numsSum
        return (totalSum - numsSum);
    }

    public int getSum(int start, int end) {
        int sum = 0;
        for (int i = start; i <= end; i++) {
            //0,1,3
            sum += i;
        }
        return sum;
    }
}
