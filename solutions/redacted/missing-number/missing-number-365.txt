class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        //   Arrays.sort(nums);
        //     for ( int i =0 ;i<n;i++){
        //         if ( nums[i]!=i) return i;
        //     }
        int sum = n * (n + 1) / 2;
        int sum1 = 0;
        for (int i = 0; i < n; i++) {
            sum1 += nums[i];
        }
        return sum - sum1;
    }
}
