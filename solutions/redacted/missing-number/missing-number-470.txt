class Solution {

    public int method1(int[] nums) {
        Map<Integer, Boolean> map = new HashMap<>();
        // intialize map with nums and boolean value say true
        for (int i = 0; i < nums.length; i++) {
            map.put(nums[i], true);
        }
        // check till nums length + 1 if map has all values in nums. If not means that number is missing from nums array so return it
        for (int i = 0; i <= nums.length; i++) {
            if (!(map.getOrDefault(i, false))) {
                return i;
            }
        }
        return -1;
    }
}
