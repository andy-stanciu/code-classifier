class Solution {

    public int method1(int[] nums) {
        // n is the length of the array
        int n = nums.length;
        // Sum of numbers from 0 to n
        int expectedSum = n * (n + 1) / 2;
        // Initialize actual sum
        int actualSum = 0;
        for (int num : nums) {
            // Sum the numbers in the array
            actualSum += num;
        }
        // The missing number
        return expectedSum - actualSum;
    }
}
