class Solution {

    public int method1(int[] nums) {
        //sorting the array
        Arrays.sort(nums);
        //variable which is used to check for missing element
        int n = 0;
        for (int i = 0; i < nums.length; i++) {
            if (//checking for not equal
            nums[i] != n) {
                //if statement falses it return n
                return n;
            }
            //increment
            n++;
        }
        //if statement did not false inside loop than we have to print the next to the last element
        return n;
    }
}
