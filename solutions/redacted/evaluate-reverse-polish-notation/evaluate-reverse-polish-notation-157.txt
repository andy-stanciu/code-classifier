public class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String token : tokens) {
            // Check if the token is an operator
            if (token.equals("+") || token.equals("-") || token.equals("*") || token.equals("/")) {
                // Pop the last two numbers from the stack
                int b = stack.pop();
                int a = stack.pop();
                // Apply the operator and push the result back onto the stack
                switch(token) {
                    case "+":
                        stack.push(a + b);
                        break;
                    case "-":
                        stack.push(a - b);
                        break;
                    case "*":
                        stack.push(a * b);
                        break;
                    case "/":
                        stack.push(a / b);
                        break;
                }
            } else {
                // If the token is a number, push it onto the stack
                stack.push(Integer.parseInt(token));
            }
        }
        // The result will be the only number left in the stack
        return stack.pop();
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        // Example usage: Creating test cases directly as String arrays
        String[] tokens1 = { "2", "1", "+", "3", "*" };
        // Output: 9
        System.out.println("Result: " + sol.evalRPN(tokens1));
        String[] tokens2 = { "4", "13", "5", "/", "+" };
        // Output: 6
        System.out.println("Result: " + sol.evalRPN(tokens2));
        String[] tokens3 = { "10", "6", "9", "3", "+", "-11", "*", "/", "*", "17", "+", "5", "+" };
        // Output: 22
        System.out.println("Result: " + sol.evalRPN(tokens3));
    }
}
