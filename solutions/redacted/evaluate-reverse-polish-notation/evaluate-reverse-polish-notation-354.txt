class Solution {

    public int evalRPN(String[] input) {
        Stack<Integer> stack = new Stack<>();
        int firstPop;
        int secondPop;
        for (int i = 0; i < input.length; i++) {
            switch(input[i]) {
                case "+":
                    stack.push(stack.pop() + stack.pop());
                    break;
                case "-":
                    firstPop = stack.pop();
                    secondPop = stack.pop();
                    stack.push(secondPop - firstPop);
                    break;
                case "*":
                    stack.push(stack.pop() * stack.pop());
                    break;
                case "/":
                    firstPop = stack.pop();
                    secondPop = stack.pop();
                    stack.push(secondPop / firstPop);
                    break;
                default:
                    stack.push(Integer.parseInt(input[i]));
            }
        }
        return stack.pop();
    }
}
