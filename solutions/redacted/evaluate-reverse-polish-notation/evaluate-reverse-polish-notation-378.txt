class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<Integer>();
        for (String elem : tokens) {
            switch(elem) {
                case "+":
                    int operand = stack.pop();
                    int operand2 = stack.pop();
                    int value = operand + operand2;
                    stack.push(value);
                    break;
                case "*":
                    operand = stack.pop();
                    operand2 = stack.pop();
                    value = operand * operand2;
                    stack.push(value);
                    break;
                case "/":
                    operand = stack.pop();
                    operand2 = stack.pop();
                    value = operand2 / operand;
                    stack.push(value);
                    break;
                case "-":
                    operand = stack.pop();
                    operand2 = stack.pop();
                    value = operand2 - operand;
                    stack.push(value);
                    break;
                default:
                    stack.push((Integer.parseInt(elem)));
            }
        }
        return stack.pop();
    }
}
