class Solution {

    int i;

    public int evalRPN(String[] tokens) {
        i = tokens.length - 1;
        return operation(tokens);
    }

    public int operation(String[] tokens) {
        String op = tokens[i--];
        int left, right;
        switch(op) {
            case "*":
                right = operation(tokens);
                left = operation(tokens);
                return left * right;
            case "/":
                right = operation(tokens);
                left = operation(tokens);
                return left / right;
            case "-":
                right = operation(tokens);
                left = operation(tokens);
                return left - right;
            case "+":
                right = operation(tokens);
                left = operation(tokens);
                return left + right;
            default:
                return Integer.parseInt(op);
        }
    }
}
