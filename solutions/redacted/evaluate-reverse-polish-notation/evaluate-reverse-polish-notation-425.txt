class Solution {

    public int evalRPN(String[] tokens) {
        Deque<Integer> st = new ArrayDeque<>();
        for (String str : tokens) {
            if (str.equals("+")) {
                int val2 = st.pop();
                int val1 = st.pop();
                st.push(val1 + val2);
            } else if (str.equals("-")) {
                int val2 = st.pop();
                int val1 = st.pop();
                st.push(val1 - val2);
            } else if (str.equals("*")) {
                int val2 = st.pop();
                int val1 = st.pop();
                st.push(val1 * val2);
            } else if (str.equals("/")) {
                int val2 = st.pop();
                int val1 = st.pop();
                st.push(val1 / val2);
            } else {
                st.push(Integer.parseInt(str));
            }
        }
        return st.pop();
    }
}
