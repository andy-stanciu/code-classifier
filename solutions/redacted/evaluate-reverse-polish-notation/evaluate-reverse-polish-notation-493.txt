class Solution {

    public int evalRPN(String[] tokens) {
        // Initialized a stack.
        Stack<String> stack = new Stack<>();
        // for-loop to iterate through each tokens of the string.
        for (String str : tokens) {
            // checking if the token is an operator or not.
            if (!isOperator(str)) {
                // if NO, then push the token into stack as it is.
                stack.push(str);
            } else {
                // if YES, then pop the first two elements from stack
                // convert them into integers for operations.
                int num2 = Integer.parseInt(stack.pop());
                int num1 = Integer.parseInt(stack.pop());
                // perform operations of the num1 and num2 based on type of operator.
                int result = 0;
                if (str.equals("+")) {
                    result = num1 + num2;
                } else if (str.equals("-")) {
                    result = num1 - num2;
                } else if (str.equals("*")) {
                    result = num1 * num2;
                } else if (str.equals("/")) {
                    result = num1 / num2;
                }
                // convert result back into string.
                // Push the result obtained into the stack.
                stack.push(Integer.toString(result));
            }
        }
        // the only/1st element present in stack will be the result.
        return Integer.parseInt(stack.peek());
    }

    // helper function to check whether the token is operator or operand.
    private boolean isOperator(String str) {
        if (str.equals("+") || str.equals("-") || str.equals("*") || str.equals("/")) {
            return true;
        }
        return false;
    }
}
