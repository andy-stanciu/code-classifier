class Solution {

    public int evalRPN(String[] token) {
        Stack<Integer> s = new Stack<>();
        for (int i = 0; i < token.length; i++) {
            if (token[i].equals("+")) {
                int a = s.pop();
                int b = s.pop();
                s.push(a + b);
            } else if (token[i].equals("-")) {
                int a = s.pop();
                int b = s.pop();
                s.push(b - a);
            } else if (token[i].equals("*")) {
                int a = s.pop();
                int b = s.pop();
                s.push(a * b);
            } else if (token[i].equals("/")) {
                int a = s.pop();
                int b = s.pop();
                s.push(b / a);
            } else
                s.push(Integer.valueOf(token[i]));
        }
        return s.peek();
    }
}
