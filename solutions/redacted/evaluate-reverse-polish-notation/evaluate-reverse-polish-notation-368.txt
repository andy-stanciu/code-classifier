class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        Set<String> set = new HashSet<>();
        set.add("+");
        set.add("-");
        set.add("*");
        set.add("/");
        int val = 0;
        String s = "";
        int n = tokens.length;
        for (int i = 0; i < n; i++) {
            s = tokens[i];
            if (!set.contains(s)) {
                val = Integer.parseInt(s);
            } else {
                int a = stack.pop();
                int b = stack.pop();
                switch(s) {
                    case "+":
                        val = a + b;
                        break;
                    case "-":
                        val = b - a;
                        break;
                    case "*":
                        val = a * b;
                        break;
                    case "/":
                        val = b / a;
                        break;
                }
            }
            stack.push(val);
        }
        return stack.pop();
    }
}
