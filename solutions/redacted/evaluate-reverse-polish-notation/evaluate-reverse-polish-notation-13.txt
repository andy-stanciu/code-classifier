class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> stack = new Stack<>();
        for (String token : tokens) {
            if ("+-*/".contains(token)) {
                int b = stack.pop();
                int a = stack.pop();
                int result;
                switch(token) {
                    case "+" ->
                        result = a + b;
                    case "-" ->
                        result = a - b;
                    case "*" ->
                        result = a * b;
                    case "/" ->
                        result = a / b;
                    default ->
                        throw new IllegalArgumentException("Unknown operator: " + token);
                }
                stack.push(result);
            } else {
                stack.push(Integer.parseInt(token));
            }
        }
        return stack.pop();
    }
}
