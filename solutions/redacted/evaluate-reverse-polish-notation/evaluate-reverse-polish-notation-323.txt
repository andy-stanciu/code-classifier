class Solution {

    public int evalRPN(String[] tokens) {
        if (tokens.length == 1)
            return Integer.parseInt(tokens[0]);
        Stack<Integer> stack = new Stack<>();
        for (String s : tokens) {
            compute(stack, s);
        }
        return stack.pop();
    }

    void compute(Stack<Integer> stack, String s) {
        if (s.equals("*")) {
            stack.push(stack.pop() * stack.pop());
        } else if (s.equals("/")) {
            int op1 = stack.pop();
            int op2 = stack.pop();
            stack.push(op2 / op1);
        } else if (s.equals("+")) {
            stack.push(stack.pop() + stack.pop());
        } else if (s.equals("-")) {
            int op1 = stack.pop();
            int op2 = stack.pop();
            stack.push(op2 - op1);
        } else {
            stack.push(Integer.parseInt(s));
        }
        // System.out.println(stack);
    }
}
