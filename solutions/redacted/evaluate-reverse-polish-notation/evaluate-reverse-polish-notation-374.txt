class Solution {

    public int evalRPN(String[] tokens) {
        Deque<Integer> stack = new ArrayDeque<>();
        for (String token : tokens) {
            switch(token) {
                case "+" ->
                    stack.push(stack.pop() + stack.pop());
                case "-" ->
                    {
                        int last = stack.pop();
                        stack.push(stack.pop() - last);
                    }
                case "*" ->
                    stack.push(stack.pop() * stack.pop());
                case "/" ->
                    {
                        int last = stack.pop();
                        stack.push(stack.pop() / last);
                    }
                default ->
                    stack.push(Integer.parseInt(token));
            }
        }
        return stack.getLast();
    }
}
