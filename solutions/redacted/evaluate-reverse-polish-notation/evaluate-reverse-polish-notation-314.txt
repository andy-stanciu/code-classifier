class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> st = new Stack<>();
        int n = 0;
        int m = 0;
        for (String s : tokens) {
            if (s.equals("+")) {
                n = st.pop();
                m = st.pop();
                st.push(n + m);
            } else if (s.equals("-")) {
                n = st.pop();
                m = st.pop();
                st.push(m - n);
            } else if (s.equals("*")) {
                n = st.pop();
                m = st.pop();
                st.push(m * n);
            } else if (s.equals("/")) {
                n = st.pop();
                m = st.pop();
                st.push(m / n);
            } else {
                st.push(Integer.parseInt(s));
            }
        }
        return st.pop();
    }
}
