class Solution {

    public int evalRPN(String[] tokens) {
        Stack<Integer> memoryStack = new Stack<>();
        for (String token : tokens) {
            if (token.equals("+")) {
                int second = memoryStack.pop();
                int first = memoryStack.pop();
                memoryStack.push(first + second);
            } else if (token.equals("-")) {
                int second = memoryStack.pop();
                int first = memoryStack.pop();
                memoryStack.push(first - second);
            } else if (token.equals("/")) {
                int second = memoryStack.pop();
                int first = memoryStack.pop();
                memoryStack.push(first / second);
            } else if (token.equals("*")) {
                int second = memoryStack.pop();
                int first = memoryStack.pop();
                memoryStack.push(first * second);
            } else {
                memoryStack.push(Integer.parseInt(token));
            }
        }
        return memoryStack.pop();
    }
}
