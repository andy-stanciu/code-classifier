class Solution {

    public int evalRPN(String[] tokens) {
        Stack<String> stack = new Stack<>();
        for (String s : tokens) {
            switch(s) {
                case "+":
                    {
                        int x = Integer.parseInt(stack.pop());
                        int y = Integer.parseInt(stack.pop());
                        int result = x + y;
                        stack.add(String.valueOf(result));
                        break;
                    }
                case "-":
                    {
                        int x = Integer.parseInt(stack.pop());
                        int y = Integer.parseInt(stack.pop());
                        int result = y - x;
                        stack.add(String.valueOf(result));
                        break;
                    }
                case "*":
                    {
                        int x = Integer.parseInt(stack.pop());
                        int y = Integer.parseInt(stack.pop());
                        int result = x * y;
                        stack.add(String.valueOf(result));
                        break;
                    }
                case "/":
                    {
                        int x = Integer.parseInt(stack.pop());
                        int y = Integer.parseInt(stack.pop());
                        int result = y / x;
                        stack.add(String.valueOf(result));
                        break;
                    }
                default:
                    {
                        stack.add(s);
                    }
            }
        }
        return Integer.parseInt(stack.pop());
    }
}
