class Solution {

    public int evalRPN(String[] a) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < a.length; i++) {
            if (a[i].equals("+")) {
                stack.push(stack.pop() + stack.pop());
            } else if (a[i].equals("-")) {
                int x = stack.pop();
                stack.push(stack.pop() - x);
            } else if (a[i].equals("/")) {
                int x = stack.pop();
                stack.push(stack.pop() / x);
            } else if (a[i].equals("*")) {
                stack.push(stack.pop() * stack.pop());
            } else {
                stack.push(Integer.parseInt(a[i]));
            }
        }
        return stack.pop();
    }
}
