class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        // nums1 = {5,7,8,11} m
        // nums2 = {1,2,4,12} n
        // nums1 = {1,2,4,5,7,8,11,12} (merged) m+n
        int i = m - 1;
        int j = n - 1;
        int k = nums1.length - 1;
        while (j >= 0) {
            if (i >= 0 && nums1[i] > nums2[j]) {
                nums1[k] = nums1[i];
                k--;
                i--;
            } else {
                nums1[k] = nums2[j];
                k--;
                j--;
            }
        }
    }
}
