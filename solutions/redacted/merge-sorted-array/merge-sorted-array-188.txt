class Solution {

    //  Time Complexity = O(n log(n)) for Arrays.sort method
    //  Space Complexity = O(1)
    public void method1(int[] nums1, int m, int[] nums2, int n) {
        //  Removing the 0s of nums1 with nums2 values by merging them
        for (int i = n - 1, j = m + n - 1; i >= 0; --i, --j) {
            nums1[j] = nums2[i];
        }
        //  Sorting the merged array
        Arrays.sort(nums1);
    }
}
