class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        // create a new result array
        int[] result = new int[m + n];
        // pointer for nums1
        int i = 0;
        // pointer for nums2
        int j = 0;
        // pointer for result array
        int k = 0;
        while (i < m && j < n) {
            if (nums1[i] < nums2[j]) {
                result[k] = nums1[i];
                i++;
            } else {
                result[k] = nums2[j];
                j++;
            }
            k++;
        }
        // copy remaining elements of nums1, if any
        while (i < m) {
            result[k] = nums1[i];
            i++;
            k++;
        }
        // copy remaining elements of nums2, if any
        while (j < n) {
            result[k] = nums2[j];
            j++;
            k++;
        }
        // copy result array into nums1
        for (int l = 0; l < m + n; l++) {
            nums1[l] = result[l];
        }
    }
}
