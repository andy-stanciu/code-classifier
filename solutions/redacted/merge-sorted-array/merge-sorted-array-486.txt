class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] A = new int[m + n];
        int i, j, k = 0;
        for (i = 0, j = 0; i < m && j < n; ) {
            if (nums1[i] <= nums2[j])
                A[k++] = nums1[i++];
            else
                A[k++] = nums2[j++];
        }
        for (; i < m; ) A[k++] = nums1[i++];
        for (; j < n; ) A[k++] = nums2[j++];
        for (i = 0; i < m + n; i++) nums1[i] = A[i];
    }
}
