class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        if (m == 0) {
            for (int i = 0; i < n; i++) nums1[i] = nums2[i];
            return;
        }
        if (n == 0)
            return;
        int index = m + n - 1;
        while (m > 0 && n > 0) nums1[index--] = nums1[m - 1] > nums2[n - 1] ? nums1[--m] : nums2[--n];
        if (m == 0) {
            for (int i = 0; i < n; i++) nums1[i] = nums2[i];
        }
    }
}
