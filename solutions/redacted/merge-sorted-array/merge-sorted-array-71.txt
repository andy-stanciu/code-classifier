class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int l = m + n;
        m--;
        n--;
        for (int j = l - 1; j >= 0; j--) {
            if (n < 0)
                break;
            else if (m < 0 || nums1[m] < nums2[n])
                nums1[j] = nums2[n--];
            else
                nums1[j] = nums1[m--];
        }
    }
}
