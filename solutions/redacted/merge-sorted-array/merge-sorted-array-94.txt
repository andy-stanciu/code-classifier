class Solution {

    public void method1(int[] num1, int m, int[] num2, int n) {
        m--;
        n--;
        int pointer = num1.length - 1;
        while (m >= 0 && n >= 0) {
            if (num1[m] >= num2[n])
                num1[pointer--] = num1[m--];
            else
                num1[pointer--] = num2[n--];
        }
        while (n >= 0) num1[pointer--] = num2[n--];
        while (m >= 0) num1[pointer--] = num1[m--];
    }
}
