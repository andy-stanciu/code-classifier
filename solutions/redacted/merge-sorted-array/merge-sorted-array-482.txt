class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int k = 0;
        for (int i = 0; i < m; i++) {
            nums1[k++] = nums1[i];
        }
        for (int i = 0; i < n; i++) {
            nums1[k++] = nums2[i];
        }
        Arrays.sort(nums1);
    }
}
