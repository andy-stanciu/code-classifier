class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int mptr = m - 1;
        int nptr = n - 1;
        int right = m + n - 1;
        while (nptr >= 0) {
            if (mptr >= 0 && nums1[mptr] > nums2[nptr]) {
                nums1[right] = nums1[mptr];
                mptr--;
            } else {
                nums1[right] = nums2[nptr];
                nptr--;
            }
            right--;
        }
        return;
    }
}
