class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int last = nums1.length - 1;
        m -= 1;
        n -= 1;
        while (m >= 0 && n >= 0) {
            if (nums1[m] < nums2[n]) {
                nums1[last] = nums2[n--];
            } else {
                nums1[last] = nums1[m--];
            }
            last -= 1;
        }
        while (n >= 0) {
            nums1[last] = nums2[n--];
            last -= 1;
        }
    }
}
