class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] var1 = new int[m + n];
        int k = 0, j = 0, i = 0;
        //early returns
        if (n == 0)
            return;
        if (m == 0) {
            for (i = 0; i < m + n; i++) {
                nums1[i] = nums2[i];
            }
            return;
        }
        while (i < m && j < n) {
            if (nums1[i] <= nums2[j]) {
                var1[k++] = nums1[i++];
            } else {
                var1[k++] = nums2[j++];
            }
        }
        while (i < m) {
            var1[k++] = nums1[i++];
        }
        while (j < n) {
            var1[k++] = nums2[j++];
        }
        //copy the array to nums1:
        for (i = 0; i < m + n; i++) {
            nums1[i] = var1[i];
        }
    }
}
