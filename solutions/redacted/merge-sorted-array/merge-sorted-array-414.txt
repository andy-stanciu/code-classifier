class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        Integer[] var1 = Stream.concat(Arrays.stream(nums1, 0, m).boxed(), Arrays.stream(nums2).boxed()).sorted().toArray(Integer[]::new);
        for (int i = 0; i < var1.length; i++) {
            nums1[i] = var1[i];
        }
    }
}
