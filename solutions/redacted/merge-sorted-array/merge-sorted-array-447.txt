class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int nextCursor = 0;
        for (int i = 0; i < nums1.length; i++) {
            if (nums1[i] == 0 && nextCursor < n) {
                nums1[i] = nums2[nextCursor];
                nextCursor++;
            }
        }
        Arrays.sort(nums1);
    }
}
