class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        for (int i = 0; i < n; i++) {
            nums1[m + i] = nums2[i];
        }
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 1; j < nums1.length - i; j++) {
                if (nums1[j] < nums1[j - 1]) {
                    int t = nums1[j];
                    nums1[j] = nums1[j - 1];
                    nums1[j - 1] = t;
                }
            }
        }
    }
}
