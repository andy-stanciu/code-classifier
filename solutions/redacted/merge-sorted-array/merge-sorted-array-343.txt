class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        if (nums2.length == 0) {
            return;
        }
        if (nums1.length == 1 && nums1[0] != 1) {
            nums1[0] = nums2[0];
            return;
        }
        for (int j = m, i = n - 1; j < nums1.length; j++, i--) {
            nums1[j] = nums2[i];
        }
        Arrays.sort(nums1);
    }
}
