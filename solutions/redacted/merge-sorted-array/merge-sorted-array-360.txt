class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        //size of nums1
        int p1 = m - 1;
        //size of nums2
        int p2 = n - 1;
        //Size of nums1 and nums2 combined
        int p3 = m + n - 1;
        while (//While second array is not fully traversed.
        p2 >= 0) {
            if (//Compare the elements of nums1 and nums2, starts with the final. If the final element in nums1 is greater, place it at the end of the array. Else, place the element at nums2 at the end of the array.
            p1 >= 0 && nums1[p1] > nums2[p2]) {
                //If the element in nums1 is greater than nums2, store it at the end of the array.
                nums1[p3--] = nums1[p1--];
            } else {
                //Store element in num2 at the end of the array.
                nums1[p3--] = nums2[p2--];
            }
        }
    }
}
