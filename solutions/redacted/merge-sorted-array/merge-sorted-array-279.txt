class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int index1 = m - 1;
        int index2 = n - 1;
        int resultIndex = nums1.length - 1;
        while (index2 >= 0) {
            if (index1 >= 0 && nums1[index1] > nums2[index2]) {
                nums1[resultIndex] = nums1[index1--];
            } else {
                nums1[resultIndex] = nums2[index2--];
            }
            resultIndex--;
        }
    }
}
