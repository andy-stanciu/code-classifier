class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int ptr = m + n - 1;
        while (m > 0 && n > 0) {
            if (nums1[m - 1] > nums2[n - 1]) {
                nums1[ptr] = nums1[m - 1];
                ptr -= 1;
                m -= 1;
            } else {
                nums1[ptr] = nums2[n - 1];
                ptr -= 1;
                n -= 1;
            }
        }
        while (m > 0) {
            nums1[ptr] = nums1[m - 1];
            ptr -= 1;
            m -= 1;
        }
        while (n > 0) {
            nums1[ptr] = nums2[n - 1];
            ptr -= 1;
            n -= 1;
        }
    }
}
