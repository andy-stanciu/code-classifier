class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int len = m + n;
        int gap = len / 2 + len % 2;
        for (int i = 0; i < n; i++) {
            nums1[m + i] = nums2[i];
        }
        while (gap > 0) {
            int l = 0, r = l + gap;
            while (r < len) {
                if (nums1[l] > nums1[r])
                    swap(nums1, l, nums1, r);
                l++;
                r++;
            }
            if (gap == 1)
                break;
            gap = gap / 2 + gap % 2;
        }
    }

    static void swap(int[] arr1, int i, int[] arr2, int j) {
        int temp = arr1[i];
        arr1[i] = arr2[j];
        arr2[j] = temp;
    }
}
