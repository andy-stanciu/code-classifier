class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int p1 = m - 1;
        int p2 = n - 1;
        int var1 = m + n - 1;
        while (p2 >= 0) {
            if (p1 >= 0 && nums1[p1] > nums2[p2]) {
                nums1[var1--] = nums1[p1--];
            } else {
                nums1[var1--] = nums2[p2--];
            }
        }
    }
}
