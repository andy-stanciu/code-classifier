class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int i1 = m - 1;
        int i2 = n - 1;
        while (i2 >= 0) {
            int targetIdx = i1 + i2 + 1;
            if (i1 < 0 || nums2[i2] >= nums1[i1]) {
                nums1[targetIdx] = nums2[i2];
                i2--;
            } else {
                nums1[targetIdx] = nums1[i1];
                i1--;
            }
        }
    }
}
