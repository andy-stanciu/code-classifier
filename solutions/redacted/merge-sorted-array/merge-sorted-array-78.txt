class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int currentIndex = nums1.length - 1;
        int index1 = m - 1;
        int index2 = n - 1;
        while (index1 >= 0 && index2 >= 0) {
            nums1[currentIndex--] = (nums1[index1] > nums2[index2]) ? nums1[index1--] : nums2[index2--];
        }
        // Copy remaining elements of nums2, if any
        while (index2 >= 0) {
            nums1[currentIndex--] = nums2[index2--];
        }
        // No need to copy remaining elements of nums1 as they are already in place
    }
}
