class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int n1 = nums1.length, n2 = nums2.length;
        if (n != 0 || n1 != 1) {
            int j = 0;
            for (int i = m; i < n1; i++) {
                System.out.println(i);
                if (j < n) {
                    nums1[i] = nums2[j];
                }
                j++;
            }
            Arrays.sort(nums1);
        }
    }
}
