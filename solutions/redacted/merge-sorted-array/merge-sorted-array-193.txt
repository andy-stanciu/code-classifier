class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        // pointer for nums1 starting from last index
        int i = m - 1;
        // pointer for numx=s2 starting from last index
        int j = n - 1;
        // pointer for the whole array starting from last index
        int k = m + n - 1;
        while (i >= 0 && j >= 0) {
            if (nums1[i] >= nums2[j]) {
                nums1[k--] = nums1[i--];
            } else {
                nums1[k--] = nums2[j--];
            }
        }
        while (j >= 0) {
            // remeining element for nums2 shifted to nums1
            nums1[k--] = nums2[j--];
        }
    }
}
