class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int[] ans = new int[m + n];
        for (int i = 0; i < m; i++) {
            // 1st for loop
            ans[i] = nums1[i];
        }
        for (int i = 0; i < n; i++) {
            // 2nd for loop
            ans[m + i] = nums2[i];
        }
        Arrays.sort(ans);
        for (int i = 0; i < nums1.length; i++) {
            // 3rd for loop
            nums1[i] = ans[i];
        }
    }
}
