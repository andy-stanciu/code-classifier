class Solution {

    public void method1(int[] nums1, int m, int[] nums2, int n) {
        int idx1 = 0;
        int idx2 = 0;
        int idx = 0;
        int[] var1 = new int[m + n];
        while (idx1 < m && idx2 < n) {
            if (nums1[idx1] < nums2[idx2]) {
                var1[idx++] = nums1[idx1++];
            } else {
                var1[idx++] = nums2[idx2++];
            }
        }
        while (idx1 < m) {
            var1[idx++] = nums1[idx1++];
        }
        while (idx2 < n) {
            var1[idx++] = nums2[idx2++];
        }
        for (int i = 0; i < var1.length; i++) {
            nums1[i] = var1[i];
        }
    }
}
