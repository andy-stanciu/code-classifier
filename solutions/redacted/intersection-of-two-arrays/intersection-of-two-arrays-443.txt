class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] arr = new int[1000];
        for (int i : nums1) {
            arr[i]++;
        }
        int[] ans = new int[1000];
        int ptr = 0;
        for (int i : nums2) {
            if (arr[i] != 0) {
                arr[i] = 0;
                ans[ptr] = i;
                ptr++;
            }
        }
        return Arrays.copyOfRange(ans, 0, ptr);
    }
}
