class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> res = new HashSet<>();
        HashSet<Integer> store = new HashSet<>();
        //      Store one array value in hashmap
        for (int i = 0; i < nums1.length; i++) {
            res.add(nums1[i]);
        }
        // check that element is present or not
        for (int i = 0; i < nums2.length; i++) {
            if (res.contains(nums2[i])) {
                store.add(nums2[i]);
            }
        }
        int[] arr = new int[store.size()];
        int k = 0;
        for (int i : store) {
            arr[k] = i;
            k++;
        }
        return arr;
    }
}
