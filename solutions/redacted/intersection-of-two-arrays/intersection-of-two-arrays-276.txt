class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set = new HashSet<>();
        for (int n : nums1) {
            set.add(n);
        }
        HashSet<Integer> method1 = new HashSet<>();
        for (int n : nums2) {
            if (set.contains(n)) {
                method1.add(n);
            }
        }
        int[] result = new int[method1.size()];
        int i = 0;
        for (int n : method1) {
            result[i++] = n;
        }
        return result;
    }
}
