class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> sets = new HashSet<>();
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 0; j < nums2.length; j++) {
                if (nums1[i] == nums2[j]) {
                    sets.add(nums1[i]);
                }
            }
        }
        int[] result = new int[sets.size()];
        int index = 0;
        for (int num : sets) {
            result[index++] = num;
        }
        return result;
    }
}
