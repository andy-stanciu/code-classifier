class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set = new HashSet<>();
        HashSet<Integer> intersectEle = new HashSet<>();
        for (int num1 : nums1) {
            set.add(num1);
        }
        for (int num2 : nums2) {
            if (set.contains(num2)) {
                intersectEle.add(num2);
            }
        }
        int[] res = new int[intersectEle.size()];
        int j = 0;
        for (int i : intersectEle) {
            res[j++] = i;
        }
        return res;
    }
}
