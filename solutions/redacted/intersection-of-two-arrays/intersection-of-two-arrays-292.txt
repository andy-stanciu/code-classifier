class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] count = new int[1001];
        int[] result = new int[Math.min(nums1.length, nums2.length)];
        int k = 0;
        for (int n : nums1) {
            count[n]++;
        }
        for (int x : nums2) {
            if (count[x] != 0) {
                result[k] = x;
                k++;
                count[x] = 0;
            }
        }
        return Arrays.copyOf(result, k);
    }
}
