class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums1) {
            set.add(num);
        }
        List<Integer> resList = new ArrayList<>();
        for (int num : nums2) {
            if (set.contains(num)) {
                resList.add(num);
                set.remove(num);
            }
        }
        int[] result = new int[resList.size()];
        for (int i = 0; i < resList.size(); i++) {
            result[i] = resList.get(i);
        }
        return result;
    }
}
