class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> L = new HashSet<>();
        for (int i : nums1) {
            for (int j : nums2) {
                if (i == j) {
                    L.add(i);
                }
            }
        }
        int[] arr = new int[L.size()];
        int index = 0;
        for (int value : L) {
            arr[index] = value;
            index++;
        }
        return arr;
    }
}
