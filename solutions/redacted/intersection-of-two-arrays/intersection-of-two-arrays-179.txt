class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> newStorage1 = new HashSet<>();
        Set<Integer> newStorage2 = new HashSet<>();
        for (int i : nums1) newStorage1.add(i);
        for (int i : nums2) {
            if (newStorage1.contains(i))
                newStorage2.add(i);
        }
        int[] newArr = new int[newStorage2.size()];
        int j = 0;
        for (int i : newStorage2) newArr[j++] = i;
        return newArr;
    }
}
