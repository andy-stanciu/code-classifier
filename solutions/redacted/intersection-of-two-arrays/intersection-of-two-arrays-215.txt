class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set = new HashSet<>();
        HashSet<Integer> res = new HashSet<>();
        // Store elements of nums1 in a HashSet
        for (int n : nums1) {
            set.add(n);
        }
        // Find intersection with nums2
        for (int n : nums2) {
            if (set.contains(n)) {
                res.add(n);
            }
        }
        // Convert HashSet to int array
        return res.stream().mapToInt(Integer::intValue).toArray();
    }
}
