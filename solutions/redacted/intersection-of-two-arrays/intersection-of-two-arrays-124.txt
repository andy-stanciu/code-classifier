class Solution {

    public int[] method1(int[] arr1, int[] arr2) {
        Set<Integer> set1 = new HashSet<>();
        Set<Integer> method1 = new HashSet<>();
        for (int num : arr1) {
            set1.add(num);
        }
        for (int num : arr2) {
            if (set1.contains(num)) {
                method1.add(num);
            }
        }
        int[] result = new int[method1.size()];
        int index = 0;
        for (int num : method1) {
            result[index++] = num;
        }
        return result;
    }
}
