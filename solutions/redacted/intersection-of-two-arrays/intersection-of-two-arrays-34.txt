class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        // copy first array into hashset
        HashSet<Integer> set = new HashSet<Integer>();
        for (int i : nums1) {
            set.add(i);
        }
        // create a new hashset to store teh intersection
        HashSet<Integer> intersec = new HashSet<>();
        for (int i : nums2) {
            // loop thourgh second array
            if (// check that elements is present int set or not if it present then add it.
            set.contains(i))
                intersec.add(i);
        }
        // now convert the hashset to array
        int[] res = new int[intersec.size()];
        int in = 0;
        for (int i : intersec) {
            res[in++] = i;
        }
        return res;
    }
}
