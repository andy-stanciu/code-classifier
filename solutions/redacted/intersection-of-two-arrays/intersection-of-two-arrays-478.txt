class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set = new HashSet<>();
        HashSet<Integer> intersect = new HashSet<>();
        for (int nums : nums1) {
            set.add(nums);
        }
        for (int nums : nums2) {
            if (set.contains(nums))
                intersect.add(nums);
        }
        int[] ans = new int[intersect.size()];
        int i = 0;
        for (int num : intersect) {
            ans[i++] = num;
        }
        return ans;
    }
}
