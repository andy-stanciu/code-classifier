class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int c = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 0; j < nums2.length; j++) {
                if (nums1[i] == nums2[j]) {
                    c++;
                }
            }
        }
        int[] a = new int[c];
        int k = 0;
        for (int i = 0; i < nums1.length; i++) {
            for (int j = 0; j < nums2.length; j++) {
                if (nums1[i] == nums2[j]) {
                    a[k] = nums1[i];
                    k++;
                }
            }
        }
        int[] a1 = Arrays.stream(a).boxed().distinct().mapToInt(Integer::intValue).toArray();
        return a1;
    }
}
