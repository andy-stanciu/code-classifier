class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashMap<Integer, Boolean> map = new HashMap<>();
        List<Integer> resultList = new ArrayList<>();
        // Add all elements of nums1 to the hashmap
        for (int i = 0; i < nums1.length; i++) {
            if (!map.containsKey(nums1[i])) {
                map.put(nums1[i], true);
            }
        }
        // Check if elements of nums2 are in the hashmap and add to result list
        for (int i = 0; i < nums2.length; i++) {
            if (map.containsKey(nums2[i]) && map.get(nums2[i])) {
                resultList.add(nums2[i]);
                // Ensure each element is only added once
                map.put(nums2[i], false);
            }
        }
        // Convert the result list to an array
        int[] resultArray = new int[resultList.size()];
        for (int i = 0; i < resultList.size(); i++) {
            resultArray[i] = resultList.get(i);
        }
        return resultArray;
    }
}
