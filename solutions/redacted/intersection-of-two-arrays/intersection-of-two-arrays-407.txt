class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        // Use a HashSet to store unique elements from nums1
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums1) {
            set.add(num);
        }
        // Use another HashSet to store the intersection
        HashSet<Integer> intersected = new HashSet<>();
        for (int num : nums2) {
            if (set.contains(num)) {
                intersected.add(num);
            }
        }
        // Convert the result set to an array
        int[] result = new int[intersected.size()];
        int i = 0;
        for (int num : intersected) {
            result[i++] = num;
        }
        return result;
    }
}
