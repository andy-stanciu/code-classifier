class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        // to count the frequency of numbers
        boolean[] count = new boolean[1001];
        // here we are making true for every number
        for (int i : nums1) {
            count[i] = true;
        }
        int index = 0;
        // new array of length which we will take in loop
        int[] ans = new int[nums2.length];
        //loop from 2nd array if number is true will take the number in array and mark it false
        for (int i : nums2) {
            if (count[i]) {
                ans[index++] = i;
                // set.remove(i);
                count[i] = false;
            }
        }
        // here we are returning a copy of array (ans till index)
        return Arrays.copyOf(ans, index);
    }
}
