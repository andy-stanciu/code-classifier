class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> hm = new HashSet<>();
        for (int i = 0; i < nums1.length; i++) {
            hm.add(nums1[i]);
        }
        HashSet<Integer> var1 = new HashSet<>();
        for (int j = 0; j < nums2.length; j++) {
            if (hm.contains(nums2[j])) {
                var1.add(nums2[j]);
                hm.remove(nums2[j]);
            }
        }
        int[] result = new int[var1.size()];
        int index = 0;
        for (int i : var1) {
            result[index++] = i;
        }
        return result;
    }
}
