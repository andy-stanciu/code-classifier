class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> resSet = new HashSet<>();
        for (int num1 : nums1) {
            for (int i = 0; i < nums2.length; i++) {
                if (num1 == nums2[i]) {
                    resSet.add(num1);
                    // mark as found
                    nums2[i] = Integer.MIN_VALUE;
                    break;
                }
            }
        }
        int[] result = new int[resSet.size()];
        int index = 0;
        for (int num : resSet) {
            result[index++] = num;
        }
        return result;
    }
}
