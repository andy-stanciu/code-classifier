class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> nums1Unrepeated = new HashSet<>();
        HashSet<Integer> nums2Processed = new HashSet<>();
        ArrayList<Integer> answ = new ArrayList<>();
        for (int num : nums1) {
            nums1Unrepeated.add(num);
        }
        for (int num : nums2) {
            if (!nums2Processed.contains(num)) {
                nums2Processed.add(num);
                if (nums1Unrepeated.contains(num)) {
                    answ.add(num);
                }
            }
        }
        int[] answInt = new int[answ.size()];
        for (int i = 0; i < answInt.length; i++) {
            answInt[i] = answ.get(i);
        }
        return answInt;
    }
}
