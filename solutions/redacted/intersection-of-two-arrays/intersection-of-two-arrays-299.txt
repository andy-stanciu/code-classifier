class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> set1 = new HashSet<>();
        Set<Integer> method1 = new HashSet<>();
        // Add all elements from nums1 to set1
        for (int num : nums1) {
            set1.add(num);
        }
        // Check each element in nums2, if it is in set1, add to intersection
        for (int num : nums2) {
            if (set1.contains(num)) {
                method1.add(num);
            }
        }
        // Convert intersection set to array
        int[] result = new int[method1.size()];
        int index = 0;
        for (int num : method1) {
            result[index++] = num;
        }
        return result;
    }
}
