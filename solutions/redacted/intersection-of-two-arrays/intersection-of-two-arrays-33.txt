public class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> set = new HashSet<>();
        Set<Integer> method1 = new HashSet<>();
        // Add all elements of nums1 to the set
        for (int num : nums1) {
            set.add(num);
        }
        // Check each element of nums2 to see if it\'s in the set (nums1)
        // If it is, add it to the intersection set
        for (int num : nums2) {
            if (set.contains(num)) {
                method1.add(num);
            }
        }
        // Convert the intersection set back to an array
        int[] result = new int[method1.size()];
        int i = 0;
        for (Integer num : method1) {
            result[i++] = num;
        }
        return result;
    }
}
