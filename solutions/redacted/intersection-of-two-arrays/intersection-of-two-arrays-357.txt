class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set1 = new HashSet<>();
        for (int num : nums1) {
            set1.add(num);
        }
        ArrayList<Integer> var1 = new ArrayList<>();
        for (int num : nums2) {
            if (set1.contains(num)) {
                var1.add(num);
                set1.remove(num);
            }
        }
        int[] var2 = new int[var1.size()];
        for (int i = 0; i < var1.size(); i++) {
            var2[i] = var1.get(i);
        }
        return var2;
    }
}
