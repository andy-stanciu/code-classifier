class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> set1 = new HashSet<>();
        Set<Integer> var1 = new HashSet<>();
        for (Integer i : nums1) {
            set1.add(i);
        }
        for (Integer i : nums2) {
            if (set1.contains(i)) {
                var1.add(i);
            }
        }
        int size = var1.size();
        int[] ans = new int[size];
        int index = 0;
        for (Integer i : var1) {
            ans[index++] = i;
        }
        return ans;
    }
}
