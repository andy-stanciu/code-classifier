class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        Set<Integer> result = new HashSet<>();
        Set<Integer> resultSet = new HashSet<>();
        for (int val : nums1) {
            result.add(val);
        }
        for (int val : nums2) {
            if (result.contains(val)) {
                resultSet.add(val);
            }
        }
        int[] finalResult = new int[resultSet.size()];
        int i = 0;
        for (int val : resultSet) {
            finalResult[i++] = val;
        }
        return finalResult;
    }
}
