class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int[] arr = new int[1001];
        for (int num : nums1) {
            arr[num] = 1;
        }
        int n = 0;
        for (int num : nums2) {
            if (arr[num] == 1) {
                arr[num] = 2;
                n++;
            }
        }
        int[] ans = new int[n];
        for (int i = 0, j = 0; i < arr.length; i++) {
            if (arr[i] == 2) {
                ans[j++] = i;
            }
        }
        return ans;
    }
}
