class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        int i;
        int[] result;
        Set<Integer> set = new HashSet<>();
        Set<Integer> method1 = new HashSet<>();
        for (i = 0; i < nums1.length; i++) {
            set.add(nums1[i]);
        }
        for (i = 0; i < nums2.length; i++) {
            if (set.contains(nums2[i])) {
                method1.add(nums2[i]);
            }
        }
        result = new int[method1.size()];
        i = 0;
        for (int num : method1) {
            result[i++] = num;
        }
        return result;
    }
}
