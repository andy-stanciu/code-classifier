class Solution {

    public int[] method1(int[] arr1, int[] arr2) {
        Set<Integer> se = new HashSet<>();
        for (int i = 0; i < arr1.length; i++) {
            se.add(arr1[i]);
        }
        List<Integer> li = new ArrayList<>();
        for (int i = 0; i < arr2.length; i++) {
            int a = arr2[i];
            if (se.contains(a)) {
                li.add(a);
                se.remove(a);
            }
        }
        int[] ans = new int[li.size()];
        for (int i = 0; i < ans.length; i++) {
            ans[i] = li.get(i);
        }
        return ans;
    }
}
