class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        return hasmap(nums1, nums2);
    }

    int[] hasmap(int[] nums1, int[] nums2) {
        HashMap<Integer, Integer> hm = new HashMap<Integer, Integer>();
        ArrayList<Integer> a = new ArrayList<Integer>();
        int n = nums1.length;
        for (int i = 0; i < n; i++) {
            hm.put(nums1[i], nums1[i]);
        }
        int m = nums2.length;
        for (int i = 0; i < m; i++) {
            if (hm.containsKey(nums2[i])) {
                a.add(nums2[i]);
                hm.remove(nums2[i]);
            }
        }
        int[] arr1 = new int[a.size()];
        for (int k = 0; k < a.size(); k++) {
            arr1[k] = a.get(k);
        }
        return arr1;
    }
}
