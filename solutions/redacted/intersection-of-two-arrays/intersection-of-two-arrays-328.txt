class Solution {

    public int[] method1(int[] nums1, int[] nums2) {
        HashSet<Integer> set1 = new HashSet<>();
        HashSet<Integer> set2 = new HashSet<>();
        for (int ele : nums1) set1.add(ele);
        for (int ele : nums2) set2.add(ele);
        set1.retainAll(set2);
        int[] ans = new int[set1.size()];
        int i = 0;
        for (int ele : set1) {
            ans[i] = ele;
            i++;
        }
        return ans;
    }
}
