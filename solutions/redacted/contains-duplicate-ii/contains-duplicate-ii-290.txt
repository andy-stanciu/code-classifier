class Solution {

    public boolean method1(int[] nums, int k) {
        // Key = the number,
        // Value = index of the number
        HashMap<Integer, Integer> count = new HashMap();
        for (int i = 0; i < nums.length; i++) {
            if (count.containsKey(nums[i]) && i - count.get(nums[i]) <= k) {
                return true;
            } else {
                count.put(nums[i], i);
            }
        }
        return false;
    }
}
