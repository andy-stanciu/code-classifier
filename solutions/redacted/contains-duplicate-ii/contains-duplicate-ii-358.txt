class Solution {

    public boolean method1(int[] nums, int k) {
        // Create a HashMap to store the value and its latest index
        Map<Integer, Integer> map = new HashMap<>();
        // Iterate through the array
        for (int i = 0; i < nums.length; i++) {
            // Check if the value is already in the map
            if (map.containsKey(nums[i])) {
                // Check the distance between the current index and the last index
                if (i - map.get(nums[i]) <= k) {
                    // Found a nearby duplicate
                    return true;
                }
            }
            // Update the index of the current value in the map
            map.put(nums[i], i);
        }
        // helps us to remove unused references for improvement of Space Complexity
        System.gc();
        // If no nearby duplicate found, return false
        return false;
    }
}
