class Solution {

    public boolean method1(int[] nums, int k) {
        HashSet<Integer> hm = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (i > k) {
                hm.remove(nums[i - k - 1]);
            }
            if (hm.contains(nums[i])) {
                return true;
            }
            hm.add(nums[i]);
        }
        return false;
    }
}
