class Solution {

    public boolean method1(int[] nums, int k) {
        int n = nums.length;
        HashSet<Integer> set = new HashSet();
        for (int i = 0; i < n; i++) {
            if (set.contains(nums[i])) {
                return true;
            } else {
                set.add(nums[i]);
                if (i - k >= 0)
                    set.remove(nums[i - k]);
            }
        }
        return false;
    }
}
