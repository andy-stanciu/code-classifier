class Solution {

    public boolean method1(int[] nums, int k) {
        // Initialize a HashMap to store the index of elements
        HashMap<Integer, Integer> indexMap = new HashMap<>();
        // Iterate through each element in the array using index
        for (int i = 0; i < nums.length; i++) {
            // Check if the current element has been seen before
            if (indexMap.containsKey(nums[i])) {
                // Check if the difference between the indices is less than or equal to k
                if (i - indexMap.get(nums[i]) <= k) {
                    return true;
                }
            }
            // Update the index of the current element
            indexMap.put(nums[i], i);
        }
        // If no such pair is found, return false
        return false;
    }
}
