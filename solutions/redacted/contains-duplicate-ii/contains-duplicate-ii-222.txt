class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> m = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (!m.containsKey(nums[i])) {
                m.put(nums[i], i);
            } else {
                if (Math.abs(m.get(nums[i]) - i) <= k) {
                    return true;
                } else {
                    m.replace(nums[i], i);
                }
            }
        }
        return false;
    }
}
