class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> map1 = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (map1.containsKey(nums[i])) {
                if (i - map1.get(nums[i]) <= k) {
                    return true;
                }
            }
            map1.put(nums[i], i);
        }
        return false;
    }
}
