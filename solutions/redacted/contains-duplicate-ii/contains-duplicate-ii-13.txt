class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> digVsIndex = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            Integer index = digVsIndex.get(nums[i]);
            if (index != null) {
                if (i - index <= k) {
                    return true;
                }
            }
            digVsIndex.put(nums[i], i);
        }
        return false;
    }
}
