class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> index = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            // put() returns previous value associated with the given key or null if key didn\'t exist
            Integer j = index.put(nums[i], i);
            if (j != null && (i - j) <= k) {
                return true;
            }
        }
        return false;
    }
}
