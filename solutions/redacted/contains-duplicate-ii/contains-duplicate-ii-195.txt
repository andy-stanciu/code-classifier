class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> numMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (numMap.containsKey(nums[i])) {
                int j = numMap.get(nums[i]);
                if (i - j <= k) {
                    return true;
                }
            }
            // Update the latest index of nums[i]
            numMap.put(nums[i], i);
        }
        return false;
    }
}
