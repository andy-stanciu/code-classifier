class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> valueIndicesMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (!valueIndicesMap.containsKey(nums[i]))
                valueIndicesMap.put(nums[i], i);
            else {
                int diff = valueIndicesMap.get(nums[i]) - i;
                if (Math.abs(diff) <= k)
                    return true;
                valueIndicesMap.put(nums[i], i);
            }
        }
        return false;
    }
}
