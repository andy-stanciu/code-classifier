class Solution {

    public boolean method1(int[] nums, int k) {
        // Create a HashSet to store the elements in the window
        Set<Integer> window = new HashSet<>();
        // Initialize a pointer j to track the start of the window
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            // Iterate through the array nums
            // If the absolute difference between the current index i and j is greater than k
            if (Math.abs(i - j) > k) {
                // Remove the element at index j from the window
                window.remove(nums[j]);
                // Move the start pointer j to the next position
                j++;
            }
            // If the current element at index i is already in the window, return true
            if (window.contains(nums[i])) {
                return true;
            }
            // Add the current element at index i to the window
            window.add(nums[i]);
        }
        // If no duplicates were found within the window, return false
        return false;
    }
}
