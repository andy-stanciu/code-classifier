class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> var1 = new HashMap<>();
        int i = 0;
        while (i < nums.length) {
            if (var1.containsKey(nums[i])) {
                if ((i - var1.get(nums[i]) <= k)) {
                    return true;
                }
            }
            var1.put(nums[i], i);
            i++;
        }
        return false;
    }
}
