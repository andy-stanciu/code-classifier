class Solution {

    public boolean method1(int[] nums, int k) {
        //first of all create a map to store value of each index as key and index as value
        Map<Integer, Integer> map = new HashMap<>();
        //Run a for loop to traverse through the array
        for (int i = 0; i < nums.length; i++) {
            //now condition for checking duplicates under given criterion
            if (map.containsKey(nums[i]) && i - map.get(nums[i]) <= k) {
                return true;
            }
            map.put(nums[i], i);
        }
        // fingers crossed!
        return false;
    }
}
