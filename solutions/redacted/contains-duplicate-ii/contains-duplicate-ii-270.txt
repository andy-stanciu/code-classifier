class Solution {

    private int abs(int num) {
        return (num < 0) ? -num : num;
    }

    public boolean method1(int[] nums, int k) {
        //using hashmap
        Map<Integer, Integer> hm = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; i++) {
            int ele = nums[i];
            if (hm.containsKey(ele)) {
                if (abs(hm.get(ele) - i) <= k)
                    return true;
                else
                    hm.put(ele, i);
            } else
                hm.put(ele, i);
        }
        return false;
    }
}
