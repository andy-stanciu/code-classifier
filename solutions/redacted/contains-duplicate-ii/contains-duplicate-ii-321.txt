class Solution {

    public boolean method1(int[] nums, int k) {
        HashSet<Integer> hs = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (hs.contains(nums[i])) {
                return true;
            }
            hs.add(nums[i]);
            if (hs.size() > k) {
                hs.remove(nums[i - k]);
            }
        }
        return false;
    }
}
