class Solution {

    public boolean method1(int[] nums, int k) {
        Set<Integer> arr = new HashSet<>();
        for (int i = 0; i < nums.length; ++i) {
            // If the element is already in the set, we\'ve found a duplicate within the window
            if (!arr.add(nums[i]))
                return true;
            // Ensure the size of the sliding window is maintained
            if (i >= k)
                // Remove the element that is no longer within the window
                arr.remove(nums[i - k]);
        }
        return false;
    }
}
