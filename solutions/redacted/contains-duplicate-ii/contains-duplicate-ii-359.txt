class Solution {

    public boolean method1(int[] nums, int k) {
        HashSet<Integer> numsWindow = new HashSet<>();
        int begin = 0;
        for (int end = 0; end < nums.length; end++) {
            if (end - begin > k) {
                numsWindow.remove(nums[begin]);
                begin++;
            }
            if (numsWindow.contains(nums[end])) {
                return true;
            }
            numsWindow.add(nums[end]);
        }
        return false;
    }
}
