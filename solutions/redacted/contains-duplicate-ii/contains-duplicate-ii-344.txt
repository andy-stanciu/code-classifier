class Solution {

    public boolean method1(int[] nums, int k) {
        //create hashmap to store key and index pairs.
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            //checks map contains the key or not if not then add it.
            if (!map.containsKey(nums[i])) {
                map.put(nums[i], i);
            } else //if map contains the key, it stores past index in a variable
            {
                int pastIndex = map.get(nums[i]);
                // check the abs(i-j)<=k condition,if satisfied returns true
                if (Math.abs(pastIndex - i) <= k) {
                    return true;
                } else
                    // if condition not satisfied update the hashmap with latest value of i.
                    map.put(nums[i], i);
            }
        }
        return false;
    }
}
