class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> hashmap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (hashmap.containsKey(nums[i])) {
                if (Math.abs(hashmap.get(nums[i]) - i) <= k) {
                    return true;
                }
                hashmap.put(nums[i], i);
            } else {
                hashmap.put(nums[i], i);
            }
        }
        return false;
    }
}
