class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> hset = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (hset.containsKey(nums[i])) {
                int j = hset.get(nums[i]);
                if ((i - j) <= k) {
                    return true;
                }
            }
            hset.put(nums[i], i);
        }
        return false;
    }
}
