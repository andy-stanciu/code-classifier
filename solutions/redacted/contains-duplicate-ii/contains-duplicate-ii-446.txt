class Solution {

    public boolean method1(int[] nums, int k) {
        //Number , Last Occurence
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i])) {
                int lastOcc = map.get(nums[i]);
                if (Math.abs(i - lastOcc) <= k) {
                    return true;
                }
            }
            map.put(nums[i], i);
        }
        return false;
    }
}
