class Solution {

    public boolean method1(int[] nums, int k) {
        //Sliding window
        int l = 0, r = 0;
        int n = nums.length;
        Set<Integer> window = new HashSet<>();
        while (r < n) {
            if (r - l > k) {
                window.remove(nums[l]);
                l = l + 1;
            }
            if (!window.add(nums[r])) {
                return true;
            }
            r++;
        }
        return false;
    }
}
