class Solution {

    public boolean method1(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap();
        int diff;
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i])) {
                diff = i - map.get(nums[i]);
                if (diff <= k)
                    return true;
                map.remove(nums[i]);
            }
            map.put(nums[i], i);
        }
        return false;
    }
}
