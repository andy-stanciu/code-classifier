class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> idx = new HashMap<>();
        for (int currIdx = 0; currIdx < nums.length; currIdx++) {
            if (idx.containsKey(nums[currIdx])) {
                if (currIdx - idx.get(nums[currIdx]) <= k)
                    return true;
            }
            idx.put(nums[currIdx], currIdx);
        }
        return false;
    }
}
