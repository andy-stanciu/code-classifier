class Solution {

    public boolean method1(int[] nums, int k) {
        Hashtable<Integer, Integer> ht = new Hashtable<>();
        for (int i = 0; i < nums.length; i++) {
            if (ht.containsKey(nums[i]) && (i - ht.get(nums[i]) <= k)) {
                return true;
            }
            ht.put(nums[i], i);
        }
        return false;
    }
}
