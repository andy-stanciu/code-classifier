class Solution {

    public boolean method1(int[] nums, int k) {
        HashMap<Integer, Integer> liczby = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (liczby.containsKey(nums[i]) && i - liczby.get(nums[i]) <= k) {
                return true;
            } else {
                liczby.put(nums[i], i);
            }
        }
        return false;
    }
}
