class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] prod = new int[n];
        prod[0] = 1;
        for (int i = 1; i < n; i++) prod[i] = prod[i - 1] * nums[i - 1];
        int suffix = 1;
        for (int i = n - 1; i >= 0; i--) {
            prod[i] = prod[i] * suffix;
            suffix = suffix * nums[i];
        }
        // for(int i=0;i<n;i++){
        //     for(int j=0;j<n;j++){
        //         if(i == j)
        //             continue;
        //         prod[i] = prod[i] * nums[j];
        //     }
        // }
        return prod;
    }
}
