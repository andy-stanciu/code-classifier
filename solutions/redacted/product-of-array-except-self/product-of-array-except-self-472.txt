class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int var1 = 1;
        int var2 = 0;
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                var2++;
            }
            var1 = var1 * nums[i];
        }
        if (var2 > 1) {
            return ans;
        }
        if (var2 == 1) {
            var1 = 1;
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] != 0) {
                    var1 = var1 * nums[i];
                } else {
                    index = i;
                }
            }
            ans[index] = var1;
            return ans;
        } else {
            for (int i = 0; i < nums.length; i++) {
                int a = 0;
                a = var1 / nums[i];
                ans[i] = a;
            }
            return ans;
        }
    }
}
