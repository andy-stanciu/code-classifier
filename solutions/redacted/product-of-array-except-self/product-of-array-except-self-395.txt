class Solution {

    public int[] method1(int[] nums) {
        int left = 1;
        int right = 1;
        int[] ans = new int[nums.length];
        ans[0] = 1;
        int var1 = 1;
        for (int i = 1; i < nums.length; i++) {
            var1 = nums[i - 1] * var1;
            ans[i] = var1;
        }
        var1 = 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            var1 = var1 * nums[i + 1];
            ans[i] = ans[i] * var1;
        }
        return ans;
    }
}
