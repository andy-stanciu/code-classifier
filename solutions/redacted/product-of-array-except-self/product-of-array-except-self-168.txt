class Solution {

    public static int method1(int[] arr) {
        int ans = 1;
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 0) {
                count++;
                continue;
            }
            ;
            ans *= arr[i];
        }
        return count > 1 ? 0 : ans;
    }

    public int[] method2(int[] nums) {
        int method1 = method1(nums);
        for (int i = 0; i < nums.length; i++) {
            int temp = nums[i];
            if (temp == 0) {
                Arrays.fill(nums, 0);
                nums[i] = method1;
                return nums;
            } else {
                nums[i] = method1 / temp;
            }
        }
        return nums;
    }
}
