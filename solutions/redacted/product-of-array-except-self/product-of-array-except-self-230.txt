class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        int beforecurrent = 1;
        int aftercurrent = 1;
        for (int i = 0; i < n; i++) {
            arr[i] = beforecurrent;
            beforecurrent *= nums[i];
        }
        for (int i = n - 1; i >= 0; i--) {
            arr[i] *= aftercurrent;
            aftercurrent *= nums[i];
        }
        return arr;
    }
}
