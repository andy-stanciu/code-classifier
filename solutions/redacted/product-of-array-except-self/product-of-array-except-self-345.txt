class Solution {

    public int[] method1(int[] nums) {
        int[] result = new int[nums.length];
        int[] var1 = new int[nums.length];
        int[] var2 = new int[nums.length];
        var2[nums.length - 1] = nums[nums.length - 1];
        for (int i = 0; i < var1.length; i++) {
            int prefixVar = i - 1 == -1 ? 1 : var1[i - 1];
            var1[i] = prefixVar * nums[i];
        }
        for (int i = nums.length - 2; i >= 0; i--) {
            var2[i] = var2[i + 1] * nums[i];
        }
        for (int i = 0; i < result.length; i++) {
            int prefixVar = i - 1 == -1 ? 1 : var1[i - 1];
            int suffixVar = i + 1 == result.length ? 1 : var2[i + 1];
            result[i] = prefixVar * suffixVar;
        }
        return result;
    }
}
