class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        // Compute prefix multipliers for every position.
        // Compute suffix multipliers for every position.
        // multiply them together to get the final output.
        int[] prefixMultiplier = new int[n];
        int[] suffixMultiplier = new int[n];
        int[] result = new int[n];
        prefixMultiplier[0] = 1;
        suffixMultiplier[n - 1] = 1;
        for (int i = 1; i < n; i++) {
            prefixMultiplier[i] = prefixMultiplier[i - 1] * nums[i - 1];
        }
        for (int i = n - 2; i >= 0; i--) {
            suffixMultiplier[i] = suffixMultiplier[i + 1] * nums[i + 1];
        }
        for (int i = 0; i < n; i++) {
            result[i] = prefixMultiplier[i] * suffixMultiplier[i];
        }
        return result;
    }
}
