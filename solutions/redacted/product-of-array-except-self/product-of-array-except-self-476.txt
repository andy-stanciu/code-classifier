class Solution {

    public int[] method1(int[] nums) {
        int t = 1;
        int[] r = new int[nums.length];
        // straight multiplication
        for (int i = 0; i < nums.length; i++) {
            r[i] = t;
            t = t * nums[i];
        }
        //r={1,1,2,6}
        //reverse multiplication
        t = 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            r[i] = r[i] * t;
            t = t * nums[i];
        }
        return r;
    }
}
