class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        // Compute prefix multipliers for every position.
        // Compute suffix multipliers for every position.
        // multiply them together to get the final output.
        // If you combine these steps and use one output array for both, we can save memory
        int[] result = new int[n];
        result[0] = 1;
        for (int i = 1; i < n; i++) {
            result[i] = result[i - 1] * nums[i - 1];
        }
        int suffix = 1;
        for (int i = n - 1; i >= 0; i--) {
            result[i] *= suffix;
            suffix *= nums[i];
        }
        return result;
    }
}
