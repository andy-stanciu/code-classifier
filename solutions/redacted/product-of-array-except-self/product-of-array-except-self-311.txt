class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int mult = 1;
        int count0 = 0;
        int index0 = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                count0++;
                index0 = i;
            } else {
                mult *= nums[i];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (count0 > 1) {
                ans[i] = 0;
            } else if (count0 == 1 && i != index0) {
                ans[i] = 0;
            } else if (count0 == 1 && i == index0) {
                ans[i] = mult;
            } else
                ans[i] = mult / nums[i];
        }
        return ans;
    }
}
