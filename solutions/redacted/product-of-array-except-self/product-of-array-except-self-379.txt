class Solution {

    public int[] method1(int[] nums) {
        //storing total product of nums
        int mul = 1;
        //counting total zeros
        int zeros = 0;
        //Calculating product and counting zeros
        for (int num : nums) {
            if (num == 0) {
                zeros++;
                continue;
            }
            mul *= num;
        }
        //total zeros is greater than 1 then all will be zeros;
        if (zeros > 1)
            Arrays.fill(nums, 0);
        else {
            for (int i = 0; i < nums.length; i++) {
                //if total zeros is 1 and num is zero then it\'s value will be total product
                //else total product divided by current value
                if (zeros == 1)
                    nums[i] = nums[i] == 0 ? mul : 0;
                else
                    nums[i] = mul / nums[i];
            }
        }
        return nums;
    }
}
