class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        int var1 = 1;
        int zeroCount = 0;
        for (int num : nums) {
            if (num == 0) {
                zeroCount++;
            } else {
                var1 *= num;
            }
        }
        for (int j = 0; j < n; j++) {
            if (zeroCount > 1) {
                answer[j] = 0;
            } else if (zeroCount == 1) {
                answer[j] = (nums[j] == 0) ? var1 : 0;
            } else {
                answer[j] = var1 / nums[j];
            }
        }
        return answer;
    }
}
