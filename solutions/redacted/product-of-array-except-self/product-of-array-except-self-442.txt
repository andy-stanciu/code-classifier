class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] res = new int[n];
        // Prefix pass
        int prefix = 1;
        for (int i = 0; i < n; i++) {
            res[i] = prefix;
            prefix *= nums[i];
        }
        // Suffix pass
        int suffix = 1;
        for (int i = n - 1; i >= 0; i--) {
            res[i] *= suffix;
            suffix *= nums[i];
        }
        return res;
    }
}
