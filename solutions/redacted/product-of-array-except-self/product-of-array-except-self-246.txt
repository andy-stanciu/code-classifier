class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] l = new int[n];
        int x = 1;
        for (int i = 0; i < n; i++) {
            l[i] = x;
            x = nums[i] * x;
        }
        int y = 1;
        for (int i = n - 1; i >= 0; i--) {
            l[i] = y * l[i];
            y = nums[i] * y;
        }
        return l;
    }
}
