class Solution {

    public int[] method1(int[] param1) {
        int[] var1 = new int[param1.length];
        int var2 = 1;
        for (int i = 0; i < param1.length; i++) {
            var1[i] = var2;
            var2 *= param1[i];
        }
        int var3 = 1;
        for (int i = param1.length - 1; i >= 0; i--) {
            var1[i] *= var3;
            var3 *= param1[i];
        }
        return var1;
    }
}
