class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] var1 = new int[n];
        int var2 = 1;
        for (int i = 0; i < n; i++) {
            var1[i] = var2;
            var2 *= nums[i];
        }
        var2 = 1;
        for (int i = n - 1; i >= 0; i--) {
            var1[i] *= var2;
            var2 *= nums[i];
        }
        return var1;
    }
}
