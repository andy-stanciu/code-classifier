class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            int var1 = 1;
            for (int j = 0; j < n; j++) {
                if (i == j)
                    continue;
                var1 = var1 * nums[j];
            }
            ans[i] = var1;
        }
        return ans;
    }
}
