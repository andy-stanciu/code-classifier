class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        // Initialize the answer array
        int[] answer = new int[n];
        // Step 1: Fill answer array with left products
        // answer[i] will contain the product of all elements to the left of nums[i]
        // There are no elements to the left of the first element
        answer[0] = 1;
        for (int i = 1; i < n; i++) {
            answer[i] = nums[i - 1] * answer[i - 1];
        }
        // Step 2: Traverse from the right and update answer array with the product
        // of elements to the right of nums[i]
        // There are no elements to the right of the last element
        int var1 = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] = answer[i] * var1;
            // Update right product for the next iteration
            var1 *= nums[i];
        }
        return answer;
    }
}
