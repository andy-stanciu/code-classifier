class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int cnt = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                cnt++;
            } else {
                var1 *= nums[i];
            }
        }
        int[] ans = new int[nums.length];
        if (cnt > 1) {
            return ans;
        }
        for (int i = 0; i < nums.length; i++) {
            if (cnt == 1 && nums[i] != 0) {
                ans[i] = 0;
                continue;
            }
            if (nums[i] == 0) {
                ans[i] = var1;
            } else {
                ans[i] = var1 / nums[i];
            }
        }
        return ans;
    }
}
