class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        //agr koi element 0 hua to product for all other places 0 ho jayega and for only that place product of baki nos hoga so see kitna 0 ha
        int zerocount = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                //agr 0 nhi ha to simply product all elements
                var1 = var1 * nums[i];
            } else {
                //zero hai to count krlo
                zerocount++;
            }
        }
        int[] ans = new int[nums.length];
        if (zerocount > 1) {
            //agr multiple zeros ha to fir to har element 0,empty array dedo
            return new int[nums.length];
        }
        for (int i = 0; i < nums.length; i++) {
            if (zerocount == 0) {
                //ab agr koi 0 na h to simply us element ka alawa(removing by dividing) bakio ka product krke assign
                ans[i] = var1 / nums[i];
            } else {
                //agr 0 ha
                if (nums[i] == 0) {
                    //jis place pr 0 h vha vakio ka product assign krdo
                    ans[i] = var1;
                } else {
                    //and baki places ka lia product 0 hoga
                    ans[i] = 0;
                }
            }
        }
        return ans;
    }
}
