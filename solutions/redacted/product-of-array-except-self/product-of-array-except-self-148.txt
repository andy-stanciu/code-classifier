class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n];
        arr[0] = 1;
        for (int i = 1; i < n; i++) {
            arr[i] = arr[i - 1] * nums[i - 1];
        }
        int prod = 1;
        for (int i = n - 1; i >= 0; i--) {
            arr[i] *= prod;
            prod *= nums[i];
        }
        return arr;
    }
}
