class Solution {

    public int[] method1(int[] nums) {
        int total = 1;
        int count = 0;
        int n = nums.length;
        for (int num : nums) {
            if (num != 0) {
                total *= num;
            } else {
                count++;
            }
        }
        int[] output = new int[n];
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                output[i] = (count >= 1) ? 0 : total / nums[i];
            } else {
                output[i] = (count > 1) ? 0 : total;
            }
        }
        return output;
    }
}
