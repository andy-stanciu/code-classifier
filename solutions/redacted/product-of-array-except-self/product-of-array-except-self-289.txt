class Solution {

    public static int[] method1(int[] nums) {
        int n = nums.length;
        int var1 = 1;
        int[] answer = new int[n];
        if (containszero(nums)) {
            int countZeros = 0;
            int var2 = 1;
            for (int i = 0; i < n; i++) {
                if (nums[i] != 0) {
                    var2 *= nums[i];
                } else {
                    countZeros++;
                }
            }
            if (countZeros > 1) {
                Arrays.fill(answer, 0);
            } else {
                for (int i = 0; i < n; i++) {
                    if (nums[i] == 0) {
                        answer[i] = var2;
                    }
                }
            }
        } else {
            for (int i = 0; i < n; i++) {
                var1 *= nums[i];
            }
            for (int i = 0; i < n; i++) {
                answer[i] = var1 / nums[i];
            }
        }
        return answer;
    }

    public static boolean containszero(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                return true;
            }
        }
        return false;
    }
}
