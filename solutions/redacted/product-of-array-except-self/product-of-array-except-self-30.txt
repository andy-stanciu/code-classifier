class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int n = nums.length;
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            ans[i] = var1;
            var1 *= nums[i];
        }
        var1 = 1;
        for (int i = n - 1; i >= 0; i--) {
            ans[i] = var1 * ans[i];
            var1 *= nums[i];
        }
        System.gc();
        return ans;
    }
}
