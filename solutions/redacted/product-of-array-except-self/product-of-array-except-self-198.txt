class Solution {

    public int[] method1(int[] nums) {
        int length = nums.length;
        int[] ans = new int[length];
        int prefix = 1;
        int postfix = 1;
        Arrays.fill(ans, 1);
        for (int i = 0; i < length; i++) {
            ans[i] = prefix;
            prefix *= nums[i];
        }
        for (int i = length - 1; i >= 0; i--) {
            ans[i] *= postfix;
            postfix *= nums[i];
        }
        return ans;
    }
}
