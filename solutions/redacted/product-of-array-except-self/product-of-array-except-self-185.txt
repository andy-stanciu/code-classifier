class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int suffix = 1;
        int[] result = new int[n];
        result[0] = 1;
        for (int i = 1; i < n; i++) {
            result[i] = result[i - 1] * nums[i - 1];
        }
        for (int i = n - 1; i >= 0; i--) {
            result[i] *= suffix;
            suffix *= nums[i];
        }
        System.gc();
        return result;
    }
}
