class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] result = new int[n];
        // Calculate the product of all elements to the left of each index
        int var1 = 1;
        for (int i = 0; i < n; i++) {
            result[i] = var1;
            var1 *= nums[i];
        }
        // Calculate the product of all elements to the right of each index
        int var2 = 1;
        for (int i = n - 1; i >= 0; i--) {
            result[i] *= var2;
            var2 *= nums[i];
        }
        return result;
    }
}
