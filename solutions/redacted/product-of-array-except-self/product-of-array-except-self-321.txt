class Solution {

    public int[] method1(int[] A) {
        //create 2 arrays to store left and right products of the given array respectively
        int[] left = new int[A.length];
        int[] right = new int[A.length];
        //fill out the left array/ since, the first element in the A doesn\'t have anything to the left
        //we will treat the product as 1
        left[0] = 1;
        for (int i = 1; i < left.length; i++) {
            left[i] = left[i - 1] * A[i - 1];
        }
        //fill out the right array. Since we are traversing from the right, we will assign the
        //last element to 1
        right[right.length - 1] = 1;
        for (int j = right.length - 2; j >= 0; j--) {
            right[j] = right[j + 1] * A[j + 1];
        }
        //create a result array
        int[] result = new int[A.length];
        for (int i = 0; i < result.length; i++) {
            result[i] = left[i] * right[i];
        }
        return result;
    }
}
