class Solution {

    public int[] method1(int[] nums) {
        int p = 1;
        int c = 0;
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                k = 1;
                p = p * nums[i];
            } else {
                c = c + 1;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (c > 1) {
                nums[i] = 0;
            } else if (c == 1 && nums[i] != 0) {
                nums[i] = 0;
            } else if (nums[i] != 0) {
                nums[i] = p / nums[i];
            } else if (nums[i] == 0 && k == 1) {
                nums[i] = p;
            } else {
                nums[i] = 0;
            }
        }
        return nums;
    }
}
