class Solution {

    public int[] method1(int[] nums) {
        int[] output = new int[nums.length];
        int var1 = nums[0];
        for (int i = 1; i < nums.length; i++) {
            output[i] = var1;
            var1 = var1 * nums[i];
        }
        var1 = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i == 0) {
                output[i] = var1;
            } else {
                output[i] = output[i] * var1;
            }
            var1 = var1 * nums[i];
        }
        return output;
    }
}
