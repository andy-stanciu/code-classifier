class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        int prodBeforeCurr = 1;
        int prodAfterCurr = 1;
        for (int i = 0; i < n; i++) {
            ans[i] = prodBeforeCurr;
            prodBeforeCurr *= nums[i];
        }
        for (int i = n - 1; i >= 0; i--) {
            ans[i] *= prodAfterCurr;
            prodAfterCurr *= nums[i];
        }
        return ans;
    }
}
