class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int var2 = 1;
        int[] result = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            result[i] = var1;
            var1 *= nums[i];
        }
        for (int i = nums.length - 1; i >= 0; i--) {
            result[i] *= var2;
            var2 *= nums[i];
        }
        return result;
    }
}
