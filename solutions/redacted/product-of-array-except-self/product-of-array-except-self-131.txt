class Solution {

    public int[] method1(int[] param1) {
        int N = param1.length;
        int[] var1 = new int[N];
        var1[0] = 1;
        for (int i = 1; (i < N); i++) {
            var1[i] = (var1[(i - 1)] * param1[(i - 1)]);
        }
        int var2 = 1;
        for (int i = (N - 1); (i >= 0); i--) {
            var1[i] *= var2;
            var2 *= param1[i];
        }
        return var1;
    }
}
