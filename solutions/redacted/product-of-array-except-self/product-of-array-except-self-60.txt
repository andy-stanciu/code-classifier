class Solution {

    public int[] method1(int[] nums) {
        int x = nums.length;
        int[] answer = new int[x];
        int var1 = 1;
        for (int i = 0; i < x; i++) {
            answer[i] = var1;
            var1 *= nums[i];
        }
        int var2 = 1;
        for (int i = x - 1; i >= 0; i--) {
            answer[i] *= var2;
            var2 *= nums[i];
        }
        return answer;
    }
}
