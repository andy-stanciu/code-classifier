class Solution {

    public int[] method1(int[] nums) {
        int total = nums.length;
        int prefProd = 1;
        int[] ans = new int[total];
        for (int i = 0; i < total; i++) {
            ans[i] = prefProd;
            prefProd *= nums[i];
        }
        int sufProd = 1;
        for (int i = total - 1; i >= 0; i--) {
            ans[i] *= sufProd;
            sufProd *= nums[i];
        }
        return ans;
    }
}
