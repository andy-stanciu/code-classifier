class Solution {

    public int[] method1(int[] nums) {
        int[] ans = new int[nums.length];
        int var1 = 1;
        for (int i = 0; i < nums.length; i++) {
            var1 *= nums[i];
            ans[i] = var1;
        }
        var1 = 1;
        for (int i = nums.length - 1; i > 0; i--) {
            ans[i] = ans[i - 1] * var1;
            var1 *= nums[i];
        }
        ans[0] = var1;
        return ans;
    }
}
