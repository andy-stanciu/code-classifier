class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] premul = new int[n];
        int pro = 1;
        for (int i = 0; i < n; i++) {
            pro *= nums[i];
            premul[i] = pro;
        }
        int[] ans = new int[n];
        int s = 1;
        for (int i = n - 1; i >= 1; i--) {
            ans[i] = premul[i - 1] * s;
            s = s * nums[i];
        }
        ans[0] = s;
        return ans;
    }
}
