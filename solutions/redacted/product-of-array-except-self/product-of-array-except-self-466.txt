class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        // Step 1: Calculate left products
        // The first element has no left elements
        answer[0] = 1;
        for (int i = 1; i < n; i++) {
            answer[i] = answer[i - 1] * nums[i - 1];
        }
        // Step 2: Calculate right products on the go and multiply with left products
        // Initially, there\'s no right element for the last element
        int var1 = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] = answer[i] * var1;
            // Update rightProduct for the next iteration
            var1 *= nums[i];
        }
        return answer;
    }
}
