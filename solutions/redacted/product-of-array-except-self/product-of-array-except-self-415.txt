class Solution {

    public int[] method1(int[] nums) {
        int a = 1;
        int b = 1;
        int countZ = 0;
        int[] result = new int[nums.length];
        for (int i : nums) {
            if (i == 0) {
                b = a;
                countZ++;
            }
            if (i != 0) {
                b *= i;
            }
            a *= i;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0 && countZ == 1) {
                result[i] = b;
            } else if (nums[i] == 0 && countZ > 1) {
                result[i] = 0;
            } else {
                result[i] = a / nums[i];
            }
        }
        return result;
    }
}
