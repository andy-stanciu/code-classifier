class Solution {

    public int[] method1(int[] nums) {
        int[] result = new int[nums.length];
        int prefix = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            result[i + 1] = prefix * nums[i];
            prefix = result[i + 1];
        }
        int postfix = 1;
        for (int i = result.length - 1; i > 0; i--) {
            result[i] *= postfix;
            postfix *= nums[i];
        }
        result[0] = postfix;
        return result;
    }
}
