class Solution {

    public int[] method1(int[] arr) {
        int var1 = 1;
        int zero = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != 0) {
                var1 *= arr[i];
            } else {
                zero++;
            }
        }
        for (int j = 0; j < arr.length; j++) {
            if (zero > 1) {
                arr[j] = 0;
            } else if (zero == 1) {
                if (arr[j] == 0) {
                    arr[j] = var1;
                } else {
                    arr[j] = 0;
                }
            } else {
                arr[j] = var1 / arr[j];
            }
        }
        return arr;
    }
}
