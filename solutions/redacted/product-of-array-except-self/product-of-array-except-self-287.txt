class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        int[] output = new int[n];
        if (n < 1)
            return output;
        int var1 = 1;
        for (int i = 0; i < n; i++) {
            var1 *= nums[i];
            output[i] = var1;
        }
        var1 = 1;
        for (int i = n - 1; i > 0; i--) {
            output[i] = output[i - 1] * var1;
            var1 *= nums[i];
        }
        output[0] = var1;
        return output;
    }
}
