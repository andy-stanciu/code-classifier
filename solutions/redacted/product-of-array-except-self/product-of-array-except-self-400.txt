class Solution {

    public int[] method1(int[] nums) {
        int N = nums.length;
        int[] ans = new int[N];
        Arrays.fill(ans, 1);
        int prefPdt = 1, suffPdt = 1;
        for (int i = 0, j = N - 1; i < N && j >= 0; i++, j--) {
            ans[i] *= prefPdt;
            prefPdt *= nums[i];
            ans[j] *= suffPdt;
            suffPdt *= nums[j];
        }
        return ans;
    }
}
