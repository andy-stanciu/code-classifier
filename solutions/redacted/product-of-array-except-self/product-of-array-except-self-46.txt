class Solution {

    public int[] method1(int[] nums) {
        return method2(nums);
    }

    public static int[] method2(int[] nums) {
        int[] arr = new int[nums.length];
        int pre = 1, post = 1;
        for (int i = 0; i < nums.length; ++i) {
            arr[i] = pre;
            pre *= nums[i];
        }
        for (int i = nums.length - 1; i >= 0; --i) {
            arr[i] = post * arr[i];
            post *= nums[i];
        }
        return arr;
    }
}
