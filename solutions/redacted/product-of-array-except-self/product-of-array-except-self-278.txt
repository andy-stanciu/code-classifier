class Solution {

    public int[] method1(int[] nums) {
        int[] front = new int[nums.length];
        int[] back = new int[nums.length];
        int frontTracker = 1;
        int backTracker = 1;
        int l = nums.length;
        for (int i = l - 1; i >= 0; i--) {
            front[l - i - 1] = frontTracker * nums[l - i - 1];
            frontTracker = front[l - i - 1];
            back[i] = backTracker * nums[i];
            backTracker = back[i];
        }
        for (int i = 0; i < l; i++) {
            if (i == 0)
                nums[i] = back[i + 1];
            else if (i == l - 1)
                nums[i] = front[i - 1];
            else
                nums[i] = back[i + 1] * front[i - 1];
        }
        return nums;
    }
}
