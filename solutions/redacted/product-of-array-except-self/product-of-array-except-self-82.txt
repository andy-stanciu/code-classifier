class Solution {

    public int[] method1(int[] nums) {
        // Initilize array variable ans size of nums.length.
        int[] ans = new int[nums.length];
        // Make ans\'s 0th position as a 1, because we make prefix multiplication except ith position.(we count i-1th position prefix)
        ans[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            // Itearte over the loop and add prefix multiplication of array nums.
            ans[i] = ans[i - 1] * nums[i - 1];
        }
        // Initilize  int variable suffix as a 1, because we need to add suffix multiplication to ans array.(Multiplication by 1 makes no difference)
        int suffix = 1;
        for (int i = ans.length - 2; i >= 0; i--) {
            // Loop through the length of ans from 2 less to 0 and count the suffix multiplication and add it to the ith position of ans.
            suffix *= nums[i + 1];
            ans[i] *= suffix;
        }
        // return array ans.
        return ans;
    }
}
