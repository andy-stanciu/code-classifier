class Solution {

    public int[] method1(int[] nums) {
        int var1 = 1;
        int zeros = 0;
        for (int n : nums) {
            if (n == 0) {
                zeros++;
            }
            if (n != 0) {
                var1 *= n;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (zeros > 0 && nums[i] != 0) {
                nums[i] = 0;
            } else if (zeros == 1) {
                if (nums[i] == 0) {
                    nums[i] = var1;
                } else {
                    nums[i] = 0;
                }
            } else if (zeros == 0) {
                nums[i] = var1 / nums[i];
            }
        }
        return nums;
    }
}
