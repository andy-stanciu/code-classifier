class Solution {

    public int[] method1(int[] nums) {
        int[] res = new int[nums.length];
        int zero = 0;
        int mul = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zero++;
            } else {
                mul *= nums[i];
            }
        }
        if (zero == 0) {
            for (int i = 0; i < nums.length; i++) {
                res[i] = mul / nums[i];
            }
        } else if (zero == 1) {
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] == 0) {
                    res[i] = mul;
                }
            }
        }
        return res;
    }
}
