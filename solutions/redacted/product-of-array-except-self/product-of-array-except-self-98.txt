class Solution {

    public int[] method1(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return new int[0];
        int[] answer = new int[n];
        int var1 = 1;
        for (int i = 0; i < n; i++) {
            answer[i] = var1;
            var1 *= nums[i];
        }
        int var2 = 1;
        for (int i = n - 1; i >= 0; i--) {
            answer[i] *= var2;
            var2 *= nums[i];
        }
        return answer;
    }
}
