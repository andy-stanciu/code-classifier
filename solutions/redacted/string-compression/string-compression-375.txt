class Solution {

    public int compress(char[] chars) {
        StringBuilder s = new StringBuilder();
        int count = 1;
        for (int i = 0; i < chars.length; i++) {
            if (i < chars.length - 1 && chars[i] == chars[i + 1]) {
                count++;
            } else {
                s.append(chars[i]);
                if (count > 1) {
                    s.append(count);
                }
                count = 1;
            }
        }
        for (int i = 0; i < s.length(); i++) {
            chars[i] = s.charAt(i);
        }
        return s.length();
    }
}
