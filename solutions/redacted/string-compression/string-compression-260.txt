class Solution {

    public int compress(char[] chars) {
        int i = 0;
        int j = 0;
        int cnt = 1;
        do {
            i++;
            if (i < chars.length && chars[i] == chars[i - 1]) {
                cnt++;
            } else {
                chars[j++] = chars[i - 1];
                if (cnt > 1) {
                    for (char c : String.valueOf(cnt).toCharArray()) chars[j++] = c;
                    cnt = 1;
                }
            }
        } while (i < chars.length);
        return j;
    }
}
