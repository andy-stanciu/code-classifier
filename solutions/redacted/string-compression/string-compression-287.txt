class Solution {

    public int compress(char[] arr) {
        StringBuilder s = new StringBuilder("");
        int i = 0;
        int j = 0;
        while (j < arr.length) {
            if (arr[i] == arr[j])
                j++;
            else {
                s.append(arr[i]);
                int length = j - i;
                if (length > 1)
                    s.append(length);
                i = j;
            }
        }
        s.append(arr[i]);
        int length = j - i;
        if (length > 1)
            s.append(length);
        for (i = 0; i < s.length(); i++) {
            arr[i] = s.charAt(i);
        }
        return s.length();
    }
}
