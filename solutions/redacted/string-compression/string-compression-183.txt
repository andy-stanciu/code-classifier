class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int index = 0;
        int i = 0;
        while (i < n) {
            //take current char and
            char curr_char = chars[i];
            //take current count
            int count = 0;
            // Find count of duplicates
            while (i < n && chars[i] == curr_char) {
                count++;
                i++;
            }
            //Now Assign it to chars and move index ahead to add the count
            chars[index] = curr_char;
            index++;
            //Now Add the count
            if (count > 1) {
                //to convert the integer count into a string so that each digit can be individually stored in the chars array.
                String count_str = Integer.toString(count);
                //suppose if we get count 10 then extract each digit of the number and assign it to next index.
                for (char ch : count_str.toCharArray()) {
                    chars[index] = ch;
                    index++;
                }
            }
        }
        return index;
    }
}
