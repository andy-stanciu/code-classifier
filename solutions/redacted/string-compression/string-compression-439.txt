class Solution {

    public int compress(char[] chars) {
        // Current index to write compressed characters
        int index = 0;
        // Current index for iterating through chars array
        int i = 0;
        while (i < chars.length) {
            char currentChar = chars[i];
            // Count of consecutive characters
            int count = 0;
            // Count consecutive characters
            while (i < chars.length && chars[i] == currentChar) {
                i++;
                count++;
            }
            // Write the character
            chars[index++] = currentChar;
            // Write the count if it\'s greater than 1
            if (count > 1) {
                for (char c : Integer.toString(count).toCharArray()) {
                    chars[index++] = c;
                }
            }
        }
        // Length of the compressed array
        return index;
    }
}
