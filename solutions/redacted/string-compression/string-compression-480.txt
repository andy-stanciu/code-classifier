class Solution {

    public int compress(char[] chars) {
        StringBuilder s = new StringBuilder();
        int count = 0;
        char c = 0;
        for (int i = 0; i < chars.length; i++) {
            if (c != 0 && c != chars[i]) {
                s.append(c);
                if (count > 1) {
                    s.append(count);
                }
                count = 0;
            }
            c = chars[i];
            count++;
            if (i == chars.length - 1) {
                s.append(c);
                if (count > 1) {
                    s.append(count);
                }
                count = 0;
            }
        }
        // Copy compressed string back to chars array
        char[] compressedArray = s.toString().toCharArray();
        System.arraycopy(compressedArray, 0, chars, 0, compressedArray.length);
        return s.length();
    }
}
