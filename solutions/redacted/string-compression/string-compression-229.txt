class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        int i = 1;
        int count = 1;
        //append the first character to SB
        sb.append(chars[0]);
        while (i < chars.length) {
            //if current character is same as prev charactrer
            if (chars[i] == chars[i - 1]) {
                count++;
            } else {
                //if current character is diff and count is greater than one
                if (count > 1) {
                    sb.append(count);
                }
                //append new character to SB
                sb.append(chars[i]);
                count = 1;
            }
            i++;
        }
        //append the count of last group of characters if it greater than 1
        if (count > 1) {
            sb.append(count);
        }
        //copy characters of SB back to char array
        for (int j = 0; j < sb.length(); j++) {
            chars[j] = sb.charAt(j);
        }
        //return the new length of compressed array
        return sb.length();
    }
}
