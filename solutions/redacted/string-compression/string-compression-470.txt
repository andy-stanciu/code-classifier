class Solution {

    public int compress(char[] chars) {
        int current = 0;
        int count = 0;
        for (int index = 0; index < chars.length; index++) {
            count++;
            if (index == chars.length - 1 || chars[index] != chars[index + 1]) {
                chars[current++] = chars[index];
                if (count != 1) {
                    for (char value : Integer.toString(count).toCharArray()) {
                        chars[current++] = value;
                    }
                }
                count = 0;
            }
        }
        return current;
    }
}
