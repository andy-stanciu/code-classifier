class Solution {

    public int compress(char[] chars) {
        int i = 0, j = 0, index = 0;
        while (i < chars.length) {
            j = i + 1;
            while (j < chars.length && chars[i] == chars[j]) j++;
            chars[index++] = chars[i];
            int count = j - i;
            if (count > 1) {
                String st = Integer.toString(count);
                for (int k = 0; k < st.length(); k++) chars[index++] = st.charAt(k);
            }
            i = j;
        }
        return index;
    }
}
