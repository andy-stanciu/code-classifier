class Solution {

    public int compress(char[] chars) {
        // use i pointer to traverse and j pointer to write into the array
        int count = 0, i = 0, j = 0;
        // traverse the array
        while (i < chars.length) {
            char ch = chars[i];
            count = 0;
            // get count of every char
            while (i < chars.length && chars[i] == ch) {
                i++;
                count++;
            }
            // first write the char
            chars[j] = ch;
            j++;
            // if count is more than 1 then write the count
            if (count > 1) {
                String val = String.valueOf(count);
                for (int k = 0; k < val.length(); k++) {
                    chars[j] = val.charAt(k);
                    j++;
                }
            }
        }
        // j is the final length of chars
        return j;
    }
}
