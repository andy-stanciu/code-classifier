class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        if (n == 0)
            return 0;
        // This will track the position in the chars array where we write the next character.
        int index = 0;
        // Initialize count to 1 for the first character.
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (chars[i] == chars[i - 1]) {
                // Increment count if the current character is the same as the previous one.
                count++;
            } else {
                // Write the character and its count (if > 1) to the chars array.
                chars[index++] = chars[i - 1];
                if (count > 1) {
                    for (char c : Integer.toString(count).toCharArray()) {
                        chars[index++] = c;
                    }
                }
                // Reset count to 1 for the new character.
                count = 1;
            }
        }
        // Handle the last group of characters.
        chars[index++] = chars[n - 1];
        if (count > 1) {
            for (char c : Integer.toString(count).toCharArray()) {
                chars[index++] = c;
            }
        }
        return index;
    }
}
