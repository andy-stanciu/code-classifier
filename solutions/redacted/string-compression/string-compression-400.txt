class Solution {

    public int compress(char[] chars) {
        // Index for writing compressed characters
        int i = 0;
        // Initialize count for consecutive occurrences
        int count = 1;
        for (int j = 1; j <= chars.length; j++) {
            if (j == chars.length || chars[j] != chars[j - 1]) {
                // Store the current character
                chars[i++] = chars[j - 1];
                // If count is greater than 1, convert it to a string and store each digit
                if (count > 1) {
                    for (char digit : Integer.toString(count).toCharArray()) {
                        chars[i++] = digit;
                    }
                }
                // Reset count for the next character
                count = 1;
            } else {
                // Increment count for consecutive occurrences
                count++;
            }
        }
        // Return the length of the compressed array
        return i;
    }
}
