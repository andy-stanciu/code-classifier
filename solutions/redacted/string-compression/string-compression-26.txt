class Solution {

    public int compress(char[] chars) {
        //final string
        String str = "";
        //number of occurence encountered
        int count = 0;
        //contains the first element
        char current = chars[0];
        for (char val : chars) {
            if (val != current) {
                //diffrent character found!!
                if (//more than one value found
                count > 1)
                    str += current + "" + count;
                else
                    //only one value found
                    str += current;
                //refresh
                count = 1;
                //refresh
                current = val;
            } else
                count++;
        }
        // still the last char encountered is not added to the str
        if (//more than one value found
        count > 1)
            str += current + "" + count;
        else
            //only one value found
            str += current;
        // time to add the final str to chars
        for (int i = 0; i < str.length(); i++) {
            chars[i] = str.charAt(i);
        }
        return str.length();
    }
}
