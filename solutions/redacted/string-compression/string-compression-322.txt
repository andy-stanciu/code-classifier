class Solution {

    public int compress(char[] chars) {
        int count = 0;
        if (chars.length == 1) {
            return 1;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < chars.length; i++) {
            count++;
            if (i + 1 >= chars.length || chars[i] != chars[i + 1]) {
                sb.append(chars[i]);
                if (count == 1) {
                    count = 0;
                    continue;
                }
                sb.append(count);
                count = 0;
            }
        }
        int ans = sb.length();
        for (int i = 0; i < ans; i++) {
            chars[i] = sb.charAt(i);
        }
        return ans;
    }
}
