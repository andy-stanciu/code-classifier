class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        if (n == 1)
            return 1;
        int i = 0;
        int iprint = 0;
        while (i < n) {
            int count = 0;
            char curr = chars[i];
            while (i < n && chars[i] == curr) {
                count++;
                i++;
            }
            chars[iprint] = curr;
            iprint++;
            if (count > 1) {
                char[] arr = Integer.toString(count).toCharArray();
                for (char dig : arr) {
                    chars[iprint] = dig;
                    iprint++;
                }
            }
        }
        return iprint;
    }
}
