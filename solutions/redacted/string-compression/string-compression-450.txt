class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        StringBuilder sb = new StringBuilder();
        int freq = 1;
        // Iterate through the characters in the input array
        for (int i = 0; i < n; i++) {
            char c = chars[i];
            // Count consecutive occurrences of the current character
            while (i < n - 1 && chars[i] == chars[i + 1]) {
                freq++;
                i++;
            }
            // Append the current character to the StringBuilder
            sb.append(c);
            // If the frequency is greater than 1, append it to the StringBuilder
            if (freq > 1) {
                sb.append(freq);
            }
            // Reset the frequency for the next character
            freq = 1;
        }
        // Convert the compressed string to an array of characters
        String compressedStr = sb.toString();
        char[] compressedChars = compressedStr.toCharArray();
        // Copy the compressed characters back to the original array
        for (int i = 0; i < compressedChars.length; i++) {
            chars[i] = compressedChars[i];
        }
        // Return the length of the compressed array
        return compressedChars.length;
    }
}
