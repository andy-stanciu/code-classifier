class Solution {

    public int compress(char[] chars) {
        int len = chars.length;
        int left = 0;
        int right = 0;
        int index = 0;
        while (left < len) {
            char c = chars[left];
            int count = 0;
            while (right < len && chars[right] == c) {
                count++;
                right++;
            }
            if (count == 1) {
                chars[index++] = c;
            } else {
                chars[index++] = c;
                for (char j : String.valueOf(count).toCharArray()) {
                    chars[index++] = j;
                }
            }
            left = right;
        }
        return index;
    }
}
