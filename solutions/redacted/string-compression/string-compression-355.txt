class Solution {

    public int compress(char[] chars) {
        int iterator = 0;
        int count = 0;
        StringBuilder result = new StringBuilder();
        while (iterator < chars.length) {
            char current = chars[iterator];
            count = 1;
            result.append(current);
            iterator++;
            while (iterator < chars.length && chars[iterator] == current) {
                count++;
                iterator++;
            }
            if (count > 1)
                result.append(count);
        }
        result.getChars(0, result.length(), chars, 0);
        return result.length();
    }
}
