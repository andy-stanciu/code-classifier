class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        int count = 1;
        // per description spaces are not allowed
        char previous = chars[0];
        for (int i = 1; i < chars.length; i++) {
            // if the char is not previous add it
            if (previous != chars[i]) {
                sb.append(previous);
                if (count > 1) {
                    sb.append(Integer.toString(count));
                }
                previous = chars[i];
                count = 1;
            } else {
                count++;
            }
        }
        sb.append(previous);
        if (count > 1) {
            sb.append(Integer.toString(count));
        }
        sb.getChars(0, sb.length(), chars, 0);
        return sb.length();
    }
}
