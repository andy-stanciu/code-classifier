class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder("");
        for (int i = 0; i < chars.length; i++) {
            Integer count = 1;
            while (i < chars.length - 1 && chars[i] == chars[i + 1]) {
                count++;
                i++;
            }
            sb.append(chars[i]);
            if (count > 1) {
                sb.append(count);
            }
        }
        char[] comp = sb.toString().toCharArray();
        for (int i = 0; i < comp.length; i++) {
            chars[i] = comp[i];
        }
        return comp.length;
    }
}
