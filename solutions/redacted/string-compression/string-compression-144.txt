class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        if (n == 1) {
            return 1;
        }
        char target = chars[0];
        int count = 1;
        StringBuilder result = new StringBuilder();
        for (int i = 1; i < n; i++) {
            if (chars[i] == target) {
                count++;
            } else {
                if (count == 1) {
                    result.append(target);
                } else {
                    result.append(target);
                    result.append(count);
                }
                target = chars[i];
                count = 1;
            }
        }
        if (count == 1) {
            result.append(target);
        } else {
            result.append(target);
            result.append(count);
        }
        //chars = new char[result.length()];
        for (int i = 0; i < result.length(); i++) {
            chars[i] = result.charAt(i);
        }
        return result.length();
    }
}
