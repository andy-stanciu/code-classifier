class Solution {

    public int compress(char[] chars) {
        int n = chars.length;
        int write = 0, read = 0;
        if (n == 1)
            return 1;
        while (read < n) {
            char cur = chars[read];
            int count = 0;
            while (read < n && chars[read] == cur) {
                read++;
                count++;
            }
            chars[write++] = cur;
            if (count > 1) {
                for (char digit : Integer.toString(count).toCharArray()) {
                    chars[write++] = digit;
                }
            }
        }
        return write;
    }
}
