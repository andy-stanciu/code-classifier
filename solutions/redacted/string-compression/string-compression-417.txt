class Solution {

    public int compress(char[] chars) {
        StringBuilder sb = new StringBuilder();
        int i = 0;
        int j = 1;
        int count = 1;
        while (j < chars.length) {
            if (chars[i] == chars[j]) {
                count++;
                j++;
            } else {
                sb.append(chars[i]);
                if (count > 1) {
                    sb.append(count);
                }
                count = 1;
                i = j;
                j++;
            }
        }
        sb.append(chars[i]);
        if (count > 1) {
            sb.append(count);
        }
        for (int k = 0; k < sb.length(); k++) {
            chars[k] = sb.charAt(k);
        }
        return sb.length();
    }
}
