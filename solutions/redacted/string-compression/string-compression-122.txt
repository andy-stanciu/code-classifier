class Solution {

    public int compress(char[] chars) {
        // Index where we\'ll write the compressed character
        int writeIndex = 0;
        // Count the occurrences of a character
        int count = 1;
        for (int i = 0; i < chars.length; i++) {
            // If we\'re at the last character or the next character is different
            if (i == chars.length - 1 || chars[i] != chars[i + 1]) {
                // Write the current character
                chars[writeIndex++] = chars[i];
                // If the count is more than 1, write the count
                if (count > 1) {
                    // Convert count to a string and write each digit
                    for (char c : Integer.toString(count).toCharArray()) {
                        chars[writeIndex++] = c;
                    }
                }
                // Reset count for the next character
                count = 1;
            } else {
                // If the next character is the same, increase the count
                count++;
            }
        }
        // The length of the compressed string
        return writeIndex;
    }
}
