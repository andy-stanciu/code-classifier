class Solution {

    public int compress(char[] chars) {
        if (chars.length < 2)
            return chars.length;
        int i = 0;
        int j = 0;
        while (i < chars.length) {
            char ch = chars[i];
            int count = 0;
            while (i < chars.length && chars[i] == ch) {
                count++;
                i++;
            }
            chars[j++] = ch;
            if (count > 1) {
                for (char digit : Integer.toString(count).toCharArray()) {
                    chars[j++] = digit;
                }
            }
        }
        return j;
    }
}
