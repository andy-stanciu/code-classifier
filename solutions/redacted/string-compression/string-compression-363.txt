class Solution {

    public int compress(char[] chars) {
        StringBuilder str = new StringBuilder();
        int cnt = 1, i;
        int len = 0;
        for (i = 0; i < chars.length - 1; i++) {
            if (chars[i] == chars[i + 1])
                cnt++;
            else {
                len++;
                str.append(chars[i]);
                if (cnt > 1) {
                    str.append(cnt);
                }
                cnt = 1;
            }
        }
        str.append(chars[i]);
        if (cnt > 1) {
            str.append(cnt);
        }
        for (i = 0; i < str.length(); i++) {
            chars[i] = str.charAt(i);
        }
        return str.length();
    }
}
