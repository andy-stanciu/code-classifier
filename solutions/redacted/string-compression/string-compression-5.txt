class Solution {

    public int compress(char[] chars) {
        if (chars.length == 0) {
            // Handle empty input
            return 0;
        }
        // create writing index and count as 1
        int writeIdx = 0;
        int count = 1;
        // compare against this char.
        char ch = chars[0];
        for (int i = 1; i < chars.length; i++) {
            // current char is same as start index, increase count
            if (chars[i] == ch) {
                count++;
            } else {
                // Write the previous character
                chars[writeIdx++] = chars[i - 1];
                // append the count
                writeIdx = appendCount(chars, writeIdx, count);
                // Reset count for the next character
                count = 1;
                // Reset char for next comparison
                ch = chars[i];
            }
        }
        // Handle the last group : add the last char
        chars[writeIdx++] = chars[chars.length - 1];
        // append the count
        if (count > 1) {
            writeIdx = appendCount(chars, writeIdx, count);
        }
        return writeIdx;
    }

    private int appendCount(char[] chars, int writeIdx, int count) {
        if (count > 1) {
            String countStr = String.valueOf(count);
            for (char digit : countStr.toCharArray()) {
                chars[writeIdx++] = digit;
            }
        }
        return writeIdx;
    }
}
