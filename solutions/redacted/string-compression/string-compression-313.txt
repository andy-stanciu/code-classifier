class Solution {

    public int compress(char[] chars) {
        if (chars == null || chars.length == 0) {
            return 0;
        }
        // Index to write compressed characters
        int writeIndex = 0;
        // Index to read characters from the original array
        int readIndex = 0;
        // Count of consecutive characters
        int count = 1;
        for (readIndex = 0; readIndex < chars.length; readIndex++) {
            if (readIndex + 1 < chars.length && chars[readIndex] == chars[readIndex + 1]) {
                count++;
            } else {
                chars[writeIndex++] = chars[readIndex];
                if (count > 1) {
                    for (char c : Integer.toString(count).toCharArray()) {
                        chars[writeIndex++] = c;
                    }
                }
                count = 1;
            }
        }
        // Return the length of the compressed array
        return writeIndex;
    }
}
