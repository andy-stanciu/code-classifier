class Solution {

    public int compress(char[] chars) {
        StringBuffer sb = new StringBuffer();
        ;
        for (int i = 0; i < chars.length; i++) {
            char ch = chars[i];
            int startIndex = i;
            while ((i + 1) < chars.length) {
                if (chars[startIndex] == chars[i + 1]) {
                    i++;
                } else {
                    break;
                }
            }
            if (startIndex == i) {
                sb.append(ch);
            } else {
                int number = i - startIndex + 1;
                sb.append(ch);
                sb.append(number);
            }
        }
        for (int i = 0; i < sb.length(); i++) {
            chars[i] = sb.charAt(i);
        }
        return sb.length();
    }
}
