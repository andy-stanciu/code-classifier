class Solution {

    public int compress(char[] chars) {
        String s = "";
        int i = 1, count = 1, k = 0;
        while (i != chars.length) {
            if (chars[i] == chars[i - 1])
                count++;
            else {
                s += chars[i - 1];
                if (count != 1)
                    s += count;
                count = 1;
            }
            i++;
        }
        s += chars[i - 1];
        if (count != 1)
            s += count;
        for (i = 0; i < s.length(); i++) {
            chars[i] = s.charAt(i);
        }
        return s.length();
    }
}
