class Solution {

    public int compress(char[] chars) {
        int put = 0;
        for (int i = 0; i < chars.length; ) {
            //to count the number of occurences of each char
            int count = 0;
            //finding the number of occurences of each char
            while (i + count < chars.length && chars[i] == chars[i + count]) count++;
            //putting chars[i] at initial place, put++ will give the o value first and later increments to 1
            chars[put++] = chars[i];
            if (count > 1) {
                //1 is not considered
                char[] countstr = Integer.toString(count).toCharArray();
                for (char c : countstr) chars[put++] = c;
            }
            //incrementing
            i += count;
        }
        return put;
    }
}
