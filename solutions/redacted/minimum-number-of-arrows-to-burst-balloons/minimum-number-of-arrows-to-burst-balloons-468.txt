class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int minArrow = 0;
        minArrow = 1;
        int xEnd = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > xEnd) {
                minArrow++;
                xEnd = points[i][1];
            }
        }
        return minArrow;
    }
}
