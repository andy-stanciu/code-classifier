class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        // for (int[] arr : points) System.out.println(Arrays.toString(arr));
        // return 2;
        int i = 1;
        int count = 1;
        int end = points[0][1];
        while (i < points.length) {
            if (points[i][0] > end) {
                count++;
                end = points[i][1];
            }
            i++;
        }
        return count;
    }
}
