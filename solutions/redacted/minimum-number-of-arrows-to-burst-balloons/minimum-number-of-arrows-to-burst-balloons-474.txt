class Solution {

    public int method1(int[][] points) {
        //O(NlogN)
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int minArrow = 1;
        int prev = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > prev) {
                minArrow++;
                prev = points[i][1];
            } else {
                prev = Math.min(prev, points[i][1]);
            }
        }
        return minArrow;
    }
}
