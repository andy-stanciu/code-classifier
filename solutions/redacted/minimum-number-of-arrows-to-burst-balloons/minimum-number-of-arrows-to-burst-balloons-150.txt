class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        int shots = 0;
        for (int i = 0; i < points.length; i++) {
            int[] point = points[i];
            int shortestRightPos = point[1];
            for (int j = i + 1; j < points.length; j++) {
                int[] nextP = points[j];
                if (nextP[0] <= shortestRightPos && nextP[0] >= point[0] && nextP[0] <= point[1]) {
                    if (nextP[1] < shortestRightPos) {
                        shortestRightPos = nextP[1];
                    }
                    i++;
                } else
                    break;
            }
            shots++;
        }
        return shots;
    }
}
