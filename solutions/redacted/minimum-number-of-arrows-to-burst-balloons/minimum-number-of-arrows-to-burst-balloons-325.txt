class Solution {

    public int method1(int[][] points) {
        //sort the balloons based on their end coordinates
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int var1 = 1;
        int end = points[0][1];
        //count the number of non-overlapping intervals
        for (int i = 0; i < points.length; i++) {
            if (points[i][0] > end) {
                var1++;
                end = points[i][1];
            }
        }
        return var1;
    }
}
