class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int pointsLen = points.length;
        int arrowCount = 1;
        int arrowLimit = points[0][1];
        for (int i = 1; i < pointsLen; i++) {
            if (points[i][0] > arrowLimit) {
                arrowLimit = points[i][1];
                arrowCount++;
            }
        }
        return arrowCount;
    }
}
