class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingDouble(o -> o[1]));
        int count = 1;
        int endTime = points[0][1];
        for (int i = 1; i < points.length; i++) {
            int startTime = points[i][0];
            if (startTime > endTime) {
                count++;
                endTime = points[i][1];
            }
        }
        return count;
    }
}
