class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int var1 = 1, end = points[0][1];
        for (int[] point : points) {
            if (point[0] > end) {
                var1++;
                end = point[1];
            }
        }
        return var1;
    }
}
