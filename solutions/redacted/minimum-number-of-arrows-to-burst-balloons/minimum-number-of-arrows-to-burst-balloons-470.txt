class Solution {

    public int method1(int[][] points) {
        if (points.length == 1) {
            return 1;
        }
        Arrays.sort(points, Comparator.comparingInt(o -> o[0]));
        int q = 1;
        int start = points[0][0], end = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] <= end) {
                start = Math.max(points[i][0], start);
                end = Math.min(points[i][1], end);
            } else {
                q++;
                start = points[i][0];
                end = points[i][1];
            }
        }
        return q;
    }
}
