class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int n = points.length;
        int count = 1;
        int val = points[0][1];
        for (int i = 1; i < n; i++) {
            if (points[i][0] > val) {
                count++;
                val = points[i][1];
            }
        }
        return count;
    }
}
