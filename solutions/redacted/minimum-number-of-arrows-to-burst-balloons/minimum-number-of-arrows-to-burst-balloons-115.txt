class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (p1, p2) -> Integer.compare(p1[0], p2[0]));
        int var1 = 1;
        int end = Integer.MAX_VALUE;
        for (int[] point : points) {
            if (point[0] <= end) {
                end = Math.min(end, point[1]);
            } else {
                var1++;
                end = point[1];
            }
        }
        return var1;
    }
}
