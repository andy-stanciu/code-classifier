class Solution {

    public int method1(int[][] points) {
        if (points == null || points.length == 0) {
            // No balloons to burst
            return 0;
        }
        // Sort by ending coordinates
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        // At least one arrow is needed
        int var1 = 1;
        // Ending coordinate of the first balloon
        int end = points[0][1];
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > end) {
                // Current balloon is not overlapping with the previous one
                // Need a new arrow
                var1++;
                // Update the ending coordinate
                end = points[i][1];
            }
            // Otherwise, the current balloon overlaps with the previous one, so continue
        }
        return var1;
    }
}
