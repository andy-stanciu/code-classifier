class Solution {

    public int method1(int[][] points) {
        // sort according to ascending order of Xstart
        Arrays.sort(points, Comparator.comparingInt(a -> a[0]));
        int count = 0;
        int tempS = points[0][0], tempE = points[0][1];
        for (int i = 1; i < points.length; i++) {
            // check if intersections present
            if (points[i][0] <= tempE) {
                tempS = Math.max(tempS, points[i][0]);
                tempE = Math.min(tempE, points[i][1]);
            } else {
                // reset tempS and tempE and add count
                count++;
                tempS = points[i][0];
                tempE = points[i][1];
            }
        }
        // do count the current intersection
        return count + 1;
    }
}
