class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int ans = 1;
        int max = points[0][1];
        int size = points.length;
        for (int i = 1; i < size; i++) {
            if (max < points[i][0]) {
                ans++;
                max = points[i][1];
            }
        }
        return ans;
    }
}
