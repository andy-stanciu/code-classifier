class Solution {

    public int method1(int[][] points) {
        if (points == null || points.length == 0) {
            return 0;
        }
        // Sort balloons based on end points
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int var1 = 1;
        int end = points[0][1];
        // Iterate through balloons
        for (int i = 1; i < points.length; i++) {
            // If current balloon\'s start point is beyond the end point of the previous balloon
            // we need to shoot a new arrow
            if (points[i][0] > end) {
                var1++;
                // Update end point
                end = points[i][1];
            }
        }
        return var1;
    }
}
