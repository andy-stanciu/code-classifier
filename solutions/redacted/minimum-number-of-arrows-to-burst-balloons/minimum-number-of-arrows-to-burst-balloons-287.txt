class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, Comparator.comparingInt(a -> a[1]));
        int n = points.length;
        int i = 0;
        int ans = 0;
        while (i < n) {
            int j = i + 1;
            // If the current balloon\'s end coordinate is greater than or equal
            // to the next balloon\'s start coordinate, move to the next balloon.
            while (i < n && j < n && points[i][1] >= points[j][0]) j++;
            // Increment the arrow count as we burst all balloons in this range.
            ans++;
            // Move to the next balloon that doesn\'t overlap with the current one.
            i = j;
        }
        return ans;
    }
}
