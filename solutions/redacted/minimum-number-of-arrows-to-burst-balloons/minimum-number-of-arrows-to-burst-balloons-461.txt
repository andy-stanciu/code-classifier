class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> (a[0] == b[0] ? a[1] - b[1] : a[0] - b[0]));
        int count = 0;
        int n = points.length;
        for (int i = 0; i < n; i++) {
            int min = points[i][0];
            int max = points[i][1];
            int j = i + 1;
            for (j = i + 1; j < n; j++) {
                int min1 = points[j][0];
                int max1 = points[j][1];
                if ((min1 < min && max1 < min) || (min1 > max && max1 > max)) {
                    break;
                }
                min = Math.max(min, min1);
                max = Math.min(max, max1);
            }
            i = j - 1;
            count++;
        }
        return count;
    }
}
