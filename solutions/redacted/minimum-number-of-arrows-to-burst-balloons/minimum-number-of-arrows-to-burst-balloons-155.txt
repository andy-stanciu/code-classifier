class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1;
        int end = points[0][1];
        int index = 1;
        while (index < points.length) {
            if (points[index][0] > end) {
                end = points[index][1];
                count++;
            }
            index++;
        }
        return count;
    }
}
