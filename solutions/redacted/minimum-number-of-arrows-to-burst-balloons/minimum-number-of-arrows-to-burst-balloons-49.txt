class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int var1 = 1;
        int lastEndPoint = points[0][1];
        for (int i = 0; i < points.length; ++i) {
            if (points[i][0] > lastEndPoint) {
                ++var1;
                lastEndPoint = points[i][1];
            }
        }
        return var1;
    }
}
