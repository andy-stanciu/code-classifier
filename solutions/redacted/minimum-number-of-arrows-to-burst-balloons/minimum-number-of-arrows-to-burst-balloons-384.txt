class Solution {

    public int method1(int[][] points) {
        int n = points.length;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int prev = 0;
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (points[i][0] > points[prev][1]) {
                prev = i;
                count++;
            }
        }
        return count;
    }
}
