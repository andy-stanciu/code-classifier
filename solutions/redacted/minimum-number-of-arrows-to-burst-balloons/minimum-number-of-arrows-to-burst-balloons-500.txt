class Solution {

    public int method1(int[][] points) {
        // sorting array on the basis of end point of balloons
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int arrow = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (count == 0 || points[i][0] > arrow) {
                // increasing the count if starting of new balloon is less than current marker
                count++;
                arrow = points[i][1];
            }
        }
        return count;
    }
}
