class Solution {

    public int method1(int[][] points) {
        //int l=points[0][0];
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int r = points[0][1];
        int count = 1;
        int i = 1;
        while (i < points.length) {
            if (points[i][0] > r) {
                count++;
                r = points[i][1];
            } else {
                i++;
            }
        }
        return count;
    }
}
