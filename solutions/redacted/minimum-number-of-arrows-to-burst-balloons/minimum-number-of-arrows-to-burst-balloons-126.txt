class Solution {

    public int method1(int[][] p) {
        Arrays.sort(p, (a, b) -> Integer.compare(a[1], b[1]));
        if (p.length == 1)
            return 1;
        int arrow = p[0][1], count = 1;
        for (int i = 1; i < p.length; i++) {
            if (!(p[i][0] <= arrow && arrow <= p[i][1])) {
                count++;
                arrow = p[i][1];
            }
        }
        return count;
    }
}
