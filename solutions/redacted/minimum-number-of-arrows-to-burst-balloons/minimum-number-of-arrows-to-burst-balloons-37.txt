class Solution {

    public int method1(int[][] arr) {
        Arrays.sort(arr, (a, b) -> Integer.compare(a[1], b[1]));
        int i = 0;
        int cnt = 0;
        int x = arr[0][1];
        while (i < arr.length) {
            if (arr[i][0] > x) {
                cnt++;
                x = arr[i][1];
            }
            i++;
        }
        return cnt + 1;
    }
}
