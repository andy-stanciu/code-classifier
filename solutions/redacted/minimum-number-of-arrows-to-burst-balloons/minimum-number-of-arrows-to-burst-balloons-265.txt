class Solution {

    public int method1(int[][] points) {
        int count = 1;
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int prev = 0;
        for (int cur = 1; cur < points.length; cur++) {
            if (points[cur][0] > points[prev][1]) {
                count++;
                prev = cur;
            }
        }
        return count;
    }
}
