class Solution {

    public int method1(int[][] points) {
        if (points.length == 0)
            return 0;
        // sorting ballons according to end points
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        //position of first balloon
        int pos = points[0][1];
        int ans = 1;
        for (int i = 1; i < points.length; i++) {
            if (pos < points[i][0]) {
                //updating the new position because 2nd arrow need to be shooted
                pos = points[i][1];
                // increasing the no. of shot by 1
                ans++;
            }
        }
        return ans;
    }
}
