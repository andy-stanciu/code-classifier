class Solution {

    public int method1(int[][] points) {
        if (points == null || points.length == 0) {
            return 0;
        }
        // Sort the balloons based on their end position
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        // Start with one arrow
        int var1 = 1;
        // Initialize the end position with the first balloon\'s end position
        int end = points[0][1];
        // Iterate through the sorted balloons
        for (int i = 1; i < points.length; i++) {
            // If the start position of the current balloon is greater than the current end position,
            // shoot a new arrow and update the end position
            if (points[i][0] > end) {
                var1++;
                end = points[i][1];
            }
        }
        return var1;
    }
}
