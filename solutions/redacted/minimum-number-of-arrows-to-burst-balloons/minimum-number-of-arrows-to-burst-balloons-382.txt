class Solution {

    public int method1(int[][] points) {
        final int len = points.length;
        Arrays.sort(points, (p, q) -> Integer.compare(p[1], q[1]));
        int var1 = 1;
        int end = points[0][1];
        for (int i = 1; i < len; ++i) {
            int[] point = points[i];
            if (end < point[0]) {
                ++var1;
                end = point[1];
            }
        }
        return var1;
    }
}
