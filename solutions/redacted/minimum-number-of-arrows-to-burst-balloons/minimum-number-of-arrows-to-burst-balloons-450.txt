class Solution {

    public int method1(int[][] points) {
        //sort the array
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));
        // we need atleast one arrow to burst
        int var1 = 1;
        // getting the first limit as the first intervals last element
        int limit = points[0][1];
        for (int i = 0; i < points.length; i++) {
            if (// if the first element of the next interval is not in the range then we need an another arrow to buyrs that arrow;
            // and the range will be the next ballons diamter to cover
            points[i][0] > limit) {
                var1++;
                limit = points[i][1];
            } else {
                //if the limit of the next interval is less than the limit the we have to update the minimum limit
                limit = Math.min(limit, points[i][1]);
            }
        }
        return var1;
    }
}
