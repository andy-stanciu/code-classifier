class Solution {

    public int method1(int[][] arr) {
        Arrays.sort(arr, (a, b) -> Integer.compare(a[0], b[0]));
        int ans = 0;
        long max = Long.MIN_VALUE;
        for (int i = 0; i < arr.length; ++i) {
            if (arr[i][0] > max) {
                ++ans;
                max = arr[i][1];
            } else {
                max = Math.min(max, arr[i][1]);
            }
        }
        return ans;
    }
}
