class Solution {

    public int method1(int[][] p) {
        Arrays.sort(p, (p1, p2) -> Integer.compare(p1[1], p2[1]));
        int arrow = 1;
        int pos = p[0][1];
        for (int i = 1; i < p.length; i++) {
            if (pos >= p[i][0])
                continue;
            arrow++;
            pos = p[i][1];
        }
        return arrow;
    }
}
