class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1;
        int lastEnd = points[0][1];
        int i = 1;
        while (i < points.length) {
            if (points[i][0] > lastEnd) {
                count++;
                lastEnd = points[i][1];
            }
            i++;
        }
        return count;
    }
}
