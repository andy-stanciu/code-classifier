class Solution {

    public int method1(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
        int var1 = 1;
        int end = points[0][1];
        if (points.length == 0)
            return 0;
        for (int i = 1; i < points.length; i++) {
            if (points[i][0] > end) {
                var1++;
                end = points[i][1];
            }
        }
        return var1;
    }
}
