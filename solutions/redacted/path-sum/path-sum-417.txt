class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return param1 == root.val;
        boolean a = method1(root.left, param1 - root.val);
        boolean b = method1(root.right, param1 - root.val);
        root.left = null;
        root.right = null;
        return a || b;
    }
}
