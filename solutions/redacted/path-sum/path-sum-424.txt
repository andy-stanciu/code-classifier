class Solution {

    public boolean method1(TreeNode root, int param1) {
        return sol(0, root, param1);
    }

    public boolean sol(int param2, TreeNode root, int target) {
        if (root == null)
            return false;
        param2 = param2 + root.val;
        if (root.left == null && root.right == null) {
            if (param2 == target)
                return true;
        }
        return sol(param2, root.left, target) || sol(param2, root.right, target);
    }
}
