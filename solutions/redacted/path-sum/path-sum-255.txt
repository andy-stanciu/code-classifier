/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return usignStack(root, param1);
    }

    boolean usignStack(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        Stack<TreeNode> stack = new Stack<>();
        Stack<Integer> var1 = new Stack<>();
        stack.push(root);
        var1.push(root.val);
        while (!stack.isEmpty()) {
            TreeNode node = stack.pop();
            int var2 = var1.pop();
            if (node.left == null && node.right == null) {
                if (var2 == param1) {
                    return true;
                }
            }
            if (node.right != null) {
                stack.push(node.right);
                var1.push(var2 + node.right.val);
            }
            if (node.left != null) {
                stack.push(node.left);
                var1.push(var2 + node.left.val);
            }
        }
        return false;
    }
}
