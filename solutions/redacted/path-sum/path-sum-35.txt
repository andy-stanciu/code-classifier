class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, 0, param1);
    }

    boolean method2(TreeNode node, int param2, int param1) {
        if (node == null)
            return false;
        param2 += node.val;
        //we are adding the condition to identify leaf node. If node.left and node.right==null indicates the "Leaf Node"
        if (node.left == null && node.right == null && param2 == param1)
            return true;
        return method2(node.left, param2, param1) || method2(node.right, param2, param1);
    }
}
