/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean helper(TreeNode root, int param1, int param2) {
        //base
        if (root == null) {
            return false;
        }
        //work
        param1 = (param1 + root.val);
        //leaf node
        if (root.left == null && root.right == null) {
            if (param1 == param2) {
                return true;
            }
            return false;
        }
        //Check for the same in your left and right subtree
        boolean left = helper(root.left, param1, param2);
        boolean right = helper(root.right, param1, param2);
        return (left || right);
    }

    public boolean method1(TreeNode root, int param2) {
        return helper(root, 0, param2);
    }
}
