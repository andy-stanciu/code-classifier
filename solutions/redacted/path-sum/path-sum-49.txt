class Solution {

    boolean result;

    public boolean method1(TreeNode root, int param1) {
        result = false;
        traverseBT(root, 0, param1);
        return result;
    }

    private void traverseBT(TreeNode node, int param2, int param1) {
        if (node == null)
            return;
        param2 += node.val;
        if (node.left == null && node.right == null && param2 == param1) {
            result = true;
            return;
        }
        traverseBT(node.left, param2, param1);
        traverseBT(node.right, param2, param1);
    }
}
