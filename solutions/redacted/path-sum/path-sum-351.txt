/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method1(root, param1, 0);
    }

    private boolean method1(TreeNode root, int param1, int acc) {
        if ((param1 == acc + root.val) && root.left == null && root.right == null) {
            return true;
        }
        if (root.left != null && method1(root.left, param1, acc + root.val)) {
            return true;
        }
        if (root.right != null && method1(root.right, param1, acc + root.val)) {
            return true;
        }
        return false;
    }
}
