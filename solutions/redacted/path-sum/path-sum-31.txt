/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1, 0);
    }

    public boolean method2(TreeNode root, int target, int param2) {
        if (root == null)
            return false;
        param2 += root.val;
        //This means we reached the leaf node
        if (root.left == null && root.right == null && param2 == target)
            return true;
        Boolean left = method2(root.left, target, param2);
        Boolean right = method2(root.right, target, param2);
        return left || right;
    }
}
