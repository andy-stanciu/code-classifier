/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1);
    }

    public boolean method2(TreeNode node, int param1) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            return node.val == param1;
        return method2(node.left, param1 - node.val) || method2(node.right, param1 - node.val);
    }
}
