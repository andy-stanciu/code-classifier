/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public boolean method1(TreeNode root, int param1) {
        //   we will subtract the targetsum with currNode value
        // the moment we reach targetSum =0 , we have found our path , however
        // if we reach lead node(i.e. null) and targetSum is not 0 , that is not the path
        if (root == null) {
            return false;
        }
        Stack<TreeNode> node_stack = new Stack();
        Stack<Integer> var2 = new Stack();
        node_stack.add(root);
        var2.add(param1 - root.val);
        while (!node_stack.isEmpty()) {
            TreeNode currNode = node_stack.pop();
            int var3 = var2.pop();
            if (currNode.left == null && currNode.right == null && var3 == 0)
                return true;
            if (currNode.left != null) {
                node_stack.add(currNode.left);
                var2.add(var3 - currNode.left.val);
            }
            if (currNode.right != null) {
                node_stack.add(currNode.right);
                var2.add(var3 - currNode.right.val);
            }
        }
        // it means you have not found any path
        return false;
    }
}
