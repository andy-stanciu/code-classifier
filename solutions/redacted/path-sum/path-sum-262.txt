/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Pair {

    TreeNode node;

    int var1;

    Pair(TreeNode node, int var1) {
        this.node = node;
        this.var1 = var1;
    }
}

class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        Stack<Pair> s = new Stack<>();
        s.push(new Pair(root, 0));
        while (!s.isEmpty()) {
            Pair pair = s.pop();
            TreeNode node = pair.node;
            int var1 = pair.var1;
            var1 += node.val;
            if (node.left == null && node.right == null) {
                if (var1 == param1)
                    return true;
            }
            if (node.left != null)
                s.push(new Pair(node.left, var1));
            if (node.right != null)
                s.push(new Pair(node.right, var1));
        }
        return false;
    }
}
