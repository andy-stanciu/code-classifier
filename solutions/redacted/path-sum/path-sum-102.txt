/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        // Base case: no tree
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            // Leaf node check
            // Check if leaf value matches target
            return param1 == root.val;
        }
        // Recursive calls for left and right subtrees
        boolean var1 = method1(root.right, param1 - root.val);
        boolean var2 = method1(root.left, param1 - root.val);
        // Return true if either subtree has a valid path
        return var1 || var2;
    }
}
