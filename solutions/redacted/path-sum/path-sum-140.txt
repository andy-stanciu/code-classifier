/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public boolean method1(TreeNode tree, int param1, int total) {
        if (tree == null) {
            return false;
        }
        total = total + tree.val;
        if (tree.left == null && tree.right == null) {
            return total == param1;
        }
        return method1(tree.left, param1, total) || method1(tree.right, param1, total);
    }

    public boolean method2(TreeNode root, int param1) {
        return method1(root, param1, 0);
    }
}
