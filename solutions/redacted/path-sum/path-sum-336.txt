/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean ans = false;

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if ((root.left == null) && (root.right == null) && (root.val == param1))
            return true;
        boolean result = fun(root, param1, 0);
        return result;
    }

    boolean fun(TreeNode root, int param1, int param2) {
        if (ans == true)
            return ans;
        if (root == null)
            return true;
        param2 = param2 + root.val;
        if ((param2 == param1) && (root.left == null) && (root.right == null))
            ans = true;
        fun(root.left, param1, param2);
        fun(root.right, param1, param2);
        return ans;
    }
}
