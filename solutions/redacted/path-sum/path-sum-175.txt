/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        //if it is the leaf node and at the leaf node the value is equal to the targetSum then return true
        if (root.val == param1 && root.left == null && root.right == null) {
            return true;
        }
        //made recursion call for left or right node and at each node reduce the value by the value of the node
        return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
    }
}
