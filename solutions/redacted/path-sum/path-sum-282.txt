/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean Total(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        param1 += root.val;
        if (param1 == param2 && root.left == null && root.right == null)
            return true;
        return Total(root.left, param1, param2) || Total(root.right, param1, param2);
    }

    public boolean method1(TreeNode root, int param2) {
        return Total(root, 0, param2);
    }
}
