class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return helper(root, param1, 0);
    }

    private boolean helper(TreeNode root, int target, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (param2 + root.val == target) {
                return true;
            }
            return false;
        }
        boolean left = false, right = false;
        left = helper(root.left, target, param2 + root.val);
        if (left == false)
            right = helper(root.right, target, param2 + root.val);
        return left || right;
    }
}
