/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int k, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            param1 = param1 + root.val;
            if (param1 == k) {
                return true;
            }
        }
        boolean leftl = method1(root.left, k, param1 + root.val);
        boolean rightl = method1(root.right, k, param1 + root.val);
        return (leftl || rightl);
    }

    public boolean method2(TreeNode root, int param2) {
        if (root == null) {
            return false;
        }
        return method1(root, param2, 0);
    }
}
