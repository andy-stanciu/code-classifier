class Solution {

    public boolean method1(TreeNode root, int param1) {
        // base case
        if (root == null) {
            return false;
        }
        // recursive case
        param1 -= root.val;
        if (param1 == 0 && root.left == null && root.right == null) {
            return true;
        }
        return method1(root.left, param1) || method1(root.right, param1);
    }
}
