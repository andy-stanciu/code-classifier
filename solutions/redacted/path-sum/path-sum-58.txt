class Solution {

    // Has Path Sum Function
    public boolean method1(TreeNode root, int param1) {
        // if, root has no value return false
        if (root == null)
            return false;
        // root left is equal to null & root right is null
        // & root value is equal to targetSum, return true
        if (root.left == null && root.right == null && root.val == param1)
            return true;
        // targetSum = targetSum - root value
        param1 = param1 - root.val;
        // recursive call for left & right subtree
        // if one is true, return true
        return method1(root.left, param1) || method1(root.right, param1);
    }
}
// Output -
/*
Input: root = [5,4,8,11,null,13,4,7,2,null,null,null,1], targetSum = 22
Output: true
Explanation: The root-to-leaf path with the target sum is shown.
*/
// Time & Space Complexity -
/*
Time - O(n)
Space - O(1)
*/
