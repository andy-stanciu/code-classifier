/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return method2(var1, root, param1);
    }

    public boolean method2(int var1, TreeNode root, int target) {
        if (root == null)
            return false;
        var1 += root.val;
        if (var1 == target && root.left == null && root.right == null)
            return true;
        return method2(var1, root.left, target) || method2(var1, root.right, target);
    }
}
