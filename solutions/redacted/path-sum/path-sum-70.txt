class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return method2(root, var1, param1);
    }

    public boolean method2(TreeNode root, int var1, int param1) {
        if (root == null) {
            return false;
        }
        var1 += root.val;
        if (root.left == null && root.right == null) {
            if (var1 == param1) {
                return true;
            } else {
                return false;
            }
        }
        boolean left = method2(root.left, var1, param1);
        boolean right = method2(root.right, var1, param1);
        return left || right;
    }
}
