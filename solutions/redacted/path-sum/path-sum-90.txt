/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, param1, 0);
    }

    private boolean helper(TreeNode root, int param2, int param3) {
        if (root == null)
            return false;
        param3 = param3 + root.val;
        if (root.left == null && root.right == null)
            return param3 == param2;
        return helper(root.left, param2, param3) || helper(root.right, param2, param3);
    }
}
