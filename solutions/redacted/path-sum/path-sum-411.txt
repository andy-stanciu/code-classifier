/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return dfs(root, param1, 0, false);
    }

    static boolean dfs(TreeNode root, int param1, int param2, boolean flag) {
        if (root == null)
            return flag;
        if (root.left == null && root.right == null) {
            return param2 + root.val == param1;
        }
        return dfs(root.left, param1, param2 + root.val, flag) || dfs(root.right, param1, param2 + root.val, flag);
    }
}
