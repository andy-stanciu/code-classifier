class Solution {

    public boolean method1(TreeNode root, int param1) {
        // If the tree is empty i.e. root is NULL, return false...
        if (root == null)
            return false;
        // If there is only a single root node and the value of root node is equal to the targetSum...
        if (root.val == param1 && (root.left == null && root.right == null))
            return true;
        // Call the same function recursively for left and right subtree...
        return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
    }
}
