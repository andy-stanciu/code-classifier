class Solution {

    /**
     * @author Brian A. Miller
     * @date Tuesday, August 27, 2024
     *
     * @leetcode.result Success
     * @leetcode.runtime 0 ms, faster than 100.00% of Java submissions
     * @leetcode.memory 42.8 MB, less than 77.80% of Java submissions
     */
    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if ((root.left == null) && (root.right == null)) {
            // leaf node
            return (root.val == param1);
        } else {
            return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
        }
    }
}
