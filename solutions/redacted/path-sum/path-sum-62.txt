class Solution {

    public boolean method1(TreeNode root, int param1) {
        //check if root is null or not
        if (root == null)
            return false;
        //check if current number is equals to target and it\'s the end of the Tree
        if (param1 == root.val && root.left == null && root.right == null) {
            return true;
        }
        // recursion
        return method1(root.right, param1 - root.val) || method1(root.left, param1 - root.val);
    }
}
