/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean f(TreeNode node, int k) {
        if (node == null) {
            return false;
        }
        if (node.left == null && node.right == null && k - node.val == 0)
            return true;
        return f(node.left, k - node.val) || f(node.right, k - node.val);
    }

    public boolean method1(TreeNode root, int param1) {
        return f(root, param1);
    }
}
