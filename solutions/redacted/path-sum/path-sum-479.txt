/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, param1, 0);
    }

    public static boolean method2(TreeNode root, int ref, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            if (param2 + root.val == ref)
                return true;
        }
        if (method2(root.left, ref, param2 + root.val))
            return true;
        if (method2(root.right, ref, param2 + root.val))
            return true;
        return false;
    }
}
