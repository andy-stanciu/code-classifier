/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            return param1 == root.val ? true : false;
            // Checking last value - present node value == 0
        }
        boolean left = false, right = false;
        if (root.left != null)
            left = method1(root.left, param1 - root.val);
        if (root.right != null)
            right = method1(root.right, param1 - root.val);
        return left || right;
    }
}
