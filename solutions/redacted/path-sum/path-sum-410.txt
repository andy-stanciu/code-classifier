/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean flag = false;

    public void method1(TreeNode root, int param1, int target) {
        param1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == target) {
                flag = true;
            }
            return;
        }
        if (root.left != null)
            method1(root.left, param1, target);
        if (root.right != null)
            method1(root.right, param1, target);
    }

    public boolean method2(TreeNode root, int param2) {
        if (root != null)
            method1(root, 0, param2);
        return flag;
    }
}
