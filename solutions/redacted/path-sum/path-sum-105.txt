/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return dfs(root, param1);
    }

    private boolean dfs(TreeNode root, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            return root.val == param2;
        }
        return dfs(root.left, param2 - root.val) || dfs(root.right, param2 - root.val);
    }
}
