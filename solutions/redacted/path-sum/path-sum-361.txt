/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode node, int param1) {
        if (node == null)
            return false;
        param1 -= node.val;
        if (isLeaf(node) && param1 == 0)
            return true;
        return method1(node.left, param1) || method1(node.right, param1);
    }

    private boolean isLeaf(TreeNode node) {
        return node != null && node.left == null && node.right == null ? true : false;
    }
}
