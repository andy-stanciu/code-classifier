/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return traverseTree(root, param1, 0);
    }

    public boolean traverseTree(TreeNode root, int param1, int total) {
        if (root == null) {
            return false;
        }
        total = root.val + total;
        if (total == param1 && root.left == null && root.right == null) {
            return true;
        }
        return traverseTree(root.left, param1, total) || traverseTree(root.right, param1, total);
    }
}
