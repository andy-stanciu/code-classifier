/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method1(root, param1, 0);
    }

    private boolean method1(TreeNode node, int param1, int param2) {
        if (node == null) {
            return false;
        }
        param2 += node.val;
        if (isLeaf(node)) {
            return param2 == param1;
        }
        return (node.left != null && method1(node.left, param1, param2)) || (node.right != null && method1(node.right, param1, param2));
    }

    private boolean isLeaf(TreeNode node) {
        return node != null && node.right == null && node.left == null;
    }
}
