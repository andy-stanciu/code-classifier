/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        boolean ans = check(root, param1, var1);
        return ans;
    }

    boolean check(TreeNode root, int param1, int var1) {
        if (root == null)
            return false;
        var1 += root.val;
        if (root.left == null && root.right == null) {
            if (var1 == param1)
                return true;
            return false;
        }
        boolean left = check(root.left, param1, var1);
        boolean right = check(root.right, param1, var1);
        return left || right;
    }
}
