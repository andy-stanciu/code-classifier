class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null && param1 == 0)
            return false;
        else
            return method2(root, param1);
    }

    public boolean method2(TreeNode trav, int param1) {
        if (trav == null && param1 == 0)
            return true;
        else if (trav == null)
            return false;
        if (trav.left != null && trav.right != null)
            return method2(trav.left, param1 - trav.val) || method2(trav.right, param1 - trav.val);
        if (trav.left != null)
            return method2(trav.left, param1 - trav.val);
        else
            return method2(trav.right, param1 - trav.val);
    }
}
