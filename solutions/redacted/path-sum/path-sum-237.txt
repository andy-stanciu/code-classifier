/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return check(root, param1, 0);
    }

    boolean check(TreeNode root, int tar, int param2) {
        if (root == null)
            return false;
        param2 += root.val;
        if (root.left == null && root.right == null)
            return param2 == tar;
        return check(root.left, tar, param2) || check(root.right, tar, param2);
    }
}
