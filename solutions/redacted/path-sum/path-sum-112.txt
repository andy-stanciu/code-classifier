/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1);
    }

    public boolean method2(TreeNode root, int target) {
        if (root == null)
            return false;
        target -= root.val;
        // If it\'s a leaf node and target is 0, return true
        if (root.left == null && root.right == null && target == 0) {
            return true;
        }
        // Recursively check left and right subtrees
        return method2(root.left, target) || method2(root.right, target);
    }
}
