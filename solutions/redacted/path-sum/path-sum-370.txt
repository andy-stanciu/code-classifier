class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, param1);
    }

    public boolean method2(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        param1 -= root.val;
        if (param1 == 0 && root.left == null && root.right == null) {
            return true;
        }
        if (method2(root.left, param1)) {
            return true;
        }
        if (method2(root.right, param1)) {
            return true;
        }
        return false;
    }
}
