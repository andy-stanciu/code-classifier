/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int[] var1 = new int[1];
        return has(root, param1, var1);
    }

    public boolean has(TreeNode root, int target, int[] var1) {
        if (root == null) {
            return false;
        }
        var1[0] += root.val;
        if (root.left == null && root.right == null && var1[0] == target) {
            return true;
        }
        boolean left = has(root.left, target, var1);
        boolean right = has(root.right, target, var1);
        var1[0] -= root.val;
        return left || right;
    }
}
