/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean helper(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        param1 = param1 + root.val;
        if (root.left == null && root.right == null)
            return param1 == param2;
        boolean l = helper(root.left, param1, param2);
        boolean r = helper(root.right, param1, param2);
        return l || r;
    }

    public boolean method1(TreeNode root, int param2) {
        return helper(root, 0, param2);
    }
}
