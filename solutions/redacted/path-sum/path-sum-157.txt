/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        List<Integer> var1 = new ArrayList();
        method2(root, param1, var1, 0);
        return var1.contains(param1);
    }

    private void method2(TreeNode node, int param1, List<Integer> var1, int param2) {
        param2 += node.val;
        if (node.left == null && node.right == null) {
            var1.add(param2);
            return;
        }
        if (node.left != null) {
            method2(node.left, param1, var1, param2);
        }
        if (node.right != null) {
            method2(node.right, param1, var1, param2);
        }
    }
}
