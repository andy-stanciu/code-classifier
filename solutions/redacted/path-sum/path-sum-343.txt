/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        return check(root, param1);
    }

    private boolean check(TreeNode node, int remain) {
        if (node == null) {
            return false;
        }
        if (node.left == null && node.right == null && remain - node.val == 0) {
            return true;
        }
        return check(node.left, remain - node.val) || check(node.right, remain - node.val);
    }
}
