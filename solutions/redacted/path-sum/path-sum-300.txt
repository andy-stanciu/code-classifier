class Solution {

    public boolean method1(TreeNode root, int param1) {
        return dfs(root, 0, param1);
    }

    public static boolean dfs(TreeNode root, int param2, int target) {
        if (root == null) {
            return false;
        }
        param2 += root.val;
        if (root.left == null && root.right == null)
            return param2 == target;
        return dfs(root.left, param2, target) || dfs(root.right, param2, target);
    }
}
