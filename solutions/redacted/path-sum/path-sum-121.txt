/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        boolean[] isThere = new boolean[1];
        preOrderBinaryTree(root, 0, isThere, param1);
        return isThere[0];
    }

    public static void preOrderBinaryTree(TreeNode root, int param2, boolean[] isThere, int param1) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if (param2 + root.val == param1) {
                isThere[0] = true;
            }
        }
        preOrderBinaryTree(root.left, param2 + root.val, isThere, param1);
        preOrderBinaryTree(root.right, param2 + root.val, isThere, param1);
    }
}
