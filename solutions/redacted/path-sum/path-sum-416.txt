class Solution {

    boolean result = false;

    public boolean method1(TreeNode root, int param1) {
        dfs(root, param1);
        return result;
    }

    private void dfs(TreeNode root, int param1) {
        if (root == null) {
            return;
        }
        param1 -= root.val;
        //leaf Node
        if (root.left == null && root.right == null) {
            if (param1 == 0) {
                result = true;
            }
            return;
        }
        //path not found yet
        if (result == false) {
            dfs(root.left, param1);
            dfs(root.right, param1);
        }
    }
}
