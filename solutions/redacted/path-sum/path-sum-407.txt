/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int target) {
        if (root == null) {
            return false;
        }
        return helper(root, 0, target);
    }

    public boolean helper(TreeNode root, int param1, int target) {
        if (root == null) {
            return false;
        }
        param1 = param1 + root.val;
        if (root.left == null && root.right == null) {
            return param1 == target;
        }
        boolean left = helper(root.left, param1, target);
        boolean right = helper(root.right, param1, target);
        if (left || right) {
            return true;
        }
        return false;
    }
}
