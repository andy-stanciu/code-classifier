/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        Boolean flag = false;
        if (root != null && param1 == root.val && (root.left == null && root.right == null))
            flag = true;
        if (root != null) {
            flag = flag || method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
        }
        return flag;
    }
}
