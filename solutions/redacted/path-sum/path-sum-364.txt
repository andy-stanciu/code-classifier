/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return dfs(root, param1, 0);
    }

    public boolean dfs(TreeNode root, int target, int param2) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null && param2 + root.val == target)
            return true;
        if (root.left == null && root.right == null)
            return false;
        return dfs(root.left, target, param2 + root.val) || dfs(root.right, target, param2 + root.val);
    }
}
