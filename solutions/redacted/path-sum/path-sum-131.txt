/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, param1, 0);
    }

    public boolean helper(TreeNode root, int target, int current) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && current + root.val == target) {
            return true;
        }
        return helper(root.left, target, root.val + current) || helper(root.right, target, root.val + current);
    }
}
