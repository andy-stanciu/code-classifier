/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        param1 = param1;
        if (root != null) {
            param1 -= root.val;
            boolean l = false;
            boolean r = false;
            if (param1 == 0 && root.left == null && root.right == null)
                return true;
            if (root.left != null) {
                l = method1(root.left, param1);
            }
            if (root.right != null) {
                r = method1(root.right, param1);
            }
            return l || r;
        }
        return false;
    }
}
