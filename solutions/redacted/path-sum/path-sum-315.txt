/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return function(root, param1, 0);
    }

    static boolean function(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            param2 = param2 + root.val;
            if (param2 == param1)
                return true;
            return false;
        }
        param2 = param2 + root.val;
        boolean f1 = function(root.left, param1, param2);
        boolean f2 = function(root.right, param1, param2);
        return f1 || f2;
    }
}
