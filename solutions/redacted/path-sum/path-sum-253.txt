/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        // Check if the binary tree is null
        if (root == null) {
            return false;
        }
        // Add the value
        var1 += root.val;
        // Check if the target is the same as the calculated one and if we are on a leaf in the subtree
        if (var1 == param1 && root.left == null && root.right == null) {
            return true;
        }
        // Substract the current value of the from the target
        int var2 = param1 - root.val;
        // Use recursion to traverse the other subtrees
        return method1(root.left, var2) || method1(root.right, var2);
    }
}
