/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        // If it\'s a leaf node, check if its value equals the remaining sum
        if (root.left == null && root.right == null) {
            return param1 == root.val;
        }
        // Recursively check left and right subtrees with reduced target sum
        return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
    }
}
