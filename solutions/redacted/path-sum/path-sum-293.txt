/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root != null && param1 - root.val == 0 && root.left == null && root.right == null) {
            return true;
        }
        if (root == null) {
            return false;
        }
        boolean l = false;
        boolean r = false;
        if (root.left != null) {
            l = method1(root.left, param1 - root.val);
        }
        if (root.right != null) {
            r = method1(root.right, param1 - root.val);
        }
        return l || r;
    }
}
