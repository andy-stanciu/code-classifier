/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        //edge case - empty tree
        if (root == null)
            return false;
        //if root
        if (root.left == null && root.right == null) {
            if (root.val == param1)
                return true;
            else
                return false;
        }
        //return hasPathSum for the left OR the right
        if (root.left == null)
            return method1(root.right, param1 - root.val);
        if (root.right == null)
            return method1(root.left, param1 - root.val);
        return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
    }
}
