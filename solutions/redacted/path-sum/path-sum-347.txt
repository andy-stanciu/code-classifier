/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return helper(root, param1, 0);
    }

    public boolean helper(TreeNode root, int target, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            param2 += root.val;
            return param2 == target;
        }
        param2 += root.val;
        return helper(root.left, target, param2) || helper(root.right, target, param2);
    }
}
