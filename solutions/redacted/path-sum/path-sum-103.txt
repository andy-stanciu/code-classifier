/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, 0, param1);
    }

    boolean method2(TreeNode root, int param2, int target) {
        if (root == null) {
            return false;
        }
        param2 += root.val;
        if (root.right == null && root.left == null) {
            return param2 == target;
        }
        boolean left = method2(root.left, param2, target);
        boolean right = method2(root.right, param2, target);
        return left || right;
    }
}
