/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean flag;

    public boolean method1(TreeNode root, int param1) {
        helper(root, 0, param1);
        return flag;
    }

    public void helper(TreeNode root, int curr, int param1) {
        if (root == null) {
            return;
        }
        curr += root.val;
        if (root.left == null && root.right == null && curr == param1) {
            flag = true;
        }
        helper(root.left, curr, param1);
        helper(root.right, curr, param1);
    }
}
