class Solution {

    public boolean method1(TreeNode root, int param1) {
        // Base case: if the root is null, there\'s no path
        if (root == null)
            return false;
        // If it\'s a leaf node, check if the target sum is achieved
        if (root.left == null && root.right == null) {
            return root.val == param1;
        }
        // Recursively check the left and right subtree
        int var1 = param1 - root.val;
        return method1(root.left, var1) || method1(root.right, var1);
    }
}
