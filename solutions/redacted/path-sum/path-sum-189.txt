/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int target) {
        if (root == null)
            return false;
        target -= root.val;
        if (target == 0 && (root.left == null && root.right == null))
            return true;
        if (target != 0 && root == null)
            return false;
        return method1(root.left, target) || method1(root.right, target);
    }
}
