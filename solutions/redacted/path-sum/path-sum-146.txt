/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, param1, 0);
    }

    private boolean method2(TreeNode root, int param1, int param2) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null)
            return (root.val + param2) == param1;
        param2 += root.val;
        if (method2(root.left, param1, param2))
            return true;
        return method2(root.right, param1, param2);
    }
}
