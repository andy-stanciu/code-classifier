/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean findAns(int param1, TreeNode t, TreeNode tHead, int target) {
        //Checking if the obtained path is correct or wrong by checking if it is leaf node or not
        if (param1 == target) {
            if (tHead.left == null && tHead.right == null) {
                return true;
            }
        }
        //if we have reached the leaf node and target sum is not found ...that means no target sum in that path
        if (t == null) {
            return false;
        }
        //Checking with the left path of the current node to find targetSum
        boolean flag1 = findAns(param1 + t.val, t.left, t, target);
        //Checking with the right path of the current node to find targetSum
        boolean flag2 = findAns(param1 + t.val, t.right, t, target);
        //If target sum is found either from left path or right path, it returns true
        return (flag1 || flag2);
    }

    public boolean method1(TreeNode root, int param2) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null && root.val == param2) {
            return true;
        }
        boolean flag1 = findAns(root.val, root.left, root, param2);
        boolean flag2 = findAns(root.val, root.right, root, param2);
        return (flag1 || flag2);
    }
}
