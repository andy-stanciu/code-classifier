class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method2(root, 0, param1);
    }

    public boolean method2(TreeNode root, int param2, int param1) {
        if (root == null) {
            return false;
        }
        param2 += root.val;
        if (root.left == null && root.right == null) {
            return param2 == param1;
        }
        return method2(root.left, param2, param1) || method2(root.right, param2, param1);
    }
}
