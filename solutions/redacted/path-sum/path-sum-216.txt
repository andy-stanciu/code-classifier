/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public boolean method1(TreeNode root, int param1) {
        //base case
        if (root == null) {
            return false;
        }
        var1 += root.val;
        if (root.left == null && root.right == null && var1 == param1) {
            return true;
        }
        if (method1(root.left, param1)) {
            return true;
        }
        if (method1(root.right, param1)) {
            return true;
        }
        var1 = var1 - root.val;
        return false;
    }
}
