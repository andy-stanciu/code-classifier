/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean find(TreeNode root, int param1, int param2) {
        if (root == null) {
            return false;
        }
        param2 += root.val;
        if (root.left == null && root.right == null && param1 == param2) {
            return true;
        }
        return find(root.left, param1, param2) || find(root.right, param1, param2);
    }

    public boolean method1(TreeNode root, int param1) {
        return find(root, param1, 0);
    }
}
