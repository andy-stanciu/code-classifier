/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return check(root, param1);
    }

    boolean check(TreeNode root, int param1) {
        if (root == null)
            return false;
        param1 -= root.val;
        if (root.left == null && root.right == null)
            return 0 == param1;
        boolean left = method1(root.left, param1);
        boolean right = method1(root.right, param1);
        return left || right;
    }
}
