/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        //Check if root is null then return false;
        if (root == null) {
            return false;
        }
        //If left and right both are null, means leaf node and along with that check whether sum is equal to targetSum
        if (root.left == null && root.right == null && param1 == root.val) {
            return true;
        }
        return method1(root.left, param1 - root.val) || method1(root.right, param1 - root.val);
    }
}
