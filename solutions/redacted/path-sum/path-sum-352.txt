/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(final TreeNode root, final int param1) {
        if (root == null) {
            return false;
        }
        final int var1 = param1 - root.val;
        final boolean isLeaf = root.left == null && root.right == null;
        if (isLeaf && var1 == 0) {
            return true;
        }
        final boolean left = method1(root.left, var1);
        final boolean right = method1(root.right, var1);
        return left || right;
        /* If one of these values goes to zero at the leaf then it has a path sum! */
    }
}
