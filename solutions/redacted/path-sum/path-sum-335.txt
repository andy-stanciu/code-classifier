/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = method1(root.left);
        int var2 = method1(root.right);
        return var1 + var2 + root.val;
    }

    public boolean method2(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        int var3 = method1(root);
        if (var3 == param1 && root.left == null && root.right == null) {
            return true;
        } else {
            return method2(root.left, param1 - root.val) || method2(root.right, param1 - root.val);
        }
    }
}
