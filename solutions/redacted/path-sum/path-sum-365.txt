/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean check(TreeNode root, int k, int curr) {
        if (root == null)
            return false;
        int var1 = root.val + curr;
        if (root.left == null && root.right == null) {
            if (var1 == k)
                return true;
            else
                return false;
        }
        return check(root.left, k, var1) || check(root.right, k, var1);
    }

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return check(root, param1, 0);
    }
}
