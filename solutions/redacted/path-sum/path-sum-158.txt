/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return method2(root, param1, var1);
    }

    public boolean method2(TreeNode root, int param1, int var1) {
        if (root == null) {
            return false;
        }
        if (root != null) {
            var1 += root.val;
            if (root.left == null && root.right == null && var1 == param1) {
                return true;
            }
        }
        return method2(root.left, param1, var1) || method2(root.right, param1, var1);
    }
}
