public class Solution {

    public bool method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        //minus value of current node
        param1 -= root.val;
        //check if now a leaf node
        if (root.left == null && root.right == null) {
            return param1 == 0;
        }
        return method1(root.left, param1) || method1(root.right, param1);
    }
}
