/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null && param1 != 0) {
            return false;
        }
        if (root == null) {
            return false;
        }
        if (root.val == param1 && root.left == null && root.right == null) {
            return true;
        }
        boolean left = method1(root.left, param1 - root.val);
        boolean right = method1(root.right, param1 - root.val);
        if (left || right) {
            return true;
        }
        return false;
    }
}
