class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return helper(root, param1, 0);
    }

    public boolean helper(TreeNode node, int target, int param2) {
        if (node == null && target == param2)
            return true;
        if (node == null)
            return false;
        // there is possibility node\'s left is empty
        if (node.left == null)
            return helper(node.right, target, param2 + node.val);
        // node\'s right is empty
        if (node.right == null)
            return helper(node.left, target, param2 + node.val);
        return helper(node.left, target, param2 + node.val) || helper(node.right, target, param2 + node.val);
    }
}
