/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        Stack<TreeNode> nodeStack = new Stack<>();
        Stack<Integer> var1 = new Stack<>();
        nodeStack.push(root);
        var1.push(root.val);
        while (!nodeStack.isEmpty()) {
            TreeNode currentNode = nodeStack.pop();
            int var2 = var1.pop();
            // Check if the current node is a leaf node
            if (currentNode.left == null && currentNode.right == null && var2 == param1) {
                return true;
            }
            // Add the right child and updated sum to the stack
            if (currentNode.right != null) {
                nodeStack.push(currentNode.right);
                var1.push(var2 + currentNode.right.val);
            }
            // Add the left child and updated sum to the stack
            if (currentNode.left != null) {
                nodeStack.push(currentNode.left);
                var1.push(var2 + currentNode.left.val);
            }
        }
        return false;
    }
}
