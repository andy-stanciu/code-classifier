/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        var1 += root.val;
        boolean left = method1(root.left, param1);
        boolean right = method1(root.right, param1);
        if (left == false && right == false) {
            if (var1 == param1 && root.left == null && root.right == null) {
                return true;
            }
            var1 -= root.val;
        } else if (left == false) {
            return right;
        } else if (right == false) {
            return left;
        }
        return false;
    }
}
