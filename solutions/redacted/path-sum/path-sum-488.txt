/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1, int param2) {
        if (root.left == null && root.right == null) {
            //currentSum += root.val;
            if (param1 == param2 + root.val)
                return true;
            else
                return false;
        }
        return (root.left != null && method1(root.left, param1, param2 + root.val)) || (root.right != null && method1(root.right, param1, param2 + root.val));
    }

    public boolean method2(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method1(root, param1, 0);
    }
}
