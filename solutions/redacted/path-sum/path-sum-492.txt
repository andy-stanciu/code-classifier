class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        Stack<Pair<TreeNode, Integer>> stack = new Stack<>();
        stack.push(new Pair<>(root, root.val));
        while (!stack.isEmpty()) {
            Pair<TreeNode, Integer> now = stack.pop();
            TreeNode nowNode = now.getKey();
            int var1 = now.getValue();
            if (nowNode.left == null && nowNode.right == null) {
                if (var1 == param1) {
                    return true;
                }
            }
            if (nowNode.left != null)
                stack.push(new Pair<>(nowNode.left, var1 + nowNode.left.val));
            if (nowNode.right != null)
                stack.push(new Pair<>(nowNode.right, var1 + nowNode.right.val));
        }
        return false;
    }
}
