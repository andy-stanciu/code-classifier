class Solution {

    boolean b = false;

    public boolean method1(TreeNode root, int param1) {
        method2(root, param1, 0);
        return b;
    }

    void method2(TreeNode root, int param1, int param2) {
        if (root == null || b)
            return;
        param2 += root.val;
        if (root.left == null && root.right == null && param2 == param1) {
            b = true;
            return;
            // System.out.println(sum);
        }
        method2(root.left, param1, param2);
        method2(root.right, param1, param2);
    }
}
