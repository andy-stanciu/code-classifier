class Solution {

    public boolean method1(TreeNode root, int target) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null && root.val == target)
            return true;
        if (method1(root.left, target - root.val) || method1(root.right, target - root.val))
            return true;
        else
            return false;
    }
}
