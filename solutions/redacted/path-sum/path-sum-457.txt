class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        return method1(root, param1, 0);
    }

    private boolean method1(TreeNode node, int param1, int param2) {
        if (node == null)
            return false;
        if (node.left == null && node.right == null)
            if (param1 == (param2 + node.val))
                return true;
        return method1(node.left, param1, param2 + node.val) || method1(node.right, param1, param2 + node.val);
    }
}
