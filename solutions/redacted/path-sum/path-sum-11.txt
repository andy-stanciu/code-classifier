/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, 0, param1);
    }

    public boolean method2(TreeNode root, int param2, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == null && root.right == null) {
            param2 = param2 + root.val;
            if (param2 == param1) {
                return true;
            }
        }
        return method2(root.left, param2 + root.val, param1) || method2(root.right, param2 += root.val, param1);
    }
}
