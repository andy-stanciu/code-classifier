/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null)
            return false;
        Stack<TreeNode> nodeSt = new Stack();
        Stack<Integer> var1 = new Stack();
        nodeSt.add(root);
        var1.add(param1 - root.val);
        while (!nodeSt.isEmpty()) {
            TreeNode currNode = nodeSt.pop();
            int var2 = var1.pop();
            if (currNode.left == null && currNode.right == null && var2 == 0) {
                return true;
            }
            if (currNode.left != null) {
                nodeSt.add(currNode.left);
                var1.add(var2 - currNode.left.val);
            }
            if (currNode.right != null) {
                nodeSt.add(currNode.right);
                var1.add(var2 - currNode.right.val);
            }
        }
        return false;
    }
}
