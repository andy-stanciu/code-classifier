/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean ans = false;

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        int var1 = 0;
        dfs(root, param1, var1);
        return ans;
    }

    public void dfs(TreeNode root, int param1, int var1) {
        if (root == null) {
            return;
        }
        var1 += root.val;
        if (root.left == null && root.right == null) {
            if (var1 == param1) {
                ans |= true;
            } else {
                ans |= false;
            }
        }
        dfs(root.left, param1, var1);
        dfs(root.right, param1, var1);
    }
}
