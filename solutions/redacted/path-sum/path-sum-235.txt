/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return method2(root, var1, param1);
    }

    public static boolean method2(TreeNode root, int var1, int target) {
        if (root == null)
            return false;
        var1 += root.val;
        if (root.left == null && root.right == null)
            return var1 == target;
        return method2(root.left, var1, target) || method2(root.right, var1, target);
    }
    //BY GANESH JAGGINENI
}
