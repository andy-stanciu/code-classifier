/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        // Check all paths with DFS
        // No bookkeeping is needed, just reduce the target by the current node as we go.
        // Leaf
        if (root.left == null && root.right == null) {
            return root.val == param1;
        }
        // Non-leaf
        boolean var1 = method1(root.left, param1 - root.val);
        boolean var2 = method1(root.right, param1 - root.val);
        return var1 || var2;
    }
}
