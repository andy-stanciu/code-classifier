/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return fun(root, param1);
    }

    private boolean fun(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        if (root.left == root.right) {
            //targetSum -= root.val;
            return param1 == root.val;
        }
        param1 -= root.val;
        boolean t1 = fun(root.left, param1);
        // targetSum += root.val;
        boolean t2 = fun(root.right, param1);
        /// targetSum += root.val;
        return t1 || t2;
    }
}
