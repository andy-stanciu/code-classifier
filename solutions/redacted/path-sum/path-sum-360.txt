/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        return method2(root, 0, param1);
    }

    private boolean method2(TreeNode node, int param2, int param1) {
        if (node == null) {
            return false;
        }
        // Add the current node\'s value to the running sum
        param2 += node.val;
        // Check if it\'s a leaf node and if the current sum equals the target sum
        if (node.left == null && node.right == null) {
            return param2 == param1;
        }
        // Recursively check the left and right subtrees
        return method2(node.left, param2, param1) || method2(node.right, param2, param1);
    }
}
