/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        if (root == null) {
            return false;
        }
        param1 = param1 - root.val;
        if (param1 == 0 && root.right == null && root.left == null) {
            return true;
        }
        if (root.right == null && root.left != null) {
            return method1(root.left, param1);
        }
        if (root.left == null && root.right != null) {
            return method1(root.right, param1);
        }
        if (root.right != null & root.left != null) {
            if (param1 - root.right.val == 0 || param1 - root.left.val == 0) {
                if (root.right.right == null && root.right.left == null && root.left.left == null && root.right.right == null)
                    return true;
            }
            return method1(root.right, param1) || method1(root.left, param1);
        }
        return false;
    }
}
