/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return check(root, param1, var1);
    }

    public boolean check(TreeNode root, int param1, int var1) {
        if (root == null)
            return false;
        if (param1 == var1 + root.val && root.left == null && root.right == null)
            return true;
        return check(root.left, param1, var1 + root.val) || check(root.right, param1, var1 + root.val);
    }
}
