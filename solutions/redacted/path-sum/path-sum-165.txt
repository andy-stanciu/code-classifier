/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean traversal(TreeNode root, int param1, int target) {
        if (root == null)
            return false;
        param1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == target)
                return true;
        }
        return traversal(root.left, param1, target) || traversal(root.right, param1, target);
    }

    public boolean method1(TreeNode root, int param2) {
        //int sum[]=new int [1];
        return traversal(root, 0, param2);
    }
}
