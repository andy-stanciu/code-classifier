class Solution {

    public void traverse(TreeNode node, int param1, int param2) {
        if (result)
            return;
        int var1 = param2 + node.val;
        if ((node.left == null && node.right == null) && var1 == param1)

            traverse(node.left, param1, var1);
        if (node.right != null)
            traverse(node.right, param1, var1);
    }
}
