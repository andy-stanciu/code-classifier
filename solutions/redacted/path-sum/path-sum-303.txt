/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    boolean var1;

    Stack<TreeNode> stack;

    Solution() {
        stack = new Stack<>();
        var1 = false;
    }

    void traversal(TreeNode root, int param1, int param2) {
        if (root == null)
            return;
        // Update the running sum by adding the current node\'s value
        param1 += root.val;
        // Check if it\'s a leaf node
        if (root.left == null && root.right == null) {
            if (param1 == param2) {
                var1 = true;
                // Early termination if the path sum is found
                return;
            }
        }
        // Continue to traverse the tree if not found
        if (!var1) {
            // Only continue if path sum hasn\'t been found yet
            traversal(root.left, param1, param2);
            traversal(root.right, param1, param2);
        }
    }

    public boolean method1(TreeNode root, int param2) {
        // Ensure hasSum is reset before each new call
        var1 = false;
        // Start traversal with initial sum of 0
        traversal(root, 0, param2);
        return var1;
    }
}
