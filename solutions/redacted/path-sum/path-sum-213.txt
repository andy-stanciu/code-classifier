/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private boolean helper(TreeNode root, int target, int param1) {
        if (root == null) {
            return false;
        }
        param1 += root.val;
        // check  if it is leaf node and if sum == targget then return true
        //else in other  if statement which is also for leaf node  but this
        //timme sum is not equal to target value so  will subtract its value
        //from sum
        if (root.left == null && root.right == null && param1 == target) {
            return true;
        }
        if (root.left == null && root.right == null && param1 != target) {
            param1 -= root.val;
        }
        return helper(root.left, target, param1) || helper(root.right, target, param1);
    }

    public boolean method1(TreeNode root, int param2) {
        return helper(root, param2, 0);
    }
}
