/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean result = false;

    public int method1(TreeNode root, int param1, int param2) {
        if (root == null) {
            return 0;
        }
        param1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == param2) {
                result = true;
                return param1;
            }
        }
        method1(root.left, param1, param2);
        method1(root.right, param1, param2);
        return param1;
    }

    public boolean method2(TreeNode root, int param2) {
        method1(root, 0, param2);
        return result;
    }
}
