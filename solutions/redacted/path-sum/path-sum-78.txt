/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        boolean check = helper(root, param1);
        return check;
    }

    public boolean helper(TreeNode root, int param1) {
        if (root == null)
            return false;
        if (param1 - root.val == 0 && root.left == null && root.right == null)
            return true;
        return helper(root.left, param1 - root.val) || helper(root.right, param1 - root.val);
    }
}
