/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        // base case if root is null
        if (root == null) {
            return false;
        }
        param1 = param1 - root.val;
        // check the condition
        if (param1 == 0 && root.left == null && root.right == null)
            return true;
        boolean checkleft = method1(root.left, param1);
        boolean checkright = method1(root.right, param1);
        return checkleft || checkright;
    }
}
