class Solution {

    private ListNode[] method1(ListNode head, int k) {
        ListNode curr = head;
        ListNode prev = null, next = null;
        for (int i = 0; i < k; i++) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return new ListNode[] { prev, curr };
    }

    public ListNode method2(ListNode head, int k) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode ptr = dummy;
        while (ptr != null) {
            ListNode tracker = ptr;
            for (int i = 0; i < k; i++) {
                if (tracker == null)
                    break;
                tracker = tracker.next;
            }
            if (tracker == null) {
                break;
            }
            ListNode[] var1 = method1(ptr.next, k);
            ListNode prev = var1[0];
            ListNode curr = var1[1];
            ListNode var2 = ptr.next;
            var2.next = curr;
            ptr.next = prev;
            ptr = var2;
        }
        return dummy.next;
    }
}
