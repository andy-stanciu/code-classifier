class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null || head.next == null || k == 1) {
            return head;
        }
        ListNode curr = head, prev = null, next = null;
        int count = 0;
        while (count < k && curr != null) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            count++;
        }
        head.next = method1(next, k);
        if (count < k) {
            return method1(prev, count);
        }
        return prev;
    }
}
