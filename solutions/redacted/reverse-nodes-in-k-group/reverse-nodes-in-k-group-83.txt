class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode cur = head;
        int i = 0;
        while (cur != null && i < k) {
            cur = cur.next;
            i++;
        }
        if (i < k)
            return head;
        i = 0;
        cur = head;
        ListNode firstNode = head, prev = null;
        while (i < k) {
            ListNode temp = cur.next;
            cur.next = prev;
            prev = cur;
            cur = temp;
            i++;
        }
        firstNode.next = method1(cur, k);
        return prev;
    }
}
