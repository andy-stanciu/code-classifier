class Solution {

    public ListNode method1(ListNode head, int k) {
        //check if we have k elements left
        ListNode temp = head;
        for (int i = 0; i < k; i++) {
            if (temp == null)
                return head;
            temp = temp.next;
        }
        ListNode prev = null;
        ListNode curr = head;
        ListNode next = head.next;
        for (int i = 0; i < k; i++) {
            curr.next = prev;
            prev = curr;
            curr = next;
            if (curr != null)
                next = curr.next;
        }
        head.next = method1(curr, k);
        return prev;
    }
}
