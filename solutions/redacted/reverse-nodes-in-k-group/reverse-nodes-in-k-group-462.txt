class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = new ListNode(0);
        temp.next = head;
        method2(temp, k);
        return temp.next;
    }

    public void method2(ListNode temp, int k) {
        ListNode sample = temp;
        ListNode p = temp.next;
        int x = k;
        while (p != null) {
            ListNode l1 = p;
            while (--x > 0 && l1.next != null) {
                l1 = l1.next;
            }
            if (l1 != null && x == 0) {
                ListNode l2 = l1.next;
                l1.next = null;
                method3(p);
                temp.next = l1;
                p.next = l2;
                temp = p;
                p = l2;
                x = k;
            } else if (x > 0) {
                return;
            }
        }
    }

    public ListNode method3(ListNode head) {
        ListNode newHead = null;
        while (head != null) {
            ListNode next = head.next;
            head.next = newHead;
            newHead = head;
            head = next;
        }
        return newHead;
    }
}
