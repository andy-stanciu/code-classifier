class Solution {

    public ListNode method1(ListNode head, int k) {
        if (head == null)
            return head;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy;
        ListNode curr = dummy.next;
        int num = 0;
        while (curr != null) {
            //Find number of nodes
            curr = curr.next;
            num++;
        }
        curr = dummy.next;
        //Start point
        ListNode kPrev = prev;
        for (int i = 0; i < num / k; i++) {
            // No of reversions
            for (int j = 0; j < k; j++) {
                //Reverse set of K nodes
                ListNode temp = curr.next;
                curr.next = prev;
                prev = curr;
                //prev=> reversed start node; curr=> start of next k-nodes - end of last iter
                curr = temp;
            }
            //Connect reversed nodes to previous start point
            kPrev.next = prev;
            for (//Skip to last of the reversed node
            //Skip to last of the reversed node
            //Skip to last of the reversed node
            int j = 0; j < k - 1; j++) prev = prev.next;
            //Connect last reversed nodes to next set of k-Nodes
            prev.next = curr;
            //Update the K-previous node - start point for next iter
            kPrev = prev;
        }
        return dummy.next;
    }
}
