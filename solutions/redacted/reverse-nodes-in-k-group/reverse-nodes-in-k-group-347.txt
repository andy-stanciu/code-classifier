class Solution {

    public ListNode method1(ListNode head, int k) {
        ListNode temp = head;
        int count = 0;
        while (temp != null) {
            count++;
            temp = temp.next;
        }
        return method2(head, count, k);
    }

    private ListNode method2(ListNode head, int count, int k) {
        if (head == null || head.next == null || count < k)
            return head;
        ListNode curr = head;
        ListNode prev = null;
        ListNode next = null;
        int i = k;
        while (i > 0 && curr != null) {
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
            i--;
        }
        ListNode dummy = head;
        ListNode head1 = method2(curr, count - k, k);
        dummy.next = head1;
        return prev;
    }
}
