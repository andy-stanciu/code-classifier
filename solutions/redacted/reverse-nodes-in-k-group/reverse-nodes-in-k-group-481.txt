/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode start, int k) {
        ListNode next = start;
        int flag = 1;
        boolean exist = true;
        ListNode temp = null;
        ListNode var1 = null;
        ListNode var2 = null;
        if (next == null) {
            return null;
        }
        do {
            ListNode prev = null;
            var1 = next;
            int i;
            for (i = 1; i <= k; i++) {
                temp = next.next;
                next.next = prev;
                prev = next;
                next = temp;
            }
            if (i == k + 1) {
                var1.next = next;
                if (flag == 1) {
                    start = prev;
                    flag++;
                } else {
                    var2.next = prev;
                }
                var2 = var1;
            }
            // check if next group of size k exist
            ListNode t = next;
            for (i = 1; i <= k; i++) {
                if (t == null) {
                    exist = false;
                    break;
                } else {
                    t = t.next;
                }
            }
        } while (exist);
        return start;
    }
}
