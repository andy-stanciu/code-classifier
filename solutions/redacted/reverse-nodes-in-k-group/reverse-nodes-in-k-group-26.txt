class Solution {

    private ListNode method1(ListNode param1, ListNode end) {
        ListNode prev = null, next;
        while (param1 != end) {
            next = param1.next;
            param1.next = prev;
            prev = param1;
            param1 = next;
        }
        return prev;
    }

    public ListNode method2(ListNode head, int k) {
        if (head == null || head.next == null || k == 1)
            return head;
        ListNode dummy = new ListNode(0, head);
        ListNode back = dummy, forward = head;
        while (back != null) {
            int var1 = 0;
            while (var1 < k && forward != null) {
                forward = forward.next;
                ++var1;
            }
            if (var1 != k)
                break;
            ListNode last = back.next;
            back.next = method1(back.next, forward);
            last.next = forward;
            back = last;
        }
        return dummy.next;
    }
}
