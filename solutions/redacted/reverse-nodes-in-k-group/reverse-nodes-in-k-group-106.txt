class Solution {

    public ListNode method1(ListNode head, int k) {
        int count = 0;
        ListNode curr = head;
        ListNode prev = null;
        ListNode tmp;
        while (curr != null && count < k) {
            count++;
            curr = curr.next;
        }
        if (count < k) {
            return head;
        }
        curr = head;
        count = 0;
        while (count < k) {
            tmp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = tmp;
            count++;
        }
        head.next = method1(curr, k);
        return prev;
    }
}
