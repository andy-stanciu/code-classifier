class Solution {

    public ListNode method1(ListNode head) {
        ListNode prev = null;
        ListNode current = head;
        while (current != null) {
            // Store next node
            ListNode next = current.next;
            // Reverse the link
            current.next = prev;
            // Move prev forward
            prev = current;
            // Move current forward
            current = next;
        }
        // New head of the reversed list
        return prev;
    }
}
