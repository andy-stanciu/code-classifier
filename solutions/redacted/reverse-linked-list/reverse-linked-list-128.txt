class Solution {

    public ListNode method1(ListNode head) {
        // base case
        if (head == null || head.next == null) {
            return head;
        }
        // reversed head
        ListNode end = method1(head.next);
        // changed the direction
        head.next.next = head;
        // reversed end points to null
        head.next = null;
        // return new head
        return end;
    }
}
