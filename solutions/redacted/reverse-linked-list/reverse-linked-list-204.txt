class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode pre = null;
        ListNode curr = head;
        ListNode nex = head;
        while (nex != null) {
            nex = curr.next;
            curr.next = pre;
            pre = curr;
            curr = nex;
        }
        return pre;
    }
}
