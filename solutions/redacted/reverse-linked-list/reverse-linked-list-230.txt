class Solution {

    public ListNode method1(ListNode head) {
        ListNode current = head;
        ListNode previous = null;
        ListNode next = head;
        while (current != null) {
            // storing the next element so that we don\'t loose that
            next = current.next;
            // actual reversing (reversing the link to previous)
            current.next = previous;
            previous = current;
            current = next;
        }
        head = previous;
        return head;
    }
}
