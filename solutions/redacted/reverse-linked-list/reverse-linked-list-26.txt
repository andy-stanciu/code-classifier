class Solution {

    public ListNode method1(ListNode head) {
        // arraylist to store linkedlist values
        List<Integer> var1 = new ArrayList();
        for (ListNode curr = head; curr != null; curr = curr.next) {
            var1.add(curr.val);
        }
        for (ListNode curr = head; curr != null; curr = curr.next) {
            // remove last item from list & asign to it linkedlist
            curr.val = var1.remove(var1.size() - 1);
        }
        return head;
    }
}
