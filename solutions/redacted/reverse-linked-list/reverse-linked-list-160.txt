class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode curr = head.next;
        ListNode prevHead = head;
        // Set tail\'s next as null
        head.next = null;
        while (curr != null) {
            ListNode temp = curr.next;
            curr.next = prevHead;
            prevHead = curr;
            curr = temp;
        }
        return prevHead;
    }
}
