class Solution {

    public ListNode method1(ListNode head) {
        return method2(head);
    }

    private ListNode method2(ListNode head) {
        ListNode curr = head, prev = null;
        while (curr != null) {
            ListNode nextNode = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nextNode;
        }
        return prev;
    }
}
