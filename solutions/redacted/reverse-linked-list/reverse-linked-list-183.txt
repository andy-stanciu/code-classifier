class Solution {

    public ListNode method1(ListNode head) {
        ListNode previous = null, next = null;
        return method2(previous, head, next);
    }

    public ListNode method2(ListNode previous, ListNode head, ListNode next) {
        if (head == null) {
            return previous;
        }
        next = head.next;
        head.next = previous;
        previous = head;
        head = next;
        return method2(previous, head, next);
    }
}
