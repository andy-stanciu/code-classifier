/**
 *  206. Reverse Linked List **
 */
class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = head.next;
        ListNode smallInput = method1(head.next);
        var1.next = head;
        head.next = null;
        return smallInput;
    }
}
