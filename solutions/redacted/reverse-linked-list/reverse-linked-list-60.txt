//Recursive DFS reversePreOrder
class Solution {

    public List<Integer> rightSideView(TreeNode root) {
        List<Integer> var1 = new ArrayList<Integer>();
        rightSideView(root, var1, 0);
        return var1;
    }

    private void rightSideView(TreeNode root, List<Integer> var1, int currentLevel) {
        if (root == null)
            return;
        if (currentLevel >= var1.size()) {
            var1.add(root.val);
        }
        rightSideView(root.right, var1, currentLevel + 1);
        rightSideView(root.left, var1, currentLevel + 1);
    }
}
