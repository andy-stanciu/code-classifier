class Solution {

    public ListNode method1(ListNode head) {
        // Make 1->2->3->4->5->null to null<-1<-2<-3<-4<-5
        ListNode prev = null;
        ListNode curr = head;
        while (curr != null) {
            ListNode temp = curr.next;
            curr.next = prev;
            prev = curr;
            curr = temp;
        }
        return prev;
    }
}
