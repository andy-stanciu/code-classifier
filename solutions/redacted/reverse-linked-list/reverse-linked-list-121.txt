class Solution {

    static ListNode newHead = new ListNode(-1);

    ListNode temp;

    private void method1(ListNode head) {
        if (head == null) {
            return;
        }
        method1(head.next);
        temp.next = head;
        temp = temp.next;
    }

    public ListNode method2(ListNode head) {
        temp = newHead;
        method1(head);
        temp.next = null;
        return newHead.next;
    }
}
