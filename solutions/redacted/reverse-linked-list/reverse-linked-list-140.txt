class Solution {

    public ListNode revL(ListNode pre, ListNode cur) {
        if (// Tail of LinkedList
        cur == null)
            //because pre on last node and cur on null
            return pre;
        //store the reference to next Node
        ListNode nxt = cur.next;
        //reversing Link
        cur.next = pre;
        //Trust Recursion and apply it on current and next Node
        return revL(cur, nxt);
    }

    public ListNode method1(ListNode head) {
        return revL(null, head);
    }
}
