class Solution {

    public ListNode method1(ListNode head) {
        return method2(head, new ListNode(0)).next;
    }

    public ListNode method2(ListNode head, ListNode dummy) {
        if (head == null)
            return dummy;
        ListNode previous = dummy.next;
        dummy.next = head;
        ListNode next = head.next;
        head.next = previous;
        return method2(next, dummy);
    }
}
