class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            // Return null if the list is empty
            return null;
        }
        ListNode dummy = null;
        ListNode node = head;
        while (node != null) {
            ListNode prev = new ListNode(node.val);
            prev.next = dummy;
            dummy = prev;
            node = node.next;
        }
        return dummy;
    }
}
