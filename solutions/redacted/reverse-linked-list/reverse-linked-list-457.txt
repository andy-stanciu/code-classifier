class Solution {

    public ListNode method1(ListNode head) {
        return helper(null, head);
    }

    ListNode helper(ListNode prevNode, ListNode currentNode) {
        if (currentNode == null) {
            return prevNode;
        }
        ListNode temp = currentNode.next;
        currentNode.next = prevNode;
        return helper(currentNode, temp);
    }
}
