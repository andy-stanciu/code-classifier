class Solution {

    public ListNode method1(ListNode head) {
        return method2(null, head);
    }

    static ListNode method2(ListNode prev, ListNode head) {
        if (head == null)
            return null;
        if (head.next == null) {
            head.next = prev;
            return head;
        }
        ListNode tempNext = head.next;
        head.next = prev;
        return method2(head, tempNext);
    }
}
