class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode fastIterNode = head.next;
        ListNode slowIterNode = head;
        ListNode tempNode;
        head.next = null;
        while (fastIterNode != null) {
            tempNode = fastIterNode.next;
            fastIterNode.next = slowIterNode;
            slowIterNode = fastIterNode;
            fastIterNode = tempNode;
        }
        return slowIterNode;
    }
}
