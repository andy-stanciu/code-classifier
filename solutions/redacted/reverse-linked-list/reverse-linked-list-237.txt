//Iterative
class Solution {

    public ListNode method1(ListNode head) {
        ListNode curr = head, prev = null;
        while (curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
}

//Recursive
class Solution {

    public ListNode method1(ListNode head) {
        return rev(head, null);
    }

    private ListNode rev(ListNode head, ListNode newHead) {
        if (head == null)
            return newHead;
        ListNode next = head.next;
        head.next = newHead;
        return rev(next, head);
    }
}
