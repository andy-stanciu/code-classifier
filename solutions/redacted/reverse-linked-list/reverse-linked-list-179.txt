class Solution {

    public ListNode method1(ListNode head) {
        //previous pointer
        ListNode prev = null;
        //current pointer
        ListNode cPtr = head;
        //head pointer
        ListNode nxt = null;
        return method2(cPtr, prev, nxt);
    }

    public ListNode method2(ListNode cPtr, ListNode prev, ListNode nxt) {
        if (cPtr == null) {
            //base condition
            return prev;
        }
        nxt = cPtr.next;
        cPtr.next = prev;
        prev = cPtr;
        cPtr = nxt;
        return method2(cPtr, prev, nxt);
    }
}
