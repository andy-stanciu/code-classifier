class Solution {

    public int method1(TreeNode root) {
        int[] arr = { 0 };
        method2(root, arr);
        return arr[0];
    }

    private void method2(TreeNode root, int[] arr) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null && root.left.right == null)
            arr[0] += root.left.val;
        method2(root.left, arr);
        method2(root.right, arr);
    }
}
