/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, TreeNode parent) {
        if (root == null)
            return 0;
        if (root.right == null && root.left == null && parent.right != root)
            return root.val;
        return method1(root.left, root) + method1(root.right, root);
    }

    public int method2(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        return method1(root.left, root) + method1(root.right, root);
    }
}
