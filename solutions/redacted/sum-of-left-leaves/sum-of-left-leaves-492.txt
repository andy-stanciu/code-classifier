/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1;

    public int method1(TreeNode root) {
        var1 = 0;
        return method2(root, false);
    }

    private int method2(TreeNode root, boolean param1) {
        if (root == null) {
            return var1;
        }
        if (param1 && root.param1 == null && root.right == null) {
            var1 += root.val;
        }
        method2(root.param1, true);
        method2(root.right, false);
        return var1;
    }
}
