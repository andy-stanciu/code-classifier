/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return solve(root, false);
    }

    private int solve(TreeNode root, boolean param1) {
        if (root.left == null && root.right == null) {
            return param1 ? root.val : 0;
        } else {
            int var1 = 0;
            if (root.left != null)
                var1 += solve(root.left, true);
            if (root.right != null)
                var1 += solve(root.right, false);
            return var1;
        }
    }
}
