/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int var1 = 0;

    public int method1(TreeNode root) {
        var1 = 0;
        method2(root, -1);
        return var1;
    }

    public static void method2(TreeNode root, int v) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && v == 0) {
            var1 = var1 + root.val;
            return;
        }
        method2(root.left, 0);
        method2(root.right, 1);
    }
}
