/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        preOrderBinaryTree(root, var1);
        return var1[0];
    }

    public static void preOrderBinaryTree(TreeNode root, int[] var1) {
        if (root == null) {
            return;
        }
        if (root.left != null && (root.left.left == null && root.left.right == null)) {
            var1[0] += root.left.val;
        }
        preOrderBinaryTree(root.left, var1);
        preOrderBinaryTree(root.right, var1);
    }
}
