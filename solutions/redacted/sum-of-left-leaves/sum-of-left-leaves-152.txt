/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        myDfsFunc(root);
        return var1;
    }

    public TreeNode myDfsFunc(TreeNode node) {
        if (node == null) {
            return null;
        }
        if (node.left != null) {
            if (node.left.left == null && node.left.right == null) {
                var1 += node.left.val;
            }
        }
        node.left = myDfsFunc(node.left);
        node.right = myDfsFunc(node.right);
        return null;
    }
}
