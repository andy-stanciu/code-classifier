class Solution {

    public int method1(TreeNode root) {
        return method2(root, 0);
    }

    int method2(TreeNode t, int d) {
        // if the node is null return 0
        if (t == null)
            return 0;
        // if this node is a leaf node (we check if both left and right nodes are null)
        // if this node is left node (d = -1 for left node, d stands for direction that we took from parent node)
        if (t.left == null && t.right == null && d == -1)
            return t.val;
        // we add the ans from left subtree and right subtree then return it
        return method2(t.left, -1) + method2(t.right, 1);
    }
}
