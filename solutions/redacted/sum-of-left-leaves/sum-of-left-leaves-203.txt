/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int result;

    public int method1(TreeNode root) {
        result = 0;
        if (root == null)
            return 0;
        traverse(root, false);
        return result;
    }

    void traverse(TreeNode node, boolean param1) {
        if (node == null)
            return;
        if (node.left == null && node.right == null && param1)
            result = result + node.val;
        traverse(node.left, true);
        traverse(node.right, false);
    }
}
