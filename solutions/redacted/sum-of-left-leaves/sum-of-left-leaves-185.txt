/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return method2(root, 0);
    }

    public int method2(TreeNode root, int s) {
        if (root == null)
            return 0;
        if (root.left == root.right)
            return root.val * s;
        int a = method2(root.left, 1);
        int b = method2(root.right, 0);
        return a + b;
    }
}
