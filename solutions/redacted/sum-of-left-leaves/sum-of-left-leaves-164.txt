/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    /*
    sumOfLeftLeaves Method:

Base Case: If the root is null, return 0.
Sum Initialization: Initialize sum to 0.
Check for Left Leaf: If the left child of the current node exists and is a leaf (both its left and right children are null), add its value to sum.
Recursive Sum Calculation: Recursively calculate the sum of left leaves in the left and right subtrees and add these to sum.
Return: Return the total sum.
     */
    public int method1(TreeNode root) {
        // Base case: if the tree is empty, return 0
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        // Check if the left child is a leaf node
        if (root.left != null && root.left.left == null && root.left.right == null) {
            // Add the value of the left leaf node
            var1 += root.left.val;
        }
        // Recursively calculate the sum of left leaves in the left and right subtrees
        var1 += method1(root.left);
        var1 += method1(root.right);
        return var1;
    }
}
