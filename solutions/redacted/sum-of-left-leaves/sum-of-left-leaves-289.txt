/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return method2(root.left, 0) + method2(root.right, 1);
    }

    public int method2(TreeNode root, int param1) {
        if (root == null) {
            return 0;
        }
        if (param1 == 0 && (root.left == null && root.right == null)) {
            return root.val;
        }
        return method2(root.left, 0) + method2(root.right, 1);
    }
}
