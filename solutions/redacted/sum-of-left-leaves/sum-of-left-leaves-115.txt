class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        method2(root);
        if (root.left == null && root.right == null)
            return 0;
        return var1;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            var1 += root.val;
            return;
        }
        method2(root.left);
        if (root.right == null)
            return;
        if (root.right.left != null || root.right.right != null)
            method2(root.right);
    }
}
