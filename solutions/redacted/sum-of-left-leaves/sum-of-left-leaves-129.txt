/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int method1(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        // Check if this node is a leaf node and is a left child
        if (node.left == null && node.right == null && param1) {
            return node.val;
        }
        int var1 = method1(node.left, true);
        int var2 = method1(node.right, false);
        return var1 + var2;
    }

    public int method2(TreeNode root) {
        return method1(root, false);
    }
}
