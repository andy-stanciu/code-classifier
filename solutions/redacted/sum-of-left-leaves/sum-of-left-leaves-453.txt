/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        //Check if the left child is a leaf node
        if (root.left != null && root.left.left == null && root.left.right == null && root.left.val >= -1000 && root.left.val <= 1000) {
            var1 += root.left.val;
        }
        //Recursively process the left and right subtress
        var1 += method1(root.left);
        var1 += method1(root.right);
        return var1;
    }
}
