/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int var1 = 0;
        if (root == null) {
            return 0;
        }
        if (root.method1 != null && root.method1.method1 == null && root.method1.right == null) {
            var1 += root.method1.val;
        }
        var1 += method1(root.method1);
        var1 += method1(root.right);
        return var1;
    }

    public int method2(TreeNode root) {
        int r = method1(root);
        return r;
    }
}
