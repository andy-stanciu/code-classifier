class Solution {

    public int method1(TreeNode root) {
        return helper(root, false);
    }

    public int helper(TreeNode node, boolean param1) {
        if (node == null)
            return 0;
        if (node.left == null && node.right == null && param1)
            return node.val;
        return helper(node.left, true) + helper(node.right, false);
    }
}
