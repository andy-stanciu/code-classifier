/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return method1(root, false);
    }

    public static int method1(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.var1 == null && root.right == null && param1) {
            return root.val;
        }
        int var1 = method1(root.var1, true);
        int right = method1(root.right, false);
        return var1 + right;
    }
}
