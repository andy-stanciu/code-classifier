/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public void helper(TreeNode node, boolean param1) {
        if (node == null) {
            return;
        }
        if (param1 == true && node.left == null && node.right == null) {
            var1 = var1 + node.val;
        }
        helper(node.left, true);
        helper(node.right, false);
    }

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        helper(root, false);
        return var1;
    }
}
