/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1;

    public void traverse(TreeNode root, int flag) {
        // base condition
        if (root == null)
            return;
        // flag validation for leaf node and node is leaf or not
        if (flag == 1 && root.left == null && root.right == null) {
            var1 += root.val;
            return;
        }
        // else call for left and right subtree
        traverse(root.left, 1);
        traverse(root.right, 0);
    }

    public int method1(TreeNode root) {
        // initialise it to zero for more testcases
        var1 = 0;
        traverse(root, 0);
        return var1;
    }
}
