/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int var1 = 0;

    public void solve(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        solve(root.left, true);
        if (param1 && root.left == null && root.right == null) {
            var1 += root.val;
        }
        solve(root.right, false);
    }

    public int method1(TreeNode root) {
        solve(root, false);
        return var1;
    }
}
