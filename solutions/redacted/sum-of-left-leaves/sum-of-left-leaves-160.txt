/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        return recursive(root, false);
    }

    int recursive(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        int var1 = 0;
        if (param1 && node.left == null && node.right == null) {
            // Add to sum only if it\'s a left leaf
            var1 += node.val;
            return var1;
        }
        // Recursively calculate sum of left leaves in left and right subtrees
        // Mark current node as left child
        var1 += recursive(node.left, true);
        // Mark current node as right child
        var1 += recursive(node.right, false);
        return var1;
    }
}
