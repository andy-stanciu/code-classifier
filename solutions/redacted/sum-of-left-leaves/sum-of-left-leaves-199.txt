/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return helper(root);
    }

    public int helper(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = 0;
        int right = 0;
        if (isLeaf(root.var1)) {
            return root.var1.val + helper(root.right);
        } else {
            System.out.println("Root: " + root.val);
            var1 = helper(root.var1);
            right = helper(root.right);
        }
        return var1 + right;
    }

    public boolean isLeaf(TreeNode root) {
        if (root == null)
            return false;
        if (root.var1 == null && root.right == null) {
            return true;
        }
        return false;
    }
}
