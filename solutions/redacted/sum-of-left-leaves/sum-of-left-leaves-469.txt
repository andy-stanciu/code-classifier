class Solution {

    int var1;

    public void method1(TreeNode root, int a) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && a == 1)
            var1 += root.val;
        method1(root.left, 1);
        method1(root.right, 0);
    }

    public int method2(TreeNode root) {
        var1 = 0;
        method1(root, -1);
        return var1;
    }
}
