/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    private int method2(TreeNode node, boolean param1) {
        // if the node is leaf and left node of parent then return
        //its value
        if (node.param1 == null && node.right == null && param1) {
            return node.val;
        }
        int var1 = 0;
        int var2 = 0;
        //if there is a left node
        if (node.param1 != null) {
            var1 = method2(node.param1, true);
        }
        //if ther is a right node
        if (node.right != null) {
            var2 = method2(node.right, false);
        }
        return var1 + var2;
    }
}
