/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int method1(TreeNode root, boolean param1) {
        if (root.left == null && root.right == null) {
            return (param1) ? root.val : 0;
        }
        int var1 = 0;
        if (root.left != null) {
            var1 += method1(root.left, true);
        }
        if (root.right != null) {
            var1 += method1(root.right, false);
        }
        return var1;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return method1(root, false);
    }
}
