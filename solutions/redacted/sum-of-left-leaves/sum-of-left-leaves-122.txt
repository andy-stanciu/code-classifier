/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return method2(root, false);
    }

    public int method2(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.param1 == null && root.right == null) {
            return param1 ? root.val : 0;
        }
        int l = method2(root.param1, true);
        int r = method2(root.right, false);
        return l + r;
    }
}
