public class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = 0;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            var1 += root.left.val;
        } else {
            var1 += method1(root.left);
        }
        var1 += method1(root.right);
        return var1;
    }
}
