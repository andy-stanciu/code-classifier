class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        // Reset sum to 0 for each new call
        var1 = 0;
        // Traverse left and right subtrees
        helper(root.left, true);
        helper(root.right, false);
        return var1;
    }

    private void helper(TreeNode root, boolean param1) {
        if (root == null) {
            return;
        }
        // Check if it\'s a left leaf node
        if (root.left == null && root.right == null && param1) {
            var1 += root.val;
        }
        // Recurse on the left and right children
        helper(root.left, true);
        helper(root.right, false);
    }
}
