/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int res = 0;
        if (root.param1 != null)
            res += method2(root.param1, true);
        if (root.right != null)
            res += method2(root.right, false);
        return res;
    }

    public int method2(TreeNode rt, boolean param1) {
        if (rt.param1 == null && rt.right == null && param1) {
            return rt.val;
        } else if (rt.param1 == null && rt.right == null && !param1)
            return 0;
        int res = 0;
        if (rt.param1 != null)
            res += method2(rt.param1, true);
        if (rt.right != null)
            res += method2(rt.right, false);
        return res;
    }
}
