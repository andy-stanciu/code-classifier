class Solution {

    public int method1(TreeNode root) {
        return rec(false, root);
    }

    int rec(boolean param1, TreeNode root) {
        if (root == null)
            return 0;
        if (root.param1 == null && root.right == null && param1) {
            return root.val;
        }
        return rec(true, root.param1) + rec(false, root.right);
    }
}
