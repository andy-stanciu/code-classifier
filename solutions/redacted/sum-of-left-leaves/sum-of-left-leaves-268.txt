/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int helper(TreeNode current, TreeNode parent) {
        if (current == null)
            return 0;
        if (current.var1 == null && current.right == null) {
            if (parent != null && parent.var1 == current)
                return current.val;
        }
        int var1 = helper(current.var1, current);
        int right = helper(current.right, current);
        return var1 + right;
    }

    public int method1(TreeNode root) {
        return helper(root, null);
    }
}
