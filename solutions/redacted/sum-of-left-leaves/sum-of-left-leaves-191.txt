/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        if (checkleaf(root))
            return 0;
        if (root.left != null) {
            if (checkleaf(root.left))
                var1 += root.left.val;
        }
        method1(root.left);
        method1(root.right);
        return var1;
    }

    public boolean checkleaf(TreeNode root) {
        if (root.left == null && root.right == null)
            return true;
        else
            return false;
    }
}
