class Solution {

    public int method1(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.var1 == null && root.right == null) {
            if (param1) {
                return root.val;
            } else {
                return 0;
            }
        }
        int var1 = method1(root.var1, true);
        int right = method1(root.right, false);
        return var1 + right;
    }

    public int method2(TreeNode root) {
        return method1(root, false);
    }
}
