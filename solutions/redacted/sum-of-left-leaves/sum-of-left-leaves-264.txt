/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return solve(root, 0);
    }

    int solve(TreeNode node, int i) {
        if (node == null)
            return 0;
        if (node.left == null && node.right == null) {
            if (i == 1)
                return node.val;
            return 0;
        }
        return solve(node.left, 1) + solve(node.right, 2);
    }
}
