class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        return method1(root.left, true) + method1(root.right, false);
    }

    private int method1(TreeNode node, boolean param1) {
        if (node == null)
            return 0;
        if (node.left == null && node.right == null && param1) {
            return node.val;
        }
        return method1(node.left, true) + method1(node.right, false);
    }
}
