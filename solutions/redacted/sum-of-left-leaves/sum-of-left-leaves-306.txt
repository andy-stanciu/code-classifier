/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int result = 0;

    public int method1(TreeNode root) {
        add(root.left, true);
        add(root.right, false);
        return result;
    }

    private void add(TreeNode node, boolean param1) {
        if (node == null)
            return;
        if (param1 && node.left == null && node.right == null) {
            result += node.val;
        }
        add(node.left, true);
        add(node.right, false);
    }
}
