/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return solveUsingDFS(root, false, false);
    }

    public int solveUsingDFS(TreeNode root, boolean param1, boolean rightNode) {
        if (root == null)
            return 0;
        if (param1 && root.left == null && root.right == null)
            return root.val;
        return solveUsingDFS(root.left, true, false) + solveUsingDFS(root.right, false, true);
    }
}
