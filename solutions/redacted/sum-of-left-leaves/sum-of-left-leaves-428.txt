class Solution {

    int var1 = 0;

    void check(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && param1) {
            var1 += root.val;
        }
        check(root.left, true);
        check(root.right, false);
    }

    public int method1(TreeNode root) {
        check(root, false);
        return var1;
    }
}
