/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int method1(TreeNode root) {
        if (root == null)
            return 0;
        else if (root.left == null && root.right == null)
            return root.val;
        return method1(root.left) + rightSide(root.right);
    }

    int rightSide(TreeNode root) {
        if (root == null || (root.left == null && root.right == null))
            return 0;
        return method1(root.left) + rightSide(root.right);
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        return method1(root.left) + rightSide(root.right);
    }
}
