class Solution {

    public int method1(TreeNode root) {
        return method2(root);
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = 0;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            var1 += root.left.val;
        }
        int var2 = method2(root.left);
        int var3 = method2(root.right);
        return var1 + var2 + var3;
    }
}
