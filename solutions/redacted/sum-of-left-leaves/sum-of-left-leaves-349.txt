/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return solve(root, null);
    }

    public int solve(TreeNode currNode, TreeNode parent) {
        if (currNode == null) {
            return 0;
        }
        if (currNode.var1 == null && currNode.right == null) {
            if (parent != null && currNode == parent.var1) {
                return currNode.val;
            }
        }
        int var1 = solve(currNode.var1, currNode);
        int right = solve(currNode.right, currNode);
        return var1 + right;
    }
}
