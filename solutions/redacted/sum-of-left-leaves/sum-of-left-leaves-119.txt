class Solution {

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return 0;
        }
        return method2(root, true);
    }

    public int method2(TreeNode root, boolean flag) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null && flag) {
            return root.val;
        }
        return method2(root.left, true) + method2(root.right, false);
    }
}
