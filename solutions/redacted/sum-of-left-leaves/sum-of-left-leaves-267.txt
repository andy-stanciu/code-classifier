/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static int method1(TreeNode root, int method1, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            if (param1) {
                return root.val;
            } else {
                return 0;
            }
        }
        int var1 = method1(root.left, method1, true);
        int var2 = method1(root.right, method1, false);
        return method1 + var1 + var2;
        // sum += sum(root.right, sum);
        // return sum;
    }

    public int method2(TreeNode root) {
        if (root.left == null && root.right == null) {
            return 0;
        }
        return method1(root.left, 0, true) + method1(root.right, 0, false);
    }
}
