/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        method2(root, root);
        return var1;
    }

    public void method2(TreeNode root, TreeNode prev) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && prev.left == root) {
            var1 += root.val;
            return;
        }
        method2(root.left, root);
        method2(root.right, root);
    }
}
