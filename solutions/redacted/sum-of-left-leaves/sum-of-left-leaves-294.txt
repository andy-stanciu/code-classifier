class Solution {

    public int method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int var1 = 0;
        while (!q.isEmpty()) {
            TreeNode t = q.poll();
            if (t.left != null) {
                if (t.left.left == null && t.left.right == null) {
                    var1 += t.left.val;
                }
                q.add(t.left);
            }
            if (t.right != null) {
                q.add(t.right);
            }
        }
        return var1;
    }
}
