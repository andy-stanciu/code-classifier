/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        boolean var1 = false;
        return rec(root, var1);
    }

    public int rec(TreeNode root, boolean var1) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && var1 == true)
            return root.val;
        int l = rec(root.left, true);
        int r = rec(root.right, false);
        return l + r;
    }
}
