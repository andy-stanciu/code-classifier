class Solution {

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null) {
            return 0;
        }
        return method2(root, true);
    }

    private int method2(TreeNode node, Boolean flag) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null && flag) {
            return node.val;
        }
        int l = method2(node.left, true);
        int r = method2(node.right, false);
        return l + r;
    }
}
