/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int var1 = 0;

    public int method1(TreeNode root) {
        dfs(root, false);
        return var1;
    }

    private void dfs(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && param1) {
            var1 += root.val;
        }
        dfs(root.left, true);
        dfs(root.right, false);
    }
}
