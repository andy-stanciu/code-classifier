/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // static int count=0;
    public int method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return 0;
        return method2(root, false);
    }

    public int method2(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            return param1 ? root.val : 0;
        }
        return method2(root.left, true) + method2(root.right, false);
    }
}
