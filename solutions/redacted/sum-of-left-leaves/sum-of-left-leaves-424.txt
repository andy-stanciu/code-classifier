/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
// time complexity : O(n) space complexity :O(height of tree)
class Solution {

    public int var1;

    public void helper(TreeNode root, boolean param1) {
        if (root == null)
            return;
        if (param1 && root.left == null && root.right == null) {
            var1 += root.val;
        }
        helper(root.left, true);
        helper(root.right, false);
    }

    public int method1(TreeNode root) {
        var1 = 0;
        helper(root, false);
        return var1;
    }
}
