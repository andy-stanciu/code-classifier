/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int method1(TreeNode root, boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.param1 == null && root.right == null && param1) {
            return root.val;
        }
        return method1(root.param1, true) + method1(root.right, false);
    }

    public int method1(TreeNode root) {
        return method1(root.param1, true) + method1(root.right, false);
    }
}
