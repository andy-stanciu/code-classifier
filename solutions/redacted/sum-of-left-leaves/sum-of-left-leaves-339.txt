class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        return method2(root.left) + method3(root.right);
    }

    private int method2(TreeNode root) {
        if (root == null)
            return 0;
        if (root.right == null && root.left == null)
            return root.val;
        //return value beacause its left leaf
        return method2(root.left) + method3(root.right);
    }

    private int method3(TreeNode root) {
        if (root == null)
            return 0;
        //Go ahead with the dfs instead of returning node val
        return method2(root.left) + method3(root.right);
    }
}
