class Solution {

    public int method1(TreeNode root) {
        return solve(root);
    }

    public int solve(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = 0, right = 0;
        if (root.var1 != null) {
            var1 = solve(root.var1);
            // if this left node is leaf then include into ans.
            if (isLeaf(root.var1)) {
                var1 += root.var1.val;
            }
        }
        if (root.right != null) {
            right = solve(root.right);
        }
        return var1 + right;
    }

    public boolean isLeaf(TreeNode root) {
        if (root.var1 == null && root.right == null)
            return true;
        return false;
    }
}
