/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    int var1 = 0;

    public void method1(TreeNode root, boolean param1) {
        if (root.left == null && root.right == null && param1)
            var1 += root.val;
        if (root.left != null)
            method1(root.left, true);
        if (root.right != null)
            method1(root.right, false);
    }

    public int method2(TreeNode root) {
        if (root == null)
            return 0;
        method1(root, false);
        return var1;
    }
}
