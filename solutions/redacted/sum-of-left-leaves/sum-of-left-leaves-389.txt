class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = 0;
        if (root.left != null && root.left.left == null && root.left.right == null) {
            // Add value if the left child is a leaf
            var1 += root.left.val;
        }
        // Recursively calculate sum for left and right subtrees
        return var1 + method1(root.left) + method1(root.right);
    }
}
