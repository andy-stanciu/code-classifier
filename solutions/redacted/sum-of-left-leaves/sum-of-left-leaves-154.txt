/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int var1 = method2(root, false);
        return var1;
    }

    public int method2(TreeNode root, Boolean param1) {
        if (root == null) {
            return 0;
        }
        if (root.var2 == null && root.right == null && param1) {
            return root.val;
        }
        int var2 = method2(root.var2, true);
        int right = method2(root.right, false);
        return var2 + right;
    }
}
