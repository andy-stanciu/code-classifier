/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        return ifLeafGetValueOtherwiseZero(root.left) + method1(root.left) + method1(root.right);
    }

    private int ifLeafGetValueOtherwiseZero(TreeNode root) {
        return root != null && root.left == null && root.right == null ? root.val : 0;
    }
}
