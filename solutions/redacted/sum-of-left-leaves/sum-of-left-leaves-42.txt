class Solution {

    // Function to calculate the sum of left leaves in a binary tree
    public int method1(TreeNode root, boolean param1) {
        // If the root is null, return 0
        if (root == null)
            return (0);
        // If the root is a leaf node, return its value only if it\'s a left leaf
        if (root.left == null && root.right == null)
            return ((param1) ? root.val : 0);
        // Recursively calculate the sum of left leaves in the left and right subtrees
        return (method1(root.left, true) + method1(root.right, false));
    }

    public int method2(TreeNode root) {
        return method1(root, false);
    }
}
