/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return method2(root, null);
    }

    private int method2(TreeNode node, TreeNode root) {
        if (node == null) {
            return 0;
        }
        ;
        if (node.left == null && node.right == null && root != null && root.left == node) {
            return node.val;
        }
        return method2(node.left, node) + method2(node.right, node);
    }
}
