/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, TreeNode pre) {
        if (root == null || pre == null)
            return 0;
        if (pre.var1 == root && (root.var1 == null && root.right == null))
            return root.val;
        int var1 = method1(root.var1, root);
        int right = method1(root.right, root);
        return var1 + right;
    }

    public int method2(TreeNode root) {
        int var2 = method1(root, root);
        return var2;
    }
}
