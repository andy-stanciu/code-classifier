/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] arr = new int[1];
        func(root, arr, false);
        return arr[0];
    }

    void func(TreeNode root, int[] arr, boolean side) {
        if (root == null) {
            return;
        } else {
            func(root.left, arr, true);
            func(root.right, arr, false);
        }
        if (side == true && root.left == null && root.right == null) {
            arr[0] += root.val;
        }
    }
}
