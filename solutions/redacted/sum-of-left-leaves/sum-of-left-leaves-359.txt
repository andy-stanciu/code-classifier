/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int[] param1) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null)
            return root.val;
        int var1 = method1(root.left, param1);
        param1[0] += var1;
        method1(root.right, param1);
        return 0;
    }

    public int method2(TreeNode root) {
        int[] param1 = new int[1];
        method1(root, param1);
        return param1[0];
    }
}
