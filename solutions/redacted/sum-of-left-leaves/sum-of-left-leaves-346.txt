/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int rec(TreeNode root, int n) {
        if (root.var1 == null && root.right == null && n == 0)
            return root.val;
        int var1 = 0, right = 0;
        if (root.var1 != null)
            var1 = rec(root.var1, 0);
        if (root.right != null)
            right = rec(root.right, 1);
        return var1 + right;
    }

    public int method1(TreeNode root) {
        return rec(root, -1);
    }
}
