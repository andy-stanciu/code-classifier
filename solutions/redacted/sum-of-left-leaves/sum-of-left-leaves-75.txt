/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int f(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (param1 && root.param1 == null && root.right == null)
            return root.val;
        return f(root.param1, true) + f(root.right, false);
    }

    public int method1(TreeNode root) {
        return f(root, false);
    }
}
