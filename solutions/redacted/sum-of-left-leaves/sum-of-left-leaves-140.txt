/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        ArrayList<Integer> l = new ArrayList<Integer>();
        int var1 = 0;
        trv(root, l);
        for (int i : l) {
            var1 = var1 + i;
        }
        return var1;
    }

    public static void trv(TreeNode root, ArrayList<Integer> l) {
        if (root == null)
            return;
        if (root.left != null && root.left.left == null & root.left.right == null)
            l.add(root.left.val);
        trv(root.left, l);
        trv(root.right, l);
    }
}
