/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] var1 = { 0 };
        helper(root, var1, false);
        return var1[0];
    }

    private void helper(TreeNode root, int[] var1, boolean isRight) {
        if (root == null) {
            return;
        }
        // Making sure that it is a leaf node
        // add sum only if it is not a part of right sub-tree
        if (root.left == null && root.right == null && isRight) {
            var1[0] = var1[0] + root.val;
        }
        helper(root.left, var1, true);
        helper(root.right, var1, false);
    }
}
