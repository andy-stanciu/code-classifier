/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = 0;

    public int method1(TreeNode root) {
        method2(root, false);
        return ans;
    }

    private void method2(TreeNode node, boolean param1) {
        if (node == null) {
            return;
        }
        method2(node.left, true);
        method2(node.right, false);
        if (param1 && node.left == null && node.right == null) {
            ans += node.val;
        }
    }
}
