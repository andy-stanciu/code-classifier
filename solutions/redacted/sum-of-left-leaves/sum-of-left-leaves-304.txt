/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return dfs(root, 0, false);
    }

    public static int dfs(TreeNode root, int result, boolean flag) {
        if (root == null)
            return 0;
        if (flag && root.left == null && root.right == null)
            return root.val;
        result += dfs(root.left, 0, true);
        result += dfs(root.right, 0, false);
        return result;
    }
}
