/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return dfs(root, false);
    }

    int dfs(TreeNode node, boolean param1) {
        if (node == null)
            return 0;
        if (node.left == null && node.right == null)
            return param1 ? node.val : 0;
        int var1 = dfs(node.left, true);
        var1 += dfs(node.right, false);
        return var1;
    }
}
