/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return method1(root, false);
    }

    private static int method1(TreeNode n, boolean param1) {
        if (n == null)
            return 0;
        if (n.left == null && n.right == null)
            return param1 ? n.val : 0;
        return method1(n.left, true) + method1(n.right, false);
    }
}
