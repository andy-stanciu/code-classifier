/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return helper(root, 0, false);
    }

    public int helper(TreeNode root, int ans, boolean param1) {
        if (root == null)
            return ans;
        if (root.left == null && root.right == null && param1) {
            return ans + root.val;
        }
        return helper(root.left, ans, true) + helper(root.right, ans, false);
    }
}
