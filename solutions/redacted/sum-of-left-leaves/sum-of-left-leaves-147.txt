/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        return helper(root, false);
    }

    public int helper(TreeNode root, boolean param1) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null && param1) {
            return root.val;
        }
        return helper(root.left, true) + helper(root.right, false);
    }
}
