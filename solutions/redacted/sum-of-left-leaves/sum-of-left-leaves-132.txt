/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int s = 0;

    public int method1(TreeNode a) {
        if (a == null)
            return 0;
        if (a.left != null && a.left.left == null && a.left.right == null)
            s += a.left.val;
        method1(a.left);
        method1(a.right);
        return s;
    }

    public int method2(TreeNode root) {
        s = 0;
        return method1(root);
    }
}
