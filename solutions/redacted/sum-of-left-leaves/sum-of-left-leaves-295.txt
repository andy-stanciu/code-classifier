/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        int x = 0;
        q.add(root);
        while (!q.isEmpty()) {
            TreeNode present = q.remove();
            if (present.left != null) {
                if (present.left.left == null && present.left.right == null) {
                    x = x + present.left.val;
                }
                q.add(present.left);
            }
            if (present.right != null) {
                q.add(present.right);
            }
        }
        return x;
    }
}
