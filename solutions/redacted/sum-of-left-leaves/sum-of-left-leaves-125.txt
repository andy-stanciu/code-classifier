/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return method2(root, root);
    }

    public int method2(TreeNode root, TreeNode prev) {
        if (root == null)
            return 0;
        if (root.left == null && root.right == null) {
            if (prev.left == root)
                return root.val;
        } else {
            return method2(root.left, root) + method2(root.right, root);
        }
        return 0;
    }
}
