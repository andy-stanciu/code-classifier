/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        return solve(root, false);
    }

    int solve(TreeNode cur, boolean param1) {
        if (cur == null)
            return 0;
        if (cur.left == null && cur.right == null && param1) {
            return cur.val;
        }
        return solve(cur.left, true) + solve(cur.right, false);
    }
}
