class Solution {

    public int method1(TreeNode root) {
        return dfs(root, false);
    }

    private int dfs(TreeNode node, boolean param1) {
        if (node == null) {
            return 0;
        }
        if (node.left == null && node.right == null) {
            // Leaf node
            return param1 ? node.val : 0;
        }
        // Traverse left child
        int var1 = dfs(node.left, true);
        // Traverse right child
        int var2 = dfs(node.right, false);
        return var1 + var2;
    }
}
