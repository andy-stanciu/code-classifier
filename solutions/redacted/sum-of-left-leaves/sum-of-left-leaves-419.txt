class Solution {

    int var1 = 0;

    public int method1(TreeNode root) {
        return method2(root, null);
    }

    public int method2(TreeNode root, TreeNode parent) {
        if (root == null)
            return 0;
        if (parent != null && parent.left == root && root.left == null && root.right == null) {
            return root.val;
        }
        return method2(root.left, root) + method2(root.right, root);
    }
}
