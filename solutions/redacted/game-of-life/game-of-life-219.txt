class Solution {

    public void method1(int[][] board) {
        //first making a clone of this board to make a correct judjment
        int[][] clone = new int[board.length][board[0].length];
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                clone[i][j] = board[i][j];
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (clone[i][j] == 1) {
                    int live = findLive(i, j, clone);
                    if (live == 2 || live == 3) {
                        board[i][j] = 1;
                    } else {
                        board[i][j] = 0;
                    }
                } else {
                    if (findLive(i, j, clone) == 3) {
                        board[i][j] = 1;
                    } else {
                        board[i][j] = 0;
                    }
                }
            }
        }
    }

    int findLive(int i, int j, int[][] clone) {
        int count = 0;
        if (j - 1 >= 0 && clone[i][j - 1] == 1)
            count++;
        // right
        if (j + 1 < clone[0].length && clone[i][j + 1] == 1)
            count++;
        // top
        if (i - 1 >= 0 && clone[i - 1][j] == 1)
            count++;
        // bottom
        if (i + 1 < clone.length && clone[i + 1][j] == 1)
            count++;
        // top left diagonal
        if (i - 1 >= 0 && j - 1 >= 0 && clone[i - 1][j - 1] == 1)
            count++;
        // bottom left diagonal
        if (i + 1 < clone.length && j - 1 >= 0 && clone[i + 1][j - 1] == 1)
            count++;
        // top right diagonal
        if (i - 1 >= 0 && j + 1 < clone[0].length && clone[i - 1][j + 1] == 1)
            count++;
        // bottom right diagonal
        if (i + 1 < clone.length && j + 1 < clone[0].length && clone[i + 1][j + 1] == 1)
            count++;
        return count;
    }
}
