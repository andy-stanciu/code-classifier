class Solution {

    public void method1(int[][] board) {
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                boolean isLive = Math.abs(board[i][j]) == 1;
                int liveNeigbors = 0;
                if (j - 1 >= 0) {
                    //check left
                    liveNeigbors = Math.abs(board[i][j - 1]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                    //top left
                    if (i - 1 >= 0) {
                        liveNeigbors = Math.abs(board[i - 1][j - 1]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                    }
                    //bottom left
                    if (i + 1 < board.length) {
                        liveNeigbors = Math.abs(board[i + 1][j - 1]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                    }
                }
                if (j + 1 < board[0].length) {
                    //check right
                    liveNeigbors = Math.abs(board[i][j + 1]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                    //top right
                    if (i - 1 >= 0) {
                        liveNeigbors = Math.abs(board[i - 1][j + 1]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                    }
                    //bottom right
                    if (i + 1 < board.length) {
                        liveNeigbors = Math.abs(board[i + 1][j + 1]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                    }
                }
                //top
                if (i - 1 >= 0) {
                    liveNeigbors = Math.abs(board[i - 1][j]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                }
                //bottom
                if (i + 1 < board.length) {
                    liveNeigbors = Math.abs(board[i + 1][j]) == 1 ? liveNeigbors + 1 : liveNeigbors;
                }
                if (isLive) {
                    if (liveNeigbors < 2) {
                        board[i][j] = -1;
                    }
                    if (liveNeigbors > 3) {
                        board[i][j] = -1;
                    }
                } else {
                    if (liveNeigbors == 3) {
                        board[i][j] = 2;
                    }
                }
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j] <= 0) {
                    board[i][j] = 0;
                } else {
                    board[i][j] = 1;
                }
            }
        }
    }
}
