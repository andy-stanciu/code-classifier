class Solution {

    public void method1(int[][] board) {
        int r = board.length, c = board[0].length;
        int[][] temp = new int[r][c];
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                int live = neighbor(board, i - 1, j - 1) + neighbor(board, i - 1, j) + neighbor(board, i - 1, j + 1) + neighbor(board, i, j - 1) + neighbor(board, i, j + 1) + neighbor(board, i + 1, j - 1) + neighbor(board, i + 1, j) + neighbor(board, i + 1, j + 1);
                if (board[i][j] == 1) {
                    temp[i][j] = (live < 2 || live > 3) ? 0 : 1;
                } else {
                    temp[i][j] = live == 3 ? 1 : 0;
                }
            }
        }
        for (int i = 0; i < r; i++) {
            board[i] = temp[i].clone();
        }
    }

    private int neighbor(int[][] board, int r, int c) {
        if (r < 0 || c < 0 || r >= board.length || c >= board[0].length || board[r][c] == 0)
            return 0;
        return 1;
    }
}
