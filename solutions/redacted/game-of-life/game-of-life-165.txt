class Solution {

    public void method1(int[][] board) {
        int[][] liveNums = new int[board.length][board[0].length];
        for (int i = 0; i < liveNums.length; i++) {
            Arrays.fill(liveNums[i], 0);
        }
        for (int i = 0; i < liveNums.length; i++) {
            for (int j = 0; j < liveNums[i].length; j++) {
                if (board[i][j] == 1) {
                    incrementLives(liveNums, i - 1, j - 1);
                    incrementLives(liveNums, i - 1, j);
                    incrementLives(liveNums, i - 1, j + 1);
                    incrementLives(liveNums, i, j - 1);
                    incrementLives(liveNums, i, j + 1);
                    incrementLives(liveNums, i + 1, j - 1);
                    incrementLives(liveNums, i + 1, j);
                    incrementLives(liveNums, i + 1, j + 1);
                }
            }
        }
        for (int i = 0; i < liveNums.length; i++) {
            for (int j = 0; j < liveNums[i].length; j++) {
                if (board[i][j] == 1) {
                    if (liveNums[i][j] < 2 || liveNums[i][j] > 3) {
                        board[i][j] = 0;
                    }
                } else {
                    if (liveNums[i][j] == 3) {
                        board[i][j] = 1;
                    }
                }
            }
        }
    }

    private void incrementLives(int[][] liveNums, int i, int j) {
        if (i >= 0 && j >= 0 && i < liveNums.length && j < liveNums[0].length) {
            liveNums[i][j] += 1;
        }
    }
}
