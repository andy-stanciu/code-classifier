// transition states is based on how we simulate this problem in our mind...
// since there are two possible states of any cell i,j in matrix we can have 2^2 = 4 state transitions
// dead -> alive
// dead -> dead
// alive -> alive
// alive -> dead
class Solution {

    // for counting alive bodies around current neighbour...
    public static int countAliveAround(int i, int j, int[][] board) {
        int count = 0;
        // top left
        if (i - 1 >= 0 && j - 1 >= 0) {
            count += board[i - 1][j - 1] % 2;
        }
        // top center
        if (j - 1 >= 0) {
            count += board[i][j - 1] % 2;
        }
        // top right
        if (i + 1 < board.length && j - 1 >= 0) {
            count += board[i + 1][j - 1] % 2;
        }
        // right
        if (i + 1 < board.length) {
            count += board[i + 1][j] % 2;
        }
        // bottom right
        if (i + 1 < board.length && j + 1 < board[0].length) {
            count += board[i + 1][j + 1] % 2;
        }
        // bottom center
        if (j + 1 < board[0].length) {
            count += board[i][j + 1] % 2;
        }
        // bottom left
        if (i - 1 >= 0 && j + 1 < board[0].length) {
            count += board[i - 1][j + 1] % 2;
        }
        // left
        if (i - 1 >= 0) {
            count += board[i - 1][j] % 2;
        }
        return count;
    }

    // for checking that whether cell can survive or not...
    public static boolean canSurvive(int i, int j, int[][] board) {
        int state = board[i][j];
        int count = countAliveAround(i, j, board);
        // System.out.println("For i,j : ("+i+","+j+"): "+count);
        if (state == 1) {
            if (count >= 2 && count <= 3) {
                return true;
            } else {
                return false;
            }
        } else {
            if (count == 3) {
                return true;
            } else {
                return false;
            }
        }
    }

    public void method1(int[][] board) {
        int cols = board.length;
        int rows = board[0].length;
        for (int i = 0; i < cols; i++) {
            for (int j = 0; j < rows; j++) {
                if (canSurvive(i, j, board)) {
                    if (board[i][j] == 1) {
                        board[i][j] = 3;
                    } else if (board[i][j] == 0) {
                        board[i][j] = 2;
                    }
                }
            }
        }
        // for updating values...
        for (int i = 0; i < cols; i++) {
            for (int j = 0; j < rows; j++) {
                board[i][j] /= 2;
            }
        }
    }
}
