class Solution {

    public void method1(int[][] board) {
        int[][] newBoard = new int[board.length][board[0].length];
        int sum = 0;
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                sum = 0;
                if (i > 0 && j > 0) {
                    if (board[i - 1][j - 1] == 1) {
                        sum++;
                    }
                }
                if (i > 0) {
                    if (board[i - 1][j] == 1) {
                        sum++;
                    }
                }
                if (i > 0 && j < board[0].length - 1) {
                    if (board[i - 1][j + 1] == 1) {
                        sum++;
                    }
                }
                if (j > 0) {
                    if (board[i][j - 1] == 1) {
                        sum++;
                    }
                }
                if (j < board[0].length - 1) {
                    if (board[i][j + 1] == 1) {
                        sum++;
                    }
                }
                if (i < board.length - 1 && j > 0) {
                    if (board[i + 1][j - 1] == 1) {
                        sum++;
                    }
                }
                if (i < board.length - 1) {
                    if (board[i + 1][j] == 1) {
                        sum++;
                    }
                }
                if (i < board.length - 1 && j < board[0].length - 1) {
                    if (board[i + 1][j + 1] == 1) {
                        sum++;
                    }
                }
                if (board[i][j] == 1) {
                    if (sum == 2 || sum == 3) {
                        newBoard[i][j] = 1;
                    } else {
                        newBoard[i][j] = 0;
                    }
                } else {
                    if (sum == 3) {
                        newBoard[i][j] = 1;
                    } else {
                        newBoard[i][j] = 0;
                    }
                }
            }
        }
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                board[i][j] = newBoard[i][j];
            }
        }
    }
}
