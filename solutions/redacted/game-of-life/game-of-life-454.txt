class Solution {

    public void method1(int[][] board) {
        int m = board.length;
        int n = board[0].length;
        int[][] ans = new int[m][n];
        for (int i = 0; i < m; i++) {
            int live_neighbours = 0;
            int dead_neighbours = 0;
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 1) {
                    //1. top
                    if (i > 0) {
                        if (board[i - 1][j] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //2. top_left
                    if (i > 0 && j > 0) {
                        if (board[i - 1][j - 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //3. top right
                    if (i > 0 && j + 1 < n) {
                        if (board[i - 1][j + 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //4. left
                    if (j - 1 >= 0) {
                        if (board[i][j - 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //5. right
                    if (j + 1 < n) {
                        if (board[i][j + 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //6. bottom
                    if (i + 1 < m) {
                        if (board[i + 1][j] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //7. bottom left
                    if (i + 1 < m && j - 1 >= 0) {
                        if (board[i + 1][j - 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //8. bottom right
                    if (i + 1 < m && j + 1 < n) {
                        if (board[i + 1][j + 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //if board[i][j] == 1 and live neighbours are fewer than 2
                    if (live_neighbours < 2)
                        ans[i][j] = 0;
                    //if board[i][j] == 1 and live neighbours are equal to 2 or 3
                    if (live_neighbours == 2 || live_neighbours == 3)
                        ans[i][j] = 1;
                    //if board[i][j] == 1 and live neighbours are greater than 3
                    if (live_neighbours > 3)
                        ans[i][j] = 0;
                } else {
                    //1. top
                    if (i > 0) {
                        if (board[i - 1][j] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //2. top_left
                    if (i > 0 && j > 0) {
                        if (board[i - 1][j - 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //3. top right
                    if (i > 0 && j + 1 < n) {
                        if (board[i - 1][j + 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //4. left
                    if (j - 1 >= 0) {
                        if (board[i][j - 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //5. right
                    if (j + 1 < n) {
                        if (board[i][j + 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //6. bottom
                    if (i + 1 < m) {
                        if (board[i + 1][j] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //7. bottom left
                    if (i + 1 < m && j - 1 >= 0) {
                        if (board[i + 1][j - 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //8. bottom right
                    if (i + 1 < m && j + 1 < n) {
                        if (board[i + 1][j + 1] == 1) {
                            live_neighbours++;
                        } else {
                            dead_neighbours++;
                        }
                    }
                    //if board[i][j] == 0 and live neighbours are equals to 3
                    if (live_neighbours == 3)
                        ans[i][j] = 1;
                }
                //else
                live_neighbours = 0;
                dead_neighbours = 0;
            }
            //inner loop
        }
        //outer loop
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                board[i][j] = ans[i][j];
            }
        }
        //If this helped you, then please don\'t forget to upvote :) Happy problem solving
    }
}
