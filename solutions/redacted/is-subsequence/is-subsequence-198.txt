class Solution {

    public boolean method1(String s, String t) {
        // Pointer for s
        int k = 0;
        for (int i = 0; i < t.length() && k < s.length(); i++) {
            if (t.charAt(i) == s.charAt(k)) {
                // Move k to the next character in s
                k++;
            }
        }
        // If we matched all characters in s, return true
        return k == s.length();
    }
}
