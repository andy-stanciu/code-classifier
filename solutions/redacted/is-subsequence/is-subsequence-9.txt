class Solution {

    public boolean method1(String s, String t) {
        if (s == null)
            return false;
        int n = t.length();
        int len = s.length();
        if (len == 0)
            return true;
        int count = 0;
        for (int i = 0; i < n; i++) {
            if (t.charAt(i) == s.charAt(count)) {
                count++;
                if (count == len) {
                    return true;
                }
            }
        }
        return false;
    }
}
