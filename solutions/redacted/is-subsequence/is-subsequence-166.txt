class Solution {

    public boolean method1(String s, String t) {
        int n = s.length(), m = t.length();
        if (n > m)
            return false;
        if (n == 0)
            return true;
        if (n == 0 && m == 0)
            return true;
        int j = 0;
        for (int i = 0; i < m; i++) {
            if (s.charAt(j) == t.charAt(i))
                j += 1;
            if (j == n)
                return true;
        }
        return false;
    }
}
