class Solution {

    public boolean method1(String s, String t) {
        int c = 0;
        int index = 0;
        int cur = -1;
        for (int i = 0; i < s.length(); i++) {
            index = cur;
            for (int j = index + 1; j < t.length(); j++) {
                if (s.charAt(i) == t.charAt(j)) {
                    c++;
                    cur = j;
                    break;
                }
            }
        }
        if (c == s.length()) {
            return true;
        }
        return false;
    }
}
