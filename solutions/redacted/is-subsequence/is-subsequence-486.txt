class Solution {

    int max = 0;

    int i = 0;

    int c = 0;

    public boolean method1(String s, String t) {
        String f = t;
        while (i < s.length()) {
            if ((f.indexOf(s.charAt(i))) == -1) {
                return false;
            } else {
                max = (f.indexOf(s.charAt(i)));
                i += 1;
                f = f.substring(max + 1);
            }
        }
        return true;
    }
}
