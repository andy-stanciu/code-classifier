class Solution {

    public boolean method1(String s, String t) {
        int i = 0, j = -1;
        int slen = s.length();
        while (i < slen) {
            j = t.indexOf(s.charAt(i++), j + 1);
            if (j == -1) {
                return false;
            }
        }
        return true;
    }
}
