class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        while (j < t.length() && i < s.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                j++;
                i++;
            } else {
                j++;
            }
        }
        if (i == s.length()) {
            return true;
        } else {
            return false;
        }
    }
}
