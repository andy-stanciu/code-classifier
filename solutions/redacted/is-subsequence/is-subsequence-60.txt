class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        int var1 = 0, it = 0;
        while (it < t.length()) {
            if (t.charAt(it) == s.charAt(var1)) {
                var1++;
                if (var1 == s.length()) {
                    return true;
                }
            }
            it++;
        }
        return false;
    }
}
