class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        if (t.length() == 0) {
            return false;
        }
        int i = 0;
        for (char c : t.toCharArray()) {
            if (i == s.length()) {
                return true;
            }
            if (c == s.charAt(i)) {
                i++;
            }
        }
        return i == s.length() ? true : false;
    }
}
