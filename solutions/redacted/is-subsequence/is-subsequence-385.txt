class Solution {

    // Two-pointer apporach
    public boolean method1(String s, String t) {
        int sLength = s.length();
        int tLength = t.length();
        if (sLength > tLength) {
            return false;
        }
        int s_index = 0;
        int t_index = 0;
        while (s_index < sLength && t_index < tLength) {
            if (s.charAt(s_index) == t.charAt(t_index)) {
                s_index++;
            }
            t_index++;
        }
        // In above while loop s_index becomes equal to length of s string before loop break.
        if (s_index == sLength) {
            return true;
        }
        return false;
    }
}
