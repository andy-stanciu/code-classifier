class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        int si = 0;
        for (int i = 0; i < t.length(); i++) {
            if (si < s.length() && t.charAt(i) == s.charAt(si)) {
                si++;
            }
        }
        if (si == s.length())
            return true;
        return false;
    }
}
