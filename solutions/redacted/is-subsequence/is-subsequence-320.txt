class Solution {

    public boolean method1(String s, String t) {
        //1. make each string into an char array
        char[] tChar = t.toCharArray();
        char[] sChar = s.toCharArray();
        //2. keep track of indexes for both arrays (i for t array and j for s array)
        int j = 0;
        //3. then, loop through t array to check if it\'s equal with s array\'s current char. If it is the same, iterate j to another element
        for (int i = 0; i < t.length(); i++) {
            if (j == sChar.length) {
                // stop when you already checked through the whole s array for subsequence; good when what t array is large
                return true;
            }
            if (sChar[j] == tChar[i]) {
                j += 1;
            }
        }
        //4. if j equals length of the s array or s length, it should return true
        return j == sChar.length;
    }
}
