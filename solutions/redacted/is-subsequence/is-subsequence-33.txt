class Solution {

    public boolean method1(String s, String t) {
        // i for long String
        int i = 0;
        // j for short String
        int j = 0;
        while (i < t.length() && j < s.length()) {
            // move short pointer
            if (t.charAt(i) == s.charAt(j))
                j++;
            // no matter above condition true or false move long pointer
            i++;
        }
        // returns true if it matches else false
        return j == s.length();
    }
}
