class Solution {

    public boolean method1(String shortStr, String longStr) {
        if (shortStr.equals(""))
            return true;
        char[] shortChar = shortStr.toCharArray();
        char[] longChar = longStr.toCharArray();
        int indexShort = 0;
        for (char val : longChar) {
            if (val == shortChar[indexShort])
                indexShort++;
            if (indexShort == shortChar.length)
                return true;
        }
        return false;
    }
}
