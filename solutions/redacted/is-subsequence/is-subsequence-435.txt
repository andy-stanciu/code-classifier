class Solution {

    public boolean method1(String s2, String s1) {
        int j = 0;
        int n = s1.length();
        int m = s2.length();
        for (int i = 0; i < n && j < m; i++) {
            if (s1.charAt(i) == s2.charAt(j)) {
                j++;
            }
        }
        return (j == m);
    }
}
