class Solution {

    public boolean method1(String word, String word1) {
        char[] a = word.toCharArray();
        char[] b = word1.toCharArray();
        int aPointer = 0;
        int bPointer = 0;
        while (aPointer < a.length && bPointer < b.length) {
            if (a[aPointer] == b[bPointer]) {
                aPointer++;
            }
            bPointer++;
        }
        return aPointer == a.length;
    }
}
