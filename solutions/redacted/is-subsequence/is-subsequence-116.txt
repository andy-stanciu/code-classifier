class Solution {

    public boolean method1(String s, String t) {
        int len1 = s.length(), len2 = t.length();
        return check(s, len1 - 1, t, len2 - 1);
    }

    private static boolean check(String s, int index1, String t, int index2) {
        if (index1 < 0)
            return true;
        if (index2 < 0)
            return false;
        if (s.charAt(index1) == t.charAt(index2)) {
            return check(s, index1 - 1, t, index2 - 1) || check(s, index1, t, index2 - 1);
        } else {
            return check(s, index1, t, index2 - 1);
        }
    }
}
// Recursive Code
// Time Complexity : O(2^(len1+len2))
// Space Compelxity : O(max(len1,len2))
