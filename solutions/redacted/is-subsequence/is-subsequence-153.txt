class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        for (int i = 0; i < t.length(); i++) {
            if (j == s.length()) {
                break;
            }
            char s_cur = s.charAt(j);
            if (t.charAt(i) == s_cur) {
                j++;
            }
        }
        return j == s.length();
    }
}
