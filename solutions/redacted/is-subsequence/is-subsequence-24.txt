class Solution {

    public boolean method1(String s, String t) {
        int slen = s.length();
        if (slen == 0)
            return true;
        int i, j;
        for (j = i = 0; j < slen && i < t.length(); i++) {
            if (t.charAt(i) == s.charAt(j)) {
                j++;
            }
        }
        return j == slen;
    }
}
