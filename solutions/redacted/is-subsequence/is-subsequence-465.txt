class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0)
            return true;
        for (int i = 0; i < t.length(); i++) {
            if (s.charAt(0) == t.charAt(i)) {
                return method1(s.substring(1, s.length()), t.substring(i + 1, t.length()));
            }
        }
        return false;
    }
}
