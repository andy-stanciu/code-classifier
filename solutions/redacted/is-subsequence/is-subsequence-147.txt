class Solution {

    public boolean method1(String s, String t) {
        int i = 0, n = s.length();
        if (n == 0)
            return true;
        for (char c : t.toCharArray()) {
            if (s.charAt(i) == c)
                i++;
            if (i == n)
                return true;
        }
        return false;
    }
}
