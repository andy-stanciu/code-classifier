class Solution {

    public boolean method1(String s, String t) {
        if (s.isEmpty())
            return true;
        int ps = 0;
        char[] schar = s.toCharArray();
        char[] tchar = t.toCharArray();
        for (int i = 0; i < tchar.length; i++) {
            if (schar[ps] == tchar[i]) {
                ps++;
            }
            if (ps == schar.length) {
                return true;
            }
        }
        return false;
    }
}
