class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0) {
            // An empty string is always a subsequence.
            return true;
        }
        if (t.length() < s.length()) {
            // If `t` is shorter than `s`, `s` cannot be a subsequence.
            return false;
        }
        int cntr = 0;
        for (int i = 0; i < t.length(); i++) {
            if (s.charAt(cntr) == t.charAt(i)) {
                cntr++;
            }
            if (cntr == s.length())
                return true;
        }
        return false;
    }
}
