class Solution {

    public boolean method1(String s, String t) {
        int count = 0;
        char[] arr1 = s.toCharArray();
        char[] arr2 = t.toCharArray();
        int i = 0;
        int k = 0;
        while (i < arr2.length && arr1.length != 0 && k < arr1.length) {
            if (arr1[k] == arr2[i]) {
                count++;
                k++;
            }
            i++;
        }
        if (count == arr1.length) {
            return true;
        } else {
            return false;
        }
    }
}
