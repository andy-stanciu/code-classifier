class Solution {

    public boolean method1(String s, String t) {
        int count = 0;
        int l1 = 0;
        int l2 = 0;
        while (count <= s.length() && l1 < s.length() && l2 < t.length()) {
            if (s.charAt(l1) == t.charAt(l2)) {
                l1++;
                l2++;
                count++;
            } else {
                l2++;
            }
        }
        if (count == s.length()) {
            return true;
        } else {
            return false;
        }
    }
}
