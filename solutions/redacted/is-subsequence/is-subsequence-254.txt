class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        char[] s2 = s.toCharArray();
        char[] t2 = t.toCharArray();
        if (s2.length == 0)
            return true;
        for (int i = 0; i < t2.length; i++) {
            if (s2[j] == t2[i]) {
                if (j == s2.length - 1) {
                    return true;
                }
                j++;
            }
        }
        return false;
    }
}
