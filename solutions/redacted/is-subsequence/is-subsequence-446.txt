class Solution {

    public boolean method1(String s, String t) {
        //check if the string is empty or null, then return true
        if (s.isEmpty())
            return true;
        int i = 0;
        //we will iterate the string and with the increasing order we will check for char of subsequence
        for (int j = 0; j < t.length(); j++) {
            if (i < s.length() && t.charAt(j) == s.charAt(i)) {
                //if the char is found in main string then we start comparinf the next cahr of subsequence from the next index of main as we what in order
                i++;
            }
        }
        //at the end if the count is equal to length of subsequence we can say the subsequence is present in main string and in increaseing order
        if (i == s.length()) {
            return true;
        }
        return false;
    }
}
