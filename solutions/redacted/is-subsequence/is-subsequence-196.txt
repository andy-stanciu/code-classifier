class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        int var1 = s.length(), jsize = t.length();
        while (i < var1 && j < jsize) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        return i == var1;
    }
}
