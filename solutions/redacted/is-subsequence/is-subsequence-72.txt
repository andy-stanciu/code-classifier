class Solution {

    public boolean method1(String s, String t) {
        int pointerS = 0, pointerT = 0;
        while (pointerS < s.length() && pointerT < t.length()) {
            char charS = s.charAt(pointerS);
            char charT = t.charAt(pointerT);
            if (charS == charT)
                pointerS++;
            pointerT++;
        }
        return pointerS == s.length();
    }
}
