class Solution {

    public boolean method1(String s, String t) {
        int n = s.length();
        int m = t.length();
        int c = 0;
        int j = 0;
        for (int i = 0; i < m && j < n; i++) {
            if (t.charAt(i) == s.charAt(j)) {
                c++;
                j++;
            }
        }
        if (c == n) {
            return true;
        } else {
            return false;
        }
    }
}
