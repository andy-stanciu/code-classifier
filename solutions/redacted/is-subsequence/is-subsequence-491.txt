class Solution {

    public boolean method1(String s, String t) {
        int m = s.length();
        if (m == 0)
            return true;
        int n = t.length();
        int j = 0;
        for (int i = 0; i < n; i++) {
            if (s.charAt(j) == t.charAt(i)) {
                j++;
                if (j == m)
                    return true;
            }
        }
        return j == m;
    }
}
