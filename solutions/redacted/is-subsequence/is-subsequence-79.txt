class Solution {

    public boolean method1(String s, String t) {
        // i is for long string
        int i = 0;
        // j is for short string
        int j = 0;
        while (i < t.length() && j < s.length()) {
            if (t.charAt(i) == s.charAt(j))
                j++;
            i++;
        }
        // if j reaches end returns true else false
        return j == s.length();
    }
}
