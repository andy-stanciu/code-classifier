class Solution {

    public boolean method1(String s, String t) {
        for (int i = 0; i < s.length(); i++) {
            if (t.indexOf(s.charAt(i)) >= 0) {
                t = t.substring(t.indexOf(s.charAt(i)) + 1, t.length());
            } else
                return false;
        }
        return true;
    }
}
