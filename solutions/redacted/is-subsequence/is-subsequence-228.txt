class Solution {

    public boolean method1(String s, String t) {
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        int i = 0;
        int j = 0;
        int count = 0;
        while (i < a.length && j < b.length) {
            if (a[i] == b[j]) {
                i++;
                j++;
                count++;
            } else {
                j++;
            }
        }
        if (count == a.length) {
            return true;
        } else
            return false;
    }
}
