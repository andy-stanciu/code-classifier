class Solution {

    public boolean method1(String s, String t) {
        int comparePtr = 0;
        int traversalPtr = 0;
        int tLength = t.length();
        int sLength = s.length();
        if (tLength == 0 && sLength == 0)
            return true;
        while (traversalPtr < tLength && comparePtr < sLength) {
            if (s.charAt(comparePtr) == t.charAt(traversalPtr)) {
                comparePtr++;
            }
            traversalPtr++;
        }
        return comparePtr == sLength;
    }
}
