class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            boolean found = false;
            while (j < t.length()) {
                char ch2 = t.charAt(j);
                if (ch == ch2) {
                    found = true;
                    // Subsequent / left over string search or check.
                    j++;
                    break;
                }
                // If not found move to next character for a particular char in i loop.
                j++;
            }
            if (!found) {
                return false;
            }
        }
        return true;
    }
}
