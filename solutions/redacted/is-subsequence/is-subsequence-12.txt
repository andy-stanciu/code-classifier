class Solution {

    public boolean method1(String s, String t) {
        char[] charS = s.toCharArray();
        char[] charT = t.toCharArray();
        int i = 0, j = 0;
        // loop s and for every character in s, find corresponding in t in sequence
        while (i < charS.length) {
            while (j < charT.length) {
                // if found a match, increment i and j and break inner loop
                if (charS[i] == charT[j]) {
                    i++;
                    j++;
                    break;
                }
                // if not, increment j to find what we are looking for
                j++;
            }
            // if we reach end of string t before we reach end of string s, we failed
            if (j == charT.length && i != charS.length) {
                return false;
            }
        }
        return true;
    }
}
