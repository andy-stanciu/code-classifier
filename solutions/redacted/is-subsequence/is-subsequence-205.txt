class Solution {

    public boolean method1(String s, String t) {
        int subStringLen = 0;
        int len = t.length();
        int i = 0;
        while (i < len && subStringLen < s.length()) {
            if (t.charAt(i) == s.charAt(subStringLen)) {
                subStringLen++;
                i++;
            } else
                i++;
        }
        return (subStringLen == s.length());
    }
}
