class Solution {

    public boolean method1(String s, String t) {
        int sPointer = 0, tPointer = -1;
        while (++tPointer < t.length() && sPointer < s.length()) {
            if (t.charAt(tPointer) == s.charAt(sPointer)) {
                sPointer++;
            }
        }
        return sPointer == s.length();
    }
}
