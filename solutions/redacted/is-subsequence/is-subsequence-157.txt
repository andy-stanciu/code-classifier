class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        if (s.length() == 0) {
            // every empty string is a subsequence of ANY string ( even empty string)
            return true;
        }
        while (j < t.length()) {
            if (t.charAt(j) == s.charAt(i)) {
                i++;
            }
            j++;
            if (i == s.length()) {
                return true;
            }
        }
        if (i == s.length()) {
            // if all the elements are present in the \'t\' , then i will be just greater than last index of \'s\'
            return true;
        } else {
            return false;
        }
    }
}
