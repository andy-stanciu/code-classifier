class Solution {

    public boolean method1(String s, String t) {
        int prevIndex = -1;
        for (char ch : s.toCharArray()) {
            int indexOfCh = t.indexOf(ch, prevIndex + 1);
            if (indexOfCh <= prevIndex) {
                return false;
            }
            prevIndex = indexOfCh;
        }
        return true;
    }
}
