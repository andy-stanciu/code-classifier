class Solution {

    public boolean method1(String s, String t) {
        int x = 0;
        for (int i = 0; i < s.length(); i++) {
            if (!t.substring(x).contains("" + s.charAt(i))) {
                return false;
            } else {
                x = t.indexOf("" + s.charAt(i), x) + 1;
            }
        }
        return true;
    }
}
