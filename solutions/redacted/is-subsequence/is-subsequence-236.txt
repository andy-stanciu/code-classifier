class Solution {

    public boolean method1(String s, String t) {
        // save the size of the arrays
        int dimS = s.length();
        int dimT = t.length();
        // convert the Strings to char arrays (so i can access them faster)
        char[] sChars = s.toCharArray();
        char[] tChars = t.toCharArray();
        // declare array idexes
        int var1 = 0;
        int iT = 0;
        while (iT < dimT && var1 < dimS) {
            //if i find a match ,increment in one the S index
            var1 += (tChars[iT] == sChars[var1]) ? 1 : 0;
            //traverse the T array until i end it or i\'ve found all S elements
            iT++;
        }
        // if I reach the end of the S string, it means i found all the characters in order
        return (var1 == dimS);
    }
}
