class Solution {

    public boolean method1(String s, String t) {
        // Get the lengths of the strings s and t
        int slen = s.length(), tlen = t.length();
        // Initialize pointers for s (i) and t (j)
        int i = 0, j = 0;
        // Loop through t using pointer j until the end of either s or t
        while (i < slen && j < tlen) {
            // If the current characters of s and t match, move the pointer i for s
            if (s.charAt(i) == t.charAt(j)) {
                i++;
            }
            // Always move the pointer j for t
            j++;
        }
        // If i has reached the end of s, then s is a subsequence of t
        return i == slen;
    }
}
