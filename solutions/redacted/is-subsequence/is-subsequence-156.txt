class Solution {

    public boolean method1(String s, String t) {
        int count = 0;
        int j = 0;
        if (s.length() == 0) {
            return (true);
        }
        for (int i = 0; i < t.length(); i++) {
            if ((j < s.length()) && (s.charAt(j) == t.charAt(i))) {
                count++;
                j++;
                if (count == s.length()) {
                    return (true);
                }
            }
        }
        return (false);
    }
}
