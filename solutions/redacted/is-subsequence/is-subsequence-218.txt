class Solution {

    public boolean method1(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        int prev = 0;
        int x;
        for (char curr : s.toCharArray()) {
            if ((x = t.indexOf(curr, prev)) != -1) {
                prev = x + 1;
            } else {
                return false;
            }
        }
        return true;
    }
}
