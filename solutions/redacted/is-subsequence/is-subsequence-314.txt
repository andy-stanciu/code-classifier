class Solution {

    public boolean method1(String s, String t) {
        int j = 0;
        if (t.length() < s.length())
            return false;
        for (int i = 0; i < t.length() && j < s.length(); i++) {
            if (t.charAt(i) == s.charAt(j)) {
                j++;
            }
        }
        return j == s.length();
    }
}
