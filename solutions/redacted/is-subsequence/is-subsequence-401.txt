class Solution {

    public boolean method1(String param1, String targetSequence) {
        char[] var1 = param1.toCharArray();
        char[] targetSequenceArr = targetSequence.toCharArray();
        int j = 0;
        for (int i = 0; i < targetSequenceArr.length; i++) {
            if (j == var1.length)
                break;
            if (targetSequenceArr[i] == var1[j]) {
                j++;
            }
        }
        return j == var1.length;
    }
}
