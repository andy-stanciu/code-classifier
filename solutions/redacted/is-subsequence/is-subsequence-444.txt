class Solution {

    public boolean method1(String s, String t) {
        char[] s1 = s.toCharArray();
        char[] s2 = t.toCharArray();
        int p1 = 0, p2 = 0;
        while (p1 < s.length() && p2 < t.length()) {
            if (s1[p1] == s2[p2]) {
                p1++;
                if (p1 == s.length()) {
                    return true;
                }
            }
            p2++;
        }
        return p1 == s.length();
    }
}
