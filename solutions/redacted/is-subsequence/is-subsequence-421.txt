class Solution {

    public boolean method1(String s, String t) {
        int i, j = 0;
        if (s.equals(""))
            return true;
        for (i = 0; i < t.length(); i++) {
            if (s.charAt(j) == t.charAt(i))
                j++;
            if (j >= s.length())
                break;
        }
        if (j >= s.length())
            return true;
        return false;
    }
}
