class Solution {

    public boolean method1(String s, String t) {
        int sIndex = 0;
        int tIndex = 0;
        while (tIndex < t.length()) {
            if (sIndex == s.length()) {
                return true;
            }
            if (t.charAt(tIndex) == s.charAt(sIndex)) {
                sIndex++;
            }
            tIndex++;
        }
        return (sIndex == s.length());
    }
}
