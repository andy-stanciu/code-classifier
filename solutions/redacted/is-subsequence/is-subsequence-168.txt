class Solution {

    public boolean method1(String s, String t) {
        int count = 0, sizeS = s.length(), sizeT = t.length();
        char cs, ct;
        if (sizeS == 0)
            return true;
        if (s.equals(t))
            return true;
        for (int i = 0; i < sizeT; i++) {
            cs = s.charAt(count);
            ct = t.charAt(i);
            if (cs == ct) {
                count++;
                if (count == sizeS)
                    return true;
            }
        }
        return count == sizeS;
    }
}
