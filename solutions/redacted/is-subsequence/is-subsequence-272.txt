class Solution {

    public boolean method1(String s, String t) {
        if (s.length() > t.length()) {
            return false;
        }
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        int i = 0, j = 0;
        while (i < a.length && j < b.length) {
            if (a[i] == b[j]) {
                i++;
            }
            j++;
        }
        return s.length() == i;
    }
}
