class Solution {

    public boolean method1(String s, String t) {
        if (s.length() > t.length())
            return false;
        int var1 = 0;
        for (int i = 0; i < t.length(); i++) {
            if (s.length() == var1)
                return true;
            if (s.charAt(var1) == (t.charAt(i)))
                var1++;
        }
        return var1 == s.length();
    }
}
