class Solution {

    public boolean method1(String s, String t) {
        if (s == null || s == "" || s.length() > t.length())
            return false;
        int indx = 0;
        for (int i = 0; i < s.length(); ) {
            indx = t.indexOf(s.charAt(i), indx);
            if (indx == -1)
                return false;
            i++;
            indx++;
        }
        return true;
    }
}
