class Solution {

    public boolean method1(String s, String t) {
        int i = 0;
        int j = 0;
        int count = 0;
        while (i != s.length() && j != t.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                count += 1;
                i++;
            }
            j++;
        }
        if (count != s.length())
            return false;
        return true;
    }
}
