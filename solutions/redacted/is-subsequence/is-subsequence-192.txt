class Solution {

    public boolean method1(String s, String t) {
        int sl = s.length();
        int tl = t.length();
        if (sl == 0)
            return true;
        for (int i = 0, j = 0; i < tl; i++) {
            if (t.charAt(i) == s.charAt(j)) {
                j++;
            }
            if (j == sl)
                return true;
        }
        return false;
    }
}
