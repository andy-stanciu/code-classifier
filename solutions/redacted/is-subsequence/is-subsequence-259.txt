class Solution {

    public boolean method1(String s, String t) {
        // Empty string is a subsequence of any string
        if (s.length() == 0)
            return true;
        int sIndex = 0, tIndex = 0;
        while (tIndex < t.length()) {
            if (s.charAt(sIndex) == t.charAt(tIndex)) {
                // Move pointer in s if there\'s a match
                sIndex++;
            }
            if (sIndex == s.length()) {
                // All characters of s are matched
                return true;
            }
            // Move pointer in t
            tIndex++;
        }
        // s is not a subsequence of t
        return false;
    }
}
