class Solution {

    public int[] method1(int[] param1) {
        int n = param1.length;
        Stack<Integer> st = new Stack<>();
        for (int ast : param1) {
            //collision will hapen when:
            while (!st.isEmpty() && ast < 0 && st.peek() > 0) {
                int sum = ast + st.peek();
                if (sum < 0) {
                    st.pop();
                } else if (sum > 0) {
                    ast = 0;
                } else {
                    st.pop();
                    ast = 0;
                }
            }
            if (ast != 0) {
                st.push(ast);
            }
        }
        int[] res = new int[st.size()];
        for (int i = st.size() - 1; i >= 0; i--) {
            res[i] = st.pop();
        }
        return res;
    }
}
