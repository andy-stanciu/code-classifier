class Solution {

    public int[] method1(int[] ast) {
        int n = ast.length;
        Stack<Integer> st = new Stack<>();
        int i = 0;
        while (i < n) {
            if (st.isEmpty() || ast[i] > 0 || st.peek() < 0) {
                st.push(ast[i]);
                i++;
            } else {
                int top = st.peek();
                st.pop();
                if (-ast[i] == top) {
                    i++;
                } else if (-ast[i] < top) {
                    st.push(top);
                    i++;
                }
            }
        }
        int[] ans = new int[st.size()];
        for (int j = ans.length - 1; j >= 0; j--) {
            ans[j] = st.pop();
        }
        return ans;
    }
}
