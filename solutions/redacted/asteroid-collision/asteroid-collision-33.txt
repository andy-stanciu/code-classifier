class Solution {

    public int[] method1(int[] param1) {
        int len = param1.length;
        int j = -1;
        for (int i = 0; i < len; ) {
            int a = param1[i];
            if (j == -1 || a > 0 || param1[j] < 0) {
                param1[++j] = a;
                i++;
            } else if (param1[j] + a > 0)
                i++;
            else if (param1[j] + a == 0) {
                j--;
                i++;
            } else if (param1[j] + a < 0)
                j--;
        }
        return Arrays.copyOfRange(param1, 0, j + 1);
    }
}
