class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> astr = new Stack<>();
        for (int var1 : param1) {
            if (var1 > 0)
                astr.push(var1);
            else {
                while (!astr.isEmpty() && astr.peek() > 0 && astr.peek() < Math.abs(var1)) astr.pop();
                if (!astr.isEmpty() && astr.peek() == Math.abs(var1))
                    astr.pop();
                else if (astr.isEmpty() || astr.peek() < 0)
                    astr.push(var1);
            }
        }
        int n = astr.size();
        int[] res = new int[n];
        for (int i = n - 1; i >= 0; i--) res[i] = astr.pop();
        return res;
    }
}
