class Solution {

    public int[] method1(int[] param1) {
        int top = -1;
        for (int var1 : param1) {
            boolean destroyed = false;
            while (top >= 0 && var1 < 0 && param1[top] > 0) {
                if (param1[top] < -var1) {
                    top--;
                } else if (param1[top] == -var1) {
                    top--;
                    destroyed = true;
                    break;
                } else {
                    destroyed = true;
                    break;
                }
            }
            if (!destroyed) {
                param1[++top] = var1;
            }
        }
        int[] res = new int[top + 1];
        System.arraycopy(param1, 0, res, 0, top + 1);
        return res;
    }
}
