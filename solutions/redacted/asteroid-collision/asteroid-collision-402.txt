class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int num : param1) {
            if (st.isEmpty() || num > 0) {
                st.push(num);
            } else {
                while (!st.isEmpty() && st.peek() > 0 && Math.abs(st.peek()) < Math.abs(num)) {
                    st.pop();
                }
                if (!st.isEmpty() && (-st.peek() == num || st.peek() == -num)) {
                    st.pop();
                } else if ((st.isEmpty()) || (!st.isEmpty() && st.peek() < 0 && num < 0)) {
                    st.push(num);
                }
            }
        }
        int[] arr = new int[st.size()];
        int j = st.size() - 1;
        while (!st.isEmpty()) {
            arr[j--] = st.pop();
        }
        return arr;
    }
}
