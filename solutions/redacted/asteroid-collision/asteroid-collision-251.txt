class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] > 0) {
                st.push(param1[i]);
            } else {
                while (st.size() > 0 && st.peek() > 0 && st.peek() < -1 * param1[i]) {
                    st.pop();
                }
                if (st.size() > 0 && st.peek() == -1 * param1[i]) {
                    st.pop();
                } else if (st.isEmpty() || st.peek() < 0) {
                    st.push(param1[i]);
                }
            }
        }
        int[] ans = new int[st.size()];
        int i = st.size() - 1;
        while (st.size() > 0) {
            int ele = st.pop();
            ans[i] = ele;
            i--;
        }
        return ans;
    }
}
