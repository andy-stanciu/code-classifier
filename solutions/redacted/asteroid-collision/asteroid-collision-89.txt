class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> s = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            int current = param1[i];
            boolean var1 = false;
            // Handle collision scenarios
            while (!s.isEmpty() && s.peek() > 0 && current < 0) {
                int top = s.peek();
                if (top < -current) {
                    s.pop();
                } else if (top == -current) {
                    s.pop();
                    var1 = true;
                    break;
                } else {
                    var1 = true;
                    break;
                }
            }
            // If no collision happened, push the current asteroid
            if (!var1) {
                s.push(current);
            }
        }
        // Convert stack to array
        int[] ans = new int[s.size()];
        for (int i = s.size() - 1; i >= 0; i--) {
            ans[i] = s.pop();
        }
        return ans;
    }
}
