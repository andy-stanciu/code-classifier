class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] > 0) {
                stack.push(param1[i]);
            } else {
                while (!stack.empty() && stack.peek() > 0 && stack.peek() < Math.abs(param1[i])) {
                    stack.pop();
                }
                if (!stack.empty() && stack.peek() == Math.abs(param1[i])) {
                    stack.pop();
                } else if (stack.empty() || stack.peek() < 0) {
                    stack.push(param1[i]);
                }
            }
        }
        int[] ans = new int[stack.size()];
        int i = stack.size() - 1;
        while (!stack.empty()) {
            ans[i] = stack.pop();
            i--;
        }
        return ans;
    }
}
