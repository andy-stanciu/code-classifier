class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int ast : param1) {
            boolean exploded = false;
            // Process only if there is a collision scenario
            while (!stack.isEmpty() && stack.peek() > 0 && ast < 0) {
                int top = stack.peek();
                if (top < Math.abs(ast)) {
                    // Smaller positive asteroid explodes
                    stack.pop();
                } else if (top == Math.abs(ast)) {
                    // Both asteroids explode
                    stack.pop();
                    exploded = true;
                    break;
                } else {
                    // Incoming negative asteroid explodes
                    exploded = true;
                    break;
                }
            }
            // If the incoming asteroid wasn\'t destroyed
            if (!exploded) {
                stack.push(ast);
            }
        }
        // Collect remaining asteroids in stack to an array
        int[] result = new int[stack.size()];
        for (int i = stack.size() - 1; i >= 0; i--) {
            result[i] = stack.pop();
        }
        return result;
    }
}
