class Solution {

    public int[] method1(int[] param1) {
        int n = param1.length;
        Stack<Integer> stack = new Stack<>();
        for (int i = n - 1; i >= 0; i--) {
            if (param1[i] < 0) {
                stack.push(param1[i]);
            } else {
                while (!stack.isEmpty()) {
                    if (stack.peek() < 0 && param1[i] > -(stack.peek()))
                        stack.pop();
                    else {
                        break;
                    }
                }
                if (stack.isEmpty() || stack.peek() > 0) {
                    stack.push(param1[i]);
                } else if (param1[i] == Math.abs(stack.peek())) {
                    stack.pop();
                }
            }
        }
        int[] ans = new int[stack.size()];
        int i = 0;
        while (!stack.isEmpty()) {
            ans[i] = stack.pop();
            i++;
        }
        return ans;
    }
}
