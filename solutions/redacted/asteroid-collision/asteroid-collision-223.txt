class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        int i = 0;
        while (i < param1.length && param1[i] < 0) {
            stack.push(param1[i]);
            i++;
        }
        while (i < param1.length) {
            if (param1[i] > 0)
                stack.push(param1[i]);
            else {
                while (!stack.isEmpty() && stack.peek() > 0 && Math.abs(param1[i]) > stack.peek()) {
                    stack.pop();
                }
                if (!stack.isEmpty() && param1[i] + stack.peek() == 0) {
                    stack.pop();
                } else if (stack.isEmpty() || (!stack.isEmpty() && stack.peek() < 0)) {
                    stack.push(param1[i]);
                }
            }
            i++;
        }
        if (stack.isEmpty())
            return new int[] {};
        int[] result = new int[stack.size()];
        int idx = result.length - 1;
        while (!stack.isEmpty()) {
            result[idx--] = stack.pop();
        }
        return result;
    }
}
