class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        int n = param1.length;
        all: for (int i = n - 1; i >= 0; --i) {
            if (param1[i] > 0) {
                while (stack.size() > 0) {
                    int curr = stack.peek();
                    if (curr == -param1[i]) {
                        stack.pop();
                        continue all;
                    }
                    if (curr > 0) {
                        stack.add(param1[i]);
                        break;
                    } else {
                        curr = -curr;
                        if (param1[i] < curr)
                            break;
                        else
                            stack.pop();
                    }
                }
                if (stack.size() == 0) {
                    stack.add(param1[i]);
                }
            } else {
                stack.add(param1[i]);
            }
        }
        int[] rs = new int[stack.size()];
        int i = 0;
        while (!stack.isEmpty()) {
            rs[i++] = stack.pop();
        }
        return rs;
    }
}
