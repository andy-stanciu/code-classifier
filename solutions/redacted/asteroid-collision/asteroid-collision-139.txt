class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            int curast = param1[i];
            while (!st.isEmpty() && ((st.peek() > 0 && curast < 0))) {
                int temp = st.pop();
                if (Math.abs(temp) > Math.abs(curast)) {
                    curast = temp;
                } else if (Math.abs(temp) == Math.abs(param1[i])) {
                    curast = 0;
                }
            }
            if (curast != 0)
                st.push(curast);
        }
        int[] res = new int[st.size()];
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = st.pop();
        }
        return res;
    }
}
