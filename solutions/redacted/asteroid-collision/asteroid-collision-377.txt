class Solution {

    public int[] method1(int[] param1) {
        // collection case
        Stack<Integer> s = new Stack<>();
        for (int var1 : param1) {
            while (!s.isEmpty() && s.peek() > 0 && var1 < 0) {
                int prev = s.pop();
                if (prev == -var1) {
                    var1 = 0;
                } else if (prev > -var1) {
                    s.push(prev);
                    var1 = 0;
                }
            }
            // if not then put it into the stack
            if (var1 != 0) {
                s.push(var1);
            }
        }
        int[] res = new int[s.size()];
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = s.pop();
        }
        return res;
    }
}
