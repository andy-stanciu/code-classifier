class Solution {

    public int[] method1(int[] param1) {
        int n = param1.length;
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < n; i++) {
            if (param1[i] > 0) {
                st.push(param1[i]);
            } else {
                while (!st.isEmpty() && st.peek() < Math.abs(param1[i]) && st.peek() > 0) {
                    st.pop();
                }
                if (!st.isEmpty() && st.peek() == Math.abs(param1[i])) {
                    st.pop();
                } else if (st.isEmpty() || st.peek() < 0) {
                    st.push(param1[i]);
                }
            }
        }
        int size = st.size();
        int[] ans = new int[size];
        for (int i = ans.length - 1; i >= 0; i--) {
            ans[i] = st.pop();
        }
        return ans;
    }
}
