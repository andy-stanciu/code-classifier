class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        int n = param1.length;
        st.push(param1[0]);
        for (int i = 1; i < n; i++) {
            if (!st.isEmpty() && param1[i] < 0 && st.peek() >= 0) {
                while (!st.isEmpty() && Math.abs(param1[i]) > st.peek() && st.peek() > 0) {
                    st.pop();
                }
                if (!st.isEmpty() && Math.abs(param1[i]) < st.peek()) {
                    continue;
                } else if (!st.isEmpty() && Math.abs(param1[i]) == st.peek()) {
                    st.pop();
                    continue;
                } else
                    st.push(param1[i]);
            } else
                st.push(param1[i]);
        }
        int p = st.size();
        int[] ans = new int[p];
        for (int i = p - 1; i >= 0; i--) {
            ans[i] = st.pop();
        }
        return ans;
    }
}
