class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> s = new Stack<>();
        s.push(param1[0]);
        for (int i = 1; i < param1.length; i++) {
            if (s.isEmpty() == true) {
                s.push(param1[i]);
                continue;
            }
            while (!s.isEmpty()) {
                if ((s.peek() > 0 && param1[i] > 0) || (s.peek() < 0 && param1[i] < 0)) {
                    s.push(param1[i]);
                    break;
                } else {
                    if (s.peek() < 0 && param1[i] > 0) {
                        s.push(param1[i]);
                        break;
                    } else if (s.peek() > 0 && param1[i] < 0) {
                        if (Math.abs(s.peek()) > Math.abs(param1[i])) {
                            break;
                        } else if (Math.abs(s.peek()) < Math.abs(param1[i])) {
                            s.pop();
                            if (s.isEmpty() == true) {
                                s.push(param1[i]);
                                break;
                            }
                        } else {
                            s.pop();
                            break;
                        }
                    }
                }
            }
        }
        int[] ans = new int[s.size()];
        int size = s.size();
        for (int i = size - 1; i >= 0; i--) {
            ans[i] = s.pop();
        }
        return ans;
    }
}
