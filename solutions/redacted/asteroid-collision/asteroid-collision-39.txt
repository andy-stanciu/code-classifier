class Solution {

    public int[] method1(int[] param1) {
        // Stack storing the asteroids in the continuous manner
        Stack<Integer> stack = new Stack<>();
        for (int var2 : param1) {
            boolean canPush = true;
            while (!stack.isEmpty() && (stack.peek() * var2) < 0) {
                // If the top asteroid has negative value and upcoming asteroid has positive value then no collision will happen, as direction of both asteroids is opposite
                if (stack.peek() < 0 && var2 > 0)
                    break;
                int var1 = Math.abs(var2);
                int absTop = Math.abs(stack.peek());
                // If the upcoming asteroid is bigger than the top asteroid
                // then explode the top asteroid
                if (var1 > absTop)
                    stack.pop();
                else // If both asteroid have same sizes then both asteroids will explode
                if (var1 == absTop) {
                    stack.pop();
                    canPush = false;
                    break;
                } else // If the upcoming asteroid is smaller than the top asteroid
                {
                    canPush = false;
                    break;
                }
            }
            if (canPush)
                stack.push(var2);
        }
        // Put the remaining asteroids in array to return the result
        int size = stack.size();
        int[] result = new int[size];
        for (int i = size - 1; i >= 0; i--) result[i] = stack.pop();
        return result;
    }
}
