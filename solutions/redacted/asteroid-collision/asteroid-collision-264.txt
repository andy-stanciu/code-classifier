class Solution {

    public int[] method1(int[] a) {
        Stack<Integer> stack = new Stack<>();
        int i = 1;
        int len = a.length;
        stack.push(a[0]);
        while (i < len) {
            int peek = stack.size() > 0 ? stack.peek() : 0;
            if ((peek > 0 && a[i] < 0)) {
                if (Math.abs(peek) < Math.abs(a[i])) {
                    stack.pop();
                } else if (Math.abs(peek) == Math.abs(a[i])) {
                    stack.pop();
                    i++;
                } else {
                    i++;
                }
            } else {
                stack.push(a[i]);
                i++;
            }
        }
        int[] res = new int[stack.size()];
        for (i = res.length - 1; i >= 0; i--) {
            res[i] = stack.pop();
        }
        return res;
    }
}
