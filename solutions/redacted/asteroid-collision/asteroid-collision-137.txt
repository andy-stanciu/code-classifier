class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            if (param1[i] > 0 || st.isEmpty()) {
                st.push(param1[i]);
            } else {
                while (!st.isEmpty() && st.peek() > 0 && st.peek() < Math.abs(param1[i])) {
                    st.pop();
                }
                if (!st.isEmpty() && st.peek() == Math.abs(param1[i])) {
                    st.pop();
                } else if (st.isEmpty() || st.peek() < 0) {
                    st.push(param1[i]);
                }
            }
        }
        int len = st.size();
        int[] arr = new int[len];
        for (int i = len - 1; i >= 0; i--) {
            arr[i] = st.pop();
        }
        return arr;
    }
}
