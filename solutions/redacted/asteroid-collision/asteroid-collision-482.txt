class Solution {

    public int[] method1(int[] param1) {
        int count = 0;
        int len = param1.length;
        ArrayList<Integer> list = new ArrayList<>();
        int j = 1;
        list.add(param1[0]);
        for (int i = 1; i < len; i++) {
            if (list.size() > 0 && (param1[i] < 0 && list.get(list.size() - 1) > 0)) {
                int check = 0;
                while (list.size() > 0 && (param1[i] < 0 && list.get(list.size() - 1) > 0)) {
                    if (Math.abs(param1[i]) < Math.abs(list.get(list.size() - 1))) {
                        check = 0;
                        break;
                    } else if (Math.abs(param1[i]) == Math.abs(list.get(list.size() - 1))) {
                        list.remove(list.size() - 1);
                        check = 0;
                        break;
                    } else {
                        list.remove(list.size() - 1);
                        check++;
                    }
                }
                if (check > 0) {
                    list.add(param1[i]);
                }
                continue;
            }
            list.add(param1[i]);
        }
        int[] arr = new int[list.size()];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = list.get(i);
        }
        return arr;
    }
}
