class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int var1 : param1) {
            boolean destroyed = false;
            while (!stack.isEmpty() && (stack.peek() > 0 && var1 < 0)) {
                if (stack.peek() > Math.abs(var1)) {
                    destroyed = true;
                    break;
                } else if (stack.peek() == Math.abs(var1)) {
                    stack.pop();
                    destroyed = true;
                    break;
                } else
                    stack.pop();
            }
            if (!destroyed)
                stack.push(var1);
        }
        int[] result = new int[stack.size()];
        int index = 0;
        for (int item : stack) result[index++] = item;
        return result;
    }
}
