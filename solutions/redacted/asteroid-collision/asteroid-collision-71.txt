class Solution {

    public int[] method1(int[] param1) {
        ArrayList<Integer> stack = new ArrayList<>();
        for (int ast : param1) {
            // 8
            boolean ignore = false;
            if (ast < 0) {
                while (!stack.isEmpty() && stack.getLast() > 0) {
                    if (stack.getLast() > Math.abs(ast)) {
                        ignore = true;
                        break;
                    } else if (stack.getLast() == Math.abs(ast)) {
                        ignore = true;
                        stack.removeLast();
                        break;
                    } else {
                        stack.removeLast();
                    }
                }
                if (!ignore) {
                    stack.addLast(ast);
                }
            } else {
                stack.addLast(ast);
            }
        }
        int[] res = new int[stack.size()];
        for (int i = 0; i < stack.size(); ++i) {
            res[i] = stack.get(i);
        }
        return res;
    }
}
