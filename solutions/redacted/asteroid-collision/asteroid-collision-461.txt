class Solution {

    public int[] method1(int[] ast) {
        Stack<Integer> s = new Stack<>();
        for (int i = 0; i < ast.length; i++) {
            if (s.empty() || ast[i] >= 0)
                s.push(ast[i]);
            else {
                while (!s.empty() && s.peek() > 0 && s.peek() < Math.abs(ast[i])) s.pop();
                if (!s.empty() && s.peek() == Math.abs(ast[i]))
                    s.pop();
                else if (s.empty() || s.peek() < 0)
                    s.push(ast[i]);
            }
        }
        int[] ans = new int[s.size()];
        int i = s.size() - 1;
        while (!s.empty()) ans[i--] = s.pop();
        return ans;
    }
}
