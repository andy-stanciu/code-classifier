class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int a : param1) {
            // when collision happens ??
            while (!st.isEmpty() && a < 0 && st.peek() > 0) {
                int sum = st.peek() + a;
                if (sum < 0) {
                    st.pop();
                } else if (sum > 0) {
                    // asteroid futt gaya
                    a = 0;
                } else {
                    st.pop();
                    // asteroid futt gaya
                    a = 0;
                }
            }
            if (a != 0) {
                // add if asteroid nhi futaa hai or no collision happens
                st.push(a);
            }
        }
        int s = st.size();
        int[] result = new int[s];
        for (int i = s - 1; i >= 0; i--) {
            result[i] = st.pop();
        }
        return result;
    }
}
