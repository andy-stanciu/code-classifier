class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int cur : param1) {
            boolean isDestroyed = false;
            while (!stack.isEmpty() && stack.peek() > 0 && cur < 0) {
                if (stack.peek() < -cur) {
                    stack.pop();
                    continue;
                } else if (stack.peek() == -cur) {
                    stack.pop();
                }
                isDestroyed = true;
                break;
            }
            if (!isDestroyed) {
                stack.push(cur);
            }
        }
        return stack.stream().mapToInt(i -> i).toArray();
    }
}
