class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] > 0 || stack.isEmpty()) {
                stack.push(param1[i]);
            } else {
                while (!stack.isEmpty() && stack.peek() > 0 && stack.peek() < (-1 * param1[i])) {
                    stack.pop();
                }
                if (!stack.isEmpty() && stack.peek() == (-1 * param1[i])) {
                    stack.pop();
                } else {
                    if (stack.isEmpty() || stack.peek() < 0) {
                        stack.push(param1[i]);
                    }
                }
            }
        }
        int[] arr = new int[stack.size()];
        for (int k = 0; k < arr.length; k++) {
            arr[k] = stack.get(k);
        }
        return arr;
    }
}
