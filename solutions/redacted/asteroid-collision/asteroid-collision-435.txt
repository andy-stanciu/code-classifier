class Solution {

    public int[] method1(int[] param1) {
        /*
        Scenarios:
        (+, +) -> no collision
        (+, -) -> collision
        (-, +) -> no collision
        (-, -) -> no collision 
        */
        Stack<Integer> st = new Stack<>();
        for (int var1 : param1) {
            // no collision no matter what is on top
            if (var1 > 0) {
                st.push(var1);
                continue;
            }
            // +ve on top/ incoming -ve
            // keep popping till st.peek() < -(asteroid)
            while (!st.isEmpty() && st.peek() > 0 && st.peek() < -var1) st.pop();
            // if peek is equal to incoming: destroy both
            if (!st.isEmpty() && st.peek() == -var1) {
                st.pop();
            } else if (st.isEmpty() || st.peek() < 0) {
                // if stack becomes empty or top has a -ve value: push incoming
                st.push(var1);
            } else {
                // if incoming < peek : destroy it i.e. continue
            }
        }
        int[] ans = new int[st.size()];
        for (int i = ans.length - 1; i >= 0; i--) {
            ans[i] = st.pop();
        }
        return ans;
    }
}
