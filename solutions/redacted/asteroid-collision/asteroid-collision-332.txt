class Solution {

    public int[] method1(int[] param1) {
        ArrayDeque<Integer> var1 = new ArrayDeque<>();
        for (int var2 : param1) {
            // When there are asteroids in the stack and the new asteroid not explodes.
            while (!var1.isEmpty() && var2 != 0) {
                // Peek the last asteroid in the stack.
                int last = var1.peekLast();
                // Only when they are facing each other, they collides.
                if (var2 < 0 && last > 0) {
                    // If the old one wins, set the new one to 0. If the new one wins, remove the old one from the stack.
                    if (Math.abs(var2) == last) {
                        var2 = 0;
                        var1.removeLast();
                    } else if (Math.abs(var2) < last) {
                        var2 = 0;
                    } else {
                        var1.removeLast();
                    }
                } else {
                    break;
                }
            }
            // After all the collision, if the new asteroid still not explodes, add it to the stack.
            if (var2 != 0) {
                var1.addLast(var2);
            }
        }
        int[] res = new int[var1.size()];
        int index = 0;
        for (int var2 : var1) {
            res[index++] = var2;
        }
        return res;
    }
}
