class Solution {

    public int[] method1(int[] param1) {
        int[] rest = Arrays.copyOf(param1, param1.length);
        int count = 0;
        for (int i = 0; i < rest.length; i++) {
            if (count == 0 || rest[count - 1] < 0 || (rest[count - 1] > 0 && param1[i] > 0)) {
                rest[count++] = param1[i];
            } else {
                while (count > 0 && rest[count - 1] > 0 && Math.abs(rest[count - 1]) < Math.abs(param1[i])) {
                    // Current asteroid exploded
                    count--;
                }
                if (count == 0 || rest[count - 1] < 0) {
                    rest[count++] = param1[i];
                } else if (Math.abs(rest[count - 1]) == Math.abs(param1[i])) {
                    // Both asteroids exploded
                    count--;
                }
                // Else: current asteroid exploded, and previous asteroid survives
            }
        }
        return Arrays.copyOfRange(rest, 0, count);
    }
}
