class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            while (!st.isEmpty() && param1[st.peek()] > 0 && param1[i] < 0) {
                int sum = param1[st.peek()] + param1[i];
                if (sum < 0) {
                    st.pop();
                } else if (sum > 0) {
                    param1[i] = 0;
                } else {
                    st.pop();
                    param1[i] = 0;
                }
            }
            if (param1[i] != 0) {
                st.push(i);
            }
        }
        int[] a = new int[st.size()];
        int i = a.length - 1;
        while (!st.isEmpty()) {
            a[i--] = param1[st.pop()];
        }
        return a;
    }
}
