class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int var1 : param1) {
            if (stack.isEmpty() || var1 > 0) {
                // Positive asteroid or stack is empty, no collision, push to stack
                stack.push(var1);
            } else {
                // Negative asteroid, check for collisions with positive asteroids in the stack
                while (!stack.isEmpty() && stack.peek() > 0 && stack.peek() < Math.abs(var1)) {
                    // Smaller positive asteroid explodes
                    stack.pop();
                }
                if (stack.isEmpty() || stack.peek() < 0) {
                    // Negative asteroid survives collision or stack is empty
                    stack.push(var1);
                } else if (stack.peek() == Math.abs(var1)) {
                    // Both asteroids explode
                    stack.pop();
                }
            }
        }
        int[] result = new int[stack.size()];
        for (int i = stack.size() - 1; i >= 0; i--) {
            result[i] = stack.pop();
        }
        return result;
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        int[] var2 = { 5, 10, -5 };
        int[] result1 = solution.method1(var2);
        // Output: [5, 10]
        System.out.println(Arrays.toString(result1));
        int[] var3 = { 8, -8 };
        int[] result2 = solution.method1(var3);
        // Output: []
        System.out.println(Arrays.toString(result2));
        int[] var4 = { 10, 2, -5 };
        int[] result3 = solution.method1(var4);
        // Output: [10]
        System.out.println(Arrays.toString(result3));
    }
}
