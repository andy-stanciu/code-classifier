class Solution {

    public int[] method1(int[] param1) {
        int j = -1;
        for (int i = 0; i < param1.length; i++) {
            int ast = param1[i];
            while (j > -1 && ast < 0 && param1[j] > 0 && -ast > param1[j]) j--;
            if (j == -1 || ast > 0 || param1[j] < 0)
                param1[++j] = ast;
            else if (-ast == param1[j])
                j--;
        }
        return Arrays.copyOfRange(param1, 0, j + 1);
    }
}
