class Solution {

    public int[] method1(int[] param1) {
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < param1.length; i++) {
            //if stack is empty or positive value insert element directly to the stack
            if (stack.isEmpty() || param1[i] > 0) {
                stack.push(param1[i]);
            } else {
                while (!stack.isEmpty()) {
                    int top = stack.peek();
                    if (top < 0) {
                        stack.push(param1[i]);
                        break;
                    }
                    int modVal = Math.abs(param1[i]);
                    if (modVal == top) {
                        stack.pop();
                        break;
                    } else if (modVal < top) {
                        break;
                    } else {
                        stack.pop();
                        if (stack.isEmpty()) {
                            stack.push(param1[i]);
                            break;
                        }
                    }
                }
            }
        }
        int len = stack.size();
        int[] ansArr = new int[len];
        for (int i = len - 1; i >= 0; i--) {
            ansArr[i] = stack.pop();
        }
        return ansArr;
    }
}
