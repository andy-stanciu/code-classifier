class Solution {

    public int method1(int[] nums) {
        // Edge case: empty array
        if (nums.length == 0)
            return 0;
        // `k` will represent the length of the array with unique elements
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                // Place the unique element in the next position
                nums[k] = nums[i];
                // Increment the count of unique elements
                k++;
            }
        }
        // Return the length of the array with unique elements
        return k;
    }
}
