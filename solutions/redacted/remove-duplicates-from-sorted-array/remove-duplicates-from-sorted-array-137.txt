class Solution {

    public static int method1(int[] nums, int val) {
        int n = nums.length;
        int count = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == val) {
                count++;
                if (count == 1) {
                    continue;
                }
                for (int j = i; j < n - 1; j++) {
                    int temp = nums[j];
                    nums[j] = nums[j + 1];
                    nums[j + 1] = temp;
                }
                n--;
            }
            if (nums[i] == val) {
                i--;
            }
        }
        return n;
    }

    public static int count(int[] arr, int val) {
        int n = arr.length;
        int co = 0;
        for (int i = 0; i < n; i++) {
            if (arr[i] == val) {
                co++;
            }
        }
        return co;
    }

    public int method2(int[] nums) {
        int n = nums.length;
        int x = 0, sum = 0;
        for (int i = 0; i < n; i++) {
            int c = count(nums, nums[i]);
            if (c > 1) {
                sum = sum + (c - 1);
                x = method1(nums, nums[i]);
            }
            n = n - (c - 1);
        }
        return n;
    }
}
