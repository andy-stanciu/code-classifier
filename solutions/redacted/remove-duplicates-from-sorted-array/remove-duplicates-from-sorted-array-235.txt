class Solution {

    public int method1(int[] nums) {
        // Points to the last unique element
        int i = 0;
        for (int j = 1; j < nums.length; j++) {
            // j explores new elements
            if (nums[i] != nums[j]) {
                // Found a new unique element
                // Move it next to last unique element
                nums[i + 1] = nums[j];
                // Increment i to the next unique position
                i++;
            }
        }
        // i is the last index of unique elements, so return i + 1 as the length
        return i + 1;
    }
}
