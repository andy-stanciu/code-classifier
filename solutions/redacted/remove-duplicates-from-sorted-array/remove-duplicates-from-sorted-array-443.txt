class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        int j = 0;
        for (int i = 1; i < len; i++) {
            if (nums[i] != nums[j]) {
                j++;
                if (i != j) {
                    nums[j] = nums[i];
                }
            }
        }
        return j + 1;
    }
}
