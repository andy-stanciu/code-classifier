class Solution {

    public int method1(int[] nums) {
        // declare idx variable
        int idx = 1;
        // iterate with pointer on nums array
        for (int p = 1; p < nums.length; p++) {
            // check if curr and prev values are unique
            if (nums[p - 1] != nums[p]) {
                // if true swap current value to idx value
                nums[idx] = nums[p];
                // increment idx
                idx++;
            }
        }
        // return idx
        return idx;
    }
}
