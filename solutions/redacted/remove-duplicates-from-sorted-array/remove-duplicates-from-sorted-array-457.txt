class Solution {

    public int method1(int[] nums) {
        List<Integer> indexes = new ArrayList<>();
        indexes.add(0);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                indexes.add(i);
            }
        }
        for (int j = 0; j < indexes.size(); j++) {
            nums[j] = nums[indexes.get(j)];
        }
        return indexes.size();
    }
}
