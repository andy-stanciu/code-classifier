class Solution {

    public int method1(int[] nums) {
        int i = 0, j = i + 1;
        while (j < nums.length) {
            if (nums[j] != nums[i]) {
                if (j != i + 1) {
                    int temp = nums[i + 1];
                    nums[i + 1] = nums[j];
                    nums[j] = temp;
                }
                i++;
            }
            j++;
        }
        return i + 1;
    }
}
