class Solution {

    public int method1(int[] nums) {
        int[] num = new int[nums.length];
        int a = 0;
        for (int i = 0; i < num.length; i++) {
            num[i] = 101;
        }
        for (int i = 0; i < nums.length; i++) {
            if (check(nums[i], num) == false) {
                num[a] = nums[i];
                a++;
            }
        }
        int count = 0;
        for (int i = 0; i < num.length; i++) {
            if (num[i] != 101) {
                count++;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            nums[i] = num[i];
        }
        return count;
    }

    static boolean check(int numb, int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            if (numb == nums[i]) {
                return true;
            }
        }
        return false;
    }
}
