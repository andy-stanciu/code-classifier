class Solution {

    public int method1(int[] nums) {
        int index = 1;
        int n = nums.length;
        if (n > 1 && nums[0] != nums[1]) {
            nums[index++] = nums[1];
        }
        for (int i = 1; i < n - 1; i++) {
            if (nums[i] != nums[i + 1]) {
                nums[index++] = nums[i + 1];
            }
        }
        if (nums[n - 1] != nums[index - 1]) {
            nums[index++] = nums[n - 1];
        }
        return index;
    }
}
