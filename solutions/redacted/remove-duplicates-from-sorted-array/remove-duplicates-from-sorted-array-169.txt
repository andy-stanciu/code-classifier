// Method 1 (Using Extra Space)
class Solution {

    public int method1(int[] arr) {
        int n = arr.length, j = 0;
        int[] temp = new int[n];
        for (int i = 0; i < (n - 1); i++) if (arr[i] != arr[i + 1])
            temp[j++] = arr[i];
        temp[j++] = arr[n - 1];
        for (int i = 0; i < j; i++) arr[i] = temp[i];
        return j;
    }
}
