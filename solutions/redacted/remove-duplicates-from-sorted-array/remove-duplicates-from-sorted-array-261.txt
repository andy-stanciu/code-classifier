class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> sec = new HashSet<>();
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (!sec.contains(nums[i])) {
                sec.add(nums[i]);
                nums[var1++] = nums[i];
            }
        }
        return var1;
    }
}
