class Solution {

    public int method1(int[] nums) {
        int k = 1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != nums[k - 1]) {
                nums[k] = nums[i];
                k++;
            }
        }
        nums = Arrays.copyOf(nums, k);
        return nums.length;
    }
}
