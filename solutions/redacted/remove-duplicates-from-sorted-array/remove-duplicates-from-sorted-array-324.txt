class Solution {

    public int method1(int[] nums) {
        // keeping track of duplicate elements count
        int dupCount = 0;
        // since the contraints says that array will be
        // of atleast one length,
        // hence minimum unique length will be 1.
        int uniqueCount = 1;
        // starting of the 1st index, since the 0th index
        // element will always unique first time.
        for (int i = 1; i < nums.length; i++) {
            // if the current ith index value is same
            // as previous index value then a duplicate element is found,
            // hence increasing the dupCount
            if (nums[i] == nums[i - 1])
                dupCount++;
            else {
                // if it\'s a unique element
                // increase the uniqueCount and
                // go back dupCount from current index and
                // replace that value with current index value.
                uniqueCount++;
                nums[i - dupCount] = nums[i];
            }
        }
        return uniqueCount;
    }
}
