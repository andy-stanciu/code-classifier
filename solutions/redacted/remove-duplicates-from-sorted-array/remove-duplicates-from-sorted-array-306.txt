class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        // Pointer for the position of unique elements
        int i = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i]) {
                // Move to the next position
                i++;
                // Update the unique position
                nums[i] = nums[j];
            }
        }
        // Length is index + 1
        return i + 1;
    }
}
