class Solution {

    public int method1(int[] nums) {
        int newIndex = 0;
        int temp = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if ((nums[i] != temp) || i == 0) {
                nums[newIndex] = nums[i];
                temp = nums[i];
                newIndex++;
            }
        }
        return newIndex;
    }
}
