class Solution {

    public int method1(int[] nums) {
        int count = 1;
        int i = 0, j = 1;
        while (j < nums.length) {
            if (nums[i] != nums[j]) {
                nums[count] = nums[j];
                count++;
                i = j;
                j++;
            } else {
                j++;
            }
        }
        return count;
    }
}
