class Solution {

    public int method1(int[] nums) {
        //This is like a counter use to keep the track of the unique index start from one as the first element is always unique
        int DiffElement = 1;
        for (int i = 1; i < nums.length; i++) {
            if (//Checks whether the current element is distinct from the previous
            nums[i] != nums[i - 1]) {
                //If Yes then at that index assign that element
                nums[DiffElement] = nums[i];
                //Increment the counter i.e Index to place the next unique element
                DiffElement++;
            }
        }
        //Return the number of unique element
        return DiffElement;
    }
}
