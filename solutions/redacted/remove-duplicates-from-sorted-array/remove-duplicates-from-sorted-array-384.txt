class Solution {

    public int method1(int[] nums) {
        int i = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[i] != nums[j]) {
                i++;
                nums[i] = nums[j];
            }
        }
        // After completion of the loop return i+1, i.e size of the array of unique elements.
        return i + 1;
    }
}
