class Solution {

    public int method1(int[] nums) {
        int Befindex = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[Befindex]) {
                Befindex++;
                nums[Befindex] = nums[i];
            }
        }
        return Befindex + 1;
    }
}
