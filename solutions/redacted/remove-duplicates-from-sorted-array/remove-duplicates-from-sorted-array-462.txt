class Solution {

    public int method1(int[] nums) {
        int k = 1;
        int j = 0;
        for (int i = 1; i < nums.length; i++) {
            // we are no longer on an equal number, so we have a new unique number
            if (nums[i] != nums[j]) {
                // j++, then set our newly found number to the next element in the arry
                j++;
                nums[j] = nums[i];
                // add to unique count
                k++;
            }
        }
        return k;
    }
}
