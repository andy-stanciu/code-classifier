class Solution {

    public int method1(int[] nums) {
        LinkedHashSet<Integer> hashSet = new LinkedHashSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            hashSet.add(nums[i]);
        }
        Iterator iterator = hashSet.iterator();
        int index = 0;
        while (iterator.hasNext()) {
            nums[index] = (int) iterator.next();
            index++;
        }
        return hashSet.size();
    }
}
