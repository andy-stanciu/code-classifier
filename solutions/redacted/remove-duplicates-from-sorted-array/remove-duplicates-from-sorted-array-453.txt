class Solution {

    public int method1(int[] nums) {
        // Handle edge case
        if (nums.length == 0)
            return 0;
        // Pointer to track unique elements
        // Start from 1 since the first element is always unique
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            // If we find a unique element (i.e., different from the previous one)
            if (nums[i] != nums[i - 1]) {
                // Place it at the position k
                nums[k] = nums[i];
                // Move the pointer k forward
                k++;
            }
        }
        // k represents the number of unique elements
        return k;
    }
}
