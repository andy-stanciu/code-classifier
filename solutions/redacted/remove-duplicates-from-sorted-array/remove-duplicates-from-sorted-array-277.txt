class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        // This pointer will keep track of the position of the next unique element.
        int k = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                // Compare the current element with the previous one
                // Update the position with the unique element
                nums[k] = nums[i];
                // Move to the next position
                k++;
            }
        }
        // k represents the number of unique elements
        return k;
    }
}
