class Solution {

    public int method1(int[] nums) {
        // If the array is empty, there are no unique elements.
        if (nums.length == 0) {
            return 0;
        }
        // Initialize the k pointer to 1, since the first element is always unique.
        int k = 1;
        // Start iterating from the second element
        for (int i = 1; i < nums.length; i++) {
            // If current element is not equal to the previous one
            if (nums[i] != nums[i - 1]) {
                // Place it at the k-th position and increment k
                nums[k] = nums[i];
                k++;
            }
        }
        // Return the number of unique elements
        return k;
    }
}
