class Solution {

    public int method1(int[] nums) {
        // i and j pointer will be initialized from 0.
        int i = 0;
        int j = 0;
        while (true) {
            while (j < nums.length && nums[j] <= nums[i]) {
                // j will increase untill it points to
                // non-repeated value.
                j++;
            }
            if (j < nums.length)
                // we will check whether j is out of range.
                nums[++i] = nums[j];
            else
                break;
        }
        // Since we have to return position so will add 1 to index i.
        return i + 1;
    }
}
