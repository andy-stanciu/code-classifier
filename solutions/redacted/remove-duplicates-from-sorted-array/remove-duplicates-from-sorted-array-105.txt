class Solution {

    public int method1(int[] nums) {
        Set<Integer> uniqueSet = new HashSet<>();
        for (int num : nums) uniqueSet.add(num);
        int ans = uniqueSet.size();
        int i = 0;
        for (int x : uniqueSet) {
            nums[i] = x;
            i++;
        }
        return ans;
    }
}
