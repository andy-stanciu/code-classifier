class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> map = new HashSet<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (map.add(nums[i])) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
