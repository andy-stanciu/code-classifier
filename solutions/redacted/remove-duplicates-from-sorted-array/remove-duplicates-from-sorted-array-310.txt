class Solution {

    public int method1(int[] nums) {
        int[] expectedNums = new int[nums.length];
        int k = 0;
        System.out.println(Arrays.toString(expectedNums));
        for (int i = 0; i <= nums.length - 1; i++) {
            if ((isNumbUnique(nums, i) == true)) {
                // comparing array with an element of an array
                expectedNums[k++] = nums[i];
                System.out.println(k);
            }
        }
        //deep copying of an array
        for (int i = 0; i <= nums.length - 1; i++) {
            nums[i] = expectedNums[i];
        }
        System.out.println(Arrays.toString(nums));
        return k;
    }

    //defining uniqueNumber function
    public static boolean isNumbUnique(int[] nums, int a) {
        boolean found = false;
        for (int i = 0; i <= nums.length - 1; i++) {
            if (nums[i] == nums[a]) {
                if ((found == true) && (i > a)) {
                    return false;
                }
                found = true;
            }
        }
        System.out.println("unique numbers" + a);
        return true;
    }
}
