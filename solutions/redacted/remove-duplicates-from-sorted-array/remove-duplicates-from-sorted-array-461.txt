class Solution {

    public int method1(int[] nums) {
        if (null == nums || nums.length == 0) {
            return 0;
        }
        int uniqueIndex = 0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[uniqueIndex] != nums[j]) {
                uniqueIndex++;
                nums[uniqueIndex] = nums[j];
            }
        }
        return uniqueIndex + 1;
    }
}
