class Solution {

    public int method1(int[] nums) {
        int temp = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (temp != nums[i]) {
                temp = nums[i];
                nums[count] = temp;
                count++;
            }
        }
        return count;
    }
}
