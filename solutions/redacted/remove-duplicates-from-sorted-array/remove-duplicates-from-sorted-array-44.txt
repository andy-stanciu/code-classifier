class Solution {

    public int method1(int[] nums) {
        int ptr = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] != nums[i + 1]) {
                //When we get unique No.
                //update previous pointer with new element
                nums[ptr] = nums[i + 1];
                ptr++;
            }
        }
        return ptr;
    }
}
