class Solution {

    public int method1(int[] nums) {
        //we will use two pointer approach
        //first pointer for previous value  i
        int i = 0;
        //pointer j for next number cheking
        for (int j = 1; j < nums.length; j++) {
            //in each  case wheter element is unique or not we will increment as to go on cheking
            if (nums[i] != nums[j]) {
                //when we find unique elemnt increment count of i
                i++;
                //update the value that repeting with unique we found
                nums[i] = nums[j];
            }
        }
        //as we have to count first also we write i+1;
        return i + 1;
    }
}
