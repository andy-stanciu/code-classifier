class Solution {

    public int method1(int[] nums) {
        int k = 0;
        int current = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != current) {
                k++;
                nums[k] = nums[i];
                current = nums[i];
            }
        }
        return k + 1;
    }
}
