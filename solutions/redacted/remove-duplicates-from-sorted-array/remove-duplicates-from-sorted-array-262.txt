class Solution {

    public int method1(int[] nums) {
        //--------- Optimized Approach ---------//
        // Initialize a variable \'rd\' (read pointer) to keep track of the
        //current position where  we will write the next unique element
        int rd = 0;
        // Initialize a variable \'count\' to keep track of the count of
        // distinct elements (initialized to 1 because the first element is always unique)
        int count = 1;
        // Loop through the input array starting from the second element
        for (int i = 1; i < nums.length; i++) {
            // Check if the current element is different from the element at \'rd\' position
            if (nums[i] != nums[rd]) {
                // If it is different, increment \'rd\' to move to the next position and update the count
                rd++;
                count++;
                // Update the element at \'rd\' position with the current element to maintain the uniqueness
                nums[rd] = nums[i];
            }
        }
        // Return the count of distinct elements
        return count;
    }
}
