class Solution {

    public int method1(int[] nums) {
        Stack<Integer> stack = new Stack<>();
        stack.push(nums[0]);
        int i = 1;
        int index = 1;
        while (i < nums.length) {
            if (nums[i] == stack.peek()) {
            } else {
                nums[index] = nums[i];
                stack.push(nums[i]);
                index++;
            }
            i++;
        }
        return stack.size();
    }
}
