class Solution {

    public int method1(int[] nums) {
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i < nums.length - 1 && nums[i] == nums[i + 1]) {
                k += 0;
            } else {
                nums[k] = nums[i];
                k += 1;
            }
        }
        return (k);
    }
}
