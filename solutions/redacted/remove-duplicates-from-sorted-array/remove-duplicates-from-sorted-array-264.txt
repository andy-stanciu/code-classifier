class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        int index = 0;
        for (int i = 1; i < n; i++) {
            // Check if the current element is different from index ele
            if (nums[i] != nums[index]) {
                // Increment index and add ele to index
                nums[++index] = nums[i];
            }
        }
        return index + 1;
    }
}
