class Solution {

    public int method1(int[] nums) {
        int pointer = 1;
        int curr = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != curr) {
                curr = nums[i];
                nums[pointer] = curr;
                pointer++;
            }
        }
        return pointer;
    }
}
