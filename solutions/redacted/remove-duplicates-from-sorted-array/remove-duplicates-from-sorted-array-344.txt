class Solution {

    public int method1(int[] nums) {
        // If the array has 0 or 1 element, no duplicates can exist
        if (nums.length == 0) {
            return 0;
        }
        // Initialize the first pointer
        int i = 0;
        // Iterate with the second pointer
        for (int j = 1; j < nums.length; j++) {
            // When a unique element is found (i.e., nums[j] != nums[i])
            if (nums[j] != nums[i]) {
                // Move the first pointer and update nums[i+1]
                i++;
                nums[i] = nums[j];
            }
        }
        // Return the number of unique elements
        return i + 1;
    }
}
