class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0 || nums.length == 1) {
            return nums.length;
        }
        TreeSet<Integer> set = new TreeSet<>();
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        int i = 0;
        for (Integer s : set) {
            nums[i++] = s;
        }
        return set.size();
    }
}
