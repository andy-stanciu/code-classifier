class Solution {

    public int method1(int[] nums) {
        int i = 0;
        for (int j = 0; j < nums.length; ++j) {
            if (nums[j] != nums[i]) {
                //\u5148\u52A0\u518D\u8D4B\u503C\u662F\u4ED6\u7684\u7CBE\u9AD3
                nums[++i] = nums[j];
            }
        }
        return i + 1;
    }
}
