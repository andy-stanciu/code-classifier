class Solution {

    public int method1(int[] nums) {
        int x = 1;
        switch(nums.length) {
            case (0):
                return 0;
            case (1):
                return x;
        }
        for (int i = 1; i < nums.length; i++) {
            if (nums[x - 1] != nums[i]) {
                // delets duplicates
                nums[x] = nums[i];
                //x counts the amount of non-duplicates
                x++;
            }
        }
        return x;
    }
}
