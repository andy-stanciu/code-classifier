class Solution {

    public int method1(int[] nums) {
        // Index for the next unique element
        int j = 1;
        for (int i = 1; i < nums.length; i++) {
            // If the current element is different from the previous one
            if (nums[i] != nums[i - 1]) {
                // Copy the current element to the next position in the array
                nums[j] = nums[i];
                // Move to the next position for the next unique element
                j++;
            }
        }
        // Return the length of the resulting array
        return j;
    }
}
