class Solution {

    public static int method1(int[] nums) {
        int j = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[j] && j < i) {
                j++;
                Swapping(nums, i, j);
            }
        }
        return j + 1;
    }

    public static int[] Swapping(int[] arr, int i, int j) {
        int temp = arr[j];
        arr[j] = arr[i];
        arr[i] = temp;
        return arr;
    }
}
