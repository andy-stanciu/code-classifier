class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        int lim = (int) Math.floor((double) nums.length / 3.0);
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i : nums) {
            hm.put(i, hm.getOrDefault(i, 0) + 1);
        }
        for (int keys : hm.keySet()) {
            if (hm.get(keys) > lim) {
                ans.add(keys);
            }
        }
        return ans;
    }
}
