class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        for (int x : nums) {
            map.put(x, map.getOrDefault(x, 0) + 1);
        }
        List<Integer> ans = new ArrayList<Integer>();
        for (int x : map.keySet()) {
            if (map.get(x) > n / 3)
                ans.add(x);
        }
        return ans;
    }
}
