class Solution {

    public List<Integer> method1(int[] nums) {
        // Initialize an empty list to store the majority elements
        List<Integer> var1 = new ArrayList<>();
        // Iterate through the input array
        for (int i = 0; i < nums.length; i++) {
            // Check if the current element is already in the list
            boolean isFound = false;
            for (int j = 0; j < var1.size(); j++) {
                if (nums[i] == var1.get(j)) {
                    isFound = true;
                    break;
                }
            }
            // If the element is not found, count its occurrences
            if (!isFound) {
                int count = 1;
                for (int j = 0; j < nums.length; j++) {
                    if (j != i && nums[j] == nums[i]) {
                        count++;
                    }
                }
                // If the count is more than n/3, add the element to the list
                if (count > nums.length / 3) {
                    var1.add(nums[i]);
                }
            }
        }
        return var1;
    }
}
