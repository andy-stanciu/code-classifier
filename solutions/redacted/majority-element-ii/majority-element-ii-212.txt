class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        int count1 = 0, count2 = 0;
        int var1 = 0, var2 = 0;
        List<Integer> result = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (count1 == 0 && nums[i] != var2) {
                count1++;
                var1 = nums[i];
            } else if (count2 == 0 && nums[i] != var1) {
                count2++;
                var2 = nums[i];
            } else if (nums[i] == var1) {
                count1++;
            } else if (nums[i] == var2) {
                count2++;
            } else {
                count1--;
                count2--;
            }
        }
        count1 = 0;
        count2 = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == var1)
                count1++;
            if (nums[i] == var2)
                count2++;
        }
        if (count1 > n / 3 && !result.contains(var1))
            result.add(var1);
        if (count2 > n / 3 && !result.contains(var2))
            result.add(var2);
        return result;
    }
}
