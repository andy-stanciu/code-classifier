class Solution {

    public List<Integer> method1(int[] nums) {
        int var1 = 0;
        int var2 = 0;
        int count1 = 0;
        int count2 = 0;
        List<Integer> list = new ArrayList<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (var1 == nums[i]) {
                count1++;
            } else if (var2 == nums[i]) {
                count2++;
            } else if (count1 == 0) {
                var1 = nums[i];
                count1 = 1;
            } else if (count2 == 0) {
                var2 = nums[i];
                count2 = 1;
            } else {
                count1--;
                count2--;
            }
        }
        int freq1 = 0;
        int freq2 = 0;
        for (int num : nums) {
            if (num == var1) {
                freq1++;
            } else if (num == var2) {
                freq2++;
            }
        }
        if (freq1 > (n / 3)) {
            list.add(var1);
        }
        if (freq2 > (n / 3)) {
            list.add(var2);
        }
        return list;
    }
}
