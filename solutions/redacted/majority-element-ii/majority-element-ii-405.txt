class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Integer> map = new HashMap<>();
        List<Integer> ans = new ArrayList<>();
        for (int v : nums) {
            map.put(v, map.getOrDefault(v, 0) + 1);
            if (map.get(v) > n / 3) {
                if (!ans.contains(v))
                    ans.add(v);
            }
        }
        return ans;
    }
}
