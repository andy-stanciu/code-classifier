class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> var1 = new ArrayList<>();
        for (int num : nums) {
            var1.add(num);
        }
        Map<Integer, Long> distinct = var1.stream().collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        Set<Integer> finalValues = var1.stream().filter(number -> distinct.get(number) > var1.size() / 3).collect(Collectors.toSet());
        return new ArrayList<>(finalValues);
    }
}
