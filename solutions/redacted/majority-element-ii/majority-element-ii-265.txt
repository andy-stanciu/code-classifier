class Solution {

    public List<Integer> method1(int[] nums) {
        ArrayList<Integer> ans = new ArrayList<>();
        int n = nums.length;
        int th = n / 3;
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int a : nums) {
            map.put(a, map.getOrDefault(a, 0) + 1);
        }
        for (int k : map.keySet()) {
            int val = map.get(k);
            if (val > th) {
                ans.add(k);
            }
        }
        return ans;
    }
}
