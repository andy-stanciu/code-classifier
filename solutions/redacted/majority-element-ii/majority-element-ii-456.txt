class Solution {

    public ArrayList<Integer> method1(int[] nums) {
        // Declare a new ArrayList to return values, as return type of function is List
        ArrayList<Integer> lst = new ArrayList<Integer>();
        // Create first pointer with variable name strt, initialize with 0
        int strt = 0;
        // Sort the Array nums
        Arrays.sort(nums);
        //*Loop through all elements of nums (starting from second element< so that index i-1 doesn\'t go out of bounds) *//
        for (int i = 1; i < nums.length; i++) {
            // Identify indexes where new number (i.e. the number which is not repeated), appears
            if (nums[i] != nums[i - 1]) {
                // Check if current subsequence qualifies the condition, if yes then add to list
                if (i - strt > nums.length / 3) {
                    lst.add(nums[i - 1]);
                }
                // Reset value of strt to i
                strt = i;
            }
        }
        // Check if last subsequence qualifies the condition, if yes then add to list
        if (nums.length - strt > nums.length / 3) {
            lst.add(nums[nums.length - 1]);
        }
        // Return the list
        return lst;
    }
}
