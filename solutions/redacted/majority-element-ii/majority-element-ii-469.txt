class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        int c1 = 0;
        int c2 = 0;
        int var1 = Integer.MIN_VALUE;
        int var2 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (c1 == 0 && var2 != nums[i]) {
                c1 = 1;
                var1 = nums[i];
            } else if (c2 == 0 && var1 != nums[i]) {
                c2 = 1;
                var2 = nums[i];
            } else if (var1 == nums[i]) {
                c1++;
            } else if (var2 == nums[i]) {
                c2++;
            } else {
                c1--;
                c2--;
            }
        }
        int count1 = 0;
        int count2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (var1 == nums[i]) {
                count1++;
            }
            if (var2 == nums[i]) {
                count2++;
            }
        }
        int min = (int) (nums.length / 3) + 1;
        if (count1 >= min) {
            ans.add(var1);
        }
        if (count2 >= min) {
            ans.add(var2);
        }
        return ans;
    }
}
