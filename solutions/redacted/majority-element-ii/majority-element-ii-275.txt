class Solution {

    public List<Integer> method1(int[] nums) {
        Arrays.sort(nums);
        List<Integer> li = new ArrayList<>();
        int i = 0, j = 0;
        while (j < nums.length) {
            int x = nums[j];
            int c = 0;
            while (j < nums.length && nums[j] == x) {
                c++;
                j++;
            }
            if (c > (nums.length / 3)) {
                li.add(x);
            }
            i = j;
        }
        return li;
    }
}
