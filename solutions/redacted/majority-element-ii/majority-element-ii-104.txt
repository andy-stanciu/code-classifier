class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> al = new ArrayList<>();
        int n = nums.length;
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            if (e.getValue() > n / 3) {
                al.add(e.getKey());
            }
        }
        return al;
    }
}
