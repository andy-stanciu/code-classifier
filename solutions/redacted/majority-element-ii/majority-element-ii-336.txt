class Solution {

    public List<Integer> method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int tms = (n < 3) ? 1 : (n / 3) + 1;
        int count = 1;
        List<Integer> list = new ArrayList<>();
        if (n < 3) {
            for (int i = 0; i < n; i++) {
                if (!list.contains(nums[i])) {
                    list.add(nums[i]);
                }
            }
        } else {
            for (int i = 1; i <= n - 1; i++) {
                if (nums[i - 1] == nums[i]) {
                    count++;
                } else {
                    count = 1;
                }
                if (count >= tms && !list.contains(nums[i])) {
                    list.add(nums[i]);
                }
            }
        }
        return list;
    }
}
