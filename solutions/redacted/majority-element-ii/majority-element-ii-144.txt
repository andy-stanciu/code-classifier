class Solution {

    public List<Integer> method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        int n = nums.length / 3;
        Set<Integer> keys = map.keySet();
        ArrayList<Integer> list = new ArrayList<>();
        for (int key : keys) {
            if (map.get(key) > n)
                list.add(key);
        }
        return list;
    }
}
