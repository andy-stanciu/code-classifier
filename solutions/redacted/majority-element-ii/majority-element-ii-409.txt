class Solution {

    public List<Integer> method1(int[] a) {
        List<Integer> ls = new ArrayList<>();
        int el1 = Integer.MIN_VALUE;
        int el2 = Integer.MIN_VALUE;
        int cnt1 = 0;
        int cnt2 = 0;
        for (int i = 0; i < a.length; i++) {
            if (cnt1 == 0 && a[i] != el2) {
                el1 = a[i];
                cnt1 = 1;
            } else if (cnt2 == 0 && a[i] != el1) {
                el2 = a[i];
                cnt2 = 1;
            } else if (el1 == a[i])
                cnt1++;
            else if (el2 == a[i])
                cnt2++;
            else {
                cnt1--;
                cnt2--;
            }
        }
        cnt1 = 0;
        cnt2 = 0;
        for (int i = 0; i < a.length; i++) {
            if (a[i] == el1)
                cnt1++;
            if (a[i] == el2)
                cnt2++;
        }
        if (cnt1 > a.length / 3)
            ls.add(el1);
        if (cnt2 > a.length / 3)
            ls.add(el2);
        return ls;
    }
}
