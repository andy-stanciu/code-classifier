class Solution {

    public List<Integer> method1(int[] nums) {
        // Sort the array to potentially group majority elements together
        Arrays.sort(nums);
        List<Integer> list = new ArrayList<>();
        int i = 0;
        int j = 0;
        // Loop through the sorted array
        while (j < nums.length) {
            // Current element
            int x = nums[i];
            // Count of current element
            int count = 0;
            // Count occurrences of the current element
            while (j < nums.length && nums[j] == x) {
                count++;
                j++;
            }
            // Check if count is greater than N/3 (potential majority element)
            if (count > nums.length / 3) {
                // Add to result list
                list.add(x);
            }
            // Move to the next unique element
            i = j;
        }
        return list;
    }
}
