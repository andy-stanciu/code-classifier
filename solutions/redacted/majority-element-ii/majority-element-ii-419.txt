class Solution {

    public List<Integer> method1(int[] nums) {
        double s = nums.length / 3;
        Map<Integer, Integer> valueCountMap = new HashMap<>();
        for (int value : nums) {
            valueCountMap.put(value, valueCountMap.getOrDefault(value, 0) + 1);
        }
        List<Integer> ans = new ArrayList<>();
        for (Map.Entry<Integer, Integer> entry : valueCountMap.entrySet()) {
            if ((double) entry.getValue() > s) {
                ans.add(entry.getKey());
            }
        }
        return ans;
    }
}
