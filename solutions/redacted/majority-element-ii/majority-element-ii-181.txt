class Solution {

    public List<Integer> method1(int[] nums) {
        Map<Integer, Integer> m = new HashMap<>();
        int n = nums.length / 3;
        Set<Integer> l = new HashSet<>();
        for (int i : nums) {
            m.put(i, m.getOrDefault(i, 0) + 1);
            if (m.get(i) > n) {
                l.add(i);
            }
        }
        return new ArrayList(l);
    }
}
