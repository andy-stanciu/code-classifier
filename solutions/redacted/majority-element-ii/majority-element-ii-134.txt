class Solution {

    public List<Integer> method1(int[] nums) {
        ArrayList<Integer> result = new ArrayList<>();
        HashMap<Integer, Integer> store = new HashMap<>();
        int n = nums.length;
        int count1 = 0;
        int count2 = 0;
        int el1 = 0;
        int el2 = 0;
        for (int i = 0; i < n; i++) {
            if (count1 == 0 && nums[i] != el2) {
                count1 = 1;
                el1 = nums[i];
            } else if (count2 == 0 && nums[i] != el1) {
                count2 = 1;
                el2 = nums[i];
            } else if (el1 == nums[i]) {
                count1++;
            } else if (el2 == nums[i]) {
                count2++;
            } else {
                count1--;
                count2--;
            }
        }
        int count3 = 0;
        int count4 = 0;
        for (int i = 0; i < n; i++) {
            if (el1 == nums[i]) {
                count3++;
            } else if (el2 == nums[i]) {
                count4++;
            }
        }
        if (count3 > n / 3) {
            result.add(el1);
        }
        if (count4 > n / 3) {
            result.add(el2);
        }
        return result;
    }
}
