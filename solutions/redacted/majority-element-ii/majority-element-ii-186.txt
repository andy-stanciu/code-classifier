class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> list = new ArrayList<>();
        int n = nums.length;
        int s = n / 3;
        // Check each element
        for (int i = 0; i < n; i++) {
            // Reset the count for each number
            int c = 0;
            // Count occurrences of nums[i]
            for (int j = 0; j < n; j++) {
                if (nums[i] == nums[j]) {
                    c++;
                }
            }
            // If the count is greater than n/3 and it\'s not already in the list, add it
            if (c > s && !list.contains(nums[i])) {
                list.add(nums[i]);
            }
        }
        // Return the list of majority elements
        return list;
    }
}
