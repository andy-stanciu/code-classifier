class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> list = new ArrayList<>();
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int e : nums) {
            map.put(e, map.getOrDefault(e, 0) + 1);
        }
        int a = nums.length / 3;
        for (int key : map.keySet()) {
            if (map.get(key) > a)
                list.add(key);
        }
        return list;
    }
}
