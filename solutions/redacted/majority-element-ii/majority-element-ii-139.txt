class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        List<Integer> ans = new ArrayList<>();
        Map<Integer, Integer> mpp = new HashMap<>();
        for (int i = 0; i < n; i++) {
            mpp.put(nums[i], mpp.getOrDefault(nums[i], 0) + 1);
            if (mpp.get(nums[i]) == n / 3 + 1)
                ans.add(nums[i]);
            if (ans.size() == 2)
                break;
        }
        return ans;
    }
}
