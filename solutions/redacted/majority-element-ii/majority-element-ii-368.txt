class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        Arrays.sort(nums);
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                // cheack the elememets with previous one
                count++;
            } else {
                if (count > nums.length / 3) {
                    // the previous elements not same for the current
                    ans.add(nums[i - 1]);
                }
                count = 1;
            }
        }
        if (count > nums.length / 3) {
            // If it exceeds n/3, add it to the result list
            ans.add(nums[nums.length - 1]);
        }
        // Return the result list
        return ans;
    }
}
