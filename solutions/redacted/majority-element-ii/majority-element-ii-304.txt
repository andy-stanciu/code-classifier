class Solution {

    public List<Integer> method1(int[] nums) {
        int n = nums.length;
        Map<Integer, Integer> mp = new HashMap<>();
        for (int x : nums) {
            mp.put(x, mp.getOrDefault(x, 0) + 1);
        }
        List<Integer> res = mp.entrySet().stream().filter(o -> o.getValue() > n / 3).map(Map.Entry::getKey).collect(Collectors.toList());
        return res;
    }
}
