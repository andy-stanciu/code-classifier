class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> result = new ArrayList<>();
        Arrays.sort(nums);
        int boundary = nums.length / 3;
        for (int i = 0; i < nums.length; i++) {
            int temp = 1;
            while (i < nums.length - 1 && nums[i] == nums[i + 1]) {
                temp++;
                i++;
            }
            if (temp > boundary)
                result.add(nums[i]);
        }
        return result;
    }
}
