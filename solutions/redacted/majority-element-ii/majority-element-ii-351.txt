class Solution {

    public ArrayList method1(int[] nums) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        int n = nums.length;
        ArrayList<Integer> al = new ArrayList<Integer>();
        for (int num : nums) {
            hm.put(num, hm.getOrDefault(num, 0) + 1);
        }
        int var1 = n / 3;
        for (int key : hm.keySet()) {
            if (hm.get(key) > var1) {
                al.add(key);
            }
        }
        return al;
    }
}
