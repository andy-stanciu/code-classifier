class Solution {

    public List<Integer> method1(int[] nums) {
        Arrays.sort(nums);
        int count = 1;
        List<Integer> res = new ArrayList<>();
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                count++;
            } else {
                if (count > nums.length / 3) {
                    res.add(nums[i - 1]);
                }
                count = 1;
            }
        }
        if (count > nums.length / 3)
            res.add(nums[nums.length - 1]);
        return res;
    }
}
