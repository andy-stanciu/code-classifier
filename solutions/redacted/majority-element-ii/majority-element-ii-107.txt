class Solution {

    public List<Integer> method1(int[] nums) {
        List<Integer> result = new ArrayList<>();
        int n = nums.length;
        int threshold = n / 3;
        // Step 1: Sort the array
        Arrays.sort(nums);
        // Step 2: Count occurrences of each element
        // To count occurrences of current element
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (nums[i] == nums[i - 1]) {
                // Increment count if it\'s the same element as the previous one
                count++;
            } else {
                // Check if the previous element count exceeds the threshold
                if (count > threshold) {
                    result.add(nums[i - 1]);
                }
                // Reset count for the new element
                count = 1;
            }
        }
        // Check the last element\'s count after the loop
        if (count > threshold) {
            result.add(nums[n - 1]);
        }
        return result;
    }
}
