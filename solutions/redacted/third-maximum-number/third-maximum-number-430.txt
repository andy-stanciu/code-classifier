class Solution {

    public int method1(int[] nums) {
        int c = 0;
        Arrays.sort(nums);
        int max = nums[nums.length - 1];
        for (int i = nums.length - 1 - 1; i >= 0; i--) {
            if (max > nums[i]) {
                if (nums[i] != nums[i + 1])
                    c++;
                else
                    continue;
            }
            if (c > 1)
                return nums[i];
        }
        return max;
    }
}
