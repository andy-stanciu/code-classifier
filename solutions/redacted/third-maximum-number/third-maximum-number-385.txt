class Solution {

    public int method1(int[] nums) {
        if (nums.length < 3) {
            Arrays.sort(nums);
            return nums[nums.length - 1];
        }
        long max1, max2, max3;
        max1 = max2 = max3 = Long.MIN_VALUE;
        max1 = Max(max1, max2, nums);
        max2 = Max(max1, max2, nums);
        max3 = Max(max1, max2, nums);
        if (max3 == Long.MIN_VALUE) {
            return (int) max1;
        }
        return (int) max3;
    }

    public long Max(long max1, long max2, int[] arr) {
        long max = Long.MIN_VALUE;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == max1 || arr[i] == max2) {
                continue;
            } else if (arr[i] > max) {
                max = arr[i];
            }
        }
        return max;
    }
}
