class Solution {

    public int method1(int[] nums) {
        PriorityQueue<Integer> pq = new PriorityQueue<Integer>();
        int max = Integer.MIN_VALUE;
        for (int n : nums) {
            if (!pq.contains(n))
                pq.offer(n);
            if (pq.size() > 3)
                pq.poll();
            max = Math.max(max, n);
        }
        return pq.size() < 3 ? max : pq.peek();
    }
}
