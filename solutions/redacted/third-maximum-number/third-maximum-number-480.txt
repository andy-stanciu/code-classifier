class Solution {

    public int method1(int[] nums) {
        Set<Integer> s = new TreeSet<>(Arrays.stream(nums).boxed().toList());
        int n = s.size() - 3;
        if (n < 0) {
            return Arrays.stream(nums).max().getAsInt();
        } else {
            int res = 0;
            int i = 0;
            for (Integer z : s) {
                if (n == i) {
                    res = z;
                }
                i++;
            }
            return res;
        }
    }
}
