class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        //1 2 3
        int count = 1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                count++;
            }
            if (count == 3) {
                return nums[i - 1];
            }
        }
        return nums[nums.length - 1];
    }
}
