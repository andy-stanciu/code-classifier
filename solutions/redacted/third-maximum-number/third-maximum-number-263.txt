class Solution {

    public int method1(int[] nums) {
        //TreeSet to ensure distinct elements
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) {
            set.add(num);
            // Remove the smallest number if the size exceeds 3
            if (set.size() > 3)
                set.pollFirst();
        }
        //If size is less than 3
        if (set.size() < 3)
            return set.last();
        return set.first();
    }
}
