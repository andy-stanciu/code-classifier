class Solution {

    public int method1(int[] nums) {
        Set<Integer> s = new TreeSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            s.add(nums[i]);
        }
        List<Integer> l = new ArrayList<Integer>(s);
        if (l.size() < 3)
            return (Collections.max(l));
        else
            return (l.get(l.size() - 3));
    }
}
