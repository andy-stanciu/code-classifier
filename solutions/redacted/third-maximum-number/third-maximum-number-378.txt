class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        //ascending sort
        Arrays.sort(nums);
        //reverse the order of the sorted array to descending
        for (int i = 0; i < n / 2; i++) {
            int temp = nums[i];
            nums[i] = nums[n - 1 - i];
            nums[n - 1 - i] = temp;
        }
        int index = 0;
        int dup = 0;
        for (int i = 1; i < n; i++) {
            index++;
            if (nums[i] == nums[i - 1])
                dup++;
            if (index - dup == 2)
                return nums[index];
        }
        return nums[0];
    }
}
