class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int[] arr = Arrays.stream(nums).distinct().toArray();
        Arrays.sort(arr);
        int max1 = arr[0];
        int max2 = arr[0];
        int max3 = arr[0];
        if (arr.length == 1) {
            return arr[0];
        } else if (arr.length == 2) {
            return arr[1];
        }
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] > max1) {
                max3 = max2;
                max2 = max1;
                max1 = arr[i];
            } else if (arr[i] > max2) {
                max3 = max2;
                max2 = arr[i];
            } else if (arr[i] > max3) {
                max3 = arr[i];
            }
        }
        return max3;
    }
}
