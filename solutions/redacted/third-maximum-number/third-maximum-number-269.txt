class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int var1 = 1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                var1++;
            }
            if (var1 == 3) {
                return nums[i - 1];
            }
        }
        if (var1 == 2) {
            return nums[nums.length - 1];
        } else
            return nums[var1 - 1];
    }
}
