class Solution {

    public static int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; ++i) {
            set.add(nums[i]);
        }
        return (set.size() >= 3) ? set.stream().sorted().collect(Collectors.toList()).get(set.size() - 3) : set.stream().sorted().collect(Collectors.toList()).get(set.size() - 1);
    }
}
