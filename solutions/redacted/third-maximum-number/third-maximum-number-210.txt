class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int k = 0;
        if (nums.length < 3)
            return nums[nums.length - 1];
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] == nums[i - 1]) {
            } else
                k++;
            if (k == 2)
                return nums[i - 1];
        }
        return nums[nums.length - 1];
    }
}
