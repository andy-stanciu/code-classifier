class Solution {

    public int method1(int[] nums) {
        long max = Long.MIN_VALUE;
        long smax = Long.MIN_VALUE;
        long tmax = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == max || nums[i] == smax || nums[i] == tmax) {
                // Skip duplicates of max, smax, and tmax
                continue;
            }
            if (nums[i] > max) {
                tmax = smax;
                smax = max;
                max = nums[i];
            } else if (nums[i] > smax) {
                tmax = smax;
                smax = nums[i];
            } else if (nums[i] > tmax) {
                tmax = nums[i];
            }
        }
        // Check if there are fewer than 3 distinct maximum values
        if (tmax == Long.MIN_VALUE) {
            return (int) max;
        }
        return (int) tmax;
    }
}
