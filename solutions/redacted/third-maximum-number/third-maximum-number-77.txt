class Solution {

    public int method1(int[] nums) {
        int l = nums.length;
        // Sort the array in descending order
        for (int i = 0; i < l - 1; i++) {
            for (int j = 0; j < l - i - 1; j++) {
                if (nums[j] < nums[j + 1]) {
                    // Swap nums[j] and nums[j + 1]
                    int temp = nums[j];
                    nums[j] = nums[j + 1];
                    nums[j + 1] = temp;
                }
            }
        }
        // Use a set to track unique elements
        int distinctCount = 0;
        int lastMax = Integer.MIN_VALUE;
        for (int num : nums) {
            if (num != lastMax) {
                distinctCount++;
                lastMax = num;
            }
            if (distinctCount == 3) {
                // Return the third distinct maximum
                return num;
            }
        }
        // If there are less than 3 distinct numbers, return the maximum
        // Return the maximum number
        return nums[0];
    }
}
