class Solution {

    public int method1(int[] nums) {
        // Convert nums array into a set to remove duplicates
        Set<Integer> hashSet = new HashSet<Integer>();
        for (int num : nums) {
            hashSet.add(num);
        }
        // Convert set into list to sorting
        List<Integer> list = new ArrayList<Integer>(hashSet);
        Collections.sort(list, Collections.reverseOrder());
        // if size is equal to or bigger than 3 then return 2 index of sorted list
        if (list.size() >= 3) {
            return list.get(2);
        }
        // otherwise return 0th index element.
        return list.get(0);
    }
}
