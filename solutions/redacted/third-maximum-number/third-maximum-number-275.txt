class Solution {

    public int method1(int[] nums) {
        long m1 = Long.MIN_VALUE;
        long m2 = m1, m3 = m1;
        for (int num : nums) {
            if (num > m1) {
                m3 = m2;
                m2 = m1;
                m1 = num;
            } else if (num > m2 && num < m1) {
                m3 = m2;
                m2 = num;
            } else if (num > m3 && num < m2) {
                m3 = num;
            }
        }
        return m3 == Long.MIN_VALUE ? (int) m1 : (int) m3;
    }
}
