class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        if (nums.length == 2) {
            int[] ans = new int[3];
            ans[0] = nums[nums.length - 1];
            ans[1] = nums[nums.length - 1];
            ans[2] = nums[nums.length - 1];
            for (int i = 0; i < nums.length - 1; i++) {
                if (nums[i] > ans[2]) {
                    int temp = ans[2];
                    ans[2] = nums[i];
                    ans[0] = ans[1];
                    ans[1] = temp;
                }
            }
            return ans[0];
        }
        int[] ans = new int[3];
        ans[0] = nums[0];
        ans[1] = nums[0];
        ans[2] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] > ans[2]) {
                int temp = ans[2];
                ans[2] = nums[i];
                ans[0] = ans[1];
                ans[1] = temp;
            }
        }
        if (ans[0] == ans[1])
            return ans[2];
        return ans[0];
    }
}
