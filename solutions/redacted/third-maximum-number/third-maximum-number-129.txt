class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int firstindex = 0;
        int[] newarray = new int[nums.length];
        newarray[firstindex++] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                newarray[firstindex++] = nums[i];
            }
        }
        if (firstindex < 3) {
            return newarray[firstindex - 1];
        } else {
            return newarray[firstindex - 3];
        }
    }
}
