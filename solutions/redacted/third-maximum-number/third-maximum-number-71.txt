class Solution {

    public int method1(int[] nums) {
        int max = nums[0];
        //this below loop will find the max element from the array
        for (int i = 1; i < nums.length; i++) {
            if (max < nums[i])
                max = nums[i];
        }
        //this will return the max value if the array size is less then 3
        if (nums.length < 3)
            return max;
        TreeSet<Integer> hs = new TreeSet<>();
        //this will sort the array and arrange the elements in order then we can directly print the elements through its array
        for (int i = 0; i < nums.length; i++) hs.add(nums[i]);
        ArrayList<Integer> as = new ArrayList<>(hs);
        if (as.size() < 3)
            return as.get(as.size() - 1);
        return as.get(as.size() - 3);
    }
}
