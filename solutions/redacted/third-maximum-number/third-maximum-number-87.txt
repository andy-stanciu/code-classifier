class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        Set<Integer> li = new HashSet<Integer>();
        for (int i = 0; i < nums.length; i++) {
            li.add(nums[i]);
        }
        List<Integer> res = new ArrayList<Integer>(li);
        Collections.sort(res, Collections.reverseOrder());
        int len = res.size();
        if (len < 3) {
            return res.get(0);
        } else {
            return res.get(2);
        }
    }
}
