class Solution {

    public int method1(int[] nums) {
        // Sort the array
        Arrays.sort(nums);
        // Start with the first distinct number
        int distinctCount = 1;
        // Start with the maximum number
        int currentMax = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            // Start from the second last element
            if (nums[i] != currentMax) {
                distinctCount++;
                currentMax = nums[i];
            }
            if (distinctCount == 3) {
                // We need the third distinct maximum
                return nums[i];
            }
        }
        // If there are less than 3 distinct numbers, return the maximum
        return nums[nums.length - 1];
    }
}
