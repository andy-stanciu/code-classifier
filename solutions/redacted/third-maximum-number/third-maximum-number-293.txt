class Solution {

    public int method1(int[] nums) {
        //TreeSet removes any duplicate if exist
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) {
            //This line adds the current element (num) to the TreeSet.
            set.add(num);
            //  Since TreeSet does not allow duplicates, if the element is already present in the set, it won\'t be added again.
            if (set.size() > 3) {
                set.remove(set.first());
            }
        }
        return set.size() < 3 ? set.last() : set.first();
    }
}
