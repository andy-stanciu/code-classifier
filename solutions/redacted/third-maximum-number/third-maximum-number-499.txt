class Solution {

    public int method1(int[] nums) {
        Integer firstMax = null;
        Integer secondMax = null;
        Integer method1 = null;
        for (Integer num : nums) {
            if (num.equals(firstMax) || num.equals(secondMax) || num.equals(method1)) {
                // Skip duplicates
                continue;
            }
            if (firstMax == null || num > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = num;
            } else if (secondMax == null || num > secondMax) {
                method1 = secondMax;
                secondMax = num;
            } else if (method1 == null || num > method1) {
                method1 = num;
            }
        }
        return method1 != null ? method1 : firstMax;
    }
}
