class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int count = 1;
        int flag = 0;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] == nums[i - 1]) {
                // for checking how many times max or second max value repeating
                flag = flag + 1;
            } else {
                // for getting distinct max value count (until 3rd max value)
                count = count + 1;
            }
            if (count == 3) {
                // n=total value , count = 3, flag = how many times max or second max repeating
                return nums[n - count - flag];
            }
        }
        return nums[n - 1];
    }
}
