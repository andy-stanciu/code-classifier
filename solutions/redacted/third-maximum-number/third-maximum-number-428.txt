class Solution {

    public int method1(int[] nums) {
        int first = Integer.MIN_VALUE;
        int second = Integer.MIN_VALUE;
        int var1 = Integer.MIN_VALUE;
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        for (int num : nums) {
            if (num > first) {
                var1 = second;
                second = first;
                first = num;
            } else if (num > second && num < first) {
                var1 = second;
                second = num;
            } else if (num > var1 && num < second) {
                var1 = num;
            }
        }
        if (var1 == Integer.MIN_VALUE && set.size() == 3) {
            return Integer.MIN_VALUE;
        }
        return (var1 != Integer.MIN_VALUE) && set.size() > 2 ? var1 : first;
    }
}
