class Solution {

    public int method1(int[] nums) {
        long firstMax = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int ele : nums) {
            if (ele > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = ele;
            } else if (ele > secondMax && firstMax != ele) {
                method1 = secondMax;
                secondMax = ele;
            } else if (ele > method1 && secondMax != ele && firstMax != ele) {
                method1 = ele;
            }
        }
        if (method1 == Long.MIN_VALUE)
            return (int) firstMax;
        else
            return (int) method1;
    }
}
