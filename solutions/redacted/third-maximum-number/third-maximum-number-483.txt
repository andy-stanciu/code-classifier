class Solution {

    public int method1(int[] nums) {
        int size = nums.length - 1;
        Arrays.sort(nums);
        HashSet<Integer> ans = new HashSet<>();
        for (int i = size; i >= 0; i--) {
            ans.add(nums[i]);
            if (ans.size() == 3) {
                return nums[i];
            }
        }
        return nums[size];
    }
}
