class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        Integer max1 = nums[nums.length - 1];
        Integer max2 = null;
        Integer max3 = null;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] != max1 && max2 == null) {
                max2 = nums[i];
            } else if (max2 != null && nums[i] != max2) {
                return nums[i];
            }
        }
        return max1;
    }
}
