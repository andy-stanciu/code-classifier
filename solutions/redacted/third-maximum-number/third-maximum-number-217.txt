class Solution {

    public int method1(int[] nums) {
        long firstMax = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = nums[i];
            } else if (nums[i] > secondMax && nums[i] < firstMax) {
                method1 = secondMax;
                secondMax = nums[i];
            } else if (nums[i] > method1 && nums[i] < secondMax && nums[i] < firstMax) {
                method1 = nums[i];
            }
        }
        return (method1 == Long.MIN_VALUE ? (int) firstMax : (int) method1);
    }
}
