class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> ans = new HashSet<>();
        for (int num : nums) {
            ans.add(num);
        }
        int[] arr = new int[ans.size()];
        int i = 0;
        for (int num : ans) {
            arr[i++] = num;
        }
        Arrays.sort(arr);
        if (arr.length == 3) {
            return arr[0];
        } else if (arr.length > 3) {
            return arr[arr.length - 3];
        }
        return arr[arr.length - 1];
    }
}
