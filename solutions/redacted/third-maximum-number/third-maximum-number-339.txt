class Solution {

    public int method1(int[] nums) {
        int firstMax = Integer.MIN_VALUE;
        int secMax = Integer.MIN_VALUE;
        int method1 = Integer.MIN_VALUE;
        boolean is = false;
        for (int num : nums) {
            if (Integer.MIN_VALUE == num) {
                is = true;
            }
            if (firstMax == num || secMax == num) {
                continue;
            }
            if (firstMax < num) {
                method1 = secMax;
                secMax = firstMax;
                // 2
                firstMax = num;
            } else if (secMax < num) {
                method1 = secMax;
                secMax = num;
            } else if (method1 < num) {
                method1 = num;
            }
        }
        // To handle Integer.MIN_VALUE
        if (method1 != secMax & is || method1 != Integer.MIN_VALUE) {
            return method1;
        }
        return firstMax;
    }
}
