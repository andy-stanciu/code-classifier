class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new TreeSet<>();
        for (int n : nums) {
            set.add(n);
        }
        int size = set.size();
        int[] arr = new int[size];
        int j = 0;
        for (int i : set) {
            arr[j++] = i;
        }
        if (size < 3) {
            return arr[size - 1];
        } else {
            return arr[size - 3];
        }
    }
}
