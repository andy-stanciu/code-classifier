class Solution {

    public int method1(int[] nums) {
        Set<Integer> s = new HashSet<>();
        for (int i : nums) {
            s.add(i);
        }
        int[] a = new int[s.size()];
        int p = 0;
        for (int i : s) {
            a[p++] = i;
        }
        Arrays.sort(a);
        if (a.length <= 2) {
            return a[a.length - 1];
        } else {
            return a[a.length - 3];
        }
    }
}
