class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        // now we have sorted the array now time for the apporach
        // now we can create the count in which if it reaches the count of 3 then we will
        // the current index at which we are
        // c ka value one as in the case of 1 2 3 it is having c = 3 only if c start form 1
        int c = 1;
        int n = nums.length - 1;
        while (n > 0 && c < 3) {
            // c should less than 3 because in some case its\' value get updated  to 4 then it get out of loop so it dose\'nt satistfy the condition so we have to make it less than 3 so it\'s\' value updated max to 3
            if (nums[n] != nums[n - 1]) {
                c++;
            }
            n--;
        }
        // so now if count is 3 then three distnic elemtn are present in the arraay
        if (c == 3) {
            return nums[n];
        }
        // else we are returning the max intiger in the array
        return nums[nums.length - 1];
    }
}
