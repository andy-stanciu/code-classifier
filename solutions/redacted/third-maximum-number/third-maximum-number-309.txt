class Solution {

    public int method1(int[] nums) {
        int first, second, var1;
        first = second = var1 = Integer.MIN_VALUE;
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        Iterator it = set.iterator();
        while (it.hasNext()) {
            int item = (int) it.next();
            if (item > first) {
                if (second != Integer.MIN_VALUE) {
                    var1 = second;
                    second = first;
                    first = item;
                } else {
                    second = first;
                    first = item;
                }
            } else if (item > second) {
                var1 = second;
                second = item;
            } else if (item > var1) {
                var1 = item;
            }
        }
        if (set.size() < 3)
            return first;
        return var1;
    }
}
