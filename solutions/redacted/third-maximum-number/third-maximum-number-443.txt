class Solution {

    public int method1(int[] nums) {
        //taking hashset bcz some of the array might have dups elements
        Set<Integer> set = new HashSet<>();
        for (int i : nums) {
            //adding ele to set
            set.add(i);
        }
        //storing them to list bcz cant directly sort hashset
        List<Integer> list = new ArrayList<>(set);
        // sort in asc
        Collections.sort(list);
        if (list.size() >= 3) {
            return list.get(list.size() - 3);
        } else {
            return list.get(list.size() - 1);
        }
    }
}
