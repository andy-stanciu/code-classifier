class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new TreeSet<Integer>(Collections.reverseOrder());
        int g = Integer.MIN_VALUE;
        for (int i : nums) {
            // for storing non repeated value
            set.add(i);
            //for storing the maximum value
            if (i > g) {
                g = i;
            }
        }
        // if size less than 3 return the maximum number
        if (set.size() < 3) {
            return g;
        }
        g = 0;
        // otherwise count upto 3 and return the number itself
        for (int i : set) {
            if (g == 2) {
                return i;
            }
            g += 1;
        }
        return 0;
    }
}
