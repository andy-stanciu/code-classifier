class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 1)
            return nums[0];
        Integer max1 = null;
        Integer max2 = null;
        Integer max3 = null;
        for (int num : nums) {
            if (max1 == null)
                max1 = num;
            else if (max2 == null) {
                if (num != max1) {
                    max2 = Math.min(num, max1);
                    max1 = Math.max(num, max1);
                }
            } else if (max3 == null) {
                if (num != max1 && num != max2) {
                    if (num > max1) {
                        max3 = max2;
                        max2 = max1;
                        max1 = num;
                    } else if (num > max2) {
                        max3 = max2;
                        max2 = num;
                    } else {
                        max3 = num;
                    }
                }
            } else {
                if (num == max1 || num == max2 || num == max3)
                    continue;
                if (num > max1) {
                    max3 = max2;
                    max2 = max1;
                    max1 = num;
                } else if (num > max2) {
                    max3 = max2;
                    max2 = num;
                } else if (num > max3) {
                    max3 = num;
                }
            }
        }
        if (max3 == null)
            return max1;
        return max3;
    }
}
