class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int size = set.size();
        int[] arr = new int[size];
        int idx = 0;
        for (int element : set) {
            arr[idx] = element;
            idx++;
        }
        Arrays.sort(arr);
        if (size < 3) {
            return arr[size - 1];
        }
        return arr[size - 3];
    }
}
