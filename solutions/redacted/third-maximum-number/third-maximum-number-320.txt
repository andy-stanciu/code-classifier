class Solution {

    public int method1(int[] nums) {
        // Initialize three variables to keep track of the first, second, and third distinct maximum numbers
        Integer firstMax = null;
        Integer secondMax = null;
        Integer method1 = null;
        // Iterate through the array
        for (Integer num : nums) {
            // Update firstMax
            if (firstMax == null || num > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = num;
            } else if (num.equals(firstMax)) {
                // Skip duplicates
                continue;
            } else // Update secondMax
            if (secondMax == null || num > secondMax) {
                method1 = secondMax;
                secondMax = num;
            } else if (num.equals(secondMax)) {
                // Skip duplicates
                continue;
            } else // Update thirdMax
            if (method1 == null || num > method1) {
                method1 = num;
            }
        }
        // Return the third distinct maximum number
        return method1 != null ? method1 : firstMax;
    }
}
