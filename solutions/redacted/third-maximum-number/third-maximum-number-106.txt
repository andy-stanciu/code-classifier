class Solution {

    public int method1(int[] nums) {
        // A set to keep track of unique maximums
        TreeSet<Integer> maxSet = new TreeSet<>();
        // Add elements to the set and maintain the top 3 distinct maximums
        for (int num : nums) {
            maxSet.add(num);
            if (maxSet.size() > 3) {
                // Remove the smallest element if the set exceeds size 3
                maxSet.pollFirst();
            }
        }
        // If the set has 3 elements, return the smallest (third maximum); otherwise, return the largest
        return maxSet.size() == 3 ? maxSet.first() : maxSet.last();
    }
}
