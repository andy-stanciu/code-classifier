class Solution {

    public int method1(int[] nums) {
        Set<Integer> var1 = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            var1.add(nums[i]);
        }
        List<Integer> var2 = new ArrayList<Integer>(var1);
        Collections.sort(var2, Collections.reverseOrder());
        int lengthOfList = var2.size();
        if (lengthOfList <= 2) {
            Integer max = var2.get(0);
            int maxEle = Integer.valueOf(max);
            return maxEle;
        }
        Integer method1 = var2.get(2);
        int var3 = Integer.valueOf(method1);
        return var3;
    }
}
