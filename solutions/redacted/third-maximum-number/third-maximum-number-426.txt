class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        //COUNTING THE NUMBER OF MAXIMUM NUMBERS
        int count = 1;
        // COUNTING THE NUMBERS THAT ARE REPEATED AGAIN
        int flag = 0;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] == nums[i - 1]) {
                flag = flag + 1;
            } else {
                count++;
            }
            if (count == 3) {
                // RETURNING THE NUMS[NUMS.LENGTH-COUNTED MAX NUMBERS - REPEATED MAX NUMBERS]
                return nums[n - count - flag];
            }
        }
        return nums[n - 1];
    }
}
