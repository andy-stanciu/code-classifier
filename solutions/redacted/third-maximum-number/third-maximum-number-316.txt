class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        List<Integer> list = new ArrayList<>();
        for (int num : set) list.add(num);
        Collections.sort(list);
        if (list.size() < 3)
            return list.get(list.size() - 1);
        return list.get(list.size() - 3);
    }
}
