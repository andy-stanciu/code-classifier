class Solution {

    public int method1(int[] nums) {
        int firstMax = Integer.MIN_VALUE;
        int secondMax = Integer.MIN_VALUE;
        int method1 = Integer.MIN_VALUE;
        int numChanges = 0;
        boolean isMinIntPresent = false;
        for (int num : nums) {
            if (num > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = num;
                numChanges++;
            } else if (num > secondMax && num < firstMax) {
                method1 = secondMax;
                secondMax = num;
                numChanges++;
            } else if (num > method1 && num < secondMax) {
                method1 = num;
                numChanges++;
            } else if (!isMinIntPresent && num == Integer.MIN_VALUE && method1 == Integer.MIN_VALUE) {
                isMinIntPresent = true;
                numChanges++;
            }
        }
        int returnValue = method1;
        if (numChanges < 3) {
            returnValue = firstMax;
        }
        return returnValue;
    }
}
