public class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> top3 = new TreeSet<>();
        for (int num : nums) {
            top3.add(num);
            if (top3.size() > 3) {
                // Remove the smallest element
                top3.remove(top3.first());
            }
        }
        if (top3.size() < 3) {
            // If less than 3 distinct maxima, return the maximum element
            return top3.last();
        } else {
            // Otherwise, return the third distinct maximum
            return top3.first();
        }
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        // Example 1
        int[] nums1 = { 3, 2, 1 };
        // Output: 1
        System.out.println(solution.method1(nums1));
        // Example 2
        int[] nums2 = { 1, 2 };
        // Output: 2
        System.out.println(solution.method1(nums2));
        // Example 3
        int[] nums3 = { 2, 2, 3, 1 };
        // Output: 1
        System.out.println(solution.method1(nums3));
    }
}
