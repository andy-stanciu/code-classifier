class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        long firstmax = Long.MIN_VALUE;
        long secondmax = Long.MIN_VALUE;
        long var1 = Long.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] > firstmax) {
                var1 = secondmax;
                secondmax = firstmax;
                firstmax = nums[i];
            } else if (nums[i] > secondmax && nums[i] < firstmax) {
                var1 = secondmax;
                secondmax = nums[i];
            } else if (nums[i] > var1 && nums[i] < secondmax) {
                var1 = nums[i];
            }
        }
        return (int) (var1 == Long.MIN_VALUE ? firstmax : var1);
    }
}
