class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int cnt = 1;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                cnt++;
                if (cnt == 3)
                    return nums[i - 1];
            }
        }
        return nums[n - 1];
        // If there is no third maximum, return the maximum
    }
}
