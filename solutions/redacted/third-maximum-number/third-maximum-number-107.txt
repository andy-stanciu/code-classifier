class Solution {

    public int method1(int[] nums) {
        // to start from the last or highest element
        Arrays.sort(nums);
        int count = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1]) {
                // condition tells that we have encountered the new distinct number
                // so we count++
                count++;
            }
            if (count == 2) {
                // we hit count 2 so the number nuns -1 would be our 3 rd distinct
                return nums[i - 1];
            }
        }
        // maxim number
        return nums[nums.length - 1];
    }
}
