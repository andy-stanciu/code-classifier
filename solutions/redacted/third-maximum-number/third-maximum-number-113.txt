class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) {
            set.add(num);
            if (set.size() > 3) {
                set.pollFirst();
            }
        }
        if (set.size() < 3) {
            // Return the largest element
            return set.last();
        }
        // Otherwise, return the third maximum
        return set.first();
    }
}
