class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new TreeSet<>();
        for (int i = 0; i < nums.length; i++) set.add(nums[i]);
        List<Integer> list = new ArrayList<>(set);
        if (list.size() >= 3)
            return list.get(list.size() - 3);
        else
            return list.get(list.size() - 1);
    }
}
