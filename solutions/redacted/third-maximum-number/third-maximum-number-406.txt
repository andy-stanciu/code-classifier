class Solution {

    public int method1(int[] nums) {
        //sort the array first
        Arrays.sort(nums);
        //if the array size is less than 3 then we just return the largest
        //element which will be at the end of the array since it is sorted
        if (nums.length < 3) {
            return nums[nums.length - 1];
        }
        //I used j as my counter so I know once I got to the 3rd largest element
        int j = 0;
        //used double max value to solve for the edge case where max and min integer values are used
        double min = Double.MAX_VALUE;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (nums[i] < min) {
                min = nums[i];
                j++;
            }
            if (//reached third largest element
            j == 3) {
                return (int) min;
            }
        }
        //if you reach the end of your loop and j still does not equal 3 then
        // it is safe to assume that there is no third largest so just return largest
        return nums[nums.length - 1];
    }
}
