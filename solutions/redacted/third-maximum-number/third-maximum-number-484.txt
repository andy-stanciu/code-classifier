class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int k = 0, n = 0;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] != nums[i - 1])
                ++k;
            if (k == 2)
                return nums[i - 1];
        }
        if (nums.length < 3 || k < 2)
            return (nums[nums.length - 1]);
        return -1;
    }
}
