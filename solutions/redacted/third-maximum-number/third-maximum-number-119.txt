class Solution {

    public int method1(int[] nums) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : nums) set.add(num);
        if (set.size() < 3)
            return set.pollLast();
        for (int i = 0; i < 2; i++) set.pollLast();
        return set.pollLast();
    }
}
