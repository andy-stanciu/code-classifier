class Solution {

    public int method1(int[] nums) {
        long firstMax = Long.MIN_VALUE;
        long secondMax = Long.MIN_VALUE;
        long method1 = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > firstMax) {
                method1 = secondMax;
                secondMax = firstMax;
                firstMax = nums[i];
            } else if (nums[i] > secondMax && nums[i] < firstMax) {
                method1 = secondMax;
                secondMax = nums[i];
            } else if (nums[i] > method1 && nums[i] < secondMax)
                method1 = nums[i];
        }
        if (method1 == Long.MIN_VALUE)
            return (int) firstMax;
        return (int) method1;
    }
}
