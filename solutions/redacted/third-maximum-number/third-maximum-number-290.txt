class Solution {

    public int method1(int[] nums) {
        int first = Integer.MIN_VALUE;
        int second = Integer.MIN_VALUE;
        int var1 = Integer.MIN_VALUE;
        int count = 0;
        for (int i : nums) {
            if (i > first || (count == 0)) {
                count++;
                var1 = count < 3 ? i : second;
                second = first;
                first = i;
                continue;
            }
            if (i != first && ((i > second) || (i == second && count == 1))) {
                count++;
                var1 = count < 3 ? first : second;
                second = i;
                continue;
            }
            if ((i != second && i != first) && (count == 2 || i > var1)) {
                count++;
                var1 = i;
            }
        }
        return var1;
    }
}
