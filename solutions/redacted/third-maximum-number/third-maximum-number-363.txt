class Solution {

    public int method1(int[] nums) {
        int a = 0;
        Arrays.sort(nums);
        int n = nums.length;
        List<Integer> ans = new ArrayList<>();
        ans.add(nums[a]);
        for (int i = 0; i < n; i++) {
            if (ans.get(a) != nums[i]) {
                ans.add(nums[i]);
                a++;
            }
        }
        int n1 = ans.size();
        if (n1 <= 2) {
            return ans.get(n1 - 1);
        }
        return ans.get(n1 - 3);
    }
}
