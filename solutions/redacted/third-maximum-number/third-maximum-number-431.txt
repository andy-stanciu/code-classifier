class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        List<Integer> lint = new ArrayList<>();
        for (int i = nums.length - 1; i >= 0; i--) {
            if (!lint.contains(nums[i])) {
                lint.add(nums[i]);
            }
        }
        if (lint.size() <= 2) {
            return lint.get(0);
        } else {
            return lint.get(2);
        }
    }
}
