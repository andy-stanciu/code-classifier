class Solution {

    public int method1(int[] nums) {
        long highest = Long.MIN_VALUE, secondHighest = Long.MIN_VALUE, var1 = Long.MIN_VALUE;
        boolean method1 = false;
        for (int i = 0; i < nums.length; i++) {
            long num = (long) nums[i];
            if (num > highest) {
                var1 = secondHighest;
                secondHighest = highest;
                highest = num;
            } else if (num != highest && num > secondHighest) {
                var1 = secondHighest;
                secondHighest = num;
            } else if (num != highest && num != secondHighest && num >= var1) {
                var1 = num;
            }
            method1 = true;
        }
        return method1 ? (var1 != Long.MIN_VALUE ? (int) var1 : (int) highest) : -1;
    }
}
