class Solution {

    public int method1(int[] nums) {
        long larg = Long.MIN_VALUE;
        long seclarg = Long.MIN_VALUE;
        long thrdlarg = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > larg) {
                thrdlarg = seclarg;
                seclarg = larg;
                larg = nums[i];
            } else if (nums[i] < larg && nums[i] > seclarg) {
                thrdlarg = seclarg;
                seclarg = nums[i];
            } else if (nums[i] < seclarg && nums[i] > thrdlarg) {
                thrdlarg = nums[i];
            }
        }
        if (thrdlarg == Long.MIN_VALUE) {
            return (int) larg;
        }
        return (int) thrdlarg;
    }
}
