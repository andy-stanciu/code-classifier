class Solution {

    public String method1(String s) {
        // Find the longest odd-length palindrome in the string
        String odd = odd(s);
        // Find the longest even-length palindrome in the string
        String even = even(s);
        // Return the longer of the two palindromes
        return odd.length() > even.length() ? odd : even;
    }

    private String odd(String st) {
        // Initialize the answer to an empty string
        String ans = "";
        // Iterate through each character in the string
        for (int i = 0; i < st.length(); i++) {
            // Initialize a temporary string with the current character
            String temp = Character.toString(st.charAt(i));
            int j = i - 1;
            int k = i + 1;
            // Expand around the center character
            while (j >= 0 && k < st.length()) {
                // Check if the characters at j and k are equal
                if (st.charAt(j) == st.charAt(k)) {
                    // If equal, expand the palindrome
                    temp = Character.toString(st.charAt(j--)) + temp + Character.toString(st.charAt(k++));
                } else {
                    // If not equal, break the loop
                    break;
                }
            }
            // Update the answer if the found palindrome is longer
            if (temp.length() > ans.length())
                ans = temp;
        }
        return ans;
    }

    private String even(String st) {
        // Initialize the answer to an empty string
        String ans = "";
        // Iterate through each character in the string
        for (int i = 0; i < st.length(); i++) {
            // Initialize a temporary string as empty
            String temp = "";
            int j = i;
            int k = i + 1;
            // Expand around the center characters
            while (j >= 0 && k < st.length()) {
                // Check if the characters at j and k are equal
                if (st.charAt(j) == st.charAt(k)) {
                    // If equal, expand the palindrome
                    temp = Character.toString(st.charAt(j--)) + temp + Character.toString(st.charAt(k++));
                } else {
                    // If not equal, break the loop
                    break;
                }
            }
            // Update the answer if the found palindrome is longer
            if (temp.length() > ans.length())
                ans = temp;
        }
        return ans;
    }
}
