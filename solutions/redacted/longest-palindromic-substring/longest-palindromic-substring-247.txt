class Solution {

    int isPalindrome(String word, int left, int right) {
        int l = left, r = right;
        while (left <= right) {
            if (word.charAt(left) == word.charAt(right)) {
                left++;
                right--;
            } else
                return 0;
        }
        return r - l;
    }

    public String method1(String s) {
        int value = 0, max = 0, j;
        String subing;
        int[] num = new int[2];
        for (int i = 0; i < s.length(); i++) {
            for (j = i; j < s.length(); j++) {
                value = isPalindrome(s, i, j);
                if (max < value) {
                    max = value;
                    num[0] = i;
                    num[1] = j;
                }
            }
        }
        subing = s.substring(num[0], num[1] + 1);
        return subing;
    }
}
