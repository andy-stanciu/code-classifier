class Solution {

    public String method1(String s) {
        if (s.length() <= 1) {
            return s;
        }
        String maxLen = s.substring(0, 1);
        for (int i = 0; i < s.length(); i++) {
            String oddLen = checkPalindrome(s, i, i);
            String evenLen = checkPalindrome(s, i, i + 1);
            if (oddLen.length() > maxLen.length()) {
                maxLen = oddLen;
            }
            if (evenLen.length() > maxLen.length()) {
                maxLen = evenLen;
            }
        }
        return maxLen;
    }

    public String checkPalindrome(String s, int left, int right) {
        while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
            left--;
            right++;
        }
        return s.substring(left + 1, right);
    }
}
