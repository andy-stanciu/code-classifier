class Solution {

    public boolean pali(String s) {
        int l = 0;
        int r = s.length() - 1;
        while (l < r) {
            char c = s.charAt(l);
            char c1 = s.charAt(r);
            if (c != c1)
                return false;
            l++;
            r--;
        }
        return true;
    }

    public String method1(String s) {
        //brute force is to check each substring
        String ans = "";
        for (int i = 0; i < s.length(); i++) {
            for (int j = i + 1; j <= s.length(); j++) {
                String cur = s.substring(i, j);
                boolean result = pali(cur);
                if (result && cur.length() > ans.length()) {
                    ans = cur;
                }
            }
        }
        return ans;
    }
}
