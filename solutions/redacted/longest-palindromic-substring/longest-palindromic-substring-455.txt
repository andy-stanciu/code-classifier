class Solution {

    public String method1(String s) {
        int len = s.length();
        int l = 0, r = s.length() - 1, max = 0;
        String maxStr = "";
        while (l + len <= s.length()) {
            String temp = s.substring(l, l + len);
            //System.out.println(temp+" "+len);
            if (isPalindrome(temp)) {
                maxStr = temp;
                break;
            }
            //System.out.println((l+len)+" "+(s.length()-1));
            if (l + len > s.length() - 1) {
                l = 0;
                len--;
            } else {
                l++;
            }
        }
        return maxStr;
    }

    public boolean isPalindrome(String s) {
        int l = 0, r = s.length() - 1;
        while (l <= r) {
            if (s.charAt(l) != s.charAt(r)) {
                return false;
            }
            l++;
            r--;
        }
        return true;
    }
}
