class Solution {

    int max = 0, start = 0, end = 0;

    public boolean isPallendrome(String s, int i, int j) {
        while (i < j) {
            char ch1 = s.charAt(i);
            char ch2 = s.charAt(j);
            if (ch1 != ch2)
                return false;
            i++;
            j--;
        }
        return true;
    }

    public String method1(String s) {
        int i;
        int j;
        int n = s.length();
        for (i = 0; i < n; i++) {
            for (j = i; j < n; j++) {
                if (isPallendrome(s, i, j) == true) {
                    if ((j - i + 1) > max) {
                        max = j - i + 1;
                        start = i;
                        end = j;
                    }
                }
            }
        }
        return s.substring(start, end + 1);
    }
}
