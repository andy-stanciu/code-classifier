class Solution {

    public boolean isPalindrome(String sb) {
        int i = 0;
        int j = sb.length() - 1;
        while (sb.charAt(i) == sb.charAt(j)) {
            i++;
            j--;
            if (i >= j) {
                return true;
            }
        }
        return false;
    }

    public String method1(String s) {
        int maxLen = 0;
        // start index
        int si = 0;
        //last index
        int li = 0;
        for (int i = 0; i < s.length(); i++) {
            for (int j = s.length() - 1; j > 0; j--) {
                int currLen = j - i + 1;
                if (currLen < maxLen) {
                    break;
                }
                if (isPalindrome(s.substring(i, j + 1))) {
                    maxLen = currLen;
                    si = i;
                    li = j;
                }
            }
        }
        return s.substring(si, li + 1);
    }
}
