public class Solution {

    public String method1(String s) {
        if (s == null || s.length() < 1) {
            // Return empty string for null or empty input
            return "";
        }
        int start = 0, end = 0;
        for (int i = 0; i < s.length(); i++) {
            // Check for longest odd-length palindrome (single character center)
            int len1 = expandAroundCenter(s, i, i);
            // Check for longest even-length palindrome (two identical characters center)
            int len2 = expandAroundCenter(s, i, i + 1);
            int len = Math.max(len1, len2);
            // Update start and end indices of the longest palindrome found
            if (len > end - start) {
                start = i - (len - 1) / 2;
                end = i + len / 2;
            }
        }
        return s.substring(start, end + 1);
    }

    private int expandAroundCenter(String s, int left, int right) {
        while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
            left--;
            right++;
        }
        // Length of the palindrome
        return right - left - 1;
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        // Test cases
        // Output: "bab" or "aba"
        System.out.println(solution.method1("babad"));
        // Output: "bb"
        System.out.println(solution.method1("cbbd"));
    }
}
