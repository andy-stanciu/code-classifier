class Solution {

    public int[] expand(String s, int p1, int p2) {
        while (p1 >= 0 && p2 < s.length() && s.charAt(p1) == s.charAt(p2)) {
            p1--;
            p2++;
        }
        return new int[] { p1 + 1, p2 - 1 };
    }

    public String method1(String s) {
        int maxLen = 0, len, a = 0, b = 0;
        int[] ptx = new int[2];
        for (int c = 0; c < s.length(); ++c) {
            ptx = expand(s, c, c);
            len = ptx[1] - ptx[0] + 1;
            if (len > maxLen) {
                maxLen = len;
                a = ptx[0];
                b = ptx[1];
            }
        }
        for (int c = 0; c < s.length(); ++c) {
            ptx = expand(s, c, c + 1);
            len = ptx[1] - ptx[0] + 1;
            if (len > maxLen) {
                maxLen = len;
                a = ptx[0];
                b = ptx[1];
            }
        }
        return s.substring(a, b + 1);
    }
}
