class Solution {

    public String method1(String s) {
        StringBuilder sb = new StringBuilder(s);
        int i = 0, l = sb.length();
        int chances = 1;
        int l1 = 0, l2 = 0;
        while (i < l) {
            int j = 0;
            int m = i, n = i;
            while (j < chances) {
                if (m >= 0 && n < l && s.charAt(m) == s.charAt(n)) {
                    // System.out.println(s.substring(m,n+1));
                    if (l2 - l1 <= n - m) {
                        l1 = m;
                        l2 = n;
                    }
                } else {
                    break;
                }
                m--;
                n++;
                j++;
            }
            if (i <= l / 2)
                chances++;
            else
                chances--;
            i++;
        }
        int x = 0;
        int x1 = 0, x2 = 0;
        while (x < l - 1) {
            int j = 0;
            int m = x, n = x + 1;
            while (j < chances) {
                if (m >= 0 && n < l && s.charAt(m) == s.charAt(n)) {
                    if (x2 - x1 <= n - m) {
                        x1 = m;
                        x2 = n;
                    }
                } else
                    break;
                m--;
                n++;
                j++;
            }
            if (x <= l / 2)
                chances++;
            else
                chances--;
            x++;
        }
        if (l2 - l1 < x2 - x1) {
            ;
        } else {
            x2 = l2;
            x1 = l1;
        }
        return s.substring(x1, x2 + 1);
    }
}
