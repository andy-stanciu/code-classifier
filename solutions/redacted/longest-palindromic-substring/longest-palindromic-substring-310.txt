class Solution {

    public static boolean isPalli(String str) {
        int l = 0;
        int r = str.length() - 1;
        while (l < r) {
            if (str.charAt(l) == str.charAt(r)) {
                l++;
                r--;
            } else {
                return false;
            }
        }
        return true;
    }

    public String method1(String s) {
        String res = "";
        for (int i = 0; i < s.length(); i++) {
            for (int j = i; j < s.length(); j++) {
                String var1 = s.var1(i, j + 1);
                if (isPalli(var1) && var1.length() > res.length()) {
                    res = var1;
                }
            }
        }
        return res;
    }
}
