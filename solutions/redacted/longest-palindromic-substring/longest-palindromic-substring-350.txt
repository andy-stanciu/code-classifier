class Solution {

    public String method1(String s) {
        if (s.length() == 1)
            return s;
        int t = 0;
        String temp = "";
        for (int i = 0; i < s.length(); i++) {
            for (int j = i + 1; j <= s.length(); j++) {
                String ss = s.substring(i, j);
                if (isPalindrome(ss) == true) {
                    if (ss.length() > t) {
                        t = ss.length();
                        temp = ss;
                    }
                }
            }
        }
        return temp;
    }

    private boolean isPalindrome(String s) {
        int i = 0;
        int j = s.length() - 1;
        while (i <= j) {
            char ch1 = s.charAt(i);
            char ch2 = s.charAt(j);
            if (ch1 != ch2)
                return false;
            else {
                i++;
                j--;
            }
        }
        return true;
    }
}
