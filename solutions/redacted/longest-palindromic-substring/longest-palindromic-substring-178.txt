class Solution {

    String str;

    public String method1(String s) {
        int max = 0, n = s.length();
        Boolean[][] dp = new Boolean[n][n];
        str = s;
        String ans = "";
        for (int i = 0; i < n; i++) {
            if (max > n - i)
                break;
            for (int j = i; j < n; j++) {
                if (solve(i, j, dp)) {
                    if (j - i + 1 > max) {
                        max = j - i + 1;
                        ans = str.substring(i, j + 1);
                    }
                }
            }
        }
        return ans;
    }

    boolean solve(int i, int j, Boolean[][] dp) {
        if (i >= j) {
            return true;
        }
        if (dp[i][j] != null) {
            return dp[i][j];
        }
        if (str.charAt(i) == str.charAt(j)) {
            return dp[i][j] = solve(i + 1, j - 1, dp);
        } else {
            return dp[i][j] = false;
        }
    }
}
