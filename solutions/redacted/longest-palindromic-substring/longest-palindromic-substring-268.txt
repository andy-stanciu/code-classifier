class Solution {

    public String method1(String s) {
        if (s.length() < 1)
            return "";
        int spoint = 0;
        int epoint = 0;
        for (int i = 0; i < s.length(); i++) {
            int len1 = determine(s, i, i);
            int len2 = determine(s, i, i + 1);
            int maxlen = Math.max(len1, len2);
            if (maxlen > epoint - spoint) {
                spoint = i - (maxlen - 1) / 2;
                epoint = i + maxlen / 2;
            }
        }
        return s.substring(spoint, epoint + 1);
    }

    private int determine(String s, int l, int r) {
        while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
            l--;
            r++;
        }
        return r - l - 1;
    }
}
