class Solution {

    public String method1(String s) {
        int a = s.length() - 1;
        int b = a + 1, c = a;
        for (int j = 0; j <= a; j++) {
            for (int i = 0; i <= s.length() - b; i++) {
                String sub = s.substring(i, i + b);
                if (ispali(sub)) {
                    return sub;
                }
            }
            b--;
        }
        return null;
    }

    public static boolean ispali(String s) {
        int left = 0;
        int right = s.length() - 1;
        while (left < right) {
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
