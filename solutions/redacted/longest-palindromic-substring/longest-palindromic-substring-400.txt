class Solution {

    public String method1(String s) {
        String method1 = s.charAt(0) + "";
        for (int i = 0; i < s.length() - 1; i++) {
            for (int j = s.length(); j > i; j--) {
                String var1 = s.var1(i, j);
                if (isPalindrome(var1) && var1.length() > method1.length()) {
                    method1 = var1;
                }
            }
        }
        return method1;
    }

    public boolean isPalindrome(String s) {
        int left = 0;
        int right = s.length() - 1;
        while (left < right) {
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
