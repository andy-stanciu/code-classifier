class Solution {

    public String method1(String s) {
        String result = "";
        for (int i = 0; i < s.length(); i++) {
            // the center will be on the same character
            String oddLengths = expand(i, i, s);
            // the center will be between two characters
            String evenLengths = expand(i, i + 1, s);
            if (oddLengths.length() > result.length()) {
                result = oddLengths;
            }
            if (evenLengths.length() > result.length()) {
                result = evenLengths;
            }
        }
        return result;
    }

    public String expand(int l, int r, String s) {
        while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
            l--;
            r++;
        }
        return s.substring(l + 1, r);
    }
}
