class Solution {

    public String method1(String s) {
        int n = s.length();
        String res = "";
        int reslen = 0;
        for (int i = 0; i < n; i++) {
            int l = i;
            int r = i;
            while (l >= 0 && r < n && s.charAt(l) == s.charAt(r)) {
                if (r - l + 1 > reslen) {
                    res = s.substring(l, r + 1);
                    reslen = r - l + 1;
                }
                l--;
                r++;
            }
            l = i;
            r = i + 1;
            while (l >= 0 && r < n && s.charAt(l) == s.charAt(r)) {
                if (r - l + 1 > reslen) {
                    res = s.substring(l, r + 1);
                    reslen = r - l + 1;
                }
                l--;
                r++;
            }
        }
        return res;
    }
}
