class Solution {

    public int method1(int[] piles, int h) {
        int maxSpeed = Integer.MIN_VALUE;
        int len = piles.length;
        for (int i = 0; i < len; i++) {
            maxSpeed = Math.max(piles[i], maxSpeed);
        }
        int ans = maxSpeed;
        int i = 1, j = maxSpeed;
        while (i <= j) {
            int mid = i + (j - i) / 2;
            int hour = 0;
            int l = 0;
            for (; l < len; l++) {
                // hour += (piles[l] / mid);
                // if (piles[l] % mid != 0) {
                //     hour +=1 ;
                // }
                hour += (piles[l] - 1) / mid + 1;
                if (hour > h)
                    break;
            }
            if (l == len) {
                ans = Math.min(ans, mid);
                j = mid - 1;
            } else {
                i = mid + 1;
            }
        }
        return ans;
    }
}
