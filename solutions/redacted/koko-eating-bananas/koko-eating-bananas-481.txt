class Solution {

    public int method1(int[] piles, int h) {
        int maxPileSize = Arrays.stream(piles).max().getAsInt();
        var left = 1;
        var right = maxPileSize;
        while (left < right) {
            var mid = (left + right) >> 1;
            var var1 = mid;
            var canEat = canEatAllPilesWithinH(var1, h, piles);
            if (canEat) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return right;
    }

    public int method2(int[] piles, int h) {
        int maxPileSize = Arrays.stream(piles).max().getAsInt();
        //valid eating speeds
        var results = new ArrayList<Integer>();
        for (int i = 1; i <= maxPileSize; i++) {
            var var1 = i;
            var canEat = canEatAllPilesWithinH(i, h, piles);
            if (canEat) {
                results.add(var1);
                //comment this out if you want to see the duplicated work
                return var1;
            }
        }
        System.out.println(results);
        //        return results.getFirst(); uncommend if using java 21
        return results.iterator().next();
    }

    private boolean canEatAllPilesWithinH(int var1, int numHours, int[] piles) {
        var hourSpent = 0;
        for (var pile : piles) {
            hourSpent += Math.ceil((double) pile / var1);
        }
        if (hourSpent <= numHours) {
            return true;
        }
        return false;
    }
}
