class Solution {

    public int method1(int[] piles, int h) {
        // h must be >= length of piles { As per the analysis }
        // Ans can lie in the range k :  1  to  max of piles
        // for whichever value of k starting from 1, hours needed to eat all bananas is <= h, will be the Ans value of k
        // Youtube explanation :   https://www.youtube.com/watch?v=U2SozAs9RzA&list=LL&index=1
        Arrays.sort(piles);
        int max = piles[piles.length - 1];
        int start = 1;
        int end = max;
        while (start < end) {
            int mid = start + (end - start) / 2;
            int sum = 0;
            for (int p : piles) {
                if (p % mid == 0)
                    sum = sum + (p / mid);
                else
                    sum = sum + (p / mid) + 1;
            }
            if (sum <= h)
                end = mid;
            else
                start = mid + 1;
        }
        return end;
    }
}
