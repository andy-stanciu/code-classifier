class Solution {

    public int method1(int[] piles, int h) {
        int max = 0;
        int n = piles.length;
        for (int i = 0; i < n; i++) {
            if (max < piles[i]) {
                max = piles[i];
            }
        }
        int s = 1;
        int e = max;
        while (s < e) {
            int mid = s + (e - s) / 2;
            if (hours(mid, piles) <= h) {
                e = mid;
            } else {
                s = mid + 1;
            }
        }
        return s;
    }

    public int hours(int k, int[] piles) {
        int H = 0;
        for (int i = 0; i < piles.length; i++) {
            H = H + (piles[i] - 1) / k + 1;
        }
        return H;
    }
}
