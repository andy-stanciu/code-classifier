class Solution {

    int findMax(int[] a) {
        int maxi = Integer.MIN_VALUE;
        for (int n : a) maxi = Math.max(n, maxi);
        return maxi;
    }

    public int method1(int[] piles, int h) {
        int low = 1, high = findMax(piles);
        while (low <= high) {
            int mid = (low + high) / 2;
            int totalH = calculateTotalHour(piles, mid);
            if (totalH <= h)
                high = mid - 1;
            else
                low = mid + 1;
        }
        return low;
    }

    int calculateTotalHour(int[] a, int hourly) {
        int hour = 0;
        for (int i = 0; i < a.length; i++) {
            hour += Math.ceil((double) a[i] / (double) hourly);
        }
        return hour;
    }
}
