class Solution {

    public int method1(int[] piles, int h) {
        int start = 1;
        // for finding largest value in piles
        int end = Integer.MIN_VALUE;
        // finding max
        for (int i = 0; i < piles.length; i++) {
            if (piles[i] > end) {
                end = piles[i];
            }
        }
        int ans = -1;
        // apply binary search on the possible values of K -> (1, max(piles))
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (method2(piles, mid, h)) {
                ans = mid;
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return ans;
    }

    boolean method2(int[] piles, int param1, int hours) {
        int hoursSpent = 0;
        for (int i = 0; i < piles.length; i++) {
            hoursSpent += piles[i] / param1;
            if (piles[i] % param1 != 0) {
                hoursSpent += 1;
            }
            if (hoursSpent > hours)
                return false;
        }
        return true;
    }
}
