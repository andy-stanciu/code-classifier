class Solution {

    public int method1(int[] array, int hours) {
        int low = 1;
        int high = findMax(array);
        while (low <= high) {
            int mid = (low + high) / 2;
            int totalHours = findTotalHours(array, mid);
            if (totalHours <= hours) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }

    public static int findMax(int[] array) {
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < array.length; i++) {
            if (array[i] > max) {
                max = array[i];
            }
        }
        return max;
    }

    public static int findTotalHours(int[] array, int param1) {
        int totalHours = 0;
        for (int i = 0; i < array.length; i++) {
            totalHours += Math.ceil((double) (array[i]) / (double) (param1));
        }
        return totalHours;
    }
}
