class Solution {

    public int totalH(int[] piles, int k) {
        int total = 0;
        for (int i = 0; i < piles.length; i++) {
            total += Math.ceil((double) piles[i] / (double) k);
        }
        return total;
    }

    public int method1(int[] piles, int h) {
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < piles.length; i++) {
            max = Math.max(max, piles[i]);
        }
        int start = 1;
        int end = max;
        while (start <= end) {
            int mid = (start + end) / 2;
            int result = totalH(piles, mid);
            if (result > h) {
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return start;
    }
}
