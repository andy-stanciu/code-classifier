class Solution {

    public int method1(int[] piles, int h) {
        int start = 1;
        int end = Integer.MIN_VALUE;
        for (int i = 0; i < piles.length; i++) {
            if (piles[i] > end) {
                end = piles[i];
            }
        }
        int ans = -1;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (method2(piles, mid, h)) {
                ans = mid;
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return ans;
    }

    boolean method2(int[] piles, int param1, int hrs) {
        int hrspent = 0;
        for (int i = 0; i < piles.length; i++) {
            hrspent += piles[i] / param1;
            if (piles[i] % param1 != 0) {
                hrspent += 1;
            }
            if (hrspent > hrs)
                return false;
        }
        return true;
    }
}
