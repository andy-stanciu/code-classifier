class Solution {

    public int method1(int[] piles, int h) {
        int left = 1;
        int right = getMaxPile(piles);
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (canEatAll(piles, h, mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

    private int getMaxPile(int[] piles) {
        int maxPile = 0;
        for (int pile : piles) {
            if (pile > maxPile) {
                maxPile = pile;
            }
        }
        return maxPile;
    }

    private boolean canEatAll(int[] piles, int h, int k) {
        int hours = 0;
        for (int pile : piles) {
            // Equivalent to ceiling(pile / k)
            hours += (pile + k - 1) / k;
        }
        return hours <= h;
    }
}
