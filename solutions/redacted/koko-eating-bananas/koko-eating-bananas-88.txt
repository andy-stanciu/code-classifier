class Solution {

    public int findhours(int mid, int[] arr) {
        int tot = 0;
        for (int i = 0; i < arr.length; i++) tot += Math.ceil((double) arr[i] / (double) mid);
        return tot;
    }

    public int method1(int[] arr, int h) {
        int maxele = arr[0];
        for (int i = 0; i < arr.length; i++) maxele = Math.max(arr[i], maxele);
        int low = 1;
        int high = maxele;
        while (low <= high) {
            int mid = (low + high) / 2;
            int midans = findhours(mid, arr);
            if (midans <= h)
                high = mid - 1;
            else
                low = mid + 1;
        }
        return low;
    }
}
