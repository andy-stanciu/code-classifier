class Solution {

    public int method1(int[] piles, int hr) {
        int l = 1;
        int h = 0;
        // range of our answer
        for (int i : piles) h = Math.max(i, h);
        // binary search
        while (l < h) {
            int mid = l + (h - l) / 2;
            if (helper(piles, mid, hr))
                h = mid;
            else
                l = mid + 1;
        }
        return l;
    }

    boolean helper(int[] arr, int h, int hr) {
        //here we ate h banana in an hour
        // so total hr neded to finish all will be....
        int ans = 0;
        for (int i : arr) {
            ans += i / h;
            if (i % h != 0)
                ans++;
        }
        // if our time to finish banana is less than given time
        return ans <= hr;
        // then we have eted more banana than required in an hour so need to
        // decrease banana by  h=mid
    }
}
