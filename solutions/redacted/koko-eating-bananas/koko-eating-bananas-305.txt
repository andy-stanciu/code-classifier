class Solution {

    public int method1(int[] piles, int h) {
        // b-search
        int left = 1, right = 0;
        for (int val : piles) right = Math.max(right, val);
        while (left < right) {
            int mid = left + (right - left) / 2;
            long hrSum = 0;
            for (int pile : piles) {
                hrSum += ((pile - 1) / mid + 1);
            }
            //System.out.println(mid + "|" + hrSum);
            if (hrSum > h)
                left = mid + 1;
            else
                right = mid;
        }
        return left;
    }
}
