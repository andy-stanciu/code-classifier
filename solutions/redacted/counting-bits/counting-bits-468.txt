class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        ans[0] = 0;
        for (int i = 1; i < ans.length; i++) {
            int count = 0;
            int j = i;
            while (j != 0) {
                if ((j & 1) == 1)
                    count++;
                j = j >> 1;
            }
            ans[i] = count;
        }
        return ans;
    }
}
