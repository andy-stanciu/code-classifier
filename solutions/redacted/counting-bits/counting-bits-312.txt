class Solution {

    public int[] method1(int n) {
        if (n == 0) {
            return new int[] { 0 };
        }
        int[] var1 = new int[n + 1];
        var1[0] = 0;
        var1[1] = 1;
        for (int i = 2; i < n + 1; i++) {
            var1[i] = var1[i / 2] + (i % 2 == 0 ? 0 : 1);
        }
        return var1;
    }
}
