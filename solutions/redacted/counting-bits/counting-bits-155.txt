class Solution {

    public int[] method1(int n) {
        //Initializing the dp array with n+1 len as we include 0 too.
        int[] dp = new int[n + 1];
        dp[0] = 0;
        //Offset here refers to 2^k {1,2,4,8,16}
        int offset = 1;
        for (int i = 1; i <= n; i++) {
            if (offset * 2 == i) {
                offset = i;
            }
            //Calculation of 1s
            dp[i] = 1 + dp[i - offset];
        }
        return dp;
    }
}
