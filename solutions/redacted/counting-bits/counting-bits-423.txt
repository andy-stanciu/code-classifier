class Solution {

    public int[] method1(int n) {
        int[] pq = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            if (i % 2 == 0) {
                pq[i] = pq[i / 2];
            } else {
                pq[i] = pq[i - 1] + 1;
            }
        }
        return pq;
    }
}
