class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            arr[i] = countone(i);
        }
        return arr;
    }

    private int countone(int k) {
        int count = 0;
        while (k > 0) {
            if ((k & 1) == 1) {
                count++;
            }
            k = k >> 1;
        }
        return count;
    }
}
