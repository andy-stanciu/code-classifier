class Solution {

    public int[] method1(int n) {
        int[] dp = new int[n + 1];
        if (n == 0)
            return dp;
        dp[1] = 1;
        int nextPow = 2;
        int prePow = 2;
        for (int i = 2; i <= n; ++i) {
            if (i == nextPow) {
                dp[i] = 1;
                prePow = nextPow;
                nextPow *= 2;
            } else {
                dp[i] = 1 + dp[i - prePow];
            }
        }
        return dp;
    }
}
