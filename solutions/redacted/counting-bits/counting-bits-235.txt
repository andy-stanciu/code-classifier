class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            var1[i] = counter(i);
        }
        return var1;
    }

    private int counter(int n) {
        if (n == 0) {
            return 0;
        }
        int c = counter(n / 2);
        return n % 2 == 1 ? c + 1 : c;
    }
}
