class Solution {

    public int[] method1(int n) {
        if (n == 0) {
            return new int[] { 0 };
        }
        if (n == 1) {
            return new int[] { 0, 1 };
        }
        int[] cb = new int[n + 1];
        cb[0] = 0;
        cb[1] = 1;
        for (int i = 2; i <= n; i++) {
            cb[i] = cb[i >> 1] + (i & 1);
        }
        return cb;
    }
}
