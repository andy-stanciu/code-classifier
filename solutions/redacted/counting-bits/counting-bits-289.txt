class Solution {

    public int[] method1(int n) {
        int[] a = new int[n + 1];
        a[0] = 0;
        for (int i = 1; i <= n; i++) {
            int c = 0;
            int k = i;
            while (k != 0) {
                if (k % 2 != 0)
                    c++;
                k >>= 1;
            }
            a[i] = c;
        }
        return a;
    }
}
