class Solution {

    public int[] method1(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 0;
        int val = 0;
        for (int i = 1; i <= n; i++) {
            if (i == (int) Math.pow(2, val)) {
                val += 1;
            }
            dp[i] = dp[i - (int) Math.pow(2, val - 1)] + 1;
        }
        return dp;
    }
}
