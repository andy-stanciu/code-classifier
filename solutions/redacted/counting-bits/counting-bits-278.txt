class Solution {

    public int[] method1(int n) {
        int[] dp = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int count = 0;
            int k = i;
            for (int j = 0; j < 32; j++) {
                count += k & 1;
                k = k >> 1;
            }
            dp[i] = count;
        }
        return dp;
    }
}
