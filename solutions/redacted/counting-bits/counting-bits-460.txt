class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) arr[i] = method2(i);
        return arr;
    }

    static int method2(int n) {
        int c = 0;
        while (n != 0) {
            c++;
            n = n & (n - 1);
        }
        return c;
    }
}
