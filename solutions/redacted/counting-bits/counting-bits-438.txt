class Solution {

    public int[] method1(int n) {
        int[] countOnes = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            // uses half of the number plus the remainder divided by two to yields the number of 1s
            countOnes[i] = countOnes[i / 2] + (i % 2);
        }
        return countOnes;
    }
}
