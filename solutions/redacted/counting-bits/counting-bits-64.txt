class Solution {

    //creating a function to calculate the number of ones in the binary representation of an integer
    public static int countOne(int i) {
        int count = 0;
        while (i != 0) {
            //comparing the last digit of binary representation of the number with the binary representation of 1
            int lastbit = i & 1;
            //adding 1 if the lastBit is 1, otherwise adding 0
            count += lastbit;
            i = i >>> 1;
        }
        return count;
    }

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            ans[i] = countOne(i);
        }
        return ans;
    }
}
