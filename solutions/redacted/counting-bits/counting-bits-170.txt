class Solution {

    public int[] method1(int n) {
        int[] result = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            //i >> 1 shifts the bits of i to the right by 1, effectively dividing i by 2
            //i & 1 checks if the last bit of i is 1 (odd number) or even
            result[i] = result[i >> 1] + (i & 1);
        }
        return result;
    }
}
