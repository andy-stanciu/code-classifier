class Solution {

    public int[] method1(int in) {
        int[] result = new int[in + 1];
        for (int i = 0; i < result.length; i++) {
            int n = i;
            int count = 0;
            while (n != 0) {
                if ((n & 1) == 1)
                    count++;
                n = n >> 1;
            }
            result[i] = count;
        }
        return result;
    }
}
