class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        var1[0] = 0;
        if (n == 0)
            return var1;
        var1[1] = 1;
        for (int i = 2; i < n + 1; i++) {
            if (i % 2 == 1)
                var1[i] = Math.min(var1[i - 1], var1[i - 2]) + 1;
            else
                var1[i] = var1[i / 2];
        }
        return var1;
    }
}
