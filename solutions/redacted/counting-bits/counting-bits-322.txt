class Solution {

    public int count(int n) {
        int count = 0;
        while (n != 0) {
            count += n & 1;
            n >>>= 1;
        }
        return count;
    }

    public int[] method1(int n) {
        int[] a = new int[n + 1];
        int num = 0;
        while (num <= n) {
            int x = count(num);
            a[num] = x;
            num++;
        }
        return a;
    }
}
