class Solution {

    public int[] method1(int num) {
        int[] result = new int[num + 1];
        //for each number 0 to num
        for (int i = 0; i < result.length; i++) {
            //count the number of 1 in i
            result[i] = result[i >> 1] + (i & 1);
        }
        return result;
    }
}
