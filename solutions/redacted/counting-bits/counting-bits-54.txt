class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        ans[0] = 0;
        for (int i = 1; i <= n; i++) {
            // O(N)
            // depending on whether i is even or odd, we can get 0 or 1
            // O(1)
            ans[i] = ans[i / 2] + i % 2;
        }
        return ans;
    }
}
