class Solution {

    public int[] method1(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 0;
        if (n == 0)
            return dp;
        dp[1] = 1;
        if (n == 1)
            return dp;
        for (int i = 0; i <= n; i++) {
            if (i % 2 == 0) {
                int half = i >> 1;
                dp[i] = dp[half];
            } else {
                int val = dp[i - 1];
                dp[i] = val + 1;
            }
        }
        return dp;
    }
}
