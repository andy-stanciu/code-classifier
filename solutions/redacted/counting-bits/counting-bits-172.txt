class Solution {

    public int[] method1(int n) {
        if (n == 0)
            return new int[] { 0 };
        else if (n == 1)
            return new int[] { 0, 1 };
        int[] arr = new int[n + 1];
        arr[0] = 0;
        arr[1] = 1;
        for (int i = 2; i <= n; i++) arr[i] = ((i & 1) == 1 ? arr[i / 2] + 1 : arr[i / 2]);
        return arr;
    }
}
