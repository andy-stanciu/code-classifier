class Solution {

    public int[] method1(int n) {
        int[] x = new int[n + 1];
        if (n == 0)
            return x;
        x[0] = 0;
        x[1] = 1;
        for (int i = 2; i <= n; i++) {
            int n1 = i;
            int a = 0;
            int p = 0;
            while (n1 != 0) {
                a = n1 % 2;
                if (a == 1)
                    p++;
                n1 = n1 / 2;
            }
            x[i] = p;
        }
        return x;
    }
}
