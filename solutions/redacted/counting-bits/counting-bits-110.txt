class Solution {

    public int[] method1(int n) {
        int[] nums = new int[n + 1];
        for (int i = 1; i < nums.length; i++) {
            nums[i] = i;
        }
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            int number = nums[i];
            while (number != 0) {
                number = number & (number - 1);
                count++;
            }
            nums[i] = count;
            count = 0;
        }
        return nums;
    }
}
