class Solution {

    public int[] method1(int n) {
        // Initialize an array to store the count of set bits from 0 to n
        int[] result = new int[n + 1];
        // Iterate through each number from 0 to n
        for (int i = 1; i <= n; i++) {
            // result[i >> 1] gives the number of 1\'s in i / 2
            // i & 1 checks if the number is odd (i.e., if the least significant bit is 1)
            result[i] = result[i >> 1] + (i & 1);
        }
        return result;
    }
}
