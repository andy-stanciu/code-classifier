class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 1; i <= n; ++i) {
            int j = i, c = 0;
            while (j > 0) {
                if (j % 2 != 0)
                    c++;
                j /= 2;
            }
            arr[i] = c;
        }
        return arr;
    }
}
