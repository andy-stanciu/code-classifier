class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            ans[i] = method2(i);
        }
        return ans;
    }

    public static int method2(int n) {
        int c = 0;
        while (n > 0) {
            if ((n & 1) == 1)
                c++;
            n = n >> 1;
        }
        return c;
    }
}
