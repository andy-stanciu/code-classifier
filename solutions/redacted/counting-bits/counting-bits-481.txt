class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        int i = 0;
        while (true) {
            if (i * 2 > n) {
                break;
            } else {
                res[i * 2] = res[i] + 0;
            }
            if (i * 2 + 1 > n) {
                break;
            } else {
                res[i * 2 + 1] = res[i] + 1;
            }
            i++;
        }
        return res;
    }
}
