class Solution {

    public int[] method1(int n) {
        //using number division
        int[] arr = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            int count = 0;
            int num = i;
            while (num > 0) {
                if (num % 2 == 1)
                    count++;
                num /= 2;
            }
            arr[i] = count;
        }
        return arr;
    }
}
