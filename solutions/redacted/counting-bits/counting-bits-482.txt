class Solution {

    public int[] method1(int n) {
        int counter = 0;
        int powerCounter = 1;
        int[] ans = new int[n + 1];
        ans[0] = 0;
        for (int i = 1; i < ans.length; i++) {
            if (powerCounter == i) {
                powerCounter *= 2;
                counter = 0;
            }
            ans[i] = 1 + ans[counter++];
        }
        return ans;
    }
}
