class Solution {

    public int[] method1(int n) {
        int[] a = new int[n + 1];
        int o = 1, x = 1;
        for (int i = 1; i <= n; i++) {
            a[i] = a[i - o] + 1;
            if (i + 1 == (int) Math.pow(2, x)) {
                o = i + 1;
                x++;
            }
        }
        return a;
    }
}
