class Solution {

    public int[] method1(int n) {
        int[] a = new int[n + 1];
        int m = 2;
        for (int i = 0; i <= n && i <= 1; i++) a[i] = i;
        for (int i = 2; i <= n; i++) {
            if (i == m) {
                m *= 2;
            }
            a[i] = 1 + a[i % (m / 2)];
        }
        return a;
    }
}
