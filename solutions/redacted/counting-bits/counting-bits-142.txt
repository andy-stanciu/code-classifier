class Solution {

    public int[] method1(int n) {
        int[] count = new int[n + 1];
        count[0] = 0;
        if (n == 0) {
            return count;
        }
        count[1] = 1;
        // f(x) = f(x/2) + x%2;
        for (int i = 2; i <= n; i++) {
            count[i] = count[i / 2] + i % 2;
        }
        return count;
    }
}
