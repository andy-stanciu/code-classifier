class Solution {

    public int[] method1(int n) {
        // Array to store result from 0 to n
        int[] arr = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            // For each number i, the number of 1\'s is:
            // arr[i >> 1] is the number of 1\'s in i/2, and (i & 1) adds 1 if i is odd
            arr[i] = arr[i >> 1] + (i & 1);
        }
        return arr;
    }
}
