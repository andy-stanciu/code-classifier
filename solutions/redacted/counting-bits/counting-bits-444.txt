class Solution {

    public int[] method1(int n) {
        int[] res = new int[n + 1];
        // int res =0;
        for (int i = 0; i <= n; i++) {
            res[i] = check(i);
        }
        return res;
    }

    public int check(int n) {
        int count = 0;
        while (n > 0) {
            if (n % 2 == 1) {
                count++;
            }
            n = n / 2;
        }
        return count;
    }
}
