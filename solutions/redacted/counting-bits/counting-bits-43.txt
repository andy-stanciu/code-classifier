class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        var1[0] = 0;
        for (int i = 1; i <= n; i++) {
            var1[i] = method2(i);
        }
        return var1;
    }

    public int method2(int n) {
        String s = Integer.toBinaryString(n).replaceAll("0", "");
        return s.length();
    }
}
// TC: O(n^2 * logn), SC: O(n)
