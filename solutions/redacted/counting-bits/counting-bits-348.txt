class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i < n + 1; i++) {
            int res = 0;
            int num = i;
            while (num != 0) {
                int lastBit = num & 1;
                if (lastBit == 1) {
                    res++;
                }
                num = num >>> 1;
            }
            arr[i] = res;
        }
        return arr;
    }
}
