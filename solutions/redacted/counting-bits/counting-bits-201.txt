class Solution {

    public int[] method1(int n) {
        int[] a = new int[n + 1];
        int rr = 0;
        for (int i = 0; i <= n; i++) {
            int c = 0;
            int r = i;
            while (r > 0) {
                if (r % 2 == 1)
                    c++;
                r /= 2;
            }
            a[rr++] = c;
            System.out.print(a[rr - 1]);
        }
        return a;
    }
}
