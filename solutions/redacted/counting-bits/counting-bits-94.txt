class Solution {

    public int[] method1(int n) {
        int[] dp = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            // Even number = (half number << 1) + 0 = (half number << 1) | 0
            // Odd number = (half number << 1) + 1 = (half number << 1) | 1
            dp[i] = dp[i / 2] + (i & 1);
        }
        return dp;
    }
}
