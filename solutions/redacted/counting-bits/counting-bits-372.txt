class Solution {

    public int[] method1(int n) {
        int[] var1 = new int[n + 1];
        String var2;
        for (int i = 0; i < var1.length; i++) {
            var2 = Integer.toBinaryString(i);
            var1[i] = var2.length() - var2.replace("1", "").length();
        }
        return var1;
    }
}
