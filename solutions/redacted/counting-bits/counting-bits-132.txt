class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= (n / 2); i++) {
            int n1 = 2 * i;
            int n2 = 2 * i + 1;
            arr[n1] = arr[i];
            if (n2 <= n) {
                arr[n2] = arr[i] + 1;
            }
        }
        return arr;
    }
}
