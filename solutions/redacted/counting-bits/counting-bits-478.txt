class Solution {

    public int[] method1(int n) {
        int[] list = IntStream.range(0, n + 1).toArray();
        int[] ans = Arrays.stream(list).map(i -> calculateOnes(i)).toArray();
        return ans;
    }

    private int calculateOnes(int i) {
        int count = 0;
        while (i > 0) {
            count += i & 1;
            i >>= 1;
        }
        return count;
    }
}
