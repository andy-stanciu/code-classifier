class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            Long num = Long.valueOf(Integer.toBinaryString(i));
            int sum = 0;
            while (num > 0) {
                if (num % 10 == 1) {
                    sum++;
                }
                num = num / 10;
            }
            arr[i] = sum;
        }
        return arr;
    }
}
