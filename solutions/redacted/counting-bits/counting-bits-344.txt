class Solution {

    private int count(int x) {
        int cnt = 0;
        while (x > 0) {
            int rem = x % 2;
            x = x / 2;
            if (rem == 1) {
                cnt++;
            }
        }
        return cnt;
    }

    public int[] method1(int n) {
        int[] num = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            num[i] = count(i);
        }
        return num;
    }
}
