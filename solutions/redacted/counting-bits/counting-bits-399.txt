class Solution {

    public int[] method1(int n) {
        int[] arr = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            arr[i] = count(i);
        }
        return arr;
    }

    public int count(int n) {
        if (n == 0)
            return 0;
        if (n == 1)
            return 1;
        if (n % 2 == 0)
            return count(n / 2);
        else {
            return 1 + count(n / 2);
        }
    }
}
