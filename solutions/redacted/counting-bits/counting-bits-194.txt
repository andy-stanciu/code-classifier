class Solution {

    public int[] method1(int n) {
        int[] result = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            if (i == 0) {
                result[i] = 0;
            }
            if (i == 1) {
                result[i] = i;
            }
            int num = i;
            int sum = 0;
            while (num != 0) {
                int rem = num % 2;
                sum += rem;
                num = num / 2;
            }
            result[i] = sum;
        }
        return result;
    }
}
