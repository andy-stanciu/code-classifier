class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        int offset = 1, curr = offset * 2;
        for (int i = 1; i <= n; i++) {
            if (i == curr) {
                offset = i;
                curr = offset * 2;
            }
            ans[i] = ans[i - offset] + 1;
        }
        return ans;
    }
}
