class Solution {

    public int[] method1(int n) {
        int[] ans = new int[n + 1];
        dfs(ans, 1, 1);
        return ans;
    }

    private static void dfs(int[] ans, int index, int param1) {
        if (index >= ans.length)
            return;
        ans[index] = param1;
        dfs(ans, index * 2, param1);
        dfs(ans, index * 2 + 1, param1 + 1);
    }
}
