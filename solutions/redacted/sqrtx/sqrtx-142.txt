class Solution {

    public int mySqrt(int x) {
        int start = 0, end = x, sqrt = 0;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            //this multiplication can go integer overflow, hence typecasting to long integer
            long midSquare = (long) mid * mid;
            if (midSquare == x) {
                sqrt = mid;
                break;
            } else if (midSquare < x) {
                sqrt = mid;
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return sqrt;
    }
}
