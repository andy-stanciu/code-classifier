class Solution {

    public int mySqrt(int number) {
        if (number == 0 || number == 1) {
            return number;
        }
        double x = number;
        double root;
        while (true) {
            root = 0.5 * (x + (number / x));
            if (Math.abs(root - x) == 0.0) {
                break;
            }
            x = root;
        }
        return (int) root;
    }
}
