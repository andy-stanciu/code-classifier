public class Solution {

    public int mySqrt(int x) {
        // Special case for 0
        if (x == 0)
            return 0;
        int left = 1, right = x, ans = 0;
        while (left <= right) {
            // Prevent overflow by using this formula for mid
            int mid = left + (right - left) / 2;
            if (mid <= x / mid) {
                // Equivalent to mid * mid <= x but avoids overflow
                // Store mid as potential answer
                ans = mid;
                // Try to find a larger square root
                left = mid + 1;
            } else {
                // Look for a smaller value
                right = mid - 1;
            }
        }
        return ans;
    }
}
