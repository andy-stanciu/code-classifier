class Solution {

    public int mySqrt(int n) {
        if (n == 0 || n == 1) {
            return n;
        }
        int s = 0;
        int e = n;
        int root = 0;
        while (s <= e) {
            int m = s + (e - s) / 2;
            if (m <= n / m) {
                // Update root only when m*m <= n
                root = m;
                s = m + 1;
            } else {
                e = m - 1;
            }
        }
        return root;
    }
}
