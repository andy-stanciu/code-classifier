class Solution {

    public int mySqrt(int x) {
        int ans = sq_roo(x);
        return ans;
    }

    static int sq_roo(int target) {
        if (target == 0) {
            return 0;
        }
        int start = 0;
        // Adding 1 to handle edge cases for large values
        int end = target / 2 + 1;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            // Use long to avoid overflow
            long midSquare = (long) mid * mid;
            if (midSquare == target) {
                return mid;
            } else if (midSquare < target) {
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return end;
    }
}
