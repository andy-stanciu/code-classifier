class Solution {

    public int mySqrt(int x) {
        int p = (int) Math.sqrt(x);
        return p;
    }

    public static void main(String[] args) {
        Solution obj = new Solution();
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter a number:");
        int n = sc.nextInt();
        int p = obj.mySqrt(n);
        System.out.print(p);
    }
}
