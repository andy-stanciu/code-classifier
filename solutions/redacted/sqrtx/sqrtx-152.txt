class Solution {

    public int mySqrt(int x) {
        int result = 0;
        for (int i = 1; i <= x; i++) {
            if (i <= x / i) {
                // Avoid overflow by using division
                result = i;
            } else {
                // Stop if i exceeds the square root of x
                break;
            }
        }
        return result;
    }
}
