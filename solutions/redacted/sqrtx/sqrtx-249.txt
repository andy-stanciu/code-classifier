class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        int left = 1, right = x, result = 0;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            //To avoid overflow, checking mid <= x/mid instead of mid * mid with x
            if (mid <= x / mid) {
                //mid is a possible answer
                result = mid;
                //Also check the next number if it\'s a smaller it\'s still possible to be the answer
                left = mid + 1;
            } else {
                //mid is too large, search in the left half
                right = mid - 1;
            }
        }
        return result;
    }
}
