class Solution {

    public int mySqrt(int x) {
        if (x <= 1) {
            return x;
        }
        int l = 0;
        int r = x;
        int res = 0;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            long temp = x / mid;
            if (temp == mid) {
                return mid;
            } else if (temp > mid) {
                res = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return res;
    }
}
