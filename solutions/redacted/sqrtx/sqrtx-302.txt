class Solution {

    public int mySqrt(int x) {
        if (x == 1 || x == 0)
            return x;
        int l = 1;
        int r = x / 2;
        int mid;
        while (l <= r) {
            mid = l + (r - l) / 2;
            if (mid == (x / mid)) {
                return mid;
            }
            if (mid < (x / mid)) {
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return r;
    }
}
