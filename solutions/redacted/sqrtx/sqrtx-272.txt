class Solution {

    public int mySqrt(int x) {
        int ans = 0;
        long st = 0;
        long end = x;
        while (st <= end) {
            long mid = st + (end - st) / 2;
            long val = mid * mid;
            if (val == x)
                return (int) mid;
            else if (val > x) {
                end = mid - 1;
            } else {
                st = mid + 1;
                ans = (int) mid;
            }
        }
        return ans;
    }
}
