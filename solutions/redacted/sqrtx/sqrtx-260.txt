class Solution {

    public int mySqrt(int x) {
        long longx = (long) x;
        long l = 0, r = longx;
        while (l < r) {
            long m = (l + r + 1) / 2;
            if (m * m > longx) {
                r = m - 1;
            }
            if (m * m <= longx) {
                l = m;
            }
        }
        return (int) l;
    }
}
