class Solution {

    public int mySqrt(int x) {
        // The square root of 0 is 0
        if (x == 0)
            return 0;
        int st = 1;
        int end = x;
        int ans = 0;
        while (st <= end) {
            int mid = st + (end - st) / 2;
            // Use multiplication to avoid precision errors
            if (mid <= x / mid) {
                // mid could be a potential answer
                ans = mid;
                // try for a larger mid
                st = mid + 1;
            } else {
                // mid is too large
                end = mid - 1;
            }
        }
        return ans;
    }
}
