class Solution {

    public int mySqrt(int x) {
        int start = 0;
        int end = x;
        int result = 0;
        if (x < 2) {
            return x;
        }
        while (start <= end) {
            int mid = start + (end - start) / 2;
            long midsquare = (long) mid * mid;
            if (midsquare == x) {
                return mid;
            } else if (midsquare < x) {
                start = mid + 1;
                result = mid;
            } else {
                end = mid - 1;
            }
        }
        return result;
    }
}
