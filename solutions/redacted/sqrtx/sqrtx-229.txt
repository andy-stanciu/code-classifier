class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            return x;
        }
        int left = 1, right = x, result = 0;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (mid <= x / mid) {
                result = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return result;
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        int test1 = 4;
        int test2 = 8;
        System.out.println("Square root of " + test1 + " is : " + solution.mySqrt(test1));
        System.out.println("Square root of " + test2 + " is : " + solution.mySqrt(test2));
    }
}
