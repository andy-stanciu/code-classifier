class Solution {

    public int mySqrt(int x) {
        if (// To avoid division by 0
        x == 0)
            return 0;
        // The squareroot can only lie in this range
        int start = 1, end = x / 2 + 1;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (// Not multiplying as the product for a large number may be greater than the max value of int
            mid == x / mid)
                return mid;
            else if (mid < x / mid)
                start = mid + 1;
            else
                end = mid - 1;
        }
        // end will be the largest integer such that end^2 < x
        return end;
    }
}
