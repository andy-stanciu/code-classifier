class Solution {

    public int mySqrt(int x) {
        long start = 1;
        long end = x;
        while (start <= end) {
            long mid = start + (end - start) / 2;
            long sqrt = mid * mid;
            if (sqrt > x) {
                end = mid - 1;
            } else if (sqrt < x) {
                start = mid + 1;
            } else {
                return (int) mid;
            }
        }
        return (int) end;
    }
}
