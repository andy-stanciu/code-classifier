class Solution {

    public int mySqrt(long target) {
        long start = 1;
        long end = target;
        long possible = 0;
        while (start <= end) {
            long mid = start + (end - start) / 2;
            if (mid * mid == target) {
                return (int) mid;
            } else if (mid * mid < target) {
                possible = mid;
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return (int) possible;
    }
}
