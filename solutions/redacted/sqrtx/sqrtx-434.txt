class Solution {

    public int mySqrt(int x) {
        long s = 1;
        long i = 0;
        while (i <= x) {
            i = s * s;
            if (i >= x)
                break;
            s++;
        }
        if (i > x)
            return (int) s - 1;
        else
            return (int) s;
    }
}
