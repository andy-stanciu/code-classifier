class Solution {

    public int mySqrt(int x) {
        int low = 1, high = x;
        int res = 0;
        if (x == 0 || x == 1) {
            return x;
        }
        while (low <= high) {
            int mid = low + ((high - low) / 2);
            if ((long) mid * mid <= (long) x) {
                res = mid;
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return res;
    }
}
