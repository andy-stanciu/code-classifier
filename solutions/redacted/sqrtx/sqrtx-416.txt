class Solution {

    public int mySqrt(int x) {
        // Edge case: the square root of 0 is 0
        if (x == 0) {
            return 0;
        }
        // Initialize the search range
        int left = 1;
        int right = x;
        int result = 0;
        // Binary search loop
        while (left <= right) {
            // Calculate the midpoint of the current range
            int mid = left + (right - left) / 2;
            // Check if the square of the midpoint is less than, equal to, or greater than x
            if (mid <= x / mid) {
                // mid * mid <= x, so mid is a potential result
                result = mid;
                // Move the left pointer to narrow the range
                left = mid + 1;
            } else {
                // mid * mid > x, so move the right pointer to narrow the range
                right = mid - 1;
            }
        }
        // Return the result
        return result;
    }
}
