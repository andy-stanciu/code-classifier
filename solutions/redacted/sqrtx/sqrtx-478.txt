class Solution {

    public int mySqrt(int x) {
        // handle edge case where mid could be zero (dividing by 0)
        if (x < 2) {
            return x;
        }
        // start from 1 since we have handled x < 2 above
        int left = 1;
        int right = x;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            // To prevent overflow for large x, compare using division
            if (mid <= x / mid) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return right;
    }
}
