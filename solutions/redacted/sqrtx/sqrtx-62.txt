class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        long l = 1, h = x / 2;
        int output = 1;
        while (l <= h) {
            long mid = l + (h - l) / 2;
            long sqrt = mid * mid;
            if (sqrt == x)
                return (int) mid;
            else if (sqrt < x) {
                output = (int) mid;
                l = mid + 1;
            } else
                h = mid - 1;
        }
        return output;
    }
}
