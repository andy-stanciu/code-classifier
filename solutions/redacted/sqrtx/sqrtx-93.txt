class Solution {

    public int mySqrt(int x) {
        if (x == 0)
            return 0;
        int start = 1, end = x;
        while (start < end) {
            int mid = start + (end - start) / 2;
            if (// Found the result
            mid <= x / mid && (mid + 1) > x / (mid + 1))
                return mid;
            else if (// Keep checking the left part
            mid > x / mid)
                end = mid;
            else
                // Keep checking the right part
                start = mid + 1;
        }
        return start;
    }
}
