class Solution {

    public int mySqrt(int x) {
        //handle simple cases directly
        if (x < 2) {
            return x;
        }
        int left = 0, right = x;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            //use long to avoid overflow
            long midSquared = (long) mid * mid;
            if (midSquared == x) {
                return mid;
            } else if (midSquared < x) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return right;
    }
}
