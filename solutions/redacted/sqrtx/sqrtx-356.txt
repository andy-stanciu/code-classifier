class Solution {

    public int mySqrt(int x) {
        if (x <= 1)
            return x;
        int lo = 0;
        int hi = x;
        int result = 0;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (mid <= x / mid) {
                lo = mid + 1;
                result = mid;
            } else if (mid > x / mid)
                hi = mid - 1;
        }
        return result;
    }
}
