class Solution {

    public int mySqrt(int x) {
        // repeated subtraction method: subtract successive odd numbers from x and count
        int count = 0;
        int odd = 1;
        while (x > 0) {
            x -= odd;
            odd += 2;
            count++;
        }
        return (x == 0) ? count : count - 1;
    }
}
