class Solution {

    public int mySqrt(int x) {
        int low = 1;
        int high = x;
        int result = 0;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if ((long) mid * (long) mid <= x) {
                low = mid + 1;
                // Update result when mid^2 is less than or equal to x
                result = mid;
            } else {
                high = mid - 1;
            }
        }
        return result;
    }
}
