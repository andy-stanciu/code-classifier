class Solution {

    public int mySqrt(int x) {
        //Base case of x happening to be equal to 1 or 0
        if (x == 1 || x == 0) {
            return x;
        }
        int start = 0;
        // Declaring end as x initially only used for decreasing mid value
        int end = x;
        int result = 0;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            //if mid square is less than x than there is need to increase the mid value
            if (mid <= x / mid) {
                result = mid;
                start = mid + 1;
            } else //Exact opposite of the previous comment
            {
                end = mid - 1;
            }
        }
        return result;
    }
}
