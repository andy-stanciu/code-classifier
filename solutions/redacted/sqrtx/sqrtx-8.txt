class Solution {

    public int mySqrt(int x) {
        double a = Math.pow(2, 31) - 1;
        for (int i = 0; i < a; i++) {
            double b = Math.pow(i, 2);
            if (x < b) {
                return i - 1;
            }
            if (x == b) {
                return i;
            }
        }
        return 0;
    }
}
