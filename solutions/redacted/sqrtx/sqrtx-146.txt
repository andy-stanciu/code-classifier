class Solution {

    public int mySqrt(int x) {
        return (int) Math.pow(x, 0.5);
    }

    public int pow(int x, int n) {
        if (n == 0) {
            return 1;
        } else {
            if (n % 2 == 0) {
                return pow(x * x, n / 2);
            } else
                return x * pow(x * x, (n - 1) / 2);
        }
    }
}
