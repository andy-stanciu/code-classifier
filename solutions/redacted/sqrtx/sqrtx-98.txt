class Solution {

    public int mySqrt(int x) {
        if (x == 1)
            return 1;
        int l = 0;
        int r = x;
        while (l < r - 1) {
            // ** not l < r => think about the case r = l + 2
            int mid = l + (r - l) / 2;
            if (// prevent from mid * mid overflow
            mid == x / mid)
                // prevent from mid * mid overflow
                return mid;
            else if (// prevent from mid * mid overflow
            mid > x / mid)
                // prevent from mid * mid overflow
                r = mid;
            else
                // *** not l = mid + 1
                l = mid;
        }
        return l;
    }
}
