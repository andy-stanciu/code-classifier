class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            // handle edge cases where x is 0 or 1
            return x;
        }
        int left = 1, right = x;
        int result = 0;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            // To avoid overflow, use division instead of multiplication for comparison
            if (mid <= x / mid) {
                // This mid value might be our answer
                result = mid;
                // search in the right half for a larger number
                left = mid + 1;
            } else {
                // search in the left half for a smaller number
                right = mid - 1;
            }
        }
        return result;
    }
}
