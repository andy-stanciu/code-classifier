class Solution {

    public int mySqrt(int x) {
        // Special case for 0
        if (x == 0)
            return 0;
        int l = 1;
        int r = x;
        // To store the result
        int result = 0;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (mid <= x / mid) {
                // This avoids potential overflow from mid * mid
                // Update result if mid*mid is still less than or equal to x
                result = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return result;
    }
}
