class Solution {

    public int mySqrt(int x) {
        if (x == 0) {
            return 0;
        }
        if (x == 1) {
            return 1;
        }
        long start = 1;
        long end = x;
        while (start <= end) {
            long midl = start + (end - start) / 2;
            long powe = midl * midl;
            if (powe == x) {
                return (int) midl;
            } else if (powe > x) {
                end = midl - 1;
            } else {
                start = midl + 1;
            }
        }
        return (int) end;
    }
}
