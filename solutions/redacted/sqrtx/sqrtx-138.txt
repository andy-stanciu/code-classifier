class Solution {

    public int mySqrt(int x) {
        if (x == 0 || x == 1) {
            // Edge cases: sqrt(0) = 0, sqrt(1) = 1
            return x;
        }
        // Initial guess
        long guess = x;
        // Newton\'s method loop
        while (guess * guess > x) {
            guess = (guess + x / guess) / 2;
        }
        // Cast guess back to integer (Newton\'s method guarantees it will converge to the integer part)
        return (int) guess;
    }
}
