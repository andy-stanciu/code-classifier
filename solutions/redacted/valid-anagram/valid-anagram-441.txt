class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int sumS = 0;
        long productS = 1;
        int sumT = 0;
        long productT = 1;
        for (int i = 0; i < s.length(); i++) {
            sumS += (int) s.charAt(i);
            productS *= (int) s.charAt(i);
            sumT += (int) t.charAt(i);
            productT *= (int) t.charAt(i);
        }
        return sumS == sumT && productS == productT;
    }
}
