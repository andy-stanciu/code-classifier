class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        int[] freq = new int[123];
        for (int i = 0; i < s.length(); i++) freq[s.charAt(i)]++;
        for (int i = 0; i < t.length(); i++) freq[t.charAt(i)]--;
        for (int i = 0; i < 123; i++) if (freq[i] != 0)
            return false;
        return true;
    }
}
