class Solution {

    public boolean method1(String s, String t) {
        // length
        if (s.length() != t.length())
            return false;
        //change to charArray
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        // sort
        Arrays.sort(a);
        Arrays.sort(b);
        // check value
        for (int i = 0; i < a.length; i++) {
            if (a[i] != b[i]) {
                return false;
            }
        }
        return true;
    }
}
