class Solution {

    public boolean method1(String s, String t) {
        //ANAGRAMS WILL HAVE SAME ORDER OF CHAR ARRAY WHEN SORTED
        char[] set1 = s.toCharArray();
        char[] set2 = t.toCharArray();
        Arrays.sort(set1);
        Arrays.sort(set2);
        if (Arrays.equals(set1, set2)) {
            return true;
        }
        return false;
    }
}
