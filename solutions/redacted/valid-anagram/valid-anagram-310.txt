class Solution {

    public boolean method1(String s, String t) {
        char[] a1 = s.toCharArray();
        Arrays.sort(a1);
        String s1 = new String(a1);
        char[] a2 = t.toCharArray();
        Arrays.sort(a2);
        String s2 = new String(a2);
        return s1.equals(s2);
    }
}
