class Solution {

    public boolean method1(String s, String t) {
        int s1 = s.length();
        int t1 = t.length();
        if (s1 == t1) {
            char[] sch = s.toCharArray();
            char[] tch = t.toCharArray();
            Arrays.sort(sch);
            Arrays.sort(tch);
            return Arrays.equals(sch, tch);
        }
        return false;
    }
}
