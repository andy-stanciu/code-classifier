public class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] al = new int[256];
        for (char ch : s.toCharArray()) {
            int idx = (int) ch;
            al[idx]++;
        }
        for (char ch : t.toCharArray()) {
            int idx = (int) ch;
            al[idx]--;
        }
        for (int i = 0; i < 256; i++) {
            if (al[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
