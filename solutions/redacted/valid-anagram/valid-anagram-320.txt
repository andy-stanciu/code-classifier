class Solution {

    public boolean method1(String s, String t) {
        boolean method1 = false;
        int[] array = new int[128];
        if (s.length() != t.length())
            return method1;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == t.charAt(i))
                continue;
            array[s.charAt(i)]++;
            array[t.charAt(i)]--;
        }
        return Arrays.stream(array).allMatch(c -> c == 0);
    }
}
