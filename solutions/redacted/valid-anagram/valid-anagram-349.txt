class Solution {

    public boolean method1(String s, String t) {
        Map<Character, Integer> hs = new HashMap<>();
        for (char x : s.toCharArray()) {
            hs.put(x, hs.getOrDefault(x, 0) + 1);
        }
        for (char x : t.toCharArray()) {
            hs.put(x, hs.getOrDefault(x, 0) - 1);
        }
        for (int val : hs.values()) {
            if (val != 0)
                return false;
        }
        return true;
    }
}
