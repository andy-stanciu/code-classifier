class Solution {

    public boolean method1(String s, String t) {
        if (t.length() != s.length())
            return false;
        HashMap<Character, Integer> h = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (h.containsKey(ch)) {
                int k = h.get(ch);
                h.replace(ch, ++k);
            } else
                h.put(ch, 1);
        }
        for (int i = 0; i < t.length(); i++) {
            char ch = t.charAt(i);
            if (h.containsKey(ch)) {
                int k = h.get(ch);
                h.replace(ch, --k);
            } else
                h.put(ch, 1);
        }
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (h.get(ch) != 0)
                return false;
        }
        return true;
    }
}
