class Solution {

    public boolean method1(String s, String t) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (char singleChar : s.toCharArray()) {
            map.put(singleChar, map.getOrDefault(singleChar, 0) + 1);
        }
        for (char singleChar : t.toCharArray()) {
            if (map.containsKey(singleChar)) {
                map.put(singleChar, map.get(singleChar) - 1);
            } else {
                return false;
            }
        }
        for (char key : map.keySet()) {
            if (map.get(key) != 0) {
                return false;
            }
        }
        return true;
    }
}
