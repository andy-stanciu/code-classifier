class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        int[] l1 = new int[s.length()];
        for (int i = 0; i < s.length(); i++) l1[i] = s.charAt(i);
        Arrays.sort(l1);
        int[] l2 = new int[t.length()];
        for (int i = 0; i < t.length(); i++) l2[i] = t.charAt(i);
        Arrays.sort(l2);
        boolean x = Arrays.equals(l1, l2);
        return x;
    }

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        Solution s = new Solution();
        System.out.print("Enter first string value: ");
        String a = scan.nextLine();
        System.out.print("Enter first string value: ");
        String b = scan.nextLine();
        System.out.print("The two strings are anagram: " + s.method1(a, b));
    }
}
