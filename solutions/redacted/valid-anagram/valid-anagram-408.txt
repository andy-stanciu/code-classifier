class Solution {

    public boolean method1(String s, String t) {
        int[] letters = new int[128];
        for (char c : s.toCharArray()) {
            letters[c]++;
        }
        for (char c : t.toCharArray()) {
            letters[c]--;
        }
        for (int letter : letters) {
            if (letter != 0)
                return false;
        }
        return true;
    }
}
