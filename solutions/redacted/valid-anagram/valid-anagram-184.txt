class Solution {

    public boolean method1(String s, String t) {
        // if both string have different length
        if (s.length() != t.length()) {
            return false;
        }
        HashMap<Character, Integer> hm = new HashMap<>();
        // put the elements in HashMap
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (hm.containsKey(ch)) {
                hm.put(ch, hm.get(ch) + 1);
            } else {
                hm.put(ch, 1);
            }
        }
        // remove the elements from HashMap
        for (int i = 0; i < t.length(); i++) {
            char ch = t.charAt(i);
            if (hm.containsKey(ch)) {
                if (hm.get(ch) == 1) {
                    hm.remove(ch);
                } else {
                    hm.put(ch, hm.get(ch) - 1);
                }
            } else {
                return false;
            }
        }
        // for(int key:hm.keySet()){
        //     if(hm.get(key)!=0){
        //         return false;
        //     }
        // }
        return hm.size() == 0 ? true : false;
    }
}
