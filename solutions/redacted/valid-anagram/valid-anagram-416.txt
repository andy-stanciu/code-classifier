class Solution {

    public boolean method1(String s, String t) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (String str : s.split("")) {
            map.put(str.charAt(0), map.getOrDefault(str.charAt(0), 0) + 1);
        }
        for (String str : t.split("")) {
            map.put(str.charAt(0), map.getOrDefault(str.charAt(0), 0) - 1);
            if (map.get(str.charAt(0)) < 0 || map.get(str.charAt(0)) == null)
                return false;
        }
        if (map.values().stream().filter(val -> val != 0).toList().isEmpty())
            return true;
        return false;
    }
}
