class Solution {

    public boolean method1(String s, String t) {
        Map<Character, Integer> result = new HashMap<>();
        for (char x : s.toCharArray()) {
            result.put(x, result.getOrDefault(x, 0) + 1);
        }
        for (char x : t.toCharArray()) {
            result.put(x, result.getOrDefault(x, 0) - 1);
        }
        for (int i : result.values()) {
            if (i != 0) {
                return false;
            }
        }
        return true;
    }
}
