class Solution {

    public boolean method1(String s, String t) {
        int n = s.length();
        int m = t.length();
        if (n != m)
            return false;
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        for (int i = 0; i < n; i++) {
            if (a[i] != b[i])
                return false;
        }
        return true;
    }
}
