class Solution {

    public boolean method1(String s, String t) {
        int n1 = s.length();
        int n2 = t.length();
        char[] arr1 = s.toCharArray();
        char[] arr2 = t.toCharArray();
        Arrays.sort(arr1);
        Arrays.sort(arr2);
        if (n1 != n1) {
            return false;
        }
        if (Arrays.equals(arr1, arr2)) {
            return true;
        }
        return false;
    }
}
