class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int a = s.length();
        //int b=t.length();
        char[] ch1 = s.toCharArray();
        char[] ch2 = t.toCharArray();
        Arrays.sort(ch1);
        Arrays.sort(ch2);
        int i = 0;
        int j = 0;
        while (i < a && j < a) {
            if (ch1[i] != ch2[j]) {
                return false;
            }
            i++;
            j++;
        }
        return true;
    }
}
