class Solution {

    public boolean method1(String s, String t) {
        // Edge case checking
        if (s == null && t != null)
            return false;
        if (t == null && s != null)
            return false;
        if (t.length() != s.length())
            return false;
        Map<Character, Integer> map = new HashMap<>();
        for (char key : s.toCharArray()) {
            map.merge(key, 1, Integer::sum);
        }
        for (char check : t.toCharArray()) {
            if (map.containsKey(check)) {
                map.put(check, map.get(check) - 1);
                if (map.get(check) == 0) {
                    map.remove(check);
                }
            }
        }
        return map.isEmpty();
    }
}
