class Solution {

    public boolean method1(String s, String t) {
        int chk = 0;
        int count = 0;
        List<Character> list = new ArrayList();
        if (s.length() == t.length()) {
            for (int i = 0; i < s.length(); i++) {
                if (list.size() == 0) {
                    char value = s.charAt(i);
                    list.add(value);
                    for (int j = 0; j < s.length(); j++) {
                        if (value == s.charAt(j)) {
                            count++;
                        }
                    }
                    for (int tIndex = 0; tIndex < t.length(); tIndex++) {
                        if (t.charAt(tIndex) == value) {
                            count--;
                        }
                    }
                    if (count == 0) {
                        chk = 1;
                    } else {
                        chk = 0;
                    }
                    if (chk == 0) {
                        break;
                    }
                } else {
                    int chkIn = 0;
                    char value = s.charAt(i);
                    //list.add(value);
                    for (char c : list) {
                        if (c == value) {
                            chkIn = 0;
                            break;
                        } else {
                            chkIn = 1;
                        }
                    }
                    if (chkIn == 1) {
                        list.add(value);
                        for (int j = i; j < s.length(); j++) {
                            if (value == s.charAt(j)) {
                                count++;
                            }
                        }
                        for (int tIndex = 0; tIndex < t.length(); tIndex++) {
                            if (t.charAt(tIndex) == value) {
                                count--;
                            }
                        }
                        if (count == 0) {
                            chk = 1;
                        } else {
                            chk = 0;
                        }
                        if (chk == 0) {
                            break;
                        }
                    }
                }
            }
            if (chk == 1) {
                return true;
            } else {
                return false;
            }
        } else {
            return false;
        }
    }
}
