class Solution {

    public boolean method1(String s, String t) {
        // chacking if the length of the string is same or  not
        if (s.length() != t.length())
            return false;
        //converting string s into a character array named "a"
        char[] a = s.toCharArray();
        //converting string t into character array named "b"
        char[] b = t.toCharArray();
        //sorting the character array "a"
        Arrays.sort(a);
        //sorting the character array "b"
        Arrays.sort(b);
        //running a loop for  comparing the two sorted array
        for (int i = 0; i < a.length; i++) {
            if (//comparing if the two sorted array have same character at the same index or not
            a[i] != b[i])
                return false;
        }
        return true;
    }
}
