class Solution {

    public boolean method1(String s, String t) {
        // not the same lenght, they can\'t be anagrams
        if (s.length() != t.length()) {
            return false;
        }
        // counts letters in 1 word
        Map<Character, Integer> map = new HashMap<>();
        for (char c : t.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        // decrease the counters and fails if we decrease too much, ie letters were not in the other string
        for (char c : s.toCharArray()) {
            if (map.getOrDefault(c, 0) == 0)
                return false;
            map.put(c, map.get(c) - 1);
        }
        // check if all letters have been consumed
        for (Integer nb : map.values()) {
            if (nb != 0)
                return false;
        }
        return true;
    }
}
