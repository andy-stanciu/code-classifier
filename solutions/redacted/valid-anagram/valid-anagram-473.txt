class Solution {

    public boolean method1(String s, String t) {
        int l1 = s.length();
        int l2 = t.length();
        if (l1 != l2) {
            return false;
        }
        char[] c1 = s.toCharArray();
        char[] c2 = t.toCharArray();
        Arrays.sort(c1);
        Arrays.sort(c2);
        if (Arrays.equals(c1, c2)) {
            return true;
        }
        return false;
    }
}
