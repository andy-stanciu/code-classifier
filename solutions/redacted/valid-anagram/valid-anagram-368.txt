class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        // Convert to char arrays
        char[] arrS = s.toCharArray();
        char[] arrT = t.toCharArray();
        // Sort the arrays
        Arrays.sort(arrS);
        Arrays.sort(arrT);
        // Directly compare the sorted arrays
        return Arrays.equals(arrS, arrT);
    }
}
