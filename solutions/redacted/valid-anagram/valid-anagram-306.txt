class Solution {

    public boolean method1(String s, String t) {
        char[] ch = s.toCharArray();
        char[] dh = t.toCharArray();
        Arrays.sort(ch);
        Arrays.sort(dh);
        if (s.length() != t.length()) {
            return false;
        }
        return Arrays.equals(ch, dh);
    }
}
