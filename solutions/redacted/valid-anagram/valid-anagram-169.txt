public class Solution {

    public static boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            // Anagrams must be of the same length
            return false;
        }
        // Use a HashMap to count character frequencies in string s
        Map<Character, Integer> charCount = new HashMap<>();
        // Count characters in s
        for (char c : s.toCharArray()) {
            charCount.put(c, charCount.getOrDefault(c, 0) + 1);
        }
        // Subtract character counts using string t
        for (char c : t.toCharArray()) {
            if (!charCount.containsKey(c) || charCount.get(c) == 0) {
                return false;
            }
            charCount.put(c, charCount.get(c) - 1);
        }
        return true;
    }

    public static void main(String[] args) {
        String s1 = "anagram";
        String t1 = "nagaram";
        // Output: true
        System.out.println(method1(s1, t1));
        String s2 = "rat";
        String t2 = "car";
        // Output: false
        System.out.println(method1(s2, t2));
    }
}
