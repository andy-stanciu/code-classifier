class Solution {

    public boolean method1(String s, String t) {
        // if the length of both strings is not equal, it cannot be an anagram --> return false
        if (s.length() != t.length())
            return false;
        int n = s.length();
        // Convert the string to char array
        char[] s1 = s.toCharArray();
        char[] t1 = t.toCharArray();
        // Sort the arrays
        Arrays.sort(s1);
        Arrays.sort(t1);
        for (int i = 0; i < n; i++) {
            // If both sorted arrays are not equal --> return false
            if (s1[i] != t1[i])
                return false;
        }
        return true;
    }
}
