class Solution {

    public boolean method1(String s, String t) {
        char[] arr1 = s.toCharArray();
        char[] arr2 = t.toCharArray();
        Arrays.sort(arr1);
        Arrays.sort(arr2);
        boolean flag = false;
        if (Arrays.equals(arr1, arr2)) {
            //if arr2 contains all characters of arr1
            // make flag true that means it is  an anagram
            flag = true;
        }
        //else return flase that means  not an anagram
        return flag;
    }
}
