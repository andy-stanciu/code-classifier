class Solution {

    public boolean method1(String s, String t) {
        String s1 = s.toLowerCase();
        String t1 = t.toLowerCase();
        char[] a = s1.toCharArray();
        char[] b = t1.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        String str1 = new String(a);
        String str2 = new String(b);
        if (str1.equals(str2)) {
            return true;
        }
        return false;
    }
}
