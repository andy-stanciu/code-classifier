class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        char[] arr = s.toCharArray();
        Arrays.sort(arr);
        char[] ar = t.toCharArray();
        Arrays.sort(ar);
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] != ar[i]) {
                return false;
            }
        }
        return true;
    }
}
