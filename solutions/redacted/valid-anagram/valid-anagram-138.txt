class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        int i, freq;
        int n = s.length();
        for (i = 97; i <= 122; i++) {
            freq = 0;
            for (int j = 0; j < n; j++) {
                if (s.charAt(j) == i)
                    freq++;
                if (t.charAt(j) == i)
                    freq--;
            }
            if (freq != 0)
                return false;
        }
        return true;
    }
}
