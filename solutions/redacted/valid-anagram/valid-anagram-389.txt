class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        Map<Character, Integer> hp1 = new HashMap<>();
        Map<Character, Integer> hp2 = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            hp1.put(s.charAt(i), hp1.getOrDefault(s.charAt(i), 0) + 1);
            hp2.put(t.charAt(i), hp2.getOrDefault(t.charAt(i), 0) + 1);
        }
        return hp1.equals(hp2);
    }
}
