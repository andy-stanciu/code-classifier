class Solution {

    public boolean method1(String s, String t) {
        //convert--> String --> Array
        char[] ch = s.toCharArray();
        char[] ar = t.toCharArray();
        // Sort---> Both of the converted Array
        Arrays.sort(ch);
        Arrays.sort(ar);
        // Compare both of them and return true or false
        return Arrays.equals(ch, ar);
    }
}
