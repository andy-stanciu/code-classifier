class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] array = new int[26];
        for (int i = 0; i < s.length(); i++) {
            array[(int) s.charAt(i) - 97]++;
        }
        for (int i = 0; i < t.length(); i++) {
            array[(int) t.charAt(i) - 97]--;
        }
        for (int i = 0; i < 26; i++) {
            if (array[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
