class Solution {

    public boolean method1(String s, String t) {
        Map<Character, Integer> countMap = new HashMap<>();
        Map<Character, Integer> sCountMap = new HashMap<>();
        if (s.length() != t.length()) {
            return false;
        }
        for (int i = 0; i < s.length(); i++) {
            if (!countMap.containsKey(s.charAt(i))) {
                countMap.put(s.charAt(i), 1);
            } else {
                countMap.put(s.charAt(i), countMap.get(s.charAt(i)) + 1);
            }
        }
        for (int i = 0; i < t.length(); i++) {
            if (!sCountMap.containsKey(t.charAt(i))) {
                sCountMap.put(t.charAt(i), 1);
            } else {
                sCountMap.put(t.charAt(i), sCountMap.get(t.charAt(i)) + 1);
            }
        }
        return countMap.equals(sCountMap);
    }
}
