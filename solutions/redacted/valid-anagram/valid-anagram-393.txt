class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length())
            return false;
        HashMap<Character, Integer> elements = new HashMap<>();
        for (char c : s.toCharArray()) {
            elements.put(c, elements.getOrDefault(c, 0) + 1);
        }
        for (char c : t.toCharArray()) {
            if (elements.containsKey(c)) {
                elements.put(c, elements.get(c) - 1);
                if (elements.get(c) == 0) {
                    elements.remove(c);
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
