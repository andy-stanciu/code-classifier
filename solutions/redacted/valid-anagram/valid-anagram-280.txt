class Solution {

    public boolean method1(String s, String t) {
        s = s.toLowerCase();
        t = t.toLowerCase();
        char[] c1 = s.toCharArray();
        char[] c2 = t.toCharArray();
        Arrays.sort(c1);
        Arrays.sort(c2);
        s = String.valueOf(c1);
        t = String.valueOf(c2);
        if (s.equals(t)) {
            return true;
        } else {
            return false;
        }
    }
}
