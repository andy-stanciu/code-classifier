class Solution {

    public boolean method1(String s, String t) {
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        String ss = new String(a);
        String tt = new String(b);
        Set<String> set = new HashSet<>();
        set.add(ss);
        set.add(tt);
        if (set.size() == 1) {
            return true;
        }
        return false;
    }
}
