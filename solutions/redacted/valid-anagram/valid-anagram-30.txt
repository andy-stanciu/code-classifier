class Solution {

    public boolean method1(String s, String t) {
        char[] charS = s.toCharArray();
        char[] charT = t.toCharArray();
        Arrays.sort(charS);
        Arrays.sort(charT);
        String stringS = new String(charS);
        String stringT = new String(charT);
        return stringS.equals(stringT);
    }
}
