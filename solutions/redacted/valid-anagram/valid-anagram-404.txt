class Solution {

    public boolean method1(String s, String t) {
        char[] a = s.toCharArray();
        char[] b = t.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        if (a.length != b.length) {
            return false;
        }
        int i = 0;
        boolean ana = true;
        for (i = 0; i < a.length; i++) {
            if (a[i] != b[i]) {
                ana = false;
                break;
            }
        }
        return ana;
    }
}
