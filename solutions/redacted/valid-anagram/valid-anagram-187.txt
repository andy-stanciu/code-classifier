class Solution {

    public boolean method1(String s, String t) {
        char[] ch1 = s.toCharArray();
        char[] ch2 = t.toCharArray();
        Arrays.sort(ch1);
        Arrays.sort(ch2);
        String newS1 = new String(ch1);
        String newS2 = new String(ch2);
        if (newS1.length() == newS2.length()) {
            if (newS1.contains(newS2)) {
                return true;
            }
        }
        return false;
    }
}
