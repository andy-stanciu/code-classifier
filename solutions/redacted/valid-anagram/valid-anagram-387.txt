class Solution {

    public boolean method1(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        // Array to store frequency of characters
        int[] freqTable = new int[256];
        for (int i = 0; i < s.length(); i++) {
            freqTable[s.charAt(i)]++;
            freqTable[t.charAt(i)]--;
        }
        for (int i = 0; i < 256; i++) {
            if (freqTable[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
