class Solution {

    public boolean method1(String s, String t) {
        Map<Character, Integer> mapS = new HashMap<Character, Integer>();
        Map<Character, Integer> mapT = new HashMap<Character, Integer>();
        char[] charS = s.toCharArray();
        char[] charT = t.toCharArray();
        if (charS.length != charT.length) {
            return false;
        }
        for (char sc : charS) {
            mapS.put(sc, mapS.containsKey(sc) ? (mapS.get(sc) + 1) : 1);
        }
        for (char tc : charT) {
            mapT.put(tc, mapT.containsKey(tc) ? (mapT.get(tc) + 1) : 1);
        }
        for (Map.Entry<Character, Integer> entry : mapS.entrySet()) {
            char key = entry.getKey();
            if (!mapT.containsKey(key) || (int) mapT.get(key) != (int) entry.getValue()) {
                return false;
            }
        }
        return true;
    }
}
