class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val == key) {
            if (root.right != null) {
                TreeNode var1 = root.right;
                while (var1.left != null) {
                    var1 = var1.left;
                }
                var1.left = root.left;
            }
            if (root.right == null)
                return root.left;
            return root.right;
        }
        if (root.val > key)
            root.left = method1(root.left, key);
        if (root.val < key)
            root.right = method1(root.right, key);
        return root;
    }
}
