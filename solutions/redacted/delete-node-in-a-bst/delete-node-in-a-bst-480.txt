/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        return method2(root, key);
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            if (root.left == null)
                return root.right;
            if (root.right == null)
                return root.left;
            TreeNode var1 = root.left;
            TreeNode var2 = root.right;
            TreeNode temp = method3(root.right);
            root.val = temp.val;
            root.right = method2(root.right, root.val);
            return root;
        } else if (key > root.val) {
            root.right = method2(root.right, key);
        } else {
            root.left = method2(root.left, key);
        }
        return root;
    }

    public TreeNode method3(TreeNode root) {
        while (root.left != null) root = root.left;
        return root;
    }
}
