/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // iterative
    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val == key)
            return method1(root);
        TreeNode var1 = root;
        while (var1 != null) {
            if (var1.left != null && var1.left.val == key)
                var1.left = method1(var1.left);
            else if (var1.right != null && var1.right.val == key)
                var1.right = method1(var1.right);
            else
                var1 = key < var1.val ? var1.left : var1.right;
        }
        return root;
    }

    private TreeNode method1(TreeNode var1) {
        TreeNode left = var1.left;
        TreeNode right = var1.right;
        if (right == null)
            return left;
        else if (left == null)
            return right;
        else {
            TreeNode leftMost = right;
            while (leftMost.left != null) leftMost = leftMost.left;
            leftMost.left = left;
            return right;
        }
    }

    // recursive
    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val == key) {
            TreeNode left = root.left;
            TreeNode right = root.right;
            if (right == null)
                return left;
            else if (left == null)
                return right;
            else {
                TreeNode leftMost = right;
                while (leftMost.left != null) leftMost = leftMost.left;
                leftMost.left = left;
                return right;
            }
        } else {
            if (key < root.val)
                root.left = method1(root.left, key);
            else
                root.right = method1(root.right, key);
        }
        return root;
    }
}
