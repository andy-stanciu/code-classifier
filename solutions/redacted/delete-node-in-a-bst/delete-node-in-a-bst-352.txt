/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            // Node with the key to delete is found
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            // Node to delete has two children
            // Find the minimum value in the right subtree
            TreeNode var1 = root.right;
            while (var1.left != null) {
                var1 = var1.left;
            }
            // Update the current node\'s value with the minimum value found
            root.val = var1.val;
            // Recursively delete the minimum node from the right subtree
            root.right = method1(root.right, var1.val);
        }
        return root;
    }
}
