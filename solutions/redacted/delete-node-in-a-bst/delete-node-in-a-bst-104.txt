/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int x) {
        if (root == null)
            return null;
        if (root.val > x)
            root.left = method1(root.left, x);
        else if (root.val < x)
            root.right = method1(root.right, x);
        else {
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            TreeNode succ = getSucc(root);
            root.val = succ.val;
            root.right = method1(root.right, succ.val);
        }
        return root;
    }

    private TreeNode getSucc(TreeNode root) {
        TreeNode curr = root.right;
        while (curr != null && curr.left != null) curr = curr.left;
        return curr;
    }
}
