/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val > key) {
            root.left = method1(root.left, key);
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else // Reached at the key
        {
            // When we have just one right child
            if (root.left == null)
                return root.right;
            else // when we have just one left child
            if (root.right == null)
                return root.left;
            else // when we have both the child
            {
                TreeNode var1 = method2(root.right);
                root.val = var1.val;
                root.right = method1(root.right, var1.val);
            }
        }
        return root;
    }

    public TreeNode method2(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }
}
