/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root.val;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key < root.val) {
            root.left = method2(root.left, key);
        } else if (key > root.val) {
            root.right = method2(root.right, key);
        } else {
            // founded
            // Case 1 -At leaf Node
            if (root.left == null && root.right == null) {
                return null;
            }
            // Case 2
            if (root.left == null) {
                //1 node in right only
                return root.right;
            } else if (root.right == null) {
                //1 node in left only
                return root.left;
            } else // Case 3 - Both right and left have val
            {
                root.val = method1(root.right);
                root.right = method2(root.right, root.val);
            }
        }
        return root;
    }
}
