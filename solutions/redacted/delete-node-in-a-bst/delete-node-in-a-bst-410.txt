/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int smallest(TreeNode root) {
        int var1 = root.val;
        while (root.left != null) {
            root = root.left;
            var1 = root.val;
        }
        return var1;
    }

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val) {
            if (root.left != null)
                root.left = method1(root.left, key);
        } else if (key > root.val) {
            if (root.right != null)
                root.right = method1(root.right, key);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            root.val = smallest(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }
}
