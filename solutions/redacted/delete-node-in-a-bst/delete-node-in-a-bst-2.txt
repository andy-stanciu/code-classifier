class Solution {

    public TreeNode method1(TreeNode root, int key) {
        //search
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        } else if (root.val > key) {
            root.left = method1(root.left, key);
        } else {
            //case 1: leaf node
            if (root.left == null && root.right == null) {
                return null;
            }
            //case 2 : one child
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            //case 3 : two children
            TreeNode IS = method2(root.right);
            root.val = IS.val;
            root.right = method1(root.right, IS.val);
        }
        return root;
    }

    private TreeNode method2(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }
}
