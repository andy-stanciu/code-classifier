/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int target) {
        if (root == null) {
            return null;
        }
        if (root.val > target) {
            root.left = method1(root.left, target);
        } else if (root.val < target) {
            root.right = method1(root.right, target);
        } else {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                int leftSubTreeMax = method2(root.left);
                root.val = leftSubTreeMax;
                root.left = method1(root.left, leftSubTreeMax);
            }
        }
        return root;
    }

    private int method2(TreeNode root) {
        if (root == null) {
            return Integer.MIN_VALUE;
        }
        int right = method2(root.right);
        return Math.max(root.val, right);
    }
}
