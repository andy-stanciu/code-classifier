/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int x) {
        if (root == null)
            return root;
        if (root.val > x)
            root.left = method1(root.left, x);
        else if (root.val < x)
            root.right = method1(root.right, x);
        else {
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            root.val = value(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    public static int value(TreeNode root) {
        int val = root.val;
        while (root.left != null) {
            val = root.left.val;
            root = root.left;
        }
        return val;
    }
}
