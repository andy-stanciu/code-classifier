class Solution {

    public TreeNode method1(TreeNode root) {
        if (root.left == null) {
            return root;
        } else {
            return method1(root.left);
        }
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val > key) {
            root.left = method2(root.left, key);
        } else if (key > root.val) {
            root.right = method2(root.right, key);
        } else {
            if (root.left != null && root.right != null) {
                // Deleting node which have 2 children
                TreeNode var1 = method1(root.right);
                root.val = var1.val;
                root.right = method2(root.right, var1.val);
            } else if (root.left != null) {
                // have only left child
                root = root.left;
            } else if (root.right != null) {
                // have only right child
                root = root.right;
            } else {
                // deleting leaf node
                root = null;
            }
        }
        return root;
    }
}
