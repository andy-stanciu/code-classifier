/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static TreeNode method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val > key) {
            root.left = method2(root.left, key);
        } else if (root.val < key) {
            root.right = method2(root.right, key);
        } else {
            // key == root.data;
            //case 1  ---> while both node null
            if (root.left == null && root.right == null) {
                return null;
            } else //case - 2  -----> while a single node is null
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else //case - 3  -----> while it has both nodes
            {
                TreeNode ios = method1(root.right);
                root.val = ios.val;
                root.right = method2(root.right, ios.val);
            }
        }
        return root;
    }
}
