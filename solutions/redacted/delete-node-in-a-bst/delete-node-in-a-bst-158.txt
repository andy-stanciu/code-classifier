/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            // Case 1: Node with only one child or no child
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            }
            // Case 2: Node with two children
            root.val = method2(root.left);
            root.left = method1(root.left, root.val);
        }
        return root;
    }

    // Why root->right bcz root ke right me subse maximum hugi isko hi replace krna ha
    private int method2(TreeNode root) {
        while (root.right != null) {
            root = root.right;
        }
        return root.val;
    }
}
