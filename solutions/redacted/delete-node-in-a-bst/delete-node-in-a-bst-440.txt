/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val == key) {
            // for no children case
            if (root.left == null && root.right == null)
                return null;
            // for 1 child case
            if (root.left != null && root.right == null)
                return root.left;
            if (root.right != null && root.left == null)
                return root.right;
            //for 2 children case
            TreeNode var1 = helper(root, root.right);
            root.val = var1.val;
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else {
            root.left = method1(root.left, key);
        }
        return root;
    }

    public TreeNode helper(TreeNode parent, TreeNode root) {
        TreeNode var2 = parent, curr = root;
        while (curr.left != null) {
            var2 = curr;
            curr = curr.left;
        }
        method1(var2, curr.val);
        return curr;
    }
}
