class Solution {

    public TreeNode method1(TreeNode param1, int key) {
        if (// base case
        param1 == null)
            return null;
        if (param1.val > key)
            // left call bst condition
            param1.left = method1(param1.left, key);
        else if (param1.val < key)
            // right call bst condition
            param1.right = method1(param1.right, key);
        else {
            if (param1.left != null && param1.right != null) {
                // both child present
                // find max from left
                int lmax = max(param1.left);
                // replace data of node with max
                param1.val = lmax;
                // remove max
                param1.left = method1(param1.left, lmax);
                return param1;
            } else if (// remove when only left child present
            param1.left != null)
                return param1.left;
            else if (// remove when only right child present
            param1.right != null)
                return param1.right;
            else
                // leaf remove
                return null;
        }
        return param1;
    }

    public int max(TreeNode param1) {
        // returns max
        if (param1.right != null)
            return max(param1.right);
        else
            return param1.val;
    }
}
