/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode param1, int value) {
        if (param1 == null)
            return null;
        if (value < param1.val) {
            param1.left = method1(param1.left, value);
        } else if (value > param1.val) {
            param1.right = method1(param1.right, value);
        } else {
            if (param1.left == null && param1.right == null) {
                return null;
            } else if (param1.left == null) {
                param1 = param1.right;
            } else if (param1.right == null) {
                param1 = param1.left;
            } else {
                int var1 = method2(param1.right);
                param1.val = var1;
                param1.right = method1(param1.right, var1);
            }
        }
        return param1;
    }

    public int method2(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root.val;
    }
}
