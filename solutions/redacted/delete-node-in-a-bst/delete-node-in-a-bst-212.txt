class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val == key) {
            if (root.right == null)
                return root.left;
            TreeNode prev = root.right;
            while (prev.left != null) {
                prev = prev.left;
            }
            prev.left = root.left;
            return root.right;
        }
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else {
            root.right = method1(root.right, key);
        }
        return root;
    }
}
