/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val > key) {
            root.left = method1(root.left, key);
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else {
            //Found the key to delete
            //case1 - no child
            if (root.left == null && root.right == null) {
                return null;
            }
            //Case 2 -  when there is 1 child
            if (root.left == null) {
                return root.right;
            }
            if (root.right == null) {
                return root.left;
            }
            //Case 2
            TreeNode IS = immediateSuccessor(root.right);
            root.val = IS.val;
            root.right = method1(root.right, IS.val);
        }
        return root;
    }

    public TreeNode immediateSuccessor(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }
}
