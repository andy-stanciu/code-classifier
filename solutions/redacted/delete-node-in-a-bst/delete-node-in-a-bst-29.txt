public class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        TreeNode temp = null;
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            if (root.left == null) {
                temp = root.right;
                root.right = null;
                return temp;
            } else if (root.right == null) {
                temp = root.left;
                root.left = null;
                return temp;
            }
            temp = successor(root.right);
            root.val = temp.val;
            root.right = method1(root.right, temp.val);
        }
        return root;
    }

    public static TreeNode successor(TreeNode cur) {
        while (cur.left != null) {
            cur = cur.left;
        }
        return cur;
    }
}
