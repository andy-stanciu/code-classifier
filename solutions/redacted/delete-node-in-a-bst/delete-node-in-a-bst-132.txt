/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode param1) {
        int max = Integer.MIN_VALUE;
        while (param1 != null) {
            max = Math.max(max, param1.val);
            param1 = param1.right;
        }
        return max;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key > root.val) {
            root.right = method2(root.right, key);
        } else if (key < root.val) {
            root.left = method2(root.left, key);
        } else {
            if (root.left == null && root.right == null) {
                return null;
            } else if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                int max = method1(root.left);
                root.val = max;
                root.left = method2(root.left, max);
                return root;
            }
        }
        return root;
    }
}
