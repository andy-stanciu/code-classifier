/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int method1(TreeNode root) {
        TreeNode curr = root;
        while (curr.left != null) {
            curr = curr.left;
        }
        return curr.val;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val > key) {
            root.left = method2(root.left, key);
        } else if (root.val < key) {
            root.right = method2(root.right, key);
        } else {
            if (root.left == null && root.right == null)
                return null;
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            root.val = method1(root.right);
            root.right = method2(root.right, root.val);
        }
        return root;
    }
}
