/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static TreeNode method1(TreeNode root) {
        if (root.left == null)
            return root;
        return method1(root.left);
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key > root.val)
            root.right = method2(root.right, key);
        else if (key < root.val)
            root.left = method2(root.left, key);
        else {
            if (root.left == null && root.right == null)
                return null;
            else if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            else {
                TreeNode temp = method1(root.right);
                root.val = temp.val;
                root.right = method2(root.right, temp.val);
            }
        }
        return root;
    }
}
