/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method1(root.right, key);
        } else if (root.val > key) {
            root.left = method1(root.left, key);
        } else {
            if (root.left != null && root.right != null) {
                // find the max node from left side of that node
                int lmax = method2(root.left);
                // replace this node with lmax
                root.val = lmax;
                // then remove that lmax node from the tree
                root.left = method1(root.left, lmax);
                return root;
            } else if (root.left != null) {
                return root.left;
            } else if (root.right != null) {
                return root.right;
            } else {
                return null;
            }
        }
        return root;
    }

    public int method2(TreeNode param1) {
        while (param1.right != null) {
            param1 = param1.right;
        }
        return param1.val;
    }
}
