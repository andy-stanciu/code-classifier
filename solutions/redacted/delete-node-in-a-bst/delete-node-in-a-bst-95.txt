/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MAX_VALUE;

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        // post-order Traversal
        root.left = method1(root.left, key);
        root.right = method1(root.right, key);
        if (root.val == key) {
            // save the current node (saved as Del) and make the current node as null
            TreeNode Del = root;
            root = null;
            // firstcase : find the delete node of right subtree\'s smallest val node and change the delete node val as smallest val node ;
            if (Del.right != null && Del.left != null) {
                method3(Del.right);
                Del.val = var1;
                Del.right = method2(Del.right, var1);
                root = Del;
            } else //secondcase: if node has left child and not right child change node as node.left
            if (Del.left != null && Del.right == null) {
                Del = Del.left;
                root = Del;
            } else //thirdcase: if node has right child and not left child change node as node.right
            if (Del.left == null && Del.right != null) {
                Del = Del.right;
                root = Del;
            }
        }
        return root;
    }

    private TreeNode method2(TreeNode param1, int var1) {
        if (param1 == null) {
            return param1;
        }
        param1.left = method2(param1.left, var1);
        param1.right = method2(param1.right, var1);
        // before deleting that check the node has children or not
        //if it has just skip the node
        if (var1 == param1.val) {
            if (param1.right != null) {
                return param1.right;
            }
            if (param1.left != null) {
                return param1.left;
            }
            param1 = null;
        }
        return param1;
    }

    // this is function is used for finding the smallest node of a
    // subtree
    private void method3(TreeNode param1) {
        if (param1 == null) {
            return;
        }
        if (param1.val < var1) {
            var1 = param1.val;
        }
        method3(param1.left);
        method3(param1.right);
    }
}
