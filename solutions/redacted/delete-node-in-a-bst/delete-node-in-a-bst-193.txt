class Solution {

    public TreeNode method1(TreeNode param1, int key) {
        if (param1 == null)
            return null;
        if (param1.val == key) {
            if (param1.right == null)
                return param1.left;
            // get left most node of right subtree
            TreeNode temp = param1.right;
            while (temp.left != null) {
                temp = temp.left;
            }
            temp.left = param1.left;
            return param1.right;
        } else if (param1.val > key) {
            param1.left = method1(param1.left, key);
        } else {
            param1.right = method1(param1.right, key);
        }
        return param1;
    }
}
