class Solution {

    public TreeNode method1(TreeNode root) {
        if (root.left == null) {
            return root.right;
        } else if (root.right == null) {
            return root.left;
        } else {
            TreeNode rightChild = root.right;
            TreeNode lastRight = method2(root.left);
            lastRight.right = rightChild;
            return root.left;
        }
    }

    public TreeNode method2(TreeNode root) {
        if (root.right == null) {
            return root;
        }
        return method2(root.right);
    }

    public TreeNode method3(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val == key) {
            return method1(root);
        }
        TreeNode dummy = root;
        while (root != null) {
            if (root.val > key) {
                if (root.left != null && root.left.val == key) {
                    root.left = method1(root.left);
                    break;
                } else {
                    root = root.left;
                }
            } else if (root.val < key) {
                if (root.right != null && root.right.val == key) {
                    root.right = method1(root.right);
                    break;
                } else {
                    root = root.right;
                }
            }
        }
        return dummy;
    }
}
