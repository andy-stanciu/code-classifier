class Solution {

    public static TreeNode method1(TreeNode root) {
        while (root != null && root.left != null) {
            root = root.left;
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        // Base case: if the root is null, return null
        if (root == null)
            return null;
        // searching for the key
        if (root.val < key) {
            root.right = method2(root.right, key);
        } else if (root.val > key) {
            root.left = method2(root.left, key);
        } else {
            // Case 1 - Leaf Node
            if (root.left == null && root.right == null)
                return null;
            else // Case 2 - Single Child
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            else // Case 3 - Both Children
            {
                TreeNode succ = method1(root.right);
                root.val = succ.val;
                root.right = method2(root.right, succ.val);
            }
        }
        return root;
    }
}
