/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int ans = root.val;
        while (root.left != null) {
            root = root.left;
            ans = root.val;
        }
        return ans;
    }

    public TreeNode helper(TreeNode root, int key) {
        if (root == null)
            return null;
        if (root.val > key) {
            root.left = helper(root.left, key);
        } else if (root.val < key) {
            root.right = helper(root.right, key);
        } else {
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            else {
                int val1 = method1(root.right);
                root.val = val1;
                root.right = helper(root.right, val1);
            }
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        return helper(root, key);
    }
}
