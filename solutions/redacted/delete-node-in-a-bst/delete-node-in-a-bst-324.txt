/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (key > root.val) {
            root.right = method1(root.right, key);
        } else if (key < root.val) {
            root.left = method1(root.left, key);
        } else {
            // Check childs
            if (root.left == null)
                return root.right;
            if (root.right == null)
                return root.left;
            // Have both childs, use the successor in-order approach
            root.val = this.getSuccessorValue(root);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    private int getSuccessorValue(TreeNode param1) {
        param1 = param1.right;
        while (param1.left != null) {
            param1 = param1.left;
        }
        return param1.val;
    }
}
