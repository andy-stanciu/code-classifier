class Solution {

    public TreeNode method1(TreeNode root, int key) {
        return helper(root, key);
    }

    public TreeNode helper(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val == key) {
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else {
                TreeNode temp = method2(root.right);
                root.val = temp.val;
                root.right = helper(root.right, temp.val);
            }
        } else if (root.val > key) {
            root.left = helper(root.left, key);
        } else {
            root.right = helper(root.right, key);
        }
        return root;
    }

    private TreeNode method2(TreeNode param1) {
        while (param1.left != null) {
            param1 = param1.left;
        }
        return param1;
    }
}
