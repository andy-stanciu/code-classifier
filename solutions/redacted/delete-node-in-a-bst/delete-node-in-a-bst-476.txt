class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        return method2(root, key);
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return null;
        if (key < root.val)
            root.left = method2(root.left, key);
        else if (key > root.val)
            root.right = method2(root.right, key);
        else
            return method3(root);
        return root;
    }

    public TreeNode method3(TreeNode root) {
        if (root.left == null) {
            return root.right;
        } else if (root.right == null) {
            return root.left;
        } else {
            TreeNode temp = method4(root.right);
            root.val = temp.val;
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    private TreeNode method4(TreeNode param1) {
        while (param1.left != null) {
            param1 = param1.left;
        }
        return param1;
    }
}
