/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            // Node found - now delete it
            // Case 1: Node has no child (leaf node)
            if (root.left == null && root.right == null) {
                return null;
            } else // Case 2: Node has only one child (either left or right)
            if (root.left == null) {
                return root.right;
            } else if (root.right == null) {
                return root.left;
            } else // Case 3: Node has two children
            {
                TreeNode var1 = method2(root.right);
                root.val = var1.val;
                root.right = method1(root.right, var1.val);
            }
        }
        return root;
    }

    private TreeNode method2(TreeNode param1) {
        while (param1.left != null) {
            param1 = param1.left;
        }
        return param1;
    }
}
