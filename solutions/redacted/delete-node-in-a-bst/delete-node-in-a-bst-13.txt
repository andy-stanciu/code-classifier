class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return root;
        }
        if (root.val == key) {
            // no child
            if (root.left == null && root.right == null) {
                return null;
            } else if (root.right == null || root.left == null) {
                // 1 child
                return root.right == null ? root.left : root.right;
            } else {
                // both child. find left most node of right child.
                TreeNode var1 = root;
                var1 = var1.right;
                while (var1.left != null) {
                    var1 = var1.left;
                }
                //update value
                root.val = var1.val;
                //now delete the leaf node of right child.
                root.right = method1(root.right, root.val);
            }
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else {
            root.left = method1(root.left, key);
        }
        return root;
    }
}
