class Solution {

    public TreeNode method1(TreeNode root, int key) {
        // case 1
        if (root == null) {
            return null;
        }
        if (root.val > key) {
            root.left = method1(root.left, key);
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else // case 2
        {
            if (root.right == null) {
                return root.left;
            } else if (root.left == null) {
                return root.right;
            } else // case 3
            {
                TreeNode Is = method2(root.right);
                root.val = Is.val;
                root.right = method1(root.right, Is.val);
            }
        }
        return root;
    }

    public TreeNode method2(TreeNode root) {
        if (root == null) {
            return null;
        }
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }
}
