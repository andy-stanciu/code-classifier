/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        TreeNode dummy = root;
        if (root.val == key) {
            return helper(root);
        }
        while (root != null) {
            if (key > root.val) {
                if (root.right != null && root.right.val == key) {
                    root.right = helper(root.right);
                } else {
                    root = root.right;
                }
            } else {
                if (root.left != null && root.left.val == key) {
                    root.left = helper(root.left);
                } else {
                    root = root.left;
                }
            }
        }
        return dummy;
    }

    private static TreeNode helper(TreeNode param1) {
        if (param1.left == null)
            return param1.right;
        if (param1.right == null)
            return param1.left;
        TreeNode var1 = param1.left;
        TreeNode var2 = param1.right;
        TreeNode attach = getRightMost(var1);
        attach.right = var2;
        return var1;
    }

    private static TreeNode getRightMost(TreeNode param1) {
        while (param1.right != null) param1 = param1.right;
        return param1;
    }
}
