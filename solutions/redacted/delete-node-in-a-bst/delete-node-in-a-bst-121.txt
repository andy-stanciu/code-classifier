/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        if (key < root.val) {
            root.left = method1(root.left, key);
        } else if (key > root.val) {
            root.right = method1(root.right, key);
        } else {
            //3 condition to delete
            if (root.left == null)
                return root.right;
            else if (root.right == null)
                return root.left;
            //right side se min element to replace and delete
            root.val = method2(root.right);
            root.right = method1(root.right, root.val);
        }
        return root;
    }

    static int method2(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root.val;
    }
}
