/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int maxi(TreeNode root) {
        if (root.right != null) {
            return maxi(root.right);
        } else {
            return root.val;
        }
    }

    public TreeNode method1(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (key > root.val) {
            root.right = method1(root.right, key);
        } else if (key < root.val) {
            root.left = method1(root.left, key);
        } else {
            if (root.left != null && root.right != null) {
                int max = maxi(root.left);
                root.val = max;
                root.left = method1(root.left, max);
                return root;
            } else if (root.left != null) {
                return root.left;
            } else if (root.right != null) {
                return root.right;
            } else {
                return null;
            }
        }
        System.gc();
        return root;
    }
}
