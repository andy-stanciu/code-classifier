/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (key < root.val)
            root.left = method1(root.left, key);
        else if (key > root.val)
            root.right = method1(root.right, key);
        else {
            TreeNode var1 = var1(root.right);
            if (var1 == null)
                return root.left;
            root.right = method2(root.right);
            var1.right = root.right;
            var1.left = root.left;
            root = var1;
        }
        return root;
    }

    private TreeNode method2(TreeNode root) {
        if (root == null)
            return null;
        if (root.left == null)
            return root.right;
        root.left = method2(root.left);
        return root;
    }

    private TreeNode var1(TreeNode root) {
        if (root == null)
            return null;
        if (root.left == null)
            return root;
        return var1(root.left);
    }
}
