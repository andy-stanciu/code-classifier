/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        if (key > root.val) {
            root.right = method1(root.right, key);
        } else if (key < root.val) {
            root.left = method1(root.left, key);
        } else {
            //if we want to delete leaf node
            if (root.left == null && root.right == null) {
                return null;
            }
            //if root has one child
            if (root.left == null) {
                return root.right;
            }
            if (root.right == null) {
                return root.left;
            }
            //voila case
            TreeNode IS = method2(root.right);
            if (IS != null) {
                root.val = IS.val;
                root.right = method1(root.right, IS.val);
            }
        }
        return root;
    }

    public static TreeNode method2(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }
}
