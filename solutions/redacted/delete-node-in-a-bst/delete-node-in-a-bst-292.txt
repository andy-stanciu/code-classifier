/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        return method2(root, key);
    }

    private TreeNode method2(TreeNode root, int key) {
        if (root == null)
            return root;
        if (root.val < key)
            root.right = method2(root.right, key);
        if (root.val > key)
            root.left = method2(root.left, key);
        if (root.val == key) {
            if (root.left == null) {
                root = root.right;
            } else if (root.right == null) {
                root = root.left;
            } else {
                TreeNode var1 = method3(root.right);
                root.val = var1.val;
                root.right = method2(root.right, root.val);
            }
        }
        return root;
    }

    private TreeNode method3(TreeNode root) {
        if (root.left != null) {
            return method3(root.left);
        }
        return root;
    }
}
