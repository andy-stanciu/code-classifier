class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return null;
        else if (// case 1
        root.val > key)
            root.left = method1(root.left, key);
        else if (// case 2
        root.val < key)
            root.right = method1(root.right, key);
        else {
            // case 3
            if (//case 1
            root.left != null && root.right != null) {
                int v = method2(root.left, Integer.MIN_VALUE);
                root.val = v;
                root.left = method1(root.left, v);
            } else if (//case 2
            root.left != null)
                return root.left;
            else if (//case3
            root.right != null)
                return root.right;
            else
                return null;
        }
        return root;
    }

    public int method2(TreeNode root, int max) {
        while (root != null) {
            max = Math.max(root.val, max);
            root = root.right;
        }
        return max;
    }
}
