/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static TreeNode method1(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }

    public TreeNode method2(TreeNode root, int key) {
        if (root == null) {
            return null;
        }
        if (root.val < key) {
            root.right = method2(root.right, key);
        } else if (root.val > key) {
            root.left = method2(root.left, key);
        } else if (root.val == key) {
            // with 0 child
            if (root.right == null && root.left == null) {
                return null;
            }
            // with 1 child
            if (root.right == null) {
                return root.left;
            } else if (root.left == null) {
                return root.right;
            }
            // with 2 child
            TreeNode tn = method1(root.right);
            root.val = tn.val;
            root.right = method2(root.right, tn.val);
        }
        return root;
    }
}
