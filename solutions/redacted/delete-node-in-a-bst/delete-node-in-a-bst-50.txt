class Solution {

    public TreeNode method1(TreeNode root, int key) {
        if (root == null)
            return root;
        else if (root.val > key) {
            root.left = method1(root.left, key);
        } else if (root.val < key) {
            root.right = method1(root.right, key);
        } else {
            //case1 leafnode
            if (root.right == null && root.left == null) {
                root = null;
                return root;
            } else // single sided node
            if (root.right == null) {
                TreeNode temp = root.left;
                root = null;
                return temp;
            } else if (root.left == null) {
                TreeNode temp = root.right;
                root = null;
                return temp;
                //sub complete bst
            } else {
                //we can also find for max in left sub tree
                TreeNode temp = method2(root.right);
                root.val = temp.val;
                root.right = method1(root.right, temp.val);
            }
        }
        return root;
    }

    public TreeNode method2(TreeNode root) {
        while (root.left != null) {
            root = root.left;
        }
        return root;
    }
}
