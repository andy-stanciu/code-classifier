/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public TreeNode method1(TreeNode root, int key) {
        TreeNode temp;
        if (root == null)
            return null;
        else if (key < root.val)
            root.left = method1(root.left, key);
        else if (key > root.val)
            root.right = method1(root.right, key);
        else if (root.left != null && root.right != null) {
            temp = method2(root.right);
            root.val = temp.val;
            root.right = method1(root.right, temp.val);
        } else {
            temp = root;
            if (root.left == null)
                root = root.right;
            else if (root.right == null)
                root = root.left;
        }
        return root;
    }

    public TreeNode method2(TreeNode t) {
        if (t == null)
            return null;
        else if (t.left == null)
            return t;
        else
            return method2(t.left);
    }
}
