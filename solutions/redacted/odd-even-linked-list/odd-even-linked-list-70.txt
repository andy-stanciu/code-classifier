class Solution {

    ListNode var1, var2, ot, et;

    public ListNode method1(ListNode head) {
        var2 = et = var1 = ot = null;
        go(head);
        method2(var2);
        return var1;
    }

    void method2(ListNode node) {
        if (ot == null) {
            var1 = ot = node;
        } else {
            ot.next = node;
            ot = node;
        }
    }

    void pusheve(ListNode node) {
        if (et == null) {
            var2 = et = node;
        } else {
            et.next = node;
            et = node;
        }
    }

    void go(ListNode node) {
        while (true) {
            if (node == null)
                return;
            method2(node);
            pusheve(node.next);
            if (node.next == null)
                return;
            node = node.next.next;
        }
    }
}
