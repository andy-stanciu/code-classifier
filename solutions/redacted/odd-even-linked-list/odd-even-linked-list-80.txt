class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode temp = var2;
        while (var1.next != null && var2.next != null) {
            var1.next = var2.next;
            var1 = var2.next;
            var2.next = var1.next;
            var2 = var1.next;
        }
        var1.next = temp;
        return head;
    }
}
