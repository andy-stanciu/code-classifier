class Solution {

    public ListNode method1(ListNode head) {
        // for list of 0,1,or2 length, the original list is the same
        if (head == null || head.next == null || head.next.next == null) {
            return head;
        }
        ListNode var1 = head;
        ListNode var2 = head;
        ListNode var3 = head.next;
        ListNode var4 = head.next;
        int i = 3;
        ListNode curr = head.next.next;
        while (curr != null) {
            if (i % 2 == 0) {
                var4.next = curr;
                var4 = curr;
            } else {
                var2.next = curr;
                var2 = curr;
            }
            i++;
            curr = curr.next;
        }
        // remove circular loop
        var4.next = null;
        var2.next = var3;
        return var1;
    }
}
