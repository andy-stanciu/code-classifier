class Solution {

    public ListNode method1(ListNode head) {
        //no need for comment to understand this code, its self explanatory
        ListNode var1 = new ListNode(-1);
        ListNode var2 = new ListNode(-1);
        ListNode var3 = var1;
        ListNode var4 = var2;
        ListNode curr = head;
        int idx = 1;
        while (curr != null) {
            if (idx % 2 == 0) {
                var3.next = curr;
                var3 = var3.next;
            } else {
                var4.next = curr;
                var4 = var4.next;
            }
            curr = curr.next;
            idx++;
        }
        var3.next = null;
        var4.next = var1.next;
        return var2.next;
    }
}
