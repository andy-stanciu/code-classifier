class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode a = new ListNode(0);
        ListNode b = new ListNode(0);
        ListNode var1 = a;
        ListNode var2 = b;
        ListNode temp = head;
        int idx = 1;
        while (temp != null) {
            if (idx % 2 == 0) {
                var1.next = temp;
                var1 = var1.next;
            } else {
                var2.next = temp;
                var2 = var2.next;
            }
            temp = temp.next;
            idx++;
        }
        var1.next = null;
        var2.next = a.next;
        return b.next;
    }
}
