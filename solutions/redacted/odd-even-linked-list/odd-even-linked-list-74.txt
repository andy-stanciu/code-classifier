class Solution {

    public ListNode method1(ListNode head) {
        // edge cases
        if (head == null || head.next == null) {
            return head;
        }
        // initialization
        ListNode var1 = head;
        ListNode var2 = head.next;
        ListNode var3 = var2;
        // iterate over the good linked list
        while (var2 != null && var2.next != null) {
            var1.next = var2.next;
            var1 = var1.next;
            var2.next = var1.next;
            var2 = var2.next;
        }
        // connect odd and even
        var1.next = var3;
        return head;
    }
}
