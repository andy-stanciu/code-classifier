class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode var1 = null, var2 = null;
        ListNode var3 = null, var4 = null;
        ListNode curr = head;
        int i = 1;
        while (curr != null) {
            // generate the odd indices list
            if (i % 2 == 1) {
                if (var1 == null) {
                    var1 = curr;
                    var2 = curr;
                } else {
                    var2.next = curr;
                    var2 = var2.next;
                }
            } else // generate the even indices list
            {
                if (var3 == null) {
                    var3 = curr;
                    var4 = curr;
                } else {
                    var4.next = curr;
                    var4 = var4.next;
                }
            }
            curr = curr.next;
            i++;
        }
        // there should not be any node after even tail
        var4.next = null;
        // join even list after odd
        var2.next = var3;
        return var1;
    }
}
