/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode current) {
        if (current == null)

        int index = 1;
        ListNode next = null;

        ListNode var3 = null;
        ListNode var4 = null;
        ListNode var5 = null;
        ListNode var6 = null;
        while (current != null) {
            next = current.next;
            current.next = null;
            if (index % 2 == 0) {
                if (var3 == null) {
                    var1 = current;
                    var3 = current;
                    var5 = current;
                } else {
                    var1.next = current;
                    var1 = current;
                    var3.next = current;
                    var3 = var3.next;
                }

                    var2 = current;
                    var4 = current;
                    var6 = current;

                    var2.next = current;
                    var2 = current;
                    var4.next = current;
                    var4 = var4.next;
                }
            }
            current = next;

        }

    }
}
