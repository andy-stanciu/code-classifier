/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        // Will take the Optimal Approach
        if (head == null || head.next == null)
            return head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        //this is taken for remembering the even as this is required for connecting after the odd Nodes
        ListNode var3 = head.next;
        while (var2 != null && var2.next != null) {
            // Skip 1 step and traverse from the odd nodes
            var1.next = var1.next.next;
            var1 = var1.next;
            // traverse the even odd
            var2.next = var2.next.next;
            var2 = var2.next;
        }
        // Connecxt the even after the odd nodes
        var1.next = var3;
        return head;
    }
}
