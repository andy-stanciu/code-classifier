class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        // CREATING ODD AND EVEN POINTERS
        ListNode var1 = head;
        ListNode var2 = head.next;
        // CREATING EVENHEAD TO CONNECT WITH ODD LIST LATER
        ListNode var3 = var2;
        while (var2 != null && var2.next != null) {
            //point odd\'s next to even\'s next
            var1.next = var2.next;
            //point even\'s next to even\'s next\'s next
            var2.next = var2.next.next;
            //move odd and even pointers
            var1 = var1.next;
            var2 = var2.next;
        }
        // CONNECT ODD WITH EVEN
        var1.next = var3;
        // return Head.
        return head;
    }
    // don\'t forget to upvote solution.
}
