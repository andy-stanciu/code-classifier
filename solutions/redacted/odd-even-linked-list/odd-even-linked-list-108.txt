class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode var1 = new ListNode(-1);
        ListNode var2 = new ListNode(-1);
        ListNode var3 = var2, var4 = var1, curr = head;
        int x = 1;
        while (curr != null) {
            if (x % 2 != 0) {
                var4.next = curr;
                var4 = var4.next;
            } else {
                var3.next = curr;
                var3 = var3.next;
            }
            curr = curr.next;
            x++;
        }
        var4.next = var2.next;
        var3.next = null;
        return var1.next;
    }
}
