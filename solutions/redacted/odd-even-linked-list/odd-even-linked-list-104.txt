/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        ListNode var1 = new ListNode(-1);
        ListNode var2 = var1;
        ListNode var3 = new ListNode(-1);
        ListNode var4 = var3;
        ListNode curr = head;
        int i = 1;
        while (curr != null) {
            if (i % 2 == 0) {
                var3.next = curr;
                var3 = var3.next;
                if (curr.next == null)
                    var1.next = null;
            } else {
                var1.next = curr;
                var1 = var1.next;
                if (curr.next == null)
                    var3.next = null;
            }
            curr = curr.next;
            i++;
        }
        var1.next = var4.next;
        return var2.next;
    }
}
