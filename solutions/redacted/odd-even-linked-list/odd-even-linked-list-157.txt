/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode var1 = null;
        ListNode var2 = null;
        var2 = head;
        if (head.next != null) {
            var1 = head.next;
        }
        ListNode var3 = head.next;
        while (var2.next != null) {
            if (var2.next.next == null) {
                break;
            }
            var2.next = var1.next;
            var2 = var1.next;
            if (var2 != null) {
                var1.next = var2.next;
                var1 = var2.next;
            }
        }
        var2.next = var3;
        return head;
    }
}
