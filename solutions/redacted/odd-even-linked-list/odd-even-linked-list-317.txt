/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        if (null == head || null == head.next) {
            return head;
        }
        ListNode result = head;
        ListNode var1 = head;
        ListNode var2 = head.next;
        boolean flag = true;
        head = head.next.next;
        while (null != head) {
            ListNode node = head;
            head = head.next;
            if (flag) {
                node.next = var1.next;
                var1.next = node;
                var1 = node;
            } else {
                node.next = null;
                var2.next = node;
                var2 = node;
            }
            flag = !flag;
        }
        var2.next = null;
        return result;
    }
}
