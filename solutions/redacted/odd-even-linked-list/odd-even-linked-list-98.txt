class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null || head.next.next == null)
            return head;
        ListNode copy1 = head.next.next;
        ListNode copy2 = head.next;
        ListNode var1 = new ListNode(head.val);
        ListNode o = var1;
        while (copy1 != null) {
            var1.next = new ListNode(copy1.val);
            var1 = var1.next;
            copy1 = copy1.next;
            if (copy1 != null)
                copy1 = copy1.next;
        }
        ListNode var2 = new ListNode(0);
        ListNode e = var2;
        while (copy2 != null) {
            var2.next = new ListNode(copy2.val);
            var2 = var2.next;
            copy2 = copy2.next;
            if (copy2 != null)
                copy2 = copy2.next;
        }
        var1.next = e.next;
        return o;
    }
}
