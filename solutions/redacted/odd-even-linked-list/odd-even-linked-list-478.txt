class Class1 {

    int val;

    Class1 next;

    Class1(int x) {
        val = x;
    }
}

class Solution {

    public Class1 method1(Class1 head) {
        if (head == null)
            return null;
        Class1 var1 = head;
        Class1 var2 = head.next;
        Class1 var3 = var2;
        while (var2 != null && var2.next != null) {
            var1.next = var2.next;
            var1 = var1.next;
            var2.next = var1.next;
            var2 = var2.next;
        }
        var1.next = var3;
        return head;
    }
}
