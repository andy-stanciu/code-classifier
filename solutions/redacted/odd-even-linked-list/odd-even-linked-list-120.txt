/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        int count = 2;
        ListNode var1 = head, var2 = head.next, ans = var1, var3 = var2;
        ListNode curr = head.next.next;
        while (curr != null) {
            count++;
            if (count % 2 == 0) {
                var2.next = curr;
                var2 = var2.next;
            } else {
                var1.next = curr;
                var1 = var1.next;
            }
            curr = curr.next;
        }
        var2.next = null;
        var1.next = var3;
        return ans;
    }
}
