class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp = head;
        int count = 0;
        ListNode var1 = new ListNode();
        ListNode curr = var1;
        ListNode var2 = new ListNode();
        ListNode var3 = var2;
        while (temp != null) {
            count++;
            if (count % 2 == 0) {
                curr.next = new ListNode(temp.val);
                curr = curr.next;
            } else {
                var3.next = new ListNode(temp.val);
                var3 = var3.next;
            }
            temp = temp.next;
        }
        temp = var2.next;
        while (temp != null) {
            if (temp.next == null) {
                temp.next = var1.next;
                break;
            }
            temp = temp.next;
        }
        return var2.next;
    }
}
