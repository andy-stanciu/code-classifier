class Solution {

    public void method1(int[] nums) {
        //initialize two pointers both at the beginning of the list
        int i = 0;
        int j = 0;
        //loop throught array using j as the pointer to populate the array
        for (; i < nums.length; i++) {
            //restrict the population to int\'s that are not equal to 0
            if (nums[i] != 0) {
                nums[j] = nums[i];
                j++;
            }
        }
        //put zero\'s in the remaining open positions in the array
        while (j < nums.length) {
            nums[j] = 0;
            j++;
        }
    }
}
