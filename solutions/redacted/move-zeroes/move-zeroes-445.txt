class Solution {

    public void method1(int[] nums) {
        int lastIndexFound = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                int temp = nums[lastIndexFound];
                nums[lastIndexFound] = nums[i];
                nums[i] = temp;
                lastIndexFound++;
            }
        }
    }
}
