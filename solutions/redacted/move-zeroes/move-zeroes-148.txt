class Solution {

    // keep two pointers one maintains non zroes at start second traverses full array
    public void method1(int[] nums) {
        int i = 0;
        int j = 0;
        while (i < nums.length && j < nums.length) {
            if (nums[j] != 0) {
                // move non zeroes at start
                nums[i] = nums[j];
                i++;
                j++;
            } else {
                // if zero found ignore for now
                j++;
            }
        }
        while (i < nums.length) {
            // fill remaining with zeros
            nums[i] = 0;
            i++;
        }
    }
}
