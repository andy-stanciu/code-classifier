class Solution {

    public void method1(int[] nums) {
        //temp will contain all non zero elements in starting and zeroes at last
        int[] temp = new int[nums.length];
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                temp[index] = nums[i];
                index++;
            }
        }
        //Replacing the elements of temp to original array nums
        for (int i = 0; i < nums.length; i++) {
            nums[i] = temp[i];
        }
    }
}
