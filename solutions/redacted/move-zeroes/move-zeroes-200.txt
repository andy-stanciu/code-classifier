class Solution {

    public void method1(int[] nums) {
        int length = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[length++] = nums[i];
            }
        }
        while (length < nums.length) {
            nums[length++] = 0;
        }
    }
}
