class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int idx = 0;
        int countZeros = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                countZeros++;
            } else if (nums[i] != 0) {
                nums[idx] = nums[i];
                idx++;
            }
        }
        for (int i = 0; i < countZeros; i++) {
            nums[idx] = 0;
            idx++;
        }
    }
}
