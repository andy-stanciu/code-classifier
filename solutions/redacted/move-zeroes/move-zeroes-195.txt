class Solution {

    public void method1(int[] nums) {
        int pos_zero = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0 && pos_zero != -1) {
                int temp = nums[i];
                nums[i] = 0;
                nums[pos_zero] = temp;
                pos_zero++;
            } else {
                if (nums[i] == 0 && pos_zero == -1) {
                    pos_zero = i;
                }
            }
        }
    }
}
