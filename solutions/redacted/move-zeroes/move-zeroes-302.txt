class Solution {

    public void method1(int[] nums) {
        int index = method2(nums);
        method3(nums, index);
    }

    private int method2(int[] nums) {
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[index++] = nums[i];
            }
        }
        return index;
    }

    private void method3(int[] nums, int index) {
        for (int i = index; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
