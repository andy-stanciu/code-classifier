class Solution {

    public void method1(int[] nums) {
        int s = nums.length;
        int i = 0;
        int j = 0;
        while (i < s) {
            if (nums[i] == 0) {
                i++;
                continue;
            }
            nums[j] = nums[i];
            j++;
            i++;
        }
        for (int k = j; k < s; k++) {
            nums[k] = 0;
        }
    }
}
