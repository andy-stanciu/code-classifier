class Solution {

    public void method1(int[] nums) {
        int lp = 0;
        int rp = 0;
        while (rp != nums.length) {
            if (nums[rp] != 0) {
                int temp = nums[lp];
                nums[lp] = nums[rp];
                nums[rp] = temp;
                lp++;
            }
            rp++;
        }
    }
}
