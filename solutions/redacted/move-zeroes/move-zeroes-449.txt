class Solution {

    public void method1(int[] nums) {
        int i = 0;
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] != 0) {
                nums[i++] = nums[j];
            }
        }
        // int s=i;
        for (i = i; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
