class Solution {

    public void method1(int[] nums) {
        int zero = 0;
        int j = 0;
        int[] res = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zero++;
            } else {
                res[j++] = nums[i];
            }
        }
        for (int i = 0; i < zero; i++) {
            res[j++] = 0;
        }
        for (int i = 0; i < res.length; i++) {
            nums[i] = res[i];
        }
    }
}
