class Solution {

    public void method1(int[] nums) {
        int j = 0;
        int temp = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[j] == 0 && nums[i] != 0) {
                temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                j++;
            } else if (nums[j] != 0) {
                j++;
            }
        }
        return;
    }
}
