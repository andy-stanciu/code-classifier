class Solution {

    public void method1(int[] nums) {
        int availableSpot = 0;
        while (nums[availableSpot] != 0 && availableSpot < nums.length - 1) availableSpot++;
        for (int i = availableSpot + 1; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[availableSpot] = nums[i];
                nums[i] = 0;
                availableSpot++;
            }
        }
    }
}
