class Solution {

    public void method1(int[] nums) {
        int value = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[value] = nums[i];
                value++;
            }
        }
        for (int i = value; i < nums.length; i++) {
            nums[i] = 0;
        }
    }
}
