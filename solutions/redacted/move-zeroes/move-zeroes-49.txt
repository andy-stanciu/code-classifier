class Solution {

    public void method1(int[] nums) {
        int i = 0;
        for (int l = 1; l < nums.length; l++) {
            if (nums[i] != 0) {
                i++;
            } else {
                if (nums[l] != 0) {
                    int temp = nums[i];
                    nums[i] = nums[l];
                    nums[l] = temp;
                    i++;
                }
            }
        }
    }
}
