class Solution {

    public void method1(int[] nums) {
        int left = -1;
        //find the first zero number in the array
        for (int right = 0; right < nums.length; right++) {
            if (nums[right] == 0) {
                left = right;
                break;
            }
        }
        //if no zero element found, return from here
        if (left == -1)
            return;
        //now swap the zero element with the first non zero element that is found
        //after swapping move the left pointer up by one place, to signify next zero position
        for (int right = left + 1; right < nums.length; right++) {
            if (nums[right] != 0) {
                swap(nums, right, left);
                left++;
            }
        }
    }

    private void swap(int[] nums, int l, int r) {
        int temp = nums[l];
        nums[l] = nums[r];
        nums[r] = temp;
    }
}
