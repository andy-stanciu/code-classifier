class Solution {

    public void method1(int[] arr) {
        int i = 0;
        int j = 0;
        while (i < arr.length) {
            if (arr[i] != 0) {
                int t = arr[i];
                arr[i] = arr[j];
                arr[j] = t;
                j++;
                i++;
            } else {
                i++;
            }
        }
    }
}
