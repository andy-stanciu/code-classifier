class Solution {

    public void method1(int[] nums) {
        int writerBoy = 0;
        // Searches for a non 0
        for (int searchBoy = 0; searchBoy < nums.length; searchBoy++) {
            if (nums[searchBoy] != 0) {
                nums[writerBoy] = nums[searchBoy];
                writerBoy++;
            }
        }
        while (writerBoy < nums.length) {
            nums[writerBoy] = 0;
            writerBoy++;
        }
    }
}
