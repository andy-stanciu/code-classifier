class Solution {

    public void method1(int[] nums) {
        int i, j, k = 0, l = nums.length, temp = 0, f = 0;
        int[] res = new int[l];
        for (i = 0; i < l; i++) {
            res[i] = nums[i];
        }
        if (l == 1) {
            System.out.println(res[0]);
        }
        for (i = 0; i < l; i++) {
            for (j = 0; j < l - i - 1; j++) {
                if (res[j] == 0) {
                    temp = res[j];
                    res[j] = res[j + 1];
                    res[j + 1] = temp;
                    f = 1;
                }
            }
            if (f == 0) {
                break;
            }
        }
        for (i = 0; i < res.length; i++) {
            nums[i] = res[i];
            System.out.print(nums[i] + ",");
        }
    }
}
