class Solution {

    public void method1(int[] nums) {
        //pointer to keep track of non zero element ;
        int i = 0;
        //another pointer j for traversing array
        for (int j = 0; j < nums.length; j++) {
            // If nums[j] is non-zero, move it to the position pointed by \'i\'.
            if (nums[j] != 0) {
                // Place the non-zero element at index \'i\'.
                nums[i] = nums[j];
                //increment i for next
                i++;
            }
        }
        while (i < nums.length) {
            //fills all the remaining indexex with zero
            nums[i] = 0;
            i++;
        }
    }
}
