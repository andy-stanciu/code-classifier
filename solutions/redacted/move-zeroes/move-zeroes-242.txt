class Solution {

    public void method1(int[] nums) {
        int c = nums.length - 1;
        int[] b = new int[c + 1];
        int d = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                b[c] = 0;
                c--;
            } else {
                b[d] = nums[i];
                d++;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            nums[i] = b[i];
        }
    }
}
