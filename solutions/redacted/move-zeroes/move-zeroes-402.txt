class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        if (n == 0 || n == 1)
            return;
        int index = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[index++] = nums[i];
            }
        }
        for (int i = index; i < n; i++) {
            nums[i] = 0;
        }
    }
}
