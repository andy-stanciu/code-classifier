class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int k = 0;
        int count = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] != 0) {
                nums[k++] = nums[i];
            } else {
                count++;
            }
        }
        for (int i = 0; i < count; i++) {
            nums[k++] = 0;
        }
    }
}
