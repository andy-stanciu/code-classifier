class Solution {

    public void method1(int[] nums) {
        int numsSize = nums.length;
        if (numsSize == 0 || numsSize == 1)
            return;
        int fast = 0, slow = 0;
        while (fast < numsSize) {
            if (nums[fast] != 0) {
                nums[slow] = nums[fast];
                slow++;
            }
            fast++;
        }
        while (slow < nums.length) {
            nums[slow] = 0;
            slow++;
        }
    }
}
