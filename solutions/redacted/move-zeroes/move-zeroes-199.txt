class Solution {

    public void method1(int[] arr) {
        int non_zero = 0;
        int len = arr.length;
        for (int i = 0; i < len; i++) {
            if (arr[i] != 0) {
                arr[non_zero] = arr[i];
                non_zero++;
            }
        }
        for (int i = non_zero; i < len; i++) {
            arr[i] = 0;
        }
    }
}
