class Solution {

    public void method1(int[] nums) {
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            // If array at i index is not zero then swap and increment j by one
            if (nums[i] != 0) {
                int x = nums[i];
                nums[i] = nums[j];
                nums[j] = x;
                j++;
            }
        }
    }
}
