class Solution {

    public void method1(int[] nums) {
        // Length of the array
        int lengthArr = nums.length;
        // Tracks the position where the next non-zero element should go
        int initialPointer = 0;
        // Iterate through the array
        for (int i = 0; i < lengthArr; i++) {
            // If the current element is non-zero, swap it with the element at \'initialPointer\'
            if (nums[i] != 0) {
                int temp = nums[i];
                nums[i] = nums[initialPointer];
                nums[initialPointer] = temp;
                // Move the pointer to the next position for non-zero elements
                initialPointer++;
            }
        }
    }
}
