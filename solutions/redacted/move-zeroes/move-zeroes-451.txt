class Solution {

    public void method1(int[] nums) {
        //using 2 pointer approach, track index of last non zero value and put all non zero elements together.
        int lastNonZeroIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[lastNonZeroIndex] = nums[i];
                lastNonZeroIndex++;
            }
        }
        //Fill the blank position with zeros.
        for (int j = lastNonZeroIndex; j < nums.length; j++) {
            nums[j] = 0;
        }
    }
}
