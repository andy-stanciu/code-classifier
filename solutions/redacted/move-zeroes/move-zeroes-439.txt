class Solution {

    public void method1(int[] nums) {
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[k++] = nums[i];
            }
        }
        if (k < nums.length) {
            for (int j = k; j < nums.length; j++) {
                nums[j] = 0;
            }
        }
    }
}
