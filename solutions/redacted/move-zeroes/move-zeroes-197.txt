class Solution {

    public void method1(int[] nums) {
        int zero = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                zero++;
            } else {
                nums[i - zero] = nums[i];
            }
        }
        for (int i = 0, j = nums.length - 1; i < zero; i++, j--) {
            nums[j] = 0;
        }
    }
}
