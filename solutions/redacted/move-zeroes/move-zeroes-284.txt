class Solution {

    public void method1(int[] nums) {
        int leftPointer = 0;
        for (int rightPointer = 0; rightPointer < nums.length; rightPointer++) {
            if (nums[rightPointer] != 0) {
                int temp = nums[rightPointer];
                nums[rightPointer] = nums[leftPointer];
                nums[leftPointer] = temp;
                leftPointer++;
            }
        }
    }
}
