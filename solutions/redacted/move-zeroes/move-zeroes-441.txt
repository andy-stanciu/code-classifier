class Solution {

    public void method1(int[] nums) {
        int c = 0;
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                nums[index] = nums[i];
                index++;
            } else {
                c++;
            }
        }
        for (int i = 0; i < c; i++) {
            nums[index] = 0;
            index++;
        }
    }
}
