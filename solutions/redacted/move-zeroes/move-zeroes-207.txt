class Solution {

    public void method1(int[] nums) {
        ArrayList<Integer> newArr = new ArrayList<>();
        ArrayList<Integer> ZeroArr = new ArrayList<>();
        // Separate non-zero and zero elements
        for (int num : nums) {
            if (num == 0) {
                ZeroArr.add(num);
            } else {
                newArr.add(num);
            }
        }
        // Combine non-zero elements with zero elements at the end
        newArr.addAll(ZeroArr);
        // Update the original nums array
        for (int i = 0; i < nums.length; i++) {
            nums[i] = newArr.get(i);
        }
        System.out.println(Arrays.toString(nums));
    }
}
