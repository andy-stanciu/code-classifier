class Solution {

    public void method1(int[] nums) {
        int lastNonZeroIdx = 0, zeroCount = 0;
        int x = 0;
        while (x < nums.length) {
            if (nums[x] == 0) {
                zeroCount++;
            } else if (zeroCount > 0) {
                nums[lastNonZeroIdx++] = nums[x];
                nums[x] = 0;
            } else
                lastNonZeroIdx++;
            x++;
        }
    }
}
