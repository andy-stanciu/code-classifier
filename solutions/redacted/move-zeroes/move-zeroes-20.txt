class Solution {

    public void method1(int[] nums) {
        // Pointer for iterating through the array
        int left = 0;
        // Pointer for placing non-zero elements
        int right = 0;
        // Move non-zero elements to the front of the array
        while (left < nums.length) {
            if (nums[left] != 0) {
                // Swap non-zero element to the right position
                int temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
                right++;
            }
            left++;
        }
    }
}
