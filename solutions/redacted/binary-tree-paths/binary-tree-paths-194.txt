class Solution {

    public List<String> method1(TreeNode root) {
        List<String> results = new ArrayList<>();
        if (root == null)
            return results;
        backTracking(root, "", results);
        return results;
    }

    private void backTracking(TreeNode node, String curResult, List<String> results) {
        if (node == null)
            return;
        if (curResult.length() == 0) {
            curResult += node.val;
        } else {
            curResult += "->" + node.val;
        }
        if (node.left == null && node.right == null) {
            results.add(curResult);
            return;
        }
        backTracking(node.left, curResult, results);
        backTracking(node.right, curResult, results);
    }
}
