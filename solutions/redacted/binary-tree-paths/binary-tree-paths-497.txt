class Solution {

    public List<String> method1(TreeNode root) {
        if (root == null) {
            List<String> empty = new ArrayList<>();
            return empty;
        }
        List<String> returnedList;
        List<String> leftList = binaryTreePathsAllowingNullRoot(root.left);
        List<String> rightList = binaryTreePathsAllowingNullRoot(root.right);
        if (leftList.size() == 0 && rightList.size() == 0) {
            String s = "" + root.val;
            returnedList = new ArrayList<String>();
            returnedList.add(s);
            return returnedList;
        } else {
            returnedList = leftList;
            returnedList.addAll(rightList);
        }
        for (int i = 0; i < returnedList.size(); i++) {
            returnedList.set(i, root.val + "->" + returnedList.get(i));
        }
        return returnedList;
    }
}
