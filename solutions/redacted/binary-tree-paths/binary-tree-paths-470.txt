class Solution {

    List<String> list = new ArrayList<String>();

    public void helperMethod(TreeNode root, String runningString) {
        if (runningString.equals("")) {
            runningString = String.valueOf(root.val);
        } else {
            runningString = runningString + "->" + String.valueOf(root.val);
        }
        if (root.left == null && root.right == null) {
            list.add(runningString);
            return;
        }
        if (root.left != null) {
            helperMethod(root.left, runningString);
        }
        if (root.right != null) {
            helperMethod(root.right, runningString);
        }
    }

    public List<String> method1(TreeNode root) {
        helperMethod(root, "");
        return list;
    }
}
