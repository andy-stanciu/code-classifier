/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        addString(root, ans, "");
        return ans;
    }

    public void addString(TreeNode root, List<String> ans, String str) {
        if (root == null) {
            return;
        }
        str += Integer.toString(root.val);
        str += "->";
        if (root.left == null && root.right == null) {
            String substr = str.substring(0, str.length() - 2);
            ans.add(substr);
        }
        addString(root.left, ans, str);
        addString(root.right, ans, str);
    }
}
