class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        String str = "";
        helper(list, root, str);
        return list;
    }

    public void helper(List<String> list, TreeNode curr, String str) {
        if (// not a valid path, don\'t add
        curr == null)
            return;
        if (curr.right == null && curr.left == null) {
            // leaf node
            str += Integer.toString(curr.val);
            list.add(str);
            return;
        }
        // add current node to str
        str += Integer.toString(curr.val) + "->";
        // left
        helper(list, curr.left, str);
        // right
        helper(list, curr.right, str);
    }
}
