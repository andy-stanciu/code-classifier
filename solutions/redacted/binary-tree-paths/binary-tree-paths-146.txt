/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void recurse(TreeNode root, List<String> res, String arr) {
        if (root == null)
            return;
        if (arr.isEmpty())
            arr += (root.val);
        else
            arr += ("->" + root.val);
        if (root.left == null && root.right == null) {
            res.add(arr);
        } else {
            recurse(root.left, res, arr);
            recurse(root.right, res, arr);
        }
    }

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        recurse(root, res, "");
        return res;
    }
}
