class Solution {

    List<String> list;

    public List<String> method1(TreeNode root) {
        list = new LinkedList<String>();
        method2(root, "");
        return list;
    }

    void method2(TreeNode root, String str) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            list.add(str + root.val);
        }
        method2(root.left, str + root.val + "->");
        method2(root.right, str + root.val + "->");
    }
}
