/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
public class Solution {

    List<String> list = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return list;
        }
        helper(root, new StringBuilder());
        return list;
    }

    void helper(TreeNode node, StringBuilder sb) {
        if (node == null) {
            return;
        }
        int len = sb.length();
        if (len != 0) {
            sb.append("->");
        }
        sb.append(node.val);
        if (node.left == null && node.right == null) {
            // if it\'s a leaf
            list.add(sb.toString());
        } else {
            // create a new StringBuilder for left subtree
            helper(node.left, new StringBuilder(sb));
            // create a new StringBuilder for right subtree
            helper(node.right, new StringBuilder(sb));
        }
    }
}
