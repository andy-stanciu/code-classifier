/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void method1(List<String> res, TreeNode root, StringBuilder str) {
        int lenBefore = str.length();
        str.append(root.val);
        if (root.left == null && root.right == null) {
            res.add(str.toString());
        } else {
            str.append("->");
            if (root.left != null) {
                method1(res, root.left, str);
            }
            if (root.right != null) {
                method1(res, root.right, str);
            }
            str.setLength(lenBefore + String.valueOf(root.val).length());
        }
        str.setLength(lenBefore);
    }

    public List<String> method2(TreeNode root) {
        List<String> res = new ArrayList<>();
        if (root != null) {
            method1(res, root, new StringBuilder());
        }
        return res;
    }
}
