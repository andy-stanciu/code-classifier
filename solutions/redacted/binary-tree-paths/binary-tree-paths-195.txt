/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        if (root == null)
            return Collections.emptyList();
        if (root.left == null && root.right == null)
            return Collections.singletonList(String.valueOf(root.val));
        List<List<String>> var1 = new ArrayList<>();
        List<String> visited = new ArrayList<>();
        List<String> result = new ArrayList<>();
        dfs(root, visited, var1);
        for (List<String> path : var1) {
            String join = String.join("->", path);
            result.add(join);
        }
        return result;
    }

    public void dfs(TreeNode node, List<String> vis, List<List<String>> var1) {
        vis.add(String.valueOf(node.val));
        if (node.left == null && node.right == null) {
            var1.add(new ArrayList<>(vis));
            return;
        }
        if (node.left != null) {
            dfs(node.left, vis, var1);
            vis.remove(vis.size() - 1);
        }
        if (node.right != null) {
            dfs(node.right, vis, var1);
            vis.remove(vis.size() - 1);
        }
    }
}
