class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        recur(root, "");
        return ans;
    }

    public void recur(TreeNode root, String curr) {
        if (root == null)
            return;
        curr = curr + root.val;
        if (root.left == null && root.right == null)
            ans.add(curr);
        recur(root.left, curr + "->");
        recur(root.right, curr + "->");
    }
}
