/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        ArrayList<String> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        method1(root, "", list);
        return list;
    }

    private void method1(TreeNode root, String details, ArrayList<String> list) {
        if (root.left == null && root.right == null) {
            list.add(details + root.val);
            return;
        }
        // list.add(details +root.val+"->");
        // list.addAll(binaryTreePaths(root.left, details ));
        // list.addAll(binaryTreePaths(root.right, details));
        if (root.left != null) {
            method1(root.left, details + root.val + "->", list);
        }
        if (root.right != null) {
            method1(root.right, details + root.val + "->", list);
        }
    }
}
