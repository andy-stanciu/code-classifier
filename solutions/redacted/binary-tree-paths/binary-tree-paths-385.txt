/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new LinkedList<String>();
        method2(root, "", result);
        return result;
    }

    private void method2(TreeNode root, String curr, List<String> result) {
        if (root == null) {
            return;
        }
        String next = curr + "->" + root.val;
        if (root.left == null && root.right == null) {
            result.add(next.substring(2));
        }
        method2(root.left, next, result);
        method2(root.right, next, result);
    }
}
