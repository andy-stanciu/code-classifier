/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList();
        if (root == null)
            return var1;
        method2(root, "", var1);
        return var1;
    }

    public void method2(TreeNode root, String currentPath, List<String> var1) {
        currentPath += root.val;
        if (root.left == null && root.right == null) {
            var1.add(currentPath);
            return;
        }
        if (root.left != null) {
            method2(root.left, currentPath + "->", var1);
        }
        if (root.right != null) {
            method2(root.right, currentPath + "->", var1);
        }
    }
}
