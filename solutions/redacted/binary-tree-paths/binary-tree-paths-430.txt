/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> path(TreeNode root, List<String> lis, String s) {
        if (root == null)
            return null;
        if (root.right == null && root.left == null) {
            s = s + root.val;
            lis.add(s);
        }
        s = s + root.val + "->";
        path(root.right, lis, s);
        path(root.left, lis, s);
        return lis;
    }

    public List<String> method1(TreeNode root) {
        if (root == null)
            return null;
        return path(root, new LinkedList(), "");
    }
}
