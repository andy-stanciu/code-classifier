class Solution {

    List<String> list;

    public List<String> method1(TreeNode root) {
        list = new ArrayList<>();
        findPath(root, "");
        return list;
    }

    void findPath(TreeNode root, String s) {
        s += root.val;
        if (//it will be our base condition
        root.left == null && root.right == null) {
            // if it is the leaf node then we will return.
            list.add(s);
            return;
        }
        s += "->";
        if (//calling left
        root.left != null)
            findPath(root.left, s);
        if (//calling right
        root.right != null)
            findPath(root.right, s);
    }
}
