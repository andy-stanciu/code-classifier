/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static void solve(TreeNode root, String str, List<String> li) {
        if (root == null) {
            return;
        }
        if (str == "") {
            str = str + Integer.toString(root.val);
        } else {
            str = str + "->" + Integer.toString(root.val);
        }
        if (root.left == null && root.right == null) {
            li.add(str);
        }
        solve(root.left, str, li);
        solve(root.right, str, li);
    }

    public List<String> method1(TreeNode root) {
        List<String> li = new ArrayList<>();
        String str = "";
        solve(root, str, li);
        return li;
    }
}
