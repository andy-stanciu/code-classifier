class Solution {

    public void helper(TreeNode root, String str, List<String> ans) {
        if (root == null)
            return;
        String a = String.valueOf(root.val);
        if (root.left == null && root.right == null) {
            str += a;
            ans.add(str);
            return;
        }
        helper(root.left, str + a + "->", ans);
        helper(root.right, str + a + "->", ans);
    }

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        helper(root, "", ans);
        return ans;
    }
}
