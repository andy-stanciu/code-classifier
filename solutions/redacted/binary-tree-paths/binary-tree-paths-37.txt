class Solution {

    public List<String> method1(TreeNode root) {
        List<String> answer = new ArrayList<>();
        pathFinder(root, "", answer);
        return answer;
    }

    private void pathFinder(TreeNode node, String path, List<String> result) {
        if (node == null) {
            return;
        }
        path += String.valueOf(node.val);
        if (node.left == null && node.right == null) {
            result.add(path);
            return;
        }
        if (node.left != null) {
            pathFinder(node.left, path + "->", result);
        }
        if (node.right != null) {
            pathFinder(node.right, path + "->", result);
        }
    }
}
