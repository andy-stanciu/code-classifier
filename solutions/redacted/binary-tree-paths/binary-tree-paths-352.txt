/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        method2(root, "", res);
        return res;
    }

    public void method2(TreeNode node, String curPath, List<String> res) {
        if (node == null) {
            return;
        }
        if (node.left == null && node.right == null) {
            res.add(curPath + node.val);
            return;
        }
        method2(node.left, curPath + node.val + "->", res);
        method2(node.right, curPath + node.val + "->", res);
    }
}
