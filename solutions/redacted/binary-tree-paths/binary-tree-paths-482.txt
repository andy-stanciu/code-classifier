class Solution {

    public List<String> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        List<String> list = new ArrayList<>();
        StringBuilder sb = new StringBuilder("");
        return PathFinder(root, list, sb);
    }

    public List<String> PathFinder(TreeNode root, List<String> param1, StringBuilder sb) {
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            // if it is a "Leaf Node" then add the path in the list.
            param1.add(sb.toString());
            return param1;
        } else
            // if it is not a leaf node then there must be a path which exists further so append the arrow "->".
            sb.append("->");
        if (root.left != null)
            PathFinder(root.left, param1, new StringBuilder(sb));
        if (root.right != null)
            PathFinder(root.right, param1, new StringBuilder(sb));
        return param1;
    }
}
