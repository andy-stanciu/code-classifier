/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<String> ans = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        rec(root, "");
        return ans;
    }

    public void rec(TreeNode root, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            str += root.val + "";
            ans.add(str);
            return;
        }
        str += root.val + "->";
        rec(root.left, str);
        rec(root.right, str);
        return;
    }
}
