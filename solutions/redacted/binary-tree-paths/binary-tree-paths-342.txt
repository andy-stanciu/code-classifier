class Solution {

    List<String> ret;

    private void method1(String s, TreeNode root) {
        s += "->";
        s += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            ret.add(s);
            return;
        }
        if (root.left != null)
            method1(s, root.left);
        if (root.right != null)
            method1(s, root.right);
    }

    public List<String> method2(TreeNode root) {
        ret = new ArrayList<>();
        String s = Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            ret.add(s);
            return ret;
        }
        if (root.left != null)
            method1(s, root.left);
        if (root.right != null)
            method1(s, root.right);
        return ret;
    }
}
