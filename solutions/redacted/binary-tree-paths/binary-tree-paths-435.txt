class Solution {

    public List<String> method1(TreeNode root) {
        List<String> l = new ArrayList<>();
        if (root == null)
            return l;
        rv(root, l, "");
        return l;
    }

    public void rv(TreeNode root, List<String> l, String s) {
        if (root.left == null && root.right == null) {
            if (s.length() != 0)
                s = s + "->" + Integer.toString(root.val);
            else
                s = s + Integer.toString(root.val);
            l.add(s);
        }
        if (s.length() != 0)
            s = s + "->" + Integer.toString(root.val);
        else
            s = s + Integer.toString(root.val);
        if (root.left != null)
            rv(root.left, l, s);
        if (root.right != null)
            rv(root.right, l, s);
    }
}
