/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void method1(TreeNode root, String smallans, List<String> ans) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            smallans = smallans + root.val;
            ans.add(smallans);
            return;
        }
        method1(root.left, smallans + root.val + "->", ans);
        method1(root.right, smallans + root.val + "->", ans);
    }

    public List<String> method2(TreeNode root) {
        List<String> ans = new ArrayList<>();
        method1(root, "", ans);
        return ans;
    }
}
