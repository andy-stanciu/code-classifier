class Solution {

    private static final String arrow = "->";

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        StringBuilder sb = new StringBuilder();
        return topDown(root, var1, sb);
    }

    private List<String> topDown(TreeNode root, List<String> var1, StringBuilder sb) {
        // recursive helper function
        // error checking for empty trees
        if (root == null)
            return var1;
        // regardless of the type of node, append the value
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            // leaf node, so add to List and return
            var1.add(sb.toString());
            return var1;
        }
        // new StringBuilder is used to prevent unwanted mutations
        // not a leaf node, so append arrow
        if (root.left != null)
            topDown(root.left, var1, new StringBuilder(sb).append(this.arrow));
        // not a leaf node, so append arrow
        if (root.right != null)
            topDown(root.right, var1, new StringBuilder(sb).append(this.arrow));
        return var1;
    }
}
