class Solution {

    public List<String> method1(TreeNode root) {
        List<String> list = new ArrayList<>();
        StringBuilder path = new StringBuilder();
        method2(root, path, list);
        return list;
    }

    public void method2(TreeNode root, StringBuilder path, List<String> list) {
        if (root == null)
            return;
        // inital length nikal li path ki taki path.setLength se backtracking kar sake
        int len = path.length();
        path.append(root.val);
        if (root.left == null && root.right == null) {
            list.add(path.toString());
            // backtracking before return
            path.setLength(len);
            return;
        }
        path.append("->");
        method2(root.left, path, list);
        method2(root.right, path, list);
        // backtracking before return
        path.setLength(len);
    }
}
