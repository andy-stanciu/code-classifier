/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static List<String> method1(TreeNode root) {
        List<String> a = new ArrayList<>();
        helper(root, a, "");
        return a;
    }

    public static void helper(TreeNode root, List<String> h, String currentPath) {
        currentPath += Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            h.add(currentPath);
            return;
        }
        currentPath += "->";
        if (root.left != null) {
            helper(root.left, h, currentPath);
        }
        if (root.right != null) {
            helper(root.right, h, currentPath);
        }
    }
}
