class Solution {

    public List<String> method1(TreeNode root) {
        List<String> result = new ArrayList();
        String path = "";
        method2(root, result, path);
        return result;
    }

    public void method2(TreeNode node, List<String> list, String path) {
        String t = path + node.val;
        if (node.left == null && node.right == null) {
            list.add(t);
        }
        if (node.left != null) {
            method2(node.left, list, t + "->");
        }
        if (node.right != null) {
            method2(node.right, list, t + "->");
        }
    }
}
