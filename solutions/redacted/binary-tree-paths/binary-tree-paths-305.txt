/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<String> list = new ArrayList();

    public List<String> method1(TreeNode root) {
        if (root == null)
            return list;
        String s = Integer.toString(root.val);
        if (root.left == null && root.right == null) {
            list.add(s);
            return list;
        }
        if (root.left != null)
            method(root.left, s);
        if (root.right != null)
            method(root.right, s);
        return list;
    }

    public void method(TreeNode root, String s) {
        if (root == null)
            return;
        s = s + "->" + root.val;
        if (root.left == null && root.right == null)
            list.add(s);
        method(root.right, s);
        method(root.left, s);
    }
}
