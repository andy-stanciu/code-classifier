class Solution {

    public List<String> method1(TreeNode root) {
        List<String> all_path = new ArrayList();
        if (root == null)
            return all_path;
        String curr_path = Integer.toString(root.val);
        if (root.left == null && root.right == null)
            all_path.add(curr_path);
        if (root.left != null)
            dfs(root.left, curr_path, all_path);
        if (root.right != null)
            dfs(root.right, curr_path, all_path);
        return all_path;
    }

    public void dfs(TreeNode root, String curr_path, List all_path) {
        curr_path += "->" + root.val;
        if (root.left == null && root.right == null) {
            all_path.add(curr_path);
            return;
        }
        if (root.left != null)
            dfs(root.left, curr_path, all_path);
        if (root.right != null)
            dfs(root.right, curr_path, all_path);
    }
}
