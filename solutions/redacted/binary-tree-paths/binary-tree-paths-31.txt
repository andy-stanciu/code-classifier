class Solution {

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        method2(root, "", res);
        return res;
    }

    public void method2(TreeNode node, String str, List<String> res) {
        if (node == null)
            return;
        if (node.left == null && node.right == null) {
            res.add(str + node.val);
            return;
        }
        method2(node.left, str + node.val + "->", res);
        method2(node.right, str + node.val + "->", res);
    }
}
