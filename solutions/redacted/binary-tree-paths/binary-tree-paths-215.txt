class Solution {

    List<String> res;

    public List<String> method1(TreeNode root) {
        res = new ArrayList<>();
        method1(root, "");
        return res;
    }

    private void method1(TreeNode root, String path) {
        if (root.left == null && root.right == null) {
            path = path + root.val;
            res.add(path);
            return;
        }
        path = path + root.val + "->";
        if (root.left != null)
            method1(root.left, path);
        if (root.right != null)
            method1(root.right, path);
    }
}
