class Solution {

    List<String> list = new ArrayList<>();

    StringBuilder sb = new StringBuilder();

    public List<String> method1(TreeNode root) {
        if (root == null)
            return null;
        int len = sb.length();
        sb.append(root.val);
        if (root.left == null && root.right == null) {
            list.add(sb.toString());
        } else {
            sb.append("->");
            method1(root.left);
            method1(root.right);
        }
        sb.setLength(len);
        return list;
    }
}
