/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void method1(TreeNode node, List<Integer> path, List<List<Integer>> param1) {
        if (node.left == null && node.right == null) {
            path.add(node.val);
            param1.add(new ArrayList<>(path));
            path.remove(path.size() - 1);
            return;
        }
        path.add(node.val);
        if (node.left != null)
            method1(node.left, path, param1);
        if (node.right != null)
            method1(node.right, path, param1);
        path.remove(path.size() - 1);
    }

    public List<String> method2(TreeNode root) {
        List<Integer> path = new ArrayList<>();
        List<List<Integer>> param1 = new ArrayList<>();
        List<String> ans = new ArrayList<>();
        method1(root, path, param1);
        for (List<Integer> arr : param1) {
            StringBuffer str = new StringBuffer();
            for (int i = 0; i < arr.size() - 1; i++) {
                str.append(String.valueOf(arr.get(i)) + "->");
            }
            str.append(String.valueOf(arr.get(arr.size() - 1)));
            ans.add(str.toString());
        }
        return ans;
    }
}
