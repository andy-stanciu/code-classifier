class Solution {

    List<String> list = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        help(root, new StringBuilder());
        return list;
    }

    public void help(TreeNode root, StringBuilder s) {
        if (root == null)
            return;
        int v = s.length();
        if (v != 0)
            s.append("->");
        s.append(root.val);
        if (root.left == null && root.right == null)
            list.add(s.toString());
        help(root.left, s);
        help(root.right, s);
        s.setLength(v);
    }
}
