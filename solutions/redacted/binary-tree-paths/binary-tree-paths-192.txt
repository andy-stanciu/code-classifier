/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> var1 = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        String path = "";
        String default_string = "" + root.val + "->";
        order(root, path, default_string);
        return var1;
    }

    public void order(TreeNode root, String path, String default_string) {
        if (root == null)
            return;
        if (root != null) {
            if (root.left == null && root.right == null) {
                path += root.val;
                var1.add(path);
                path = default_string;
            }
            path += root.val + "->";
        }
        order(root.left, path, default_string);
        order(root.right, path, default_string);
    }
}
