class Solution {

    public List<String> method1(TreeNode root) {
        List<String> var1 = new ArrayList<>();
        if (root != null) {
            method2(root, "", var1);
        }
        return var1;
    }

    public void method2(TreeNode root, String ans, List<String> var1) {
        // Append the current node\'s value to the current path
        ans += root.val;
        if (root.left == null && root.right == null) {
            // If leaf node, add the current path to paths
            var1.add(ans);
            return;
        }
        if (root.left != null) {
            // Recursively call for left child
            method2(root.left, ans + "->", var1);
        }
        if (root.right != null) {
            // Recursively call for right child
            method2(root.right, ans + "->", var1);
        }
    }
}
