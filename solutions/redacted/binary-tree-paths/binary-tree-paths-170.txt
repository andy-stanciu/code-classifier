/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<String> var1 = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        if (root == null) {
            return var1;
        }
        StringBuilder str = new StringBuilder();
        method2(root, str);
        return var1;
    }

    public void method2(TreeNode root, StringBuilder str) {
        int length = str.length();
        str.append(root.val);
        if (root.left == null && root.right == null) {
            var1.add(str.toString());
        }
        str.append("->");
        if (root.left != null) {
            method2(root.left, str);
        }
        if (root.right != null) {
            method2(root.right, str);
        }
        str.setLength(length);
    }
}
