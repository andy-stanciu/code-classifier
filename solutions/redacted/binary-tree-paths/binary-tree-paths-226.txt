/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        if (root == null)
            return ans;
        String temp = "";
        postOrder(root, temp, ans);
        return ans;
    }

    public void postOrder(TreeNode root, String temp, List<String> ans) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
            //leaf
            temp += root.val;
            ans.add(temp);
            return;
        }
        temp += root.val + "->";
        if (root.left != null)
            postOrder(root.left, temp, ans);
        if (root.right != null)
            postOrder(root.right, temp, ans);
    }
}
