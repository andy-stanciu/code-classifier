/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    void check(List<String> ans, TreeNode root, String value, boolean first) {
        if (root == null) {
            return;
        }
        if (!first) {
            value += "->";
        }
        first = false;
        value += root.val;
        if (root.left == null && root.right == null) {
            ans.add(value);
            return;
        }
        check(ans, root.left, value, false);
        check(ans, root.right, value, false);
    }

    public List<String> method1(TreeNode root) {
        List<String> ans = new ArrayList<>();
        check(ans, root, "", true);
        return ans;
    }
}
