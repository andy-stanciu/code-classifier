/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<String> list = new ArrayList<>();

    public List<String> method1(TreeNode root) {
        String path = "";
        if (root.left == null && root.right == null) {
            path += root.val;
            list.add(path);
            return list;
        }
        method2(root, path);
        return list;
    }

    void method2(TreeNode root, String str) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            str += root.val;
            list.add(str);
            return;
        } else {
            str += root.val + "->";
        }
        method2(root.left, str);
        method2(root.right, str);
    }
}
