/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> al = new ArrayList<>();

    public void solve(TreeNode root, String x) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            String ans = x + "->" + root.val;
            ans = ans.substring(2);
            al.add(ans);
            return;
        }
        solve(root.left, x + "->" + root.val);
        solve(root.right, x + "->" + root.val);
    }

    public List<String> method1(TreeNode root) {
        solve(root, "");
        return al;
    }
}
