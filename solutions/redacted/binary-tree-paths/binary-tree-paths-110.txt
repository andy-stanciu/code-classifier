/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<String>();
        if (root.left == null && root.right == null)
            return new ArrayList<String>() {

                {
                    add(Integer.toString(root.val));
                }
            };
        List<String> rightList = method1(root.right);
        List<String> leftList = method1(root.left);
        List<String> result = new ArrayList();
        for (String s : leftList) result.add(Integer.toString(root.val) + "->" + s);
        for (String s : rightList) result.add(Integer.toString(root.val) + "->" + s);
        return result;
    }
}
