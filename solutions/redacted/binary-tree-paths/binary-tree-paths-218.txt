/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static void s1(TreeNode t, List<String> l, String s) {
        if (t == null) {
            return;
        }
        if (t.left == null && t.right == null) {
            s += Integer.toString(t.val);
            l.add(s);
            int g = (Integer.toString(t.val)).length();
            s = s.substring(0, s.length() - g);
            return;
        }
        s += Integer.toString(t.val) + "->";
        s1(t.left, l, s);
        s1(t.right, l, s);
    }

    public List<String> method1(TreeNode root) {
        List<String> l = new ArrayList<>();
        s1(root, l, "");
        return l;
    }
}
