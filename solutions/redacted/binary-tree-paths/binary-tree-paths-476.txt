/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {

    public List<String> method1(TreeNode root) {
        List<String> res = new ArrayList<>();
        method2(root, res, "");
        return res;
    }

    public void method2(TreeNode root, List<String> res, String curr) {
        if (root == null)
            return;
        curr += root.val;
        method2(root.left, res, curr + "->");
        method2(root.right, res, curr + "->");
        if (root.left == null && root.right == null)
            res.add(curr);
    }
}
