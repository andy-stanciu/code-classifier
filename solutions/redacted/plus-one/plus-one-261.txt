class Solution {

    public int[] method1(int[] digits) {
        // Traverse the digits array from the end towards the start
        for (int i = digits.length - 1; i >= 0; i--) {
            // If the digit is less than 9, we can safely add 1 and return
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the digit is 9, it becomes 0 and we continue to the next digit
            digits[i] = 0;
        }
        // If we have reached this point, it means all the digits were 9
        // Hence, we need to create a new array with an extra digit (e.g., 999 + 1 = 1000)
        int[] newNumber = new int[digits.length + 1];
        // The first digit is 1, and the rest are already 0 by default
        newNumber[0] = 1;
        return newNumber;
    }
}
