class Solution {

    public int[] method1(int[] nums) {
        int l = nums.length;
        for (int i = l - 1; i >= 0; i--) {
            if (nums[i] == 9) {
                nums[i] = 0;
                continue;
            }
            nums[i]++;
            return nums;
        }
        int[] n = new int[l + 1];
        n[0] = 1;
        return n;
    }
}
