class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Start from the end of the digits array
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                // No carry needed, just increment the digit and return the result
                digits[i]++;
                return digits;
            }
            // Set current digit to 0 and move to the next significant digit
            digits[i] = 0;
        }
        // If we are here, it means all digits were 9, so we need an extra digit
        int[] newDigits = new int[n + 1];
        // Set the most significant digit to 1
        newDigits[0] = 1;
        return newDigits;
    }
}
