class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        if (n == 0) {
            return digits;
        }
        int i = n - 1;
        while (i >= 0) {
            if (digits[i] < 9) {
                digits[i] += 1;
                return digits;
            }
            digits[i] = 0;
            i--;
        }
        digits = new int[n + 1];
        digits[0] = 1;
        return digits;
    }
}
