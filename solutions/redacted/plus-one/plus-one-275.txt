class Solution {

    public int[] method1(int[] digits) {
        // Start with a carry of 1 because we are adding one
        int carry = 1;
        // Traverse the digits array from the last element to the first
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] == 9 && carry == 1) {
                // If digit is 9, it becomes 0, and we carry over 1
                digits[i] = 0;
            } else {
                // Add one to the current digit
                digits[i] += 1;
                // No carry needed anymore, return the digits
                carry = 0;
                return digits;
            }
        }
        // If we exit the loop with carry still being 1, we need a larger array
        int[] number = new int[digits.length + 1];
        // Set the first element to 1, rest are 0s by default
        number[0] = 1;
        return number;
    }
}
