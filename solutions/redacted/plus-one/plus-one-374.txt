class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Start from the last digit
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        // If all digits were 9, we need a new array
        int[] newNumber = new int[n + 1];
        // The first digit will be 1
        newNumber[0] = 1;
        return newNumber;
    }
}
