class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            // If the current digit is less than 9, just increment it and return the array
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the current digit is 9, set it to 0 (carry over)
            digits[i] = 0;
        }
        // If all digits were 9, we need a new array with an additional digit
        int[] newDigits = new int[digits.length + 1];
        // Set the first digit to 1, rest are already 0 by default
        newDigits[0] = 1;
        return newDigits;
    }
}
