class Solution {

    public int[] method1(int[] arr) {
        //4,3,2,9,9
        int n = arr.length;
        if (arr[n - 1] < 9) {
            arr[n - 1] = arr[n - 1] + 1;
            return arr;
        } else {
            String ans = "";
            int i = 0;
            for (i = n - 1; i >= 0; i--) {
                if (arr[i] == 9) {
                    arr[i] = 0;
                } else {
                    break;
                }
            }
            if (i >= 0) {
                arr[i] = arr[i] + 1;
            } else {
                int[] temp = new int[n + 1];
                temp[0] = 1;
                return temp;
            }
        }
        return arr;
    }
}
