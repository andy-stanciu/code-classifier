class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] == 9) {
                digits[i] = 0;
            } else {
                digits[i] = digits[i] + 1;
                return digits;
            }
        }
        int[] Digits = new int[n + 1];
        Digits[0] = 1;
        return Digits;
    }
}
