class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Start from the last and iterate backwards
        for (int i = n - 1; i >= 0; i--) {
            // If the digit < 9, increment it and return the array
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the digit is 9, it becomes 0 and the carry goes to the next digit
            digits[i] = 0;
        }
        // If all the digits were 9, the result will be 100
        int[] newNumber = new int[n + 1];
        newNumber[0] = 1;
        return newNumber;
    }
}
