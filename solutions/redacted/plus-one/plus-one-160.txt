class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Start from the last digit and process towards the first
        for (int i = n - 1; i >= 0; i--) {
            // If the current digit is less than 9, just increment it and return
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the current digit is 9, set it to 0 and continue to the next digit
            digits[i] = 0;
        }
        // If all digits were 9, we need an extra space for the new digit
        int[] newNumber = new int[n + 1];
        // Set the first digit to 1
        newNumber[0] = 1;
        // The rest are already initialized to 0
        return newNumber;
    }
}
