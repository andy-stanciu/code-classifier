class Solution {

    public int[] method1(int[] digits) {
        int i = digits.length - 1;
        while (i > -1 && digits[i] == 9) {
            //Tring to find till which position continious 9 is present
            i--;
        }
        if (i == -1) {
            // when complete array is of value 9.
            // we need new array because previous array memory is not enough.
            int[] A = new int[digits.length + 1];
            A[0] = 1;
            return A;
        } else {
            i = digits.length - 1;
            while (digits[i] == 9) {
                //making all 9 as 0
                digits[i] = 0;
                i--;
            }
            //incresing +1 in the value present just     beside last 9.
            digits[i] = digits[i] + 1;
        }
        return digits;
    }
}
