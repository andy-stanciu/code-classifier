class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] == 9) {
                digits[i] = 0;
            } else {
                digits[i] += 1;
                return digits;
            }
        }
        // If we reached here, it means all digits were 9, so we need to create a new array
        int[] newDigit = new int[digits.length + 1];
        // Set the first element to 1, the rest will be 0 by default
        newDigit[0] = 1;
        return newDigit;
    }
}
