class Solution {

    public int[] method1(int[] digits) {
        // Iterate over digits from last elem to first
        for (int i = digits.length - 1; i >= 0; i--) {
            // if the digit is 9 then remove it 0 (to make it 10) else add 1.
            if (digits[i] == 9) {
                digits[i] = 0;
            } else {
                digits[i]++;
                break;
            }
        }
        // In case the most significant bit was a 9 which became \'0\' in previous loop
        if (digits[0] == 0) {
            int[] ans = new int[digits.length + 1];
            ans[0] = 1;
            ans[1] = 0;
            for (int i = 2; i < digits.length; i++) {
                ans[i] = digits[i];
            }
            return ans;
        }
        return digits;
    }
}
