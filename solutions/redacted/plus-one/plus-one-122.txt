class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            // current digit is less than 9 => increment it and return the array
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // Otherwise, set the current digit to 0 and continue the loop
            digits[i] = 0;
        }
        // If here, it means all the digits were 9, create a larger int[] first digit to 1, all others are 0s
        int[] result = new int[digits.length + 1];
        result[0] = 1;
        return result;
    }
}
