class Solution {

    public int[] method1(int[] digits) {
        int[] ans = new int[digits.length + 1];
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] == 9) {
                if (i != 0) {
                    digits[i] = 0;
                    ans[i] = 0;
                } else {
                    ans[i] = 1;
                    ans[ans.length - 1] = 0;
                    return ans;
                }
            } else {
                digits[i] = digits[i] + 1;
                return digits;
            }
        }
        return digits;
    }
}
