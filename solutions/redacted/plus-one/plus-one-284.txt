class Solution {

    public int[] method1(int[] digits) {
        //Traverse from  arrays (digits)last index to first
        for (int farukIndex = digits.length - 1; farukIndex >= 0; farukIndex--) {
            // increment the index value
            digits[farukIndex]++;
            // Check the current digit is less than 10
            if (digits[farukIndex] < 10) {
                // No carry, return the result
                return digits;
            }
            // If digit is 10, set it to 0 and continue to the next digit
            digits[farukIndex] = 0;
        }
        // If the loop completes, all digits were 9, and we need a carry
        int[] result = new int[digits.length + 1];
        // here Set the most significant digit to 1
        result[0] = 1;
        // here The rest of the array is already initialized  0
        return result;
    }
}
