class Solution {

    public int[] method1(int[] digits) {
        int carry = 1;
        int n = digits.length;
        // loop to add the digits
        for (int i = n - 1; i >= 0; i--) {
            // add carry in every digit
            digits[i] += carry;
            // update carry for next digit
            carry = digits[i] / 10;
            //  put remainder in the digit
            digits[i] = digits[i] % 10;
        }
        if (carry == 1) {
            // in case of carry, we need to return n+1 length array
            int[] res = new int[n + 1];
            res[0] = 1;
            // assinging values of digit in new array
            for (int i = 1; i <= n; i++) {
                res[i] = digits[i - 1];
            }
            return res;
        }
        // not a carry, we can send updated digit array
        return digits;
    }
}
