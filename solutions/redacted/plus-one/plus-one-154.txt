class Solution {

    public int[] method1(int[] digits) {
        int rem = 1;
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] != 9) {
                digits[i]++;
                rem = 0;
            } else {
                digits[i] = 0;
                rem = 1;
            }
            if (rem == 0) {
                return digits;
            }
        }
        if (rem == 1) {
            int[] arr = new int[digits.length + 1];
            arr[0] = 1;
            for (int i = 1; i < arr.length; i++) {
                arr[i] = digits[i - 1];
            }
            return arr;
        }
        return digits;
    }
}
