class Solution {

    public int[] method1(int[] digits) {
        int counter = digits.length;
        while (counter > 0) {
            if (digits[counter - 1] + 1 == 10) {
                digits[counter - 1] = 0;
                counter--;
            } else {
                digits[counter - 1] = digits[counter - 1] + 1;
                return digits;
            }
        }
        digits = Arrays.copyOf(digits, digits.length + 1);
        digits[0] = 1;
        return digits;
    }
}
