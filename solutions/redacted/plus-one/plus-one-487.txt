class Solution {

    public int[] method1(int[] digits) {
        // store length of the digits array for easy use
        int x = digits.length;
        // iterate through
        for (int i = x - 1; i >= 0; i--) {
            // this is the base case and simple case
            if (digits[i] != 9) {
                // simply just add one and we are done
                digits[i]++;
                return digits;
            } else {
                // if it is nine that means we go up a place
                digits[i] = 0;
                // we go back into the loop to add 1 to another place
            }
        }
        // if it exits the loop, that we means we have a case where we have to add another place
        // can\'t figure out elegant solution to simply add 1 to the beginning
        int[] edgeDigits = new int[x + 1];
        edgeDigits[0] = 1;
        // i am simply just copying the digits into a new array with just 1 in the 0th index
        for (int j = 0; j < x; j++) {
            edgeDigits[j + 1] = digits[j];
        }
        return edgeDigits;
    }
}
