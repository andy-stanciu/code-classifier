class Solution {

    public int[] method1(int[] digits) {
        int carry = 0;
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] == 9) {
                digits[i] = 0;
                carry = 1;
            } else {
                digits[i] = digits[i] + 1;
                carry = 0;
            }
            if (carry == 0)
                break;
        }
        if (carry == 1) {
            int[] ans = new int[digits.length + 1];
            for (int i = 0; i < digits.length; i++) {
                ans[i + 1] = digits[i];
            }
            ans[0] = 1;
            return ans;
        }
        return digits;
    }
}
