class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            // Check if the current digit is less than 9
            if (digits[i] < 9) {
                // Increment the digit and return the updated array
                digits[i]++;
                return digits;
            }
            // Set the current digit to 0 if it was 9
            digits[i] = 0;
        }
        // If all digits were 9, create a new array with an additional digit
        digits = new int[digits.length + 1];
        // Set the first digit to 1
        digits[0] = 1;
        // Return the new array
        return digits;
    }
}
