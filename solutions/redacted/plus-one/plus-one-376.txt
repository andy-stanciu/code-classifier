class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        int[] result = new int[n + 1];
        int i = n - 1;
        while (i != -1 && digits[i] == 9) {
            digits[i] = 0;
            i--;
        }
        if (i == -1) {
            result[0] = 1;
            return result;
        }
        digits[i] += 1;
        return digits;
    }
}
