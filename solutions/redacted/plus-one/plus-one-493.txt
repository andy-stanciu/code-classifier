class Solution {

    public int[] method1(int[] inp) {
        int[] op = new int[inp.length];
        int carry = 0;
        int add = 1;
        for (int i = inp.length - 1; i >= 0; i--) {
            //
            int update = inp[i] + add + carry;
            op[i] = update % 10;
            if ((update) >= 10) {
                carry = 1;
            } else {
                carry = 0;
            }
            add = 0;
        }
        int[] res = new int[inp.length + 1];
        if (carry == 1) {
            res[0] = carry;
            for (int i = 0; i < op.length; i++) {
                res[i + 1] = op[i];
            }
        }
        return carry == 1 ? res : op;
    }
}
