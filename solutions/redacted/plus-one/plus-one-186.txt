class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        if (n == 0) {
            return digits;
        }
        return method2(digits, n - 1);
    }

    private int[] method2(int[] digits, int idx) {
        if (idx == -1) {
            int[] newDigits = new int[digits.length + 1];
            // Insert 1 at the start
            newDigits[0] = 1;
            // Rest of the array will be 0 by default
            return newDigits;
        } else {
            if (digits[idx] < 9) {
                digits[idx]++;
                return digits;
            } else {
                digits[idx] = 0;
                return method2(digits, idx - 1);
            }
        }
    }
}
