class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        // this is the last case ([9,9,9]) that all elements in digits were 9 so we know initailly every element in the array is 0 so we just need to put the first element as 1 in the new array
        int[] arr = new int[n + 1];
        arr[0] = 1;
        return arr;
    }
}
