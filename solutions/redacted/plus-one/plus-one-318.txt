class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        // if the array contains only 9
        int[] edgecase = new int[n + 1];
        edgecase[0] = 1;
        return edgecase;
    }
}
