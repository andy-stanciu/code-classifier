public class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Traverse the array from the last digit to the first
        for (int i = n - 1; i >= 0; i--) {
            // If the current digit is less than 9, just increment it and return the array
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // If the digit is 9, it becomes 0 (carry over)
            digits[i] = 0;
        }
        // If we have exited the loop, it means all digits were 9 and we need an extra 1 at the beginning
        int[] newNumber = new int[n + 1];
        // The first digit is 1 all others are already 0
        newNumber[0] = 1;
        return newNumber;
    }
}
