class Solution {

    public int[] method1(int[] digits) {
        int originalArraySize = digits.length;
        int[] result = digits.clone();
        int[] var1;
        int i = originalArraySize - 1;
        while (true) {
            boolean var2 = false;
            if (digits[i] == 9) {
                var2 = true;
                result[i] = 0;
            } else
                result[i] += 1;
            i--;
            if (var2 && i < 0) {
                var1 = new int[originalArraySize + 1];
                var1[0] = 1;
                for (int j = 0; j < originalArraySize; j++) {
                    var1[j + 1] = result[j];
                }
                return var1;
            }
            if (!var2)
                break;
        }
        return result;
    }
}
