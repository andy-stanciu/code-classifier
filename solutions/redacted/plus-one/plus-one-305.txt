class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                /*
                each iteration starts from end, 
                check if every digit from end is less than 9,
                then just increment by 1 and just return the same,
                task is done.
                */
                digits[i] += 1;
                return digits;
            }
            /* 
            if we get digit equal to 9 in iteration just make that 
            element to 0 bcz adding +1 to 9 becomes \'10\' as last digit 
            in \'10\' is \'0\' and 1st digit \'1\' will be checked in next 
            iteration as iteration happening reverse 
            */
            digits[i] = 0;
        }
        /*
        if you reach till here means, your number array contains only 
        9 in all digits, bcz of that you haven\'t entered into if cond
        in for loop so that your code executed only line-19 in all 
        iteration and end up in not reaching return statement at line-13.
        
        which means now you have to handle situation if your array
        contains only 9999\'s. which means your OP will be one extra number 
        at starting (10000), which will increase size of current array 
        by 1. so create a new array from past array size + 1 and add 
        0(Zero) to 0th index as thats the only place where your arr size 
        incresed and return that new array.
        */
        int[] res = new int[digits.length + 1];
        res[0] = 1;
        return res;
    }
}
