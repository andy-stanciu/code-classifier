class Solution {

    public int[] method1(int[] digits) {
        int nl = digits.length;
        digits[nl - 1]++;
        for (int i = nl - 1; i >= 0; i--) {
            if (digits[i] == 10) {
                digits[i] = 0;
                if (i == 0) {
                    int[] newDigits = new int[nl + 1];
                    newDigits[0] = 1;
                    for (int j = 1; j < nl; j++) {
                        newDigits[j] = digits[j - 1];
                    }
                    return newDigits;
                }
                digits[i - 1]++;
            }
        }
        return digits;
    }
}
