class Solution {

    public int[] method1(int[] digits) {
        //start from the back of the list
        for (int i = digits.length - 1; i >= 0; i--) {
            //handles what happens when the digit is LESS than 9
            if (digits[i] < 9) {
                //add 1 to the last number and return it
                digits[i] += 1;
                return digits;
            }
            // if digit is \'9\' make it \'0\'
            digits[i] = 0;
        }
        //account for the carry by making new list with extra digit
        int[] result = new int[digits.length + 1];
        //if the for loop had been completed that mean the array was all \'9\'s\'
        //meanig the first digit should be 1
        result[0] = 1;
        return result;
    }
}
