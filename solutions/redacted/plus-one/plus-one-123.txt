class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        int sum = 0;
        int carry = 1;
        // Iterate through the digits in reverse order
        for (int i = n - 1; i >= 0; i--) {
            sum = digits[i] + carry;
            digits[i] = sum % 10;
            carry = sum / 10;
        }
        // If there is a carry after processing all digits, create a new array
        if (carry > 0) {
            int[] result = new int[n + 1];
            result[0] = carry;
            System.arraycopy(digits, 0, result, 1, n);
            return result;
        }
        return digits;
    }
}
