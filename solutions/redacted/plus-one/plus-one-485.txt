class Solution {

    public int[] method1(int[] digits) {
        for (int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] != 9) {
                // If the digit is less than 9
                // Increment the digit by one
                digits[i]++;
                // Return the modified array
                return digits;
            }
            // If the digit is 9, set it to 0 and continue to the next digit
            digits[i] = 0;
        }
        // If all digits were 9, create a new array with an extra digit
        int[] newDigits = new int[digits.length + 1];
        // Set the most significant digit to 1
        newDigits[0] = 1;
        // Remaining digits are already 0 by default
        return newDigits;
    }
}
