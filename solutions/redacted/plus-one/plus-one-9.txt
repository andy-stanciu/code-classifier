class Solution {

    public int[] method1(int[] digits) {
        int carry = 1;
        int digit = 0;
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            digit = digits[i] + carry;
            if (digit > 9) {
                carry = 1;
            } else {
                carry = 0;
            }
            digits[i] = digit % 10;
        }
        int[] res = new int[n + 1];
        if (carry == 1) {
            res[0] = 1;
            int j = 1;
            for (int i = 0; i < n; i++) {
                res[j] = digits[i];
                j++;
            }
            return res;
        }
        return digits;
    }
}
