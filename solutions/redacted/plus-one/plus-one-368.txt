class Solution {

    public int[] method1(int[] digits) {
        int carry = 1;
        List<Integer> list = new LinkedList<>();
        int n = digits.length;
        for (int i = n - 1; i >= 0; i--) {
            if (carry + digits[i] >= 10) {
                list.add(0, 0);
                carry = 1;
            } else {
                list.add(0, carry + digits[i]);
                carry = 0;
            }
        }
        if (carry == 1) {
            list.add(0, carry);
        }
        System.out.println(list);
        // Object[] arr=list.toArray();
        return list.stream().mapToInt(Integer::intValue).toArray();
    }
}
