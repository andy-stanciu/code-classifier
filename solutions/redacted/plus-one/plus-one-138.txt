class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        digits[n - 1]++;
        if (digits[n - 1] < 10)
            return digits;
        int i = n - 1;
        while (i > 0) {
            if (digits[i] < 10) {
                break;
            } else {
                digits[i] = 0;
                digits[i - 1]++;
                i--;
            }
        }
        if (digits[0] > 9) {
            int[] ans = new int[n + 1];
            System.arraycopy(digits, 0, ans, 1, n);
            ans[0] = 1;
            ans[1] = 0;
            return ans;
        } else
            return digits;
    }
}
