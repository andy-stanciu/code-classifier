class Solution {

    public static int[] method1(int[] digits) {
        int n = digits.length;
        //Traverse from the last digit to the first
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            //if the current digit is 9, set it to 0 and continue tht loop
            digits[i] = 0;
        }
        //if all elements were 9 , we need an extra digit at beginning
        int[] newDigits = new int[n + 1];
        //set the first digit to 1, reset are 0 by default
        newDigits[0] = 1;
        return newDigits;
    }

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int n = scan.nextInt();
        int[] ar = new int[n];
        for (int i = 0; i < ar.length; i++) {
            ar[i] = scan.nextInt();
        }
        int[] result = method1(ar);
        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i] + " ");
        }
    }
}
