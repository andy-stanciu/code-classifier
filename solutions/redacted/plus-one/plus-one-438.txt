class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Start from the last digit and work backwards
        for (int i = n - 1; i >= 0; i--) {
            // If the current digit is less than 9, just increment and return
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            // Set the current digit to 0 if it\'s 9
            digits[i] = 0;
        }
        // If we exited the loop, all digits were 9 and became 0
        // We need an additional array with one more digit (e.g., 999 + 1 = 1000)
        int[] newNumber = new int[n + 1];
        // The first digit is 1, rest are 0s
        newNumber[0] = 1;
        return newNumber;
    }
}
