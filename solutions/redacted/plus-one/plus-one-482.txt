class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        // Traverse from the last digit to the first digit
        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                // Increment the digit
                digits[i]++;
                // Return the updated digits
                return digits;
            }
            // Set the digit to 0 if it was 9
            digits[i] = 0;
        }
        // If we are here, it means all digits were 9
        int[] newNumber = new int[n + 1];
        // Set the first digit to 1
        newNumber[0] = 1;
        // The rest are 0s by default in a new int array
        return newNumber;
    }
}
