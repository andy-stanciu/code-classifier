class Solution {

    public int[] method1(int[] digits) {
        int[] ans = digits;
        // start at the end, increment the last digit and then each after that
        for (int i = digits.length - 1; i >= 0; i--) {
            ans[i]++;
            if (ans[i] == 10) {
                ans[i] = 0;
                if (i == 0) {
                    // start of the array, need bigger array
                    ans = increaseArraySize(digits);
                }
            } else {
                break;
            }
        }
        return ans;
    }

    private int[] increaseArraySize(int[] digits) {
        // create new array
        int[] ans = new int[digits.length + 1];
        // set first number to 1
        ans[0] = 1;
        // start at the end, copy over the digits
        for (int i = digits.length - 1; i >= 0; i--) {
            ans[i + 1] = digits[i];
        }
        return ans;
    }
}
