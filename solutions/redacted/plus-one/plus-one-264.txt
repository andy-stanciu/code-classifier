class Solution {

    public int[] method1(int[] digits) {
        int i = digits.length - 1;
        int nums = digits[i] + 1;
        int carry = 0;
        if (nums <= 9) {
            digits[i] = nums;
        } else {
            digits[i] = 0;
            carry = 1;
        }
        for (int j = i - 1; j >= 0; j--) {
            if (carry != 0) {
                digits[j] = digits[j] + 1;
                if (digits[j] > 9) {
                    digits[j] = 0;
                } else {
                    carry = 0;
                }
            }
        }
        if (carry != 0) {
            int[] newDigits = new int[digits.length + 1];
            newDigits[0] = 1;
            for (int a = 0; a < digits.length; a++) {
                newDigits[a + 1] = digits[a];
            }
            return newDigits;
        }
        return digits;
    }
}
