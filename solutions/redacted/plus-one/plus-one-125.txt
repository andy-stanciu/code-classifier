class Solution {

    public int[] method1(int[] digits) {
        int n = digits.length;
        if (digits[n - 1] != 9) {
            digits[n - 1]++;
            return digits;
        } else {
            int[] ans = new int[n + 1];
            int i = n - 1;
            while (digits[i] == 9 && i > 0) {
                if (!(digits[i - 1] == 9)) {
                    digits[i] = 0;
                    digits[i - 1]++;
                    return digits;
                } else {
                    digits[i] = 0;
                    ans[i] = 0;
                }
                i--;
            }
            ans[0] = 1;
            return ans;
        }
    }
}
