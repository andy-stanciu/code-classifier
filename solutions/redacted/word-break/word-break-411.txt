class Solution {

    Boolean[] dp = null;

    public boolean method1(String s, List<String> param1) {
        dp = new Boolean[s.length() + 1];
        return method2(s, 0, param1);
    }

    public boolean method2(String s, int i, List<String> param1) {
        if (i == s.length())
            return true;
        if (dp[i] != null)
            return dp[i];
        for (int j = i; j < s.length(); j++) {
            if (method3(s, i, j + 1, param1) && method2(s, j + 1, param1)) {
                return dp[i] = true;
            }
        }
        return dp[i] = false;
    }

    public boolean method3(String s, int l, int r, List<String> param1) {
        String var1 = s.substring(l, r);
        return param1.contains(var1);
    }
}
