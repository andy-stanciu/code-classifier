class Solution {

    boolean[] seen;

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        seen = new boolean[n + 1];
        seen[0] = true;
        return dfs(0, s, param1, n);
    }

    public boolean dfs(int curNode, String s, List<String> param1, int n) {
        if (curNode == n)
            return true;
        for (String var1 : param1) {
            int next = curNode + var1.length();
            if (next <= n && s.substring(curNode, next).equals(var1) && !seen[next]) {
                seen[next] = true;
                if (dfs(next, s, param1, n)) {
                    return true;
                }
            }
        }
        return false;
    }
}
