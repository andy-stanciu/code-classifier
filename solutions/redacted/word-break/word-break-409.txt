class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        int[] dp = new int[n];
        Arrays.fill(dp, -1);
        return solve(0, s, param1, dp);
    }

    boolean solve(int index, String str, List<String> param1, int[] dp) {
        int n = str.length();
        if (index >= str.length())
            return true;
        if (dp[index] != -1) {
            if (dp[index] == 1) {
                return true;
            } else {
                return false;
            }
        }
        boolean res = false;
        for (int i = 0; i < param1.size(); i++) {
            boolean temp = false;
            if (validate(index, str, param1.get(i))) {
                temp = solve(index + param1.get(i).length(), str, param1, dp);
            }
            if (temp == true) {
                res = true;
                break;
            }
        }
        if (res == true) {
            dp[index] = 1;
        } else {
            dp[index] = 0;
        }
        return res;
    }

    boolean validate(int index, String str, String param2) {
        if (index + param2.length() - 1 >= str.length())
            return false;
        boolean res = true;
        for (int i = 0; i < param2.length(); i++) {
            if (param2.charAt(i) != str.charAt(index + i)) {
                res = false;
                break;
            }
        }
        return res;
    }
}
