class Solution {

    HashMap<String, Boolean> inDict;

    public Solution() {
        inDict = new HashMap<>();
    }

    public boolean method1(String s, List<String> param1) {
        if (s.length() == 0) {
            return true;
        }
        if (inDict.containsKey(s)) {
            return inDict.get(s);
        }
        for (String var1 : param1) {
            if (s.startsWith(var1)) {
                if (method1(s.substring(var1.length()), param1)) {
                    inDict.put(s, true);
                    return true;
                }
                // inDict.put(s, false);
                // return false;
            }
        }
        inDict.put(s, false);
        return false;
    }
}
