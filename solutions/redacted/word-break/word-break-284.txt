class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> var1 = new HashSet<>(param1);
        int n = s.length();
        boolean[] f = new boolean[n + 1];
        // base case
        f[0] = true;
        for (int i = 1; i <= n; i++) {
            String newInput = s.substring(0, i);
            for (int j = 0; j < i; j++) {
                // time complexity is n^2 / 2, i.e. n ^ 2
                if (var1.contains(newInput.substring(j)) && f[j]) {
                    // n ^ 3
                    f[i] = true;
                    break;
                }
            }
        }
        return f[n];
    }
}
