class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        Set<String> set = new HashSet(param1);
        boolean[] dp = new boolean[n + 1];
        dp[n] = true;
        if (set.contains("" + s.charAt(n - 1)))
            dp[n - 1] = true;
        for (int i = n - 2; i >= 0; i--) {
            for (int j = i + 1; j <= n; j++) {
                if (dp[j]) {
                    if (set.contains(s.substring(i, j))) {
                        dp[i] = true;
                        break;
                    }
                }
            }
        }
        return dp[0];
    }
}
