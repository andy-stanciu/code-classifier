class Solution {

    private String s;

    private List<String> var1;

    private boolean[] arr;

    private int n;

    public boolean method1(String s, List<String> var1) {
        this.s = s;
        this.var1 = var1;
        n = s.length();
        arr = new boolean[n + 1];
        return solve(0);
    }

    private boolean solve(int start) {
        if (start == n)
            return true;
        for (String var2 : var1) {
            if (s.startsWith(var2, start)) {
                int len = start + var2.length();
                if (!arr[len]) {
                    if (solve(len))
                        return true;
                    arr[len] = true;
                }
            }
        }
        return false;
    }
}
