public class Solution {

    private HashMap<String, Boolean> memo = new HashMap<>();

    public boolean method1(String s, List<String> param1) {
        return fio(s, param1);
    }

    boolean fio(String s, List<String> param1) {
        if (s.isEmpty()) {
            return true;
        }
        if (memo.containsKey(s)) {
            return memo.get(s);
        }
        for (String var1 : param1) {
            if (s.startsWith(var1)) {
                if (fio(s.substring(var1.length()), param1)) {
                    memo.put(s, true);
                    return true;
                }
            }
        }
        memo.put(s, false);
        return false;
    }
}
