class Solution {

    public boolean method1(String s, List<String> l1) {
        boolean[] dp = new boolean[s.length() + 1];
        dp[0] = true;
        for (int i = 1; i <= s.length(); i++) {
            for (int j = 0; j < i; j++) {
                if (l1.contains(s.substring(j, i)) && dp[j]) {
                    dp[i] = true;
                    System.out.println(i);
                }
            }
        }
        return dp[s.length()];
    }
}
