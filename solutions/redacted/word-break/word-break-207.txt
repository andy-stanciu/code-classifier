class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        Set<String> var1 = new HashSet<>(param1);
        // dp[i] represents whether the substring s[0...i] can be segmented into words from the dictionary
        boolean[] dp = new boolean[n + 1];
        // Base case: an empty string can be segmented
        dp[0] = true;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                if (dp[j] && var1.contains(s.substring(j, i))) {
                    // Update dp[i] if s[j...i) exists in the dictionary and dp[j] is true
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[n];
    }
}
