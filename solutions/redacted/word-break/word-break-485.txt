class Solution {

    HashMap<String, Boolean> dp = new HashMap<>();

    public boolean method1(String s, List<String> param1) {
        if (s.length() == 0)
            return true;
        if (dp.containsKey(s))
            return dp.get(s);
        boolean canForm = false;
        for (String var1 : param1) {
            if (var1.length() > s.length())
                continue;
            if (s.startsWith(var1)) {
                if (method1(s.substring(var1.length()), param1)) {
                    canForm = true;
                    break;
                }
            }
        }
        dp.put(s, canForm);
        return dp.get(s);
    }
}
