class Solution {

    public boolean method1(String s, List<String> param1) {
        Boolean[] memo = new Boolean[s.length()];
        return helper(s, param1, "", 0, memo);
    }

    public boolean helper(String s, List<String> param1, String curStr, int index, Boolean[] memo) {
        if (index == s.length()) {
            return curStr == "";
        }
        if (memo[index] != null) {
            return memo[index];
        }
        for (int i = index; i < s.length(); i++) {
            curStr += s.charAt(i);
            if (param1.contains(curStr)) {
                if (helper(s, param1, "", i + 1, memo))
                    return memo[index] = true;
            }
        }
        return memo[index] = false;
    }
}
