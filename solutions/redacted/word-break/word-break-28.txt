class Solution {

    public boolean method1(String s, List<String> param1) {
        Boolean[] dp = new Boolean[s.length()];
        return helper(0, s, param1, dp);
    }

    public boolean helper(int index, String s, List<String> param1, Boolean[] dp) {
        //step 3
        int n = s.length();
        if (index == n)
            return true;
        if (dp[index] != null)
            return dp[index];
        for (int k = index + 1; k <= n; k++) {
            //step 2
            if (param1.contains(s.substring(index, k)) && helper(k, s, param1, dp))
                return dp[index] = true;
        }
        return dp[index] = false;
    }
}
