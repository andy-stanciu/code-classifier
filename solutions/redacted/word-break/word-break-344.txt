class Solution {

    public boolean compare(String s1, String s2) {
        return s1.equals(s2);
    }

    public boolean recur(String s, List<String> param1, int i, Boolean[] dp) {
        if (i >= s.length())
            return true;
        if (dp[i] != null)
            return dp[i];
        for (String s2 : param1) {
            if (i + s2.length() <= s.length() && compare(s.substring(i, i + s2.length()), s2)) {
                boolean ans = recur(s, param1, i + s2.length(), dp);
                if (ans)
                    return dp[i] = true;
            }
        }
        return dp[i] = false;
    }

    public boolean method1(String s, List<String> param1) {
        Boolean[] dp = new Boolean[s.length() + 1];
        return recur(s, param1, 0, dp);
    }
}
