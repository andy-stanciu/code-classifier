class Solution {

    public boolean method1(String s, List<String> param1) {
        if (param1.contains(s))
            return true;
        for (int i = 1; i <= s.length(); i++) {
            String left = s.substring(0, i);
            if (param1.contains(left) && method1(s.substring(i), param1)) {
                return true;
            }
        }
        return false;
    }
}
