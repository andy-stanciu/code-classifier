class Solution {

    private boolean solve(String s, List<String> param1, HashSet<String> set) {
        if (s.length() == 0)
            return true;
        if (set.contains(s))
            return false;
        for (int j = 0; j < param1.size(); j++) {
            String curr = param1.get(j);
            if (curr.length() <= s.length() && s.substring(0, curr.length()).equals(curr)) {
                boolean isTrue = solve(s.substring(curr.length()), param1, set);
                if (isTrue)
                    return true;
            }
        }
        set.add(s);
        return false;
    }

    public boolean method1(String s, List<String> param1) {
        return solve(s, param1, new HashSet<>());
    }
}
