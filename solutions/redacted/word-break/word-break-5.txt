class Solution {

    public boolean method1(String s, List<String> param1) {
        // if we found word break for whole string, return true
        if (s.isEmpty())
            return true;
        for (String var1 : param1) {
            int n = var1.length();
            if (s.length() >= n) {
                // if word is prefix of s, then break s and look for word break in remaining string
                if (s.startsWith(var1)) {
                    boolean b = method1(s.substring(n), param1);
                    if (b)
                        return b;
                }
            }
        }
        return false;
    }
}
