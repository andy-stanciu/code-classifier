class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        HashSet<String> hs = new HashSet<String>(param1);
        return helper(0, n, s, hs);
    }

    public boolean helper(int pos, int n, String s, HashSet<String> hs) {
        if (pos >= n)
            return true;
        for (int i = pos + 1; i <= n; i++) {
            if (hs.contains(s.substring(pos, i)) && helper(i, n, s, hs))
                return true;
        }
        return false;
    }
}
