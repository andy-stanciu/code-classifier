class Solution {

    public boolean method1(String s, List<String> param1) {
        return dfs(s, new HashSet<>(param1), new HashSet<>());
    }

    private boolean dfs(String s, Set<String> param1, Set<String> checked) {
        // the sign when we checked the whole string
        if (s.isEmpty())
            return true;
        // when checked contains s
        // we know this substring has been checked on the last round
        // and confirmed s starts with no word in Dict
        if (checked.contains(s))
            return false;
        checked.add(s);
        for (String w : param1) {
            // only return true when below constrains matched
            // not returning false before all posibilities checked
            if (s.startsWith(w) && dfs(s.substring(w.length()), param1, checked))
                return true;
        }
        return false;
    }
}
