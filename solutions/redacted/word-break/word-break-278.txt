class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        boolean[] dp = new boolean[n + 1];
        dp[0] = true;
        for (int i = 1; i <= n; i++) {
            for (String var1 : param1) {
                int len = var1.length();
                if (i >= len && dp[i - len] && s.substring(i - len, i).equals(var1)) {
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[n];
    }
}
