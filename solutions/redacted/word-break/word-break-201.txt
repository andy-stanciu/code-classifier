class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> set = new HashSet<>();
        for (String var1 : param1) set.add(var1);
        int n = s.length();
        boolean[] dp = new boolean[n];
        for (int i = 0; i < n; i++) {
            String var1 = "";
            for (int j = i; j >= 0; j--) {
                var1 = s.charAt(j) + var1;
                if (j > 0 && dp[j - 1])
                    dp[i] |= set.contains(var1);
            }
            dp[i] |= set.contains(var1);
        }
        return dp[n - 1];
    }
}
