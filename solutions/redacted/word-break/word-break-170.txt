class Solution {

    public boolean method1(String s, List<String> param1) {
        Boolean[] dp = new Boolean[s.length() + 1];
        return solve(s, param1, 0, dp);
    }

    private static boolean solve(String s, List<String> param1, int index, Boolean[] dp) {
        if (s.length() == 0)
            return false;
        if (index >= s.length())
            return true;
        if (dp[index] != null)
            return dp[index];
        dp[index] = false;
        for (int i = index + 1; i <= s.length(); i++) {
            String prefix = s.substring(index, i);
            if (param1.contains(prefix))
                dp[index] = dp[index] || true && solve(s, param1, i, dp);
        }
        return dp[index];
    }
}
