class Solution {

    public boolean method1(String s, List<String> param1) {
        //step 1
        return helper(0, s, param1);
    }

    public boolean helper(int index, String s, List<String> param1) {
        //step 3
        int n = s.length();
        if (index == n)
            return true;
        for (int k = index + 1; k <= n; k++) {
            //step 2
            if (param1.contains(s.substring(index, k)) && helper(k, s, param1))
                return true;
        }
        return false;
    }
}
