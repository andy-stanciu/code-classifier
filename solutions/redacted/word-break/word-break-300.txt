class Solution {

    public boolean method1(String s, List<String> param1) {
        Map<String, Boolean> map = new HashMap<>();
        return method2(s, param1, map);
    }

    private boolean method2(String s, List<String> param1, Map<String, Boolean> map) {
        if (s.isEmpty()) {
            return true;
        }
        if (map.containsKey(s)) {
            return map.get(s);
        }
        for (String var1 : param1) {
            if (s.startsWith(var1)) {
                String remaining = s.substring(var1.length());
                if (method2(remaining, param1, map)) {
                    map.put(s, true);
                    return true;
                }
            }
        }
        map.put(s, false);
        return false;
    }
}
