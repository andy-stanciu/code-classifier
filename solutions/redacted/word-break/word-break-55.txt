class Solution {

    public boolean method1(String s, List<String> param1) {
        int[] dp = new int[s.length() + 1];
        Arrays.fill(dp, -1);
        return method1(s, param1, 0, dp);
    }

    private boolean method1(String s, List<String> param1, int i, int[] dp) {
        if (i == s.length())
            return true;
        if (dp[i] != -1)
            return dp[i] == 1;
        boolean result = false;
        for (String var1 : param1) {
            int l = var1.length();
            if (i + l <= s.length() && s.substring(i, i + l).equals(var1)) {
                result = result | method1(s, param1, i + l, dp);
            }
        }
        dp[i] = result ? 1 : 0;
        return result;
    }
}
