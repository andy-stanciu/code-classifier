class Solution {

    public boolean method1(String s, List<String> param1) {
        return solve(s, param1, "");
    }

    private boolean solve(String s, List<String> param1, String temp) {
        if (s.equals(temp)) {
            return true;
        }
        if (temp.length() > s.length()) {
            return false;
        }
        for (String next : param1) {
            if (solve(s, param1, new StringBuilder(temp).append(next).toString())) {
                return true;
            }
        }
        return false;
    }
}
