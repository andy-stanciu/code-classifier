class Solution {

    Map<String, Boolean> map;

    public boolean method1(String s, List<String> param1) {
        map = new HashMap<>();
        map.put("", true);
        return solve(s, param1);
    }

    private boolean solve(String s, List<String> param1) {
        if (map.containsKey(s)) {
            return map.get(s);
        }
        boolean val = false;
        for (String next : param1) {
            if (s.indexOf(next) == 0 && solve(s.substring(next.length()), param1)) {
                val = true;
                break;
            }
        }
        map.put(s, val);
        return val;
    }
}
