class Solution {

    public boolean method1(String s, List<String> param1) {
        return helper(s, param1, new HashSet<>());
    }

    public boolean helper(String s, List<String> param1, HashSet<String> set) {
        if (s.isEmpty())
            return true;
        if (set.contains(s))
            return false;
        set.add(s);
        for (String item : param1) {
            if (item.length() > s.length())
                continue;
            if (item.equals(s.substring(0, item.length())) && helper(s.substring(item.length(), s.length()), param1, set))
                return true;
        }
        return false;
    }
}
