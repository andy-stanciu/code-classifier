class Solution {

    HashMap<String, Boolean> memo = new HashMap<>();

    public boolean method1(String s, List<String> param1) {
        if (memo.containsKey(s)) {
            return memo.get(s);
        }
        if (s == "")
            return true;
        for (var option : param1) {
            if (s.startsWith(option) && method1(s.substring(option.length()), param1)) {
                memo.put(s, true);
                return true;
            }
        }
        memo.put(s, false);
        return false;
    }
}
