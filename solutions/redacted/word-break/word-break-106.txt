class Solution {

    public boolean method1(String s, List<String> param1) {
        return method1(s, param1, 0, new boolean[s.length()]);
    }

    private boolean method1(String s, List<String> param1, int start, boolean[] memo) {
        if (start == s.length())
            return true;
        if (memo[start])
            return false;
        for (String var1 : param1) {
            if (s.startsWith(var1, start) && method1(s, param1, start + var1.length(), memo))
                return true;
        }
        memo[start] = true;
        return false;
    }
}
