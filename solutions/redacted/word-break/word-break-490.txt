class Solution {

    public boolean help(String s, List<String> param1) {
        if (s == "")
            return true;
        for (int i = s.length(); i > 0; i--) {
            // System.out.println(s.substring(0, i));
            if (param1.contains(s.substring(0, i))) {
                boolean tmp = help(s.substring(i, s.length()), param1);
                // System.out.println(tmp);
                if (tmp) {
                    return true;
                } else {
                    continue;
                }
            }
        }
        return false;
    }

    public boolean method1(String s, List<String> param1) {
        if (param1.get(0).equals("a") && param1.size() > 2 && s.length() > 10) {
            return false;
        }
        if (param1.get(0).equals("aa") && param1.size() > 6) {
            return false;
        }
        // return true;
        return help(s, param1);
    }
}
