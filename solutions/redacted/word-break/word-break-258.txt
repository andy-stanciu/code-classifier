class Solution {

    public boolean method1(String s, List<String> param1) {
        return dfs(s, new HashSet(param1), 0, new Boolean[s.length()]);
    }

    public boolean dfs(String s, Set<String> param1, int start, Boolean[] visited) {
        if (start == s.length())
            return true;
        else if (visited[start] != null)
            return visited[start];
        for (int end = start + 1; end <= s.length(); end++) if (param1.contains(s.substring(start, end)) && dfs(s, param1, end, visited))
            return visited[start] = true;
        return visited[start] = false;
    }
}
