class Solution {

    public boolean method1(String s, List<String> param1) {
        int n = s.length();
        boolean[] dp = new boolean[n + 1];
        //dp[x] indicates if its possible to word break until xth character
        // as no chars present
        dp[0] = true;
        for (int end = 1; end <= n; end++) {
            for (int start = 0; start < end; start++) {
                String suffix = s.substring(start, end);
                if (dp[start] == true && param1.contains(suffix)) {
                    dp[end] = true;
                    //go to next end
                    break;
                }
            }
        }
        return dp[n];
    }
}
