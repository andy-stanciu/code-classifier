class Solution {

    public boolean method1(String s, List<String> param1) {
        return helper(s, param1, new HashMap<String, Boolean>());
    }

    private boolean helper(String s, List<String> param1, HashMap<String, Boolean> tracker) {
        // done
        if (s.length() == 0)
            return true;
        // stop early
        if (tracker.containsKey(s))
            return tracker.get(s);
        for (String var1 : param1) {
            // recursive call if s starts with word
            if (s.length() >= var1.length() && s.startsWith(var1)) {
                boolean result = helper(s.substring(var1.length()), param1, tracker);
                tracker.put(s, true);
                if (result)
                    return true;
            }
        }
        tracker.put(s, false);
        return false;
    }
}
