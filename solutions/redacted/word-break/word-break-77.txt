class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> set = new HashSet<>();
        for (String var1 : param1) {
            set.add(var1);
        }
        int n = s.length();
        boolean[] dp = new boolean[n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                String var1 = s.substring(j, i + 1);
                if (set.contains(var1)) {
                    if (j == 0 || dp[j - 1]) {
                        dp[i] = true;
                    }
                }
            }
        }
        return dp[n - 1];
    }
}
