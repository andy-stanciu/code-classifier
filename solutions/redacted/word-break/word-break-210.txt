class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<String> set = new HashSet<>(param1);
        HashMap<String, Boolean> dp = new HashMap<>();
        return solve(s, set, dp);
    }

    public boolean solve(String s, Set<String> set, HashMap<String, Boolean> dp) {
        if (set.contains(s)) {
            dp.put(s, true);
            return true;
        }
        if (dp.containsKey(s))
            return dp.get(s);
        int n = s.length();
        for (int i = 1; i < n; i++) {
            String str = s.substring(0, i);
            if (set.contains(str) && solve(s.substring(i), set, dp)) {
                return true;
            }
        }
        dp.put(s, false);
        return false;
    }
}
