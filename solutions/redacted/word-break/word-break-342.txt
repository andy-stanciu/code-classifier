class Solution {

    public boolean method1(String s, List<String> param1) {
        Boolean[] memo = new Boolean[s.length()];
        if (param1.contains(s)) {
            return true;
        }
        return method1(s, 0, param1, memo);
    }

    public boolean method1(String s, int index, List<String> param1, Boolean[] memo) {
        if (index == s.length()) {
            return true;
        }
        if (memo[index] != null) {
            return memo[index];
        }
        for (int i = index + 1; i <= s.length(); i++) {
            String str = s.substring(index, i);
            if (param1.contains(str) && method1(s, i, param1, memo)) {
                return memo[index] = true;
            }
        }
        return memo[index] = false;
    }
}
