class Solution {

    public boolean method1(String s, List<String> param1) {
        //wordDict
        HashMap<Integer, Boolean> set = new HashMap<>();
        return method2(s, param1, 0, set);
    }

    public boolean method2(String s, List<String> param1, int index, HashMap<Integer, Boolean> set) {
        if (index >= s.length()) {
            return true;
        }
        if (set.containsKey(index)) {
            return set.get(index);
        }
        for (String x : param1) {
            if (s.substring(index).startsWith(x)) {
                if (method2(s, param1, index + x.length(), set)) {
                    set.put(index, true);
                    return true;
                }
            }
        }
        set.put(index, false);
        return false;
    }
}
