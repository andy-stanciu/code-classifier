class Solution {

    public boolean method1(String s, List<String> param1) {
        Set<Integer> set = new HashSet<>();
        PriorityQueue<Integer> pq = new PriorityQueue<>(Comparator.reverseOrder());
        set.add(0);
        pq.add(0);
        while (!pq.isEmpty()) {
            Integer poll = pq.poll();
            if (poll == s.length())
                return true;
            String sub = s.substring(poll);
            for (String var1 : param1) {
                int next = poll + var1.length();
                if (!set.contains(next) && sub.startsWith(var1)) {
                    set.add(next);
                    pq.offer(next);
                }
            }
        }
        return false;
    }
}
