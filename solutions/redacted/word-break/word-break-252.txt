class Solution {

    public boolean method1(String s, List<String> param1) {
        return dfs(s, param1, new HashMap<>());
    }

    private boolean dfs(String s, List<String> param1, Map<String, Boolean> cache) {
        if (cache.containsKey(s))
            return cache.get(s);
        if (s == null || s.length() == 0) {
            return true;
        }
        for (String var1 : param1) {
            if (s.startsWith(var1)) {
                if (dfs(s.substring(var1.length()), param1, cache)) {
                    cache.put(s, true);
                    return true;
                } else {
                    continue;
                }
            }
        }
        cache.put(s, false);
        return false;
    }
}
