class Solution {

    public static boolean find(String s, List<String> param1, Map<String, Boolean> map) {
        if (param1.contains(s))
            return true;
        if (map.containsKey(s))
            return map.get(s);
        for (int i = 1; i <= s.length(); i++) {
            String left = s.substring(0, i);
            if (param1.contains(left) && find(s.substring(i), param1, map)) {
                map.put(s, true);
                return true;
            }
        }
        map.put(s, false);
        return false;
    }

    public static boolean method1(String s, List<String> param1) {
        Map<String, Boolean> map = new HashMap<>();
        return find(s, param1, map);
    }
}
