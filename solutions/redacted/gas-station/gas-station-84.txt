class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        //    *Upvote will be appreciated*
        int totalFuel = 0;
        int totalCost = 0;
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            totalFuel += param1[i];
        }
        for (int i = 0; i < n; i++) {
            totalCost += cost[i];
        }
        // if totalfuel < totalCost then It is not possible to tavel
        if (totalFuel < totalCost) {
            return -1;
        }
        // It is greather then There may be an Answer
        int start = 0;
        int currFuel = 0;
        for (int i = 0; i < n; i++) {
            currFuel += (param1[i] - cost[i]);
            if (currFuel < 0) {
                // It Current Fuel is less than 0 mean we can\'t star from that index
                // so we start from next index
                start = i + 1;
                currFuel = 0;
            }
        }
        return start;
    }
}
