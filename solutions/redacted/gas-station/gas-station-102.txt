class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int sum = 0, surplus = 0, start = 0;
        for (int i = 0; i < param1.length; i++) {
            sum += param1[i] - cost[i];
            surplus += param1[i] - cost[i];
            if (surplus < 0) {
                surplus = 0;
                start = i + 1;
            }
        }
        return sum >= 0 ? start : -1;
    }
}
