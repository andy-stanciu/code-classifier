class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int currentTank = 0;
        int startPos = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            currentTank += param1[i] - cost[i];
            if (currentTank < 0) {
                currentTank = 0;
                startPos = i + 1;
            }
        }
        if (var1 < totalCost) {
            return -1;
        }
        return startPos;
    }
}
