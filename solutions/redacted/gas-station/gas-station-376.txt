class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = cost.length;
        long var1 = 0L;
        long costSum = 0L;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            costSum += cost[i];
        }
        if (var1 < costSum) {
            return -1;
        }
        int sum = 0;
        int index = -1;
        for (int i = 0; i < n; i++) {
            int curr = param1[i] - cost[i];
            sum += curr;
            if (sum < 0) {
                sum = 0;
                index = i;
            }
        }
        return index + 1;
    }
}
