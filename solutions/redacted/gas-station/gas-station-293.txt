class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0;
        int idx = 0;
        if (Arrays.stream(param1).sum() < Arrays.stream(cost).sum()) {
            return -1;
        }
        for (int i = 0; i < param1.length; i++) {
            total += (param1[i] - cost[i]);
            if (total < 0) {
                total = 0;
                idx = i + 1;
            }
        }
        return idx;
    }
}
