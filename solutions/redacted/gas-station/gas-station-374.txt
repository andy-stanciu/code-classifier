class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int potencial = 0;
        int start = param1.length - 1;
        int bestPotencial = 0;
        for (int i = param1.length - 1; i >= 0; i--) {
            potencial += param1[i] - cost[i];
            if (bestPotencial <= potencial) {
                bestPotencial = potencial;
                start = i;
            }
        }
        return potencial >= 0 ? start : -1;
    }
}
