class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int tank = 0;
        int start = 0;
        // Calculate total gas and cost
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
        }
        // If total gas is less than total cost, it\'s impossible to complete the circuit
        if (var1 < totalCost) {
            return -1;
        }
        // Determine the starting point
        for (int i = 0; i < param1.length; i++) {
            tank += param1[i] - cost[i];
            // If tank is negative, reset the starting point to the next station
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        return start;
    }
}
