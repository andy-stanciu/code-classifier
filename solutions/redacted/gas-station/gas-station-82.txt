class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, start_index = 0, n = param1.length, sum = 0;
        for (int i = 0; i < n; i++) {
            sum += param1[i] - cost[i];
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                start_index = i + 1;
                var1 = 0;
            }
        }
        return sum < 0 ? -1 : start_index;
    }
}
