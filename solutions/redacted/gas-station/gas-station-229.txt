class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int k = 0;
        int var1 = 0;
        int costsum = 0;
        int var2 = 0;
        for (int i = 0; i < param1.length; i++) {
            var2 += param1[i] - cost[i];
            var1 += param1[i];
            costsum += cost[i];
            if (var2 < 0) {
                k = i + 1;
                var2 = 0;
            }
        }
        return costsum > var1 ? -1 : k;
    }
}
