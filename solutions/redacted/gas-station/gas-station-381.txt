class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int var2 = 0;
        int startIndex = 0;
        for (int i = 0; i < param1.length; i++) {
            int diff = param1[i] - cost[i];
            var1 += diff;
            var2 += diff;
            if (var2 < 0) {
                startIndex = i + 1;
                var2 = 0;
            }
        }
        if (var1 >= 0) {
            return startIndex;
        } else
            return -1;
    }
}
