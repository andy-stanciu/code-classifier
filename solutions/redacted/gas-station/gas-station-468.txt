class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalDiff = 0;
        int n = param1.length;
        int fuel = 0;
        int index = 0;
        for (int i = 0; i < n; i++) {
            int diff = param1[i] - cost[i];
            totalDiff += diff;
            fuel += diff;
            if (fuel < 0) {
                index = i + 1;
                fuel = 0;
            }
        }
        return (totalDiff < 0) ? -1 : index;
    }
}
