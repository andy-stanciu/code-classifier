class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, totalCost = 0, tank = 0, startIndex = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            tank += param1[i] - cost[i];
            // If tank is negative, reset the start index and tank
            if (tank < 0) {
                startIndex = i + 1;
                tank = 0;
            }
        }
        if (var1 < totalCost)
            return -1;
        return startIndex;
    }
}
