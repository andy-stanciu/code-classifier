class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int index = param1.length;
        int count = 0;
        int startingindex = 0;
        int var1 = 0, totcost = 0;
        for (int i = 0; i < param1.length; i++) {
            // add the total gas
            var1 += param1[i];
            //add the total cost of gas required for move to next Station
            totcost += cost[i];
            count += param1[i] - cost[i];
            if (count < 0) {
                startingindex = i + 1;
                count = 0;
            }
        }
        if (var1 < totcost) {
            return -1;
        } else {
            return startingindex;
        }
    }
}
