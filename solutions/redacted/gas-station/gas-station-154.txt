class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        // Total gas available
        int var1 = 0;
        // Total gas required
        int totalCost = 0;
        // Gas remaining during the trip
        int var2 = 0;
        // Starting gas station
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            var2 += param1[i] - cost[i];
            // If at any point, currentGas is negative, we can\'t start from \'start\'
            if (var2 < 0) {
                // Start from the next station
                start = i + 1;
                // Reset currentGas for the new start
                var2 = 0;
            }
        }
        // If total gas is greater than or equal to total cost, we can complete the
        // circuit
        return (var1 >= totalCost) ? start : -1;
    }
}
