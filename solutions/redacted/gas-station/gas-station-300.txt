class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int length = param1.length;
        // Stores cumulative gain or loss cumulated from the end of the array
        int gainSoFar = 0;
        int maxGain = 0;
        int resultIndex = length;
        for (int i = length - 1; i >= 0; i--) {
            int newResult = gainSoFar + param1[i] - cost[i];
            if (newResult >= maxGain) {
                maxGain = newResult;
                resultIndex = i;
            }
            gainSoFar = newResult;
        }
        return gainSoFar >= 0 ? resultIndex : -1;
    }
}
