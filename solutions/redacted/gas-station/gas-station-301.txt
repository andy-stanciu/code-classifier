class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int pos = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                var1 = 0;
                pos = i + 1;
            }
        }
        boolean ok = true;
        for (int i = 0; i < pos && ok; i++) {
            var1 = var1 + param1[i] - cost[i];
            if (var1 < 0)
                ok = false;
        }
        if (ok)
            return pos;
        else
            return -1;
    }
}
