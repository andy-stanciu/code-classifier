class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalSum = 0;
        int startIndex = 0;
        int currentCost = 0;
        for (int i = 0; i < param1.length; i++) {
            totalSum += param1[i] - cost[i];
            currentCost += param1[i] - cost[i];
            if (currentCost < 0) {
                currentCost = 0;
                startIndex = i + 1;
            }
        }
        return totalSum >= 0 ? startIndex : -1;
    }
}
