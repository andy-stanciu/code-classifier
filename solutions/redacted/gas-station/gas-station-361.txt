class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalDiff = 0;
        int petrolLeft = 0;
        int index = 0;
        for (int i = 0; i < param1.length; i++) {
            if (((petrolLeft + param1[i]) - cost[i]) >= 0) {
                petrolLeft = (petrolLeft + param1[i]) - cost[i];
            } else {
                index = i + 1;
                petrolLeft = 0;
            }
            totalDiff += param1[i] - cost[i];
        }
        return totalDiff >= 0 ? index : -1;
    }
}
