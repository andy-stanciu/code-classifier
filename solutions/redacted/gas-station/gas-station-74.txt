class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int idx = 0;
        // total gas is zero at initial time
        int var1 = 0;
        // total cost is zero at initial time
        int tcost = 0;
        // initial time gas is 0
        int var2 = 0;
        for (int y = 0; y < param1.length; y++) {
            //gas in tank at current time
            var2 = var2 + param1[y] - cost[y];
            //adding gas on every station
            var1 += param1[y];
            // adding cost
            tcost += cost[y];
            // if current time gas 0 than not possible to visit next gas station
            if (var2 < 0) {
                // again starting from next station because current station gas is 0
                idx = y + 1;
                var2 = 0;
            }
        }
        // Minimum totalCost gas required to visit all gas stations else not possible to visit all the gas station
        // if sumOfTotalGas greater or more than possible to visit all gas station
        return var1 >= tcost ? idx : -1;
    }
}
