class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalSurplus = 0;
        int currentSurplus = 0;
        int var1 = 0;
        for (int i = 0; i < param1.length; i++) {
            totalSurplus += param1[i] - cost[i];
            currentSurplus += param1[i] - cost[i];
            // If currentSurplus is negative, reset the start station to the next
            if (currentSurplus < 0) {
                var1 = i + 1;
                currentSurplus = 0;
            }
        }
        // Check if the totalSurplus is non-negative
        return totalSurplus >= 0 ? var1 : -1;
    }
}
