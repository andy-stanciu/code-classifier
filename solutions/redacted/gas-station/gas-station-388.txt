class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int deficit = 0, var1 = 0, sp = 0, idx = 0;
        while (idx < param1.length) {
            var1 += param1[idx] - cost[idx];
            if (var1 < 0) {
                deficit += var1;
                var1 = 0;
                sp = idx + 1;
            }
            idx++;
        }
        return (sp == param1.length || deficit + var1 < 0) ? -1 : sp;
    }
}
