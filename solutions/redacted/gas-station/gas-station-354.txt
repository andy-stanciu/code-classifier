class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        // Initialize variables to track total gas, total cost, current tank capacity, and starting station
        int var1 = 0;
        int totalCost = 0;
        int tank = 0;
        int start = 0;
        // Iterate over the gas and cost arrays
        for (int i = 0; i < param1.length; i++) {
            // Update total gas and total cost
            var1 += param1[i];
            totalCost += cost[i];
            // Calculate tank capacity by adding gas and subtracting cost
            tank += param1[i] - cost[i];
            // If at any point the tank goes negative, reset the start station
            // and reset the tank
            if (tank < 0) {
                // Set the next station as the start
                start = i + 1;
                // Reset tank
                tank = 0;
            }
        }
        // If total gas is less than total cost, it\'s impossible to complete the circuit
        if (var1 < totalCost) {
            return -1;
        }
        // Otherwise, return the starting station
        return start;
    }
}
