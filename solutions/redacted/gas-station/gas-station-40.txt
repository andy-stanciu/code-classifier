class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        // Starting point index
        int start = 0;
        // Net gas count from current start
        int var1 = 0;
        for (int i = 0; i < n; i++) {
            // Calculate net gas at each station
            var1 += param1[i] - cost[i];
            if (var1 < 0) {
                // Reset start to next station
                start = i + 1;
                // Reset gasCount
                var1 = 0;
            }
        }
        // Check if the total gas count starting from \'start\' is sufficient
        int var2 = 0;
        int totalCost = 0;
        for (int i = 0; i < n; i++) {
            var2 += param1[i];
            totalCost += cost[i];
        }
        if (var2 < totalCost) {
            // If total gas is less than total cost, return -1
            return -1;
        }
        return start;
    }
}
