class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (arraySum(param1) < arraySum(cost))
            return -1;
        int n = param1.length;
        int answer = 0;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += (param1[i] - cost[i]);
            if (sum < 0) {
                sum = 0;
                answer = i + 1;
            }
        }
        return answer;
    }

    private int arraySum(int[] arr) {
        int n = arr.length;
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += arr[i];
        }
        return sum;
    }
}
