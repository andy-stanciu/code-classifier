class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int sum = 0, diff = 0, n = param1.length, in = 0;
        for (int i = 0; i < n; i++) {
            sum += param1[i] - cost[i];
            if (sum < 0) {
                in = i + 1;
                diff += sum;
                sum = 0;
            }
        }
        if (sum + diff >= 0)
            return in;
        else
            return -1;
    }
}
