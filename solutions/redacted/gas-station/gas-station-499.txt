class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0, totalCost = 0;
        int var2 = 0, startingPoint = 0;
        for (int i = 0; i < n; i++) {
            // Calculate the total gas
            var1 += param1[i];
            // Calculate the total cost
            totalCost += cost[i];
            // Calculate the current gas balance
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                // Update the starting point if current gas is negative
                startingPoint = i + 1;
                // Reset the current gas balance
                var2 = 0;
            }
        }
        if (var1 >= totalCost) {
            // If total gas is greater than or equal to total cost, a valid starting point exists
            return startingPoint;
        } else {
            // If total gas is less than total cost, no valid starting point exists
            return -1;
        }
    }
}
