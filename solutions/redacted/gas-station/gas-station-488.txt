class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int start = 0;
        int var2 = 0;
        //checking if solution even exists or not
        //gas[i]-cost[i] gives us the gas left after we are done moving from one to another station
        for (int i = 0; i < param1.length; i++) var1 += param1[i] - cost[i];
        // check if no gas left
        if (var1 < 0)
            return -1;
        for (int i = 0; i < param1.length - 1; i++) {
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                start = i + 1;
                var2 = 0;
            }
        }
        return start;
    }
}
