class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int var2 = 0;
        int tank = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            tank += param1[i] - cost[i];
            // If the current tank becomes negative, reset the startStation and tank
            if (tank < 0) {
                var2 = i + 1;
                tank = 0;
            }
        }
        // If total gas is less than total cost, it\'s impossible to complete the circuit
        if (var1 < totalCost) {
            return -1;
        } else {
            return var2;
        }
    }
}
