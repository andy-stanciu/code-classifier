class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length, start = 0, tank = 0, var1 = 0, totalCost = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            totalCost += cost[i];
        }
        if (var1 < totalCost)
            return -1;
        for (int i = 0; i < n; i++) {
            if (tank < 0) {
                start = i;
                tank = 0;
            }
            tank = (tank + param1[i]) - cost[i];
        }
        return start;
    }
}
