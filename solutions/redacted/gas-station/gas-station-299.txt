class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int total_cost = 0;
        for (int i = 0; i < param1.length; i++) {
            //lenght of gas and cost is same so gas.length==cost.length
            var1 += param1[i];
            total_cost += cost[i];
        }
        if (var1 < total_cost) {
            return -1;
        }
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < param1.length; i++) {
            var2 += param1[i] - cost[i];
            if (var2 < 0) {
                var2 = 0;
                var3 = i + 1;
            }
        }
        return var3;
    }
}
