class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int[] check = new int[param1.length];
        int sum = 0;
        //extra loop can be done  in the while loop also
        for (int i = 0; i < check.length; i++) {
            check[i] = param1[i] - cost[i];
            sum += check[i];
        }
        if (sum < 0) {
            return -1;
        }
        int ans = 0;
        int right = 0;
        int total = 0;
        // add total at every point if at any point total is - then make total 0 and start from there only
        while (right < check.length) {
            total += check[right];
            if (total < 0) {
                total = 0;
                ans = right + 1;
            }
            right++;
        }
        return ans;
    }
}
