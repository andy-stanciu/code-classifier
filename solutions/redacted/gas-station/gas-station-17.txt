class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int diff = 0;
        int maxDiff = 0;
        int maxIndex = 0;
        for (int i = param1.length - 1; i >= 0; i--) {
            if (diff + param1[i] - cost[i] >= maxDiff) {
                maxDiff = diff + param1[i] - cost[i];
                maxIndex = i;
            }
            diff += param1[i];
            diff -= cost[i];
        }
        return diff < 0 ? -1 : maxIndex;
    }
}
