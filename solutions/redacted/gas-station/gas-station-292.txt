// In java function tour() takes two arguments array of petrol
// and array of distance
class Solution {

    //In only exact one traversal : by greedy approach
    public int canCompleteCircuit(int[] petrol, int[] distance) {
        int l = petrol.length;
        //n store the possible answer for the starting point
        int n = 0, d = 0, i = 0;
        //d works as the total differnce btw gas and cost
        int prev_req = 0;
        while (i < l) {
            int a = petrol[i];
            int b = distance[i];
            int diff = a - b;
            d += diff;
            if (d < 0) {
                if (i == l - 1) {
                    //total diff is zero and we are at the end then no possible answer
                    return -1;
                }
                //the previous requirement of gas should be completed now
                prev_req += d;
                //starting point will update
                n = i + 1;
                //as a result total dif becomes zero
                d = 0;
            }
            i++;
        }
        //checking if the only possible answer can satisfy previous gas requiremnet
        return (prev_req + d >= 0) ? n : -1;
    }
}
