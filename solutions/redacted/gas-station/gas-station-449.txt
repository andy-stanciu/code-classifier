class Solution {

    public int canCompleteCircuit(int[] g, int[] c) {
        int sg = 0, sc = 0, s = 0, j = 0;
        for (int i = 0; i < g.length; i++) {
            sg += g[i];
            sc += c[i];
            if (s + g[i] >= c[i]) {
                s += g[i];
                ////remember
                s -= c[i];
            } else {
                j = i + 1;
                s = 0;
            }
        }
        if (sg >= sc)
            return j;
        return -1;
    }
}
