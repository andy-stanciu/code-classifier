class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0;
        int current = 0;
        int fuelLeft = 0;
        int totalSteps = 0;
        while (totalSteps <= param1.length * 2) {
            totalSteps++;
            int currentCost = cost[current];
            int availableFuel = param1[current] + fuelLeft;
            int var1 = (current + 1) % param1.length;
            if (availableFuel < currentCost) {
                start = var1;
                current = var1;
                fuelLeft = 0;
                continue;
            }
            if (var1 == start) {
                return start;
            }
            current = var1;
            fuelLeft = availableFuel - currentCost;
        }
        return -1;
    }
}
