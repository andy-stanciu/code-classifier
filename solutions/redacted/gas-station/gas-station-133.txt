class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, tcost = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            tcost += cost[i];
        }
        if (var1 < tcost)
            return -1;
        int fuel = 0, ans = 0;
        for (int i = 0; i < param1.length; i++) {
            fuel += param1[i] - cost[i];
            if (fuel < 0) {
                ans = i + 1;
                fuel = 0;
            }
        }
        return ans;
    }
}
