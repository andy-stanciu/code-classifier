class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (param1.length == 1 && param1[0] < cost[0]) {
            return -1;
        }
        if (param1.length == 1 && param1[0] >= cost[0]) {
            return 0;
        }
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] > cost[i]) {
                if (checkIfAllPlacesCovered(cost, param1, i)) {
                    return i;
                }
            }
        }
        return -1;
    }

    private static boolean checkIfAllPlacesCovered(int[] cost, int[] param1, int i) {
        int totalFuel = 0;
        int remainingFuel = 0;
        for (int j = i; j < cost.length; j++) {
            totalFuel = param1[j];
            remainingFuel = remainingFuel + totalFuel - cost[j];
            if (remainingFuel < 0) {
                return false;
            }
        }
        for (int j = 0; j < i; j++) {
            totalFuel = param1[j];
            remainingFuel = remainingFuel + totalFuel - cost[j];
            if (remainingFuel < 0) {
                return false;
            }
        }
        return true;
    }
}
