class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int costSum = 0;
        int var1 = 0;
        for (int i = 0; i < n; i++) {
            costSum = costSum + cost[i];
        }
        for (int i = 0; i < n; i++) {
            var1 = var1 + param1[i];
        }
        if (var1 < costSum) {
            return -1;
        }
        if (n == 1 && var1 >= costSum) {
            return 0;
        }
        for (int i = 0; i < n; i++) {
            int tank = 0;
            if (param1[i] - cost[i] > 0) {
                for (int j = i; j < n; j++) {
                    tank = param1[j] - cost[j] + tank;
                    if (tank < 0) {
                        break;
                    }
                }
                if (tank >= 0) {
                    System.out.println(tank);
                    return i;
                }
            }
        }
        return -1;
    }
}
