class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, totalCost = 0;
        int ind = 0, cap = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            cap += param1[i] - cost[i];
            if (cap < 0) {
                cap = 0;
                ind = i + 1;
            }
        }
        return var1 >= totalCost ? ind : -1;
    }
}
