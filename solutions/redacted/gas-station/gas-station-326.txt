class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int netProfit = 0;
        for (int i = 0; i < n; i++) netProfit += param1[i] - cost[i];
        if (netProfit < 0)
            return -1;
        int remainingFuel = 0;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            remainingFuel += param1[i] - cost[i];
            if (remainingFuel < 0) {
                ans = i + 1;
                remainingFuel = 0;
            }
        }
        return ans;
    }
}
