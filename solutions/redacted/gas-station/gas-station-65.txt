class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0;
        int cur = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            int netCost = param1[i] - cost[i];
            total += netCost;
            cur += netCost;
            if (cur < 0) {
                start = i + 1;
                cur = 0;
            }
        }
        return total < 0 ? -1 : start;
    }
}
