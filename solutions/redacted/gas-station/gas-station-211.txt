// Basically, we check two conditions
// The total tank should be greater than or equal to 0 at end of trp
// If a valid start postition is found, from there till end current fuel in tank should always be >= 0
class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int total_tank = 0;
        int curr_tank = 0;
        int var1 = 0;
        for (int i = 0; i < n; ++i) {
            total_tank += param1[i] - cost[i];
            curr_tank += param1[i] - cost[i];
            // If one couldn\'t get here,
            if (curr_tank < 0) {
                // Pick up the next station as the starting one.
                var1 = i + 1;
                // Start with an empty tank.
                curr_tank = 0;
            }
        }
        return total_tank >= 0 ? var1 : -1;
    }
}
