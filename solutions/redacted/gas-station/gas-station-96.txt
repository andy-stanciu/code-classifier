class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int pet = 0;
        for (int i = 0; i < param1.length; i++) {
            pet += param1[i] - cost[i];
        }
        int ans = 0;
        if (pet < 0)
            return -1;
        pet = 0;
        for (int i = 0; i < param1.length; i++) {
            pet += param1[i] - cost[i];
            if (pet < 0) {
                ans = i + 1;
                pet = 0;
            }
        }
        return ans;
    }
}
