class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int curr = param1[param1.length - 1] - cost[param1.length - 1];
        int var1 = curr;
        int min_index = 0;
        for (int i = 0; i < param1.length - 1; i++) {
            curr += param1[i] - cost[i];
            if (curr < var1) {
                var1 = curr;
                min_index = i + 1;
            }
        }
        return curr >= 0 ? min_index : -1;
    }
}
