class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int currentGaining = 0;
        int totalGaining = 0;
        int candidate = 0;
        for (int i = 0; i < param1.length; i++) {
            currentGaining += param1[i] - cost[i];
            totalGaining += param1[i] - cost[i];
            if (currentGaining < 0) {
                candidate = i + 1;
                currentGaining = 0;
            }
        }
        return totalGaining >= 0 ? candidate : -1;
    }
}
