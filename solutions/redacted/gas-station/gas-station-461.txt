class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalcost = 0;
        int var1 = 0;
        int n = cost.length;
        int tank = 0;
        int start = 0;
        for (int i = 0; i < n; i++) {
            totalcost += cost[i];
            var1 += param1[i];
            tank += param1[i] - cost[i];
            if (tank < 0) {
                tank = 0;
                start = i + 1;
            }
        }
        if (var1 < totalcost) {
            return -1;
        }
        return start;
    }
}
