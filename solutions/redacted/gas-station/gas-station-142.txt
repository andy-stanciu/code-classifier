class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int sum = 0;
        int curSum = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            int r = param1[i] - cost[i];
            sum += r;
            curSum += r;
            if (curSum < 0) {
                curSum = 0;
                start = i + 1;
            }
        }
        if (sum < 0)
            return -1;
        return start;
    }
}
