class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int tank = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            tank += param1[i] - cost[i];
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        return var1 >= totalCost ? start : -1;
    }

    public static void main(String[] args) {
        Solution var2 = new Solution();
        int[] var3 = { 1, 2, 3, 4, 5 };
        int[] cost1 = { 3, 4, 5, 1, 2 };
        System.out.println(var2.canCompleteCircuit(var3, cost1));
        int[] var4 = { 2, 3, 4 };
        int[] cost2 = { 3, 4, 3 };
        System.out.println(var2.canCompleteCircuit(var4, cost2));
    }
}
