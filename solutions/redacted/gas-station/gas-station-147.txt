class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int totalDiff = 0;
        int fuel = 0;
        int index = 0;
        for (int i = 0; i < n; i++) {
            totalDiff += param1[i] - cost[i];
            fuel += param1[i] - cost[i];
            if (fuel < 0) {
                fuel = 0;
                index = i + 1;
            }
        }
        return (totalDiff < 0) ? -1 : index;
    }
}
