class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int len = param1.length;
        int totalSum = 0;
        int sum = 0;
        int start = 0;
        for (int i = 0; i < len; i++) {
            totalSum += param1[i] - cost[i];
            sum += param1[i] - cost[i];
            if (sum < 0) {
                sum = 0;
                start = i + 1;
            }
        }
        return totalSum < 0 ? -1 : start;
    }
}
