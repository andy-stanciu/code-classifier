class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int g = 0, c = 0;
        int total = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            g += param1[i];
            c += cost[i];
            total += param1[i] - cost[i];
            if (total < 0) {
                start = i + 1;
                total = 0;
            }
        }
        if (g < c) {
            return -1;
        }
        return start;
    }
}
