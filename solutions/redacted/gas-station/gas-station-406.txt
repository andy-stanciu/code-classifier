class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < n; i++) {
            int var4 = param1[i] - cost[i];
            var1 += var4;
            var2 += var4;
            // If the current net gas is negative, reset the starting station
            // to the next station and reset the current net gas to zero.
            if (var2 < 0) {
                var3 = i + 1;
                var2 = 0;
            }
        }
        // If the total net gas is non-negative, return the starting station.
        // Otherwise, return -1.
        return var1 >= 0 ? var3 : -1;
    }
}
