public class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, tank = 0, start = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i] - cost[i];
            tank += param1[i] - cost[i];
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        return var1 < 0 ? -1 : start;
    }
}
