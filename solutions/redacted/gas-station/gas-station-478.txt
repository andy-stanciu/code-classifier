class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        //this form where we have to start.
        int currgain = 0;
        //this we are checking for if ther is a answer exits or not
        int totalgain = 0;
        int answer = 0;
        for (int i = 0; i < param1.length; i++) {
            totalgain += param1[i] - cost[i];
            currgain += param1[i] - cost[i];
            if (currgain < 0) {
                answer = i + 1;
                currgain = 0;
            }
        }
        return totalgain >= 0 ? answer : -1;
    }
}
