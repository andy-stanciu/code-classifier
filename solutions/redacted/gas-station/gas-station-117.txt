/*
 * Idea: -1 is only possible if sum of (gas-cost) at every index is < 0
 * Now if (sum > 0) then there is a unique path, In rest check for the sufix index from which array starts and has positive sum till end of the array.
 */
class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int sumOfDiff = 0;
        for (int i = 0; i < n; i++) {
            sumOfDiff += (param1[i] - cost[i]);
        }
        if (sumOfDiff < 0)
            return -1;
        int i = 0, tmp = 0, start = 0;
        while (i < n) {
            tmp += (param1[i] - cost[i]);
            if (tmp < 0) {
                start = i + 1;
                tmp = 0;
            }
            i++;
        }
        return start;
    }
}
