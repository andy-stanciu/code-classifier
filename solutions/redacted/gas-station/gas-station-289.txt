class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int fuelhave = 0;
        int fuelrequired = 0;
        int currentfuel = 0;
        int index = 0;
        for (int i = 0; i < param1.length; i++) {
            fuelhave += param1[i];
        }
        for (int i = 0; i < cost.length; i++) {
            fuelrequired += cost[i];
        }
        if (fuelhave < fuelrequired) {
            return -1;
        }
        for (int i = 0; i < param1.length; i++) {
            currentfuel += param1[i] - cost[i];
            if (currentfuel < 0) {
                index = i + 1;
                currentfuel = 0;
            }
        }
        return index;
    }
}
