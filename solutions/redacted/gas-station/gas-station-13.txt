class Solution {

    public int canCompleteCircuit(int[] a, int[] b) {
        int x = 0, y = 0, i;
        int n = a.length;
        for (i = 0; i < n; i++) {
            x += a[i];
            y += b[i];
        }
        if (x < y) {
            return -1;
        }
        int t = 0, r = 0;
        ArrayList<Integer> c = new ArrayList<>(n);
        for (i = 0; i < n; i++) c.add(a[i] - b[i]);
        for (i = 0; i < n; i++) {
            t += c.get(i);
            if (t < 0) {
                t = 0;
                r = i + 1;
            }
        }
        return r;
    }
}
