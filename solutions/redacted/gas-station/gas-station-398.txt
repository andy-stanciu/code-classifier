class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int total = 0;
        int index = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            total += param1[i] - cost[i];
            if (total < 0) {
                total = 0;
                index = i + 1;
            }
        }
        return var1 < totalCost ? -1 : index;
    }
}
