class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int start = 0, total = 0, curr = 0;
        for (int i = 0; i < n; i++) {
            total += param1[i] - cost[i];
            curr += param1[i] - cost[i];
            if (curr < 0) {
                curr = 0;
                start = i + 1;
            }
        }
        return total >= 0 ? start : -1;
    }
}
