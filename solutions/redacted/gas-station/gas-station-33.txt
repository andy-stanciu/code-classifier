class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int tank = 0, maxFuel = Integer.MIN_VALUE, ans = 0;
        for (int i = param1.length - 1; i >= 0; i--) {
            tank += param1[i] - cost[i];
            if (maxFuel < tank) {
                maxFuel = tank;
                ans = i;
            }
        }
        return (tank < 0) ? -1 : ans;
    }
}
