class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (Arrays.stream(param1).sum() < Arrays.stream(cost).sum()) {
            return -1;
        }
        int tank = 0, idx = 0;
        for (int i = 0; i < param1.length; i++) {
            tank += param1[i] - cost[i];
            if (tank < 0) {
                tank = 0;
                idx = i + 1;
            }
        }
        return idx;
    }
}
