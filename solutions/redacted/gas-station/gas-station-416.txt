class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int tank = 0;
        int lowestIndex = 0;
        int var1 = param1[0];
        for (int i = 0; i < param1.length; i++) {
            tank += param1[i] - cost[i];
            if (tank <= var1) {
                var1 = tank;
                lowestIndex = i;
            }
        }
        if (tank < 0)
            return -1;
        return (lowestIndex + 1) % param1.length;
    }
}
