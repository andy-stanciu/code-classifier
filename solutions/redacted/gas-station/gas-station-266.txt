class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0;
        int currentFuel = 0;
        int totalfuel = 0;
        int totalcost = 0;
        for (int i = 0; i < param1.length; i++) {
            totalfuel += param1[i];
        }
        for (int i = 0; i < cost.length; i++) {
            totalcost += cost[i];
        }
        if (totalfuel < totalcost)
            return -1;
        for (int i = 0; i < param1.length; i++) {
            currentFuel = currentFuel + param1[i] - cost[i];
            if (currentFuel < 0) {
                //we have chosen the wrong startting index
                //with that starting indez we won\'t be able to complete the cycle\'
                //hence we need to reset
                start = i + 1;
                currentFuel = 0;
            }
        }
        return start;
    }
}
