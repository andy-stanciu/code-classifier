class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = param1[i] - cost[i];
        }
        int sum = 0, len = 0;
        for (int i = 0; i < 2 * n; i++) {
            sum += arr[i % n];
            len++;
            if (len == n && sum >= 0)
                return i - n + 1;
            if (sum < 0) {
                sum = 0;
                len = 0;
            }
        }
        return -1;
    }
}
