class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0;
        int cap = 0;
        int diff = 0;
        for (int i = 0; i < param1.length; i++) {
            cap += param1[i] - cost[i];
            if (cap < 0) {
                start = i + 1;
                diff += cap;
                cap = 0;
            }
        }
        return (cap + diff >= 0) ? start : -1;
    }
}
