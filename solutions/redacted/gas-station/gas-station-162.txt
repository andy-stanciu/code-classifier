class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int ts = 0;
        int s = 0;
        int st = 0;
        for (int i = 0; i < n; i++) {
            ts += param1[i] - cost[i];
            s += param1[i] - cost[i];
            if (s < 0) {
                st = i + 1;
                s = 0;
            }
        }
        return (ts >= 0) ? st : -1;
    }
}
