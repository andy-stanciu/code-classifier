class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int totalCost = 0;
        int var1 = 0;
        int tank = 0;
        int start = 0;
        for (int i = 0; i < n; i++) {
            totalCost += cost[i];
            var1 += param1[i];
            /**
             * Add current gas[i] & reduce the cost[i] to move to the next station
             */
            tank += (param1[i] - cost[i]);
            /**
             * If tank < 0, We can\'t move to the next station, hence set start as i+1
             *            & tank =0
             */
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        /**
         * Circular trip not possible if totalGas < totalCost
         */
        if (var1 < totalCost)
            return -1;
        /**
         * Return the start index, If circular route is possible
         */
        return start;
    }
}
