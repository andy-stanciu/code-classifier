class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int[] diff = new int[param1.length];
        int fuel = 0, index = 0, totalDiff = 0;
        for (int i = 0; i < cost.length; i++) {
            diff[i] = param1[i] - cost[i];
            fuel = fuel + diff[i];
            totalDiff = totalDiff + diff[i];
            if (fuel < 0) {
                index = i + 1;
                fuel = 0;
            }
        }
        if (totalDiff >= 0)
            return index;
        else
            return -1;
    }
}
