class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0;
        int totalCost = 0;
        int var2 = 0;
        int var3 = 0;
        for (int i = 0; i < param1.length; i++) {
            var1 += param1[i];
            totalCost += cost[i];
            var2 += param1[i] - cost[i];
            // If currentGas is negative, it means we cannot start from the current station.
            // So, we reset the startStation to the next station and reset currentGas.
            if (var2 < 0) {
                var3 = i + 1;
                var2 = 0;
            }
        }
        // If the totalGas is less than totalCost, it means there is no solution.
        if (var1 < totalCost) {
            return -1;
        }
        return var3;
    }
}
