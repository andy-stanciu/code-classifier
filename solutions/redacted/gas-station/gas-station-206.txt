class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int totalTank = 0;
        int currentTank = 0;
        int var1 = 0;
        for (int i = 0; i < param1.length; i++) {
            totalTank += param1[i] - cost[i];
            currentTank += param1[i] - cost[i];
            if (currentTank < 0) {
                var1 = i + 1;
                currentTank = 0;
            }
        }
        return totalTank >= 0 ? var1 : -1;
    }
}
