class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0, avail = 0, ind = 0;
        for (int i = 0; i < param1.length; i++) {
            total += param1[i] - cost[i];
            avail += param1[i] - cost[i];
            if (avail < 0) {
                avail = 0;
                ind = i + 1;
            }
        }
        return total < 0 ? -1 : ind;
    }
}
