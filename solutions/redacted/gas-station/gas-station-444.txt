class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int total = 0, var1 = 0, start = 0;
        for (int i = 0; i < param1.length; i++) {
            int cur = param1[i] - cost[i];
            total += cur;
            var1 += cur;
            // we cal from the start/index_gasSum==0
            // if gas sum < 0, then we cannot reach to the next from start/index_gasSum==0
            // if gas sum < 0. then we start from the next
            if (var1 < 0) {
                var1 = 0;
                // if to the end, then also means cannot goal
                start = i + 1;
            }
        }
        if (total < 0)
            return -1;
        return start;
    }
}
