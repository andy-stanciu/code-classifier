class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int start = 0, currentFuel = 0, var1 = 0, totalCost = 0;
        for (int i = 0; i < param1.length; i++) var1 += param1[i];
        for (int i = 0; i < cost.length; i++) totalCost += cost[i];
        if (var1 < totalCost)
            return -1;
        for (int i = 0; i < param1.length; i++) {
            currentFuel += (param1[i] - cost[i]);
            if (currentFuel < 0) {
                start = i + 1;
                currentFuel = 0;
            }
        }
        return start;
    }
}
