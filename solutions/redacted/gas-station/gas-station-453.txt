class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, i = 0, j = param1.length - 1;
        while (i <= j) {
            if (var1 >= 0)
                var1 += param1[i] - cost[i++];
            else
                var1 += param1[j] - cost[j--];
        }
        return var1 < 0 ? -1 : (j + 1) % param1.length;
    }
}
