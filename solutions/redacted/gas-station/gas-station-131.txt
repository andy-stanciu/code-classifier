class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int currGain = 0, totalGain = 0, startIndex = 0;
        for (int i = 0; i < param1.length; ++i) {
            int pathGain = param1[i] - cost[i];
            totalGain += pathGain;
            currGain += pathGain;
            // If we meet a "valley", start over from the next station
            // with 0 initial gas.
            if (currGain < 0) {
                // start from next index
                startIndex = i + 1;
                currGain = 0;
            }
        }
        return totalGain >= 0 ? startIndex : -1;
    }
}
