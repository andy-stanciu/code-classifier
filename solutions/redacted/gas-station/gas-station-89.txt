class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int running = 0;
        int max = Integer.MIN_VALUE;
        int maxIndex = -1;
        for (int i = param1.length - 1; i >= 0; i--) {
            running += param1[i] - cost[i];
            if (running > max) {
                max = running;
                maxIndex = i;
            }
        }
        return running < 0 ? -1 : maxIndex;
    }
}
