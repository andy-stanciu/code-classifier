class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0, totalCost = 0;
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            totalCost += cost[i];
        }
        if (var1 < totalCost)
            return -1;
        int var2 = 0, start = 0;
        for (int i = 0; i < n; i++) {
            var2 += param1[i];
            var2 -= cost[i];
            if (var2 < 0) {
                start = i + 1;
                var2 = 0;
            }
        }
        return start;
    }
}
