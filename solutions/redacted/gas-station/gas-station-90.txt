class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        // Initialize variables to keep track of the total amount of gas and the current
        // amount of gas in the tank
        int var1 = 0;
        int var2 = 0;
        // Initialize a variable to keep track of the starting index
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            // Calculate the difference between the gas at the current station and the cost
            // of traveling to the next station
            int diff = param1[i] - cost[i];
            // Add the difference to the total gas and the current gas in the tank
            var1 += diff;
            var2 += diff;
            // If the current gas in the tank falls below zero, update the starting index
            // to the next station and reset the current gas to zero
            if (var2 < 0) {
                start = i + 1;
                var2 = 0;
            }
        }
        // If the total gas is greater than or equal to zero, return the starting index,
        // otherwise return -1
        return var1 >= 0 ? start : -1;
    }
}
