class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        for (int i = 0; i < param1.length; ) {
            int distance = getDistance(i, param1, cost);
            if (distance >= param1.length) {
                return i;
            }
            i = i + distance + 1;
        }
        return -1;
    }

    int getDistance(int start, int[] param1, int[] cost) {
        int distance = 0;
        int var1 = param1[start];
        while (var1 >= cost[start] && distance < param1.length) {
            distance++;
            var1 -= cost[start];
            start = (start + 1) % param1.length;
            var1 += param1[start];
        }
        return distance;
    }
}
