public class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int currRemaining = 0;
        int totalRemaining = 0;
        int start = 0;
        for (int i = 0; i < param1.length; i++) {
            int remaining = param1[i] - cost[i];
            if (currRemaining < 0) {
                start = i;
                currRemaining = remaining;
            } else {
                currRemaining += remaining;
            }
            totalRemaining += remaining;
        }
        if (totalRemaining < 0)
            return -1;
        else
            return start;
    }
}
