class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        // Total gas available
        int var1 = 0;
        // Gas in the current tank
        int var2 = 0;
        // Starting gas station index
        int var3 = 0;
        for (int i = 0; i < param1.length; i++) {
            // Difference between gas and cost at the current station
            int diff = param1[i] - cost[i];
            var1 += diff;
            var2 += diff;
            if (var2 < 0) {
                // If the current tank is empty, update the start station to the next station
                var3 = i + 1;
                var2 = 0;
            }
        }
        // If totalGas is negative, it means the total cost of gas is greater than the total available gas,
        // and it is not possible to complete the circuit
        if (var1 < 0) {
            return -1;
        }
        return var3;
    }
}
