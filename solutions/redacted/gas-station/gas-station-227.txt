class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        if (param1 == null || cost == null || param1.length == 0 || cost.length == 0 || param1.length != cost.length) {
            return -1;
        }
        int var1 = Integer.MIN_VALUE;
        int var2 = -1;
        int var3 = 0;
        for (int i = param1.length - 1; i >= 0; i--) {
            var3 += param1[i];
            var3 -= cost[i];
            if (var3 > var1) {
                var1 = var3;
                var2 = i;
            }
        }
        return var3 >= 0 ? var2 : -1;
    }
}
