class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int curr = 0;
        int tf = 0;
        int tc = 0;
        int start = 0;
        int n = param1.length;
        for (int i = 0; i < n; i++) {
            tf += param1[i];
            tc += cost[i];
        }
        if (tf < tc)
            return -1;
        for (int i = 0; i < n; i++) {
            curr += param1[i] - cost[i];
            if (curr < 0) {
                start = i + 1;
                curr = 0;
            }
        }
        return start;
    }
}
