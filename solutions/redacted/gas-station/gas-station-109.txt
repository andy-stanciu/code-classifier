class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int n = param1.length;
        int var1 = 0;
        int sumCost = 0;
        // Calculate the total gas and total cost
        for (int i = 0; i < n; i++) {
            var1 += param1[i];
            sumCost += cost[i];
        }
        // If total gas is less than total cost, we can\'t complete the circuit
        if (var1 < sumCost) {
            return -1;
        }
        int total = 0;
        int result = 0;
        // Iterate through the stations to find the start station
        for (int i = 0; i < n; i++) {
            total += param1[i] - cost[i];
            // If at any point, we can\'t proceed, reset the starting point to the next station
            if (total < 0) {
                total = 0;
                // set the start station to the next one
                result = i + 1;
            }
        }
        return result;
    }
}
