class Solution {

    public int canCompleteCircuit(int[] param1, int[] cost) {
        int var1 = 0, costSum = 0, n = cost.length, ind = -1, prevDiff = 0;
        for (int i = 0; i < n; i++) {
            costSum += cost[i];
            var1 += param1[i];
            // finding the lowest possible diff between cost and gas sum
            if (prevDiff > (var1 - costSum)) {
                ind = i;
                prevDiff = var1 - costSum;
            }
        }
        // if cost sum is more than gas sum then there so no path
        if (costSum > var1)
            return -1;
        // if there is a path we will return the next ind of lowest possible value
        return ind + 1;
    }
}
