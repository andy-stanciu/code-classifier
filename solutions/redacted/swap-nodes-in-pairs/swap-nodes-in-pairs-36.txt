/**
 * class ListNode {
 *     int val;
 *     ListNode next;
 *
 *     ListNode() {
 *         val = 0;
 *         next = null;
 *     }
 *
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 *
 *     ListNode(int x, ListNode next) {
 *         val = x;
 *         this.next = next;
 *     }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode temp = head.next;
        head.next = method1(head.next.next);
        temp.next = head;
        return temp;
    }
}
