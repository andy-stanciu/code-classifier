class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode tu = dummy;
        while (head != null && head.next != null) {
            ListNode a = head;
            ListNode b = head.next;
            tu.next = b;
            a.next = b.next;
            b.next = a;
            tu = a;
            head = a.next;
        }
        return dummy.next;
    }
}
