/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        ListNode ans = head, arr = head, first = null, prev = null, second = null, third = null;
        while (arr != null) {
            first = arr;
            if (arr.next != null)
                second = arr.next;
            else
                break;
            third = arr.next.next;
            first.next = third;
            second.next = first;
            if (prev != null) {
                prev.next = second;
            } else
                ans = second;
            prev = first;
            arr = arr.next;
        }
        return ans;
    }
}
