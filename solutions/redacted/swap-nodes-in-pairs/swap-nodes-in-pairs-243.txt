class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0, head);
        ListNode prev = dummy;
        ListNode curr = head;
        while (curr != null && curr.next != null) {
            ListNode npn = curr.next.next;
            ListNode second = curr.next;
            second.next = curr;
            curr.next = npn;
            prev.next = second;
            prev = curr;
            curr = npn;
        }
        return dummy.next;
    }
}
