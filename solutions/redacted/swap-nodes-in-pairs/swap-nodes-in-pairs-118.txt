class Solution {

    public static ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head.next == null)
            return head;
        var1 = head.next;
        method2(head, head.next, null);
        return var1;
    }

    static ListNode var1 = null;

    private static ListNode method2(ListNode head, ListNode next, ListNode param1) {
        if (head == null) {
            return null;
        }
        if (next == null) {
            return head;
        }
        ListNode var2 = next.next;
        method3(head, next);
        if (param1 != null) {
            param1.next = next;
        }
        ListNode nextPairConnect = head;
        if (var2 == null) {
            return method2(var2, null, nextPairConnect);
        } else {
            method2(var2, var2.next, nextPairConnect);
        }
        return null;
    }

    static void method3(ListNode one, ListNode two) {
        ListNode temp = two.next;
        two.next = one;
        one.next = temp;
    }
}
