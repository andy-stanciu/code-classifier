class Solution {

    public ListNode method1(ListNode head) {
        ListNode temp;
        if (head != null && head.next == null)
            return head;
        if (head == null || head.next == null) {
            return null;
        }
        temp = head.next;
        head.next = temp.next;
        temp.next = head;
        head = temp;
        head.next.next = method1(head.next.next);
        return head;
    }
}
