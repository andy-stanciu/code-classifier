class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode var1 = new ListNode();
        var1.next = head;
        ListNode first = var1;
        ListNode second = head;
        ListNode third = head.next;
        while (third != null) {
            second.next = third.next;
            third.next = second;
            first.next = third;
            first = second;
            second = first.next;
            third = second != null ? second.next : null;
        }
        return var1.next;
    }
}
