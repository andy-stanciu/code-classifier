class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode prev = head;
        ListNode curr = prev.next;
        prev.next = curr.next;
        curr.next = prev;
        ListNode lastprev = prev;
        prev = prev.next;
        ListNode var1 = curr;
        while (prev != null && prev.next != null) {
            curr = prev.next;
            prev.next = curr.next;
            curr.next = prev;
            lastprev.next = curr;
            lastprev = prev;
            prev = prev.next;
        }
        return var1;
    }
}
