class Solution {

    public ListNode method1(ListNode head) {
        ListNode current = head, prev = null, next = null;
        if (current != null)
            next = current.next;
        if (next != null)
            head = next;
        while (next != null) {
            current.next = next.next;
            next.next = current;
            if (prev != null)
                prev.next = next;
            prev = current;
            current = current.next;
            next = (current == null) ? null : current.next;
        }
        return head;
    }
}
