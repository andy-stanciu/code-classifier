/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        ListNode curr = head;
        ListNode nex = null;
        ListNode newNode = new ListNode(-1);
        ListNode temp = newNode;
        while (head != null) {
            if (head.next == null) {
                temp.next = head;
                head = null;
            } else {
                nex = head.next;
                temp.next = new ListNode(nex.val);
                temp.next.next = new ListNode(head.val);
                head = head.next.next;
                temp = temp.next.next;
            }
        }
        return newNode.next;
    }
}
