public class Solution {

    public ListNode method1(ListNode head) {
        if (//Base case
        (head == null) || (head.next == null))
            return head;
        // subproblem
        ListNode n = head.next;
        // subproblem swapping done through recursion
        head.next = method1(head.next.next);
        //merging the subproblem
        n.next = head;
        return n;
    }
}
