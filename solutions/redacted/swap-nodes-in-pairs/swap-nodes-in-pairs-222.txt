class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0, head);
        for (ListNode node = dummy; node != null; ) {
            if (node.next == null || node.next.next == null)
                break;
            ListNode next = node.next, nextnext = next.next;
            next.next = nextnext.next;
            nextnext.next = next;
            node.next = nextnext;
            node = next;
        }
        return dummy.next;
    }
}
