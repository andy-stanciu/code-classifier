/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        //pointer to traverse the list
        ListNode var1 = dummy;
        //loop until there are at least two nodes to swap
        while (var1.next != null && var1.next.next != null) {
            // 2 nodes to be swapped
            ListNode var2 = var1.next;
            ListNode var3 = var1.next.next;
            // swapping both nodes
            var2.next = var3.next;
            var3.next = var2;
            //updating the next of previous node to the new head of the swapped pair
            var1.next = var3;
            //updating pointer for next iteration
            var1 = var2;
        }
        return dummy.next;
    }
}
