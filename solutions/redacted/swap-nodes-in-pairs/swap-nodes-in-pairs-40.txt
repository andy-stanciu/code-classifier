class Solution {

    int cnt = 0;

    public ListNode method1(ListNode node) {
        if (node == null)
            return null;
        cnt++;
        var ret = method1(node.next);
        if (cnt % 2 == 1 && ret != null) {
            node.next = ret.next;
            ret.next = node;
        } else {
            node.next = ret;
            ret = node;
        }
        cnt--;
        return ret;
    }
}
