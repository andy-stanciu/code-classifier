/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode head) {
        Stack<ListNode> var1 = new Stack<>();
        ListNode temp = head;
        ListNode newHead = null, tempNewList = null;
        if (temp == null || temp.next == null) {
            return head;
        }
        var1.add(head);
        temp = head.next;
        while (temp != null) {
            var1.add(temp);
            temp = temp.next;
            if (var1.size() == 2) {
                if (newHead == null) {
                    newHead = var1.pop();
                    newHead.next = var1.pop();
                    newHead.next.next = null;
                    tempNewList = newHead.next;
                } else {
                    tempNewList.next = var1.pop();
                    tempNewList.next.next = var1.pop();
                    tempNewList.next.next.next = null;
                    tempNewList = tempNewList.next.next;
                }
            }
        }
        if (var1.size() == 1) {
            tempNewList.next = var1.pop();
            tempNewList.next.next = null;
        }
        return newHead;
    }
}
