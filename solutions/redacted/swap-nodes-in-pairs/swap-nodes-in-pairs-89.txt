class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return head;
        ListNode tempNext = head.next.next;
        ListNode tempFirst = head.next;
        ListNode tempSecond = head;
        head = tempFirst;
        head.next = tempSecond;
        head.next.next = method1(tempNext);
        return head;
    }
}
