class Solution {

    public ListNode method1(ListNode head) {
        //base case
        if (head == null || head.next == null) {
            return head;
        } else {
            ListNode current = head.next;
            ListNode pre = head;
            //swapping starts
            int temp = pre.val;
            pre.val = current.val;
            current.val = temp;
            method1(current.next);
            return head;
        }
    }
}
