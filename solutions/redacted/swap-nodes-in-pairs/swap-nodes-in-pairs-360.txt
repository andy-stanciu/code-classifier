class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode first = head;
        ListNode second = head.next;
        ListNode temp = head;
        ListNode answer = head;
        while (true) {
            if (first == head) {
                first.next = second.next;
                second.next = first;
                temp = first;
                answer = second;
            } else {
                first.next = second.next;
                second.next = first;
                temp.next = second;
                temp = first;
            }
            if (first.next == null || first.next.next == null) {
                return answer;
            } else {
                first = first.next;
                second = first.next;
            }
        }
    }
}
