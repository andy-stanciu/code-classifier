class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> list = new ArrayList<>();
        combination(list, new ArrayList<>(), n, k, 1);
        return list;
    }

    public void combination(List<List<Integer>> k, ArrayList<Integer> temp, int n, int k1, int index) {
        if (temp.size() == k1) {
            k.add(new ArrayList<>(temp));
            return;
        }
        if (index > n)
            return;
        for (int i = index; i <= n; i++) {
            temp.add(i);
            combination(k, temp, n, k1, i + 1);
            temp.remove(temp.size() - 1);
        }
    }
}
