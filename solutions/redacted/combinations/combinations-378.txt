class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> combine(int n, int k) {
        for (int i = 1; i <= n; i++) {
            ArrayList<Integer> list = new ArrayList<>();
            list.add(i);
            helper(i, list, k, n);
        }
        return ans;
    }

    void helper(int i, ArrayList<Integer> list, int k, int n) {
        if (list.size() == k) {
            ans.add(new ArrayList<>(list));
            return;
        }
        for (int j = i + 1; j <= n; j++) {
            list.add(j);
            helper(j, list, k, n);
            list.remove(list.size() - 1);
        }
    }
}
