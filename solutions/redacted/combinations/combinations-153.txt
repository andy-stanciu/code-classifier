class Solution {

    public void solve(List<List<Integer>> res, List<Integer> list, int idx, int n, int k) {
        if (list.size() == k) {
            res.add(new ArrayList(list));
            return;
        }
        // list.add(idx);
        for (int i = idx; i <= n; i++) {
            list.add(i);
            solve(res, list, i + 1, n, k);
            list.remove(list.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        solve(res, list, 1, n, k);
        return res;
    }
}
