class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> list = new ArrayList<>();
        int[] nums = new int[n];
        int d = 1;
        int i = 0;
        while (n != i) {
            nums[i] = d;
            i++;
            d++;
        }
        helper(nums, 0, k, new ArrayList<>(), list);
        return list;
    }

    public static void helper(int[] nums, int idx, int k, List<Integer> list1, List<List<Integer>> list) {
        //if idx is equal to the n then return
        if (idx == nums.length) {
            //and if k equl to zero then return
            if (k == 0) {
                list.add(new ArrayList(list1));
            }
            return;
        }
        // add element in list temp list1
        list1.add(nums[idx]);
        // call the helper function with increasing index and decreasing k value
        helper(nums, idx + 1, k - 1, list1, list);
        // remove the element in list1 greater then k value
        list1.remove(list1.size() - 1);
        // call again helper function
        helper(nums, idx + 1, k, list1, list);
    }
}
