class Solution {

    private List<List<Integer>> var1 = new ArrayList<>();

    private ArrayList<Integer> nums = new ArrayList<>();

    private int len;

    public void method1(ArrayList<Integer> current, int start) {
        if (current.size() == len) {
            var1.add(current);
            return;
        }
        for (int i = start; i < nums.size(); i++) {
            ArrayList<Integer> copy = new ArrayList<>(current);
            copy.add(nums.get(i));
            method1(copy, i + 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        len = k;
        for (int i = 1; i <= n; i++) {
            nums.add(i);
        }
        method1(new ArrayList<Integer>(), 0);
        System.out.println(nums);
        return var1;
    }
}
