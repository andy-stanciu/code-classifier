class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        solver(ans, temp, n, k, 1);
        return ans;
    }

    public void solver(List<List<Integer>> ans, List<Integer> temp, int n, int k, int i) {
        if (i == n + 1) {
            if (temp.size() == k) {
                ans.add(new ArrayList<>(temp));
            }
            return;
        }
        temp.add(i);
        solver(ans, temp, n, k, i + 1);
        temp.remove(temp.size() - 1);
        solver(ans, temp, n, k, i + 1);
    }
}
