class Solution {

    List<List<Integer>> res = new ArrayList<>();

    private void solve(int start, int end, int k, List<Integer> ans) {
        if (start > end) {
            if (ans.size() == k)
                res.add(new ArrayList<>(ans));
            return;
        }
        ans.add(start);
        solve(start + 1, end, k, ans);
        //backtraacking
        ans.remove(ans.size() - 1);
        solve(start + 1, end, k, ans);
    }

    public List<List<Integer>> combine(int n, int k) {
        solve(1, n, k, new ArrayList<Integer>());
        return res;
    }
}
