class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        rec(k, 0, n, ans, new ArrayList<Integer>());
        return ans;
    }

    public void rec(int k, int currDigit, int n, List<List<Integer>> ans, ArrayList<Integer> curr) {
        if (currDigit > n)
            return;
        if (curr.size() == k) {
            ans.add(new ArrayList<>(curr));
            return;
        }
        curr.add(currDigit + 1);
        rec(k, currDigit + 1, n, ans, curr);
        curr.remove(curr.size() - 1);
        rec(k, currDigit + 1, n, ans, curr);
    }
}
