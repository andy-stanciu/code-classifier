class Solution {

    List<List<Integer>> res = new ArrayList<>();

    void solve(int i, int n, int k, List<Integer> al) {
        if (k == 0) {
            res.add(new ArrayList<>(al));
            return;
        }
        for (int j = i + 1; j <= n; j++) {
            al.add(j);
            solve(j, n, k - 1, al);
            al.remove(al.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<Integer> al = new ArrayList<>();
        solve(0, n, k, al);
        return res;
    }
}
