class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> comb = new ArrayList<List<Integer>>();
        List<Integer> list = new ArrayList<Integer>();
        getALLCombination1(1, n, comb, list, k);
        return comb;
    }

    void getALLCombination1(int index, int n, List<List<Integer>> comb, List<Integer> list, int k) {
        if (list.size() == k) {
            comb.add(new ArrayList<Integer>(list));
            return;
        }
        for (int i = index; i <= n; i++) {
            list.add(i);
            getALLCombination1(i + 1, n, comb, list, k);
            list.remove(list.size() - 1);
        }
    }
}
