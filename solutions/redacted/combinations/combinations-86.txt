class Solution {

    public void combineHelper(int n, int k, int curr, List<Integer> currList, List<List<Integer>> result) {
        // base case: reached list size
        if (currList.size() == k) {
            result.add(currList);
            return;
        }
        // for the current number, try all the numbers from curr to n
        // and add them to the list
        for (int i = curr; i <= n; i++) {
            var newList = new ArrayList<>(currList);
            newList.add(i);
            combineHelper(n, k, i + 1, newList, result);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        var result = new ArrayList<List<Integer>>();
        combineHelper(n, k, 1, new ArrayList<Integer>(), result);
        return result;
    }
}
