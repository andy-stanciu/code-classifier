class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        permute(0, n, k, new ArrayList<>(), result);
        return result;
    }

    public void permute(int start, int end, int k, List<Integer> tmp, List<List<Integer>> result) {
        if (tmp.size() == k) {
            result.add(new ArrayList<>(tmp));
            return;
        }
        int index = 0;
        for (int i = start; i < end; i++) {
            tmp.add(i + 1);
            index = index + 1;
            permute(index + start, end, k, tmp, result);
            tmp.remove(tmp.size() - 1);
        }
    }
}
