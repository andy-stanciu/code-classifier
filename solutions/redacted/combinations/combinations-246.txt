class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList<>();
        helperCombine(1, n, k, new ArrayList<>(), result);
        return result;
    }

    public static void helperCombine(int start, int end, int k, List<Integer> current, List<List<Integer>> result) {
        if (current.size() == k) {
            result.add(new ArrayList<>(current));
            return;
        }
        for (int i = start; i <= end; i++) {
            current.add(i);
            // choose another value
            // Here, index will be i + 1; It should start from the current i
            helperCombine(i + 1, end, k, current, result);
            // remove the value we just added
            current.remove(current.size() - 1);
        }
    }
}
