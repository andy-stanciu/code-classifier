class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public void check(int n, int k, int index, List<Integer> tlist) {
        if (k < 0) {
            return;
        } else if (k == 0) {
            list.add(tlist);
        } else {
            for (int i = index + 1; i <= n; i++) {
                List<Integer> temp = new ArrayList<>(tlist);
                temp.add(i);
                check(n, k - 1, i, temp);
            }
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<Integer> tlist = new ArrayList<>();
        check(n, k, 0, tlist);
        return list;
    }
}
