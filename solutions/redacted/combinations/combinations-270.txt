class Solution {

    List<List<Integer>> res = new LinkedList<>();

    LinkedList<Integer> track = new LinkedList<>();

    // combine problem = find the subsets of n, which have k elements
    public List<List<Integer>> combine(int n, int k) {
        // the number chosen from the range [1,n]
        backtrack(n, k, 1);
        return res;
    }

    public void backtrack(int n, int k, int start) {
        // base case: select the subsets whose length are k
        if (track.size() == k) {
            res.add(new LinkedList(track));
        }
        // backtrack framework
        for (int i = start; i <= n; i++) {
            track.addLast(i);
            backtrack(n, k, i + 1);
            track.removeLast();
        }
    }
}
