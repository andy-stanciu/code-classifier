class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = i + 1;
        helper(res, new ArrayList<>(), arr, k, 0);
        return res;
    }

    public void helper(List<List<Integer>> res, List<Integer> ans, int[] arr, int k, int i) {
        if (ans.size() == k) {
            res.add(new ArrayList<>(ans));
            return;
        }
        if (i < arr.length) {
            ans.add(arr[i]);
            helper(res, ans, arr, k, i + 1);
            ans.remove(ans.size() - 1);
            helper(res, ans, arr, k, i + 1);
        }
    }
}
