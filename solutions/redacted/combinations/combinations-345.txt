class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> solutions = new ArrayList<>();
        List<Integer> subsol = new ArrayList<>();
        method1(solutions, subsol, n, k, 1);
        return solutions;
    }

    public void method1(List<List<Integer>> solutions, List<Integer> subsol, int n, int k, int a) {
        if (k == 0) {
            solutions.add(new ArrayList<>(subsol));
            return;
        }
        for (int i = a; i <= n; i++) {
            subsol.add(i);
            method1(solutions, subsol, n, k - 1, i + 1);
            subsol.remove(subsol.size() - 1);
        }
    }
}
