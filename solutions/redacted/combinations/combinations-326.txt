class Solution {

    //use reoccur
    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> result = new ArrayList();
        combine(n, k, 1, new ArrayList(k), result);
        return result;
    }

    private void combine(int n, int k, int start, ArrayList<Integer> list, List<List<Integer>> result) {
        if (list.size() == k) {
            result.add(list);
            return;
        }
        // the remaining must larger than k
        for (int i = start; i <= n && (list.size() + (n - i + 1)) >= k; i++) {
            ArrayList<Integer> newList = (ArrayList<Integer>) list.clone();
            newList.add(i);
            combine(n, k, i + 1, newList, result);
        }
    }
}
