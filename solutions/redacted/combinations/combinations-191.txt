class Solution {

    public void solve(int st, int n, List<List<Integer>> ans, List<Integer> list, int k) {
        if (list.size() == k) {
            ans.add(new ArrayList<>(list));
            return;
        }
        for (int i = st; i <= n; i++) {
            list.add(i);
            solve(i + 1, n, ans, list, k);
            list.remove(list.size() - 1);
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        solve(1, n, ans, list, k);
        return ans;
    }
}
