class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> set = new ArrayList<>();
        rec(set, 1, n, k, new ArrayList<>());
        return set;
    }

    public void rec(List<List<Integer>> set, int curr, int n, int k, List<Integer> l) {
        if (l.size() == k) {
            set.add(new ArrayList<>(l));
            return;
        }
        if (curr > n) {
            return;
        }
        l.add(curr);
        rec(set, curr + 1, n, k, l);
        l.remove(l.size() - 1);
        rec(set, curr + 1, n, k, l);
    }
}
