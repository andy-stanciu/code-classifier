class Solution {

    public void generate(int i, int n, int k, List<Integer> arr, List<List<Integer>> result) {
        if (k == 0) {
            result.add(new ArrayList(arr));
            return;
        }
        if (i > n) {
            return;
        }
        arr.add(i);
        generate(i + 1, n, k - 1, arr, result);
        arr.remove(arr.size() - 1);
        generate(i + 1, n, k, arr, result);
    }

    public List<List<Integer>> combine(int n, int k) {
        int i = 1;
        ArrayList<List<Integer>> result = new ArrayList<List<Integer>>();
        List<Integer> arr = new ArrayList<Integer>();
        generate(i, n, k, arr, result);
        return result;
    }
}
