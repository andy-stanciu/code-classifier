class Solution {

    public List<List<Integer>> combine(int n, int k) {
        if (k > n)
            return new ArrayList<List<Integer>>();
        if (k == 1) {
            List<List<Integer>> result = new ArrayList<>(n);
            for (int i = 1; i <= n; ++i) result.add(List.of(i));
            return result;
        }
        final List<List<Integer>> result = combine(n - 1, k);
        List<List<Integer>> intermediate = combine(n - 1, k - 1);
        for (List<Integer> l : intermediate) {
            List<Integer> ll = new ArrayList<Integer>(l);
            ll.add(n);
            result.add(ll);
        }
        return result;
    }
}
