class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        backTrack(1, n, k, res, new ArrayList<>());
        return res;
    }

    private static void backTrack(int idx, int n, int k, List<List<Integer>> res, List<Integer> current) {
        if (current.size() == k) {
            res.add(new ArrayList<>(current));
            return;
        }
        for (int i = idx; i <= n; i++) {
            current.add(i);
            backTrack(i + 1, n, k, res, current);
            current.remove(current.size() - 1);
        }
    }
}
