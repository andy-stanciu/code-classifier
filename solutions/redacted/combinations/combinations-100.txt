class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> res = new ArrayList<>();
        method1(n, k, 1, new ArrayList<>(), res);
        return res;
    }

    public void method1(int n, int k, int pos, List<Integer> temp, List<List<Integer>> res) {
        if (temp.size() == k) {
            res.add(new ArrayList<>(temp));
            return;
        }
        for (int i = pos; i <= n; i++) {
            temp.add(i);
            method1(n, k, i + 1, temp, res);
            temp.remove(temp.size() - 1);
        }
    }
}
