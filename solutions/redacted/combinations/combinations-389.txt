class Solution {

    List<List<Integer>> ans = new ArrayList();

    public List<List<Integer>> combine(int n, int k) {
        rec(1, n, k, new ArrayList());
        return ans;
    }

    public void rec(int i, int n, int k, List<Integer> list) {
        if (k == 0) {
            ans.add(new ArrayList(list));
            return;
        }
        if (i > n)
            return;
        //case where curr number is ignored
        rec(i + 1, n, k, list);
        list.add(i);
        //case where curr number is considered
        rec(i + 1, n, k - 1, list);
        //remove curr number for backtracking
        list.remove(list.size() - 1);
    }
}
