class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> var1 = new ArrayList<>();
        f(1, n, k, var1, new ArrayList<>());
        return var1;
    }

    private void f(int i, int n, int k, List<List<Integer>> var1, ArrayList<Integer> list) {
        if (list.size() == k) {
            var1.add(new ArrayList<>(list));
            return;
        }
        for (int j = i; j <= n; j++) {
            list.add(j);
            f(j + 1, n, k, var1, list);
            list.remove(list.size() - 1);
        }
    }
}
