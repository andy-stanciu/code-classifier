class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        backTracking_Combination_Finder(1, n, k, new ArrayList<>(), ans);
        return ans;
    }

    public void backTracking_Combination_Finder(int ind, int n, int k, List<Integer> list, List<List<Integer>> ans) {
        if (list.size() == k) {
            ans.add(new ArrayList<>(list));
            return;
        }
        if (ind > n)
            return;
        list.add(ind);
        backTracking_Combination_Finder(ind + 1, n, k, list, ans);
        list.remove(list.size() - 1);
        backTracking_Combination_Finder(ind + 1, n, k, list, ans);
    }
}
