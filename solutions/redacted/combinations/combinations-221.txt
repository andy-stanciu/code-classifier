class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> list = new ArrayList<>();
        method1(n, 1, k, new ArrayList<>(), list);
        return list;
    }

    void method1(int n, int start, int k, List<Integer> temp, List<List<Integer>> list) {
        if (temp.size() == k) {
            list.add(new ArrayList<>(temp));
            return;
        }
        for (int i = start; i <= n; i++) {
            temp.add(i);
            method1(n, i + 1, k, temp, list);
            temp.remove(temp.size() - 1);
        }
    }
}
