class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> var1 = new LinkedList<>();
        method1(var1, new ArrayList<>(k), 1, 1, n, k);
        return var1;
    }

    private void method1(List<List<Integer>> var1, ArrayList<Integer> currentArray, int currentN, int currentK, int n, int k) {
        if (currentK > k) {
            var1.add(new ArrayList<>(currentArray));
            return;
        }
        for (int i = currentN; i <= n; i++) {
            currentArray.add(i);
            method1(var1, currentArray, i + 1, currentK + 1, n, k);
            currentArray.remove(currentArray.size() - 1);
        }
    }
}
