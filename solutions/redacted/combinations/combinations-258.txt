class Solution {

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<List<Integer>>();
        List<Integer> currans = new ArrayList<Integer>();
        ans = combineB(n, k, ans, currans, 1);
        return ans;
    }

    public List<List<Integer>> combineB(int n, int k, List<List<Integer>> ans, List<Integer> currans, int count) {
        if (currans.size() == k) {
            List<Integer> mylist = new ArrayList<Integer>();
            mylist.addAll(currans);
            ans.add(mylist);
            return ans;
        }
        List<List<Integer>> myans = new ArrayList<List<Integer>>();
        for (int i = count; i <= n; i++) {
            currans.add(i);
            combineB(n, k, ans, currans, i + 1);
            currans.remove(currans.size() - 1);
        }
        return ans;
    }
}
