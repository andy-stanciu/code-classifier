class Solution {

    public String convert(String s, int numRows) {
        char[] sChar = s.toCharArray();
        Map<Integer, String> sq = new HashMap<>();
        int n = 0;
        boolean inc = true;
        for (char c : sChar) {
            if (n == 0) {
                inc = true;
            } else if (n == numRows - 1) {
                inc = false;
            }
            String currentValue = sq.getOrDefault(n, "");
            sq.put(n, currentValue + c);
            if (inc) {
                n++;
            } else {
                n--;
            }
        }
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < sq.size(); i++) {
            result.append(sq.get(i));
        }
        String concatenatedString = result.toString();
        return result.toString();
    }
}
