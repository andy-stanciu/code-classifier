class Solution {

    public String convert(String s, int numsRows) {
        if (numsRows == 1) {
            return s;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < numsRows; i++) {
            int idx = i;
            int south = 2 * (numsRows - 1 - i);
            int north = 2 * i;
            boolean goingSouth = true;
            while (idx < s.length()) {
                sb.append(s.charAt(idx));
                if (i == 0) {
                    idx += south;
                } else if (i == numsRows - 1) {
                    idx += north;
                } else {
                    if (goingSouth) {
                        idx += south;
                    } else {
                        idx += north;
                    }
                    goingSouth = !goingSouth;
                }
            }
        }
        return sb.toString();
    }
}
