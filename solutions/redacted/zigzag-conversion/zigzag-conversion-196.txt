class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1 || s.length() <= numRows)
            return s;
        StringBuilder res = new StringBuilder();
        int step = 2 * numRows - 2;
        for (int i = 0; i < numRows; i++) {
            for (int j = i; j < s.length(); j += step) {
                res.append(s.charAt(j));
                int diagonal = j + step - 2 * i;
                if (i != 0 && i != numRows - 1 && diagonal < s.length()) {
                    res.append(s.charAt(diagonal));
                }
            }
        }
        return res.toString();
    }

    public static void main(String[] args) {
        Solution sln = new Solution();
        System.out.println(sln.convert("PAYPALISHIRING", 3));
        System.out.println(sln.convert("PAYPALISHIRING", 4));
        System.out.println(sln.convert("A", 1));
    }
}
