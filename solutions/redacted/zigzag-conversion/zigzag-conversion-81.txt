class Solution {

    public String convert(String s, int numRows) {
        int l = s.length();
        if (l == 0 || l == 1)
            return s;
        else if (numRows > l || numRows == 1)
            return s;
        int hop = (numRows * 2) - 2;
        StringBuilder res = new StringBuilder();
        int pos = 0;
        for (int row = 0; row < numRows; row++) {
            pos = row;
            if (row == 0 || row == numRows - 1) {
                while (pos < l) {
                    res.append(s.charAt(pos));
                    pos += hop;
                }
            } else {
                int gap = hop - (2 * row);
                while (pos < l) {
                    res.append(s.charAt(pos));
                    // Below we use if condition to check if the gap is within range of string length else ignore in which happens in certain cases.
                    if (pos + gap < l)
                        res.append(s.charAt(pos + gap));
                    pos += hop;
                }
            }
        }
        return res.toString();
    }
}
