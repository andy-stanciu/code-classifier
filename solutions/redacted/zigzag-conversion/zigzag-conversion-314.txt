class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        StringBuilder s1 = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            int ind = i;
            //deltasouth means niche se uper jate time   kitane elements hai
            int deltasouth = 2 * (numRows - i - 1);
            int deltanorth = 2 * i;
            boolean goingsouth = true;
            while (ind < s.length()) {
                s1.append(s.charAt(ind));
                // System.out.println(i+"-"+ind+"-"+s1);
                if (i == 0) {
                    ind += deltasouth;
                } else if (i == numRows - 1) {
                    ind += deltanorth;
                } else {
                    if (goingsouth) {
                        ind += deltasouth;
                    } else {
                        ind += deltanorth;
                    }
                }
                goingsouth = !goingsouth;
            }
        }
        return s1.toString();
    }
}
