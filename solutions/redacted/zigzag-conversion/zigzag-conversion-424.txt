class Solution {

    public String convert(String s, int numRows) {
        char[] seq = new char[s.length()];
        int it = 0;
        for (int i = 0; i < numRows; i++) {
            boolean b = true;
            int a = i;
            while (a < s.length()) {
                seq[it++] = s.charAt(a);
                if (i == 0 || i == numRows - 1) {
                    a += Math.max(1, 2 * (numRows - 1));
                } else {
                    if (b) {
                        b = false;
                        a += 2 * (numRows - i - 1);
                    } else {
                        b = true;
                        a += 2 * i;
                    }
                }
            }
        }
        return String.valueOf(seq);
    }
}
