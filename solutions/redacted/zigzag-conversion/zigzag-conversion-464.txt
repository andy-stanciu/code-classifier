class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        char[][] answer = new char[numRows][s.length()];
        for (int i = 0, letter = 0; letter < s.length(); i++) {
            for (int j = 0; j < numRows && letter < s.length(); j++) {
                if (i % (numRows - 1) == 0) {
                    answer[j][i] = s.charAt(letter++);
                } else {
                    answer[numRows - 1 - (i % (numRows - 1))][i] = s.charAt(letter++);
                    break;
                }
            }
        }
        StringBuilder stringBuilder = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            for (int j = 0; j < s.length(); j++) {
                stringBuilder.append(answer[i][j] != 0 ? answer[i][j] : "");
            }
        }
        return stringBuilder.toString();
    }
}
