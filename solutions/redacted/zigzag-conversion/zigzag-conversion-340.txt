class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        StringBuilder[] temp = new StringBuilder[numRows];
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < numRows; i++) {
            temp[i] = new StringBuilder();
        }
        int i = 0;
        int tempIdx = 0;
        //True : Down & False : UP
        boolean flag = true;
        while (i < s.length()) {
            //Logic
            temp[tempIdx].append(s.charAt(i));
            i++;
            if (flag == true) {
                tempIdx++;
            } else {
                tempIdx--;
            }
            if (tempIdx > numRows - 1) {
                flag = false;
                tempIdx = tempIdx - 2;
            }
            if (tempIdx < 0) {
                flag = true;
                tempIdx = tempIdx + 2;
            }
        }
        for (int j = 0; j < numRows; j++) {
            ans.append(temp[j]);
        }
        return ans.toString();
    }
}
