class Solution {

    public String convert(String s, int n) {
        //if n equals 1, the string remains same.
        if (n == 1)
            return s;
        StringBuilder res = new StringBuilder();
        //increments values add up to t
        int t = (n - 1) * 2;
        for (int i = 0; i < n; i++) {
            //t1 and t2 are initialized
            int t1 = t - 2 * i, t2 = t - t1;
            //if t1 or t2 is 0, same character is printed again,
            //so correct it
            if (t1 == 0)
                t1 = t2;
            if (t2 == 0)
                t2 = t1;
            //set the target value as i
            int target = i;
            boolean flag = true;
            while (target < s.length()) {
                //appends target to result
                res.append(s.charAt(target));
                //incrementing t1 and t2 alternatively
                if (flag)
                    target += t1;
                else
                    target += t2;
                flag = !flag;
            }
        }
        return res.toString();
    }
}
