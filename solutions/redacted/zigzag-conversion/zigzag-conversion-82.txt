class Solution {

    public String convert(String s, int numRows) {
        if (s.length() == 0) {
            return "";
        }
        if (numRows == 1) {
            return s;
        }
        StringBuilder resultBuilder = new StringBuilder();
        int repetitionStep = 2 * numRows - 2;
        for (int i = 0; i < numRows; i++) {
            for (int j = i, nextCharStep = i == numRows - 1 ? repetitionStep : repetitionStep - i * 2; j < s.length(); j += nextCharStep, nextCharStep = repetitionStep == nextCharStep ? nextCharStep : repetitionStep - nextCharStep) {
                resultBuilder.append(s.charAt(j));
            }
        }
        return resultBuilder.toString();
    }
}
