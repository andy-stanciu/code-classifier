class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        StringBuilder sb = new StringBuilder();
        int jump = (numRows - 1) * 2;
        for (int i = 0; i < numRows; i++) {
            for (int j = i; j < s.length(); j += jump) {
                // Add character from the current row
                sb.append(s.charAt(j));
                // Calculate the position of the next character to be added from the current row
                int nextCharPos = j + jump - 2 * i;
                // Ensure that we do not add a character from the first and last rows twice
                if (i != 0 && i != numRows - 1 && nextCharPos < s.length()) {
                    sb.append(s.charAt(nextCharPos));
                }
            }
        }
        return sb.toString();
    }
}
