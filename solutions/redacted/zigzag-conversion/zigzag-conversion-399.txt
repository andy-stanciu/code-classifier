class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        String[] str = new String[numRows];
        for (int i = 0; i < str.length; i++) str[i] = new String();
        int flag = 0;
        int val = -1;
        for (int i = 0; i < s.length(); i++) {
            if (flag == 0) {
                val++;
                if (val == numRows - 1) {
                    flag = 1;
                }
            } else {
                val--;
                if (val == 0) {
                    flag = 0;
                }
            }
            str[val] += s.charAt(i);
        }
        String ans = "";
        for (int i = 0; i < str.length; i++) ans += str[i];
        return ans;
    }
}
