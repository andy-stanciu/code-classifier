class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1)
            return s;
        char[] series = s.toCharArray();
        int len = s.length();
        int step = (numRows - 1) * 2;
        StringBuilder str = new StringBuilder();
        for (//construct the first row, contains only 1 character per cycle (on the vertical line of the Zigzag)
        //construct the first row, contains only 1 character per cycle (on the vertical line of the Zigzag)
        int i = 0; //construct the first row, contains only 1 character per cycle (on the vertical line of the Zigzag)
        i < len; i += step) str.append(series[i]);
        int limit = numRows - 1;
        for (int row = 1; row < limit; row++) {
            //construct rows from 1 to numRows-1 (the middle rows contain 2 characters, one on the vertical line and one on the diagonal line of the zigzag)
            int calc = 2 * (numRows - row - 1);
            for (int cycle = row; cycle < len; cycle += step) {
                str.append(series[cycle]);
                int ind = cycle + calc;
                if (ind < len)
                    str.append(series[ind]);
            }
        }
        for (//construct the last row, contains only 1 character per cycle (on the vertical line of the Zigzag)
        //construct the last row, contains only 1 character per cycle (on the vertical line of the Zigzag)
        int i = numRows - 1; //construct the last row, contains only 1 character per cycle (on the vertical line of the Zigzag)
        i < len; i += step) str.append(series[i]);
        return str.toString();
    }
}
