class Solution {

    public String convert(String s, int numRows) {
        StringBuilder output = new StringBuilder();
        int j = 0;
        if (numRows == 1) {
            return s;
        }
        for (int i = 0; i < numRows; i++) {
            j = i;
            if (j == 0 || j == numRows - 1) {
                while (j < s.length()) {
                    output.append(s.charAt(j));
                    j = j + 2 * numRows - 2;
                }
            } else {
                while (j < s.length()) {
                    output.append(s.charAt(j));
                    if (j + 2 * numRows - 2 - 2 * i < s.length()) {
                        j = j + 2 * numRows - 2 - 2 * i;
                        output.append(s.charAt(j));
                        j = j + 2 * i;
                    } else {
                        j = j + 2 * numRows - 2;
                    }
                }
            }
        }
        return output.toString();
    }
}
