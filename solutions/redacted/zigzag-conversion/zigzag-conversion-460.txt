class Solution {

    public String convert(String s, int numRows) {
        // Length of the input string
        int n = s.length();
        // Create an array of StringBuffers to store characters in each row
        StringBuffer[] sb = new StringBuffer[numRows];
        // Initialize each StringBuffer
        for (int i = 0; i < numRows; i++) sb[i] = new StringBuffer();
        int i = 0;
        while (i < n) {
            // Traverse down the rows and append characters to each row
            for (int j = 0; j < numRows && i < n; j++) {
                sb[j].append(s.charAt(i++));
            }
            // Traverse up the rows (excluding the first and last rows) and append characters
            for (int j = numRows - 2; j > 0 && i < n; j--) {
                sb[j].append(s.charAt(i++));
            }
        }
        // Combine the StringBuffers to form the final result
        StringBuffer ans = new StringBuffer();
        for (StringBuffer res : sb) {
            ans.append(res);
        }
        // Convert the StringBuffer to a String and return
        return ans.toString();
    }
}
