class Solution {

    public static String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        int n = s.length();
        StringBuilder sb = new StringBuilder();
        int i, j;
        for (i = 0; i < numRows; i++) {
            j = i;
            while (j < n) {
                sb.append(s.charAt(j));
                j += (numRows - 1 - j % (numRows - 1)) * 2;
            }
        }
        return sb.toString();
    }
}
