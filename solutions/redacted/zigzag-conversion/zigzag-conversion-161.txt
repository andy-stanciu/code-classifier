class Solution {

    public String convert(String s, int numRows) {
        //  Special cases
        if (s.length() == 1 || numRows == 1)
            return s;
        StringBuilder[] var1 = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            var1[i] = new StringBuilder();
        }
        //  Keeps track of row of zigzag array to add to
        int row = 0;
        //  -1 for down, 1 for up
        int direction = -1;
        //  Add characters one by one to zigzag
        for (char c : s.toCharArray()) {
            var1[row].append(c);
            //  Pointer will reverse direction at either end of zigzag
            if (row == 0 || row == numRows - 1)
                direction *= -1;
            //  Move pointer in direction
            row += direction;
        }
        //  A final stringbuilder is used to turn the array of
        //  stringbuilders to a single stringbuilder
        StringBuilder result = new StringBuilder();
        for (StringBuilder r : var1) result.append(r);
        return result.toString();
    }
}
