class Solution {

    public String convert(String s, int numRows) {
        String[] s1 = new String[numRows];
        Arrays.fill(s1, "");
        int k = 0;
        int d = 1;
        if (numRows == 1)
            return s;
        for (int i = 0; i < s.length(); i++) {
            s1[k] += s.charAt(i);
            k = k + d;
            if (k == numRows) {
                k = numRows - 2;
                d = -1;
            } else if (k == -1) {
                k = 1;
                d = 1;
            }
        }
        String s2 = "";
        for (int i = 0; i < numRows; i++) {
            s2 = s2 + s1[i];
        }
        return s2;
    }
}
