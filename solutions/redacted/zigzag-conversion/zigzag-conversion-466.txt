class Solution {

    public String convert(String s, int numRows) {
        if (numRows <= 1)
            return s;
        String[] rowStrings = new String[numRows];
        Arrays.fill(rowStrings, "");
        int l = s.length();
        int c = 0;
        int i = 0;
        while (c < l) {
            for (; i < numRows; i++) {
                if (c < l) {
                    rowStrings[i] = rowStrings[i].concat(String.valueOf(s.charAt(c)));
                    c++;
                } else {
                    break;
                }
            }
            i--;
            for (i = i - 1; i >= 0; i--) {
                if (c < l) {
                    rowStrings[i] = rowStrings[i].concat(String.valueOf(s.charAt(c)));
                    c++;
                } else {
                    break;
                }
            }
            i = 1;
        }
        return String.join("", rowStrings);
    }
}
