class Solution {

    public String convert(String s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        int n = s.length();
        String sol = "";
        for (int i = 0; i < numRows; i++) {
            int j = i;
            while (j < n) {
                sol += s.charAt(j);
                if (i != 0 && i != numRows - 1) {
                    j = j + 2 * (numRows - i - 1);
                    if (j < n) {
                        sol += s.charAt(j);
                    }
                    j = j + 2 * (i);
                } else {
                    j = j + 2 * (numRows - 1);
                }
            }
        }
        return sol;
    }
}
