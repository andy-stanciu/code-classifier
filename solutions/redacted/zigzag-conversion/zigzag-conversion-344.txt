class Solution {

    public String convert(String s, int numRows) {
        int length = s.length();
        if (numRows == 1 || numRows == length)
            return s;
        char[] ans = new char[length];
        int count = 0;
        int inc = 2 * (numRows - 1);
        for (int i = 0; i < numRows; i++) {
            int j = i;
            while (j < length) {
                ans[count++] = s.charAt(j);
                if (i == 0 || i == numRows - 1)
                    j += inc;
                else {
                    j += (inc - (2 * i));
                    if (j < length) {
                        ans[count++] = s.charAt(j);
                        j += (2 * i);
                    }
                }
            }
        }
        return new String(ans);
    }
}
