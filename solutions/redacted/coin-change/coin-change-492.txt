class Solution {

    public int method1(int[] param1, int amount) {
        if (amount < 1)
            return 0;
        return helper(param1, amount, new int[amount]);
    }

    private int helper(int[] param1, int rem, int[] count) {
        // rem: remaining coins after the last step; count[rem]: minimum number of coins to sum up to rem
        // not valid
        if (rem < 0)
            return -1;
        // completed
        if (rem == 0)
            return 0;
        // already computed, so reuse
        if (count[rem - 1] != 0)
            return count[rem - 1];
        int min = Integer.MAX_VALUE;
        for (int var1 : param1) {
            int res = helper(param1, rem - var1, count);
            if (res >= 0 && res < min)
                min = 1 + res;
        }
        count[rem - 1] = (min == Integer.MAX_VALUE) ? -1 : min;
        return count[rem - 1];
    }
}
