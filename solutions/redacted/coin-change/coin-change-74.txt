class Solution {

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        for (int i = 1; i <= amount; i++) {
            dp[i] = -1;
        }
        for (int i = 1; i <= amount; i++) {
            for (int var1 : param1) {
                if (i >= var1 && dp[i - var1] > -1) {
                    dp[i] = dp[i] == -1 ? dp[i - var1] + 1 : Math.min(dp[i], dp[i - var1] + 1);
                }
            }
        }
        return dp[amount];
    }
}
