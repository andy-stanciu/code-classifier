class Solution {

    public int method1(int[] param1, int amount) {
        int n = param1.length;
        int m = amount;
        int[] dp = new int[m + 1];
        Arrays.fill(dp, (int) (1e9));
        dp[0] = 0;
        for (int var1 : param1) {
            for (int j = var1; j <= m; j++) {
                dp[j] = Math.min(dp[j], dp[j - var1] + 1);
            }
        }
        return dp[m] == (int) (1e9) ? -1 : dp[m];
    }
}
