class Solution {

    public int solveDp(int[] param1, int amount, int[] dp) {
        if (amount == 0) {
            return 0;
        }
        if (amount < 0) {
            return Integer.MAX_VALUE;
        }
        if (dp[amount] != -1) {
            return dp[amount];
        }
        int mini = Integer.MAX_VALUE;
        for (int i = 0; i < param1.length; i++) {
            int var1 = solveDp(param1, amount - param1[i], dp);
            if (var1 != Integer.MAX_VALUE)
                mini = Math.min(var1 + 1, mini);
        }
        return dp[amount] = mini;
    }

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        for (int i = 0; i < dp.length; i++) {
            dp[i] = -1;
        }
        int ans = solveDp(param1, amount, dp);
        if (ans != Integer.MAX_VALUE) {
            return ans;
        } else {
            return -1;
        }
    }
}
