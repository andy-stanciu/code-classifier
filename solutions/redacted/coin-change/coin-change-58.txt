class Solution {

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        for (int target = 1; target < dp.length; target++) {
            dp[target] = dp.length;
            for (int var1 : param1) {
                if (target >= var1) {
                    dp[target] = Math.min(dp[target], dp[target - var1] + 1);
                }
            }
        }
        return dp[amount] == dp.length ? -1 : dp[amount];
    }
}
