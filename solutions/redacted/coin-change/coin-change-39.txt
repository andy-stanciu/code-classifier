class Solution {

    public int method1(int[] arr, int target) {
        if (target == 0)
            return target;
        int[] dp = new int[target + 1];
        for (int i = 1; i <= target; i++) {
            int mn = Integer.MAX_VALUE;
            for (int k : arr) {
                if (k <= i && dp[i - k] != -1) {
                    mn = Math.min(dp[i - k], mn);
                }
            }
            dp[i] = (mn == Integer.MAX_VALUE ? -1 : 1 + mn);
        }
        return dp[target];
    }
}
