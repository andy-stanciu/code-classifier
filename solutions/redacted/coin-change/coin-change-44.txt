class Solution {

    public int method1(int[] param1, int amount) {
        // Create DP table
        int[][] dp = new int[param1.length + 1][amount + 1];
        for (int i = 0; i < param1.length + 1; i++) for (int j = 0; j < amount + 1; j++) {
            if (// First row initialization
            i == 0)
                dp[i][j] = Integer.MAX_VALUE - 1;
            else if (// First column initialization
            j == 0)
                dp[i][j] = 0;
            else if (// Second row initialization
            i == 1)
                if (j % param1[0] == 0)
                    dp[i][j] = j / param1[0];
                else
                    dp[i][j] = Integer.MAX_VALUE - 1;
            else if (//Rest of the rows and columns calculation
            i > 1 && param1[i - 1] > j)
                dp[i][j] = dp[i - 1][j];
            else if (i > 1)
                dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - param1[i - 1]] + 1);
        }
        return dp[param1.length][amount] == Integer.MAX_VALUE - 1 ? -1 : dp[param1.length][amount];
    }
}
