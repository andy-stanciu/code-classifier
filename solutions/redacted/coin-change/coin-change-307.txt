class Solution {

    public int method1(int[] param1, int amount) {
        int[] amt = new int[amount + 1];
        Arrays.fill(amt, amount + 1);
        amt[0] = 0;
        for (int i = 1; i <= amount; i++) {
            for (int j = 0; j < param1.length; j++) {
                if (i >= param1[j])
                    amt[i] = Math.min(amt[i], amt[i - param1[j]] + 1);
            }
        }
        if (amt[amount] < amount + 1)
            return amt[amount];
        return -1;
    }
}
