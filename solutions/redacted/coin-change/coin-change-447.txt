class Solution {

    public int method1(int[] param1, int amount) {
        int[][] dp = new int[param1.length + 1][amount + 1];
        for (int[] d : dp) Arrays.fill(d, -1);
        int ans = rec(param1.length, param1, 0, amount, dp);
        return ans == Integer.MAX_VALUE - 1 ? -1 : ans;
    }

    public static int rec(int n, int[] param1, int i, int amount, int[][] dp) {
        if (amount == 0)
            return dp[i][amount] = 0;
        if (i == n || amount < 0)
            return Integer.MAX_VALUE - 1;
        if (dp[i][amount] != -1)
            return dp[i][amount];
        int a1 = rec(n, param1, i + 1, amount, dp);
        int a2 = 1 + rec(n, param1, i, amount - param1[i], dp);
        return dp[i][amount] = Math.min(a1, a2);
    }
}
