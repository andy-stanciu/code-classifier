class Solution {

    public int method1(int[] param1, int amount) {
        // Array to store minimum number of coins for each amount
        int[] dp = new int[amount + 1];
        // Initialize with a value greater than the maximum possible
        Arrays.fill(dp, amount + 1);
        // It takes 0 coins to make amount 0
        dp[0] = 0;
        for (int var1 : param1) {
            for (int i = var1; i <= amount; i++) {
                dp[i] = Math.min(dp[i], dp[i - var1] + 1);
            }
        }
        return dp[amount] > amount ? -1 : dp[amount];
    }
}
