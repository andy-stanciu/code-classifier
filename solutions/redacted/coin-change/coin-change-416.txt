class Solution {

    public int method1(int[] param1, int amount) {
        if (amount == 0)
            return 0;
        // Arrays.sort(coins);
        int[] dp = new int[amount + 1];
        int MAX = 2 * amount;
        //keeping infinity as 2*amount to avoid 1+ infinity leading to overflows
        Arrays.fill(dp, MAX);
        //dp[i] -> min no. of coins to form i
        //no. of coins to form 0 is 0
        dp[0] = 0;
        for (int i = 1; i <= amount; i++) {
            for (int var1 : param1) {
                if (i >= var1) {
                    dp[i] = Math.min(dp[i], 1 + dp[i - var1]);
                }
            }
        }
        return dp[amount] == MAX ? -1 : dp[amount];
    }
}
