class Solution {

    public int method1(int[] param1, int amount) {
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, amount + 1);
        dp[0] = 0;
        for (int var1 : param1) {
            for (int i = var1; i <= amount; i++) {
                dp[i] = Math.min(dp[i], dp[i - var1] + 1);
            }
        }
        return dp[amount] <= amount ? dp[amount] : -1;
    }
}
