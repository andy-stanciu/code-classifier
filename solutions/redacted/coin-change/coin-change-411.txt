class Solution {

    List<Integer> selected;

    int min = Integer.MAX_VALUE;

    public int method1(int[] param1, int amount) {
        selected = new ArrayList<>();
        backtrack(param1, amount, 0);
        return min != Integer.MAX_VALUE ? min : -1;
    }

    private void backtrack(int[] param1, int amount, int curAmount) {
        if (curAmount == amount) {
            min = Math.min(min, selected.size());
            return;
        } else if (selected.size() >= min) {
            return;
        }
        for (int i = param1.length - 1; i >= 0; i--) {
            if (curAmount + param1[i] > amount) {
                continue;
            }
            selected.add(param1[i]);
            backtrack(param1, amount, curAmount + param1[i]);
            selected.remove(selected.size() - 1);
        }
    }
}
