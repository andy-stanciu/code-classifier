class Solution {

    public int method1(int[] param1, int amount) {
        int n = param1.length;
        int[] dp = new int[amount + 1];
        for (int i = 0; i <= amount; i++) {
            dp[i] = -1;
        }
        dp[0] = 0;
        for (int i = 0; i < n; i++) {
            for (int a = param1[i]; a <= amount; a++) {
                if (dp[a] == -1) {
                    if (dp[a - param1[i]] != -1) {
                        dp[a] = dp[a - param1[i]] + 1;
                    }
                } else {
                    if (dp[a - param1[i]] != -1) {
                        dp[a] = Math.min(dp[a], dp[a - param1[i]] + 1);
                    }
                }
            }
        }
        return dp[amount];
    }
}
