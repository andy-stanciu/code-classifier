class Solution {

    private void recurPermute(int[] nums, List<Integer> ds, List<List<Integer>> ans, boolean[] freq) {
        if (ds.size() == nums.length) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!freq[i]) {
                // Skip duplicates in the same level of recursion
                if (i > 0 && nums[i] == nums[i - 1] && !freq[i - 1]) {
                    continue;
                }
                freq[i] = true;
                ds.add(nums[i]);
                recurPermute(nums, ds, ans, freq);
                ds.remove(ds.size() - 1);
                freq[i] = false;
            }
        }
    }

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        boolean[] freq = new boolean[nums.length];
        // Sort the array to easily skip duplicates
        Arrays.sort(nums);
        recurPermute(nums, ds, ans, freq);
        return ans;
    }
}
