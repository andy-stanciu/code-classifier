class Solution {

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(nums);
        backTracking(new boolean[nums.length], new ArrayList<>(), result, nums);
        return result;
    }

    public static void backTracking(boolean[] vis, List<Integer> currList, List<List<Integer>> result, int[] nums) {
        if (currList.size() == nums.length) {
            result.add(new ArrayList<>(currList));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (vis[i])
                continue;
            if (i > 0 && nums[i] == nums[i - 1] && !vis[i - 1])
                continue;
            vis[i] = true;
            currList.add(nums[i]);
            backTracking(vis, currList, result, nums);
            currList.remove(currList.size() - 1);
            vis[i] = false;
        }
    }
}
