class Solution {

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        // Sort the array to handle duplicates
        Arrays.sort(nums);
        // Correct length
        backTracking(result, new ArrayList<>(), nums, new boolean[nums.length]);
        return result;
    }

    public static void backTracking(List<List<Integer>> result, List<Integer> tempList, int[] nums, boolean[] vis) {
        if (tempList.size() == nums.length) {
            // Add the current permutation to the result
            result.add(new ArrayList<>(tempList));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            // Skip used elements
            if (vis[i])
                continue;
            // Skip duplicates
            if (i > 0 && nums[i] == nums[i - 1] && !vis[i - 1])
                continue;
            // Mark the element as used
            vis[i] = true;
            // Add the element to the current permutation
            tempList.add(nums[i]);
            // Recur to build the next part
            backTracking(result, tempList, nums, vis);
            // Backtrack, unmark the element
            vis[i] = false;
            // Remove the last element to try the next one
            tempList.remove(tempList.size() - 1);
        }
    }
}
