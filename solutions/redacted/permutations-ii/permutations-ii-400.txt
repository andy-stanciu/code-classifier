class Solution {

    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        for (int n : nums) temp.add(n);
        permute(res, new ArrayList<>(), temp);
        return res;
    }

    public void permute(List<List<Integer>> res, List<Integer> list, List<Integer> temp) {
        if (temp.size() == 0) {
            if (!res.contains(new ArrayList<>(list)))
                res.add(list);
            return;
        }
        for (Integer t : temp) {
            List<Integer> l1 = new ArrayList<>(list);
            List<Integer> l2 = new ArrayList<>(temp);
            l1.add(t);
            l2.remove(t);
            permute(res, l1, l2);
        }
    }
}
