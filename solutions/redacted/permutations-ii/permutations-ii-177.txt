class Solution {

    List<List<Integer>> l = new ArrayList<>();

    public List<List<Integer>> permuteUnique(int[] nums) {
        boolean[] b = new boolean[nums.length];
        help(nums, new ArrayList<>(), b);
        return l;
    }

    void help(int[] n, List<Integer> ll, boolean[] b) {
        if (ll.size() == n.length) {
            if (!l.contains(ll))
                l.add(new ArrayList<>(ll));
            return;
        }
        for (int i = 0; i < n.length; i++) {
            if (!b[i]) {
                b[i] = true;
                ll.add(n[i]);
                help(n, ll, b);
                ll.remove(ll.size() - 1);
                b[i] = false;
            }
        }
    }
}
