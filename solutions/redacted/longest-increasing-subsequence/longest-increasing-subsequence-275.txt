class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        return recur(0, -1, n, nums);
    }

    public int recur(int i, int prev, int n, int[] nums) {
        if (i >= n)
            return 0;
        int nonPick = recur(i + 1, prev, n, nums);
        int pick = 0;
        if (prev == -1 || nums[i] > nums[prev]) {
            pick = 1 + recur(i + 1, i, n, nums);
        }
        return Math.max(pick, nonPick);
    }
}
