class Solution {

    public int lengthOfLIS(int[] nums) {
        // Create an array T to store the length of the longest subsequence at each index
        int[] T = new int[nums.length];
        // Start main pointer
        for (int i = 1; i < nums.length; i++) {
            // Start second pointer
            for (int j = 0; j < i; j++) {
                if (nums[i] > nums[j]) {
                    // Check if the current number is greater than the previous one
                    if (T[j] + 1 > T[i]) {
                        // Update the LIS length at index i
                        T[i] = T[j] + 1;
                    }
                }
            }
        }
        // Find the maximum value in the T array
        int maxIndex = 0;
        for (int i = 0; i < T.length; i++) {
            if (T[i] > T[maxIndex]) {
                maxIndex = i;
            }
        }
        // Return the longest increasing subsequence length
        return T[maxIndex] + 1;
    }
}
