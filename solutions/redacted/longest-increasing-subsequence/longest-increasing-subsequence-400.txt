class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        int maxResult = 1;
        int startIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            int index = binarySearch(dp, nums[i], 0, startIndex - 1);
            dp[index] = nums[i];
            if (index == startIndex) {
                startIndex++;
            }
            maxResult = Math.max(index + 1, maxResult);
        }
        return maxResult;
    }

    int binarySearch(int[] dp, int numberToSearch, int start, int end) {
        while (start <= end) {
            int mid = (start + end) / 2;
            if (dp[mid] >= numberToSearch) {
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return start;
    }
}
