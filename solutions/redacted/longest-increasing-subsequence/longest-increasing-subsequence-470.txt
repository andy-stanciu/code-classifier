class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] DP = new int[n];
        Arrays.fill(DP, 1);
        int max = 1;
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    DP[i] = Math.max(DP[i], DP[j] + 1);
                }
            }
            max = Math.max(max, DP[i]);
        }
        return max;
    }
}
