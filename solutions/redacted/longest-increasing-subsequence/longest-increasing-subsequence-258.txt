class Solution {

    public int lengthOfLIS(int[] nums) {
        int max = 0;
        int[] arrList = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            arrList[i] = 1;
        }
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] > nums[j] && arrList[i] <= arrList[j]) {
                    arrList[i] = arrList[j] + 1;
                }
            }
            if (max < arrList[i])
                max = arrList[i];
        }
        return max;
    }
}
