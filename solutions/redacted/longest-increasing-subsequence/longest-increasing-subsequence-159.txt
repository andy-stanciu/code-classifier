class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[][] dp = new int[n + 1][n + 1];
        for (int index = n - 1; index >= 0; index--) {
            for (int prev = -1; prev < index; prev++) {
                int nottaken = dp[index + 1][prev + 1];
                int taken = 0;
                if (prev == -1 || nums[index] > nums[prev]) {
                    taken = 1 + dp[index + 1][index + 1];
                }
                dp[index][prev + 1] = Math.max(taken, nottaken);
            }
        }
        return dp[0][0];
    }
}
