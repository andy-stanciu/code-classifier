class Solution {

    public int lengthOfLIS(int[] nums) {
        // this will store the length of LIS for every single element in the array
        int[] dp = new int[nums.length];
        Arrays.fill(dp, 1);
        int result = 0;
        // check for every element in the given array
        for (int i = nums.length - 1; i >= 0; i--) {
            // check for every possible combination from current to end of array
            for (int j = i + 1; j < nums.length; j++) {
                // check if next element is greater than current
                if (nums[i] < nums[j]) {
                    // take the maximum length b/w current and next
                    dp[i] = Math.max(dp[i], 1 + dp[j]);
                }
            }
            // store the max length
            result = Math.max(result, dp[i]);
        }
        return result;
    }
}
