class Solution {

    public int lengthOfLIS(int[] nums) {
        return lengthOfLIS(nums, 0, Integer.MIN_VALUE, 0);
    }

    public int lengthOfLIS(int[] nums, int i, int prev, int curr_len) {
        if (i == nums.length) {
            return curr_len;
        }
        int take = 0;
        if (nums[i] > prev)
            take = lengthOfLIS(nums, i + 1, nums[i], curr_len + 1);
        int not_take = lengthOfLIS(nums, i + 1, prev, curr_len);
        return Math.max(take, not_take);
    }
}
