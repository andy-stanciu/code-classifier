class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] var1 = new int[nums.length];
        int ans = 1;
        var1[0] = 1;
        for (int i = 1; i < nums.length; i++) {
            int max = 0;
            for (int j = 0; j < i; j++) {
                if (var1[j] > max && nums[j] < nums[i])
                    max = var1[j];
            }
            var1[i] = max + 1;
            ans = Math.max(var1[i], ans);
        }
        return ans;
    }
}
