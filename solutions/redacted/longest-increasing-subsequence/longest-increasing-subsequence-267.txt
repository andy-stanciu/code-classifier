class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length, maxValue = 1;
        int[] dp = new int[n];
        for (int i = 0; i < n; i++) {
            dp[i] = 1;
            for (int prevInd = 0; prevInd < i; prevInd++) {
                if (nums[prevInd] < nums[i]) {
                    dp[i] = Math.max(dp[i], dp[prevInd] + 1);
                    maxValue = Math.max(maxValue, dp[i]);
                }
            }
        }
        return maxValue;
    }
}
