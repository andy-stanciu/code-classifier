class Solution {

    public int lengthOfLIS(int[] nums) {
        // f[i] = max(f[j]) j in [0, i]
        int n = nums.length;
        int[] f = new int[n];
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                if (nums[j] < nums[i])
                    f[i] = Math.max(f[i], f[j]);
            }
            f[i] += 1;
            ans = Math.max(ans, f[i]);
        }
        return ans;
    }
}
