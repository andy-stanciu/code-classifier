class Solution {

    public int solveLIS(int[] LIS, int[] nums, int i) {
        if (i == 0) {
            return 1;
        }
        int maxLIS = 0;
        for (int j = 0; j < i; j++) {
            if (nums[j] < nums[i]) {
                maxLIS = Math.max(maxLIS, LIS[j]);
            }
        }
        return 1 + maxLIS;
    }

    public int lengthOfLIS(int[] nums) {
        int ans = 1;
        int[] LIS = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            LIS[i] = solveLIS(LIS, nums, i);
            ans = Math.max(ans, LIS[i]);
        }
        return ans;
    }
}
