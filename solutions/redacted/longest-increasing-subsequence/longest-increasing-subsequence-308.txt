class Solution {

    public int lengthOfLIS(int[] nums) {
        int arrSize = nums.length;
        if (arrSize == 1)
            return 1;
        int[] lis = new int[arrSize];
        lis[arrSize - 1] = 1;
        int max = 1;
        for (int i = arrSize - 2; i >= 0; i--) {
            lis[i] = 1 + getMax(nums, lis, i + 1, nums[i]);
            max = Math.max(max, lis[i]);
        }
        return max;
    }

    public int getMax(int[] nums, int[] lis, int startIndex, int element) {
        int max = 0;
        for (int j = startIndex; j < nums.length; j++) {
            if (element < nums[j] && lis[j] > max)
                max = lis[j];
        }
        return max;
    }
}
