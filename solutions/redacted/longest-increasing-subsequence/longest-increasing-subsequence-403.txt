class Solution {

    public int lengthOfLIS(int[] nums) {
        // using binary search
        int[] dp = new int[nums.length];
        int size = 0;
        for (int current : nums) {
            int i = 0;
            int j = size;
            // we loop through the current increasing array dp to see
            // where shall we put the current element, if it needs to be
            // place at the end of the array, we will increase the size by 1
            while (i != j) {
                int mid = (i + j) / 2;
                if (dp[mid] < current) {
                    i = mid + 1;
                } else {
                    j = mid;
                }
            }
            dp[i] = current;
            // in this case, it means the current number should be place at the end of the dp array
            // so we need to increase the size of the array by 1
            if (i == size) {
                size++;
            }
        }
        return size;
    }
}
