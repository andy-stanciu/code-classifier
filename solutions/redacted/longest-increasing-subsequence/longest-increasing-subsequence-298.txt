class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            // because every number of array represents increasing subsequence of 1.
            dp[i] = 1;
        }
        int ans = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[j] > nums[i]) {
                    if (dp[j] < dp[i] + 1)
                        dp[j] = dp[i] + 1;
                }
            }
        }
        for (int i = 0; i < dp.length; i++) {
            if (ans < dp[i])
                ans = dp[i];
        }
        return ans;
    }
}
