class Solution {

    public int lengthOfLIS(int[] nums) {
        int maxOrig = 0;
        for (int i = 1; i < nums.length; ++i) {
            if (nums[i] > nums[maxOrig]) {
                ++maxOrig;
                nums[maxOrig] = nums[i];
                continue;
            }
            if (nums[i] < nums[0]) {
                nums[0] = nums[i];
                continue;
            }
            for (int j = maxOrig - 1; j >= 0; --j) {
                if (nums[i] > nums[j] && nums[i] < nums[j + 1]) {
                    nums[j + 1] = nums[i];
                    continue;
                }
            }
            // System.out.println(nums[i]);
        }
        return maxOrig + 1;
    }
}
