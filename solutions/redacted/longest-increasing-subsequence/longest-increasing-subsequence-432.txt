class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] arr = new int[nums.length];
        Arrays.fill(arr, 1);
        for (int movable1 = 1; movable1 < nums.length; movable1++) {
            for (int iterate = 0; iterate < movable1; iterate++) {
                if (nums[iterate] < nums[movable1])
                    arr[movable1] = Math.max(arr[movable1], arr[iterate] + 1);
            }
        }
        Arrays.sort(arr);
        return arr[nums.length - 1];
    }
}
