class Solution {

    public int lengthOfLIS(int[] a) {
        int n = a.length;
        int[] dp = new int[n];
        int max = 0;
        for (int num : a) {
            int i = Arrays.binarySearch(dp, 0, max, num);
            if (i < 0) {
                i = -(i + 1);
            }
            dp[i] = num;
            if (i == max)
                max++;
        }
        return max;
    }
}
