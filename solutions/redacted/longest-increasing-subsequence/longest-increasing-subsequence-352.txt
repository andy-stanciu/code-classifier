class Solution {

    public int lengthOfLIS(int[] nums) {
        List<Integer> arr = new ArrayList<>();
        arr.add(nums[0]);
        int len = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] > arr.get(arr.size() - 1)) {
                arr.add(nums[i]);
                len++;
            } else {
                int ind = Collections.binarySearch(arr, nums[i]);
                if (ind < 0) {
                    ind = -(ind + 1);
                }
                arr.set(ind, nums[i]);
            }
        }
        return len;
    }
}
