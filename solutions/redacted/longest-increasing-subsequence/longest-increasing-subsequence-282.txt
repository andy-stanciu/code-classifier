class Solution {

    public int lengthOfLIS(int[] nums) {
        int maxLen = 0;
        int[] dp = new int[nums.length + 1];
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i] && dp[i] < (dp[j] + 1)) {
                    dp[i] = dp[j] + 1;
                }
            }
            maxLen = Math.max(dp[i], maxLen);
        }
        return maxLen + 1;
    }
}
