class Solution {

    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        // Length of longest increasing subsequence
        int len = 0;
        for (int num : nums) {
            int idx = Arrays.binarySearch(dp, 0, len, num);
            if (idx < 0) {
                idx = -(idx + 1);
            }
            dp[idx] = num;
            if (idx == len) {
                len++;
            }
        }
        return len;
    }
}
