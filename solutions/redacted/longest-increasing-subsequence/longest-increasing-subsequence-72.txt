class Solution {

    public int lengthOfLIS(int[] nums) {
        var set = new TreeSet<Integer>();
        for (var i = 0; i < nums.length; i++) {
            var currElem = set.ceiling(nums[i]);
            if (currElem != null) {
                set.remove(currElem);
            }
            set.add(nums[i]);
        }
        return set.size();
    }
}
