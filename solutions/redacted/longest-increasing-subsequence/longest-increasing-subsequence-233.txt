class Solution {

    public int lengthOfLIS(int[] nums) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int num : nums) {
            Integer ceilingKey = map.ceilingKey(num);
            if (ceilingKey != null) {
                map.remove(ceilingKey);
            }
            map.put(num, 0);
        }
        return map.size();
    }
}
