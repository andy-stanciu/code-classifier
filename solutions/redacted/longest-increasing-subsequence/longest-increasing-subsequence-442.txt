class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] temp = new int[nums.length];
        int len = 0;
        for (int num : nums) {
            int i = Arrays.binarySearch(temp, 0, len, num);
            if (i < 0) {
                i = -(i + 1);
            }
            temp[i] = num;
            if (i == len)
                len++;
        }
        return len;
    }
}
