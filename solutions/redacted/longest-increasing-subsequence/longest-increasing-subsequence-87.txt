class Solution {

    public int lengthOfLIS(int[] nums) {
        int[] length = new int[nums.length];
        for (int i = 0; i < nums.length; i++) {
            length[i] = 1;
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i])
                    length[i] = Math.max(length[i], length[j] + 1);
            }
        }
        int max = length[0];
        for (int s : length) {
            if (s > max)
                max = s;
        }
        return max;
    }
}
