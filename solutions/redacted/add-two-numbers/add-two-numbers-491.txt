class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        // Correct return
        if (l1 == null && l2 == null)
            return new ListNode(0);
        ListNode result = null, current = null;
        ListNode temp1 = l1;
        ListNode temp2 = l2;
        int carry = 0;
        int var1 = 0;
        while (temp1 != null || temp2 != null) {
            // Handle case when temp1 is null
            int val1 = (temp1 != null) ? temp1.val : 0;
            // Handle case when temp2 is null
            int val2 = (temp2 != null) ? temp2.val : 0;
            var1 = val1 + val2 + carry;
            if (var1 >= 10) {
                // Carry condition fixed
                carry = 1;
                // Set the actual digit value
                var1 = var1 % 10;
            } else {
                carry = 0;
            }
            // Just a NODE CREATION which stores the value
            ListNode newNode = new ListNode(var1);
            if (result == null) {
                result = newNode;
                current = result;
            } else {
                // Add the newNode to the list that we have as answer(result list)
                current.next = newNode;
                current = current.next;
            }
            // traversing in the l1
            if (temp1 != null)
                temp1 = temp1.next;
            // traversing in the l2
            if (temp2 != null)
                temp2 = temp2.next;
        }
        if (carry != 0) {
            // the case where we get a carry after all the numbers-thus, a new NODE for storing that leading 1
            current.next = new ListNode(carry);
        }
        return result;
    }
}
