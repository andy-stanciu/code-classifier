class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        int counter = 0;
        ListNode ret = new ListNode(0);
        ListNode head = ret;
        int var1 = 0;
        while (l1 != null || l2 != null) {
            var1 = counter;
            if (l1 != null) {
                var1 += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                var1 += l2.val;
                l2 = l2.next;
            }
            if (var1 >= 10) {
                ret.next = new ListNode(var1 % 10);
                ret = ret.next;
                counter = 1;
            } else {
                ret.next = new ListNode(var1);
                ret = ret.next;
                counter = 0;
            }
        }
        if (counter == 1) {
            ret.next = new ListNode(1);
            ret = ret.next;
        }
        return head.next;
    }
}
