class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        ListNode head = new ListNode();
        ListNode current = head;
        do {
            if (l1 != null) {
                current.val += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                current.val += l2.val;
                l2 = l2.next;
            }
            if (current.val > 9) {
                current.val -= 10;
                current.next = new ListNode(1);
            } else if (l1 != null || l2 != null) {
                current.next = new ListNode();
            }
            current = current.next;
        } while (current != null);
        return head;
    }
}
