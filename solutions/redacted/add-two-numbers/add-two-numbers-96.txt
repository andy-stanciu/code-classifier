class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        //final ListNode
        ListNode result = new ListNode(0);
        ListNode temp = result;
        //Becasue we want to find the sum of two ListNode, so
        int sum = 0;
        // if both of the lists are not null, i.e. it has definitely some elements
        while (l1 != null || l2 != null) {
            sum /= 10;
            // we have to check list1
            if (l1 != null) {
                // sum -> 0   0+9
                sum = sum + l1.val;
                l1 = l1.next;
            }
            // we have to check list2
            if (l2 != null) {
                // sum -> 0   0+9
                sum = sum + l2.val;
                l2 = l2.next;
            }
            temp.next = new ListNode(sum % 10);
            temp = temp.next;
        }
        if (sum / 10 == 1) {
            temp.next = new ListNode(1);
        }
        return result.val == 0 ? result.next : result;
    }
}
