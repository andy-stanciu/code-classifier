class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        // Head of the new linked list - this is the head of the resultant list
        ListNode head = null;
        // Reference of head which is null at this point..
        ListNode tmp = null;
        // we have to initialize carry c
        int c = 0;
        while (l1 != null || l2 != null) {
            // At the start of each iteration, we should add carry c from the last iteration
            int sum = c;
            // if the current node is null for one of the lists
            if (l1 != null) {
                sum += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                sum += l2.val;
                l2 = l2.next;
            }
            // we will add the total sum % 10 to the new node in the resultant list
            ListNode node = new ListNode(sum % 10);
            // Carry c to be added in the next iteration
            c = sum / 10;
            // If this is the first node or head
            if (tmp == null) {
                tmp = head = node;
            } else {
                //for other node..
                tmp.next = node;
                tmp = tmp.next;
            }
        }
        // If carry c left then we will create a new node and add it
        if (c > 0) {
            tmp.next = new ListNode(c);
        }
        return head;
    }
}
