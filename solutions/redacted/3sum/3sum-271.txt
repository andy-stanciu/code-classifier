class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        // Sort the array to enable the two-pointer technique and to handle duplicates easily
        Arrays.sort(nums);
        List<List<Integer>> result = new ArrayList<>();
        // Iterate through the array
        for (int i = 0; i < nums.length - 2; i++) {
            // Skip duplicate elements for the first element
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            // Two pointers: left starting at i + 1 and right starting at the end of the array
            int left = i + 1, right = nums.length - 1;
            // Use two-pointer technique to find valid triplets
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum < 0) {
                    // We need a larger sum, move the left pointer to the right
                    left++;
                } else if (sum > 0) {
                    // We need a smaller sum, move the right pointer to the left
                    right--;
                } else {
                    // We found a triplet that sums to 0
                    result.add(Arrays.asList(nums[i], nums[left], nums[right]));
                    // Skip duplicates for the left pointer
                    while (left < right && nums[left] == nums[left + 1]) {
                        left++;
                    }
                    // Skip duplicates for the right pointer
                    while (left < right && nums[right] == nums[right - 1]) {
                        right--;
                    }
                    // Move both pointers after finding a valid triplet
                    left++;
                    right--;
                }
            }
        }
        return result;
    }
}
