class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        ArrayList<List<Integer>> listArr = new ArrayList<>();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            if (i == 0 || (i > 0 && nums[i] != nums[i - 1])) {
                int low = i + 1;
                int end = nums.length - 1;
                int sum = 0 - nums[i];
                while (low < end) {
                    if (nums[low] + nums[end] == sum) {
                        listArr.add(Arrays.asList(nums[i], nums[low], nums[end]));
                        while (low < end && nums[low] == nums[low + 1]) low++;
                        while (low < end && nums[end] == nums[end - 1]) end--;
                        low++;
                        end--;
                    } else if (nums[low] + nums[end] > sum) {
                        end--;
                    } else {
                        low++;
                    }
                }
            }
        }
        return listArr;
    }
}
