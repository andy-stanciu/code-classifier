class Solution {

    public int method1(int[] nums) {
        int right_total = 0;
        int left_total = 0;
        for (int i = 0; i < nums.length; i++) {
            right_total += nums[i];
        }
        int ans = -1;
        for (int i = 0; i < nums.length; i++) {
            right_total -= nums[i];
            if (i > 0)
                left_total += nums[i - 1];
            if (left_total == right_total) {
                ans = i;
                break;
            }
        }
        return ans;
    }
}
