class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum + nums[i];
        }
        int left = 0;
        int right;
        for (int j = 0; j < nums.length; j++) {
            right = sum - left - nums[j];
            if (right == left)
                return j;
            left = left + nums[j];
        }
        return -1;
    }
}
