class Solution {

    public int method1(int[] nums) {
        int sumleft = 0;
        int sumright = Arrays.stream(nums).sum();
        int i = 0;
        System.out.print(sumright);
        for (i = 0; i < nums.length; i++) {
            sumright -= nums[i];
            if (sumleft == sumright) {
                return i;
            }
            sumleft += nums[i];
        }
        return -1;
    }
}
