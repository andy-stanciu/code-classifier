class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int[] psum = new int[n + 1];
        psum[0] = 0;
        for (int i = 0; i < n; i++) {
            psum[i + 1] = psum[i] + nums[i];
        }
        for (int i = 0; i < n; i++) {
            int leftsum = psum[i];
            int rightsum = psum[n] - psum[i + 1];
            if (leftsum == rightsum) {
                return i;
            }
        }
        return -1;
    }
}
