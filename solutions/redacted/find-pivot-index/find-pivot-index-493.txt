class Solution {

    public static int method1(int[] nums) {
        int leftSum = 0;
        int rightSum = 0;
        int i = 0;
        while (i < nums.length) {
            rightSum = rightSum + nums[i];
            i++;
        }
        i = 0;
        while (i < nums.length) {
            rightSum = rightSum - nums[i];
            if (leftSum == rightSum) {
                return i;
            }
            leftSum = leftSum + nums[i];
            i++;
        }
        return -1;
    }
}
