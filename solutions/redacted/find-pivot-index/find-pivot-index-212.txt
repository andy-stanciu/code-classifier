class Solution {

    public int method1(int[] nums) {
        int l = 0;
        int r = 0;
        for (int i : nums) {
            r += i;
        }
        for (int i = 0; i < nums.length; i++) {
            r -= nums[i];
            if (r == l) {
                return i;
            } else {
                l += nums[i];
            }
        }
        return -1;
    }
}
