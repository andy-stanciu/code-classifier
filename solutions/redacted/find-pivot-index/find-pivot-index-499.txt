class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i : nums) {
            sum += i;
        }
        int sum1 = 0;
        for (int i = 0; i < nums.length; i++) {
            sum1 = sum1 + nums[i];
            if (sum == sum1) {
                return i;
            }
            sum = sum - nums[i];
        }
        return -1;
    }
}
