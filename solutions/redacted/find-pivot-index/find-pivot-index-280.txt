class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int rightSum = 0;
        for (int element : nums) rightSum += element;
        for (int i = 0; i < nums.length; i++) {
            rightSum -= nums[i];
            if (rightSum == leftSum)
                return i;
            leftSum += nums[i];
        }
        return -1;
    }
}
