class Solution {

    public int method1(int[] nums) {
        // prefix sum
        int pre = 0;
        // total sum of elements
        int total = 0;
        for (int i = 0; i < nums.length; i++) {
            total += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (pre == total - nums[i] - pre) {
                return i;
            }
            pre += nums[i];
        }
        return -1;
    }
}
