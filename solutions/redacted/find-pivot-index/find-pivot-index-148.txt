class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        for (int num : nums) {
            // Calculate the total sum of the array
            totalSum += num;
        }
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            // If left sum equals the right sum, the current index is the pivot
            if (leftSum == totalSum - leftSum - nums[i]) {
                return i;
            }
            // Update the left sum for the next index
            leftSum += nums[i];
        }
        // If no pivot index found, return -1
        return -1;
    }
}
