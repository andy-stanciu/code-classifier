class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            int leftSum = sum;
            int rightSum = 0;
            for (int j = i + 1; j < nums.length; j++) {
                rightSum += nums[j];
            }
            if (leftSum == rightSum)
                return i;
            sum += nums[i];
        }
        return -1;
    }
}
