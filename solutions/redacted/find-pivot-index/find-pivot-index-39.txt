class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int totalSum = 0;
        for (int n : nums) {
            totalSum += n;
        }
        for (int method1 = 0; method1 < nums.length; method1++) {
            if (leftSum * 2 == totalSum - nums[method1]) {
                return method1;
            }
            leftSum += nums[method1];
        }
        return -1;
    }
}
