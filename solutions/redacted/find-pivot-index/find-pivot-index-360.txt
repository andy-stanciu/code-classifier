class Solution {

    public int[] getPrefixSum(int[] nums) {
        int[] prefixSum = new int[nums.length];
        prefixSum[0] = nums[0];
        for (int i = 1; i < nums.length; i += 1) {
            prefixSum[i] = nums[i] + prefixSum[i - 1];
        }
        return prefixSum;
    }

    public int[] getSuffixSum(int[] nums) {
        int[] suffixSum = new int[nums.length];
        suffixSum[suffixSum.length - 1] = nums[nums.length - 1];
        for (int i = (nums.length - 2); i >= 0; i -= 1) {
            suffixSum[i] = nums[i] + suffixSum[i + 1];
        }
        return suffixSum;
    }

    public int method1(int[] nums) {
        int var1 = -1;
        int[] prefixSum = getPrefixSum(nums);
        int[] suffixSum = getSuffixSum(nums);
        for (int idx = 0; idx < prefixSum.length; idx += 1) {
            if (prefixSum[idx] == suffixSum[idx]) {
                var1 = idx;
                break;
            }
        }
        return var1;
    }
}
