class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int a : nums) sum += a;
        int left_sum = 0;
        int remaining_sum = sum;
        for (int i = 0; i < nums.length; i++) {
            if (i == 0)
                left_sum = 0;
            else
                left_sum += nums[i - 1];
            remaining_sum = remaining_sum - nums[i];
            if (left_sum == remaining_sum)
                return i;
        }
        return -1;
    }
}
