class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int[] left_sum = new int[n];
        int[] right_sum = new int[n];
        int rs = 0;
        int ls = 0;
        for (int i = 1; i < n; i++) {
            ls += nums[i - 1];
            left_sum[i] = ls;
        }
        left_sum[0] = 0;
        for (int i = n - 2; i >= 0; i--) {
            rs += nums[i + 1];
            right_sum[i] = rs;
        }
        right_sum[n - 1] = 0;
        for (int i = 0; i < n; i++) {
            if (right_sum[i] == left_sum[i])
                return i;
        }
        return -1;
    }
}
