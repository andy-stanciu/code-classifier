class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        // Iterate over each index to check for the pivot index
        for (int i = 0; i < n; i++) {
            // Calculate leftSum as the sum of elements from 0 to i-1
            int leftSum = 0;
            for (int j = 0; j < i; j++) {
                leftSum += nums[j];
            }
            // Calculate rightSum as the sum of elements from i+1 to n-1
            int rightSum = 0;
            for (int j = i + 1; j < n; j++) {
                rightSum += nums[j];
            }
            // Check if the leftSum is equal to rightSum
            if (leftSum == rightSum) {
                return i;
            }
        }
        // If no pivot index is found, return -1
        return -1;
    }
}
