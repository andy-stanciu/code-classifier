class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        for (int i = 1; i < nums.length; i++) totalSum += nums[i];
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (sum == totalSum)
                return i;
            sum += nums[i];
            if (i != nums.length - 1)
                totalSum -= nums[i + 1];
        }
        return -1;
    }
}
