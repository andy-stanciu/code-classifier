class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        double leftsum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (leftsum == (sum - nums[i]) / 2D) {
                return i;
            }
            leftsum += nums[i];
        }
        return -1;
    }
}
