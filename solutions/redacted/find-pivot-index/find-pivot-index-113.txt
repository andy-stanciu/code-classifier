class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (// Calculating sum of all elements of arrray
        // Calculating sum of all elements of arrray
        int i : nums) sum += i;
        // Right sum is equal to sum of array
        int right_sum = sum;
        int left_sum = 0;
        for (int i = 0; i < nums.length; i++) {
            // Substracting the element from right_sum one by one
            right_sum -= nums[i];
            if (// Checking if left_sum is equal or not
            left_sum == right_sum)
                return i;
            left_sum += nums[i];
        }
        return -1;
    }
}
