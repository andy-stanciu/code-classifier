class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int suffixSum = 0;
        for (int i = n - 1; i >= 0; --i) {
            suffixSum += nums[i];
        }
        int prefixSum = 0;
        for (int i = 0; i < n; ++i) {
            prefixSum += nums[i];
            if (prefixSum == suffixSum) {
                return i;
            }
            suffixSum -= nums[i];
        }
        return -1;
    }
}
