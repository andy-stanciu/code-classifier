class Solution {

    public int method1(int[] nums) {
        int lSum = 0;
        int sum = 0;
        for (int x : nums) {
            sum += x;
        }
        for (int i = 0; i < nums.length; i++) {
            if (lSum == sum - lSum - nums[i]) {
                return i;
            } else {
                lSum += nums[i];
            }
        }
        return -1;
    }
}
