class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int sum = 0;
        for (int i : nums) sum += i;
        for (int curr = 0; curr < nums.length; curr++) {
            if (sum - nums[curr] == left)
                return curr;
            left += nums[curr];
            sum -= nums[curr];
        }
        return -1;
    }
}
