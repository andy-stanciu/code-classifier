class Solution {

    public int method1(int[] nums) {
        int ans = 0;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        for (int j = 0; j < nums.length; j++) {
            if (ans == (sum - nums[j] - ans))
                return j;
            ans += nums[j];
        }
        return -1;
    }
}
