class Solution {

    public int method1(int[] nums) {
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            res += nums[i];
        }
        int l = 0;
        for (int i = 0; i < nums.length; i++) {
            res -= nums[i];
            if (l == res) {
                return i;
            }
            l += nums[i];
        }
        return -1;
    }
}
