class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return 1;
        int totalsum = 0;
        for (int i = 0; i < nums.length; i++) {
            // calculating total sum of array
            totalsum += nums[i];
        }
        // initially left sum is 0
        int leftsum = 0;
        for (int i = 0; i < nums.length; i++) {
            // this will act as rightsum now
            totalsum = totalsum - nums[i];
            //comparing right sum with left sum
            if (leftsum == totalsum)
                return i;
            // moving forward and increasing left sum
            leftsum = leftsum + nums[i];
        }
        // no such index exists then
        return -1;
    }
}
