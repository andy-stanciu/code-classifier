class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int left = 0;
        int right = 0;
        for (int i = 0; i < nums.length; i++) {
            right = sum - (left + nums[i]);
            if (right == left) {
                return i;
            }
            left += nums[i];
        }
        return -1;
    }
}
