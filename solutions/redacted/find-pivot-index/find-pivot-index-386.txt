class Solution {

    public int method1(int[] nums) {
        int total = 0;
        for (int i : nums) total += i;
        int sumLeft = 0;
        for (int i = 0; i < nums.length; i++) {
            int sumRight = total - sumLeft - nums[i];
            if (sumRight == sumLeft)
                return i;
            sumLeft += nums[i];
        }
        return -1;
    }
}
