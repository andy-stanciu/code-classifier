class Solution {

    public int method1(int[] nums) {
        int totalarr = 0;
        for (int num : nums) {
            totalarr += num;
        }
        int leftsum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (leftsum == totalarr - nums[i])
                return i;
            leftsum += nums[i];
            totalarr -= nums[i];
        }
        return -1;
    }
}
