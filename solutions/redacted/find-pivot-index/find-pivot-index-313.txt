class Solution {

    public int method1(int[] nums) {
        int sumLeft = 0;
        int sumRight = 0;
        int totalSum = 0;
        for (int num : nums) {
            totalSum += num;
        }
        for (int i = 0; i < nums.length; i++) {
            sumRight = totalSum - sumLeft - nums[i];
            if (sumLeft == sumRight) {
                return i;
            }
            sumLeft += nums[i];
        }
        return -1;
    }
}
