class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        for (int i = 0; i < len; i++) {
            int sum1 = 0;
            int sum2 = 0;
            for (int k = 0; k < i; k++) {
                sum1 = sum1 + nums[k];
            }
            for (int k = i + 1; k < len; k++) {
                sum2 = sum2 + nums[k];
            }
            if (sum1 == sum2) {
                return i;
            }
        }
        return -1;
    }
}
