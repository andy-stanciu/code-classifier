class Solution {

    public int method1(int[] nums) {
        var sum = 0;
        for (var num : nums) {
            sum += num;
        }
        var prefixSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (prefixSum == sum - nums[i] - prefixSum) {
                return i;
            }
            prefixSum += nums[i];
        }
        return -1;
    }
}
