class Solution {

    public int method1(int[] nums) {
        int leftSum = 0, var1;
        int rightSum = 0;
        for (int i = nums.length - 1; i >= 0; i--) rightSum += nums[i];
        for (var1 = 0; var1 < nums.length; var1++) {
            if (var1 - 1 != -1)
                leftSum += nums[var1 - 1];
            rightSum -= nums[var1];
            if (leftSum == rightSum)
                return var1;
        }
        return -1;
    }
}
