class Solution {

    public int method1(int[] nums) {
        int totalRight = 0;
        int totalLeft = 0;
        for (int n : nums) {
            totalRight += n;
        }
        int i = 0;
        while (i < nums.length) {
            totalRight -= nums[i];
            if (totalRight == totalLeft) {
                return i;
            }
            totalLeft += nums[i];
            i++;
        }
        return -1;
    }
}
