class Solution {

    public int method1(int[] nums) {
        // Initialize two variables to keep track of the sum of elements on the left and right of the current index
        int lSum = 0, rSum = 0;
        // Calculate the sum of all elements in the array
        for (// Add each element to the right sum
        // Add each element to the right sum
        int x : nums) rSum += x;
        // Iterate through the array from the start
        for (int i = 0; i < nums.length; i++) {
            // Subtract the current element from the right sum
            rSum -= nums[i];
            // Check if the left sum is equal to the right sum
            if (lSum == rSum)
                // If they are equal, return the current index
                return i;
            // Add the current element to the left sum
            lSum += nums[i];
        }
        // If no index is found that satisfies the condition, return -1
        return -1;
    }
}
