class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int totalSum = 0;
        for (int n : nums) {
            totalSum += n;
        }
        int rightSum = totalSum - nums[0];
        if (leftSum == rightSum)
            return 0;
        for (int i = 1; i < nums.length; i++) {
            leftSum += nums[i - 1];
            rightSum -= nums[i];
            if (leftSum == rightSum)
                return i;
        }
        return -1;
    }
}
