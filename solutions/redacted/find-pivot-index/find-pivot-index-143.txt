class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (leftSum == sum - leftSum - nums[i]) {
                return i;
            } else // else if ( leftSum > sum -leftSum -nums[i]){
            //     System.out.println(i);
            //     return -1;
            // }
            {
                leftSum += nums[i];
            }
        }
        return -1;
    }
}
