class Solution {

    public int method1(int[] dp) {
        int n = dp.length;
        for (int i = 1; i < n; i++) {
            dp[i] += dp[i - 1];
        }
        if (dp[n - 1] - dp[0] == 0)
            return 0;
        for (int i = 1; i < n - 1; i++) {
            if (dp[i - 1] == dp[n - 1] - dp[i])
                return i;
        }
        if (dp[n - 2] == 0)
            return n - 1;
        return -1;
    }
}
