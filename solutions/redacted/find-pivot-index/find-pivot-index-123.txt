class Solution {

    public int method1(int[] nums) {
        int[] sumLeft = new int[nums.length];
        int previousSum = 0;
        for (int i = 0; i < nums.length; i++) {
            sumLeft[i] = previousSum + nums[i];
            previousSum = sumLeft[i];
        }
        for (int i = 0; i < nums.length; i++) {
            int difference = sumLeft[nums.length - 1] - sumLeft[i];
            if (i == 0 && difference == 0) {
                return 0;
            } else if (i > 0 && difference == sumLeft[i - 1]) {
                return i;
            }
        }
        return -1;
    }
}
