class Solution {

    public int method1(int[] nums) {
        int n = nums.length, leftSum = 0, rightSum = 0, sum = 0;
        for (int i = 0; i < n; i++) rightSum += nums[i];
        for (int i = 0; i < n; i++) {
            rightSum = rightSum - nums[i];
            if (leftSum == rightSum) {
                return i;
            }
            leftSum += nums[i];
        }
        return -1;
    }
}
