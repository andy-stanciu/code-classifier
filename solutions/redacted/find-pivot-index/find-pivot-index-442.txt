class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int totaltSum = 0;
        int rightSum = 0;
        for (int num : nums) {
            totaltSum += num;
        }
        for (int i = 0; i < nums.length; i++) {
            rightSum = totaltSum - leftSum - nums[i];
            if (leftSum == rightSum) {
                return i;
            }
            leftSum += nums[i];
        }
        return -1;
    }
}
