class Solution {

    public int method1(int[] nums) {
        int leftSum = 0, rightSum = 0;
        // rightSum at this point indicates total array sum
        for (int i = 0; i < nums.length; i++) {
            rightSum += nums[i];
        }
        // traverse array, at each index, rightSum is sum after that index and left sum is sum till before that index
        for (int i = 0; i < nums.length; i++) {
            // since we are at index i, element at i is removed from sum
            rightSum -= nums[i];
            if (leftSum == rightSum) {
                return i;
            }
            // for next index, leftSum must be added with current index element
            leftSum += nums[i];
        }
        return -1;
    }
}
