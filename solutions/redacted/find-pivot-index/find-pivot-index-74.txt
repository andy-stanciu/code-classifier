class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int[] to = new int[n];
        int[] back = new int[n];
        to[0] = nums[0];
        back[n - 1] = nums[n - 1];
        for (int i = 1; i < n; i++) {
            to[i] = to[i - 1] + nums[i];
        }
        for (int i = n - 2; i >= 0; i--) {
            back[i] = back[i + 1] + nums[i];
        }
        for (int i = 0; i < n; i++) {
            if (to[i] == back[i]) {
                return i;
            }
        }
        return -1;
    }
}
