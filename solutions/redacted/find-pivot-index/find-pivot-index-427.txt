class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return 0;
        int sum = 0;
        for (int i = 0; i < nums.length; i++) sum += nums[i];
        int left = 0;
        for (int j = 0; j < nums.length; j++) {
            if (j > 0)
                left += nums[j - 1];
            if (left == sum - left - nums[j])
                return j;
        }
        return -1;
    }
}
