class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 1) {
            return 0;
        }
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum = sum + nums[i];
        }
        int lsum = 0;
        for (int i = 0; i < n; i++) {
            sum = sum - nums[i];
            if (lsum == sum) {
                return i;
            }
            lsum = lsum + nums[i];
        }
        return -1;
    }
}
