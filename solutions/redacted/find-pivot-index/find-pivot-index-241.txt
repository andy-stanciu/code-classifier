class Solution {

    public int method1(int[] nums) {
        int total = Arrays.stream(nums).reduce(0, Integer::sum);
        int leftTotal = 0;
        for (int i = 0; i < nums.length; i++) {
            total -= nums[i];
            if (leftTotal == total) {
                return i;
            }
            leftTotal += nums[i];
        }
        return -1;
    }
}
