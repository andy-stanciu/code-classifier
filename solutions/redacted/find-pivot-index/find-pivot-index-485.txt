class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int num : nums) sum = sum + num;
        int leftsum = 0;
        if (sum - nums[0] == 0)
            return 0;
        for (int i = 1; i < nums.length; i++) {
            leftsum = leftsum + nums[i - 1];
            System.out.println(leftsum + " " + sum);
            if (leftsum == (sum - leftsum - nums[i])) {
                return i;
            }
        }
        return -1;
    }
}
