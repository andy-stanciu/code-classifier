class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        for (int e : nums) {
            totalSum += e;
        }
        int prefixSum = 0;
        for (int i = 0; i < nums.length; i++) {
            prefixSum += nums[i];
            if (prefixSum - nums[i] == totalSum - prefixSum)
                return i;
        }
        return -1;
    }
}
