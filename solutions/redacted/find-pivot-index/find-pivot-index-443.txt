class Solution {

    public int method1(int[] nums) {
        int leftSum = 0, rightSum = 0;
        for (int i = 0; i < nums.length; i++) {
            rightSum += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (i == 0) {
                rightSum -= nums[i];
            } else {
                rightSum -= nums[i];
                leftSum += nums[i - 1];
            }
            if (rightSum == leftSum) {
                return i;
            }
        }
        return -1;
    }
}
