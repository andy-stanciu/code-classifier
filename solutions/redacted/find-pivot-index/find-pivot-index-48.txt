class Solution {

    public int method1(int[] nums) {
        int total = 0, left = 0;
        for (int i : nums) {
            total += i;
        }
        for (int i = 0; i < nums.length; i++) {
            if (left == total - left - nums[i])
                return i;
            left += nums[i];
        }
        return -1;
    }
}
