class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int rightSum = getTotalSum(nums);
        for (int i = 0; i < nums.length; i++) {
            rightSum = rightSum - nums[i];
            if (leftSum == rightSum) {
                return i;
            } else {
                leftSum = leftSum + nums[i];
            }
        }
        return -1;
    }

    private int getTotalSum(int[] nums) {
        int totalSum = 0;
        for (int i : nums) {
            totalSum += i;
        }
        return totalSum;
    }
}
