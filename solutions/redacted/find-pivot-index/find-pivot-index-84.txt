class Solution {

    public int method1(int[] nums) {
        int total = 0;
        int left = 0;
        for (int i : nums) {
            total += i;
        }
        for (int i = 0; i < nums.length; i++) {
            if (i > 0) {
                // here we add if because for index 0 there is no left elem
                left += nums[i - 1];
            }
            if (total - left - nums[i] == left) {
                return i;
            }
        }
        return -1;
    }
}
