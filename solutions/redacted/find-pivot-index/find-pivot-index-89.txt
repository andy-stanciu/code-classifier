class Solution {

    public int method1(int[] nums) {
        if (nums.length < 1)
            return -1;
        int rsum = 0, lsum = 0;
        for (//calculating sum of the array
        //calculating sum of the array
        int x : nums) rsum += x;
        for (int x = 0; x < nums.length; x++) {
            rsum -= nums[x];
            if (rsum == lsum)
                return x;
            lsum += nums[x];
        }
        return -1;
    }
}
