class Solution {

    public int method1(int[] nums) {
        int left = 0, sum = 0;
        for (int i = 0; i < nums.length; i++) {
            //summation of all the elements in nums array
            sum = sum + nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            //decreasing sum value
            sum = sum - nums[i];
            if (sum == left) {
                // we check if sum==left, if equal return that index
                return i;
            }
            //increasing left value
            left = left + nums[i];
        }
        return -1;
    }
}
/*

sum=sum+nums[i] =28
left = 0
last for loop will work like this : 
for 1st iteration : sum=27 (28-1)
                    sum!=left(28!=0)
                       left=0+1
after 2 iterations, sum=11 and left=11
therefore ,         returning index of 6 i.e. 3 which is pivot index.
