class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        for (int i : nums) {
            sum += i;
        }
        int left = 0;
        for (int i = 0; i < n; i++) {
            left += nums[i];
            if (left == sum) {
                return i;
            }
            sum -= nums[i];
        }
        return -1;
    }
}
