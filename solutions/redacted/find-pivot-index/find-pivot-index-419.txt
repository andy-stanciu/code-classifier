class Solution {

    public int method1(int[] nums) {
        //TC: O(N)
        //SC: O(1)
        int totalSum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalSum += nums[i];
        }
        int leftSum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalSum -= nums[i];
            if (totalSum == leftSum)
                return i;
            else
                leftSum += nums[i];
        }
        return -1;
    }
}
