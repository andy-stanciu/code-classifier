class Solution {

    public int method1(int[] nums) {
        int leftSum = 0, var1 = 0, rightSum = 0;
        for (int i = 1; i < nums.length; i++) {
            rightSum += nums[i];
        }
        while ((var1 < nums.length - 1) && rightSum != leftSum) {
            var1++;
            rightSum -= nums[var1];
            leftSum += nums[var1 - 1];
        }
        return (leftSum == rightSum) ? var1 : -1;
    }
}
