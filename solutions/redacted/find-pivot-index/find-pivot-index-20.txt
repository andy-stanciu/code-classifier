class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        int leftSum = 0;
        for (int val : nums) {
            totalSum += val;
        }
        for (int i = 0; i < nums.length; ) {
            if (leftSum * 2 == totalSum - nums[i]) {
                //as we want left and right sum to be equal :-  rightsum = leftsum => 2*leftsum;  and we want
                //left sum not including the ith element value , and right sum without including the ith element value .
                //so we subtracting value og ith element from total sum;
                return i;
            }
            leftSum += nums[i++];
        }
        return -1;
    }
}
