class Solution {

    public int method1(int[] nums) {
        int ans = -1;
        int n = nums.length;
        // Conpute the prfix sum of all item
        // with padding left and right with 0
        // the padding is useful in the caase if of the pivot index at the edges
        int[] prefixSum = new int[n + 2];
        for (int i = 0; i < n; i++) {
            prefixSum[i + 1] = prefixSum[i] + nums[i];
        }
        // check if at index i to the prefix sum, left sum equal right sum
        // if the case return i-1 since i was shifted by one regarding to nums.
        for (int i = 1; i <= n; i++) {
            if (prefixSum[i - 1] == prefixSum[n] - prefixSum[i]) {
                return i - 1;
            }
        }
        return ans;
    }
}
