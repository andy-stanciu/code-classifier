class Solution {

    public int method1(int[] nums) {
        int right = 0;
        for (int i = 0; i < nums.length; i++) {
            right += nums[i];
        }
        int left = 0;
        for (int i = 0; i < nums.length; i++) {
            left += nums[i];
            if (left == right) {
                return i;
            }
            right -= nums[i];
        }
        return -1;
    }
}
