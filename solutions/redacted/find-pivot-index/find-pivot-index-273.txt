class Solution {

    public int method1(int[] nums) {
        int RightSum = 0;
        for (int num : nums) RightSum += num;
        int LeftSum = 0;
        int size = nums.length;
        for (int i = 0; i < size; i++) {
            RightSum -= nums[i];
            if (LeftSum == RightSum)
                return i;
            LeftSum += nums[i];
        }
        return -1;
    }
}
