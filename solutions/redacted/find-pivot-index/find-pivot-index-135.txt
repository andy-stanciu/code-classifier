class Solution {

    public int method1(int[] nums) {
        int totalSum = 0;
        for (int i = 0; i < nums.length; i++) {
            totalSum += nums[i];
        }
        int currSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (currSum == totalSum - (currSum + nums[i])) {
                return i;
            }
            currSum += nums[i];
        }
        return -1;
    }
}
