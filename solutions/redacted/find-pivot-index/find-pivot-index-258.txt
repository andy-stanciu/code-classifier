class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return 0;
        }
        for (int i = nums.length - 1; i >= 0; i--) {
            if (i != nums.length - 1) {
                nums[i] = nums[i] + nums[i + 1];
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (i == 0) {
                if (nums[i + 1] == 0) {
                    return i;
                }
            } else if (i == nums.length - 1) {
                if (nums[i - 1] == 0) {
                    return i;
                }
            } else {
                if (nums[i + 1] == nums[i - 1]) {
                    return i;
                }
            }
            if (i != nums.length - 1) {
                nums[i] = nums[i] - nums[i + 1];
            }
            if (i != 0) {
                nums[i] += nums[i - 1];
            }
        }
        return -1;
    }
}
