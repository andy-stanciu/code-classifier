class Solution {

    public int method1(int[] nums) {
        int leftSum = 0;
        int i = 0;
        int rightSum = 0;
        rightSum = this.returnRightSum(nums, i, rightSum);
        return method2(leftSum, rightSum, nums, i);
    }

    public int returnRightSum(int[] nums, int i, int rightSum) {
        // base case
        if (i >= nums.length) {
            return rightSum;
        }
        // recursive case
        return returnRightSum(nums, i + 1, rightSum + nums[i]);
    }

    public int method2(int leftSum, int rightSum, int[] nums, int i) {
        // base case
        if (i >= nums.length) {
            return -1;
        }
        rightSum = rightSum - nums[i];
        if (leftSum == rightSum) {
            return i;
        }
        leftSum = leftSum + nums[i];
        // recursive call
        return method2(leftSum, rightSum, nums, i + 1);
    }
}
