class Solution {

    public ListNode method1(ListNode[] param1) {
        var head = new ListNode();
        var ptr = head;
        var queue = new PriorityQueue<int[]>((a, b) -> Integer.compare(a[0], b[0]));
        for (var i = 0; i < param1.length; i++) if (param1[i] != null)
            queue.offer(new int[] { param1[i].val, i });
        while (!queue.isEmpty()) {
            var a = queue.poll();
            var val = a[0];
            var idx = a[1];
            ptr.next = new ListNode(val);
            ptr = ptr.next;
            param1[idx] = param1[idx].next;
            if (param1[idx] != null)
                queue.offer(new int[] { param1[idx].val, idx });
        }
        return head.next;
    }
}
