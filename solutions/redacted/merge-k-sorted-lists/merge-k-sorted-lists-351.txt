class Solution {

    public ListNode method1(ListNode[] param1) {
        PriorityQueue<Integer> mnheap = new PriorityQueue<>();
        for (int i = 0; i < param1.length; i++) {
            ListNode temp = param1[i];
            while (temp != null) {
                mnheap.add(temp.val);
                temp = temp.next;
            }
        }
        ListNode res = new ListNode(-1);
        ListNode rhead = res;
        while (!mnheap.isEmpty()) {
            ListNode node = new ListNode(mnheap.poll());
            rhead.next = node;
            rhead = rhead.next;
        }
        return res.next;
    }
}
