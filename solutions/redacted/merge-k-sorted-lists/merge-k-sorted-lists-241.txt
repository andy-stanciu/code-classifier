class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode dummy = new ListNode(0);
        ListNode previous = dummy;
        while (true) {
            ListNode current = null;
            Integer selectedI = null;
            for (int i = 0; i < param1.length; i++) {
                ListNode list = param1[i];
                // if this list is empty
                if (list == null)
                    continue;
                // is the current val a good match?
                if (current == null || current.val > list.val) {
                    current = list;
                    selectedI = i;
                }
            }
            // no selection == end of all lists
            if (selectedI == null) {
                break;
            }
            // update previous
            previous.next = current;
            // move forward
            previous = previous.next;
            param1[selectedI] = param1[selectedI].next;
        }
        return dummy.next;
    }
}
