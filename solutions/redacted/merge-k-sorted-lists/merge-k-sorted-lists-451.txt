/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1 == null || param1.length == 0)
            return null;
        ListNode head = new ListNode();
        ListNode tail = head;
        Integer min = null;
        while ((min = getMinIndex(param1)) != null) {
            tail.next = param1[min];
            tail = tail.next;
            param1[min] = param1[min].next;
        }
        return head.next;
    }

    //Time Complexity of getMinIndex O(K)
    public Integer getMinIndex(ListNode[] param1) {
        Integer minIndex = null;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i] != null && (minIndex == null || param1[i].val < param1[minIndex].val)) {
                minIndex = i;
            }
        }
        return minIndex;
    }
}
