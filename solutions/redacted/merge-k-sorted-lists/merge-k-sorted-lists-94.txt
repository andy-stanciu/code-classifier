class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0)
            return null;
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (ListNode ls : param1) {
            while (ls != null) {
                pq.add(ls.val);
                ls = ls.next;
            }
        }
        ListNode head;
        if (!pq.isEmpty())
            head = new ListNode(pq.poll());
        else
            return null;
        var temp = head;
        while (!pq.isEmpty()) {
            temp.next = new ListNode(pq.poll());
            temp = temp.next;
        }
        temp.next = null;
        return head;
    }
}
