class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0 || param1 == null)
            return null;
        /* Just to simplfy returning refrence */
        ListNode dummyHead = new ListNode(0);
        ListNode temp = dummyHead;
        Queue<ListNode> heap = new PriorityQueue<>((a, b) -> (a.val - b.val));
        for (ListNode list : param1) {
            if (list != null)
                heap.add(list);
        }
        while (!heap.isEmpty()) {
            /* taking out heap\'s top element and adding to Linked List(ListNode) */
            ListNode top = heap.poll();
            temp.next = top;
            temp = temp.next;
            /* if the linked list whose element was polled out has any other element left add that to heap */
            if (top.next != null)
                heap.add(top.next);
        }
        return dummyHead.next;
    }
}
