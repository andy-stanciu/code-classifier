public class Solution {

    public ListNode method1(ListNode[] param1) {
        // List to store all values from ListNode arrays
        List<Integer> values = new ArrayList<>();
        // Extract values from each ListNode and add them to the values list
        for (ListNode node : param1) {
            while (node != null) {
                values.add(node.val);
                node = node.next;
            }
        }
        // Sort the list of values
        Collections.sort(values);
        // Create a dummy ListNode to build the merged list
        ListNode dummy = new ListNode(-1);
        ListNode current = dummy;
        // Create ListNode objects with sorted values and link them together
        for (int val : values) {
            current.next = new ListNode(val);
            current = current.next;
        }
        // Return the head of the merged list (excluding the dummy node)
        return dummy.next;
    }
}
