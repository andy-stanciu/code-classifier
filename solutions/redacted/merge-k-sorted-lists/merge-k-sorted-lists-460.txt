class Solution {

    public ListNode method1(ListNode[] param1) {
        ListNode head = null, curr = null, out = null, tail = null;
        if (param1.length == 1)
            return param1[0];
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode listNode : param1) {
            while (null != listNode) {
                pq.offer(listNode);
                listNode = listNode.next;
            }
        }
        while (!pq.isEmpty()) {
            if (null == out) {
                out = pq.poll();
                tail = out;
            } else {
                tail.next = pq.poll();
                tail = tail.next;
                tail.next = null;
            }
        }
        return out;
    }
}
