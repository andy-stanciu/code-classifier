class Solution {

    public ListNode method1(ListNode l1, ListNode l2) {
        if (l1 == null)
            return l2;
        if (l2 == null)
            return l1;
        ListNode t;
        if (l1.val < l2.val) {
            t = l1;
            l1 = l1.next;
        } else {
            t = l2;
            l2 = l2.next;
        }
        ListNode head = t;
        while (l1 != null && l2 != null) {
            if (l1.val <= l2.val) {
                t.next = l1;
                l1 = l1.next;
            } else {
                t.next = l2;
                l2 = l2.next;
            }
            t = t.next;
        }
        if (l1 == null)
            t.next = l2;
        else
            t.next = l1;
        return head;
    }

    public ListNode method2(ListNode[] param1) {
        ListNode head = null;
        for (int i = 0; i < param1.length; i++) {
            head = method1(head, param1[i]);
        }
        return head;
    }
}
