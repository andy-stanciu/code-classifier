class Solution {

    public ListNode method1(ListNode[] param1) {
        if (param1.length == 0 || param1 == null) {
            return null;
        }
        if (param1.length == 1) {
            return param1[0];
        }
        ListNode answer = null;
        for (ListNode list : param1) {
            answer = method2(answer, list);
        }
        return answer;
    }

    private ListNode method2(ListNode list1, ListNode list2) {
        ListNode thing = new ListNode(0);
        ListNode curr = thing;
        while (list1 != null && list2 != null) {
            if (list1.val < list2.val) {
                curr.next = list1;
                list1 = list1.next;
                curr = curr.next;
            } else {
                curr.next = list2;
                list2 = list2.next;
                curr = curr.next;
            }
        }
        if (list1 != null) {
            curr.next = list1;
        }
        if (list2 != null) {
            curr.next = list2;
        }
        return thing.next;
    }
}
