class Solution {

    public List<Integer> method1(TreeNode root) {
        // create arraylist (returntype)
        List<Integer> list = new ArrayList();
        // call preorder funtn
        method2(root, list);
        // return arraylist
        return list;
    }

    public void method2(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        // step 1: store value
        list.add(root.val);
        // step 2: call left subtree
        method2(root.left, list);
        // step 3: call right subtree
        method2(root.right, list);
    }
}
