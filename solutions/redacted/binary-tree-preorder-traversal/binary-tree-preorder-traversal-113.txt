class Solution {

    ArrayList<Integer> pre = new ArrayList<>();

    public void method1(TreeNode node) {
        if (node == null) {
            return;
        }
        pre.add(node.val);
        method1(node.left);
        method1(node.right);
        return;
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return pre;
    }
}
