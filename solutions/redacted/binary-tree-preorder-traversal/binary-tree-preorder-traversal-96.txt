class Solution {

    void method1(TreeNode root, List<Integer> answer) {
        if (root == null)
            return;
        answer.add(root.val);
        if (root.left != null)
            method1(root.left, answer);
        if (root.right != null)
            method1(root.right, answer);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> answer = new ArrayList<>();
        method1(root, answer);
        return answer;
    }
}
