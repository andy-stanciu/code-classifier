/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void solve(List<Integer> arr, TreeNode root) {
        if (root == null)
            return;
        arr.add(root.val);
        solve(arr, root.left);
        solve(arr, root.right);
    }

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        if (root == null)
            return arr;
        solve(arr, root);
        return arr;
    }
}
