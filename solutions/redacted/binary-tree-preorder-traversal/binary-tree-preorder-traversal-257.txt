/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if (root == null) {
            return list;
        }
        TreeNode curr = root;
        while (curr != null) {
            if (curr.left == null) {
                list.add(curr.val);
                curr = curr.right;
            } else {
                TreeNode currLeft = curr.left;
                while (currLeft.right != null && currLeft.right != curr) {
                    currLeft = currLeft.right;
                }
                if (currLeft.right == null) {
                    list.add(curr.val);
                    currLeft.right = curr;
                    curr = curr.left;
                }
                if (currLeft.right == curr) {
                    currLeft.right = null;
                    curr = curr.right;
                }
            }
        }
        return list;
    }
}
