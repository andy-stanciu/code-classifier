class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        traverse(root, list);
        return list;
    }

    public void traverse(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        list.add(node.val);
        traverse(node.left, list);
        traverse(node.right, list);
    }
}
