class Solution {

    //to store ans in this list
    List<Integer> ans = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        //return the list
        if (root == null)
            return ans;
        //add val to the ans List
        ans.add(root.val);
        //left call by recursion
        method1(root.left);
        // right call by recurtion
        method1(root.right);
        //rsturn the stored list of ans
        return ans;
    }
}
