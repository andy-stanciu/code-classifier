/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list1 = new ArrayList<>();
        if (root == null) {
            return list1;
        }
        return method2(root, list1);
    }

    public List<Integer> method2(TreeNode root, ArrayList<Integer> list1) {
        if (root == null) {
            return null;
        }
        list1.add(root.val);
        method2(root.left, list1);
        method2(root.right, list1);
        return list1;
    }
}
