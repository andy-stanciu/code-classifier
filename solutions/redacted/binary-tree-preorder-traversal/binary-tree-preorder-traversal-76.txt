class Solution {

    public void method1(TreeNode node, List<Integer> pre) {
        if (node == null)
            return;
        pre.add(node.val);
        method1(node.left, pre);
        method1(node.right, pre);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        method1(root, pre);
        return pre;
    }
}
