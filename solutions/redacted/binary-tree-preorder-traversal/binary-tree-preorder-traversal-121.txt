/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private List<Integer> res = new ArrayList();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return res;
        res.add(root.val);
        if (root.left != null)
            method1(root.left);
        if (root.right != null)
            method1(root.right);
        return res;
    }
}
