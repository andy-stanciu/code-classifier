/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return var1;
        Stack<TreeNode> sta = new Stack<TreeNode>();
        sta.push(root);
        while (!sta.isEmpty()) {
            root = sta.pop();
            var1.add(root.val);
            if (root.right != null)
                sta.push(root.right);
            if (root.left != null)
                sta.push(root.left);
        }
        return var1;
    }
}
