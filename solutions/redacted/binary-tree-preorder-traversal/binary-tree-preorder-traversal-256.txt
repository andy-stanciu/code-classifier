class Solution {

    List<Integer> list;

    public List<Integer> method1(TreeNode root) {
        list = new ArrayList<Integer>();
        solver(root);
        return list;
    }

    public void solver(TreeNode root) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left != null)
            solver(root.left);
        if (root.right != null)
            solver(root.right);
        return;
    }
}
