class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    void method2(TreeNode node, ArrayList<Integer> ls) {
        if (node == null)
            return;
        ls.add(node.val);
        method2(node.left, ls);
        method2(node.right, ls);
    }
}
