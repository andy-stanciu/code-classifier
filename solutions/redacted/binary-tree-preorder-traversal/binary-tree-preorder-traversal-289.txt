/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void method1(TreeNode node, List<Integer> method1) {
        if (node == null) {
            return;
        }
        method1.add(node.val);
        method1(node.left, method1);
        method1(node.right, method1);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> method1 = new ArrayList<>();
        if (root == null) {
            return method1;
        }
        method1(root, method1);
        return method1;
    }
}
