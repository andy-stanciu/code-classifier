/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new ArrayList<>();
        method2(output, root);
        return output;
    }

    private void method2(List<Integer> l, TreeNode root) {
        if (root == null)
            return;
        l.add(root.val);
        method2(l, root.left);
        method2(l, root.right);
    }
}
