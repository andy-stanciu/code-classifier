/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l1 = new ArrayList<Integer>();
        return method2(l1, root);
    }

    List<Integer> method2(List<Integer> l1, TreeNode root) {
        if (root != null) {
            l1.add(root.val);
            method2(l1, root.left);
            method2(l1, root.right);
        }
        return l1;
    }
}
