/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        if (root == null)
            return ls;
        while (!st.isEmpty() || root != null) {
            if (root != null) {
                ls.add(root.val);
                st.push(root);
                root = root.left;
            } else {
                root = st.pop();
                if (!st.isEmpty() && root.left == null && root.right == null) {
                    root = st.pop();
                }
                root = root.right;
            }
        }
        return ls;
    }
}
