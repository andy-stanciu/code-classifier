/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList();
        return get(root, list);
    }

    public List<Integer> get(TreeNode root, List<Integer> list) {
        if (root == null)
            return list;
        list.add(root.val);
        get(root.left, list);
        get(root.right, list);
        return list;
    }
}
