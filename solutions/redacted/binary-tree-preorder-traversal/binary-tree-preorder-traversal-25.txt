class Solution {

    public List<Integer> Answer = new ArrayList<Integer>();

    //Consider Global List For Answer
    public List<Integer> method1(TreeNode root) {
        if (//if root of tree is null than tree dont have any element so return empty list of Answer
        root == null)
            return Answer;
        //Travers Tree in  preorder(root,left sub-tree,right sub-tree)
        Travers(root);
        return Answer;
    }

    public void Travers(TreeNode root) {
        // first add value of rood into list of Answer
        Answer.add(root.val);
        if (//if root have Left sub-tree than Travers it in preorder
        root.left != null)
            Travers(root.left);
        if (//if root have Right sub-tree than Travers it in preorder
        root.right != null)
            Travers(root.right);
    }
}
