/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> l1 = new ArrayList<>();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        l1.add(root.val);
        method1(root.left);
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return l1;
    }
}
