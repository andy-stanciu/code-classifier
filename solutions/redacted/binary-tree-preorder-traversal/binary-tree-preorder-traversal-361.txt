/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new LinkedList<>();
        method2(root, var1);
        return var1;
    }

    private static void method2(TreeNode root, List<Integer> var1) {
        if (root == null) {
            return;
        }
        var1.add(root.val);
        method2(root.left, var1);
        method2(root.right, var1);
    }
}
