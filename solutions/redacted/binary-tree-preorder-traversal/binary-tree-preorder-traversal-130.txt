class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method2(root, list);
        return list;
    }

    private void method2(TreeNode root, List<Integer> list) {
        TreeNode temp = root;
        if (temp == null) {
            return;
        }
        list.add(temp.val);
        method2(temp.left, list);
        method2(temp.right, list);
    }
}
