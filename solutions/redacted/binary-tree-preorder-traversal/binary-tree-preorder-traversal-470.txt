/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> lst = new ArrayList<>();
        ans(lst, root);
        return lst;
    }

    static void ans(ArrayList<Integer> lst, TreeNode root) {
        if (root == null) {
            return;
        } else {
            lst.add(root.val);
            ans(lst, root.left);
            ans(lst, root.right);
        }
    }
}
