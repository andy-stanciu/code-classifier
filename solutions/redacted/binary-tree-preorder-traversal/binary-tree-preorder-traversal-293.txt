/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root, List<Integer> l) {
        if (root == null)
            return;
        l.add(root.val);
        method1(root.left, l);
        method1(root.right, l);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        method1(root, l);
        return l;
    }
}
