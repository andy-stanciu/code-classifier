/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> ans;

    public void method1(TreeNode root) {
        // if node is null then return
        if (root == null)
            return;
        //add root first
        ans.add(root.val);
        // call left child first
        method1(root.left);
        // then call right child
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        ans = new ArrayList<>();
        method1(root);
        return ans;
    }
}
