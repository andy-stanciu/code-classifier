/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> lists = new ArrayList<>();
        method2(lists, root);
        return lists;
    }

    public void method2(ArrayList lists, TreeNode root) {
        if (root == null) {
            return;
        } else {
            lists.add(root.val);
            method2(lists, root.left);
            method2(lists, root.right);
        }
    }
}
