class Solution {

    List<Integer> var1 = new ArrayList<Integer>();

    public static List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    private static void method2(TreeNode node, List<Integer> result) {
        if (node == null) {
            // Base case: If the node is null, do nothing
            return;
        }
        result.add(node.val);
        method2(node.left, result);
        method2(node.right, result);
    }
}
