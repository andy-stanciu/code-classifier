class Solution {

    List<Integer> var1 = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return var1;
    }

    // A recursive function to find the preorder list of the tree
    private void method2(TreeNode root) {
        if (root == null)
            return;
        var1.add(root.val);
        method2(root.left);
        method2(root.right);
    }
}
