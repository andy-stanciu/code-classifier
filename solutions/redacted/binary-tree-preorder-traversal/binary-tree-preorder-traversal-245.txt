class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> a = new ArrayList<>();
        List<Integer> ans = method2(root, a);
        return ans;
    }

    public List<Integer> method2(TreeNode node, List<Integer> a) {
        if (node == null)
            return a;
        a.add(node.val);
        method2(node.left, a);
        method2(node.right, a);
        return a;
    }
}
