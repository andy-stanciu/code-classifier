class Solution {

    public int method1(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        for (int r = 0; r < m; r++) {
            for (int c = 0; c < n; c++) {
                if (obstacleGrid[r][c] == 1) {
                    obstacleGrid[r][c] = 0;
                } else if (r == 0 && c == 0) {
                    obstacleGrid[r][c] = 1;
                } else {
                    obstacleGrid[r][c] = ((r == 0) ? 0 : obstacleGrid[r - 1][c]) + ((c == 0) ? 0 : obstacleGrid[r][c - 1]);
                }
            }
        }
        return obstacleGrid[m - 1][n - 1];
    }
}
