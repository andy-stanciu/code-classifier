class Solution {

    public int method1(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        //   int [][] dp= new int[m][n];
        if (grid[m - 1][n - 1] == 1)
            return 0;
        if (m == 1 && n == 1)
            return 1;
        for (int i = m - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                if (i == m - 1 && j == n - 1)
                    grid[i][j] = 1;
                else if (grid[i][j] == 1) {
                    grid[i][j] = 0;
                    continue;
                } else if (j == n - 1) {
                    grid[i][j] = grid[i + 1][j];
                } else if (i == m - 1) {
                    grid[i][j] = grid[i][j + 1];
                } else {
                    grid[i][j] = grid[i + 1][j] + grid[i][j + 1];
                }
            }
        }
        return grid[0][0];
    }
}
