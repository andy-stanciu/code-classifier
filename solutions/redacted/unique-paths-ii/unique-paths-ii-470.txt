class Solution {

    public int method1(int[][] g) {
        for (int i = 0; i < g.length; ++i) {
            for (int j = 0; j < g[0].length; ++j) {
                g[i][j] ^= 1;
            }
        }
        for (int i = 0; i < g.length; ++i) {
            for (int j = 0; j < g[0].length; ++j) {
                if (g[i][j] == 0 || (i == 0 && j == 0))
                    continue;
                else if (i == 0)
                    g[i][j] *= g[i][j - 1];
                else if (j == 0)
                    g[i][j] *= g[i - 1][j];
                else
                    g[i][j] = g[i - 1][j] + g[i][j - 1];
            }
        }
        return g[g.length - 1][g[0].length - 1];
    }
}
