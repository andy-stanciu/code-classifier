class Solution {

    public int method1(int[][] nums) {
        int r = nums.length;
        int c = nums[0].length;
        int[][] memo = new int[r][c];
        for (int[] i : memo) {
            Arrays.fill(i, -1);
        }
        return method2(nums, r - 1, c - 1, 0, 0, memo);
    }

    private int method2(int[][] nums, int row, int col, int r, int c, int[][] memo) {
        if (r > row || c > col || c < 0 || r < 0 || nums[r][c] == 1) {
            return 0;
        }
        if (r == row && c == col) {
            return 1;
        }
        if (memo[r][c] != -1)
            return memo[r][c];
        memo[r][c] = method2(nums, row, col, r, c + 1, memo) + method2(nums, row, col, r + 1, c, memo);
        return memo[r][c];
    }
}
