class Solution {

    public int method1(int[][] a) {
        int n = a.length;
        int m = a[0].length;
        int[][] dp = new int[n][m];
        return cal(n, m, a, dp);
    }

    public int cal(int n, int m, int[][] a, int[][] dp) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (a[i][j] == 0) {
                    if (i == 0 && j == 0)
                        a[i][j] = 1;
                    else {
                        if (i - 1 >= 0)
                            a[i][j] = a[i - 1][j];
                        if (j - 1 >= 0)
                            a[i][j] += a[i][j - 1];
                    }
                } else
                    a[i][j] = 0;
            }
        }
        return a[n - 1][m - 1];
    }
}
