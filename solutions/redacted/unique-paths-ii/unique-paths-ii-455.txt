class Solution {

    public int findPath(int i, int j, int[][] way, int[][] dp) {
        if ((i >= 0 && j >= 0) && way[i][j] == 1)
            return 0;
        if (i == 0 && j == 0)
            return 1;
        if (i < 0 || j < 0)
            return 0;
        if (dp[i][j] != -1)
            return dp[i][j];
        int up = 0, left = 0;
        if (i > 0) {
            if (way[i - 1][j] != 1)
                up = findPath(i - 1, j, way, dp);
        }
        if (j > 0) {
            if (way[i][j - 1] != 1)
                left = findPath(i, j - 1, way, dp);
        }
        dp[i][j] = up + left;
        return up + left;
    }

    public int method1(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        int[][] dp = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] = -1;
            }
        }
        return findPath(m - 1, n - 1, obstacleGrid, dp);
    }
}
