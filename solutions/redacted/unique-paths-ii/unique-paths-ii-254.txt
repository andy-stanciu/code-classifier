class Solution {

    private static int OBSTACLE = 1;

    public int method1(int[][] obstacleGrid) {
        int gridLen = obstacleGrid.length;
        int gridWid = obstacleGrid[0].length;
        int[] steps = new int[gridWid];
        steps[0] = 1;
        for (int row = 0; row < gridLen; row++) {
            for (int col = 0; col < gridWid; col++) {
                if (obstacleGrid[row][col] == OBSTACLE)
                    steps[col] = 0;
                else if (col == 0)
                    continue;
                else
                    steps[col] += steps[col - 1];
            }
        }
        return steps[gridWid - 1];
    }
}
