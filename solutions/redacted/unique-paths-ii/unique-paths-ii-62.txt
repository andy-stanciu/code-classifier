class Solution {

    public int method1(int[][] obstacleGrid) {
        int rows = obstacleGrid.length;
        int cols = obstacleGrid[0].length;
        int[] dp = new int[cols];
        for (int i = 0; i < rows; i++) {
            int[] cur = new int[cols];
            for (int j = 0; j < cols; j++) {
                if (obstacleGrid[i][j] == 1)
                    cur[j] = 0;
                else if (i == 0 && j == 0)
                    cur[j] = 1;
                else {
                    int bottom = 0, right = 0;
                    if (i > 0)
                        bottom = dp[j];
                    if (j > 0)
                        right = cur[j - 1];
                    cur[j] = bottom + right;
                }
            }
            dp = cur;
        }
        return dp[cols - 1];
    }
}
