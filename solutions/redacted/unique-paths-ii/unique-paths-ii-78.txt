class Solution {

    public int method1(int[][] arr) {
        int m = arr.length, n = arr[0].length;
        if (arr[0][0] == 1)
            return 0;
        int[] prev = new int[n];
        int[] curr = new int[n];
        curr[0] = 1;
        for (int row = 0; row < m; row++) {
            for (int col = 0; col < n; col++) {
                if (row == 0 && col == 0)
                    continue;
                else if (arr[row][col] == 1)
                    curr[col] = 0;
                else {
                    int down = (row > 0) ? prev[col] : 0;
                    int right = (col > 0) ? curr[col - 1] : 0;
                    curr[col] = down + right;
                }
            }
            prev = curr.clone();
        }
        return prev[n - 1];
    }
}
