/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
// merge two sorted linklist.
class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode firstlist = list1;
        ListNode secondlist = list2;
        if (firstlist == null) {
            if (secondlist != null)
                return secondlist;
        } else if (secondlist == null) {
            if (firstlist != null)
                return firstlist;
        }
        ListNode head = new ListNode(100);
        ListNode temp = head;
        while (firstlist != null && secondlist != null) {
            if (firstlist.val < secondlist.val) {
                ListNode k = new ListNode(firstlist.val);
                temp.next = k;
                temp = k;
                firstlist = firstlist.next;
            } else {
                ListNode k = new ListNode(secondlist.val);
                temp.next = k;
                temp = k;
                secondlist = secondlist.next;
            }
            if (secondlist == null) {
                temp.next = firstlist;
            } else {
                temp.next = secondlist;
            }
        }
        return head.next;
    }
}
