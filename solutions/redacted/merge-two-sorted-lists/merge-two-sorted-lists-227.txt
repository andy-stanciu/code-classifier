class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        // Base cases aka stop condition
        if (list1 == null) {
            // l1 is empty, directly return l2
            return list2;
        } else if (list2 == null) {
            // l2 is empty, directly return l1
            return list1;
        }
        // General cases
        if (list1.val < list2.val) {
            // l1 is smaller than l2
            list1.next = method1(list1.next, list2);
            return list1;
        } else {
            // l2 is smaller than l1
            list2.next = method1(list1, list2.next);
            return list2;
        }
    }
}
