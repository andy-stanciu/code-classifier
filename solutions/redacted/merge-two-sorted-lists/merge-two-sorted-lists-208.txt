class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode head;
        if (list1.val <= list2.val) {
            head = list1;
            head.next = method1(list1.next, list2);
        } else {
            head = list2;
            head.next = method1(list1, list2.next);
        }
        return head;
    }
}

// list2:
// list1.next : (1)-> (1)-> (2)->(3)-> (4)-> 4

// List1.next :
// list2.next :
