class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode res = new ListNode();
        ListNode list3 = res;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                list3.next = list1;
                list1 = list1.next;
            } else {
                list3.next = list2;
                list2 = list2.next;
            }
            list3 = list3.next;
        }
        if (list1 != null && list2 == null) {
            list3.next = list1;
        } else if (list2 != null && list1 == null) {
            list3.next = list2;
        }
        return res.next;
    }
}
