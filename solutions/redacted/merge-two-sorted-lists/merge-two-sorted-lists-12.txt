class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode newList = new ListNode(0);
        ListNode travel = newList;
        while (list1 != null || list2 != null) {
            if (list2 == null || list1 != null && list1.val < list2.val) {
                travel.next = new ListNode(list1.val);
                travel = travel.next;
                list1 = list1.next;
            } else {
                travel.next = new ListNode(list2.val);
                travel = travel.next;
                list2 = list2.next;
            }
        }
        return newList.next;
    }
}
