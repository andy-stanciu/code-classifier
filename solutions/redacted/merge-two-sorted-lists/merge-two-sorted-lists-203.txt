class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode start = null;
        ListNode head = null;
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                if (head == null) {
                    start = list1;
                    head = list1;
                } else {
                    head.next = list1;
                    head = head.next;
                }
                list1 = list1.next;
            } else if (list1.val > list2.val) {
                if (head == null) {
                    head = list2;
                    start = list2;
                } else {
                    head.next = list2;
                    head = head.next;
                }
                list2 = list2.next;
            }
        }
        if (list1 == null)
            head.next = list2;
        if (list2 == null)
            head.next = list1;
        return start;
    }
}
