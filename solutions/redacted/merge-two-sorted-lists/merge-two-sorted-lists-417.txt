class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode newlist = new ListNode();
        newlist = null;
        //return the alternative list if any list is empty
        if (list1 == null) {
            return list2;
        }
        if (list2 == null) {
            return list1;
        }
        //temp for list1
        ListNode h1 = list1;
        //temp for list2
        ListNode h2 = list2;
        //temp for final list
        ListNode h3 = newlist;
        //initial check values
        if (h1.val <= h2.val) {
            h3 = h1;
            h1 = h1.next;
        } else {
            h3 = h2;
            h2 = h2.next;
        }
        newlist = h3;
        while (h1 != null && h2 != null) {
            //loop terminates once any list gets empty
            if (h1.val <= h2.val) {
                h3.next = h1;
                h3 = h1;
                h1 = h1.next;
            } else {
                h3.next = h2;
                h3 = h2;
                h2 = h2.next;
            }
        }
        if (h1 != null) {
            //condition satisfies when there is  remaining elements in list1
            h3.next = h1;
        }
        if (h2 != null) {
            //condition satisfies when there is  remaining elements in list2
            h3.next = h2;
        }
        //return the final list
        return newlist;
    }
}
