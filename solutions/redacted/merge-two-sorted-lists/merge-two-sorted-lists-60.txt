class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 != null && list2 != null) {
            return list1.val > list2.val ? new ListNode(list2.val, method1(list1, list2.next)) : new ListNode(list1.val, method1(list1.next, list2));
        } else {
            return list1 == null ? list2 : list1;
        }
    }
}
