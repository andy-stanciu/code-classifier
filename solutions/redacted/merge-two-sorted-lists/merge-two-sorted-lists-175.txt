class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        ListNode temp = null;
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        if (list1.val > list2.val) {
            temp = list2;
            temp.next = method1(list1, list2.next);
        } else {
            temp = list1;
            temp.next = method1(list1.next, list2);
        }
        return temp;
    }
}
