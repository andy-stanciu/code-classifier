class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        return combinedList(list1, list2);
    }

    public ListNode combinedList(ListNode list1, ListNode list2) {
        ListNode dummy = new ListNode(0);
        ListNode temp = dummy;
        ListNode i = list1;
        ListNode j = list2;
        while (i != null && j != null) {
            if (i.val < j.val) {
                temp.next = i;
                temp = temp.next;
                i = i.next;
            } else {
                temp.next = j;
                temp = temp.next;
                j = j.next;
            }
        }
        while (i != null) {
            temp.next = i;
            temp = temp.next;
            i = i.next;
        }
        while (j != null) {
            temp.next = j;
            temp = temp.next;
            j = j.next;
        }
        return dummy.next;
    }
}
