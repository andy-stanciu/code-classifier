class Solution {

    public ListNode method1(ListNode list1, ListNode list2) {
        if (list1 == null) {
            return list2;
        } else if (list2 != null && list1.val > list2.val) {
            return method1(list2, list1);
        }
        return new ListNode(list1.val, method1(list1.next, list2));
    }
}
