class Solution {

    public void method1(int[] nums) {
        int[] count = new int[3];
        for (int i : nums) {
            count[i]++;
        }
        int i = 0;
        int j = 0;
        while (i < nums.length && j < count.length) {
            if (count[j] > 0) {
                nums[i] = j;
                count[j]--;
                i++;
            } else {
                j++;
            }
        }
    }
}
