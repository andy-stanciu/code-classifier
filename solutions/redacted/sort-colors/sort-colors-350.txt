class Solution {

    public void method1(int[] nums) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i : nums) {
            hm.put(i, hm.getOrDefault(i, 0) + 1);
        }
        int j = 0;
        for (int key : hm.keySet()) {
            int idx = hm.get(key);
            for (int i = 0; i < idx; i++) {
                nums[j] = key;
                j++;
            }
        }
    }
}
