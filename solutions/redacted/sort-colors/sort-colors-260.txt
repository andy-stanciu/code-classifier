class Solution {

    static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    public void method1(int[] nums) {
        int l = 0, mid = 0, r = nums.length - 1;
        while (mid <= r) if (nums[mid] == 0)
            swap(nums, l++, mid++);
        else if (nums[mid] == 1)
            mid++;
        else
            swap(nums, mid, r--);
    }
}
