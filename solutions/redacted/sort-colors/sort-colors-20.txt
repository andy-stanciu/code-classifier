class Solution {

    public void method1(int[] nums) {
        int[] count = new int[3];
        for (int i : nums) {
            count[i]++;
        }
        int counter = 0;
        for (int i = 0; i < 3; i++) {
            while (count[i]-- > 0) {
                nums[counter++] = i;
            }
        }
    }
}
