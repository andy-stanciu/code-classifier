class Solution {

    public void method1(int[] nums) {
        int zero = 0;
        int one = 0;
        int two = 0;
        for (int num : nums) {
            if (num == 0) {
                zero++;
            } else if (num == 1) {
                one++;
            } else {
                two++;
            }
        }
        // int[] array = new int[nums.length];
        int ce = 0;
        for (int i = 0; i < nums.length; i++) {
            if (zero > 0) {
                zero--;
                ce = 0;
            } else if (one > 0) {
                one--;
                ce = 1;
            } else if (two > 0) {
                two--;
                ce = 2;
            }
            nums[i] = ce;
        }
    }
}
