class Solution {

    public void method1(int[] nums) {
        int i = 0;
        int j = 0;
        int k = nums.length - 1;
        int temp = 0;
        while (j <= k) {
            if (nums[j] == 0) {
                temp = nums[j];
                nums[j] = nums[i];
                nums[i] = temp;
                i++;
                j++;
                continue;
            }
            if (nums[j] == 1) {
                j++;
                continue;
            }
            if (nums[j] == 2) {
                temp = nums[j];
                nums[j] = nums[k];
                nums[k] = temp;
                k--;
                continue;
            }
        }
    }
}
