class Solution {

    public void method1(int[] nums) {
        int zero_index = -1;
        int two_index = nums.length;
        int index = 0;
        while (index < two_index) {
            if (nums[index] == 0) {
                ++zero_index;
                swap(nums, zero_index, index);
                ++index;
            } else if (nums[index] == 2) {
                --two_index;
                swap(nums, two_index, index);
            } else {
                ++index;
            }
        }
    }

    private void swap(int[] array, int a, int b) {
        int temp = array[a];
        array[a] = array[b];
        array[b] = temp;
    }
}
