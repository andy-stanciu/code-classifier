class Solution {

    public void method1(int[] nums) {
        int[] var1 = new int[3];
        for (int i = 0; i < nums.length; i++) {
            int x = nums[i];
            var1[x] += 1;
        }
        int j = 0;
        for (int i = 0; i < var1.length; i++) {
            int n = var1[i];
            while (n > 0) {
                nums[j] = i;
                n--;
                j++;
            }
        }
    }
}
