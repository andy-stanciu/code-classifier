class Solution {

    public void method1(int[] arr) {
        int n = arr.length;
        // for(int i=0;i<n;i++){
        //     for(int j=i+1;j<n;j++){
        //         if(arr[j]<arr[i]){
        //             int temp = arr[i];
        //             arr[i] = arr[j];
        //             arr[j] = temp;
        //         }
        //     }
        // }
        int i = 0;
        int j = 0;
        int c1 = 0;
        int c2 = 0;
        int c3 = 0;
        while (i < n) {
            if (arr[i] == 0) {
                c1++;
            } else if (arr[i] == 1) {
                c2++;
            } else {
                c3++;
            }
            i++;
        }
        while (c1 > 0) {
            arr[j] = 0;
            j++;
            c1--;
        }
        while (c2 > 0) {
            arr[j] = 1;
            j++;
            c2--;
        }
        while (c3 > 0) {
            arr[j] = 2;
            j++;
            c3--;
        }
    }
}
