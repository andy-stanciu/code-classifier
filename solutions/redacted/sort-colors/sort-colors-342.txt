class Solution {

    public void method1(int[] nums) {
        for (int o = 0, t = nums.length - 1, i = 0; i < nums.length; i++) {
            // System.out.println(Arrays.toString(nums)+"||| "+o+" "+i+" "+t);
            if (nums[i] == 0 && i > o) {
                int temp = nums[o];
                nums[o] = nums[i];
                nums[i] = temp;
                // if((o+1)<(i-1)){
                o++;
                i--;
                // }
            } else if (nums[i] == 2 && i < t) {
                int temp = nums[t];
                nums[t] = nums[i];
                nums[i] = temp;
                // if((t-1)>(i-1)){
                t--;
                i--;
                // }
            }
        }
    }
}
