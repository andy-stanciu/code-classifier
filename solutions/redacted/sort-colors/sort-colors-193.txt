class Solution {

    //Link to image explanation: https://www.autodraw.com/share/41XBBLHOIA77
    public void method1(int[] nums) {
        int low = 0, mid = 0, high = nums.length - 1;
        int temp;
        while (mid <= high) {
            switch(nums[mid]) {
                case 0:
                    {
                        temp = nums[low];
                        nums[low] = nums[mid];
                        nums[mid] = temp;
                        low++;
                        mid++;
                        break;
                    }
                case 1:
                    mid++;
                    break;
                case 2:
                    {
                        temp = nums[high];
                        nums[high] = nums[mid];
                        nums[mid] = temp;
                        high--;
                        break;
                    }
            }
        }
    }
}
