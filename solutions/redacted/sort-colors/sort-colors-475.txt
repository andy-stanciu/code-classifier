class Solution {

    public void method1(int[] nums) {
        int low = 0;
        int mid = 0;
        int high = nums.length - 1;
        int temp1 = 0;
        int temp2 = 0;
        while (mid <= high) {
            switch(nums[mid]) {
                case 0:
                    temp1 = nums[low];
                    nums[low] = nums[mid];
                    nums[mid] = temp1;
                    low++;
                    mid++;
                    break;
                case 1:
                    mid++;
                    break;
                case 2:
                    temp2 = nums[high];
                    nums[high] = nums[mid];
                    nums[mid] = temp2;
                    high--;
                    break;
            }
        }
    }
}
