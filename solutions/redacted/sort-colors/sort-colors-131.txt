class Solution {

    public void method1(int[] nums) {
        int z = 0, o = 0, t = 0;
        for (int i : nums) {
            if (i == 0)
                z++;
            if (i == 1)
                o++;
            if (i == 2)
                t++;
        }
        System.out.println(z + "z" + o + "o" + t);
        for (int i = 0; i < z; i++) nums[i] = 0;
        for (int i = z; i < z + o; i++) nums[i] = 1;
        for (int i = o + z; i < t + o + z; i++) nums[i] = 2;
    }
}
