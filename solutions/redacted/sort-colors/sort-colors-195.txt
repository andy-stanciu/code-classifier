class Solution {

    public void method1(int[] nums) {
        int lastZero = 0, lastTwo = nums.length - 1;
        int i = 0;
        while (i <= lastTwo) {
            if (nums[i] == 0) {
                nums[i++] = nums[lastZero];
                nums[lastZero++] = 0;
            } else if (nums[i] == 2) {
                nums[i] = nums[lastTwo];
                nums[lastTwo--] = 2;
            } else
                i++;
        }
    }
}
