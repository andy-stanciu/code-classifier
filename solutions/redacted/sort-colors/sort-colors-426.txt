class Solution {

    public void method1(int[] arr) {
        int zeros = 0, ones = 0, n = arr.length;
        for (int i = 0; i < n; i++) {
            if (arr[i] == 0)
                zeros++;
            if (arr[i] == 1)
                ones++;
        }
        for (int i = 0; i < zeros; i++) {
            arr[i] = 0;
        }
        for (int i = zeros; i < zeros + ones; i++) {
            arr[i] = 1;
        }
        for (int i = zeros + ones; i < arr.length; i++) {
            arr[i] = 2;
        }
    }
}
