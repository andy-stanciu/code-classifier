class Solution {

    public void method1(int[] nums) {
        //all the indicies strictly less than left contain 0
        int left = 0;
        //all the indicies between mid and right are unsorted
        int mid = 0;
        //all the indicies strictly greater left contain 0
        int right = nums.length - 1;
        while (mid <= right) {
            if (nums[mid] == 0) {
                int temp = nums[mid];
                nums[mid] = nums[left];
                nums[left] = temp;
                left++;
                mid++;
            } else if (nums[mid] == 2) {
                //we can\'t increase mid partition because we don\'t know what the number we pulled from the right it
                int temp = nums[mid];
                nums[mid] = nums[right];
                nums[right] = temp;
                right--;
            } else {
                mid++;
            }
        }
    }
}
