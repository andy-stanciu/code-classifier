class Solution {

    public void method1(int[] nums) {
        // Pointer for the next position of 0
        int low = 0;
        // Current element pointer
        int mid = 0;
        // Pointer for the next position of 2
        int high = nums.length - 1;
        while (mid <= high) {
            switch(nums[mid]) {
                case 0:
                    swap(nums, low, mid);
                    low++;
                    mid++;
                    break;
                case 1:
                    mid++;
                    break;
                case 2:
                    swap(nums, mid, high);
                    high--;
                    break;
            }
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
