class Solution {

    public void method1(int[] arr) {
        // Arrays.sort(nums);//We can also solve using this inbuilt library function but since we aren\'t allowed to use, so using bubble sort technique
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) for (int j = 0; j < n - i - 1; j++) if (arr[j] > arr[j + 1]) {
            // swap arr[j+1] and arr[j]
            int temp = arr[j];
            arr[j] = arr[j + 1];
            arr[j + 1] = temp;
        }
    }
}
