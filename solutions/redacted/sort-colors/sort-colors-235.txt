class Solution {

    public void method1(int[] nums) {
        int count0 = 0, count1 = 0, n = nums.length;
        for (int i = 0; i < n; i++) {
            switch(nums[i]) {
                case 0:
                    count0++;
                    break;
                case 1:
                    count1++;
                    break;
            }
        }
        for (int i = 0; i < n; i++) {
            if (count0 > 0) {
                nums[i] = 0;
                count0--;
            } else if (count1 > 0) {
                nums[i] = 1;
                count1--;
            } else
                nums[i] = 2;
        }
    }
}
