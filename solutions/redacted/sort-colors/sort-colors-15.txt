class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int[] c_array = { 0, 0, 0 };
        for (int i = 0; i < n; i++) {
            if (nums[i] == 0) {
                c_array[0]++;
            } else if (nums[i] == 1) {
                c_array[1]++;
            } else {
                c_array[2]++;
            }
        }
        int value = 0;
        int count = 0;
        for (int x : c_array) {
            for (int i = 0; i < x; i++) {
                nums[count] = value;
                count++;
            }
            value++;
        }
    }
}
