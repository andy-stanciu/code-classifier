class Solution {

    public void method1(int[] nums) {
        int s = 0, mid = 0, l = nums.length - 1;
        while (mid <= l) {
            if (nums[mid] == 0) {
                swap(nums, s, mid);
                s++;
                mid++;
            } else if (nums[mid] == 1) {
                mid++;
            } else {
                swap(nums, mid, l);
                l--;
            }
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
