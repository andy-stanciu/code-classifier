class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> var1 = new ArrayList<>();
        method1(var1, new ArrayList<>(), nums);
        return var1;
    }

    public void method1(List<List<Integer>> var1, List<Integer> permutation, int[] nums) {
        if (permutation.size() == nums.length) {
            var1.add(new ArrayList<>(permutation));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!permutation.contains(nums[i])) {
                permutation.add(nums[i]);
                method1(var1, permutation, nums);
                permutation.remove(permutation.size() - 1);
            }
        }
    }
}
