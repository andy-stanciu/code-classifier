class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> l = new ArrayList<>();
        fn(l, nums, 0);
        return l;
    }

    public void fn(List<List<Integer>> l, int[] ar, int i) {
        if (i == ar.length) {
            List<Integer> mn = new ArrayList<>();
            for (int n : ar) {
                mn.add(n);
            }
            l.add(mn);
            return;
        }
        for (int ind = i; ind < ar.length; ind++) {
            swap(ar, ind, i);
            fn(l, ar, i + 1);
            swap(ar, ind, i);
        }
    }

    public void swap(int[] ar, int i1, int i2) {
        int temp = ar[i1];
        ar[i1] = ar[i2];
        ar[i2] = temp;
    }
}
