class Solution {

    public void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public void listAll(int index, int[] nums, List<List<Integer>> ans) {
        if (index == nums.length) {
            List<Integer> l = new ArrayList<>();
            for (int i = 0; i < nums.length; i++) {
                l.add(nums[i]);
            }
            ans.add(l);
        }
        for (int i = index; i < nums.length; i++) {
            swap(nums, i, index);
            listAll(index + 1, nums, ans);
            swap(nums, i, index);
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        listAll(0, nums, ans);
        return ans;
    }
}
