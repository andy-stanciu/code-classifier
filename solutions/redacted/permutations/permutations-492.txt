class Solution {

    List<List<Integer>> ll = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            find(new ArrayList<>(), i, nums);
        }
        return ll;
    }

    public void find(List<Integer> li, int j, int[] nums) {
        li.add(nums[j]);
        if (li.size() == nums.length) {
            ll.add(new ArrayList<>(li));
        } else {
            for (int i = 0; i < nums.length; i++) {
                if (!li.contains(nums[(i + j) % nums.length])) {
                    find(li, (i + j) % nums.length, nums);
                }
            }
        }
        li.remove(li.size() - 1);
    }
}
