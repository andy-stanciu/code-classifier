class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        boolean[] map = new boolean[nums.length];
        method1(nums, ans, new ArrayList<>(), map);
        return ans;
    }

    private void method1(int[] nums, List<List<Integer>> ans, List<Integer> ds, boolean[] map) {
        if (nums.length == ds.size()) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!map[i]) {
                map[i] = true;
                ds.add(nums[i]);
                method1(nums, ans, ds, map);
                ds.remove(ds.size() - 1);
                map[i] = false;
            }
        }
    }
}
