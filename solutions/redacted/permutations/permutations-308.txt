class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ll = new ArrayList<>();
        backtrack(ll, nums, new ArrayList<>(), 0);
        return ll;
    }

    public void backtrack(List<List<Integer>> ll, int[] nums, List<Integer> al, int s) {
        if (al.size() == nums.length) {
            ll.add(new ArrayList<>(al));
        } else {
            for (int i = 0; i < nums.length; i++) {
                if (al.contains(nums[i]))
                    continue;
                al.add(nums[i]);
                backtrack(ll, nums, al, i + 1);
                al.remove(al.size() - 1);
            }
        }
    }
}
