class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        boolean[] fre = new boolean[nums.length];
        recusrionPermution(nums, ds, fre, ans);
        return ans;
    }

    private void recusrionPermution(int[] nums, List<Integer> ds, boolean[] fre, List<List<Integer>> ans) {
        if (ds.size() == nums.length) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!fre[i]) {
                fre[i] = true;
                ds.add(nums[i]);
                recusrionPermution(nums, ds, fre, ans);
                ds.remove(ds.size() - 1);
                fre[i] = false;
            }
        }
    }
}
