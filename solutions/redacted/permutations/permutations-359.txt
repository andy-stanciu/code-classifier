class Solution {

    public static void solve(int[] nums, int start, List<List<Integer>> list) {
        if (start == nums.length - 1) {
            List<Integer> a = new ArrayList<>();
            for (int x : nums) {
                a.add(x);
            }
            list.add(a);
            return;
        }
        for (int i = start; i < nums.length; i++) {
            int temp = nums[start];
            nums[start] = nums[i];
            nums[i] = temp;
            solve(nums, start + 1, list);
            int tem2 = nums[start];
            nums[start] = nums[i];
            nums[i] = tem2;
        }
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> list = new ArrayList<>();
        solve(nums, 0, list);
        return list;
    }
}
