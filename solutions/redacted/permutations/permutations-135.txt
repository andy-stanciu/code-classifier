class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ip = new ArrayList<>();
        for (int ele : nums) ip.add(ele);
        solve(new ArrayList<>(), ip, ans);
        return ans;
    }

    public void solve(List<Integer> op, List<Integer> ip, List<List<Integer>> ans) {
        if (ip.isEmpty()) {
            ans.add(op);
            return;
        }
        for (int i = 0; i < ip.size(); i++) {
            List<Integer> newOp = new ArrayList<>();
            List<Integer> newIp = new ArrayList<>();
            for (int ele : ip) newIp.add(ele);
            for (int ele : op) newOp.add(ele);
            newOp.add(ip.get(i));
            newIp.remove(i);
            solve(newOp, newIp, ans);
        }
    }
}
