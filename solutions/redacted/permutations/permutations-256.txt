class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> collector = new ArrayList<>();
        permutation(nums, collector, new LinkedHashSet<>());
        return collector;
    }

    private void permutation(int[] nums, List<List<Integer>> collector, Set<Integer> iter) {
        if (iter.size() == nums.length) {
            collector.add(new ArrayList<>(iter));
            return;
        }
        for (int j = 0; j < nums.length; j++) {
            if (!iter.contains(nums[j])) {
                iter.add(nums[j]);
                permutation(nums, collector, iter);
                iter.remove(nums[j]);
            }
        }
    }
}
