class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> var1 = new ArrayList<>();
        LinkedList<Integer> start = new LinkedList<>();
        start.add(nums[0]);
        var1.add(start);
        for (int i = 1; i < nums.length; i++) {
            List<List<Integer>> var2 = new ArrayList<>();
            for (List permutation : var1) {
                for (int j = 0; j <= permutation.size(); j++) {
                    LinkedList<Integer> newPermutation = new LinkedList<>(permutation);
                    newPermutation.add(j, nums[i]);
                    var2.add(newPermutation);
                }
            }
            var1 = var2;
        }
        return var1;
    }
}
