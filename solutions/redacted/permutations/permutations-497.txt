class Solution {

    Set<Integer> used = new HashSet<>();

    List<List<Integer>> result = new ArrayList<>();

    List<Integer> cp = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        p(nums);
        return result;
    }

    public void p(int[] nums) {
        if (used.size() == nums.length) {
            result.add(new ArrayList<>(cp));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (used.contains(i)) {
                continue;
            }
            used.add(i);
            cp.add(nums[i]);
            permute(nums);
            used.remove(i);
            cp.remove(cp.size() - 1);
        }
    }
}
