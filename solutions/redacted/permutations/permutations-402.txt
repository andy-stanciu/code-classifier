class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        Set<Integer> s = new HashSet<>();
        helper(nums, ans, s, new ArrayList<>());
        return ans;
    }

    public void helper(int[] a, List<List<Integer>> ans, Set<Integer> s, ArrayList<Integer> ds) {
        if (ds.size() == a.length) {
            ans.add(new ArrayList<>(ds));
        }
        for (int i = 0; i < a.length; i++) {
            if (s.add(a[i])) {
                ds.add(a[i]);
                helper(a, ans, s, ds);
                ds.remove(ds.size() - 1);
                s.remove(a[i]);
            }
        }
    }
}
