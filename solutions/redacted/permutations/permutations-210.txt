class Solution {

    public void permutation(int[] nums, ArrayList<Integer> list, List<List<Integer>> temp, int i) {
        if (list.size() == nums.length) {
            temp.add(new ArrayList<>(list));
            return;
        }
        if (i == nums.length) {
            return;
        }
        if (!list.contains(nums[i])) {
            list.add(nums[i]);
            permutation(nums, list, temp, 0);
            list.remove(list.size() - 1);
        }
        permutation(nums, list, temp, i + 1);
    }

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> temp = new ArrayList<>();
        permutation(nums, new ArrayList<>(), temp, 0);
        return temp;
    }
}
