class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> var1 = new ArrayList();
        List<Integer> permutation = new ArrayList();
        solve(nums, permutation, var1, new HashSet<>());
        return var1;
    }

    public void solve(int[] nums, List<Integer> permutation, List<List<Integer>> var1, HashSet<Integer> set) {
        if (permutation.size() == nums.length) {
            var1.add(new ArrayList<>(permutation));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!set.contains(nums[i])) {
                permutation.add(nums[i]);
                set.add(nums[i]);
                solve(nums, permutation, var1, set);
                permutation.remove(permutation.size() - 1);
                set.remove(nums[i]);
            }
        }
    }
}
