class Solution {

    private List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        List<Integer> listNums = new ArrayList<>();
        for (int num : nums) {
            listNums.add(num);
        }
        helper(new ArrayList<>(), listNums);
        return result;
    }

    private void helper(List<Integer> base, List<Integer> nums) {
        if (nums.size() == 0) {
            result.add(base);
            return;
        }
        for (int num : nums) {
            List<Integer> baseTemp = new ArrayList<>(base);
            baseTemp.add(num);
            List<Integer> numsTemp = new ArrayList<>(nums);
            numsTemp.remove(Integer.valueOf(num));
            helper(baseTemp, numsTemp);
        }
    }
}
