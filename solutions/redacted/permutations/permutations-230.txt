class Solution {

    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> s = new ArrayList<List<Integer>>();
        List<Integer> t = new ArrayList<>();
        r(nums, t, s);
        return s;
    }

    public void r(int[] n, List<Integer> t, List<List<Integer>> s) {
        if (t.size() == n.length)
            s.add(new ArrayList<>(t));
        for (int i = 0; i < n.length; i++) {
            if (t.contains(n[i]))
                continue;
            t.add(n[i]);
            r(n, t, s);
            t.remove(t.size() - 1);
        }
    }
}
