class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> permute(int[] nums) {
        // so this question is similar to the permutation question just the simple pattern of the backtraking question
        // so now here
        List<Integer> curr = new ArrayList<>();
        boolean[] used = new boolean[nums.length];
        helper(nums, curr, used);
        return list;
    }

    void helper(int[] nums, List<Integer> curr, boolean[] used) {
        // so nwo here
        if (curr.size() == nums.length) {
            list.add(new ArrayList<>(curr));
            return;
        }
        // so nwo here otherwise we have to
        for (int i = 0; i < nums.length; i++) {
            // so now here we have skip it
            if (used[i] || (i > 0 && nums[i] == nums[i - 1] && used[i - 1])) {
                continue;
            }
            // so nwo here adding here
            curr.add(nums[i]);
            used[i] = true;
            helper(nums, curr, used);
            curr.remove(curr.size() - 1);
            used[i] = false;
        }
    }
}
