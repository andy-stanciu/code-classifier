class Solution {

    public static List<List<Integer>> recursionpermute(int[] nums, List<Integer> ds, List<List<Integer>> ans, boolean[] freq) {
        // Base Case : If the size of data structure(i.e ds) is equal to length of nums Array it means the permutation in ds is one of possible permutation.
        if (ds.size() == nums.length) {
            ans.add(new ArrayList(ds));
            return ans;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!freq[i]) {
                // !freq[i] Means current element is not present in our data structure.
                freq[i] = true;
                // Add current element into the data Structure (i.e ds).
                ds.add(nums[i]);
                recursionpermute(nums, ds, ans, freq);
                ds.remove(ds.size() - 1);
                freq[i] = false;
            }
        }
        return ans;
    }

    public List<List<Integer>> permute(int[] nums) {
        // It store all possible permutation.
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        // It is used to track our current element is present in our ds(i.e data structure) or not.
        boolean[] freq = new boolean[nums.length];
        ans = recursionpermute(nums, ds, ans, freq);
        return ans;
    }
}
