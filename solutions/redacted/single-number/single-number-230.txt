class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> numOccurences = new HashMap<>();
        for (int num : nums) {
            numOccurences.put(num, numOccurences.getOrDefault(num, 0) + 1);
        }
        for (int num : numOccurences.keySet()) {
            if (numOccurences.get(num) == 1) {
                return num;
            }
        }
        return 0;
    }
}
