class Solution {

    public int method1(int[] nums) {
        //1. create a HashMap with Integer Key & Integer Value
        HashMap<Integer, Integer> map = new HashMap<Integer, Integer>();
        //2. for loop to iterate through the given array
        for (int i = 0; i < nums.length; i++) {
            //3. check if the map contains the key same as nums[i]
            if (map.containsKey(nums[i])) {
                //3.1 if it does, increment the value of the key by +1
                map.put(nums[i], map.get(nums[i]) + 1);
            } else //4. else put the nums[i] value as a new key and assign 1 as initial value
            {
                map.put(nums[i], 1);
            }
        }
        //5. initialize the answer to be returned as 0
        int val = 0;
        //6. check through each of the key in the map
        for (int j : map.keySet()) {
            //7. check if the map has a Key(j) with 1 as value
            if (map.get(j) == 1) {
                //7.1 if it does, change the answer (val) to the key (j)
                val = j;
            }
        }
        //8. return the answer(val)
        return val;
    }
}
