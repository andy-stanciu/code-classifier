class Solution {

    public int method1(int[] nums) {
        //Usig Bit Manupulation
        int bitmask = 0;
        //we know that a^a=0
        //so the elments which are appeared twice will become
        //zero except the elemnt which was appears only once
        //PLEASE UPVOTE IF YOU FOUND THIS HELPFUL
        for (int val : nums) bitmask ^= val;
        return bitmask;
    }
}
