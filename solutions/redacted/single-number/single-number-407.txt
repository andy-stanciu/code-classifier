class Solution {

    public int method1(int[] nums) {
        int d = 0;
        for (int i = 0; i <= nums.length - 1; i++) {
            int count = 0;
            for (int j = 0; j <= nums.length - 1; j++) {
                if (nums[i] == nums[j]) {
                    count++;
                }
            }
            if (count == 1) {
                d = nums[i];
            }
        }
        return d;
    }
}
