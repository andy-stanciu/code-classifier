class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int i = 0;
        if (nums.length == 1) {
            return nums[0];
        }
        while (i < nums.length - 1) {
            int j = i + 1;
            if (nums[i] == nums[j]) {
                i += 2;
            } else {
                return nums[i];
            }
        }
        return nums[nums.length - 1];
    }
}
