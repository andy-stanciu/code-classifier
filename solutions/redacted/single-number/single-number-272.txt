class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            // XOR of any Number with itself gives 0
            //Xor of any Number with 0 gives number itself
            var1 = var1 ^ nums[i];
        }
        return var1;
    }
}
