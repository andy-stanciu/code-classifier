class Solution {

    public int method1(int[] nums) {
        int a = 0;
        for (int i = 0; i < nums.length; i++) {
            a = nums[i];
            for (int j = 0; j < nums.length; j++) {
                if (i == j)
                    continue;
                if (a == nums[j])
                    break;
                if (j == nums.length - 1)
                    return a;
            }
        }
        return a;
    }
}
