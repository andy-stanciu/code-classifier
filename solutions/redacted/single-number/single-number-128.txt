class Solution {

    public int method1(int[] nums) {
        //taking the first value of array as reference
        int temp = nums[0];
        /*  traversing along array and doing Xor of elements from  
        2nd element that is element with index 0  */
        for (int i = 1; i < nums.length; i++) {
            temp ^= nums[i];
        }
        //the left out element will be stored in temp variable
        return temp;
    }
}
