class Solution {

    public int method1(int[] nums) {
        int unique = 0;
        for (int num : nums) {
            // a ^ a = 0 (any number XORed with itself results in 0).
            // a ^ 0 = a (any number XORed with 0 results in the number itself).
            unique ^= num;
        }
        return unique;
    }
}
