class Solution {

    public int method1(int[] nums) {
        Set<Integer> distinctNum = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (!distinctNum.contains(nums[i])) {
                distinctNum.add(nums[i]);
            } else {
                distinctNum.remove(nums[i]);
            }
        }
        // Explicit garbage collection
        System.gc();
        int element = 0;
        for (int item : distinctNum) {
            element = item;
        }
        return element;
    }
}
