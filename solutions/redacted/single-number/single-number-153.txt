/*
Time Complexity  : O(n) 
Space Complexity : O(1)
Example :
-----------
Input : nums[]={2,3,2}
xor(2,3)= 010^011 =001=1
xor(2,1)= 010^001 =011=3
Output=3
*/
class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum = sum ^ nums[i];
        }
        return sum;
    }
}
