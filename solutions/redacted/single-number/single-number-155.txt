class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            if (map.containsKey(i))
                map.put(i, map.get(i) + 1);
            else
                map.put(i, 1);
        }
        int num = 0;
        for (int j : nums) {
            if (map.get(j) == 1) {
                num = j;
            }
        }
        return num;
    }
}
