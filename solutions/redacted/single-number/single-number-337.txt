class Solution {

    public int method1(int[] nums) {
        // sort the array so the duplicates appear adjacently
        Arrays.sort(nums);
        int ans = 0;
        if (nums.length == 1 || nums[0] != nums[0 + 1]) {
            // edge case if the first number is unique
            ans = nums[0];
        }
        if (nums.length > 1 && nums[nums.length - 1] != nums[nums.length - 2]) {
            //edge case if the last number is unique (must make sure the length of the array is greater than 1 )
            ans = nums[nums.length - 1];
        }
        for (int i = 1; i < nums.length - 1; i++) {
            if (nums[i] != nums[i + 1] && nums[i] != nums[i - 1]) {
                // iterate the array and check if the nums[i] is not equal to the number before and the number after it
                ans = nums[i];
            }
        }
        // return the answer.
        return ans;
    }
}
