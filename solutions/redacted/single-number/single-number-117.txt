class Solution {

    public int method1(int[] nums) {
        boolean cond = false;
        for (int i = 0; i < nums.length; i++) {
            test1: for (int j = 0; j < nums.length; j++) {
                if (i != j && nums[i] == nums[j]) {
                    cond = false;
                    break test1;
                }
                cond = true;
            }
            if (cond)
                return nums[i];
        }
        return 0;
    }
}
