class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        } else {
            for (int i = 0; i < nums.length; i++) {
                int c = 0;
                for (int j = 0; j < nums.length; j++) {
                    if (nums[i] == nums[j]) {
                        c++;
                    } else {
                        continue;
                    }
                }
                if (c == 1) {
                    return nums[i];
                }
            }
            return 0;
        }
    }
}
