class Solution {

    public int method1(int[] nums) {
        int num = 0;
        for (int a : nums) {
            //here the xor operator used
            // xor has property a ^ a=0
            num = num ^ a;
        }
        // a ^ 0=a so when same numbers comes it cancels
        // for different numbers it gives output
        return num;
    }
}
