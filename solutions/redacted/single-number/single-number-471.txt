class Solution {

    public int method1(int[] nums) {
        int num;
        int n = nums.length;
        int count = 0;
        for (int i = 0; i < n; i++) {
            num = nums[i];
            count = 0;
            for (int j = 0; j < n; j++) {
                if (nums[j] == num) {
                    count++;
                }
            }
            if (count == 1) {
                return num;
            }
        }
        return -1;
    }
}
