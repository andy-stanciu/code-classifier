class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> countMap = new HashMap<>();
        for (Integer e : nums) {
            countMap.put(e, countMap.getOrDefault(e, 0) + 1);
        }
        List<Map.Entry<Integer, Integer>> collect = countMap.entrySet().stream().collect(Collectors.toList());
        for (Map.Entry<Integer, Integer> entry : collect) {
            if (entry.getValue() == 1) {
                return entry.getKey();
            }
        }
        return -1;
    }
}
