class Solution {

    public int method1(int[] nums) {
        int countofOnes = 0;
        int result = 0;
        for (int i = 0; i <= 31; i++) {
            int temp = 1 << i;
            countofOnes = 0;
            for (int n : nums) {
                if ((n & temp) != 0) {
                    countofOnes++;
                }
            }
            if (countofOnes % 2 != 0) {
                result = result | temp;
            }
        }
        return result;
    }
}
