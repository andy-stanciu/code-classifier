class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> hash = new <Integer, Integer> HashMap();
        for (int x : nums) {
            if (hash.containsKey(x)) {
                //if hashmap contains \'element\', then increase correspoding val
                int count = hash.get(x);
                hash.put(x, ++count);
            } else
                //if hashmap does not contain  \'element\', then put it in hashmap
                hash.put(x, 1);
        }
        for (int x : hash.keySet()) {
            //Iterate through \'keys\' of HashMap
            if (//if \'value\' for \'key\' is \'1\', then return that \'key\' element
            hash.get(x) == 1)
                return x;
        }
        return -1;
    }
}
