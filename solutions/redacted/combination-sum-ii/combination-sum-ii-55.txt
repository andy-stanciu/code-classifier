class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ans = new ArrayList<>();
        method2(candidates, target, 0, new ArrayList<>(), ans);
        return ans;
    }

    private void method2(int[] nums, int target, int index, List<Integer> subset, List<List<Integer>> ans) {
        if (target == 0) {
            ans.add(new ArrayList(subset));
            return;
        }
        if (target < 0)
            return;
        for (int i = index; i < nums.length; i++) {
            if (i > index && nums[i] == nums[i - 1])
                continue;
            subset.add(nums[i]);
            method2(nums, target - nums[i], i + 1, subset, ans);
            subset.remove(subset.size() - 1);
        }
    }
}
