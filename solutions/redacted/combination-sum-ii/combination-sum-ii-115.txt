class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> list = new ArrayList<>();
        check(list, candidates, target, 0, new ArrayList<>(), 0);
        return list;
    }

    public void check(List<List<Integer>> list, int[] candidates, int target, int param1, List<Integer> arr, int x) {
        if (param1 == target) {
            list.add(new ArrayList<>(arr));
            return;
        }
        if (param1 > target) {
            return;
        }
        for (int i = x; i < candidates.length && param1 < target; i++) {
            if (i > x && candidates[i] == candidates[i - 1])
                continue;
            arr.add(candidates[i]);
            check(list, candidates, target, param1 + candidates[i], arr, i + 1);
            arr.remove(arr.size() - 1);
        }
    }
}
