//Approach - 3 (Tabulation)
public class Solution {

    public int method1(int[] nums, int target) {
        int[] dp = new int[target + 1];
        dp[0] = 1;
        for (int t = 1; t <= target; t++) {
            int res = 0;
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] <= t) {
                    res = res + dp[t - nums[i]];
                }
            }
            dp[t] = res;
        }
        return dp[target];
    }
}
