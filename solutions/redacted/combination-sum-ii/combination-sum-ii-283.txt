class Solution {

    public static List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ans = new ArrayList<>();
        method2(candidates, target, 0, ans, new ArrayList<>());
        return ans;
    }

    public static void method2(int[] cands, int target, int index, List<List<Integer>> ans, List<Integer> arr) {
        if (target == 0) {
            ans.add(new ArrayList<>(arr));
            return;
        }
        if (index >= cands.length || cands[index] > target) {
            return;
        }
        arr.add(cands[index]);
        method2(cands, target - cands[index], index + 1, ans, arr);
        arr.remove(arr.size() - 1);
        while (index + 1 < cands.length && cands[index] == cands[index + 1]) {
            index++;
        }
        method2(cands, target, index + 1, ans, arr);
    }
}
