class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        // n logn
        Arrays.sort(candidates);
        method2(0, 0, new ArrayList<>(), candidates, target, result);
        return result;
    }

    private void method2(int index, int param1, List<Integer> list, int[] candidates, int target, List<List<Integer>> result) {
        // base case
        if (param1 == target) {
            result.add(new ArrayList<>(list));
            return;
        }
        if (param1 > target) {
            return;
        }
        for (int i = index; i < candidates.length; i++) {
            // n
            if (i > index && candidates[i] == candidates[i - 1])
                continue;
            int num = candidates[i];
            list.add(num);
            method2(i + 1, param1 + num, list, candidates, target, result);
            list.remove(list.size() - 1);
        }
    }
}
