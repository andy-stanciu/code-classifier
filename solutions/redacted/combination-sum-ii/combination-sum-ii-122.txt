class Solution {

    public static List<List<Integer>> method1(int[] arr, int t) {
        List<List<Integer>> ansList = new ArrayList<>();
        Arrays.sort(arr);
        method2(ansList, new ArrayList<>(), arr, 0, t, 0);
        return ansList;
    }

    private static void method2(List<List<Integer>> ansList, ArrayList<Integer> param1, int[] arr, int param2, int target, int index) {
        if (param2 == target) {
            ansList.add(new ArrayList<>(param1));
            return;
        }
        int prev = -1;
        // for loop for each index
        for (int i = index; i < arr.length; i++) {
            if (arr[i] == prev) {
                continue;
            }
            if (arr[i] + param2 > target) {
                break;
            }
            param1.add(arr[i]);
            method2(ansList, param1, arr, param2 + arr[i], target, i + 1);
            param1.removeLast();
            prev = arr[i];
        }
    }
}
