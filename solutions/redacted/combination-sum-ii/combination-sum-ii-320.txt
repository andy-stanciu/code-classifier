//Approach - 1
public class Solution {

    public int method1(int[] nums, int target) {
        if (target == 0) {
            return 1;
        }
        int res = 0;
        for (int num : nums) {
            if (num <= target) {
                res += method1(nums, target - num);
            }
        }
        return res;
    }
}
