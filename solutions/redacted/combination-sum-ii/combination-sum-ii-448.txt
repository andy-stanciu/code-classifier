class Solution {

    private void fComb(int idx, int[] arr, int t, List<List<Integer>> sol, List<Integer> temp) {
        if (t == 0) {
            sol.add(new ArrayList<>(temp));
            return;
        }
        for (int i = idx; i < arr.length; i++) {
            if (i > idx && arr[i] == arr[i - 1])
                continue;
            if (arr[i] > t)
                break;
            temp.add(arr[i]);
            fComb(i + 1, arr, t - arr[i], sol, temp);
            temp.remove(temp.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> sol = new ArrayList<>();
        Arrays.sort(candidates);
        fComb(0, candidates, target, sol, new ArrayList<>());
        return sol;
    }
}
