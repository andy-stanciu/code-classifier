class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ComB = new ArrayList<>();
        method2(0, candidates, new ArrayList(), ComB, target);
        return ComB;
    }

    void method2(int start, int[] nums, List<Integer> Current, List<List<Integer>> ComB, int target) {
        if (target == 0) {
            ComB.add(new ArrayList(Current));
        }
        if (target < 0) {
            return;
        }
        for (int i = start; i < nums.length; i++) {
            if (i > start && nums[i] == nums[i - 1])
                continue;
            Current.add(nums[i]);
            method2(i + 1, nums, Current, ComB, target - nums[i]);
            Current.remove(Current.size() - 1);
        }
    }
}
