class Solution {

    private static void method1(int ind, int[] arr, int target, List<List<Integer>> ans, List<Integer> ds) {
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
        }
        for (int i = ind; i < arr.length; i++) {
            //If a Element is a repetitive element do not pick it
            if (i > ind && arr[i] == arr[i - 1])
                continue;
            //If this condition is satisfied then add the element to the list
            if (arr[i] > target)
                break;
            ds.add(arr[i]);
            //Move the next index and reduce the target and find if the target is present in array add to the Data structure
            method1(i + 1, arr, target - arr[i], ans, ds);
            ds.remove(ds.size() - 1);
        }
    }

    public List<List<Integer>> method2(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        method1(0, candidates, target, ans, new ArrayList<>());
        return ans;
    }
}
