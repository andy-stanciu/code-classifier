class Solution {

    public void solve(int[] candidates, int target, List<List<Integer>> sol, List<Integer> temp, int index, int param1) {
        if (param1 > target)
            return;
        if (target == param1)
            sol.add(new ArrayList<>(temp));
        for (int i = index; i < candidates.length; i++) {
            if (i > index && candidates[i] == candidates[i - 1])
                continue;
            temp.add(candidates[i]);
            solve(candidates, target, sol, temp, i + 1, param1 + candidates[i]);
            temp.remove(temp.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> sol = new ArrayList<>();
        int param1 = 0;
        Arrays.sort(candidates);
        solve(candidates, target, sol, new ArrayList<>(), 0, param1);
        return sol;
    }
}
