class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ans = new ArrayList<>();
        helper(0, target, candidates, ans, new ArrayList<>());
        return ans;
    }

    private void helper(int index, int target, int[] candidates, List<List<Integer>> ans, List<Integer> ds) {
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
        }
        for (int i = index; i < candidates.length; i++) {
            if (candidates[i] > target) {
                break;
            }
            // for avoiding duplicates call and for first call to occur
            if (i != index && candidates[i] == candidates[i - 1])
                continue;
            ds.add(candidates[i]);
            helper(i + 1, target - candidates[i], candidates, ans, ds);
            ds.remove(ds.size() - 1);
        }
    }
}
