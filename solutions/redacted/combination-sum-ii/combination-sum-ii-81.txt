class Solution {

    public void solve(int[] candidates, List<List<Integer>> ans, int target, int start, int param1, ArrayList<Integer> arr) {
        if (target == param1) {
            ans.add(new ArrayList<>(arr));
            return;
        }
        int n = candidates.length;
        if (start == n || target < param1) {
            return;
        }
        for (int i = start; i < n; i++) {
            if (i != start && candidates[i] == candidates[i - 1])
                continue;
            arr.add(candidates[i]);
            solve(candidates, ans, target, i + 1, param1 + candidates[i], arr);
            arr.remove(arr.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] candidates, int target) {
        List<List<Integer>> list = new ArrayList<List<Integer>>();
        Arrays.sort(candidates);
        solve(candidates, list, target, 0, 0, new ArrayList<>());
        return list;
    }
}
