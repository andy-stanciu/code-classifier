class Solution {

    public void find_all(int ind, int[] nums, int target, List<List<Integer>> ans, List<Integer> ds) {
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        if (target < 0) {
            return;
        }
        for (int i = ind; i < nums.length; i++) {
            if (i > ind && nums[i] == nums[i - 1]) {
                continue;
            }
            ds.add(nums[i]);
            find_all(i + 1, nums, target - nums[i], ans, ds);
            ds.remove(ds.size() - 1);
        }
    }

    public List<List<Integer>> method1(int[] nums, int target) {
        Arrays.sort(nums);
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> ds = new ArrayList<>();
        find_all(0, nums, target, ans, ds);
        return ans;
    }
}
