class Solution {

    public List<List<Integer>> method1(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> l1 = new ArrayList<>();
        fun(0, candidates, new ArrayList(), l1, target);
        return l1;
    }

    void fun(int start, int[] nums, ArrayList<Integer> curr, List<List<Integer>> l1, int target) {
        if (target == 0) {
            l1.add(new ArrayList(curr));
            return;
        }
        if (target < 0) {
            return;
        }
        for (int i = start; i < nums.length; i++) {
            if (i > start && nums[i] == nums[i - 1])
                continue;
            curr.add(nums[i]);
            fun(i + 1, nums, curr, l1, target - nums[i]);
            curr.remove(curr.size() - 1);
        }
    }
}
