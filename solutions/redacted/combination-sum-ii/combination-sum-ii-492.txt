class Solution {

    public static void method1(int index, int target, int[] arr, List<Integer> ds, List<List<Integer>> ans) {
        if (target == 0) {
            ans.add(new ArrayList<>(ds));
            return;
        }
        for (int i = index; i < arr.length; i++) {
            if (i > index && arr[i] == arr[i - 1])
                continue;
            if (arr[i] > target)
                break;
            ds.add(arr[i]);
            method1(i + 1, target - arr[i], arr, ds, ans);
            ds.remove(ds.size() - 1);
        }
    }

    public static List<List<Integer>> method2(int[] arr, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(arr);
        method1(0, target, arr, new ArrayList<>(), ans);
        return ans;
    }
}
