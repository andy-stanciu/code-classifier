class Solution {

    public boolean canFinish(int param1, int[][] prerequisites) {
        // As there are no pre requisite
        if (prerequisites.length == 0)
            return true;
        // if there are prerequisite
        // create a dependency graph (prerequisiste graph)
        List<List<Integer>> var1 = new ArrayList<>();
        for (int i = 0; i < param1; i++) {
            var1.add(new ArrayList<Integer>());
        }
        // graph bi -> ai
        for (int i = 0; i < prerequisites.length; i++) {
            var1.get(prerequisites[i][1]).add(prerequisites[i][0]);
        }
        // this array is to track the visited ones
        int[] visited = new int[param1];
        // visit each course using dfs and is there is a cycle return false
        // cycle means you can\'t complete the prerequisite
        for (int i = 0; i < param1; i++) {
            if (!dfs(var1, visited, i))
                return false;
        }
        return true;
    }

    // dfs function
    public boolean dfs(List<List<Integer>> var1, int[] visited, int param2) {
        // mark the current node as visited
        visited[param2] = 1;
        // traverse all its adjacent and unmarked node and
        // call recursive function with index of adjacent node
        List<Integer> var2 = var1.get(param2);
        for (int i = 0; i < var2.size(); i++) {
            int var3 = var2.get(i);
            // we are visiting it again which means cycle
            if (visited[var3] == 1)
                return false;
            if (visited[var3] == 0) {
                if (!dfs(var1, visited, var3))
                    return false;
            }
        }
        // done
        visited[param2] = 2;
        return true;
    }
}
