class Solution {

    public boolean canFinish(int param1, int[][] prerequisites) {
        if (param1 <= 1 || prerequisites == null || prerequisites.length <= 1) {
            return true;
        }
        int[][] matrix = new int[param1][param1];
        int[] requirements = new int[param1];
        for (int[] each : prerequisites) {
            int var1 = each[0];
            int pre = each[1];
            if (matrix[pre][var1] == 0) {
                requirements[var1]++;
            }
            matrix[pre][var1] = 1;
        }
        java.util.Queue<Integer> stack = new java.util.LinkedList<>();
        for (int i = 0; i < param1; i++) {
            if (requirements[i] == 0) {
                stack.add(i);
            }
        }
        int counter = 0;
        while (!stack.isEmpty()) {
            int var1 = stack.poll();
            counter++;
            for (int i = 0; i < param1; i++) {
                if (matrix[var1][i] != 0) {
                    requirements[i]--;
                    if (requirements[i] == 0) {
                        stack.add(i);
                    }
                }
            }
        }
        return counter == param1;
    }
}
