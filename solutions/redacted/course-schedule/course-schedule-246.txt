class Solution {

    Map<Integer, List<Integer>> map = new HashMap<>();

    public boolean canFinish(int param1, int[][] prerequisites) {
        for (int i = 0; i < param1; i++) {
            map.put(i, new ArrayList<>());
        }
        for (int i = 0; i < prerequisites.length; i++) {
            map.get(prerequisites[i][0]).add(prerequisites[i][1]);
        }
        int[] visited = new int[param1];
        for (int i = 0; i < param1; i++) {
            if (!dfs(i, visited)) {
                return false;
            }
        }
        return true;
    }

    private boolean dfs(int param2, int[] visited) {
        if (visited[param2] == 1) {
            return false;
        }
        if (visited[param2] == 2) {
            return true;
        }
        visited[param2] = 1;
        for (int n : map.get(param2)) {
            if (!dfs(n, visited)) {
                return false;
            }
        }
        visited[param2] = 2;
        return true;
    }
}
