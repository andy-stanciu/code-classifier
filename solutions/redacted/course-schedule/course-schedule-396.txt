class Solution {

    public boolean canFinish(int param1, int[][] prerequisites) {
        List<Integer>[] map = getMap(param1, prerequisites);
        boolean[] visited = new boolean[param1];
        boolean[] hasInpath = new boolean[param1];
        for (int i = 0; i < param1; i++) {
            if (!visited[i] && dfs(i, map, visited, hasInpath)) {
                return false;
            }
        }
        return true;
    }

    private boolean dfs(int start, List<Integer>[] map, boolean[] visited, boolean[] hasInpath) {
        visited[start] = true;
        hasInpath[start] = true;
        for (int child : map[start]) {
            if (hasInpath[child])
                return true;
            if (!visited[child] && dfs(child, map, visited, hasInpath)) {
                return true;
            }
        }
        hasInpath[start] = false;
        return false;
    }

    private List<Integer>[] getMap(int param1, int[][] prerequisites) {
        List<Integer>[] map = new ArrayList[param1];
        for (int i = 0; i < param1; i++) {
            map[i] = new ArrayList<>();
        }
        for (int i = 0; i < prerequisites.length; i++) {
            int start = prerequisites[i][0];
            int end = prerequisites[i][1];
            map[end].add(start);
        }
        return map;
    }
}
