class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String str : strs) {
            String strReverse = str;
            char[] reverseArr = strReverse.toCharArray();
            Arrays.sort(reverseArr);
            // System.out.println(str+" "+strReverse+" "+anagramGroupsMap);
            var1.computeIfAbsent(new String(reverseArr), k -> new ArrayList<>()).add(str);
        }
        return var1.values().stream().collect(Collectors.toList());
    }
}
