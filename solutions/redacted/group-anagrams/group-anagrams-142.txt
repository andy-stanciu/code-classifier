class Solution {

    // time complexity ON(logN) and space O(N)
    public List<List<String>> method1(String[] strs) {
        var var1 = new HashMap<String, List<String>>();
        for (var str : strs) {
            var charData = str.toCharArray();
            Arrays.sort(charData);
            var key = new String(charData);
            if (var1.containsKey(key)) {
                var1.get(key).add(str);
            } else {
                var1.put(key, new ArrayList<>(List.of(str)));
            }
        }
        return new ArrayList<>(var1.values());
    }
}
