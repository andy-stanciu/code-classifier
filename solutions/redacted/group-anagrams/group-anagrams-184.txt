class Solution {

    public List<List<String>> method1(String[] strs) {
        // A map to group anagrams based on their sorted string
        Map<String, List<String>> strMap = new HashMap<>();
        List<List<String>> result = new ArrayList<>();
        // Loop through each string
        for (String str : strs) {
            // Convert the string to a character array, sort it, and convert it back to a string
            char[] charArray = str.toCharArray();
            Arrays.sort(charArray);
            String sortedStr = new String(charArray);
            // If the sorted string doesn\'t exist in the map, add it with a new list
            if (!strMap.containsKey(sortedStr)) {
                strMap.put(sortedStr, new ArrayList<>());
            }
            // Add the original string to the corresponding anagram group
            strMap.get(sortedStr).add(str);
        }
        // Collect all the grouped anagrams from the map
        result.addAll(strMap.values());
        return result;
    }
}
