class Solution {

    public List<List<String>> method1(String[] strs) {
        //Map contains unique sorted word as key and list of original words as value.
        // Example: words = [EAT, ATE, TEA] the sorted chars of all these words would be same i.e AET.
        //so AET will be the key of map and all these original words would become value and this is what exactly the ask is.
        Map<String, List<String>> map = new HashMap<>();
        //Iterate over original words
        for (String word : strs) {
            //Rearrange the chars of the given word in sorted form to make the key of map
            char[] charArray = word.toCharArray();
            Arrays.sort(charArray);
            String sortedStr = String.copyValueOf(charArray);
            // Put such words into map
            map.putIfAbsent(sortedStr, new ArrayList<>());
            map.get(sortedStr).add(word);
        }
        //Since map.values() return collections type, convert that to list.
        return map.values().stream().collect(Collectors.toList());
    }
}
