class Solution {

    private String stringSort(String str) {
        char[] temp = str.toCharArray();
        Arrays.sort(temp);
        return new String(temp);
    }

    public List<List<String>> method1(String[] strs) {
        int n = strs.length;
        HashMap<String, List<String>> var1 = new HashMap<>();
        for (String str : strs) {
            String sortedString = stringSort(str);
            if (!var1.containsKey(sortedString)) {
                var1.put(sortedString, new ArrayList<>());
            }
            var1.get(sortedString).add(str);
        }
        return new ArrayList<>(var1.values());
    }
}
