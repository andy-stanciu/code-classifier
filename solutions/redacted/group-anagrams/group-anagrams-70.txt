class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> result = new HashMap<>();
        for (String word : strs) {
            char[] Ana = word.toCharArray();
            Arrays.sort(Ana);
            String keyAna = Arrays.toString(Ana);
            if (result.containsKey(keyAna)) {
                result.get(keyAna).add(word);
            } else {
                result.put(keyAna, new ArrayList<>());
                result.get(keyAna).add(word);
            }
        }
        return new ArrayList<>(result.values());
    }
}
