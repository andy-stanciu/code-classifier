class Solution {

    public List<List<String>> method1(String[] strs) {
        // hashmap created
        Map<String, List<String>> mp1 = new HashMap<>();
        for (int k = 0; k < strs.length; k++) {
            char[] ref = strs[k].toCharArray();
            Arrays.sort(ref);
            String t1 = new String(ref);
            if (// if it does not contain the key
            !mp1.containsKey(t1)) {
                String to_push = strs[k];
                List<String> temp = new ArrayList<>();
                temp.add(to_push);
                mp1.put(t1, temp);
            } else //if it contains the key
            {
                List<String> temp2 = mp1.get(t1);
                temp2.add(strs[k]);
                mp1.put(t1, temp2);
            }
        }
        List<List<String>> res = new ArrayList<>();
        for (Map.Entry m : mp1.entrySet()) {
            List<String> lis2 = (List<String>) m.getValue();
            res.add(lis2);
        }
        return res;
    }
}
