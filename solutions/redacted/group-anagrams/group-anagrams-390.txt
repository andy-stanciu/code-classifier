class Solution {

    public List<List<String>> method1(String[] strs) {
        // If the array is empty return an empty ArrayList
        if (strs.length == 0)
            return new ArrayList<>();
        // Create a HashMap to store the sorted character array as keys and original arrays as values.
        Map<String, List<String>> map = new HashMap<>();
        // Iterate over the strs array
        for (String s : strs) {
            // Convert each element to a character array
            char[] ca = s.toCharArray();
            // Sort the character array
            Arrays.sort(ca);
            // Make a string out of the sorted array
            String key = String.valueOf(ca);
            // If the map doesn\'t contains the key created then put the key to map.
            if (!map.containsKey(key))
                map.put(key, new ArrayList<>());
            // Add the original string to the map under sorted key.
            map.get(key).add(s);
        }
        // Return the map as ArrayList
        return new ArrayList<>(map.values());
    }
}
