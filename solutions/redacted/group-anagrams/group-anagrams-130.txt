class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String word : strs) {
            //convert to char array
            var charString = word.toCharArray();
            //now sort it
            Arrays.sort(charString);
            //now convert back after sorting
            var chartostring = new String(charString);
            //Now save this sorted string as key in our list
            var1.putIfAbsent(chartostring, new ArrayList<>());
            //now get the key
            var1.get(chartostring).add(word);
        }
        return new ArrayList<>(var1.values());
    }
}
