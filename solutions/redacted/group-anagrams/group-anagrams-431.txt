class Solution {

    public List<List<String>> method1(String[] strs) {
        // our hashmap has key with string and values with array list
        HashMap<String, List<String>> map = new HashMap<>();
        for (String word : strs) {
            //for every word in the string
            char[] chars = word.toCharArray();
            //changing word into chararray as strings are immutable
            //sorting the word
            Arrays.sort(chars);
            //again changing to string
            String sortedWord = new String(chars);
            if (!map.containsKey(sortedWord)) {
                //check whether sorted word is in key or not
                //if not create a empty array list
                map.put(sortedWord, new ArrayList<>());
            }
            //if there add the word in it
            map.get(sortedWord).add(word);
        }
        //return that array list
        return new ArrayList<>(map.values());
    }
}
