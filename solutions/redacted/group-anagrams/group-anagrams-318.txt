class Solution {

    public List<List<String>> method1(String[] strs) {
        //Create a hashmap to store the sorted strings as keys and List of corresponding strings from strs
        Map<String, List<String>> map = new HashMap<>();
        //Sort the strs array of strings
        for (String str : strs) {
            //Create a new char Array and store str in it
            char[] charArr = str.toCharArray();
            //Sort this char Array
            Arrays.sort(charArr);
            //Store this sorted char Array in the strs original Array
            String sortedCharArr = new String(charArr);
            //Check if there exists any key of the current value of sortedCharArr
            if (!(map.containsKey(sortedCharArr))) {
                //If there is no existing key of sortedCharArr create one
                map.put(sortedCharArr, new ArrayList<>());
            }
            //Add the current value of str to the ArrayList of the current key(sortedCharArr)
            map.get(sortedCharArr).add(str);
        }
        // Create an ArrayList to store the lists of anagrams
        List<List<String>> ans = new ArrayList<>();
        // Iterate over the entries in the map
        for (Map.Entry<String, List<String>> entry : map.entrySet()) {
            // Get the list of anagrams and add it to the ArrayList
            ans.add(entry.getValue());
        }
        return ans;
    }
}
