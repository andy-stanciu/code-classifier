class Solution {

    public List<List<String>> method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            return new ArrayList<>();
        }
        Map<String, List<String>> map = new HashMap<String, List<String>>();
        for (String string : strs) {
            //String string2=Arrays.stream(string.split("")).sorted().collect(Collectors.joining());
            char[] cs = string.toCharArray();
            Arrays.sort(cs);
            String string2 = new String(cs);
            if (!map.containsKey(string2)) {
                map.put(string2, new ArrayList<>());
            }
            map.get(string2).add(string);
        }
        return new ArrayList<>(map.values());
    }
}
