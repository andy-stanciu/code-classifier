class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> h = new HashMap<>();
        for (String s : strs) {
            char[] cl = s.toCharArray();
            Arrays.sort(cl);
            String ss = new String(cl);
            if (!h.containsKey(ss)) {
                h.put(ss, new ArrayList<>());
            }
            h.get(ss).add(s);
        }
        return new ArrayList<>(h.values());
    }
}
