class Solution {

    public List<List<String>> method1(String[] strs) {
        // we start by initializing an hashmap
        Map<String, List<String>> result = new HashMap<>();
        // here we iterate over all the strings that we have
        for (String s : strs) {
            // convert the current string to char array to be rearranged
            char[] ch = s.toCharArray();
            // sort the given array to form the key
            Arrays.sort(ch);
            // the key is formed
            String key = new String(ch);
            if (!result.containsKey(key)) {
                // check if the key already exists
                result.put(key, new ArrayList<>());
            }
            // add the current string to the required key
            result.get(key).add(s);
        }
        // return all the key values
        return new ArrayList<>(result.values());
    }
}
