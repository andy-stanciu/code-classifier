class Solution {

    public List<List<String>> method1(String[] strs) {
        // Map to group anagrams using frequency map as key
        HashMap<Map<Character, Integer>, List<String>> map = new HashMap<>();
        for (String str : strs) {
            // Frequency map for each string
            HashMap<Character, Integer> freqMap = new HashMap<>();
            for (char c : str.toCharArray()) {
                freqMap.put(c, freqMap.getOrDefault(c, 0) + 1);
            }
            // If the frequency map is not in the map, add a new list
            if (!map.containsKey(freqMap)) {
                map.put(freqMap, new ArrayList<>());
            }
            // Add the current string to the corresponding list
            map.get(freqMap).add(str);
        }
        // Prepare the result list from map values
        List<List<String>> result = new ArrayList<>();
        for (List<String> val : map.values()) {
            result.add(val);
        }
        return result;
    }
}
