class Solution {

    public List<List<String>> method1(String[] strs) {
        // some validation at the beginning
        // assueme all the constrains are passed
        Map<String, List<String>> var1 = new HashMap<>();
        for (int i = 0; i < strs.length; ++i) {
            String currentStr = strs[i];
            String convertedStr = bubbleSortString(currentStr);
            List currentList = var1.get(convertedStr);
            if (currentList == null) {
                currentList = new ArrayList();
            }
            currentList.add(currentStr);
            var1.put(convertedStr, currentList);
        }
        List var2 = new ArrayList();
        var1.forEach((key, value) -> var2.add(value));
        return var2;
    }

    public String bubbleSortString(String rawString) {
        char[] charArray = rawString.toCharArray();
        for (int i = 0; i < charArray.length - 1; ++i) {
            for (int j = 0; j < charArray.length - 1; ++j) {
                if (charArray[j] > charArray[j + 1]) {
                    char temp = charArray[j + 1];
                    charArray[j + 1] = charArray[j];
                    charArray[j] = temp;
                }
            }
        }
        System.out.println(new String(charArray));
        return new String(charArray);
    }
}
