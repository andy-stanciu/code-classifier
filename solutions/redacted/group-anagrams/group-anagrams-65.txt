class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> var1 = new HashMap<>();
        for (String str : strs) {
            char[] charArray = str.toCharArray();
            Arrays.sort(charArray);
            String key = new String(charArray);
            if (!var1.containsKey(key)) {
                var1.put(key, new ArrayList<>());
            }
            var1.get(key).add(str);
        }
        return new ArrayList(var1.values());
    }
}
