class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> hm = new HashMap<>();
        for (String st : strs) {
            char[] CharArray = st.toCharArray();
            Arrays.sort(CharArray);
            String snew = new String(CharArray);
            if (!hm.containsKey(snew)) {
                hm.put(snew, new ArrayList<>());
            }
            hm.get(snew).add(st);
        }
        List<List<String>> result = new ArrayList<>();
        for (Map.Entry<String, List<String>> h : hm.entrySet()) {
            result.add(h.getValue());
        }
        return result;
    }
}
