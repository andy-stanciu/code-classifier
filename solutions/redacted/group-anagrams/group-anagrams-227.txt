class Solution {

    public List<List<String>> method1(String[] strs) {
        // Create a hashmap to group anagrams using their sorted character string as the key
        Map<String, List<String>> map = new HashMap<>();
        // Iterate through each string in the input array
        for (String s : strs) {
            // Convert the string to a character array and sort it
            char[] chars = s.toCharArray();
            Arrays.sort(chars);
            // Create the sorted string from the sorted character array
            String sorted = new String(chars);
            // If the key does not exist in the map, initialize it with an empty list
            if (!map.containsKey(sorted)) {
                map.put(sorted, new ArrayList<>());
            }
            // Add the original string to the list of anagrams in the hashmap
            map.get(sorted).add(s);
        }
        // Convert the values in the hashmap to a list of lists to return the grouped anagrams
        return new ArrayList<>(map.values());
    }
}
