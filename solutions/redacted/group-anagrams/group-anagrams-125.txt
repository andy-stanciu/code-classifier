public class Solution {

    public List<List<String>> method1(String[] strs) {
        // Initialize a HashMap where the key is a sorted string and value is a list of anagrams.
        Map<String, List<String>> var1 = new HashMap<>();
        // Iterate through each string in the array.
        for (String str : strs) {
            // Convert the string to a char array, sort it, and then convert it back to a string.
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            String sortedStr = new String(chars);
            // Check if the sorted string is already a key in the map.
            // If not, initialize a new list for this key.
            var1.putIfAbsent(sortedStr, new ArrayList<>());
            // Add the original string to the list corresponding to the sorted key.
            var1.get(sortedStr).add(str);
        }
        // Collect all the values (lists of anagrams) from the map into a single list to return.
        return new ArrayList<>(var1.values());
    }
}
