class Solution {

    public List<List<String>> method1(String[] strs) {
        // Resultant list to store lists of anagrams
        List<List<String>> ans = new ArrayList<>();
        // HashMap to store canonical form as key and list of anagrams as value
        Map<String, List<String>> hs = new HashMap<>();
        // Har string ko process karte hain
        for (int i = 0; i < strs.length; i++) {
            // String ko character array me convert karte hain
            char[] temp = strs[i].toCharArray();
            // Characters ko sort karte hain
            Arrays.sort(temp);
            // Sorted characters ko ek string me convert karte hain (canonical form)
            String key = new String(temp);
            // Agar key map me already exist karti hai, to us list me string ko add karte hain
            if (hs.containsKey(key)) {
                hs.get(key).add(strs[i]);
            } else {
                // Agar key nahi milti hai, to ek new list create karte hain aur string ko add karte hain
                hs.put(key, new ArrayList<>());
                hs.get(key).add(strs[i]);
            }
        }
        // HashMap ki values ko result list me add karte hain
        for (List<String> l : hs.values()) {
            ans.add(l);
        }
        // Result list ko return karte hain
        return ans;
    }
}
