class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> res = new HashMap<>();
        for (int i = 0; i < strs.length; ++i) {
            String currKey = Arrays.stream(strs[i].split("")).sorted().collect(Collectors.joining());
            res.computeIfAbsent(currKey, k -> new ArrayList<>()).add(strs[i]);
        }
        return res.values().stream().collect(Collectors.toList());
    }
}
