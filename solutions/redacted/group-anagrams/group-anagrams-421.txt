class Solution {

    public List<List<String>> method1(String[] strs) {
        // Create a map to store anagrams
        Map<String, List<String>> map = new HashMap<>();
        // Iterate through each string in the input array
        for (String str : strs) {
            // Convert the string to a character array, sort it, and convert it back to string
            char[] charArray = str.toCharArray();
            Arrays.sort(charArray);
            String sortedStr = new String(charArray);
            // If the sorted string is already in the map, add the original string to its list
            if (map.containsKey(sortedStr)) {
                map.get(sortedStr).add(str);
            } else {
                // If the sorted string is not in the map, create a new list and add the original string to it
                List<String> list = new ArrayList<>();
                list.add(str);
                map.put(sortedStr, list);
            }
        }
        // Convert the map values to a list and return
        return new ArrayList<>(map.values());
    }
}
