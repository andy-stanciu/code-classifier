class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<HashMap<Character, Integer>, List<String>> mainMap = new HashMap<HashMap<Character, Integer>, List<String>>();
        for (String str : strs) {
            HashMap<Character, Integer> frequencyMap = new HashMap<Character, Integer>();
            for (int i = 0; i < str.length(); i++) {
                char ch = str.charAt(i);
                frequencyMap.put(ch, frequencyMap.getOrDefault(ch, 0) + 1);
            }
            if (!mainMap.containsKey(frequencyMap)) {
                List<String> list = new ArrayList<String>();
                list.add(str);
                mainMap.put(frequencyMap, list);
            } else {
                List<String> list = mainMap.get(frequencyMap);
                list.add(str);
            }
        }
        List<List<String>> result = new ArrayList<>();
        for (List<String> val : mainMap.values()) {
            result.add(val);
        }
        return result;
    }
}
