class Solution {

    public List<List<String>> method1(String[] strs) {
        ArrayList<String> toReturn = new ArrayList<>();
        HashMap<String, ArrayList<String>> vals = new HashMap<>();
        for (int i = 0; i < strs.length; i++) {
            String str = strs[i];
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            String sortedStr = new String(chars);
            ArrayList<String> words = new ArrayList<String>();
            if (vals.containsKey(sortedStr)) {
                words = vals.get(sortedStr);
            } else {
                // Only add a new list once, when it\'s genuinely a new list.
                toReturn.add(words);
            }
            words.add(str);
            vals.put(sortedStr, words);
        }
        return toReturn;
    }
}
