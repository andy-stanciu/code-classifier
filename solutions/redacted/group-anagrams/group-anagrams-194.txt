class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (var str : strs) {
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            String sortedStr = String.valueOf(chars);
            List<String> var2 = var1.get(sortedStr);
            if (var2 == null) {
                List<String> adds = new ArrayList<>();
                adds.add(str);
                var1.put(sortedStr, adds);
            } else
                var2.add(str);
        }
        return new ArrayList<>(var1.values());
    }
}
