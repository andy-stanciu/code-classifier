class Solution {

    public List<List<String>> method1(String[] strs) {
        // Create a HashMap to hold sorted words as keys and their anagrams as values
        Map<String, List<String>> map = new HashMap<>();
        // Iterate through each word in the input array
        for (String word : strs) {
            // Convert the word to a character array and sort it
            char[] chars = word.toCharArray();
            Arrays.sort(chars);
            // Create a new string from the sorted character array
            String sortedWord = new String(chars);
            // If the sorted word is not already a key in the map, add it
            if (!map.containsKey(sortedWord)) {
                map.put(sortedWord, new ArrayList<>());
            }
            // Add the original word to the list corresponding to the sorted key
            map.get(sortedWord).add(word);
        }
        // Return a list of all the grouped anagrams
        return new ArrayList<>(map.values());
    }
}
