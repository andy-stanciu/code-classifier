class Solution {

    public List<List<String>> method1(String[] strs) {
        // Create a hashmap to store the sorted string as key and list of anagrams as value
        Map<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            // Convert the string to a char array, sort it, then convert it back to a string
            char[] charArray = str.toCharArray();
            Arrays.sort(charArray);
            String sortedStr = new String(charArray);
            // If this sorted string is not in the map, add it
            if (!map.containsKey(sortedStr)) {
                map.put(sortedStr, new ArrayList<>());
            }
            // Add the original string to the corresponding anagram group
            map.get(sortedStr).add(str);
        }
        // Return all the grouped anagrams
        return new ArrayList<>(map.values());
    }

    // Main method to test the solution
    public static void main(String[] args) {
        Solution solution = new Solution();
        // Example 1
        String[] strs1 = { "eat", "tea", "tan", "ate", "nat", "bat" };
        System.out.println(solution.method1(strs1));
        // Example 2
        String[] strs2 = { "" };
        System.out.println(solution.method1(strs2));
        // Example 3
        String[] strs3 = { "a" };
        System.out.println(solution.method1(strs3));
    }
}
