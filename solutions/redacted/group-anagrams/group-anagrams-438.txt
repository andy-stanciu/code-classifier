class Solution {

    public List<List<String>> method1(String[] strs) {
        /*
            Option 1:
                1.sort each word O(N*mlogm) = O(N)
                2. add to map with original index
                3. convert map to list of lists by mapping via original index
            Option 2: is it possible to make a char order agnostic checksum?
        */
        Map<String, List<String>> var1 = new HashMap<>();
        for (String word : strs) {
            char[] sortedChars = word.toCharArray();
            Arrays.sort(sortedChars);
            String wordKey = String.valueOf(sortedChars);
            List<String> var2 = var1.getOrDefault(wordKey, new ArrayList<String>());
            var2.add(word);
            var1.put(wordKey, var2);
        }
        List<List<String>> var3 = new ArrayList(var1.size());
        for (List<String> var2 : var1.values()) {
            var3.add(var2);
        }
        return var3;
    }
}
