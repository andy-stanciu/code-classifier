class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> set = new HashMap<>();
        for (String newstrs : strs) {
            char[] ch = newstrs.toCharArray();
            Arrays.sort(ch);
            String newString = new String(ch);
            if (!set.containsKey(newString)) {
                set.put(newString, new ArrayList<>());
            }
            set.get(newString).add(newstrs);
        }
        return new ArrayList<>(set.values());
    }
}
