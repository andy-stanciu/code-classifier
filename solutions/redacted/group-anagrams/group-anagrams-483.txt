class Solution {

    public List<List<String>> method1(String[] strs) {
        // Create a HashMap to store sorted strings as keys and lists of anagrams as
        // values
        HashMap<String, List<String>> hm = new HashMap<>();
        // Iterate through each string in the input array
        for (String str : strs) {
            // Convert the string to a character array and sort it
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            // Convert the sorted character array back to a string
            String sortedStr = new String(chars);
            // Compute the value for the sorted string key using computeIfAbsent
            // If the key is not present in the HashMap, create a new ArrayList and
            // associate it with the key
            // Then add the original string to the list
            // If the key is already present, retrieve the associated list and add the
            // original string to it
            hm.computeIfAbsent(sortedStr, key -> new ArrayList<>()).add(str);
        }
        // Return a new ArrayList containing all the lists of anagrams (values) from the
        // HashMap
        return new ArrayList<>(hm.values());
    }
}
