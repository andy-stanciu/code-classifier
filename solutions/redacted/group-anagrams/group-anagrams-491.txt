class Solution {

    public List<List<String>> method1(String[] strs) {
        //I have take HashMap where key is String and value is List<String>
        Map<String, List<String>> map = new HashMap<>();
        //iterating each and every string
        for (String str : strs) {
            //converting String str to character array
            char[] ch = str.toCharArray();
            //sorting the array ch in ascending order
            Arrays.sort(ch);
            //converting sorted array ch to String
            String sortedString = new String(ch);
            //if the above sorted String is not in the map already then add a new empty ArrayList
            map.putIfAbsent(sortedString, new ArrayList<>());
            // add the original string to the list
            map.get(sortedString).add(str);
        }
        //we are returning after converting the above values of the map to a list of lists
        return new ArrayList<>(map.values());
    }
}
