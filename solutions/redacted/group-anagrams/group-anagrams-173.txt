class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            // Sort the string
            String sortedStr = sortstr(str);
            // If the sorted string is not already a key in the map, add it
            if (!map.containsKey(sortedStr)) {
                map.put(sortedStr, new ArrayList<>());
            }
            // Add the original string to the list corresponding to its sorted form
            map.get(sortedStr).add(str);
        }
        // Return all the grouped anagrams
        return new ArrayList<>(map.values());
    }

    public String sortstr(String str) {
        char[] charArr = str.toCharArray();
        Arrays.sort(charArr);
        return new String(charArr);
    }
}
