class Solution {

    public List<List<String>> method1(String[] words) {
        // Map to store anagram groups
        Map<String, Integer> map = new HashMap<>();
        List<List<String>> ans = new ArrayList<>();
        // Iterate through each word in the input array
        for (String word : words) {
            char[] array = word.toCharArray();
            Arrays.sort(array);
            String str = new String(array);
            // If key is already present, append the word to the existing list
            if (map.containsKey(str)) {
                ans.get(map.get(str)).add(word);
            } else // If key is not in the map, add a new entry with the word in a list
            // and add a entry in a map such that the size of list will treat as index of string.
            {
                map.put(str, ans.size());
                List<String> subList = new ArrayList<>();
                subList.add(word);
                ans.add(subList);
            }
        }
        return ans;
    }
}
