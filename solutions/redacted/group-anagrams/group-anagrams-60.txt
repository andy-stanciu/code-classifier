// Algo Used: Hashing and Linear Traversal
class Solution {

    public List<List<String>> method1(String[] strs) {
        HashMap<String, List<String>> map = new HashMap<>();
        for (String word : strs) {
            // convert string to char array
            char[] ch = word.toCharArray();
            // sort it so all anagrams string will become same
            Arrays.sort(ch);
            // take it now as a String
            String anagram = new String(ch);
            if (!map.containsKey(anagram)) {
                // if this anagram is not present then add a List<String> against that anagram
                map.put(anagram, new ArrayList<>());
            }
            // put the orignal word in the List
            map.get(anagram).add(word);
        }
        // Add all the List from the Map in a outer List and return
        return new ArrayList<>(map.values());
    }
}
