class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> var1 = new HashMap<>();
        for (String str : strs) {
            char[] chars = str.toCharArray();
            Arrays.sort(chars);
            String sortedStr = String.valueOf(chars);
            var1.computeIfAbsent(sortedStr, k -> new ArrayList()).add(str);
        }
        return var1.values().stream().collect(Collectors.toList());
    }
}
