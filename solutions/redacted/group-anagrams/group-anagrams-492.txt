class Solution {

    public List<List<String>> method1(String[] strs) {
        Map<String, List<String>> map = new HashMap<>();
        for (String anagram : strs) {
            //convert to char array to get individual chars
            char[] temporary = anagram.toCharArray();
            //sort the chars to form a unique key
            Arrays.sort(temporary);
            String key = String.valueOf(temporary);
            //make a list of lists associated to every key
            if (!map.containsKey(key))
                map.put(key, new ArrayList<String>());
            map.get(key).add(anagram);
        }
        List<List<String>> result = new ArrayList<>();
        for (String s : map.keySet()) result.add(map.get(s));
        return result;
    }
}
