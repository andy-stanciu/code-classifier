class Solution {

    // my solution
    public int[] intersect(int[] nums1, int[] nums2) {
        int[] arr = new int[1001];
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums1.length; i++) {
            arr[nums1[i]]++;
        }
        for (int i = 0; i < nums2.length; i++) {
            if (arr[nums2[i]] > 0) {
                list.add(nums2[i]);
                arr[nums2[i]]--;
            }
        }
        return list.stream().mapToInt(Integer::intValue).toArray();
    }
}
