class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        List<Integer> listResult = new ArrayList<>();
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int nums1Length = nums1.length;
        int nums2Length = nums2.length;
        int startNums1 = 0;
        int startNums2 = 0;
        for (; startNums1 < nums1Length; ++startNums1) {
            for (; startNums2 < nums2Length; ++startNums2) {
                if (nums2[startNums2] == nums1[startNums1]) {
                    listResult.add(nums1[startNums1]);
                    ++startNums2;
                    break;
                } else if (nums2[startNums2] > nums1[startNums1]) {
                    break;
                }
            }
        }
        return listResult.stream().mapToInt(Integer::intValue).toArray();
    }
}
