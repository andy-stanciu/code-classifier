class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        ArrayList<Integer> list = new ArrayList<>();
        int a, b = 0;
        if (nums1.length > nums2.length) {
            a = nums2.length;
            b = nums1.length;
        } else {
            a = nums1.length;
            b = nums2.length;
        }
        outerloop1: for (int i = 0; i < a; i++) {
            outerloop: for (int j = 0; j < b; j++) {
                if (nums1.length < nums2.length) {
                    if (nums1[i] == nums2[j]) {
                        list.add(nums1[i]);
                        nums1[i] = -1;
                        nums2[j] = -1;
                        break outerloop;
                    }
                } else {
                    if (nums1[j] == nums2[i]) {
                        list.add(nums2[i]);
                        nums1[j] = -1;
                        nums2[i] = -1;
                        break outerloop;
                    }
                }
            }
            if (list.size() > nums1.length || list.size() > nums2.length) {
                list.remove(list.size() - 1);
                break outerloop1;
            }
        }
        int[] answer = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            answer[i] = list.get(i);
        }
        return answer;
    }
}
