public class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        // Edge case: if any array is empty, return empty array
        if (nums1.length == 0 || nums2.length == 0) {
            return new int[0];
        }
        // Use HashMap to count frequency of elements in nums1
        Map<Integer, Integer> countMap = new HashMap<>();
        for (int num : nums1) {
            countMap.put(num, countMap.getOrDefault(num, 0) + 1);
        }
        // List to store intersecting elements
        List<Integer> var1 = new ArrayList<>();
        // Iterate through nums2 and find intersecting elements
        for (int num : nums2) {
            if (countMap.containsKey(num) && countMap.get(num) > 0) {
                var1.add(num);
                countMap.put(num, countMap.get(num) - 1);
            }
        }
        // Convert List to array
        int[] result = new int[var1.size()];
        for (int i = 0; i < var1.size(); i++) {
            result[i] = var1.get(i);
        }
        return result;
    }
}
