class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        List<Integer> nums1List = new ArrayList<>(nums1.length);
        List<Integer> resultList = new LinkedList<>();
        for (int i : nums1) nums1List.add(i);
        for (int j : nums2) {
            if (nums1List.contains(j)) {
                resultList.add(j);
                // remove method by default will remove an element based on index so we convert into object
                nums1List.remove((Object) j);
            }
        }
        int[] result = new int[resultList.size()];
        for (int i = 0; i < resultList.size(); i++) {
            result[i] = resultList.get(i);
        }
        return result;
    }
}
