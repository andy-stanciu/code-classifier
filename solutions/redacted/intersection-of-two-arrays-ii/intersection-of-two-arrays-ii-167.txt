class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        ArrayList<Integer> list = new ArrayList<Integer>();
        //Iterate nums1[], and check is nums1[i] exist in nums2[]
        for (int i = 0; i < nums1.length; i++) {
            if (searchOverlap(nums1[i], nums2)) {
                list.add(nums1[i]);
            }
        }
        //Convert ArrayList to Array
        int[] result = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            result[i] = list.get(i);
        }
        return result;
    }

    static boolean searchOverlap(int nums1, int[] nums2) {
        for (int i = 0; i < nums2.length; i++) {
            //If found it, replace as INT_MAX to avoid duplicate counting
            if (nums2[i] == nums1) {
                nums2[i] = Integer.MAX_VALUE;
                return true;
            }
        }
        //Not found
        return false;
    }
}
