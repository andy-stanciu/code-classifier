class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        // to count the freq of each element and size is declared according to the constraints
        int[] arr = new int[1001];
        // to return the result array we are using it
        int[] res = new int[1001];
        for (int num : nums1) {
            // counting the frequencies of each
            arr[num]++;
        }
        int cnt = 0;
        for (int num : nums2) {
            if (arr[num] > 0) {
                // if the num exist in nums1
                // store the num in res
                res[cnt++] = num;
                // reduce the freq of num in "arr" array
                arr[num]--;
            }
        }
        // to return using a copyOfRange(int[],start_ind,end_ind)
        return Arrays.copyOfRange(res, 0, cnt);
    }
}
