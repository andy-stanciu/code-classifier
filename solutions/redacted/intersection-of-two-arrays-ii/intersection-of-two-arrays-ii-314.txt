class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] arr = new int[1001];
        int[] res = new int[1001];
        for (int i : nums1) {
            arr[i]++;
        }
        int ctr = 0;
        for (int i : nums2) {
            if (arr[i] > 0) {
                res[ctr++] = i;
                arr[i]--;
            }
        }
        return Arrays.copyOfRange(res, 0, ctr);
    }
}
