class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] cntArr1 = new int[1001];
        int[] cntArr2 = new int[1001];
        int N1 = nums1.length;
        int N2 = nums2.length;
        for (int i = 0; i < N1; i++) {
            cntArr1[nums1[i]]++;
        }
        for (int i = 0; i < N2; i++) {
            cntArr2[nums2[i]]++;
        }
        int l = 0, r = 0, k = 0;
        while (l < 1001 && r < 1001) {
            while (cntArr1[l] != 0 && cntArr2[r] != 0) {
                cntArr1[l]--;
                cntArr2[r]--;
                nums1[k++] = l;
                System.out.println(l);
            }
            l++;
            r++;
        }
        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = nums1[i];
        }
        return res;
    }
}
