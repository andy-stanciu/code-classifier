class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        int[] freq = new int[1001];
        int[] result = new int[1001];
        for (int n : nums1) {
            freq[n]++;
        }
        int index = 0;
        for (int n : nums2) {
            if (freq[n] > 0) {
                result[index++] = n;
                freq[n]--;
            }
        }
        return Arrays.copyOfRange(result, 0, index);
    }
}
