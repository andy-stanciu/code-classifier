class Solution {

    public int[] intersect(int[] a, int[] b) {
        int[] exist = new int[1001];
        int[] inter = new int[Math.max(a.length, b.length)];
        int k = 0;
        for (int A : a) {
            exist[A]++;
        }
        for (int B : b) {
            if (exist[B] > 0) {
                inter[k] = B;
                k++;
                exist[B]--;
            }
        }
        return Arrays.copyOfRange(inter, 0, k);
    }
}
