class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        ArrayList<Integer> al = new ArrayList<>();
        ArrayList<Integer> alans = new ArrayList<>();
        for (int i = 0; i < nums1.length; i++) {
            al.add(nums1[i]);
        }
        for (int i = 0; i < nums2.length; i++) {
            if (al.contains(nums2[i])) {
                int idx = al.indexOf(nums2[i]);
                alans.add(nums2[i]);
                al.remove(idx);
            }
        }
        int[] ans = new int[alans.size()];
        int i = 0;
        Iterator itr = alans.iterator();
        while (itr.hasNext()) {
            ans[i] = (int) itr.next();
            i++;
        }
        return ans;
    }
}
