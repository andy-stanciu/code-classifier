class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        //SORT BOTH THE ARRAYS FOR COMPARISION
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int i = 0, j = 0;
        //MAKING A LIST TO STORE COMMON ELEMENTS
        //BECAUSE WE INITIALLY DO NOT KNOW HOW MANY ELEMENTS ARE COMMON
        ArrayList<Integer> list = new ArrayList<>();
        while (i < nums1.length && j < nums2.length) {
            //IF COMMON ELEMENT FOUND ADD IT INTO THE LIST
            if (nums1[i] == nums2[j]) {
                list.add(nums1[i]);
                i++;
                j++;
            } else //ELSE MOVE THE POINTERS ACCORDINGLY
            if (nums1[i] < nums2[j]) {
                i++;
            } else {
                j++;
            }
        }
        //ADDING THE LIST ELEMENTS TO THE ARRAY!
        int[] ans = new int[list.size()];
        for (int k = 0; k < list.size(); k++) {
            ans[k] = list.get(k);
        }
        //RETURN THE FINAL ARRAY
        return ans;
    }
}
