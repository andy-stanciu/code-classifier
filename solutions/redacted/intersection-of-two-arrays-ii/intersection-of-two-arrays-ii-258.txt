class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        // Frequency count array for nums1 elements
        int[] count = new int[1001];
        // List to store intersection elements
        List<Integer> var1 = new ArrayList<>();
        // Populate the frequency count array with nums1 elements
        for (int num : nums1) {
            count[num]++;
        }
        // Check nums2 elements against the frequency count array
        for (int num : nums2) {
            if (count[num] > 0) {
                var1.add(num);
                count[num]--;
            }
        }
        // Convert the list to an array and return
        int[] var2 = new int[var1.size()];
        for (int i = 0; i < var1.size(); i++) {
            var2[i] = var1.get(i);
        }
        return var2;
    }
}
