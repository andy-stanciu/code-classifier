class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        //Create an ArrayList to store the common elements easily..
        List<Integer> ans = new ArrayList<>();
        //Traverse any 1 array and store the count of the elements in it..
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums1.length; i++) {
            int x = nums1[i];
            map.put(x, map.getOrDefault(x, 0) + 1);
        }
        //Now traverse the 2nd array to find out which elements are common with the 1st array.
        for (int i = 0; i < nums2.length; i++) {
            int y = nums2[i];
            if (map.containsKey(y)) {
                //A common element found
                //Decrease its count in the map by 1.
                map.put(y, map.get(y) - 1);
                if (map.get(y) == 0) {
                    //If the count becomes 0,remove the element from map.
                    map.remove(y);
                }
                //Add the common element to the ArrayList.
                ans.add(y);
            }
        }
        //Finally transfer all elements of ArrayList to an Array.
        int[] answer = new int[ans.size()];
        for (int i = 0; i < ans.size(); i++) {
            answer[i] = ans.get(i);
        }
        return answer;
    }
}
