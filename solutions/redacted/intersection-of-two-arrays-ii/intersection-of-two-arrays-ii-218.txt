class Solution {

    public int[] intersect(int[] nums1, int[] nums2) {
        Map<Integer, Integer> ans = new HashMap<>();
        for (int n : nums1) {
            ans.put(n, ans.getOrDefault(n, 0) + 1);
        }
        List<Integer> var1 = new ArrayList<>();
        for (int n2 : nums2) {
            if (ans.containsKey(n2) && ans.get(n2) >= 1) {
                var1.add(n2);
                ans.put(n2, ans.getOrDefault(n2, 0) - 1);
                ;
            }
        }
        int[] result = new int[var1.size()];
        for (int i = 0; i < var1.size(); i++) {
            result[i] = var1.get(i);
        }
        return result;
    }
}
