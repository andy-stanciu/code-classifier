class Solution {

    public int[] method1(int[] nums, int target) {
        int[] num1 = new int[2];
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[i] + nums[j] == target) {
                    num1[0] = i;
                    num1[1] = j;
                    return num1;
                }
            }
        }
        return num1;
    }
}
