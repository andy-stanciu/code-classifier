class Solution {

    public int[] method1(int[] nums, int target) {
        // Start from the first two numbers
        int i = 0, j = 1;
        int length = nums.length;
        while (nums[i] + nums[j] != target) {
            // Keep checking pairs until we find the right one
            // Since, we enter the loop, then prev sum is not desired, Move to the next number
            j++;
            if (j >= length) {
                // If we reach the end of the array for second position, move to the next number for first position
                i++;
                j = i + 1;
            }
            if (// If we\'ve checked all numbers and still haven\'t found the sum, stop
            i >= length)
                break;
        }
        // Return the indices of the pair that adds up to the target
        return new int[] { i, j };
    }
}
