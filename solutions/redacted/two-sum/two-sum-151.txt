class Solution {

    public int[] method1(int[] nums, int target) {
        Map<Integer, Integer> lookUp = new HashMap<>();
        int indexOne = 0, var1 = 0;
        for (int num : nums) {
            if (lookUp.containsKey(target - num)) {
                var1 = lookUp.get(target - num);
                break;
            }
            lookUp.put(num, indexOne);
            indexOne++;
        }
        return new int[] { indexOne, var1 };
    }
}
