class Solution {

    public int[] method1(int[] nums, int target) {
        Map<Integer, Integer> hashMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int num = nums[i];
            int potentialValue = target - num;
            if (hashMap.containsKey(potentialValue)) {
                return new int[] { i, hashMap.get(potentialValue) };
            } else {
                hashMap.put(num, i);
            }
        }
        return new int[] {};
    }
}
