class Solution {

    public int[] method1(int[] nums, int target) {
        // Initialize an array to store the result with two indices.
        int[] ans = new int[2];
        // Get the length of the input array.
        int n = nums.length;
        // Nested loop to iterate through all pairs of elements in the array.
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                // Check if the sum of elements at indices i and j equals the target.
                if (nums[j] == target - nums[i]) {
                    // Store the first index in the result array.
                    ans[0] = i;
                    // Store the second index in the result array.
                    ans[1] = j;
                    // Return the result array and exit the function.
                    return ans;
                }
            }
        }
        // If no solution is found, return the initialized \'ans\' array.
        return ans;
    }
}
