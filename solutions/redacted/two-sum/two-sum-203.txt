class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> hashstore = new HashMap<>();
        //Array to return
        int[] ret = new int[2];
        for (int i = 0; i < nums.length; i++) {
            int required = target - nums[i];
            if (hashstore.containsKey(required)) {
                //Check if there is a number in the Hashmap so that: nums[i]+x=target
                ret[0] = hashstore.get(required);
                ret[1] = i;
                return ret;
            }
            //putting the values of the Array as keys, because in a HashMap we cant search for Values
            hashstore.put(nums[i], i);
        }
        return null;
    }
}
