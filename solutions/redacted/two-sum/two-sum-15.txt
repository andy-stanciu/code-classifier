class Solution {

    public int[] method1(int[] nums, int target) {
        int s = nums.length;
        HashMap<Integer, Integer> mp = new HashMap<>();
        for (int i = 0; i < s; i++) {
            //getting the value we need by subtracting the target with the current value
            int n = target - nums[i];
            if (//if the map already contains the "n" we are looking for
            mp.containsKey(n)) {
                //print there indexes
                return new int[] { i, mp.get(n) };
            }
            //putting the value and its index onto the map
            mp.put(nums[i], i);
        }
        return new int[0];
    }
}
