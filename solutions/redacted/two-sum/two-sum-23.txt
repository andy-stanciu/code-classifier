class Solution {

    public int[] method1(int[] nums, int target) {
        for (int i = 0; i < nums.length - 1; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                int var1 = nums[i] + nums[j];
                if (var1 == target) {
                    return new int[] { i, j };
                }
            }
        }
        return new int[] {};
    }
}
