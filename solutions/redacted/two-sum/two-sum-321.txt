class Solution {

    public int[] method1(int[] nums, int target) {
        for (int row = 0; row < nums.length; row++) {
            int value = nums[row];
            int var1 = 0;
            for (int col = row + 1; col < nums.length; col++) {
                var1 = value + nums[col];
                if (target == var1) {
                    return new int[] { row, col };
                }
            }
        }
        return new int[] { 0 };
    }
}
