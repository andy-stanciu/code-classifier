class Solution {

    public int[] method1(int[] nums, int target) {
        int n = nums.length;
        int[] ans = new int[2];
        HashMap<Integer, Integer> has = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (has.containsKey(target - nums[i])) {
                ans[0] = i;
                ans[1] = has.get(target - nums[i]);
                return ans;
            }
            has.put(nums[i], i);
        }
        return ans;
    }
}
