class Solution {

    public int[] method1(int[] nums, int target) {
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                // Change \'num\' to \'nums\' and j to i+1 to avoid duplicates
                if (nums[i] + nums[j] == target) {
                    // Use \'new int[]{}\' for returning an array
                    return new int[] { i, j };
                }
            }
        }
        // Same here
        return new int[] { -1, -1 };
    }
}
