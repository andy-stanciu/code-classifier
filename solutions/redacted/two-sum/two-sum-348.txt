class Solution {

    public int[] method1(int[] nums, int target) {
        //used to store and return array indices
        int[] indices = new int[2];
        //nested for loop to compare all possible index value combinations
        for (int x = 0; x < nums.length; x++) {
            for (int y = x + 1; y < nums.length; y++) {
                if (nums[x] + nums[y] == target) {
                    indices[0] = x;
                    indices[1] = y;
                    break;
                }
            }
        }
        return indices;
    }
}
