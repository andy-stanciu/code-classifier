class Solution {

    public int[] method1(int[] nums, int target) {
        // Nested loops to check all possible pairs.
        for (int i = 1; i < nums.length; ++i) {
            for (int j = i; j < nums.length; ++j) {
                // Calculate the sum of elements at indices j-i and j.
                int var1 = nums[j - i] + nums[j];
                // Check if the calculated sum matches the target.
                if (var1 == target) {
                    // If a match is found, return an array with the indices of the two numbers.
                    return new int[] { j - i, j };
                }
            }
        }
        // If no valid pair is found, return null to indicate that there is no solution.
        return null;
    }
}
