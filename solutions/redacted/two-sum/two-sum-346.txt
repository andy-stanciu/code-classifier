class Solution {

    public int[] method1(int[] nums, int target) {
        int[] ans;
        ans = new int[2];
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] + nums[j] == target) {
                    ans[0] = j;
                    ans[1] = i;
                    break;
                }
            }
        }
        return (ans);
    }
}
