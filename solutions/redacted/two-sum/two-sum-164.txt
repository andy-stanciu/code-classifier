class Solution {

    public int[] method1(int[] a, int t) {
        int n = a.length;
        int[] x = new int[2];
        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                if (a[i] + a[j] == t) {
                    x[0] = i;
                    x[1] = j;
                }
            }
        }
        return x;
    }
}
