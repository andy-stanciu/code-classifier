class Solution {

    public int[] method1(int[] nums, int target) {
        // we are using hashmap to store the number value and its index as "key value pair"
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (hm.containsKey(target - nums[i])) {
                // we are checking whether a key exists in hashmap
                // if yes we will return it
                return new int[] { hm.get(target - nums[i]), i };
            }
            // else we will store in hashmap
            hm.put(nums[i], i);
        }
        // worst case to target sum is found then just send null
        return null;
    }
}
