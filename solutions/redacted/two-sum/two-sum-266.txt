class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> ans = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int num = nums[i];
            if (ans.containsKey(target - num)) {
                //return the indices
                return new int[] { i, ans.get(target - num) };
            }
            ans.put(num, i);
        }
        // No Solution Found
        return null;
    }
}
