class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> hs = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (hs.containsKey(nums[i])) {
                return new int[] { i, hs.get(nums[i]) };
            }
            hs.put(target - nums[i], i);
        }
        return new int[] { 0, 0 };
    }
}
