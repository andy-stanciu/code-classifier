class Solution {

    public int[] method1(int[] nums, int target) {
        int diff = 0;
        HashMap<Integer, Integer> hash = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            diff = target - nums[i];
            if (hash.containsKey(diff)) {
                return new int[] { i, hash.get(diff) };
            } else
                hash.put(nums[i], i);
        }
        return null;
    }
}
