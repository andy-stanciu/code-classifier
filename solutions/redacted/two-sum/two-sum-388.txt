class Solution {

    public int[] method1(int[] nums, int target) {
        // Create a map to store the values and their indices
        Map<Integer, Integer> map = new HashMap<>();
        // Loop through the array and add each value and index to the map
        for (int i = 0; i < nums.length; i++) {
            int value = nums[i];
            map.put(value, i);
        }
        // Loop through the array again and check if the difference is in the map
        for (int i = 0; i < nums.length; i++) {
            int value = nums[i];
            int diff = target - value;
            if (map.containsKey(diff) && map.get(diff) != i) {
                // If the difference is in the map, return the indices
                return new int[] { i, map.get(diff) };
            }
        }
        // If no indices were found, return an empty array
        return new int[0];
    }
}
