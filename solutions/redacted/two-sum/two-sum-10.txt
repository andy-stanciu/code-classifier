class Solution {

    public int[] method1(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            // finding the value we need to check in the map if we get duplicate element which is the answer then the value in the map wont be overriden if the answer is not the duplicate element then it is ignored
            int complement = target - nums[i];
            if (map.containsKey(complement)) {
                return new int[] { map.get(complement), i };
            }
            map.put(nums[i], i);
        }
        return new int[] {};
    }
}
