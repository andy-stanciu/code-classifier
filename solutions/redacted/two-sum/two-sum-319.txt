class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> inverses = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (inverses.containsKey(nums[i])) {
                return new int[] { inverses.get(nums[i]), i };
            }
            inverses.put(target - nums[i], i);
        }
        return new int[] { -1, -1 };
    }
}
