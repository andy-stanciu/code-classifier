class Solution {

    public int[] method1(int[] nums, int target) {
        // to store the indices of the two numbers
        int[] ans = new int[2];
        for (int i = 0; i < nums.length; i++) {
            // loop through each element in the array
            for (int j = i + 1; j < nums.length; j++) {
                // loop through the rest of the array to find the other number
                // calculate the sum of the two numbers
                int var1 = nums[i] + nums[j];
                if (var1 == target) {
                    // check if the sum equals the target value
                    // store the indices of the two numbers
                    ans[0] = i;
                    ans[1] = j;
                    // return the answer
                    return ans;
                }
            }
        }
        // if no such numbers exist, return an empty vector
        return ans;
    }
}
