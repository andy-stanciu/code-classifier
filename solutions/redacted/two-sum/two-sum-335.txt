class Solution {

    public int[] method1(int[] nums, int target) {
        // Create a HashMap to store elements from the nums array as keys and their indices as values.
        HashMap<Integer, Integer> hash = new HashMap<>();
        // Iterate through the nums array.
        for (int i = 0; i < nums.length; i++) {
            // Calculate the difference needed to achieve the target sum.
            int diff = target - nums[i];
            // Check if the difference is present in the HashMap.
            if (hash.containsKey(diff)) {
                // If found, return the indices of the two numbers that sum up to the target.
                // The first index is retrieved from the HashMap, and the second index is the current loop variable \'i\'.
                return new int[] { hash.get(diff), i };
            }
            // If the difference is not found, put the current number and its index into the HashMap.
            hash.put(nums[i], i);
        }
        // If no valid pair is found, return null to indicate that there is no solution.
        return null;
    }
}
