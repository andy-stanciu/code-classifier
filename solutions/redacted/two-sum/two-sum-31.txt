class Solution {

    public int[] method1(int[] nums, int target) {
        int i, j, m;
        m = 0;
        int l = nums.length;
        int var1 = 0;
        for (i = 0; i < l; i++) {
            m = i;
            for (j = i + 1; j < l; j++) {
                var1 = (nums[i] + nums[j]);
                if (var1 == target) {
                    return new int[] { m, j };
                }
            }
        }
        throw new IllegalArgumentException("no match found");
    }
}
