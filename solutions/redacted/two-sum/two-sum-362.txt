class Solution {

    public int[] method1(int[] nums, int target) {
        HashMap<Integer, Integer> prevMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (prevMap.containsKey(nums[i])) {
                return new int[] { prevMap.get(nums[i]), i };
            }
            prevMap.put(target - nums[i], i);
        }
        return new int[] {};
    }
}
