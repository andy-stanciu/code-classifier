class Solution {

    public int[] method1(int[] nums, int target) {
        int[] a = new int[2];
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (i != j) {
                    if (nums[i] + nums[j] == target) {
                        a = new int[] { i, j };
                        return a;
                    }
                }
            }
        }
        return a = new int[] { -1 };
    }
}
