class Solution {

    public int[] method1(int[] nums, int target) {
        /*Brute Force Solution time coplexity really matters we need to loop
        on our array one time and get rid of the nested for loop
        we will be using a Hashmap that stores key value 
        */
        // for(int i = 0; i < nums.length; i++){
        //     for(int j = i+1;j < nums.length; j++){
        //         if(nums[i]+nums[j] == target){
        //             return new int[]{i,j};
        //         }
        //     }
        // }
        // return nums;
        Map<Integer, Integer> complements = new HashMap();
        for (int i = 0; i < nums.length; i++) {
            Integer indexComplement = complements.get(nums[i]);
            if (indexComplement != null) {
                return new int[] { i, indexComplement };
            }
            complements.put(target - nums[i], i);
        }
        return nums;
    }
}
