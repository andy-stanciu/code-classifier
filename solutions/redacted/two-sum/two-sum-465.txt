class Solution {

    public int[] method1(int[] nums, int target) {
        Map<Integer, Integer> indexs = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int numCom = target - nums[i];
            if (indexs.containsKey(numCom)) {
                return new int[] { indexs.get(numCom), i };
            }
            indexs.put(nums[i], i);
        }
        return new int[] {};
    }
}
