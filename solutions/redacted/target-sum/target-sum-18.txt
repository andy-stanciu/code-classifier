class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        if ((var1 - param1) < 0 || (var1 - param1) % 2 == 1) {
            return 0;
        }
        int k = ((var1 - param1) / 2);
        int[] prev = new int[k + 1];
        if (nums[0] == 0) {
            prev[0] = 2;
        } else {
            prev[0] = 1;
        }
        if (nums[0] != 0 && nums[0] <= k) {
            prev[nums[0]] = 1;
        }
        for (int ind = 1; ind < nums.length; ind++) {
            int[] curr = new int[k + 1];
            for (int var2 = 0; var2 <= k; var2++) {
                int pick = 0;
                if (nums[ind] <= var2) {
                    pick = prev[var2 - nums[ind]];
                }
                int notPick = prev[var2];
                curr[var2] = pick + notPick;
            }
            prev = curr;
        }
        return prev[k];
    }
}
