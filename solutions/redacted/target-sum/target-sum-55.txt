class Solution {

    public int func(int[][] dp, int[] nums, int param1, int ind) {
        if (param1 < 0)
            return 0;
        if (ind == 0) {
            if (param1 == 0 && nums[0] == 0)
                return 2;
            if (param1 == 0 || nums[0] == param1)
                return 1;
            return 0;
        }
        if (dp[ind][param1] != -1) {
            return dp[ind][param1];
        }
        int notTake = func(dp, nums, param1, ind - 1);
        int take = 0;
        if (param1 >= nums[ind]) {
            take += func(dp, nums, param1 - nums[ind], ind - 1);
        }
        return dp[ind][param1] = (take + notTake);
    }

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        int var1 = 0;
        for (int x : nums) {
            var1 += x;
        }
        if (var1 - param1 < 0 || (var1 - param1) % 2 == 1)
            return 0;
        int s2 = (var1 - param1) / 2;
        int[][] dp = new int[n][s2 + 1];
        for (int[] row : dp) {
            Arrays.fill(row, -1);
        }
        return func(dp, nums, s2, n - 1);
    }
}
