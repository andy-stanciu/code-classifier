class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, nums.length - 1, param1);
    }

    public int helper(int[] nums, int index, int param1) {
        if (index < 0) {
            if (param1 == 0) {
                return 1;
            } else {
                return 0;
            }
        }
        int plus = helper(nums, index - 1, param1 - nums[index]);
        int minus = helper(nums, index - 1, nums[index] + param1);
        return plus + minus;
    }
}
