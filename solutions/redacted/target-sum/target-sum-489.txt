class Solution {

    public int findCount(int[] nums, int param1, int idx, int param2) {
        if (idx == nums.length) {
            if (param2 == param1)
                return 1;
            return 0;
        }
        return findCount(nums, param1, idx + 1, param2 + nums[idx]) + findCount(nums, param1, idx + 1, param2 - nums[idx]);
    }

    public int method1(int[] nums, int param1) {
        return findCount(nums, param1, 0, 0);
    }
}
