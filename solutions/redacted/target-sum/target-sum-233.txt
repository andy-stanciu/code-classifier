class Solution {

    private int final_ans = 0;

    public int method1(int[] nums, int param1) {
        helper(nums, param1, 0, 0);
        return final_ans;
    }

    private void helper(int[] nums, int param1, int local, int index) {
        if (index == nums.length && local == param1)
            final_ans++;
        if (index == nums.length)
            return;
        helper(nums, param1, local + nums[index], index + 1);
        helper(nums, param1, local - nums[index], index + 1);
    }
}
