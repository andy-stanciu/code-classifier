class Solution {

    int res = 0;

    int t;

    public int method1(int[] nums, int param1) {
        t = param1;
        dfs(nums, 0, 0);
        return res;
    }

    public void dfs(int[] nums, int param2, int index) {
        if (index == nums.length) {
            if (param2 == t) {
                res++;
            }
            return;
        }
        dfs(nums, param2 + nums[index], index + 1);
        dfs(nums, param2 - nums[index], index + 1);
    }
}
