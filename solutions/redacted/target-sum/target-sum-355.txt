class Solution {

    public int method1(int[] nums, int param1) {
        return fun(0, 0, param1, nums);
    }

    public int fun(int index, int param2, int param1, int[] nums) {
        if (index == nums.length) {
            if (param2 == param1)
                return 1;
            return 0;
        }
        return fun(index + 1, param2 - nums[index], param1, nums) + fun(index + 1, param2 + nums[index], param1, nums);
    }
}
