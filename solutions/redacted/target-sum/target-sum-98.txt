class Solution {

    public int count;

    public int method1(int[] nums, int param1) {
        count = 0;
        backtrack(nums, param1, 0, 0);
        return count;
    }

    public void backtrack(int[] nums, int param1, int param2, int index) {
        if (index == nums.length) {
            if (param2 == param1)
                ++count;
        } else {
            backtrack(nums, param1, param2 + nums[index], index + 1);
            backtrack(nums, param1, param2 - nums[index], index + 1);
        }
    }
}
