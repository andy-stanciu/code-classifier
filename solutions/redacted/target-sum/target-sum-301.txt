class Solution {

    public int method1(int[] nums, int param1) {
        int ans = helper(nums, param1, 0, 0);
        return ans;
    }

    int helper(int[] nums, int param1, int index, int param2) {
        if (index >= nums.length && param2 == param1)
            return 1;
        if (index >= nums.length && param2 != param1)
            return 0;
        return helper(nums, param1, index + 1, param2 + nums[index]) + helper(nums, param1, index + 1, param2 - nums[index]);
    }
}
