class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0);
    }

    private int method2(int[] nums, int param1, int idx) {
        if (idx == nums.length) {
            if (param1 == 0)
                return 1;
            return 0;
        }
        return method2(nums, param1 - nums[idx], idx + 1) + method2(nums, param1 + nums[idx], idx + 1);
    }
}
