// Approach - 3 Space optimized
class Solution {

    int mod = (int) (Math.pow(10, 9) + 7);

    public int method1(int[] arr, int param1) {
        int n = arr.length;
        int var1 = 0;
        // Calculate the total sum of elements in the array
        for (int i = 0; i < n; i++) {
            var1 += arr[i];
        }
        // Checking for edge cases
        if (var1 - param1 < 0 || (var1 - param1) % 2 == 1)
            return 0;
        return findWays(arr, (var1 - param1) / 2);
    }

    int findWays(int[] num, int tar) {
        int n = num.length;
        // Create an array to store results of subproblems for the current element
        int[] prev = new int[tar + 1];
        // Initialize the prev array for the first element of the array
        if (num[0] == 0)
            // 2 cases - pick and not pick
            prev[0] = 2;
        else
            // 1 case - not pick
            prev[0] = 1;
        if (num[0] != 0 && num[0] <= tar)
            // 1 case - pick
            prev[num[0]] = 1;
        // Fill the prev array using dynamic programming
        for (int ind = 1; ind < n; ind++) {
            int[] cur = new int[tar + 1];
            for (int param1 = 0; param1 <= tar; param1++) {
                int notTaken = prev[param1];
                int taken = 0;
                if (num[ind] <= param1)
                    taken = prev[param1 - num[ind]];
                cur[param1] = (notTaken + taken) % mod;
            }
            prev = cur;
        }
        return prev[tar];
    }
}
