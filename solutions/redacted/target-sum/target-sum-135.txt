class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        // they adjust the negative and odd value also thats why we calculate separately.
        int diff = var1 - param1;
        if (diff < 0 || diff % 2 != 0) {
            return 0;
        }
        int var2 = diff / 2;
        return method2(nums, var2);
    }

    private int method2(int[] nums, int var2) {
        int n = nums.length;
        int[] dp = new int[var2 + 1];
        dp[0] = 1;
        for (int num : nums) {
            for (int j = var2; j >= num; j--) {
                dp[j] += dp[j - num];
            }
        }
        return dp[var2];
    }
}
