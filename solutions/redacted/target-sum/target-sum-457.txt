class Solution {

    Map<String, Integer> memo = new HashMap<>();

    public int method1(int[] nums, int param1) {
        return dfs(nums, 0, param1, 0);
    }

    public int dfs(int[] nums, int i, int param1, int param2) {
        if (i == nums.length) {
            if (param1 == param2)
                return 1;
            else
                return 0;
        }
        String key = String.valueOf(i) + "," + String.valueOf(param2);
        if (memo.containsKey(key))
            return memo.get(key);
        int add = dfs(nums, i + 1, param1, param2 + nums[i]);
        int minus = dfs(nums, i + 1, param1, param2 - nums[i]);
        memo.put(key, add + minus);
        return memo.get(key);
    }
}
