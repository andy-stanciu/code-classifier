class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, 0, 0, param1);
    }

    private int helper(int[] nums, int index, int param2, int param1) {
        // Base case: when we reach the end of the array
        if (index == nums.length) {
            // Check if we have reached the target sum
            if (param2 == param1) {
                // Return 1 to indicate that we have found a valid combination
                return 1;
            } else {
                // Return 0 to indicate that we have not found a valid combination
                return 0;
            }
        }
        // Recursive case: we can either add or subtract the current number to the current sum
        // Add the current number to the current sum
        int left = helper(nums, index + 1, param2 + nums[index], param1);
        // Subtract the current number from the current sum
        int right = helper(nums, index + 1, param2 - nums[index], param1);
        // Return the sum of the results obtained from the left and right subproblems
        return left + right;
    }
}
