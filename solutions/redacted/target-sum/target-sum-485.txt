class Solution {

    public int method1(int[] nums, int param1) {
        return f(0, 0, param1, nums);
    }

    public int f(int idx, int s, int param1, int[] nums) {
        if (idx == nums.length) {
            if (s == param1)
                return 1;
            return 0;
        }
        s += nums[idx];
        int l = f(idx + 1, s, param1, nums);
        s = s - 2 * nums[idx];
        int r = f(idx + 1, s, param1, nums);
        return l + r;
    }
}
