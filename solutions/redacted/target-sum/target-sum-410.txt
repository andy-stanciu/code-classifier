class Solution {

    int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int n : nums) var1 += n;
        // \u8FD9\u4E24\u79CD\u60C5\u51B5\uFF0C\u4E0D\u53EF\u80FD\u5B58\u5728\u5408\u6CD5\u7684\u5B50\u96C6\u5212\u5206
        if (var1 < Math.abs(param1) || (var1 + param1) % 2 == 1) {
            return 0;
        }
        return subsets(nums, (var1 + param1) / 2);
    }

    //         return dp[n][target];

    int subsets(int[] nums, int param1) {
        int[] dp = new int[param1 + 1];
        dp[0] = 1;
        for (int i = 0; i < nums.length; i++) {
            for (int j = param1; j >= 0; j--) {
                if (j - nums[i] < 0) {
                    dp[j] = dp[j];
                } else {
                    dp[j] = dp[j] + dp[j - nums[i]];
                }
                // if( j - nums[i] >= 0) {
                //     dp[j] = dp[j] + dp[j-nums[i]];
                // }
            }
        }
        return dp[param1];
    }
}
