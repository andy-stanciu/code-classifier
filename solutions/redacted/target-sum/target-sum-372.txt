class Solution {

    public int method1(int[] nums, int param1) {
        return f(nums, nums.length, 0, 0, param1);
    }

    public int f(int[] nums, int n, int i, int param2, int k) {
        if (i == n) {
            if (param2 == k)
                return 1;
            return 0;
        }
        return f(nums, n, i + 1, param2 + nums[i], k) + f(nums, n, i + 1, param2 + (-1 * nums[i]), k);
    }
}
