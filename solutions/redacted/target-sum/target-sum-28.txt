class Solution {

    public int method1(int[] nums, int param1) {
        return rres(nums, nums.length, param1);
    }

    private int rres(int[] A, int N, int param1) {
        if (N == 0 && param1 == 0)
            return 1;
        if (N == 0)
            return 0;
        return rres(A, N - 1, param1 - A[N - 1]) + rres(A, N - 1, param1 + A[N - 1]);
    }
}
