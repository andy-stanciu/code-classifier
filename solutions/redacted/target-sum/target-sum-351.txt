class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = SoA(nums);
        if (var1 < Math.abs(param1))
            return 0;
        // if(sum==target)
        //     return 1;
        if ((param1 + var1) % 2 != 0)
            return 0;
        param1 = Math.abs((param1 + var1) / 2);
        int[][] t = new int[nums.length + 1][param1 + 1];
        for (int j = 0; j < t[0].length; j++) {
            t[0][j] = 0;
        }
        int flag = 0;
        for (int i = 0; i < t.length; i++) {
            if (i != 0 && nums[i - 1] == 0)
                flag++;
            t[i][0] = (int) Math.pow(2, flag);
        }
        for (int i = 1; i < t.length; i++) {
            for (int j = 1; j < t[0].length; j++) {
                if (nums[i - 1] <= j) {
                    t[i][j] = t[i - 1][j] + t[i - 1][j - nums[i - 1]];
                } else
                    t[i][j] = t[i - 1][j];
            }
        }
        return t[t.length - 1][t[0].length - 1];
    }

    int SoA(int[] arr) {
        int var1 = 0;
        for (int num : arr) {
            var1 += num;
        }
        return var1;
    }
}
