class Solution {

    public int method1(int[] nums, int param1) {
        return rec(0, nums, param1, 0);
    }

    int rec(int i, int[] n, int param1, int t) {
        if (i >= n.length) {
            if (param1 == t) {
                return 1;
            } else {
                return 0;
            }
        }
        int su = rec(i + 1, n, param1, t + n[i]);
        int sb = rec(i + 1, n, param1, t - n[i]);
        return su + sb;
    }
}
