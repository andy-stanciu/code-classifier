class Solution {

    public int method1(int[] nums, int param1) {
        //passing an array because it will not be
        int[] count = { 0 };
        //passed as call by value
        method2(nums, param1, 0, count, 0);
        return count[0];
    }

    public void method2(int[] nums, int param1, int ptr, int[] count, int value) {
        //base condition
        if (ptr == nums.length) {
            if (value == param1) {
                count[0]++;
            }
            return;
        }
        //checking both the possibilities for all
        method2(nums, param1, ptr + 1, count, value - nums[ptr]);
        method2(nums, param1, ptr + 1, count, value + nums[ptr]);
    }
}
