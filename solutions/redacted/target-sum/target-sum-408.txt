class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0);
    }

    private int method2(int[] nums, int param1, int i) {
        if (nums.length == i)
            return param1 == 0 ? 1 : 0;
        return method2(nums, param1 + nums[i], i + 1) + method2(nums, param1 - nums[i], i + 1);
    }
}
