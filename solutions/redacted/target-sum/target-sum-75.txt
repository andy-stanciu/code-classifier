class Solution {

    public int result = 0;

    public int method1(int[] nums, int param1) {
        rec(0, nums, 0, param1);
        return result;
    }

    public void rec(int n, int[] nums, int s, int param1) {
        if (n == nums.length) {
            if (s == param1) {
                result++;
            }
            return;
        }
        //can be +
        rec(n + 1, nums, s + nums[n], param1);
        //can be -
        rec(n + 1, nums, s - nums[n], param1);
    }
}
