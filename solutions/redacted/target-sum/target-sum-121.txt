class Solution {

    public int method1(int[] nums, int param1) {
        return calculate(nums, 0, param1);
    }

    private int calculate(int[] nums, int index, int param1) {
        // Base case: If all elements are processed
        if (index == nums.length) {
            // If target is 0, return 1 (valid way)
            return param1 == 0 ? 1 : 0;
        }
        // Recursive cases: add or subtract nums[index] to/from target
        int ways = 0;
        // Add nums[index]
        ways += calculate(nums, index + 1, param1 - nums[index]);
        // Subtract nums[index]
        ways += calculate(nums, index + 1, param1 + nums[index]);
        return ways;
    }
}
