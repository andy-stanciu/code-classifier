class Solution {

    public int method1(int[] nums, int param1) {
        //Sum of array.
        int range = method2(nums);
        if (range < Math.abs(param1) || (range + param1) % 2 != 0)
            return 0;
        // 0 - 1 Knapsack Problem
        int var1 = (range + param1) / 2;
        if (var1 < 0 && var1 > range)
            return 0;
        int n = nums.length;
        int[][] dp = new int[n + 1][var1 + 1];
        for (int i = 0; i <= var1; i++) {
            dp[0][i] = 0;
        }
        for (int i = 0; i <= n; i++) {
            dp[i][0] = 1;
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= var1; j++) {
                int pick = 0, notPick = 0;
                notPick = dp[i - 1][j];
                if (nums[i - 1] <= j) {
                    pick = dp[i - 1][j - nums[i - 1]];
                    dp[i][j] = pick + notPick;
                } else {
                    dp[i][j] = notPick;
                }
            }
        }
        return dp[n][var1];
    }

    int method2(int[] nums) {
        int var2 = 0;
        for (int num : nums) {
            var2 += num;
        }
        return var2;
    }
}
