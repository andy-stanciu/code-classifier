class Solution {

    public static int method1(int[] nums, int param1) {
        //taking array of size 1 for pass by reference
        int[] res = new int[1];
        findPaths(0, nums, param1, 0, res);
        return res[0];
    }

    public static void findPaths(int ind, int[] arr, int param1, int param2, int[] res) {
        //base case - end of array
        if (ind == arr.length) {
            if (param2 == param1)
                res[0]++;
            return;
        }
        //take- +ve path
        findPaths(ind + 1, arr, param1, param2 + arr[ind], res);
        //not take- -ve path
        findPaths(ind + 1, arr, param1, param2 - arr[ind], res);
        return;
    }
}
