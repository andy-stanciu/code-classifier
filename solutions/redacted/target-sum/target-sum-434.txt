class Solution {

    private int knapsackMemoizationUtil2(int[] nums, int index, int param1, int[][] dp) {
        if (index == nums.length) {
            if (param1 == 0) {
                return 1;
            }
            return 0;
        }
        if (dp[index][param1] != -1) {
            return dp[index][param1];
        }
        if (nums[index] > param1) {
            dp[index][param1] = knapsackMemoizationUtil2(nums, index + 1, param1, dp);
        } else {
            dp[index][param1] = knapsackMemoizationUtil2(nums, index + 1, param1 - nums[index], dp) + knapsackMemoizationUtil2(nums, index + 1, param1, dp);
        }
        return dp[index][param1];
    }

    private int memoization2(int[] nums, int diff) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        if ((var1 + diff) % 2 != 0) {
            return 0;
        }
        int param1 = (var1 + diff) / 2;
        if (param1 < 0) {
            return 0;
        }
        int[][] dp = new int[nums.length + 1][param1 + 1];
        for (int[] row : dp) {
            Arrays.fill(row, -1);
        }
        return knapsackMemoizationUtil2(nums, 0, param1, dp);
    }

    private int backtrack(int[] nums, int index, int currValue, int param1) {
        if (index == nums.length) {
            return currValue == param1 ? 1 : 0;
        }
        int ans = backtrack(nums, index + 1, currValue + nums[index], param1);
        ans += backtrack(nums, index + 1, currValue - nums[index], param1);
        return ans;
    }

    private int knapsackTabulation(int[] nums, int param1) {
        int var1 = 0;
        for (int num : nums) {
            var1 += num;
        }
        if ((var1 + param1) % 2 != 0) {
            return 0;
        }
        int s1 = (var1 + param1) / 2;
        if (s1 < 0) {
            return 0;
        }
        int n = nums.length;
        int s2 = (var1 - param1) / 2;
        if (s2 < 0) {
            return 0;
        }
        int[][] dp = new int[nums.length + 1][s2 + 1];
        dp[0][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= s2; j++) {
                if (nums[i - 1] > j) {
                    dp[i][j] = dp[i - 1][j];
                } else {
                    dp[i][j] = dp[i - 1][j] + dp[i - 1][j - nums[i - 1]];
                }
            }
        }
        return dp[n][s2];
    }

    private int knapsackMemoizationUtil(int[] nums, int n, int param1, int[][] dp) {
        if (n == 0) {
            return param1 == 0 ? 1 : 0;
        }
        if (dp[n][param1] != -1) {
            return dp[n][param1];
        }
        if (nums[n - 1] > param1) {
            dp[n][param1] = knapsackMemoizationUtil(nums, n - 1, param1, dp);
        } else {
            dp[n][param1] = knapsackMemoizationUtil(nums, n - 1, param1 - nums[n - 1], dp) + knapsackMemoizationUtil(nums, n - 1, param1, dp);
        }
        return dp[n][param1];
    }

    private int memoization(int[] nums, int diff) {
        int var1 = 0;
        for (int i : nums) {
            var1 += i;
        }
        if ((var1 + diff) % 2 != 0) {
            return 0;
        }
        int param1 = (var1 + diff) / 2;
        if (param1 < 0) {
            return 0;
        }
        int[][] dp = new int[nums.length + 1][param1 + 1];
        for (int[] row : dp) {
            Arrays.fill(row, -1);
        }
        return knapsackMemoizationUtil(nums, nums.length, param1, dp);
    }

    public int method1(int[] nums, int param1) {
        // return backtrack(nums, 0, 0, target);
        // return memoization(nums, target);
        return memoization2(nums, param1);
        // return knapsackTabulation(nums, target);
    }
}
