class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0, 0);
    }

    private int method2(int[] arr, int param1, int counter, int index) {
        if (index == arr.length) {
            if (counter == param1) {
                return 1;
            }
            return 0;
        }
        return method2(arr, param1, counter + arr[index], index + 1) + method2(arr, param1, counter - arr[index], index + 1);
    }
}
