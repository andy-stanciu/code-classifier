class Solution {

    int answer = 0;

    public int method1(int[] nums, int param1) {
        recur(nums, param1, 0);
        return answer;
    }

    public void recur(int[] nums, int param1, int i) {
        if (i >= nums.length && 0 == param1) {
            answer++;
            return;
        } else if (i >= nums.length) {
            return;
        }
        recur(nums, param1 + (nums[i] * (-1)), i + 1);
        recur(nums, param1 + (nums[i]), i + 1);
    }
}
