class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0, 0);
    }

    public int method2(int[] nums, int param1, int idx, int param2) {
        if (idx == nums.length) {
            if (param2 == param1)
                return 1;
            return 0;
        }
        int pos = method2(nums, param1, idx + 1, param2 + nums[idx]);
        int neg = method2(nums, param1, idx + 1, param2 -= nums[idx]);
        return pos + neg;
    }
}
