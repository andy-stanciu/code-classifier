class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int n : nums) var1 += n;
        var1 += param1;
        if (var1 < 0 || var1 % 2 == 1)
            return 0;
        var1 /= 2;
        int[] dp = new int[var1 + 1];
        dp[0] = 1;
        for (int n : nums) {
            for (int j = var1; j >= n; j--) {
                dp[j] += dp[j - n];
            }
        }
        return dp[var1];
    }
}
