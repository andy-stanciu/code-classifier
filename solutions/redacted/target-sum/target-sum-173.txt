class Solution {

    int ans = 0;

    public int method1(int[] nums, int param1) {
        solver(nums, 0, 0, param1);
        return ans;
    }

    public void solver(int[] nums, int index, int param2, int param1) {
        if (index == nums.length) {
            if (param2 == param1)
                ans += 1;
            return;
        }
        solver(nums, index + 1, param2 + nums[index], param1);
        solver(nums, index + 1, param2 - nums[index], param1);
    }
}
