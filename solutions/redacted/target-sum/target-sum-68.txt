class Solution {

    int ways = 0;

    public int method1(int[] nums, int param1) {
        way(nums, param1, 0);
        return ways;
    }

    void way(int[] nums, int param1, int idx) {
        if (idx == nums.length) {
            if (param1 == 0)
                ways += 1;
            return;
        }
        way(nums, param1 - nums[idx], idx + 1);
        way(nums, param1 + nums[idx], idx + 1);
    }
}
