class Solution {

    int nWays = 0;

    public int method1(int[] nums, int param1) {
        dfs(0, 0, param1, nums);
        return nWays;
    }

    void dfs(int index, int param2, int param1, int[] nums) {
        if (index == nums.length) {
            if (param2 == param1) {
                nWays++;
            }
            return;
        }
        dfs(index + 1, param2 - nums[index], param1, nums);
        dfs(index + 1, param2 + nums[index], param1, nums);
    }
}
