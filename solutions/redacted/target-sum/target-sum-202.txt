class Solution {

    int total = 0;

    public int helper(int[] nums, int i, int curr, int param1, int[][] dp) {
        if (i == nums.length) {
            if (curr == param1) {
                return 1;
            }
            return 0;
        }
        if (dp[i][curr + total] != Integer.MIN_VALUE)
            return dp[i][curr + total];
        int var1 = 0;
        var1 += helper(nums, i + 1, curr + nums[i], param1, dp);
        var1 += helper(nums, i + 1, curr - nums[i], param1, dp);
        return dp[i][curr + total] = var1;
    }

    public int method1(int[] nums, int param1) {
        for (int i = 0; i < nums.length; i++) {
            total += (nums[i]);
        }
        int[][] dp = new int[nums.length][2 * total + 1];
        for (int[] row : dp) {
            Arrays.fill(row, Integer.MIN_VALUE);
        }
        return helper(nums, 0, 0, param1, dp);
    }
}
