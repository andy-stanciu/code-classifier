class Solution {

    int count = 0;

    private void countWays(int[] nums, int param1, int n, int param2) {
        if (n == nums.length) {
            if (param1 == param2) {
                count++;
            }
        } else {
            countWays(nums, param1 + nums[n], n + 1, param2);
            countWays(nums, param1 - nums[n], n + 1, param2);
        }
    }

    public int method1(int[] nums, int param2) {
        countWays(nums, 0, 0, param2);
        return count;
    }
}
