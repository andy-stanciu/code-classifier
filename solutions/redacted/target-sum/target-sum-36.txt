class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0, 0, 0);
    }

    public static int method2(int[] nums, int param1, int param2, int index, int count) {
        if (index == nums.length) {
            if (param2 == param1) {
                count++;
            }
            return count;
        }
        // adding the positive operator
        int answerPositive = method2(nums, param1, param2 + nums[index], index + 1, count);
        // adding the negative operator
        int answerNegative = method2(nums, param1, param2 - nums[index], index + 1, count);
        return answerPositive + answerNegative;
    }
}
