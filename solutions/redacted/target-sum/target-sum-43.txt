// This is count of subset sum with given diff question in fancy language
class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        int var1 = 0;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
        }
        if ((var1 - param1) % 2 == 1 || (param1 > var1))
            return 0;
        int var2 = (var1 + param1) / 2;
        if (var2 < 0)
            return 0;
        int[][] dp = new int[n + 1][var2 + 1];
        for (int i = 0; i < n + 1; i++) {
            for (int j = 0; j < var2 + 1; j++) {
                if (i == 0)
                    dp[i][j] = 0;
                if (j == 0)
                    dp[i][j] = 1;
            }
        }
        for (int i = 1; i < n + 1; i++) {
            for (int j = 0; j < var2 + 1; j++) {
                if (j >= nums[i - 1]) {
                    dp[i][j] = dp[i - 1][j - nums[i - 1]] + dp[i - 1][j];
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][var2];
    }
}
