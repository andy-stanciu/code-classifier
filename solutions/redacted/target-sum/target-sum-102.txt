class Solution {

    public int method1(int[] nums, int param1) {
        return method1(nums, 0, 0, param1);
    }

    public int method1(int[] nums, int i, int curr, int param1) {
        // base case
        if (i == nums.length) {
            if (curr == param1) {
                return 1;
            }
            return 0;
        }
        return method1(nums, i + 1, curr + nums[i], param1) + method1(nums, i + 1, curr - nums[i], param1);
    }
}
