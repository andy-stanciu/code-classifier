class Solution {

    int ans = 0;

    public int method1(int[] nums, int param1) {
        helper(nums, param1, 0, 0);
        return ans;
    }

    public void helper(int[] nums, int param1, int param2, int idx) {
        if (idx == nums.length) {
            if (param2 == param1)
                ans += 1;
            return;
        }
        int a = nums[idx];
        helper(nums, param1, param2 + a, idx + 1);
        helper(nums, param1, param2 - a, idx + 1);
    }
}
