class Solution {

    private int count = 0;

    public int method1(int[] nums, int param1) {
        help(nums, param1, 0, 0);
        return count;
    }

    private void help(int[] nums, int param1, int index, int total) {
        if (index >= nums.length) {
            if (param1 == total)
                count++;
            return;
        }
        help(nums, param1, index + 1, total - nums[index]);
        help(nums, param1, index + 1, total + nums[index]);
    }
}
