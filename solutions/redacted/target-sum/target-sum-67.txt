class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length, k = 0;
        int ans = method2(nums, param1, n - 1, k);
        return ans;
    }

    public int method2(int[] nums, int param1, int n, int k) {
        if (n < 0 && param1 == k) {
            return 1;
        }
        if (n < 0 && param1 != k)
            return 0;
        return method2(nums, param1, n - 1, k + nums[n]) + method2(nums, param1, n - 1, k - nums[n]);
    }
}
