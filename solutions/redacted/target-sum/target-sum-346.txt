class Solution {

    int count = 0;

    public int method1(int[] nums, int param1) {
        solve(nums, 0, 0, param1);
        return count;
    }

    private void solve(int[] nums, int i, int param2, int param1) {
        if (i == nums.length) {
            if (param2 == param1)
                count += 1;
            return;
        } else {
            solve(nums, i + 1, param2 + nums[i], param1);
            solve(nums, i + 1, param2 - nums[i], param1);
        }
    }
}
