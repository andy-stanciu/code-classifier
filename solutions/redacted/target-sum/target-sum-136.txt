class Solution {

    int res = 0;

    public int method1(int[] nums, int param1) {
        backTracking(nums, param1, 0, 0);
        return res;
    }

    public void backTracking(int[] nums, int param1, int idx, int param2) {
        if (idx == nums.length) {
            if (param2 == param1)
                res++;
            return;
        }
        backTracking(nums, param1, idx + 1, param2 + nums[idx]);
        backTracking(nums, param1, idx + 1, param2 - nums[idx]);
    }
}
