class Solution {

    public int method1(int[] nums, int param1) {
        return ways(nums, 0, param1);
    }

    public int ways(int[] nums, int index, int param1) {
        if (index == nums.length) {
            return param1 == 0 ? 1 : 0;
        }
        int add = ways(nums, index + 1, param1 - nums[index]);
        int subtract = ways(nums, index + 1, param1 + nums[index]);
        return add + subtract;
    }
}
