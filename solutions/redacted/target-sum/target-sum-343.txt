class Solution {

    public int method1(int[] nums, int param1) {
        return method2(nums, param1, 0, 0);
    }

    public int method2(int[] nums, int param1, int param2, int start) {
        if (start == nums.length) {
            return param2 == param1 ? 1 : -1;
        }
        int positiveNum = method2(nums, param1, param2 + nums[start], start + 1);
        int negativeNum = method2(nums, param1, param2 - nums[start], start + 1);
        return (positiveNum < 0 ? 0 : positiveNum) + (negativeNum < 0 ? 0 : negativeNum);
    }
}
