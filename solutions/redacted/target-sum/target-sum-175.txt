class Solution {

    public int method1(int[] nums, int param1) {
        return findWays(nums, 0, 0, param1);
    }

    public int findWays(int[] nums, int index, int param2, int param1) {
        int count = 0;
        if (index == nums.length) {
            if (param2 == param1)
                count++;
            return count;
        }
        count += findWays(nums, index + 1, param2 - nums[index], param1);
        count += findWays(nums, index + 1, param2 + nums[index], param1);
        return count;
    }
}
