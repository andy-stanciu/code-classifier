class Solution {

    public int method1(int[] nums, int param1) {
        return helper(nums, 0, param1, 0);
    }

    private int helper(int[] nums, int ans, int param1, int index) {
        if (index == nums.length && ans != param1)
            return 0;
        if (index == nums.length && ans == param1) {
            return 1;
        }
        int left = helper(nums, ans + nums[index], param1, index + 1);
        int right = helper(nums, ans - nums[index], param1, index + 1);
        return left + right;
    }
}
