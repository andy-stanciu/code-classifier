class Solution {

    public int method1(int[] nums, int param1) {
        // since target is negative also so use hashMap
        int n = nums.length;
        return solve(nums, n - 1, param1);
    }

    public static int solve(int[] nums, int idx, int param1) {
        if (idx == -1)
            return (param1 == 0) ? 1 : 0;
        int minus = solve(nums, idx - 1, param1 + nums[idx]);
        int plus = 0;
        if (Math.abs(param1) >= 0)
            plus = solve(nums, idx - 1, param1 - nums[idx]);
        return minus + plus;
    }
}
