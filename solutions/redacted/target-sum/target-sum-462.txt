class Solution {

    public int method1(int[] nums, int param1) {
        return solve(nums, nums.length - 1, 0, param1);
    }

    private int solve(int[] nums, int i, int param2, int param1) {
        if (i < 0 && param2 == param1)
            return 1;
        else if (i < 0)
            return 0;
        //total number of ways if you added nums[i]
        int positive = solve(nums, i - 1, param2 + nums[i], param1);
        //total number of ways if you subtracted nums[i]
        int negative = solve(nums, i - 1, param2 - nums[i], param1);
        return positive + negative;
    }
}
