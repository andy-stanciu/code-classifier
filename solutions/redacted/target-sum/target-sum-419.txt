class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        System.out.println(n);
        if (n == 1) {
            if (nums[0] == Math.abs(param1))
                return 1;
            else
                return 0;
        }
        int range = 0;
        for (int x : nums) range += x;
        if ((range + param1) % 2 != 0)
            return 0;
        if (range % 2 != 0 && param1 == 0)
            return 0;
        if (range == 0) {
            return (int) Math.pow(2, n);
        }
        if (range == param1) {
            int power = nums.length - 1;
            double exp = Math.pow(2, power);
            return (int) exp;
        }
        int var1 = (range + param1) / 2;
        int[][] dp = new int[n + 1][var1 + 1];
        for (int i = 0; i <= n; i++) dp[i][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= var1; j++) {
                if (nums[i - 1] <= j) {
                    dp[i][j] = dp[i - 1][j - nums[i - 1]] + dp[i - 1][j];
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][var1];
    }
}
