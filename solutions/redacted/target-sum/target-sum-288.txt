class Solution {

    public static int fun(int ind, int param1, int[] arr, int[][] dp) {
        if (ind == 0) {
            if (param1 == 0 && arr[0] == 0)
                return 2;
            if (param1 == 0 || param1 == arr[0])
                return 1;
            return 0;
        }
        if (dp[ind][param1] != -1)
            return dp[ind][param1];
        int notTaken = fun(ind - 1, param1, arr, dp);
        int taken = 0;
        if (arr[ind] <= param1)
            taken = fun(ind - 1, param1 - arr[ind], arr, dp);
        return dp[ind][param1] = (notTaken + taken);
    }

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        int var1 = 0;
        int d = param1;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
        }
        //Checking for edge cases
        if (var1 - d < 0)
            return 0;
        if ((var1 - d) % 2 == 1)
            return 0;
        int s2 = (var1 - d) / 2;
        int[][] dp = new int[n][s2 + 1];
        for (int[] row : dp) Arrays.fill(row, -1);
        return fun(n - 1, s2, nums, dp);
    }
}
