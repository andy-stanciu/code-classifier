class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        int n = nums.length;
        //find sum
        for (int i : nums) {
            var1 += i;
        }
        //s1 - s2 = target
        //s1 + s2 = sum
        //2s1 = target+sum
        //s1  = (target+sum)/2;
        if (((var1 - param1) % 2 == 1) || (param1 > var1))
            return 0;
        int var2 = (var1 - param1) / 2;
        //solution 1
        /*
        int[][] dp = new int[n+1][sum1+1];
       
        dp[0][0] = 1;

        for(int i=1; i<=nums.length;i++){
            for(int j=0; j<=sum1; j++){
                if(nums[i-1] <=j)
                    dp[i][j] = dp[i-1][j] + dp[i-1][j-nums[i-1]];
                else
                    dp[i][j] = dp[i-1][j];
            }

        }
        return dp[nums.length][sum1];
        */
        //Collapse the 2D array into
        int[] dp = new int[var2 + 1];
        dp[0] = 1;
        for (int num : nums) {
            for (int j = var2; j >= 0; j--) {
                if (num <= j)
                    dp[j] = dp[j] + dp[j - num];
            }
        }
        return dp[var2];
    }
}
