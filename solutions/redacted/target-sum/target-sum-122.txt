class Solution {

    public int method1(int[] nums, int param1) {
        // Arrays.sort(nums);
        int a = method2(nums, param1, 0, 0);
        return a;
    }

    public int method2(int[] nums, int param1, int index, int param2) {
        if (param2 == param1 && index == nums.length) {
            return 1;
        }
        if (index >= nums.length && param2 != param1) {
            return 0;
        }
        int a = method2(nums, param1, index + 1, param2 + nums[index]);
        int b = method2(nums, param1, index + 1, param2 - nums[index]);
        return a + b;
    }
}
