class Solution {

    public int method1(int[] nums, int param1) {
        int n = nums.length;
        int var1 = 0;
        for (int i = 0; i < n; i++) var1 += nums[i];
        Map<String, Integer> dp2 = new HashMap<>();
        for (int i = n; i >= 0; i--) {
            for (int j = var1; j >= -var1; j--) {
                if (i == n && param1 == j)
                    dp2.put(i + "_" + j, 1);
                else if (i != n)
                    dp2.put(i + "_" + j, dp2.getOrDefault((i + 1) + "_" + (j + nums[i]), 0) + dp2.getOrDefault((i + 1) + "_" + (j - nums[i]), 0));
            }
        }
        return dp2.get("0_0");
    }
}
