class Solution {

    public int method1(int[] nums, int param1) {
        int var1 = 0;
        for (int i : nums) var1 += i;
        return countNumberOfWays(nums, (param1 + var1) / 2, nums.length - 1);
    }

    public int countNumberOfWays(int[] nums, int param1, int index) {
        if (index == 0) {
            if (nums[0] == 0 && param1 == 0)
                return 2;
            if (param1 == 0)
                return 1;
            if (nums[0] == param1)
                return 1;
            return 0;
        }
        //not pick
        int notPick = countNumberOfWays(nums, param1, index - 1);
        //pick
        int pick = 0;
        if (nums[index] <= param1)
            pick = countNumberOfWays(nums, param1 - nums[index], index - 1);
        return pick + notPick;
    }
}
