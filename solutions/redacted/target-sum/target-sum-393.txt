class Solution {

    public int method1(int[] a, int tar) {
        int xx = f(a.length - 1, tar, a);
        if (a[0] == 0)
            return 2 * xx;
        return xx;
    }

    public int f(int ind, int tar, int[] a) {
        if (ind == 0) {
            if (tar == a[0] || tar == -a[0])
                return 1;
            return 0;
        }
        int pos = f(ind - 1, tar - a[ind], a);
        int neg = f(ind - 1, tar + a[ind], a);
        return pos + neg;
    }
}
