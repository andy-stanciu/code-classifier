class Solution {

    int count = 0;

    public int method1(int[] nums, int param1) {
        dfs(nums, 0, 0, param1);
        return count;
    }

    public void dfs(int[] nums, int index, int param2, int param1) {
        if (index == nums.length) {
            if (param2 == param1)
                count++;
            return;
        }
        int pos = nums[index];
        int neg = -1 * pos;
        //System.out.println(sum+pos+" "+index);
        dfs(nums, index + 1, param2 + pos, param1);
        dfs(nums, index + 1, param2 + neg, param1);
    }
}
