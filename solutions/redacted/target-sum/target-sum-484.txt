class Solution {

    public int method1(int[] nums, int param1) {
        Map<Integer, Integer> var1 = new HashMap();
        var1.put(0, 1);
        for (int num : nums) {
            Map<Integer, Integer> var2 = new HashMap();
            for (int k : var1.keySet()) {
                // Increment for when adding num to subtotal
                int var3 = var2.getOrDefault(k + num, 0);
                var2.put(k + num, var3 + var1.get(k));
                // Increment for when subtracting num from subtotal
                int currSub = var2.getOrDefault(k - num, 0);
                var2.put(k - num, currSub + var1.get(k));
            }
            var1 = var2;
        }
        return var1.getOrDefault(param1, 0);
    }
}
