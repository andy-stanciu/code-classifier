class Solution {

    public int method1(int[] arr, int param1) {
        int n = arr.length;
        int var1 = 0;
        for (int i : arr) var1 += i;
        if (param1 > var1 || param1 < -var1 || (param1 + var1) % 2 != 0)
            return 0;
        int s = (param1 + var1) / 2;
        int[][] t = new int[n + 1][s + 1];
        for (int i = 0; i < t.length; i++) t[i][0] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= s; j++) {
                if (j - arr[i - 1] >= 0)
                    t[i][j] = t[i - 1][j - arr[i - 1]] + t[i - 1][j];
                else
                    t[i][j] = t[i - 1][j];
            }
        }
        return t[n][s];
    }
}
