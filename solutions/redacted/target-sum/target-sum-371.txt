class Solution {

    public int method1(int[] nums, int param1) {
        int total = 0;
        for (int num : nums) total += num;
        if (total < param1)
            return 0;
        if ((total - param1) % 2 != 0)
            return 0;
        int var1 = (total - param1) / 2;
        int var2 = 0;
        int[] dp = new int[var1 + 1];
        dp[0] = 1;
        for (int num : nums) {
            var2 += num;
            for (int i = (Math.min(var2, var1)); i >= num; i--) {
                if (i - num >= 0 && dp[i - num] > 0)
                    dp[i] += dp[i - num];
            }
        }
        return dp[var1];
    }
}
