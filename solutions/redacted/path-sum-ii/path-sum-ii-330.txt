/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans;

    void method1(List<List<Integer>> l, List<Integer> temp, TreeNode root, int param1, int curr) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && curr == param1) {
            l.add(new ArrayList<>(temp));
            return;
        }
        if (root.left != null) {
            temp.add(root.left.val);
            method1(l, temp, root.left, param1, curr + root.left.val);
            temp.remove(temp.size() - 1);
        }
        if (root.right != null) {
            temp.add(root.right.val);
            method1(l, temp, root.right, param1, curr + root.right.val);
            temp.remove(temp.size() - 1);
        }
    }

    public List<List<Integer>> method2(TreeNode root, int param2) {
        ans = new ArrayList<List<Integer>>();
        if (root == null)
            return ans;
        ArrayList<Integer> temp = new ArrayList<>();
        temp.add(root.val);
        method1(ans, temp, root, param2, root.val);
        return ans;
    }
}
