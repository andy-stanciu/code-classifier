/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return result;
        method2(root, 0, param1, new ArrayList<>());
        return result;
    }

    private void method2(TreeNode root, int param2, int param1, List<Integer> param3) {
        if (root == null)
            return;
        param3.add(root.val);
        param2 += root.val;
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                result.add(new ArrayList<>(param3));
            }
        } else {
            method2(root.left, param2, param1, param3);
            method2(root.right, param2, param1, param3);
        }
        param3.remove(param3.size() - 1);
    }
}
