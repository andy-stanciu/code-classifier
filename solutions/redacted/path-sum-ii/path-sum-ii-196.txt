class Solution {

    List<List<Integer>> l = new ArrayList<>();

    List<Integer> t = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        sol(0, root, param1);
        return l;
    }

    public void sol(int param2, TreeNode root, int target) {
        if (root == null)
            return;
        param2 += root.val;
        t.add(root.val);
        if (root.left == null && root.right == null && param2 == target) {
            l.add(new ArrayList<>(t));
            // Backtrack
            t.remove(t.size() - 1);
            return;
        }
        sol(param2, root.left, target);
        sol(param2, root.right, target);
        // Backtrack: remove the last element from the list
        t.remove(t.size() - 1);
    }
}
