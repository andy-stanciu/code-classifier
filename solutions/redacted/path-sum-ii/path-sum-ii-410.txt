/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int target;

    List<List<Integer>> ans;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        this.target = param1;
        ans = new LinkedList();
        backtrack(new LinkedList(), root, 0);
        return ans;
    }

    void backtrack(List<Integer> list, TreeNode root, int param2) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && param2 + root.val == target) {
            ans.add(new LinkedList(list));
            list.remove(list.size() - 1);
            return;
        }
        backtrack(list, root.left, param2 + root.val);
        backtrack(list, root.right, param2 + root.val);
        list.remove(list.size() - 1);
    }
}
