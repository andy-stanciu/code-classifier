/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    List<Integer> ls = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        dfs(root, param1);
        return ans;
    }

    public void dfs(TreeNode root, int param2) {
        if (root == null) {
            return;
        }
        ls.add(root.val);
        if (root.left == null && root.right == null && root.val == param2) {
            ans.add(new ArrayList<>(ls));
        }
        dfs(root.left, param2 - root.val);
        dfs(root.right, param2 - root.val);
        ls.remove(ls.size() - 1);
    }
}
