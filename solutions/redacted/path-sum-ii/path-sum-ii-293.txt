/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> list = new ArrayList<>();
        List<Integer> curr = new ArrayList<>();
        method2(root, param1, list, curr);
        return list;
    }

    public void method2(TreeNode root, int param1, List<List<Integer>> list, List<Integer> curr) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            curr.add(root.val);
            list.add(new ArrayList<Integer>(curr));
            curr.remove(curr.size() - 1);
        }
        curr.add(root.val);
        method2(root.left, param1 - root.val, list, curr);
        method2(root.right, param1 - root.val, list, curr);
        curr.remove(curr.size() - 1);
    }
}
