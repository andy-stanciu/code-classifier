/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void dfs(TreeNode root, List<List<Integer>> ans, LinkedList<Integer> v, int t, int param1) {
        if (root == null)
            return;
        param1 += root.val;
        if (root.left == null && root.right == null) {
            if (param1 == t) {
                v.add(root.val);
                ans.add(new ArrayList<>(v));
                v.removeLast();
            }
            return;
        }
        v.add(root.val);
        dfs(root.left, ans, v, t, param1);
        dfs(root.right, ans, v, t, param1);
        v.removeLast();
    }

    public List<List<Integer>> method1(TreeNode root, int t) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null)
            return ans;
        List<Integer> v = new LinkedList<>();
        dfs(root, ans, new LinkedList<>(), t, 0);
        return ans;
    }
}
