/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var1 = new ArrayList<>();
        List<List<Integer>> result = new ArrayList<>();
        method2(root, param1, var1, result);
        return result;
    }

    public void method2(TreeNode root, int param1, List var1, List<List<Integer>> result) {
        if (root == null)
            return;
        var1.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            result.add(new ArrayList<>(var1));
        }
        method2(root.left, param1 - root.val, var1, result);
        method2(root.right, param1 - root.val, var1, result);
        var1.remove(var1.size() - 1);
    }
}
