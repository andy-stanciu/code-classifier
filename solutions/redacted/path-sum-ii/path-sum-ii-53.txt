/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        helper(root, param1, ans, new ArrayList<>());
        return new ArrayList<>(ans);
    }

    private void helper(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> l) {
        if (root == null)
            return;
        l.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            ans.add(l);
            return;
        }
        helper(root.left, param1 - root.val, ans, new ArrayList<>(l));
        helper(root.right, param1 - root.val, ans, new ArrayList<>(l));
    }
}
