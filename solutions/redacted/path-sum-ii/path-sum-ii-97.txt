class Solution {

    List<List<Integer>> var1 = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return Collections.emptyList();
        List<Integer> list = new ArrayList<>();
        findList(root, 0, param1, list);
        return var1;
    }

    private void findList(TreeNode root, int param2, int param1, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        param2 += root.val;
        if (param2 == param1 && root.left == null && root.right == null) {
            var1.add(new ArrayList<>(list));
        }
        findList(root.left, param2, param1, list);
        findList(root.right, param2, param1, list);
        list.remove(list.size() - 1);
    }
}
