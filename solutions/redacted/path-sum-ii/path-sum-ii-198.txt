/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null) {
            return new ArrayList<List<Integer>>();
        }
        List<Integer> var1 = new ArrayList<Integer>();
        List<List<Integer>> out = new ArrayList<List<Integer>>();
        helper(root, param1, var1, out);
        return out;
    }

    private void helper(TreeNode root, int param1, List<Integer> param2, List<List<Integer>> out) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1 - root.val == 0) {
            param2.add(root.val);
            out.add(param2);
            return;
        }
        param2.add(root.val);
        List<Integer> copy = new ArrayList<>();
        copy.addAll(param2);
        helper(root.left, param1 - root.val, param2, out);
        helper(root.right, param1 - root.val, copy, out);
    }
}
