/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //helper function
    private void dfs(TreeNode node, int target, List<Integer> param1, List<List<Integer>> result) {
        if (node == null)
            return;
        //add that node\'s value to currentPath
        param1.add(node.val);
        //if we reach a leaf node, check if the sum is equal to target
        if (node.left == null && node.right == null && target == node.val)
            result.add(new ArrayList<>(param1));
        else {
            dfs(node.left, target - node.val, param1, result);
            dfs(node.right, target - node.val, param1, result);
        }
        //remove that current path or backtrack
        param1.remove(param1.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        //stores current paths
        List<Integer> param1 = new ArrayList<>();
        //stores all the paths that have sum equal to target
        List<List<Integer>> result = new ArrayList<>();
        dfs(root, param2, param1, result);
        return result;
    }
}
