/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        helper(root, 0, param1, result, new ArrayList<>());
        return result;
    }

    void helper(TreeNode root, int ans, int param1, List<List<Integer>> result, List<Integer> list) {
        if (root == null)
            return;
        int var1 = ans + root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && var1 == param1) {
            result.add(new ArrayList<>(list));
        }
        helper(root.left, var1, param1, result, list);
        helper(root.right, var1, param1, result, list);
        list.remove(list.size() - 1);
    }
}
