/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static void method1(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> temp) {
        // if root==null return...
        if (root == null)
            return;
        temp.add(root.val);
        if (root.left == null && root.right == null && root.val == param1)
            ans.add(new ArrayList<>(temp));
        // checking on the left subtree of the tree...
        method1(root.left, param1 - root.val, ans, temp);
        // checking on the right subtree of the tree...
        method1(root.right, param1 - root.val, ans, temp);
        // backtracking step...
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> method2(TreeNode root, int param2) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        // calling the paths function to get all the answers...
        method1(root, param2, ans, temp);
        return ans;
        // Time Complexity : O(N)
        // Space Complexity : O(N)
    }
}
