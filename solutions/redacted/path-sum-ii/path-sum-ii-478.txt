class Solution {

    static List<List<Integer>> finalResult;

    public void helper(TreeNode root, String param1, int param2, int target) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            param2 += root.val;
            if (param2 == target) {
                param1 += root.val;
                String[] var1 = param1.split(" ");
                List<Integer> tempList = new ArrayList<>();
                for (int i = 0; i < var1.length; i++) {
                    tempList.add(Integer.valueOf(var1[i]));
                }
                finalResult.add(tempList);
            }
        }
        helper(root.left, param1 + root.val + " ", param2 + root.val, target);
        helper(root.right, param1 + root.val + " ", param2 + root.val, target);
    }

    public List<List<Integer>> method1(TreeNode root, int param3) {
        finalResult = new ArrayList<>();
        helper(root, "", 0, param3);
        return finalResult;
    }
}
