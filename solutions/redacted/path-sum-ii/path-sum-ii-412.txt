/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> subResult = new ArrayList<>();
        Integer value;
        method2(root, result, subResult, 0, param1);
        return result;
    }

    private void method2(TreeNode node, List<List<Integer>> result, List<Integer> subResult, Integer value, int param1) {
        if (node == null)
            return;
        if (node.left == null && node.right == null) {
            value += node.val;
            if (value == param1) {
                subResult.add(node.val);
                result.add(subResult);
            }
        } else {
            value += node.val;
            subResult.add(node.val);
            method2(node.left, result, new ArrayList<>(subResult), value, param1);
            method2(node.right, result, new ArrayList<>(subResult), value, param1);
        }
    }
}
