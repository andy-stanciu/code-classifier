/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var1 = new ArrayList<>();
        List<List<Integer>> var2 = new ArrayList<>();
        helper(root, param1, var1, var2);
        return var2;
    }

    void helper(TreeNode root, int param1, List<Integer> var1, List<List<Integer>> var2) {
        if (root == null) {
            return;
        }
        //add the node to the list
        var1.add(root.val);
        //when i am at the leaf node add it to paths
        if (root.val == param1 && root.left == null && root.right == null) {
            var2.add(new ArrayList<>(var1));
        } else {
            //recursion call
            helper(root.left, param1 - root.val, var1, var2);
            helper(root.right, param1 - root.val, var1, var2);
        }
        //do backtrack because i am updating the same list instead to creating a new one
        var1.remove(var1.size() - 1);
    }
}
