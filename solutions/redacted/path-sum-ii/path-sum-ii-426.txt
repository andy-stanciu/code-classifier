/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> temp = new ArrayList<>();
        dfs(root, param1, 0, temp);
        return result;
    }

    public void dfs(TreeNode root, int param1, int param2, List<Integer> temp) {
        if (root == null)
            return;
        temp.add(root.val);
        param2 += root.val;
        //if leaf node
        if (root.left == null && root.right == null) {
            if (param2 == param1)
                result.add(new ArrayList<>(temp));
        } else {
            dfs(root.left, param1, param2, temp);
            dfs(root.right, param1, param2, temp);
        }
        temp.remove(temp.size() - 1);
    }
}
