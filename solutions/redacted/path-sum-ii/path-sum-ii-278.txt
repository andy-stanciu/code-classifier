/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result = new ArrayList<>();
        helper(root, 0, param1, new ArrayList<>());
        return result;
    }

    private void helper(TreeNode root, int param2, int param1, List<Integer> param3) {
        if (root == null)
            return;
        param2 += root.val;
        // Action
        param3.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                result.add(new ArrayList<>(param3));
            }
        }
        // Recursion
        helper(root.left, param2, param1, param3);
        helper(root.right, param2, param1, param3);
        // BackTrack
        param3.remove(param3.size() - 1);
    }
}
