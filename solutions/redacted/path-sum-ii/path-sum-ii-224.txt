/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(root, param1, new LinkedList());
        return list;
    }

    public void method2(TreeNode root, int param1, LinkedList<Integer> current) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && root.val == param1) {
            current.add(root.val);
            list.add(current);
            return;
        }
        current.add(root.val);
        method2(root.left, param1 - root.val, new LinkedList(current));
        method2(root.right, param1 - root.val, new LinkedList(current));
    }
}
