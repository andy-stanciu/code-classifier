class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList();
        method2(root, param1, ans, new ArrayList<Integer>());
        return ans;
    }

    private void method2(TreeNode node, int param2, List<List<Integer>> ans, List<Integer> param3) {
        if (node == null)
            return;
        param3.add(node.val);
        if (node.left == null && node.right == null && param2 - node.val == 0) {
            List<Integer> list = new ArrayList(param3);
            ans.add(list);
        }
        method2(node.left, param2 - node.val, ans, param3);
        method2(node.right, param2 - node.val, ans, param3);
        param3.remove(param3.size() - 1);
    }
}
