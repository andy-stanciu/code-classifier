/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        List<Integer> acc = new ArrayList<Integer>();
        helper(root, acc, res, param1);
        return res;
    }

    void helper(TreeNode node, List<Integer> acc, List<List<Integer>> res, int param1) {
        if (node == null)
            return;
        param1 -= node.val;
        acc.add(node.val);
        if (node.left == node.right) {
            if (param1 == 0)
                res.add(new ArrayList<Integer>(acc));
            acc.remove(acc.size() - 1);
            return;
        }
        helper(node.left, acc, res, param1);
        helper(node.right, acc, res, param1);
        acc.remove(acc.size() - 1);
    }
}
