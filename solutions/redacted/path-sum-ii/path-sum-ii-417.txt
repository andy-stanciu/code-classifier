/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> answer = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> sl = new ArrayList<>();
        if (root == null)
            return answer;
        helper(root, 0, param1, sl);
        return answer;
    }

    public void helper(TreeNode root, int param2, int ts, List<Integer> sl) {
        if (root.left == null && root.right == null) {
            if (param2 + root.val == ts) {
                sl.add(root.val);
                answer.add(new ArrayList<>(sl));
                sl.remove(sl.size() - 1);
            }
            return;
        }
        sl.add(root.val);
        param2 += root.val;
        if (root.left != null)
            helper(root.left, param2, ts, sl);
        if (root.right != null)
            helper(root.right, param2, ts, sl);
        sl.remove(sl.size() - 1);
        param2 -= root.val;
    }
}
