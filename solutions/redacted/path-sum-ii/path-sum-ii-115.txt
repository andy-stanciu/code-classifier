/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> finallist = new ArrayList<>();

    public void method1(TreeNode root, int param1, List<Integer> list1) {
        if (root == null) {
            return;
        }
        list1.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            finallist.add(new ArrayList<>(list1));
        } else {
            method1(root.left, param1 - root.val, list1);
            method1(root.right, param1 - root.val, list1);
        }
        list1.remove(list1.size() - 1);
        return;
    }

    public List<List<Integer>> method2(TreeNode root, int param1) {
        method1(root, param1, new ArrayList<>());
        return finallist;
    }
}
