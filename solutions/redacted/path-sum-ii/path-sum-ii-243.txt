/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> li = new ArrayList<>();

    public List<List<Integer>> list(TreeNode root, int param1, int target, List<Integer> ans) {
        if (root == null)
            return li;
        // List<Integer> l = new ArrayList<>();
        param1 += root.val;
        ans.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == target) {
                li.add(new ArrayList<>(ans));
            } else //backtracking
            {
                param1 -= root.val;
            }
        }
        list(root.left, param1, target, ans);
        list(root.right, param1, target, ans);
        ans.remove(ans.size() - 1);
        //to move to other values recursively
        return li;
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        if (root == null)
            return li;
        return list(root, 0, param2, new ArrayList<>());
    }
}
