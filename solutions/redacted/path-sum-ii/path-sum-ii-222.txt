/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void helper(TreeNode root, int target, ArrayList<Integer> ds, List<List<Integer>> ans) {
        if (root == null)
            return;
        if (target == root.val && root.left == null && root.right == null) {
            ds.add(root.val);
            ans.add(new ArrayList<>(ds));
            ds.remove(ds.size() - 1);
            return;
        }
        ds.add(root.val);
        helper(root.left, target - root.val, ds, ans);
        helper(root.right, target - root.val, ds, ans);
        ds.remove(ds.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        helper(root, param1, new ArrayList<>(), ans);
        return ans;
    }
}
