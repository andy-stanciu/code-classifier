/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> li = new ArrayList<>();
        ArrayList<Integer> list = new ArrayList<>();
        method2(root, param1, li, list);
        return li;
    }

    public void method2(TreeNode root, int param1, List<List<Integer>> li, ArrayList<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == root.val) {
                li.add(new ArrayList<>(list));
            }
        } else {
            method2(root.left, param1 - root.val, li, list);
            method2(root.right, param1 - root.val, li, list);
        }
        list.remove(list.size() - 1);
    }
}
