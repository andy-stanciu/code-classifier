/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int target) {
        List<List<Integer>> result = new ArrayList<>();
        method1(root, target, result, new ArrayList<Integer>());
        return result;
    }

    private void method1(TreeNode root, int param1, List<List<Integer>> result, List<Integer> curr) {
        if (root == null)
            return;
        curr.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            result.add(new ArrayList<>(curr));
        }
        method1(root.left, param1 - root.val, result, curr);
        method1(root.right, param1 - root.val, result, curr);
        curr.remove(curr.size() - 1);
    }
}
