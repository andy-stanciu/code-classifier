/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        boolean a = solve(root, 0, param1, res, temp);
        return res;
    }

    public boolean solve(TreeNode root, int param2, int target, List<List<Integer>> res, List<Integer> temp) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null) {
            param2 += root.val;
            temp.add(root.val);
            if (param2 == target) {
                res.add(new ArrayList<>(temp));
                temp.remove(temp.size() - 1);
                return true;
            } else {
                temp.remove(temp.size() - 1);
                return false;
            }
        }
        temp.add(root.val);
        boolean left = solve(root.left, param2 + root.val, target, res, temp);
        boolean right = solve(root.right, param2 + root.val, target, res, temp);
        temp.remove(temp.size() - 1);
        return left || right;
    }
}
