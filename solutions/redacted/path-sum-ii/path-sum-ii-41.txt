/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
public class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        method2(root, result, var1, param1, 0);
        return result;
    }

    private void method2(TreeNode root, List<List<Integer>> result, List<Integer> var1, int target, int method2) {
        if (root == null) {
            return;
        }
        //add root value to currentPath
        var1.add(root.val);
        //Calculate sum
        method2 += root.val;
        //check whether both roots are null and sum equal to target
        if (root.left == null && root.right == null && method2 == target) {
            result.add(new ArrayList<>(var1));
            //if yes add currentPath to result List
        }
        method2(root.left, result, var1, target, method2);
        method2(root.right, result, var1, target, method2);
        // To BackTrack the element
        var1.remove(var1.size() - 1);
    }
}
