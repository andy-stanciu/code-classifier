class Solution {

    List<List<Integer>> listOfLists = new ArrayList();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> list = new ArrayList();
        helper(root, param1, list);
        return listOfLists;
    }

    public void helper(TreeNode root, int param1, List list) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            listOfLists.add(new ArrayList<>(list));
        }
        helper(root.left, param1 - root.val, list);
        helper(root.right, param1 - root.val, list);
        list.remove(list.size() - 1);
    }
}
