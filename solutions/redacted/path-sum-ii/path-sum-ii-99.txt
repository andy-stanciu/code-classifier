/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        dfs(result, new ArrayList<>(), root, param1);
        return result;
    }

    private void dfs(List<List<Integer>> result, List<Integer> dfsList, TreeNode root, int param1) {
        if (root == null)
            return;
        dfsList.add(root.val);
        param1 -= root.val;
        if (root.left == null && root.right == null) {
            //System.out.println(dfsList);
            if (param1 == 0)
                result.add(new ArrayList<>(dfsList));
            param1 += root.val;
        }
        dfs(result, dfsList, root.left, param1);
        dfs(result, dfsList, root.right, param1);
        dfsList.remove(dfsList.size() - 1);
    }
}
