class Solution {

    List<List<Integer>> ans;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        ans = new ArrayList();
        solve(root, param1, new ArrayList());
        return ans;
    }

    public void solve(TreeNode root, int param2, List<Integer> temp) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && root.val == param2) {
            temp.add(root.val);
            ans.add(new ArrayList(temp));
            temp.remove(temp.size() - 1);
        }
        temp.add(root.val);
        solve(root.left, param2 - root.val, temp);
        solve(root.right, param2 - root.val, temp);
        temp.remove(temp.size() - 1);
    }
}
