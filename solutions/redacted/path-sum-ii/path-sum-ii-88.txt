/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void solve(TreeNode root, int target, int param1, List<List<Integer>> res, List<Integer> arr) {
        if (root == null) {
            return;
        }
        param1 += root.val;
        arr.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == target) {
                res.add(new ArrayList<>(arr));
            }
        } else {
            solve(root.left, target, param1, res, arr);
            solve(root.right, target, param1, res, arr);
        }
        arr.remove(arr.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null) {
            return res;
        }
        solve(root, param2, 0, res, new ArrayList<>());
        return res;
    }
}
