class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        helper(root, param1, new ArrayList<>(), res);
        return res;
    }

    private void helper(TreeNode node, int param1, List<Integer> param2, List<List<Integer>> res) {
        if (node == null)
            return;
        if (node.val == param1 && node.left == null && node.right == null) {
            param2.add(node.val);
            res.add(new ArrayList<>(param2));
            param2.remove(param2.size() - 1);
            return;
        }
        param2.add(node.val);
        helper(node.left, param1 - node.val, param2, res);
        helper(node.right, param1 - node.val, param2, res);
        param2.remove(param2.size() - 1);
    }
}
