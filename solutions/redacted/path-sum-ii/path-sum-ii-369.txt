/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList();
        List<Integer> li = new ArrayList();
        method2(root, param1, ans, li);
        return ans;
    }

    static void method2(TreeNode root, int param2, List<List<Integer>> ans, List<Integer> li) {
        if (root == null) {
            return;
        }
        li.add(root.val);
        if (root.left == null && root.right == null) {
            param2 -= root.val;
            if (param2 == 0) {
                ans.add(new ArrayList<>(li));
            }
        }
        method2(root.left, param2 - root.val, ans, li);
        method2(root.right, param2 - root.val, ans, li);
        li.remove(li.size() - 1);
    }
}
