class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        LinkedList<Integer> var1 = new LinkedList<>();
        traverse(root, param1, var1);
        return res;
    }

    private void traverse(TreeNode root, int param1, LinkedList<Integer> var1) {
        if (root == null)
            return;
        var1.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == root.val) {
                res.add(new ArrayList<>(var1));
            }
        }
        traverse(root.left, param1 - root.val, var1);
        traverse(root.right, param1 - root.val, var1);
        var1.removeLast();
    }
}
