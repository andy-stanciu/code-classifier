/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans;

    public void dfs(TreeNode root, int target, List<Integer> t, int param1) {
        if (root == null)
            return;
        param1 += root.val;
        t.add(root.val);
        if (root.left == null && root.right == null && param1 == target) {
            List<Integer> temp = new ArrayList<Integer>(t);
            ans.add(temp);
        }
        dfs(root.left, target, t, param1);
        dfs(root.right, target, t, param1);
        t.remove(t.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        ans = new ArrayList<>();
        List<Integer> t = new ArrayList<>();
        dfs(root, param2, t, 0);
        return ans;
    }
}
