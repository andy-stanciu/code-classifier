/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static void dfs(TreeNode root, int param1, List<Integer> currentList, List<List<Integer>> result) {
        if (root == null) {
            return;
        }
        currentList.add(root.val);
        if (root.val == param1 && root.left == null && root.right == null) {
            result.add(new ArrayList<Integer>(currentList));
        } else {
            dfs(root.left, param1 - root.val, currentList, result);
            dfs(root.right, param1 - root.val, currentList, result);
        }
        currentList.remove(currentList.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> currentList = new ArrayList<>();
        dfs(root, param1, currentList, result);
        return result;
    }
}
