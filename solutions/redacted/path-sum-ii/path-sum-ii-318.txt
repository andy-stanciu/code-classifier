/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(root, param1, 0, new ArrayList<>());
        return result;
    }

    List<List<Integer>> result = new ArrayList<>();

    public void helper(TreeNode node, int param1, int param2, List<Integer> nodes) {
        if (node == null) {
            return;
        }
        if (node.left == null && node.right == null) {
            if (param2 + node.val == param1) {
                List<Integer> ls = new ArrayList<>();
                ls.addAll(nodes);
                ls.add(node.val);
                result.add(ls);
            }
        }
        if (node.left != null) {
            nodes.add(node.val);
            helper(node.left, param1, param2 + node.val, nodes);
            nodes.remove(nodes.size() - 1);
        }
        if (node.right != null) {
            nodes.add(node.val);
            helper(node.right, param1, param2 + node.val, nodes);
            nodes.remove(nodes.size() - 1);
        }
    }
}
