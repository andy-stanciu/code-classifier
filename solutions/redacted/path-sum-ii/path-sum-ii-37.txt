/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean method1(TreeNode root, int param1) {
        int var1 = 0;
        return helper(root, param1, var1);
    }

    public boolean helper(TreeNode root, int param1, int var1) {
        if (root == null)
            return false;
        if (root.left == null && root.right == null && param1 == var1 + root.val)
            return true;
        var1 += root.val;
        boolean left = helper(root.left, param1, var1);
        boolean right = helper(root.right, param1, var1);
        return (left || right);
    }
}
