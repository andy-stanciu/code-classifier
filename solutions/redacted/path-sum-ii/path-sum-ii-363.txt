/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        int var1 = 0;
        int idx = -1;
        rec(root, param1, ans, temp, var1, idx);
        return ans;
    }

    void rec(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> temp, int var1, int idx) {
        if (root == null)
            return;
        temp.add(root.val);
        idx++;
        var1 = var1 + root.val;
        if (root.left == null && root.right == null) {
            if (var1 == param1) {
                ans.add(new ArrayList<>(temp));
                temp.remove(idx);
                idx--;
                var1 = var1 - root.val;
                return;
            } else {
                temp.remove(idx);
                idx--;
                var1 = var1 - root.val;
                return;
            }
        }
        if (root.left != null) {
            rec(root.left, param1, ans, temp, var1, idx);
        }
        if (root.right != null) {
            rec(root.right, param1, ans, temp, var1, idx);
        }
        temp.remove(idx);
        idx--;
        var1 = var1 - root.val;
        return;
    }
}
