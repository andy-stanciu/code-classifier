class Solution {

    private static void dfs(TreeNode root, int param1, int curr, ArrayList<Integer> temp, List<List<Integer>> ans) {
        if (root == null)
            return;
        curr += root.val;
        temp.add(root.val);
        if (curr == param1 && root.left == null && root.right == null) {
            ans.add(new ArrayList<>(temp));
        }
        dfs(root.left, param1, curr, temp, ans);
        dfs(root.right, param1, curr, temp, ans);
        // Backtrack: remove the last element before returning from the current call
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        List<List<Integer>> ans = new ArrayList<>();
        ArrayList<Integer> temp = new ArrayList<>();
        dfs(root, param2, 0, temp, ans);
        return ans;
    }
}
