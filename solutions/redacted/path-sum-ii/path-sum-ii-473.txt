class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root != null)
            dfs(root, param1, 0, new ArrayList<>());
        return result;
    }

    private void dfs(TreeNode node, int target, int param2, List<Integer> currentList) {
        param2 += node.val;
        currentList.add(node.val);
        if (node.left == null && node.right == null && param2 == target)
            result.add(currentList);
        if (node.left != null)
            dfs(node.left, target, param2, new ArrayList<>(currentList));
        if (node.right != null)
            dfs(node.right, target, param2, new ArrayList<>(currentList));
    }
}
