/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> res;

    int var1;

    public List<List<Integer>> method1(TreeNode root, int var1) {
        this.var1 = var1;
        this.res = new ArrayList<>();
        if (root != null) {
            List<Integer> list = new ArrayList<>();
            list.add(root.val);
            dfs(root, list, root.val);
        }
        return res;
    }

    private void dfs(TreeNode root, List<Integer> list, int param1) {
        if (root.left == null && root.right == null) {
            if (param1 == var1) {
                res.add(new ArrayList<>(list));
            }
            return;
        }
        if (root.left != null) {
            list.add(root.left.val);
            dfs(root.left, list, param1 + root.left.val);
            list.remove(list.size() - 1);
        }
        if (root.right != null) {
            list.add(root.right.val);
            dfs(root.right, list, param1 + root.right.val);
            list.remove(list.size() - 1);
        }
    }
}
