class Solution {

    public List<List<Integer>> res;

    public void solve(TreeNode root, int param1, List<Integer> vals) {
        if (root == null)
            return;
        vals.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            res.add(vals);
            return;
        }
        solve(root.left, param1 - root.val, new ArrayList<Integer>(vals));
        solve(root.right, param1 - root.val, vals);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        res = new ArrayList<>();
        solve(root, param1, new ArrayList<>());
        return res;
    }
}
