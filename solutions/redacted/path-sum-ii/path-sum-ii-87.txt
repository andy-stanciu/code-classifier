/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        calc(root, param1, new ArrayList<>());
        return result;
    }

    void calc(TreeNode node, int param2, List<Integer> temp) {
        if (node == null)
            return;
        temp.add(node.val);
        if (node.val == param2 && node.left == null && node.right == null) {
            result.add(new ArrayList<>(temp));
        }
        calc(node.left, param2 - node.val, temp);
        calc(node.right, param2 - node.val, temp);
        temp.remove(temp.size() - 1);
    }
}
