class Solution {

    int var1;

    List<List<Integer>> res;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        res = new ArrayList<>();
        this.var1 = param1;
        travel(root, 0, new ArrayList<>());
        return res;
    }

    public void travel(TreeNode root, int param2, List<Integer> list) {
        if (root == null)
            return;
        param2 += root.val;
        list.add(root.val);
        if (root.left == null && root.right == null && param2 == var1) {
            res.add(new ArrayList<>(list));
        }
        travel(root.left, param2, list);
        travel(root.right, param2, list);
        list.remove(list.size() - 1);
        param2 -= root.val;
    }
}
