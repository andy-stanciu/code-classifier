class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        solve(root, param1, 0, res, new ArrayList<Integer>());
        return res;
    }

    public void solve(TreeNode root, int param2, int param3, List<List<Integer>> res, ArrayList<Integer> temp) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param3 + root.val == param2) {
            temp.add(root.val);
            res.add(temp);
            return;
        }
        temp.add(root.val);
        solve(root.left, param2, param3 + root.val, res, new ArrayList<Integer>(temp));
        solve(root.right, param2, param3 + root.val, res, new ArrayList<Integer>(temp));
    }
}
