/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return ans;
        preorder(ans, var1, root, param1);
        return ans;
    }

    void preorder(List<List<Integer>> ans, List<Integer> var1, TreeNode root, int param2) {
        if (root == null)
            return;
        param2 -= root.val;
        // process current node and recur for left and right;
        var1.add(root.val);
        //bc sum==0 and left anf right child is null
        if (root.right == null && root.left == null && param2 == 0) {
            ans.add(var1);
        }
        // y new  ? because we need a a path, if not ,mentions to new then we cannot have added this unique path in our ans 2d array.
        preorder(ans, new ArrayList<>(var1), root.left, param2);
        preorder(ans, new ArrayList<>(var1), root.right, param2);
    }
}
