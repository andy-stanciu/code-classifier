/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        helper(root, param1, var1, result);
        return result;
    }

    private void helper(TreeNode root, int param1, List<Integer> var1, List<List<Integer>> result) {
        if (root == null) {
            return;
        }
        var1.add(root.val);
        param1 -= root.val;
        if (root.left == null && root.right == null && param1 == 0) {
            // If the current node is a leaf and targetSum is 0, add the path to the result
            result.add(new ArrayList<>(var1));
        } else {
            // Continue exploring left and right subtrees
            helper(root.left, param1, var1, result);
            helper(root.right, param1, var1, result);
        }
        // Remove the last node value to backtrack and explore other paths
        var1.remove(var1.size() - 1);
    }
}
