class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return new ArrayList<>();
        return helper(root, 0, param1, new ArrayList<>(), new ArrayList<>());
    }

    List<List<Integer>> helper(TreeNode node, int param2, int target, List<List<Integer>> ans, List<Integer> list) {
        if (node == null)
            return ans;
        list.add(node.val);
        if (node.left == null && node.right == null && param2 + node.val == target) {
            ans.add(new ArrayList<>(list));
        }
        helper(node.left, param2 + node.val, target, ans, list);
        helper(node.right, param2 + node.val, target, ans, list);
        list.remove(list.size() - 1);
        return ans;
    }
}
