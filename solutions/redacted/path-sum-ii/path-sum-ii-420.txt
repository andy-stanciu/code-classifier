/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        helper(res, var1, root, param1);
        return res;
    }

    void helper(List<List<Integer>> res, List<Integer> var1, TreeNode root, int param1) {
        if (root == null)
            return;
        var1.add(root.val);
        if (root.val == param1 && root.left == null && root.right == null)
            res.add(new ArrayList<>(var1));
        helper(res, var1, root.left, param1 - root.val);
        helper(res, var1, root.right, param1 - root.val);
        var1.remove(var1.size() - 1);
    }
}
