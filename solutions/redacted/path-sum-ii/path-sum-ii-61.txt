class Solution {

    List<List<Integer>> al = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(new ArrayList<>(), root, param1);
        return al;
    }

    void helper(List<Integer> li, TreeNode root, int param1) {
        if (root == null)
            return;
        li.add(root.val);
        if (root.left == null && root.right == null) {
            int var1 = 0;
            for (int i : li) var1 += i;
            if (var1 == param1)
                al.add(li);
        }
        helper(new ArrayList<>(li), root.left, param1);
        helper(new ArrayList<>(li), root.right, param1);
    }
}
