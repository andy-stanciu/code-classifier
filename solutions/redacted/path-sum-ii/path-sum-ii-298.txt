/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        method2(root, param1, new ArrayList<>(), result);
        return result;
    }

    private void method2(TreeNode root, int param1, List<Integer> param2, List<List<Integer>> result) {
        if (root == null)
            return;
        param2.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            result.add(param2);
        }
        method2(root.left, param1 - root.val, new ArrayList(param2), result);
        method2(root.right, param1 - root.val, new ArrayList(param2), result);
    }
}
