/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        var var1 = new ArrayList<List<Integer>>();
        depthFirstSearch(root, param1, new ArrayList<>(), var1);
        return var1;
    }

    private static void depthFirstSearch(TreeNode root, int param2, List<Integer> param3, List<List<Integer>> var1) {
        if (root == null) {
            return;
        }
        param3.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 - root.val == 0) {
                var1.add(param3.stream().toList());
            }
            param3.remove(param3.size() - 1);
            return;
        }
        depthFirstSearch(root.left, param2 - root.val, param3, var1);
        depthFirstSearch(root.right, param2 - root.val, param3, var1);
        param3.remove(param3.size() - 1);
    }
}
