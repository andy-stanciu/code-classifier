/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        dfs(new ArrayList<>(), param1, root);
        return res;
    }

    private void dfs(List<Integer> subList, int param2, TreeNode cur) {
        if (cur == null) {
            return;
        }
        param2 -= cur.val;
        subList.add(cur.val);
        if (cur.left == null && cur.right == null && param2 == 0) {
            res.add(new ArrayList<>(subList));
        }
        dfs(subList, param2, cur.left);
        dfs(subList, param2, cur.right);
        param2 += cur.val;
        subList.remove(subList.size() - 1);
    }
}
