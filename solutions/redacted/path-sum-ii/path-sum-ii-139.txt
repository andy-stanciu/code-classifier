class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(root, param1, 0, new ArrayList<Integer>());
        return ans;
    }

    private void method2(TreeNode root, int param1, int count, List<Integer> lt) {
        if (root == null)
            return;
        lt.add(root.val);
        if (root.left == null && root.right == null) {
            if (root.val + count == param1) {
                ans.add(new ArrayList<>(lt));
            }
        }
        method2(root.left, param1, count + root.val, lt);
        method2(root.right, param1, count + root.val, lt);
        lt.remove(lt.size() - 1);
    }
}
