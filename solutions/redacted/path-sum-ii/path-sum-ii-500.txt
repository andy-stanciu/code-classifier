class Solution {

    // stores all paths that add to targetSum
    List<List<Integer>> var1 = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return var1;
        method2(root, param1, 0, new ArrayList<Integer>());
        return var1;
    }

    public void method2(TreeNode node, int target, int param2, ArrayList<Integer> param3) {
        param2 += node.val;
        param3.add(node.val);
        // if we are at a leaf node, check if the path sum == target
        if (node.left == null && node.right == null) {
            if (param2 == target)
                var1.add(new ArrayList<Integer>(param3));
            // we don\'t call a child, so we don\'t have to remove a node from the list
            return;
        }
        if (node.left != null) {
            // add the left child to the path
            method2(node.left, target, param2, param3);
            // after it finishes executing, remove the node that was added by node.left
            param3.remove(param3.size() - 1);
        }
        // do the same for the right child
        if (node.right != null) {
            method2(node.right, target, param2, param3);
            param3.remove(param3.size() - 1);
        }
    }
}
