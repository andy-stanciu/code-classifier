class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> al = new ArrayList<>();
        method2(root, param1, al);
        return ans;
    }

    public void method2(TreeNode root, int val, List<Integer> param2) {
        if (root == null) {
            return;
        }
        param2.add(root.val);
        if (root.left == null && root.right == null && root.val == val) {
            ans.add(new ArrayList<>(param2));
        }
        method2(root.left, val - root.val, param2);
        method2(root.right, val - root.val, param2);
        param2.remove(param2.size() - 1);
    }
}
