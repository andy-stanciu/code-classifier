/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        int var1 = 0;
        method2(root, param1, var1, ans, list);
        return ans;
    }

    public void method2(TreeNode root, int param1, int var1, List<List<Integer>> ans, List<Integer> list) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        var1 += root.val;
        if (root.left == null && root.right == null && var1 == param1) {
            ans.add(new ArrayList<>(list));
        }
        method2(root.left, param1, var1, ans, list);
        method2(root.right, param1, var1, ans, list);
        list.remove(list.size() - 1);
    }
}
