/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result = new ArrayList<>();
        dfs(root, param1, new ArrayList<>());
        return result;
    }

    void dfs(TreeNode root, int param1, List<Integer> param2) {
        if (root == null)
            return;
        //Add current node in the path
        param2.add(root.val);
        //If the current node is leaf node and value of this node is equal to remaining sum then this is the valid path
        if (root.left == null && root.right == null && root.val == param1) {
            result.add(new ArrayList<>(param2));
        }
        //Go to both the childrens
        dfs(root.left, param1 - root.val, param2);
        dfs(root.right, param1 - root.val, param2);
        //Backtrack and remove current node
        param2.remove(param2.size() - 1);
    }
}
