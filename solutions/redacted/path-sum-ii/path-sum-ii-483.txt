class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return result;
        List<Integer> var1 = new ArrayList<>();
        helper(var1, root, param1);
        return result;
    }

    public void helper(List<Integer> var1, TreeNode root, int param1) {
        if (root == null)
            return;
        var1.add(root.val);
        if (root.left == null && root.right == null && root.val == param1) {
            result.add(new ArrayList<>(var1));
        } else {
            helper(var1, root.left, param1 - root.val);
            helper(var1, root.right, param1 - root.val);
        }
        var1.remove(var1.size() - 1);
    }
}
