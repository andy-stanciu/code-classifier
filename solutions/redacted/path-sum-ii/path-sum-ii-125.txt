/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        method1(root, param1, new ArrayList<>(), result);
        return result;
    }

    private void method1(TreeNode root, int param1, List<Integer> list, List<List<Integer>> result) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            result.add(new ArrayList<>(list));
        } else {
            method1(root.left, param1 - root.val, list, result);
            method1(root.right, param1 - root.val, list, result);
        }
        //backtracking, remove the last element
        list.remove(list.size() - 1);
    }
}
