/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        traverse(root, param1, 0, res, new ArrayList<>());
        return res;
    }

    void traverse(TreeNode root, int param1, int param2, List<List<Integer>> res, List<Integer> param3) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            if (param2 + root.val == param1) {
                //Create a new list with current path and add the new list to the result
                List<Integer> newList = new ArrayList<>();
                newList.addAll(param3);
                newList.add(root.val);
                res.add(newList);
            }
            return;
        }
        param3.add(root.val);
        param2 += root.val;
        traverse(root.left, param1, param2, res, param3);
        traverse(root.right, param1, param2, res, param3);
        //Remove the last node in the path as we are done with processing this node in the current path
        param3.remove(param3.size() - 1);
    }
}
