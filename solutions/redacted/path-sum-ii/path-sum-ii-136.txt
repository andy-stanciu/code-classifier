/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        targetvalue(root, param1, ans, new ArrayList<>());
        return ans;
    }

    private void targetvalue(TreeNode root, int param1, List<List<Integer>> ans, List<Integer> innerList) {
        if (root == null) {
            return;
        }
        innerList.add(root.val);
        if (root.right == null && root.left == null && root.val == param1) {
            ans.add(new ArrayList<>(innerList));
        } else {
            targetvalue(root.left, param1 - root.val, ans, innerList);
            targetvalue(root.right, param1 - root.val, ans, innerList);
        }
        innerList.remove(innerList.size() - 1);
    }
}
