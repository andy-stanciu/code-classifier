/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        method2(root, param1, new ArrayList<Integer>());
        return result;
    }

    public void method2(TreeNode root, int param1, List<Integer> param2) {
        if (root == null)
            return;
        param2.add(root.val);
        if (root.val == param1 && root.left == null && root.right == null) {
            result.add(new ArrayList<>(param2));
        } else {
            method2(root.left, param1 - root.val, param2);
            method2(root.right, param1 - root.val, param2);
        }
        param2.remove(param2.size() - 1);
    }
}
