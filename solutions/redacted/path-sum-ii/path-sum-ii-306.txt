/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ll = new ArrayList<>();
        List<Integer> lld = new ArrayList<>();
        method2(root, param1, lld, ll);
        return ll;
    }

    public void method2(TreeNode root, int param1, List<Integer> lld, List<List<Integer>> ll) {
        if (root == null)
            return;
        lld.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            ll.add(new ArrayList<Integer>(lld));
        } else {
            method2(root.left, param1 - root.val, lld, ll);
            method2(root.right, param1 - root.val, lld, ll);
        }
        lld.remove(lld.size() - 1);
    }
}
