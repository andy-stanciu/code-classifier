/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var1 = new ArrayList<>();
        dfs(root, var1, param1);
        return ans;
    }

    void dfs(TreeNode root, List<Integer> var1, int param1) {
        if (root == null)
            return;
        var1.add(root.val);
        // check the leaf node
        if (root.val == param1 && root.left == null && root.right == null) {
            ans.add(new ArrayList<>(var1));
        }
        // traverse left node
        dfs(root.left, var1, param1 - root.val);
        // traverse right node
        dfs(root.right, var1, param1 - root.val);
        // backtrack
        var1.remove(var1.size() - 1);
    }
}
