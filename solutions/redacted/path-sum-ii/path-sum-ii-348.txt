/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> finalList = new ArrayList<>();

    List<Integer> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return finalList;
        int var1 = 0;
        dfs(root, param1, var1);
        return finalList;
    }

    private void dfs(TreeNode root, int param1, int var1) {
        if (root == null) {
            return;
        }
        ans.add(root.val);
        var1 += root.val;
        if (var1 == param1 && root.left == null && root.right == null) {
            finalList.add(new ArrayList<>(ans));
        }
        dfs(root.left, param1, var1);
        dfs(root.right, param1, var1);
        ans.remove(ans.size() - 1);
    }
}
