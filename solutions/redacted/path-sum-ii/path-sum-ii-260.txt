/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        method2(root, param1, new ArrayList<Integer>(), result);
        return result;
    }

    private void method2(TreeNode node, int param2, List<Integer> list, List<List<Integer>> result) {
        if (node == null) {
            return;
        }
        list.add(node.val);
        if (param2 == node.val && node.left == null && node.right == null) {
            result.add(new ArrayList<Integer>(list));
        }
        method2(node.left, param2 - node.val, list, result);
        method2(node.right, param2 - node.val, list, result);
        list.remove(list.size() - 1);
    }
}
