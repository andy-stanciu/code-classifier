class TreeNode {

    int val;

    TreeNode left;

    TreeNode right;

    TreeNode(int x) {
        val = x;
    }
}

class Solution {

    public static List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        result = new ArrayList<>();
        fill(root, 0, new ArrayList<>(), param1);
        return result;
    }

    public static void fill(TreeNode root, int param2, List<Integer> temp, int param1) {
        if (root == null) {
            return;
        }
        param2 += root.val;
        temp.add(root.val);
        if (root.left == null && root.right == null) {
            if (param2 == param1) {
                result.add(new ArrayList<>(temp));
            }
        } else {
            fill(root.left, param2, temp, param1);
            fill(root.right, param2, temp, param1);
        }
        // Remove the current node from the path to backtrack
        temp.remove(temp.size() - 1);
    }
}
