class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        traverse(root, new ArrayList<>(), 0, param1);
        return ans;
    }

    public void traverse(TreeNode node, List<Integer> list, int param2, int param1) {
        if (node == null)
            return;
        list.add(node.val);
        if (node.left == null && node.right == null && param2 + node.val == param1) {
            ans.add(list);
            return;
        }
        traverse(node.left, new ArrayList<>(list), param2 + node.val, param1);
        traverse(node.right, new ArrayList<>(list), param2 + node.val, param1);
    }
}
