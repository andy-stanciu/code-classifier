/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null)
            return result;
        helper(result, new ArrayList<Integer>(), root, param1);
        return result;
    }

    private void helper(List<List<Integer>> result, List<Integer> curr, TreeNode node, int param1) {
        if (node.left == null && node.right == null) {
            if (param1 == node.val) {
                curr.add(node.val);
                result.add(new ArrayList<>(curr));
                curr.remove(curr.size() - 1);
            }
            return;
        }
        if (node.left != null) {
            curr.add(node.val);
            helper(result, curr, node.left, param1 - node.val);
            curr.remove(curr.size() - 1);
        }
        if (node.right != null) {
            curr.add(node.val);
            helper(result, curr, node.right, param1 - node.val);
            curr.remove(curr.size() - 1);
        }
    }
}
