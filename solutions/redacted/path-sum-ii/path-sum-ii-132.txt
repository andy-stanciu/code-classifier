/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        help(root, param1, new ArrayList<>(), 0);
        return result;
    }

    public void help(TreeNode root, int target, List<Integer> ans, int param2) {
        if (root == null)
            return;
        param2 += root.val;
        ans.add(root.val);
        if (param2 == target && root.left == null && root.right == null) {
            result.add(new ArrayList<>(ans));
        }
        help(root.left, target, ans, param2);
        help(root.right, target, ans, param2);
        ans.remove(ans.size() - 1);
    }
}
