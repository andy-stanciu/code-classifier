/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private List<List<Integer>> res;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        this.res = new ArrayList<>();
        DFS(root, param1, new ArrayList<>());
        return res;
    }

    public void DFS(TreeNode node, int param1, List<Integer> list) {
        if (node == null)
            return;
        list.add(node.val);
        if (node.left == null && node.right == null && param1 == node.val) {
            res.add(new ArrayList(list));
        }
        DFS(node.left, param1 - node.val, list);
        DFS(node.right, param1 - node.val, list);
        list.remove(list.size() - 1);
    }
}
