/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        method1(root, new LinkedList<Integer>(), res, param1);
        return res;
    }

    //memory optimized
    public void method1(TreeNode node, LinkedList<Integer> param2, List<List<Integer>> res, int target) {
        if (node == null)
            return;
        if (node.val == target && node.left == null && node.right == null) {
            //found path
            LinkedList<Integer> var1 = new LinkedList<Integer>(param2);
            var1.add(node.val);
            res.add(var1);
        } else {
            param2.add(node.val);
            method1(node.left, param2, res, target - node.val);
            method1(node.right, param2, res, target - node.val);
            param2.removeLast();
        }
    }
