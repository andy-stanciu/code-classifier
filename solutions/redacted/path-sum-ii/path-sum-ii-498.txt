class Solution {

    List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return result;
        method2(root, param1, new ArrayList<>());
        return result;
    }

    private void method2(TreeNode root, int param2, List<Integer> list) {
        if (root == null)
            return;
        list.add(root.val);
        if (root.left == null && root.right == null && param2 == root.val)
            result.add(new ArrayList<>(list));
        else {
            method2(root.left, param2 - root.val, list);
            method2(root.right, param2 - root.val, list);
        }
        list.remove(list.size() - 1);
    }
}
