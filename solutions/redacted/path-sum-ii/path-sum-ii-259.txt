/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> list = new LinkedList<>();
        method2(root, param1, new LinkedList<Integer>(), list);
        return list;
    }

    public void method2(TreeNode root, int param1, List<Integer> param2, List<List<Integer>> finalList) {
        if (root == null)
            return;
        count += root.val;
        param2.add(root.val);
        if (count == param1 && root.left == null && root.right == null)
            finalList.add(new LinkedList<>(param2));
        method2(root.left, param1, param2, finalList);
        method2(root.right, param1, param2, finalList);
        count -= root.val;
        param2.remove(param2.size() - 1);
    }
}
