/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> answer = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        helper(root, param1, 0, new ArrayList<>());
        return answer;
    }

    void helper(TreeNode root, int param1, int param2, List<Integer> param3) {
        //base case
        if (root == null)
            return;
        //backtracking logic
        //1. action
        param2 += root.val;
        param3.add(root.val);
        if (root.left == null && root.right == null && param1 == param2) {
            answer.add(new ArrayList<>(param3));
        }
        //2. recurse
        helper(root.left, param1, param2, param3);
        helper(root.right, param1, param2, param3);
        //3. backtrack
        param3.remove(param3.size() - 1);
    }
}
