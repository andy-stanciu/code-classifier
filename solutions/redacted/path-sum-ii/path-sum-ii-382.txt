/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int target) {
        List<Integer> nums = new ArrayList<>();
        helper(root, target, 0, nums);
        return list;
    }

    public void helper(TreeNode root, int target, int param1, List<Integer> nums) {
        if (root == null)
            return;
        param1 += root.val;
        nums.add(root.val);
        if (root.left == null && root.right == null && target == param1) {
            list.add(new ArrayList<>(nums));
        }
        helper(root.left, target, param1, nums);
        helper(root.right, target, param1, nums);
        nums.remove(nums.size() - 1);
    }
}
