class Solution {

    public boolean isLeaf(TreeNode root) {
        return root != null && root.left == null && root.right == null;
    }

    public void dfs(TreeNode root, int param1, List<List<Integer>> result, List<Integer> list) {
        if (root == null)
            return;
        //System.out.println(list);
        list.add(root.val);
        if (param1 == root.val) {
            //list.add(root.val);
            if (isLeaf(root))
                result.add(new ArrayList<>(list));
        }
        dfs(root.left, param1 - root.val, result, list);
        dfs(root.right, param1 - root.val, result, list);
        list.remove(list.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        dfs(root, param1, result, new ArrayList<>());
        return result;
    }
}
