/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> l = new ArrayList<>();
        f(root, ans, l, param1);
        return ans;
    }

    public static void f(TreeNode root, List<List<Integer>> ans, List<Integer> l, int param2) {
        if (root == null)
            return;
        l.add(root.val);
        param2 -= root.val;
        if (root.left == null && root.right == null) {
            if (param2 == 0) {
                ans.add(new ArrayList<>(l));
            }
        }
        f(root.left, ans, l, param2);
        f(root.right, ans, l, param2);
        l.remove(l.size() - 1);
    }
}
