/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> MainList = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        return method1(root, param1, list);
    }

    public List<List<Integer>> method1(TreeNode root, int param1, List<Integer> list) {
        if (root == null)
            return new ArrayList<>();
        param1 -= root.val;
        list.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == 0)
                MainList.add(new ArrayList<>(list));
        }
        method1(root.left, param1, new ArrayList<>(list));
        method1(root.right, param1, new ArrayList<>(list));
        list.remove(list.size() - 1);
        return MainList;
    }
}
