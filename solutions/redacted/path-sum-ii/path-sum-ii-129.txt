/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        ArrayList<Integer> res = new ArrayList<>();
        helper(root, param1, res);
        return list;
    }

    void helper(TreeNode node, int param2, List<Integer> res) {
        if (node == null) {
            return;
        }
        res.add(node.val);
        if (node.right == null && node.left == null && param2 == node.val) {
            list.add(new ArrayList(res));
        }
        helper(node.left, param2 - node.val, res);
        helper(node.right, param2 - node.val, res);
        res.remove(res.size() - 1);
    }
}
