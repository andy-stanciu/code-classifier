class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> returnList = new ArrayList<>();
        List<Integer> innerList = new ArrayList<>();
        helper(root, innerList, returnList, 0, param1);
        return returnList;
    }

    public void helper(TreeNode root, List<Integer> innerList, List<List<Integer>> returnList, int param2, int param1) {
        if (root != null) {
            param2 += root.val;
            innerList.add(root.val);
            if (root.left == null && root.right == null) {
                if (param2 == param1) {
                    returnList.add(new ArrayList<>(innerList));
                }
            }
            helper(root.left, innerList, returnList, param2, param1);
            helper(root.right, innerList, returnList, param2, param1);
            innerList.remove(innerList.size() - 1);
        }
    }
}
