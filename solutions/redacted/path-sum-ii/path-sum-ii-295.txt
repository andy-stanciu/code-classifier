/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> var1 = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<Integer> var2 = new ArrayList<>();
        method2(root, var2, 0, param1);
        return var1;
    }

    private void method2(TreeNode node, List<Integer> var2, int param2, int param1) {
        // Base Cases
        if (node == null) {
            return;
        }
        var2.add(node.val);
        param2 += node.val;
        if (node.left == null && node.right == null) {
            if (param2 == param1) {
                // Caution: Do not add paths directly to the rootToLeafPaths lists, rootToLeafPaths.add(paths).
                // Only the reference to paths will be added so when the algorithm backtracks below, the values will be
                // removed from the rootToLeafPaths lists.
                var1.add(var2.stream().toList());
            }
        }
        // Recursion
        method2(node.left, var2, param2, param1);
        method2(node.right, var2, param2, param1);
        // Backtracking
        var2.remove(var2.size() - 1);
    }
}
