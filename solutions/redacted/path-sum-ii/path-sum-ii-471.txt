class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new LinkedList<>();
        LinkedList<Integer> var1 = new LinkedList<>();
        traverse(root, var1, res, param1);
        return res;
    }

    public void traverse(TreeNode root, LinkedList<Integer> var1, List<List<Integer>> res, int remain) {
        if (root == null)
            return;
        remain -= root.val;
        var1.add(root.val);
        if (root.left == null && root.right == null) {
            if (remain == 0) {
                res.add(new LinkedList<Integer>(var1));
            }
        }
        traverse(root.left, var1, res, remain);
        traverse(root.right, var1, res, remain);
        var1.removeLast();
    }
}
