/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> ans = new ArrayList<>();

    void find(TreeNode node, int param1, List<Integer> res, int param2) {
        if (node == null) {
            return;
        }
        if (node.right == null && node.left == null) {
            if (param1 + node.val == param2) {
                res.add(node.val);
                ans.add(new ArrayList(res));
                res.remove(res.size() - 1);
            }
            return;
        }
        res.add(node.val);
        find(node.left, param1 + node.val, res, param2);
        find(node.right, param1 + node.val, res, param2);
        res.remove(res.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        find(root, 0, new ArrayList<Integer>(), param2);
        return ans;
    }
}
