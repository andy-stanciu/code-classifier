/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        if (root == null)
            return res;
        dfs(root, res, var1, param1);
        return res;
    }

    public void dfs(TreeNode node, List<List<Integer>> res, List<Integer> var1, int param1) {
        if (node == null)
            return;
        var1.add(node.val);
        // Check if it\'s a leaf node and targetSum equals the node\'s value
        if (node.left == null && node.right == null && param1 == node.val) {
            // Make a copy of the current path
            res.add(new ArrayList<>(var1));
        } else {
            // Subtract the current node value from targetSum and recurse
            dfs(node.left, res, var1, param1 - node.val);
            dfs(node.right, res, var1, param1 - node.val);
        }
        // Backtrack after exploring both subtrees
        var1.remove(var1.size() - 1);
    }
}
