class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        traversal(root, param1, result, new ArrayList<>());
        return result;
    }

    private void traversal(TreeNode root, int param1, List<List<Integer>> result, List<Integer> param2) {
        if (root == null)
            return;
        param1 -= root.val;
        param2.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == 0) {
                result.add(new ArrayList<>(param2));
            }
        } else {
            traversal(root.left, param1, result, param2);
            traversal(root.right, param1, result, param2);
        }
        param2.remove(param2.size() - 1);
    }
}
