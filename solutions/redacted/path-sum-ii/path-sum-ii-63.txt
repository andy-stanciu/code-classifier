class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public void method1(TreeNode root, int param1, List<Integer> list1) {
        list1.add(root.val);
        if (root.left == null && root.right == null) {
            if (param1 == root.val)
                list.add(new LinkedList<>(list1));
            list1.remove(list1.size() - 1);
            return;
        }
        if (root.left != null) {
            method1(root.left, param1 - root.val, list1);
        }
        if (root.right != null) {
            method1(root.right, param1 - root.val, list1);
        }
        list1.remove(list1.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param1) {
        if (root == null)
            return list;
        method1(root, param1, new ArrayList<>());
        return list;
    }
}
