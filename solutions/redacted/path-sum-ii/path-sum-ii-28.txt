class Solution {

    public void fn(TreeNode root, List<List<Integer>> ans, List<Integer> temp, int param1, int param2) {
        if (root == null)
            return;
        if (root.left == null && root.right == null && param1 + root.val == param2) {
            temp.add(root.val);
            ans.add(new ArrayList<>(temp));
            temp.remove(temp.size() - 1);
            return;
        }
        param1 += root.val;
        temp.add(root.val);
        fn(root.left, ans, temp, param1, param2);
        fn(root.right, ans, temp, param1, param2);
        temp.remove(temp.size() - 1);
    }

    public List<List<Integer>> method1(TreeNode root, int param2) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> temp = new ArrayList<>();
        int param1 = 0;
        fn(root, ans, temp, param1, param2);
        return ans;
    }
}
