/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
public class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> var1 = new ArrayList<>();
        dfs(root, param1, var1, result);
        return result;
    }

    private void dfs(TreeNode node, int param1, List<Integer> var1, List<List<Integer>> result) {
        if (node == null) {
            return;
        }
        var1.add(node.val);
        if (node.left == null && node.right == null && param1 - node.val == 0) {
            result.add(new ArrayList<>(var1));
        }
        dfs(node.left, param1 - node.val, var1, result);
        dfs(node.right, param1 - node.val, var1, result);
        var1.remove(var1.size() - 1);
    }
}
