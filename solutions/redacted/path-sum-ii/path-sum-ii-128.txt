/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> local = new ArrayList<>();
        helper(root, param1, local, ans);
        return ans;
    }

    static void helper(TreeNode root, int param1, List<Integer> local, List<List<Integer>> ans) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null && param1 == root.val) {
            List<Integer> copy = new ArrayList<>(local);
            copy.add(root.val);
            ans.add(copy);
            return;
        }
        local.add(root.val);
        helper(root.left, param1 - root.val, local, ans);
        helper(root.right, param1 - root.val, local, ans);
        local.remove(local.size() - 1);
        return;
    }
}
