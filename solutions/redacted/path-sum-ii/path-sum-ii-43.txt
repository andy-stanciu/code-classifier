class Solution {

    List<List<Integer>> list;

    public List<List<Integer>> method1(TreeNode root, int param1) {
        list = new ArrayList<>();
        method2(root, param1, new ArrayList<>());
        return list;
    }

    void method2(TreeNode root, int method2, List<Integer> curr) {
        if (root == null)
            return;
        if (isLeaf(root)) {
            if (method2 == root.val) {
                curr.add(root.val);
                list.add(new ArrayList<>(curr));
                curr.remove(curr.size() - 1);
            }
            return;
        }
        curr.add(root.val);
        method2(root.left, method2 - root.val, curr);
        method2(root.right, method2 - root.val, curr);
        curr.remove(curr.size() - 1);
    }

    static boolean isLeaf(TreeNode root) {
        return root.left == root.right;
    }
}
