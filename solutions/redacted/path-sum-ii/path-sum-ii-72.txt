class Solution {

    final List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root, int param1) {
        dfs(root, new ArrayList<>(), param1, 0);
        return res;
    }

    void dfs(TreeNode root, List<Integer> temp, int target, int param2) {
        if (root == null) {
            return;
        }
        param2 += root.val;
        temp.add(root.val);
        if (root.left == null && root.right == null && param2 == target) {
            res.add(new ArrayList<>(temp));
        }
        dfs(root.left, temp, target, param2);
        dfs(root.right, temp, target, param2);
        temp.remove(temp.size() - 1);
    }
}
