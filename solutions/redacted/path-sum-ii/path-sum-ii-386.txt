/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root, int param1) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        List<Integer> ans = new ArrayList<>();
        helper(root, param1, ans, res);
        return res;
    }

    public void helper(TreeNode root, int param1, List<Integer> ans, List<List<Integer>> res) {
        if (root == null)
            return;
        ans.add(root.val);
        if (root.left == null && root.right == null && param1 == root.val) {
            res.add(new ArrayList(ans));
            ans.remove(ans.size() - 1);
            return;
        } else {
            helper(root.left, param1 - root.val, ans, res);
            helper(root.right, param1 - root.val, ans, res);
        }
        ans.remove(ans.size() - 1);
    }
}
