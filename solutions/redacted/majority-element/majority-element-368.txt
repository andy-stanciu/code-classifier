class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> hash = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            hash.put(nums[i], hash.get(nums[i]) == null ? 1 : hash.get(nums[i]) + 1);
            if (hash.get(nums[i]) > (n / 2)) {
                return nums[i];
            }
        }
        return 0;
    }
}
