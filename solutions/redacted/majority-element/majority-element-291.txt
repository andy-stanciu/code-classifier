class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> hash = new HashMap<>();
        int res = 0;
        int var1 = 0;
        for (int n : nums) {
            hash.put(n, 1 + hash.getOrDefault(n, 0));
            if (hash.get(n) > var1) {
                res = n;
                var1 = hash.get(n);
            }
        }
        return res;
    }
}
