class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int var1 = n / 2;
        for (int i = 0; i < n - var1; i++) {
            if (nums[i] == nums[i + var1]) {
                return nums[i];
            }
        }
        return -1;
    }
}
