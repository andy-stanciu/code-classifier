class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> m = new HashMap<>();
        int s = nums.length / 2;
        for (int i = 0; i < nums.length; i++) {
            int n = nums[i];
            m.put(n, m.getOrDefault(n, 0) + 1);
        }
        for (int j : m.keySet()) {
            if (m.get(j) > s) {
                return j;
            }
        }
        return -1;
    }
}
