class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> result = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            result.put(nums[i], result.getOrDefault(nums[i], 0) + 1);
        }
        for (Map.Entry<Integer, Integer> res : result.entrySet()) {
            if (res.getValue() > n / 2) {
                return res.getKey();
            }
        }
        return 1;
    }
}
