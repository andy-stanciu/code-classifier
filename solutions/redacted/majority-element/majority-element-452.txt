class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> ctr = new HashMap<>();
        for (int i : nums) {
            ctr.put(i, ctr.getOrDefault(i, 0) + 1);
        }
        int half = nums.length % 2 == 0 ? nums.length / 2 : nums.length / 2 + 1;
        for (Map.Entry<Integer, Integer> entry : ctr.entrySet()) {
            if (entry.getValue() >= half)
                return entry.getKey();
        }
        return 0;
    }
}
