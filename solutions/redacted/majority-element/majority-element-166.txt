class Solution {

    public int method1(int[] nums) {
        int i = 1, var1 = nums[0], count = 1;
        while (i < nums.length) {
            if (nums[i++] != var1) {
                count--;
            } else {
                count++;
            }
            if (count < 0) {
                var1 = nums[i - 1];
                count = 0;
            }
        }
        return var1;
    }
}
