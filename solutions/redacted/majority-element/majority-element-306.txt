class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                var1 = nums[i];
                count = 1;
            } else if (var1 == nums[i]) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
