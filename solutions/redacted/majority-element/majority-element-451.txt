class Solution {

    public int method1(int[] nums) {
        // Initially set the first element as the candidate
        int person = nums[0];
        // Start the count at 1 for the first element
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            // Iterate through the array starting from the second element
            if (count == 0) {
                // Set a new candidate
                person = nums[i];
            }
            if (nums[i] == person) {
                // Increment count if the current number matches the candidate
                count++;
            } else {
                // Decrement count otherwise
                count--;
            }
        }
        // Return the majority element
        return person;
    }
}
