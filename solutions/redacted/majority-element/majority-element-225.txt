class Solution {

    public int method1(int[] nums) {
        int candidate = nums[0];
        if (nums.length == 1) {
            return candidate;
        }
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (candidate == nums[i]) {
                count++;
            } else {
                count--;
            }
            if (count == 0) {
                candidate = nums[i + 1];
            }
        }
        return candidate;
    }
}
