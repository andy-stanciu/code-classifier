class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        int count = 0;
        for (int x : nums) {
            if (count == 0) {
                var1 = x;
            }
            if (var1 == x) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
