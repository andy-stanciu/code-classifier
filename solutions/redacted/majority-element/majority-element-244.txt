class Solution {

    public int method1(int[] arr) {
        HashMap<Integer, Integer> hp = new HashMap<>();
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            if (hp.containsKey(arr[i]))
                hp.put(arr[i], hp.get(arr[i]) + 1);
            else
                hp.put(arr[i], 1);
        }
        int x = n / 2;
        int ans = -1;
        for (var a : hp.keySet()) {
            if (hp.get(a) > x)
                ans = a;
        }
        return ans;
    }
}
