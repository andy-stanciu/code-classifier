class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Integer> hpp = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int val = hpp.getOrDefault(nums[i], 0);
            hpp.put(nums[i], val + 1);
        }
        for (Map.Entry<Integer, Integer> it : hpp.entrySet()) {
            if (it.getValue() > (n / 2))
                return it.getKey();
        }
        return -1;
    }
}
