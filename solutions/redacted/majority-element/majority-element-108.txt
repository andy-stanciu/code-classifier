class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int max = 0;
        int res = 0;
        for (int i = 0, n = nums.length; i < n; i++) {
            int count = map.getOrDefault(nums[i], 0) + 1;
            map.put(nums[i], count);
            if (count > max) {
                max = count;
                res = nums[i];
            }
        }
        return res;
    }
}
