class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int candidate = 0;
        // Phase 1: Find the candidate
        for (int num : nums) {
            if (count == 0) {
                candidate = num;
            }
            count += (num == candidate) ? 1 : -1;
        }
        // Phase 2: Verify the candidate (optional, since the problem guarantees a majority element)
        count = 0;
        for (int num : nums) {
            if (num == candidate) {
                count++;
            }
        }
        // or return candidate only if count > nums.length / 2;
        return candidate;
    }
}
