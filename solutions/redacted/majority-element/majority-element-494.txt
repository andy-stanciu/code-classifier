class Solution {

    public int method1(int[] nums) {
        // Boyer-Moore Voting Algorithm
        int candidate = 0;
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                candidate = num;
            }
            count += (num == candidate) ? 1 : -1;
        }
        // The candidate will be the majority element
        return candidate;
    }
}
