class Solution {

    public int method1(int[] nums) {
        int candidate = 0;
        int count = 0;
        // First pass to find a potential majority candidate
        for (int num : nums) {
            if (count == 0) {
                // Set new candidate
                candidate = num;
            }
            // Increment or decrement count
            count += (num == candidate) ? 1 : -1;
        }
        // Return the majority element
        return candidate;
    }
}
