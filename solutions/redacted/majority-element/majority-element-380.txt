class Solution {

    public int method1(int[] nums) {
        //initializing the winner as the nums[0].
        int winner = nums[0], vote = 0;
        //iterating through the array.
        for (int i = 0; i < nums.length; i++) {
            //vote counter increment.
            if (nums[i] == winner)
                vote++;
            else
                //decrement the vote counter.
                vote--;
            //we take into consieration a potential winner and set  vote counter to 1.
            if (vote == 0) {
                winner = nums[i];
                vote = 1;
            }
        }
        //return the winner
        return winner;
    }
}
