class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        int voter = nums[0];
        int voterCount = 1;
        for (int i = 0; i < len; i++) {
            if (nums[i] == voter)
                voterCount++;
            else
                voterCount--;
            if (voterCount == 0) {
                voter = nums[i];
                voterCount++;
            }
        }
        return voter;
    }
}
