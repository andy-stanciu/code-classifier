class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int var1 = 0;
        for (int num : nums) {
            if (count == 0) {
                var1 = num;
            }
            if (var1 == num) {
                count += 1;
            } else {
                count -= 1;
            }
        }
        return var1;
    }
}
