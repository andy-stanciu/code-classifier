class Solution {

    //Boyer\u2013Moore majority vote algorithm
    //TC- O(N)
    public int method1(int[] nums) {
        int count = 0;
        int var1 = 0;
        for (int num : nums) {
            if (count == 0) {
                var1 = num;
            }
            if (num == var1) {
                count += 1;
            } else {
                count -= 1;
            }
        }
        return var1;
    }
}
