//Boyer-Moore Majority Voting Algo
class Solution {

    public int method1(int[] nums) {
        //max element
        int me = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == me) {
                count++;
            } else {
                count--;
            }
            if (count == 0) {
                me = nums[i];
                count = 1;
            }
        }
        return me;
    }
}
