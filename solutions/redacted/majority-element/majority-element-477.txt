class Solution {

    public int method1(int[] nums) {
        int c = 0;
        int v = 0;
        for (int i = 0; i < nums.length; i++) {
            if (c == 0) {
                v = nums[i];
            }
            if (v == nums[i]) {
                c++;
            } else {
                c--;
            }
        }
        return v;
    }
}
