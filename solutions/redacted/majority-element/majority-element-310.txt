class Solution {

    public int method1(int[] nums) {
        int res = 0;
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int ele : nums) {
            if (map.containsKey(ele)) {
                map.put(ele, map.get(ele) + 1);
            } else {
                map.put(ele, 1);
            }
        }
        for (int ele : nums) {
            if (map.get(ele) > (nums.length / 2))
                res = ele;
        }
        return res;
    }
}
