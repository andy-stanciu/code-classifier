class Solution {

    public int method1(int[] nums) {
        int count = 0, number = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0 && number != nums[i]) {
                number = nums[i];
                count = 1;
            } else {
                count = (number == nums[i]) ? ++count : --count;
            }
        }
        return number;
    }
}
