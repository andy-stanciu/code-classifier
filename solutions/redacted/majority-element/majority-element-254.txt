class Solution {

    public int method1(int[] nums) {
        return method2(nums, 0, nums.length - 1);
    }

    public int method2(int[] nums, int l, int h) {
        if (l == h)
            return nums[l];
        int m = l + (h - l) / 2;
        int left = method2(nums, l, m);
        int right = method2(nums, m + 1, h);
        if (left == right)
            return left;
        int lCount = count(nums, left, l, h);
        int rCount = count(nums, right, l, h);
        if (lCount > rCount)
            return left;
        else
            return right;
    }

    public int count(int[] nums, int n, int l, int h) {
        int c = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == n)
                c++;
        }
        return c;
    }
}
