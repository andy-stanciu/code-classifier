class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        int voter = 1;
        for (int i = 1; i < nums.length; i++) {
            if (voter == 0) {
                var1 = nums[i];
                voter++;
            } else if (nums[i] == var1) {
                voter++;
            } else {
                voter--;
            }
        }
        return var1;
    }
}
