class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        Map<Integer, Integer> map = new HashMap<>();
        for (//to put element in Hashmap and count each
        //to put element in Hashmap and count each
        //to put element in Hashmap and count each
        int i = 0; i < n; i++) {
            if (map.containsKey(nums[i])) {
                map.put(nums[i], map.get(nums[i]) + 1);
            } else {
                map.put(nums[i], 1);
            }
        }
        for (//to cheak count of element
        //to cheak count of element
        int key : map.keySet()) {
            if (map.get(key) > (n / 2)) {
                return key;
            }
        }
        return 0;
    }
}
