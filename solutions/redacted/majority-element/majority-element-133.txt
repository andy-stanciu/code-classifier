class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                var1 = nums[i];
            }
            if (nums[i] == var1) {
                count++;
            } else {
                count--;
            }
        }
        return var1;
    }
}
