class Solution {

    public int method1(int[] num) {
        int result = num[0];
        int start = 1;
        for (int i = 1; i < num.length; i++) {
            if (start == 0) {
                result = num[i];
                start = 1;
            } else {
                if (num[i] == result)
                    start++;
                else
                    start--;
            }
        }
        //for i
        return result;
    }
}
