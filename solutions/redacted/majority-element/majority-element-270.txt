class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            map.put(num, map.getOrDefault(num, 0) + 1);
            // Check if the current element\'s count is more than half of the array length
            if (map.get(num) > nums.length / 2) {
                return num;
            }
        }
        // In case no majority element is found (though it is guaranteed in the problem)
        return -1;
    }
}
