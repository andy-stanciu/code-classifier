class Solution {

    public int method1(int[] nums) {
        int cnt = 0;
        Integer var1 = null;
        for (int num : nums) {
            if (cnt == 0) {
                var1 = num;
            }
            if (num == var1) {
                cnt++;
            } else {
                cnt--;
            }
        }
        return var1;
    }
}
