class Solution {

    public int method1(int[] nums) {
        // Initialize candidate as the first element
        int candidate = nums[0];
        // Initialize count of candidate
        int count = 0;
        // Traverse through the array
        for (int num : nums) {
            // If the current number matches the candidate, increment the count
            if (num == candidate) {
                count++;
            } else {
                // If the current number doesn\'t match the candidate, decrement the count
                count--;
                // If count becomes 0, update the candidate to the current number
                if (count == 0) {
                    candidate = num;
                    count = 1;
                }
            }
        }
        // Return the candidate, which is the majority element
        return candidate;
    }
}
