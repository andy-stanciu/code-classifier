class Solution {

    public int method1(int[] nums) {
        int arr = nums.length;
        Map<Integer, Integer> data = new HashMap<>();
        for (int i = 0; i < arr; i++) {
            data.put(nums[i], data.getOrDefault(nums[i], 0) + 1);
        }
        arr = arr / 2;
        for (Map.Entry<Integer, Integer> entry : data.entrySet()) {
            if (entry.getValue() > arr) {
                return entry.getKey();
            }
        }
        return 0;
    }
}
