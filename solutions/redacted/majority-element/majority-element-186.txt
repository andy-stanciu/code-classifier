class Solution {

    public int method1(int[] nums) {
        int count = 0;
        int first = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (count == 0) {
                first = nums[i];
                count++;
            } else if (first == nums[i]) {
                count++;
            } else {
                count--;
            }
        }
        return first;
    }
}
