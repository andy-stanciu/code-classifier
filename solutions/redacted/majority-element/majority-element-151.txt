class Solution {

    public int method1(int[] nums) {
        // Boyer-Moore algorithm approach (for majority element problems)
        int count = 0;
        int var1 = -1;
        for (int curr : nums) {
            // if count reaches 0, set the current element as the majority element
            if (count == 0)
                var1 = curr;
            // increment count if curr does not change; else, decrement count
            if (curr == var1)
                count++;
            else
                count--;
        }
        return var1;
        // time complexity: O(n)
        // we iterate through an array with n items
        // space complexity: O(1)
        // we only store data in local variables
    }
}
