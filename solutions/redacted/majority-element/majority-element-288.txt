class Solution {

    public int method1(int[] nums) {
        int[] eleFreq = new int[] { 0, 0 };
        for (int num : nums) {
            if (eleFreq[0] == num)
                ++eleFreq[1];
            else if (--eleFreq[1] <= 0) {
                eleFreq[0] = num;
                eleFreq[1] = 1;
            }
        }
        return eleFreq[0];
    }
}
