class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> hashMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            hashMap.put(nums[i], hashMap.getOrDefault(nums[i], 0) + 1);
        }
        int result = 0;
        int n = nums.length / 2;
        for (int res : hashMap.keySet()) {
            if (hashMap.get(res) > n) {
                return res;
            }
        }
        return result;
    }
}
