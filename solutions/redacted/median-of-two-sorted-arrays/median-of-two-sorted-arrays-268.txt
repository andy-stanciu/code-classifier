class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int[] merged = method2(nums1, nums2);
        int n = merged.length;
        if (n % 2 != 0) {
            //if length of the array is odd then the middle element is the median of the array
            return (double) merged[n / 2];
        }
        //if the array is of even length then the average of middle two elements is the median
        int prevmid = merged[(n - 1) / 2];
        int mid = merged[n / 2];
        //average of both middle elements
        return (double) (prevmid + mid) / 2;
    }

    public int[] method2(int[] arr1, int[] arr2) {
        //first we have to merge those array and return sorted
        int[] mergedArray = new int[arr1.length + arr2.length];
        // Copy elements from the first array
        System.arraycopy(arr1, 0, mergedArray, 0, arr1.length);
        // Copy elements from the second array
        System.arraycopy(arr2, 0, mergedArray, arr1.length, arr2.length);
        //after merging return the sorted version of merged array
        Arrays.sort(mergedArray);
        return mergedArray;
    }
}
