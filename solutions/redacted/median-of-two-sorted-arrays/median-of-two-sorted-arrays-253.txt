class Solution {

    public double method1(int[] nums1, int[] nums2) {
        //getting length of arrays
        int m = nums1.length;
        int n = nums2.length;
        //adding an array so that it contains merged array
        int[] merge = new int[n + m];
        //for loop to merge two arrays
        //for adding the values
        int k = 0;
        for (int i = 0; i < m; i++) {
            //giving nums1 array to merge array
            merge[k++] = nums1[i];
        }
        for (int i = 0; i < n; i++) {
            //giving nums2 array to merge array
            merge[k++] = nums2[i];
        }
        //sorting merged array
        Arrays.sort(merge);
        int len = (m + n);
        if (len % 2 == 1) {
            //for odd number of arrays
            return (double) merge[len / 2];
        } else {
            int middle1 = merge[len / 2];
            int middle2 = merge[len / 2 - 1];
            //for even number of arrays
            return (double) (middle1 + middle2) / 2;
        }
    }
}
