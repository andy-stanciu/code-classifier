class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        //Creating the new array calles nums with size m+n.
        int[] nums = new int[m + n];
        int low = 0;
        int h = 0, k = 0;
        while (low < m && h < n) {
            if (nums1[low] > nums2[h])
                nums[k++] = nums2[h++];
            else
                nums[k++] = nums1[low++];
        }
        while (low < m) nums[k++] = nums1[low++];
        while (h < n) nums[k++] = nums2[h++];
        if ((m + n) % 2 == 0) {
            int mid1 = (m + n) / 2;
            int mid2 = mid1 - 1;
            return (nums[mid1] + nums[mid2]) / 2.0;
        }
        return nums[(m + n) / 2];
    }
}
