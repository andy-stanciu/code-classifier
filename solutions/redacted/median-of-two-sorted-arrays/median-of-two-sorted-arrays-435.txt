class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;
        int[] mergeArr = new int[len1 + len2];
        System.arraycopy(nums1, 0, mergeArr, 0, len1);
        System.arraycopy(nums2, 0, mergeArr, len1, len2);
        Arrays.sort(mergeArr);
        int len3 = mergeArr.length;
        double res = 0;
        if (len3 % 2 == 0) {
            int m = (len3 / 2) - 1;
            int n = len3 / 2;
            res = (mergeArr[m] + mergeArr[n]) / 2.0;
        } else {
            res = mergeArr[len3 / 2];
        }
        return res;
    }
}
