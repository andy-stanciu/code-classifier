class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int i1 = 0, i2 = 0;
        int[] var1 = new int[nums1.length + nums2.length];
        for (int i = 0; i < var1.length; i++) {
            if (i1 < nums1.length && i2 < nums2.length) {
                if (nums1[i1] < nums2[i2]) {
                    var1[i] = nums1[i1];
                    i1++;
                } else if (nums1[i1] > nums2[i2]) {
                    var1[i] = nums2[i2];
                    i2++;
                } else {
                    var1[i] = nums1[i1];
                    i1++;
                    var1[++i] = nums2[i2];
                    i2++;
                }
            } else if (i1 < nums1.length) {
                var1[i] = nums1[i1];
                i1++;
            } else {
                var1[i] = nums2[i2];
                i2++;
            }
        }
        if (var1.length % 2 == 0) {
            return (var1[var1.length / 2 - 1] + var1[var1.length / 2]) / 2.0f;
        }
        return var1[var1.length / 2];
    }
}
