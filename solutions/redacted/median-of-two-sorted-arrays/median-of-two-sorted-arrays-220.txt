class Solution {

    public double method1(int[] arr1, int[] arr2) {
        int m = arr1.length;
        int n = arr2.length;
        //pointer for first array
        int i = 0;
        //pointer for second array
        int j = 0;
        int[] merged = new int[m + n];
        //pointer for merged array
        int x = 0;
        while (i < m && j < n) {
            if (arr1[i] < arr2[j]) {
                merged[x++] = arr1[i++];
            } else {
                merged[x++] = arr2[j++];
            }
        }
        while (i < m) {
            merged[x++] = arr1[i++];
        }
        while (j < n) {
            merged[x++] = arr2[j++];
        }
        //Merged is ready
        int mid = (m + n) / 2;
        double var1;
        //even
        if ((m + n) % 2 == 0) {
            //indexes
            int prev = mid - 1;
            var1 = Double.valueOf(merged[mid] + merged[prev]) / 2;
        } else //odd
        {
            var1 = merged[mid];
        }
        return var1;
    }
}
