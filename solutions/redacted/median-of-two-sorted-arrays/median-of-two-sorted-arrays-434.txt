class Solution {

    public double method1(int[] nums1, int[] nums2) {
        List<Integer> nums3 = new ArrayList<>();
        int n1 = nums1.length;
        int n2 = nums2.length;
        int i = 0, j = 0;
        while (i < n1 && j < n2) {
            if (nums1[i] < nums2[j]) {
                nums3.add(nums1[i]);
                i++;
            } else {
                nums3.add(nums2[j]);
                j++;
            }
        }
        // for remaining elements :
        while (i < n1) {
            nums3.add(nums1[i]);
            i++;
        }
        while (j < n2) {
            nums3.add(nums2[j]);
            j++;
        }
        int n = n1 + n2;
        if (n % 2 == 1) {
            return nums3.get(n / 2);
        } else {
            return (nums3.get(n / 2) + nums3.get(n / 2 - 1)) / 2.0;
        }
    }
}
