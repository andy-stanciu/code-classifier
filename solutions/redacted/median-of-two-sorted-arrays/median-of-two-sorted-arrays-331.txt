class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int n = nums1.length;
        int m = nums2.length;
        //Create a new Array that holds the numbers after merging
        int[] merged = new int[n + m];
        int k = 0;
        // add nums1 array numbers to new array
        for (int i = 0; i < n; i++) {
            merged[k++] = nums1[i];
        }
        // add nums2 array numbers to new array
        for (int i = 0; i < m; i++) {
            merged[k++] = nums2[i];
        }
        // After merging both nums1 and nums2 now Sort them.
        Arrays.sort(merged);
        int total = merged.length;
        // Use basic median Formula here
        if (total % 2 == 1) {
            return (double) merged[total / 2];
        } else {
            int middle1 = merged[total / 2 - 1];
            int middle2 = merged[total / 2];
            return ((double) middle1 + (double) middle2) / 2.0;
        }
    }
}
