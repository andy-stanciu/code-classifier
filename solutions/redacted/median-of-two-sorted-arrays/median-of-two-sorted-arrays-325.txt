class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int m = nums1.length;
        int n = nums2.length;
        List<Integer> l = new ArrayList<>();
        int k = 0;
        for (int num : nums1) {
            l.add(num);
        }
        for (int num : nums2) {
            l.add(num);
        }
        Collections.sort(l);
        int total = l.size();
        if (total % 2 != 0) {
            return (double) l.get(total / 2);
        } else {
            int m1 = l.get(total / 2);
            int m2 = l.get((total / 2) - 1);
            return (double) ((m1 + m2) / 2.0);
        }
    }
}
