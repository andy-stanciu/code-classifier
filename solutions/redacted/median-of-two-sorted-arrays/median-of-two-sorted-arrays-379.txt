class Solution {

    public double method1(int[] nums1, int[] nums2) {
        // Massiv uzunliklarini tekshirish
        if (nums1.length > 1000 || nums2.length > 1000 || nums1.length + nums2.length > 2000) {
            throw new IllegalArgumentException("Massiv uzunliklari cheklovlarga mos kelmaydi");
        }
        // Kichikroq massivni nums1 qilib olamiz
        if (nums1.length > nums2.length) {
            return method1(nums2, nums1);
        }
        int x = nums1.length;
        int y = nums2.length;
        int low = 0;
        int high = x;
        while (low <= high) {
            int partitionX = (low + high) / 2;
            int partitionY = (x + y + 1) / 2 - partitionX;
            int maxLeftX = (partitionX == 0) ? Integer.MIN_VALUE : nums1[partitionX - 1];
            int minRightX = (partitionX == x) ? Integer.MAX_VALUE : nums1[partitionX];
            int maxLeftY = (partitionY == 0) ? Integer.MIN_VALUE : nums2[partitionY - 1];
            int minRightY = (partitionY == y) ? Integer.MAX_VALUE : nums2[partitionY];
            if (maxLeftX <= minRightY && maxLeftY <= minRightX) {
                // To\'g\'ri bo\'linish nuqtasini topdik
                if ((x + y) % 2 == 0) {
                    return ((double) Math.max(maxLeftX, maxLeftY) + Math.min(minRightX, minRightY)) / 2;
                } else {
                    return (double) Math.max(maxLeftX, maxLeftY);
                }
            } else if (maxLeftX > minRightY) {
                // X ning o\'ng tomoniga siljish kerak
                high = partitionX - 1;
            } else {
                // X ning chap tomoniga siljish kerak
                low = partitionX + 1;
            }
        }
        throw new IllegalArgumentException("Massivlar tartiblangan emas");
    }
}
