class Solution {

    public double method1(int[] nums1, int[] nums2) {
        double ans = 0, ans1 = 0;
        int[] arr = new int[nums1.length + nums2.length];
        int j = 0;
        for (int i : nums1) {
            arr[j++] = i;
        }
        for (int i : nums2) {
            arr[j++] = i;
        }
        Arrays.sort(arr);
        if (arr.length % 2 == 0) {
            ans = (arr[(arr.length / 2) - 1] + arr[(arr.length / 2)]);
            ans1 = (double) ans / 2;
        } else {
            ans1 = arr[(arr.length / 2)];
        }
        return ans1;
    }
}
