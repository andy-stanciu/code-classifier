class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int n1 = nums1.length;
        int n2 = nums2.length;
        int m = n1 + n2;
        int[] a = new int[m];
        for (int i = 0, j = 0; i < m && j < n1; i++, j++) {
            a[i] = nums1[j];
        }
        for (int i = n1, j = 0; i < m && j < n2; i++, j++) {
            a[i] = nums2[j];
        }
        Arrays.sort(a);
        double var1;
        if (m % 2 != 0) {
            int mid = m / 2;
            var1 = a[mid];
        } else {
            int mid1 = m / 2;
            int mid2 = m / 2 - 1;
            var1 = (double) (a[mid1] + a[mid2]) / 2;
        }
        return var1;
    }
}
