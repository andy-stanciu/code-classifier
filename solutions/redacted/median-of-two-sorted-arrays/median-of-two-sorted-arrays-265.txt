class Solution {

    public double method1(int[] nums1, int[] nums2) {
        ArrayList<Integer> l = new ArrayList<>();
        for (int i : nums1) {
            l.add(i);
        }
        for (int i : nums2) {
            l.add(i);
        }
        Collections.sort(l);
        int s = l.size();
        if (s % 2 == 0) {
            return (double) (l.get(s / 2 - 1) + l.get(s / 2)) / 2;
        }
        return l.get(s / 2);
    }
}
