class Solution {

    public double method1(int[] nums1, int[] nums2) {
        int arr1 = nums1.length;
        int arr2 = nums2.length;
        int len = arr1 + arr2;
        int l = 0;
        int r = 0;
        int m = len / 2;
        int me = 0;
        int me1 = 0;
        for (int i = 0; i <= m; i++) {
            me1 = me;
            if (l < arr1 && (r >= arr2 || nums1[l] <= nums2[r])) {
                me = nums1[l++];
            } else {
                me = nums2[r++];
            }
        }
        if (len % 2 == 0) {
            return (me1 + me) / 2.0;
        }
        return me * 1.0;
    }
}
