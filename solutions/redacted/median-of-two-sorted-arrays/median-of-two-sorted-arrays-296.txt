class Solution {

    public int[] method1(int[] arr1, int[] arr2) {
        int[] mergedArray = new int[arr1.length + arr2.length];
        System.arraycopy(arr1, 0, mergedArray, 0, arr1.length);
        System.arraycopy(arr2, 0, mergedArray, arr1.length, arr2.length);
        return mergedArray;
    }

    public static double method2(int[] array) {
        Arrays.sort(array);
        int n = array.length;
        if (n % 2 == 1) {
            return array[n / 2];
        } else {
            return (array[(n / 2) - 1] + array[n / 2]) / 2.0;
        }
    }

    public double method3(int[] nums1, int[] nums2) {
        int[] mergedArray = method1(nums1, nums2);
        return method2(mergedArray);
    }
}
