class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        //make a hashset so it doesn\'t copy the same element in an ans list
        Set<List<Integer>> set = new HashSet<>();
        Arrays.sort(nums);
        //run a loop starting from ith index to 4 indices by taking j,k,t elements
        for (int i = 0; i < nums.length - 3; i++) {
            for (int j = i + 1; j < nums.length - 2; j++) {
                int k = j + 1;
                int t = nums.length - 1;
                while (t > k) {
                    long sum = (long) nums[i] + nums[j] + nums[k] + nums[t];
                    //check the sum if it is equal then add elements in set list
                    if (sum == target) {
                        set.add(Arrays.asList(nums[i], nums[j], nums[k], nums[t]));
                        t--;
                        k++;
                        // if element at t index is equal to t+1 index then skip that index thereby, runtime will take less time.then same goes for k indices too.
                        while (t > k && nums[t] == nums[t + 1]) t--;
                        while (t > k && nums[k] == nums[k - 1]) k++;
                    } else // if sum is less then increase kth index or else decrease tth index
                    if (sum < target)
                        k++;
                    else
                        t--;
                }
            }
        }
        //copy set elements into list bcoz return must be in list<list<Integer>>
        ans.addAll(set);
        return ans;
    }
}
