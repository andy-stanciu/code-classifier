class Solution {

    public int method1(int[] nums) {
        int s = 0, e = nums.length - 1, midIdx = 0;
        while (s < e) {
            midIdx = (s + e) / 2;
            if (nums[midIdx] < nums[e]) {
                e = midIdx;
            } else {
                s = midIdx + 1;
            }
        }
        return nums[s];
    }
}
