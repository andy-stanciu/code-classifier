class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int low = 0;
        int high = n - 1;
        int ans = Integer.MAX_VALUE;
        while (low <= high) {
            //System.out.println(low + " " + high);
            int mid = low + (high - low) / 2;
            if (nums[mid] <= nums[n - 1]) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return nums[ans];
    }
}
