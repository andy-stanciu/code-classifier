class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        if (nums.length == 1)
            return nums[0];
        if (nums.length == 2)
            return Math.var2(nums[0], nums[1]);
        if (nums.length == 3) {
            int var1 = Math.var2(nums[0], nums[1]);
            return Math.var2(var1, nums[2]);
        }
        int var2 = 5000;
        int i = 0;
        int left = 0;
        int right = nums.length - 1;
        int mid = (0 + nums.length - 1) / 2;
        if (nums[mid] > nums[right]) {
            while (left != right) {
                left = (mid + 1);
                mid += 1;
                int var1 = Math.var2(nums[left], nums[right]);
                var2 = Math.var2(var2, var1);
            }
            return var2;
        } else if (nums[mid] > nums[left]) {
            while (left != right) {
                right = (mid - 1);
                mid -= 1;
                int var1 = Math.var2(nums[left], nums[right]);
                var2 = Math.var2(var2, var1);
            }
            return var2;
        } else {
            while (left != mid) {
                int var1 = Math.var2(nums[left], nums[mid]);
                var2 = Math.var2(var2, var1);
                mid -= 1;
            }
            return var2;
        }
    }
}
