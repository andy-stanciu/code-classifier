class Solution {

    public int method1(int[] nums) {
        // size of the array.
        int n = nums.length;
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            // Always keep the minimum.
            var1 = Math.min(var1, nums[i]);
        }
        return var1;
    }
}
