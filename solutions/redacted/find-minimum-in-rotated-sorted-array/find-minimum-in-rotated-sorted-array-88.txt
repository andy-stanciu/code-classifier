class Solution {

    public int method1(int[] nums) {
        if (nums.length == 2) {
            return Math.min(nums[0], nums[1]);
        } else if (nums.length == 1) {
            return nums[0];
        }
        int var1 = Integer.MAX_VALUE;
        int l = 0;
        int h = nums.length - 1;
        while (l <= h) {
            int mid = (l + h) / 2;
            if (nums[l] < nums[h]) {
                var1 = Math.min(var1, nums[l]);
                // Corrected: Added semicolon
                break;
            } else if (nums[l] <= nums[mid]) {
                var1 = Math.min(var1, nums[l]);
                l = mid + 1;
            } else {
                var1 = Math.min(var1, nums[mid]);
                h = mid - 1;
            }
        }
        return var1;
    }
}
