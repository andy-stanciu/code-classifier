class Solution {

    public int method1(int[] nums) {
        int pivot = method2(nums);
        if (pivot == -1) {
            // The array is not rotated, return the first element
            return nums[0];
        }
        // Return the element right after the pivot, which is the minimum element
        return nums[pivot + 1];
    }

    public int method2(int[] arr) {
        int start = 0;
        int end = arr.length - 1;
        // Handle the case where the array is not rotated
        if (arr[start] < arr[end]) {
            return -1;
        }
        while (start <= end) {
            int mid = start + (end - start) / 2;
            // Check if mid element is the pivot
            if (mid < end && arr[mid] > arr[mid + 1]) {
                return mid;
            }
            if (mid > start && arr[mid] < arr[mid - 1]) {
                return mid - 1;
            }
            // Decide which part to search next
            if (arr[start] >= arr[mid]) {
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        // No pivot found, array is not rotated
        return -1;
    }
}
