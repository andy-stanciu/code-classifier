class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int ans = 2147483647;
        int low = 0;
        int high = n - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (nums[low] <= nums[mid]) {
                ans = Math.min(ans, nums[low]);
                low = mid + 1;
            } else {
                ans = Math.min(ans, nums[mid]);
                high = high - 1;
            }
        }
        return ans;
    }
}
