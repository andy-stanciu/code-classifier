class Solution {

    public int method1(int[] nums) {
        int res = nums[0];
        int left = 0, right = nums.length - 1;
        while (left <= right) {
            if (nums[left] < nums[right]) {
                res = Math.min(res, nums[left]);
            }
            int mid = left + (right - left) / 2;
            res = Math.min(res, nums[mid]);
            if (nums[mid] >= nums[left]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return res;
    }
}
