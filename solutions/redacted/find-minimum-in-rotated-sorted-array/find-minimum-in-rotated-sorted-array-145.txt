class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (i + 1 < nums.length && nums[i] > nums[i + 1]) {
                var1 = nums[i + 1];
                break;
            }
        }
        return var1;
    }
}
