class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int s = 0, e = nums.length - 1, mid, ans = nums[0];
        while (s <= e) {
            mid = s + (e - s) / 2;
            if (nums[mid] > nums[mid + 1]) {
                ans = nums[mid + 1];
                break;
            } else if (nums[mid] < nums[e])
                e = mid;
            else
                s = mid + 1;
        }
        return ans;
    }
}
