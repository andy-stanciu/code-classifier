class Solution {

    int pivot(int[] nums) {
        int lo = 0;
        int hi = nums.length - 1;
        int mid;
        while (lo <= hi) {
            mid = (lo + hi) >> 1;
            if (mid + 1 > hi) {
                return hi;
            }
            if (nums[mid] > nums[mid + 1]) {
                return mid;
            }
            if (nums[mid] < nums[0]) {
                hi = mid - 1;
            } else {
                lo = mid + 1;
            }
        }
        return 0;
    }

    public int method1(int[] nums) {
        int pivot = pivot(nums);
        if (pivot == nums.length - 1)
            return nums[0];
        return nums[pivot + 1];
    }
}
