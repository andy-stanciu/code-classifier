class Solution {

    public int method1(int[] nums) {
        int l = 0, h = nums.length - 1, res = Integer.MAX_VALUE;
        while (l <= h) {
            int m = l + (h - l) / 2;
            if (nums[l] <= nums[m]) {
                //left subarray is sorted
                res = Math.min(res, nums[l]);
                l = m + 1;
            } else {
                //right subarray is sorted
                res = Math.min(res, nums[m]);
                h = m - 1;
            }
        }
        return res;
    }
}
