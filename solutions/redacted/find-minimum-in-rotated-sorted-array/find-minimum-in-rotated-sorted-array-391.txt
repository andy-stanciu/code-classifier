class Solution {

    public int method1(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = (high + low) / 2;
            if (//if 0th element is smaller than mid means left portion is sorted..
            nums[low] <= nums[mid]) {
                var1 = Math.var1(var1, nums[low]);
                low = mid + 1;
            } else if (//if mid element is smaller than n-1th element so right side is sorted and mid is smallest.
            nums[mid] <= nums[high]) {
                var1 = Math.var1(var1, nums[mid]);
                high = mid - 1;
            }
        }
        return var1;
    }
}
