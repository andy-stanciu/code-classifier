class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] < nums[i + 1])
                continue;
            else {
                if (nums[i + 1] < var1)
                    var1 = nums[i + 1];
                i++;
            }
        }
        return var1;
    }
}
