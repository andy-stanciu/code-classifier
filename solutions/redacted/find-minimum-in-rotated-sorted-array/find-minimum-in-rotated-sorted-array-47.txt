class Solution {

    public int method1(int[] a) {
        int l = 0, r = a.length - 1;
        int curr = 0;
        int var1 = Integer.MAX_VALUE;
        while (l < r) {
            int mid = (l + r) / 2;
            if (a[mid] < a[r]) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return a[l];
    }
}
