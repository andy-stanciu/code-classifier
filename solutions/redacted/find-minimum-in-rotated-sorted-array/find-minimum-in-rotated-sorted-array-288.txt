class Solution {

    public int method1(int[] nums) {
        return method2(nums, 0, nums.length - 1);
    }

    int method2(int[] nums, int i, int j) {
        int p = (i + j) / 2;
        if (i == j) {
            return nums[i];
        }
        return Math.min(method2(nums, i, p), method2(nums, p + 1, j));
    }
}
