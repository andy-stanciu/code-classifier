class Solution {

    public int method1(int[] nums, int param1, int param2) {
        // base case: array between leftIndex and rightIndex is sorted ?
        if (nums[param1] <= nums[param2]) {
            return nums[param1];
        }
        int key = param1 + (param2 - param1) / 2;
        return nums[key] > nums[param2] ? method1(nums, key + 1, param2) : method1(nums, param1, key);
    }

    public int method1(int[] nums) {
        // binary search , find range where nums [i] > nums [j]
        return method1(nums, 0, nums.length - 1);
    }
}
