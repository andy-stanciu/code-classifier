class Solution {

    int[] arr;

    public int method1(int[] nums) {
        this.arr = nums;
        //    StartIndex = 0, EndIndex = nums.length-1
        return method1(0, nums.length - 1);
    }

    int method1(int si, int ei) {
        if (si == ei) {
            return arr[si];
        }
        if (arr[si] < arr[ei])
            return arr[si];
        // when only two elements, to avoid false recursion
        if (ei - si == 1) {
            return Math.min(arr[si], arr[ei]);
        }
        int mid = (si + ei + 1) / 2;
        if (arr[si] < arr[mid])
            return method1(mid + 1, ei);
        else
            return method1(si, mid);
    }
}
