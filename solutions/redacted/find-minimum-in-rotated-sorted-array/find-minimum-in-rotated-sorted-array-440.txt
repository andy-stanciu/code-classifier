class Solution {

    public int method1(int[] arr) {
        int start = 0, end = arr.length - 1, mid = 0, var1 = 5001;
        while (start <= end) {
            mid = start + (end - start) / 2;
            if (arr[start] <= arr[mid]) {
                var1 = Math.var1(var1, arr[start]);
                start = mid + 1;
            } else {
                var1 = Math.var1(var1, arr[mid + 1]);
                end = mid;
            }
        }
        return var1;
    }
}
