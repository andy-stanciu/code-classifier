class Solution {

    public int method1(int[] arr) {
        int low = 0, high = arr.length - 1;
        int ans = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = (low + high) / 2;
            //Check for right sorted or not
            if (arr[low] <= arr[mid]) {
                //if sorted, then take arr[low] as min;
                ans = Math.min(ans, arr[low]);
                //incre low;
                low = mid + 1;
            } else //Check if right is sorted
            {
                //Then obviously the arr[mid] will be the smallest value;
                ans = Math.min(ans, arr[mid]);
                high = mid - 1;
            }
        }
        return ans;
    }
}
