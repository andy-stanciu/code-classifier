class Solution {

    public int method1(int[] a) {
        int low = 0, high = a.length - 1, var1 = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = (low + high) >> 1;
            if (a[low] <= a[high]) {
                var1 = Math.var1(var1, a[low]);
                break;
            }
            if (a[low] <= a[mid]) {
                var1 = Math.var1(var1, a[low]);
                low = mid + 1;
            } else {
                var1 = Math.var1(var1, a[mid]);
                high = mid - 1;
            }
        }
        return var1;
    }
}
