class Solution {

    public int method1(int[] nums) {
        int start = 0, end = nums.length - 1, mid, ans = nums[0];
        while (start <= end) {
            mid = start + (end - start) / 2;
            //left rotate check
            if (nums[mid] >= nums[0]) {
                start = mid + 1;
            } else //right side check
            {
                ans = nums[mid];
                end = mid - 1;
            }
        }
        return ans;
    }
}
