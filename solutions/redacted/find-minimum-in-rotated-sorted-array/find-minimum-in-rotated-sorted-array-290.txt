class Solution {

    public int method1(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            if (nums[mid] >= nums[start]) {
                var1 = Math.var1(var1, nums[start]);
                start = mid + 1;
            } else {
                var1 = Math.var1(var1, nums[mid]);
                end = mid - 1;
            }
        }
        return var1;
    }
}
