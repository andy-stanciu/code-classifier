class Solution {

    public int method1(int[] nums) {
        int len = nums.length - 1;
        // if the array is in ascending order then first element is the min
        if (nums[len] > nums[0])
            return nums[0];
        int i = 0, j = len;
        while (i < j) {
            int mid = (i + j) / 2;
            if (//discard left elements
            nums[mid] > nums[0] || nums[mid] > nums[len])
                i = mid + 1;
            else if (//discard right elements
            nums[mid] < nums[0])
                j = mid;
        }
        return nums[i];
    }
}
