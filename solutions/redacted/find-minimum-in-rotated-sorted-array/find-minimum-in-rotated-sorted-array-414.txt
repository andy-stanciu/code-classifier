class Solution {

    public int method1(int[] nums) {
        int si = 0;
        int ei = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (si <= ei) {
            int mid = si + (ei - si) / 2;
            if (nums[si] <= nums[mid]) {
                var1 = Math.var1(nums[si], var1);
                si++;
            } else {
                var1 = Math.var1(nums[mid], var1);
                ei--;
            }
        }
        return var1;
    }
}
