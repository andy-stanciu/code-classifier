class Solution {

    public int method1(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        while (start < end) {
            int mid = start + (end - start) / 2;
            if (nums[mid] > nums[end]) {
                // The minimum element must be in the right half
                start = mid + 1;
            } else {
                // The minimum element could be the mid element or in the left half
                end = mid;
            }
        }
        return nums[start];
    }
}
