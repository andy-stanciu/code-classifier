class Solution {

    public int method1(int[] nums) {
        // start the minimum with the first element
        int var1 = nums[0];
        // left and right pointers for binary search
        int left = 0;
        int right = nums.length - 1;
        while (left <= right) {
            int middle = (right + left) / 2;
            int currNum = nums[middle];
            // update minimum
            if (currNum < var1) {
                var1 = currNum;
            }
            // update the left and right pointers based
            // on where in the rotated array we are at
            if (currNum < nums[right]) {
                right = middle - 1;
            } else {
                left = middle + 1;
            }
        }
        return var1;
    }
}
