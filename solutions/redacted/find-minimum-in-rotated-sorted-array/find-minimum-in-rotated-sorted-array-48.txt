class Solution {

    public int method1(int[] nums) {
        int l = 0;
        int r = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (nums[mid] >= nums[l]) {
                var1 = Math.var1(var1, nums[l]);
                l = mid + 1;
            } else {
                var1 = Math.var1(var1, nums[mid]);
                r = mid - 1;
            }
        }
        return var1;
    }
}
