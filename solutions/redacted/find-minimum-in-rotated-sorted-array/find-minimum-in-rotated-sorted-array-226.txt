class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int ans = nums[0];
        int mid = (right + left) / 2;
        while (left <= right) {
            mid = (right + left) / 2;
            ans = ans > nums[left] ? nums[left] : ans;
            if (nums[mid] >= nums[left]) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return ans;
    }
}
