class Solution {

    public int method1(int[] nums) {
        int l = 0;
        int h = nums.length - 1;
        int ans = Integer.MAX_VALUE;
        while (l <= h) {
            int m = l + (h - l) / 2;
            if (nums[l] <= nums[m]) {
                ans = Math.min(ans, nums[l]);
                l = m + 1;
            } else {
                ans = Math.min(ans, nums[m]);
                h = m - 1;
            }
        }
        return ans;
    }
}
