class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int mid = 0;
        int var1 = Integer.MAX_VALUE;
        while (left < right) {
            mid = left + (right - left) / 2;
            if (nums[right] > nums[mid]) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return nums[left];
    }
}
