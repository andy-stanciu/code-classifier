class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int ret = nums[0];
        int ref = nums[n - 1];
        int l = 0, r = n - 1;
        while (l <= r) {
            int k = (l + r) / 2;
            if (k != 0 && nums[k - 1] > nums[k]) {
                ret = nums[k];
                break;
            }
            if (nums[k] < ref) {
                r = k - 1;
            } else
                l = k + 1;
        }
        return ret;
    }
}
