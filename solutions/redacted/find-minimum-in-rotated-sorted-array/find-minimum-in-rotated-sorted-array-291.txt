class Solution {

    public int method1(int[] nums) {
        int len = nums.length;
        if (len == 1) {
            return nums[0];
        }
        int l = 0, r = len - 1;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (mid > 0 && nums[mid] < nums[mid - 1]) {
                return nums[mid];
            } else if (nums[mid] > nums[len - 1]) {
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return nums[0];
    }
}
