class Solution {

    public int method1(int[] nums) {
        int start = 0, end = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            //Eliminate the sorted half, choose min from the sorted part
            //The sorted part of rotated sorted array MAY contain your answer, so eliminate it after choosing min.
            if (nums[start] <= nums[mid]) {
                // "<=" if array has two element
                var1 = Math.var1(var1, nums[start]);
                start = mid + 1;
            } else {
                var1 = Math.var1(var1, nums[mid]);
                end = mid - 1;
            }
        }
        return var1;
    }
}
