class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        // Loop through the array
        for (int i = 1; i < nums.length; i++) {
            // If a smaller value is found, update min
            if (nums[i] < var1) {
                var1 = nums[i];
            }
        }
        return var1;
    }
}
