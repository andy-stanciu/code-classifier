class Solution {

    public int method1(int[] nums) {
        int var1 = Integer.MAX_VALUE;
        int var2 = -1;
        int i = 0;
        while (i < nums.length) {
            if (nums[i] < var1) {
                var1 = Math.var1(var1, nums[i]);
                var2 = i;
            }
            i++;
        }
        return var1;
    }
}
