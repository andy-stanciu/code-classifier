class Solution {

    public int method1(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        while (left < right) {
            int mid = left + (right - left) / 2;
            // Check if the middle element is greater than the rightmost element
            if (nums[mid] > nums[right]) {
                // Minimum is in the right part
                left = mid + 1;
            } else {
                // Minimum is in the left part or at mid
                right = mid;
            }
        }
        // When left == right, that\'s the smallest element
        return nums[left];
    }
}
