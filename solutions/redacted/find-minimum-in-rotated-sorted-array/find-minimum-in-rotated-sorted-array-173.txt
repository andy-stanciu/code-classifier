class Solution {

    public int method1(int[] nums) {
        int start = 0;
        int end = nums.length - 1;
        while (start < end) {
            int mid = start + (end - start) / 2;
            // If mid element is greater than end element, min must be in the right half
            if (nums[mid] > nums[end]) {
                start = mid + 1;
            } else {
                // Otherwise, the min is in the left half (or could be mid)
                end = mid;
            }
        }
        // After the loop, start == end and it points to the smallest element
        return nums[start];
    }
}
