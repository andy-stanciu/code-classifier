class Solution {

    public int method1(int[] nums) {
        int low = 0, high = nums.length - 1, var1 = nums[0];
        while (low <= high) {
            // to handle the case when array is already sorted
            if (nums[low] < nums[high]) {
                // not putting <= because the array contains unique elements
                var1 = Math.var1(var1, nums[low]);
                break;
            }
            int mid = low + (high - low) / 2;
            var1 = Math.var1(var1, nums[mid]);
            if (// doing >= because of the edge case when mid and low points at the same index
            nums[mid] >= nums[high])
                low = mid + 1;
            else
                high = mid - 1;
        }
        return var1;
    }
}
