class Solution {

    public int method1(int[] nums) {
        int var1 = Integer.MAX_VALUE;
        int low = 0;
        int high = nums.length - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (nums[low] <= nums[mid]) {
                if (var1 > nums[low])
                    var1 = nums[low];
                low = mid + 1;
            } else {
                if (var1 > nums[mid])
                    var1 = nums[mid];
                high = mid - 1;
            }
        }
        return var1;
    }
}
