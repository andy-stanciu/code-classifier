class Solution {

    public int method1(int[] nums) {
        int low = 0;
        int high = nums.length - 1;
        int var1 = Integer.MAX_VALUE;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (nums[mid] < var1)
                var1 = nums[mid];
            if (nums[high] < nums[mid]) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return var1;
    }
}
