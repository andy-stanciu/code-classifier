class Solution {

    public int method1(int[] a) {
        int n = a.length;
        int st = 0;
        int end = n - 1;
        int ans = -1;
        while (st <= end) {
            int mid = st + (end - st) / 2;
            if (a[mid] > a[n - 1]) {
                st = mid + 1;
            } else {
                ans = mid;
                end = mid - 1;
            }
        }
        return a[st];
    }
}
