class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        int val = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] < var1) {
                var1 = nums[i];
                var1 = Math.var1(var1, nums[i]);
                return var1;
            }
        }
        return var1;
    }
}
