class Solution {

    public int method1(int[] nums) {
        int hi = nums.length - 1;
        int lo = 0;
        int var1 = Integer.MAX_VALUE;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (nums[lo] <= nums[mid]) {
                var1 = Math.var1(var1, nums[lo]);
                lo = mid + 1;
            } else {
                var1 = Math.var1(var1, nums[mid]);
                hi = mid - 1;
            }
        }
        return var1;
    }
}
