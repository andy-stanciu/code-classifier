class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int max = 0;
        int count = 1;
        // return 0 if nums is empty
        if (nums.length == 0)
            return 0;
        for (int i = 0; i < nums.length - 1; i++) {
            // check if same num is repeated
            if (nums[i + 1] == nums[i])
                continue;
            else //if value increments, increase count value
            if (nums[i + 1] == nums[i] + 1)
                count++;
            else //if both conditions break, update max and reset count
            {
                max = Math.max(count, max);
                count = 1;
            }
        }
        return Math.max(count, max);
    }
}
