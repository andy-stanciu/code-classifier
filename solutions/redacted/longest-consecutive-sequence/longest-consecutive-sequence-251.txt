class Solution {

    public int method1(int[] nums) {
        Set<Integer> st = new HashSet<>();
        //add to set to remove duplicates
        Arrays.stream(nums).forEach(st::add);
        int count = 0;
        for (int x : st) {
            //move forward only if it has no number before x
            if (!st.contains(x - 1)) {
                int length = 1;
                //start counting
                while (st.contains(x + length)) length++;
                //update the count
                count = Math.max(length, count);
            }
        }
        return count;
    }
}
