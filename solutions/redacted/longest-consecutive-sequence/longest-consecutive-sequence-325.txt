class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        // Sort the array
        Arrays.sort(nums);
        // To keep track of the length of the longest consecutive sequence
        int max = 1;
        // To keep track of the current consecutive sequence length
        int temp = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                // Skip duplicates
                continue;
            }
            if (nums[i] == nums[i - 1] + 1) {
                // Continuation of the consecutive sequence
                temp++;
            } else {
                // Update max and reset temp
                max = Math.max(max, temp);
                // Reset temp for a new sequence
                temp = 1;
            }
        }
        // Final check for the last sequence
        max = Math.max(max, temp);
        return max;
    }
}
