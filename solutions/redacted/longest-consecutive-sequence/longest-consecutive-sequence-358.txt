class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int var1 = 1;
        int var2 = 1;
        int n = nums.length;
        if (n == 0)
            return 0;
        for (int i = 1; i < n; i++) {
            if (nums[i] == nums[i - 1])
                continue;
            if (nums[i] - 1 == nums[i - 1]) {
                var2++;
                if (var2 > var1)
                    var1 = var2;
            } else
                var2 = 1;
        }
        return var1;
    }
}
