class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            set.add(num);
        }
        int var1 = 0;
        for (int num : nums) {
            if (!set.contains(num))
                continue;
            int backward = num - 1;
            int forward = num + 1;
            while (set.contains(backward)) {
                set.remove(backward);
                backward--;
            }
            while (set.contains(forward)) {
                set.remove(forward);
                forward++;
            }
            var1 = Math.max(var1, forward - backward - 1);
        }
        return var1;
    }
}
