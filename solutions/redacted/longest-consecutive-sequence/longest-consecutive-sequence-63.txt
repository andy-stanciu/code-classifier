class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            //if length is 0
            return 0;
        }
        //sorted array
        Arrays.sort(nums);
        // number of elements found consecutive
        int var1 = 1;
        // current degree of consecutive numbers
        int currentStreak = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                if (nums[i] == nums[i - 1] + 1) {
                    currentStreak += 1;
                } else {
                    var1 = Math.max(var1, currentStreak);
                    currentStreak = 1;
                }
            }
        }
        //output is determined by which of the 2 is greater.
        return Math.max(var1, currentStreak);
    }
}
