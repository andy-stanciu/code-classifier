class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        Set<Integer> hs = new HashSet<>();
        for (int i : nums) {
            hs.add(i);
        }
        int result = 0;
        for (int j : hs) {
            if (!hs.contains(j - 1)) {
                int temp = 0;
                while (hs.contains(j + 1)) {
                    j++;
                    temp++;
                }
                result = Math.max(result, temp);
            }
        }
        return result + 1;
    }
}
