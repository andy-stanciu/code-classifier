class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        if (nums.length == 1)
            return 1;
        Arrays.sort(nums);
        int cnt = 1, max = 1, start = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (start + 1 == nums[i]) {
                start = nums[i];
                cnt++;
            } else if (start == nums[i]) {
                continue;
            } else {
                start = nums[i];
                cnt = 1;
            }
            max = Math.max(cnt, max);
        }
        return max;
    }
}
