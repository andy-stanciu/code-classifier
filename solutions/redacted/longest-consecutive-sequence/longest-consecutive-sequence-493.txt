class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        if (n == 0)
            return 0;
        int var1 = 1;
        int currentcount = 0;
        int lastsmaller = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastsmaller) {
                currentcount += 1;
                lastsmaller = nums[i];
            } else if (nums[i] != lastsmaller) {
                currentcount = 1;
                lastsmaller = nums[i];
            }
            var1 = Math.max(var1, currentcount);
        }
        return var1;
    }
}
