class Solution {

    public int method1(int[] nums) {
        if (nums.length < 1) {
            return 0;
        }
        Arrays.sort(nums);
        int cc = 1;
        int ans = 1;
        int it = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                continue;
            }
            if (nums[i] - 1 == it) {
                cc++;
                it = nums[i];
            } else {
                cc = 1;
                it = nums[i];
            }
            ans = ans > cc ? ans : cc;
        }
        return ans;
    }
}
