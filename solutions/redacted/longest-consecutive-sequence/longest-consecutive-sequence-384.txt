class Solution {

    public int method1(int[] nums) {
        // If the array is empty, return 0
        if (nums.length == 0)
            return 0;
        //sort the nums Array
        Arrays.sort(nums);
        // Set<Integer> set = new HashSet<>();
        int var1 = 1;
        int currentStreak = 1;
        for (int i = 1; i < nums.length; i++) {
            // Iterate the sorted array starting from the second element
            if (nums[i] != nums[i - 1]) {
                // Skip duplicates
                if (nums[i] == nums[i - 1] + 1) {
                    // If the current number is exactly one more than theprevious number
                    currentStreak++;
                } else {
                    if (currentStreak > var1) {
                        // Update the longest streak
                        var1 = currentStreak;
                    }
                    //if not
                    // Reset current streak
                    currentStreak = 1;
                }
            }
        }
        // look to this exemple he will explain it [0,3,7,2,5,8,4,6,0,1]
        if (currentStreak > var1)
            return currentStreak;
        return var1;
        // or   return Math.max(longestStreak, currentStreak);
    }
}
