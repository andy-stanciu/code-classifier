class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        Arrays.sort(nums);
        int var1 = 1;
        int prev = Integer.MIN_VALUE;
        int var2 = 1;
        for (int i : nums) {
            if (i == prev)
                continue;
            if (i == prev + 1) {
                var2++;
                if (var2 > var1)
                    var1 = var2;
            } else {
                var2 = 1;
            }
            prev = i;
        }
        return var1;
    }
}
