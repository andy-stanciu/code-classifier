class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int l = Integer.MIN_VALUE;
        int count = 0;
        int lon = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] - 1 == l) {
                count++;
                l = nums[i];
            } else if (nums[i] != l) {
                count = 1;
                l = nums[i];
            }
            lon = Math.max(lon, count);
        }
        return lon;
    }
}
