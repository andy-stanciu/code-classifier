class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        int streak = 1;
        int maxStreak = 1;
        Arrays.sort(nums);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                continue;
            }
            if ((nums[i]) == nums[i - 1] + 1) {
                streak++;
            } else {
                maxStreak = Math.max(maxStreak, streak);
                streak = 1;
            }
            maxStreak = Math.max(maxStreak, streak);
        }
        return maxStreak;
    }
}
