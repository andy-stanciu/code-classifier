class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1) {
            return 1;
        }
        Set<Integer> HM = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            HM.add(nums[i]);
        }
        int length = 0;
        int var1 = 0;
        int lowValue = 0;
        int upValue = 0;
        for (int j = 0; j < nums.length; j++) {
            int k = nums[j];
            if (HM.remove(k)) {
                lowValue = k - 1;
                upValue = k + 1;
                var1 += 1;
            }
            while (HM.remove(lowValue)) {
                lowValue -= 1;
                var1 += 1;
            }
            while (HM.remove(upValue)) {
                var1 += 1;
                upValue += 1;
            }
            length = Math.max(length, var1);
            var1 = 0;
        }
        return length;
    }
}
