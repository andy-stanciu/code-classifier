class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        int max = 0;
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            hm.put(nums[i], i);
        }
        for (int i = 0; i < nums.length; i++) {
            if (!hm.containsKey(nums[i] - 1)) {
                int count = 1;
                int check = nums[i];
                while (hm.containsKey(check + 1)) {
                    count++;
                    check++;
                }
                max = Math.max(count, max);
            }
        }
        return max;
    }
}
