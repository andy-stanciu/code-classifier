class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return 0;
        // Sort the array
        Arrays.sort(nums);
        int lastSmaller = Integer.MIN_VALUE;
        int count = 0;
        int var1 = 1;
        // Iterate through the sorted array
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastSmaller) {
                // Current number is consecutive
                count += 1;
                lastSmaller = nums[i];
            } else if (nums[i] != lastSmaller) {
                // Reset count for a new sequence
                count = 1;
                lastSmaller = nums[i];
            }
            // Update longest sequence length
            var1 = Math.max(var1, count);
        }
        return var1;
    }
}
