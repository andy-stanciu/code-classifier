class Solution {

    public int method1(int[] nums) {
        int max = 1;
        Arrays.sort(nums);
        System.out.println(Arrays.toString(nums));
        int counter = 1;
        if (nums.length == 0) {
            return 0;
        }
        for (int i = 0; i < nums.length - 1; i++) {
            if ((nums[i]) == nums[i + 1]) {
                continue;
            }
            if ((nums[i] + 1) == nums[i + 1]) {
                counter++;
                if (counter > max) {
                    max = counter;
                }
            } else {
                counter = 1;
            }
        }
        return max;
    }
}
