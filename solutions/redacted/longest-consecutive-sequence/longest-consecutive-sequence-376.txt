class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        Arrays.sort(nums);
        int count = 1, max = 1;
        for (int i = 1; i < nums.length; i++) {
            // skip duplicates
            if (nums[i] == nums[i - 1])
                continue;
            if (nums[i] - 1 == nums[i - 1])
                count++;
            else
                // reset count var
                count = 1;
            max = Math.max(max, count);
        }
        return max;
    }
}
