class Solution {

    public int method1(int[] nums) {
        // Edge case: empty array
        if (nums.length == 0)
            return 0;
        Set<Integer> uniqueNumbers = new HashSet<>();
        for (int num : nums) {
            // Add all numbers to the set
            uniqueNumbers.add(num);
        }
        int var1 = 0;
        // Loop through the set to find only the starting points of sequences
        for (int num : uniqueNumbers) {
            if (!uniqueNumbers.contains(num - 1)) {
                // Only start a sequence if `num` is the start (no `num - 1`)
                int currentNum = num;
                int var2 = 1;
                // Count consecutive numbers starting from `num`
                while (uniqueNumbers.contains(currentNum + 1)) {
                    currentNum++;
                    var2++;
                }
                // Update the max length of the sequence
                var1 = Math.max(var1, var2);
            }
        }
        return var1;
    }
}
