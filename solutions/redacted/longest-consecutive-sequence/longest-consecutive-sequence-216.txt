class Solution {

    public int method1(int[] nums) {
        int var1 = 0;
        HashSet<Integer> set = new HashSet<>();
        for (int i : nums) set.add(i);
        for (int i : nums) {
            if (!set.contains(i - 1)) {
                int length = 1;
                while (set.contains(i + length)) {
                    length++;
                }
                var1 = Math.max(length, var1);
            }
        }
        return var1;
    }
}
