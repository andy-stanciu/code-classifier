class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        Arrays.sort(nums);
        if (n == 0) {
            return 0;
        }
        int var1 = 1, count = 0, lastSmall = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastSmall) {
                count += 1;
                lastSmall = nums[i];
            } else if (nums[i] != lastSmall) {
                count = 1;
                lastSmall = nums[i];
            }
            var1 = Math.max(var1, count);
        }
        return var1;
    }
}
