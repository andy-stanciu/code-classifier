class Solution {

    public int method1(int[] nums) {
        Set<Integer> hs = new HashSet<>();
        for (int num : nums) {
            hs.add(num);
        }
        int var1 = 0;
        for (int val : nums) {
            if (!hs.contains(val - 1)) {
                int count = 1;
                int currVal = val;
                while (hs.contains(currVal + 1)) {
                    count++;
                    currVal++;
                }
                var1 = Math.max(var1, count);
            }
        }
        return var1;
    }
}
