class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int k = nums.length;
        int maxCount = 1;
        int count = 1;
        if (nums.length == 0)
            return 0;
        if (nums.length == 1)
            return 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i - 1] != nums[i]) {
                if (nums[i] == nums[i - 1] + 1)
                    count++;
                else {
                    maxCount = Math.max(maxCount, count);
                    count = 1;
                }
            }
        }
        return Math.max(maxCount, count);
    }
}
