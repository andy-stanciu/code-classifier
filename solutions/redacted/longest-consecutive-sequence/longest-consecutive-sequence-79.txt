class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        Arrays.sort(nums);
        int max = 0;
        int valPrev = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1] + 1) {
                valPrev += 1;
            } else if (nums[i] != nums[i - 1]) {
                valPrev = 0;
            }
            max = Math.max(valPrev, max);
        }
        return max + 1;
    }
}
