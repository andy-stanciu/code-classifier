class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        Map<Integer, Integer> m = new HashMap<>();
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            int temp = m.getOrDefault(nums[i] - 1, 0) + 1;
            m.put(nums[i], temp);
            result = Math.max(result, temp);
        }
        return result;
    }
}
