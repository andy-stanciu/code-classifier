class Solution {

    public int method1(int[] nums) {
        int cnt = 1;
        int max = 0;
        int val = 0;
        if (nums.length == 0) {
            return 0;
        }
        Arrays.sort(nums);
        for (int i = 1; i < nums.length; i++) {
            val = nums[i] - nums[i - 1];
            if (val == 0) {
                continue;
            }
            if (val == 1) {
                cnt++;
            }
            if (val > 1) {
                if (cnt > max) {
                    max = cnt;
                }
                cnt = 1;
            }
        }
        if (cnt > max) {
            return cnt;
        }
        return max;
    }
}
