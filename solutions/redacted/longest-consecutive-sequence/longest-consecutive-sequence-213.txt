class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        // add all element in set
        for (int a : nums) set.add(a);
        int count = 0, ans = 0;
        // traverse on set
        for (Integer a : set) {
            // if a is starting of any sequence then count++ until the sequence break
            // if a is not stating element of any sequence, ignore it
            if (!set.contains(a - 1)) {
                count = 0;
                while (set.contains(a)) {
                    count++;
                    a++;
                }
                ans = Math.max(ans, count);
            }
        }
        return ans;
    }
}
