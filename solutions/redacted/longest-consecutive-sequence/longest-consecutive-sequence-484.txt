class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return 1;
        Arrays.sort(nums);
        int len = 1;
        int max = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] == 1) {
                len++;
            } else if (nums[i] - nums[i - 1] != 0) {
                len = 1;
            }
            if (len > max) {
                max = len;
            }
        }
        return max;
    }
}
