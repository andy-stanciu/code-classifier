class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int i : nums) set.add(i);
        int max = 0;
        for (int i : nums) {
            int x = i;
            if (!set.contains(x - 1)) {
                int cnt = 0;
                while (set.contains(x)) {
                    set.remove(x);
                    x += 1;
                    cnt++;
                }
                max = Math.max(max, cnt);
            }
        }
        return max;
    }
}
