class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        Arrays.sort(nums);
        int count = 0, max = 0, temp = 0;
        for (int i = 1; i < nums.length; i++) {
            while (i < nums.length && nums[i] - nums[i - 1] == 1) {
                count++;
                i++;
            }
            //temp = count;
            max = Math.max(max, count);
            if (i < nums.length && nums[i] != nums[i - 1])
                count = 0;
            //System.out.println(temp);
        }
        return max + 1;
        //return nums.length;
    }
}
