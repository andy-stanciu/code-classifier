class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 0)
            return 0;
        int var1 = 1;
        HashSet<Integer> st = new HashSet<>();
        for (int i = 0; i < n; i++) {
            st.add(nums[i]);
        }
        for (int it : st) {
            if (!st.contains(it - 1)) {
                int x = it;
                int cnt = 1;
                while (st.contains(x + 1)) {
                    cnt++;
                    x = x + 1;
                }
                var1 = Math.max(var1, cnt);
            }
        }
        return var1;
    }
}
