class Solution {

    public int method1(int[] nums) {
        Arrays.sort(nums);
        int var1 = 0;
        int currcount = 0;
        int n = nums.length;
        int lastsmall = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            if (nums[i] - 1 == lastsmall) {
                currcount += 1;
                lastsmall = nums[i];
            } else if (nums[i] != lastsmall) {
                currcount = 1;
                lastsmall = nums[i];
            }
            var1 = Math.max(var1, currcount);
        }
        return var1;
    }
}
