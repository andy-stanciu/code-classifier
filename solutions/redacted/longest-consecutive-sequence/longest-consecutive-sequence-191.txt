class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        HashMap<Integer, Boolean> present = new HashMap<>();
        HashMap<Integer, Boolean> checked = new HashMap<>();
        for (int i = 0; i < n; i++) {
            present.put(nums[i], true);
        }
        int longcount = 0;
        for (int i = 0; i < n; i++) {
            if (!checked.containsKey(nums[i]) && !present.containsKey(nums[i] - 1)) {
                int curr = 0;
                int start = nums[i];
                while (present.containsKey(start)) {
                    curr++;
                    checked.put(start, true);
                    start++;
                }
                longcount = Math.max(longcount, curr);
            }
        }
        return longcount;
    }
}
