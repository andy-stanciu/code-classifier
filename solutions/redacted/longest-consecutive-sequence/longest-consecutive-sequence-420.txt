class Solution {

    public int method1(int[] nums) {
        Set<Integer> m = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            m.add(nums[i]);
        }
        int l, lo;
        lo = 0;
        for (int i = 0; i < nums.length; i++) {
            if (!m.contains(nums[i] - 1)) {
                l = 0;
                while (m.contains(nums[i] + l)) {
                    l += 1;
                    lo = Math.max(lo, l);
                }
            }
        }
        return lo;
    }
}
