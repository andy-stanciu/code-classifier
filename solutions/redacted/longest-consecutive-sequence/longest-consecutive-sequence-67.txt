class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int i = 0; i < nums.length; i++) set.add(nums[i]);
        int max = 0;
        // we are taking the first encountered element of the hashSet and checking for the lower and higher value
        // by difference of 1 ("+1 and -1"), and counting the length and storing in the max variable (removing the elements as well).
        for (int i = 0; i < nums.length; i++) {
            int length = 1;
            //checking smaller(left-side) elements
            int var = nums[i];
            while (set.contains(--var)) {
                set.remove(var);
                length++;
            }
            //checking larger(right-ride) elements
            var = nums[i];
            while (set.contains(++var)) {
                set.remove(var);
                length++;
            }
            max = Math.max(max, length);
        }
        return max;
    }
}
