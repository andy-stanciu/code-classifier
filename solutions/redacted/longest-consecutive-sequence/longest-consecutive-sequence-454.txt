class Solution {

    public int method1(int[] nums1) {
        TreeSet<Integer> h1 = new TreeSet<>();
        for (int num : nums1) {
            h1.add(num);
        }
        List<Integer> nums = new ArrayList<Integer>(h1);
        int maxcount = 1;
        if (nums.size() == 0) {
            return 0;
        }
        int count = 1;
        for (int i = 0; i < h1.size() - 1; i++) {
            if ((nums.get(i) + 1 == nums.get(i + 1)) || (nums.get(i) == nums.get(i + 1))) {
                count++;
                maxcount = Math.max(maxcount, count);
            } else
                count = 1;
        }
        return maxcount;
    }
}
