class Solution {

    public int method1(int[] nums) {
        // Handle empty array
        if (nums.length == 0)
            return 0;
        // Sort the array
        Arrays.sort(nums);
        // To track the longest streak
        int var1 = 1;
        // To track the current consecutive streak
        int currentStreak = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                // Ignore duplicates
                if (nums[i] == nums[i - 1] + 1) {
                    // Increment current streak if consecutive
                    currentStreak++;
                } else {
                    // Update the longest streak
                    var1 = Math.max(var1, currentStreak);
                    // Reset current streak
                    currentStreak = 1;
                }
            }
        }
        // Return the maximum streak length
        return Math.max(var1, currentStreak);
    }
}
