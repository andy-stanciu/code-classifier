class Solution {

    public int method1(int[] nums) {
        // for we make a hashtable for the element. the hashtable values will be its respective length consequetive sequence
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (!map.containsKey(nums[i])) {
                map.put(nums[i], 1);
            }
        }
        int var1 = 0;
        // after that we again iterate through the array
        for (int i = 0; i < nums.length; i++) {
            if (!map.containsKey(nums[i] - 1)) {
                int currentValue = nums[i];
                int count = 1;
                while (true) {
                    if (!map.containsKey(currentValue + 1))
                        break;
                    else {
                        currentValue += 1;
                        count += 1;
                    }
                }
                var1 = Math.max(var1, count);
            }
        }
        return var1;
    }
}
