class Solution {

    public int method1(int[] nums) {
        int size = nums.length;
        if (size <= 1)
            return size;
        int maxSeq = 1;
        Set<Integer> set = new HashSet();
        for (int num : nums) {
            set.add(num);
        }
        for (int num : nums) {
            if (!set.contains(num - 1)) {
                int curSeq = 1;
                set.remove(num++);
                while (set.contains(num)) {
                    curSeq += 1;
                    set.remove(num++);
                }
                if (curSeq > maxSeq)
                    maxSeq = curSeq;
            }
        }
        return maxSeq;
    }
}
