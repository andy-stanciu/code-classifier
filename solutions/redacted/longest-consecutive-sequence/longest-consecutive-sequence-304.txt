class Solution {

    public int method1(int[] nums) {
        int max = 0;
        int sum = 1;
        int lE = Integer.MIN_VALUE;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == lE + 1) {
                sum++;
                lE++;
            } else if (nums[i] == lE) {
                continue;
            } else {
                sum = 1;
                lE = nums[i];
            }
            max = Math.max(sum, max);
        }
        return max;
    }
}
