class Solution {

    public int method1(int[] n) {
        if (n.length <= 1)
            return n.length;
        Arrays.sort(n);
        int max = 0;
        int c = 0;
        for (int i = 0; i < n.length - 1; i++) {
            if (n[i + 1] - n[i] == 1) {
                c++;
            } else if (n[i + 1] - n[i] == 0) {
            } else {
                c = 0;
            }
            max = Math.max(max, c);
        }
        return max + 1;
    }
}
