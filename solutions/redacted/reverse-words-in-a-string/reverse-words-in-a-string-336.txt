class Solution {

    public String method1(String s) {
        //String is converted into array
        String[] data = s.split(" +");
        //for answer
        StringBuilder ans = new StringBuilder();
        //looped in reverse order
        for (int a = data.length - 1; a >= 0; a--) {
            //added to answer string
            ans.append(data[a]);
            //space also added to string
            ans.append(" ");
        }
        //stringBuilder obj converted to string class and
        // trailing and leading spaces chopped
        return ans.toString().trim();
    }
}
