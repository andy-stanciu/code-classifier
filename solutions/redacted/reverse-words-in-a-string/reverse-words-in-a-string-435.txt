class Solution {

    public String method1(String s) {
        // Step 1: Trim leading and trailing spaces
        s = s.trim();
        // Step 2: Split the string by spaces
        String[] var1 = s.split("\\\\s+");
        // Step 3: Reverse the order of words
        int left = 0, right = var1.length - 1;
        while (left < right) {
            // Swap words[left] and words[right]
            String temp = var1[left];
            var1[left] = var1[right];
            var1[right] = temp;
            left++;
            right--;
        }
        // Step 4: Join the words back into a single string
        return String.join(" ", var1);
    }
}
