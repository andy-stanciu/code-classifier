class Solution {

    public String method1(String s) {
        String ss = s.strip();
        String[] s1 = ss.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = s1.length - 1; i >= 0; i--) {
            if (s1[i].strip() != "") {
                sb.append(s1[i]);
                if (i != 0) {
                    sb.append(" ");
                }
            }
        }
        return sb.toString();
    }
}
