class Solution {

    public String method1(String s) {
        String trimmed = s.trim().replaceAll("\\\\s{2,}", " ");
        String[] var1 = trimmed.split(" ");
        StringBuilder var2 = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            var2.append(var1[i]);
            if (i > 0) {
                var2.append(" ");
            }
        }
        return var2.toString();
    }
}
