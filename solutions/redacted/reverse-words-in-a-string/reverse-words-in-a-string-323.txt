class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" ");
        StringBuffer var2 = new StringBuffer();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].isEmpty()) {
                var2.append(var1[i]).append(" ");
            }
        }
        return var2.toString().strip();
    }
}
