class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().replaceAll("\\\\s+", " ").split(" ");
        String result = "";
        for (int i = var1.length - 1; i >= 0; i--) {
            result = result + " " + var1[i];
        }
        return result.trim();
    }
}
