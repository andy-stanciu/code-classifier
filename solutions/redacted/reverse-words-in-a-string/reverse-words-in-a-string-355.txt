class Solution {

    public String method1(String s) {
        String trimmed = s.trim();
        String[] var1 = trimmed.split("\\\\s+");
        StringBuilder st = new StringBuilder();
        int l = var1.length;
        for (int i = l - 1; i > 0; i--) {
            st.append(var1[i]);
            st.append(" ");
        }
        st.append(var1[0]);
        return st.toString();
    }
}
