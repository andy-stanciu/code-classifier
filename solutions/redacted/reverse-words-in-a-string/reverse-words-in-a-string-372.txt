class Solution {

    public String method1(String s) {
        //
        s = s.trim();
        String[] str = s.split("\\\\s+");
        StringBuilder sc = new StringBuilder();
        for (int i = str.length - 1; i >= 0; i--) {
            sc.append(str[i]);
            if (i > 0) {
                sc.append(" ");
            }
        }
        System.out.println(sc);
        return sc.toString();
    }
}
