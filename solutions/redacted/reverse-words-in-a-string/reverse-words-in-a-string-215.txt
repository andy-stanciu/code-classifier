class Solution {

    public String method1(String s) {
        String[] newStr = s.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = newStr.length - 1; i > 0; i--) {
            sb.append(newStr[i]).append(" ");
        }
        return sb.toString() + newStr[0];
    }
}
