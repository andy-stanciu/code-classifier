class Solution {

    public String method1(String s) {
        if (s == null || s.length() == 0) {
            return "";
        }
        s = s.trim();
        String[] var1 = s.split(" ");
        String answer = "";
        for (int i = var1.length - 1; i >= 0; i--) {
            String currentWord = var1[i].trim();
            if (currentWord.length() > 0) {
                answer += currentWord + " ";
            }
        }
        return answer.trim();
    }
}
