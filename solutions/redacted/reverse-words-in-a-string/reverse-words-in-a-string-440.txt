class Solution {

    public String method1(String s) {
        s = s.stripLeading();
        String[] s1 = s.split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = s1.length - 1; i >= 0; i--) {
            sb.append(s1[i] + " ");
        }
        sb.deleteCharAt(sb.length() - 1);
        return sb.toString();
    }
}
