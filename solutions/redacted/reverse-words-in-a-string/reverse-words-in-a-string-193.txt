class Solution {

    public String method1(String s) {
        String[] str = s.split("\\\\s+");
        String reserseArray = "";
        int j = 0;
        for (int i = str.length - 1; i >= 0; i--) {
            reserseArray += (str[i] + " ");
        }
        return reserseArray.trim();
    }
}
