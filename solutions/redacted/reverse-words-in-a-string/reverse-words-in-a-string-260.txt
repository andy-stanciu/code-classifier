class Solution {

    public String method1(String s) {
        var arr = s.split(" ");
        int len = arr.length - 1;
        StringBuilder sb = new StringBuilder();
        for (int i = len; i >= 0; i--) {
            if (!arr[i].equals("")) {
                sb.append(arr[i]);
                sb.append(" ");
            }
        }
        return sb.toString().trim();
    }
}
