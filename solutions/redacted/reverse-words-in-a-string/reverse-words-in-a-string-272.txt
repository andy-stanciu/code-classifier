class Solution {

    public String method1(String s) {
        String var1 = method2(s);
        // Print the resulting string
        // System.out.println("Reversed String:");
        // System.out.println(reversedString);
        return var1;
    }

    // Method to process the string: trim, split, reverse, and join
    public static String method2(String s) {
        // Trim leading and trailing spaces
        String var2 = s.trim();
        // Split the trimmed string into words
        String[] var3 = var2.split("\\\\s+");
        // Convert the array to an ArrayList
        ArrayList<String> var4 = new ArrayList<>(Arrays.asList(var3));
        // Convert the reversed ArrayList to a single string
        String var1 = method3(var4);
        return var1;
    }

    // Method to convert reversed ArrayList to a single string
    public static String method3(ArrayList<String> list) {
        StringBuilder sb = new StringBuilder();
        for (int i = list.size() - 1; i >= 0; i--) {
            sb.append(list.get(i));
            if (i > 0) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
