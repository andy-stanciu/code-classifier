class Solution {

    public String method1(String str) {
        String[] s = str.trim().split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = s.length - 1; i >= 0; i--) {
            sb.append(s[i]);
            if (i > 0)
                sb.append(" ");
        }
        return sb.toString();
    }
}
