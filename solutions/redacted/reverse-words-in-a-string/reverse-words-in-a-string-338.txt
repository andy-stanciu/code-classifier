class Solution {

    public String method1(String s) {
        Stack<String> st = new Stack<String>();
        StringBuilder sb = new StringBuilder();
        for (String a : s.trim().split(" ")) {
            if (!a.isEmpty())
                st.push(a);
        }
        while (!st.isEmpty()) {
            sb.append(st.pop());
            if (!st.isEmpty())
                sb.append(" ");
        }
        return sb.toString();
    }
}
