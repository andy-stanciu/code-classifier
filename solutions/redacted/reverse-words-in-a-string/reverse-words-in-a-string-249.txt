class Solution {

    public String method1(String s) {
        //trim the string
        s = s.trim();
        //Split the string
        String[] var1 = s.split("\\\\s+");
        //reversing a string using stringbuilder
        StringBuilder Rev = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            Rev.append(var1[i]);
            if (i != 0)
                Rev.append(" ");
        }
        return Rev.toString();
    }
}
