class Solution {

    public static String method1(String s) {
        // Split the string into words
        String[] var1 = s.trim().split("\\\\s+");
        // Reverse the words array
        int left = 0;
        int right = var1.length - 1;
        while (left < right) {
            String temp = var1[left];
            var1[left] = var1[right];
            var1[right] = temp;
            left++;
            right--;
        }
        // Join the reversed words into a single string
        return String.join(" ", var1);
    }
}
