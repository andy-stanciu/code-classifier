class Solution {

    public String method1(String s) {
        String[] k = s.trim().split("\\\\s+");
        StringBuilder m = new StringBuilder();
        for (int i = k.length - 1; i >= 0; i--) {
            m.append(k[i]);
            if (i != 0) {
                m.append(" ");
            }
        }
        return m.toString();
    }
}
