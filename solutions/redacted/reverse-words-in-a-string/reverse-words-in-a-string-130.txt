class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        // using StringBuilder which is more efficient for repeated concatenations.
        StringBuilder output = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            output.append(var1[i]);
            if (i != 0) {
                output.append(" ");
            }
        }
        return output.toString();
    }
}
