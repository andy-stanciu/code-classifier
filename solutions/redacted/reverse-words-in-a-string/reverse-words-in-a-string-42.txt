class Solution {

    public String method1(String s) {
        List<String> var1 = Arrays.asList(s.trim().split(" "));
        Collections.reverse(var1);
        StringBuilder sb = new StringBuilder();
        for (String w : var1) if (w.length() > 0)
            sb.append(w).append(" ");
        sb.setLength(sb.length() - 1);
        return sb.toString();
    }
}
