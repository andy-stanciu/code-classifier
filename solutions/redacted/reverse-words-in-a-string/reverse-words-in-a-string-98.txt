class Solution {

    public String method1(String t) {
        String s = t.trim().replaceAll("\\\\s+", " ");
        String[] var1 = s.split(" ");
        StringBuilder var2 = new StringBuilder();
        int i = var1.length - 1;
        while (i >= 0) {
            var2.append(var1[i]);
            if (i != 0) {
                var2.append(" ");
            }
            i--;
        }
        return var2.toString();
    }
}
