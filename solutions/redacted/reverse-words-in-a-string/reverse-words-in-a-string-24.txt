class Solution {

    public String method1(String s) {
        String[] str = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = str.length - 1; i >= 0; i--) {
            if (str[i] != "") {
                // add word
                sb.append(str[i]);
                // add space after that word
                sb.append(" ");
            }
        }
        //delete last space by reducing size
        sb.setLength(sb.length() - 1);
        return sb.toString();
    }
}
