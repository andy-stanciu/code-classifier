class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" ");
        int left = 0;
        int right = var1.length - 1;
        while (left < right) {
            String temp = var1[left];
            var1[left] = var1[right];
            var1[right] = temp;
            left++;
            right--;
        }
        StringBuilder result = new StringBuilder();
        for (String word : var1) {
            if (!word.isEmpty()) {
                if (result.length() > 0) {
                    result.append(" ");
                }
                result.append(word);
            }
        }
        return result.toString();
    }
}
