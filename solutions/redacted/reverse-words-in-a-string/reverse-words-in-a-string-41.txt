class Solution {

    public String method1(String s) {
        //  Using StringTokenizer;
        StringTokenizer st = new StringTokenizer(s, " ");
        String var1 = " ";
        int i = 0;
        while (st.hasMoreTokens()) {
            if (i == 0) {
                var1 = st.nextToken();
            } else {
                var1 = st.nextToken() + " " + var1;
            }
            i++;
        }
        return var1;
    }
}
