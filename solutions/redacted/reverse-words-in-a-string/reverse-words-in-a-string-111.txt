class Solution {

    public String method1(String s) {
        StringBuilder sb = new StringBuilder();
        s = s.trim().replaceAll("\\\\s+", " ");
        String[] strs = s.split(" ");
        for (int i = strs.length - 1; i >= 0; i--) {
            sb.append(strs[i]);
            if (i != 0) {
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
