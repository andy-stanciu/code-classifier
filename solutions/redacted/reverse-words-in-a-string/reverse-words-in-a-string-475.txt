class Solution {

    public String method1(String s) {
        var var1 = s.split(" ");
        var sb = new StringBuilder();
        for (var i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].isEmpty()) {
                sb.append(var1[i] + " ");
            }
        }
        sb.deleteCharAt(sb.length() - 1);
        return sb.toString();
    }
}
