class Solution {

    public String method1(String s) {
        String s2 = s.trim();
        String s1 = s2.replaceAll("\\\\s+", " ");
        String[] arr = s1.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = arr.length - 1; i >= 0; i--) {
            sb.append(arr[i]);
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
