class Solution {

    public String method1(String s) {
        // Split the string by spaces, this will handle multiple spaces
        String[] var1 = s.trim().split("\\\\s+");
        // Create a StringBuilder to build the result
        StringBuilder var2 = new StringBuilder();
        // Iterate over the words array in reverse order
        for (int i = var1.length - 1; i >= 0; i--) {
            var2.append(var1[i]);
            if (i != 0) {
                var2.append(" ");
            }
        }
        return var2.toString();
    }
}
