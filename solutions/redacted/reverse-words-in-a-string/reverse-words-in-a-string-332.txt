class Solution {

    public String method1(String s) {
        // Split the string into words, handling one or more spaces between them
        String[] str = s.split("\\\\s+");
        // Initialize a StringBuilder to construct the reversed sentence
        StringBuilder sb = new StringBuilder();
        // Loop through the array of words in reverse order
        for (int i = str.length - 1; i >= 0; i--) {
            // Trim any leading or trailing spaces from each word
            String word = str[i].trim();
            // Append the word and a space to the StringBuilder
            sb.append(word).append(" ");
        }
        // Convert StringBuilder to a string and trim any trailing space
        return sb.toString().trim();
    }
}
