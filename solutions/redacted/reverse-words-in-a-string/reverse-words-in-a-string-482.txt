class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" ");
        String rev = "";
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].equals("")) {
                rev += var1[i];
                if (i != 0 && !var1[i - 1].equals("")) {
                    rev += " ";
                }
            }
        }
        return rev;
    }
}
