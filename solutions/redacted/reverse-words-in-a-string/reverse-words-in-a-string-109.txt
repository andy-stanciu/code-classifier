class Solution {

    public String method1(String s) {
        //convert string in to array of strings
        // use space as a delimeter
        // print the word of this array in reverse order, while printing eleminate all the sapces
        {
            StringBuilder sb = new StringBuilder();
            String[] var1 = s.split(" ");
            for (int i = var1.length - 1; i >= 0; i--) {
                if (var1[i] != "") {
                    sb.append(var1[i].trim());
                    if (i != 0)
                        sb.append(" ");
                }
            }
            return sb.toString().trim();
        }
    }
}
