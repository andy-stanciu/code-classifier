class Solution {

    public String method1(String s) {
        String[] arr = s.split(" ");
        String ans = "";
        for (int i = arr.length - 1; i >= 0; i--) {
            if (!arr[i].isBlank()) {
                ans = ans + arr[i];
                ans += " ";
            }
        }
        return ans.substring(0, ans.length() - 1);
    }
}
