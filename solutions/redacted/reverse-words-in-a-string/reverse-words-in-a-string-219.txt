class Solution {

    public String method1(String s) {
        // Split the string on one or more spaces, trimming leading/trailing spaces
        String[] arr = s.trim().split("[ ]+");
        // Reverse the array using built-in collections functionality
        List<String> list = Arrays.asList(arr);
        Collections.reverse(list);
        // Join the reversed list back into a string with spaces and return the result
        return String.join(" ", list);
    }
}
