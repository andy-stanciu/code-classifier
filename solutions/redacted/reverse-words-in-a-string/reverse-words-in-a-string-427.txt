class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (var1[i].length() > 0) {
                sb.append(var1[i]);
                sb.append(" ");
            }
        }
        return sb.substring(0, sb.length() - 1).toString();
    }
}
