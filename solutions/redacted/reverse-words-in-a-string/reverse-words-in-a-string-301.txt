class Solution {

    public String method1(String s) {
        //splits the sentence based on spaces and all
        String[] splited = s.split("\\\\s+");
        //as we do not know anything about overall words
        StringBuilder var1 = new StringBuilder();
        for (int i = splited.length - 1; i >= 0; i--) {
            // we add space so that the words are separated
            var1.append(splited[i]).append(" ");
        }
        // reconverting back to string from stringbuilder
        return var1.toString().trim();
    }
}
