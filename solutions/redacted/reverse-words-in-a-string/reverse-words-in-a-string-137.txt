class Solution {

    public String method1(String s) {
        String str = s.trim();
        StringBuilder var1 = new StringBuilder();
        StringBuilder builder = new StringBuilder(str);
        String x = builder.reverse().toString();
        String[] s1 = x.split(" ");
        for (int i = 0; i < s1.length; i++) {
            StringBuilder builder1 = new StringBuilder(s1[i]);
            if (!builder1.toString().isEmpty()) {
                var1.append(" ");
                var1.append(builder1.reverse());
            }
        }
        return var1.toString().trim();
    }
}
