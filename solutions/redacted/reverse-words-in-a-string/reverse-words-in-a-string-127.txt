class Solution {

    public String method1(String s) {
        String[] var1 = s.split("\\\\s+");
        String result = "";
        for (int i = var1.length - 1; i >= 0; i--) {
            if (i == var1.length - 1) {
                result = result + var1[i];
            } else {
                if (var1[i] != "") {
                    result = result + " " + var1[i];
                }
            }
        }
        return result;
    }
}
