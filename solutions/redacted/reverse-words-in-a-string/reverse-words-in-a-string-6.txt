class Solution {

    public String method1(String s) {
        String[] str = s.split("\\\\s+");
        int n = str.length;
        String[] arr = new String[n];
        int var1 = 0;
        for (int i = n - 1; i >= 0; i--) {
            arr[var1++] = str[i];
        }
        String result = String.join(" ", arr).trim();
        return result;
    }
}
