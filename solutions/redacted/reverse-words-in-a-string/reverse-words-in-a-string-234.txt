class Solution {

    public String method1(String s) {
        s.trim();
        String[] var1 = s.split("\\\\s+");
        StringBuilder sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            sb.append(var1[i]);
            if (i > 0) {
                sb.append(" ");
            }
        }
        return sb.toString().trim();
    }
}
