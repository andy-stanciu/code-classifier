class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder res = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            res.append(var1[i]);
            if (i != 0) {
                res.append(" ");
            }
        }
        return res.toString();
    }
}
