class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" ");
        StringBuilder result = new StringBuilder();
        for (int i = var1.length - 1; i > 0; i--) {
            if (var1[i] != "") {
                result.append(var1[i]);
                result.append(" ");
            }
        }
        result.append(var1[0]);
        return result.toString().trim();
    }
}
