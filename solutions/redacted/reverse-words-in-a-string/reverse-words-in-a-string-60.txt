class Solution {

    public String method1(String s) {
        if (s.length() == 1)
            return s;
        String[] var1 = s.trim().split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].trim().isEmpty()) {
                sb.append(var1[i]);
                if (i > 0)
                    sb.append(" ");
            }
        }
        return sb.toString();
    }
}
