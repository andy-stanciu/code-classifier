class Solution {

    public String method1(String s) {
        StringBuffer sb = new StringBuffer();
        String[] st = s.trim().split("\\\\s+");
        for (int i = st.length - 1; i >= 0; i--) {
            String var1 = st[i];
            sb = sb.append(var1).append(" ");
        }
        return sb.toString().trim();
    }
}
