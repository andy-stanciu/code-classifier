class Solution {

    public String method1(String s) {
        StringBuilder builder = new StringBuilder();
        s.strip();
        String[] arr = s.split(" +");
        int n = arr.length - 1;
        for (int i = 0; i < arr.length; i++) {
            builder.append(arr[n - i] + " ");
        }
        return builder.toString().strip();
    }
}
