class Solution {

    public String method1(String s) {
        // Trim and split the string into words, then reverse and join
        String[] var1 = s.trim().split("\\\\s+");
        StringBuilder result = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            result.append(var1[i]);
            if (i != 0)
                result.append(" ");
        }
        return result.toString();
    }
}
