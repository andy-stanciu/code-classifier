class Solution {

    public String method1(String word) {
        List<String> var1 = new ArrayList<>();
        String[] strArray = word.split("\\\\s+");
        for (int i = strArray.length - 1; i >= 0; i--) {
            var1.add(strArray[i]);
        }
        String var2 = var1.stream().collect(Collectors.joining(" "));
        return var2.trim();
    }
}
