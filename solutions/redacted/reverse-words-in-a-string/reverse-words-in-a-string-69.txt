class Solution {

    public String method1(String s) {
        //removes leading and trailling spaces
        s = s.trim();
        //spliting the string into array of strings based on one or mores spaces
        String[] splited = s.split("\\\\s+");
        int len = splited.length;
        s = "";
        //traversing the array from last and creating a new string
        for (int i = len - 1; i > 0; i--) {
            s += splited[i];
            s += " ";
        }
        s += splited[0];
        return s;
    }
}
