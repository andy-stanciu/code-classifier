class Solution {

    public String method1(String s) {
        String[] parts = s.trim().split(" ");
        StringBuilder b = new StringBuilder();
        String part;
        for (int i = parts.length - 1; i >= 0; i--) {
            part = parts[i];
            if (part != null && part != "" && part != " ") {
                b.append(part);
                b.append(" ");
            }
        }
        return b.toString().trim();
    }
}
