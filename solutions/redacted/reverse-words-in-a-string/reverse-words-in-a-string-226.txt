class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] a = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = a.length - 1; i >= 0; i--) {
            if (!a[i].isEmpty())
                sb.append(a[i] + " ");
        }
        return sb.toString().trim();
    }
}
