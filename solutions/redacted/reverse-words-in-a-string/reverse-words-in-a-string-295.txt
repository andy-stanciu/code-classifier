class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" +");
        String var2 = new String();
        StringBuilder var3 = new StringBuilder();
        for (int i = var1.length - 1; i >= 0; i--) {
            var3.append(var1[i]);
            var3.append(" ");
        }
        return var3.toString().trim();
    }
}
