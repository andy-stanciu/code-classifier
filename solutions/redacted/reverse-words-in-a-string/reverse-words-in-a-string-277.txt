class Solution {

    public String method1(String s) {
        //using two pointers
        String[] var1 = s.trim().split("\\\\s+");
        int start = 0;
        int end = var1.length - 1;
        while (start < end) {
            String var2 = var1[start];
            var1[start] = var1[end];
            var1[end] = var2;
            start++;
            end--;
        }
        return String.join(" ", var1);
    }
}
