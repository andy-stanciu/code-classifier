class Solution {

    public String method1(String s) {
        //array
        String[] st = s.trim().split("\\\\s+");
        //string
        String output = "";
        for (int i = st.length - 1; i > 0; i--) {
            //reverse iteration
            output += st[i] + " ";
        }
        return output + st[0];
    }
}
