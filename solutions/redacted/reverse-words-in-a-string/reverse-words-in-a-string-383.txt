class Solution {

    public String method1(String s) {
        s = s.trim();
        String[] arr = s.split(" ");
        StringBuffer res = new StringBuffer();
        for (int n = arr.length - 1; n >= 0; n--) {
            if (arr[n].length() > 0) {
                res.append(arr[n] + " ");
            }
        }
        return res.toString().trim();
    }
}
