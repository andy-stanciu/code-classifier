class Solution {

    public String method1(String s) {
        String[] arr = s.split(" ");
        StringBuilder n = new StringBuilder();
        for (int i = arr.length - 1; i >= 0; i--) {
            if (arr[i].length() == 0) {
                continue;
            }
            if (n.length() == 0) {
                n.append(arr[i]);
            } else {
                n.append(" ");
                n.append(arr[i]);
            }
        }
        return n.toString();
    }
}
