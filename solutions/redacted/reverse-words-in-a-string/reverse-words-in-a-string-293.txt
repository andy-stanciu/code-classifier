class Solution {

    public String method1(String s) {
        StringBuilder str1 = new StringBuilder();
        String[] var1 = s.trim().split("\\\\s+");
        for (int i = var1.length - 1; i >= 0; i--) {
            str1.append(var1[i]);
            if (i != 0) {
                str1.append(" ");
            }
        }
        return str1.toString();
    }
}
