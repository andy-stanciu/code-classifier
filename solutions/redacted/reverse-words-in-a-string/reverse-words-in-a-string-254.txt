class Solution {

    public String method1(String s) {
        String[] var1 = s.replaceAll("\\\\s+", " ").trim().split(" ");
        int start = 0;
        int end = var1.length - 1;
        while (start < end) {
            String temp = var1[start];
            var1[start] = var1[end];
            var1[end] = temp;
            start++;
            end--;
        }
        String answer = String.join(" ", var1);
        return answer;
    }
}
