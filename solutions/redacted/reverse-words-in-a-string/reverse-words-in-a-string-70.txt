class Solution {

    public String method1(String s) {
        String rev = "";
        String[] arr = s.trim().split("\\\\s* \\\\s*");
        for (int i = arr.length - 1; i >= 0; i--) {
            if (i != 0)
                rev = arr[i] + " ";
            else
                rev += arr[i];
        }
        return rev;
    }
}
