class Solution {

    public String method1(String s) {
        String[] arr = s.split(" ");
        String a = "";
        List<String> l = new ArrayList<>();
        for (int i = arr.length - 1; i >= 0; i--) {
            if (!arr[i].equals("")) {
                l.add(arr[i]);
                a += arr[i] + " ";
            }
        }
        a = a.trim();
        return a;
    }
}
