class Solution {

    public String method1(String s) {
        // Split by spaces after trimming the string
        String[] var1 = s.trim().split("\\\\s+");
        // Create an array to store reversed words
        String[] rs = new String[var1.length];
        // Index for reversed array
        int j = 0;
        // Reverse the words
        for (int i = var1.length - 1; i >= 0; i--) {
            rs[j] = var1[i];
            j++;
        }
        // Join reversed words with a single space
        String ns = String.join(" ", rs);
        return ns;
    }
}
