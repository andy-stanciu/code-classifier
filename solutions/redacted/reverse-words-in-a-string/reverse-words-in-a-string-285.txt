class Solution {

    public String method1(String s) {
        String sTrimmed = s.replaceAll("\\\\s+", " ").trim();
        String[] var1 = sTrimmed.split(" ");
        String[] result = new String[var1.length];
        int j = 0;
        for (int i = var1.length - 1; i >= 0; i--) {
            result[j] = var1[i];
            j++;
        }
        System.out.println(result);
        String var2 = String.join(" ", result);
        return var2;
    }
}
