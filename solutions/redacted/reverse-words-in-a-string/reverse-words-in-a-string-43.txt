class Solution {

    public String method1(String s) {
        String[] strArray = s.split(" ");
        String ans = "";
        for (int i = strArray.length - 1; i >= 0; i--) {
            if (strArray[i].trim() == "") {
                continue;
            }
            ans = ans + " " + strArray[i];
        }
        return ans.substring(1).trim();
    }
}
