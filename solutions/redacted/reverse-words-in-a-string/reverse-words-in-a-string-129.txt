class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        // using StringBuilder which is more efficient for repeated concatenations.
        StringBuilder output = new StringBuilder();
        int left = 0;
        int right = var1.length - 1;
        while (left < right) {
            String temp = var1[left];
            var1[left++] = var1[right];
            var1[right--] = temp;
        }
        for (String word : var1) {
            if (output.length() > 0) {
                output.append(" ");
            }
            output.append(word);
        }
        return output.toString();
    }
}
