Solution {

    public String method1(String s) {
        String[] sarray = s.trim().split(" ");
        int left = 0;
        int right = sarray.length - 1;
        while (left < right) {
            String temp = sarray[left];
            sarray[left] = sarray[right];
            sarray[right] = temp;
            left++;
            right--;
        }
        return Arrays.stream(sarray).filter(k -> !k.isEmpty()).collect(Collectors.joining(" "));
    }
}
