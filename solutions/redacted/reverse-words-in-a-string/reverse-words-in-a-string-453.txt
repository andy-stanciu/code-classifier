class Solution {

    public String method1(String s) {
        s = s.trim().replaceAll("( )+", " ");
        System.out.println(s);
        String[] newstr = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for (int i = newstr.length - 1; i >= 0; i--) {
            if (i == 0) {
                sb.append(newstr[i]);
            } else {
                sb.append(newstr[i] + " ");
            }
        }
        return sb.toString();
    }
}
