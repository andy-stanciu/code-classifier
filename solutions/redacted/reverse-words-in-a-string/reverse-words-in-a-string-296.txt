class Solution {

    public String method1(String s) {
        StringBuilder var1 = new StringBuilder();
        String temp = s.trim();
        String[] arr = temp.split(" ");
        for (int i = arr.length - 1; i >= 0; i--) {
            String str = arr[i];
            if (str.isBlank()) {
                continue;
            }
            var1.append(str);
            if (i != 0) {
                var1.append(" ");
            }
        }
        return var1.toString().trim();
    }
}
