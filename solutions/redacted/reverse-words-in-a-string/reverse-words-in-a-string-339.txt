class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\s+");
        StringBuilder result = new StringBuilder();
        result.append(var1[var1.length - 1]);
        for (int i = var1.length - 2; i >= 0; i--) {
            result.append(" " + var1[i]);
        }
        return result.toString();
    }
}
