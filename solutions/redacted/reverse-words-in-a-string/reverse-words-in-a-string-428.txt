class Solution {

    public String method1(String s) {
        StringBuilder result = new StringBuilder();
        String[] var1 = s.split(" ");
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].isEmpty()) {
                result.append(" ");
                result.append(var1[i]);
            }
        }
        result.deleteCharAt(0);
        return result.toString();
    }
}
