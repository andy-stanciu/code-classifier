class Solution {

    public String method1(String s) {
        var var1 = s.split(" ");
        var result = new ArrayList<String>();
        for (var i = var1.length - 1; i >= 0; i--) {
            if (var1[i].isEmpty())
                continue;
            result.add(var1[i]);
        }
        return String.join(" ", result);
    }
}
