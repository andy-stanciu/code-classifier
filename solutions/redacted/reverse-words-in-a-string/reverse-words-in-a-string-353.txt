class Solution {

    public String method1(String s) {
        StringBuilder answer = new StringBuilder();
        String[] var1 = s.split(" ");
        for (int i = var1.length - 1; i >= 0; i--) {
            if (var1[i] != "")
                answer.append((var1[i]).trim()).append(" ");
        }
        answer.deleteCharAt(answer.length() - 1);
        return answer.toString();
    }
}
