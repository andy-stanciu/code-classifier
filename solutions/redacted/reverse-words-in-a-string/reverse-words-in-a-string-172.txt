class Solution {

    public String method1(String s) {
        String[] var1 = s.trim().split("\\\\s+");
        StringBuffer var2 = new StringBuffer();
        for (int i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].isEmpty()) {
                // Check for empty strings
                var2.append(var1[i]);
                if (i > 0) {
                    // Add space between words
                    var2.append(" ");
                }
            }
        }
        return var2.toString();
    }
}
