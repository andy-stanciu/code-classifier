class Solution {

    public String method1(String s) {
        String[] var1 = s.split(" ");
        StringBuilder res = new StringBuilder();
        int i;
        for (i = var1.length - 1; i >= 0; i--) {
            if (!var1[i].isEmpty()) {
                if (res.length() > 0) {
                    res.append(" ");
                }
                res.append(var1[i]);
            }
        }
        String result = res.toString();
        return result;
    }
}
