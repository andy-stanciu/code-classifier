class Solution {

    public String method1(String s) {
        String result = "";
        String[] var1 = s.split("\\\\W+");
        int left = 0;
        int right = var1.length - 1;
        while (left < right) {
            String j = var1[left];
            var1[left] = var1[right];
            var1[right] = j;
            left++;
            right--;
        }
        result = String.join(" ", var1).trim();
        return result;
    }
}
