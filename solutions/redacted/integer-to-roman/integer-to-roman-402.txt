class Solution {

    public String method1(int num) {
        int number = num;
        final int[] arab = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        final String[] var1 = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        final StringBuilder var2 = new StringBuilder();
        int var3 = 0;
        while (number > 0) {
            if (number >= arab[var3]) {
                number -= arab[var3];
                var2.append(var1[var3]);
            } else {
                var3++;
            }
        }
        return var2.toString();
    }
}
