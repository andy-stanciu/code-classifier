class Solution {

    public String method1(int num) {
        StringBuilder ans = new StringBuilder();
        String[] strarr = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] arr1 = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        for (int i = 0; num > 0; ++i) {
            while (num >= arr1[i]) {
                ans.append(strarr[i]);
                num -= arr1[i];
            }
        }
        return ans.toString();
    }
}
