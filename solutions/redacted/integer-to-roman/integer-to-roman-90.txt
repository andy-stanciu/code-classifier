class Solution {

    public String method1(int num) {
        int[] Num = { 1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000 };
        String[] var1 = { "I", "IV", "V", "IX", "X", "XL", "L", "XC", "C", "CD", "D", "CM", "M" };
        String var2 = "";
        int a = 12;
        while (num > 0) {
            while (num / Num[a] > 0) {
                var2 += var1[a];
                num -= Num[a];
            }
            a--;
        }
        return var2;
    }
}
