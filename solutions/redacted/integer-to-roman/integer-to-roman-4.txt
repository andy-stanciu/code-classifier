class Solution {

    public String method1(int num) {
        String var1 = "";
        int[][] var2 = { { 1000, "M" }, { 900, "CM" }, { 500, "D" }, { 400, "CD" }, { 100, "C" }, { 90, "XC" }, { 50, "L" }, { 40, "XL" }, { 10, "X" }, { 9, "IX" }, { 5, "V" }, { 4, "IV" }, { 1, "I" } };
        for (int i = 0; i < var2.length; i++) {
            while (num >= var2[i][0]) {
                var1 += var2[i][1];
                num -= var2[i][0];
            }
        }
        return var1;
    }
}
