class Solution {

    public String method1(int num) {
        int[] div = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] rom = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        StringBuilder res = new StringBuilder();
        int i = 0;
        while (num > 0) {
            while (div[i] > num) i++;
            num -= div[i];
            res.append(rom[i]);
        }
        return res.toString();
    }
}
