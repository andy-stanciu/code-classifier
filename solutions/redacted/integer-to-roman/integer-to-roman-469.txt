class Solution {

    public String method1(int num) {
        int[] v = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] r = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        StringBuilder s = new StringBuilder();
        int i = 0;
        while (num > 0) {
            if (num >= v[i]) {
                s.append(r[i]);
                num -= v[i];
            } else {
                i++;
            }
        }
        return s.toString();
    }
}
