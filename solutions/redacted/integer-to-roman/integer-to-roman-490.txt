class Solution {

    public String method1(int num) {
        String var1 = "";
        int[] intArray = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] var2 = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        for (int i = 0; i < intArray.length; i++) {
            while (num >= intArray[i]) {
                var1 = var1 + var2[i];
                num = num - intArray[i];
            }
        }
        return var1;
    }
}
