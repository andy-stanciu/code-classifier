class Solution {

    public String method1(int num) {
        StringBuilder sb = new StringBuilder();
        String[] var1 = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int[] nums = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        for (int i = 0; i < 13; i++) {
            while (num >= nums[i]) {
                num = num - nums[i];
                sb.append(var1[i]);
            }
        }
        return sb.toString();
    }
}
