class Solution {

    public String method1(int num) {
        // Defini\xE7\xE3o dos valores e s\xEDmbolos romanos em ordem decrescente
        int[] valores = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] simbolos = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        // StringBuilder para construir o numeral romano
        StringBuilder resultado = new StringBuilder();
        // Itera sobre os valores de maior para menor
        for (int i = 0; i < valores.length; i++) {
            // Enquanto o n\xFAmero for maior ou igual ao valor atual
            while (num >= valores[i]) {
                // Subtrai o valor do n\xFAmero
                num -= valores[i];
                // Adiciona o s\xEDmbolo correspondente ao resultado
                resultado.append(simbolos[i]);
            }
        }
        // Retorna o numeral romano como string
        return resultado.toString();
    }
}
