class Solution {

    public String method1(int num) {
        int[] nums = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        String[] str = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        // String ans = "";
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < nums.length; i++) {
            while (num >= nums[i]) {
                ans.append(str[i]);
                num -= nums[i];
            }
        }
        return ans.toString();
    }
}
