class Solution {

    // 1994-1000=994-900=94-90=4
    public String method1(int num) {
        int[] values = { 1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000 };
        String[] var1 = { "I", "IV", "V", "IX", "X", "XL", "L", "XC", "C", "CD", "D", "CM", "M" };
        StringBuilder sb = new StringBuilder();
        for (int i = values.length - 1; i >= 0 && num >= 0; i--) {
            while (num >= values[i]) {
                num -= values[i];
                sb.append(var1[i]);
            }
        }
        return sb.toString();
    }
}
