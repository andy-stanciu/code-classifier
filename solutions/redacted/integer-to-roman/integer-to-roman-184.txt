class Solution {

    public String method1(int num) {
        int[] val = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        StringBuilder s = new StringBuilder();
        String[] symbols = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        for (int i = 0; i < val.length; i++) {
            while (num >= val[i]) {
                s.append(symbols[i]);
                num -= val[i];
            }
        }
        return s.toString();
    }
}
