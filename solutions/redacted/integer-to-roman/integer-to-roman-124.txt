class Solution {

    public String method1(int num) {
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        StringBuilder ans = new StringBuilder();
        String[] roms = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" };
        int i = 0;
        while (num > 0) {
            if (num >= values[i]) {
                num -= values[i];
                ans.append(roms[i]);
            } else {
                i++;
            }
        }
        return ans.toString();
    }
}
