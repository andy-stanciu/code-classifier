class Solution {

    public String method1(int num) {
        String var1 = "";
        while (num >= 1000) {
            var1 += "M";
            num -= 1000;
        }
        while (num >= 900) {
            var1 += "CM";
            num -= 900;
        }
        while (num >= 500) {
            var1 += "D";
            num -= 500;
        }
        while (num >= 400) {
            var1 += "CD";
            num -= 400;
        }
        while (num >= 100) {
            var1 += "C";
            num -= 100;
        }
        while (num >= 90) {
            var1 += "XC";
            num -= 90;
        }
        while (num >= 50) {
            var1 += "L";
            num -= 50;
        }
        while (num >= 40) {
            var1 += "XL";
            num -= 40;
        }
        while (num >= 10) {
            var1 += "X";
            num -= 10;
        }
        while (num >= 9) {
            var1 += "IX";
            num -= 9;
        }
        while (num >= 5) {
            var1 += "V";
            num -= 5;
        }
        while (num >= 4) {
            var1 += "IV";
            num -= 4;
        }
        while (num >= 1) {
            var1 += "I";
            num -= 1;
        }
        return var1;
    }
}
