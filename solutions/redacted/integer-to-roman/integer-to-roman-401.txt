class Solution {

    public String method1(int num) {
        // Define a mapping of integer values to Roman numeral symbols
        Map<Integer, String> map = new HashMap<>();
        map.put(1, "I");
        map.put(4, "IV");
        map.put(5, "V");
        map.put(9, "IX");
        map.put(10, "X");
        map.put(40, "XL");
        map.put(50, "L");
        map.put(90, "XC");
        map.put(100, "C");
        map.put(400, "CD");
        map.put(500, "D");
        map.put(900, "CM");
        map.put(1000, "M");
        // Define an array of all possible values in descending order
        int[] values = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 };
        StringBuilder var1 = new StringBuilder();
        // Iterate through each possible value
        for (int value : values) {
            // Repeat while the input number is greater than or equal to the current value
            while (num >= value) {
                // Append the corresponding Roman numeral symbol to the result
                var1.append(map.get(value));
                // Subtract the current value from the input number
                num -= value;
            }
        }
        return var1.toString();
    }
}
