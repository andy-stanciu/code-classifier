class Solution {

    List<Integer> l = new ArrayList<Integer>();

    public void find(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        method1(root.right);
        l.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        find(root);
        return l;
    }
}
