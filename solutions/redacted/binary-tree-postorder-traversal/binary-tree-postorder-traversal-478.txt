class Solution {

    public List<Integer> method1(TreeNode root) {
        // create list (returntype)
        List<Integer> list = new ArrayList();
        // call postorder funtn
        method2(root, list);
        // return the list
        return list;
    }

    public void method2(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        // step 1: call left subtree
        method2(root.left, list);
        // step 2: call right subtree
        method2(root.right, list);
        // step 3: store val in list
        list.add(root.val);
    }
}
