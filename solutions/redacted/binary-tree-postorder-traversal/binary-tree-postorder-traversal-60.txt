/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> preorder = new ArrayList<>();
        method2(root, preorder);
        return preorder;
    }

    void method2(TreeNode curr, ArrayList<Integer> param1) {
        if (curr == null) {
            return;
        }
        method2(curr.left, param1);
        method2(curr.right, param1);
        param1.add(curr.val);
    }
}
