/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> p = new ArrayList();
        check(root, p);
        return p;
    }

    void check(TreeNode root, List<Integer> p) {
        if (root == null)
            return;
        check(root.left, p);
        check(root.right, p);
        p.add(root.val);
    }
}
