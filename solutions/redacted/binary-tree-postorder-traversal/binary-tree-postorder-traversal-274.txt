/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(Class1 root) {
        List<Integer> result = new ArrayList<>();
        if (root == null)
            return result;
        Stack<Class1> stack = new Stack<>();
        Stack<Class1> output = new Stack<>();
        stack.push(root);
        while (!stack.isEmpty()) {
            Class1 node = stack.pop();
            output.push(node);
            if (node.left != null)
                stack.push(node.left);
            if (node.right != null)
                stack.push(node.right);
        }
        while (!output.isEmpty()) {
            result.add(output.pop().val);
        }
        return result;
    }
}

class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int x) {
        val = x;
    }
}
