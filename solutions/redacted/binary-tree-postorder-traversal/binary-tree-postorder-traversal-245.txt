/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> ar = new ArrayList<Integer>();
        po(root, ar);
        return ar;
    }

    public static void po(TreeNode root, ArrayList<Integer> ar) {
        if (root == null)
            return;
        po(root.left, ar);
        po(root.right, ar);
        ar.add(root.val);
    }
}
