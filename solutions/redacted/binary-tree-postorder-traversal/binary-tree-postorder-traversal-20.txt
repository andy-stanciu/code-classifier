/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root, List<Integer> arr) {
        if (root != null) {
            int data = root.val;
            method1(root.left, arr);
            method1(root.right, arr);
            arr.add(data);
        }
        return arr;
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        return method1(root, arr);
    }
}
