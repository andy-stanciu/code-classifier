class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<Integer>();
        if (root == null)
            return list;
        List<Integer> left = method1(root.left);
        List<Integer> right = method1(root.right);
        list.addAll(left);
        list.addAll(right);
        list.add(root.val);
        return list;
    }
}
