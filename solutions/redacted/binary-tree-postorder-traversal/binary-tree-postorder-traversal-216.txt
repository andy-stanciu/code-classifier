class Solution {

    public List<Integer> l = new ArrayList<>();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left == null && root.right == null) {
            l.add(root.val);
            return;
        }
        if (root.left != null) {
            method1(root.left);
        }
        if (root.right != null) {
            method1(root.right);
        }
        l.add(root.val);
        return;
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return l;
    }
}
