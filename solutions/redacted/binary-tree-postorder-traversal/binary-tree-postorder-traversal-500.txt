class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> post = new ArrayList<>();
        method2(post, root);
        return post;
    }

    static void method2(List<Integer> post, TreeNode node) {
        if (node == null)
            return;
        TreeNode temp = node;
        method2(post, node.left);
        method2(post, node.right);
        post.add(temp.val);
    }
}
