class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        method2(root, res);
        return res;
    }

    private void method2(TreeNode node, List<Integer> res) {
        if (node != null) {
            method2(node.left, res);
            method2(node.right, res);
            res.add(node.val);
        }
    }
}
