/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        dfs(root, var1);
        return var1;
    }

    public void dfs(TreeNode node, List<Integer> var1) {
        if (node == null)
            return;
        dfs(node.left, var1);
        dfs(node.right, var1);
        var1.add(node.val);
    }
}
