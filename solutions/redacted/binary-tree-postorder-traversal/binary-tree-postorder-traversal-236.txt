class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l1 = new ArrayList();
        return method2(root, l1);
    }

    public List<Integer> method2(TreeNode root, List<Integer> arr) {
        if (root != null) {
            method2(root.left, arr);
            method2(root.right, arr);
            arr.add(root.val);
        }
        return arr;
    }
}
