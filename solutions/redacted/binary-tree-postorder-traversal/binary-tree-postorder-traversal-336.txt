/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> order = new ArrayList<>();
        method1(root, order);
        return order;
    }

    public void method1(TreeNode root, List<Integer> order) {
        if (root != null) {
            method1(root.left, order);
            method1(root.right, order);
            order.add(root.val);
        }
    }
}
