/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static void method1(TreeNode node, ArrayList<Integer> list1) {
        if (node == null) {
            return;
        }
        method1(node.left, list1);
        method1(node.right, list1);
        list1.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        method1(root, list);
        return list;
    }
}
