class Solution {

    public void method1(TreeNode node, List<Integer> post) {
        if (node == null)
            return;
        method1(node.left, post);
        method1(node.right, post);
        post.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> post = new ArrayList<>();
        method1(root, post);
        return post;
    }
}
