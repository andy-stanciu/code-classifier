class Solution {

    private void method1(TreeNode node, List<Integer> result) {
        if (node == null) {
            return;
        }
        method1(node.left, result);
        method1(node.right, result);
        result.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method1(root, result);
        return result;
    }
}
