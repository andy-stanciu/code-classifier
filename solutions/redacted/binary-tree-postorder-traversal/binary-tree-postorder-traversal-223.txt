/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int c = 0;

    List<Integer> list = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        //calling a another function to store the postorder in list array
        method2(root);
        return list;
    }

    public void method2(TreeNode node) {
        //return if reached the end of the tree
        if (node == null)
            return;
        // To  print the left side of the binary tree first
        method2(node.left);
        // To print the right side of the binary tree
        method2(node.right);
        list.add(node.val);
    }
}
