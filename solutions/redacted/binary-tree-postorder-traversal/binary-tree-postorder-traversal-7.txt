/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        TreeNode node = root;
        // Initalise the ArrayList to store node values;
        List<Integer> result = new ArrayList<>();
        findOrder(node, result);
        return result;
    }

    private void findOrder(TreeNode node, List<Integer> result) {
        if (node == null) {
            return;
        }
        // Recursive call for left
        findOrder(node.left, result);
        // Recursive call for right
        findOrder(node.right, result);
        // Add the values
        result.add(node.val);
    }
}
