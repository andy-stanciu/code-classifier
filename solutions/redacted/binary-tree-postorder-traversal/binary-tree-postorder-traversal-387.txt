/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    public void method2(TreeNode root, List<Integer> result) {
        //base case
        if (root == null)
            return;
        //postorder of LST
        method2(root.left, result);
        //postorder of RST
        method2(root.right, result);
        //add the root to the result
        result.add(root.val);
    }
}
