class Solution {

    List<Integer> res = new ArrayList<>();

    void method1(TreeNode root) {
        if (root != null) {
            method1(root.left);
            method1(root.right);
            res.add(root.val);
        }
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return res;
    }
}
