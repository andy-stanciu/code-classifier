class Solution {

    static List<Integer> result;

    public List<Integer> method1(TreeNode root) {
        result = new ArrayList<>();
        method2(root);
        return result;
    }

    private static void method2(TreeNode node) {
        if (node == null)
            return;
        if (node.left != null) {
            method2(node.left);
        }
        if (node.right != null) {
            method2(node.right);
        }
        result.add(node.val);
    }
}
