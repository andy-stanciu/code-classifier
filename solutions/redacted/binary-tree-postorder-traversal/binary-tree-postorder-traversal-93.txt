class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans;
    }

    public void method2(TreeNode r, List<Integer> ans) {
        if (r == null)
            return;
        method2(r.left, ans);
        method2(r.right, ans);
        ans.add(r.val);
    }
}
