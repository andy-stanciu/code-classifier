class Solution {

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    private void method2(TreeNode node, List<Integer> result) {
        if (node == null)
            return;
        method2(node.left, result);
        method2(node.right, result);
        result.add(node.val);
    }
}
