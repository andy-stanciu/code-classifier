/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<Integer>();
        method1(root, result);
        return result;
    }

    private void method1(TreeNode root, List<Integer> result) {
        if (root == null)
            return;
        method1(root.left, result);
        method1(root.right, result);
        result.add(root.val);
    }
}
