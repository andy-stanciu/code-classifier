/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> list = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            return list;
        } else {
            pp(root, list);
        }
        return list;
    }

    public static void pp(TreeNode root, List<Integer> list) {
        if (root == null) {
            return;
        }
        pp(root.left, list);
        pp(root.right, list);
        list.add(root.val);
    }
}
