/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        Stack<TreeNode> st = new Stack<>();
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode var1 = st.peek();
            if (var1.left != null) {
                //left tree exists
                TreeNode var3 = var1.left;
                var1.left = null;
                st.push(var3);
            } else if (var1.right != null) {
                //right tree exists
                TreeNode var2 = var1.right;
                var1.right = null;
                st.push(var2);
            } else {
                result.add(var1.val);
                st.pop();
            }
        }
        return result;
    }
}
