/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void method1(TreeNode node, List<Integer> list) {
        if (node == null)
            return;
        method1(node.left, list);
        method1(node.right, list);
        list.add(node.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> ans = new ArrayList();
        method1(root, ans);
        return ans;
    }
}
