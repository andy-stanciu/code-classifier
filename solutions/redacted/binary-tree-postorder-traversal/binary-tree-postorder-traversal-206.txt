class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int val) {
        this.val = val;
    }
}

public class Solution {

    public List<Integer> method1(Class1 root) {
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    private void method2(Class1 node, List<Integer> result) {
        if (node == null) {
            return;
        }
        // Traverse left subtree
        method2(node.left, result);
        // Traverse right subtree
        method2(node.right, result);
        // Visit the root
        result.add(node.val);
    }

    public static void main(String[] args) {
        // Example usage:
        Class1 root = new Class1(1);
        root.right = new Class1(2);
        root.right.left = new Class1(3);
        Solution solution = new Solution();
        // Output: [3, 2, 1]
        System.out.println(solution.method1(root));
    }
}
