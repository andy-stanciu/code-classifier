/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
/*
1.) InOrder   <-> LNR
2.) PreOrder  <-> NLR
3.) PostOrder <-> LRN
4.) LevelOrder <-> level wise level 


4, 5, 2, 6, 3, 1


*/
class Solution {

    private void util(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        util(root.left, res);
        util(root.right, res);
        res.add(root.val);
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        util(root, res);
        return res;
    }
}
