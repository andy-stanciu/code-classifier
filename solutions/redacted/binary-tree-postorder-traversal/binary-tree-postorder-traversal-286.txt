/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void post(List<Integer> output, TreeNode root) {
        if (root == null) {
            return;
        }
        post(output, root.left);
        post(output, root.right);
        output.add(root.val);
        return;
    }

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new ArrayList<>();
        post(output, root);
        return output;
    }
}
