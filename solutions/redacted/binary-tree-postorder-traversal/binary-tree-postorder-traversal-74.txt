class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> output = new ArrayList<>();
        method2(root, output);
        return output;
    }

    public void method2(TreeNode root, List<Integer> output) {
        if (root == null) {
            return;
        }
        method2(root.left, output);
        method2(root.right, output);
        output.add(root.val);
    }
}
