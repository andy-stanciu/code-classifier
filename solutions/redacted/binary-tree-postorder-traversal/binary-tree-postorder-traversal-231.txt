/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> listt = new ArrayList<>();
        tra(root, listt);
        return listt;
    }

    public void tra(TreeNode root, ArrayList<Integer> list) {
        if (root == null) {
            return;
        }
        tra(root.left, list);
        tra(root.right, list);
        list.add(root.val);
    }
}
