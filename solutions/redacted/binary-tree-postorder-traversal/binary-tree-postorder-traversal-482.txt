/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList();
        if (root == null) {
            return list;
        }
        Stack<TreeNode> st = new Stack();
        st.push(root);
        while (!st.isEmpty()) {
            TreeNode currnt = st.pop();
            list.add(0, currnt.val);
            if (currnt.left != null) {
                st.push(currnt.left);
            }
            if (currnt.right != null) {
                st.push(currnt.right);
            }
        }
        return list;
    }
}
