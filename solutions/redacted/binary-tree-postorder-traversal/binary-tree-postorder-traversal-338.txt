class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        method2(root, result);
        return result;
    }

    // Helper method for recursion
    private void method2(TreeNode root, List<Integer> result) {
        if (root == null) {
            return;
        }
        // Traverse the left subtree
        method2(root.left, result);
        // Traverse the right subtree
        method2(root.right, result);
        // Visit the root node
        result.add(root.val);
    }
}
