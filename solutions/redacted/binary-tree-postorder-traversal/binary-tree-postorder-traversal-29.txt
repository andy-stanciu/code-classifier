/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        method2(root, ans);
        return ans;
    }

    void method2(TreeNode node, List<Integer> list) {
        if (node == null) {
            return;
        }
        method2(node.left, list);
        method2(node.right, list);
        list.add(node.val);
    }
}
