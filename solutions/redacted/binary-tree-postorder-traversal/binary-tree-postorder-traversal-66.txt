class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l1 = new ArrayList<Integer>();
        traverse(l1, root);
        return l1;
    }

    public void traverse(List<Integer> l1, TreeNode root) {
        if (root == null)
            return;
        traverse(l1, root.left);
        traverse(l1, root.right);
        l1.add(root.val);
    }
}
