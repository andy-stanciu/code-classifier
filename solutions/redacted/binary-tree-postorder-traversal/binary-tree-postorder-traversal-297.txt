/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void method1(TreeNode root, List<Integer> post) {
        if (root == null)
            return;
        method1(root.left, post);
        method1(root.right, post);
        post.add(root.val);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> post = new ArrayList<>();
        method1(root, post);
        return post;
    }
}
