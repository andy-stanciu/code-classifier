class Solution {

    public int method1(int n) {
        int one = 1;
        int two = 1;
        //only have to compute up to n-1 because at n, one will be equal to n
        for (int i = 0; i < n - 1; i++) {
            int temp = one;
            one = one + two;
            two = temp;
        }
        return one;
    }
}
