class Solution {

    public int method1(int n) {
        int[] memo = new int[n + 1];
        Arrays.fill(memo, -1);
        memo[0] = 1;
        memo[1] = 1;
        return method2(n, memo);
    }

    public int method2(int n, int[] memo) {
        if (memo[n] != -1) {
            return memo[n];
        }
        memo[n] = method2(n - 1, memo) + method2(n - 2, memo);
        return memo[n];
    }
}
