class Solution {

    public int method1(int n) {
        if (n == 1) {
            return 1;
        } else if (n == 2) {
            return 2;
        }
        // Initialize the base cases
        // ways to climb 1 step
        int prev2 = 1;
        // ways to climb 2 steps
        int prev1 = 2;
        // Start from step 3 up to n
        for (int i = 3; i <= n; i++) {
            // Current step is the sum of the previous two
            int current = prev1 + prev2;
            // Shift the values for next iteration
            prev2 = prev1;
            prev1 = current;
        }
        return prev1;
    }
}
