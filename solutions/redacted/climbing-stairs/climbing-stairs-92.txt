class Solution {

    public int method1(int n) {
        if (n <= 0) {
            return 0;
        }
        if (n == 1) {
            return 1;
        }
        if (n == 2) {
            return 2;
        }
        //how many step possibilities there are with 1 stairs
        int first = 1;
        //how many step possibilities there are with 2 stairs
        int second = 2;
        int third = 0;
        for (int i = 3; i <= n; i++) {
            third = first + second;
            //initially start with what we have at 2 steps
            //and continually add the previous two step
            //possibilities as the loop runs
            first = second;
            second = third;
        }
        return third;
    }
}
