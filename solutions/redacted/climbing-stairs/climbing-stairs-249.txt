class Solution {

    // Create new array based on size given
    int[] var1 = new int[46];

    public int method1(int n) {
        // Base Condition
        if (n < 0)
            return 0;
        if (n == 0)
            return 1;
        if (var1[n] != 0)
            return var1[n];
        // Simple Concept (FIBONACCI)
        return var1[n] = method1(n - 1) + method1(n - 2);
    }
}
