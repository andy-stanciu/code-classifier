class Solution {

    public int method1(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        // Number of ways to climb 1 stair
        int prev1 = 1;
        // Number of ways to climb 2 stairs
        int prev2 = 1;
        // Number of ways to climb the current stair
        int current = 0;
        for (int i = 2; i <= n; i++) {
            // Calculate the number of ways to climb the current stair
            current = prev1 + prev2;
            // Update prev2 to store the number of ways to climb i-2 stairs for the next iteration
            prev2 = prev1;
            // Update prev1 to store the number of ways to climb i-1 stairs for the next iteration
            prev1 = current;
        }
        // Return the number of ways to climb n stairs
        return current;
    }
}
