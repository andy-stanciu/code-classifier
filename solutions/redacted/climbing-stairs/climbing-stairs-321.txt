class Solution {

    public int method1(int n) {
        if (n <= 2)
            return n;
        // Ways to reach the first step
        int first = 1;
        // Ways to reach the second step
        int second = 2;
        int current = 0;
        for (int i = 3; i <= n; i++) {
            current = first + second;
            first = second;
            second = current;
        }
        return current;
    }
}
