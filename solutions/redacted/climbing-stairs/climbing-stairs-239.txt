class Solution {

    public int method1(int d) {
        double f = 0;
        int dou = d;
        double n = d;
        if (n / 2 == 0) {
            return (1);
        } else {
            if (n % 2 == 0) {
                for (double i = n; i >= n / 2; i--) {
                    f = f + (factorial(i) / ((factorial(n - i)) * factorial(i - (n - i))));
                }
            } else {
                for (double i = n; i >= (n + 1) / 2; i--) {
                    f = f + (((factorial(i)) / ((factorial(n - i)) * (factorial(i - (n - i))))));
                }
            }
        }
        int value = (int) f;
        return (value);
    }

    static double factorial(double c) {
        if (c == 0) {
            return (1);
        }
        double fact = 1;
        for (double i = 1; i <= c; i++) {
            fact = fact * i;
        }
        return fact;
    }
}
