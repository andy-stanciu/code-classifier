class Solution {

    public int method1(int n) {
        // base case
        // since there is one way to go to ground. i.e . to be there itself
        int prev1 = 1;
        // since we cant go underground
        int prev2 = 0;
        for (int i = 1; i <= n; i++) {
            int curr = prev1 + prev2;
            prev2 = prev1;
            prev1 = curr;
        }
        return prev1;
    }
}
