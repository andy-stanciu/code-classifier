class Solution {

    public int method1(int n) {
        if (n == 1) {
            return 1;
        }
        int[] opts = new int[n + 1];
        opts[1] = 1;
        opts[2] = 2;
        for (int i = 3; i <= n; i++) {
            opts[i] = opts[i - 1] + opts[i - 2];
        }
        return opts[n];
    }
}
