class Solution {

    public int method1(int n) {
        if (n <= 3)
            return n;
        //Represents the number of ways to climb to step n-1
        int prev1 = 1;
        //Represents the number of ways to climb to step n-2
        int prev2 = 2;
        for (int i = 3; i <= n; i++) {
            // Current step is the sum of the previous two steps
            int current = prev1 + prev2;
            //Move prev1 up to the next step
            prev1 = prev2;
            //Move prev2 up to the next step
            prev2 = current;
        }
        return prev2;
    }
}
