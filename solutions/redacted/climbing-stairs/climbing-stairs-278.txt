class Solution {

    public int method1(int n) {
        if (n == 0 || n == 1) {
            return n;
        }
        int prev = 1, prev2 = 1;
        for (int i = 2; i <= n; i++) {
            int curr = prev + prev2;
            prev2 = prev;
            prev = curr;
        }
        return prev;
    }
}
