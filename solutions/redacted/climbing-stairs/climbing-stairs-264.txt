class Solution {

    HashMap<Integer, Integer> map = new HashMap<>();

    public int method1(int n) {
        if (n == 1 || n == 2) {
            return n;
        }
        map.put(-1, 0);
        map.put(1, 1);
        map.put(2, 2);
        stairPath(n);
        return map.get(n);
    }

    int stairPath(int n) {
        if (map.containsKey(n) && map.get(n) > 0) {
            return map.get(n);
        }
        int l = stairPath(n - 1);
        int r = stairPath(n - 2);
        map.put(n, l + r);
        return l + r;
    }
}
