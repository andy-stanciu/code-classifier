class Solution {

    int[] mem = new int[1000];

    public int method1(int n) {
        mem[0] = 1;
        mem[1] = 1;
        mem[2] = 2;
        //set mem
        if (n == 1) {
            return 1;
        }
        if (n == 2) {
            return 2;
        }
        for (int i = 3; i <= n; i++) {
            mem[i] = mem[i - 1] + mem[i - 2];
        }
        return mem[n];
    }
}
