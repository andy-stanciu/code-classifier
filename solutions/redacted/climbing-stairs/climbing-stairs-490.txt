class Solution {

    public int method1(int n) {
        if (n == 1)
            // Base case
            return n;
        // initialize new array of size n
        int[] res = new int[n];
        //value for case when n = 1, we have only 1 way
        res[0] = 1;
        // when n = 2, we have two ways of going
        res[1] = 2;
        // for the rest of the cases we know that the number of ways will be (n - 1) + (n - 2)
        for (int i = 2; i < n; i++) {
            res[i] = res[i - 1] + res[i - 2];
        }
        return res[n - 1];
    }
}
