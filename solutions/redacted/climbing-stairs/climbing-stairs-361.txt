class Solution {

    private int[] unique;

    public int method1(int n) {
        this.unique = new int[n + 1];
        // set base case
        unique[0] = 1;
        unique[1] = unique[0];
        // bot up dp
        for (int i = 2; i <= n; i++) {
            unique[i] = unique[i - 1] + unique[i - 2];
        }
        return unique[n];
    }
}
