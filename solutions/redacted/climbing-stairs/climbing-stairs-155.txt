class Solution {

    public int method1(int n) {
        int[] climbStore = new int[n + 1];
        // Base Conditions of Recursion
        climbStore[0] = 1;
        // Base Conditions of Recursion
        climbStore[1] = 1;
        for (int i = 2; i <= n; i++) {
            // Recurring Condition of Recursion
            climbStore[i] = climbStore[i - 1] + climbStore[i - 2];
        }
        // Final Value
        return climbStore[n];
    }
}
