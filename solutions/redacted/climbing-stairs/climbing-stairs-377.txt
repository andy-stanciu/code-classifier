class Solution {

    public int method1(int n) {
        int[] f = new int[n + 1];
        return method1(n, f);
    }

    public static int method1(int n, int[] f) {
        if (n < 0) {
            return 0;
        }
        if (n == 0) {
            return 1;
        }
        if (f[n] != 0) {
            return f[n];
        }
        f[n] = method1(n - 1, f) + method1(n - 2, f);
        return f[n];
    }
}
