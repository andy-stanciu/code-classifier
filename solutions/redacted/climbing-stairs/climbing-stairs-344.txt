class Solution {

    public int method1(int n) {
        int[] arr = new int[n + 1];
        return climbStair(n, 0, arr);
    }

    public static int climbStair(int dest, int current, int[] arr) {
        if (current == dest + 1) {
            return 0;
        }
        if (arr[current] != 0) {
            return arr[current];
        } else if (current == dest) {
            return 1;
        }
        arr[current] = climbStair(dest, current + 1, arr) + climbStair(dest, current + 2, arr);
        return arr[current];
    }
}
