class Solution {

    public int method1(int n) {
        // Base cases
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        // Create an array to store the number of ways to reach each step
        int[] a = new int[n];
        // There\'s only one way to reach the first step (take 1 step)
        a[0] = 1;
        // There are two ways to reach the second step (take 1 step twice or take 2 steps)
        a[1] = 2;
        // Fill the array for steps starting from the third step
        for (int i = 2; i < n; i++) {
            a[i] = a[i - 1] + a[i - 2];
        }
        // Return the number of ways to reach the top (last) step
        return a[n - 1];
    }
}
