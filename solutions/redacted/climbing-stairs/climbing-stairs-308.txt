class Solution {

    public int method1(int n) {
        int[] steps = new int[n + 1];
        if (n == 1) {
            return 1;
        } else if (n == 2) {
            return 2;
        } else if (n == 3) {
            return 3;
        }
        steps[0] = 0;
        steps[1] = 1;
        steps[2] = 2;
        steps[3] = 3;
        for (int i = 4; i <= n; i++) {
            steps[i] = steps[i - 1] + steps[i - 2];
        }
        return steps[n];
    }
}
