class Solution {

    Hashtable<Integer, Integer> steps = new Hashtable<>();

    public int method1(int n) {
        if (steps.containsKey(n)) {
            return steps.get(n);
        }
        if (n == 1 || n == 0) {
            return 1;
        }
        steps.put(n, method1(n - 1) + method1(n - 2));
        return steps.get(n);
    }
}
