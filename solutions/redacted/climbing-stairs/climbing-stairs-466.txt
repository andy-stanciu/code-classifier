class Solution {

    public int method1(int n) {
        int[] memo = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            memo[i] = -1;
        }
        return fib(n, memo);
    }

    public int fib(int n, int[] memo) {
        if (n == 1) {
            return 1;
        }
        if (n == 2)
            return 2;
        if (memo[n] != -1)
            return memo[n];
        memo[n] = fib(n - 1, memo) + fib(n - 2, memo);
        return memo[n];
    }
}
