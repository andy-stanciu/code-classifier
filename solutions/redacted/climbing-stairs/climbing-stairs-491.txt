class Solution {

    public int method1(int n) {
        Map<Integer, Integer> m = new HashMap<>();
        return method1(n, m);
    }

    private int method1(int n, Map<Integer, Integer> m) {
        if (n == 0 || n == 1)
            return 1;
        if (!m.containsKey(n))
            m.put(n, method1(n - 1, m) + method1(n - 2, m));
        return m.get(n);
    }
}
