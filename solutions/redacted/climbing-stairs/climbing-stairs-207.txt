class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1);
        return slove(n, dp);
    }

    public int slove(int n, int[] dp) {
        //base case
        if (n <= 2) {
            return n;
        }
        if (dp[n] != -1) {
            return dp[n];
        }
        return dp[n] = slove(n - 1, dp) + slove(n - 2, dp);
    }
}
