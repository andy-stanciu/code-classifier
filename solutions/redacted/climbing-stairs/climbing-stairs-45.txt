class Solution {

    int[] dp;

    public int method1(int n) {
        dp = new int[n + 1];
        dp[0] = 1;
        for (int i = 0; i < n + 1; i++) {
            if (i >= 2) {
                dp[i] += dp[i - 2];
            }
            if (i >= 1)
                dp[i] += dp[i - 1];
        }
        return dp[n];
    }
}
