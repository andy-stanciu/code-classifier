class Solution {

    public int method1(int n) {
        if (n == 1) {
            return 1;
        }
        if (n == 2) {
            return 2;
        }
        int[] fib = { 1, 2 };
        int bucket = 0;
        int[] waysToClimb = new int[n - 2];
        for (int i = 0; i < waysToClimb.length; i++) {
            waysToClimb[i] = fib[0] + fib[1];
            fib[0] = fib[1];
            fib[1] = waysToClimb[i];
        }
        return waysToClimb[n - 3];
    }
}
