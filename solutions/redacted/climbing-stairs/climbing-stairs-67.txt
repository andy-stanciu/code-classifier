class Solution {

    public int method1(int n) {
        int[] dp = new int[n + 1];
        Arrays.fill(dp, -1);
        return f(dp, n);
    }

    public int f(int[] dp, int n) {
        if (n == 0 || n == 1)
            return 1;
        if (dp[n] != -1)
            return dp[n];
        return dp[n] = f(dp, n - 1) + f(dp, n - 2);
    }
}
