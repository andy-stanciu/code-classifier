class Solution {

    public int method1(int n) {
        int total = 0;
        if (n == 1 || n == 2 || n == 3) {
            //default values
            return n;
        } else {
            //using factorial logic
            int x = 2;
            int y = 3;
            int z = 0;
            for (int i = 3; i < n; i++) {
                z = x + y;
                x = y;
                y = z;
            }
            return z;
        }
    }
}
