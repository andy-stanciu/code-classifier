class Solution {

    public int method1(int n) {
        // If there is only one step, there is only one way to climb it
        if (n == 1) {
            return 1;
        }
        // Initialize the number of ways to reach the first and second steps
        // One way to reach the first step
        int first = 1;
        // Two ways to reach the second step: (1+1) or (2)
        int second = 2;
        // Iterate from the third step to the nth step
        for (int i = 3; i <= n; ++i) {
            // The number of ways to reach the current step is the sum of the ways
            // to reach the previous two steps
            int third = first + second;
            // Update the values for the next iteration
            first = second;
            second = third;
        }
        // Return the number of ways to reach the nth step
        return second;
    }
}
