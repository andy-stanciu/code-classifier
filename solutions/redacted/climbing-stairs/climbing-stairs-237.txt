class Solution {

    public int method1(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        int n1 = 0;
        int n2 = 1;
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            ans = n1 + n2;
            n1 = n2;
            n2 = ans;
        }
        return ans;
    }
}
