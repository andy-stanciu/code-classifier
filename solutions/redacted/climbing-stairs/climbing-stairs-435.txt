class Solution {

    public int method1(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        int temp0 = 1;
        int temp1 = 1;
        for (int i = 2; i < n; i++) {
            int temp2 = temp1 + temp0;
            temp0 = temp1;
            temp1 = temp2;
        }
        return temp0 + temp1;
    }
}
