class Solution {

    //Function to find the number of ways to climb the staircase
    public int method1(int n) {
        //array to store the number of ways to climb to each step
        //n+1 - As we need to store for the nth step too
        int[] ways = new int[n + 1];
        //base case: there is only one way to climb 0 steps (i.e, do nothing)
        ways[0] = 1;
        //base case: there is only one way to climb 1 step (i.e, take a single step)
        ways[1] = 1;
        //compute the number of ways to climb the n steps
        for (int i = 2; i <= n; i++) {
            ways[i] = ways[i - 1] + ways[i - 2];
            //ways[i-1]: climb by taking a single step
            //ways[i-2]: climb by taking two steps
        }
        //return the number of ways to climb the n steps
        return ways[n];
    }
}
