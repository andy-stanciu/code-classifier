class Solution {

    public int method1(int n) {
        int num1 = 1;
        int num2 = 2;
        int nextnum = 0;
        if (n == 1)
            return num1;
        if (n == 2)
            return num2;
        for (int i = 3; i <= n; i++) {
            nextnum = num1 + num2;
            num1 = num2;
            num2 = nextnum;
        }
        return nextnum;
    }
}
