class Solution {

    //would do n+1 but n isnt global and didn\'t want to use helper function to pass n
    int[] memo = new int[1000];

    public int method1(int n) {
        if (n < 0) {
            return 0;
        }
        if (n == 0) {
            return 1;
        }
        if (memo[n] != 0) {
            return memo[n];
        }
        memo[n] = method1(n - 1) + method1(n - 2);
        return memo[n];
    }
}
