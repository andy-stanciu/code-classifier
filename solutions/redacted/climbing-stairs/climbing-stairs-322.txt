class Solution {

    static HashMap<Integer, Integer> memo = new HashMap<>();

    public int method1(int n) {
        if (n <= 2)
            return n;
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        int res = method1(n - 1) + method1(n - 2);
        memo.put(n, res);
        return res;
    }
}
