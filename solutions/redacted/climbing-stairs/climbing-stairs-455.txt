class Solution {

    public int method1(int n) {
        if (n == 1) {
            return 1;
        }
        // Represents the ways to reach the first step
        int first = 1;
        // Represents the ways to reach the second step
        int second = 2;
        // For n >= 3, calculate the ways iteratively
        for (int i = 3; i <= n; i++) {
            // The ways to reach the current step
            int current = first + second;
            // Update first to the previous second
            first = second;
            // Update second to the current
            second = current;
        }
        return second;
    }
}
