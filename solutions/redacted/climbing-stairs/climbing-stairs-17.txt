class Solution {

    private Map<Integer, Integer> memo = new HashMap<>();

    public int method1(int n) {
        if (n <= 0)
            return 0;
        if (n == 1 || n == 2 || n == 3)
            return n;
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        int ways = method1(n - 1) + method1(n - 2);
        memo.put(n, ways);
        return ways;
    }
}
