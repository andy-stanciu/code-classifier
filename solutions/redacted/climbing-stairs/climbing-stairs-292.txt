class Solution {

    public int method1(int n) {
        if (n <= 1) {
            return 1;
        }
        int a = 1;
        int b = 1;
        int output = 0;
        for (int i = 2; i <= n; i++) {
            output = a + b;
            b = a;
            a = output;
        }
        return output;
    }
}
