class Solution {

    public int method1(int n) {
        if (n == 0 || n == 1)
            return 1;
        return dp(n - 1) + dp(n - 2);
    }

    public int dp(int n) {
        if (n == 1 || n == 0) {
            return 1;
        }
        int[] dp = new int[n + 1];
        dp[0] = dp[1] = 1;
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }
}
