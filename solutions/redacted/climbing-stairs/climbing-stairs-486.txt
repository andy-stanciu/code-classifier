class Solution {

    public int ways(int[] dp, int n) {
        if (n <= 3)
            return n;
        if (dp[n] != 0)
            return dp[n];
        return dp[n] = ways(dp, n - 1) + ways(dp, n - 2);
    }

    public int method1(int n) {
        int[] dp = new int[n + 1];
        return ways(dp, n);
    }
}
