class Solution {

    public int method1(int n) {
        return method2(n, new int[n + 1]);
    }

    private int method2(int n, int[] memoize) {
        memoize[0] = 1;
        memoize[1] = memoize[0];
        for (int i = 2; i < memoize.length; i++) {
            memoize[i] = memoize[i - 1] + memoize[i - 2];
        }
        return memoize[n];
    }
}
