class Solution {

    public int method1(int n) {
        if (n == 1)
            return 1;
        int[] dp = new int[n];
        //first step pattern
        dp[0] = 1;
        //second pattern
        dp[1] = 2;
        for (int i = 2; i < n; i++) {
            //first + second patterns
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        //return the last one
        return dp[n - 1];
    }
}
