class Solution {

    public int method1(int n) {
        //DP: MEMOIZATION + RECURSION (TOP-DOWN)
        int[] dp = new int[n + 1];
        return countSteps(n, dp);
    }

    static int countSteps(int n, int[] dp) {
        if (n == 0)
            return 1;
        if (n < 0)
            return 0;
        //found the answer then return ans --> removing repetition
        if (dp[n] > 0)
            return dp[n];
        int step1 = countSteps(n - 1, dp);
        int step2 = countSteps(n - 2, dp);
        int ans = step1 + step2;
        dp[n] = ans;
        return ans;
    }
}
