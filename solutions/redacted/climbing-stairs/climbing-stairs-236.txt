class Solution {

    int count = 0;

    public void solve(int n) {
        if (n < 0)
            return;
        if (n == 0) {
            count += 1;
            return;
        }
        solve(n - 1);
        solve(n - 2);
    }

    public int method1(int n) {
        solve(n);
        return count;
    }
}
