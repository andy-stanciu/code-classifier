class Solution {

    public int method1(int n) {
        if (n <= 1) {
            return n;
        }
        //current corresponds to the value in current index, the one that we are supposed to calculate at the moment
        int current = 0;
        //prev1 is initialised with the value in index - 0
        int prev1 = 1;
        //prev2 is initialised with the value in index - 1
        int prev2 = 1;
        //prev1 and prev2 already account for first two numbers in the sequence. So, i can start from 2
        for (int i = 2; i <= n; i++) {
            current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        return current;
    }
}
