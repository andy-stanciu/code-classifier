class Solution {

    public int method1(int n) {
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        // T(n) = T(n-1) + T(n-2)
        int last = 2, secondLast = 1, next = 0;
        for (int i = 3; i <= n; i++) {
            next = last + secondLast;
            secondLast = last;
            last = next;
        }
        return next;
    }
}
