class Solution {

    public int method1(int n) {
        int prev2 = 1, prev = 2, curr = 0;
        if (n == 1)
            return prev2;
        if (n == 2)
            return prev;
        for (int i = 3; i <= n; i++) {
            curr = prev + prev2;
            prev2 = prev;
            prev = curr;
        }
        return curr;
    }
}
