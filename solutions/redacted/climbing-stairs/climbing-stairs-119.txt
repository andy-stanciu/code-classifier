class Solution {

    public int method1(int n) {
        int[] memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return countWays(n, memo);
    }

    private int countWays(int n, int[] memo) {
        if (n <= 2)
            return n;
        if (memo[n] != -1)
            return memo[n];
        return memo[n] = countWays(n - 1, memo) + countWays(n - 2, memo);
    }
}
