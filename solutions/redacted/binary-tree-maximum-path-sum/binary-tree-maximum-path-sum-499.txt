/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxValue = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        method2(root);
        return maxValue;
    }

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(0, method2(root.left));
        int var2 = Math.max(0, method2(root.right));
        maxValue = Math.max(maxValue, var1 + var2 + root.val);
        return Math.max(var1, var2) + root.val;
    }
}
