/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] maxi = { Integer.MIN_VALUE };
        method2(root, maxi);
        return maxi[0];
    }

    int method2(TreeNode root, int[] maxi) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(0, method2(root.left, maxi));
        int var2 = Math.max(0, method2(root.right, maxi));
        // Update the maximum path sum with the larger of the current max and the new path sum including the current node
        maxi[0] = Math.max(maxi[0], var1 + var2 + root.val);
        // Return the maximum path sum including the current node and either its left or right child
        return Math.max(var1, var2) + root.val;
    }
}
