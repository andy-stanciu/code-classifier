class Solution {

    private static int ans = Integer.MIN_VALUE;

    private static int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = method1(root.left);
        int right = method1(root.right);
        int ms = Math.max(Math.max(left, right) + root.val, root.val);
        int ms_1 = Math.max((left + right + root.val), ms);
        ans = Math.max(ans, ms_1);
        return ms;
    }

    public int method2(TreeNode root) {
        ans = Integer.MIN_VALUE;
        method1(root);
        return ans;
    }
}
