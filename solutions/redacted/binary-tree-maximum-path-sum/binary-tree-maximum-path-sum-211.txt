/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int method1(TreeNode root, int[] param1) {
        if (root == null)
            return 0;
        int ls = Math.max(0, method1(root.left, param1));
        int rs = Math.max(0, method1(root.right, param1));
        param1[0] = Math.max(param1[0], ls + rs + root.val);
        return Math.max(ls, rs) + root.val;
    }

    public int method2(TreeNode root) {
        int[] param1 = { Integer.MIN_VALUE };
        method1(root, param1);
        return param1[0];
    }
}
