/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root, 0);
        return var1;
    }

    private int method2(TreeNode root, int param1) {
        if (root == null)
            return 0;
        int left = Math.max(method2(root.left, root.val), 0);
        int right = Math.max(method2(root.right, root.val), 0);
        var1 = Math.max(var1, root.val + left + right);
        return root.val + Math.max(left, right);
    }
}
