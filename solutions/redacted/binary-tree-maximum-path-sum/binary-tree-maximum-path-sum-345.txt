/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int var1(TreeNode root) {
        method1(root);
        return var1;
    }

    private int method1(TreeNode root) {
        if (root == null)
            return 0;
        int var2 = method1(root.left);
        int var3 = method1(root.right);
        int result = root.val + Math.max(var2, var3);
        result = Math.max(result, root.val);
        var1 = Math.max(var1, root.val + var2 + var3);
        var1 = Math.max(var1, result);
        return result;
    }
}
