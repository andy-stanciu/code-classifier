/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        this.var1 = Integer.MIN_VALUE;
        maxContrib(root);
        return this.var1;
    }

    private int maxContrib(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftMax = maxContrib(root.left);
        int rightMax = maxContrib(root.right);
        int var2 = 0;
        int var3 = 0;
        if (leftMax > 0)
            var2 = leftMax;
        if (rightMax > 0)
            var3 = rightMax;
        int var4 = root.val + var2 + var3;
        this.var1 = Math.max(var1, var4);
        return root.val + Math.max(var2, var3);
    }
}
