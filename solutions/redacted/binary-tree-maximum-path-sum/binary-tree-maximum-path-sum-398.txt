/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        AtomicInteger var1 = new AtomicInteger(Integer.MIN_VALUE);
        helper(root, var1);
        return var1.get();
    }

    private static int helper(TreeNode root, AtomicInteger var1) {
        if (root == null) {
            return 0;
        }
        int left = Math.max(helper(root.left, var1), 0);
        int right = Math.max(helper(root.right, var1), 0);
        int newMax = Math.max(var1.get(), left + right + root.val);
        var1.set(newMax);
        return Math.max(left + root.val, right + root.val);
    }
}
