class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return var1;
    }

    public int helper(TreeNode root) {
        if (root == null)
            return 0;
        int leftVal = helper(root.left);
        int rightVal = helper(root.right);
        int totalVal = root.val + leftVal + rightVal;
        //pathMax value with will be used in the recursive process
        //its either (roor+left) or (root+right) or (root)
        int var2 = Math.max(leftVal + root.val, rightVal + root.val);
        var2 = Math.max(var2, root.val);
        // find the value (left +root+right) it can be the max value path
        int max = Math.max(var2, totalVal);
        // Which ever is max ,update the global variable
        var1 = Math.max(var1, max);
        return var2;
    }
}
