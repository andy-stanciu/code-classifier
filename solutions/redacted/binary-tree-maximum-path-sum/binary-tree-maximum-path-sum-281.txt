class Solution {

    private int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        return Math.max(dfs(root), var1);
    }

    private int dfs(TreeNode node) {
        // Termination condition.
        if (node == null)
            return 0;
        if (node.left == null && node.right == null) {
            var1 = Math.max(var1, node.val);
            return node.val;
        }
        // Traverse condition.
        int var2 = dfs(node.left);
        int var3 = dfs(node.right);
        // Return value
        int var4 = Math.max(Math.max(node.val, var3 + node.val), var2 + node.val);
        var1 = Math.max(var1, Math.max(var4, var2 + node.val + var3));
        return var4;
    }
}
