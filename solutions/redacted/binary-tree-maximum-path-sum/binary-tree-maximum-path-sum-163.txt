/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        recursive(root);
        return ans;
    }

    private int recursive(TreeNode node) {
        if (node == null)
            return 0;
        int lv = recursive(node.left);
        int rv = recursive(node.right);
        int ret = node.val;
        ret = Math.max(ret, ret + Math.max(lv, rv));
        ans = Math.max(ans, Math.max(ret, node.val + lv + rv));
        return ret;
    }
}
