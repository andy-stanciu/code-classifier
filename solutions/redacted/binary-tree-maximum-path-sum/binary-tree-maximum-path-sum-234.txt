/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int method1(TreeNode root, int[] param1) {
        if (root == null)
            return 0;
        int left = Math.max(0, method1(root.left, param1));
        int right = Math.max(0, method1(root.right, param1));
        param1[0] = Math.max(param1[0], root.val + left + right);
        return root.val + Math.max(left, right);
    }

    public int method2(TreeNode root) {
        int[] param1 = new int[1];
        param1[0] = Integer.MIN_VALUE;
        method1(root, param1);
        return param1[0];
    }
}
