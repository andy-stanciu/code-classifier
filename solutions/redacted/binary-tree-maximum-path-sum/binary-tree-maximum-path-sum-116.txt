/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[1];
        // set 0-th index to minValue, to handle negative values
        var1[0] = Integer.MIN_VALUE;
        // findMaxi
        findMaxi(root, var1);
        // return maxiPathSum
        return var1[0];
    }

    private int findMaxi(TreeNode node, int[] var1) {
        if (node == null) {
            return Integer.MIN_VALUE;
        }
        // find leftMax & rightMax
        int leftMax = findMaxi(node.left, var1);
        int rightMax = findMaxi(node.right, var1);
        // update maxiPathSum
        var1[0] = Math.max(var1[0], Math.max(node.val, node.val + Math.max(0, leftMax) + Math.max(0, rightMax)));
        // return to prevNode
        return Math.max(node.val, Math.max(node.val + Math.max(0, leftMax), node.val + Math.max(0, rightMax)));
    }
}
