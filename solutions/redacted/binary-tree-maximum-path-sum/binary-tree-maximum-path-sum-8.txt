class Solution {

    int max = -9999;

    public int method1(TreeNode root) {
        if (root.left == null && root.right == null)
            return root.val;
        method2(root);
        return max;
    }

    int method2(TreeNode root) {
        if (root == null)
            return -9999;
        if (root.left == null && root.right == null)
            return root.val;
        int var1 = method2(root.left);
        int var2 = method2(root.right);
        int var3 = Math.max(var1, var2);
        int var4 = Math.max(var3 + root.val, root.val);
        int var5 = Math.max(var4, var3);
        max = Math.max(var5, max);
        max = Math.max(max, var1 + var2 + root.val);
        return var4;
    }
}
