/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        solve(root);
        return max;
    }

    public int solve(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = solve(root.left);
        int right = solve(root.right);
        //here i was comparing current poition,current+left max,current+right max,current+right max+left max
        max = Math.max(max, Math.max(Math.max(Math.max(left + right + root.val, root.val + left), root.val + right), root.val));
        // at the time of returning just return max ( Max(left,right)+current val , current val)
        return Math.max(Math.max(root.val + left, root.val + right), root.val);
    }
}
