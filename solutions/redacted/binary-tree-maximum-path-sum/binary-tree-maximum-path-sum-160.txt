/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int[] maxi) {
        if (root == null)
            return 0;
        int maxl = Math.max(0, method1(root.left, maxi));
        int maxr = Math.max(0, method1(root.right, maxi));
        maxi[0] = Math.max(maxl + maxr + root.val, maxi[0]);
        return root.val + Math.max(maxl, maxr);
    }

    public int method2(TreeNode root) {
        int[] maxi = new int[1];
        maxi[0] = Integer.MIN_VALUE;
        method1(root, maxi);
        return maxi[0];
    }
}
