/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int[] maxi) {
        if (root == null)
            return 0;
        int var1 = Math.max(0, method1(root.left, maxi));
        int var2 = Math.max(0, method1(root.right, maxi));
        maxi[0] = Math.max(maxi[0], var1 + var2 + root.val);
        return Math.max(var2, var1) + root.val;
    }

    public int method2(TreeNode root) {
        int[] maxi = { Integer.MIN_VALUE };
        int x = method1(root, maxi);
        return maxi[0];
    }
}
