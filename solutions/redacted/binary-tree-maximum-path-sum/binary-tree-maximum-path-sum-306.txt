/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] maxValue = { Integer.MIN_VALUE };
        maxdown(root, maxValue);
        return maxValue[0];
    }

    public static int maxdown(TreeNode root, int[] maxValue) {
        if (root == null)
            return 0;
        int left = Math.max(0, maxdown(root.left, maxValue));
        int right = Math.max(0, maxdown(root.right, maxValue));
        maxValue[0] = Math.max(maxValue[0], root.val + left + right);
        return Math.max(left, right) + root.val;
    }
}
