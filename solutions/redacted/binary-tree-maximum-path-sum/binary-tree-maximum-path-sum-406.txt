/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        solve(root);
        // TODO
        return max;
    }

    public int solve(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = solve(root.left);
        int var2 = solve(root.right);
        int leftAndRoot = var1 + root.val;
        int rightAndRoot = var2 + root.val;
        int var3 = Math.max(Math.max(leftAndRoot, rightAndRoot), root.val);
        max = Math.max(max, Math.max(var3, var1 + var2 + root.val));
        return var3;
    }
}
