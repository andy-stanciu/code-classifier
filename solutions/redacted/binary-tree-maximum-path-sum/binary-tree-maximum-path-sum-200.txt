/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return max;
    }

    public int method2(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int left = method2(node.left);
        int right = method2(node.right);
        int self = node.val;
        // returnVal is max sum if this node is part of path
        // returnVal = max(left + self, right + self, self)
        int returnVal = self;
        if (node.left != null) {
            returnVal = Math.max(returnVal, left + self);
        }
        if (node.right != null) {
            returnVal = Math.max(returnVal, right + self);
        }
        // check if returnVal is max
        max = Math.max(max, returnVal);
        // check if current sub tree is max
        max = Math.max(max, left + self + right);
        return returnVal;
    }
}
