class Solution {

    public int res = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return res;
    }

    public int helper(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = helper(root.left);
        int var2 = helper(root.right);
        res = Math.max(res, Math.max(Math.max(var1 + var2 + root.val, root.val), Math.max(root.val + var1, root.val + var2)));
        return Math.max(Math.max(var1, var2) + root.val, root.val);
    }
}
