/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static int method1(TreeNode root, int[] Maxval) {
        if (root == null)
            return 0;
        int left = Math.max(0, method1(root.left, Maxval));
        int right = Math.max(0, method1(root.right, Maxval));
        Maxval[0] = Math.max(Maxval[0], (left + right + root.val));
        return Math.max(left, right) + root.val;
    }

    public int method2(TreeNode root) {
        int[] maxVal = new int[1];
        maxVal[0] = Integer.MIN_VALUE;
        method1(root, maxVal);
        return maxVal[0];
    }
}
