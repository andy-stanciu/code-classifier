 * public class TreeNode {
 *     int val;
 *     TreeNode left;

 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

        int var2 = traverse(root);

    public int traverse(TreeNode node) {
        if (node == null)
            return 0;
        else {
            int var3 = traverse(node.left);
            int var4 = traverse(node.right);
            int var5 = Math.max(var3, var4);
            // 1st path = node.val
            // 2nd path = node.val+maxChildSum
            // 3rd path = node.val+leftSum+rightSum
            // pathSum = max of all 3
            int var6 = Math.max(node.val, node.val + var5);
            var6 = Math.max(var6, node.val + var3 + var4);
            // updated maxSum for every node.
            var1 = Math.max(var1, var6);
            // returning curr max path
            return Math.max(node.val, node.val + var5);
        }
    }
}
