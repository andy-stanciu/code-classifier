/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // Initialize maxSum to the smallest possible integer value
    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        inorderTraversal(root);
        return var1;
    }

    private int inorderTraversal(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var2 = Math.max(0, inorderTraversal(root.left));
        int var3 = Math.max(0, inorderTraversal(root.right));
        // Update maxSum by considering the current subtree\'s path sum
        var1 = Math.max(var1, var2 + var3 + root.val);
        // Return the maximum path sum for the current subtree
        return Math.max(var2, var3) + root.val;
    }
}
