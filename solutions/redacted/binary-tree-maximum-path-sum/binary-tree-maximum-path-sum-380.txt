/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxx;

    public int method1(TreeNode root) {
        maxx = -2000;
        traverse(root);
        return maxx;
    }

    private int traverse(TreeNode root) {
        if (root == null)
            return 0;
        int leftT = Math.max(traverse(root.left), 0);
        int rightT = Math.max(traverse(root.right), 0);
        maxx = Math.max(maxx, leftT + rightT + root.val);
        return Math.max(leftT, rightT) + root.val;
    }
}
