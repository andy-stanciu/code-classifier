/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        int ans = helper(root);
        return max;
    }

    public int helper(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(helper(root.left), 0);
        int var2 = Math.max(helper(root.right), 0);
        max = Math.max(max, root.val + var1 + var2);
        return Math.max(var1, var2) + root.val;
    }
}
