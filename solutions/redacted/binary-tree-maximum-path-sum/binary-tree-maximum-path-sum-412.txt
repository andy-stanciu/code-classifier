/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    Map<TreeNode, Integer> map = new HashMap<TreeNode, Integer>();

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int lh = 0;
        if (map.containsKey(root.left)) {
            lh = map.get(root.left);
        } else {
            lh = method1(root.left);
            map.put(root.left, lh);
        }
        int rh = 0;
        if (map.containsKey(root.right)) {
            rh = map.get(root.right);
        } else {
            rh = method1(root.right);
            map.put(root.right, rh);
        }
        int var2 = lh + rh + (root.val);
        var2 = Math.max(var2, Math.max(lh + root.val, rh + root.val));
        var1 = Math.max(Math.max(var1, var2), root.val);
        return Math.max(Math.max(lh, rh) + (root.val), root.val);
    }

    public int var1(TreeNode root) {
        if (root == null)
            return 0;
        var1 = Integer.MIN_VALUE;
        int var3 = method1(root);
        return Math.max(var1, var3);
    }
}
