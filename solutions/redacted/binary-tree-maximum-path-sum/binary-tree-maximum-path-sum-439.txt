/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return var1;
    }

    public int helper(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var2 = helper(root.left);
        if (var2 < 0)
            var2 = 0;
        int var3 = helper(root.right);
        if (var3 < 0)
            var3 = 0;
        var1 = Math.max(var1, var2 + var3 + root.val);
        return root.val + Math.max(var2, var3);
    }
}
