/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] maxvalue = new int[1];
        maxvalue[0] = Integer.MIN_VALUE;
        method2(root, maxvalue);
        return maxvalue[0];
    }

    public int method2(TreeNode root, int[] maxvalue) {
        if (root == null) {
            return 0;
        }
        int leftside = Math.max(0, method2(root.left, maxvalue));
        int rightside = Math.max(0, method2(root.right, maxvalue));
        maxvalue[0] = Math.max(maxvalue[0], leftside + rightside + root.val);
        return Math.max(leftside, rightside) + root.val;
    }
}
