/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int data;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int val) { data = val; left = null, right = null }
 * }
 */
class Solution {

    public static int method1(TreeNode root, int[] maxi) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(0, method1(root.left, maxi));
        int var2 = Math.max(0, method1(root.right, maxi));
        maxi[0] = Math.max(maxi[0], var1 + var2 + root.val);
        return Math.max(var1, var2) + root.val;
    }

    public int method2(TreeNode root) {
        //your code goes here
        int[] maxi = { Integer.MIN_VALUE };
        method1(root, maxi);
        return maxi[0];
    }
}
