/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        find(root);
        return max;
    }

    private int find(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = find(root.left);
        int var2 = find(root.right);
        int self = root.val;
        int var3 = var1 + var2 + self;
        int var4 = Math.max(var1, var2);
        max = Math.max(max, self + Math.max(0, Math.max(var1 + var2, var4)));
        return self + Math.max(0, var4);
    }
}
