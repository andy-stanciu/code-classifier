/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    private int method1(TreeNode root) {
        if (root == null)
            return 0;
        int var2 = Math.max(0, method1(root.left));
        int var3 = Math.max(0, method1(root.right));
        var1 = Math.max(var1, root.val + var2 + var3);
        return root.val + Math.max(var2, var3);
    }

    public int method2(TreeNode root) {
        method1(root);
        return var1;
    }
}
