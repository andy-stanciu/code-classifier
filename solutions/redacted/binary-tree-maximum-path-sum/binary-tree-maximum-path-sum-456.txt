/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static List<Integer> var1;

    public static int dfs(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int left, right;
        left = dfs(node.left);
        right = dfs(node.right);
        int var2 = left + node.val + right;
        left = left > right ? left + node.val : right + node.val;
        left = node.val > left ? node.val : left;
        var2 = var2 > left ? var2 : left;
        var1.add(var2);
        return left;
    }

    public int method1(TreeNode root) {
        var1 = new ArrayList<>();
        dfs(root);
        int var3 = Integer.MIN_VALUE;
        for (int i : var1) {
            if (i > var3) {
                var3 = i;
            }
        }
        return var3;
    }
}
