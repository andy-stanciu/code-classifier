/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        maxGain(root);
        return var1;
    }

    private int maxGain(TreeNode node) {
        if (node == null)
            return 0;
        // Recursively call maxGain on left and right children
        // max of left subtree or 0
        int leftGain = Math.max(maxGain(node.left), 0);
        // max of right subtree or 0
        int rightGain = Math.max(maxGain(node.right), 0);
        // Current node\'s max path sum could be part of the maximum path sum
        int var2 = node.val + leftGain + rightGain;
        // Update maxSum if the current path sum is greater
        var1 = Math.max(var1, var2);
        // Return the max gain if we continue the same path including this node
        return node.val + Math.max(leftGain, rightGain);
    }
}
