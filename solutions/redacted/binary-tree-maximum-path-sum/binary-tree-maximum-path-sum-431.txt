class Solution {

    private int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        maxGain(root);
        return var1;
    }

    private int maxGain(TreeNode node) {
        if (node == null)
            return 0;
        // Calculate max sum of paths through the left and right children
        int leftGain = Math.max(maxGain(node.left), 0);
        int rightGain = Math.max(maxGain(node.right), 0);
        // Update global max sum if the path through this node is greater
        int var2 = node.val + leftGain + rightGain;
        var1 = Math.max(var1, var2);
        // Return the max gain for parent (considering only one child)
        return node.val + Math.max(leftGain, rightGain);
    }
}
