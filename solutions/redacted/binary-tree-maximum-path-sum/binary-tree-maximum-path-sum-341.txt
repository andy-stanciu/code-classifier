/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] res = new int[1];
        res[0] = Integer.MIN_VALUE;
        helper(root, res);
        return res[0];
    }

    public int helper(TreeNode root, int[] res) {
        if (root == null) {
            return 0;
        }
        int var1 = helper(root.left, res);
        int var2 = helper(root.right, res);
        res[0] = Math.max(res[0], var1 + var2 + root.val);
        int ans = Math.max(var1, var2) + root.val;
        return Math.max(ans, 0);
    }
}
