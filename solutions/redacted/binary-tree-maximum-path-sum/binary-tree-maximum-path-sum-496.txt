/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int answer = -1000000000;

    public int method1(TreeNode root) {
        helper(root);
        return answer;
    }

    int helper(TreeNode node) {
        if (node == null) {
            return 0;
        }
        int var1 = Math.max(helper(node.left), 0);
        int var2 = Math.max(helper(node.right), 0);
        answer = Math.max(answer, var1 + var2 + node.val);
        return Math.max(var1, var2) + node.val;
    }
}
