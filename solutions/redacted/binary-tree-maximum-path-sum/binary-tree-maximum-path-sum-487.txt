/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int[] result = new int[1];
        result[0] = Integer.MIN_VALUE;
        method2(root, result);
        return result[0];
    }

    public int method2(TreeNode root, int[] result) {
        if (root == null)
            return 0;
        int lh = Math.max(0, method2(root.left, result));
        int rh = Math.max(0, method2(root.right, result));
        result[0] = Math.max(result[0], lh + rh + root.val);
        return root.val + Math.max(lh, rh);
    }
}
