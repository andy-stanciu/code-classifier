/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        this.method2(root);
        return finalMax;
    }

    int finalMax = Integer.MIN_VALUE;

    public int method2(TreeNode root) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            if (root.val > finalMax) {
                finalMax = root.val;
            }
            return root.val;
        }
        int var1 = method2(root.left);
        int var2 = method2(root.right);
        int currVal = root.val;
        int maxx = Math.max(currVal, Math.max(currVal + var2, currVal + var1));
        int max = Math.max(maxx, currVal + var2 + var1);
        if (max > finalMax) {
            finalMax = max;
        }
        return max == currVal + var2 + var1 ? maxx : max;
    }
}
