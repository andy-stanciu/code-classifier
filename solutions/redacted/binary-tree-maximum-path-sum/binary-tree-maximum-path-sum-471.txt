/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int temp = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        method2(root);
        return temp;
    }

    int method2(TreeNode root) {
        if (root == null)
            return 0;
        int left = Math.max(0, method2(root.left));
        int right = Math.max(0, method2(root.right));
        temp = Math.max(left + right + root.val, temp);
        return Math.max(left, right) + root.val;
    }
}
