/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //or we can take array and pass as reference to pathSum
    int maxi = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return maxi;
    }

    public int method2(TreeNode node) {
        if (node == null) {
            return 0;
        }
        //If any negative value in left or right skip that.Dont change maxi
        int left = Math.max(0, method2(node.left));
        int right = Math.max(0, method2(node.right));
        maxi = Math.max(maxi, left + right + node.val);
        return Math.max(left, right) + node.val;
    }
}
