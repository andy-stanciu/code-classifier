 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}

 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int var1;

    public int method1(TreeNode root) {
        var1 = Integer.MIN_VALUE;
        var1(root);

    public int var1(TreeNode root) {
        if (root == null) {

        }
        int var2 = Math.max(0, var1(root.left));
        int var3 = Math.max(0, var1(root.right));
        int var4 = var2 + var3 + root.val;
        var1 = Math.max(var1, var4);

    }
