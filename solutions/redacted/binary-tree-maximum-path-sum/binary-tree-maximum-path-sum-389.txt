/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int maxi = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        findMax(root, 0);
        return maxi;
    }

    public int findMax(TreeNode root, int max) {
        if (root == null) {
            return 0;
        }
        int l = findMax(root.left, max);
        int r = findMax(root.right, max);
        max = Math.max(Math.max(root.val, root.val + l), Math.max(root.val, root.val + r));
        maxi = Math.max(maxi, Math.max(root.val, Math.max(l + r + root.val, max)));
        return max;
    }
}
