/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return var1;
    }

    private int method2(TreeNode node) {
        if (node == null)
            return 0;
        // Recursively compute the maximum sum from left and right subtree
        // If it\'s negative, we don\'t include it (hence the Math.max with 0)
        int var2 = Math.max(method2(node.left), 0);
        int var3 = Math.max(method2(node.right), 0);
        // Compute the maximum path sum that includes the current node
        // This path can include both left and right branches
        int var4 = node.val + var2 + var3;
        // Update the maximum if this path sum is larger
        var1 = Math.max(var1, var4);
        // Return the maximum sum of a path that can be extended to the parent
        // We can only choose one branch (left or right) to extend upwards
        return node.val + Math.max(var2, var3);
    }
}
