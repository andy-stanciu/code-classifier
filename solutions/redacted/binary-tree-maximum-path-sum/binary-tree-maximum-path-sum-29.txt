class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        update(root);
        return ans;
    }

    private int update(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(update(root.left), 0);
        int var2 = Math.max(update(root.right), 0);
        ans = Math.max(ans, var1 + var2 + root.val);
        return Math.max(var1 + root.val, var2 + root.val);
    }
}
