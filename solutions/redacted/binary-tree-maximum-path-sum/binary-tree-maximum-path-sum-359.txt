/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        inorder(root, root);
        return var1;
    }

    public int inorder(TreeNode root, TreeNode org) {
        if (root == null) {
            return 0;
        }
        int left = inorder(root.left, org);
        int right = inorder(root.right, org);
        int var2 = root.val + left;
        int var3 = root.val + right;
        int var4 = root.val + left + right;
        int var5 = Math.max(Math.max(var2, var4), Math.max(var3, root.val));
        int var6 = Math.max(var2, Math.max(root.val, var3));
        var1 = Math.max(var1, var5);
        return var6;
    }
}
