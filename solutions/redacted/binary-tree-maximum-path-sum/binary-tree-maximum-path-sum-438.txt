/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int dfs(TreeNode curr, int[] maxi) {
        if (curr == null)
            return 0;
        //Find the sum of left and right value
        int lh = Math.max(0, dfs(curr.left, maxi));
        int rh = Math.max(0, dfs(curr.right, maxi));
        //comparing it with maximum value
        maxi[0] = Math.max(lh + rh + curr.val, maxi[0]);
        return curr.val + Math.max(lh, rh);
    }

    public int method1(TreeNode root) {
        int[] maxi = new int[1];
        //Value can be negative so taking manimum value
        maxi[0] = Integer.MIN_VALUE;
        dfs(root, maxi);
        return maxi[0];
    }
}
