/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return ans;
    }

    public int helper(TreeNode root) {
        if (root == null) {
            return Integer.MIN_VALUE;
        }
        int x = helper(root.left);
        int y = helper(root.right);
        if (x == Integer.MIN_VALUE) {
            x = 0;
        }
        if (y == Integer.MIN_VALUE) {
            y = 0;
        }
        if (ans < Math.max(root.val + x + y, Math.max(root.val, Math.max(root.val + x, root.val + y)))) {
            ans = Math.max(root.val + x + y, Math.max(root.val, Math.max(root.val + x, root.val + y)));
        }
        return Math.max(root.val, Math.max(root.val + x, root.val + y));
    }
}
