class Solution {

    public int[] method1(TreeNode root) {
        if (root == null)
            return new int[] { Integer.MIN_VALUE, Integer.MIN_VALUE };
        if (root.left == null && root.right == null) {
            return new int[] { root.val, root.val };
        }
        int[] ls = method1(root.left);
        int[] rs = method1(root.right);
        int mp = Math.max(rs[0], ls[0]);
        int p = mp > 0 ? mp + root.val : root.val;
        int mm = Math.max(rs[1], ls[1]);
        int l = ls[0] > 0 ? ls[0] : 0;
        int r = rs[0] > 0 ? rs[0] : 0;
        int nm = Math.max(l + r + root.val, mm);
        ls[0] = p;
        ls[1] = nm;
        return ls;
    }

    public int method2(TreeNode root) {
        int[] re = method1(root);
        return Math.max(re[0], re[1]);
    }
}
