/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = -1000;

    public int method1(TreeNode root) {
        method2(root);
        return max;
    }

    public int method2(TreeNode node) {
        if (node == null)
            return 0;
        int left = method2(node.left);
        int right = method2(node.right);
        int var1 = node.val + left + right;
        int var2 = node.val + Math.max(left, right);
        int localMax = Math.max(var2, var1);
        max = Math.max(max, Math.max(localMax, node.val));
        return Math.max(node.val, node.val + Math.max(left, right));
    }
}
