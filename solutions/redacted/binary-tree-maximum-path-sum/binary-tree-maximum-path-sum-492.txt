class Solution {

    int result = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return result;
    }

    public int helper(TreeNode root) {
        if (root == null)
            return 0;
        int leftChild = helper(root.left);
        int rightChild = helper(root.right);
        leftChild = Math.max(leftChild, 0);
        rightChild = Math.max(rightChild, 0);
        root.val = root.val + leftChild + rightChild;
        result = Math.max(result, root.val);
        return root.val - Math.min(leftChild, rightChild);
    }
}
