class Solution {

    int res = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        method2(root);
        return res;
    }

    private int method2(TreeNode node) {
        if (node == null)
            return 0;
        int var1 = method2(node.left), var2 = method2(node.right);
        res = Math.max(res, node.val + var1 + var2);
        return Math.max(node.val + Math.max(var1, var2), 0);
    }
}
