/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    private int solve(TreeNode root) {
        if (root == null)
            return 0;
        int var1 = Math.max(0, solve(root.left));
        int var2 = Math.max(0, solve(root.right));
        int var3 = var1 + var2 + root.val;
        max = Math.max(max, var3);
        return root.val + Math.max(var1, var2);
    }

    public int method1(TreeNode root) {
        int a = solve(root);
        return max;
    }
}
