class Solution {

    public int function(TreeNode root, int[] param1) {
        if (root == null)
            return 0;
        int left = Math.max(0, function(root.left, param1));
        int right = Math.max(0, function(root.right, param1));
        param1[0] = Math.max(param1[0], root.val + left + right);
        return root.val + Math.max(left, right);
    }

    public int method1(TreeNode root) {
        int[] param1 = new int[1];
        param1[0] = Integer.MIN_VALUE;
        function(root, param1);
        return param1[0];
    }
}
