/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        int result = method2(root);
        return var1;
    }

    int method2(TreeNode node) {
        if (node == null)
            return 0;
        int lh = method2(node.left);
        if (lh < 0)
            lh = 0;
        int rh = method2(node.right);
        if (rh < 0)
            rh = 0;
        var1 = Math.max(var1, lh + rh + node.val);
        return node.val + Math.max(lh, rh);
    }
}
