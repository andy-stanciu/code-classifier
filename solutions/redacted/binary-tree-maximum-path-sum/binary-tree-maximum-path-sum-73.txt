/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] var1 = new int[] { Integer.MIN_VALUE };
        helper(root, var1);
        return var1[0];
    }

    public int helper(TreeNode node, int[] var1) {
        if (node == null) {
            return 0;
        }
        int left = Math.max(0, helper(node.left, var1));
        int right = Math.max(0, helper(node.right, var1));
        var1[0] = Math.max(var1[0], node.val + left + right);
        return node.val + Math.max(left, right);
    }
}
