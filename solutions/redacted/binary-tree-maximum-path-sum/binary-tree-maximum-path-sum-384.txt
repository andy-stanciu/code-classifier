/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int max = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        if (root == null)
            return 0;
        int value = root.val;
        int var1 = Math.max(method1(root.left), 0);
        int var2 = Math.max(method1(root.right), 0);
        max = Math.max(max, var1 + var2 + value);
        return Math.max(var1, var2) + value;
    }

    public int method2(TreeNode root) {
        method1(root);
        return max;
    }
}
