/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int globalMax = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        rec(root);
        return globalMax;
    }

    private int rec(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int left = rec(root.left);
        int right = rec(root.right);
        int val = root.val;
        int var1 = val + left;
        int var2 = val + right;
        int var3 = val + left + right;
        // Until this node, lets find Max (val, val+left, val+right, val+right+left)
        int var4 = Math.max(val, var1);
        var4 = Math.max(var4, var2);
        var4 = Math.max(var4, var3);
        globalMax = Math.max(var4, globalMax);
        // We can return only leftPath or rightPath because currnode can be part of only one path.
        int var5 = Math.max(val, var1);
        var5 = Math.max(var5, var2);
        return var5;
    }
}
