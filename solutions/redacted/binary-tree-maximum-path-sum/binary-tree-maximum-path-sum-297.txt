/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] maxValue = new int[1];
        maxValue[0] = Integer.MIN_VALUE;
        method2(root, maxValue);
        return maxValue[0];
    }

    public static int method2(TreeNode root, int[] maxValue) {
        if (root == null) {
            return 0;
        }
        int var1 = Math.max(0, method2(root.left, maxValue));
        int var2 = Math.max(0, method2(root.right, maxValue));
        maxValue[0] = Math.max(maxValue[0], var1 + var2 + root.val);
        return Math.max(var1, var2) + root.val;
    }
}
