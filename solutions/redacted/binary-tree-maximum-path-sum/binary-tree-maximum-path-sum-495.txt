/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        helper(root);
        return ans;
    }

    int helper(TreeNode node) {
        if (node == null) {
            // check for the root node first
            return 0;
        }
        // left recursion of left tree
        int left = helper(node.left);
        // right recursion of right tree
        int right = helper(node.right);
        // if there is negative  no on left ignore
        left = Math.max(0, left);
        // if there is neg no. on right ignore
        right = Math.max(0, right);
        // maximum of both left and right
        int var1 = left + right + node.val;
        ans = Math.max(ans, var1);
        return Math.max(left, right) + node.val;
    }
}
