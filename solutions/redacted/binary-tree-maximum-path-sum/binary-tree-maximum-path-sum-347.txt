/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        postOrder(root);
        return var1;
    }

    public int postOrder(TreeNode root) {
        if (root == null)
            return 0;
        int var2 = Math.max(postOrder(root.left), 0);
        int var3 = Math.max(postOrder(root.right), 0);
        var1 = Math.max(var1, var2 + root.val + var3);
        return Math.max(var2 + root.val, var3 + root.val);
    }
}
