class Solution {

    int ans = Integer.MIN_VALUE;

    public int method1(TreeNode root) {
        dfs(root);
        return ans;
    }

    int dfs(TreeNode node) {
        if (node == null)
            return 0;
        int left = dfs(node.left);
        int right = dfs(node.right);
        //possibility 1, 2, 3
        //possiblity 1: left  + node.val
        //possiblity 2: right + node.val
        //possiblity 3: 0     + node.val
        int val = Math.max(0, Math.max(left, right)) + node.val;
        //possibility 4: (left + right + node.val)
        //ans is the max among possibilities 1,2,3,4
        ans = Math.max(ans, Math.max(val, left + right + node.val));
        //return the max among possibilities 1,2,3
        return val;
    }
}
