/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root) {
        int[] ans = new int[] { Integer.MIN_VALUE };
        int val = method1(root, ans);
        return ans[0];
    }

    private int method1(TreeNode root, int[] ans) {
        if (root == null)
            return 0;
        int l = method1(root.left, ans);
        int r = method1(root.right, ans);
        int max = Math.max(Math.max(l + root.val, r + root.val), l + r + root.val);
        ans[0] = Math.max(ans[0], Math.max(root.val, max));
        return Math.max(root.val, Math.max(root.val + l, root.val + r));
    }
}
