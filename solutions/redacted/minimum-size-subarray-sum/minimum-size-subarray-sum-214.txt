class Solution {

    public int method1(int target, int[] nums) {
        int minL = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int var1 = 0;
        while (i < nums.length) {
            // Expand the window by moving `j` to the right
            while (var1 < target && j < nums.length) {
                var1 += nums[j];
                j++;
            }
            // If the current window sum is equal to or greater than the target, update the minimum length
            if (var1 >= target && (j - i) < minL) {
                minL = j - i;
            }
            // Shrink the window by moving `i` to the right
            var1 = var1 - nums[i];
            i++;
        }
        return (minL == Integer.MAX_VALUE) ? 0 : minL;
    }
}
