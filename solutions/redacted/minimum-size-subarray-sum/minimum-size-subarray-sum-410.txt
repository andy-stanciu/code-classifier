class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int var1 = 0;
        while (j < nums.length && i < nums.length) {
            if (nums[j] == target) {
                return 1;
            }
            var1 = var1 + nums[j];
            if (var1 < target) {
                j++;
            }
            if (var1 >= target) {
                while (var1 >= target) {
                    var1 = var1 - nums[i];
                    ans = Math.min(ans, (j - i) + 1);
                    i++;
                }
                j++;
            }
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
