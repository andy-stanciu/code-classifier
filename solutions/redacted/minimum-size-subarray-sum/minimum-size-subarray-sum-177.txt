class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int i = 0;
        int min_length = Integer.MAX_VALUE;
        for (int j = 0; j < nums.length; ++j) {
            var1 += nums[j];
            while (var1 >= target) {
                min_length = Math.min(min_length, j - i + 1);
                var1 -= nums[i];
                i++;
            }
        }
        return (min_length == Integer.MAX_VALUE) ? 0 : min_length;
    }
}
