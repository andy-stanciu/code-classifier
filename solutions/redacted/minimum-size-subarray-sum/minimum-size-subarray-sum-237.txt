class Solution {

    public int method1(int target, int[] nums) {
        int shortLenght = 0;
        int var1 = 0;
        int left = 0;
        for (int right = 0; right < nums.length; ++right) {
            // Add the right number into window to check the sum
            var1 += nums[right];
            //If sum exceed the target reduce the left pointer check the minimum length of the subArray
            while (var1 >= target) {
                // if shortLenght is 0 get the max window otherwise get the smallest one
                shortLenght = (shortLenght == 0 ? (right - left) + 1 : Math.min(shortLenght, (right - left) + 1));
                // redurce the left number from the Array
                var1 -= nums[left];
                ++left;
            }
        }
        return shortLenght;
    }
}
