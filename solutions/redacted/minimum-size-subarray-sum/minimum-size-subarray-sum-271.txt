class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, total = 0;
        int res = Integer.MAX_VALUE;
        for (int r = 0; r < nums.length; r++) {
            // subarray total
            total += nums[r];
            while (total >= target) {
                // keep update the min size subarray
                res = Math.min(res, r - l + 1);
                //subtract left pointer and move one spot right
                total -= nums[l++];
            }
        }
        return (res == Integer.MAX_VALUE) ? 0 : res;
    }
}
