class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int var1 = 0;
        int min = 100001;
        for (int r = 0; r < nums.length; r++) {
            var1 += nums[r];
            while (var1 >= target) {
                min = Math.min((r - l) + 1, min);
                var1 -= nums[l];
                l++;
            }
        }
        if (l == 0) {
            return 0;
        } else {
            return min;
        }
    }
}
