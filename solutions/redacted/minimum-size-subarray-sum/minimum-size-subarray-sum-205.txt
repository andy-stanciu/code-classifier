class Solution {

    public int method1(int target, int[] nums) {
        int total = 0;
        int minLength = nums.length;
        int left = 0;
        int var1 = 0;
        for (int right = 0; right < nums.length; right++) {
            total += nums[right];
            var1 += nums[right];
            while (var1 >= target) {
                minLength = Math.min(minLength, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        if (total < target)
            return 0;
        return minLength;
    }
}
