class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int right = 0;
        int var1 = 0;
        int minSubArrLen = Integer.MAX_VALUE;
        while (right < n) {
            var1 += nums[right];
            while (var1 >= target) {
                minSubArrLen = Math.min(minSubArrLen, right - left + 1);
                var1 -= nums[left];
                left++;
            }
            right++;
        }
        return minSubArrLen != Integer.MAX_VALUE ? minSubArrLen : 0;
    }
}
