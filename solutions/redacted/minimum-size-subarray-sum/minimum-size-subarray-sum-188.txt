class Solution {

    public int method1(int target, int[] nums) {
        int cs = 0, cnt = 0, mCnt = Integer.MAX_VALUE;
        int n = nums.length;
        int l = 0;
        for (int r = 0; r < n; r++) {
            cs += nums[r];
            while (cs >= target) {
                mCnt = Math.min(mCnt, r - l + 1);
                cs -= nums[l];
                l++;
            }
        }
        // If sum comes to be < target then return 0
        return (mCnt == Integer.MAX_VALUE) ? 0 : mCnt;
    }
}
