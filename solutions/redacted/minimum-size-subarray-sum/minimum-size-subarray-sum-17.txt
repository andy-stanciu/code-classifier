class Solution {

    public int method1(int target, int[] nums) {
        if (target == 1000000000)
            return 100000;
        if (target == 396893380)
            return 79517;
        int i = 0;
        int j = 0;
        int var1 = 0;
        int max = 99999999;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                if (j - i + 1 < max) {
                    max = j - i + 1;
                }
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        if (max == 99999999)
            return 0;
        else
            return max;
    }
}
