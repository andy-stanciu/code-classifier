class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int var2 = Integer.MAX_VALUE;
        for (int j = 0, i = 0; j < nums.length; ++j) {
            var1 += nums[j];
            while (var1 >= target) {
                var1 -= nums[i];
                var2 = Math.min(var2, (j - i + 1));
                ++i;
            }
        }
        return var2 == Integer.MAX_VALUE ? 0 : var2;
    }
}
