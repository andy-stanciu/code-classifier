class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int n = nums.length;
        int len = Integer.MAX_VALUE, var1 = 0;
        while (i < n) {
            var1 += nums[i];
            while (var1 >= target) {
                len = Math.min(len, i - j + 1);
                var1 -= nums[j++];
            }
            i++;
        }
        if (len == Integer.MAX_VALUE) {
            return 0;
        }
        return len;
    }
}
