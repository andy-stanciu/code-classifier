class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int left = 0, right = 0;
        int ans = Integer.MAX_VALUE;
        while (right < nums.length) {
            var1 += nums[right];
            while (var1 >= target) {
                ans = Math.min(ans, right - left + 1);
                var1 -= nums[left];
                ++left;
            }
            ++right;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
