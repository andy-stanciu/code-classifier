class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int left = 0;
        int right = 0;
        int var1 = 0;
        for (right = 0; right < nums.length; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                ans = Math.min(ans, right - left + 1);
                var1 -= nums[left];
                left++;
            }
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        }
        return ans;
    }
}
