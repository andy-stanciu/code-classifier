class Solution {

    public int method1(int target, int[] nums) {
        int start = 0;
        int var1 = 0;
        int result = Integer.MAX_VALUE;
        for (int end = 0; end < nums.length; end++) {
            var1 += nums[end];
            while (var1 >= target) {
                result = Math.min(result, end - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        if (result == Integer.MAX_VALUE)
            return 0;
        else
            return result;
    }
}
