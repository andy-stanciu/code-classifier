class Solution {

    public int method1(int target, int[] nums) {
        // declare left pointer
        int left = 0;
        // declare sum of sliding window with 0
        int var1 = 0;
        // declare minLen and set it to max value
        int minLen = Integer.MAX_VALUE;
        // iterate with right pointer over nums array
        for (int right = 0; right < nums.length; right++) {
            // add num at right pointer to sum
            var1 += nums[right];
            // if sum more than target:
            while (var1 >= target) {
                // compare minLen with current len of window and set min to minLen
                minLen = Math.min(minLen, right + 1 - left);
                // subtract from sum number at left pointer
                var1 -= nums[left];
                // increment left pointer
                left++;
            }
        }
        // check if minLen are default value that we set at start, if yes return 0 else minLen
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
