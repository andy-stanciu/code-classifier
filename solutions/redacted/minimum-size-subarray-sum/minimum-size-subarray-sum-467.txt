class Solution {

    public int method1(int target, int[] nums) {
        int start = 0;
        int var1 = 0;
        int minLen = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                minLen = Math.min(minLen, i - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        return minLen < Integer.MAX_VALUE ? minLen : 0;
    }
}
