class Solution {

    public int method1(int tg, int[] nums) {
        int var1 = 0, n = nums.length, cnt = Integer.MAX_VALUE, j = 0;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
            while (var1 >= tg) {
                cnt = Math.min(cnt, i - j + 1);
                var1 -= nums[j++];
            }
        }
        return cnt == Integer.MAX_VALUE ? 0 : cnt;
    }
}
