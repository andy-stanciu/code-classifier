class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length, min = Integer.MAX_VALUE, var1 = 0;
        int i = 0, j = 0;
        while (i < n) {
            while (j < n && var1 < target) var1 += nums[j++];
            if (var1 >= target)
                min = Math.min(min, j - i);
            var1 -= nums[i++];
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
