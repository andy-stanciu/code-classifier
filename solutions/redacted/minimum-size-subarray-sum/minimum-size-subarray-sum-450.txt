class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = i, mini = Integer.MAX_VALUE, var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                mini = Math.min(mini, j - i + 1);
                var1 -= nums[i++];
            }
            j++;
        }
        return mini == Integer.MAX_VALUE ? 0 : mini;
    }
}
