class Solution {

    public int method1(int target, int[] a) {
        int l = 0, r = 0, len = a.length, var1 = 0, ans = Integer.MAX_VALUE;
        while (r < len) {
            var1 += a[r];
            // System.out.println(sum);
            while (var1 >= target) {
                ans = Math.min(ans, r - l + 1);
                var1 -= a[l];
                l++;
            }
            r++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
