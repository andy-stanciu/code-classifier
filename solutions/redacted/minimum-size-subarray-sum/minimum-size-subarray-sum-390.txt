class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int var1 = 0;
        int minLength = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                if (i - left + 1 < minLength || minLength == 0) {
                    minLength = i - left + 1;
                }
                var1 -= nums[left++];
            }
        }
        return minLength;
    }
}
