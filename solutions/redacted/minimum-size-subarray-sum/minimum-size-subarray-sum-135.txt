class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int var1 = 0;
        int end = 0;
        int minLen = Integer.MAX_VALUE;
        for (int start = 0; start < n; start++) {
            var1 += nums[start];
            while (var1 >= target) {
                minLen = Math.min(minLen, start - end + 1);
                var1 -= nums[end];
                end++;
            }
        }
        return (minLen != Integer.MAX_VALUE) ? minLen : 0;
    }
}
