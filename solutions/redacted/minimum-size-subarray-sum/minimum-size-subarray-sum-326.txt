class Solution {

    public int method1(int target, int[] nums) {
        int left = 0, ans = nums.length + 1, var1 = 0;
        for (int right = 0; right < nums.length; right++) {
            var1 += nums[right];
            while (var1 >= target) {
                ans = Math.min(ans, right - left + 1);
                var1 -= nums[left++];
            }
        }
        return ans == nums.length + 1 ? 0 : ans;
    }
}
