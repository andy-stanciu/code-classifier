class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int count = Integer.MAX_VALUE, var1 = 0;
        for (int start = 0, end = 0; end < n; end++) {
            var1 += nums[end];
            while (var1 >= target) {
                count = Math.min(count, end - start + 1);
                var1 = var1 - nums[start++];
            }
        }
        return count == Integer.MAX_VALUE ? 0 : count;
    }
}
