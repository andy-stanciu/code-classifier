class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0, i = 0, j = 0;
        int min = Integer.MAX_VALUE;
        while (i < nums.length && j < nums.length) {
            if (var1 + nums[j] < target) {
                var1 += nums[j];
                j++;
            } else {
                var1 -= nums[i];
                min = Math.min(min, j - i + 1);
                i++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
