class Solution {

    public int method1(int target, int[] a) {
        int var1 = 0;
        int min = 1000000;
        int si = 0;
        for (int i = 0; i < a.length; i++) {
            var1 += a[i];
            while (var1 >= target && si <= i) {
                min = Math.min(min, i - si);
                var1 -= a[si];
                si++;
            }
        }
        if (min == 1000000)
            return 0;
        return 1 + min;
    }
}
