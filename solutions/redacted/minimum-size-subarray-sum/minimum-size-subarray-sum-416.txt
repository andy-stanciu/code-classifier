class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                var1 -= nums[i];
                ans = Math.min(ans, j - i + 1);
                i++;
            }
            j++;
        }
        if (ans > nums.length) {
            return 0;
        }
        return ans;
    }
}
