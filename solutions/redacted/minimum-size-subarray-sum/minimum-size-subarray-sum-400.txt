class Solution {

    public int method1(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int var1 = 0;
        int j = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            var1 = var1 + nums[i];
            while (var1 >= target) {
                min = Math.min(min, i - j + 1);
                var1 -= nums[j];
                j++;
            }
        }
        if (min > n) {
            return 0;
        }
        return min;
    }
}
