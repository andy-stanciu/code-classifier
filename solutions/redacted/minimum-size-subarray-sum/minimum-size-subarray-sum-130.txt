class Solution {

    public int method1(int target, int[] nums) {
        int minArrayLength = Integer.MAX_VALUE;
        int start = 0, end = 0;
        int var1 = 0;
        while (end < nums.length) {
            var1 += nums[end];
            while (var1 >= target) {
                minArrayLength = Math.min(minArrayLength, (end - start) + 1);
                var1 -= nums[start];
                start++;
            }
            end++;
        }
        return minArrayLength == Integer.MAX_VALUE ? 0 : minArrayLength;
    }
}
