class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0, j = 0, i = 0, min = Integer.MAX_VALUE;
        while (i < nums.length && j < nums.length) {
            var1 = var1 + nums[j];
            while (var1 >= target && i <= j) {
                int ans = j - i + 1;
                if (ans < min)
                    min = ans;
                var1 = var1 - nums[i];
                i++;
            }
            j++;
        }
        if (j - i + 1 < min && var1 == target)
            return Math.min(min, j - i + 1);
        if (min == Integer.MAX_VALUE)
            return 0;
        return min;
    }
}
