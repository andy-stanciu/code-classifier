class Solution {

    public int method1(int target, int[] nums) {
        int leftPointer = -1;
        int rightPointer = -1;
        int var1 = 0;
        int minLength = Integer.MAX_VALUE;
        while (true) {
            boolean loop1 = false;
            boolean loop2 = false;
            while (rightPointer < nums.length - 1 && var1 < target) {
                loop1 = true;
                rightPointer++;
                var1 += nums[rightPointer];
            }
            while (leftPointer <= rightPointer && var1 >= target) {
                loop2 = true;
                minLength = Math.min(minLength, rightPointer - leftPointer);
                leftPointer++;
                var1 -= nums[leftPointer];
            }
            if (!loop1 && !loop2) {
                break;
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
