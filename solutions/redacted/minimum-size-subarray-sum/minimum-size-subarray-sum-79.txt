class Solution {

    public int method1(int target, int[] nums) {
        int min = Integer.MAX_VALUE;
        int i = 0;
        int j = 0;
        int var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            if (var1 < target) {
                j++;
            }
            if (var1 >= target) {
                int win = j - i + 1;
                if (win < min)
                    min = win;
                while (var1 > target) {
                    var1 -= nums[i];
                    i++;
                    if (var1 >= target) {
                        if (j - i + 1 < min)
                            min = j - i + 1;
                    }
                }
                j++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
