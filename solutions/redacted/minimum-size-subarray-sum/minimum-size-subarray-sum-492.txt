class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int left = 0;
        int end = 0;
        int var1 = 0;
        int min = Integer.MAX_VALUE;
        while (end < n) {
            var1 += nums[end];
            //Finding shortest subarray
            while (var1 >= target) {
                min = Math.min(min, end - left + 1);
                var1 -= nums[left];
                left++;
            }
            end++;
        }
        //If there is no subarray
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
