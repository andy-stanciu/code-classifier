class Solution {

    public int method1(int target, int[] nums) {
        int len = Integer.MAX_VALUE;
        int j = 0, i = 0;
        int var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                len = Math.min(len, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return len == Integer.MAX_VALUE ? 0 : len;
    }
}
