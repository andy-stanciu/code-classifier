class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int l = 0, r = 0;
        int res = Integer.MAX_VALUE, var1 = 0;
        while (r < n) {
            var1 += nums[r];
            if (var1 < target)
                r++;
            else {
                while (var1 >= target) {
                    res = Math.min(res, (r - l) + 1);
                    var1 -= nums[l];
                    l++;
                }
                r++;
            }
        }
        if (l == 0)
            return 0;
        return res;
    }
}
