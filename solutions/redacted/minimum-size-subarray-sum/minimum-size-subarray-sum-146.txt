class Solution {

    public int method1(int target, int[] nums) {
        int left = 0, right = 0, mini = Integer.MAX_VALUE, var1 = 0;
        while (right < nums.length) {
            var1 += nums[right];
            while (left <= right && var1 >= target) {
                var1 -= nums[left];
                mini = Math.min(mini, right - left + 1);
                left++;
            }
            right++;
        }
        return (mini == Integer.MAX_VALUE) ? 0 : mini;
    }
}
