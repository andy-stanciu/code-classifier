class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int left = 0, right = 0;
        int var1 = 0, ans = n + 1;
        while (right < n) {
            var1 += nums[right];
            while (var1 >= target) {
                ans = Math.min(ans, right - left + 1);
                var1 -= nums[left++];
            }
            ++right;
        }
        return ans == n + 1 ? 0 : ans;
    }
}
