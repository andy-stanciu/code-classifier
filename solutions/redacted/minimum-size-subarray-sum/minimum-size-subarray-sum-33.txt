class Solution {

    public int method1(int target, int[] nums) {
        int minLen = Integer.MAX_VALUE;
        int var1 = 0;
        int startIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                minLen = Math.min(minLen, i - startIndex + 1);
                var1 -= nums[startIndex];
                ++startIndex;
            }
        }
        return minLen == Integer.MAX_VALUE ? 0 : minLen;
    }
}
