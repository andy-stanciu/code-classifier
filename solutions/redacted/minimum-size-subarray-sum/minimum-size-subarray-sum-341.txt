class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int n = nums.length;
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        while (r < n) {
            var2 += nums[r];
            while (var2 >= target) {
                var1 = Math.min(var1, r - l + 1);
                var2 -= nums[l];
                l++;
            }
            r++;
        }
        if (var1 == Integer.MAX_VALUE) {
            var1 = 0;
        }
        return var1;
    }
}
