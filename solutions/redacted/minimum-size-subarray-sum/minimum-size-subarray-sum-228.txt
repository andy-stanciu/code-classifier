class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int var1 = 0;
        int minLength = Integer.MAX_VALUE;
        int currLength = 0;
        while (right < nums.length) {
            var1 += nums[right];
            right++;
            while (var1 >= target) {
                var1 -= nums[left];
                left++;
                minLength = Math.min(minLength, right - left + 1);
            }
        }
        return minLength == Integer.MAX_VALUE ? 0 : minLength;
    }
}
