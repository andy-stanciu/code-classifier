class Solution {

    public int method1(int target, int[] nums) {
        int min = nums.length + 1;
        int start = 0;
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                min = Math.min(min, i - start + 1);
                var1 -= nums[start++];
            }
        }
        return (min == nums.length + 1) ? 0 : min;
    }
}
