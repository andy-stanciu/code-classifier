class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int n = nums.length;
        int min = Integer.MAX_VALUE;
        int var1 = 0;
        int stlength = -1;
        while (r < n) {
            var1 += nums[r];
            while (var1 >= target) {
                if (min > r - l + 1) {
                    min = r - l + 1;
                    stlength = l;
                }
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        if (stlength == -1) {
            return 0;
        }
        return min;
    }
}
