class Solution {

    public static int method1(int target, int[] nums) {
        int ret = Integer.MAX_VALUE;
        int var1 = 0;
        int i = 0;
        int j = 0;
        while (j < nums.length) {
            var1 += nums[j];
            if (var1 < target) {
                j++;
            } else {
                while (var1 >= target) {
                    ret = Math.min(ret, (j - i + 1));
                    var1 -= nums[i++];
                }
                j++;
            }
        }
        if (i != 0) {
            return ret;
        }
        return 0;
    }
}
