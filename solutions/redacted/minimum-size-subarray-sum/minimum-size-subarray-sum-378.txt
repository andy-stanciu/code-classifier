class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int end = nums.length;
        int start = 0;
        int var1 = 0;
        int minumumLength = Integer.MAX_VALUE;
        while (left < end) {
            while (start < end && var1 < target) {
                var1 = var1 + nums[start];
                start++;
            }
            if (var1 >= target) {
                minumumLength = Math.min(minumumLength, start - left);
            }
            var1 = var1 - nums[left];
            left++;
        }
        if (minumumLength == Integer.MAX_VALUE)
            return 0;
        return minumumLength;
    }
}
