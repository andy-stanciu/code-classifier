class Solution {

    public int method1(int target, int[] nums) {
        int l = nums.length;
        int minLength = Integer.MAX_VALUE, var1 = 0;
        int j = 0;
        for (int i = 0; i < l; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                minLength = Math.min(minLength, i - j + 1);
                if (j < l)
                    var1 -= nums[j++];
            }
        }
        return (minLength == Integer.MAX_VALUE) ? 0 : minLength;
    }
}
