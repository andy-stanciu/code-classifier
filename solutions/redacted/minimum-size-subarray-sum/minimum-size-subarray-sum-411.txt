class Solution {

    public int method1(int target, int[] nums) {
        int s = 0;
        int var1 = 0;
        int m = Integer.MAX_VALUE;
        int e = 0;
        int n = nums.length;
        while (e < n) {
            var1 += nums[e++];
            while (var1 >= target && s < e) {
                m = Math.min(m, e - s);
                var1 -= nums[s++];
            }
        }
        return (m == Integer.MAX_VALUE) ? 0 : m;
    }
}
