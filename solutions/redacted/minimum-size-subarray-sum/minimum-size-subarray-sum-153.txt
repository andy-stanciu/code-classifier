class Solution {

    public int method1(int tar, int[] nums) {
        int n = nums.length;
        int min = Integer.MAX_VALUE;
        int start = 0;
        int var1 = 0;
        for (int i = 0; i < n; i++) {
            var1 += nums[i];
            while (var1 >= tar) {
                min = Math.min(min, i - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
