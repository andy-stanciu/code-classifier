class Solution {

    public int method1(int target, int[] nums) {
        int start = 0;
        int end = 0;
        int var1 = 0;
        int minLength = Integer.MAX_VALUE;
        while (end < nums.length) {
            var1 += nums[end];
            while (var1 >= target) {
                minLength = Math.min(end - start + 1, minLength);
                var1 -= nums[start];
                start++;
            }
            end++;
        }
        if (minLength == Integer.MAX_VALUE) {
            return 0;
        }
        return minLength;
    }
}
