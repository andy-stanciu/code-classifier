class Solution {

    public int method1(int target, int[] nums) {
        int total = 0;
        int var1 = 0;
        int count = 0, minCount = Integer.MAX_VALUE;
        int left = 0, right = 0;
        for (int n : nums) total += n;
        if (target > total)
            return 0;
        while (right < nums.length) {
            var1 += nums[right++];
            count++;
            while (var1 >= target) {
                minCount = Math.min(minCount, count);
                var1 -= nums[left++];
                count--;
            }
        }
        return minCount;
    }
}
