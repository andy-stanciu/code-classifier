class Solution {

    public int method1(int target, int[] nums) {
        int minLength = Integer.MAX_VALUE;
        int var1 = 0;
        int start = 0;
        for (int end = 0; end < nums.length; end++) {
            var1 += nums[end];
            // Shrink the window as long as currentSum is larger than or equal to target
            while (var1 >= target) {
                minLength = Math.min(minLength, end - start + 1);
                var1 -= nums[start];
                start++;
            }
        }
        // If minLength was not updated, return 0 (no such subarray exists)
        return (minLength == Integer.MAX_VALUE) ? 0 : minLength;
    }
}
