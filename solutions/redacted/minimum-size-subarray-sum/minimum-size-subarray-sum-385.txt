class Solution {

    public int method1(int target, int[] nums) {
        //using the sliding window technique
        int counter = 0;
        int var1 = 0;
        int j = 0;
        int minLen = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            counter++;
            while (j < nums.length && var1 >= target) {
                var1 = var1 - nums[j++];
                minLen = Math.min(counter, minLen);
                counter--;
            }
        }
        if (counter == nums.length) {
            return 0;
        }
        return minLen;
    }
}
