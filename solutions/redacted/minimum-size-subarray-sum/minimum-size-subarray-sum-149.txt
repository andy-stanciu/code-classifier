class Solution {

    public int method1(int target, int[] nums) {
        int l = 0, r = 0, n = nums.length;
        int minLen = 100001;
        int var1 = 0;
        while (r < n) {
            var1 += nums[r];
            while (var1 >= target) {
                minLen = Math.min(minLen, r - l + 1);
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        return minLen == 100001 ? 0 : minLen;
    }
}
