class Solution {

    public int method1(int s, int[] a) {
        if (a == null || a.length == 0)
            return 0;
        int i = 0, j = 0, var1 = 0, min = Integer.MAX_VALUE;
        while (j < a.length) {
            var1 += a[j++];
            while (var1 >= s) {
                min = Math.min(min, j - i);
                var1 -= a[i++];
            }
        }
        return min == Integer.MAX_VALUE ? 0 : min;
    }
}
