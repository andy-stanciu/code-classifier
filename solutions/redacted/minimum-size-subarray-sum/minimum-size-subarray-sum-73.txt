class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int var1 = 0;
        int res = Integer.MAX_VALUE;
        while (j < nums.length) {
            var1 = var1 + nums[j];
            if (var1 < target) {
                j++;
            } else if (var1 == target) {
                res = Math.min(res, j - i + 1);
                j++;
            } else if (var1 > target) {
                while (var1 > target) {
                    res = Math.min(res, j - i + 1);
                    var1 = var1 - nums[i];
                    i++;
                }
                if (var1 == target) {
                    res = Math.min(res, j - i + 1);
                }
                j++;
            }
        }
        if (res == Integer.MAX_VALUE) {
            return 0;
        }
        return res;
    }
}
