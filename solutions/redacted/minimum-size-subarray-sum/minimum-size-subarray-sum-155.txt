class Solution {

    public int method1(int target, int[] nums) {
        if (nums.length == 0) {
            // Handle empty array case
            return 0;
        }
        int result = Integer.MAX_VALUE;
        int left = 0;
        int right = 0;
        int var1 = nums[0];
        while (left < nums.length) {
            // Expand the window
            if (var1 < target && right < nums.length - 1) {
                right++;
                var1 += nums[right];
            } else // Shrink the window
            if (var1 >= target) {
                // Update result before shrinking
                result = Math.min(result, right - left + 1);
                var1 -= nums[left];
                left++;
            } else // If we can\'t expand anymore and the sum is still smaller than target
            {
                break;
            }
        }
        // If result was never updated, return 0
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
