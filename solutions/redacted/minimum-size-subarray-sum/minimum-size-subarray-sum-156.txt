class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int minLen = Integer.MAX_VALUE;
        int i = 0;
        int var1 = 0;
        for (int j = 0; j < n; j++) {
            var1 += nums[j];
            while (var1 >= target) {
                minLen = Math.min(minLen, j - i + 1);
                var1 -= nums[i];
                i++;
            }
        }
        if (minLen == Integer.MAX_VALUE)
            return 0;
        return minLen;
    }
}
