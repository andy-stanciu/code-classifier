class Solution {

    public int method1(int target, int[] nums) {
        int s = 0, n = nums.length, var1 = 0, result = Integer.MAX_VALUE;
        for (int e = 0; e < n; e++) {
            var1 += nums[e];
            while (var1 >= target && s <= e) {
                result = Math.min(result, e - s + 1);
                var1 -= nums[s++];
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
