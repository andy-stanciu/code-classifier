class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int var1 = 0;
        int len = Integer.MAX_VALUE;
        while (right < nums.length) {
            var1 += nums[right];
            right++;
            while (var1 >= target) {
                len = Math.min(len, right - left);
                var1 -= nums[left];
                left++;
            }
        }
        return len == Integer.MAX_VALUE ? 0 : len;
    }
}
