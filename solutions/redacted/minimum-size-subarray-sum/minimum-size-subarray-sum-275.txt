class Solution {

    public int method1(int target, int[] nums) {
        int var1 = Integer.MAX_VALUE;
        int var2 = nums[0];
        if (var2 >= target) {
            return 1;
        }
        int lastI = 0;
        int lastJ = 0;
        int i = 0;
        int j = 1;
        while (i != nums.length && j != nums.length) {
            if (i != lastI) {
                var2 -= nums[lastI];
                lastI = i;
            } else if (j != lastJ) {
                var2 += nums[j];
                lastJ = j;
            }
            if (var2 >= target) {
                if ((j - i) + 1 < var1) {
                    var1 = (j - i) + 1;
                }
                i++;
            } else {
                if (j + 1 == nums.length) {
                    i++;
                } else {
                    j++;
                }
            }
        }
        if (var1 == Integer.MAX_VALUE) {
            return 0;
        } else {
            return var1;
        }
    }
}
