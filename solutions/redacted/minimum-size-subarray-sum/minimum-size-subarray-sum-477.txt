class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        for (int num : nums) var1 += num;
        if (var1 < target)
            return 0;
        int var2 = Integer.MAX_VALUE;
        int var3 = 0;
        int windowStart = 0;
        for (int windowEnd = 0; windowEnd < nums.length; windowEnd++) {
            var3 += nums[windowEnd];
            while (var3 >= target) {
                // +1 bcs the indexing thing.
                var2 = Math.min(var2, windowEnd - windowStart + 1);
                var3 -= nums[windowStart];
                windowStart++;
            }
        }
        return var2;
    }
}
