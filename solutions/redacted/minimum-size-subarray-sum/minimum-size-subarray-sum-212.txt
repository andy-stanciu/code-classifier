class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int var1 = 0;
        int len = Integer.MAX_VALUE;
        while (j < nums.length) {
            var1 += nums[j];
            //If sum of i to j is  greater than or equal to target
            //There might be a possible answer so store it if it\'s less
            // remove to ith element from sum to decrease the sum
            while (var1 >= target) {
                len = Math.min(len, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            //After loop sum is always less then target
            //So just Increase the j index to add jth element int sum
            j++;
        }
        return len == Integer.MAX_VALUE ? 0 : len;
    }
}
