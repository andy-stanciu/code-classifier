class Solution {

    public int method1(int target, int[] nums) {
        int low = 0;
        int high = 0;
        int var1 = 0;
        int minlen = 0;
        int lowestint = Integer.MAX_VALUE;
        while (high < nums.length) {
            var1 = var1 + nums[high];
            high++;
            while (var1 >= target) {
                minlen = high - low;
                lowestint = Math.min(lowestint, minlen);
                var1 = var1 - nums[low];
                low++;
            }
        }
        if (minlen < lowestint) {
            return 0;
        } else {
            return lowestint;
        }
    }
}
