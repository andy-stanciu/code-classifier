// Algo Used: Sliding Window
// TC: O N , SC: O 1
class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int l = 0;
        int r = 0;
        int var1 = 0;
        int ans = Integer.MAX_VALUE;
        int currLen = 0;
        while (r < n) {
            // run till the right pointer reaches end
            var1 += nums[r];
            while (var1 >= target) {
                // if found potential ans then reduce the window and keep on updating the window i.e ans
                currLen = r - l + 1;
                ans = Math.min(ans, currLen);
                var1 -= nums[l];
                // shrinkiing window
                l++;
            }
            // expanding window
            r++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
