class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int res = Integer.MAX_VALUE;
        int var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                res = Math.min(res, j - i + 1);
                var1 -= nums[i];
                i += 1;
            }
            j += 1;
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}
