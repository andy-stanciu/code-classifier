class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int min_len = Integer.MAX_VALUE;
        int var1 = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (i <= j && var1 >= target) {
                min_len = Math.min(min_len, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return min_len == Integer.MAX_VALUE ? 0 : min_len;
    }
}
