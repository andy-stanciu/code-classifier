class Solution {

    public int method1(int target, int[] nums) {
        int start = 0, end = 0, var1 = nums[0], minLen = 100_001;
        while (end < nums.length && start <= end) {
            if (var1 >= target) {
                int len = end - start + 1;
                if (minLen > len)
                    minLen = len;
                var1 -= nums[start];
                start++;
            } else {
                end++;
                var1 += end >= nums.length ? 0 : nums[end];
            }
        }
        return minLen == 100_001 ? 0 : minLen;
    }
}
