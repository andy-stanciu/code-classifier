class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int var1 = 0;
        int len = Integer.MAX_VALUE;
        while (r < nums.length) {
            var1 += nums[r];
            while (l <= r && var1 >= target) {
                len = Math.min(len, r - l + 1);
                var1 = var1 - nums[l++];
            }
            r++;
        }
        return len == Integer.MAX_VALUE ? 0 : len;
    }
}
