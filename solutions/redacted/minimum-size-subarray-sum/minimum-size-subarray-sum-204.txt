class Solution {

    public int method1(int target, int[] nums) {
        int minLength = Integer.MAX_VALUE;
        int var1 = 0;
        int i = 0, j = 0;
        while (i <= j && j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                minLength = Math.min(minLength, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return (minLength == Integer.MAX_VALUE ? 0 : minLength);
    }
}
