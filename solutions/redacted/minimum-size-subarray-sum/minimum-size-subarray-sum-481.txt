class Solution {

    public int method1(int target, int[] nums) {
        int in = 0;
        for (int el : nums) {
            in += el;
        }
        if (in < target)
            return 0;
        int len = nums.length;
        int var1 = 0;
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                len = Math.min(len, i - j + 1);
                var1 -= nums[j];
                j++;
            }
        }
        return len;
    }
}
