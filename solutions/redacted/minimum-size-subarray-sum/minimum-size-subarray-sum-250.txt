class Solution {

    public int method1(int target, int[] nums) {
        int l = 0;
        int r = 0;
        int var1 = 0;
        int ans = Integer.MAX_VALUE;
        while (r < nums.length) {
            var1 += nums[r];
            while (var1 >= target) {
                ans = Math.min(ans, r - l + 1);
                var1 -= nums[l++];
            }
            r++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
