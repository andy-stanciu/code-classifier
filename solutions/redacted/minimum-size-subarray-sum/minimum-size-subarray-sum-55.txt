class Solution {

    public int method1(int target, int[] nums) {
        int slowPtr = 0;
        int fastPtr;
        int var1 = 0;
        int minLen = Integer.MAX_VALUE;
        for (fastPtr = 0; fastPtr < nums.length; fastPtr++) {
            var1 += nums[fastPtr];
            while (var1 >= target) {
                int currentLen = fastPtr - slowPtr + 1;
                minLen = Math.min(minLen, currentLen);
                var1 -= nums[slowPtr++];
            }
        }
        if (minLen == Integer.MAX_VALUE)
            return 0;
        else
            return minLen;
    }
}
