class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int j = 0;
        int min = nums.length + 1;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                if (min > i - j + 1) {
                    min = i - j + 1;
                }
                var1 -= nums[j++];
            }
            if (min == 1) {
                return min;
            }
        }
        if (min == nums.length + 1) {
            return 0;
        }
        return min;
    }
}
