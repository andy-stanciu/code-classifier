class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int minL = 0;
        int start = 0;
        int end = 0;
        while (end < nums.length) {
            var1 += nums[end];
            if (var1 < target)
                end++;
            else {
                var1 = var1 - nums[start] - nums[end];
                minL = minL == 0 ? end - start + 1 : Math.min(minL, end - start + 1);
                start++;
            }
        }
        return minL;
    }
}
