class Solution {

    public int method1(int target, int[] nums) {
        int n = nums.length;
        int start = 0;
        int end = 0;
        int minLength = Integer.MAX_VALUE;
        int var1 = 0;
        for (end = 0; end < n; end++) {
            var1 += nums[end];
            while (var1 >= target && start <= end) {
                minLength = Math.min(minLength, end - start + 1);
                var1 = var1 - nums[start];
                start++;
            }
        }
        if (minLength == Integer.MAX_VALUE) {
            return 0;
        }
        return minLength;
    }
}
