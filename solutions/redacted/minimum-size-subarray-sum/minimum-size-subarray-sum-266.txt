class Solution {

    public int method1(int target, int[] nums) {
        int i = 0;
        int j = 0;
        int var1 = 0;
        int ans = Integer.MAX_VALUE;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                ans = Math.min(ans, j - i + 1);
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
