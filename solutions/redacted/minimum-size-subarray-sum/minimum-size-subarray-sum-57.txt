class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0, from = 0, ans = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                ans = Math.min(ans, i - from + 1);
                var1 -= nums[from++];
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}
