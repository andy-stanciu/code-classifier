class Solution {

    public int method1(int target, int[] nums) {
        int result = Integer.MAX_VALUE;
        int var1 = 0;
        int left = 0;
        int right = 0;
        while (left < nums.length) {
            while (right < nums.length && var1 < target) {
                var1 = var1 + nums[right];
                right++;
            }
            if (var1 >= target) {
                result = Math.min(result, right - left);
            }
            var1 = var1 - nums[left];
            left++;
            if (result == Integer.MAX_VALUE) {
                return 0;
            }
        }
        return result;
    }
}
