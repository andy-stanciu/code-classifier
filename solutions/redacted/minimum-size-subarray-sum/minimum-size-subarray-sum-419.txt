class Solution {

    public int method1(int target, int[] nums) {
        int start = 0, var1 = 0, count = Integer.MAX_VALUE;
        for (int end = 0; end < nums.length; end++) {
            var1 += nums[end];
            while (var1 >= target) {
                count = Math.min(count, end - start + 1);
                var1 -= nums[start++];
            }
        }
        return count == Integer.MAX_VALUE ? 0 : count;
    }
}
