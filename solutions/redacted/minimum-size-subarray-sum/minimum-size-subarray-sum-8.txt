// easy optimal aproch
class Solution {

    public int method1(int target, int[] arr) {
        int n = arr.length;
        int i = 0, var1 = 0, len = 0, minlen = Integer.MAX_VALUE;
        for (int j = 0; j < n; j++) {
            var1 += arr[j];
            while (var1 >= target) {
                len = j - i + 1;
                minlen = Math.min(len, minlen);
                var1 -= arr[i];
                i++;
            }
        }
        if (minlen == Integer.MAX_VALUE)
            return 0;
        return minlen;
    }
}
