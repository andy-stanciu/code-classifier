class Solution {

    public int method1(int target, int[] nums) {
        int windowStart = 0, windowEnd = 0;
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        for (windowEnd = 0; windowEnd < nums.length; windowEnd++) {
            var2 += nums[windowEnd];
            while (var2 >= target) {
                var1 = Math.min(var1, windowEnd - windowStart + 1);
                var2 -= nums[windowStart];
                windowStart++;
            }
        }
        if (var1 == Integer.MAX_VALUE) {
            return 0;
        }
        return var1;
    }
}
