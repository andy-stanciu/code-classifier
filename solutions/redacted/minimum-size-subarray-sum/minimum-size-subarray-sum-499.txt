class Solution {

    public int method1(int target, int[] nums) {
        int result = 100001, var1 = nums[0];
        if (var1 >= target)
            return 1;
        for (int i = 0, j = 0; i <= j; ) {
            if (var1 >= target) {
                var1 = var1 - nums[i];
                i++;
            } else {
                j++;
                if (j == nums.length)
                    break;
                var1 = var1 + nums[j];
            }
            if (var1 >= target)
                result = Math.min(result, j - i + 1);
        }
        return result == 100001 ? 0 : result;
    }
}
