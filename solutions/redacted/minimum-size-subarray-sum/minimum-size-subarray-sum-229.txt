class Solution {

    public int method1(int target, int[] nums) {
        int var1 = Integer.MAX_VALUE;
        int left = 0, right = 0, var2 = 0;
        while (right < nums.length) {
            var2 += nums[right++];
            while (var2 >= target) {
                var1 = Math.min(var1, right - left);
                var2 -= nums[left++];
            }
        }
        return var1 == Integer.MAX_VALUE ? 0 : var1;
    }
}
