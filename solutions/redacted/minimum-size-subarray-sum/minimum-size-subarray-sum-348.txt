class Solution {

    public int method1(int target, int[] nums) {
        int len = nums.length;
        int left = 0;
        int right = 0;
        int result = 0;
        int var1 = 0;
        while (right < len) {
            var1 += nums[right];
            while (var1 >= target) {
                int SubArrLen = right - left + 1;
                if (result == 0)
                    result = SubArrLen;
                else
                    result = result < SubArrLen ? result : SubArrLen;
                var1 -= nums[left];
                left++;
            }
            right++;
        }
        return result;
    }
}
