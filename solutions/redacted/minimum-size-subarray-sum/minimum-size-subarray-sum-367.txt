class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = 0;
        int var1 = 0;
        int var2 = Integer.MAX_VALUE;
        while (i <= j && j < nums.length) {
            if (var1 < target) {
                var1 += nums[j];
                if (var1 >= target) {
                    var2 = Math.min(var2, j - i + 1);
                }
                j++;
            } else {
                var1 -= nums[i];
                if (var1 >= target) {
                    var2 = Math.min(var2, j - i - 1);
                }
                i++;
            }
        }
        if (i < --j && var1 > target) {
            while (i < j && var1 > target) {
                var1 -= nums[i];
                i++;
                if (var1 >= target) {
                    var2 = Math.min(var2, j - i + 1);
                }
            }
        }
        return var2 != Integer.MAX_VALUE ? var2 : 0;
    }
}
