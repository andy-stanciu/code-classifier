class Solution {

    public int method1(int target, int[] nums) {
        int minlen = nums.length + 1;
        int var1 = 0;
        int l = 0;
        int r = 0;
        while (r < nums.length) {
            var1 += nums[r];
            while (var1 >= target) {
                minlen = Math.min(minlen, r - l + 1);
                var1 -= nums[l];
                l++;
            }
            r++;
        }
        return (minlen == nums.length + 1) ? 0 : minlen;
    }
}
