class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int window = Integer.MAX_VALUE;
        int st = 0;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                window = Math.min(window, i - st + 1);
                var1 -= nums[st];
                st++;
            }
        }
        return window == Integer.MAX_VALUE ? 0 : window;
    }
}
