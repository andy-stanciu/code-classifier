class Solution {

    public int method1(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int result = Integer.MAX_VALUE;
        int var1 = nums[0];
        while (true) {
            if (var1 >= target) {
                result = Math.min(result, right - left + 1);
                left++;
                var1 -= nums[left - 1];
            } else if (left >= right || var1 < target) {
                right++;
                if (right < nums.length) {
                    var1 += nums[right];
                } else {
                    break;
                }
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}
