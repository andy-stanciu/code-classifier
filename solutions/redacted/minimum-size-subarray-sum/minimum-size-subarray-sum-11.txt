class Solution {

    public int method1(int target, int[] nums) {
        int minlenWindow = Integer.MAX_VALUE;
        int var1 = 0;
        int low = 0;
        int high = 0;
        while (high < nums.length) {
            var1 += nums[high];
            high++;
            while (var1 >= target) {
                int var2 = high - low;
                minlenWindow = Math.min(minlenWindow, var2);
                var1 -= nums[low];
                low++;
            }
        }
        return minlenWindow == Integer.MAX_VALUE ? 0 : minlenWindow;
    }
}
