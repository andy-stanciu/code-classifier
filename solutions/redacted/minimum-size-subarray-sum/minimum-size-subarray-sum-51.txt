class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int left = 0;
        int n = nums.length;
        int right = 0;
        int shortest = Integer.MAX_VALUE;
        while (right < n) {
            var1 += nums[right];
            right++;
            if (var1 >= target) {
                while (var1 >= target) {
                    shortest = Math.min(shortest, right - left);
                    var1 -= nums[left];
                    left++;
                }
            }
        }
        return shortest == Integer.MAX_VALUE ? 0 : shortest;
    }
}
