class Solution {

    public int method1(int target, int[] nums) {
        int var1 = 0;
        int j = 0;
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            var1 += nums[i];
            while (var1 >= target) {
                var1 -= nums[j];
                min = Math.min(min, (i + 1) - j);
                j++;
            }
        }
        return (min == Integer.MAX_VALUE) ? 0 : min;
    }
}
