class Solution {

    private boolean checkArray(int[] nums, int target, int param1) {
        int left = 0;
        int var1 = 0;
        for (int right = 0; right < nums.length; right++) {
            var1 += nums[right];
            if (right - left + 1 == param1) {
                if (var1 >= target)
                    return true;
                var1 -= nums[left];
                left++;
            }
        }
        return false;
    }

    public int method1(int target, int[] nums) {
        int answer = Integer.MAX_VALUE;
        int l = 1, r = nums.length;
        int m;
        while (l <= r) {
            m = l + (r - l) / 2;
            if (checkArray(nums, target, m)) {
                r = m - 1;
                answer = m;
            } else
                l = m + 1;
        }
        if (answer == Integer.MAX_VALUE)
            return 0;
        return answer;
    }
}
