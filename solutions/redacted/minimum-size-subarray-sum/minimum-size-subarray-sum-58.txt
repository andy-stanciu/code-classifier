class Solution {

    public int method1(int target, int[] nums) {
        int ans = Integer.MAX_VALUE;
        int var1 = 0;
        int i = 0;
        int j = 0;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                ans = Math.min(ans, j - i + 1);
                var1 -= nums[i++];
            }
            j++;
        }
        if (ans == Integer.MAX_VALUE) {
            return 0;
        } else {
            return ans;
        }
    }
}
