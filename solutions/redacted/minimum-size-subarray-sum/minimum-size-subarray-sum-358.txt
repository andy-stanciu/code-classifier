class Solution {

    public int method1(int target, int[] nums) {
        int i = 0, j = 0, var1 = 0, minlen = 1000000;
        while (j < nums.length) {
            var1 += nums[j];
            while (var1 >= target) {
                minlen = (minlen > j - i + 1) ? (j - i + 1) : minlen;
                var1 -= nums[i];
                i++;
            }
            j++;
        }
        if (minlen == 1000000) {
            return 0;
        }
        return minlen;
    }
}
