/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> ans = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int n = q.size();
            double temp = 0.0;
            for (int i = 0; i < n; i++) {
                TreeNode top = q.poll();
                // q.pop();
                if (top.left != null)
                    q.add(top.left);
                if (top.right != null)
                    q.add(top.right);
                temp += top.val;
            }
            ans.add(temp / n);
        }
        return ans;
    }
}
