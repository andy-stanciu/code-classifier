/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> list = new ArrayList<>();
        // Queue is created to perform BFS traverse
        Queue<TreeNode> queue = new LinkedList<>();
        //Add root to the queue
        queue.offer(root);
        while (!queue.isEmpty()) {
            //To sum level wise calculate the size of queue.
            int size = queue.size();
            double var1 = 0;
            //For each level sum the node values
            for (int i = 0; i < size; i++) {
                TreeNode node = queue.poll();
                var1 += (double) node.val;
                //Add the next child nodes
                if (node.left != null) {
                    queue.offer(node.left);
                }
                if (node.right != null) {
                    queue.offer(node.right);
                }
            }
            //Add the average of levels to the list
            list.add((double) var1 / size);
        }
        return list;
    }
}
