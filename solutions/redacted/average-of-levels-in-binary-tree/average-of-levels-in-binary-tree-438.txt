/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> avg = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        q.offer(null);
        long var1 = 0;
        int nodeCount = 0;
        while (!q.isEmpty()) {
            TreeNode curr = q.poll();
            if (curr == null) {
                Double currAvg = var1 / (double) nodeCount;
                avg.add(currAvg);
                var1 = 0;
                nodeCount = 0;
                if (q.isEmpty()) {
                    break;
                } else {
                    q.offer(null);
                }
            } else {
                var1 += curr.val;
                nodeCount++;
                if (curr.left != null)
                    q.offer(curr.left);
                if (curr.right != null)
                    q.offer(curr.right);
            }
        }
        return avg;
    }
}
