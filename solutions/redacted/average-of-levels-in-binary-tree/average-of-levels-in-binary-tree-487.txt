class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        Queue<TreeNode> bfsQueue = new LinkedList<>();
        bfsQueue.add(root);
        while (!bfsQueue.isEmpty()) {
            int size = bfsQueue.size();
            int currenlevelItemCount = bfsQueue.size();
            double currentLevelTotal = 0;
            for (int i = size - 1; i >= 0; i -= 1) {
                TreeNode var1 = bfsQueue.poll();
                currentLevelTotal = currentLevelTotal + var1.val;
                if (var1.left != null) {
                    bfsQueue.add(var1.left);
                }
                if (var1.right != null) {
                    bfsQueue.add(var1.right);
                }
            }
            currentLevelTotal = (currentLevelTotal / currenlevelItemCount);
            result.add(currentLevelTotal);
        }
        return result;
    }
}
