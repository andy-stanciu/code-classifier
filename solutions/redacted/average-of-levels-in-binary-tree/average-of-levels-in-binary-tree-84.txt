/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> method1(TreeNode root) {
        // List to store the sum and count of nodes at each level
        List<Double> sums = new ArrayList<>();
        List<Integer> counts = new ArrayList<>();
        // Start the DFS traversal from the root node
        helper(root, 0, sums, counts);
        // Calculate averages for each level
        List<Double> var1 = new ArrayList<>();
        for (int i = 0; i < sums.size(); i++) {
            var1.add(sums.get(i) / counts.get(i));
        }
        return var1;
    }

    private void helper(TreeNode node, int level, List<Double> sums, List<Integer> counts) {
        if (node == null)
            return;
        // Check if the level is already in the lists
        if (level >= sums.size()) {
            // Initialize sums and counts for this level
            sums.add((double) node.val);
            counts.add(1);
        } else {
            // Update existing sums and counts for this level
            sums.set(level, sums.get(level) + node.val);
            counts.set(level, counts.get(level) + 1);
        }
        // Recursively process the left and right subtrees
        helper(node.left, level + 1, sums, counts);
        helper(node.right, level + 1, sums, counts);
    }
}
