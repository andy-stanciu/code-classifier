/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> method1(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        List<Double> ans = new ArrayList<>();
        queue.add(root);
        while (!queue.isEmpty()) {
            double l = queue.size();
            double sum = 0;
            for (int i = 0; i < l; i++) {
                TreeNode k = queue.poll();
                if (k.left != null) {
                    queue.add(k.left);
                }
                if (k.right != null) {
                    queue.add(k.right);
                }
                sum += k.val;
            }
            ans.add(sum / l);
        }
        return ans;
    }
}
