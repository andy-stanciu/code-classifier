/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> var1 = new ArrayList<>();

    public List<Double> method1(TreeNode root) {
        method2(root, 0);
        List<Double> valSum = new ArrayList<>();
        for (List<Integer> xVal : var1) {
            long sumValue = 0;
            for (int x : xVal) {
                sumValue += x;
            }
            if (xVal.size() != 0)
                valSum.add((double) sumValue / xVal.size());
        }
        return valSum;
    }

    public void method2(TreeNode root, int level) {
        if (root == null) {
            return;
        } else {
            if (level <= var1.size()) {
                var1.add(new ArrayList<>());
            }
            var1.get(level).add(root.val);
            method2(root.left, level + 1);
            method2(root.right, level + 1);
        }
    }
}
