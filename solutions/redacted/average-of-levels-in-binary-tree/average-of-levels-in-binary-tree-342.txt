/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            // How many nodes i have at current level
            int var1 = queue.size();
            double var2 = 0;
            for (int i = 0; i < var1; i++) {
                TreeNode currentNode = queue.poll();
                var2 = var2 + currentNode.val;
                // if left & right are not null. Add them
                if (currentNode.left != null) {
                    queue.offer(currentNode.left);
                }
                if (currentNode.right != null) {
                    queue.offer(currentNode.right);
                }
            }
            // AverageLevel will have sum of nodes at that level
            // Divide by total number of nodes at that level
            result.add(var2 / var1);
        }
        return result;
    }
}
