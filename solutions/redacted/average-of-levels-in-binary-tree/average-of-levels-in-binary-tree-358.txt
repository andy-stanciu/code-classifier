class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int nodesCount = 1;
        while (!q.isEmpty()) {
            double sum = 0;
            int count = 0;
            int newNodesCount = 0;
            for (int i = 0; i < nodesCount; i++) {
                TreeNode node = q.poll();
                sum += node.val;
                count++;
                if (node.left != null) {
                    q.add(node.left);
                    newNodesCount++;
                }
                if (node.right != null) {
                    q.add(node.right);
                    newNodesCount++;
                }
            }
            result.add(sum / count);
            nodesCount = newNodesCount;
        }
        return result;
    }
}
