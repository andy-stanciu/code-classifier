class Solution {

    public List<Double> method1(TreeNode root) {
        List<Double> result = new ArrayList<>();
        if (root == null)
            return result;
        util(root, result);
        return result;
    }

    public void util(TreeNode root, List<Double> result) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int n = q.size();
            int elements = n;
            double sum = 0;
            while (n-- != 0) {
                TreeNode node = q.poll();
                sum += node.val;
                if (node.left != null)
                    q.add(node.left);
                if (node.right != null)
                    q.add(node.right);
            }
            result.add(sum / elements);
        }
    }
}
