class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        paranthesis(n, 0, 0, result, "");
        return result;
    }

    private void paranthesis(int n, int index, int index2, List<String> s, String str) {
        if (str.length() == 2 * n) {
            s.add(str);
            return;
        }
        if (index < n) {
            paranthesis(n, index + 1, index2, s, str + "(");
        }
        if (index2 < index) {
            paranthesis(n, index, index2 + 1, s, str + ")");
        }
    }
}
