class Solution {

    public List<String> method1(int n) {
        ArrayList<String> ans = new ArrayList<>();
        recur(n, 0, ans, "");
        return ans;
    }

    public void recur(int open, int close, ArrayList<String> ans, String curr) {
        if (open == 0 && close == 0) {
            ans.add(curr);
            // curr = "";
        }
        if (open > 0) {
            recur(open - 1, close + 1, ans, curr + "(");
        }
        if (close > 0) {
            recur(open, close - 1, ans, curr + ")");
        }
    }
}
