class Solution {

    public List<String> method1(int n) {
        List<String> list = new ArrayList<>();
        f(n, 0, "", list);
        return list;
    }

    private static void f(int o, int c, String s, List<String> list) {
        if (o == 0 && c == 0)
            list.add(s);
        if (o > 0)
            f(o - 1, c + 1, s + "(", list);
        if (c > 0)
            f(o, c - 1, s + ")", list);
    }
}
