class Solution {

    public List<String> method1(int n) {
        List<String> results = new ArrayList();
        addPossibilities(results, 1, 0, "(", n);
        return results;
    }

    private void addPossibilities(List<String> results, int openCount, int closedCount, String current, int maxCount) {
        if (openCount == maxCount && closedCount == maxCount) {
            results.add(current);
            return;
        }
        if (closedCount < openCount) {
            addPossibilities(results, openCount, closedCount + 1, current + ")", maxCount);
        }
        if (openCount < maxCount) {
            addPossibilities(results, openCount + 1, closedCount, current + "(", maxCount);
        }
    }
}
