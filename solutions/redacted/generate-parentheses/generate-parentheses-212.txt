class Solution {

    List<String> ans;

    public List<String> method1(int n) {
        ans = new ArrayList<>();
        helper(n, 0, 0, "");
        return ans;
    }

    public void helper(int n, int open, int closed, String str) {
        if (open == n) {
            while (closed-- != 0) str += ")";
            ans.add(str);
            return;
        }
        if (closed != 0)
            helper(n, open, closed - 1, str + ")");
        helper(n, open + 1, closed + 1, str + "(");
    }
}
