class Solution {

    List<String> res = new ArrayList<>();

    public List<String> method1(int n) {
        int leftCount = 0, rightCount = 0;
        backtrack(n, leftCount, rightCount, new StringBuilder());
        return res;
    }

    public void backtrack(int n, int leftCount, int rightCount, StringBuilder sb) {
        if (n == rightCount && rightCount == leftCount) {
            res.add(sb.toString());
            return;
        }
        if (leftCount < n) {
            sb.append("(");
            backtrack(n, leftCount + 1, rightCount, sb);
            sb.deleteCharAt(sb.length() - 1);
        }
        if (rightCount < leftCount) {
            sb.append(")");
            backtrack(n, leftCount, rightCount + 1, sb);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
