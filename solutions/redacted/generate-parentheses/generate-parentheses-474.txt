class Solution {

    public List<String> method1(int n) {
        List<String> al = new ArrayList<String>();
        int open = 0;
        int closed = 0;
        String s = "";
        method2(al, s, open, closed, n);
        return al;
    }

    public void method2(List<String> al, String ans, int open, int closed, int n) {
        if (open == n && closed == n) {
            al.add(ans);
            return;
        }
        if (open < n) {
            ans += "(";
            open++;
            method2(al, ans + "(", open + 1, closed, n);
        }
        if (closed < open) {
            ans += ")";
            closed++;
            method2(al, ans + ")", open, closed + 1, n);
        }
    }
}
