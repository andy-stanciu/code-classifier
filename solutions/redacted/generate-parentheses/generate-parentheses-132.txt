class Solution {

    public List<String> method1(int n) {
        //  Set up helper method
        List<String> result = new ArrayList<>();
        backtrack(result, "", 0, 0, n);
        return result;
    }

    private void backtrack(List<String> result, String current, int openParen, int closeParen, int n) {
        //  Base case, add current string if all parenthesis used up
        if (closeParen == n) {
            result.add(current);
            return;
        }
        //  Add "(" if the limit for "(" hasnt been reached yet
        if (openParen < n)
            //  Add "(", update the count, and recur
            backtrack(result, current + "(", openParen + 1, closeParen, n);
        //  Add ")" if the limit for ")" hasnt been reached yet and there arent more ")" than "("
        if (closeParen < openParen)
            //  Add ")", update the count, and recur
            backtrack(result, current + ")", openParen, closeParen + 1, n);
    }
}
