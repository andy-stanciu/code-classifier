class Solution {

    public List<String> method1(int n) {
        List<String> ans = new ArrayList<>();
        String str = "";
        method2(ans, new StringBuilder(), 0, 0, n);
        return ans;
    }

    private static void method2(List<String> ans, StringBuilder str, int open, int close, int max) {
        if (open == max && close == max) {
            ans.add(str.toString());
            return;
        }
        if (open < max) {
            str.append("(");
            method2(ans, str, open + 1, close, max);
            str.deleteCharAt(str.length() - 1);
        }
        if (close < open) {
            str.append(")");
            method2(ans, str, open, close + 1, max);
            str.deleteCharAt(str.length() - 1);
        }
    }
}
