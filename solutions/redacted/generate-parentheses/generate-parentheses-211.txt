class Solution {

    List<String> res;

    private void helper(int n, int ocnt, int ccnt, String gen) {
        if (ccnt == n) {
            res.add(gen);
            return;
        }
        if (ocnt < n)
            helper(n, ocnt + 1, ccnt, gen + "(");
        if (ccnt < ocnt)
            helper(n, ocnt, ccnt + 1, gen + ")");
    }

    public List<String> method1(int n) {
        res = new ArrayList<>();
        helper(n, 0, 0, "");
        return res;
    }
}
