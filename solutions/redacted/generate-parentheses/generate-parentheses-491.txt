class Solution {

    ArrayList<String> ans = new ArrayList<>();

    int n;

    void method1(String s, int o, int c) {
        if (o > n || c > n)
            return;
        if ((o + c) == n * 2) {
            ans.add(s);
            return;
        }
        method1(s + "(", o + 1, c);
        if (c < o)
            method1(s + ")", o, c + 1);
    }

    public List<String> method2(int n) {
        this.n = n;
        method1("", 0, 0);
        return ans;
    }
}
