class Solution {

    public void solve(int n, List<String> sol, int i, int j, String res) {
        if (res.length() == n * 2) {
            sol.add(res);
            return;
        }
        if (i < n) {
            solve(n, sol, i + 1, j, res + "(");
        }
        if (j < i) {
            solve(n, sol, i, j + 1, res + ")");
        }
    }

    public List<String> method1(int n) {
        List<String> sol = new ArrayList<>();
        solve(n, sol, 0, 0, "");
        return sol;
    }
}
