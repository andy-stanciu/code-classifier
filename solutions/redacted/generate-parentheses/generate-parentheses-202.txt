class Solution {

    static List<String> ans;

    public void print(int o, int c, int n, String s) {
        if (s.length() == 2 * n) {
            ans.add(s);
            return;
        }
        if (o < n)
            print(o + 1, c, n, s + "(");
        if (c < o)
            print(o, c + 1, n, s + ")");
    }

    public List<String> method1(int n) {
        ans = new ArrayList<>();
        print(0, 0, n, "");
        return ans;
    }
}
