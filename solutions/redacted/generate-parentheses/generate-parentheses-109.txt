class Solution {

    public List<String> method1(int n) {
        List<String> ans = new ArrayList<>();
        back(n, "", ans, 0, 0);
        return ans;
    }

    public void back(int n, String temp, List<String> ans, int open, int close) {
        if (temp.length() == 2 * n) {
            ans.add(temp);
            return;
        }
        if (open < n)
            back(n, temp + "(", ans, open + 1, close);
        if (close < open)
            back(n, temp + ")", ans, open, close + 1);
    }
}
