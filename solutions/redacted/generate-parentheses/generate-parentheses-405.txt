class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        f(n, 0, 0, "", result);
        return result;
    }

    private static void f(int i, int o, int c, String s, List<String> result) {
        if (c == i) {
            result.add(s);
            return;
        }
        if (o > c)
            f(i, o, c + 1, s + ")", result);
        if (i > o)
            f(i, o + 1, c, s + "(", result);
    }
}
