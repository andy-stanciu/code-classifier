class Solution {

    public List<String> method1(int n) {
        return dfs(0, 0, n, "");
    }

    List<String> dfs(int l, int r, int n, String s) {
        if (l == n && r == n)
            return List.of(s);
        List<String> res = new ArrayList<>();
        if (l < n)
            res.addAll(dfs(l + 1, r, n, s + "("));
        if (r < l)
            res.addAll(dfs(l, r + 1, n, s + ")"));
        return res;
    }
}
