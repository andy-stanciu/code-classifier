final class Solution {

    private final List<String> res = new ArrayList<>();

    public List<String> method1(final int n) {
        method2(0, 0, "", n);
        return res;
    }

    private void method2(final int open, final int closed, final String path, final int n) {
        if (open == closed && open == n) {
            res.add(path);
            return;
        }
        if (open < n)
            method2(open + 1, closed, path + "(", n);
        if (closed < open)
            method2(open, closed + 1, path + ")", n);
    }
}
