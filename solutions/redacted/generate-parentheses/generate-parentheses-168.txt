class Solution {

    List<String> list = new ArrayList<>();

    public void genPar(String str, int openCount, int closeCount, int n) {
        if (openCount == closeCount && openCount == n) {
            list.add(str);
            return;
        }
        if (closeCount < openCount) {
            genPar(str + ")", openCount, closeCount + 1, n);
        }
        if (closeCount < n && openCount < n) {
            genPar(str + "(", openCount + 1, closeCount, n);
        }
    }

    public List<String> method1(int n) {
        genPar("(", 1, 0, n);
        return list;
    }
}
