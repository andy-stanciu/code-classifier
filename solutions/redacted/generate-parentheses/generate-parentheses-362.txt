class Solution {

    public static List<String> gp(String p, int open, int close, ArrayList<String> list) {
        if (open == 0 && close == 0) {
            list.add(p);
            return list;
        }
        if (open != 0) {
            gp(p + "(", open - 1, close, list);
        }
        if (close > open) {
            gp(p + ")", open, close - 1, list);
        }
        return list;
    }

    public List<String> method1(int n) {
        return gp("", n, n, new ArrayList<>());
    }
}
