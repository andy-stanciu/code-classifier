class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        backtrack(result, "", 0, 0, n);
        return result;
    }

    // n represents the number of pairs of parentheses you need to generate.
    // For each pair of parentheses, there are two characters: an opening parenthesis \'(\' and a closing parenthesis \')\'.
    // Therefore, for n pairs, you will have a total of 2 * n characters.
    public void backtrack(List<String> result, String temp, int open, int close, int n) {
        // base case
        if (temp.length() == n * 2) {
            result.add(temp);
            return;
        }
        if (open < n)
            backtrack(result, temp + "(", open + 1, close, n);
        if (close < open)
            backtrack(result, temp + ")", open, close + 1, n);
    }
}
