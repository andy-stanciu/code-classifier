class Solution {

    public List<String> method1(int n) {
        List<String> list = new ArrayList<>();
        StringBuilder tempString = new StringBuilder();
        method2(list, tempString, 0, 0, n);
        return list;
    }

    public void method2(List<String> list, StringBuilder tempString, int i, int j, int n) {
        if (tempString.length() == n * 2) {
            list.add(tempString.toString());
            return;
        }
        if (i < n) {
            tempString.append("(");
            method2(list, tempString, i + 1, j, n);
            tempString.deleteCharAt(tempString.length() - 1);
        }
        if (j < i) {
            tempString.append(")");
            method2(list, tempString, i, j + 1, n);
            tempString.deleteCharAt(tempString.length() - 1);
        }
    }
}
