class Solution {

    public List<String> method1(int n) {
        List<String> result = new ArrayList<>();
        method1(n, 0, 0, "", result);
        return result;
    }

    public void method1(int n, int openBrackets, int closedBrackets, String ans, List<String> result) {
        if (openBrackets == n && closedBrackets == openBrackets) {
            result.add(ans);
        } else {
            if (openBrackets < n) {
                method1(n, openBrackets + 1, closedBrackets, ans + "(", result);
            }
            if (closedBrackets < openBrackets) {
                method1(n, openBrackets, closedBrackets + 1, ans + ")", result);
            }
        }
    }
}
