class Solution {

    public List<String> method1(int n) {
        HashSet<String> result = new HashSet<>();
        method2(n, 0, 0, "", result);
        ArrayList<String> output = new ArrayList<>();
        output.addAll(result);
        return output;
    }

    private void method2(int n, int open, int close, String p, Set<String> result) {
        if (open + close == 2 * n) {
            result.add(p);
            return;
        }
        if (open < n) {
            method2(n, open + 1, close, p + "(", result);
        }
        if (close < open) {
            method2(n, open, close + 1, p + ")", result);
        }
    }
}
