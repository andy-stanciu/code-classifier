class Solution {

    Set<String> uniquePattern = new HashSet<>();

    List<String> retList = new ArrayList<>();

    public List<String> method1(int n) {
        if (n != 0)
            this.method2(n, n, "");
        return retList;
    }

    public void method2(int openCount, int closeCount, String currPattern) {
        if (openCount == 0 && closeCount == 0) {
            if (uniquePattern.add(currPattern))
                retList.add(currPattern);
            return;
        }
        if (openCount == closeCount)
            this.method2(openCount - 1, closeCount, currPattern + "(");
        else {
            if (openCount > 0)
                this.method2(openCount - 1, closeCount, currPattern + "(");
            this.method2(openCount, closeCount - 1, currPattern + ")");
        }
    }
}
