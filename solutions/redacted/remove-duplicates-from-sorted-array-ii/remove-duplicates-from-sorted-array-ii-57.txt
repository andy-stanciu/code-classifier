class Solution {

    public int method1(int[] nums) {
        Map<Integer, Integer> map = new TreeMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        int i = 0;
        for (int key : map.keySet()) {
            nums[i] = key;
            if (map.get(key) < 2)
                i++;
            else {
                nums[i + 1] = key;
                i += 2;
            }
        }
        return i;
    }
}
// TC: O(n * logn), SC: O(n)
