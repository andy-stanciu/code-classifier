class Solution {

    public int method1(int[] nums) {
        int i, j, temp, l, index = 0, k = 0, m = 0;
        l = nums.length;
        int[] res = new int[l];
        for (i = 0; i < l; i++) {
            temp = nums[m];
            for (j = m + 1; j < l; j++) {
                if (nums[j] != temp) {
                    res[index] = nums[j - 1];
                    index++;
                    if (j > 1 && nums[j - 1] == nums[j - 2]) {
                        res[index] = nums[j - 2];
                        index++;
                    }
                    m = j;
                    j = l;
                } else {
                    m = j;
                }
            }
            if (m == l - 1) {
                if (m > 0 && nums[m] == nums[m - 1]) {
                    res[index] = nums[m];
                    index++;
                    res[index] = nums[m];
                    break;
                } else {
                    res[index] = nums[m];
                    break;
                }
            }
        }
        for (i = 0; i <= index; i++) {
            nums[i] = res[i];
        }
        return index + 1;
    }
}
