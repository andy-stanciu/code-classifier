class Solution {

    public int method1(int[] nums) {
        int index = 1, cnt = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[index++] = nums[i];
                cnt = 0;
                //when cnt = 0 means there is already 1 number in the array, so we need to check whether cnt < 1
            } else if (nums[i] == nums[i - 1] && cnt < 1) {
                nums[index++] = nums[i];
                cnt++;
            }
        }
        return index;
    }
}
