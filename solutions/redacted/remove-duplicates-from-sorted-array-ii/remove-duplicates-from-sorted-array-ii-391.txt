class Solution {

    public int method1(int[] nums) {
        //to leave 2 common elements in array we are stating with 2
        int index = 2;
        for (int i = 2; i < nums.length; i++) {
            //we are assigning itself because we want only the existing elts matches condition
            if (nums[i] != nums[index - 2]) {
                nums[index] = nums[i];
                index++;
            }
        }
        return index;
    }
}
