class Solution {

    public static int method1(int[] nums) {
        HashMap<Integer, Integer> map = new HashMap<>();
        // Creating HashMap of nums
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        // Updating hashmap value == 2 as per quetion demand
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            if (entry.getValue() > 2) {
                entry.setValue(2);
            } else {
                continue;
            }
        }
        // Storing key upto value times in list called ans.
        ArrayList<Integer> ans = new ArrayList<>();
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            int key = entry.getKey();
            int value = entry.getValue();
            for (int i = 0; i < value; i++) {
                ans.add(key);
            }
        }
        // Sorting the list because of some test case are in negative
        Collections.sort(ans);
        // Updating given array nums as per que demand
        for (int i = 0; i < ans.size(); i++) {
            nums[i] = ans.get(i);
        }
        // returning the size of array.
        return ans.size();
    }
}
