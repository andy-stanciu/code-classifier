class Solution {

    public int method1(int[] nums) {
        int index = 0;
        int atMost = 2;
        int currentSeq = 0;
        for (int i = 0; i < nums.length; ) {
            while (i + currentSeq < nums.length && nums[i] == nums[i + currentSeq]) {
                nums[index + currentSeq] = nums[i];
                currentSeq++;
            }
            index += Math.min(currentSeq, atMost);
            i += currentSeq;
            currentSeq = 0;
        }
        return index;
    }
}
