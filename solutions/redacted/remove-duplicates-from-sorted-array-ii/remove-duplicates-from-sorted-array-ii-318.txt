class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int pos = 0;
        for (int i = 0; i < nums.length; i++) {
            if (pos < 2 || nums[i] != nums[pos - 2]) {
                nums[pos] = nums[i];
                pos++;
            }
        }
        return pos;
    }
}
