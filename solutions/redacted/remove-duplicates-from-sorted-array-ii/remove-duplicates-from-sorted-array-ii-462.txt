class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return 1;
        int a = nums[0];
        int b = nums[1];
        int index = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != a || nums[i] != b) {
                nums[index] = nums[i];
                a = b;
                b = nums[i];
                index++;
            }
        }
        return index;
    }
}
