class Solution {

    public int method1(int[] nums) {
        boolean isTwice = false;
        // keep length of new array
        int index = 0;
        // keep number len
        int realLen = nums.length;
        for (int i = 0; i + 1 < realLen; i++) {
            if (nums[i] == nums[i + 1] && !isTwice) {
                index++;
                isTwice = true;
            } else if (nums[i] == nums[i + 1] && isTwice) {
                //move array to left by one and reduce len
                while (i + 1 < realLen && nums[i] == nums[i + 1]) {
                    for (int j = i; j + 1 < realLen; j++) {
                        nums[j] = nums[j + 1];
                    }
                    realLen--;
                    index++;
                }
                isTwice = false;
            } else if (nums[i] != nums[i + 1]) {
                index++;
                isTwice = false;
            }
        }
        return realLen;
    }
}
