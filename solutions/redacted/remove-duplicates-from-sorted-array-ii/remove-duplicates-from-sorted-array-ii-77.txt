class Solution {

    public int method1(int[] nums) {
        int count = 0;
        ArrayList<Integer> list = new ArrayList<>();
        for (int num : nums) {
            if (list.contains(num)) {
                count++;
                if (count < 2)
                    list.add(num);
            } else if (!list.contains(num)) {
                count = 0;
                list.add(num);
            }
        }
        for (int i = 0; i < list.size(); i++) {
            nums[i] = list.get(i);
        }
        return list.size();
    }
}
