class Solution {

    public int method1(int[] nums) {
        if (nums.length <= 2)
            return nums.length;
        // Start from the third element since the first two are always allowed
        int i = 2;
        for (int j = 2; j < nums.length; j++) {
            if (nums[j] != nums[i - 2]) {
                nums[i] = nums[j];
                i++;
            }
        }
        return i;
    }
}
