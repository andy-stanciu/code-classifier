class Solution {

    public int method1(int[] nums) {
        int k = 1, m = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                m++;
                if (m <= 1)
                    nums[k++] = nums[i];
            } else {
                m = 0;
                nums[k++] = nums[i];
            }
        }
        return k;
    }
}
