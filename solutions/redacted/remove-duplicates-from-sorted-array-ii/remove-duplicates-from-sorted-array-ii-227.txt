class Solution {

    public int method1(int[] nums) {
        int idx = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[idx] == nums[i]) {
                if (idx == 0 || nums[idx - 1] != nums[i]) {
                    nums[++idx] = nums[i];
                }
            } else {
                nums[++idx] = nums[i];
            }
        }
        return idx + 1;
    }
}
