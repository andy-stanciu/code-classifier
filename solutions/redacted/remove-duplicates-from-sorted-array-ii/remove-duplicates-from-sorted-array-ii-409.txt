class Solution {

    public int method1(int[] nums) {
        int rd = 0;
        int freak = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[rd] != nums[i]) {
                rd = rd + 1;
                nums[rd] = nums[i];
                freak = 1;
            } else if (freak == 1) {
                rd = rd + 1;
                nums[rd] = nums[i];
                freak++;
            }
        }
        return rd + 1;
    }
}
