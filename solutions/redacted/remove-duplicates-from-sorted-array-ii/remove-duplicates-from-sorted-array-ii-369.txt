class Solution {

    public int method1(int[] nums) {
        int left = 0, right = 0;
        while (right < nums.length) {
            int count = 1;
            while (right + 1 < nums.length && nums[right] == nums[right + 1]) {
                right++;
                count++;
            }
            for (int i = 0; i < Math.min(count, 2); i++) {
                nums[left++] = nums[right];
            }
            right++;
        }
        return left;
    }
}
