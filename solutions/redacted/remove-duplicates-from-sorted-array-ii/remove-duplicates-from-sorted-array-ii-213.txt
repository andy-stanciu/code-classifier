class Solution {

    public int method1(int[] nums) {
        int insertionPointer = 0;
        for (int i = 0; i < nums.length; i++) {
            boolean multipleFlag = false;
            while (i < nums.length - 1 && nums[i] == nums[i + 1]) {
                i++;
                multipleFlag = true;
            }
            if (multipleFlag) {
                nums[insertionPointer++] = nums[i];
                nums[insertionPointer++] = nums[i];
            } else {
                nums[insertionPointer++] = nums[i];
            }
        }
        return insertionPointer;
    }
}
