class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        // Start from the second element
        int count = 1;
        // Count duplicates of current element
        int duplicateCount = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[count - 1]) {
                if (duplicateCount < 1) {
                    nums[count++] = nums[i];
                    duplicateCount++;
                }
                // If duplicateCount is 1 or more, skip this element
            } else {
                nums[count++] = nums[i];
                // Reset duplicate count for new element
                duplicateCount = 0;
            }
        }
        return count;
    }
}
