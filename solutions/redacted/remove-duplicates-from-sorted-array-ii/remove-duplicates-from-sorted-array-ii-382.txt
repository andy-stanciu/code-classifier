class Solution {

    public int method1(int[] nums) {
        int old = nums[0], count = 0;
        int index = 0;
        for (int num : nums) {
            if (num != old) {
                old = num;
                count = 1;
                nums[index++] = old;
            } else {
                count++;
                if (count <= 2) {
                    nums[index] = old;
                    index++;
                }
            }
        }
        return index;
    }
}
