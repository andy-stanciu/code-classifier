class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int k = 0;
        for (int j = 1; j < nums.length; j++) {
            if ((nums[i] == nums[j]) && (k == 0)) {
                nums[i + 1] = nums[j];
                i += 1;
                k += 1;
            }
            if (nums[i] != nums[j]) {
                nums[i + 1] = nums[j];
                i += 1;
                k = 0;
            }
        }
        return i + 1;
    }
}
