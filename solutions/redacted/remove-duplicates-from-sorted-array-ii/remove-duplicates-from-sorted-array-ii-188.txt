class Solution {

    public int method1(int[] nums) {
        int result = 1;
        int check = nums[0];
        int count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == check) {
                if (count < 2) {
                    nums[result] = nums[i];
                    count++;
                    result++;
                }
            } else {
                check = nums[i];
                nums[result] = nums[i];
                count = 1;
                result++;
            }
        }
        return result;
    }
}
