class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int j = 1;
        int count = 1;
        while (j <= nums.length - 1) {
            if (nums[i] == nums[j] && count < 2) {
                nums[++i] = nums[j];
                j++;
                count++;
            } else if (nums[i] != nums[j]) {
                nums[++i] = nums[j];
                j++;
                count = 1;
            } else {
                j++;
                count++;
            }
        }
        return i + 1;
    }
}
