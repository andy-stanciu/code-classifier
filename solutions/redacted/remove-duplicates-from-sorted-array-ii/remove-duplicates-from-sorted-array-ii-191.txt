class Solution {

    public int method1(int[] nums) {
        int k = 1;
        int kRep = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[k++] = nums[i];
                if (++kRep >= 2)
                    kRep = 1;
            } else if (kRep < 2) {
                nums[k++] = nums[i];
                kRep++;
            }
        }
        return k;
    }
}
