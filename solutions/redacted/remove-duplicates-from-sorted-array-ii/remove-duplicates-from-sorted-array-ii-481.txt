class Solution {

    public int method1(int[] nums) {
        if (nums.length <= 2) {
            // No duplicates can be removed if length is 0 or 1 or 2
            return nums.length;
        }
        // k will track the position to overwrite duplicate elements
        int k = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[k - 2]) {
                nums[k++] = nums[i];
            }
        }
        // k is the length of the modified array
        return k;
    }
}
