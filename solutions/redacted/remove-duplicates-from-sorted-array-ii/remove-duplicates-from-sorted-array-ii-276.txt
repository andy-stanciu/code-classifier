class Solution {

    public int method1(int[] nums) {
        if (nums.length <= 2)
            return nums.length;
        // Start from the third element
        int k = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[k - 2]) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
