class Solution {

    public int method1(int[] nums) {
        if (nums.length <= 2)
            return nums.length;
        int x = 2;
        for (int y = 2; y < nums.length; y++) {
            if (nums[y] != nums[x - 2]) {
                nums[x] = nums[y];
                x++;
            }
        }
        return x;
    }
}
