class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int k = 2;
        //Handling the edge cases of inputs
        if (nums.length >= 30000) {
            throw new IllegalArgumentException("The array is either too small or too large in length");
        }
        if (nums.length == 0) {
            return 0;
        }
        // Edge case: if the array has fewer than 2 elements, return its length
        if (nums.length < 2) {
            return nums.length;
        }
        if (-10000 >= nums[i] || nums[i] >= 10000) {
            throw new IllegalArgumentException("The values are invalid inside array nums");
        }
        //Repetitive element can be repeated at most twice, not more than that
        for (i = 2; i < nums.length; i++) {
            if (nums[i] != nums[k - 2]) {
                nums[k] = nums[i];
                k++;
            }
        }
        //prints output array and length of array k
        System.out.print(k + "is the length of the expected output new array nums");
        for (i = 0; i < k; i++) {
            System.out.print(nums[i] + " ");
        }
        return k;
    }
}
