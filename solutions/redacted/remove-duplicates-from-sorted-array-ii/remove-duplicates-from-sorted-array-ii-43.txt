class Solution {

    public int method1(int[] nums) {
        // If the array length is less than or equal to 2, return its length as is
        if (nums.length <= 2) {
            return nums.length;
        }
        // Initialize the index for the next unique element to be placed
        int ans = 2;
        // Start from the third element in the array
        for (int i = 2; i < nums.length; i++) {
            // If the current element is not equal to the element two positions before,
            // it means we can place it in the array
            if (nums[i] != nums[ans - 2]) {
                nums[ans] = nums[i];
                ans++;
            }
        }
        return ans;
    }
}
