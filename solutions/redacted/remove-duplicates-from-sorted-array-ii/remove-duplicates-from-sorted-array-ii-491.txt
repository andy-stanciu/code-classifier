class Solution {

    public int method1(int[] nums) {
        int k = 1;
        ArrayList<Integer> list = new ArrayList<>();
        int temp = nums[0];
        int j = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == temp) {
                if (k > 2)
                    continue;
                else {
                    k++;
                }
            } else {
                temp = nums[i];
                k = 1;
            }
            if (k <= 2) {
                nums[j++] = nums[i];
            }
        }
        return j;
    }
}
