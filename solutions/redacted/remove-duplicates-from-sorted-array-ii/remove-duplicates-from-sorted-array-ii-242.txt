class Solution {

    public int method1(int[] arr) {
        if (arr.length <= 2)
            return arr.length;
        int k = 2;
        for (int i = 2; i < arr.length; i++) {
            if (arr[k - 2] != arr[i]) {
                arr[k] = arr[i];
                k++;
            }
        }
        return k;
    }
}
