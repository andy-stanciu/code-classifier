class Solution {

    public int method1(int[] nums) {
        // If array has 2 or fewer elements, no need to process further
        if (nums.length <= 2) {
            return nums.length;
        }
        // Initialize write index starting from the 3rd element (index 2)
        int writeIndex = 2;
        // Loop through the array starting from the 3rd element (index 2)
        for (int i = 2; i < nums.length; i++) {
            // If the current element is not the same as the element two positions back
            if (nums[i] != nums[writeIndex - 2]) {
                // Copy the current element to the write index and move the write index forward
                nums[writeIndex] = nums[i];
                writeIndex++;
            }
        }
        // The writeIndex is now the length of the modified array
        return writeIndex;
    }
}
