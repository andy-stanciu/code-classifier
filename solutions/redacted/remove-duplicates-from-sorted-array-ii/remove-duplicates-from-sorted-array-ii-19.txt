class Solution {

    public int method1(int[] nums) {
        //index is 1 as 0th element is always unique
        int idx = 1;
        //for the duplicate count
        int dup = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                //if previous element equals the current element
                //so increase the dup count.
                dup++;
            } else {
                //if not duplicate then make the count of dup again 1
                dup = 1;
                //as we got another unique value
            }
            if (dup < 3) {
                //Since we need only 2 same values can come together so dup count should be less than 3
                //if dup is less than 3 just update the values of the array
                nums[idx] = nums[i];
                idx++;
            }
        }
        //it will return the finall array having atmost 2 duplicates of each elements
        return idx;
    }
}
