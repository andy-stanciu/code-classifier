class Solution {

    public int method1(int[] nums) {
        // Initialize \'left\' to 2 because the first two elements are always allowed.
        int left = 2, right;
        // Iterate from the third element (index 2).
        for (right = 2; right < nums.length; right++) {
            // Check if the current element is the same as the element one position to the left.
            if (nums[left - 1] == nums[right]) {
                // If it\'s the third occurrence of the element, continue to the next element to skip it.
                if (nums[left - 1] == nums[left - 2])
                    continue;
                // If it\'s not the third occurrence, store it in the \'left\' position and increment \'left.\'
                nums[left++] = nums[right];
                continue;
            }
            // If the current element is different from the element at \'left - 1\', store it in the \'left\' position and increment \'left.\'
            if (nums[left - 1] != nums[right])
                nums[left++] = nums[right];
        }
        // \'left\' now holds the count of unique elements, and the first \'left\' elements in the \'nums\' array are the modified array.
        return left;
    }
}
