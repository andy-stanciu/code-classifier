class Solution {

    public int method1(int[] nums) {
        // Pointer for the position to place the next element
        int i = 0;
        for (int n : nums) {
            if (i < 2 || n > nums[i - 2]) {
                // Place the current element and increment i
                nums[i++] = n;
            }
        }
        // Return the length of the modified array
        return i;
    }
}
