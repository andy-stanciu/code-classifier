class Solution {

    public int method1(int[] nums) {
        if (nums.length == 2)
            return 2;
        int i = 1;
        for (int j = 2; j < nums.length; j++) {
            if (nums[j] != nums[i - 1]) {
                nums[++i] = nums[j];
            }
        }
        return i + 1;
    }
}
