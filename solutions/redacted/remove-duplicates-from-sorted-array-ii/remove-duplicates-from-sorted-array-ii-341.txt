class Solution {

    public int method1(int[] nums) {
        if (nums.length == 0)
            return 0;
        int total = 0;
        int count = 1;
        for (int i = 1; i < nums.length; ++i) {
            if (nums[i] == nums[total]) {
                count++;
            } else {
                count = 1;
            }
            if (count <= 2) {
                nums[++total] = nums[i];
            }
        }
        return total + 1;
    }
}
