class Solution {

    public int method1(int[] nums) {
        int pos = 1, count = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1]) {
                if (count < 2) {
                    nums[pos++] = nums[i];
                }
                count++;
            } else {
                nums[pos++] = nums[i];
                count = 1;
            }
        }
        return pos;
    }
}
