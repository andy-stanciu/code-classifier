class Solution {

    public int method1(int[] nums) {
        if (nums.length <= 2)
            return nums.length;
        int k = 0;
        for (int i = 0; i < nums.length - 2; ++i) if (nums[i] != nums[i + 2])
            nums[k++] = nums[i];
        nums[k++] = nums[nums.length - 2];
        nums[k++] = nums[nums.length - 1];
        return k;
    }
}
