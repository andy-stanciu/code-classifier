class Solution {

    public int method1(int[] nums) {
        //not need to cheak for 1st 2 element
        int k = 2;
        //checking for remaning element
        for (int i = 2; i < nums.length; i++) {
            if (nums[i] != nums[k - 2]) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
