class Solution {

    public int method1(int[] nums) {
        int i = 1, n = nums.length;
        boolean flag = false;
        for (int j = 1; j < n; j++) {
            if (nums[j] == nums[i - 1] && !flag) {
                nums[i++] = nums[j];
                flag = true;
            } else if (nums[j] == nums[i - 1] && flag) {
                continue;
            } else if (nums[j] > nums[i - 1]) {
                nums[i++] = nums[j];
                flag = false;
            }
        }
        return i;
    }
}
