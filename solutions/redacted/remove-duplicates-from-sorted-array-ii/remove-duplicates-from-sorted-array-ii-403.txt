class Solution {

    public int method1(int[] nums) {
        //Set the pointer & occurnence  to 1 since all unique elements have to
        //have at least one occurence
        int pointer = 1;
        int uniqueCount = 1;
        for (int i = 1; i < nums.length; i++) {
            //increase the unique character count to 2 if
            //previous value is the same as current value
            if (nums[i] == nums[i - 1]) {
                uniqueCount++;
            } else {
                //If previous value not the same as current
                //value, reset unique character count to 1
                //since character count cannot be more than 2
                if (nums[i] != nums[i - 1]) {
                    uniqueCount = 1;
                }
            }
            //if the unique character count is not more than 2,
            // make the value at the current index remain the same
            if (uniqueCount <= 2) {
                nums[pointer] = nums[i];
                pointer++;
            }
        }
        //Return the amount of times the value remained the same
        return pointer;
    }
}
