class Solution {

    public int method1(int[] nums) {
        int k = 1;
        int numOccur = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[k] = nums[i];
                k++;
                numOccur = 0;
            } else {
                numOccur++;
            }
            if (numOccur == 1) {
                nums[k] = nums[i];
                k++;
            }
        }
        return k;
    }
}
