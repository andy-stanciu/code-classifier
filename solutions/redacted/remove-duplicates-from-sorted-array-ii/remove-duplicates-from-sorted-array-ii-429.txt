class Solution {

    public int method1(int[] nums) {
        int index = 2;
        for (int i = 2; i < nums.length; i++) {
            if (nums[index - 1] != nums[index - 2] || nums[i] != nums[index - 1]) {
                nums[index++] = nums[i];
            }
        }
        return index;
    }
}
