class Solution {

    public int method1(int[] nums) {
        int dup = 1;
        int index = 1;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] == nums[i - 1])
                dup++;
            else
                dup = 1;
            if (dup <= 2) {
                nums[index] = nums[i];
                index++;
            }
        }
        return index;
    }
}
