class Solution {

    public int method1(int[] nums) {
        int i = 0;
        for (int n : nums) {
            //for each element of array nums(we can write this as for(int n:nums))
            if (i < 2 || n != nums[i - 2]) {
                //we check here that if i<2 or n is not equal to i-2 means previous\'s previous position is not same then we can assign it to the array and increment the i;
                nums[i++] = n;
            }
        }
        return i;
    }
}
