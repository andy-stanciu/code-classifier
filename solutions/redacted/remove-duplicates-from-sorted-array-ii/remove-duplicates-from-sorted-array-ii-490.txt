class Solution {

    public int method1(int[] nums) {
        if (nums.length < 2) {
            return nums.length;
        }
        int prev = 1;
        int curr = 2;
        //Traverse all elements through loop
        while (curr < nums.length) {
            //If the curr index matches the previous 2 elements,skip it
            if (nums[curr] == nums[prev] && nums[curr] == nums[prev - 1]) {
                curr++;
            } else //Otherwise count that element nd update...
            {
                prev++;
                nums[prev] = nums[curr];
                curr++;
            }
        }
        // Return K after placing the final result in the first K slots of nums
        return prev + 1;
    }
}
