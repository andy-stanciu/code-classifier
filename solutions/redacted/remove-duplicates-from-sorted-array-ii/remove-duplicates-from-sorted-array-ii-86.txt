class Solution {

    public int method1(int[] nums) {
        // Index of the next unique element to add to the array
        int index = 1;
        // Count of the current element
        int count = 1;
        // Loop through the array starting from the second element
        for (int i = 1; i < nums.length; i++) {
            // If the current element is the same as the previous element, increment the count
            if (nums[i] == nums[i - 1]) {
                count++;
            } else // If the current element is different from the previous element, reset the count
            {
                count = 1;
            }
            // If the count is less than or equal to 2, add the current element to the array
            if (count <= 2) {
                nums[index] = nums[i];
                index++;
            }
        }
        // Return the index, which represents the number of unique elements in the array after removing duplicates
        return index;
    }
}
