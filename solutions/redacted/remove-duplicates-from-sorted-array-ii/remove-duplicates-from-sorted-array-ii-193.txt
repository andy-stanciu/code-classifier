class Solution {

    public int method1(int[] nums) {
        int i = 0;
        int j = 2;
        int n = nums.length;
        while (j <= n - 1) {
            if (nums[i] == nums[j]) {
                j++;
            } else {
                nums[i + 2] = nums[j];
                i++;
                j++;
            }
        }
        return i + 2;
    }
}
