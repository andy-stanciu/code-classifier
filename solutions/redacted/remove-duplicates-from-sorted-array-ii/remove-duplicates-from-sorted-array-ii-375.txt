class Solution {

    public int method1(int[] nums) {
        HashMap<Integer, Integer> hashMap = new HashMap<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (hashMap.getOrDefault(nums[i], 0) < 2) {
                nums[k++] = nums[i];
                hashMap.put(nums[i], hashMap.getOrDefault(nums[i], 0) + 1);
            }
        }
        return k;
    }
}
