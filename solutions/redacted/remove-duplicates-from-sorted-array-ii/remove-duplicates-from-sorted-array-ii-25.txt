class Solution {

    public int method1(int[] nums) {
        // Initilize two pointer i and j both as 2.
        int i = 2, j = 2;
        while (i < nums.length) {
            // Iterate loop from i till nums.length
            if (nums[i] != nums[j - 2]) {
                // Check if nums[i] is not equal to the nums[j-2] if this return true then modify nums[j] with nums[i] and increase j by 1.
                nums[j] = nums[i];
                j++;
            }
            // increase i by 1.
            i++;
        }
        // return j, because in the result array we have no. of j elements.
        return j;
    }
}
