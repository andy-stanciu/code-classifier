class Solution {

    public int method1(int[] nums) {
        int prev = 10001, count = 0;
        int pt = -1, move = 0;
        while (++pt < nums.length) {
            if (nums[pt] == prev) {
                if (++count > 2) {
                    move++;
                }
            } else {
                prev = nums[pt];
                count = 1;
            }
            nums[pt - move] = nums[pt];
        }
        return nums.length - move;
    }
}
