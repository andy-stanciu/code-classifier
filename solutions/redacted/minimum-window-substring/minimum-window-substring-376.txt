class Solution {

    public String method1(String s, String t) {
        if (s.length() < t.length() || s == null || t == null || s.length() == 0 || t.length() == 0) {
            return new String();
        }
        int left = 0, right = 0, minLen = Integer.MAX_VALUE, startIndex = 0;
        int t_size = t.length();
        int[] mp = new int[128];
        for (char c : t.toCharArray()) {
            mp[c]++;
        }
        char[] cs = s.toCharArray();
        while (right < cs.length) {
            if (mp[cs[right++]]-- > 0) {
                t_size--;
            }
            while (t_size == 0) {
                if (right - left < minLen) {
                    startIndex = left;
                    minLen = right - left;
                }
                if (mp[cs[left++]]++ == 0) {
                    t_size++;
                }
            }
        }
        return minLen == Integer.MAX_VALUE ? new String() : new String(cs, startIndex, minLen);
    }
}
