class Solution {

    public String method1(String s, String t) {
        boolean bFound = false;
        int left = 0;
        int right = 0;
        int minLeft = 0;
        int minRight = 0;
        int minLen = Integer.MAX_VALUE;
        Map<Character, Integer> tar = new HashMap<>();
        Map<Character, Integer> var1 = new HashMap<>();
        for (int i = 0; i < t.length(); i++) {
            tar.put(t.charAt(i), tar.getOrDefault(t.charAt(i), 0) + 1);
        }
        for (int i = 0; i < s.length(); i++) {
            var1.put(s.charAt(i), var1.getOrDefault(s.charAt(i), 0) + 1);
            while (isMatch(var1, tar)) {
                bFound = true;
                right = i;
                if (minLen > (i - left)) {
                    minLen = i - left;
                    minLeft = left;
                    minRight = right;
                }
                var1.put(s.charAt(left), var1.getOrDefault(s.charAt(left), 0) - 1);
                left++;
            }
        }
        if (!bFound)
            return "";
        return s.substring(minLeft, minRight + 1);
    }

    boolean isMatch(Map<Character, Integer> var1, Map<Character, Integer> tar) {
        for (Map.Entry<Character, Integer> x : tar.entrySet()) {
            if (var1.getOrDefault(x.getKey(), 0) < x.getValue()) {
                return false;
            }
        }
        return true;
    }
}
