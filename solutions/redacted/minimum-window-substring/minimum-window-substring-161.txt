class Solution {

    public String method1(String s, String t) {
        if (s.length() == 0 || t.length() == 0) {
            return "";
        }
        // Count characters in t
        Map<Character, Integer> tCount = new HashMap<>();
        for (char c : t.toCharArray()) {
            tCount.put(c, tCount.getOrDefault(c, 0) + 1);
        }
        // Sliding window
        Map<Character, Integer> var1 = new HashMap<>();
        int left = 0, right = 0, required = tCount.size(), formed = 0;
        int minLength = Integer.MAX_VALUE, minLeft = 0;
        while (right < s.length()) {
            // Add current character to window
            char c = s.charAt(right);
            var1.put(c, var1.getOrDefault(c, 0) + 1);
            // Check if current character fulfills requirement
            if (tCount.containsKey(c) && var1.get(c).intValue() == tCount.get(c).intValue()) {
                formed++;
            }
            // Try to shrink the window from the left
            while (left <= right && formed == required) {
                c = s.charAt(left);
                // Update minimum window
                if (right - left + 1 < minLength) {
                    minLength = right - left + 1;
                    minLeft = left;
                }
                // Remove character from window and update formed
                var1.put(c, var1.get(c) - 1);
                if (tCount.containsKey(c) && var1.get(c).intValue() < tCount.get(c).intValue()) {
                    formed--;
                }
                left++;
            }
            right++;
        }
        return minLength == Integer.MAX_VALUE ? "" : s.substring(minLeft, minLeft + minLength);
    }
}
