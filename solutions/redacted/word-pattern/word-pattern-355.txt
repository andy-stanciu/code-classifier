class Solution {

    public boolean method1(String param1, String s) {
        // initialize the map
        Map<Character, String> map = new HashMap<>();
        // convert the words in s into an array
        String[] s_arr = s.split(" ");
        // check if arr.length == pattern.length -> if not return false
        if (s_arr.length != param1.length()) {
            return false;
        }
        // map the chars of pattern to the words in s array
        // if mapping not present, put
        // else get and compare the value and arr element
        for (int i = 0; i < param1.length(); i++) {
            String s_str = s_arr[i];
            Character var1 = param1.charAt(i);
            if (map.containsKey(var1)) {
                if (map.get(var1).equals(s_str)) {
                    continue;
                } else {
                    return false;
                }
            } else {
                if (map.containsValue(s_str)) {
                    return false;
                } else {
                    map.put(var1, s_str);
                }
            }
        }
        return true;
    }
}
