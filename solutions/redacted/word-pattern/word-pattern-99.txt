class Solution {

    public boolean method1(String param1, String s) {
        String[] split = s.split(" ");
        char[] ch = param1.toCharArray();
        HashMap<Character, String> map = new HashMap<>();
        if (ch.length != split.length)
            return false;
        for (int i = 0; i < ch.length; i++) {
            if (map.containsKey(ch[i])) {
                if (!map.get(ch[i]).equals(split[i]))
                    return false;
            } else if (map.containsValue(split[i]))
                return false;
            else {
                map.put(ch[i], split[i]);
            }
        }
        return true;
    }
}
