class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (arr.length != param1.length())
            return false;
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            String var1 = arr[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(var1))
                    return false;
            } else {
                if (map.containsValue(var1))
                    return false;
                map.put(ch, var1);
            }
        }
        return true;
    }
}
