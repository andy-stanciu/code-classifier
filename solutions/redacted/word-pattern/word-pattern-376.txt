class Solution {

    public boolean method1(String param1, String s) {
        String[] allStrings = s.split(" ");
        if (param1.length() != allStrings.length)
            return false;
        int n = param1.length();
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (map.get(param1.charAt(i)).equals(allStrings[i]))
                    continue;
                else
                    return false;
            } else {
                if (map.containsValue(allStrings[i]))
                    return false;
                map.put(param1.charAt(i), allStrings[i]);
            }
        }
        return true;
    }
}
