class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        // separate words in string s
        if (var1.length != param1.length())
            return false;
        HashMap<Character, String> map = new HashMap();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            if (map.containsKey(c)) {
                // if map contains character(key) but is not matching to its value
                if (!map.get(c).equals(var1[i])) {
                    return false;
                }
            } else {
                // if map contains that value but corresponding to some other key
                if (map.containsValue(var1[i])) {
                    return false;
                }
                // add to map
                map.put(c, var1[i]);
            }
        }
        return true;
    }
}
