class Solution {

    public boolean method1(String param1, String values) {
        Map<Character, String> map = new HashMap<>();
        String[] var1 = values.split(" ");
        if (var1.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (!map.get(param1.charAt(i)).equals(var1[i])) {
                    return false;
                }
            } else {
                if (map.containsValue(var1[i])) {
                    return false;
                }
                map.put(param1.charAt(i), var1[i]);
            }
        }
        return true;
    }
}
