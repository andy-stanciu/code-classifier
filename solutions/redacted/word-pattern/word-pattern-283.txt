class Solution {

    public boolean method1(String param1, String s) {
        String[] arrayStrings = s.split(" ");
        if (param1.length() != arrayStrings.length) {
            return false;
        }
        Map<Character, String> var1 = new HashMap<>();
        Map<String, Character> var2 = new HashMap<>();
        for (int i = 0; i < arrayStrings.length; i++) {
            char charCurrent = param1.charAt(i);
            String var3 = arrayStrings[i];
            if (!var1.containsKey(charCurrent)) {
                var1.put(charCurrent, var3);
            } else {
                if (!var1.get(charCurrent).equals(var3)) {
                    return false;
                }
            }
            if (!var2.containsKey(var3)) {
                var2.put(var3, charCurrent);
            } else {
                if (!var2.get(var3).equals(charCurrent)) {
                    return false;
                }
            }
        }
        return true;
    }
}
