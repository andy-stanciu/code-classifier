class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> mp = new HashMap<>();
        String[] st = s.split(" ");
        String[] at = param1.split(" ");
        if (st.length < param1.length()) {
            return false;
        }
        StringBuilder sb = new StringBuilder();
        StringBuilder ans = new StringBuilder();
        for (int i = 0; i < param1.length(); i++) {
            if (!mp.containsKey(param1.charAt(i)) && !mp.containsValue(st[i])) {
                mp.put(param1.charAt(i), st[i]);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (mp.containsKey(param1.charAt(i))) {
                ans.append(mp.get(param1.charAt(i)));
            }
        }
        for (int i = 0; i < st.length; i++) {
            sb.append(st[i]);
        }
        int check = sb.compareTo(ans);
        return check == 0;
    }
}
