class Solution {

    public boolean method1(String param1, String s) {
        String[] strings = s.split(" ");
        if (param1.length() != strings.length)
            return false;
        HashMap<Character, String> map = new HashMap<Character, String>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String string = strings[i];
            if (map.containsKey(c) && !map.get(c).equals(string)) {
                return false;
            }
            if (map.containsValue(string) && !map.containsKey(c)) {
                return false;
            } else {
                map.put(c, string);
            }
        }
        return true;
    }
}
