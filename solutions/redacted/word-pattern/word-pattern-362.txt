class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        int i = 0;
        String[] str = s.split(" ");
        if (param1.length() != str.length) {
            return false;
        }
        for (String x : str) {
            if (!map.containsKey(param1.charAt(i))) {
                if (map.containsValue(x)) {
                    return false;
                }
                map.put(param1.charAt(i), x);
            } else {
                if (!map.get(param1.charAt(i)).equals(x)) {
                    return false;
                }
            }
            i++;
        }
        return true;
    }
}
