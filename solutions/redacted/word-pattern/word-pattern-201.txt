class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> hashmap = new HashMap<Character, String>();
        String[] str = s.split(" ");
        String[] array = new String[param1.length()];
        if (str.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (!hashmap.containsKey(param1.charAt(i)) && !hashmap.containsValue(str[i])) {
                hashmap.put(param1.charAt(i), str[i]);
            } else {
                if (hashmap.containsKey(param1.charAt(i))) {
                    boolean b = hashmap.get(param1.charAt(i)).equals(str[i]);
                    if (b == false)
                        return false;
                } else {
                    if (hashmap.containsValue(str[i])) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
}
