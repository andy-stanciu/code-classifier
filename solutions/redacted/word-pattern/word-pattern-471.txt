class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        if (str.length != param1.length())
            return false;
        HashMap<String, Character> hashMap = new HashMap<>();
        for (int i = 0; i < str.length; i++) {
            if (!hashMap.containsKey(str[i])) {
                if (hashMap.containsValue(param1.charAt(i)))
                    return false;
                hashMap.put(str[i], param1.charAt(i));
            } else {
                if (hashMap.get(str[i]) != param1.charAt(i))
                    return false;
            }
        }
        return true;
    }
}
