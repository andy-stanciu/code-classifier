class Solution {

    public boolean method1(String param1, String s) {
        // O(n) time | O(n) space
        HashMap<Character, String> myMap = new HashMap<>();
        String[] var1 = s.split(" ");
        // pattern = " a b c" && s = "mice cat dog chicken" then return false directly
        if (param1.length() != var1.length)
            return false;
        // update myMap in for-loop
        for (int i = 0; i < var1.length; i++) {
            char ch = param1.charAt(i);
            if (!myMap.containsKey(ch)) {
                // we need to check the case that, we dont\' have such a key in map but value already exists
                // for example, pattern = "abab" && s = "dog dog dog dog"
                if (myMap.containsValue(var1[i]))
                    return false;
                myMap.put(ch, var1[i]);
            } else if (!myMap.get(ch).equals(var1[i]))
                return false;
        }
        return true;
    }
}
