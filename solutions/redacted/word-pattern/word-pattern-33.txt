class Solution {

    static boolean helper(String param1, String str) {
        int n = param1.length();
        String[] arr = str.split(" ");
        if (n != arr.length)
            return false;
        HashMap<Character, String> hp = new HashMap<>();
        for (int i = 0; i < n; i++) {
            char ch = param1.charAt(i);
            String s = arr[i];
            if (!hp.containsKey(ch)) {
                if (hp.containsValue(s))
                    return false;
                else {
                    hp.put(ch, s);
                }
            } else {
                if (s.compareTo(hp.get(ch)) != 0)
                    return false;
            }
        }
        return true;
    }

    public boolean method1(String param1, String str) {
        return helper(param1, str);
    }
}
