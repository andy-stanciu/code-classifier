class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> map = new HashMap<>();
        Map<String, Character> smap = new HashMap<>();
        String[] var1 = s.split(" ");
        if (var1.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i)) && !map.get(param1.charAt(i)).equals(var1[i])) {
                return false;
            }
            map.put(param1.charAt(i), var1[i]);
            if (smap.containsKey(var1[i]) && !smap.get(var1[i]).equals(param1.charAt(i))) {
                return false;
            }
            smap.put(var1[i], param1.charAt(i));
        }
        return true;
    }
}
