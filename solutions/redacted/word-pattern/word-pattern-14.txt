class Solution {

    public boolean method1(String param1, String str) {
        String[] var1 = str.split(" ");
        if (var1.length != param1.length())
            return false;
        Map<Character, Integer> charIndexMap = new HashMap<>();
        Map<String, Integer> var2 = new HashMap<>();
        for (int i = 0; i < var1.length; ++i) {
            char c = param1.charAt(i);
            String var3 = var1[i];
            // Using Integer to avoid unboxing issues and null values
            Integer charIndex = charIndexMap.put(c, i);
            Integer var4 = var2.put(var3, i);
            // Check if the current mapping is different from previous occurrences
            if (!Objects.equals(charIndex, var4)) {
                // cant use other comparasion techies
                return false;
            }
        }
        return true;
    }
}
