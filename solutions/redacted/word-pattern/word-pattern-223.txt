class Solution {

    public boolean method1(String param1, String s) {
        String[] w = s.split(" ");
        if (w.length != param1.length())
            return false;
        Map i = new HashMap();
        for (Integer j = 0; j < w.length; j++) if (i.put(param1.charAt(j), j) != i.put(w[j], j))
            return false;
        return true;
    }
}
