class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            boolean containsKey = map.containsKey(ch);
            if (map.containsValue(arr[i]) && !containsKey)
                return false;
            if (containsKey && !map.get(ch).equals(arr[i]))
                return false;
            else
                map.put(ch, arr[i]);
        }
        return true;
    }
}
