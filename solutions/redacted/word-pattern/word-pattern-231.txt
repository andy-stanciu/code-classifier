class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> hashmap = new HashMap<>();
        String[] splittedText = s.split(" ");
        if (param1.length() != splittedText.length) {
            return false;
        }
        for (int i = 0; i < splittedText.length; i++) {
            if (hashmap.get(param1.charAt(i)) == null && hashmap.containsValue(splittedText[i]) || (hashmap.get(param1.charAt(i)) != null && !hashmap.get(param1.charAt(i)).equals(splittedText[i]))) {
                return false;
            } else {
                hashmap.put(param1.charAt(i), splittedText[i]);
            }
        }
        return true;
    }
}
