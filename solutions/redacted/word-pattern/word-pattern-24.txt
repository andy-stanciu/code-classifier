class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        Map<Character, String> map = new HashMap<>();
        Set<String> h = new HashSet<>();
        if (arr.length != param1.length()) {
            return false;
        }
        boolean flag = false;
        for (int i = 0; i < param1.length(); i++) {
            if (!map.containsKey(param1.charAt(i)) && !h.contains(arr[i])) {
                map.put(param1.charAt(i), arr[i]);
                h.add(arr[i]);
            }
            if (map.containsKey(param1.charAt(i)) && map.get(param1.charAt(i)).equals(arr[i])) {
                flag = true;
            } else {
                return false;
            }
        }
        return flag;
    }
}
