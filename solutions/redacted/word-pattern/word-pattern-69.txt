class Solution {

    public boolean method1(String param1, String s) {
        //to map string to characters
        Map<Character, String> map = new HashMap<>();
        //to track already visited strings
        Set<String> v = new HashSet<>();
        //splitting string s into tokens
        String[] arr = s.split(" ", s.length());
        if (arr.length != param1.length()) {
            return false;
        }
        int i = 0;
        for (String str : arr) {
            //if map already contains character
            if (map.containsKey(param1.charAt(i))) {
                //if this time has same string as previously mapped
                if (map.get(param1.charAt(i)).equals(str)) {
                    i++;
                    continue;
                } else //if not then return false
                {
                    return false;
                }
            } else //if string already visited but this time it has diffrent character
            if (!map.containsKey(param1.charAt(i)) && v.contains(str)) {
                return false;
            } else
                map.put(param1.charAt(i), str);
            i++;
            v.add(str);
        }
        return true;
    }
}
