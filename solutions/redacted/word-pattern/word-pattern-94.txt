class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<Character, String> map = new HashMap<Character, String>();
        for (int i = 0; i < arr.length; i++) {
            //if the Character is not present in the map:
            if (!map.containsKey(param1.charAt(i))) {
                //if the String is already present
                //this means it is already mapped to some other character
                if (map.containsValue(arr[i])) {
                    return false;
                }
                map.put(param1.charAt(i), arr[i]);
            } else //if the Character is already present in the map
            {
                //if the current string and the already mapped string
                //are not same, return false
                String replace = map.get(param1.charAt(i));
                if (!replace.equals(arr[i]))
                    return false;
            }
        }
        return true;
    }
}
