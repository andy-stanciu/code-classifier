class Solution {

    public boolean method1(String param1, String s) {
        var var1 = s.split(" ");
        var chars = param1.toCharArray();
        if (var1.length != chars.length) {
            return false;
        }
        var var2 = new HashMap<Character, String>();
        for (int i = 0; i < chars.length; i++) {
            var var3 = var2.get(chars[i]);
            if (var3 == null) {
                if (var2.containsValue(var1[i])) {
                    return false;
                }
                var2.put(chars[i], var1[i]);
                continue;
            }
            if (!Objects.equals(var3, var1[i])) {
                return false;
            }
        }
        return true;
    }
}
