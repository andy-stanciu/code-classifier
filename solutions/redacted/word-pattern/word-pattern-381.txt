class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        HashMap<String, Character> map = new HashMap<String, Character>();
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(arr[i])) {
                if (!map.get(arr[i]).equals(param1.charAt(i))) {
                    return false;
                }
            } else {
                if (map.containsValue(param1.charAt(i))) {
                    return false;
                }
                map.put(arr[i], param1.charAt(i));
            }
        }
        return true;
    }
}
