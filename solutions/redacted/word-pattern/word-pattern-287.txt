class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> bijection = new HashMap<>();
        HashMap<String, Character> reverseBijection = new HashMap<>();
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String var2 = var1[i];
            if (!bijection.containsKey(c)) {
                bijection.put(c, var2);
            } else if (!bijection.get(c).equals(var2)) {
                return false;
            }
            if (!reverseBijection.containsKey(var2)) {
                reverseBijection.put(var2, c);
            } else if (!reverseBijection.get(var2).equals(c)) {
                return false;
            }
        }
        return true;
    }
}
