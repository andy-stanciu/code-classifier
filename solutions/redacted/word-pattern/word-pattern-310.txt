class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        if (var1.length != param1.length())
            return false;
        Map<Character, Integer> map = new HashMap<>();
        Map<String, Integer> mapS = new HashMap<>();
        Map<Integer, Integer> var2 = new HashMap<>();
        Map<Integer, Integer> var3 = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String var4 = var1[i];
            if (!map.containsKey(c))
                map.put(c, i);
            if (!mapS.containsKey(var4))
                mapS.put(var4, i);
            var2.put(i, map.get(c));
            var3.put(i, mapS.get(var4));
        }
        for (int i = 0; i < var1.length; i++) {
            if (var2.get(i) - var3.get(i) != 0)
                return false;
        }
        return true;
    }
}
