class Solution {

    public boolean method1(String param1, String s) {
        String[] s1 = s.split(" ");
        if (s1.length != param1.length())
            return false;
        HashMap<Character, String> m1 = new HashMap<>();
        HashMap<String, Character> m2 = new HashMap<>();
        for (int i = 0; i < s1.length; i++) {
            char c = param1.charAt(i);
            String var1 = s1[i];
            if (m1.containsKey(c)) {
                if (!m1.get(c).equals(var1))
                    return false;
            } else {
                if (m2.containsKey(var1))
                    return false;
                m1.put(c, var1);
                m2.put(var1, c);
            }
        }
        return true;
    }
}
