class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        if (str.length != param1.length())
            return false;
        Map<Character, String> map = new HashMap<>();
        Map<String, Character> valTochar = new HashMap<>();
        for (int i = 0; i < str.length; i++) {
            char curr = param1.charAt(i);
            String tempStr = str[i];
            if (map.containsKey(curr)) {
                if (!tempStr.equalsIgnoreCase(map.get(curr))) {
                    return false;
                }
            } else if (valTochar.containsKey(tempStr)) {
                if (!tempStr.equalsIgnoreCase(map.get(tempStr))) {
                    return false;
                }
            } else {
                map.put(curr, tempStr);
                valTochar.put(tempStr, curr);
            }
        }
        return true;
    }
}
