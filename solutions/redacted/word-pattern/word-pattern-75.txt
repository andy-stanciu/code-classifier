// Rules ->
// two characters can\'t have same word
// one characters can\'t have two different words
class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> map = new HashMap<>();
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length)
            return false;
        for (int i = 0; i < param1.length(); i++) {
            // for case like a b b a -> dog dog dog dog
            if (map.containsValue(var1[i]) && map.containsKey(param1.charAt(i)) == false)
                return false;
            if (map.containsKey(param1.charAt(i)) == false)
                map.put(param1.charAt(i), var1[i]);
            else {
                String str = map.get(param1.charAt(i));
                if (!str.equals(var1[i]))
                    return false;
            }
        }
        return true;
    }
}
