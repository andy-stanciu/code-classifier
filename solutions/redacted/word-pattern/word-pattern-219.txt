class Solution {

    public boolean method1(String param1, String s) {
        String[] strArray = s.split(" ");
        char[] chaArray = param1.toCharArray();
        if (strArray.length != chaArray.length)
            return false;
        Map<Character, String> map = new HashMap<>();
        int index = 0;
        for (Character cha : chaArray) {
            if (map.containsKey(cha)) {
                if (!map.get(cha).equals(strArray[index]))
                    return false;
            } else {
                if (map.containsValue(strArray[index]))
                    return false;
            }
            map.put(cha, strArray[index++]);
        }
        return true;
    }
}
