class Solution {

    public boolean method1(String param1, String s) {
        var var1 = new HashMap<Character, String>();
        var var2 = s.split(" ");
        if (var2.length != param1.length())
            return false;
        for (var i = 0; i < var2.length; i++) {
            if (var1.containsKey(param1.charAt(i))) {
                if (!var1.get(param1.charAt(i)).equals(var2[i]))
                    return false;
            } else {
                if (var1.containsValue(var2[i]))
                    return false;
            }
            var1.put(param1.charAt(i), var2[i]);
        }
        return true;
    }
}
