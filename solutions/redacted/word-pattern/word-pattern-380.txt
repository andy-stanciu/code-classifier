class Solution {

    public boolean method1(String param1, String s) {
        List<Character> chars = new ArrayList<>();
        List<String> var1 = new ArrayList<>();
        String[] strings = s.split(" ");
        if (param1.length() != strings.length) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (!chars.contains(param1.charAt(i)) && !var1.contains(strings[i])) {
                chars.add(param1.charAt(i));
                var1.add(strings[i]);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (chars.indexOf(param1.charAt(i)) != var1.indexOf(strings[i])) {
                return false;
            }
        }
        return true;
    }
}
