class Solution {

    public boolean method1(String param1, String s) {
        String[] sArray = s.split(" ");
        if (sArray.length != param1.length())
            return false;
        HashMap<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            if (map.containsKey(c) && !map.get(c).equals(sArray[i]))
                return false;
            else if (map.containsValue(sArray[i]) && !map.containsKey(c))
                return false;
            else
                map.put(c, sArray[i]);
        }
        return true;
    }
}
