class Solution {

    public boolean method1(String param1, String s) {
        String[] var1 = s.split(" ");
        char[] letters = param1.toCharArray();
        // checking lengths are same or not, if different return false;
        if (var1.length != letters.length)
            return false;
        // creating map to store each letter with corresponding word
        Map<Character, String> map = new HashMap();
        Set<String> var2 = new HashSet();
        // loop through letters to fill the map
        for (int i = 0; i < letters.length; i++) {
            if (map.containsKey(letters[i])) {
                if (!map.get(letters[i]).equals(var1[i]))
                    return false;
            } else {
                // if adding new word fails then it\'s duplicated
                if (!var2.add(var1[i]))
                    return false;
                map.put(letters[i], var1[i]);
            }
        }
        return true;
    }
}
