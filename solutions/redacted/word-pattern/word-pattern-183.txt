class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> hs = new HashMap<>();
        String[] st = s.split(" ");
        if (st.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < st.length; i++) {
            char ch = param1.charAt(i);
            String ss = st[i];
            if (hs.containsKey(ch) && !hs.get(ch).equals(ss) || !hs.containsKey(ch) && hs.values().contains(ss)) {
                return false;
            }
            hs.put(ch, ss);
        }
        return true;
    }
}
