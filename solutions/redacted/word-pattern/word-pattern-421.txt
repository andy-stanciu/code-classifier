class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> h = new HashMap<>();
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            String th = arr[i];
            if (h.containsKey(ch)) {
                if (!h.get(ch).equals(th))
                    return false;
            } else if (h.containsValue(th))
                return false;
            else
                h.put(ch, th);
        }
        return true;
    }
}
