class Solution {

    public boolean method1(String param1, String s) {
        String[] ch = s.split(" ");
        Boolean flag = true;
        if (ch.length != param1.length()) {
            return false;
        } else {
            Map<Character, String> map = new HashMap<Character, String>();
            Map<String, Character> map2 = new HashMap<String, Character>();
            for (int i = 0; i < param1.length(); i++) {
                String gt = map.getOrDefault((param1.charAt(i)), null);
                Character gt2 = map2.getOrDefault(ch[i], null);
                if (gt == null && gt2 == null) {
                    map.put(param1.charAt(i), ch[i]);
                    map2.put(ch[i], param1.charAt(i));
                } else if (gt != null && gt2 != null && gt.equals(ch[i]) && (gt2 == param1.charAt(i))) {
                    flag = true;
                } else {
                    return false;
                }
            }
        }
        return flag;
    }
}
