class Solution {

    public boolean method1(String param1, String s) {
        List<String> list = List.of(s.split(" "));
        Map<Character, String> map = new HashMap<>();
        if (param1.length() != list.size()) {
            return false;
        }
        return (extracted(param1, list, map));
    }

    private static boolean extracted(String param1, List<String> list, Map<Character, String> map) {
        for (int i = 0; i < param1.length(); i++) {
            if (!map.containsKey(param1.charAt(i))) {
                if (map.containsValue(list.get(i)))
                    return false;
                map.put(param1.charAt(i), list.get(i));
            } else if (map.containsKey(param1.charAt(i)) && !map.get(param1.charAt(i)).equals(list.get(i))) {
                return false;
            }
        }
        return true;
    }
}
