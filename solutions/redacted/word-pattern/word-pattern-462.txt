class Solution {

    public boolean method1(String param1, String s) {
        String[] array_s = s.split(" ");
        if (param1.length() != array_s.length)
            return false;
        HashMap<Character, String> map = new HashMap<>();
        int index = 0;
        for (final char c : param1.toCharArray()) {
            String str = array_s[index];
            if (map.containsKey(c) && !map.get(c).equals(str) || !map.containsKey(c) && map.containsValue(str))
                return false;
            else
                map.put(c, str);
            index++;
        }
        return true;
    }
}
