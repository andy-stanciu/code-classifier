class Solution {

    public boolean method1(String param1, String s) {
        String[] newStr = s.split(" ");
        if (param1.length() != newStr.length)
            return false;
        Map<Character, String> hashy = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (hashy.containsKey(param1.charAt(i))) {
                //String check= hashy.get(pattern.charAt(i));
                // System.out.println(check+" --");
                if (!hashy.get(param1.charAt(i)).equals(newStr[i]))
                    return false;
            } else if (hashy.containsValue(newStr[i])) {
                return false;
            }
            hashy.put(param1.charAt(i), newStr[i]);
        }
        return true;
    }
}
