class Solution {

    public boolean method1(String param1, String s) {
        String[] w = s.split(" ");
        if (param1.length() != w.length) {
            return false;
        } else {
            HashMap<Character, String> m = new HashMap<>();
            for (int i = 0; i < param1.length(); i++) {
                char c = param1.charAt(i);
                if (m.containsKey(c)) {
                    if (!m.get(c).equals(w[i]))
                        return false;
                } else {
                    if (m.containsValue(w[i]))
                        return false;
                    m.put(c, w[i]);
                }
            }
        }
        return true;
    }
}
