class Solution {

    public boolean method1(String param1, String s) {
        // to map a string to its character
        Map<Character, String> m = new HashMap();
        int leftp = 0;
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        while (leftp < param1.length()) {
            if (!m.containsKey(param1.charAt(leftp))) {
                if (m.containsValue(arr[leftp])) {
                    return false;
                }
                m.put(param1.charAt(leftp), arr[leftp]);
            } else {
                if (!m.get(param1.charAt(leftp)).equals(arr[leftp]))
                    return false;
            }
            leftp++;
        }
        return true;
    }
}
