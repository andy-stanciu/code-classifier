class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        if (str.length != param1.length()) {
            return false;
        }
        Map<Character, String> patMap = new HashMap<>();
        Map<String, Character> sMap = new HashMap<>();
        int length = str.length;
        for (int i = 0; i < length; i++) {
            Character pChar = param1.charAt(i);
            String var1 = str[i];
            if (patMap.containsKey(pChar)) {
                if (!patMap.get(pChar).equals(var1)) {
                    return false;
                }
            } else {
                patMap.put(pChar, var1);
            }
            if (sMap.containsKey(var1)) {
                if (!Objects.equals(sMap.get(var1), pChar)) {
                    return false;
                }
            } else {
                sMap.put(var1, pChar);
            }
        }
        return true;
    }
}
