class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        HashMap<String, Character> reverseMap = new HashMap<>();
        String[] arr = s.split(" ");
        if (arr.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            char ch = param1.charAt(i);
            String var1 = arr[i];
            if (map.containsKey(ch)) {
                if (!map.get(ch).equals(var1)) {
                    return false;
                }
            } else {
                if (reverseMap.containsKey(var1) && reverseMap.get(var1) != ch) {
                    return false;
                }
                map.put(ch, var1);
                reverseMap.put(var1, ch);
            }
        }
        return true;
    }
}
