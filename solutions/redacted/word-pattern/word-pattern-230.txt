class Solution {

    public boolean method1(String param1, String s) {
        StringTokenizer st = new StringTokenizer(s);
        if (param1.length() != st.countTokens()) {
            return false;
        } else {
            HashMap<Character, String> hm = new HashMap<>();
            for (int i = 0; i < param1.length(); i++) {
                String temp = st.nextToken();
                if (hm.containsKey(param1.charAt(i))) {
                    if (!hm.get(param1.charAt(i)).equals(temp)) {
                        return false;
                    }
                } else {
                    if (hm.containsValue(temp)) {
                        return false;
                    }
                    hm.put(param1.charAt(i), temp);
                }
            }
        }
        return true;
    }
}
