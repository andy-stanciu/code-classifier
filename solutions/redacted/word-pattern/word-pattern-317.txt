class Solution {

    public boolean method1(String param1, String s) {
        HashMap<String, Character> hmap = new HashMap<>();
        HashMap<Character, String> hmap2 = new HashMap<>();
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        }
        for (int i = 0; i < arr.length; i++) {
            char c = param1.charAt(i);
            if (!hmap.containsKey(arr[i]) && !hmap2.containsKey(c)) {
                hmap.put(arr[i], c);
                hmap2.put(c, arr[i]);
            } else {
                if (hmap.containsKey(arr[i])) {
                    if (hmap.get(arr[i]) != c) {
                        return false;
                    }
                }
                if (hmap2.containsKey(c)) {
                    if (!hmap2.get(c).equals(arr[i])) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
}
