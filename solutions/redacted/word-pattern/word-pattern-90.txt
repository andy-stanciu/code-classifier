class Solution {

    public boolean method1(String param1, String s) {
        String[] strArr = s.split(" ");
        if (param1.length() != strArr.length)
            return false;
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < strArr.length; i++) {
            var ch = param1.charAt(i);
            if (!map.containsKey(ch)) {
                if (map.containsValue(strArr[i]))
                    return false;
                map.put(ch, strArr[i]);
            } else if (map.containsKey(ch) && !map.get(ch).equals(strArr[i]))
                return false;
        }
        return true;
    }
}
