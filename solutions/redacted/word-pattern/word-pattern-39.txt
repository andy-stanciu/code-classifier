class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        HashMap<Character, String> h = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            if (!h.containsKey(param1.charAt(i))) {
                if (!h.containsValue(arr[i])) {
                    h.put(param1.charAt(i), arr[i]);
                } else
                    return false;
            } else {
                if (arr[i].compareTo(h.get(param1.charAt(i))) != 0)
                    return false;
            }
        }
        return true;
    }
}
