class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length) {
            return false;
        } else {
            for (int i = 0; i < arr.length; i++) {
                for (int j = i + 1; j < param1.length(); j++) {
                    if (param1.charAt(j) == param1.charAt(i) && !arr[j].equals(arr[i])) {
                        return false;
                    }
                    if (arr[j].equals(arr[i]) && param1.charAt(j) != param1.charAt(i)) {
                        return false;
                    }
                }
            }
            return true;
        }
    }
}
