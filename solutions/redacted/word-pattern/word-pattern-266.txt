class Solution {

    public boolean method1(String param1, String s) {
        String[] sArr = s.split(" ");
        if (param1.length() != sArr.length)
            return false;
        Map<Character, String> hashPat = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (hashPat.containsKey(param1.charAt(i))) {
                if (!hashPat.get(param1.charAt(i)).equals(sArr[i]))
                    return false;
            } else {
                if (hashPat.containsValue(sArr[i]))
                    return false;
            }
            hashPat.put(param1.charAt(i), sArr[i]);
        }
        return true;
    }
}
