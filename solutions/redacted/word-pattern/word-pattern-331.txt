class Solution {

    public boolean method1(String param1, String s) {
        if (param1 == null || s == null || s.trim().isEmpty()) {
            return false;
        }
        s = s.trim();
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length) {
            // Patterns and words have unequal lengths
            return false;
        }
        Map<Character, String> map = new HashMap<>();
        Set<String> var2 = new HashSet<>();
        for (int i = 0; i < param1.length(); i++) {
            char currentChar = param1.charAt(i);
            String var3 = var1[i];
            if (map.containsKey(currentChar)) {
                // If pattern character is already mapped, check if it maps to the same word
                if (!map.get(param1.charAt(i)).equalsIgnoreCase(var3)) {
                    // Pattern character maps to a different word
                    return false;
                }
            } else {
                // If pattern character is not mapped yet, check if the word is already mapped
                // to another character
                if (var2.contains(var3)) {
                    // Word is mapped to another pattern character
                    return false;
                }
                map.put(currentChar, var3);
                var2.add(var3);
            }
        }
        return true;
    }
}
