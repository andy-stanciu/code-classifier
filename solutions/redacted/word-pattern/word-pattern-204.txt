class Solution {

    public boolean method1(String param1, String s) {
        String[] arr = s.split(" ");
        if (param1.length() != arr.length)
            return false;
        Map<Character, String> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (map.containsKey(param1.charAt(i))) {
                if (arr[i].equals(map.get(param1.charAt(i))))
                    continue;
                else {
                    return false;
                }
            } else {
                if (map.containsValue(arr[i]))
                    return false;
                map.put(param1.charAt(i), arr[i]);
            }
        }
        return true;
    }
}
