class Solution {

    public boolean method1(String param1, String s) {
        if ((null == param1 && null == s) || (param1.isEmpty() && s.isEmpty()))
            return true;
        String[] strs = s.split("\\\\ ");
        if (strs.length != param1.length())
            return false;
        Map<String, Character> strMap = new HashMap<>();
        Map<Character, String> charMap = new HashMap<>();
        for (int idx = 0; idx < param1.length(); idx++) {
            String curStr = strs[idx];
            char curChar = param1.charAt(idx);
            Character val = strMap.get(curStr);
            String strVal = charMap.get(curChar);
            if (val == null) {
                strMap.put(curStr, curChar);
            } else {
                if (val != curChar)
                    return false;
            }
            if (strVal == null) {
                charMap.put(curChar, curStr);
            } else {
                if (!strVal.equals(curStr))
                    return false;
            }
        }
        return true;
    }
}
