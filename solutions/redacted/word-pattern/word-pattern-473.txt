class Solution {

    public boolean method1(String param1, String s) {
        HashMap<String, Character> map = new HashMap<>();
        String[] sArr = s.split(" ");
        if (param1.length() != sArr.length)
            return false;
        for (int i = 0; i < sArr.length; i++) {
            if (map.containsKey(sArr[i])) {
                if (map.get(sArr[i]) != param1.charAt(i))
                    return false;
            } else {
                if (map.containsValue(param1.charAt(i)))
                    return false;
                map.put(sArr[i], param1.charAt(i));
            }
        }
        return true;
    }
}
