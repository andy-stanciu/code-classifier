class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> pMap = new HashMap<Character, String>();
        HashMap<String, Character> sMap = new HashMap<String, Character>();
        String[] var1 = s.split(" ");
        char[] chars = param1.toCharArray();
        if (chars.length != var1.length)
            return false;
        for (int i = 0; i < var1.length; i++) {
            char ch = chars[i];
            String var2 = var1[i];
            if (pMap.containsKey(ch) && !pMap.get(ch).equals(var2))
                return false;
            else if (sMap.containsKey(var2) && !sMap.get(var2).equals(ch))
                return false;
            else {
                pMap.put(chars[i], var1[i]);
                sMap.put(var1[i], chars[i]);
            }
        }
        return true;
    }
}
