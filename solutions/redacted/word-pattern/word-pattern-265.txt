class Solution {

    public boolean method1(String param1, String s) {
        String[] str = s.split(" ");
        if (param1.length() != str.length)
            return false;
        HashMap<Character, String> hm = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            /*
             * For any key in the HashMap, if the corresponding value is
             * not equal to the string of  str, i.e.,
             * for same key, values are different, so return false
             */
            if (hm.containsKey(param1.charAt(i))) {
                if (!hm.get(param1.charAt(i)).equals(str[i]))
                    return false;
            } else {
                /*
                 * For any new key in the HashMap, if the character of str
                 * already exist, i.e., for two different keys, values are same,
                 * so return false
                 */
                if (hm.containsValue(str[i]))
                    return false;
                else
                    hm.put(param1.charAt(i), str[i]);
            }
        }
        return true;
    }
}
