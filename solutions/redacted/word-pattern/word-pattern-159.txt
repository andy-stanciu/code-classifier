class Solution {

    public boolean method1(String param1, String s) {
        // Split the string `s` into words
        String[] var1 = s.split(" ");
        // Check if the number of words matches the number of characters in pattern
        if (param1.length() != var1.length) {
            return false;
        }
        // Create two hash maps for mapping pattern -> word and word -> pattern
        HashMap<Character, String> var2 = new HashMap<>();
        HashMap<String, Character> var3 = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String var4 = var1[i];
            // Check if there\'s already a mapping from pattern character to a word
            if (var2.containsKey(c)) {
                if (!var2.get(c).equals(var4)) {
                    // Mismatch in mapping
                    return false;
                }
            } else {
                var2.put(c, var4);
            }
            // Check if there\'s already a mapping from word to a pattern character
            if (var3.containsKey(var4)) {
                if (var3.get(var4) != c) {
                    // Mismatch in mapping
                    return false;
                }
            } else {
                var3.put(var4, c);
            }
        }
        // All mappings are consistent
        return true;
    }
}
