class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] str = s.split(" ");
        if (str.length != param1.length())
            return false;
        for (int i = 0; i < param1.length(); i++) {
            char curr = param1.charAt(i);
            String string = str[i];
            if (map.containsKey(curr)) {
                if (!string.equals(map.get(curr)))
                    return false;
            } else {
                if (map.containsValue(string))
                    return false;
                map.put(curr, string);
            }
        }
        return true;
    }
}
