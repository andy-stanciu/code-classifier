class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<Character, String>();
        String[] var1 = s.split(" ");
        if (param1.length() != var1.length) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            String str = var1[i];
            if (!map.containsValue(str)) {
                map.computeIfAbsent(param1.charAt(i), f -> new String(str));
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (map.get(param1.charAt(i)) == null || !map.get(param1.charAt(i)).equals(var1[i])) {
                return false;
            }
        }
        return true;
    }
}
