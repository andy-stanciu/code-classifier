class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] var1 = s.split(" ");
        if (var1.length != param1.length()) {
            return false;
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            String w = var1[i];
            if (!map.containsKey(c)) {
                if (map.containsValue(w)) {
                    // Word already mapped to a different character
                    return false;
                }
                map.put(c, w);
            } else {
                if (!map.get(c).equals(w)) {
                    // Character c is mapped to a different word
                    return false;
                }
            }
        }
        return true;
    }
}
