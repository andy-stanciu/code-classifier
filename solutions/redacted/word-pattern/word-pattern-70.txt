class Solution {

    public boolean method1(String param1, String s) {
        Map<Character, String> map = new HashMap<>();
        Map<String, Character> opMap = new HashMap<>();
        char[] chars = param1.toCharArray();
        String[] strs = s.split(" ");
        int pl = chars.length;
        int sl = strs.length;
        int i = 0;
        if (sl != pl) {
            return false;
        }
        for (String var1 : strs) {
            Character ch = chars[i];
            if (map.containsKey(ch) && !var1.equals(map.get(ch))) {
                return false;
            } else if (opMap.containsKey(var1) && ch != opMap.get(var1)) {
                return false;
            }
            map.put(ch, var1);
            opMap.put(var1, ch);
            i++;
        }
        return true;
    }
}
