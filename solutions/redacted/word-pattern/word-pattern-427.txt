class Solution {

    public boolean method1(String param1, String s) {
        String[] spt = s.split(" ");
        HashMap<Character, String> hm = new HashMap<>();
        Set<String> sSet = new HashSet<String>();
        Set<Character> cSet = new HashSet<Character>();
        for (String var1 : spt) sSet.add(var1);
        for (char c : param1.toCharArray()) cSet.add(c);
        int[] arr = new int[256];
        if (spt.length != param1.length() || sSet.size() != cSet.size())
            return false;
        for (int i = 0; i < spt.length; i++) {
            if (hm.containsKey(param1.charAt(i))) {
                if (!hm.get(param1.charAt(i)).equals(spt[i])) {
                    System.out.println(spt[i] + " " + param1.charAt(i));
                    return false;
                }
            } else {
                hm.put(param1.charAt(i), spt[i]);
            }
        }
        return true;
    }
}
