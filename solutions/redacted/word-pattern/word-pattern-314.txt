class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map1 = new HashMap<>();
        HashMap<String, Character> map2 = new HashMap<>();
        String[] arr = s.split(" ");
        if (arr.length != param1.length())
            return false;
        for (int i = 0; i < arr.length; i++) {
            if (map1.containsKey(param1.charAt(i)) || map2.containsKey(arr[i])) {
                if (!map1.containsKey(param1.charAt(i)) || !map2.containsKey(arr[i])) {
                    return false;
                }
                if (!map1.get(param1.charAt(i)).equals(arr[i]) || map2.get(arr[i]) != param1.charAt(i)) {
                    return false;
                }
            }
            map1.put(param1.charAt(i), arr[i]);
            map2.put(arr[i], param1.charAt(i));
        }
        return true;
    }
}
