class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        String[] ss = s.split(" ");
        if (param1.length() != ss.length)
            return false;
        for (int i = 0; i <= param1.length() - 1; i++) {
            if (map.containsKey(param1.charAt(i))) {
                String value = map.get(param1.charAt(i));
                if (ss[i].equals(value)) {
                    continue;
                } else {
                    return false;
                }
            } else {
                for (String a : map.values()) {
                    if (a.equals(ss[i])) {
                        return false;
                    }
                }
                map.put(param1.charAt(i), ss[i]);
            }
        }
        return true;
    }
}
