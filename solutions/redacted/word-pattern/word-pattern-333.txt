class Solution {

    public boolean method1(String param1, String s) {
        String[] splits = s.split(" ");
        if (splits.length != param1.length()) {
            return false;
        }
        Map<Character, Integer> ch = new HashMap<>();
        Map<String, Integer> st = new HashMap<>();
        for (Integer i = 0; i < param1.length(); ++i) {
            if (ch.put(param1.charAt(i), i) != st.put(splits[i], i))
                return false;
        }
        return true;
    }
}
