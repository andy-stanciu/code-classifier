class Solution {

    public boolean method1(String param1, String s) {
        HashMap<Character, String> map = new HashMap<>();
        int length = param1.length();
        String[] arr = s.split(" ");
        if (length != arr.length)
            return false;
        for (int i = 0; i < length; i++) {
            char c = param1.charAt(i);
            String var1 = arr[i];
            String existValue = map.get(c);
            if (existValue == null) {
                if (map.containsValue(var1))
                    return false;
                else
                    map.put(c, var1);
            } else if (!existValue.equals(var1))
                return false;
        }
        return true;
    }
}
