class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> arr = new ArrayList<>();
        method2(arr, root);
        return arr;
    }

    public static void method2(ArrayList<Integer> arr, TreeNode root) {
        if (root == null) {
            return;
        }
        //left value right
        method2(arr, root.left);
        arr.add(root.val);
        method2(arr, root.right);
    }
}
