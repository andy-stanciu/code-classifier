/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        method2(list, root);
        return list;
    }

    public void method2(ArrayList<Integer> list, TreeNode root) {
        if (root == null) {
            return;
        }
        method2(list, root.left);
        list.add(root.val);
        method2(list, root.right);
    }
}
