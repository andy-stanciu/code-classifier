class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        helper(root, result);
        return result;
    }

    public void helper(TreeNode node, List<Integer> result) {
        if (node == null)
            return;
        helper(node.left, result);
        result.add(node.val);
        helper(node.right, result);
    }
}
