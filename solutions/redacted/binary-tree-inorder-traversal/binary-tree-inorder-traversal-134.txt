class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        return var1;
    }

    private void method2(TreeNode node, List<Integer> list) {
        if (node != null) {
            method2(node.left, list);
            list.add(node.val);
            method2(node.right, list);
        }
    }
}
