/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> n = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root != null) {
            method1(root.left);
            n.add(root.val);
            method1(root.right);
        }
        return n;
    }
}
