class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> al = new ArrayList<>();
        if (root == null)
            return al;
        TreeNode curr = root;
        while (curr != null) {
            if (// LEFT HAI
            curr.left != null) {
                TreeNode pred = curr.left;
                while (pred.right != null && pred.right != curr) {
                    pred = pred.right;
                }
                if (pred.right == null) {
                    pred.right = curr;
                    curr = curr.left;
                } else {
                    al.add(curr.val);
                    pred.right = null;
                    curr = curr.right;
                }
            } else // LEFT NHI HAI
            {
                al.add(curr.val);
                curr = curr.right;
            }
        }
        return al;
    }
}
