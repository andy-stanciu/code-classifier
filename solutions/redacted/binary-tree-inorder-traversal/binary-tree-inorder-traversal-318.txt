/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        return var1;
    }

    private void method2(TreeNode root, List<Integer> result) {
        if (root == null) {
            return;
        }
        method2(root.left, result);
        result.add(root.val);
        method2(root.right, result);
    }
}
