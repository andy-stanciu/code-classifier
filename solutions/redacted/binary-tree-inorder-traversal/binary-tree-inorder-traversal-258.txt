/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        return method2(root, result);
    }

    public List<Integer> method2(TreeNode root, List<Integer> result) {
        //Left -> Root -> Right
        if (root.left != null) {
            method2(root.left, result);
        }
        result.add(root.val);
        if (root.right != null) {
            method2(root.right, result);
        }
        return result;
    }
}
