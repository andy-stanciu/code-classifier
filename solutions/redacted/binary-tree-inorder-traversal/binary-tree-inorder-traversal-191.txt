class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        Stack<TreeNode> st = new Stack<>();
        TreeNode node = root;
        while (node != null || st.size() > 0) {
            if (node != null) {
                st.push(node);
                node = node.left;
            } else {
                TreeNode temp = st.peek();
                st.pop();
                list.add(temp.val);
                node = temp.right;
            }
        }
        return list;
    }
}
