/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> arr = new ArrayList<>();
        method2(arr, root);
        return arr;
    }

    public void method2(List<Integer> arr, TreeNode root) {
        if (root != null) {
            method2(arr, root.left);
            arr.add(root.val);
            method2(arr, root.right);
        }
    }
}
