/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    ArrayList<Integer> al = new ArrayList<>();

    public void method1(TreeNode node) {
        if (node == null) {
            return;
        }
        method1(node.left);
        al.add(node.val);
        method1(node.right);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return al;
    }
}
