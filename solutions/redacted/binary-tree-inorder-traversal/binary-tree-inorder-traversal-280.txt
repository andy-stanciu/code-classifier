/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        return trav(root, new Stack<Integer>());
    }

    public Stack<Integer> trav(TreeNode b, Stack<Integer> l) {
        if (b == null) {
            return l;
        }
        if (b.left != null) {
            trav(b.left, l);
        }
        l.push(b.val);
        if (b.right != null) {
            trav(b.right, l);
        }
        return l;
    }
}
