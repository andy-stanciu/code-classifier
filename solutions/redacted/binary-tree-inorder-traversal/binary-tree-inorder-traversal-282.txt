/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> result = new ArrayList();

    public List<Integer> method1(TreeNode root) {
        if (root == null) {
            return result;
        }
        method2(root);
        return result;
    }

    public void method2(TreeNode root) {
        if (root.left != null) {
            method2(root.left);
        }
        // Add the current node value
        result.add(root.val);
        // Traverse the right subtree
        if (root.right != null) {
            method2(root.right);
        }
    }
}
