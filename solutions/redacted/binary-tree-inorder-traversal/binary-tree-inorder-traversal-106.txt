class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        helper(root, list);
        return list;
    }

    public void helper(TreeNode node, List<Integer> list) {
        if (node != null) {
            helper(node.left, list);
            list.add(node.val);
            helper(node.right, list);
        }
    }
}
