/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void method1(TreeNode root, List<Integer> arr) {
        if (root == null) {
            return;
        }
        method1(root.left, arr);
        arr.add(root.val);
        method1(root.right, arr);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> lr = new ArrayList<>();
        method1(root, lr);
        return lr;
    }
}
