/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        method1(root, list);
        return list;
    }

    void method1(TreeNode node, List<Integer> list) {
        if (node == null) {
            return;
        }
        // Traverse the left subtree
        method1(node.left, list);
        // Visit the root node
        list.add(node.val);
        // Traverse the right subtree
        method1(node.right, list);
    }
}
