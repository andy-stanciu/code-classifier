class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ll = new ArrayList<>();
        method2(root, ll);
        return ll;
    }

    public void method2(TreeNode root, List<Integer> ll) {
        if (root == null)
            return;
        method2(root.left, ll);
        ll.add(root.val);
        method2(root.right, ll);
    }
}
