/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> L = new ArrayList<>();
        method2(root, L);
        return L;
    }

    public static void method2(TreeNode root, List<Integer> L) {
        if (root == null) {
            return;
        }
        method2(root.left, L);
        L.add(root.val);
        method2(root.right, L);
    }
}
