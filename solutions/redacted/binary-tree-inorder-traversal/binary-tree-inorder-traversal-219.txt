/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static void helper(TreeNode root, ArrayList<Integer> a) {
        if (root == null)
            return;
        helper(root.left, a);
        a.add(root.val);
        helper(root.right, a);
    }

    public List<Integer> method1(TreeNode root) {
        ArrayList<Integer> a = new ArrayList<>();
        helper(root, a);
        return a;
    }
}
