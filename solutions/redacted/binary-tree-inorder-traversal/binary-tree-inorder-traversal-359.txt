class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> nums = new ArrayList<>();
        method2(root, nums);
        return nums;
    }

    public void method2(TreeNode root, List<Integer> nums) {
        if (root == null) {
            return;
        }
        method2(root.left, nums);
        nums.add(root.val);
        method2(root.right, nums);
    }
}
