class Solution {

    void method1(TreeNode root, List<Integer> ans) {
        if (root == null)
            return;
        method1(root.left, ans);
        ans.add(root.val);
        method1(root.right, ans);
    }

    public List<Integer> method2(TreeNode root) {
        List<Integer> ans = new ArrayList<Integer>();
        method1(root, ans);
        return ans;
    }
}
