class Class1 {

    int val;

    Class1 left;

    Class1 right;

    Class1(int x) {
        val = x;
    }
}

class Solution {

    public List<Integer> method1(Class1 root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        method2(root, result);
        return result;
    }

    private void method2(Class1 node, List<Integer> result) {
        if (node == null) {
            return;
        }
        method2(node.left, result);
        result.add(node.val);
        method2(node.right, result);
    }
}
