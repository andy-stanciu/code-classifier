/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ls = new ArrayList<>();
        method2(root, ls);
        return ls;
    }

    public static void method2(TreeNode root, List<Integer> ls) {
        if (root == null)
            return;
        method2(root.left, ls);
        ls.add(root.val);
        method2(root.right, ls);
    }
}
