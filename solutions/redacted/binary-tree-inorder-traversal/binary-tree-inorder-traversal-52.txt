/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        method2(l, root);
        return l;
    }

    public static void method2(List<Integer> l, TreeNode node) {
        if (node == null)
            return;
        method2(l, node.left);
        l.add(node.val);
        method2(l, node.right);
        return;
    }
}
