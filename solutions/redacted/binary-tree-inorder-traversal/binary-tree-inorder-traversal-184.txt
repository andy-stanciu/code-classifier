class Solution {

    List<Integer> li = new ArrayList();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return li;
        // Traverse the left subtree
        if (root.left != null)
            method1(root.left);
        // Process the root node
        li.add(root.val);
        // Traverse the right subtree
        if (root.right != null)
            method1(root.right);
        return li;
    }
}
