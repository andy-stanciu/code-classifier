class Solution {

    List<Integer> res = new LinkedList<Integer>();

    public void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        res.add(root.val);
        method1(root.right);
    }

    public List<Integer> method2(TreeNode root) {
        method1(root);
        return res;
    }
}
