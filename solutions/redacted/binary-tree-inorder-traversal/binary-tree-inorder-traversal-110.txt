class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        solve(res, root);
        return res;
    }

    private void solve(List<Integer> res, TreeNode root) {
        if (root == null)
            return;
        // Traverse left node
        solve(res, root.left);
        // Traverse parent node
        res.add(root.val);
        // Traverse right node
        solve(res, root.right);
    }
}
