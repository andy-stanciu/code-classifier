class Solution {

    List<Integer> ans = new ArrayList<Integer>();

    public List<Integer> method1(TreeNode root) {
        method2(root);
        return ans;
    }

    void method2(TreeNode root) {
        if (root != null) {
            method2(root.left);
            ans.add(root.val);
            method2(root.right);
        }
    }
}
