/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        method2(root, l);
        return l;
    }

    private void method2(TreeNode root, List<Integer> l) {
        if (root != null) {
            method2(root.left, l);
            l.add(root.val);
            method2(root.right, l);
        }
    }
}
