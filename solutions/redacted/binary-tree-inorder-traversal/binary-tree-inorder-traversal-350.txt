/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        // Base case: if the root is null, return an empty list
        if (root == null) {
            return new ArrayList<>();
        }
        // Initialize the result list
        List<Integer> res = new ArrayList<>();
        // Recursively traverse the left subtree and add the result to the list
        res.addAll(method1(root.left));
        // Add the current node\'s value to the list
        res.add(root.val);
        // Recursively traverse the right subtree and add the result to the list
        res.addAll(method1(root.right));
        return res;
    }
}
