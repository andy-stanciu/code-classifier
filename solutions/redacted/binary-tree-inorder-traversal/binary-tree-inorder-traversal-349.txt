/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> a1 = new ArrayList<>();
        trav(root, a1);
        return a1;
    }

    public static void trav(TreeNode root, List<Integer> a1) {
        if (root == null)
            return;
        if (root != null) {
            trav(root.left, a1);
            a1.add(root.val);
            trav(root.right, a1);
        }
    }
}
