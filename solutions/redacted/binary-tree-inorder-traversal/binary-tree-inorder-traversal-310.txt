class Solution {

    private List<Integer> output = new ArrayList<>();

    public List<Integer> method1(TreeNode root) {
        if (root == null)
            return output;
        if (root.left != null)
            method1(root.left);
        output.add(root.val);
        if (root.right != null)
            method1(root.right);
        return output;
    }
}
