/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> aws = new ArrayList<>();
        helper(root, aws);
        return aws;
    }

    public void helper(TreeNode root, List aws) {
        if (root == null) {
            return;
        } else {
            helper(root.left, aws);
            aws.add(root.val);
            helper(root.right, aws);
        }
    }
}
