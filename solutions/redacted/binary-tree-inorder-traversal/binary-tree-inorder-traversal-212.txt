class Solution {

    public List<Integer> method1(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        TreeNode curr = root;
        while (curr != null) {
            if (curr.left != null) {
                //left hai
                TreeNode pred = curr.left;
                while (pred.right != null && pred.right != curr) {
                    pred = pred.right;
                }
                if (pred.right == null) {
                    pred.right = curr;
                    curr = curr.left;
                } else {
                    ans.add(curr.val);
                    curr = curr.right;
                    pred.right = null;
                }
            } else {
                //left nahi hai
                ans.add(curr.val);
                curr = curr.right;
            }
        }
        return ans;
    }
}
