class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int count = 0;
        StringBuilder magazineStringBuilder = new StringBuilder(magazine);
        for (int i = 0; i < param1.length(); i++) {
            char choosenChar = param1.charAt(i);
            for (int j = 0; j < magazineStringBuilder.length(); j++) {
                if (choosenChar == magazineStringBuilder.charAt(j)) {
                    magazineStringBuilder.deleteCharAt(j);
                    count++;
                    break;
                }
            }
        }
        if (count == param1.length()) {
            return true;
        }
        return false;
    }
}
