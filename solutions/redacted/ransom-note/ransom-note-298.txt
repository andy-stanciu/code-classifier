class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int rlen = param1.length();
        int mlen = magazine.length();
        if (rlen > mlen) {
            return false;
        }
        for (Character c : param1.toCharArray()) {
            if (magazine.contains(c.toString())) {
                magazine = magazine.replaceFirst(c.toString(), "");
            } else {
                return false;
            }
        }
        return true;
    }
}
