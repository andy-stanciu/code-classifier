class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> map = new HashMap<>();
        HashMap<Character, Integer> map2 = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            int val = map.getOrDefault(param1.charAt(i), 0);
            map.put(param1.charAt(i), val + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            int val = map2.getOrDefault(magazine.charAt(i), 0);
            map2.put(magazine.charAt(i), val + 1);
        }
        for (Map.Entry<Character, Integer> it : map.entrySet()) {
            if (!map2.containsKey(it.getKey()))
                return false;
            else if (map2.containsKey(it.getKey()) && map2.get(it.getKey()) < it.getValue())
                return false;
        }
        return true;
    }
}
