class Solution {

    public boolean canConstruct(String param1, String magazine) {
        //Not the most elegant but its checking if letter exist at any place in the magazine
        //Time complexity O(n2) cause it needs to check magazine it every time
        //Space complexity O(n) cause it had to make a new LinkedList that it uses to check it
        LinkedList<Character> magazineLetters = new LinkedList<>();
        for (int i = 0; i < magazine.length(); i++) {
            magazineLetters.add(magazine.charAt(i));
        }
        for (int i = 0; i < param1.length(); i++) {
            char var1 = param1.charAt(i);
            if (!magazineLetters.removeFirstOccurrence(var1)) {
                //this took me a moment since i did not know linked list has a method to remove the first occurance
                return false;
            }
        }
        return true;
    }
}
