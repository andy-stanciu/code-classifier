class Solution {

    public boolean canConstruct(String param1, String magazine) {
        StringBuilder myString = new StringBuilder(magazine);
        int i, size = param1.length();
        boolean result = false;
        for (i = 0; i < size; i++) {
            int index = myString.indexOf(String.valueOf(param1.charAt(i)));
            if (index != -1) {
                myString.deleteCharAt(index);
            } else {
                return false;
            }
        }
        return true;
    }
}
