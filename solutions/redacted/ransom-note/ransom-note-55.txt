class Solution {

    public boolean canConstruct(String param1, String magazine) {
        List<Integer> temp = new ArrayList<Integer>();
        if (param1.length() > magazine.length())
            return false;
        for (char c : param1.toCharArray()) {
            if (magazine.contains(c + "")) {
                // temp.add(magazine.indexOf(c+""));
                magazine = magazine.replaceFirst(String.valueOf(c), "1");
            } else
                return false;
        }
        return true;
    }
}
