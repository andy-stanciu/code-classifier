class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int len = param1.length();
        //int lenn=magazine.length();
        HashMap<Character, Integer> hashset = new HashMap<>();
        for (char c : param1.toCharArray()) {
            hashset.put(c, hashset.getOrDefault(c, 0) + 1);
        }
        HashMap<Character, Integer> hashset2 = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            hashset2.put(c, hashset2.getOrDefault(c, 0) + 1);
        }
        for (int i = 0; i < len; i++) {
            int check = hashset.get(param1.charAt(i));
            int contains = hashset2.getOrDefault(param1.charAt(i), 0);
            if (contains < check) {
                return false;
            }
        }
        return true;
    }
}
