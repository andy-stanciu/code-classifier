class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> r = freq(param1);
        Map<Character, Integer> m = freq(magazine);
        for (Map.Entry<Character, Integer> entries : r.entrySet()) {
            if (m.containsKey(entries.getKey())) {
                if (!(m.get(entries.getKey()) >= entries.getValue())) {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }

    public Map<Character, Integer> freq(String s) {
        Map<Character, Integer> fre = new HashMap<Character, Integer>();
        for (int i = 0; i < s.length(); i++) {
            if (fre.containsKey(s.charAt(i))) {
                int val = fre.get(s.charAt(i));
                fre.put(s.charAt(i), val + 1);
            } else {
                fre.put(s.charAt(i), 1);
            }
        }
        return fre;
    }
}
