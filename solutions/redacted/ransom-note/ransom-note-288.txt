class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length())
            return false;
        Map<Character, Integer> vocabulary = new HashMap();
        for (char ch : magazine.toCharArray()) {
            vocabulary.merge(ch, 1, Integer::sum);
        }
        for (char ch : param1.toCharArray()) {
            if (vocabulary.getOrDefault(ch, 0) > 0) {
                vocabulary.merge(ch, -1, Integer::sum);
            } else {
                return false;
            }
        }
        return true;
    }
}
