class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> charactersNeed = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            charactersNeed.put(c, charactersNeed.getOrDefault(c, 0) + 1);
        }
        for (char c : param1.toCharArray()) {
            if (charactersNeed.containsKey(c)) {
                charactersNeed.put(c, charactersNeed.get(c) - 1);
                if (charactersNeed.get(c) == 0) {
                    charactersNeed.remove(c);
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
