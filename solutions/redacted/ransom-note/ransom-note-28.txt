// 3 ms runtime - 90.73% faster
// 46.2 MB memory - 59.68% better
// Metrics don\'t matter
class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        // alm -> Available Letter Map
        int[] alm = new int[26];
        // Create an int alm map since both the strings only contain lowercase letters
        for (int letter : magazine.toCharArray()) ++alm[letter - 97];
        // Check the alm
        for (int letter : param1.toCharArray()) if (--alm[letter - 97] < 0)
            return false;
        // Return true since everything is fine
        return true;
    }
}
