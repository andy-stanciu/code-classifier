class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.indexOf(param1) != -1) {
            return true;
        } else {
            char[] var1 = param1.toCharArray();
            Arrays.sort(var1);
            char[] magazineArray = magazine.toCharArray();
            Arrays.sort(magazineArray);
            int rptr = 0;
            int mptr = 0;
            while (mptr < magazineArray.length) {
                if (var1[rptr] != magazineArray[mptr]) {
                    mptr++;
                } else {
                    rptr++;
                    mptr++;
                }
                if (rptr == var1.length)
                    return true;
            }
        }
        return false;
    }
}
