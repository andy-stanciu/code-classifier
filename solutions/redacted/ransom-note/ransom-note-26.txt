class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int var1 = param1.length(), lenMaga = magazine.length();
        HashMap<Character, Integer> var2 = new HashMap<>();
        HashMap<Character, Integer> mapMaga = new HashMap<>();
        for (int i = 0; i < var1; i++) {
            var2.put(param1.charAt(i), var2.getOrDefault(param1.charAt(i), 0) + 1);
        }
        for (int j = 0; j < lenMaga; j++) {
            mapMaga.put(magazine.charAt(j), mapMaga.getOrDefault(magazine.charAt(j), 0) + 1);
        }
        for (Map.Entry<Character, Integer> element : var2.entrySet()) {
            if (mapMaga.containsKey(element.getKey())) {
                if (mapMaga.get(element.getKey()) >= element.getValue()) {
                    continue;
                } else {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
