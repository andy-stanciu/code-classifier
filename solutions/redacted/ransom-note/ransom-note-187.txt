class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> map = new HashMap<>();
        // O(n)
        for (int i = 0; i < param1.length(); i++) {
            int count = map.containsKey(param1.charAt(i)) ? map.get(param1.charAt(i)) + 1 : 1;
            map.put(param1.charAt(i), count);
        }
        HashMap<Character, Integer> m = new HashMap<>();
        // O(n)
        for (int i = 0; i < magazine.length(); i++) {
            int count = m.containsKey(magazine.charAt(i)) ? m.get(magazine.charAt(i)) + 1 : 1;
            m.put(magazine.charAt(i), count);
        }
        for (Character c : map.keySet()) {
            if (!m.containsKey(c) || map.get(c) > m.get(c))
                return false;
        }
        return true;
    }
}
