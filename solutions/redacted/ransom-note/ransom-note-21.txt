class Solution {

    public boolean canConstruct(String param1, String magazine) {
        String r = param1, m = magazine, x = "", r2 = "";
        char[] r1 = r.toCharArray();
        char[] m1 = m.toCharArray();
        Arrays.sort(r1);
        for (int p = 0; p < r1.length; p++) r2 = r2 + r1[p];
        Arrays.sort(m1);
        int i = r1.length - 1, j = 0, k = 0;
        while (i >= 0 && k < m1.length) {
            if (r1[j] == m1[k]) {
                i--;
                j++;
                x = x + m1[k];
            }
            k++;
        }
        System.out.println(x + " " + r2);
        if (x.compareTo(r2) == 0)
            return true;
        else
            return false;
    }
}
