class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] ranArray = param1.toCharArray();
        Arrays.sort(ranArray);
        char[] magArray = magazine.toCharArray();
        Arrays.sort(magArray);
        int r = 0;
        int m = 0;
        while (r < ranArray.length && m < magArray.length) {
            if (ranArray[r] == magArray[m]) {
                r++;
            }
            m++;
        }
        return r == ranArray.length;
    }
}
