class Solution {

    String del(String word, char a) {
        String newWord = "";
        int count = 0;
        int wordLength = word.length();
        for (int i = 0; i < wordLength; i++) {
            if (word.charAt(i) == a && count == 0) {
                count++;
                continue;
            } else
                newWord += word.charAt(i);
        }
        return newWord;
    }

    public boolean canConstruct(String param1, String magazine) {
        int magLength = magazine.length();
        if (magLength < param1.length())
            return false;
        for (int i = 0; i < magLength; i++) {
            param1 = del(param1, magazine.charAt(i));
            if (param1 == "")
                return true;
        }
        return false;
    }
}
