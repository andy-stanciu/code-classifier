class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] countR = new int[26];
        int[] countM = new int[26];
        for (int i = 0; i < param1.length(); i++) {
            countR[param1.charAt(i) - 97]++;
        }
        for (int i = 0; i < magazine.length(); i++) {
            countM[magazine.charAt(i) - 97]++;
        }
        for (int i = 0; i < 26; i++) {
            if (countR[i] > countM[i]) {
                return false;
            }
        }
        return true;
    }
}
