class Solution {

    public static boolean canConstruct(String param1, String magazine) {
        char[] one = param1.toCharArray();
        char[] two = magazine.toCharArray();
        Arrays.sort(one);
        Arrays.sort(two);
        param1 = new String(one);
        magazine = new String(two);
        int i = 0;
        int j = 0;
        while (i < one.length && j < two.length) {
            if (one[i] == two[j]) {
                i++;
                j++;
            } else {
                j++;
            }
        }
        if (i < param1.length()) {
            return false;
        }
        return true;
    }
}
