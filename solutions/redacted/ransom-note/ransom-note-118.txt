class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> letterCounts = new HashMap<>();
        for (char letter : magazine.toCharArray()) {
            letterCounts.put(letter, letterCounts.getOrDefault(letter, 0) + 1);
        }
        for (char letter : param1.toCharArray()) {
            int count = letterCounts.getOrDefault(letter, 0);
            if (count == 0) {
                return false;
            }
            letterCounts.put(letter, count - 1);
        }
        return true;
    }
}
