class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> var1 = new HashMap<>();
        //update ransom map
        for (char r : param1.toCharArray()) {
            var1.put(r, var1.getOrDefault(r, 0) + 1);
        }
        for (char m : magazine.toCharArray()) {
            if (var1.containsKey(m)) {
                var1.put(m, var1.get(m) - 1);
                // remove if key has no value
                if (var1.get(m) == 0) {
                    var1.remove(m);
                }
            }
        }
        return var1.isEmpty();
    }
}
