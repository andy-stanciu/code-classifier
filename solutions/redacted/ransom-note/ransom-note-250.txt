class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> m = new HashMap<>();
        for (int i = 0; i < magazine.length(); ++i) {
            m.put(magazine.charAt(i), m.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < param1.length(); ++i) {
            m.put(param1.charAt(i), m.getOrDefault(param1.charAt(i), 0) - 1);
        }
        return m.values().stream().allMatch(e -> e >= 0);
    }
}
