class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length())
            return false;
        Map<Character, Integer> magazineCharMapCount = new HashMap<>();
        char[] magArr = magazine.toCharArray();
        char[] ranArr = param1.toCharArray();
        for (int i = 0; i < magArr.length; i++) {
            magazineCharMapCount.put(magArr[i], magazineCharMapCount.getOrDefault(magArr[i], 0) + 1);
        }
        for (int i = 0; i < ranArr.length; i++) {
            if (!magazineCharMapCount.containsKey(ranArr[i]))
                return false;
            int count = magazineCharMapCount.get(ranArr[i]);
            if (count == 0)
                return false;
            magazineCharMapCount.put(ranArr[i], count - 1);
        }
        return true;
    }
}
