class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] letters = param1.toCharArray();
        for (char var1 : letters) {
            if (magazine.indexOf(var1) >= 0) {
                magazine = magazine.replaceFirst(var1 + "", "");
            } else {
                return false;
            }
        }
        return true;
    }
}
