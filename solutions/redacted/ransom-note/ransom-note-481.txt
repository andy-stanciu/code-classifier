class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> hm = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            //iterating magazine
            char c = magazine.charAt(i);
            if (!hm.containsKey(c)) {
                //if hashmap contains key increment count of key or
                //if key does not exist add key with value of 1
                hm.put(c, 1);
            } else {
                hm.put(c, hm.get(c) + 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            //iterating ransomnote
            char d = param1.charAt(i);
            if (hm.containsKey(d) && hm.get(d) > 0) {
                //if  map contains key(alphabet) of ransomNote and count of key
                //is greater than 0  then reduce the size of value(count) by 1
                hm.put(d, hm.get(d) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
