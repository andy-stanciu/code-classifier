class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> indexMapping = new HashMap<>();
        for (Character character : param1.toCharArray()) {
            int index = magazine.indexOf(character, indexMapping.getOrDefault(character, 0));
            if (index == -1) {
                return false;
            }
            indexMapping.put(character, index + 1);
        }
        return true;
    }
}
