class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] c1 = new int[128];
        int[] c2 = new int[128];
        for (int i = 0; i < param1.length(); i++) {
            c1[param1.charAt(i)]++;
        }
        for (int i = 0; i < magazine.length(); i++) {
            c2[magazine.charAt(i)]++;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (c2[param1.charAt(i)] < c1[param1.charAt(i)]) {
                return false;
            }
        }
        return true;
    }
}
