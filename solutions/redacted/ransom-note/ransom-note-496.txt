class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> nbofcharacters = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char m = magazine.charAt(i);
            int count = nbofcharacters.getOrDefault(m, 0);
            nbofcharacters.put(m, ++count);
        }
        System.out.println(nbofcharacters);
        for (int i = 0; i < param1.length(); i++) {
            char r = param1.charAt(i);
            int currentcount = nbofcharacters.getOrDefault(r, 0);
            if (currentcount == 0) {
                return false;
            }
            nbofcharacters.put(r, --currentcount);
        }
        return true;
    }
}
