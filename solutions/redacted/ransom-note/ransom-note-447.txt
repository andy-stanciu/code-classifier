class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int k = 0, C = 0;
        char[] b = param1.toCharArray();
        char[] a = magazine.toCharArray();
        Arrays.sort(a);
        Arrays.sort(b);
        for (int i = 0; i < b.length; i++) {
            char c = b[i];
            for (int j = k; j < a.length; j++) {
                if (a[j] == c) {
                    k = j + 1;
                    C++;
                    break;
                }
            }
        }
        return C == b.length ? true : false;
    }
}
