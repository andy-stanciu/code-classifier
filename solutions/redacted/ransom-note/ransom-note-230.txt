class Solution {

    public boolean canConstruct(String param1, String magazine) {
        ransom_note_loop: for (int i = 0; i < param1.length(); i++) {
            for (int m = 0; m < magazine.length(); m++) {
                if (param1.charAt(i) == magazine.charAt(m)) {
                    magazine = magazine.replaceFirst(magazine.charAt(m) + "", "");
                    continue ransom_note_loop;
                }
            }
            return false;
        }
        return true;
    }
}
