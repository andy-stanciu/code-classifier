class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> magazineMap = new HashMap<>();
        for (char key : magazine.toCharArray()) {
            if (magazineMap.containsKey(key)) {
                magazineMap.put(key, magazineMap.get(key) + 1);
            } else {
                magazineMap.put(key, 1);
            }
        }
        for (char key : param1.toCharArray()) {
            if (magazineMap.getOrDefault(key, 0) != 0) {
                magazineMap.put(key, magazineMap.get(key) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
