class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] ch = param1.toCharArray();
        for (char c : ch) {
            int i = magazine.indexOf(c);
            if (i == -1)
                return false;
            magazine = magazine.substring(0, i) + magazine.substring(i + 1);
        }
        return true;
    }
}
