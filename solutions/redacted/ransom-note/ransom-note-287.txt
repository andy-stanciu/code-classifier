class Solution {

    public boolean canConstruct(String param1, String magazine) {
        //Brute Force - Maintain two hashmaps
        HashMap<Character, Integer> var1 = new HashMap<>();
        HashMap<Character, Integer> magazineMap = new HashMap<>();
        for (char ch : magazine.toCharArray()) {
            magazineMap.put(ch, magazineMap.getOrDefault(ch, 0) + 1);
        }
        for (char ch : param1.toCharArray()) {
            var1.put(ch, var1.getOrDefault(ch, 0) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            int val = var1.get(c);
            if (magazineMap.containsKey(c)) {
                if (val > magazineMap.get(c)) {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
