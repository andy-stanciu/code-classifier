class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> magazineMap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            magazineMap.put(magazine.charAt(i), magazineMap.getOrDefault(magazine.charAt(i), 0) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            char currChar = param1.charAt(i);
            if (!magazineMap.containsKey(currChar) || magazineMap.get(currChar) == 0) {
                return false;
            }
            magazineMap.put(currChar, magazineMap.get(currChar) - 1);
        }
        return true;
    }
}
