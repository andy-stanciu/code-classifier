class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] count = new int[256];
        for (int i = 0; i < magazine.length(); i++) count[(int) magazine.charAt(i)]++;
        for (int i = 0; i < param1.length(); i++) if (count[(int) param1.charAt(i)]-- == 0)
            return false;
        return true;
    }
}
