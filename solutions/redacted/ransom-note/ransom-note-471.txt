class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int nLength = param1.length();
        int mLength = magazine.length();
        int[] var1 = new int[256];
        int[] magFreq = new int[256];
        // boundary case
        if (nLength > mLength) {
            return false;
        }
        // loop to get character & mag frequencies
        for (int i = 0; i < mLength; i++) {
            // save as positive value
            magFreq[magazine.charAt(i)]++;
        }
        for (int j = 0; j < nLength; j++) {
            // save as neg value
            var1[param1.charAt(j)]--;
        }
        // add ransom note to magazine, if any number < 0, fails
        for (int k = 0; k < 26; k++) {
            int answer = var1[k + 97] + magFreq[k + 97];
            if (answer < 0)
                return false;
        }
        return true;
    }
}
