class Solution {

    public boolean canConstruct(String param1, String magazine) {
        String var1 = param1;
        String magazineTemp = magazine;
        while (var1.length() > 0) {
            magazineTemp = hasThisChar(var1.charAt(0), magazineTemp);
            if (magazineTemp == null && var1.length() > 0) {
                return false;
            }
            var1 = var1.substring(1);
        }
        return true;
    }

    private String hasThisChar(char charX, String currentMagazine) {
        String beginPart = "";
        String wordTemp = currentMagazine;
        while (wordTemp.length() > 0) {
            if (wordTemp.charAt(0) == charX) {
                return (beginPart + wordTemp.substring(1));
            }
            beginPart += wordTemp.charAt(0);
            wordTemp = wordTemp.substring(1);
        }
        return null;
    }
}
