class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length())
            return false;
        Map<Character, Long> var1 = param1.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        Map<Character, Long> magazineCharMap = magazine.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        return var1.entrySet().stream().allMatch(e -> e.getValue() <= magazineCharMap.getOrDefault(e.getKey(), 0L));
    }
}
