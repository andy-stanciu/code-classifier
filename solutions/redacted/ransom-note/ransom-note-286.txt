class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> hm = new HashMap<>();
        char[] word = param1.toCharArray();
        for (int i = 0; i < word.length; i++) {
            if (!hm.containsKey(word[i])) {
                hm.put(word[i], 1);
            } else
                hm.put(word[i], hm.get(word[i]) + 1);
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (hm.containsKey(magazine.charAt(i))) {
                if (hm.get(magazine.charAt(i)) > 1) {
                    hm.put(magazine.charAt(i), hm.get(magazine.charAt(i)) - 1);
                } else
                    hm.remove(magazine.charAt(i));
            }
        }
        if (hm.isEmpty())
            return true;
        return false;
    }
}
