class Solution {

    public boolean canConstruct(String param1, String magazine) {
        StringBuilder sb = new StringBuilder(magazine);
        if (param1.length() <= 0) {
            return false;
        }
        System.out.println(sb);
        for (int i = 0; i < param1.length(); i++) {
            if (sb.indexOf(String.valueOf(param1.charAt(i))) == -1) {
                return false;
            } else {
                sb.deleteCharAt(sb.indexOf(String.valueOf(param1.charAt(i))));
            }
        }
        return true;
    }
}
