class Solution {

    public boolean canConstruct(String param1, String magazine) {
        // a -> 0, b -> 1, ...
        int[] var1 = new int[26];
        for (int i = 0; i < param1.length(); i++) {
            var1[param1.charAt(i) - 97] += 1;
        }
        for (int j = 0; j < magazine.length(); j++) {
            var1[magazine.charAt(j) - 97] -= 1;
        }
        for (int k = 0; k < var1.length; k++) {
            if (var1[k] > 0) {
                return false;
            }
        }
        return true;
    }
}
