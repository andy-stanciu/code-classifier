class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> map = new HashMap<>();
        for (int index = 0; index < magazine.length(); index++) {
            map.put(magazine.charAt(index), map.getOrDefault(magazine.charAt(index), 0) + 1);
        }
        for (int index = 0; index < param1.length(); index++) {
            if (map.containsKey(param1.charAt(index))) {
                if (map.get(param1.charAt(index)) <= 0) {
                    return false;
                } else {
                    map.put(param1.charAt(index), map.get(param1.charAt(index)) - 1);
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
