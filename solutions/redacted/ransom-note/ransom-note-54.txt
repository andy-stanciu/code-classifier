class Solution {

    public boolean canConstruct(String param1, String magazine) {
        // Creating HashMap
        HashMap<Character, Integer> mp = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char c = magazine.charAt(i);
            // If it char not exists then put it by 1
            if (!mp.containsKey(c)) {
                mp.put(c, 1);
            } else // already exists in map increment it by 1
            {
                mp.put(c, mp.get(c) + 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            // If char is there then decrement it by 1
            if (mp.containsKey(c) && mp.get(c) > 0) {
                mp.put(c, mp.get(c) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
