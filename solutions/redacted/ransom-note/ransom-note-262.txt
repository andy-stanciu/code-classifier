class Solution {

    public boolean canConstruct(String param1, String magazine) {
        boolean equal = false;
        for (int i = 0; i < param1.length(); i++) {
            if (magazine.contains(param1.charAt(i) + "")) {
                magazine = magazine.replaceFirst(param1.charAt(i) + "", "");
                equal = true;
                continue;
            }
            equal = false;
            break;
        }
        return equal;
    }
}
