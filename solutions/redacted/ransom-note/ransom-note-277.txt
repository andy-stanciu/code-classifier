class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length()) {
            return false;
        }
        HashMap<Character, Integer> magazine_map = new HashMap<Character, Integer>();
        for (int i = 0; i < magazine.length(); i++) {
            if (magazine_map.containsKey(magazine.charAt(i))) {
                magazine_map.put(magazine.charAt(i), magazine_map.get(magazine.charAt(i)) + 1);
            } else {
                magazine_map.put(magazine.charAt(i), 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (!magazine_map.containsKey(param1.charAt(i))) {
                return false;
            } else {
                magazine_map.put(param1.charAt(i), magazine_map.get(param1.charAt(i)) - 1);
                if (magazine_map.get(param1.charAt(i)) < 0) {
                    return false;
                }
            }
        }
        return true;
    }
}
