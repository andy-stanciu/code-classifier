class Solution {

    public boolean canConstruct(String param1, String magazine) {
        // Create a HashMap to store the frequency of characters in magazine
        Map<Character, Integer> charFrequency = new HashMap<>();
        // Count the frequency of each character in magazine
        for (char c : magazine.toCharArray()) {
            charFrequency.put(c, charFrequency.getOrDefault(c, 0) + 1);
        }
        // Check if ransomNote can be constructed
        for (char c : param1.toCharArray()) {
            // If the character is not in the magazine or its frequency is zero, return false
            if (!charFrequency.containsKey(c) || charFrequency.get(c) == 0) {
                return false;
            }
            // Decrement the frequency of the character in the magazine
            charFrequency.put(c, charFrequency.get(c) - 1);
        }
        // If all characters in ransomNote are found in magazine, return true
        return true;
    }
}
