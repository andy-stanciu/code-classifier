class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] magazine1 = magazine.toCharArray();
        char[] var1 = param1.toCharArray();
        Arrays.sort(magazine1);
        Arrays.sort(var1);
        int index = 0;
        for (int i = 0; i < magazine1.length; i++) {
            if (var1[index] == magazine1[i]) {
                index++;
                if (index == var1.length)
                    return true;
            }
        }
        return false;
    }
}
