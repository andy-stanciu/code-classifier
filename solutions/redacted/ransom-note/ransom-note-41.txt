class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] count = new int[26];
        for (int i = 0; i < param1.length(); i++) count[param1.charAt(i) - 97]++;
        for (int i = 0; i < magazine.length(); i++) count[magazine.charAt(i) - 97]--;
        for (int i = 0; i < 26; i++) if (count[i] > 0)
            return false;
        return true;
    }
}
