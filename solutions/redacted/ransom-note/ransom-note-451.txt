class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> mp = new HashMap<>();
        for (Character x : magazine.toCharArray()) {
            mp.put(x, mp.getOrDefault(x, 0) + 1);
        }
        for (Character x : param1.toCharArray()) {
            if (mp.containsKey(x) && mp.get(x) > 0) {
                mp.put(x, mp.getOrDefault(x, 0) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
