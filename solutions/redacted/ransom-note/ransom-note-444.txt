class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int rn = param1.length();
        int mn = magazine.length();
        if (mn < rn)
            return false;
        Map<Character, Integer> s1 = new HashMap<>();
        Map<Character, Integer> s2 = new HashMap<>();
        for (char x : param1.toCharArray()) {
            s1.put(x, s1.getOrDefault(x, 0) + 1);
        }
        for (char x : magazine.toCharArray()) {
            s2.put(x, s2.getOrDefault(x, 0) + 1);
        }
        int c = 0;
        for (char x : s1.keySet()) {
            for (char y : s2.keySet()) {
                if (x == y && s2.get(y) >= s1.get(x)) {
                    c++;
                }
            }
        }
        rn = s1.size();
        return c == rn;
    }
}
