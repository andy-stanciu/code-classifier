class Solution {

    public boolean canConstruct(String param1, String magazine) {
        List<String> var1 = Arrays.stream(param1.split("")).toList();
        List<String> magazines = new ArrayList<>(Arrays.stream(magazine.split("")).toList());
        for (String r : var1) {
            if (magazines.contains(r)) {
                magazines.remove(r);
            } else {
                return false;
            }
        }
        return true;
    }
}
