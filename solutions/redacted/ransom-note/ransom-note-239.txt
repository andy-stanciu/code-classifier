class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> maps = new HashMap<>();
        // Store characters from magazine and their counts in a map
        for (char c : magazine.toCharArray()) {
            maps.put(c, maps.getOrDefault(c, 0) + 1);
        }
        // Check if each character in ransomNote can be constructed from magazine
        for (char c : param1.toCharArray()) {
            if (maps.containsKey(c)) {
                int count = maps.get(c);
                if (count == 1) {
                    // Remove the character from map once used
                    maps.remove(c);
                } else {
                    // Decrease the count
                    maps.put(c, count - 1);
                }
            } else {
                // If character not found in magazine, return  false
                return false;
            }
        }
        // All characters found and counts are correct
        return true;
    }
}
