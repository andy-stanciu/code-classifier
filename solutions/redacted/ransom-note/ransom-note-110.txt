class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length()) {
            return false;
        }
        HashMap<Character, Integer> magazineHashMap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char m = magazine.charAt(i);
            int currentCount = magazineHashMap.getOrDefault(m, 0);
            magazineHashMap.put(m, currentCount + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            char r = param1.charAt(i);
            int currentCount = magazineHashMap.getOrDefault(r, 0);
            if (currentCount == 0)
                return false;
            magazineHashMap.put(r, currentCount - 1);
        }
        return true;
    }
}
