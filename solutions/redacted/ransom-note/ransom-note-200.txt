class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> alphaCount = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            if (!alphaCount.containsKey(c)) {
                alphaCount.put(c, 1);
            } else {
                alphaCount.put(c, alphaCount.get(c) + 1);
            }
        }
        for (char c : param1.toCharArray()) {
            if (alphaCount.containsKey(c) && alphaCount.get(c) > 0) {
                alphaCount.put(c, alphaCount.get(c) - 1);
            } else {
                return false;
            }
        }
        return true;
    }
}
