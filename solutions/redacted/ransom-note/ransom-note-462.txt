class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] first = new int[27];
        int[] sescond = new int[27];
        for (int i = 0; i < param1.length(); i++) {
            first[param1.charAt(i) - 96]++;
        }
        for (int i = 0; i < magazine.length(); i++) {
            sescond[magazine.charAt(i) - 96]++;
        }
        for (int i = 1; i < 27; i++) {
            if (first[i] != 0 && (first[i] > sescond[i])) {
                //System.out.println("false");
                return false;
            }
        }
        return true;
    }
}
