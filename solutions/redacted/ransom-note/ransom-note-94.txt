class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Queue<Character> var1 = new LinkedList();
        //add to queue
        for (Character c : param1.toCharArray()) {
            var1.add(c);
        }
        System.out.println(var1);
        //now iterate over magazine, take one letter at a time and remove from queue if found,
        for (Character c : magazine.toCharArray()) {
            if (var1.contains(c)) {
                var1.remove(c);
            }
        }
        if (var1.isEmpty()) {
            return true;
        } else {
            return false;
        }
    }
}
