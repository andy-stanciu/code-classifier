class Solution {

    public boolean canConstruct(String param1, String magazine) {
        // step1: generate HashMap table
        HashMap<Character, Integer> hashTable = new HashMap<Character, Integer>();
        for (int i = 0; i < magazine.length(); i++) {
            Character eachChar = magazine.charAt(i);
            Integer eachCharValue = hashTable.getOrDefault(eachChar, 0);
            hashTable.put(eachChar, eachCharValue + 1);
        }
        // iterate ransomNote for each occurences as hashTable key
        for (int i = 0; i < param1.length(); i++) {
            Character eachChar = param1.charAt(i);
            if (hashTable.getOrDefault(eachChar, 0) == 0)
                return false;
            else {
                Integer minusOne = hashTable.get(eachChar) - 1;
                hashTable.put(eachChar, minusOne);
            }
        }
        return true;
    }
}
