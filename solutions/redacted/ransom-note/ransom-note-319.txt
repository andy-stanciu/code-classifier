class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] magazineArray = magazine.toCharArray();
        char[] var1 = param1.toCharArray();
        int count = 0;
        for (int i = 0; i < var1.length; i++) {
            for (int j = 0; j < magazineArray.length; j++) {
                if (var1[i] == magazineArray[j]) {
                    count++;
                    magazineArray[j] = (char) -1;
                    break;
                }
            }
        }
        if (count == param1.length()) {
            return true;
        }
        return false;
        // return (count==ransomNoteArray.length);
    }
}
