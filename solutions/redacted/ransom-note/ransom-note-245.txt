class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> checkmap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char c = magazine.charAt(i);
            if (!checkmap.containsKey(c))
                checkmap.put(c, 1);
            else
                checkmap.put(c, checkmap.get(c) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            if (checkmap.containsKey(c) && checkmap.get(c) > 0)
                checkmap.put(c, checkmap.get(c) - 1);
            else
                return false;
        }
        return true;
    }
}
