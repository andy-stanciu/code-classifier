class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            return false;
        }
        HashMap<Character, Integer> var1 = new HashMap<>();
        HashMap<Character, Integer> magzineMap = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (var1.containsKey(param1.charAt(i))) {
                var1.put(param1.charAt(i), (var1.get(param1.charAt(i))) + 1);
            } else {
                var1.put(param1.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (magzineMap.containsKey(magazine.charAt(i))) {
                magzineMap.put(magazine.charAt(i), (magzineMap.get(magazine.charAt(i))) + 1);
            } else {
                magzineMap.put(magazine.charAt(i), 1);
            }
        }
        for (Character ch : var1.keySet()) {
            if (magzineMap.get(ch) != null) {
                if ((Integer.valueOf(var1.get(ch)) > Integer.valueOf(magzineMap.get(ch)))) {
                    return false;
                }
            } else if (magzineMap.get(ch) == null) {
                return false;
            }
        }
        return true;
    }
}
