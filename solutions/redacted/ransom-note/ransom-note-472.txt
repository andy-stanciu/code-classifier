class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int i = 0;
        int j = 0;
        char[] ran = param1.toCharArray();
        char[] mag = magazine.toCharArray();
        Arrays.sort(ran);
        Arrays.sort(mag);
        while (j < mag.length) {
            if (ran[i] == mag[j]) {
                i++;
            }
            if (i == ran.length)
                return true;
            j++;
        }
        return false;
    }
}
