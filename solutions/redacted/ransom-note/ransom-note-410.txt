class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int i = 0, j = 0;
        char[] var1 = param1.toCharArray();
        Arrays.sort(var1);
        char[] arrmaga = magazine.toCharArray();
        Arrays.sort(arrmaga);
        while (i < param1.length() && j < magazine.length()) {
            if (var1[i] == arrmaga[j++]) {
                i++;
            }
        }
        return i == param1.length();
    }
}
