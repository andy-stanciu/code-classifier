class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] a = new int[26];
        for (char c : param1.toCharArray()) {
            int i = magazine.indexOf(c, a[c % 26]);
            if (i == -1)
                return false;
            a[c % 26] = i + 1;
        }
        return true;
    }
}
