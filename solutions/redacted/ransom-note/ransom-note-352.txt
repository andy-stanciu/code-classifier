class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Long> charToFreq = magazine.chars().mapToObj(codePoint -> (char) codePoint).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        return param1.chars().mapToObj(codePoint -> (char) codePoint).allMatch(character -> charToFreq.merge(character, -1L, Long::sum) >= 0);
    }
}
