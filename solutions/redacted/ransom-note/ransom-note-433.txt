class Solution {

    public boolean canConstruct(String param1, String magazine) {
        char[] arr = param1.toCharArray();
        for (int c = 0; c < arr.length; c++) {
            if (!magazine.contains("" + arr[c]))
                return false;
            magazine = magazine.replaceFirst(("" + arr[c]), "");
        }
        System.gc();
        return true;
    }
}
