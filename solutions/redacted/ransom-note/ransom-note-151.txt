class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Hashtable<Character, Integer> hmR = new Hashtable<Character, Integer>();
        Hashtable<Character, Integer> hmM = new Hashtable<Character, Integer>();
        for (int i = 0; i < param1.length(); i++) {
            if (hmR.containsKey(param1.charAt(i))) {
                hmR.put(param1.charAt(i), hmR.get(param1.charAt(i)) + 1);
            } else {
                hmR.put(param1.charAt(i), 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (hmM.containsKey(magazine.charAt(i))) {
                hmM.put(magazine.charAt(i), hmM.get(magazine.charAt(i)) + 1);
            } else {
                hmM.put(magazine.charAt(i), 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if ((hmM.containsKey(param1.charAt(i)))) {
                if ((hmR.get(param1.charAt(i)) > hmM.get(param1.charAt(i)))) {
                    return false;
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
