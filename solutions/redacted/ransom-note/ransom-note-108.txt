class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> var1 = new HashMap<>();
        HashMap<Character, Integer> magazineMap = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            if (!var1.containsKey(param1.charAt(i))) {
                var1.put(param1.charAt(i), 1);
            } else {
                var1.put(param1.charAt(i), var1.get(param1.charAt(i)) + 1);
            }
        }
        for (int i = 0; i < magazine.length(); i++) {
            if (!magazineMap.containsKey(magazine.charAt(i))) {
                magazineMap.put(magazine.charAt(i), 1);
            } else {
                magazineMap.put(magazine.charAt(i), magazineMap.get(magazine.charAt(i)) + 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (!magazineMap.containsKey(param1.charAt(i))) {
                return false;
            } else if (magazineMap.get(param1.charAt(i)) < var1.get(param1.charAt(i))) {
                return false;
            }
            // && magazineMap.get(ransomNote.charAt(i))==)
        }
        return true;
    }
}
