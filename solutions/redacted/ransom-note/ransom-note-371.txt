class Solution {

    public boolean canConstruct(String param1, String magazine) {
        for (int i = 0; i < param1.length(); i++) {
            String aChar = String.valueOf(param1.charAt(i));
            if (!magazine.contains(aChar)) {
                return false;
            }
            magazine = magazine.replaceFirst(aChar, "");
        }
        return true;
    }
}
