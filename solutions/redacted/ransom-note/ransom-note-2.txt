class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            return false;
        }
        Set<Character> var1 = new HashSet<>();
        for (char c : param1.toCharArray()) {
            var1.add(c);
        }
        for (char c : var1) {
            if (countOccurrences(magazine, c) < countOccurrences(param1, c)) {
                return false;
            }
        }
        return true;
    }

    private int countOccurrences(String str, char c) {
        return (int) str.chars().filter(ch -> ch == c).count();
    }
}
