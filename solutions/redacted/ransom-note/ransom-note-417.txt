class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (magazine.length() < param1.length()) {
            return false;
        }
        Map<Character, Integer> magazineMap = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            char m = magazine.charAt(i);
            int currentCount = magazineMap.getOrDefault(m, 0);
            magazineMap.put(m, ++currentCount);
        }
        for (int i = 0; i < param1.length(); i++) {
            char r = param1.charAt(i);
            int currentCount = magazineMap.getOrDefault(r, 0);
            if (currentCount == 0) {
                System.gc();
                return false;
            }
            magazineMap.put(r, --currentCount);
        }
        System.gc();
        return true;
    }
}
