class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1 == null || magazine == null) {
            return false;
        }
        if (param1.length() > magazine.length()) {
            return false;
        }
        char[] var1 = param1.toCharArray();
        char[] magazineArray = magazine.toCharArray();
        Map<Character, Integer> magazineMap = new HashMap();
        for (int i = 0; i < magazineArray.length; i++) {
            magazineMap.put(magazineArray[i], magazineMap.getOrDefault(magazineArray[i], 0) + 1);
        }
        for (int i = 0; i < var1.length; i++) {
            if (magazineMap.get(var1[i]) == null || magazineMap.get(var1[i]) == 0) {
                return false;
            } else {
                magazineMap.put(var1[i], magazineMap.get(var1[i]) - 1);
            }
        }
        return true;
    }
}
