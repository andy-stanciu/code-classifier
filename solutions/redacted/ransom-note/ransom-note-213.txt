class Solution {

    public boolean canConstruct(String ran, String mag) {
        HashMap<Character, Integer> mp = new HashMap<>();
        HashMap<Character, Integer> mp1 = new HashMap<>();
        boolean flag = false;
        for (int i = 0; i < mag.length(); i++) mp.put(mag.charAt(i), mp.getOrDefault(mag.charAt(i), 0) + 1);
        for (int i = 0; i < ran.length(); i++) mp1.put(ran.charAt(i), mp1.getOrDefault(ran.charAt(i), 0) + 1);
        for (char key : mp1.keySet()) {
            int rans = mp1.getOrDefault(key, 0);
            int magn = mp.getOrDefault(key, 0);
            if (!mp.containsKey(key) || rans > magn) {
                return false;
            }
        }
        return true;
    }
}
