class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int n1 = param1.length();
        int n2 = magazine.length();
        Map<Character, Integer> H1 = new HashMap<>();
        // Map<Character,Integer>H2=new HashMap<>();
        for (int i = 0; i < n2; i++) {
            if (!H1.containsKey(magazine.charAt(i))) {
                H1.put(magazine.charAt(i), 0);
            }
            H1.put(magazine.charAt(i), H1.get(magazine.charAt(i)) + 1);
        }
        System.out.print(H1);
        for (int i = 0; i < n1; i++) {
            char t = param1.charAt(i);
            if (H1.containsKey(t)) {
                H1.put(t, H1.get(t) - 1);
                if (H1.get(t) < 1) {
                    H1.remove(t);
                }
            } else {
                System.out.print(H1);
                return false;
            }
        }
        return true;
    }
}
