class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> map = new HashMap<>();
        char[] ch = magazine.toCharArray();
        char[] ch1 = param1.toCharArray();
        for (char ele : ch) {
            if (map.containsKey(ele)) {
                int f = map.get(ele);
                map.put(ele, f + 1);
            } else
                map.put(ele, 1);
        }
        int c = 0;
        for (char ele : ch1) {
            if (map.containsKey(ele)) {
                int f = map.get(ele);
                if (f == 1)
                    map.remove(ele);
                else
                    map.put(ele, f - 1);
                c++;
            }
        }
        return c == param1.length();
    }
}
