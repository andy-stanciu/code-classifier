class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < param1.length(); i++) {
            map.put(param1.charAt(i), map.getOrDefault(param1.charAt(i), 0) + 1);
        }
        for (int j = 0; j < magazine.length(); j++) {
            if (map.containsKey(magazine.charAt(j))) {
                map.put(magazine.charAt(j), map.get(magazine.charAt(j)) - 1);
                if (map.get(magazine.charAt(j)) == 0) {
                    map.remove(magazine.charAt(j));
                }
            }
        }
        return map.size() == 0;
    }
}
