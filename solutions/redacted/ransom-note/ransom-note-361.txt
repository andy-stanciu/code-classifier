class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> magazineMap = new HashMap<>();
        for (char c : magazine.toCharArray()) {
            magazineMap.put(c, magazineMap.getOrDefault(c, 0) + 1);
        }
        HashMap<Character, Integer> var1 = new HashMap<>();
        for (char c : param1.toCharArray()) {
            var1.put(c, var1.getOrDefault(c, 0) + 1);
        }
        boolean containsSequence = true;
        for (char c : var1.keySet()) {
            if (!magazineMap.containsKey(c) || magazineMap.get(c) < var1.get(c)) {
                containsSequence = false;
                break;
            }
        }
        return containsSequence;
    }
}
// TC: O(m + n), SC: O(m + n)
