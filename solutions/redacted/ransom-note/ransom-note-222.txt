class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> magMap = new HashMap<Character, Integer>();
        for (int i = 0; i < magazine.length(); i++) {
            int count = magMap.getOrDefault(magazine.charAt(i), 0);
            magMap.put(magazine.charAt(i), ++count);
        }
        for (int i = 0; i < param1.length(); i++) {
            if (magMap.containsKey(param1.charAt(i)) && magMap.get(param1.charAt(i)) != 0) {
                int count = magMap.get(param1.charAt(i));
                magMap.put(param1.charAt(i), --count);
            } else {
                return false;
            }
        }
        return true;
    }
}
