class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<String, Integer> magazineMap = new HashMap<>();
        Map<String, Integer> var1 = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) {
            String key = String.valueOf(magazine.charAt(i));
            magazineMap.put(key, magazineMap.getOrDefault(key, 0) + 1);
        }
        for (int i = 0; i < param1.length(); i++) {
            String key = String.valueOf(param1.charAt(i));
            var1.put(key, var1.getOrDefault(key, 0) + 1);
        }
        for (Map.Entry<String, Integer> var2 : var1.entrySet()) {
            if (var2.getValue() > 1) {
                if (!magazineMap.containsKey(var2.getKey()) || magazineMap.get(var2.getKey()) < var2.getValue())
                    return false;
            } else {
                if (!magazineMap.containsKey(var2.getKey())) {
                    return false;
                }
            }
        }
        return true;
    }
}
