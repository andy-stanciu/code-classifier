class Solution {

    public boolean canConstruct(String param1, String magzine) {
        HashMap<Character, Integer> mp = new HashMap<>();
        for (int i = 0; i < magzine.length(); i++) {
            if (mp.containsKey(magzine.charAt(i))) {
                int temp = mp.get(magzine.charAt(i));
                mp.put(magzine.charAt(i), ++temp);
            } else {
                mp.put(magzine.charAt(i), 1);
            }
        }
        for (int i = 0; i < param1.length(); i++) {
            if (mp.containsKey(param1.charAt(i))) {
                if (mp.get(param1.charAt(i)) == 1) {
                    mp.remove(param1.charAt(i));
                } else {
                    int temp = mp.get(param1.charAt(i));
                    mp.put(param1.charAt(i), --temp);
                }
            } else {
                return false;
            }
        }
        return true;
    }
}
