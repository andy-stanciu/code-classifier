class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] arr = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            arr[magazine.charAt(i) - 97]++;
        }
        for (int i = 0; i < param1.length(); i++) {
            if (arr[param1.charAt(i) - 97] == 0)
                return false;
            arr[param1.charAt(i) - 97]--;
        }
        return true;
    }
}
