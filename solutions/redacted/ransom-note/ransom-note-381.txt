class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            //Check to see if ransomNote is larger than magazine
            return false;
        }
        //Create an integer array to hold occurrences of letters in the magazine
        int[] alphabet = new int[26];
        for (int i = 0; i < magazine.length(); i++) {
            //Add 1 to every alphabetic position that a letter occurs
            alphabet[magazine.charAt(i) - 97] += 1;
        }
        for (int i = 0; i < param1.length(); i++) {
            //Loop to cycle through the ransom note
            if (alphabet[param1.charAt(i) - 97] == 0) {
                //if there are no remaining letters in that bucket, return false
                return false;
            } else {
                //else, remove one letter from that bucket
                alphabet[param1.charAt(i) - 97] -= 1;
            }
        }
        //if we cycle through the whole note, return true
        return true;
    }
}
