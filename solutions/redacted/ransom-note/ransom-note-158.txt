class Solution {

    public boolean canConstruct(String param1, String magazine) {
        int[] letters = new int[26];
        for (char c : magazine.toCharArray()) {
            letters[c - 97]++;
        }
        for (int i = 0; i < param1.length(); i++) {
            char c = param1.charAt(i);
            letters[c - 97]--;
            if (letters[c - 97] < 0)
                return false;
        }
        return true;
    }
}
