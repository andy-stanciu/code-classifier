class Solution {

    public boolean canConstruct(String param1, String magazine) {
        if (param1.length() > magazine.length()) {
            return false;
        }
        char[] magazineCharArray = magazine.toCharArray();
        char[] var1 = param1.toCharArray();
        Map<Character, Integer> charIndexMap = new HashMap<>();
        boolean isCheckSuccessful = false;
        for (int i = 0; i < var1.length; i++) {
            isCheckSuccessful = false;
            char var2 = var1[i];
            for (int j = 0; j < magazineCharArray.length; j++) {
                Integer var3 = charIndexMap.get(var2);
                if (var3 != null) {
                    j = var3 + 1;
                }
                char magazineChar = magazineCharArray[j];
                if (var2 == magazineChar) {
                    isCheckSuccessful = true;
                    charIndexMap.put(var2, j);
                    break;
                } else {
                    charIndexMap.remove(var2);
                }
            }
            if (!isCheckSuccessful) {
                break;
            }
        }
        return isCheckSuccessful;
    }
}
