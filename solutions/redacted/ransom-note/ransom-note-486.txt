class Solution {

    public boolean canConstruct(String r, String m) {
        int r_len = r.length();
        int m_len = m.length();
        char[] r_char = r.toCharArray();
        char[] m_char = m.toCharArray();
        Arrays.sort(r_char);
        Arrays.sort(m_char);
        int r_temp = 0;
        int m_temp = 0;
        for (int i = m_temp; i < m_len; i++) {
            if (r_char[r_temp] == m_char[i])
                r_temp++;
            if (r_temp == r.length())
                return true;
            m_temp++;
        }
        return false;
    }
}
