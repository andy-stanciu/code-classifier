class Solution {

    public boolean canConstruct(String param1, String magazine) {
        Map<Character, Integer> mgz = new HashMap<>();
        for (int i = 0; i < magazine.length(); i++) mgz.put(magazine.charAt(i), mgz.getOrDefault(magazine.charAt(i), 0) + 1);
        for (int i = 0; i < param1.length(); i++) {
            if (mgz.containsKey(param1.charAt(i))) {
                mgz.put(param1.charAt(i), mgz.get(param1.charAt(i)) - 1);
                if (mgz.get(param1.charAt(i)) != null && mgz.get(param1.charAt(i)) < 0)
                    return false;
            } else
                return false;
        }
        return true;
    }
}
