class Solution {

    public static boolean canConstruct(String param1, String magazine) {
        HashSet<Character> list = new HashSet<>();
        for (char c : param1.toCharArray()) list.add(c);
        for (Character c : list) {
            int count1 = countChar(param1, c);
            int count2 = countChar(magazine, c);
            if (count1 > count2)
                return false;
        }
        return true;
    }

    public static int countChar(String str, char c) {
        int count = 0;
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == c) {
                count++;
            }
        }
        return count;
    }
}
