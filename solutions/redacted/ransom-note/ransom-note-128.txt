class Solution {

    public boolean canConstruct(String param1, String magazine) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (char letter : param1.toCharArray()) {
            if (!map.containsKey(letter)) {
                map.put(letter, 1);
            } else if (map.containsKey(letter)) {
                int currentVal = map.get(letter);
                map.put(letter, currentVal + 1);
            }
        }
        for (char letter : magazine.toCharArray()) {
            if (map.containsKey(letter)) {
                int currentVal = map.get(letter);
                map.put(letter, currentVal - 1);
            }
        }
        for (char key : map.keySet()) {
            if (map.get(key) > 0) {
                return false;
            }
        }
        return true;
    }
}
