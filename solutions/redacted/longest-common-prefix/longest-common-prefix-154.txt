class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs, (str1, str2) -> str1.length() - str2.length());
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            if (var1.length() == 0)
                return "";
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
