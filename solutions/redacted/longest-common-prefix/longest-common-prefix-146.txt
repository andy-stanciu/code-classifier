class Solution {

    public String method1(String[] strs) {
        //Sort : [Most prefix strings ------to ------Least common prefix strings]
        Arrays.sort(strs);
        String res = "";
        //supposedly Most Common Prefix
        String first = strs[0];
        // Least common prefix
        String last = strs[strs.length - 1];
        for (int i = 0; i < first.length(); i++) {
            if (first.charAt(i) == last.charAt(i))
                res = res + first.charAt(i);
            else
                //no point in checking remaining part of string if prefix dont match
                return res;
        }
        return res;
    }
}
