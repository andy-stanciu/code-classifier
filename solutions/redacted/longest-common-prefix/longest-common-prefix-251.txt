class Solution {

    public String method1(String[] str) {
        if (str.length == 0)
            return "";
        String ans = str[0];
        int j = 0;
        for (int i = 1; i < str.length; i++) {
            j = 0;
            while (j < ans.length() && j < str[i].length()) {
                if (ans.charAt(j) != str[i].charAt(j)) {
                    ans = ans.substring(0, j);
                    break;
                }
                j++;
            }
            ans = ans.substring(0, j);
        }
        return ans;
    }
}
