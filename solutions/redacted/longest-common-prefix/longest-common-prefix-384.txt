class Solution {

    public String method1(String[] arr) {
        String var1 = arr[0];
        for (int i = 0; i < arr.length; i++) {
            while (arr[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
                if (var1.isEmpty()) {
                    return var1;
                }
            }
        }
        return var1;
    }
}
