class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        String first = strs[0];
        String last = strs[strs.length - 1];
        int m = first.length();
        int n = last.length();
        int maxLen = Math.min(m, n);
        int count = 0;
        for (int i = 0; i < maxLen; i++) {
            if (first.charAt(i) == last.charAt(i)) {
                count++;
            } else {
                break;
            }
        }
        return first.substring(0, count);
    }
}
