class Solution {

    public String method1(String[] strs) {
        HashMap<String, Integer> map = new HashMap<>();
        String ans = "";
        for (int i = 0; i < strs.length; i++) {
            String st = strs[i];
            int size = st.length();
            for (int j = 1; j <= size; j++) {
                String sub = st.substring(0, j);
                map.put(sub, map.getOrDefault(sub, 0) + 1);
                String curreentstring = sub;
                int sizeans = 0;
                if (map.get(sub) == strs.length && sub.length() > sizeans) {
                    ans = curreentstring;
                    curreentstring = sub;
                    sizeans = sub.length();
                }
            }
        }
        return ans;
    }
}
