class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        String s1 = strs[0];
        String s2 = strs[strs.length - 1];
        int i = 0;
        int min = Math.min(s1.length(), s2.length());
        while (i < min && s1.charAt(i) == s2.charAt(i)) {
            i++;
        }
        if (i == 0) {
            return "";
        }
        return s1.substring(0, i);
    }
}
