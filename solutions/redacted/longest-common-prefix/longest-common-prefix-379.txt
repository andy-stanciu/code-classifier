class Solution {

    public String method1(String[] strs) {
        int len = strs.length;
        //List<String> list = new ArrayList<>();
        String str = "";
        Arrays.sort(strs);
        if (len == 0) {
            return "";
        } else if (len == 1) {
            return strs[0];
        } else {
            String first = strs[0];
            String second = strs[len - 1];
            int n1 = first.length();
            int n2 = second.length();
            int min = Math.min(n1, n2);
            for (int i = 0; i < min; i++) {
                if (first.charAt(i) != second.charAt(i)) {
                    break;
                } else {
                    str += String.valueOf(first.charAt(i));
                }
            }
        }
        return str;
