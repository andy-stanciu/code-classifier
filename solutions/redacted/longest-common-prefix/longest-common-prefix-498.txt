class Solution {

    public String method1(String[] strs) {
        // Initialize the prefix as the first string in the array
        String var1 = strs[0];
        // Iterate through the array starting from the second string
        for (int i = 1; i < strs.length; i++) {
            // Check if the current string does not start with the current prefix
            while (strs[i].indexOf(var1) != 0) {
                // Shorten the prefix by removing the last character
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        // Return the longest prefix
        return var1;
    }
}
