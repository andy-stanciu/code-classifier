class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        int l = strs.length;
        for (int i = 0; i < l; i++) {
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
                if (var1.isEmpty()) {
                    return "";
                }
            }
        }
        return var1;
    }
}
