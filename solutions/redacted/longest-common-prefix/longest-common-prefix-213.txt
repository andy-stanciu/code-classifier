class Solution {

    public String method1(String[] strs) {
        int n = strs.length;
        /**
         * EDGE CASE : Single String
         */
        if (n == 1)
            return strs[0];
        /**
         * First word
         */
        String firstWord = strs[0];
        int index = 0;
        /**
         * Match each character of first word to each corresponding
         *        index character of the rest of the words, If at any point
         *        characters don\'t match, We have found the LCP in range of  word[0 -> index-1]
         */
        for (int j = 0; j < firstWord.length(); j++) {
            /**
             * Process each word\'s index character
             */
            for (int i = 1; i < n; i++) {
                String word = strs[i];
                /**
                 * If the chars don\'t match or the currentWord has exhausted,
                 *                Return the result
                 */
                if (word.length() == index || firstWord.charAt(index) != word.charAt(index))
                    return firstWord.substring(0, index);
            }
            index++;
        }
        return strs[0];
    }
}
