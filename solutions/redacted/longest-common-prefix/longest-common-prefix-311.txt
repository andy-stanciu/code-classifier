class Solution {

    public String method1(String[] strs) {
        // If the input array is empty, return an empty string
        if (strs.length == 0) {
            return "";
        }
        // Assume the first string as the initial prefix
        String var1 = strs[0];
        // Loop through the rest of the strings in the array
        for (int i = 1; i < strs.length; i++) {
            // Check if the current string starts with the current prefix
            // If it doesn\'t, reduce the prefix size by one character at a time
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
                // If the prefix becomes empty, return an empty string
                if (var1.isEmpty()) {
                    return "";
                }
            }
        }
        return var1;
    }
}
