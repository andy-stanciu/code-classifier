class Solution {

    public String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            // Return an empty string if the input is invalid.
            return "";
        }
        // Assume the first string is the common prefix.
        String var1 = strs[0];
        // Compare the prefix with each string in the array.
        for (int i = 1; i < strs.length; i++) {
            // Shorten the prefix until it matches the current string.
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
                if (var1.isEmpty()) {
                    // If the prefix is empty, no common prefix exists.
                    return "";
                }
            }
        }
        return var1;
    }
}
