class Solution {

    public String method1(String[] strs) {
        java.util.Arrays.sort(strs);
        int i = 0;
        String first = strs[0];
        String last = strs[strs.length - 1];
        String n = "";
        while (i < first.length()) {
            if (first.charAt(i) == last.charAt(i)) {
                n += first.charAt(i);
                i++;
            } else {
                break;
            }
        }
        return n;
    }
}
