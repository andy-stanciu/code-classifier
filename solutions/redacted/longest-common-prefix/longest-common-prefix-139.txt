class Solution {

    public String method1(String[] strs) {
        int len = strs.length;
        //If array has only one string
        if (len == 1)
            return strs[0];
        StringBuffer var1 = new StringBuffer();
        int idx = 0;
        while (idx < 200) {
            //If the index is greater than or equal to length of the first string
            if (strs[0].length() <= idx)
                return var1.toString();
            Character ch = strs[0].charAt(idx);
            for (int i = 1; i < len; i++) {
                String s = strs[i];
                //If the index is greater than or equal to length of the ith string
                if (idx >= s.length())
                    return var1.toString();
                //If the ith string has different character at index idx
                if (s.charAt(idx) != ch)
                    return var1.toString();
            }
            //All the stings has same character at index idx, hence appended to prefix
            var1.append(ch);
            idx++;
        }
        return var1.toString();
    }
}
