class Solution {

    public String method1(String[] strs) {
        // return empty string if the string is null
        if (strs == null || strs.length == 0) {
            return "";
        }
        // math the prefix by comparing with the first string
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            // if the string of index i does not start with the prefix,then we need to remove the last letter
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
            }
            if (var1.isEmpty()) {
                return "";
            }
        }
        return var1;
    }
}
