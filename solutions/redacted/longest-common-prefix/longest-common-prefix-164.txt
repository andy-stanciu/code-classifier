class Solution {

    public String method1(String[] strs) {
        //Sorts array based on length of each string
        Arrays.sort(strs);
        String s = "";
        for (int i = 0; i < strs[0].length(); i++) {
            if (strs[0].charAt(i) == strs[strs.length - 1].charAt(i)) {
                s += (strs[strs.length - 1].charAt(i));
            } else {
                break;
            }
        }
        return s;
    }
}
