class Solution {

    public String method1(String[] strs) {
        // Return empty string if input is null or empty
        if (strs == null || strs.length == 0) {
            return "";
        }
        // Find the shortest string in the array
        String shortestStr = strs[0];
        for (String str : strs) {
            if (str.length() < shortestStr.length()) {
                shortestStr = str;
            }
        }
        // Use the shortest string to avoid iterating every character to find the longest common prefix
        for (int i = 0; i < shortestStr.length(); i++) {
            char currentChar = shortestStr.charAt(i);
            for (String str : strs) {
                // Immediately return on first char that does not match.
                if (str.charAt(i) != currentChar) {
                    return shortestStr.substring(0, i);
                }
            }
        }
        return shortestStr;
    }
}
