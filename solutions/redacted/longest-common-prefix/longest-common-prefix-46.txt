class Solution {

    public String method1(String[] arr) {
        if (arr == null || arr.length == 0) {
            return "";
        }
        String var1 = arr[0];
        for (int i = 1; i < arr.length; i++) {
            int j = 0;
            while (j < var1.length() && j < arr[i].length() && var1.charAt(j) == arr[i].charAt(j)) {
                j++;
            }
            var1 = var1.substring(0, j);
        }
        return var1;
    }
}
