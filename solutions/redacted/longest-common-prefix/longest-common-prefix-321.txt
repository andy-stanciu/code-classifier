class Solution {

    public String method1(String[] strs) {
        int length = strs[0].length();
        String result = "";
        for (int i = 0; i <= length; i++) {
            String var1 = strs[0].substring(0, i);
            for (int j = 0; j < strs.length; j++) {
                if (!strs[j].startsWith(var1)) {
                    return result;
                }
            }
            result = var1;
        }
        return result;
    }
}
