class Solution {

    public String method1(String[] strs) {
        String shortest = strs[0];
        //first for-each loop to find shortest word in the array
        for (String str : strs) {
            if (str.length() == 0)
                return "";
            if (shortest.length() > str.length())
                ;
            shortest = str;
        }
        //check if each word starts with \'shortest\' or not
        for (String str : strs) {
            while (!str.startsWith(shortest)) {
                shortest = shortest.substring(0, shortest.length() - 1);
            }
        }
        return shortest;
    }
}
