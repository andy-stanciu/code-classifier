class Solution {

    public String method1(String[] strs) {
        int n = strs.length;
        Arrays.sort(strs);
        String s1 = strs[0];
        String s2 = strs[n - 1];
        int idx = 0;
        for (; idx < s1.length(); idx++) {
            if (s1.charAt(idx) != s2.charAt(idx)) {
                break;
            }
        }
        return s1.substring(0, idx);
    }
}
