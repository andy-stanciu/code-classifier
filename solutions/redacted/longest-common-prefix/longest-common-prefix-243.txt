class Solution {

    public String method1(String[] strs) {
        if (strs.length == 1) {
            return strs[0];
        }
        String res = strs[0];
        for (int i = 0; i < strs.length; i++) {
            String str1 = strs[i];
            String str2 = res;
            res = "";
            int j = 0;
            while (j < str1.length() && j < str2.length() && str1.charAt(j) == str2.charAt(j)) {
                res += str1.charAt(j);
                j++;
            }
        }
        return res;
    }
}
