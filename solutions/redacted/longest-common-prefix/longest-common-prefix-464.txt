class Solution {

    public String method1(String[] strs) {
        if (strs.length == 1)
            return strs[0];
        int indexOfShortestString = 0;
        for (int i = 0; i < strs.length; i++) if (strs[i].length() < strs[indexOfShortestString].length())
            indexOfShortestString = i;
        if (strs[indexOfShortestString].isEmpty())
            return "";
        for (int i = 0; i < strs[indexOfShortestString].length(); i++) {
            char c = strs[0].charAt(i);
            for (int j = 1; j < strs.length; j++) {
                if (c != strs[j].charAt(i))
                    // i is exclusive
                    return strs[0].substring(0, i);
            }
        }
        return strs[indexOfShortestString];
    }
}
