class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0)
            return "";
        String base = strs[0];
        for (int i = 0; i < base.length(); i++) {
            for (String s : strs) {
                if (i == s.length() || s.charAt(i) != base.charAt(i))
                    return base.substring(0, i);
            }
        }
        return base;
    }
}
