class Solution {

    public String method1(String[] arr) {
        int n = arr.length;
        String var1 = arr[0];
        for (int i = 1; i < n; i++) {
            while (arr[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
