class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        for (String currentString : strs) {
            while ((currentString.length() < var1.length()) || (!var1.equals(currentString.substring(0, var1.length())))) {
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
