class Solution {

    public static boolean method1(String s, char ch, int i) {
        if (s.charAt(i) == ch) {
            return true;
        }
        return false;
    }

    public static int minlength(String[] strs) {
        int minLength = Integer.MAX_VALUE;
        for (String str : strs) {
            if (str.length() < minLength) {
                minLength = str.length();
            }
        }
        return minLength;
    }

    public String method2(String[] strs) {
        if (strs.length == 1) {
            return strs[0];
        }
        int n = minlength(strs);
        StringBuilder x = new StringBuilder();
        for (int i = 0; i < n; i++) {
            char ch = strs[0].charAt(i);
            boolean var1 = true;
            for (int j = 1; j < strs.length; j++) {
                String s = strs[j];
                if (!method1(s, ch, i)) {
                    var1 = false;
                    break;
                }
            }
            if (var1) {
                x.append(ch);
            } else {
                break;
            }
        }
        return x.toString();
    }
}
