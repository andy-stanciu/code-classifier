class Solution {

    public String method1(String[] strs) {
        String sol = strs[0];
        for (int i = 1; i < strs.length; i++) {
            int index = 0;
            while (index < strs[i].length() && index < sol.length() && sol.charAt(index) == strs[i].charAt(index)) {
                index++;
            }
            sol = sol.substring(0, index);
            if (sol.equals(""))
                return "";
        }
        return sol;
    }
}
