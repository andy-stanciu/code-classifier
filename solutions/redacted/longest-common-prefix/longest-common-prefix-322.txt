class Solution {

    public String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            // If the input array is empty, return an empty string.
            return "";
        }
        // Start by assuming the entire first string is the common prefix.
        String var1 = strs[0];
        // Compare this prefix with each subsequent string.
        for (int i = 1; i < strs.length; i++) {
            // Trim the prefix while it doesn\'t match the current string.
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
                // If the prefix becomes empty, return an empty string.
                if (var1.isEmpty()) {
                    return "";
                }
            }
        }
        return var1;
    }
}
