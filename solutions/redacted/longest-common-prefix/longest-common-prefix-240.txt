class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0) {
            return "";
        }
        Arrays.sort(strs);
        int k = 0;
        String str1 = strs[0];
        String str2 = strs[strs.length - 1];
        while (k < str1.length() && k < str2.length()) {
            if (str1.charAt(k) == str2.charAt(k)) {
                k++;
            } else {
                break;
            }
        }
        return str1.substring(0, k);
    }
}
