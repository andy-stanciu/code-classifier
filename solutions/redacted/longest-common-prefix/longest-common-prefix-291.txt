class Solution {

    public String method1(String[] strs) {
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            if (!strs[i].startsWith(var1)) {
                int var2 = 0;
                while (var2 < strs[i].length() && var1.charAt(var2) == strs[i].charAt(var2)) {
                    var2++;
                }
                var1 = strs[i].substring(0, var2);
            }
        }
        return var1;
    }
}
