class Solution {

    public String method1(String[] strs) {
        String var1 = "";
        for (String w : strs) {
            for (Character c : w.toCharArray()) {
                var var2 = var1 + c;
                if (!method2(strs, var2))
                    return var1;
                var1 = var2;
            }
        }
        return var1;
    }

    public Boolean method2(String[] words, String var1) {
        for (String w : words) {
            if (!w.startsWith(var1))
                return false;
        }
        return true;
    }
}
