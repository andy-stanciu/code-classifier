class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0)
            return "";
        String var1 = strs[0];
        for (String s : strs) {
            int i = 0, j = 0;
            while (i < var1.length() && j < s.length() && var1.charAt(i) == s.charAt(j)) {
                i++;
                j++;
            }
            var1 = var1.substring(0, j);
        }
        return var1;
    }
}
