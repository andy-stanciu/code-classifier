class Solution {

    private String compareTwoStrings(String s1, String s2) {
        char[] c1 = s1.toCharArray();
        char[] c2 = s2.toCharArray();
        int length = (c1.length > c2.length) ? c2.length : c1.length;
        String var1 = "";
        for (int i = 0; i < length; i++) {
            if (c1[i] == c2[i])
                var1 += String.valueOf(c1[i]);
            else
                break;
        }
        return var1;
    }

    public String method1(String[] strs) {
        String var1 = strs[0];
        if (strs.length > 1) {
            for (int i = 1; i < strs.length; i++) var1 = compareTwoStrings(var1, strs[i]);
        }
        return var1;
    }
}
