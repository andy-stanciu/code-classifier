class Solution {

    public String method1(String[] strs) {
        //if we see the string in arrays carefully we can say that if
        //arreange them in lexicogrphy order and check for the first and last
        //string for same prefix we will get the ans
        //no need to travels and check all the strings
        //sorting the strs array
        Arrays.sort(strs);
        //taking first string in sorted array
        String start = strs[0];
        //taking last string in sorted array
        String end = strs[strs.length - 1];
        //creating an empty string
        String s = "";
        for (int i = 0; i < start.length(); i++) {
            if (start.charAt(i) == end.charAt(i)) {
                //checking the elemet in first & last string
                //if same add in string s
                s += start.charAt(i);
            } else {
                //if letters are not same break the loop
                break;
            }
        }
        return s;
    }
}
