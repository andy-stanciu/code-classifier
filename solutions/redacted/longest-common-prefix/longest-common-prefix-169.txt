class Solution {

    public String method1(String[] strs) {
        //         Assign first String to prefix variable
        //         Check for 1st index to n-1 , using indexOf function , if prefix is not then removes last character and continue
        if (strs.length == 0)
            return "";
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            while (strs[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
                if (var1.isEmpty())
                    return "";
            }
        }
        return var1;
    }
}
