class Solution {

    public String method1(String[] strs) {
        // tomar el primer string como el prefijo
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            // recorrer el string a partir del segundo indice
            while (strs[i].indexOf(var1) != 0) {
                // si el string actual no comienza con el prefijo
                // reducir el prefijo en uno
                var1 = var1.substring(0, var1.length() - 1);
            }
        }
        return var1;
    }
}
