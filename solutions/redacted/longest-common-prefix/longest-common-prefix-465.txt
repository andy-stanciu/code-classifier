class Solution {

    public String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            // Return empty string if input is empty
            return "";
        }
        int c = 0;
        while (true) {
            // Check if we have reached the end of the first string
            if (c >= strs[0].length()) {
                break;
            }
            // Character to be matched in all strings
            char ch = strs[0].charAt(c);
            for (String str : strs) {
                // If we reach the end of a string or a mismatch is found
                if (c >= str.length() || str.charAt(c) != ch) {
                    return strs[0].substring(0, c);
                }
            }
            // Increment counter if all strings match at position c
            c++;
        }
        // Return the common prefix found
        return strs[0].substring(0, c);
    }
}
