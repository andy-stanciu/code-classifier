class Solution {

    public String method1(String[] arr) {
        if (arr.length == 0)
            return "";
        if (arr.length == 1)
            return arr[0];
        String var1 = arr[0];
        for (int i = 1; i < arr.length; i++) {
            while (arr[i].indexOf(var1) != 0) {
                var1 = var1.substring(0, var1.length() - 1);
            }
            if (var1.isEmpty())
                return "";
        }
        return var1;
    }
}
