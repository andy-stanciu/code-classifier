class Solution {

    public String method1(String[] strs) {
        // Sort the arrays in ascending order
        Arrays.sort(strs);
        // Shortest and alphabetically first
        String first = strs[0];
        // Longest or / and alphabetically last
        String last = strs[strs.length - 1];
        String method1 = "";
        // Iterate through the chars of the first shortest word
        for (int i = 0; i < first.length(); i++) {
            char currChar = last.charAt(i);
            if (currChar == first.charAt(i)) {
                method1 += currChar;
            } else {
                break;
            }
        }
        return method1;
    }
}
