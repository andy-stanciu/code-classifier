class Solution {

    public String method1(String[] strs) {
        Arrays.sort(strs);
        String fir = strs[0];
        String las = strs[strs.length - 1];
        int i = 0;
        while (i < fir.length() && i < las.length()) {
            if (fir.charAt(i) != las.charAt(i)) {
                break;
            }
            i++;
        }
        return fir.substring(0, i);
    }
}
