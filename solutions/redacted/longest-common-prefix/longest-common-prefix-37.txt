class Solution {

    public String method1(String[] strs) {
        String r = "";
        int n = strs.length;
        //Sorted first
        Arrays.sort(strs);
        //smallest no of char
        String s = strs[0];
        //highest no of char
        String h = strs[n - 1];
        for (//taken smallest length so that run time will be less
        int i = 0; //taken smallest length so that run time will be less
        i < s.length(); //taken smallest length so that run time will be less
        i++) {
            if (s.charAt(i) != h.charAt(i))
                break;
            r = r + s.charAt(i);
        }
        return r;
    }
}
//Time complexity would be O(n)
