class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0) {
            return " ";
        }
        String pre = strs[0];
        for (String s : strs) {
            while (s.indexOf(pre) != 0) {
                pre = pre.substring(0, pre.length() - 1);
            }
        }
        return pre;
    }
}
