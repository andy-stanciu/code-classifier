// Bruteforce
class Solution {

    public String method1(String[] strs) {
        // Take the first string as the base for comparison
        StringBuilder var1 = new StringBuilder();
        // Loop through each character of the first string
        for (int i = 0; i < strs[0].length(); i++) {
            char currentChar = strs[0].charAt(i);
            // Compare the character with all other strings
            for (int j = 1; j < strs.length; j++) {
                // If we reach the end of a string or find a mismatch, return the prefix so far
                if (i >= strs[j].length() || strs[j].charAt(i) != currentChar) {
                    return var1.toString();
                }
            }
            // If current character matches in all strings, append it to the prefix
            var1.append(currentChar);
        }
        return var1.toString();
    }
}
