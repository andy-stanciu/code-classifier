class Solution {

    public String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        // Take the first string as a reference for comparison
        String var1 = strs[0];
        for (int i = 1; i < strs.length; i++) {
            int j = 0;
            // Compare the prefix with each string character by character
            while (j < var1.length() && j < strs[i].length() && var1.charAt(j) == strs[i].charAt(j)) {
                j++;
            }
            // Reduce the prefix to the matched characters so far
            var1 = var1.substring(0, j);
            // If the prefix becomes empty, there\'s no common prefix
            if (var1.isEmpty()) {
                return "";
            }
        }
        return var1;
    }
}
