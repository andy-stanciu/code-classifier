class Solution {

    public String method1(String[] strs) {
        if (strs == null || strs.length == 0) {
            return "";
        }
        // Start with the first string as the initial prefix
        String var1 = strs[0];
        // Compare the prefix with each string in the array
        for (int i = 1; i < strs.length; i++) {
            // Find the common prefix between the current prefix and the next string
            while (strs[i].indexOf(var1) != 0) {
                // Shorten the prefix by removing the last character
                var1 = var1.substring(0, var1.length() - 1);
                // If the prefix becomes empty, return ""
                if (var1.isEmpty()) {
                    return "";
                }
            }
        }
        return var1;
    }

    // Main method for testing
    public static void main(String[] args) {
        Solution solution = new Solution();
        // Example 1
        String[] strs1 = { "flower", "flow", "flight" };
        // Output: "fl"
        System.out.println(solution.method1(strs1));
        // Example 2
        String[] strs2 = { "dog", "racecar", "car" };
        // Output: ""
        System.out.println(solution.method1(strs2));
    }
}
