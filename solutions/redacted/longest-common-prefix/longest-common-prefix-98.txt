class Solution {

    public String method1(String[] strs) {
        if (strs.length == 0)
            return "";
        // In JVM 8 Sorting algorithm is a Dual-Pivot Quicksort. So, T.C will be O(N) to O(NlogN)
        Arrays.sort(strs);
        String first = strs[0];
        String last = strs[strs.length - 1];
        int var1 = 0;
        while (var1 < first.length()) {
            if (first.charAt(var1) == last.charAt(var1)) {
                var1++;
            } else {
                break;
            }
        }
        return var1 == 0 ? "" : first.substring(0, var1);
    }
}
// Time Complexity : O(NlogN)
// Space Complexity : O(1)
