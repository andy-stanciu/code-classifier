class Solution {

    public void method1(int[] nums) {
        int idx = -1;
        int n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                idx = i;
                break;
            }
        }
        if (idx == -1) {
            reverseArray(nums, 0, n - 1);
            return;
        }
        for (int i = n - 1; i > idx; i--) {
            if (nums[i] > nums[idx]) {
                arrSwap(nums, i, idx);
                break;
            }
        }
        reverseArray(nums, idx + 1, n - 1);
    }

    public static void reverseArray(int[] nums, int start, int end) {
        while (start <= end) {
            int t = nums[start];
            nums[start] = nums[end];
            nums[end] = t;
            start++;
            end--;
        }
    }

    public static void arrSwap(int[] nums, int i, int idx) {
        int t = nums[i];
        nums[i] = nums[idx];
        nums[idx] = t;
    }
}
