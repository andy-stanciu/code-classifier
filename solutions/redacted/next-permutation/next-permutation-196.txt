class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int pivot = -1;
        // find pivot
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                pivot = i;
                break;
            }
        }
        // edge case if pivot is -1
        if (pivot == -1) {
            int i = 0, j = n - 1;
            while (i < j) {
                int temp = nums[j];
                nums[j] = nums[i];
                nums[i] = temp;
                i++;
                j--;
            }
            return;
        }
        // swap pivot with right most element
        for (int i = n - 1; i > pivot; i--) {
            if (nums[pivot] < nums[i]) {
                int temp = nums[pivot];
                nums[pivot] = nums[i];
                nums[i] = temp;
                break;
            }
        }
        // reverese array from pivot+1 to n
        int i = pivot + 1, j = n - 1;
        while (i <= j) {
            int temp = nums[j];
            nums[j] = nums[i];
            nums[i] = temp;
            i++;
            j--;
        }
    }
}
