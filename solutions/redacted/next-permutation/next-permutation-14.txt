class Solution {

    public void method1(int[] nums) {
        int swapInd = -1;
        int swapNo = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i - 1] < nums[i]) {
                swapInd = i - 1;
                swapNo = nums[i - 1];
                break;
            }
        }
        if (swapInd != -1) {
            System.out.println("swapind.." + swapInd);
            for (int i = nums.length - 1; i > swapInd; i--) {
                if (nums[i] > swapNo) {
                    System.out.println("new no.." + nums[i]);
                    nums[swapInd] = nums[i];
                    nums[i] = swapNo;
                    break;
                }
            }
        }
        int start = swapInd + 1;
        int end = nums.length - 1;
        while (start < end) {
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }
}
