class Solution {

    public void method1(int[] nums) {
        int index = -1;
        int n = nums.length;
        // Find the breakpoint
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                index = i;
                break;
            }
        }
        if (index == -1) {
            // Reverse the whole array if no breakpoint is found
            reverse(nums, 0, n - 1);
            return;
        }
        // Find the smallest number greater than nums[index] to the right of index
        for (int i = n - 1; i > index; i--) {
            if (nums[i] > nums[index]) {
                // Swap the numbers
                swap(nums, i, index);
                break;
            }
        }
        // Reverse the numbers after the index
        reverse(nums, index + 1, n - 1);
    }

    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
