class Solution {

    public void method1(int[] nums) {
        // According to wikipedia definition for how to find next permutation
        // 1. find the largest index k such that a[k] < a[k + 1]. if no such index exists, the
        // permutation is the last permutation
        int k = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                k = i - 1;
                break;
            }
        }
        if (k == -1) {
            // reverse
            for (int i = 0; i < nums.length / 2; i++) {
                int temp = nums[i];
                nums[i] = nums[nums.length - 1 - i];
                nums[nums.length - 1 - i] = temp;
            }
            return;
        }
        // 2. Find the largest index l greater than k such that a[k] < a[l]
        int l = nums.length - 1;
        for (; l > k; l--) {
            if (nums[l] > nums[k]) {
                break;
            }
        }
        // 3. Swap the value of a[k] with that of a[l]
        int temp = nums[k];
        nums[k] = nums[l];
        nums[l] = temp;
        // 4. Reverse the sequence from a[k + 1] up to and including the final element a[n]
        for (int i = k + 1, j = nums.length - 1; i < j; i++, j--) {
            temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
        }
    }
}
