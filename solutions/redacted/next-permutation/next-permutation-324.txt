class Solution {

    public void method1(int[] nums) {
        int i1 = -1, i2 = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                i1 = i - 1;
                break;
            }
        }
        if (i1 == -1)
            reverse(nums, 0);
        else {
            for (int i = nums.length - 1; i >= 0; i--) {
                if (nums[i] > nums[i1]) {
                    i2 = i;
                    break;
                }
            }
            swap(nums, i1, i2);
            reverse(nums, i1 + 1);
        }
    }

    static void swap(int[] a, int i, int j) {
        int temp = a[i];
        a[i] = a[j];
        a[j] = temp;
    }

    static void reverse(int[] a, int start) {
        int end = a.length - 1;
        while (start < end) {
            swap(a, start, end);
            start++;
            end--;
        }
    }
}
