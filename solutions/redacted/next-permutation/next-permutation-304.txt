class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int golaindex = -1;
        int swap = 0;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                golaindex = i - 1;
                break;
            }
        }
        if (golaindex != -1) {
            for (int j = n - 1; j > golaindex; j--) {
                if (nums[j] > nums[golaindex]) {
                    swap = j;
                    break;
                }
            }
            int mid = nums[swap];
            nums[swap] = nums[golaindex];
            nums[golaindex] = mid;
        }
        int left = golaindex + 1;
        int right = n - 1;
        while (left < right) {
            int temp = nums[left];
            nums[left] = nums[right];
            nums[right] = temp;
            left++;
            right--;
        }
    }
}
