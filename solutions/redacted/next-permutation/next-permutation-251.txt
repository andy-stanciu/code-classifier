class Solution {

    public void method1(int[] nums) {
        int index = -1;
        int smallestIndex = nums.length - 1;
        // finding Breakpoint
        for (int i = nums.length - 2; i >= 0; i--) {
            System.out.println(nums[i]);
            if (nums[i] < nums[i + 1]) {
                index = i;
                break;
            }
        }
        System.out.println("Breakpoint. " + index);
        if (index == -1) {
            Arrays.sort(nums);
        } else {
            for (int i = nums.length - 1; i > index; i--) {
                if (nums[i] > nums[index]) {
                    System.out.println("nums[i]. " + nums[i]);
                    System.out.println("nums[i]. " + nums[index]);
                    int element = nums[i];
                    nums[i] = nums[index];
                    nums[index] = element;
                    break;
                }
                System.out.println("nums[index]. " + nums[index]);
            }
            Arrays.sort(nums, index + 1, nums.length);
        }
    }
}
