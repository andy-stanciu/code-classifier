class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int ind = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            reverse(nums, 0, n - 1);
        } else {
            for (int i = n - 1; i > ind; i--) {
                if (nums[i] > nums[ind]) {
                    int temp = nums[i];
                    nums[i] = nums[ind];
                    nums[ind] = temp;
                    break;
                }
            }
            reverse(nums, ind + 1, n - 1);
        }
    }

    public static void reverse(int[] a, int start, int end) {
        int j = end;
        int k = (end + 1) - start;
        int m = start;
        for (int i = 0; i < k / 2; i++) {
            int temp = a[m];
            a[m] = a[j];
            a[j] = temp;
            j--;
            m++;
        }
    }
}
