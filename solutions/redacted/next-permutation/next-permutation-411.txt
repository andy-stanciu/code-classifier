class Solution {

    public void method1(int[] nums) {
        boolean change = false;
        int findIndex = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                findIndex = i;
                break;
            }
        }
        for (int i = findIndex + 1, j = nums.length - 1; i < j; i++, j--) {
            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
        }
        if (findIndex == -1) {
            return;
        }
        int findIndex2 = findIndex + 1;
        for (int i = findIndex + 1; i < nums.length; i++) {
            if (nums[i] > nums[findIndex]) {
                findIndex2 = i;
                break;
            }
        }
        int temp = nums[findIndex2];
        nums[findIndex2] = nums[findIndex];
        nums[findIndex] = temp;
    }
}
