class Solution {

    public void method1(int[] nums) {
        int index = -1, len = nums.length, start = 0, end = 0, temp = 0;
        for (int i = len - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                index = i;
                break;
            }
        }
        if (index == -1) {
            end = len - 1;
            while (start <= end) {
                temp = nums[start];
                nums[start++] = nums[end];
                nums[end--] = temp;
            }
            return;
        }
        temp = 0;
        for (int i = len - 1; i > index; i--) {
            if (nums[i] > nums[index]) {
                temp = nums[i];
                nums[i] = nums[index];
                nums[index] = temp;
                break;
            }
        }
        start = index + 1;
        end = len - 1;
        while (start <= end) {
            temp = nums[start];
            nums[start++] = nums[end];
            nums[end--] = temp;
        }
    }
}
