class Solution {

    static void swap(int i, int j, int[] a) {
        while (i < j) {
            int tmp = a[i];
            a[i] = a[j];
            a[j] = tmp;
            i++;
            j--;
        }
    }

    public void method1(int[] a) {
        int N = a.length;
        int ind = -1;
        for (int i = N - 2; i >= 0; i--) {
            if (a[i] < a[i + 1]) {
                ind = i;
                break;
            }
        }
        List<Integer> l = new ArrayList<>();
        if (ind == -1) {
            //reverse the whole array
            swap(0, N - 1, a);
        } else {
            for (int i = N - 1; i >= 0; i--) {
                if (a[i] > a[ind]) {
                    int temp = a[i];
                    a[i] = a[ind];
                    a[ind] = temp;
                    break;
                }
            }
            swap(ind + 1, N - 1, a);
        }
    }
}
