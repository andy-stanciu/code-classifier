class Solution {

    public void rev(int s, int e, int[] a) {
        while (s <= e) {
            int t = a[s];
            a[s] = a[e];
            a[e] = t;
            s++;
            e--;
        }
    }

    public void method1(int[] nums) {
        int ind = -1, n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            rev(0, n - 1, nums);
            return;
        }
        for (int i = n - 1; i > ind; i--) {
            if (nums[ind] < nums[i]) {
                int t = nums[ind];
                nums[ind] = nums[i];
                nums[i] = t;
                break;
            }
        }
        rev(ind + 1, n - 1, nums);
    }
}
