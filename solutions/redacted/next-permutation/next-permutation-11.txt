class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        for (int j = n - 1; j > 0; j--) {
            if (nums[j - 1] < nums[n - 1]) {
                for (int i = j; i < n; i++) {
                    if (nums[i] > nums[j - 1]) {
                        swap(nums, i, j - 1);
                        return;
                    }
                }
            } else {
                for (int i = j; i < n; i++) {
                    swap(nums, i - 1, i);
                }
            }
        }
    }

    private void swap(int[] nums, int x, int y) {
        int temp = nums[x];
        nums[x] = nums[y];
        nums[y] = temp;
    }
}
