class Solution {

    public void method1(int[] nums) {
        // variable to see if the nums array is in decreasing order
        int inx = -1;
        // to identify for which index we need to swap
        for (int i = nums.length - 1; i >= 1; i--) {
            if (nums[i] > nums[i - 1]) {
                inx = i - 1;
                break;
            }
        }
        // inx == -1 means the array is in decreasing order
        if (inx == -1) {
            // hence we just need to reverse the entire array
            reverse(nums, 0, nums.length - 1);
            return;
        }
        // swap with the just bigger element
        for (int i = nums.length - 1; i > inx; i--) {
            if (nums[i] > nums[inx]) {
                int temp = nums[i];
                nums[i] = nums[inx];
                nums[inx] = temp;
                break;
            }
        }
        // reverse the rest of the array to find the next permutation
        reverse(nums, inx + 1, nums.length - 1);
    }

    // reverse logic
    public void reverse(int[] nums, int inx, int n) {
        while (inx <= n) {
            int temp = nums[inx];
            nums[inx] = nums[n];
            nums[n] = temp;
            inx++;
            n--;
        }
    }
}
