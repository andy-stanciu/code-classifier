class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        // Step 1 Finding the break Point
        //Initial breakpoint
        int ind = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                // i is the  break point
                ind = i;
                break;
            }
        }
        // Edge Case - If the break point doesnt exist
        if (ind == -1) {
            // Reverse the whole array
            reverseArray(nums, 0, n - 1);
            return;
        }
        // Step 2 Find the elemenet greater than nums[i] and swap it
        for (int i = n - 1; i > ind; i--) {
            if (nums[i] > nums[ind]) {
                int temp = nums[i];
                nums[i] = nums[ind];
                nums[ind] = temp;
                break;
            }
        }
        // Step 3 Reverse the right Half
        reverseArray(nums, ind + 1, n - 1);
    }

    public static void reverseArray(int[] array, int start, int end) {
        while (start < end) {
            int temp = array[start];
            array[start] = array[end];
            array[end] = temp;
            start++;
            end--;
        }
    }
}
