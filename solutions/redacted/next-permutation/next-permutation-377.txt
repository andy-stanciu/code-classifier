class Solution {

    public void method1(int[] nums) {
        Boolean flag = false;
        int n = nums.length - 1;
        // Step 1: Find the first decreasing element from the end
        for (int i = n; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                // Step 2: Find the element just larger than nums[i - 1] from the end
                int j = n;
                for (; j >= i; j--) {
                    if (nums[j] > nums[i - 1]) {
                        break;
                    }
                }
                // Swap the found element with nums[i - 1]
                int temp = nums[i - 1];
                nums[i - 1] = nums[j];
                nums[j] = temp;
                // Step 3: Reverse the elements to the right of i-1 to get the next permutation
                reverse(nums, i, n);
                flag = true;
                break;
            }
        }
        // If no next permutation is found, reverse the entire array to get the smallest permutation
        if (!flag) {
            reverse(nums, 0, n);
        }
    }

    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }
}
