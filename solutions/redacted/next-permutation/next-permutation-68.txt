class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        int k = n - 2;
        // exceptions
        if (nums == null || nums.length == 0)
            return;
        // find k
        // 3 1 4 2
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] <= nums[i - 1]) {
                k--;
            } else {
                break;
            }
        }
        // if k is -1 then boom
        if (k == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        // 3 k1 4 n2
        for (int i = n - 1; i > 0; i--) {
            if (nums[i] > nums[k]) {
                int temp = nums[i];
                nums[i] = nums[k];
                nums[k] = temp;
                break;
            }
        }
        reverse(nums, k + 1, n - 1);
    }

    void reverse(int[] nums, int start, int end) {
        while (start < end) {
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }
}
