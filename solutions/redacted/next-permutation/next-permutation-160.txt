class Solution {

    public void method1(int[] nums) {
        int pointer = nums.length - 1;
        // Step 1: Find the first decreasing element from the end
        while (pointer > 0 && nums[pointer] <= nums[pointer - 1]) {
            pointer--;
        }
        // Step 2: If the array is completely non-increasing, sort it
        if (pointer == 0 || nums.length == 1) {
            Arrays.sort(nums);
        } else {
            // Step 3: Find the element just larger than nums[pointer - 1]
            int right = nums.length - 1;
            while (right > pointer && nums[right] <= nums[pointer - 1]) {
                right--;
            }
            // Step 4: Swap the elements
            int temp = nums[pointer - 1];
            nums[pointer - 1] = nums[right];
            nums[right] = temp;
            // Step 5: Reverse the suffix starting at pointer
            int start = pointer;
            int end = nums.length - 1;
            while (start < end) {
                temp = nums[start];
                nums[start] = nums[end];
                nums[end] = temp;
                start++;
                end--;
            }
        }
    }
}
