class Solution {

    public void method1(int[] nums) {
        boolean found = false;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                int minIndex = i + 1;
                for (int j = i + 1; j < nums.length; j++) {
                    if (nums[j] > nums[i] && nums[j] <= nums[minIndex]) {
                        minIndex = j;
                    }
                }
                int temp = nums[i];
                nums[i] = nums[minIndex];
                nums[minIndex] = temp;
                reverse(nums, i + 1);
                found = true;
                break;
            }
        }
        if (!found) {
            reverse(nums, 0);
        }
    }

    private void reverse(int[] nums, int lastIndex) {
        for (int i = 0; i < (nums.length - lastIndex) / 2; i++) {
            int temp = nums[i + lastIndex];
            nums[i + lastIndex] = nums[nums.length - 1 - i];
            nums[nums.length - 1 - i] = temp;
        }
    }
}
