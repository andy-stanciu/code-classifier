class Solution {

    public void method1(int[] nums) {
        int len = nums.length;
        int index = len - 2;
        //pivot
        while (index >= 0 && nums[index] >= nums[index + 1]) {
            index--;
        }
        // successor
        if (index >= 0) {
            int j = len - 1;
            while (nums[j] <= nums[index]) {
                j--;
            }
            // swap
            swap(nums, index, j);
        }
        reverse(nums, index + 1);
    }

    public void swap(int[] nums, int index, int j) {
        int temp = nums[index];
        nums[index] = nums[j];
        nums[j] = temp;
    }

    public void reverse(int[] nums, int start) {
        int end = nums.length - 1;
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }
}
