class Solution {

    public void method1(int[] nums) {
        int temp = 0;
        int temp1 = 0;
        int c = nums.length;
        xyz: for (int i = c - 1; i >= 0; i--) {
            temp = nums[i];
            for (int j = c - 1; j > i; j--) {
                if (nums[j] > temp) {
                    temp1 = nums[j];
                    nums[j] = temp;
                    nums[i] = temp1;
                    temp1 = i;
                    c++;
                    break xyz;
                }
            }
        }
        if (c == nums.length) {
            Arrays.sort(nums);
        }
        c = nums.length;
        for (int i = temp1 + 1; i < c; i++) {
            for (int j = temp1 + 1; j < c; j++) {
                if (nums[i] < nums[j]) {
                    temp = nums[i];
                    nums[i] = nums[j];
                    nums[j] = temp;
                }
            }
        }
    }
}
