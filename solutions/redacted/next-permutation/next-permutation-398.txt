class Solution {

    public void method1(int[] nums) {
        int b = -1;
        for (int i = nums.length - 1; i > 0; i--) {
            if (nums[i] > nums[i - 1]) {
                b = i - 1;
                break;
            }
        }
        if (b == -1) {
            int left = 0;
            int right = nums.length - 1;
            int temp;
            while (left < right) {
                temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
                left++;
                right--;
            }
        } else {
            int min = b + 1;
            for (int i = b + 1; i < nums.length; i++) {
                if (nums[i] <= nums[min] && nums[i] > nums[b]) {
                    min = i;
                }
            }
            int temp = nums[b];
            nums[b] = nums[min];
            nums[min] = temp;
            int left = b + 1;
            int right = nums.length - 1;
            while (left < right) {
                temp = nums[left];
                nums[left] = nums[right];
                nums[right] = temp;
                left++;
                right--;
            }
        }
    }
}
