class Solution {

    public void swap(int[] nums, int start, int last) {
        int temp = nums[start];
        nums[start] = nums[last];
        nums[last] = temp;
    }

    public void reverse(int[] nums, int start, int last) {
        int temp;
        if (start >= last) {
            return;
        }
        swap(nums, start, last);
        reverse(nums, start + 1, last - 1);
    }

    public void method1(int[] nums) {
        int n = nums.length;
        int k = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                k = i;
                break;
            }
        }
        if (k == -1) {
            Arrays.sort(nums);
            return;
        }
        int l = -1;
        for (int i = n - 1; i >= k; i--) {
            if (nums[i] > nums[k]) {
                l = i;
                break;
            }
        }
        swap(nums, k, l);
        reverse(nums, k + 1, n - 1);
    }
}
