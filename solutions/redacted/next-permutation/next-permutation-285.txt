class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        // Step 1: Find the break point
        int ind = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        // Step 2: If no break point, reverse the whole array
        if (ind == -1) {
            reverseArray(nums, 0, n - 1);
            return;
        }
        // Step 3: Find the next greater element and swap it with nums[ind]
        for (int i = n - 1; i > ind; i--) {
            if (nums[i] > nums[ind]) {
                int temp = nums[i];
                nums[i] = nums[ind];
                nums[ind] = temp;
                break;
            }
        }
        // Step 4: Reverse the right half of the array
        reverseArray(nums, ind + 1, n - 1);
    }

    private void reverseArray(int[] nums, int start, int end) {
        while (start < end) {
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }
}
