class Solution {

    public void method1(int[] arr) {
        int n = arr.length;
        int ind = -1;
        int ind2 = -1;
        //find the breakpoint from right side
        for (int i = n - 2; i >= 0; i--) {
            if (arr[i] < arr[i + 1]) {
                ind = i;
                break;
            }
        }
        //if there is no breakpoint
        if (ind == -1) {
            reverse(arr, 0);
        } else {
            for (int i = n - 1; i >= 0; i--) {
                if (arr[i] > arr[ind]) {
                    ind2 = i;
                    break;
                }
            }
            swap(arr, ind, ind2);
            reverse(arr, ind + 1);
        }
    }

    //swap function
    void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    //reverse function
    void reverse(int[] arr, int start) {
        int i = start;
        int j = arr.length - 1;
        while (i < j) {
            swap(arr, i, j);
            i++;
            j--;
        }
    }
}
