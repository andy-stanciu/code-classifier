class Solution {

    public void method1(int[] nums) {
        int ind = -1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            rev(nums, 0);
        } else {
            for (int j = nums.length - 1; j >= ind; j--) {
                if (nums[j] > nums[ind]) {
                    int temp = nums[ind];
                    nums[ind] = nums[j];
                    nums[j] = temp;
                    break;
                }
            }
            rev(nums, ind + 1);
        }
    }

    public void rev(int[] nums, int l) {
        int n = nums.length - 1;
        while (l < n) {
            int temp = nums[l];
            nums[l] = nums[n];
            nums[n] = temp;
            l++;
            n--;
        }
    }
}
