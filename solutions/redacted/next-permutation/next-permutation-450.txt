class Solution {

    public void method1(int[] nums) {
        int ind = -1;
        int n = nums.length;
        // Find the first decreasing element from the end
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        // If no such element is found, reverse the entire array
        if (ind == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        // Find the next larger element from the end and swap
        for (int i = n - 1; i > ind; i--) {
            if (nums[i] > nums[ind]) {
                swap(nums, i, ind);
                break;
            }
        }
        // Reverse the elements after the swapped element
        reverse(nums, ind + 1, n - 1);
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }
}
