class Solution {

    public void method1(int[] nums) {
        int b = -1, c = -1;
        int l = 0, h = nums.length - 1;
        for (int i = h - 1; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                b = i;
                break;
            }
        }
        if (b == -1) {
            while (l < h) {
                int temp1 = nums[l];
                nums[l] = nums[h];
                nums[h] = temp1;
                l++;
                h--;
            }
            return;
        }
        for (int i = h; i >= b + 1; i--) {
            if (nums[i] > nums[b]) {
                c = i;
                break;
            }
        }
        int temp = nums[b];
        nums[b] = nums[c];
        nums[c] = temp;
        l = b + 1;
        while (l < h) {
            int temp1 = nums[l];
            nums[l] = nums[h];
            nums[h] = temp1;
            l++;
            h--;
        }
    }
}
