class Solution {

    public void method1(int[] nums) {
        int i = nums.length - 2;
        int n = nums.length;
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            i--;
        }
        if (i >= 0) {
            int j = n - 1;
            while (nums[i] >= nums[j]) {
                j--;
            }
            Swap(nums, i, j);
        }
        Reverse(nums, i + 1, n - 1);
    }

    public static void Swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public static void Reverse(int[] nums, int i, int j) {
        while (i < j) {
            Swap(nums, i, j);
            i++;
            j--;
        }
    }
}
