class Solution {

    public void method1(int[] nums) {
        //1 2 5 4 3 0 0
        int n = nums.length;
        //finding break point....
        int ind = -1;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                ind = i;
                break;
            }
        }
        if (ind == -1) {
            reverse(nums, 0, n - 1);
            return;
        }
        //finding someone who is closest to el at ind and greater than that
        for (int i = n - 1; i > ind; i--) {
            if (nums[i] > nums[ind]) {
                int temp = nums[i];
                nums[i] = nums[ind];
                nums[ind] = temp;
                break;
            }
        }
        //sorting.the.element.after.ind.{reversing.as.already.sorted.in.desc}
        reverse(nums, ind + 1, n - 1);
    }

    //function of reversing..........
    public void reverse(int[] arr, int start, int end) {
        while (start < end) {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }
}
