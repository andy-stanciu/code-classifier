class Solution {

    public void method1(int[] nums) {
        int gola_idx = -1;
        int n = nums.length;
        for (int i = n - 1; i > 0; i--) {
            if (nums[i - 1] < nums[i]) {
                gola_idx = i - 1;
                break;
            }
        }
        //find the next greaterelement
        if (gola_idx != -1) {
            int swap_idx = gola_idx;
            for (int i = n - 1; i >= gola_idx + 1; i--) {
                if (nums[i] > nums[gola_idx]) {
                    swap_idx = i;
                    break;
                }
            }
            //swap both idx
            int temp = nums[gola_idx];
            nums[gola_idx] = nums[swap_idx];
            nums[swap_idx] = temp;
        }
        reverse(gola_idx + 1, n - 1, nums);
    }

    public void reverse(int i, int j, int[] nums) {
        while (i < j) {
            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
            i++;
            j--;
        }
    }
}
