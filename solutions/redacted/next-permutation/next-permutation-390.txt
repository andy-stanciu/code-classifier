class Solution {

    public void method1(int[] arr) {
        int ind = -1;
        // 1: Find the pivot index:
        for (int i = arr.length - 2; i >= 0; i--) {
            if (arr[i] < arr[i + 1]) {
                ind = i;
                break;
            }
        }
        // 2: If no such index is found, reverse the entire array:
        if (ind == -1) {
            reverse(arr, 0, arr.length - 1);
            return;
        }
        // 3: Find the smallest element on the right side of index which is greater than arr[ind] and swap :
        for (int i = arr.length - 1; i > ind; i--) {
            if (arr[i] > arr[ind]) {
                swap(arr, i, ind);
                break;
            }
        }
        // 4: Reverse the elements to the right of the pivot index:
        reverse(arr, ind + 1, arr.length - 1);
    }

    static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    static void reverse(int[] arr, int i, int j) {
        while (i < j) {
            swap(arr, i, j);
            i++;
            j--;
        }
    }
}
