class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        // Check for empty or null array
        if (nums == null || nums.length == 0)
            return;
        int k = n - 2;
        // Find the largest index k such that nums[k] < nums[k + 1]
        while (k >= 0 && nums[k] >= nums[k + 1]) {
            k--;
        }
        // If no such index k exists, the array is in descending order
        if (k < 0) {
            reverse(nums, 0, n - 1);
            return;
        }
        // Find the largest index l greater than k such that nums[k] < nums[l]
        int l = n - 1;
        while (nums[l] <= nums[k]) {
            l--;
        }
        // Swap the value of nums[k] with nums[l]
        swap(nums, k, l);
        // Reverse the sequence from nums[k + 1] up to the end
        reverse(nums, k + 1, n - 1);
    }

    // Helper method to reverse the elements in the array from start to end
    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }

    // Helper method to swap two elements in the array
    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
