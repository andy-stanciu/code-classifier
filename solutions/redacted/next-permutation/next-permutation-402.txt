class Solution {

    public void method1(int[] nums) {
        int n = nums.length;
        if (n <= 1)
            return;
        int index = n - 1;
        while (index > 0 && nums[index - 1] >= nums[index]) {
            --index;
        }
        if (index == 0) {
            reverse(nums, 0, n - 1);
            return;
        }
        --index;
        int index2 = n - 1;
        while (index2 > index && nums[index2] <= nums[index]) {
            --index2;
        }
        int tmp = nums[index];
        nums[index] = nums[index2];
        nums[index2] = tmp;
        reverse(nums, index + 1, n - 1);
    }

    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            int tmp = nums[start];
            nums[start] = nums[end];
            nums[end] = tmp;
            --end;
            ++start;
        }
    }
}
