class Solution {

    public void method1(int[] arr) {
        int j = 0;
        int flag = 0;
        int i = 0;
        for (i = arr.length - 1; i >= 0; i--) {
            for (j = arr.length - 1; j > i; j--) {
                if (arr[i] < arr[j]) {
                    int temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                    // System.out.println(arr[i]+" "+i+"  "+arr[j]+\' \'+j);
                    flag = 1;
                    break;
                }
                if (flag == 1)
                    break;
            }
            if (flag == 1)
                break;
        }
        for (int p = i + 1, q = arr.length - 1; p < q; p++, q--) {
            int temp = arr[p];
            arr[p] = arr[q];
            arr[q] = temp;
        }
    }
}
