class Solution {

    public void method1(int[] nums) {
        int ind = -1;
        int len = nums.length - 1;
        for (int k = len - 1; k >= 0; k--) {
            if (nums[k + 1] > nums[k]) {
                ind = k;
                break;
            }
        }
        if (ind == -1) {
            reverse(nums, 0, len);
            return;
        }
        for (int k = len; k > ind; k--) {
            if (nums[k] > nums[ind]) {
                swap(nums, ind, k);
                break;
            }
        }
        reverse(nums, ind + 1, len);
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    private void reverse(int[] nums, int start, int end) {
        while (start < end) {
            swap(nums, start, end);
            start++;
            end--;
        }
    }
}
