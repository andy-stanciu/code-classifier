class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int targetsum = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int currsum = nums[i] + nums[left] + nums[right];
                if (Math.abs(currsum - target) < Math.abs(targetsum - target)) {
                    targetsum = currsum;
                }
                if (currsum < target) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return targetsum;
    }
}
