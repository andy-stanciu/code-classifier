class Solution {

    public int method1(int[] nums, int target) {
        int minDifference = Integer.MAX_VALUE;
        boolean overTarget = false;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            int goal = target - nums[i];
            while (left < right) {
                int diff = goal - nums[left] - nums[right];
                if (Math.abs(diff) < minDifference) {
                    overTarget = diff < 0;
                    minDifference = Math.abs(diff);
                }
                if (diff > 0) {
                    left++;
                } else if (diff == 0) {
                    return target;
                } else {
                    right--;
                }
            }
        }
        if (overTarget)
            minDifference *= -1;
        return target - minDifference;
    }
}
