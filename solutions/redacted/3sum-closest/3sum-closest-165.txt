class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        //Take the closest Sum
        int closesSum = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < n - 2; i++) {
            if (i > 0 && nums[i] == nums[i - 1])
                continue;
            int j = i + 1;
            int k = n - 1;
            while (j < k) {
                //Take the current Sum
                int currentSum = nums[i] + nums[j] + nums[k];
                //if closest sum - target is large means current Sum is equivales to target
                if (Math.abs(currentSum - target) < Math.abs(closesSum - target)) {
                    closesSum = currentSum;
                }
                // If target greater than the current Sum means shift right to get the sum equivalent to target
                if (currentSum < target) {
                    j++;
                } else if (currentSum > target) {
                    k--;
                } else {
                    return currentSum;
                }
            }
        }
        return closesSum;
    }
}
