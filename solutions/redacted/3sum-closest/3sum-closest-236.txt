class Solution {

    public int method1(int[] nums, int target) {
        int minDiff = Integer.MAX_VALUE;
        int ans = 0;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length - 2; i++) {
            int l = i + 1, r = nums.length - 1;
            int sum = 0;
            int diff;
            while (l < r) {
                sum = nums[i] + nums[l] + nums[r];
                if (sum == target) {
                    return sum;
                }
                diff = Math.abs(sum - target);
                if (diff < minDiff) {
                    minDiff = diff;
                    ans = sum;
                }
                if (sum > target) {
                    r--;
                } else {
                    l++;
                }
            }
        }
        return ans;
    }
}
