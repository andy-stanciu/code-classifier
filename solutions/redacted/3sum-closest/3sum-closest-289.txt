class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        // Initialize closest sum
        int var1 = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1, right = nums.length - 1;
            while (left < right) {
                int currentSum = nums[i] + nums[left] + nums[right];
                if (currentSum == target) {
                    // Exact match found
                    return target;
                }
                // Update closestSum if currentSum is closer to target
                if (Math.abs(currentSum - target) < Math.abs(var1 - target)) {
                    var1 = currentSum;
                }
                if (currentSum < target) {
                    // Move left pointer to increase the sum
                    left++;
                } else {
                    // Move right pointer to decrease the sum
                    right--;
                }
            }
        }
        return var1;
    }
}
