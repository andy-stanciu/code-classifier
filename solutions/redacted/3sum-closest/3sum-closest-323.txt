class Solution {

    public int method1(int[] nums, int target) {
        if (nums == null)
            return 0;
        Arrays.sort(nums);
        int var1 = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (right > left) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == target) {
                    return sum;
                }
                if (Math.abs(target - var1) > Math.abs(target - sum)) {
                    var1 = sum;
                } else {
                    if (sum < target) {
                        left++;
                    } else
                        right--;
                }
            }
        }
        return var1;
    }
}
