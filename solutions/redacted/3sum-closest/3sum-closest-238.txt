class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int min = Integer.MAX_VALUE;
        int best = 0;
        int length = nums.length;
        for (int i = 0; i < length; i++) {
            int cur = nums[i];
            int firstPointer = i + 1;
            int secondPointer = length - 1;
            while (firstPointer < secondPointer) {
                int currentSum = cur + nums[firstPointer] + nums[secondPointer];
                int difference = Math.abs(target - currentSum);
                if (difference < min) {
                    min = difference;
                    best = currentSum;
                }
                if (currentSum == target) {
                    return (target);
                }
                if (currentSum < target) {
                    firstPointer++;
                }
                if (currentSum > target) {
                    secondPointer--;
                }
            }
        }
        return (best);
    }
}
