class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < n - 2; i++) {
            int j = i + 1;
            int k = n - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (Math.abs(ans - target) > Math.abs(sum - target)) {
                    ans = sum;
                }
                if (sum == target) {
                    return sum;
                } else if (sum > target) {
                    --k;
                } else {
                    ++j;
                }
            }
        }
        return ans;
    }
}
