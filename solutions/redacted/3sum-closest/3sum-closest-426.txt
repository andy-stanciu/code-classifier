class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int numsLength = nums.length;
        Integer var1 = null;
        for (int i = 0; i < numsLength; i++) {
            int currentNum = nums[i];
            int startCursor = i + 1;
            int endCursor = numsLength - 1;
            while (startCursor < endCursor) {
                int sum = currentNum + nums[startCursor] + nums[endCursor];
                if (var1 == null) {
                    var1 = sum;
                } else {
                    int distanceWithSum = Math.abs(target - sum);
                    int var2 = Math.abs(target - var1);
                    if (distanceWithSum < var2) {
                        var1 = sum;
                    }
                }
                if (sum > target) {
                    endCursor--;
                } else if (sum < target) {
                    startCursor++;
                } else {
                    startCursor++;
                    endCursor--;
                }
            }
        }
        return var1;
    }
}
