class Solution {

    public int method1(int[] nums, int target) {
        // Sorting the integer array
        Arrays.sort(nums);
        // Getting length of integer array
        int len = nums.length;
        // Initialize closest sum with the sum of the first three elements
        int var1 = nums[0] + nums[1] + nums[2];
        for (int current = 0; current < len - 2; current++) {
            // Creating two pointers for two sum logic
            int left = current + 1;
            int right = len - 1;
            while (left < right) {
                int sum = nums[current] + nums[left] + nums[right];
                // if we have found a 3sum that is equal to target then no need to check further just return the calculated sum
                if (sum == target) {
                    return sum;
                }
                if (Math.abs(target - sum) <= Math.abs(target - var1)) {
                    // Update closest sum if the current sum is closer to the target
                    var1 = sum;
                }
                if (sum < target) {
                    // Increment left to increase the sum
                    left++;
                } else {
                    // Decrement right to decrease the sum
                    right--;
                }
            }
        }
        return var1;
    }
}
