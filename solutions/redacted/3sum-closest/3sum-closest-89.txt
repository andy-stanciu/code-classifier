class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        int min = Integer.MAX_VALUE;
        int sum = 0;
        int sum1 = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    sum1 = nums[i] + nums[j] + nums[k];
                    int diff = Math.abs(sum1 - target);
                    if (diff < min) {
                        sum = sum1;
                        min = diff;
                    }
                }
            }
        }
        return sum;
    }
}
