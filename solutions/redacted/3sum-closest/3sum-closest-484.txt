class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int ans = 0;
        int var1 = Integer.MAX_VALUE;
        for (int i = 0; i < n - 2; i++) {
            int l = i + 1;
            int r = n - 1;
            while (l < r) {
                int sum = nums[i] + nums[l] + nums[r];
                int dist = Math.abs(target - sum);
                if (dist < var1) {
                    ans = sum;
                    var1 = dist;
                }
                if (sum < target)
                    l++;
                else
                    r--;
            }
        }
        return ans;
    }
}
