class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int closeSum = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int addition = nums[i] + nums[j] + nums[k];
                if (Math.abs(target - addition) < Math.abs(target - closeSum)) {
                    closeSum = addition;
                } else if (addition == target) {
                    return target;
                } else if (addition < target) {
                    j++;
                } else {
                    k--;
                }
            }
        }
        return closeSum;
    }
}
