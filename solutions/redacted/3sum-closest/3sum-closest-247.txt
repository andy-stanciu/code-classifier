class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int n = nums.length;
        int var1 = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < n - 2; i++) {
            int l = i + 1;
            int r = n - 1;
            while (l < r) {
                int currentSum = nums[i] + nums[l] + nums[r];
                if (Math.abs(currentSum - target) < Math.abs(var1 - target)) {
                    var1 = currentSum;
                }
                if (currentSum > target)
                    r--;
                else if (currentSum < target)
                    l++;
                else
                    return currentSum;
            }
        }
        return var1;
    }
}
