class Solution {

    public int method1(int[] arr, int target) {
        int n = arr.length;
        Arrays.sort(arr);
        int minDiff = Integer.MAX_VALUE;
        int closeSum = Integer.MAX_VALUE;
        for (int i = 0; i < n - 2; i++) {
            int j = i + 1;
            int right = n - 1;
            while (j < right) {
                int sum = arr[i] + arr[j] + arr[right];
                int diff = Math.abs(sum - target);
                if (diff < minDiff) {
                    minDiff = diff;
                    closeSum = sum;
                }
                if (target < sum) {
                    right--;
                } else {
                    j++;
                }
            }
        }
        return closeSum;
    }
}
