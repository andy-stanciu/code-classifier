class Solution {

    public int method1(int[] nums, int target) {
        // Sort the array to simplify the search
        Arrays.sort(nums);
        // Initialize with the first three elements
        int var1 = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (Math.abs(target - sum) < Math.abs(target - var1)) {
                    var1 = sum;
                }
                if (sum < target) {
                    // Increment left pointer if sum is less than target
                    left++;
                } else if (sum > target) {
                    // Decrement right pointer if sum is greater than target
                    right--;
                } else {
                    // Exact match found
                    return sum;
                }
            }
        }
        return var1;
    }
}
