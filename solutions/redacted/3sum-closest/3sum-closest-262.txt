class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        // Initialize closest sum with the sum of the first three elements
        int var1 = nums[0] + nums[1] + nums[2];
        for (int i = 0; i < nums.length - 2; i++) {
            int left = i + 1;
            int right = nums.length - 1;
            while (left < right) {
                int currSum = nums[i] + nums[left] + nums[right];
                if (Math.abs(target - currSum) < Math.abs(target - var1)) {
                    // Update closest sum if the current sum is closer to the target
                    var1 = currSum;
                }
                if (currSum < target) {
                    // Increment j to increase the sum
                    left++;
                } else {
                    // Decrement k to decrease the sum
                    right--;
                }
            }
        }
        return var1;
    }
}
