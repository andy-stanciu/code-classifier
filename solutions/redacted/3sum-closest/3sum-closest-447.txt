class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int length = nums.length;
        int cs = nums[0] + nums[1] + nums[2];
        // base case length == 3
        if (length == 3) {
            return cs;
        }
        for (int i = 0; i < length - 2; i++) {
            // pointer after i
            int j = i + 1;
            // pointer at end of array
            int k = length - 1;
            while (j < k) {
                // while j pointer is less than pointer at end of array
                int sum = nums[i] + nums[j] + nums[k];
                if (Math.abs(target - sum) < Math.abs(target - cs)) {
                    cs = sum;
                }
                if (sum < target) {
                    // array is sorted so we can keep iterating each value gets bigger
                    j++;
                } else {
                    // iterate down as values get smaller from sorted array
                    k--;
                }
            }
        }
        return cs;
    }
}
