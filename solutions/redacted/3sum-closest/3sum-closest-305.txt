class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int ans = 0;
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length - 2; i++) {
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (sum == target)
                    return sum;
                else if (sum < target)
                    j++;
                else
                    k--;
                int abs = Math.abs(sum - target);
                if (abs < min) {
                    min = abs;
                    ans = sum;
                }
            }
        }
        return ans;
    }
}
