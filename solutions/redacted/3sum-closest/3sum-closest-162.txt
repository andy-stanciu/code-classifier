class Solution {

    public int method1(int[] nums, int target) {
        Arrays.sort(nums);
        int sum = Integer.MAX_VALUE;
        int min = Integer.MAX_VALUE;
        for (int k = 0; k < nums.length; k++) {
            int l = k + 1;
            int r = nums.length - 1;
            while (l < r) {
                int calc = nums[k] + nums[l] + nums[r];
                int diff = Math.abs(target - calc);
                if (diff == 0) {
                    return target;
                }
                if (diff < min) {
                    min = diff;
                    sum = calc;
                }
                if (calc > target) {
                    r--;
                } else {
                    l++;
                }
            }
        }
        return sum;
    }
}
