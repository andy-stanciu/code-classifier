class Solution {

    public static void method1(int[][] matrix) {
        if (matrix.length != 1) {
            int countLength = 0;
            while (countLength < matrix.length) {
                int countSubLength = countLength;
                while (countSubLength < matrix[countLength].length) {
                    if (countSubLength != countLength) {
                        int tempInt = matrix[countLength][countSubLength];
                        matrix[countLength][countSubLength] = matrix[countSubLength][countLength];
                        matrix[countSubLength][countLength] = tempInt;
                    }
                    countSubLength++;
                }
                int leftSub = 0;
                int rightSub = matrix[countLength].length - 1;
                while (leftSub < rightSub) {
                    int tempNum = matrix[countLength][leftSub];
                    matrix[countLength][leftSub] = matrix[countLength][rightSub];
                    matrix[countLength][rightSub] = tempNum;
                    leftSub++;
                    rightSub--;
                }
                countLength++;
            }
        }
    }
}
