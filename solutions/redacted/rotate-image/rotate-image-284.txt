class Solution {

    public void method1(int[][] matrix) {
        // transpose of the matrix
        int row = matrix.length;
        int col = matrix[0].length;
        for (int i = 0; i < row - 1; i++) {
            // swap only when j > i this means we don\'t have to go for the last row
            for (int j = i + 1; j < col; j++) {
                //  and j is started from i+1
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        //swap the columns
        int startCol = 0, endCol = col - 1;
        while (startCol < endCol) {
            //swapping the columns\' elements
            for (int i = 0; i < row; i++) {
                int temp = matrix[i][startCol];
                matrix[i][startCol] = matrix[i][endCol];
                matrix[i][endCol] = temp;
            }
            // next pair columns
            startCol++;
            endCol--;
        }
    }
}
