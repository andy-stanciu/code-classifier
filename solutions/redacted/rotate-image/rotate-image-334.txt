class Solution {

    public void method1(int[][] mat) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] tempx = new int[m][n];
        int p = 0, q = 0;
        for (int i = 0; i < n; i++) {
            q = 0;
            for (int j = m - 1; j >= 0; j--) {
                int temp = mat[j][i];
                tempx[p][q] = temp;
                q++;
            }
            p++;
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                mat[i][j] = tempx[i][j];
            }
        }
    }
}
