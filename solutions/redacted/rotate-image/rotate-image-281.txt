class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        for (int layer = 0; layer < n / 2; layer++) {
            int layerLength = n - (2 * layer);
            int layerStart = layer;
            int layerEnd = layerStart + layerLength - 1;
            for (int i = 0; i < layerLength - 1; i++) {
                int topLeft = matrix[layerStart][layerStart + i];
                matrix[layerStart][layerStart + i] = matrix[layerEnd - i][layerStart];
                matrix[layerEnd - i][layerStart] = matrix[layerEnd][layerEnd - i];
                matrix[layerEnd][layerEnd - i] = matrix[layerStart + i][layerEnd];
                matrix[layerStart + i][layerEnd] = topLeft;
            }
        }
    }
}
