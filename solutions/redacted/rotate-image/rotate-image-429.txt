class Solution {

    public void method1(int[][] matrix) {
        for (int i = 0; i < matrix.length / 2; i++) {
            turnBox(matrix, i);
        }
    }

    public void turnBox(int[][] matrix, int layer) {
        int l = matrix.length - 1;
        for (int i = layer; i < matrix.length - 1 - layer; i++) {
            int aux = matrix[layer][i];
            matrix[layer][i] = matrix[l - i][layer];
            matrix[l - i][layer] = matrix[l - layer][l - i];
            matrix[l - layer][l - i] = matrix[i][l - layer];
            matrix[i][l - layer] = aux;
        }
    }
}
