class Solution {

    public void method1(int[][] matrix) {
        // top border
        int iMin = 0;
        // bottom border
        int iMax = matrix.length - 1;
        // left border
        int jMin = 0;
        // right border
        int jMax = matrix.length - 1;
        for (int z = 0; iMin < iMax && jMin < jMax; z++) {
            for (int i = iMin, j = jMin, l = iMax, k = jMax; i < iMax; i++, j++, k--, l--) {
                // in the first iteration, for the marix -
                // [5, 1, 9, 11]
                // [2, 4, 8, 10]
                // [13,3, 6, 7]
                // [15,14,12,16]
                // matrix[iMin][j] = 5, going towards right
                // matrix[i][jMax] = 11, going downwards
                // matrix[iMax][k] = 16, going towards left
                // matrix[l][jMin] = 15, going upwards
                int temp = matrix[iMin][j];
                matrix[iMin][j] = matrix[i][jMax];
                matrix[i][jMax] = temp;
                temp = matrix[iMax][k];
                matrix[iMax][k] = matrix[l][jMin];
                matrix[l][jMin] = temp;
                temp = matrix[iMin][j];
                matrix[iMin][j] = matrix[iMax][k];
                matrix[iMax][k] = temp;
            }
            iMin++;
            jMin++;
            iMax--;
            jMax--;
        }
    }
}
