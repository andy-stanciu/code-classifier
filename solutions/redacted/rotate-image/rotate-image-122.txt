class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        for (int i = 0; i < n; i++) for (int j = i; j < n; j++) matrix[i][j] = matrix[i][j] ^ matrix[j][i] ^ (matrix[j][i] = matrix[i][j]);
        int k;
        for (int i = 0; i < n; i++) {
            k = n / 2;
            int r = 0, c = n - 1;
            while (k != 0) {
                matrix[i][r] = matrix[i][r] ^ matrix[i][c] ^ (matrix[i][c] = matrix[i][r]);
                r++;
                c--;
                k--;
            }
        }
    }
}
