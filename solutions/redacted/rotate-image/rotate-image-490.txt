class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        // left
        int l = 0;
        // right
        int r = n - 1;
        while (l < r) {
            for (int j = 0; j < n; j++) {
                // rotation 90 deg
                int x = matrix[l][j];
                matrix[l][j] = matrix[r][j];
                matrix[r][j] = x;
            }
            // increment left
            l++;
            // decrement right
            r--;
        }
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                // transpose
                int x = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = x;
            }
        }
    }
}
