class Solution {

    public void method1(int[][] matrix) {
        int l = 0, r = matrix.length - 1;
        while (l < r) {
            for (int index = 0; index < (r - l); index++) {
                //because square matrix
                int top = l, bottom = r;
                //iteration
                // top 0th, top 1st, top 2nd
                // int index=l-r;
                //store value of top left so that it is stored before getting replaced
                int topLeft = matrix[top][l + index];
                //topleft = bottom left
                matrix[top][l + index] = matrix[bottom - index][l];
                //bottom left =bottom right
                matrix[bottom - index][l] = matrix[bottom][r - index];
                //bottom right = top right
                matrix[bottom][r - index] = matrix[top + index][r];
                //top right = top left
                matrix[top + index][r] = topLeft;
            }
            //sqaure matrix sub problem
            l++;
            r--;
        }
    }
}
