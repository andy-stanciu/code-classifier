class Solution {

    public void method1(int[][] matrix) {
        int len = matrix.length;
        //first transpose the matrix
        for (int i = 0; i < len; i++) {
            for (int j = i; j < len; j++) {
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        // now reverse the matrix;
        int left = 0;
        int right = len - 1;
        while (left != right && left < right) {
            for (int i = 0; i < len; i++) {
                int temp = matrix[i][left];
                matrix[i][left] = matrix[i][right];
                matrix[i][right] = temp;
            }
            left++;
            right--;
        }
    }
}
