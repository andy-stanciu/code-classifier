class Solution {

    public void method1(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = i; j < matrix[0].length; j++) {
                int temp = 0;
                temp = matrix[i][j];
                //reverse row and columns
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
                //here we get the transpose of the matrix using a temp variable
            }
        }
        for (int i = 0; i < matrix.length; i++) {
            for (//here using two pointers we have to just reverse the row
            //here using two pointers we have to just reverse the row
            int j = 0; //here using two pointers we have to just reverse the row
            j < matrix.length / 2; j++) {
                int temp = 0;
                temp = matrix[i][j];
                //reversing each row
                matrix[i][j] = matrix[i][matrix.length - 1 - j];
                matrix[i][matrix.length - 1 - j] = temp;
            }
        }
    }
}
