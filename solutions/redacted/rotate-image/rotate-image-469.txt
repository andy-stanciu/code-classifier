class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        int[][] res = matrix;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int temp = res[i][j];
                res[i][j] = res[j][i];
                res[j][i] = temp;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n / 2; j++) {
                int temp2 = res[i][j];
                res[i][j] = res[i][n - j - 1];
                res[i][n - j - 1] = temp2;
            }
        }
    }
}
