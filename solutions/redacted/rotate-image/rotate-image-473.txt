class Solution {

    public void method1(int[][] matrix) {
        // Step 1: Transpose the matrix (convert rows to columns).
        trandposeMatrix(matrix);
        // Step 2: Reverse each row of the transposed matrix to achieve 90-degree clockwise rotation.
        reverseMatrix(matrix);
    }

    // Method to transpose the matrix: swap elements across the diagonal.
    public void trandposeMatrix(int[][] matrix) {
        int rows = matrix.length;
        int cols = matrix.length;
        for (int row = 0; row < rows; row++) {
            for (int col = row; col < cols; col++) {
                // Skip the diagonal elements.
                if (col == row)
                    continue;
                // Swap elements across the diagonal.
                int temp = matrix[row][col];
                matrix[row][col] = matrix[col][row];
                matrix[col][row] = temp;
            }
        }
    }

    // Method to reverse each row of the matrix.
    public void reverseMatrix(int[][] matrix) {
        int rows = matrix.length;
        int cols = matrix.length;
        for (int row = 0; row < rows; row++) {
            int left = 0;
            int right = cols - 1;
            // Reverse the current row by swapping elements from both ends.
            while (left < right) {
                int temp = matrix[row][left];
                matrix[row][left] = matrix[row][right];
                matrix[row][right] = temp;
                left++;
                right--;
            }
        }
    }
}
