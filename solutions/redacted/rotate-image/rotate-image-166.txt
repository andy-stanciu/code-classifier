class Solution {

    public void method1(int[][] matrix) {
        int total = matrix.length / 2, size = matrix.length;
        for (int level = 0; level <= total; level++) {
            for (int curr = level; curr < size - level - 1; curr++) {
                int temp = matrix[level][curr];
                matrix[level][curr] = matrix[size - curr - 1][level];
                matrix[size - curr - 1][level] = matrix[size - level - 1][size - curr - 1];
                matrix[size - level - 1][size - curr - 1] = matrix[curr][size - level - 1];
                matrix[curr][size - level - 1] = temp;
            }
        }
    }
}
