class Solution {

    public void method1(int[][] matrix) {
        int n = matrix.length;
        int[][] var1 = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                var1[j][n - 1 - i] = matrix[i][j];
            }
        }
        for (int i = 0; i < n; i++) {
            System.arraycopy(var1[i], 0, matrix[i], 0, n);
        }
    }
}
