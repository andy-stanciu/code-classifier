class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        List<String> list = new ArrayList<>();
        answer(ans, list, 0, s);
        return ans;
    }

    public static void answer(List<List<String>> ans, List<String> list, int currentIndex, String s) {
        if (currentIndex == s.length()) {
            ans.add(new ArrayList<>(list));
            return;
        }
        for (int j = currentIndex; j < s.length(); j++) {
            if (method1(s.substring(currentIndex, j + 1))) {
                list.add(s.substring(currentIndex, j + 1));
                answer(ans, list, j + 1, s);
                list.remove(list.size() - 1);
            }
        }
    }

    public static boolean method1(String str) {
        return str.equals(new StringBuilder(str).reverse().toString());
    }
}
