class Solution {

    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        List<String> ll = new ArrayList<>();
        function(0, s, ll, ans);
        return ans;
    }

    public static void function(int idx, String s, List<String> ll, List<List<String>> ans) {
        if (idx == s.length()) {
            ans.add(new ArrayList<>(ll));
            return;
        }
        for (int i = idx; i < s.length(); i++) {
            if (method1(s, idx, i)) {
                ll.add(s.substring(idx, i + 1));
                function(i + 1, s, ll, ans);
                ll.remove(ll.size() - 1);
            }
        }
    }

    public static boolean method1(String s, int start, int end) {
        while (start <= end) {
            if (s.charAt(start++) != s.charAt(end--)) {
                return false;
            }
        }
        return true;
    }
}
