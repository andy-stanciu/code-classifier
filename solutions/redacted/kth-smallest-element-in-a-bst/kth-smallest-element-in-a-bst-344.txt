/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<Integer> helper(TreeNode root, ArrayList<Integer> lt) {
        if (root == null) {
            return lt;
        }
        lt.add(root.val);
        helper(root.left, lt);
        helper(root.right, lt);
        return lt;
    }

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> alt = new ArrayList<>();
        helper(root, alt);
        Collections.sort(alt);
        return alt.get(k - 1);
    }
}
