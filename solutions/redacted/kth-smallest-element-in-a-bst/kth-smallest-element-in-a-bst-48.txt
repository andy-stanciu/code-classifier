/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int var1 = Integer.MAX_VALUE;

    private int globalK;

    public int method1(TreeNode root, int k) {
        this.globalK = k;
        traverse(root);
        return var1;
    }

    private void traverse(TreeNode node) {
        if (node == null)
            return;
        traverse(node.left);
        if (globalK == 1) {
            var1 = node.val;
        }
        globalK--;
        traverse(node.right);
    }
}
