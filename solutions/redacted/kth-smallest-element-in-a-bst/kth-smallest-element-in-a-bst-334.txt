/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int res = -1;

    public int method1(TreeNode root, int k) {
        if (null == root.left && null == root.right)
            return root.val;
        method2(root, k, 0);
        return res;
    }

    int method2(TreeNode root, int k, int parcel) {
        if (root == null)
            return parcel;
        // left
        parcel = method2(root.left, k, parcel);
        // root
        parcel = parcel + 1;
        if (parcel == k) {
            res = root.val;
        }
        // right
        parcel = method2(root.right, k, parcel);
        return parcel;
    }
}
