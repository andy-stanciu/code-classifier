 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private long var1 = Long.MIN_VALUE;

    public int method1(TreeNode root, int k) {
        int i = method1(root, k, 0);
        return var1 == Long.MIN_VALUE ? i : (int) var1;
    }

    private int method1(TreeNode node, int k, int i) {
        if (node == null)
            return i;
        else if (node.left == null && node.right == null)
            return ++i;
        if (var1 > Long.MIN_VALUE)
            return i;
        int il = method1(node.left, k, i);
        if (il == k && var1 == Long.MIN_VALUE)
            var1 = node.left.val;
        il++;
        if (il == k && var1 == Long.MIN_VALUE)
            var1 = node.val;
        int ir = method1(node.right, k, il);
        if (ir == k && var1 == Long.MIN_VALUE)
            var1 = node.right.val;
        return ir;
    }
}
