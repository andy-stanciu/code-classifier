/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    int result = 0;

    public int method1(TreeNode root, int k) {
        small(root, k);
        return result;
    }

    public void small(TreeNode root, int k) {
        if (root == null)
            return;
        small(root.left, k);
        count++;
        if (count == k) {
            result = root.val;
            return;
        }
        small(root.right, k);
    }
}
