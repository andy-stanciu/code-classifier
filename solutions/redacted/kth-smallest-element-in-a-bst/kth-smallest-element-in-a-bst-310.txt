/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int size = 1;

    public boolean result = false;

    public int value = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        method1(root.left, k);
        if (size == k) {
            size++;
            result = true;
            value = root.val;
        } else {
            size++;
        }
        method1(root.right, k);
        if (result) {
            return value;
        }
        return root.val;
    }
}
