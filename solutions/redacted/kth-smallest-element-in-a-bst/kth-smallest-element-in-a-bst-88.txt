/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int val = 0;

    int d = 0;

    public int method1(TreeNode root, int k) {
        d = k;
        helper(root);
        return val;
    }

    void helper(TreeNode root) {
        if (root == null) {
            return;
        }
        helper(root.left);
        d--;
        if (d == 0) {
            val = root.val;
            return;
        }
        helper(root.right);
    }
}
