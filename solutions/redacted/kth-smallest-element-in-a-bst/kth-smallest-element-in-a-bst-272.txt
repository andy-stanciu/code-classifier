/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int count = 0;

    public int res = -1;

    public int method1(TreeNode root, int k) {
        count = 0;
        res = -1;
        method2(root, k);
        return res;
    }

    public void method2(TreeNode root, int k) {
        if (root != null) {
            method2(root.left, k);
            count++;
            if (count == k) {
                res = root.val;
                return;
            }
            method2(root.right, k);
        }
    }
}
