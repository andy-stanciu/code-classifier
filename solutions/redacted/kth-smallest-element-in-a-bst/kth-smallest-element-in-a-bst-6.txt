/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = 0;

    int x;

    public void solve(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null)
            solve(root.left);
        x--;
        if (x == 0)
            ans = root.val;
        if (root.right != null)
            solve(root.right);
    }

    public int method1(TreeNode root, int k) {
        x = k;
        solve(root);
        return ans;
    }
}
