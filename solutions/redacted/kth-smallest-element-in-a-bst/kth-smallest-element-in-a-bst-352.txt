/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        LinkedList<Integer> ll = new LinkedList<>();
        dfs(root, ll);
        return ll.get(k - 1);
    }

    public void dfs(TreeNode root, LinkedList<Integer> ll) {
        if (root == null)
            return;
        dfs(root.left, ll);
        ll.add(root.val);
        dfs(root.right, ll);
    }
}
