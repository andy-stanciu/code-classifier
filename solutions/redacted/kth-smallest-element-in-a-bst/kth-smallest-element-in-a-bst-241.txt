/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0, result = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method1(root.left, k);
        count++;
        if (count == k)
            result = root.val;
        else
            method1(root.right, k);
        return result;
    }
}
