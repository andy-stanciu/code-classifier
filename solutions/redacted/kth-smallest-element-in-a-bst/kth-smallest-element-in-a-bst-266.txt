/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> values = new ArrayList<>();
        method2(root, values);
        return values.get(k - 1);
    }

    private void method2(TreeNode node, List<Integer> values) {
        if (node == null) {
            return;
        }
        method2(node.left, values);
        values.add(node.val);
        method2(node.right, values);
    }
}
