/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;n
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        TreeNode temp = root;
        TreeNode prev = root;
        while (k != 0) {
            temp = root;
            prev = root;
            while (temp.left != null) {
                prev = temp;
                temp = temp.left;
            }
            if (temp.right != null) {
                prev.left = temp.right;
            } else {
                prev.left = null;
            }
            k--;
        }
        return temp.val;
    }
}
