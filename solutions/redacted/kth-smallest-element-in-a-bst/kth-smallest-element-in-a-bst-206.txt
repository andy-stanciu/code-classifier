/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    Queue<TreeNode> q = new LinkedList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        for (int i = 0; i < k - 1; i++) {
            q.poll();
        }
        TreeNode ans = q.poll();
        return ans.val;
    }

    TreeNode method2(TreeNode node) {
        if (node == null)
            return null;
        TreeNode left = method2(node.left);
        q.offer(node);
        TreeNode right = method2(node.right);
        return node;
    }
}
