class Solution {

    private int i;

    private int method1(TreeNode root, int k) {
        if (root == null)
            return -1;
        int left = method1(root.left, k);
        if (left != -1)
            return left;
        i++;
        if (i == k)
            return root.val;
        int right = method1(root.right, k);
        if (right != -1)
            return right;
        return -1;
    }

    public int method2(TreeNode root, int k) {
        i = 0;
        return method1(root, k);
    }
}
