class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> ans = method2(root);
        return ans.get(k - 1);
    }

    public ArrayList<Integer> list = new ArrayList<>();

    public ArrayList<Integer> method2(TreeNode root) {
        if (root != null) {
            method2(root.left);
            list.add(root.val);
            method2(root.right);
        }
        return list;
    }
}
