class Solution {

    int count;

    int s;

    public int method1(TreeNode root, int k) {
        count = 0;
        dfs(root, k);
        return s;
    }

    public void dfs(TreeNode root, int k) {
        if (root == null)
            return;
        dfs(root.left, k);
        count++;
        if (count == k) {
            s = root.val;
            return;
        }
        dfs(root.right, k);
    }
}
