class Solution {

    int ans = 0, i = 1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root != null) {
            method2(root.left, k);
            if (k == i) {
                ans = root.val;
            }
            i++;
            method2(root.right, k);
        }
    }
}
