/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    public int method1(TreeNode root, int k) {
        if (root.left != null) {
            int result = method1(root.left, k);
            if (var1 == k)
                return result;
        }
        if (++var1 == k || root.right == null)
            return root.val;
        else
            return method1(root.right, k);
    }
}
