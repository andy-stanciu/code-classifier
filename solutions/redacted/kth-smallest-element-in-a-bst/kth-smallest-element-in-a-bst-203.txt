/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int small = 0;

    int count = 0;

    public int method1(TreeNode root, int val) {
        method2(root, val);
        return small;
    }

    public void method2(TreeNode root, int val) {
        if (root == null || count > val)
            return;
        method2(root.left, val);
        count++;
        if (count == val) {
            small = root.val;
        }
        method2(root.right, val);
        return;
    }
}
