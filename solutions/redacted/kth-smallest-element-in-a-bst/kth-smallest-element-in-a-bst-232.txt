class Solution {

    public int method1(TreeNode root, int k) {
        int n = 0;
        Stack<TreeNode> stack = new Stack<>();
        TreeNode current = root;
        while (current != null || !stack.isEmpty()) {
            while (current != null) {
                stack.push(current);
                current = current.left;
            }
            current = stack.pop();
            n++;
            if (n == k)
                return current.val;
            current = current.right;
        }
        return -1;
    }
}
