/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int count = 0;

    private int result = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return result;
    }

    private void method2(TreeNode node, int k) {
        if (node == null)
            return;
        //Traverse the left subtree
        method2(node.left, k);
        //Process the right node
        count++;
        if (count == k) {
            result = node.val;
            return;
        }
        //Traverse the right subtree
        method2(node.right, k);
    }
}
