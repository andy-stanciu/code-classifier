/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        arr = method2(root, arr);
        return arr.get(k - 1);
    }

    public ArrayList<Integer> method2(TreeNode root, ArrayList<Integer> arr) {
        if (root == null) {
            arr.add(-1);
            return arr;
        }
        method3(root, arr);
        return arr;
    }

    public void method3(TreeNode root, ArrayList<Integer> arr) {
        if (root == null) {
            return;
        }
        method3(root.left, arr);
        arr.add(root.val);
        method3(root.right, arr);
    }
}
