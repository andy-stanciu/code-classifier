/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int poss;

    int ans;

    void value(TreeNode root) {
        if (root != null) {
            if (root.left != null) {
                value(root.left);
            }
            poss--;
            // System.out.println(root.val+" "+poss);
            if (poss == 0)
                ans = root.val;
            if (root.right != null) {
                value(root.right);
            }
        }
        return;
    }

    public int method1(TreeNode root, int k) {
        poss = k;
        value(root);
        return ans;
    }
}
