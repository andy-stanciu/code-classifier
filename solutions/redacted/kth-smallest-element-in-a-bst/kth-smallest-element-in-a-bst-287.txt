/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int ans = 0;

    public int method1(TreeNode root, int k) {
        int[] arr = new int[1];
        arr[0] = k;
        method2(root, arr);
        return ans;
    }

    public void method2(TreeNode root, int[] arr) {
        if (root == null)
            return;
        method2(root.left, arr);
        arr[0] = arr[0] - 1;
        if (arr[0] == 0) {
            ans = root.val;
            return;
        }
        method2(root.right, arr);
    }
}
