/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        int[] c = { 0 };
        int[] ans = { -1 };
        traverse(root, k, c, ans);
        return ans[0];
    }

    public void traverse(TreeNode root, int k, int[] c, int[] ans) {
        if (root == null)
            return;
        traverse(root.left, k, c, ans);
        c[0]++;
        if (c[0] == k) {
            ans[0] = root.val;
            return;
        }
        traverse(root.right, k, c, ans);
    }
}
