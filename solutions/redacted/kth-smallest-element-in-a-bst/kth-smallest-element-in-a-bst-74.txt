/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> ans = new ArrayList<Integer>();

    public int method1(TreeNode root, int k) {
        helper(root);
        //Collections.sort(ans);
        return ans.get(k - 1);
    }

    public void helper(TreeNode temp) {
        if (temp == null)
            return;
        helper(temp.left);
        ans.add(temp.val);
        helper(temp.right);
    }
}
