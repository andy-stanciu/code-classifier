/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int idx;

    int val;

    public int method1(TreeNode root, int k) {
        idx = 0;
        val = -1;
        method2(root, k);
        return val;
    }

    void method2(TreeNode node, int k) {
        if (node != null) {
            method2(node.left, k);
            idx++;
            if (idx == k) {
                val = node.val;
                return;
            }
            method2(node.right, k);
        }
    }
}
