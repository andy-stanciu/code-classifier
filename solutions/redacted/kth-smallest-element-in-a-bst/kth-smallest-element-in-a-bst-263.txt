class Solution {

    public List<Integer> ans = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        method2(root);
        return ans.get(k - 1);
    }

    private void method2(TreeNode root) {
        if (root == null)
            return;
        if (root.left != null)
            method2(root.left);
        ans.add(root.val);
        if (root.right != null)
            method2(root.right);
    }
}
