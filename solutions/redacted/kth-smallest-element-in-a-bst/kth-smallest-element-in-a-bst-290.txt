/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> ds = new ArrayList<>();
        method2(root, ds);
        return ds.get(k - 1);
    }

    void method2(TreeNode root, List<Integer> ds) {
        if (root == null)
            return;
        method2(root.left, ds);
        ds.add(root.val);
        method2(root.right, ds);
    }
}
