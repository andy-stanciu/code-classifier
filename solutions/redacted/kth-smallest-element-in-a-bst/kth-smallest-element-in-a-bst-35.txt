class Solution {

    int ans = 0;

    ArrayList<Integer> arr = new ArrayList<>();

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        arr.add(root.val);
        if (arr.size() == k) {
            ans = root.val;
            return;
        }
        method2(root.right, k);
    }
}
