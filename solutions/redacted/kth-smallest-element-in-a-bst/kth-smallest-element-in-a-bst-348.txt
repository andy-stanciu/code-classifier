/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> data = new ArrayList<>();

    int size = 0;

    public int method1(TreeNode root, int k) {
        size = k;
        method2(root);
        return data.get(k - 1);
    }

    public void method2(TreeNode root) {
        if (data.size() >= size)
            return;
        if (root == null)
            return;
        method2(root.left);
        data.add(root.val);
        method2(root.right);
    }
}
