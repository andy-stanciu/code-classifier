/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        if (k == 1 && root.left == null && root.right == null) {
            return root.val;
        }
        List<Integer> valuesList = new ArrayList<>();
        method2(root, valuesList);
        return valuesList.get(k - 1);
    }

    private void method2(TreeNode root, List<Integer> valuesList) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            method2(root.left, valuesList);
        }
        valuesList.add(root.val);
        if (root.right != null) {
            method2(root.right, valuesList);
        }
    }
}
