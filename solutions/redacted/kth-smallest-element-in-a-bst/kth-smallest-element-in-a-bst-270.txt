/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        int[] count = { 0 };
        int[] result = { -1 };
        method2(root, k, count, result);
        return result[0];
    }

    private void method2(TreeNode root, int k, int[] count, int[] result) {
        if (root == null) {
            return;
        }
        method2(root.left, k, count, result);
        count[0]++;
        if (count[0] == k) {
            result[0] = root.val;
            return;
        }
        method2(root.right, k, count, result);
    }
}
