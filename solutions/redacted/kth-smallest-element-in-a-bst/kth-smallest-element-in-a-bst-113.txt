/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int cnt = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return ans;
    }

    void method2(TreeNode root, int k) {
        if (root != null) {
            method2(root.left, k);
            cnt++;
            if (cnt == k) {
                ans = root.val;
            }
            method2(root.right, k);
        }
    }
}
