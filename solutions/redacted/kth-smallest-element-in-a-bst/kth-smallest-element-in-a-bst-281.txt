class Solution {

    int count = 0;

    int ans;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        method1(root.left, k);
        count++;
        if (count == k) {
            ans = root.val;
        }
        method1(root.right, k);
        if (k > count) {
            return -1;
        }
        return ans;
    }
}
