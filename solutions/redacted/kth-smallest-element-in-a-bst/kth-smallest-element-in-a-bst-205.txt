/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count;

    int result;

    public int method1(TreeNode root, int k) {
        count = 0;
        result = 0;
        method2(root, k);
        return result;
    }

    public void method2(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        method2(root.left, k);
        count += 1;
        if (count == k) {
            result = root.val;
        }
        method2(root.right, k);
    }
}
