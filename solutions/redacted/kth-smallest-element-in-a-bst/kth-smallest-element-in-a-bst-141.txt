/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    //List<Integer> li = new ArrayList<>();
    int count = 0;

    int value;

    public int method1(TreeNode root, int k) {
        calc(root, k);
        //System.out.println(li);
        return value;
    }

    public void calc(TreeNode node, int k) {
        if (node == null)
            return;
        calc(node.left, k);
        //li.add(node.val);
        count++;
        if (count == k) {
            value = node.val;
        }
        calc(node.right, k);
    }
}
