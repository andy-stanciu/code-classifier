/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    static int method1(TreeNode root, int k) {
        List<Integer> var1 = new ArrayList<>();
        method2(root, var1);
        if (k - 1 < var1.size()) {
            return var1.get(k - 1);
        } else {
            // If k is out of bounds
            return -1;
        }
    }

    static void method2(TreeNode root, List<Integer> var1) {
        if (root == null) {
            return;
        }
        method2(root.left, var1);
        var1.add(root.val);
        method2(root.right, var1);
    }
}
