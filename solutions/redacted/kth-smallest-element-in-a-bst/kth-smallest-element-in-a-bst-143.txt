/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int var1 = -1;

    private int c = 0;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return var1;
    }

    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        c++;
        if (c == k) {
            var1 = root.val;
            return;
        }
        method2(root.right, k);
    }
}
