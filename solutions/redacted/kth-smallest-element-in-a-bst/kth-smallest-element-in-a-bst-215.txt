public class Solution {

    // To keep track of the count of nodes visited
    private int count = 0;

    // To store the Kth smallest element
    private int result = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return result;
    }

    private void method2(TreeNode node, int k) {
        if (node == null) {
            return;
        }
        // Traverse the left subtree
        method2(node.left, k);
        // Process the current node
        count++;
        if (count == k) {
            result = node.val;
            return;
        }
        // Traverse the right subtree
        method2(node.right, k);
    }
}
