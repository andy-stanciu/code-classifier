/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int kCounter = 0;

    int res = -1;

    public int method1(TreeNode root, int k) {
        method2(root, k);
        return res;
    }

    public void method2(TreeNode node, int k) {
        if (node != null) {
            method2(node.left, k);
            kCounter++;
            if (kCounter == k) {
                res = node.val;
            }
            method2(node.right, k);
        }
    }
}
