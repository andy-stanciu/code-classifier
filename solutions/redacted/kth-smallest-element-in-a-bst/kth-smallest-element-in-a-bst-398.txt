/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int count = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        int l = method1(root.left, k);
        if (l != 0) {
            return l;
        }
        count++;
        if (count == k) {
            return root.val;
        }
        int r = method1(root.right, k);
        return r;
    }
}
