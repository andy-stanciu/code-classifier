/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> l = new ArrayList<>();

    public void method1(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        l.add(root.val);
        method1(root.right);
    }

    public int method2(TreeNode root, int k) {
        if (root == null)
            return 0;
        // List<Integer> list = inorder(root);
        method1(root);
        return l.get(k - 1);
    }
}
