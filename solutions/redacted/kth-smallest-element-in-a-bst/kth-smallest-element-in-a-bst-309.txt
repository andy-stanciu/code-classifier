/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    int val = -1;

    public int method1(TreeNode root, int k) {
        call(root, k);
        return val;
    }

    public void call(TreeNode root, int k) {
        if (root == null) {
            return;
        }
        call(root.left, k);
        var1++;
        if (var1 == k) {
            val = root.val;
        }
        call(root.right, k);
    }
}
