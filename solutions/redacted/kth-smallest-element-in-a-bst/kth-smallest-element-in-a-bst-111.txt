/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void method1(TreeNode root, List<Integer> res) {
        if (root == null) {
            return;
        }
        method1(root.left, res);
        res.add(root.val);
        method1(root.right, res);
    }

    public int method2(TreeNode root, int k) {
        List<Integer> res = new ArrayList<>();
        method1(root, res);
        Collections.sort(res);
        return res.get(k - 1);
    }
}
