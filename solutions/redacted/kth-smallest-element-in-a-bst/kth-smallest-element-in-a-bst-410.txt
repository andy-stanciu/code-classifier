/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int k;

    private int result;

    /**
     * Time Complexity: O(n)
     * Space Complexity: O(1)
     */
    public int method1(TreeNode root, int k) {
        this.k = k;
        method2(root);
        return result;
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        // InOrder Traversal
        method2(root.left);
        k--;
        if (k == 0) {
            result = root.val;
            return;
        }
        method2(root.right);
    }
}
