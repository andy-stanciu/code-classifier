class Solution {

    private int count = 0, ans = -1;

    public int method1(TreeNode root, int k) {
        // Reset counter
        count = 0;
        // Reset answer
        ans = -1;
        method2(root, k);
        return ans;
    }

    // Helper method to perform in-order traversal
    private void method2(TreeNode root, int k) {
        if (root == null)
            return;
        method2(root.left, k);
        count++;
        if (count == k) {
            // If count matches k, store answer and exit
            ans = root.val;
            return;
        }
        method2(root.right, k);
    }
}
