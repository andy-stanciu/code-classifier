/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        method2(root, list);
        Collections.sort(list);
        return list.get(k - 1);
    }

    public void method2(TreeNode root, ArrayList<Integer> list) {
        if (root != null) {
            method2(root.left, list);
            list.add(root.val);
            method2(root.right, list);
        }
        return;
    }
}
