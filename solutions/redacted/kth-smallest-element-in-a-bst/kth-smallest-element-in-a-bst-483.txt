class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> values = new ArrayList<>();
        method2(root, values);
        return values.get(k - 1);
    }

    private void method2(TreeNode root, List<Integer> values) {
        // base case
        if (root == null) {
            return;
        }
        // recursive case
        method2(root.left, values);
        values.add(root.val);
        method2(root.right, values);
    }
}
