class Solution {

    int[] var1;

    int var2 = 0;

    public int method1(TreeNode root, int k) {
        var1 = new int[k];
        method2(root);
        return var1[k - 1];
    }

    private void method2(TreeNode root) {
        if (root == null) {
            return;
        }
        method2(root.left);
        if (var2 > var1.length - 1) {
            return;
        }
        var1[var2++] = root.val;
        method2(root.right);
    }
}
