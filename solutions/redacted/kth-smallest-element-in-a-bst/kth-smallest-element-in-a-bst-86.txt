/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        Solution ob = new Solution();
        ArrayList<Integer> li = new ArrayList<>();
        li = ob.method2(root);
        if (li.size() == 0)
            return -1;
        else
            return li.get(k - 1);
    }

    public ArrayList<Integer> method2(TreeNode root) {
        ArrayList<Integer> list = new ArrayList<>();
        if (root == null)
            return list;
        list.addAll(method2(root.left));
        list.add(root.val);
        list.addAll(method2(root.right));
        return list;
    }
}
