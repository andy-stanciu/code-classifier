/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> a = new ArrayList<>();
        method2(root, a);
        return a.get(k - 1);
    }

    public void method2(TreeNode root, ArrayList<Integer> a) {
        if (root == null) {
            return;
        }
        method2(root.left, a);
        a.add(root.val);
        method2(root.right, a);
    }
}
