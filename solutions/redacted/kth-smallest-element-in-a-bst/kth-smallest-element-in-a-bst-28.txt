class Solution {

    int k;

    private int method1(TreeNode node) {
        if (node == null)
            return -1;
        var ans = method1(node.left);
        if (ans != -1 && --k <= 0)
            return ans;
        if (--k <= 0)
            return node.val;
        return method1(node.right);
    }

    public int method2(TreeNode root, int k) {
        this.k = k;
        return method1(root);
    }
}
