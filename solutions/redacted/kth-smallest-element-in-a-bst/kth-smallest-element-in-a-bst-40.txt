/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    ArrayList<Integer> ls = new ArrayList<>();

    private void method1(TreeNode root, ArrayList ls) {
        if (root == null) {
            return;
        }
        method1(root.left, ls);
        ls.add(root.val);
        method1(root.right, ls);
        return;
    }

    public int method2(TreeNode root, int k) {
        method1(root, ls);
        return ls.get(k - 1);
    }
}
