/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<Integer> l = new ArrayList<Integer>();

    public List<Integer> small(TreeNode node, int k) {
        if (node == null) {
            return l;
        }
        small(node.left, k);
        l.add(node.val);
        small(node.right, k);
        return l;
    }

    public int method1(TreeNode root, int k) {
        List<Integer> num = small(root, k);
        return num.get(k - 1);
    }
}
