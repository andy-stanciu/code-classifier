/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> list = new LinkedList<>();
        int[] counter = { 0 };
        method2(root, k, counter, list);
        return list.get(0);
    }

    public void method2(TreeNode node, int k, int[] counter, List<Integer> list) {
        if (node == null)
            return;
        method2(node.left, k, counter, list);
        counter[0]++;
        if (counter[0] == k) {
            list.add(node.val);
            return;
        }
        method2(node.right, k, counter, list);
    }
}
