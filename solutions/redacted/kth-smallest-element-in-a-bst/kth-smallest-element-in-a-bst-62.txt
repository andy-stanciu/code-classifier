class Solution {

    private int result = -1, k = 0;

    public int method1(TreeNode root, int k) {
        this.k = k;
        method1(root);
        return result;
    }

    private void method1(TreeNode root) {
        if (root == null) {
            return;
        }
        method1(root.left);
        if (--k == 0) {
            result = root.val;
            return;
        }
        method1(root.right);
    }
}
