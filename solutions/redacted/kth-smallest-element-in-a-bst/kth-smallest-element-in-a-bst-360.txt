/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int kCount = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        //Go in order until you hit k
        if (root == null || kCount > k) {
            return ans;
        }
        method1(root.left, k);
        //Process root
        kCount++;
        if (k == kCount) {
            ans = root.val;
            return root.val;
        }
        method1(root.right, k);
        return ans;
    }
}
