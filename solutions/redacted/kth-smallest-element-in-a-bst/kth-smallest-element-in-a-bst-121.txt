/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // To track the number of nodes visited
    private int count = 0;

    // To store the k-th smallest value
    private int result = -1;

    // Function to find the kth smallest element in the BST
    public int method1(TreeNode root, int k) {
        // Perform an in-order traversal to find the kth smallest element
        method2(root, k);
        return result;
    }

    // Helper function to perform in-order traversal
    private void method2(TreeNode node, int k) {
        // If the node is null, return (base case)
        if (node == null) {
            return;
        }
        // Traverse the left subtree
        method2(node.left, k);
        // Increment the count of visited nodes
        count++;
        // If count equals k, we\'ve found the k-th smallest element
        if (count == k) {
            result = node.val;
            // No need to continue traversal
            return;
        }
        // Traverse the right subtree
        method2(node.right, k);
    }
}
