/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int c = 0;

    int ans = -1;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return ans;
        method1(root.left, k);
        c++;
        if (c == k)
            ans = root.val;
        return c < k ? method1(root.right, k) : ans;
    }
}
