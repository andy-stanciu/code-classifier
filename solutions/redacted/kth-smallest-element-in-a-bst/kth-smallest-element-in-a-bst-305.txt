/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int currentK = 0;

    int ans = 0;

    public int method1(TreeNode root, int k) {
        currentK = k;
        method2(root);
        return ans;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        if (currentK > 0) {
            method2(root.left);
            if (currentK > 0)
                ans = root.val;
            currentK--;
            method2(root.right);
        }
    }
}
