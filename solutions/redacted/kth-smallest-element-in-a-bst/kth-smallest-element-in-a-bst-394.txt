/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int val;

    int required_k;

    int num = 0;

    public int method1(TreeNode root, int k) {
        if (root == null)
            return 0;
        required_k = k;
        method2(root);
        return val;
    }

    public void method2(TreeNode root) {
        if (root == null)
            return;
        method2(root.left);
        num++;
        if (num == required_k)
            val = root.val;
        method2(root.right);
    }
}
