class Solution {

    class Class1 {

        int value;

        Class1(int value) {
            this.value = value;
        }
    }

    public int method1(TreeNode root, int k) {
        Class1 toReturn = new Class1(0);
        Class1 iter = new Class1(0);
        method1(root, k, iter, toReturn);
        return toReturn.value;
    }

    public boolean method1(TreeNode root, int k, Class1 curr, Class1 value) {
        if (root == null)
            return false;
        if (method1(root.left, k, curr, value))
            return true;
        curr.value++;
        if (curr.value == k) {
            value.value = root.val;
            return true;
        }
        if (method1(root.right, k, curr, value))
            return true;
        return false;
    }
}
