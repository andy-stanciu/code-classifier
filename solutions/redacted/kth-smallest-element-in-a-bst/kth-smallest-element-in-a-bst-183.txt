/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private int count;

    private int result;

    public void method1(TreeNode root, int k) {
        if (root == null)
            return;
        if (count < k)
            method1(root.left, k);
        ++count;
        if (count == k) {
            result = root.val;
            return;
        }
        if (count < k)
            method1(root.right, k);
    }

    public int method2(TreeNode root, int k) {
        count = 0;
        result = 0;
        method1(root, k);
        return result;
    }
}
