class Solution {

    int count = 1, res = -1;

    public int method1(TreeNode root, int k) {
        // Traverse the left subtree
        if (root.left != null)
            method1(root.left, k);
        // Check if the current node is the kth smallest
        if (count++ == k)
            res = root.val;
        // Traverse the right subtree
        if (root.right != null)
            method1(root.right, k);
        return res;
    }
}
