class Solution {

    int cnt;

    int res;

    public int method1(TreeNode root, int k) {
        // in-order traverse
        cnt = k;
        dfs(root);
        return res;
    }

    void dfs(TreeNode node) {
        if (node == null)
            return;
        dfs(node.left);
        if (--cnt == 0)
            res = node.val;
        dfs(node.right);
    }
}
