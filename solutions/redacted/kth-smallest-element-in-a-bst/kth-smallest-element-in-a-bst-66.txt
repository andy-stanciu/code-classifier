class Solution {

    void help(TreeNode root, List<Integer> list) {
        if (root == null)
            return;
        help(root.left, list);
        list.add(root.val);
        help(root.right, list);
    }

    public int method1(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        help(root, list);
        if (k > list.size())
            return -1;
        return list.get(k - 1);
    }
}
