/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        int[] arr = { k, 0 };
        method2(root, arr);
        return arr[1];
    }

    void method2(TreeNode node, int[] arr) {
        if (node == null)
            return;
        method2(node.left, arr);
        arr[0] -= 1;
        if (arr[0] == 0) {
            arr[1] = node.val;
            return;
        }
        method2(node.right, arr);
    }
}
