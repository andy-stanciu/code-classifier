/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static int var1;

    public int method1(TreeNode root, int k) {
        var1 = 0;
        return dfs(root, k);
    }

    public int dfs(TreeNode root, int k) {
        if (root == null)
            return -1;
        int leftVal = dfs(root.left, k);
        if (k == ++var1)
            return root.val;
        int rightVal = dfs(root.right, k);
        if (leftVal != -1)
            return leftVal;
        return rightVal;
    }
}
