class Solution {

    public int method1(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        method2(list, root);
        return list.get(k - 1);
    }

    private void method2(List<Integer> list, TreeNode root) {
        if (root == null)
            return;
        method2(list, root.left);
        list.add(root.val);
        method2(list, root.right);
    }
}
// TC: O(n), SC: O(n)
