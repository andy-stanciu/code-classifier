/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    private void method1(TreeNode root, ArrayList<Integer> arr) {
        if (root == null) {
            return;
        }
        method1(root.left, arr);
        arr.add(root.val);
        method1(root.right, arr);
    }

    public int method2(TreeNode root, int k) {
        if (root == null)
            return -1;
        ArrayList<Integer> arr = new ArrayList<>();
        method1(root, arr);
        return arr.get(k - 1);
    }
}
