/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        return method2(root, new int[1], k);
    }

    public int method2(TreeNode root, int[] count, int k) {
        if (null == root) {
            return -1;
        }
        int leftValue = method2(root.left, count, k);
        if (leftValue != -1) {
            return leftValue;
        }
        count[0]++;
        if (count[0] == k) {
            return root.val;
        }
        return method2(root.right, count, k);
    }
}
