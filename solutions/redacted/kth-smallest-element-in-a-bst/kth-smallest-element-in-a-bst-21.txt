class Solution {

    public int method1(TreeNode root, int k) {
        Set<Integer> set = new TreeSet<>();
        method2(root, set);
        int count = 1;
        for (int n : set) {
            if (count++ == k)
                return n;
        }
        return k;
    }

    private void method2(TreeNode node, Set<Integer> set) {
        if (node == null)
            return;
        set.add(node.val);
        method2(node.left, set);
        method2(node.right, set);
    }
}
