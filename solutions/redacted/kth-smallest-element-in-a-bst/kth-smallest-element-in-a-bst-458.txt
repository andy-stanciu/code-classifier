/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> arr = new ArrayList<>();
        method2(root, arr);
        return arr.get(k - 1);
    }

    public void method2(TreeNode node, ArrayList<Integer> arr) {
        if (node == null)
            return;
        method2(node.left, arr);
        arr.add(node.val);
        method2(node.right, arr);
    }
}
