/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    // non- static varable
    int ans = 0;

    int count = 0;

    public int method1(TreeNode root, int k) {
        // non static method
        method2(root, k);
        return ans;
    }

    public void method2(TreeNode root, int k) {
        //non static method
        if (root == null) {
            return;
        }
        method2(root.left, k);
        //Node as LNR
        count++;
        if (count == k) {
            ans = root.val;
            return;
        }
        method2(root.right, k);
    }
}
