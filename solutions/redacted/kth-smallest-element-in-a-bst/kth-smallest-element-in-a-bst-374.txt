class Solution {

    public int method1(TreeNode root, int k) {
        TreeNode curr = root;
        int n = 0;
        while (curr != null) {
            if (curr.left != null) {
                TreeNode pred = curr.left;
                while (pred.right != null && pred.right != curr) pred = pred.right;
                if (pred.right == null) {
                    pred.right = curr;
                    curr = curr.left;
                } else {
                    pred.right = null;
                    n++;
                    if (n == k)
                        return curr.val;
                    curr = curr.right;
                }
            } else {
                n++;
                if (n == k)
                    return curr.val;
                curr = curr.right;
            }
        }
        return 0;
    }
}
