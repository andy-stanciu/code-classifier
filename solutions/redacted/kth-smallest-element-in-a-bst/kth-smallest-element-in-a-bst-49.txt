class Solution {

    int count = 0;

    //for comparison
    int res = Integer.MIN_VALUE;

    public int method1(TreeNode root, int k) {
        traverse(root, k);
        return res;
    }

    public void traverse(TreeNode root, int k) {
        //base condition
        if (root == null)
            return;
        traverse(root.left, k);
        count++;
        if (count == k)
            res = root.val;
        traverse(root.right, k);
    }
}
