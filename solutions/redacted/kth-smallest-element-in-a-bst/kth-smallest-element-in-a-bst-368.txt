/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        ArrayList<Integer> list = method2(root, new ArrayList<>());
        return list.get(k - 1);
    }

    ArrayList<Integer> method2(TreeNode node, ArrayList<Integer> list) {
        if (node == null) {
            return list;
        }
        method2(node.left, list);
        list.add(node.val);
        method2(node.right, list);
        return list;
    }
}
