/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public int method1(TreeNode root, int k) {
        Queue<Integer> q = new LinkedList<>();
        call(root, k, q);
        int i = 0;
        int ans = 0;
        while (i < k) {
            ans = q.poll();
            i++;
        }
        return ans;
    }

    public void call(TreeNode root, int k, Queue<Integer> q) {
        if (root == null || q.size() == k) {
            return;
        }
        call(root.left, k, q);
        q.add(root.val);
        call(root.right, k, q);
    }
}
