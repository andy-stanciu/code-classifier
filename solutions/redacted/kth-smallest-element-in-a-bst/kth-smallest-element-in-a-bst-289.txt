/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int var1 = 0;

    int result = -1;

    public int method1(TreeNode root, int k) {
        solve(root, k);
        return result;
    }

    void solve(TreeNode node, int k) {
        if (node == null)
            return;
        solve(node.left, k);
        var1++;
        if (var1 == k) {
            result = node.val;
            return;
        } else
            solve(node.right, k);
    }
}
