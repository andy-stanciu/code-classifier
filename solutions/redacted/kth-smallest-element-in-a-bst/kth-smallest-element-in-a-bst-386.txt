/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    ArrayList<Integer> arr = new ArrayList<>();

    private void method1(TreeNode root) {
        if (root == null)
            return;
        method1(root.left);
        arr.add(root.val);
        method1(root.right);
    }

    public int method2(TreeNode root, int k) {
        method1(root);
        return arr.get(k - 1);
    }
}
