/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public static int c = 0;

    public int method1(TreeNode root, int k) {
        if (root == null) {
            return 0;
        }
        int n = method1(root.left, k);
        c++;
        if (c == k)
            return root.val;
        int m = method1(root.right, k);
        if (m != 0) {
            return m;
        }
        return n;
    }

    public int method2(TreeNode root, int k) {
        int z = method1(root, k);
        c = 0;
        return z;
    }
}
