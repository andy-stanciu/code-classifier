class Solution {

    int i = 0;

    private int helper(TreeNode root, int k) {
        if (root == null)
            return -1;
        int x = helper(root.left, k);
        if (x != -1)
            return x;
        if (++i == k)
            return root.val;
        return helper(root.right, k);
    }

    public int method1(TreeNode root, int k) {
        i = 0;
        return helper(root, k);
    }
}
