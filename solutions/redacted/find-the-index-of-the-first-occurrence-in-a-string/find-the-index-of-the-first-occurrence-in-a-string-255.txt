class Solution {

    public int strStr(String h, String n) {
        for (int i = 0; i < h.length() - n.length() + 1; i++) {
            // System.out.println(h.substring(i,i+n.length()));
            if (n.equals(h.substring(i, i + n.length())))
                return i;
        }
        return -1;
    }
}
