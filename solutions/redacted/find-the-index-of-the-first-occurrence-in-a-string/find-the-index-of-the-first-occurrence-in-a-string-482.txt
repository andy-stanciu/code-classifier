class Solution {

    public int strStr(String haystack, String needle) {
        if (Objects.equals(haystack, needle))
            return 0;
        int len = needle.length();
        for (int i = 0; i < haystack.length() - len + 1; i++) {
            String s = haystack.substring(i, i + len);
            if (Objects.equals(s, needle))
                return i;
        }
        return -1;
    }
}
