class Solution {

    public int strStr(String s, String pattern) {
        for (int i = 0; i < s.length() - pattern.length() + 1; i++) {
            int t = i;
            for (int j = 0; j < pattern.length(); j++, t++) {
                if (s.charAt(t) != pattern.charAt(j))
                    break;
            }
            if (t - i == pattern.length())
                return i;
        }
        return -1;
    }
}
