class Solution {

    public int strStr(String haystack, String needle) {
        int n = needle.length();
        int k = 0;
        int j = haystack.length();
        if (n > j) {
            return -1;
        }
        for (int i = 0; i < haystack.length(); i++) {
            if (haystack.charAt(i) == needle.charAt(k)) {
                if (i + n <= j && haystack.substring(i, i + n).equals(needle))
                    return i;
            }
        }
        return -1;
    }
}
