class Solution {

    public static int strStr(String haystack, String needle) {
        int occurence = -1;
        for (int i = 0, j = 0; i < haystack.length() && j < needle.length(); ) {
            if (haystack.charAt(i) == needle.charAt(j)) {
                i++;
                j++;
            } else {
                i = i - j + 1;
                j = 0;
            }
            if (j == needle.length()) {
                occurence = i - j;
                break;
            }
        }
        return occurence;
    }
}
