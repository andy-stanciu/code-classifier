class Solution {

    public int strStr(String haystack, String needle) {
        int res = 0;
        String temp = "";
        int nh = haystack.length();
        if (!haystack.contains(needle)) {
            res = -1;
        }
        for (int i = 0; i < nh; i++) {
            temp += haystack.charAt(i);
            if (temp.contains(needle)) {
                res = i - (needle.length() - 1);
                break;
            }
        }
        return res;
    }
}
