class Solution {

    public int strStr(String haystack, String needle) {
        int m = haystack.length();
        int n = needle.length();
        for (int start = 0; start <= m - n; start++) {
            //need ot match 1 st charachter of needle
            if (haystack.substring(start, needle.length() + start).equals(needle)) {
                return start;
            }
        }
        return -1;
    }
}
