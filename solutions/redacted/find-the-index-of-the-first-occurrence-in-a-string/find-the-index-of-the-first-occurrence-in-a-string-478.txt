class Solution {

    public int strStr(String haystack, String needle) {
        int var1 = needle.length();
        int var2 = haystack.length();
        for (int i = 0; i <= var2 - var1; i++) {
            if (haystack.substring(i, i + var1).equals(needle)) {
                return i;
            }
        }
        return -1;
    }
}
