class Solution {

    public int strStr(String s1, String s2) {
        int m = s1.length(), n = s2.length();
        if (m < n)
            return -1;
        for (int i = 0; i < m; i++) {
            if (s1.charAt(i) == s2.charAt(0)) {
                if (i + n > m)
                    return -1;
                int temp = i + 1;
                boolean flag = true;
                for (int l = 1; l < n; l++) {
                    if (temp == m)
                        return -1;
                    else if (s2.charAt(l) != s1.charAt(temp)) {
                        flag = false;
                        break;
                    } else
                        temp++;
                }
                if (flag)
                    return i;
            }
        }
        return -1;
    }
}
