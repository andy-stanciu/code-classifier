class Solution {

    public int strStr(String haystack, String needle) {
        if (needle.isEmpty()) {
            // If needle is an empty string, return 0
            return 0;
        }
        int n = haystack.length();
        int m = needle.length();
        for (int i = 0; i <= n - m; i++) {
            // Check if the substring from i to i + m equals needle
            if (haystack.substring(i, i + m).equals(needle)) {
                return i;
            }
        }
        // Needle not found
        return -1;
    }
}
