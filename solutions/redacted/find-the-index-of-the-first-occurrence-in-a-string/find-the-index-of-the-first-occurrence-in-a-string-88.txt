class Solution {

    public int strStr(String haystack, String needle) {
        int n = haystack.length();
        int m = needle.length();
        int x = n - m;
        for (int i = 0; i <= x; i++) {
            if (needle.equals(haystack.substring(i, i + m))) {
                return i;
            }
        }
        return -1;
    }
}
