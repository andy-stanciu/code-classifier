class Solution {

    public int strStr(String haystack, String needle) {
        if (haystack.length() < needle.length()) {
            return -1;
        }
        int k = 0;
        for (int i = 0; i <= ((haystack.length()) - (needle.length())); i++) {
            k = 0;
            while (k < needle.length()) {
                if (needle.charAt(k) == haystack.charAt(k + i)) {
                    k += 1;
                } else {
                    break;
                }
            }
            if (k == needle.length()) {
                return i;
            }
        }
        return -1;
    }
}
