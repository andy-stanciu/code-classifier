class Solution {

    public int strStr(String haystack, String needle) {
        int haystackLength = haystack.length();
        int needleLength = needle.length();
        int i = 0, j = 0, k = 0;
        while (i < haystackLength && j < needleLength) {
            if (haystack.charAt(i) == needle.charAt(j)) {
                i++;
                j++;
            } else {
                i = k + 1;
                j = 0;
                k = i;
            }
        }
        if (j == needleLength)
            return k;
        return -1;
    }
}
