class Solution {

    public int strStr(String haystack, String needle) {
        // haystack pointer
        int hp = 0;
        // needle pointer
        int np = 0;
        int hLen = haystack.length();
        int nLen = needle.length();
        int idx = 0;
        while (hp < hLen) {
            // track the current haystack pointer
            idx = hp;
            // check if the needle is present in haystack
            while (np < nLen && hp < hLen) {
                if (haystack.charAt(hp) == needle.charAt(np)) {
                    np++;
                    hp++;
                } else
                    break;
            }
            // if needle is present in haystack
            if (np == nLen) {
                return idx;
            } else {
                // reset the pointers
                np = 0;
                hp = idx;
            }
            hp++;
        }
        return -1;
    }
}
