class Solution {

    public int strStr(String haystack, String needle) {
        if (haystack.length() < needle.length())
            return -1;
        int j = 0;
        for (int i = needle.length(); i <= haystack.length(); i++) {
            if (haystack.substring(j, i).equals(needle))
                return j;
            ++j;
        }
        return -1;
    }
}
