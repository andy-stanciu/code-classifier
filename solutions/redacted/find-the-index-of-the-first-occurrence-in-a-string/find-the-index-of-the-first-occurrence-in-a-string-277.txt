class Solution {

    public int strStr(String haystack, String needle) {
        if (needle.isEmpty()) {
            return 0;
        }
        int count = 0;
        int j = 0;
        for (int i = 0; i < haystack.length(); i++) {
            if (haystack.charAt(i) == needle.charAt(j)) {
                count++;
                j++;
            } else {
                i = i - count;
                count = 0;
                j = 0;
            }
            if (count == needle.length()) {
                return i - needle.length() + 1;
            }
        }
        return -1;
    }
}
