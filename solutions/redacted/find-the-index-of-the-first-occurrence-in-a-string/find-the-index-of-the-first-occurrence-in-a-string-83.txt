class Solution {

    public int strStr(String str, String s) {
        if (str.length() < s.length())
            return -1;
        String x = null;
        for (int i = s.length(); i <= str.length(); i++) {
            x = str.substring(i - s.length(), i);
            if (x.equals(s))
                return i - s.length();
        }
        return -1;
    }
}
