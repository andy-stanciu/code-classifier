class Solution {

    public int strStr(String haystack, String needle) {
        int var1 = 0;
        int needleLength = needle.length();
        int var2 = 0;
        int haystackLength = haystack.length();
        while (var2 < haystackLength) {
            //haystack.charAt(index) == needle.charAt(needleIndex)
            if (haystack.charAt(var2) == needle.charAt(var1)) {
                var1++;
            } else {
                var2 = var2 - var1;
                var1 = 0;
            }
            if (var1 >= needleLength) {
                return var2 - (needleLength - 1);
            }
            var2++;
        }
        return -1;
    }
}
