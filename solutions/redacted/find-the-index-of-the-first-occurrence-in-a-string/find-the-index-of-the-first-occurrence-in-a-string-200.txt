class Solution {

    public int strStr(String haystack, String needle) {
        int l1 = haystack.length();
        int l2 = needle.length();
        if (l1 < l2)
            return -1;
        int i = 0;
        while (i <= l1 - l2) {
            String curr = haystack.substring(i, i + l2);
            if (curr.equals(needle))
                return i;
            i++;
        }
        return -1;
    }
}
