class Solution {

    public int strStr(String haystack, String needle) {
        int left = 0;
        int right = 0;
        int n = haystack.length();
        int needleLength = needle.length();
        String hayStackNeedle = "";
        while (right < n) {
            char c = haystack.charAt(right);
            hayStackNeedle += c;
            if ((right - left) + 1 == needleLength) {
                if (needle.equals(hayStackNeedle)) {
                    return left;
                } else {
                    hayStackNeedle = hayStackNeedle.substring(1, hayStackNeedle.length());
                    left++;
                }
            }
            right++;
        }
        return -1;
    }
}
