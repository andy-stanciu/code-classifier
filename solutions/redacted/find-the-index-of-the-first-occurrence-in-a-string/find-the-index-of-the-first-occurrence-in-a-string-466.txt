class Solution {

    // Function to find the first occurrence of the substring \'needle\' in \'haystack\'
    public int strStr(String haystack, String needle) {
        // Loop through the haystack with a window of size equal to the length of needle
        for (int i = 0; i <= haystack.length() - needle.length(); i++) {
            int j = 0;
            // For each position in the haystack, check if needle matches
            for (j = 0; j < needle.length(); j++) {
                // If a mismatch is found, break out of the inner loop
                if (haystack.charAt(i + j) != needle.charAt(j)) {
                    break;
                }
            }
            // If the entire needle is matched (j == needle.length()), return the starting index
            if (j == needle.length()) {
                return i;
            }
        }
        // If no match is found, return -1
        return -1;
    }
}
