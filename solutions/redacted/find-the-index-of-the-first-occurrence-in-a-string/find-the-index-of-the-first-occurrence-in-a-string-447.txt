class Solution {

    public int strStr(String haystack, String needle) {
        return haystack.indexOf(needle);
        // In Java, the indexOf function is used to find the index of the first
        // occurrence of a specified substring or character in a string. If the substring
        // or character is found, it returns the index (starting from 0). If it\'s not found, it returns -1.
    }
}
