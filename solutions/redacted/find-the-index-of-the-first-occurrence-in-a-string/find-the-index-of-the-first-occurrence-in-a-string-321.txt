class Solution {

    public int strStr(String haystack, String needle) {
        int var1 = 0, var2 = 0;
        int lastMet = -1;
        while (var1 < needle.length() && var2 < haystack.length()) {
            if (needle.charAt(var1) == haystack.charAt(var2)) {
                var1++;
                var2++;
            } else if (var1 == 0)
                var2++;
            else {
                var2 = var2 - var1 + 1;
                var1 = 0;
            }
        }
        if (var1 == needle.length())
            lastMet = var2 - var1;
        return lastMet;
    }
}
