class Solution {

    public int strStr(String haystack, String needle) {
        int hlen = haystack.length();
        int nlen = needle.length();
        if (hlen - nlen < 0) {
            return -1;
        }
        for (int i = 0; i <= (hlen - nlen); i++) {
            for (int j = 0; j < nlen; j++) {
                if (haystack.charAt(i + j) != needle.charAt(j)) {
                    break;
                }
                if (j == (nlen - 1)) {
                    return i;
                }
            }
        }
        return -1;
    }
}
