class Solution {

    public int strStr(String haystack, String needle) {
        int i = 0, j = 0, k = needle.length();
        int var1 = -1;
        while (j < haystack.length()) {
            //  Window size is not reached just increase j.
            if (j - i + 1 < k)
                j++;
            else // Window size reached
            if (j - i + 1 == k) {
                // Check if Strings match
                if (haystack.substring(i, j + 1).equals(needle)) {
                    // If they match break since we need smallest index.
                    var1 = i;
                    // If they match break since we need smallest index.
                    break;
                }
                // Otherwise just move the window.
                i++;
                j++;
            }
        }
        return var1;
    }
}
