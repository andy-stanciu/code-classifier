class Solution {

    public boolean method1(int x) {
        int rev = 0, c = x;
        if (x >= 0) {
            while (c != 0) {
                rev *= 10;
                rev += c % 10;
                c /= 10;
            }
            if (rev == Math.abs(x)) {
                return true;
            } else {
                return false;
            }
        } else {
            return false;
        }
    }
}
