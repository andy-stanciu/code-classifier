class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int dup = x;
        int rev = 0;
        while (x > 0) {
            int dig = x % 10;
            rev = rev * 10 + dig;
            x = x / 10;
        }
        if (dup == rev) {
            return true;
        }
        return false;
    }
}
