class Solution {

    static boolean method1(int x) {
        int rev = 0;
        int x1 = x;
        while (x > 0) {
            rev = rev * 10 + (x % 10);
            x /= 10;
        }
        return rev == x1;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        System.out.println(method1(x));
    }
}
