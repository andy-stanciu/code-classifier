class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int z = x;
        int y = 0;
        while (z > 0) {
            y = y * 10 + z % 10;
            z = z / 10;
        }
        if (x == y)
            return (true);
        return (false);
    }
}
