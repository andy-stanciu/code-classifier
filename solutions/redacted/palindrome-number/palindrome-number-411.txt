class Solution {

    public boolean method1(int x) {
        boolean method1;
        int var1 = x;
        int reversed = 0;
        if (x < 0) {
            method1 = false;
        } else {
            while (x != 0) {
                reversed = reversed * 10 + x % 10;
                x /= 10;
            }
            method1 = reversed == var1;
        }
        return method1;
    }
}
