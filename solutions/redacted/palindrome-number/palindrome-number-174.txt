class Solution {

    public boolean method1(int x) {
        int originalnum = x;
        int reversedNum = 0;
        while (x != 0 && x > 0) {
            int remainder = x % 10;
            reversedNum = reversedNum * 10 + remainder;
            x = x / 10;
        }
        return originalnum == reversedNum;
    }
}
