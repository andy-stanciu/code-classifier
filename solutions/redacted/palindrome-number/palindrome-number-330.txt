class Solution {

    public boolean method1(int x) {
        // Negative numbers and numbers ending with 0 (except for 0 itself) are not palindromes
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int reversed = 0;
        while (x > reversed) {
            // Append the last digit of x to reversed
            reversed = reversed * 10 + x % 10;
            // Remove the last digit from x
            x /= 10;
        }
        // x is equal to reversed for even length numbers,
        // or x is equal to reversed/10 for odd length numbers
        return x == reversed || x == reversed / 10;
    }
}
