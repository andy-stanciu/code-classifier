class Solution {

    public boolean method1(int x) {
        // later, x will be changes in smaller value so we stored "x" in another variable
        int k = x;
        int num = 0;
        if (x < 0) {
            //for negative number
            return false;
        }
        while (x > 0) {
            //here x will be updated every time
            int rem = x % 10;
            num = rem + (num * 10);
            x = x / 10;
        }
        if (num == k) {
            return true;
        }
        return false;
    }
}
