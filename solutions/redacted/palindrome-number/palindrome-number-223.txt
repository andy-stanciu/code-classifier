class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        } else {
            int rev = 0, v = x, rem = 0;
            while (v > 0) {
                rem = v % 10;
                rev = rev * 10 + rem;
                v = v / 10;
            }
            return x == rev;
        }
    }
}
