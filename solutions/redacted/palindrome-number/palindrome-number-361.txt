class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        } else {
            int t = x;
            int r = 0, re = 0;
            while (t != 0) {
                r = t % 10;
                re = re * 10 + r;
                t = t / 10;
            }
            if (re == x) {
                return true;
            }
        }
        return false;
    }
}
