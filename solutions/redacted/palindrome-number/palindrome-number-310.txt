class Solution {

    public boolean method1(int x) {
        String var1 = String.valueOf(x);
        for (int i = 0; i < var1.length(); i++) {
            char c = var1.charAt(i);
            char d = var1.charAt(var1.length() - 1 - i);
            if (c != d) {
                return false;
            }
        }
        return true;
    }
}
