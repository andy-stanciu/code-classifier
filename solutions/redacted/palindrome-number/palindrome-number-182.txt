class Solution {

    public boolean method1(int x) {
        int ld;
        int n = x;
        int rev = 0;
        while (x != 0) {
            if (x < 0) {
                return false;
            }
            ld = x % 10;
            rev = rev * 10 + ld;
            x = x / 10;
        }
        if (n == rev) {
            return true;
        } else {
            return false;
        }
    }
}
