class Solution {

    public boolean method1(int x) {
        int reverseNum = 0;
        int remainder;
        int originalNum = x;
        if (originalNum < 0) {
            return false;
        }
        while (x != 0) {
            remainder = x % 10;
            reverseNum = reverseNum * 10 + remainder;
            x /= 10;
        }
        if (originalNum == reverseNum) {
            return true;
        } else {
            return false;
        }
    }
}
