class Solution {

    public boolean method1(int x) {
        //% ->to get last digit
        // / ->to remove last digit
        if (x < 0 || (x != 0 && x % 10 == 0))
            return false;
        //number ending with zero will not be palindrome.
        int temp = 0;
        while (x > temp) {
            temp = temp * 10 + x % 10;
            x = x / 10;
        }
        return (x == temp || x == temp / 10);
    }
}
