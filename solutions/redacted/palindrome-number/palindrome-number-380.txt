class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int num = x;
        int revNum = 0;
        while (num != 0) {
            int rem = num % 10;
            revNum = revNum * 10 + rem;
            num /= 10;
        }
        if (revNum == x) {
            return true;
        } else
            return false;
    }
}
