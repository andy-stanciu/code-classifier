class Solution {

    public boolean method1(int x) {
        int var1 = x;
        int rev = 0;
        while (var1 != 0) {
            int lastdigit = var1 % 10;
            rev = (rev * 10) + lastdigit;
            var1 = var1 / 10;
        }
        if (x > 0 && x == rev || x == 0) {
            return true;
        }
        return false;
    }
}
