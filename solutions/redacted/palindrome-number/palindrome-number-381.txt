class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int originalNum = x;
        int reverseNum = 0;
        while (x != 0) {
            //get last digit
            int lastdigit = x % 10;
            //append the lastdigit to reverseNum
            reverseNum = reverseNum * 10 + lastdigit;
            //remove the last digit from the x
            x /= 10;
        }
        return originalNum == reverseNum;
    }
}
