class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x != 0 && x % 10 == 0))
            return false;
        int num = x, rev_num = 0;
        while (num > 0) {
            rev_num = rev_num * 10 + num % 10;
            num = num / 10;
        }
        return rev_num == x;
    }
}
