class Solution {

    public boolean method1(int x) {
        List num = new ArrayList();
        if (x < 0)
            return false;
        while (x >= 1) {
            num.add(x % 10);
            x = x / 10;
        }
        int i = 0, j = num.size() - 1;
        while (i < j) {
            if (num.get(i++) != num.get(j--))
                return false;
        }
        return true;
    }
}
