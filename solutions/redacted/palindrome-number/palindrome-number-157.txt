class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int reverse = method2(x);
        return reverse == x;
    }

    private static int method2(int num) {
        int reverse = 0;
        while (num >= 1) {
            int remainder = num % 10;
            reverse = reverse * 10 + remainder;
            num = num / 10;
        }
        return reverse;
    }
}
