class Solution {

    public boolean method1(Integer x) {
        String digit1 = "";
        String digit2 = "";
        if (x < 0) {
            return false;
        }
        while (x > 0) {
            digit1 += x % 10;
            digit2 = x % 10 + digit2;
            x /= 10;
        }
        if (digit1.equals(digit2)) {
            return true;
        }
        return false;
    }
}
