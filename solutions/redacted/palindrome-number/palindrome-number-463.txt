class Solution {

    public boolean method1(int x) {
        String var1 = Integer.toString(x);
        int start = 0;
        int end = var1.length() - 1;
        while (start < end) {
            if (var1.charAt(start) != var1.charAt(end)) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
