class Solution {

    public boolean method1(int x) {
        int d = x;
        int reverse = 0;
        if (d < 0) {
            return false;
        }
        while (x != 0) {
            int lastDigit = x % 10;
            reverse = reverse * 10 + lastDigit;
            x = x / 10;
        }
        return reverse == d ? true : false;
    }
}
