class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int n = x;
        int rev = 0;
        while (n != 0) {
            int rem = (int) (n % 10);
            rev = rev * 10 + rem;
            n /= 10;
        }
        return x == rev;
    }
}
