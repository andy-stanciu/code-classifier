class Solution {

    public boolean method1(int x) {
        int r, sum = 0, temp;
        temp = x;
        for (; x > 0; x = x / 10) {
            r = x % 10;
            sum = sum * 10 + r;
        }
        if (temp == sum) {
            return true;
        } else {
            return false;
        }
    }
}
