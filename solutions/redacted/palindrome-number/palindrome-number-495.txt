public class Solution {

    public boolean method1(long x) {
        if (x < 0) {
            // Negative numbers cannot be palindromes
            return false;
        }
        long rem = 0;
        long x1 = x;
        long newNum = 0;
        while (x1 > 0) {
            // Get the last digit
            rem = x1 % 10;
            // Remove the last digit from x1
            x1 = x1 / 10;
            // Build the reversed number
            newNum = newNum * 10 + rem;
        }
        // Check if the reversed number is the same as the original
        return newNum == x;
    }
}
