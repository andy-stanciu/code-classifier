class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        // where temp is temporary variable in which value of x is stored and rev is reversed x.
        int temp = x, rev = 0;
        while (temp != 0) {
            rev = (rev * 10) + temp % 10;
            temp = temp / 10;
        }
        return rev == x;
    }
}
