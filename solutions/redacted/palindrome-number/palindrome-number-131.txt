class Solution {

    public boolean method1(int n) {
        {
            int rev = 0;
            int temp = n;
            if (// checking for neagtive no
            n < 0)
                // checking for neagtive no
                return false;
            else {
                while (temp != 0) {
                    //%10 to get last digit of temp
                    rev = rev * 10 + temp % 10;
                    //deleting the used no
                    temp /= 10;
                }
                if (rev == n)
                    return true;
                else
                    return false;
            }
        }
    }
}
