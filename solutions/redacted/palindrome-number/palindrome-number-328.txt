class Solution {

    public boolean method1(int x) {
        String num = "" + x;
        String xStr = "";
        for (int i = num.length() - 1; i >= 0; i--) {
            xStr += num.substring(i, i + 1);
        }
        System.out.println(num + " : " + xStr);
        if (num.equals(xStr)) {
            return true;
        } else {
            return false;
        }
    }
}
