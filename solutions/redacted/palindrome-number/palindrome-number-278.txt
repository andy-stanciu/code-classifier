class Solution {

    public boolean method1(int x) {
        ArrayList<Integer> arr = new ArrayList<>();
        if (x < 0) {
            return false;
        }
        while (x > 0) {
            int c = x % 10;
            x = x / 10;
            arr.add(c);
        }
        for (int i = 0; i < arr.size(); i++) {
            if (arr.get(i) != arr.get(arr.size() - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
