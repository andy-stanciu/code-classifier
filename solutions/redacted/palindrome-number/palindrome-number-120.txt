class Solution {

    public boolean method1(int x) {
        String str1 = Integer.toString(x);
        String str2 = new StringBuilder().append(x).reverse().toString();
        if (str1.equals(str2))
            return true;
        return false;
        //or in 1 lines
        // return Integer.toString(x).equals(new StringBuilder().append(x).reverse().toString());
    }
}
