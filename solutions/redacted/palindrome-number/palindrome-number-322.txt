class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int sum = 0, tmp = 0;
        int x2 = x;
        while (x != 0) {
            tmp = x % 10;
            x = x / 10;
            if (x == 0)
                sum = sum + tmp;
            else
                sum = tmp * 10 + sum * 10;
        }
        if (sum == x2)
            return true;
        else
            return false;
    }
}
