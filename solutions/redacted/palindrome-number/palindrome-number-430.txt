class Solution {

    public boolean method1(int x) {
        return x - (revNum(x)) == 0 ? true : false;
    }

    public int revNum(int num) {
        int rNum = 0;
        while (num > 0) {
            rNum = rNum * 10 + num % 10;
            num = num / 10;
        }
        return rNum;
    }
}
