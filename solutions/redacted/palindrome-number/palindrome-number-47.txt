class Solution {

    public boolean method1(int x) {
        String num = Integer.toString(x);
        int l = 0, r = num.length() - 1;
        boolean flag = false;
        //here i iterated l and r over the string num while check characters at l index and r index while reducing r after every iteration and increasing l after every iteration.
        while (l <= r) {
            if (num.charAt(l) == num.charAt(r)) {
                flag = true;
            } else {
                //even if it is false once , we return false
                flag = false;
                return false;
            }
            l++;
            r--;
        }
        return flag;
    }
}
