class Solution {

    public boolean method1(int x) {
        // Handle negative numbers and numbers ending in 0
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int var1 = 0;
        while (x > var1) {
            var1 = var1 * 10 + x % 10;
            x /= 10;
        }
        // Check if the number is a palindrome or half-palindrome
        return x == var1 || x == (var1 / 10);
    }
}
