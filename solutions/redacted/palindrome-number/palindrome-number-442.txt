class Solution {

    public boolean method1(int x) {
        int temp = x;
        int r;
        int rev = 0;
        while (x > 0) {
            r = x % 10;
            rev = rev * 10 + r;
            x = x / 10;
        }
        if (temp == rev) {
            return true;
        } else {
            return false;
        }
    }

    public static void main(String[] args) {
        Solution a = new Solution();
        boolean result = a.method1(10);
        System.out.println("" + result);
    }
}
