class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int originalNum = x;
        int reversedNum = 0;
        while (x != 0) {
            int rem = x % 10;
            reversedNum = reversedNum * 10 + rem;
            x = x / 10;
        }
        if (originalNum == reversedNum) {
            return true;
        } else {
            return false;
        }
    }
}
