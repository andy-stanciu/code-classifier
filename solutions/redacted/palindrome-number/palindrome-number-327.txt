class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int reversed = 0;
        int xCopy = x;
        while (xCopy != 0) {
            reversed = (reversed * 10) + xCopy % 10;
            xCopy = xCopy / 10;
        }
        return x == reversed;
    }
}
