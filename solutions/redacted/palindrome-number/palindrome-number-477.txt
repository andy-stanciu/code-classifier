class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int reversedX = 0;
        int num = x;
        while (num > 0) {
            int remainder = num % 10;
            num = num / 10;
            reversedX = (reversedX * 10) + remainder;
        }
        return reversedX == x;
    }
}
