class Solution {

    public boolean method1(int x) {
        // Convert x into String
        String t = String.valueOf(x);
        int n = t.length();
        // Iterate over String t for half length and compare the first & last character
        for (int i = 0; i < n / 2; i++) {
            // if at any point chars are not equal return false
            if (t.charAt(i) != t.charAt(n - i - 1))
                return false;
        }
        // if loop\'s over then return true means it\'s palindrome
        return true;
    }
}
