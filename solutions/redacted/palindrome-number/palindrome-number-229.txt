class Solution {

    public boolean method1(int x) {
        int rev = 0;
        int n = x;
        while (x != 0) {
            rev = ((rev * 10) + (x % 10));
            x /= 10;
        }
        ;
        if (rev < 0)
            return false;
        return (n == 0 || n == rev) ? true : false;
    }
}
