class Solution {

    public boolean method1(int x) {
        String str_x = String.valueOf(x);
        int j = str_x.length() - 1;
        for (int i = 0; i < Math.floor(str_x.length() / 2); i++) {
            if (!(str_x.charAt(i) == str_x.charAt(j - i))) {
                return false;
            }
            if (j <= i) {
                break;
            }
        }
        return true;
    }
}
