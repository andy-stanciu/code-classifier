class Solution {

    public boolean method1(int x) {
        int rev = 0;
        int temp = x;
        while (x > 0) {
            int rem = x % 10;
            rev = rev * 10 + rem;
            x = x / 10;
        }
        if (rev == temp) {
            return true;
        } else {
            return false;
        }
    }
}
