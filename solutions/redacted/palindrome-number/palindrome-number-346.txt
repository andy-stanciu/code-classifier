class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int i = 0;
        int o = x;
        int reverse = 0;
        while (x > 0) {
            i = x % 10;
            reverse = reverse * 10 + i;
            x /= 10;
        }
        if (reverse == o)
            return true;
        else
            return false;
    }
}
