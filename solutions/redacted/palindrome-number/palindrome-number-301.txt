class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int rev = 0;
        while (x > rev) {
            int pop = x % 10;
            rev = rev * 10 + pop;
            x = x / 10;
        }
        return x == rev || x == rev / 10;
    }
}
