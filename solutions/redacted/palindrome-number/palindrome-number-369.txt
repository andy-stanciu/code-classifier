class Solution {

    public boolean method1(int x) {
        int temp = x;
        int sum = 0;
        int rem = 0;
        while (x > 0) {
            rem = x % 10;
            sum = (sum * 10) + rem;
            x = x / 10;
        }
        if (sum == temp) {
            return true;
        } else {
            return false;
        }
    }
}
