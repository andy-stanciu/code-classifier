class Solution {

    public boolean method1(int x) {
        if (x < 0)
            return false;
        int rev = reverse(x);
        int val = x;
        while (val > 0) {
            int t1 = val % 10;
            int tr1 = rev % 10;
            if (t1 != tr1)
                return false;
            rev = rev / 10;
            val = val / 10;
        }
        return true;
    }

    int reverse(int x) {
        int res = 0;
        while (x > 0) {
            int temp = x % 10;
            res = (res * 10) + temp;
            x = x / 10;
        }
        return res;
    }
}
