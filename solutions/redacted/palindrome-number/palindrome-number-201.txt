class Solution {

    public boolean method1(int x) {
        // Negative numbers are not palindromes
        // Numbers ending with 0 are not palindromes (except for 0 itself)
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int var1 = 0;
        while (x > var1) {
            int rem = x % 10;
            var1 = var1 * 10 + rem;
            x /= 10;
        }
        // The number is a palindrome if it is equal to the reversed number,
        // or if the reversed number without the last digit is equal to x (for odd length numbers)
        return x == var1 || x == var1 / 10;
    }
}
