class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        long reversed = 0;
        long temp = x;
        while (temp != 0) {
            // reversing a number
            int digit = (int) (temp % 10);
            reversed = reversed * 10 + digit;
            temp /= 10;
        }
        // comparing the input and the reversed input
        return (reversed == x);
    }
}
