class Solution {

    public boolean method1(int x) {
        String n = Integer.toString(x);
        int i = n.length() - 1;
        int j = 0;
        while (i >= ((n.length() - 1) / 2)) {
            if (n.charAt(i) != n.charAt(j)) {
                return false;
            }
            i--;
            j++;
        }
        return true;
    }
}
