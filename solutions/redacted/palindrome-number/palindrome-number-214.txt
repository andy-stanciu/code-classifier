class Solution {

    public boolean method1(int x) {
        int y = 0;
        int val = x;
        if (x < 0) {
            return false;
        } else {
            while (val > 0) {
                y = y * 10 + val % 10;
                val = val / 10;
            }
        }
        if (y == x) {
            return true;
        }
        return false;
    }
}
