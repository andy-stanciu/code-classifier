class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        } else {
            String s = "" + x;
            StringBuilder s2 = new StringBuilder(s).reverse();
            String s1 = s2.toString();
            if (s.equals(s1)) {
                return true;
            }
        }
        return false;
    }
}
