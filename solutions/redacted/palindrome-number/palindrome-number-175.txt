class Solution {

    public boolean method1(int x) {
        int temp = x;
        int sum = 0;
        while (x > 0) {
            int t = x % 10;
            sum = (sum * 10) + t;
            x = x / 10;
        }
        if (temp == sum) {
            return true;
        } else {
            return false;
        }
    }
}
