class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int reversenum = 0;
        int originalnum = x;
        while (x != 0) {
            int reminder = x % 10;
            reversenum = reversenum * 10 + reminder;
            x /= 10;
        }
        return originalnum == reversenum;
    }
}
