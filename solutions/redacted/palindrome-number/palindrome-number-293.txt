class Solution {

    public boolean method1(int x) {
        // Is it an integer?
        if (x < Integer.MIN_VALUE || x > Integer.MAX_VALUE)
            return false;
        //Is it positive? Negtive numbers can\'t be a palindrome
        if (x < 0)
            return false;
        // Any positive number under 10 is palindromic
        if (x < 10)
            return true;
        String var1 = String.valueOf(x);
        char[] var2 = var1.toCharArray();
        int lastIndexPointer = var2.length - 1;
        for (int i = 0; (i < var2.length / 2); i++) {
            if (var2[i] != var2[lastIndexPointer])
                return false;
            lastIndexPointer--;
        }
        return true;
    }
}
