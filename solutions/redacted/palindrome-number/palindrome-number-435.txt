class Solution {

    public boolean method1(int x) {
        int res = 0;
        int temp;
        int org = x;
        while (x > 0) {
            res = res * 10;
            temp = x % 10;
            res = res + temp;
            x = x / 10;
        }
        if (org == res)
            return true;
        return false;
    }
}
