class Solution {

    public boolean method1(int x) {
        int num = 0;
        int temp = x;
        int digit = 0;
        if (x < 0)
            return false;
        while (x != 0) {
            digit = x % 10;
            num = num * 10 + digit;
            x = x / 10;
        }
        if (num == temp)
            return true;
        else
            return false;
    }
}
