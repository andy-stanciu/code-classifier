class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        //use to compare
        int copy = x;
        int ans = 0;
        while (x > 0) {
            int rem = x % 10;
            ans = ans * 10 + rem;
            x /= 10;
        }
        if (ans == copy) {
            return true;
        } else {
            return false;
        }
    }
}
