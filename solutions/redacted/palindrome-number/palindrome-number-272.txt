class Solution {

    public static boolean method1(int x) {
        //negative numbers cannot be palindrome
        if (x < 0) {
            return false;
        }
        //convert number to string
        String str = Integer.toString(x);
        //use two pointers: one starts from beginning and other starts from ending
        int left = 0;
        int right = str.length() - 1;
        //check if the digits at two pointers are equal
        while (left < right) {
            if (str.charAt(left) != str.charAt(right)) {
                // if not equal its false;
                return false;
            }
            left++;
            right--;
        }
        // if all digits matched its a palindrome
        return true;
    }
}
