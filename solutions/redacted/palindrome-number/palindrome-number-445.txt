class Solution {

    public boolean method1(int x) {
        int OriginalInt = x;
        int ReverseInt = 0;
        int LastDigit = 0;
        if (x < 0) {
            return false;
        }
        while (x != 0) {
            LastDigit = x % 10;
            ReverseInt = ReverseInt * 10 + LastDigit;
            x = x / 10;
        }
        return ReverseInt == OriginalInt ? true : false;
    }
}
