class Solution {

    public boolean method1(int x) {
        // Negative numbers and numbers ending with 0 (except 0 itself) can\'t be palindrome
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int var1 = 0;
        int var2 = x;
        // Reverse the number
        while (x > 0) {
            int lastDigit = x % 10;
            var1 = var1 * 10 + lastDigit;
            x /= 10;
        }
        // Compare the original number with the reversed number
        return var2 == var1;
    }
}
