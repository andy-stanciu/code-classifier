class Solution {

    public boolean method1(int x) {
        int n = x;
        int n1 = 0;
        while (n > 0) {
            int l = n % 10;
            n1 = n1 * 10 + l;
            n = n / 10;
        }
        if (x >= 0 && n1 == x) {
            return true;
        } else {
            return false;
        }
    }
}
