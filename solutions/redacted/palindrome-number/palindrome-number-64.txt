class Solution {

    public boolean method1(int x) {
        String var1 = Integer.toString(x);
        int length = var1.length();
        for (int i = 0; i < length / 2; i++) {
            if (var1.charAt(i) != var1.charAt(length - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
