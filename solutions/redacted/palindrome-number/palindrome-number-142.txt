public class Solution {

    public bool method1(int x) {
        string k = x.ToString();
        for (int i = 0; i < k.Length / 2; i++) {
            if (k[i] != k[k.Length - 1 - i]) {
                // if any char not the same, return false
                return false;
            }
        }
        return true;
    }
}
