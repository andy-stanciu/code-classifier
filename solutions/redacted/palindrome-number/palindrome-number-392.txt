class Solution {

    public boolean method1(int x) {
        // Negative numbers are not palindromes
        if (x < 0) {
            return false;
        }
        // Convert the number to a string
        String s = Integer.toString(x);
        // Use helper function to check recursively
        return method2(s, 0, s.length() - 1);
    }

    private boolean method2(String s, int left, int right) {
        // Base case: If we\'ve checked all characters
        if (left >= right) {
            return true;
        }
        // If the current characters don\'t match
        if (s.charAt(left) != s.charAt(right)) {
            return false;
        }
        // Recur for the next pair of characters
        return method2(s, left + 1, right - 1);
    }
}
