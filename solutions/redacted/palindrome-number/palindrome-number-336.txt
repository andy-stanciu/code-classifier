class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        // to convert a number in string
        String strX = Integer.toString(x);
        // reverse the string
        String reversedStrX = new StringBuilder(strX).reverse().toString();
        // to check string is equal to reversed string
        return strX.equals(reversedStrX);
    }
}
