class Solution {

    public boolean method1(int n) {
        int temp, rev = 0;
        temp = n;
        while (n > 0) {
            int r = n % 10;
            rev = (rev * 10) + r;
            n = n / 10;
        }
        if (temp == rev) {
            return true;
        } else {
            return false;
        }
    }
}
