class Solution {

    public boolean method1(int x) {
        int temp = x;
        if (x == 0) {
            return true;
        } else if (x % 10 == 0 || x < 0) {
            return false;
        }
        int y = 0;
        while (x > 0) {
            y = y * 10 + x % 10;
            x = x / 10;
        }
        if (temp == y) {
            return true;
        }
        return false;
    }
}
