class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        } else {
            int var1 = x;
            int reversed = 0;
            while (var1 != 0) {
                int digit = var1 % 10;
                reversed = reversed * 10 + digit;
                var1 /= 10;
            }
            return x == reversed;
        }
    }
}
