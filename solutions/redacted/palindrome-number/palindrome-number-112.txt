class Solution {

    /**
     * @param Integer $x
     * @return Boolean
     */
    public static boolean method1(int original) {
        //paso#1
        int invertido = 0;
        int aux = original;
        //paso#2
        while (original > 0) {
            int val = original % 10;
            //paso#3
            invertido = val + (invertido * 10);
            //paso#4
            original = original / 10;
        }
        //paso#5
        return invertido == aux;
    }
}
