class Solution {

    public boolean method1(int x) {
        int temp, sum = 0, rem;
        temp = x;
        while (x > 0) {
            rem = x % 10;
            sum = (sum * 10) + rem;
            x = x / 10;
        }
        if (temp == sum) {
            return true;
        }
        return false;
    }
}
