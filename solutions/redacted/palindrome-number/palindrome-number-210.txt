class Solution {

    public boolean method1(int x) {
        int rem, rev = 0;
        int n = x;
        if (x < 0) {
            return false;
        }
        while (x != 0) {
            rem = x % 10;
            rev = rev * 10 + rem;
            x = x / 10;
        }
        System.out.println(rev);
        if (n == rev) {
            return true;
        } else {
            return false;
        }
    }
}
