class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int result = 0;
        int num = x;
        while (num > 0) {
            int div = num % 10;
            result = result * 10 + div;
            num = num / 10;
        }
        return result == x;
    }
}
