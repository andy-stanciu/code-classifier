class Solution {

    public boolean method1(int x) {
        if (x < 0) {
            return false;
        }
        int var1 = 0, var2 = x, currentDigit;
        while (x > 0) {
            currentDigit = x % 10;
            var1 = var1 * 10 + currentDigit;
            x /= 10;
        }
        return var1 == var2;
    }
}
