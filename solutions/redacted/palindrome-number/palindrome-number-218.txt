class Solution {

    public boolean method1(int x) {
        int temp = x;
        System.out.println(temp);
        int sum = 0;
        if (x < 0) {
            return false;
        } else {
            while (x > 0) {
                sum = (sum * 10) + (x % 10);
                x = x / 10;
            }
            if (temp == sum) {
                return true;
            } else {
                return false;
            }
        }
    }
}
