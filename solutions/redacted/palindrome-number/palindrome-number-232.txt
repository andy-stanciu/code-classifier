class Solution {

    public boolean method1(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) {
            return false;
        }
        int reversedHalf = 0;
        // Reverse half of the number
        while (x > reversedHalf) {
            reversedHalf = reversedHalf * 10 + x % 10;
            x /= 10;
        }
        // Check if the reversed half matches the original half (even length),
        // or ignoring the middle digit (odd length)
        return x == reversedHalf || x == reversedHalf / 10;
    }
}
