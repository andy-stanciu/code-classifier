class Solution {

    //zero indexed or one indexed
    //one indexed-[3,2,1,0,4]   ((i)1->3 steps) so reached 3 and value is 0 ,false;
    public boolean method1(int[] nums) {
        int res = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (i > res) {
                return false;
            }
            res = Math.max(res, i + nums[i]);
            if (res >= n - 1) {
                return true;
            }
        }
        return false;
    }
}
