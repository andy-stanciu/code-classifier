class Solution {

    public boolean method1(int[] nums) {
        int MaxIndex = 0;
        int i = 0;
        do {
            MaxIndex = Math.max(MaxIndex, i + nums[i]);
            if (MaxIndex >= nums.length - 1) {
                return true;
            }
            i++;
        } while (i <= MaxIndex);
        return false;
    }
}
