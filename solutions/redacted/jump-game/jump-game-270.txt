class Solution {

    public boolean method1(int[] nums) {
        // we set our target to be the last index
        int targetNumIndex = nums.length - 1;
        // we move backwards to see if any of the preceding elements can reach our target
        for (int i = nums.length - 2; i >= 0; i--) {
            // if an element can reach our target ,we make it our new target.
            if (targetNumIndex <= (i + nums[i])) {
                targetNumIndex = i;
            }
        }
        // if our first index becomes our target then we can reach the last index.
        if (targetNumIndex == 0)
            return true;
        return false;
    }
}
