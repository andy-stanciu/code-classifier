class Solution {

    public boolean method1(int[] nums) {
        boolean[] dp = new boolean[nums.length];
        int last = nums.length - 1;
        for (int i = nums.length - 1; i >= 0; --i) {
            if ((nums[i] + i) >= last) {
                last = i;
            }
        }
        return (last == 0);
    }
}
