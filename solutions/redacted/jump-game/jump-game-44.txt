class Solution {

    public boolean method1(int[] nums) {
        int last = nums.length - 1;
        // Maximum index we can reach
        int maxReach = 0;
        for (int i = 0; i <= maxReach; i++) {
            maxReach = Math.max(maxReach, i + nums[i]);
            if (maxReach >= last) {
                // We can reach the last index
                return true;
            }
        }
        // Cannot reach the last index
        return false;
    }
}
