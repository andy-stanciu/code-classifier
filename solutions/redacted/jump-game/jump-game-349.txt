class Solution {

    public boolean method1(int[] nums) {
        int reachableIndex = 0;
        int lastIndex = 0;
        if (nums.length == 1) {
            return true;
        }
        for (int i = 0; i < nums.length; i++) {
            reachableIndex = i + nums[i];
            lastIndex = Math.max(lastIndex, reachableIndex);
            if (lastIndex == i) {
                break;
            }
        }
        if (lastIndex < nums.length - 1) {
            return false;
        } else {
            return true;
        }
    }
}
