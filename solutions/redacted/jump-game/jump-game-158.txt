class Solution {

    public boolean method1(int[] nums) {
        return dfs(nums, 0, new int[nums.length]);
    }

    private boolean dfs(int[] nums, int index, int[] visited) {
        // at last index = yes we can reach here
        if (index >= nums.length - 1) {
            return true;
        }
        // had already visited here
        // go back no need to recurse again
        if (visited[index] > 0) {
            return false;
        }
        // set as visited
        visited[index] = 1;
        boolean result = false;
        for (int i = 1; i <= nums[index]; i++) {
            int var1 = index + i;
            // even 1 true means, it has found a way
            // break and exit
            if (dfs(nums, var1, visited)) {
                result = true;
                break;
            }
        }
        return result;
    }
}
