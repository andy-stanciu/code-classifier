class Solution {

    public boolean method1(int[] nums) {
        int finalPos = nums.length - 1;
        // check from last and update the destination
        for (int idx = nums.length - 2; idx >= 0; idx--) {
            if (idx + nums[idx] >= finalPos) {
                finalPos = idx;
            }
        }
        return finalPos == 0;
    }
}
