class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int maxi = 0;
        for (int i = 0; i < n; i++) {
            if (i > maxi) {
                return false;
            }
            maxi = Math.max(maxi, i + nums[i]);
        }
        return true;
    }
}
