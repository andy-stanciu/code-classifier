/**
 * Idea: Calculate the maximum window thats reacheable from every index and
 * iterate over that window keeping max window updated.
 */
class Solution {

    public boolean method1(int[] nums) {
        int len = nums.length;
        if (len == 1) {
            return true;
        }
        int maxInd = nums[0];
        boolean possible = false;
        int i = 1;
        while (maxInd < len - 1 && i <= maxInd) {
            if (i + nums[i] > maxInd) {
                maxInd = i + nums[i];
            }
            i++;
        }
        if (maxInd >= len - 1)
            return true;
        else
            return false;
    }
}
