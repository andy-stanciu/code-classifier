class Solution {

    public boolean method1(int[] nums) {
        if (nums.length <= 1) {
            return true;
        }
        boolean[] dp = new boolean[nums.length];
        Arrays.fill(dp, true);
        return helper(nums, 0, dp);
    }

    private boolean helper(int[] nums, int index, boolean[] dp) {
        if (index >= nums.length - 1) {
            return true;
        }
        if (dp[index] == false) {
            return false;
        }
        for (int i = index + nums[index]; i - index > 0; i--) {
            if (helper(nums, i, dp)) {
                return true;
            } else {
                dp[i] = false;
            }
        }
        return false;
    }
}
