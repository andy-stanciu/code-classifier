class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1) {
            return true;
        }
        int maxreach = nums[0];
        int lastindex = nums.length - 1;
        for (int i = 1; i <= maxreach; i++) {
            maxreach = Math.max(maxreach, (nums[i] + i));
            if (maxreach >= lastindex) {
                return true;
            }
        }
        return false;
    }
}
