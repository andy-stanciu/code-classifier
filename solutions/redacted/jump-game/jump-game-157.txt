class Solution {

    public boolean method1(int[] nums) {
        int i = 0;
        // keeps track of the maximum index that we can reach.
        int maxInd = 0;
        int n = nums.length;
        while (i < n - 1) {
            maxInd = Math.max(maxInd, i + nums[i]);
            // if we reached the last index or greater than that in max index we return true.. instead of iterating till last point
            if (maxInd >= n - 1)
                return true;
            // if we cant move furthur from that point then we return false
            if (maxInd == i)
                return false;
            i++;
        }
        return true;
    }
}
