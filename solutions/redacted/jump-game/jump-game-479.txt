class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        // This keeps track of the furthest index we can reach.
        int reachable = 0;
        for (int i = 0; i < n; i++) {
            if (i > reachable) {
                // If the current index is greater than the furthest reachable index, return false.
                return false;
            }
            // Update the furthest reachable index.
            reachable = Math.max(reachable, i + nums[i]);
            if (reachable >= n - 1) {
                // If the last index is reachable, return true.
                return true;
            }
        }
        return false;
    }
}
