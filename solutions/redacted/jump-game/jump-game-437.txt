class Solution {

    public boolean method1(int[] nums) {
        //2,3,1,1,4
        int finalPos = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= finalPos) {
                finalPos = i;
            }
        }
        return finalPos == 0;
    }
}
