// Greedy
class Solution {

    public boolean method1(int[] nums) {
        int reach = 0;
        int i = 0;
        for (int x : nums) {
            if (reach < i) {
                return false;
            }
            reach = Math.max(reach, i + x);
            if (reach >= nums.length - 1) {
                return true;
            }
            i++;
        }
        return false;
    }
}
