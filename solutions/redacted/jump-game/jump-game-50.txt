class Solution {

    public boolean method1(int[] nums) {
        int max = 0;
        int cur = 0;
        for (int i = 0; i < nums.length; i++) {
            if (max < i) {
                return false;
            }
            cur = i + nums[i];
            max = max > cur ? max : cur;
        }
        return true;
    }
}
