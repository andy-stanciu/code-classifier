class Solution {

    public boolean method1(int[] nums) {
        int k = nums.length;
        int reachable = 0;
        for (int i = 0; i < k; i++) {
            if (i > reachable) {
                return false;
            } else {
                reachable = Math.max(reachable, i + nums[i]);
            }
            if (reachable >= k - 1) {
                return true;
            }
        }
        return true;
    }
}
