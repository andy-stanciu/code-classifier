class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0, lenMinus1 = nums.length - 1;
        for (int i = 0; var1 >= i && var1 < lenMinus1; i++) {
            if (i + nums[i] > var1) {
                //if Ai=0, maxJump = i
                var1 = i + nums[i];
            }
        }
        return var1 >= lenMinus1;
    }
}
