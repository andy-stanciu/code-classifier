class Solution {

    public boolean method1(int[] nums) {
        int maxIndex = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > maxIndex) {
                return false;
            }
            maxIndex = Math.max(maxIndex, nums[i] + i);
            if (maxIndex >= nums.length) {
                return true;
            }
        }
        return true;
    }
}
