class Solution {

    public boolean method1(int[] nums) {
        // Initialize the farthest index we can reach
        int farthestReachable = 0;
        for (int i = 0; i < nums.length; i++) {
            // If the current index is beyond the farthest reachable index, return false
            if (i > farthestReachable) {
                return false;
            }
            // Update the farthest reachable index
            farthestReachable = Math.max(farthestReachable, i + nums[i]);
            // If we can reach or exceed the last index, return true
            if (farthestReachable >= nums.length - 1) {
                return true;
            }
        }
        return false;
    }
}
