class Solution {

    public boolean method1(int[] nums) {
        int dp = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (i == 0) {
                dp = nums[i];
            } else {
                dp = Math.max(nums[i], dp);
            }
            dp--;
            if (dp < 0) {
                return false;
            }
        }
        return true;
    }
}
