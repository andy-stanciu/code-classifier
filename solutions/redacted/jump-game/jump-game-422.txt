class Solution {

    public boolean method1(int[] nums) {
        int farthest = 0;
        for (int i = 0; i < nums.length; i++) {
            int currentValue = i + nums[i];
            if (i > farthest) {
                return false;
            }
            if (currentValue > farthest) {
                farthest = currentValue;
            }
            if (farthest >= nums.length - 1) {
                return true;
            }
        }
        return false;
    }
}
