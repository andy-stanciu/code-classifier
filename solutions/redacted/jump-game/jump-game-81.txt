class Solution {

    public boolean method1(int[] nums) {
        int maxHere = 0, n = nums.length;
        if (n <= 1)
            return true;
        for (int i = 0; i < n; i++) {
            maxHere = Math.max(maxHere, i + nums[i]);
            if (i >= maxHere)
                return false;
            if (maxHere >= n - 1) {
                return true;
            }
        }
        return false;
    }
}
