class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int max1 = 0;
        for (int i = 0; i < n; i++) {
            if (i > max1) {
                return false;
            }
            max1 = Math.max(max1, i + nums[i]);
        }
        return true;
    }
}
