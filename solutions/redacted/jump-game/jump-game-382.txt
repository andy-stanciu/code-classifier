class Solution {

    public boolean method1(int[] nums) {
        boolean bool = false;
        if (nums.length == 1) {
            return true;
        }
        int s = nums.length - 2;
        int e = nums.length - 1;
        while (s != -1) {
            if (e - s <= nums[s]) {
                e = s;
                s--;
                bool = true;
            } else {
                s--;
                bool = false;
            }
        }
        return bool;
    }
}
