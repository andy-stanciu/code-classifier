class Solution {

    public boolean method1(int[] nums) {
        int counter = 1;
        boolean var1 = true;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (nums[i] >= counter) {
                var1 = true;
                counter = 1;
            } else {
                var1 = false;
                counter++;
            }
        }
        return var1;
    }
}
