class Solution {

    public boolean method1(int[] nums) {
        // O(n) time | O(1) space
        int max_reach = 0;
        for (int i = 0; i < nums.length; i++) {
            if (max_reach < i)
                return false;
            max_reach = Math.max(max_reach, i + nums[i]);
        }
        return true;
    }
}
