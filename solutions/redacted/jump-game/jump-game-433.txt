class Solution {

    public boolean method1(int[] nums) {
        // if(nums.length==1){
        //     return true;
        // }
        int index = nums.length - 1;
        int sort = nums.length - 2;
        while (sort >= 0) {
            if (nums[sort] < (index - sort)) {
                sort--;
            } else {
                index = sort;
                sort--;
            }
        }
        if (index == 0) {
            return true;
        }
        return false;
    }
}
