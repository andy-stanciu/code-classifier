class Solution {

    public boolean method1(int[] nums) {
        // Track the farthest index we can reach
        int farthest = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > farthest) {
                // If our current index is beyond the farthest we can jump, return false
                return false;
            }
            // Update the farthest index we can reach from the current index
            farthest = Math.max(farthest, i + nums[i]);
            if (farthest >= nums.length - 1) {
                // If we can reach or exceed the last index, return true
                return true;
            }
        }
        // If the loop finishes and we haven\'t reached the last index, return false
        return false;
    }
}
