class Solution {

    public boolean method1(int[] nums) {
        if (nums.length == 1)
            return true;
        int var1 = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (// Out of jumps cant reach the current index
            var1 <= 0)
                return false;
            //Recalculate jumps remaining.
            var1 = Math.max(var1 - 1, nums[i]);
            //Be greedy so you have more jumps.
        }
        return true;
    }
}
