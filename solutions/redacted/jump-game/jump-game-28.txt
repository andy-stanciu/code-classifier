class Solution {

    public boolean method1(int[] nums) {
        // Initilize variable reach as 0, to store reach of the highest index.
        int reach = 0;
        for (int i = 0; i < nums.length; i++) {
            // Iterate the nums and check if reach is smaller than i then return false else overwrite reach with max of reach and i+nums[i].
            if (reach < i) {
                return false;
            }
            reach = Math.max(reach, i + nums[i]);
        }
        // return true, beacause we reach the last value of the array nums.
        return true;
    }
}
