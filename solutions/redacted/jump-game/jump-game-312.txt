class Solution {

    public boolean method1(int[] nums) {
        boolean[] dp = new boolean[nums.length + 1];
        return create(nums, 0, dp);
    }

    public boolean create(int[] nums, int idx, boolean[] dp) {
        //int idx=0;
        if (idx == nums.length - 1)
            return true;
        if (nums[idx] == 0)
            return false;
        if (dp[idx] != false)
            return dp[idx];
        int reach = idx + nums[idx];
        for (int var1 = idx + 1; var1 <= reach; var1++) {
            if (var1 < nums.length && create(nums, var1, dp)) {
                return dp[idx] = true;
            }
        }
        return dp[idx] = false;
    }
}
