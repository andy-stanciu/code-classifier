//Approach 2
class Solution {

    public boolean method1(int[] nums) {
        int index = nums.length - 1;
        boolean answer;
        for (int i = nums.length - 2; i > -1; i--) {
            if (i + nums[i] >= index)
                index = i;
        }
        return answer = (index == 0) ? true : false;
    }
}
