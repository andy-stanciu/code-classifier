class Solution {

    public boolean method1(int[] nums) {
        int idx = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            for (int j = 0; j <= nums[i]; j++) {
                if (i + j >= idx && i + j < nums.length) {
                    idx = i;
                    break;
                }
            }
        }
        if (idx == 0) {
            return true;
        } else {
            return false;
        }
    }
}
