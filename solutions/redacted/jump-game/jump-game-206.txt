class Solution {

    public boolean method1(int[] nums) {
        int lastThatCanReach = nums.length - 1;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (i + nums[i] >= lastThatCanReach) {
                lastThatCanReach = i;
            }
        }
        return lastThatCanReach == 0;
    }
}
