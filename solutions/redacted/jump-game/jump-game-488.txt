class Solution {

    public boolean method1(int[] nums) {
        if (nums.length <= 1) {
            return true;
        }
        int var1 = 0;
        int i = 0;
        while (i <= nums.length - 1) {
            if (i > var1) {
                return false;
            }
            if (i + nums[i] > var1) {
                var1 = i + nums[i];
            }
            i++;
        }
        return true;
    }
}
