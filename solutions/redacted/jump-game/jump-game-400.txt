class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        // Maximum index we can reach so far
        int maxReach = 0;
        for (int i = 0; i < n; i++) {
            // If the current index is beyond the maxReach, we cannot move further
            if (i > maxReach) {
                return false;
            }
            // Update maxReach based on the current index
            maxReach = Math.max(maxReach, i + nums[i]);
            // If the maxReach is beyond or at the last index, return true
            if (maxReach >= n - 1) {
                return true;
            }
        }
        return false;
    }
}
