class Solution {

    public boolean method1(int[] nums) {
        int n = nums.length;
        int indi = 0;
        for (int i = 0; i < n; i++) {
            if (indi < i) {
                return false;
            }
            indi = Math.max(indi, i + nums[i]);
        }
        return true;
    }
}
