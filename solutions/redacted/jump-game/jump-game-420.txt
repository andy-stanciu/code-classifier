class Solution {

    public boolean method1(int[] nums) {
        // Initialize the target index to the last position of the array
        int targetIndex = nums.length - 1;
        // Iterate backwards from the second last index to the start
        for (int currentIndex = nums.length - 2; currentIndex >= 0; currentIndex--) {
            // Check if the current index can jump to or beyond the target index
            if (currentIndex + nums[currentIndex] >= targetIndex) {
                // If so, update the target index to the current index
                targetIndex = currentIndex;
            }
        }
        // If the target index is zero, we can reach the last index from the start
        return targetIndex == 0;
    }
}
