class Solution {

    public boolean method1(int[] nums) {
        int maxIndex = nums.length - 1;
        for (int i = nums.length - 1; i >= 0; i--) {
            if (i + nums[i] >= maxIndex) {
                maxIndex = i;
            }
        }
        if (maxIndex == 0) {
            return true;
        }
        return false;
    }
}
