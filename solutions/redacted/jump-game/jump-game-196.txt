class Solution {

    public boolean method1(int[] nums) {
        boolean isZeroBlocking = false;
        int j = 0;
        for (int i = nums.length - 2; i >= 0; i--) {
            if (isZeroBlocking && (j - i) < nums[i])
                isZeroBlocking = false;
            if (!isZeroBlocking && nums[i] == 0) {
                isZeroBlocking = true;
                j = i;
            }
        }
        return !isZeroBlocking;
    }
}
