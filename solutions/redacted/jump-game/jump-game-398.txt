public class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > var1) {
                // If we reach index which cannot be reached from previous indexes
                return false;
            }
            var1 = Math.max(var1, i + nums[i]);
        }
        return var1 >= nums.length - 1;
    }
}
