class Solution {

    public boolean method1(int[] nums) {
        int len = nums.length;
        boolean[] dp = new boolean[len];
        dp[0] = true;
        for (int i = 0; i < len; i++) {
            if (!dp[i]) {
                break;
            }
            for (int j = i; j <= i + nums[i] && j < len; j++) {
                dp[j] = true;
            }
        }
        return dp[len - 1];
    }
}
