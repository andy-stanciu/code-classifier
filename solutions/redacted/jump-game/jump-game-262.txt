class Solution {

    public boolean method1(int[] nums) {
        int var1 = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            var1--;
            if (nums[i] > var1) {
                var1 = nums[i];
            }
            if (var1 == 0) {
                return false;
            }
        }
        return true;
    }
}
