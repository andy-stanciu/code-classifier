class Solution {

    public boolean method1(int[] nums) {
        int curr = 0;
        int end = nums[curr];
        if (nums.length == 1)
            return true;
        while (curr < end) {
            curr++;
            if (curr + nums[curr] > end)
                end = curr + nums[curr];
            if (end >= nums.length - 1)
                return true;
        }
        return false;
    }
}
