class Solution {

    public boolean method1(int[] nums) {
        int max = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > max) {
                //i max ko cross krke phucha kese? not possible so false
                return false;
            }
            max = Math.max(max, i + nums[i]);
            if (max >= nums.length - 1) {
                //max already last phuch chuka
                return true;
            }
        }
        return true;
    }
}
