class Solution {

    public static boolean method1(int[] nums) {
        int furthestReachable = 0;
        for (int i = 0; i < nums.length; i++) {
            // If we reach an index beyond what we can reach, return false
            if (i > furthestReachable) {
                return false;
            }
            // Update the furthest index we can reach from this index
            furthestReachable = Math.max(furthestReachable, i + nums[i]);
            // If the furthest reachable index is beyond or at the last index, return true
            if (furthestReachable >= nums.length - 1) {
                return true;
            }
        }
        return false;
    }

    public static void main(String[] args) {
        int[] nums = { 2, 3, 1, 1, 4 };
        // Output: true
        System.out.println("Can reach last index: " + method1(nums));
    }
}
