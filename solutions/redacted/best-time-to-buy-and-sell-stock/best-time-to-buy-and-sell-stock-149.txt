class Solution {

    public int maxProfit(int[] prices) {
        int sp = 0;
        int profit = 0;
        for (int i = prices.length - 1; i >= 0; i--) {
            sp = Math.max(sp, prices[i]);
            profit = Math.max(profit, sp - prices[i]);
        }
        return profit;
    }
}
//TC:O(n)   n:size of array prices
/*
 APPROACH: profit=sp-cp
 Traverse from the last index of array prices which is aur cp and find maximum sp and projfit for each iteration
