/* We want one transaction only so buy on day 1 and we need to sell on other  day so for every
coming day , treat it as Potential selling day and compute your MaxProfit, 
also check if today\'s price of stock is less than the price on which you bought , if yes,
then update cost price and further check for max profit in future */
class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int costPrice = prices[0];
        for (int i = 1; i < prices.length; i++) {
            //this is my potential selling day
            maxProfit = Math.max(maxProfit, prices[i] - costPrice);
            //will check for profit with this cost price
            if (prices[i] < costPrice)
                costPrice = prices[i];
        }
        return maxProfit;
    }
}
