class Solution {

    public int maxProfit(int[] prices) {
        int b = prices[0], max = 0;
        int n = prices.length;
        for (int i = 0; i < n; i++) {
            if (b > prices[i]) {
                b = prices[i];
            } else if (prices[i] - b > max) {
                max = prices[i] - b;
            }
        }
        return max;
    }
}
