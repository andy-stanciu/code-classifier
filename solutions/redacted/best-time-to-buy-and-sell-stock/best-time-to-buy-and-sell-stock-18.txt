class Solution {

    public int maxProfit(int[] prices) {
        int min = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int c : prices) {
            if (c < min) {
                min = c;
            } else if (c - min > maxProfit) {
                maxProfit = c - min;
            }
        }
        return maxProfit;
    }
}
