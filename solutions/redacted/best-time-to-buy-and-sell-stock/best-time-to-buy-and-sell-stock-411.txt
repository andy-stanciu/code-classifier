class Solution {

    public int maxProfit(int[] prices) {
        //imaging every day as a selling day
        //buy a stock at a minimum day
        int var1 = prices[0];
        int maxprofit = 0;
        for (int i = 1; i < prices.length; i++) {
            // now sell the stock to the max price;
            // maxprofit : jo buyingprice hai subtract it from current price of i;
            // and most imp thing we sell the only  if prices[i] > buuying price
            if (prices[i] > var1) {
                maxprofit = Math.max(maxprofit, prices[i] - var1);
            }
            // buy the stock if it is minimum than the current buystock
            var1 = Math.min(var1, prices[i]);
        }
        return maxprofit;
    }
}
