class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        for (int var2 = 0; var2 < prices.length - 1; var2++) {
            for (int var1 = var2 + 1; var1 < prices.length; var1++) {
                if (prices[var2] < prices[var1]) {
                    profit = Math.max(profit, prices[var1] - prices[var2]);
                } else {
                    var2 = var1;
                }
            }
        }
        return profit;
    }
}
