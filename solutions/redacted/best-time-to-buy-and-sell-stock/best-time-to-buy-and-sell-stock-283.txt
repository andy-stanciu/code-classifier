class Solution {

    public int maxProfit(int[] prices) {
        // assign two pointers
        int left = 0;
        int right = 0;
        // to temporarily store the difference between stocks
        int current = 0;
        // to store and compare the maximum profit
        int result = 0;
        boolean found = false;
        // bring the left pointer to the first index where the next value increases
        while (left != prices.length - 1 && prices[left + 1] <= prices[left]) {
            left++;
        }
        // bring the right pointer in front of the left pointer
        right = left + 1;
        // while the right pointer has not reached the end
        while (right < prices.length) {
            // if there is positive profit, store it in the current variable
            if (prices[right] > prices[left]) {
                current = prices[right] - prices[left];
            }
            // if the current profit is greater than the current result, update the result
            if (current > result) {
                result = current;
            }
            // if the right pointer finds a value less than the current left pointers value, move the left pointer to the new minimum value
            if (prices[right] < prices[left]) {
                left = right;
            }
            // move the right pointer
            right++;
        }
        return result;
    }
}
