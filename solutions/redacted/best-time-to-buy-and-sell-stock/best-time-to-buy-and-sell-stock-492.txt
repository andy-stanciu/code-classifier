class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = Integer.MIN_VALUE;
        for (int i = 0; i < prices.length; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            }
            profit = Math.max(profit, prices[i] - var1);
        }
        return profit;
    }
}
