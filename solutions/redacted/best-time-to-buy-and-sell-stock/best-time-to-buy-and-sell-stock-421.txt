class Solution {

    public int maxProfit(int[] prices) {
        // Initialize minPrice with a very large value (infinity equivalent).
        int minPrice = Integer.MAX_VALUE;
        // Variable to store the maximum profit observed.
        int profit = 0;
        // Iterate through the list of prices.
        for (int i = 0; i < prices.length; i++) {
            // Update minPrice to be the minimum of the current price or the previous minPrice.
            minPrice = Math.min(prices[i], minPrice);
            // Calculate potential profit if sold at current price and update max profit.
            profit = Math.max(profit, prices[i] - minPrice);
        }
        // Return the maximum profit.
        return profit;
    }
}
