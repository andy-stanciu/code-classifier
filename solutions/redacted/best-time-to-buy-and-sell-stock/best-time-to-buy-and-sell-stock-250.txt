class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int var1 = prices[0];
        int profit = 0;
        for (int i = 1; i < n; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else if (prices[i] - var1 > profit) {
                profit = prices[i] - var1;
            }
        }
        return profit;
    }
}
