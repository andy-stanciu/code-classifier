class Solution {

    public int maxProfit(int[] prices) {
        int by = prices[0];
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            by = Math.min(by, prices[i]);
            int newprofit = prices[i] - by;
            profit = Math.max(newprofit, profit);
        }
        return profit;
    }
}
