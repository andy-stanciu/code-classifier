class Solution {

    public int maxProfit(int[] prices) {
        int res = prices[0];
        int max = 0;
        int i = 1;
        while (i < prices.length) {
            if (res > prices[i]) {
                res = prices[i];
            } else {
                max = Math.max(max, prices[i] - res);
                i++;
            }
        }
        return max;
    }
}
