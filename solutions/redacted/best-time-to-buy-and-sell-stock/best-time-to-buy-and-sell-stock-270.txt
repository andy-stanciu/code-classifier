class Solution {

    public int maxProfit(int[] prices) {
        int small = Integer.MIN_VALUE;
        int big = Integer.MAX_VALUE;
        int diff = 0;
        for (int val : prices) {
            if (val >= small)
                small = val;
            if (val <= big)
                small = big = val;
            if ((small - big) > diff)
                diff = small - big;
        }
        return diff;
    }
}
