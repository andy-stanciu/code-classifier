class Solution {

    public int maxProfit(int[] prices) {
        int start = 1;
        int i = 0;
        int max = 0;
        while (start < prices.length) {
            if (prices[start] > prices[i]) {
                max = Math.max(max, prices[start] - prices[i]);
            } else {
                i = start;
            }
            start++;
        }
        return max;
    }
}
