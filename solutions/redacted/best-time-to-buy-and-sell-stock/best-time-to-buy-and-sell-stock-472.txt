class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length < 2) {
            // No transaction is possible
            return 0;
        }
        // Initialize with the first price
        int minPrice = prices[0];
        // Initial profit
        int maxProfit = 0;
        for (int i = 1; i < prices.length; i++) {
            // Update minPrice if a new lower price is found
            if (prices[i] < minPrice) {
                minPrice = prices[i];
            }
            // Calculate the potential profit if selling at the current price
            int profit = prices[i] - minPrice;
            // Update maxProfit if a higher profit is found
            if (profit > maxProfit) {
                maxProfit = profit;
            }
        }
        return maxProfit;
    }
}
