class Solution {

    public int maxProfit(int[] prices) {
        if ((prices.length) == 0)
            return 0;
        int var1 = prices[0];
        int diff = 0;
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            diff = (prices[i] - var1);
            profit = Math.max(profit, diff);
        }
        return profit;
    }
}
