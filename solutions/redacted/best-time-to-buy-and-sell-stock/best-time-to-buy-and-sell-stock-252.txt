class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 0)
            return 0;
        int minPrice = Integer.MAX_VALUE;
        int maxValue = 0;
        for (int price : prices) {
            minPrice = Math.min(minPrice, price);
            int profit = price - minPrice;
            maxValue = Math.max(maxValue, profit);
        }
        return maxValue;
    }
}
