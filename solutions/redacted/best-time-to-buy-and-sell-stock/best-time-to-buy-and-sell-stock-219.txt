class Solution {

    public int maxProfit(int[] P) {
        int minI = 0;
        int maxP = 0;
        for (int i = 0; i < P.length; i++) {
            if (P[i] <= P[minI]) {
                minI = i;
            }
            maxP = Math.max(maxP, P[i] - P[minI]);
        }
        return maxP;
    }
}
