class Solution {

    public int maxProfit(int[] nums) {
        int profit = 0;
        int n = nums.length;
        int max = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            max = Math.max(nums[i], max);
            profit = Math.max(max - nums[i], profit);
        }
        return profit;
    }
}
