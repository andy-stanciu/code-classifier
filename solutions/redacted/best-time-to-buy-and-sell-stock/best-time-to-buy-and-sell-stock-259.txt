class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0, var2 = 1;
        int profit = 0;
        while (var2 < prices.length) {
            if (prices[var2] < prices[var1])
                var1 = var2;
            profit = Math.max(profit, prices[var2] - prices[var1]);
            var2++;
        }
        return profit;
    }
}
