class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int maximumProfit = 0, var1 = Integer.MAX_VALUE;
        int i = 0;
        while (i < n) {
            var1 = Math.min(var1, prices[i]);
            // whenever the price of current stock is greater then then the stock value which we bought then only we will sell the stock
            if (prices[i] >= var1)
                maximumProfit = Math.max(maximumProfit, prices[i] - var1);
            i++;
        }
        return maximumProfit;
    }
}
