class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int minTillNow = prices[0];
        for (int i = 1; i < prices.length; i++) {
            minTillNow = Math.min(minTillNow, prices[i]);
            profit = Math.max(profit, prices[i] - minTillNow);
        }
        return profit;
    }
}
