class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int bp = prices[0];
        int max = 0;
        for (int i = 1; i < prices.length; i++) {
            if (bp < prices[i]) {
                profit = prices[i] - bp;
                max = Math.max(max, profit);
            } else {
                bp = prices[i];
            }
        }
        return max;
    }
}
