class Solution {

    public int maxProfit(int[] prices) {
        //initially taking first element as min
        int min = prices[0];
        //max profit is 0 initially
        int max = 0;
        for (int i = 0; i < prices.length; i++) {
            if (//checking if there is any prices less than min
            prices[i] < min)
                min = prices[i];
            if (//similarly checking max
            max < prices[i] - min)
                max = prices[i] - min;
        }
        return max;
    }
}
