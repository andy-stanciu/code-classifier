class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 1)
            return 0;
        int var1 = 0;
        int max_profit = 0;
        for (int i = 1; i < prices.length; i++) {
            /*if getting stock at lower price than before then buy it*/
            if (prices[var1] > prices[i]) {
                var1 = i;
            } else /*if current price is greater than buying prices, sell it and keep mark of max profit*/
            if (prices[i] > prices[var1]) {
                max_profit = Math.max(max_profit, prices[i] - prices[var1]);
            }
        }
        return max_profit;
    }
}
