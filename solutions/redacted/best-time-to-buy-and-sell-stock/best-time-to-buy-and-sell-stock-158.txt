class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            } else {
                int var2 = prices[i] - var1;
                profit = Math.max(var2, profit);
            }
        }
        return profit;
    }
}
