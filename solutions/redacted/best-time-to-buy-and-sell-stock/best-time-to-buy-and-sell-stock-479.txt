class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null || prices.length == 0) {
            // Edge case: if the array is empty, there\'s no profit.
            return 0;
        }
        // Initialize the minimum price to the first day\'s price
        int minPrice = prices[0];
        // Initialize the maximum profit to 0
        int maxProfit = 0;
        // Loop through the array starting from the second day
        for (int i = 1; i < prices.length; i++) {
            // Update minPrice if we find a new lower price
            if (prices[i] < minPrice) {
                minPrice = prices[i];
            } else {
                // Calculate the profit if we sell on the current day
                int profit = prices[i] - minPrice;
                // Update maxProfit if we get a new maximum profit
                if (profit > maxProfit) {
                    maxProfit = profit;
                }
            }
        }
        return maxProfit;
    }
}
