class Solution {

    public int maxProfit(int[] prices) {
        int diffrence = 0;
        int min = prices[0];
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < min) {
                min = prices[i];
            }
            diffrence = Math.max(diffrence, prices[i] - min);
        }
        return diffrence;
    }
}
