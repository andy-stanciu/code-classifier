class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int size = prices.length;
        int profit = 0;
        for (int i = 1; i < size; i++) {
            if (prices[i] < min) {
                min = prices[i];
            }
            if (prices[i] - min > profit) {
                profit = prices[i] - min;
            }
        }
        return profit;
    }
}
