class Solution {

    public int maxProfit(int[] prices) {
        int max = 0;
        int minIndex = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < prices[minIndex]) {
                minIndex = i;
            }
            if (prices[i] - prices[minIndex] > max) {
                max = prices[i] - prices[minIndex];
            }
        }
        return max;
    }
}
