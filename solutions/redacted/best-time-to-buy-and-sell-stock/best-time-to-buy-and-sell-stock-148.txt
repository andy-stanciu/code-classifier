class Solution {

    public int maxProfit(int[] prices) {
        int min = prices[0];
        int maxDiff = 0;
        for (int i = 1; i < prices.length; i++) {
            if (min > prices[i]) {
                min = prices[i];
            } else if (prices[i] - min > maxDiff) {
                maxDiff = prices[i] - min;
            }
        }
        return maxDiff;
    }
}
