class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = prices[0];
        for (int i : prices) {
            if (i > var1) {
                if ((i - var1) > maxProfit)
                    maxProfit = i - var1;
            } else {
                var1 = i;
            }
        }
        return maxProfit;
    }
}
