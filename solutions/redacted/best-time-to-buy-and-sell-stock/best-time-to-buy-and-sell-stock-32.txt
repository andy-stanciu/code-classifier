class Solution {

    public int maxProfit(int[] prices) {
        if (prices == null)
            return 0;
        int minNo = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < minNo) {
                minNo = prices[i];
            } else if (prices[i] - minNo > maxProfit) {
                maxProfit = prices[i] - minNo;
            }
        }
        return maxProfit;
    }
}
