class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int p = 0, q = 1;
        while (q < prices.length) {
            if (prices[q] - prices[p] <= 0) {
                p = q;
                q++;
                continue;
            }
            if (prices[q] - prices[p] > profit)
                profit = prices[q] - prices[p];
            q++;
        }
        return profit;
    }
}
