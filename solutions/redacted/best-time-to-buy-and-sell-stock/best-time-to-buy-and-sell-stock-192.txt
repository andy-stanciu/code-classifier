class Solution {

    public int maxProfit(int[] prices) {
        int ans = 0;
        int count = 0;
        for (int i = 0; i < prices.length; i++) {
            ans = Math.max(ans, prices[i] - prices[count]);
            if (prices[i] < prices[count])
                count = i;
        }
        return ans;
    }
}
