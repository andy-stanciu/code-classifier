class Solution {

    public int maxProfit(int[] prices) {
        int curr = Integer.MAX_VALUE;
        int maxP = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < curr) {
                curr = prices[i];
            } else {
                maxP = Math.max(maxP, prices[i] - curr);
            }
        }
        return maxP;
    }
}
