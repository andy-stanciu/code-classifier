/*
Idea: Find the maximum element till that index and subtrack with the current index to get the max profit for that index, maximum of values of indexes is the max profit
 */
class Solution {

    public int maxProfit(int[] p) {
        int mxl = 0;
        int mp = 0;
        for (int i = p.length - 1; i >= 0; i--) {
            mxl = Math.max(mxl, p[i]);
            mp = Math.max(mp, mxl - p[i]);
        }
        return mp;
    }
}
