class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0, var1 = prices[0];
        for (int price : prices) {
            if (price - var1 > maxProfit)
                maxProfit = price - var1;
            else if (price < var1)
                var1 = price;
        }
        return maxProfit;
    }
}
