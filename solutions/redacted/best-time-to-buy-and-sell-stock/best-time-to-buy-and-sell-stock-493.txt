class Solution {

    public int maxProfit(int[] prices) {
        // Initial buying price
        int bp = prices[0];
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < prices.length; i++) {
            // Update price if less than bp
            if (prices[i] < bp)
                bp = prices[i];
            // Calculate profit
            int prof = prices[i] - bp;
            max = Math.max(prof, max);
        }
        return max;
    }
}
