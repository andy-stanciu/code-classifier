class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        int m = Integer.MAX_VALUE;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < m) {
                m = prices[i];
            }
            if (prices[i] - m > profit) {
                profit = prices[i] - m;
            }
        }
        return profit;
    }
}
