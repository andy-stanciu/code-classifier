class Solution {

    public int maxProfit(int[] pricesarr) {
        int indx = 0;
        int minvalue = pricesarr[0];
        int maxprofit = 0;
        for (int i = 1; i < pricesarr.length; i++) {
            if (pricesarr[i] < minvalue) {
                minvalue = pricesarr[i];
                indx = 0;
            }
            if (pricesarr[i] >= indx) {
                maxprofit = Math.max(maxprofit, pricesarr[i] - minvalue);
                indx = pricesarr[i];
            }
        }
        return maxprofit;
    }
}
