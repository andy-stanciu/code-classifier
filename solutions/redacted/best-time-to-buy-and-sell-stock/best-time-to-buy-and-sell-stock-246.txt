class Solution {

    public int maxProfit(int[] prices) {
        int left = 0;
        int right = 0;
        int profit = 0;
        if (prices.length <= 1) {
            return 0;
        }
        while (right < prices.length) {
            if (prices[left] < prices[right]) {
                profit = Math.max(profit, prices[right] - prices[left]);
            } else {
                // this is the trick
                left = right;
            }
            right++;
        }
        return profit;
    }
}
