class Solution {

    public int maxProfit(int[] prices) {
        int cheap = prices[0];
        int maxProfit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (cheap > prices[i]) {
                cheap = prices[i];
            } else {
                maxProfit = Math.max(prices[i] - cheap, maxProfit);
            }
        }
        return maxProfit;
    }
}
