class Solution {

    public int maxProfit(int[] prices) {
        // to be minimized
        int var1 = prices[0];
        // to be maximized
        int profit = 0;
        for (int i = 1; i < prices.length; i++) {
            if (prices[i] < var1) {
                // minimize buyPrice
                var1 = prices[i];
            } else if (prices[i] - var1 > profit) {
                // maximize profit
                profit = prices[i] - var1;
            }
        }
        return profit;
    }
}
