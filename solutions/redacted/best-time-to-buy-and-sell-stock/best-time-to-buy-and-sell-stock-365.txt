class Solution {

    public int maxProfit(int[] prices) {
        int profit = 0;
        if (prices.length < 2) {
            return profit;
        }
        int l = 0;
        int r = 1;
        while (true) {
            if (l >= prices.length || r >= prices.length) {
                break;
            } else if (prices[l] > prices[r]) {
                l = r;
            } else {
                profit = Math.max(profit, prices[r] - prices[l]);
                r++;
            }
        }
        return profit;
    }
}
