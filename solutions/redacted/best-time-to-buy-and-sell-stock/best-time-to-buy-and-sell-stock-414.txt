class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] > var1) {
                int profit = prices[i] - var1;
                maxProfit = Math.max(profit, maxProfit);
            } else {
                var1 = prices[i];
            }
        }
        return maxProfit;
    }
}
