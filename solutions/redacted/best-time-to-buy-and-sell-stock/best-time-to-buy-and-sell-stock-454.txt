class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 0) {
            return 0;
        }
        int t = 0, min = Integer.MAX_VALUE, max = 0;
        for (int i : prices) {
            if (i < min) {
                min = i;
            } else if (i - min > max) {
                max = i - min;
            }
        }
        return max;
    }
}
