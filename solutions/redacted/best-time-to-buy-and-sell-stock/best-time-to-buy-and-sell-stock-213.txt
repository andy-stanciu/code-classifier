class Solution {

    public int maxProfit(int[] prices) {
        int var1 = prices[0];
        int max_profit = 0;
        int var2 = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            }
            var2 = prices[i] - var1;
            if (var2 > max_profit) {
                max_profit = var2;
            }
        }
        return max_profit;
    }
}
