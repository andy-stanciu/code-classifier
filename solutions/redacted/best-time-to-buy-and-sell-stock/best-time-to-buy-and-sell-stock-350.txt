class Solution {

    public int maxProfit(int[] prices) {
        int var1 = 0;
        int profit = 0;
        for (int j = var1 + 1; j < prices.length; j++) {
            if (prices[j] < prices[var1]) {
                // && j != prices.length-1(unnecessary)
                var1 = j;
            }
            profit = Math.max(profit, prices[j] - prices[var1]);
        }
        return profit;
    }
}
