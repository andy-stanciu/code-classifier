class Solution {

    public int maxProfit(int[] prices) {
        int maxprofit = 0;
        int var1 = Integer.MAX_VALUE;
        int var2 = 0;
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < var1) {
                var1 = prices[i];
            }
            var2 = prices[i] - var1;
            if (maxprofit < var2) {
                maxprofit = var2;
            }
        }
        return maxprofit;
    }
}
