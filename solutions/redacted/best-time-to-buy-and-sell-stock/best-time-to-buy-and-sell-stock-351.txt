class Solution {

    public int maxProfit(int[] prices) {
        int var1 = Integer.MAX_VALUE;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            //Buy
            if (prices[i] < var1) {
                var1 = prices[i];
            } else //Sell
            {
                if ((prices[i] - var1) > profit)
                    profit = (prices[i] - var1);
            }
        }
        return profit;
    }
}
