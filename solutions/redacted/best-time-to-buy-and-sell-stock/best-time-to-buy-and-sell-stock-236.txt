class Solution {

    public int maxProfit(int[] prices) {
        // Initialize the minimum buying price and maximum profit
        int var1 = prices[0];
        int profit = 0;
        // Loop through each price in the list
        for (int price : prices) {
            // If the current price is lower than buyPrice, update buyPrice
            if (price < var1) {
                var1 = price;
            }
            // Calculate the current profit and update max profit if higher
            profit = Math.max(profit, price - var1);
        }
        // Return the maximum profit found
        return profit;
    }
}
