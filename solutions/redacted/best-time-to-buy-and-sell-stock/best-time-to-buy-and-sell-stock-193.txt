class Solution {

    public int maxProfit(int[] nums) {
        int mini = Integer.MAX_VALUE;
        int maxprofit = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            mini = Math.min(mini, nums[i]);
            maxprofit = Math.max(maxprofit, nums[i] - mini);
        }
        return maxprofit;
    }
}
