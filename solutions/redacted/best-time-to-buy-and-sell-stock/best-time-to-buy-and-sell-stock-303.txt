class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int profit = 0;
        int var1 = prices[0];
        for (int i = 1; i < n; i++) {
            if (var1 > prices[i]) {
                var1 = prices[i];
            }
            profit = Math.max(profit, prices[i] - var1);
        }
        return profit;
    }
}
