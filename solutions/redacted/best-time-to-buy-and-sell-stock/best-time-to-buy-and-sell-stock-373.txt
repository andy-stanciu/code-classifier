class Solution {

    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int var1 = 0;
        int var2 = 1;
        int len = prices.length;
        while (var2 < len) {
            if (prices[var1] < prices[var2]) {
                maxProfit = Math.max(maxProfit, prices[var2] - prices[var1]);
            } else {
                var1 = var2;
            }
            var2++;
        }
        return maxProfit;
    }
}
