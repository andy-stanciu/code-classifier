class Solution {

    public int maxProfit(int[] prices) {
        int n = prices.length;
        int mn = prices[0], mx = prices[0];
        int ans = 0;
        for (int i = 1; i < n; i++) {
            int x = prices[i];
            if (mn > x && i != n - 1) {
                mn = x;
                mx = 0;
            } else {
                mx = x;
            }
            ans = Math.max(ans, mx - mn);
        }
        return ans;
    }
}
