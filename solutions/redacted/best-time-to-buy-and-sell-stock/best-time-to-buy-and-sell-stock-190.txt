class Solution {

    public int maxProfit(int[] prices) {
        if (prices.length == 1)
            return 0;
        //              DYNAMIC PROGRAMMING APPROACH
        //we are going to run a loop and find selling on which day is going to give us profit, corresponding to each buying day
        //to optimize this we further find the best buying day before calculating the profit
        int var1 = prices[0];
        int maxprofit = 0;
        int profit = 0;
        for (int i = 0; i < prices.length; i++) {
            var1 = Math.min(var1, prices[i]);
            //we buy at the price of minbuysofar
            profit = prices[i] - var1;
            /* this translates to:
        minbuysofar = selling on i\'th day- buying when price is minbuysofar*/
            //Now we need to see where the profit was maximum
            maxprofit = Math.max(maxprofit, profit);
        }
        return maxprofit;
    }
}
