class Solution {

    public int maxProfit(int[] prices) {
        // Step 1: Edge case agar sirf ek din ka price diya hai to koi profit nahi hoga
        if (prices.length == 1)
            return 0;
        // Step 2: Initialize variables
        // ab tak ka sabse kam buy price
        int var1 = prices[0];
        // ab tak ka sabse zyada profit
        int maxProfit = 0;
        // Step 3: Traverse through prices array
        for (int i = 0; i < prices.length; i++) {
            // Step 4: Calculate profit agar tum minbuySoFar pe kharidte aur aaj ke din pe bechte
            int currProfit = prices[i] - var1;
            // Step 5: Agar current profit zyada hai to maxProfit ko update karo
            if (currProfit > maxProfit) {
                maxProfit = currProfit;
            }
            // Step 6: Update minbuySoFar agar current price chhoti hai
            var1 = Math.min(var1, prices[i]);
        }
        // Step 7: Return maximum profit
        return maxProfit;
    }
}
