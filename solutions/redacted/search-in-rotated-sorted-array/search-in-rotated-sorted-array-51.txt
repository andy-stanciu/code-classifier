class Solution {

    public int method1(int[] arr, int k) {
        int i = 0;
        int j = arr.length - 1;
        while (i <= j) {
            int m = (i + j) / 2;
            if (arr[m] == k)
                return m;
            if (arr[i] <= arr[m]) {
                if (k >= arr[i] && k < arr[m]) {
                    j = m - 1;
                } else {
                    i = m + 1;
                }
            } else {
                if (k > arr[m] && k <= arr[j]) {
                    i = m + 1;
                } else {
                    j = m - 1;
                }
            }
        }
        return -1;
    }
}
