class Solution {

    public int method1(int[] nums, int target) {
        //Input: nums = [  4,5,6,7,0,1,2], target = 0
        int start = 0;
        int end = nums.length - 1;
        while (start <= end) {
            int mid = (start + end) / 2;
            // find number
            if (target == nums[mid])
                return mid;
            if (nums[mid] < nums[start]) {
                //means right side array is in asc order,all the biggest number(may have) and smallest number in left
                if (target > nums[mid] && target <= nums[end]) {
                    start = mid + 1;
                } else {
                    end = mid - 1;
                }
            } else {
                //means left side array in asc order , all the biggest number (may have) and smallest number in right
                if (target >= nums[start] && target < nums[mid]) {
                    end = mid - 1;
                } else {
                    start = mid + 1;
                }
            }
        }
        return -1;
    }
}
