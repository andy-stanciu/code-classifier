class Solution {

    public int method1(int[] nums, int target) {
        return method2(nums, 0, nums.length - 1, target);
    }

    public int method2(int[] nums, int left, int right, int target) {
        if ((right - left) == 1 || left == right) {
            if (nums[left] == target) {
                return left;
            } else if (nums[right] == target) {
                return right;
            } else {
                return -1;
            }
        }
        int mid = (left + right) / 2;
        if (nums[left] <= nums[mid]) {
            if (target >= nums[left] && target <= nums[mid]) {
                return method2(nums, left, mid, target);
            } else {
                return method2(nums, mid, right, target);
            }
        } else {
            if (target >= nums[left] || target <= nums[mid]) {
                return method2(nums, left, mid, target);
            } else {
                return method2(nums, mid, right, target);
            }
        }
    }
}
