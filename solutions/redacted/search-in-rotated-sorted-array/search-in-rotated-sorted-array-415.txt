class Solution {

    public int method1(int[] nums, int target, int si, int ei) {
        if (si > ei) {
            return -1;
        }
        int mid = si + (ei - si) / 2;
        if (nums[mid] == target) {
            return mid;
        }
        if (nums[si] <= nums[mid]) {
            if (nums[si] <= target && target <= nums[mid]) {
                return method1(nums, target, si, mid - 1);
            } else {
                return method1(nums, target, mid + 1, ei);
            }
        } else {
            if (nums[mid + 1] <= target && target <= nums[ei]) {
                return method1(nums, target, mid + 1, ei);
            } else {
                return method1(nums, target, si, mid);
            }
        }
    }

    public int method2(int[] nums, int target) {
        int si = 0;
        int ei = nums.length - 1;
        int idx = method1(nums, target, si, ei);
        return idx;
    }
}
