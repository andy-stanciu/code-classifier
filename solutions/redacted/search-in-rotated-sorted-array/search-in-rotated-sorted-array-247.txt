class Solution {

    public int method1(int[] nums, int target) {
        int start = 0;
        int end = nums.length - 1;
        while (end >= start) {
            int mid = (end + start) / 2;
            if (nums[mid] == target)
                return mid;
            // left part is sorted:
            if (nums[mid] >= nums[start]) {
                if (nums[mid] >= target && nums[start] <= target)
                    end = mid - 1;
                else
                    start = mid + 1;
            } else {
                // right part is sorted:
                if (nums[mid] <= target && nums[end] >= target)
                    start = mid + 1;
                else
                    end = mid - 1;
            }
        }
        return -1;
    }
}
