class Solution {

    public int method1(int[] arr, int target) {
        int n = arr.length;
        int low = 0;
        int high = n - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (arr[mid] == target) {
                return mid;
            }
            //left sorted
            if (arr[low] <= arr[mid]) {
                if (arr[low] <= target && target <= arr[mid]) {
                    //eliminate the right half
                    high = mid - 1;
                } else {
                    //otherwise eliminate the left half and move right
                    low = mid + 1;
                }
            } else //right sorted
            {
                if (arr[mid] <= target && target <= arr[high]) {
                    //eliminate the left half
                    low = mid + 1;
                } else {
                    //eliminate the right half
                    high = mid - 1;
                }
            }
        }
        return -1;
    }
}
