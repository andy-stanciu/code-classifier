class Solution {

    /**
     * take lo and high
     *        find mid and compare
     *        else if nums[mid] >nums[low]
     *            -> if(target<nums[mid] && tagrte> nums[low])
     *                    --> hi = mid-1
     *        else
     *            -> nums[mid] <nums[lo]
     *                -->taget <num[hi] && target> num[mid]
     */
    public int method1(int[] nums, int target) {
        int left = 0;
        int right = nums.length - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target)
                return mid;
            if (nums[left] <= nums[mid]) {
                if (nums[left] <= target && target < nums[mid]) {
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            } else {
                if (nums[mid] < target && nums[right] >= target) {
                    left = mid + 1;
                } else {
                    right = right - 1;
                }
            }
        }
        return -1;
    }
}
