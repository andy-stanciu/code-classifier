class Solution {

    public int method1(int[] nums, int target) {
        int len = nums.length;
        int i = 0;
        int j = len - 1;
        while (i <= j) {
            int mid = (i + j) / 2;
            if (nums[mid] == target)
                return mid;
            else if (// checking whether first half is in sorted order or not
            nums[i] <= nums[mid]) {
                if (//if the element is in first half then we will discard second half
                target < nums[mid] && target >= nums[i])
                    j = mid - 1;
                else
                    i = mid + 1;
            } else {
                if (target > nums[mid] && target <= nums[j])
                    i = mid + 1;
                else
                    j = mid - 1;
            }
        }
        return -1;
    }
}
