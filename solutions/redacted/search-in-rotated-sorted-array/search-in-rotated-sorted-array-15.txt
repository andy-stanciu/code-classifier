class Solution {

    public int method1(int[] nums, int target) {
        int n = nums.length;
        if (n == 1) {
            if (nums[n - 1] == target) {
                return n - 1;
            }
        }
        int l = 0;
        int r = n - 1;
        while (l < r) {
            int m = l + (r - l) / 2;
            if (nums[m] > nums[r]) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        int var1 = r;
        int lt = -1;
        int rt = -1;
        if (var1 == 0) {
            lt = 0;
            rt = n - 1;
        } else if (target > nums[n - 1]) {
            lt = 0;
            rt = var1 - 1;
        } else {
            lt = var1;
            rt = n - 1;
        }
        while (lt <= rt) {
            int mid = lt + (rt - lt) / 2;
            if (target == nums[mid]) {
                return mid;
            } else if (target > nums[mid]) {
                lt = mid + 1;
            } else {
                rt = mid - 1;
            }
        }
        return -1;
    }
}
