class Solution {

    public int method1(int[] nums, int target) {
        int res = -1;
        for (int i = 0; i < nums.length; i++) {
            //res will be assigned only once as array has distinct elem
            if (nums[i] == target)
                res = i;
        }
        //means if target is there in array ->>just return that index->>if not res will not be modified means res==-1->>we will simply return it
        return res;
    }
}
