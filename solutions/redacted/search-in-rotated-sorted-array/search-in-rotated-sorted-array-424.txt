class Solution {

    public int method1(int[] nums, int target) {
        // Iterate over the array
        for (int i = 0; i < nums.length; i++) {
            // check if the element is equal to the target
            if (nums[i] == target) {
                //return the index i
                return i;
            }
        }
        //  return -1 if target is not found
        return -1;
    }
}
