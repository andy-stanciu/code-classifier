class Solution {

    private int method1(int[] nums, int n) {
        int l = 0, r = n - 1;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (nums[mid] > nums[r]) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        return r;
    }

    private int method2(int l, int r, int[] nums, int target) {
        int var1 = -1;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (nums[mid] == target) {
                return mid;
            } else if (nums[mid] < target) {
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return var1;
    }

    public int method3(int[] nums, int target) {
        int n = nums.length;
        int var2 = method1(nums, n);
        int var1 = method2(0, var2 - 1, nums, target);
        if (var1 != -1)
            return var1;
        var1 = method2(var2, n - 1, nums, target);
        return var1;
    }
}
