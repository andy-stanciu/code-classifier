class Solution {

    public int method1(int[] nums, int target) {
        return method2(nums, target, 0, nums.length - 1);
    }

    public static int method2(int[] nums, int target, int s, int e) {
        if (s > e) {
            return -1;
        }
        int mid = s + (e - s) / 2;
        if (nums[mid] == target) {
            return mid;
        }
        if (nums[s] <= nums[mid]) {
            if ((nums[s] <= target && target <= nums[mid])) {
                return method2(nums, target, s, mid - 1);
            } else {
                return method2(nums, target, mid + 1, e);
            }
        } else {
            if ((nums[mid] <= target && target <= nums[e])) {
                return method2(nums, target, mid + 1, e);
            } else {
                return method2(nums, target, s, mid - 1);
            }
        }
    }
}
//Important : consider all the 4 cases in order to find the pivot
