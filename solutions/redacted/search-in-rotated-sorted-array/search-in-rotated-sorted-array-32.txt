class Solution {

    public int method1(int[] nums, int target) {
        int low = 0, high = nums.length - 1;
        return method2(nums, low, high, target);
    }

    public int method2(int[] nums, int low, int high, int target) {
        int mid = (low + high) / 2, left = -1, right = -1;
        if (low <= high) {
            if (nums[mid] == target)
                return mid;
            else {
                left = method2(nums, low, mid - 1, target);
                right = method2(nums, mid + 1, high, target);
            }
        }
        return Math.max(left, right);
    }
}
