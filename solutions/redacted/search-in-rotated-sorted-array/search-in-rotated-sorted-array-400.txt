class Solution {

    public int method1(int[] nums, int target) {
        int k = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                k = i + 1;
                break;
            }
        }
        int a = method2(nums, target, 0, k - 1);
        if (a == -1)
            a = method2(nums, target, k, nums.length - 1);
        return a;
    }

    public int method2(int[] nums, int target, int s, int e) {
        int f = -1;
        int mid;
        while (s <= e) {
            mid = (s + e) / 2;
            if (nums[mid] == target) {
                f = mid;
                break;
            } else if (nums[mid] > target)
                e = mid - 1;
            else
                s = mid + 1;
        }
        return f;
    }
}
