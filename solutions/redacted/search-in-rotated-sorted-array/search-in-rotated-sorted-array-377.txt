class Solution {

    public int method1(int[] nums, int target) {
        // getting the length of the array
        int len = nums.length;
        int left = 0;
        int right = len - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            //check if the mid number is the target value
            if (nums[mid] == target) {
                return mid;
            } else // search the array in the left half of the array
            if (nums[left] <= nums[mid]) {
                if (target >= nums[left] && target < nums[mid]) {
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            } else // serach in the right half of the array
            {
                if (target > nums[mid] && target <= nums[right]) {
                    left = mid + 1;
                } else {
                    right = mid - 1;
                }
            }
        }
        return -1;
    }
}
