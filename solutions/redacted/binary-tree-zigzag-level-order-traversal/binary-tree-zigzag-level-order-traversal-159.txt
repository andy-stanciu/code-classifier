class Solution {

    List<List<Integer>> Arr = new ArrayList<>();

    int count = 1;

    public List<List<Integer>> method1(TreeNode root) {
        if (root == null)
            return Arr;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            List<Integer> arr = new ArrayList<>();
            int n = q.size();
            for (int i = 0; i < n; i++) {
                TreeNode ele = q.poll();
                arr.add(ele.val);
                if (ele.left != null)
                    q.add(ele.left);
                if (ele.right != null)
                    q.add(ele.right);
            }
            if (count % 2 == 0)
                Collections.reverse(arr);
            Arr.add(arr);
            count++;
        }
        return Arr;
    }
}
