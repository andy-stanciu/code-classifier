class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        var ans = new ArrayList<List<Integer>>();
        if (root == null)
            return ans;
        var zig = true;
        var queue = new ArrayDeque<TreeNode>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            var var1 = new LinkedList<Integer>();
            for (var i = queue.size(); i > 0; i--) {
                var node = queue.poll();
                if (zig)
                    var1.add(node.val);
                else
                    var1.addFirst(node.val);
                if (node.left != null)
                    queue.offer(node.left);
                if (node.right != null)
                    queue.offer(node.right);
            }
            ans.add(var1);
            zig = !zig;
        }
        return ans;
    }
}
