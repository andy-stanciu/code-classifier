/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        Deque<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        boolean reverse = false;
        while (!queue.isEmpty()) {
            int lsize = queue.size();
            List<Integer> cl = new ArrayList<>(lsize);
            for (int i = 0; i < lsize; i++) {
                if (!reverse) {
                    TreeNode cn = queue.pollFirst();
                    cl.add(cn.val);
                    if (cn.left != null) {
                        queue.addLast(cn.left);
                    }
                    if (cn.right != null) {
                        queue.addLast(cn.right);
                    }
                } else {
                    TreeNode cn = queue.pollLast();
                    cl.add(cn.val);
                    if (cn.right != null) {
                        queue.addFirst(cn.right);
                    }
                    if (cn.left != null) {
                        queue.addFirst(cn.left);
                    }
                }
            }
            reverse = !reverse;
            ans.add(cl);
        }
        return ans;
    }
}
