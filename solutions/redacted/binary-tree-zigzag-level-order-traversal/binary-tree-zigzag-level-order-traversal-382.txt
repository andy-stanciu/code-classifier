/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    List<List<Integer>> ans;

    public List<List<Integer>> method1(TreeNode root) {
        ans = new ArrayList<>();
        recursion(root, 0);
        return ans;
    }

    private void recursion(TreeNode node, int param1) {
        if (node == null)
            return;
        if (ans.size() <= param1)
            ans.add(new ArrayList<>());
        if (param1 % 2 == 0)
            ans.get(param1).add(node.val);
        else
            ans.get(param1).add(0, node.val);
        recursion(node.left, param1 + 1);
        recursion(node.right, param1 + 1);
    }
}
