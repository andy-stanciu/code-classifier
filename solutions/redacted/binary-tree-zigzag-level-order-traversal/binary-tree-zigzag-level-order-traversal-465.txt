/**
 * Definition for a binary tree node.
 * public class TreeNode {
 * int val;
 * TreeNode left;
 * TreeNode right;
 * TreeNode() {}
 * TreeNode(int val) { this.val = val; }
 * TreeNode(int val, TreeNode left, TreeNode right) {
 * this.val = val;
 * this.left = left;
 * this.right = right;
 * }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> parent = new ArrayList<>();
        if (root == null)
            return parent;
        Queue<TreeNode> q = new LinkedList<>();
        List<Integer> child = new ArrayList<>();
        q.add(root);
        q.add(null);
        boolean isLeft = true;
        while (!q.isEmpty()) {
            TreeNode curr = q.remove();
            if (curr != null) {
                if (isLeft) {
                    child.add(curr.val);
                } else {
                    child.add(0, curr.val);
                }
                if (curr.left != null)
                    q.add(curr.left);
                if (curr.right != null)
                    q.add(curr.right);
            } else {
                if (!q.isEmpty())
                    q.add(null);
                parent.add(child);
                child = new ArrayList<>();
                isLeft = !isLeft;
            }
        }
        return parent;
    }
}
