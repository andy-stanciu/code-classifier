/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> result;

    public List<List<Integer>> method1(TreeNode root) {
        result = new ArrayList<>();
        traverse(root, 0);
        return result;
    }

    private void traverse(TreeNode root, int param1) {
        if (root == null)
            return;
        if (param1 >= result.size())
            result.add(param1, new ArrayList<>());
        if (param1 % 2 == 0) {
            result.get(param1).add(root.val);
        } else {
            result.get(param1).add(0, root.val);
        }
        traverse(root.left, param1 + 1);
        traverse(root.right, param1 + 1);
    }
}
