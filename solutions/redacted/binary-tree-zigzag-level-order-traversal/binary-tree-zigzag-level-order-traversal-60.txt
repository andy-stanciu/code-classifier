/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    List<List<Integer>> list = new ArrayList<>();

    public List<List<Integer>> method1(TreeNode root) {
        if (root == null) {
            return list;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        boolean reverse = false;
        while (!q.isEmpty()) {
            List<Integer> temp = new ArrayList<>();
            int sz = q.size();
            while (sz-- > 0) {
                TreeNode res = q.poll();
                temp.add(res.val);
                if (res.left != null) {
                    q.offer(res.left);
                }
                if (res.right != null) {
                    q.offer(res.right);
                }
            }
            if (reverse) {
                Collections.reverse(temp);
            }
            list.add(temp);
            reverse = !reverse;
        }
        return list;
    }
}
