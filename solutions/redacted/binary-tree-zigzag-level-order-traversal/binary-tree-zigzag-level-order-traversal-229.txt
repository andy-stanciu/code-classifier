/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        if (root == null)
            return new LinkedList<>();
        List<List<Integer>> lst = new LinkedList<>();
        // flag is used as an indicator to perform zigzag operation
        boolean flag = true;
        LinkedList<TreeNode> q = new LinkedList<>();
        q.add(root);
        TreeNode dummy = null;
        while (!q.isEmpty()) {
            List<Integer> temp = new LinkedList<>();
            // n is the length of current queue
            int n = q.size();
            while (n > 0) {
                dummy = q.remove();
                if (dummy.left != null)
                    q.add(dummy.left);
                if (dummy.right != null)
                    q.add(dummy.right);
                // flag = true indicates that values should be added at the
                // while "false" indicates values should be added at beginning
                if (flag == true) {
                    temp.add(dummy.val);
                } else {
                    temp.addFirst(dummy.val);
                }
                n--;
            }
            flag = flag == true ? false : true;
            lst.add(temp);
        }
        return lst;
    }
}
