/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
public class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> sol = new ArrayList<>();
        travel(root, sol, 0);
        return sol;
    }

    private void travel(TreeNode curr, List<List<Integer>> sol, int param1) {
        if (curr == null)
            return;
        if (sol.size() <= param1) {
            List<Integer> var1 = new LinkedList<>();
            sol.add(var1);
        }
        List<Integer> collection = sol.get(param1);
        if (param1 % 2 == 0)
            collection.add(curr.val);
        else
            collection.add(0, curr.val);
        travel(curr.left, sol, param1 + 1);
        travel(curr.right, sol, param1 + 1);
    }
}
