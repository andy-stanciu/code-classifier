/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        List<List<Integer>> li = new LinkedList<>();
        if (root == null)
            return li;
        q.offer(root);
        int var1 = 0;
        while (!q.isEmpty()) {
            var1++;
            int lvl = q.size();
            List<Integer> tl = new ArrayList<>();
            for (int i = 0; i < lvl; i++) {
                TreeNode tmp = q.poll();
                tl.add(tmp.val);
                if (tmp.left != null) {
                    q.offer(tmp.left);
                }
                if (tmp.right != null) {
                    q.offer(tmp.right);
                }
            }
            if (var1 % 2 == 0) {
                Collections.reverse(tl);
            }
            li.add(tl);
        }
        return li;
    }
}
