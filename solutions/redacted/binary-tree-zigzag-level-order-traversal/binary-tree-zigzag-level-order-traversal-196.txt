/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        if (root == null)
            return new ArrayList<>();
        List<List<Integer>> finalList = new ArrayList<>();
        method2(root, finalList, 0);
        return finalList;
    }

    public void method2(TreeNode node, List<List<Integer>> finalList, int param1) {
        if (node == null)
            return;
        if (finalList.size() <= param1) {
            LinkedList<Integer> list = new LinkedList<>();
            if (param1 % 2 == 0)
                list.addLast(node.val);
            else
                list.addFirst(node.val);
            finalList.add(list);
        } else {
            List<Integer> list = finalList.get(param1);
            if (param1 % 2 == 0)
                list.addLast(node.val);
            else
                list.addFirst(node.val);
        }
        method2(node.left, finalList, param1 + 1);
        method2(node.right, finalList, param1 + 1);
    }
}
