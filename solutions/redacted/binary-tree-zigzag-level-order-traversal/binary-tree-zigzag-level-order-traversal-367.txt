class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null)
            return res;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            int size = q.size();
            List<Integer> var1 = new ArrayList<>();
            while (size-- > 0) {
                TreeNode cur = q.poll();
                var1.add(cur.val);
                if (cur.left != null)
                    q.offer(cur.left);
                if (cur.right != null)
                    q.offer(cur.right);
            }
            res.add(var1);
        }
        for (int i = 0; i < res.size(); i++) {
            if (i % 2 != 0)
                Collections.reverse(res.get(i));
        }
        return res;
    }
}
