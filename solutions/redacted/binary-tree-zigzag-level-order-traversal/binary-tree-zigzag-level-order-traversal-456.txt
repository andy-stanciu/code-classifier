/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public List<List<Integer>> method1(TreeNode root) {
        //queue level bfs with alternate addition of children
        if (root == null) {
            return new ArrayList<>();
        }
        Queue<TreeNode> q = new LinkedList<>();
        List<List<Integer>> result = new ArrayList<>();
        q.add(root);
        //righ to left
        boolean var1 = true;
        while (!q.isEmpty()) {
            int n = q.size();
            List<Integer> var2 = new ArrayList<>();
            for (int i = 0; i < n; i++) {
                TreeNode node = q.poll();
                if (node == null)
                    continue;
                //add children
                if (var1) {
                    var2.add(node.val);
                } else {
                    //insert in front and not append
                    var2.add(0, node.val);
                }
                if (node.left != null) {
                    q.add(node.left);
                }
                if (node.right != null) {
                    q.add(node.right);
                }
            }
            var1 = !var1;
            result.add(var2);
        }
        return result;
    }
}
