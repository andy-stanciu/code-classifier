class Solution {

    public static double findPow(double x, int n) {
        if (n == 0) {
            return 1.0;
        }
        double half = findPow(x, n / 2);
        if (n % 2 == 0) {
            return half * half;
        } else {
            return x * half * half;
        }
    }

    public double myPow(double x, int n) {
        return n >= 0 ? findPow(x, n) : 1.0 / findPow(x, -n);
    }
}
