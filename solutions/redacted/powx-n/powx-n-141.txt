class Solution {

    public double myPow(double x, int N) {
        long n = N;
        if (n < 0) {
            n = -n;
            x = 1 / x;
        }
        double ans = 1;
        double curr = x;
        for (long i = n; i > 0; i /= 2) {
            if ((i % 2) == 1) {
                ans = ans * curr;
            }
            curr = curr * curr;
        }
        return ans;
    }
}
