class Solution {

    public double myPow(double x, int n) {
        if (n < 0) {
            return 1 / powerHelper(x, -n);
        }
        return powerHelper(x, n);
    }

    private static double powerHelper(double x, int n) {
        if (n == 0) {
            return 1;
        }
        double half = powerHelper(x, n / 2);
        if (n % 2 == 0) {
            return half * half;
        } else {
            return half * half * x;
        }
    }
}
