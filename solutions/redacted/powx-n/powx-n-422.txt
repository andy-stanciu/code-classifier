class Solution {

    public double myPow(double x, int n) {
        return computePower(x, (long) n);
    }

    private double computePower(double x, long n) {
        if (n == 0) {
            return 1;
        }
        if (n < 0) {
            return 1.0 / computePower(x, -n);
        }
        if (n % 2 == 1) {
            return x * computePower(x * x, n / 2);
        } else {
            return computePower(x * x, n / 2);
        }
    }
}
