class Solution {

    public double myPow(double x, int n) {
        double ans = 1.0;
        // Store n as long to avoid overflow issues
        long m = n;
        // Handle negative exponent case
        if (m < 0) {
            x = 1 / x;
            // Work with positive exponent now
            m = -m;
        }
        // Perform exponentiation
        while (m > 0) {
            if (m % 2 == 1) {
                ans *= x;
            }
            x *= x;
            m /= 2;
        }
        return ans;
    }
}
