class Solution {

    public double myPow(double x, int n) {
        long power = n;
        if (power < 0)
            power = -1 * power;
        double ans = 1.0;
        while (power >= 1) {
            if (power % 2 == 1) {
                ans *= x;
                power = power - 1;
            }
            power /= 2;
            x *= x;
        }
        return n < 0 ? 1.0 / ans : ans;
    }
}
