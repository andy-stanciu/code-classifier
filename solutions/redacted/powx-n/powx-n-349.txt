class Solution {

    public double myPow(double x, int n) {
        double ans = 1;
        long g = n;
        long p = Math.abs(g);
        while (p > 0) {
            if ((p & 1) == 1) {
                ans = ans * x;
            }
            x = x * x;
            p = p >> 1;
        }
        if (n < 0)
            return 1 / ans;
        return ans;
    }
}
