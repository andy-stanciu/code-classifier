class Solution {

    public double myPow(double x, int n) {
        //double pow = 1.00;
        // for(int i=1; i<=n; i++){
        //     pow = pow * 2;
        // }
        double pow = 1.0;
        long N = n;
        if (N < 0) {
            N = -1 * N;
        }
        while (N > 0) {
            if (N % 2 == 1) {
                pow *= x;
                N = N - 1;
            } else {
                x *= x;
                N = N / 2;
            }
        }
        if (n < 0) {
            pow = (double) (1.0) / (double) (pow);
        }
        return pow;
    }
}
