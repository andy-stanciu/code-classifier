class Solution {

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return x;
        if (n < 0) {
            return (1 / x) * myPow(1 / x, -(n + 1));
        } else {
            double p1 = myPow(x, n / 2);
            p1 = p1 * p1;
            if (n % 2 != 0)
                p1 = p1 * x;
            return p1;
        }
    }
}
