class Solution {

    public double myPow(double x, long n) {
        boolean flag = false;
        if (n < 0) {
            flag = true;
            n = -n;
        }
        double ans = 1;
        while (n > 0) {
            if (n % 2 == 1) {
                ans *= x;
                n--;
            }
            x *= x;
            n /= 2;
        }
        return flag ? 1 / ans : ans;
    }
}
