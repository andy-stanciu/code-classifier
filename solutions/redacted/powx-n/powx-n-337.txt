class Solution {

    public double myPow(double x, int n) {
        long pow = n;
        double ans = 1;
        if (n < 0) {
            pow = -pow;
        }
        while (pow > 0) {
            if (pow % 2 == 0) {
                x = x * x;
                pow = pow / 2;
            } else {
                ans = ans * x;
                pow = pow - 1;
            }
        }
        if (n < 0) {
            ans = 1 / ans;
        }
        return ans;
    }
}
