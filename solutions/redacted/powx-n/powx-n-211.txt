class Solution {

    public double power(double x, long n) {
        if (n == 0) {
            return 1;
        }
        if (n % 2 == 0) {
            return power(x * x, n / 2);
        } else {
            return x * power(x * x, n / 2);
        }
    }

    public double myPow(double x, int n) {
        if (n < 0) {
            return power(1 / x, -(long) (n));
        }
        return power(x, n);
    }
}
