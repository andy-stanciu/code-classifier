class Solution {

    public double myPow(double x, int n) {
        if (n == Integer.MIN_VALUE) {
            n = n / 2;
            x = x * x;
        }
        if (n < 0) {
            n *= -1;
            x = 1 / x;
        }
        double ans = 1.0;
        while (n != 0) {
            if (n % 2 == 1) {
                n -= 1;
                ans = ans * x;
            }
            n = n / 2;
            x = x * x;
        }
        return ans;
    }
}
