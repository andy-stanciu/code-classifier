class Solution {

    public double myPow(double x, int n) {
        double powValue;
        if (n < 0) {
            powValue = findValue(x, -n);
            return 1 / powValue;
        } else
            powValue = findValue(x, n);
        return powValue;
    }

    public double findValue(double base, int pow) {
        if (pow == 0)
            return 1.0;
        double value = findValue(base, pow / 2);
        if (pow % 2 == 0)
            return value * value;
        else
            return value * value * base;
    }
}
