class Solution {

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        double ans;
        if (n < 0) {
            if (n == Integer.MIN_VALUE) {
                ans = power(x, -(n + 1));
                return 1 / (ans * x);
            } else {
                ans = power(x, -(n));
                return 1 / ans;
            }
        }
        return power(x, n);
    }

    public static double power(double x, int n) {
        if (n == 0)
            return 1;
        //calculating beforehand to surpass TLE for a specific testcase
        double ans = power(x, n / 2);
        if (n % 2 == 0) {
            //Make sure not to use power(x,n/2)*power(x,n/2) otherwise TLE for testcase- x=0.00001, n=2147483647
            return ans * ans;
        } else {
            return ans * ans * x;
        }
    }
}
