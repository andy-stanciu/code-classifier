class Solution {

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        if (x == 0)
            return 0;
        double ans = 1;
        long absN = Math.abs((long) n);
        while (absN > 0) {
            if ((absN & 1) != 0) {
                ans *= x;
            }
            x *= x;
            absN >>= 1;
        }
        return (n < 0) ? 1 / ans : ans;
    }
}
