class Solution {

    public double myPow(double x, int n) {
        if (x == 0) {
            return 0.0;
        }
        double res = helper(x, n);
        return n > 0 ? res : 1 / res;
    }

    private double helper(double x, int n) {
        if (n == 0) {
            return 1.0;
        }
        double res = helper(x, n / 2);
        if (n % 2 == 0) {
            res = res * res;
        } else {
            res = x * res * res;
        }
        return res;
    }
}
