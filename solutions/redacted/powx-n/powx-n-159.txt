class Solution {

    public double myPow(double x, int n) {
        if (n < 0) {
            x = 1 / x;
        }
        double ans = 1;
        while (n != 0) {
            if (n % 2 == 1 || n % 2 == -1) {
                ans *= x;
            }
            x *= x;
            n /= 2;
        }
        return ans;
    }
}
