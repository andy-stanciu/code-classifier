class Solution {

    public double myPowHelper(double x, long n) {
        if (n == 1) {
            return x;
        }
        double num = myPowHelper(x, n / 2);
        return n % 2 == 0 ? num * num : x * num * num;
    }

    public double myPow(double x, int n) {
        if (n == 0)
            return 1;
        double ans = myPowHelper(x, Math.abs((long) n));
        return n > 0 ? ans : 1 / ans;
    }
}
