class Solution {

    public double myPow(double x, long n) {
        return fun(x, n);
    }

    double fun(double x, long n) {
        if (n == 0)
            return 1;
        if (n < 0)
            return 1 / fun(x, -n);
        if (n % 2 == 1)
            return x * fun(x, n - 1);
        return fun(x * x, n / 2);
    }
}
