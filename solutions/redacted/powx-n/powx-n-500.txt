class Solution {

    public double myPow(double x, int n) {
        long nn = n;
        if (nn < 0) {
            nn = -nn;
        }
        if (n < 0) {
            return 1 / recurse(x, nn);
        }
        return recurse(x, nn);
    }

    private double recurse(double x, long nn) {
        if (nn == 0) {
            return 1;
        }
        if (nn % 2 == 0) {
            return recurse(x * x, nn / 2);
        } else {
            return x * recurse(x, nn - 1);
        }
    }
}
