class Solution {

    private double helper(double x, long n) {
        if (n == 0) {
            return 1;
        }
        if (n == 1) {
            return x;
        }
        if (n < 0) {
            return 1.0 / helper(x, -n);
        }
        if (n % 2 == 0) {
            return helper(x * x, n / 2);
        } else {
            return x * helper(x * x, (n - 1) / 2);
        }
    }

    public double myPow(double x, int n) {
        return helper(x, (long) n);
    }
}
