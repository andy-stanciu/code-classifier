class Solution {

    public double myPow(double x, int n) {
        double a = x;
        double b = n;
        double ans = 1;
        if (n < 0) {
            n = Math.abs(n);
            ans = Math.pow(a, b);
            double Result = Math.round(ans * 100000.0) / 100000.0;
            return Result;
        } else {
            ans = Math.pow(a, b);
            double Result = Math.round(ans * 100000.0) / 100000.0;
            return Result;
        }
    }
}
