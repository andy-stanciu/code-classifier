class Solution {

    public double myPow(double x, int n) {
        double ans = 1.0;
        long temp = n;
        //if power i.e n is -ve then make it positive.
        if (temp < 0) {
            temp = -temp;
        }
        while (temp > 0) {
            if (temp % 2 == 1) {
                ans = ans * x;
            }
            x = x * x;
            temp = temp / 2;
        }
        if (n < 0) {
            ans = 1 / ans;
        }
        return ans;
    }
}
