class Solution {

    public double myPow(double x, int n) {
        if (n == 0)
            return 1.0;
        else {
            double ans = 1.0;
            long pow = Math.abs((long) n);
            //System.out.println(pow);
            int i = 1;
            if (pow % 2 == 1)
                ans = x;
            pow >>= 1;
            while (pow != 0) {
                x *= x;
                if (pow % 2 == 1)
                    ans *= x;
                pow >>= 1;
            }
            if (n < 0)
                return 1.0 / ans;
            else
                return ans;
        }
    }
}
