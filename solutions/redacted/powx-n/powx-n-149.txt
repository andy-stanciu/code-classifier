class Solution {

    public double myPow(double x, int n) {
        double ans = 1;
        long num = n;
        if (num < 0) {
            num = -1 * num;
        }
        while (num > 0) {
            if ((num & 1) > 0) {
                ans = ans * x;
            }
            x = x * x;
            num = num >> 1;
        }
        if (n < 0) {
            return 1.0 / ans;
        }
        return ans;
    }
}
