class Solution {

    public double myPow(double x, int n) {
        if (n == 0) {
            return 1.0;
        }
        double result = rec(x, Math.abs(n));
        return n < 0 ? 1 / result : result;
    }

    private double rec(double x, int n) {
        if (n == 0) {
            return 1.0;
        }
        double half = rec(x, n / 2);
        if (n % 2 == 0) {
            return half * half;
        } else {
            return half * half * x;
        }
    }
}
