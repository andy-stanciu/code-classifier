class Solution {

    public double myPow(double x, int n) {
        long exp = n;
        if (exp < 0) {
            exp = -exp;
        }
        double result = 1;
        while (exp > 0) {
            if ((exp % 2) == 1) {
                result *= x;
            }
            x *= x;
            exp /= 2;
        }
        if (n < 0) {
            System.out.println(1 / result);
            return 1 / result;
        }
        System.out.println(result);
        return result;
    }
}
