class Solution {

    public double myPow(double x, int N) {
        long n = N;
        if (n < 0) {
            n = -n;
            x = 1 / x;
        }
        return helper(x, n);
    }

    public double helper(double x, long n) {
        if (n == 0) {
            return 1.0;
        }
        double partition = helper(x, n / 2);
        return (n % 2 == 0) ? partition * partition : x * partition * partition;
    }
}
