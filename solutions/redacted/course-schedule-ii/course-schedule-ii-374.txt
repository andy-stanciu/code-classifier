class Solution {

    public int[] findOrder(int param1, int[][] prerequisites) {
        // Step 1: Create the adjacency list and the indegree array
        List<List<Integer>> adjList = new ArrayList<>();
        int[] indegree = new int[param1];
        for (int i = 0; i < param1; i++) {
            adjList.add(new ArrayList<>());
        }
        for (int[] prerequisite : prerequisites) {
            int var1 = prerequisite[0];
            int var2 = prerequisite[1];
            adjList.get(var2).add(var1);
            indegree[var1]++;
        }
        // Step 2: Perform the topological sorting
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < param1; i++) {
            if (indegree[i] == 0) {
                queue.offer(i);
            }
        }
        int[] order = new int[param1];
        int index = 0;
        while (!queue.isEmpty()) {
            int var1 = queue.poll();
            order[index++] = var1;
            for (int neighbor : adjList.get(var1)) {
                indegree[neighbor]--;
                if (indegree[neighbor] == 0) {
                    queue.offer(neighbor);
                }
            }
        }
        // Step 3: Check if there is a valid course order
        if (index != param1) {
            return new int[0];
        }
        return order;
    }
}
