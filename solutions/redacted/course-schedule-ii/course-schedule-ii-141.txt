class Solution {

    public int[] findOrder(int param1, int[][] prerequisites) {
        List<List<Integer>> list = new ArrayList<>();
        for (int i = 0; i < param1; i++) {
            list.add(new ArrayList<>());
        }
        ///creating adjacency list
        for (int i = 0; i < prerequisites.length; i++) {
            list.get(prerequisites[i][1]).add(prerequisites[i][0]);
        }
        //getting indegrees of nodes, how many incoming connection to a given node forms indegree of that node
        //example a<--b, a<--c,b<---c means a has indegree 2, b has indegree 1 and c has indegree 0
        int[] indegree = new int[param1];
        for (int i = 0; i < param1; i++) {
            for (int node : list.get(i)) {
                indegree[node]++;
            }
        }
        //getting nodes with 0 indegrees
        // nodes that have no incoming connection to them have indegree 0 and nodes with indegree 0 can be placed at the start of the ordering
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < indegree.length; i++) {
            if (indegree[i] == 0)
                queue.add(i);
        }
        ArrayList<Integer> result = new ArrayList<>();
        while (!queue.isEmpty()) {
            int node = queue.remove();
            // this nodes is now taken away from graph
            result.add(node);
            for (int adjacentNode : list.get(node)) {
                // adjacentNode<---node ,since node is taken away indegree of adjacentNode will reduce by 1
                indegree[adjacentNode]--;
                // if indegree of adjacentNode becomes 0 then it can be part of ongoing ordering
                if (indegree[adjacentNode] == 0)
                    queue.add(adjacentNode);
            }
        }
        //now if the length of all the nodes in result is equal to numCourses then there is no loop in the graph, else there is a loop and we can not choose all the courses
        //in that case we will have to return an enpty array
        return result.size() == param1 ? result.stream().mapToInt(i -> i).toArray() : new int[0];
    }
}
