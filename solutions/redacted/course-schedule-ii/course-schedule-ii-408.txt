class Solution {

    public int[] findOrder(int param1, int[][] prerequisites) {
        List<List<Integer>> var1 = new ArrayList<>();
        int[] inDegree = new int[param1];
        int[] ret = new int[param1];
        int counter = 0;
        for (int i = 0; i < param1; i++) {
            var1.add(new ArrayList<>());
        }
        for (int[] prereq : prerequisites) {
            var1.get(prereq[1]).add(prereq[0]);
            inDegree[prereq[0]]++;
        }
        Queue<Integer> q = new LinkedList<>();
        for (int i = 0; i < param1; i++) {
            if (inDegree[i] == 0) {
                q.add(i);
                ret[counter] = i;
                counter++;
            }
        }
        while (!q.isEmpty()) {
            int prereq = q.poll();
            for (int var2 : var1.get(prereq)) {
                inDegree[var2]--;
                if (inDegree[var2] == 0) {
                    q.add(var2);
                    ret[counter] = var2;
                    counter++;
                }
            }
        }
        for (int i = 0; i < inDegree.length; i++) {
            if (inDegree[i] != 0) {
                return new int[] {};
            }
        }
        return ret;
    }
}
