class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int erase = 0;
        int i = 0;
        while (i < param1.length - 1) {
            int end = param1[i][1];
            int j = i + 1;
            while (j < param1.length && param1[j][0] < end) {
                j++;
                erase++;
            }
            i = j;
        }
        return erase;
    }
}
