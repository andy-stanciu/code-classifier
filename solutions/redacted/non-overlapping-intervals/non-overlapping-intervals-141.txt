class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int result = 0;
        int[] last = param1[0];
        for (int i = 1; i < param1.length; ++i) {
            int[] currentInterval = param1[i];
            if (last[1] > currentInterval[0]) {
                result++;
                continue;
            }
            last = currentInterval;
        }
        return result;
    }
}
