class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(row -> row[1]));
        int ans = -1;
        int[] prev = param1[0];
        for (int[] interval : param1) {
            if (interval[0] < prev[1])
                ans++;
            else
                prev = interval;
        }
        return ans;
    }
}
