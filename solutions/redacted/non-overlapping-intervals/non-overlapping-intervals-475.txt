class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[1]));
        ArrayList<Integer> ans = new ArrayList<>();
        int maxInt = 1;
        int lastEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= lastEnd) {
                maxInt++;
                lastEnd = param1[i][1];
            }
        }
        return param1.length - maxInt;
    }
}
