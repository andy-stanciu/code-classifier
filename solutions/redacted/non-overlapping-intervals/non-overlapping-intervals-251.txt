class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (i1, i2) -> (i1[1] - i2[1]));
        //for(int i=0;i<intervals.length;i++) System.out.println(Arrays.toString(intervals[i]));
        int overlap = 0;
        int lastin = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[lastin][1] <= param1[i][0]) {
                //System.out.println(Arrays.toString(intervals[lastin]));
                lastin = i;
            } else {
                overlap++;
            }
        }
        return overlap;
    }
}
