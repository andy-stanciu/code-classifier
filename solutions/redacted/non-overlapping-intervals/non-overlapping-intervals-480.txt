class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, (a1, a2) -> a1[1] - a2[1]);
        int ans = 1;
        int end = param1[0][1];
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= end) {
                ans++;
                end = param1[i][1];
            }
        }
        return n - ans;
    }
}
