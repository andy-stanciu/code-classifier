class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int ans = 0;
        int i = 0;
        while (i < param1.length) {
            int end = param1[i][1];
            i++;
            while (i < param1.length && param1[i][0] < end) {
                i++;
                ans++;
            }
        }
        return ans;
    }
}
