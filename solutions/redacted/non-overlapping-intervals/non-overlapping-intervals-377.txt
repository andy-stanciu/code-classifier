class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1, curr = 1, prev = 0, n = param1.length;
        while (curr < n) {
            if (param1[prev][1] <= param1[curr][0]) {
                count++;
                prev = curr;
            }
            curr++;
        }
        return n - count;
    }
}
