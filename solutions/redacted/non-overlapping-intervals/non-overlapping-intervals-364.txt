class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        // Sort by ending time
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int end = Integer.MIN_VALUE;
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (end <= param1[i][0]) {
                end = param1[i][1];
                cnt++;
            }
        }
        // Minimum deletion=n-Maximum number of non-overlapping intervals.
        return n - cnt;
    }
}
