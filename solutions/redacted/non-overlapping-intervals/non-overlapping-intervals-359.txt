class Solution {

    public int method1(int[][] param1) {
        int prev = 0;
        if (param1.length == 1) {
            return 0;
        }
        int count = 0;
        int i = 1;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        while (i < param1.length) {
            if (param1[prev][1] <= param1[i][0]) {
                prev = i;
                i++;
            } else {
                while (i < param1.length && param1[prev][1] > param1[i][0]) {
                    i++;
                    count++;
                }
            }
        }
        return count;
    }
}
