public class Solution {

    public int method1(int[][] param1) {
        // Initialize a variable to count the number of intervals to be removed
        int toBeRemoved = 0;
        Arrays.sort(param1, new Comparator<int[]>() {

            @Override
            public int compare(int[] o1, int[] o2) {
                // Sort the intervals based on their start points
                return o1[0] - o2[0];
            }
        });
        // Initialize the end time of the first interval
        int end = param1[0][1];
        // Iterate through the remaining intervals (starting from the second one)
        for (int i = 1; i < param1.length; i++) {
            int[] interval = param1[i];
            if (interval[0] < end) {
                // If the current interval\'s start time is less than the end time of the last interval,
                // it means there is an overlap between the intervals.
                // Increment the count of intervals to be removed
                toBeRemoved++;
                // Greedily choose the interval with the smaller end time to keep (to minimize overlaps)
                end = Math.min(end, interval[1]);
            } else {
                // If there is no overlap, update the end time to the current interval\'s end time.
                end = interval[1];
            }
        }
        // Return the total number of intervals to be removed
        return toBeRemoved;
    }
}
