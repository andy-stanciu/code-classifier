class Solution {

    public int method1(int[][] arr) {
        Arrays.sort(arr, (a, b) -> Integer.compare(a[1], b[1]));
        int idx = 0;
        int cnt = 1;
        for (int i = 1; i < arr.length; i++) {
            int cur = arr[i][0];
            if (arr[idx][1] <= arr[i][0]) {
                cnt++;
                idx = i;
            }
        }
        return arr.length - cnt;
    }
}
