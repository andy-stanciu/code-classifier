class Solution {

    public int method1(int[][] param1) {
        // Sort the intervals by their end points in ascending order
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        // Initialize the number of intervals to remove and the previous end point
        int count = 0;
        int prev = Integer.MIN_VALUE;
        // Loop through the intervals
        for (int[] interval : param1) {
            // If the current interval overlaps with the previous one, increment the count
            if (interval[0] < prev) {
                count++;
            } else {
                // Otherwise, update the previous end point
                prev = interval[1];
            }
        }
        // Return the number of intervals to remove
        return count;
    }
}
