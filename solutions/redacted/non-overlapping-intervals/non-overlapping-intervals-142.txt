class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        // sort by interval end, quicker a task end more room for other non overllaped ones
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int count = 1, prev = 0;
        for (int i = 1; i < n; i++) {
            // find out if the intervals are non overlapping
            if (param1[i][0] >= param1[prev][1]) {
                // count it
                count++;
                // set the current one as prev to compare with upcoming
                // interval
                prev = i;
            }
        }
        return n - count;
    }
}
