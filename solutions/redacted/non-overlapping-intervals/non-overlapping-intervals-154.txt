class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        Stack<Integer> end_time_stack = new Stack<>();
        end_time_stack.push(param1[0][1]);
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= end_time_stack.peek())
                end_time_stack.push(param1[i][1]);
        }
        return param1.length - end_time_stack.size();
    }
}
