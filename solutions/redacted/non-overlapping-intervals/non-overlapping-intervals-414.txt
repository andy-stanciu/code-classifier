class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[0]));
        int common = param1[0][1];
        int overlap = 0;
        for (int i = 1; i < param1.length; ++i) {
            if (common > param1[i][0]) {
                ++overlap;
                common = Math.min(common, param1[i][1]);
            } else {
                common = param1[i][1];
            }
        }
        return overlap;
    }
}
