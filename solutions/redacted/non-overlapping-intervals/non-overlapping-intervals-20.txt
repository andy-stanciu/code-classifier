class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int prev_end = param1[0][1];
        for (int i = 1; i < n; i++) {
            if (param1[i][0] < prev_end) {
                count++;
            } else {
                prev_end = param1[i][1];
            }
        }
        return count;
    }
}
