class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 1) {
            return 0;
        }
        // print2D(intervals);
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[1]));
        // print2D(intervals);
        int count = 0, j = 1;
        for (int i = 1; i < param1.length; i++) {
            int small = param1[j - 1][1];
            int high = param1[i][0];
            System.out.println(high + " : " + small);
            if (high < small) {
                count++;
            } else if (high >= small) {
                j = i + 1;
            }
        }
        return count;
    }
}
