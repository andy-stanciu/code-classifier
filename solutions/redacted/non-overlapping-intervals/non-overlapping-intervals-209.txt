class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (o1, o2) -> {
            return o1[1] - o2[1];
        });
        Queue<Integer> pq = new PriorityQueue<>(Comparator.reverseOrder());
        for (int i = 0; i < param1.length; i++) {
            if (pq.isEmpty() || pq.peek() <= param1[i][0]) {
                pq.add(param1[i][1]);
            }
        }
        return param1.length - pq.size();
    }
}
