class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        Stack<Integer> stack = new Stack<>();
        int len = param1.length;
        for (int i = 0; i < len; i++) {
            if (stack.empty() || param1[i][0] >= stack.peek()) {
                stack.push(param1[i][1]);
            }
        }
        return len - stack.size();
    }
}
