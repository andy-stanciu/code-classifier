class Solution {

    public int method1(int[][] param1) {
        int var1 = 0;
        int previous = 0;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        ///int current=1;
        for (int current = 1; current < param1.length; current++) {
            if (param1[previous][1] > param1[current][0]) {
                var1++;
                continue;
            }
            previous = current;
        }
        return var1;
    }
}
