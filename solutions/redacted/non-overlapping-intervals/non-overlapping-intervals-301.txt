class Solution {

    public int method1(int[][] param1) {
        int len = param1.length;
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        int count = 0;
        int curEnd = param1[0][1];
        for (int i = 1; i < len; i++) {
            if (param1[i][0] < curEnd)
                count++;
            else
                curEnd = param1[i][1];
        }
        return count;
    }
}
