class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[1]));
        int count = 1;
        int endtime = param1[0][1];
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= endtime) {
                count++;
                endtime = param1[i][1];
            }
        }
        return param1.length - count;
    }
}
