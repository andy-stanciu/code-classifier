class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int[] pre = param1[0];
        int res = 0;
        for (int i = 1; i < param1.length; i++) {
            int[] cur = param1[i];
            if (cur[0] < pre[1]) {
                //overlapping
                res++;
                if (pre[1] <= cur[1])
                    continue;
            }
            pre = cur;
        }
        return res;
    }
}
