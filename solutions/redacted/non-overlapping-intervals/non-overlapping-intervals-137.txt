class Solution {

    public int method1(int[][] param1) {
        if (param1.length <= 1)
            return 0;
        Arrays.sort(param1, new Comparator<int[]>() {

            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        int count = 0;
        int endInterval = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (endInterval > param1[i][0]) {
                count++;
                endInterval = Math.min(param1[i][1], endInterval);
            } else {
                endInterval = param1[i][1];
            }
        }
        return count;
    }
}
