class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0) {
            // No intervals to remove
            return 0;
        }
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < end) {
                // Overlapping intervals, remove the one with the larger end point
                count++;
            } else {
                // Non-overlapping interval, update the end point
                end = param1[i][1];
            }
        }
        return count;
    }
}
