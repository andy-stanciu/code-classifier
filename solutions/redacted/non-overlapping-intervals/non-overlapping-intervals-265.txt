class Solution {

    public int method1(int[][] param1) {
        if (param1 == null || param1.length == 0) {
            return 0;
        }
        // Sort intervals based on end times
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        // Number of non-overlapping intervals
        int count = 1;
        int prevEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= prevEnd) {
                // Non-overlapping interval found
                count++;
                prevEnd = param1[i][1];
            }
            // If the interval overlaps with the previous interval, skip it
        }
        // Number of intervals to remove = total intervals - non-overlapping intervals
        return param1.length - count;
    }
}
