class Solution {

    public int method1(int[][] param1) {
        int res = 0;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= param1[i - 1][1])
                continue;
            if (param1[i][1] > param1[i - 1][1]) {
                param1[i] = param1[i - 1];
            }
            res++;
        }
        return res;
    }
}
