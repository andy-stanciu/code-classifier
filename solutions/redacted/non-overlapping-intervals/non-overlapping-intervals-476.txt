class Solution {

    public int method1(int[][] param1) {
        if (param1.length <= 1)
            return 0;
        List<int[]> list = new ArrayList<>();
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        int[] start = param1[0];
        list.add(start);
        int count = 0;
        for (int i = 1; i < param1.length; i++) {
            if (start[1] > param1[i][0]) {
                count++;
            } else {
                list.add(param1[i]);
                start = param1[i];
            }
        }
        return count;
    }
}
