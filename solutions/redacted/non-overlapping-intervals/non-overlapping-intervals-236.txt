class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] == b[1] ? a[0] - b[0] : a[1] - b[1]);
        int[] prev = param1[0];
        int count = 0;
        for (int i = 1; i < param1.length; i++) {
            int[] curr = param1[i];
            if (prev[0] < curr[1] && prev[1] > curr[0])
                count++;
            else
                prev = curr;
        }
        return count;
    }
}
