class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        for (int i = 0; i < param1.length - 1; i++) {
            if (param1[i + 1][0] < param1[i][1]) {
                count++;
                param1[i + 1][0] = Math.min(param1[i][0], param1[i + 1][0]);
                param1[i + 1][1] = Math.min(param1[i][1], param1[i + 1][1]);
            }
        }
        return count;
    }
}
