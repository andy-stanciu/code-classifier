class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a, int[] b) {
                return a[1] - b[1];
            }
        });
        int prev = 0;
        int count = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < param1[prev][1]) {
                count++;
            } else {
                prev = i;
            }
        }
        return count;
    }
}
