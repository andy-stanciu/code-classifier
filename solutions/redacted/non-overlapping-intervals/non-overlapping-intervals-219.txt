class Solution {

    public int method1(int[][] param1) {
        if (param1 == null || param1.length == 0) {
            return 0;
        }
        // Sort intervals based on end times
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int removeCount = 0;
        int end = param1[0][1];
        // Iterate through sorted intervals
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < end) {
                // Overlapping interval, remove the one with larger end time
                removeCount++;
            } else {
                // Non-overlapping interval, update end time
                end = param1[i][1];
            }
        }
        return removeCount;
    }
}
