class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 1)
            return 0;
        int n = param1.length;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1;
        int lastEndTime = param1[0][1];
        for (int i = 0; i < n; i++) {
            if (param1[i][0] >= lastEndTime) {
                count += 1;
                lastEndTime = param1[i][1];
            }
        }
        return n - count;
    }
}
