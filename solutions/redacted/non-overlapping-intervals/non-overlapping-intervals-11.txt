public class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int res = 0;
        int prevEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (prevEnd > param1[i][0]) {
                res++;
            } else {
                prevEnd = param1[i][1];
            }
        }
        return res;
    }
}
