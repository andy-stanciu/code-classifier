class Solution {

    public int method1(int[][] param1) {
        int end = Integer.MIN_VALUE, count = 0;
        Arrays.sort(param1, (x, y) -> x[1] - y[1]);
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= end)
                end = param1[i][1];
            else
                count++;
        }
        return count;
    }
}
