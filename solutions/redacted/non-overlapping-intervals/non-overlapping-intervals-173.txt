public class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (x, y) -> x[1] - y[1]);
        int var1 = 0;
        int endIntervalValue = param1[0][1];
        int index = 1;
        while (index < param1.length) {
            if (endIntervalValue > param1[index][0]) {
                ++var1;
            } else {
                endIntervalValue = param1[index][1];
            }
            ++index;
        }
        return var1;
    }
}
