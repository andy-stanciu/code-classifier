class Solution {

    public int method1(int[][] param1) {
        int res = 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int prev_end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (prev_end > param1[i][0]) {
                res++;
            } else {
                prev_end = param1[i][1];
            }
        }
        return res;
    }
}
