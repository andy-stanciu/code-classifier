class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingDouble(o -> o[1]));
        int count = 0;
        int x = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (x > param1[i][0]) {
                count++;
            } else {
                x = param1[i][1];
            }
        }
        return count;
    }
}
