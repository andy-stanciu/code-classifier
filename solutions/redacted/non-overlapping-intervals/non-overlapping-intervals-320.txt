class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length;
        int pos = 0;
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= param1[pos][1]) {
                pos = i;
                count++;
            }
        }
        return n - count;
    }
}
