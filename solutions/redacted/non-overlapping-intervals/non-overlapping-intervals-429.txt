class Solution {

    public int method1(int[][] param1) {
        int overlaping = 0;
        int len = param1.length - 1;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int lastend = param1[0][1];
        for (int i = 0; i <= len; i++) {
            if (param1[i][0] >= lastend) {
                overlaping++;
                lastend = param1[i][1];
            }
        }
        System.gc();
        return len - overlaping;
    }
}
