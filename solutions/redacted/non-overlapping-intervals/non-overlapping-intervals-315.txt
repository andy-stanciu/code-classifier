class Solution {

    public int method1(int[][] param1) {
        // Sort the array first
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int prev = Integer.MIN_VALUE;
        int count = 0;
        for (int[] interval : param1) {
            int start = interval[0];
            int end = interval[1];
            if (start >= prev) {
                // No overlap
                prev = end;
            } else {
                // Overlap found
                if (end > prev) {
                    // Discard the current limit
                    count += 1;
                } else {
                    // Discard the previous limit
                    prev = end;
                    count += 1;
                }
            }
        }
        return count;
    }
}
