class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int ans = 0;
        int start = 0, end = 1;
        while (start < param1.length - 1 && end < param1.length) {
            if (param1[start][1] > param1[end][0]) {
                ans++;
                if (param1[start][1] > param1[end][1]) {
                    start = end;
                    end++;
                    continue;
                }
                if (param1[start][1] < param1[end][1]) {
                    end++;
                    continue;
                }
            }
            start = end;
            end++;
        }
        return ans;
    }
}
