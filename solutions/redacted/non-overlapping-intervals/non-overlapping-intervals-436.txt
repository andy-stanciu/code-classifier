class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (x, y) -> x[1] - y[1]);
        int n = param1.length;
        if (n <= 1)
            return 0;
        int end = param1[0][1];
        int ans = 0;
        for (int i = 1; i < n; i++) {
            if (end > param1[i][0]) {
                ans++;
            } else {
                end = param1[i][1];
            }
        }
        return ans;
    }
}
