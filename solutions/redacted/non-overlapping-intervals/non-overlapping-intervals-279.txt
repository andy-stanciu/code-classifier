class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int n = param1.length;
        int ans = 0, prev = 0;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] < param1[prev][1]) {
                ans++;
            } else {
                prev = i;
            }
        }
        return ans;
    }
}
