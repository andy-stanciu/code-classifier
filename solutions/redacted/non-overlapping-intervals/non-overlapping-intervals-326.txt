class Solution {

    public int method1(int[][] param1) {
        int ans = 0, k = Integer.MIN_VALUE;
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        for (int i = 0; i < param1.length; i++) {
            int x = param1[i][0];
            int y = param1[i][1];
            if (x >= k) {
                k = y;
            } else {
                ans++;
            }
        }
        return ans;
    }
}
