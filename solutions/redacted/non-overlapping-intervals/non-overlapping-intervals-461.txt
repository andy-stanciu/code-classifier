class Solution {

    public int method1(int[][] I) {
        Arrays.sort(I, (a, b) -> a[0] - b[0]);
        int count = 0;
        for (int i = 0; i < I.length - 1; i++) {
            if (I[i][1] > I[i + 1][0]) {
                int min = Math.min(I[i][1], I[i + 1][1]);
                // THI SIMULATES swapping with the overlapping interval with larger end value and place in i+1
                // only this need to be updated
                I[i + 1][1] = min;
                count++;
            }
        }
        return count;
    }
}
