class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int endidx = param1[0][1];
        int n = param1.length;
        int count = 1;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= endidx) {
                count++;
                endidx = param1[i][1];
            }
        }
        return n - count;
    }
}
