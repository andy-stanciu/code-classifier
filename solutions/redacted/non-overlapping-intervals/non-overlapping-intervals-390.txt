class Solution {

    public int method1(int[][] param1) {
        // for counting the overlapping intervals
        int count = 0;
        // for previous selcting interval index
        int prev = 0;
        // sorting the given array on column 1 to select as many intervals
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= param1[prev][1]) {
                // updating previous selecting interval
                prev = i;
            } else {
                count++;
            }
        }
        return count;
    }
}
