class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (int[] a, int[] b) -> a[0] - b[0]);
        int overlap_count = 0;
        for (int i = 0; i < param1.length - 1; i++) {
            if (param1[i][1] > param1[i + 1][0]) {
                param1[i + 1][1] = Math.min(param1[i][1], param1[i + 1][1]);
                overlap_count++;
            }
        }
        return (overlap_count);
    }
}
