class Solution {

    public int method1(int[][] param1) {
        int c = 0;
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int s = param1[0][1];
        int e = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < e) {
                e = Math.min(e, param1[i][1]);
                c += 1;
            } else {
                e = param1[i][1];
            }
        }
        return c;
    }
}
