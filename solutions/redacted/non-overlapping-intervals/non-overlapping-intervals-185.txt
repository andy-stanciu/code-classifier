class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 1) {
            return 0;
        }
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int count = 0;
        int i = 0;
        int j = 1;
        while (j < param1.length) {
            if (param1[i][1] <= param1[j][0]) {
                //no overlapping->no deletions // __________ ___________
                i = j;
                j++;
            } else if (param1[i][1] <= param1[j][1]) {
                //overlapping  //_____________
                //       _____________
                j++;
                count++;
            } else if (param1[i][1] > param1[j][1]) {
                //overlapping // ___________________
                //   ___________
                i = j;
                j++;
                count++;
            }
        }
        return count;
    }
}
