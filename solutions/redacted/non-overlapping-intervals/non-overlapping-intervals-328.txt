class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int prev = param1[0][0];
        int ans = 0;
        for (int i = 0; i < param1.length; i++) {
            if (prev <= param1[i][0]) {
                prev = param1[i][1];
            } else {
                ans += 1;
            }
        }
        return ans;
    }
}
