class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int count = 0;
        int prev = Integer.MIN_VALUE;
        for (int[] interval : param1) {
            int start = interval[0];
            int end = interval[1];
            if (start >= prev) {
                // We can take this interval
                prev = end;
            } else {
                // We will discard this interval
                count += 1;
            }
        }
        return count;
    }
}
