class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        if (n == 0 || n == 1) {
            return 0;
        }
        int var1 = 1;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int last_end = param1[0][1];
        for (int i = 1; i < n; i++) {
            if (param1[i][0] >= last_end) {
                var1++;
                last_end = param1[i][1];
            }
        }
        return n - var1;
    }
}
