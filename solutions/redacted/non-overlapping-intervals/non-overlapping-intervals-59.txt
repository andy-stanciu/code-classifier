class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> (a[0] - b[0]));
        int count = 0;
        //we start with the first interval
        int preEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= preEnd) {
                // if no overlap(start value of current >= end value of previous one), we just update last interval to current one
                //because we know current one end value must greater than previous one
                preEnd = param1[i][1];
            } else {
                // if overlap, we delete the one with larger end value
                count++;
                preEnd = Math.min(preEnd, param1[i][1]);
            }
        }
        return count;
    }
}
