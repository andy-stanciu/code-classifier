class Solution {

    public int method1(int[][] param1) {
        // Sort intervals by their end times
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int n = param1.length;
        // This will count the number of intervals to remove
        int count = 0;
        // Initial end time set to the end of the first interval
        int endtime = param1[0][1];
        // Start from the second interval
        for (int i = 1; i < n; i++) {
            int start = param1[i][0];
            if (start < endtime) {
                // Overlapping interval found, increment count
                count++;
            } else {
                // No overlap, update endtime to current interval\'s end time
                endtime = param1[i][1];
            }
        }
        return count;
    }
}
