class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int count = 0;
        int m = param1.length;
        int n = param1[0].length;
        int prev = param1[0][1];
        for (int i = 1; i < m; i++) {
            if (param1[i][0] < prev)
                count++;
            else
                prev = param1[i][1];
        }
        return count;
    }
}
