class Solution {

    private int canAttendMeet(int idx, int prevEnd, int totalMeets, int[][] param1) {
        if (idx == totalMeets)
            return 0;
        int currStart = param1[idx][0];
        int currEnd = param1[idx][1];
        if (currStart >= prevEnd) {
            return 1 + canAttendMeet(idx + 1, currEnd, totalMeets, param1);
        } else {
            return canAttendMeet(idx + 1, prevEnd, totalMeets, param1);
        }
    }

    public int method1(int[][] param1) {
        int len = param1.length;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        // sorting on basis of end time
        return len - (1 + canAttendMeet(1, param1[0][1], len, param1));
    }
}
