class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        int cnt = 0;
        Arrays.sort(param1, (int[] a, int[] b) -> a[1] - b[1]);
        int i = 0;
        while (i < n) {
            int b = param1[i][1];
            while (i + 1 < n && b > param1[i + 1][0]) {
                i++;
                cnt++;
            }
            i++;
        }
        return cnt;
    }
}
