class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (in1, in2) -> Integer.compare(in1[1], in2[1]));
        int cnt = 0, start;
        for (int i = param1.length - 2; i >= 0; i--) {
            start = param1[i + 1][0];
            while (i >= 0 && param1[i][1] > start) {
                cnt++;
                start = Math.max(start, param1[i][0]);
                i--;
            }
        }
        return cnt;
    }
}
