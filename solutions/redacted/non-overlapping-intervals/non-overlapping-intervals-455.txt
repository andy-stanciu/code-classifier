class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> (a[1] - b[1]));
        int cnt = 0;
        System.out.println(Arrays.deepToString(param1));
        int curr = 0;
        int start = 0;
        int end = 1;
        for (int i = 1; i < param1.length; i++) {
            //check for merge
            if (param1[i][start] < param1[curr][end]) {
                cnt++;
            } else
                curr = i;
        }
        return cnt;
    }
}
