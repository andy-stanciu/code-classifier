class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int i = 0;
        while (i < param1.length) {
            int j = i + 1;
            while (j < param1.length && param1[j][0] < param1[i][1]) {
                j++;
                count++;
            }
            i = j;
        }
        return count;
    }
}
