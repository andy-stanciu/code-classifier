class Solution {

    private boolean checkOverlap(int[] intervalOne, int[] intervalTwo) {
        return (intervalTwo[0] < intervalOne[1]);
    }

    public int method1(int[][] param1) {
        int var1 = 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int lastIntervalIndex = 0;
        int currentIntervalIndex = 1;
        while (currentIntervalIndex < param1.length) {
            if (checkOverlap(param1[lastIntervalIndex], param1[currentIntervalIndex])) {
                var1++;
            } else {
                lastIntervalIndex = currentIntervalIndex;
            }
            currentIntervalIndex++;
        }
        return var1;
    }
}
