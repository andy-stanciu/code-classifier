class Solution {

    public int method1(int[][] param1) {
        // sort intervals on basis of end
        /*
[[1,2],[2,3],[3,4],[1,3]]
[1,2], [2,3], [1,3], [3,4] // sorted
*/
        // sort
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int end = param1[0][1];
        // count the valid intervals
        int count = 1;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= end) {
                end = param1[i][1];
                count++;
            }
        }
        return param1.length - count;
    }
}
