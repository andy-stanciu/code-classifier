class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length;
        int index = 0, count = 0;
        for (int i = 1; i < n; i++) {
            if (param1[i][0] < param1[index][1])
                count++;
            else
                index = i;
        }
        return count;
    }
}
