class Solution {

    public int method1(int[][] param1) {
        int len = param1.length;
        int i = 0, j = 1;
        int count = 0;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        while (j < len) {
            var first = param1[i];
            var second = param1[j];
            if (isOverlap(first, second)) {
                count++;
                j++;
            } else {
                i = j;
                j++;
            }
        }
        return count;
    }

    private boolean isOverlap(int[] first, int[] second) {
        return first[0] < second[1] && second[0] < first[1];
    }
}
