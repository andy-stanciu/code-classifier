class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int len = param1.length;
        int ans = 0;
        int index = 1;
        int start = param1[0][0];
        int end = param1[0][1];
        while (index < len) {
            int[] next = param1[index];
            // If next slot starts before current
            if (next[0] < end) {
                end = Math.min(next[1], end);
                ans++;
            } else // Otherwise update current
            {
                start = next[0];
                end = next[1];
            }
            index++;
        }
        return ans;
    }
}
