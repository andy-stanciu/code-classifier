class Solution {

    public static void main(String[] args) {
    }

    public static int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int canAttend = 1;
        int pre = 0;
        for (int i = 1; i < param1.length; i++) {
            if (param1[pre][1] <= param1[i][0]) {
                canAttend++;
                pre = i;
            }
        }
        return param1.length - canAttend;
    }
}
