class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        if (n == 0) {
            return 0;
        }
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int prev = 0;
        int count = 0;
        for (int i = 1; i < n; i++) {
            if (param1[prev][1] > param1[i][0]) {
                if (param1[prev][1] > param1[i][1]) {
                    prev = i;
                }
                count++;
            } else
                prev = i;
        }
        return count;
    }
}
