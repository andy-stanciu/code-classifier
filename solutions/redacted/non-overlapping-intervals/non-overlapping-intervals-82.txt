class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int last = param1[0][1], c = 1;
        for (int i = 1; i < n; i++) {
            if (last <= param1[i][0]) {
                c++;
                last = param1[i][1];
            }
        }
        return n - c;
    }
}
