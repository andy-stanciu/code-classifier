class Solution {

    public int method1(int[][] param1) {
        int count = 0;
        if (param1.length == 0 || param1[0].length == 0)
            return count;
        //sort based on end val
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int prevEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < prevEnd) {
                //if current start is less than prev end it means it overlaps
                // and need to remove then increament the count
                count++;
            } else {
                //if it does not overlap update the prevEnd with current interval end
                //intervals[i][1] -- end n intervals[i][0] -- start
                prevEnd = param1[i][1];
            }
        }
        return count;
    }
}
