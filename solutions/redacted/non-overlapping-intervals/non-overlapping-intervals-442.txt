class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (i1, i2) -> Integer.compare(i1[1], i2[1]));
        int count = 0, intv = Integer.MIN_VALUE;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= intv)
                intv = param1[i][1];
            else
                count++;
        }
        return count;
    }
}
