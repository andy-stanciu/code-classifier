class Solution {

    public int method1(int[][] pairs) {
        Arrays.sort(pairs, (a, b) -> Integer.compare(a[1], b[1]));
        int c = Integer.MIN_VALUE;
        int ans = 0;
        for (int[] x : pairs) {
            if (c <= x[0]) {
                c = x[1];
                ans++;
            }
        }
        return pairs.length - ans;
    }
}
