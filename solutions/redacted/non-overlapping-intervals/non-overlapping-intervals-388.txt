class Solution {

    // Greedy Approach
    public int method1(int[][] param1) {
        // sort on basis of end time
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length;
        int intervalCount = 0;
        Pair prev = new Pair(param1[0][0], param1[0][1]);
        for (int i = 1; i < n; i++) {
            Pair curr = new Pair(param1[i][0], param1[i][1]);
            if (prev.end > curr.start) {
                intervalCount++;
            } else {
                prev = curr;
            }
        }
        return intervalCount;
    }
}

class Pair {

    int start, end;

    public Pair(int s, int e) {
        this.start = s;
        this.end = e;
    }
}
// TC: O(nlogn), SC: O(1)
