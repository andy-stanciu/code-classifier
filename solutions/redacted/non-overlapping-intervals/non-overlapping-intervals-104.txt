class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, java.util.Comparator.comparingDouble(a -> a[1]));
        int r = param1.length;
        int i = 1, n = 0, f = param1[0][1];
        while (i < r) {
            if (param1[i][0] < f)
                n++;
            else {
                f = param1[i][1];
            }
            i++;
        }
        return n;
    }
}
