class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, new Comparator<int[]>() {

            public int compare(int[] a1, int[] a2) {
                return a1[1] - a2[1];
            }
        });
        int count = 0;
        int prev = Integer.MIN_VALUE;
        for (int i = 0; i < param1.length; i++) {
            if (param1[i][0] >= prev) {
                count++;
                prev = param1[i][1];
            }
        }
        return param1.length - count;
    }
}
