class Solution {

    public int method1(int[][] v) {
        Arrays.sort(v, (a, b) -> Integer.compare(a[1], b[1]));
        int sec = v[0][1], out = 0;
        for (int i = 1; i < v.length; i++) {
            if (v[i][0] >= sec)
                sec = v[i][1];
            else
                out++;
        }
        return out;
    }
}
