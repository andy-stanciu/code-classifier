class Solution {

    public int method1(int[][] param1) {
        param1 = Arrays.stream(param1).sorted(Comparator.comparingInt(row -> row[1])).toArray(int[][]::new);
        int st = param1[0][0];
        int end = param1[0][1], n = 0;
        for (int i = 0; i < param1.length; i++) {
            if (end > param1[i][0]) {
                n++;
            } else {
                if (end == param1[i][0]) {
                    end = param1[i][1];
                } else if (end < param1[i][0]) {
                    st = param1[i][0];
                    end = param1[i][1];
                }
            }
        }
        return n - 1;
    }
}
