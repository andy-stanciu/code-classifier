class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (x, y) -> x[1] - y[1]);
        int res = 0, start = param1[0][0], end = Integer.MIN_VALUE;
        for (int[] arr : param1) {
            int s = arr[0], e = arr[1];
            if (s >= end) {
                end = e;
            } else
                res++;
        }
        return res;
    }
}
