class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0)
            return 0;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int c = 0;
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < end) {
                c++;
            } else {
                end = param1[i][1];
            }
        }
        return c;
    }
}
