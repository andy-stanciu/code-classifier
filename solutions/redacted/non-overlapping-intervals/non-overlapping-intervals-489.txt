class Solution {

    public int method1(int[][] param1) {
        int minCount = 0;
        // Empty case
        if (param1.length == 0)
            return minCount;
        // Sort intervals first - O(nlogn)
        Arrays.sort(param1, (a, b) -> {
            return a[0] - b[0];
        });
        int[] compareTo = param1[0];
        // O(n)
        for (int i = 1; i < param1.length; i++) {
            //Check for overlap
            if (compareTo[1] > param1[i][0]) {
                minCount++;
                //Overlap found, keep the interval with the smallest end value
                if (compareTo[1] > param1[i][1]) {
                    compareTo = param1[i];
                }
            } else {
                //Set compareTo as the next value, we have found all the overlap for this one
                compareTo = param1[i];
            }
        }
        return minCount;
    }
}
