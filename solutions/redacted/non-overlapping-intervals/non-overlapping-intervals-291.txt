class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (i1, i2) -> {
            int diff = Integer.compare(i1[1], i2[1]);
            // if(diff == 0){
            //     return Integer.compare(i1[0], i2[0]);
            // }
            return diff;
        });
        int cnt = 0;
        for (int i = 0; i < param1.length; ) {
            int[] prev = param1[i];
            int j = i + 1;
            for (; j < param1.length; ) {
                if (prev[1] > param1[j][0]) {
                    cnt++;
                    j++;
                } else {
                    break;
                }
            }
            i = j;
        }
        return cnt;
    }
}
