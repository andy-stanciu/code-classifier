class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int lastEnd = Integer.MIN_VALUE, ans = 0;
        for (int i = 0; i < param1.length; i++) {
            int curStart = param1[i][0];
            int curEnd = param1[i][1];
            if (curStart >= lastEnd) {
                lastEnd = curEnd;
            } else {
                ans++;
            }
        }
        return ans;
    }
}
