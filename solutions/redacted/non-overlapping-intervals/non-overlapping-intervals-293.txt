class Solution {

    public int method1(int[][] arr) {
        //sort the array with respect to column 1 of the array
        sortBycol(arr, 1);
        //first interval counted and start loop with check from 2nd interval
        int var1 = 1;
        int start = arr[0][0];
        int end = arr[0][1];
        for (int i = 1; i < arr.length; i++) {
            if (arr[i][0] >= end) {
                var1++;
                end = arr[i][1];
            }
        }
        //total intervals minus completed intervals
        return arr.length - var1;
    }

    public static void sortBycol(int[][] arr, int col) {
        Arrays.sort(arr, new Comparator<int[]>() {

            @Override
            public int compare(int[] o1, int[] o2) {
                // TODO Auto-generated method stub
                return o1[col] - o2[col];
            }
        });
    }
}
