class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (int[] x, int[] y) -> x[1] - y[1]);
        int total = param1.length;
        int lastTime = Integer.MIN_VALUE;
        int count = 0;
        for (int[] interval : param1) {
            if (interval[0] >= lastTime) {
                count++;
                lastTime = interval[1];
            }
        }
        return total - count;
    }
}
