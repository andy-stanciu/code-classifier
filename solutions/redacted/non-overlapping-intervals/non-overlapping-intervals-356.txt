class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int current = 1;
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] >= param1[0][1]) {
                param1[0][1] = param1[i][1];
                current++;
            }
        }
        return param1.length - current;
    }
}
