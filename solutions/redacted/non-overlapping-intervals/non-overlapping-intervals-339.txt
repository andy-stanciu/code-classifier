class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, Comparator.comparingInt(o -> o[0]));
        int count = 0, l = 0, r = 1;
        while (r != param1.length) {
            if (param1[r][0] < param1[l][1]) {
                ++count;
                if (param1[r][1] > param1[l][1]) {
                    ++r;
                    continue;
                }
            }
            l = r++;
        }
        return count;
    }
}
