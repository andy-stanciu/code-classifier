class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0) {
            return 0;
        }
        // Sort the intervals based on their end times
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        // Number of non-overlapping intervals
        int count = 1;
        // End time of the most recent interval
        int end = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            // If the current interval overlaps with the most recent interval, we need to remove it
            if (param1[i][0] >= end) {
                count++;
                end = param1[i][1];
            }
        }
        // Calculate the number of intervals to remove
        int toRemove = param1.length - count;
        return toRemove;
    }
}
