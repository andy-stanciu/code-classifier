class Solution {

    public int method1(int[][] param1) {
        //sort according to difference because if you start with example [2,10] and your following arrays is [2,3] and [4,5], you might end up removing more than you need
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        //last number of array
        int end = param1[0][1];
        int count = param1.length - 1;
        for (int i = 1; i < param1.length; i++) {
            //as long as start is higher or equal to end means no overlapping
            if (param1[i][0] >= end) {
                end = param1[i][1];
                count--;
            }
        }
        return count;
    }
}
