class Solution {

    public int method1(int[][] param1) {
        if (param1.length <= 1)
            return 0;
        //sort by end time
        Arrays.sort(param1, Comparator.comparingInt(a -> a[1]));
        int c = 0;
        int curr_end = param1[0][1];
        for (int[] next : param1) {
            if (curr_end <= next[0]) {
                //overlapping can be done
                curr_end = Math.max(curr_end, next[1]);
            } else {
                //since overlaping cant be done so we need to remove this interval
                c++;
            }
        }
        //inside for loop we did c++ for i =0 which was not needed so subtracted it at the end
        return c - 1;
    }
}
