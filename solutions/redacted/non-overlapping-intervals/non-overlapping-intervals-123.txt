class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[0] - b[0]);
        int count = 0;
        int last = param1[0][1];
        for (int i = 1; i < param1.length; ++i) {
            int[] interval = param1[i];
            if (interval[0] < last) {
                count++;
                last = Math.min(last, interval[1]);
            } else
                last = interval[1];
        }
        return count;
    }
}
