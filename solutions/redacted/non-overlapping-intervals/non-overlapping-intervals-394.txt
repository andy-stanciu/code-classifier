class Solution {

    public int method1(int[][] param1) {
        // sort by end date
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int n = param1.length, ans = 0, last = param1[0][1];
        for (int i = 1; i < n; i++) {
            if (// remove current
            param1[i][0] < last)
                // remove current
                ans++;
            else
                // take current
                last = param1[i][1];
        }
        return ans;
    }
}
