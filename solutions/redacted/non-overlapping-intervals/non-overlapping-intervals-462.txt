class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (o1, o2) -> o1[1] - o2[1]);
        int r = param1.length;
        int cnt = 0, i = 1;
        int[] start = param1[0];
        while (i < r) {
            if (param1[i][0] < start[1]) {
                cnt++;
            } else {
                start = param1[i];
            }
            i++;
        }
        return cnt;
    }
}
