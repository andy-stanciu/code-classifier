class Solution {

    public int method1(int[][] param1) {
        int len = param1.length;
        ArrayList<Class1> list = new ArrayList<>();
        for (int i = 0; i < len; i++) {
            list.add(new Class1(param1[i][0], param1[i][1]));
        }
        Collections.sort(list);
        int prevEnd = list.get(0).end;
        int count = 1;
        for (int i = 1; i < list.size(); i++) {
            int currStart = list.get(i).start;
            if (currStart >= prevEnd) {
                count++;
                prevEnd = list.get(i).end;
            }
        }
        return len - count;
    }

    class Class1 implements Comparable<Class1> {

        int start;

        int end;

        public Class1(int start, int end) {
            this.start = start;
            this.end = end;
        }

        @Override
        public int compareTo(Class1 o) {
            return this.end - o.end;
        }
    }
}
