class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (x, y) -> x[0] == y[0] ? x[1] - y[1] : x[0] - y[0]);
        int i = 1;
        int n = param1.length;
        int s = param1[0][0], e = param1[0][1];
        int cnt = 0;
        while (i < n) {
            if (e > param1[i][0]) {
                e = Math.min(e, param1[i][1]);
                cnt++;
            } else
                e = param1[i][1];
            i++;
        }
        return cnt;
    }
}
