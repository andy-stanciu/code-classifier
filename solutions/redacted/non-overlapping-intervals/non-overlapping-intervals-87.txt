class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int end = param1[0][1];
        int i = 1, count = 0;
        while (i < param1.length) {
            if (param1[i][0] < end) {
                count++;
            } else {
                end = param1[i][1];
            }
            i++;
        }
        return count;
    }
}
