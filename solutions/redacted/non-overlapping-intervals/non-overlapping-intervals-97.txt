class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 1;
        int end = param1[0][1];
        for (int i = 1, len = param1.length; i < len; i++) {
            if (param1[i][0] >= end) {
                count++;
                end = param1[i][1];
            }
        }
        return param1.length - count;
    }
}
