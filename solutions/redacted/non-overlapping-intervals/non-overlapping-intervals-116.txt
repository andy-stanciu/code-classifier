class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int count = 0;
        int prev = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            int[] interval = param1[i];
            if (interval[0] < prev) {
                count++;
            } else {
                prev = interval[1];
            }
        }
        return count;
    }
}
