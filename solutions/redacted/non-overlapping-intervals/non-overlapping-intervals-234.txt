class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int ans = 0;
        int prevend = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            int start = param1[i][0];
            int end = param1[i][1];
            if (prevend <= start) {
                prevend = end;
            } else {
                ans += 1;
                prevend = Math.min(end, prevend);
            }
        }
        return ans;
    }
}
