class Solution {

    public int method1(int[][] param1) {
        if (param1.length <= 1)
            return 0;
        // Sort the intervals based on the start time
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int end = param1[0][1];
        int count = 0;
        for (int i = 1; i < param1.length; i++) {
            // If overlap found
            if (param1[i][0] < end) {
                count++;
                // Update end to the smaller end time
                end = Math.min(end, param1[i][1]);
            } else {
                end = param1[i][1];
            }
        }
        return count;
    }
}
