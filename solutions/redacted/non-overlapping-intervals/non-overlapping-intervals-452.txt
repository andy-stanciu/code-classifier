class Solution {

    class IntervalComparator implements Comparator<int[]> {

        public int compare(int[] intervalA, int[] intervalB) {
            return intervalA[0] - intervalB[0];
        }
    }

    public int method1(int[][] param1) {
        // sort intervals by the first value
        Arrays.sort(param1, new IntervalComparator());
        int removals = 0;
        for (int i = 1; i < param1.length; i++) {
            // if there is an overlap
            if (param1[i - 1][1] > param1[i][0]) {
                // prefer the interval that ends sooner
                // turn the next one into the smaller one, so that we can compare
                // the next interval to the smaller one
                param1[i][1] = Math.min(param1[i][1], param1[i - 1][1]);
                removals += 1;
            }
        }
        return removals;
    }
}
