class Solution {

    public int method1(int[][] param1) {
        if (param1.length == 0) {
            return 0;
        }
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int c = 0;
        int lastEnd = param1[0][1];
        for (int i = 1; i < param1.length; i++) {
            if (param1[i][0] < lastEnd) {
                c++;
            } else {
                lastEnd = param1[i][1];
            }
        }
        return c;
    }
}
