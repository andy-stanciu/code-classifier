class Solution {

    public int method1(int[][] param1) {
        int n = param1.length;
        Arrays.sort(param1, (a, b) -> Integer.compare(a[1], b[1]));
        int prev = 0;
        int c = 1;
        for (int i = 0; i < n; i++) {
            if (param1[i][0] >= param1[prev][1]) {
                prev = i;
                c++;
            }
        }
        return n - c;
    }
}
