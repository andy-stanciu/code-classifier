class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> a[1] - b[1]);
        int ans = -1;
        int[] previous = param1[0];
        for (int[] interval : param1) {
            if (previous[1] > interval[0]) {
                ans++;
            } else {
                previous = interval;
            }
        }
        return ans;
    }
}
