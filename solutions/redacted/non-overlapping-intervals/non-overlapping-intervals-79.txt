class Solution {

    public int method1(int[][] param1) {
        Arrays.sort(param1, (a, b) -> Integer.compare(a[0], b[0]));
        int prevStart = -1;
        int prevEnd = -1;
        boolean flag = false;
        int removeCount = 0;
        for (int[] interval : param1) {
            if (!flag) {
                prevStart = interval[0];
                prevEnd = interval[1];
                flag = true;
                continue;
            }
            int start = interval[0];
            int end = interval[1];
            if (prevEnd > start) {
                prevEnd = Math.min(end, prevEnd);
                removeCount++;
            } else {
                prevEnd = end;
            }
        }
        return removeCount;
    }
}
