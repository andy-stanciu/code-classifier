class Solution {

    public int method1(int[] arr) {
        int n = arr.length;
        int i = 0;
        while (i < n) {
            int idx = arr[i] - 1;
            if (arr[i] > 0 && arr[i] <= n && arr[i] != arr[idx])
                swap(arr, i, idx);
            else
                i++;
        }
        for (int j = 0; j < n; j++) {
            if (arr[j] != j + 1)
                return j + 1;
        }
        return n + 1;
    }

    public void swap(int[] arr, int p1, int p2) {
        int temp = arr[p1];
        arr[p1] = arr[p2];
        arr[p2] = temp;
    }
}
