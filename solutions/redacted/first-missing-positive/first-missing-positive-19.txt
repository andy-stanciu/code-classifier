class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            set.add(nums[i]);
        }
        for (int i = 1; i < nums[nums.length - 1]; i++) {
            if (set.add(i)) {
                return i;
            }
        }
        if (nums[nums.length - 1] < 0) {
            return 1;
        }
        return nums[nums.length - 1] + 1;
    }
}
