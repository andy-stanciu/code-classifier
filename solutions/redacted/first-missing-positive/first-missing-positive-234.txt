class Solution {

    public static int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int num : nums) {
            if (num > 0) {
                set.add(num);
            }
        }
        int x = 1;
        while (true) {
            if (!set.contains(x)) {
                return x;
            }
            x++;
        }
    }
}
