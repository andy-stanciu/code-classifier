class Solution {

    public int method1(int[] arr) {
        boolean one = false;
        int n = arr.length;
        // Step 1: Check if \'1\' is present in the array, if not return 1
        for (int i = 0; i < n; i++) {
            if (arr[i] == 1) {
                one = true;
            }
            if (arr[i] < 1 || arr[i] > n) {
                arr[i] = 1;
            }
        }
        if (!one) {
            return 1;
        }
        // Step 2: Use the index as a hash to record the presence of numbers
        for (int i = 0; i < n; i++) {
            // Converting value to index
            int index = Math.abs(arr[i]) - 1;
            if (arr[index] > 0) {
                arr[index] = -arr[index];
            }
        }
        // Step 3: Find the first positive index
        for (int i = 0; i < n; i++) {
            if (arr[i] > 0) {
                return i + 1;
            }
        }
        // Step 4: If all indices are marked, the missing positive number is n + 1
        return n + 1;
    }
}
