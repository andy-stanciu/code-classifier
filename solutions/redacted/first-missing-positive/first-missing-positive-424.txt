class Solution {

    public int method1(int[] nums) {
        Set<Integer> s = new HashSet<>();
        for (int i : nums) {
            if (i > 0) {
                s.add(i);
            }
        }
        List<Integer> arr = new ArrayList<>(s);
        Collections.sort(arr);
        if (arr.size() == 0) {
            return 1;
        } else if (arr.get(0) > 1) {
            return 1;
        } else {
            for (int i = 0; i < arr.size() - 1; i++) {
                if (arr.get(i) + 1 != arr.get(i + 1)) {
                    return arr.get(i) + 1;
                }
            }
        }
        return arr.get(arr.size() - 1) + 1;
    }
}
