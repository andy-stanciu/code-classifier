class Solution {

    public int method1(int[] nums) {
        Set<Integer> s = new HashSet<>();
        for (int i : nums) {
            if (i > 0) {
                s.add(i);
            }
        }
        for (int i = 1; i <= nums.length; i++) {
            if (!s.contains(i)) {
                return i;
            }
        }
        return nums.length + 1;
    }
}
