class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] <= 0) {
                nums[i] = n + 2;
            }
        }
        int elem, index;
        for (int i = 0; i < n; i++) {
            elem = Math.abs(nums[i]);
            if (elem >= 1 && elem <= n) {
                index = elem - 1;
                nums[index] = -1 * Math.abs(nums[index]);
            }
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0) {
                return i + 1;
            }
        }
        return n + 1;
    }
}
