class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        // Mark non-positive integers and integers greater than n with a special value (n + 1)
        for (int i = 0; i < n; i++) {
            if (nums[i] <= 0 || nums[i] > n) {
                nums[i] = n + 1;
            }
        }
        // Mark the presence of positive integers by changing the sign of corresponding indices
        for (int i = 0; i < n; i++) {
            int num = Math.abs(nums[i]);
            if (num <= n) {
                nums[num - 1] = -Math.abs(nums[num - 1]);
            }
        }
        // Find the first index with a positive value, indicating the missing positive integer
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0) {
                return i + 1;
            }
        }
        // If all positive integers are present, return n + 1
        return n + 1;
    }
}
