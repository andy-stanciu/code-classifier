class Solution {

    public int method1(int[] nums) {
        int tmp, tmp2;
        for (int i = 0; i < nums.length; i++) {
            tmp = nums[i];
            while (tmp > 0 && tmp <= nums.length && nums[tmp - 1] != tmp) {
                tmp2 = nums[tmp - 1];
                nums[tmp - 1] = tmp;
                tmp = tmp2;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != (i + 1)) {
                return (i + 1);
            }
        }
        return nums.length + 1;
    }
}
