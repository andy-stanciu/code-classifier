class Solution {

    public int method1(int[] nums) {
        int numsLength = nums.length;
        boolean[] seen = new boolean[numsLength + 1];
        for (int num : nums) {
            if (num > 0 && num <= numsLength) {
                seen[num] = true;
            }
        }
        for (int i = 1; i <= numsLength; i++) {
            if (!seen[i]) {
                return i;
            }
        }
        return numsLength + 1;
    }
}
