class Solution {

    public int method1(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            int idx = nums[i] - 1;
            if (nums[i] < nums.length && nums[i] > 0 && nums[i] != nums[idx]) {
                swapArray(nums, i, idx);
            } else {
                i++;
            }
        }
        for (i = 0; i < nums.length; i++) {
            //The first index i where nums[i] is not equal to i + 1
            //indicates that i + 1 is the first missing positive number.
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        //if all elements are at correct position
        return nums.length + 1;
    }

    static void swapArray(int[] arr, int param1, int second) {
        int temp = arr[param1];
        arr[param1] = arr[second];
        arr[second] = temp;
    }
}
