class Solution {

    public int method1(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            int correct = nums[i] - 1;
            if (correct < 0 || correct >= nums.length) {
                i++;
            } else if (nums[i] != nums[correct]) {
                swap(nums, i, correct);
            } else {
                i++;
            }
        }
        for (i = 0; i < nums.length; i++) {
            if (nums[i] - 1 != i) {
                return i + 1;
            }
        }
        return nums.length + 1;
    }

    void swap(int[] arr, int f, int l) {
        int temp = arr[f];
        arr[f] = arr[l];
        arr[l] = temp;
    }
}
