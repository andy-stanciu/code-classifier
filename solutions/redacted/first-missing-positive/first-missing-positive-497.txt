class Solution {

    public int method1(int[] nums) {
        cyclicSorting(nums);
        for (int i = 0; i < nums.length; i++) {
            if (i != nums[i] - 1) {
                return ++i;
            }
        }
        return ++nums[nums.length - 1];
    }

    public static void cyclicSorting(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            while (i < nums.length) {
                int correct = nums[i] - 1;
                if (i != correct && correct >= 0 && correct < nums.length && nums[i] != nums[correct]) {
                    int temp = nums[i];
                    nums[i] = nums[correct];
                    nums[correct] = temp;
                } else {
                    break;
                }
            }
        }
    }
}
