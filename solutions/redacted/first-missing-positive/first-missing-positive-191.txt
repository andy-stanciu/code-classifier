class Solution {

    public int method1(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            int correct_index = nums[i] - 1;
            // conditions to ignore negative no\'s ,
            //numbers greater than arrays length
            // and swap numbers if at wrong index
            if (nums[i] < nums.length && nums[i] > 0 && nums[i] != nums[correct_index]) {
                int temp = nums[i];
                nums[i] = nums[correct_index];
                nums[correct_index] = temp;
            } else {
                i++;
            }
        }
        // check whick is number is at wrong index
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] != j + 1) {
                return j + 1;
            }
        }
        // return length if all positive numbers are entered
        return nums.length + 1;
    }
}
