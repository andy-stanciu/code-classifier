class Solution {

    public int method1(int[] nums) {
        // Sort the array to bring all positive numbers together
        Arrays.sort(nums);
        // Initialize a variable \'h\' to track the expected positive integer
        int h = 1;
        // Iterate through the sorted array
        for (int i = 0; i < nums.length; i++) {
            // If the current number matches the expected positive integer \'h\'
            if (nums[i] == h) {
                // Increment \'h\' to check for the next positive integer
                h++;
            }
        }
        // \'h\' now holds the smallest positive integer missing from the array
        return h;
    }
}
