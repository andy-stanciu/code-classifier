class Solution {

    public int method1(int[] nums) {
        int s = 1;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (!(nums[i] <= s)) {
                return s;
            } else {
                if (nums[i] == s) {
                    s++;
                }
            }
        }
        return s;
    }
}
