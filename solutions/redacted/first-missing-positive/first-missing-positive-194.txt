class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            if (nums[i] <= 0 || nums[i] >= n + 1)
                nums[i] = Integer.MAX_VALUE;
        }
        for (int i = 0; i < n; i++) {
            int no = Math.abs(nums[i]);
            if (no == Integer.MAX_VALUE)
                continue;
            if (nums[no - 1] > 0)
                nums[no - 1] = -nums[no - 1];
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0)
                return i + 1;
        }
        return n + 1;
    }
}
