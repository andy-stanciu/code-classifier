class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> set = new HashSet<>();
        for (int i : nums) set.add(i);
        Arrays.sort(nums);
        int n = nums[nums.length - 1];
        for (int i = 1; i <= nums.length; i++) {
            if (!set.contains(i))
                return i;
        }
        return n + 1;
    }
}
