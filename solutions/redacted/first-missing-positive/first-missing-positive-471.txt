class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int k = n + 5;
        for (int i = 0; i < n; i++) {
            if (nums[i] <= 0) {
                nums[i] = k;
                k++;
            }
        }
        for (int i = 0; i < n; i++) {
            int num = Math.abs(nums[i]);
            if (num <= n && nums[num - 1] > 0) {
                nums[num - 1] = -1 * nums[num - 1];
            }
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0) {
                return i + 1;
            }
        }
        return n + 1;
    }
}
