class Solution {

    public int method1(int[] a) {
        int minPos = 1;
        HashSet<Integer> hash = new HashSet<>();
        for (int i : a) {
            if (i > 0)
                hash.add(i);
            if (i > 0 && i < minPos)
                minPos = i;
        }
        while (hash.contains(minPos)) minPos++;
        return minPos;
    }
}
