class Solution {

    public int method1(int[] nums) {
        Set<Integer> set = new HashSet<>();
        int var1 = 1;
        for (int num : nums) {
            if (num > 0) {
                set.add(num);
            }
            if (num == var1) {
                var1++;
                while (set.contains(var1)) {
                    var1++;
                }
            }
        }
        return var1;
    }
}
