class Solution {

    public int method1(int[] nums) {
        HashSet<Integer> ms = new HashSet<>();
        int max = 0;
        for (int num : nums) {
            if (num > 0) {
                ms.add(num);
                if (num > max) {
                    max = num;
                }
            }
        }
        for (int i = 1; i <= max; i++) {
            if (!ms.contains(i)) {
                return i;
            }
        }
        return max + 1;
    }
}
