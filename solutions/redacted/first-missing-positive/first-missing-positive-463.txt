class Solution {

    public int method1(int[] nums) {
        int size = nums.length;
        int index = 0;
        while (index < size) {
            while (nums[index] < size && nums[index] > -1 && nums[index] != index && nums[index] != nums[nums[index]]) {
                swap(nums, index, nums[index]);
            }
            index++;
        }
        for (int i = 1; i < size; i++) {
            if (nums[i] != i)
                return i;
        }
        return nums[0] == size ? size + 1 : size;
    }

    void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
