class Solution {

    public int method1(int[] nums) {
        boolean one = false;
        //Mark out of range elements as one and find original one is exist or not
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 1)
                one = true;
            else if (nums[i] < 1 || nums[i] > nums.length)
                nums[i] = 1;
        }
        // If one is missing than return one.
        if (!one)
            return 1;
        // Mark in range elements as negative.
        for (int i = 0; i < nums.length; i++) {
            int index = Math.abs(nums[i]);
            nums[index - 1] = -Math.abs(nums[index - 1]);
        }
        // return the index+1 whose element is non negative..
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0)
                return i + 1;
        }
        //If all elements are exist than we will return n+1
        return nums.length + 1;
    }
}
