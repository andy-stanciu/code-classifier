class Solution {

    public int method1(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            if (nums[i] <= 0) {
                i++;
            } else if (nums[i] > nums.length) {
                i++;
            } else if (nums[i] != nums[nums[i] - 1]) {
                swap(nums, i, nums[i] - 1);
            } else {
                i++;
            }
        }
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] != j + 1) {
                return j + 1;
            }
        }
        return nums.length + 1;
    }

    void swap(int[] arr, int a, int b) {
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
}
