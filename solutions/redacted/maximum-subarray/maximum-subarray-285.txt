class Solution {

    public int method1(int[] nums) {
        // Initialize variables to track the current sum and maximum sum encountered.
        int sum = 0;
        // Start with the smallest possible integer value.
        int ans = Integer.MIN_VALUE;
        // Iterate through the array to calculate the maximum subarray sum.
        for (int i = 0; i < nums.length; i++) {
            // If the current sum is non-negative, add the current element.
            // If it\'s negative, reset the sum to the current element.
            if (sum >= 0) {
                sum += nums[i];
            } else {
                // Start a new subarray from the current element.
                sum = nums[i];
            }
            // Update the maximum sum encountered so far.
            ans = Math.max(ans, sum);
        }
        // Return the largest sum found.
        return ans;
    }
}
