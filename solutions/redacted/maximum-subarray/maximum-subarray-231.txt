class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        long var1 = Long.MIN_VALUE;
        long sum = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            if (var1 < sum) {
                var1 = sum;
            }
            if (sum < 0) {
                sum = 0;
            }
        }
        return (int) var1;
    }
}
