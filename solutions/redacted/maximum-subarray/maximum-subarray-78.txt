class Solution {

    public int method1(int[] arr) {
        // Initialize sum to 0, which will keep track of the current subarray sum.
        int sum = 0;
        // Initialize max to the first element of the array.
        // This is because the maximum subarray sum could be the first element itself.
        int max = arr[0];
        // Iterate through the array.
        for (int i = 0; i < arr.length; i++) {
            // Add the current element to the sum.
            sum += arr[i];
            // If the current sum is greater than the max found so far, update max.
            if (sum > max) {
                max = sum;
            }
            // If the current sum becomes negative, reset sum to 0.
            // This is because a negative sum would decrease the value of any subsequent subarray.
            if (sum < 0) {
                sum = 0;
            }
        }
        // Return the maximum subarray sum found.
        return max;
    }
}
