class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int sum = 0;
        int maxi = nums[0];
        for (int i = 0; i < n; i++) {
            //Step 1 -> Find the Sum of SubArray
            sum = sum + nums[i];
            //Step 2 -> Check for MaxiMum Sum
            maxi = Math.max(maxi, sum);
            //Step 3 -> When Sum is smaller than zero
            if (sum < 0) {
                sum = 0;
            }
        }
        return maxi;
    }
}
