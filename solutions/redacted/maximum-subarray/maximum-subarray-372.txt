class Solution {

    public int method1(int[] nums) {
        int c_sum = nums[0];
        int o_sum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            if (c_sum > 0) {
                c_sum += nums[i];
            } else {
                //make it positive
                c_sum = nums[i];
            }
            o_sum = Math.max(c_sum, o_sum);
        }
        return o_sum;
    }
}
