class Solution {

    public int method1(int[] nums) {
        boolean isNegativeArray = true;
        int currSum = 0;
        int maxSum = Integer.MIN_VALUE;
        int min = nums[0];
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0)
                isNegativeArray = false;
            min = Math.max(min, nums[i]);
            currSum += nums[i];
            if (currSum < 0)
                currSum = 0;
            maxSum = Math.max(currSum, maxSum);
        }
        if (isNegativeArray) {
            return min;
        }
        return maxSum;
    }
}
