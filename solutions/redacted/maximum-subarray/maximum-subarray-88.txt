class Solution {

    public int method1(int[] nums) {
        int max = Integer.MIN_VALUE;
        int maxEnding = 0;
        for (int num : nums) {
            max = Math.max(max, maxEnding += num);
            if (maxEnding < 0) {
                maxEnding = 0;
            }
        }
        return max;
    }
}
