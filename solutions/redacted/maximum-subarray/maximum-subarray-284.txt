class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int currSum = nums[0];
        int longestSum = nums[0];
        if (n == 1)
            return nums[0];
        for (int i = 1; i < n; i++) {
            if (currSum >= 0) {
                //if the sum is coimg potive then just add it
                currSum += nums[i];
            } else {
                //if it is negative then update the curr sum with nums[i];
                currSum = nums[i];
            }
            //when currSum becomes greater than the longetsum update it coz that is what we neeed to return
            if (currSum > longestSum) {
                longestSum = currSum;
            }
        }
        return longestSum;
    }
}
