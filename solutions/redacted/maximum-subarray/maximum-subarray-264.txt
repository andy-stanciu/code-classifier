class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        if (n == 1) {
            return nums[0];
        }
        int max = nums[0];
        int currSum = nums[0];
        for (int i = 1; i < n; i++) {
            currSum = Math.max(nums[i], currSum + nums[i]);
            max = Math.max(currSum, max);
        }
        return max;
    }
}
