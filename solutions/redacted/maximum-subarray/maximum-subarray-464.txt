class Solution {

    public int method1(int[] arr) {
        //for temporary sum
        int temp = arr[0];
        // for final sum
        int finals = arr[0];
        for (int i = 1; i < arr.length; i++) {
            temp = Math.max(arr[i], temp + arr[i]);
            finals = Math.max(temp, finals);
        }
        return finals;
    }
}
