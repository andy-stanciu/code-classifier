class Solution {

    public int method1(int[] A) {
        int current_sum = 0;
        int max_num = Integer.MIN_VALUE;
        int n = A.length;
        for (int i = 0; i < n; i++) {
            current_sum += A[i];
            if (current_sum > max_num) {
                max_num = current_sum;
            }
            if (current_sum < 0) {
                current_sum = 0;
            }
        }
        return max_num;
    }
}
