class Solution {

    public int method1(int[] nums) {
        //create a variable max to hold the max value of a sub array
        int max = Math.max(Integer.MIN_VALUE, nums[0]);
        for (int i = 1; i < nums.length; i++) {
            // check if at position i it is better continue this
            // subarray or start a new one
            nums[i] = Math.max(nums[i], nums[i] + nums[i - 1]);
            // check if we hold the best subarray
            // value at the moment
            max = Math.max(nums[i], max);
        }
        return max;
    }
}
