class Solution {

    public int method1(int[] nums) {
        // start with 1st element as a max
        int maxSum = nums[0];
        int currentSum = 0;
        for (int i = 0; i < nums.length; i++) {
            // add current num to currentsum
            currentSum += nums[i];
            // update maxsum if currentsum is greater
            maxSum = Math.max(maxSum, currentSum);
            // reset current sum is it goes below 0
            if (currentSum < 0) {
                currentSum = 0;
            }
        }
        return maxSum;
    }
}
