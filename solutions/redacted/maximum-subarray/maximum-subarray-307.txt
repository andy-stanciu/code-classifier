class Solution {

    public int method1(int[] nums) {
        int c = 0;
        int m = nums[0];
        for (int i : nums) {
            c += i;
            if (m < c) {
                m = c;
            }
            if (c < 0) {
                c = 0;
            }
        }
        return m;
    }
}
