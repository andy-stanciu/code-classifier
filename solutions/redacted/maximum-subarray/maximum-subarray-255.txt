class Solution {

    public int method1(int[] nums) {
        int maxEnding = nums[0];
        int result = nums[0];
        for (int i = 1; i < nums.length; i++) {
            maxEnding = Math.max((maxEnding + nums[i]), nums[i]);
            result = Math.max(result, maxEnding);
        }
        return result;
    }
}
