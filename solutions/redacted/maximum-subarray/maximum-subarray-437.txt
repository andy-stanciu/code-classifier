class Solution {

    public int method1(int[] nums) {
        int CurrSum = 0;
        int var1 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            CurrSum += nums[i];
            var1 = Math.max(CurrSum, var1);
            if (CurrSum < 0) {
                CurrSum = 0;
            }
        }
        return var1;
    }
}
