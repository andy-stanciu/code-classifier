class Solution {

    public int method1(int[] nums) {
        // check all element are negetive or not
        boolean isNeg = true;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > -1) {
                isNeg = false;
                break;
            }
        }
        int currSum = 0, maxSum = Integer.MIN_VALUE;
        //Case-1  it is the corner case when all element are negetive
        if (isNeg) {
            for (int i = 0; i < nums.length; i++) {
                maxSum = Math.max(nums[i], maxSum);
            }
        } else //Case-2 if atleast one value is non-negative
        {
            for (int i = 0; i < nums.length; i++) {
                currSum += nums[i];
                if (currSum < 0)
                    currSum = 0;
                if (currSum > maxSum)
                    maxSum = currSum;
            }
        }
        return maxSum;
    }
}
