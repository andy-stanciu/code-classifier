class Solution {

    public int method1(int[] nums) {
        int maxSum = Integer.MIN_VALUE;
        int maxValue = 0;
        for (int currentValue : nums) {
            maxValue += currentValue;
            maxSum = maxValue > maxSum ? maxValue : maxSum;
            if (maxValue < 0)
                maxValue = 0;
        }
        return maxSum;
    }
}
