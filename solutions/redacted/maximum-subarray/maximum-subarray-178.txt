class Solution {

    public int method1(int[] nums) {
        int maxi = Integer.MIN_VALUE;
        int sum = 0;
        int ansstart = -1;
        int ansend = -1;
        int start = 0;
        for (int i = 0; i < nums.length; i++) {
            if (sum == 0) {
                start = i;
            }
            sum = sum + nums[i];
            if (sum > maxi) {
                maxi = sum;
                ansstart = start;
                ansend = i;
            }
            if (sum < 0) {
                sum = 0;
            }
        }
        int[] arr = new int[ansend - ansstart + 1];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = nums[ansstart + i];
        }
        System.out.println(Arrays.toString(arr));
        return maxi;
    }
}
