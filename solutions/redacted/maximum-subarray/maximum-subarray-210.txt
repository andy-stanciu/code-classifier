class Solution {

    public int method1(int[] nums) {
        int maxSum = Integer.MIN_VALUE;
        int currSum = 0;
        for (int i = 0; i < nums.length; i++) {
            currSum = Math.max((currSum + nums[i]), nums[i]);
            maxSum = Math.max(maxSum, currSum);
        }
        return maxSum;
    }
}
