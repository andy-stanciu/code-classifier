class Solution {

    public int method1(int[] nums) {
        int currSum = 0;
        int ans = Integer.MIN_VALUE;
        int right = 0;
        while (right < nums.length) {
            currSum += nums[right];
            ans = Math.max(ans, currSum);
            if (currSum < 0) {
                currSum = 0;
            }
            right++;
        }
        return ans;
    }
}
