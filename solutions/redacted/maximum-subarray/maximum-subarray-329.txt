class Solution {

    public int method1(int[] nums) {
        int currMax = nums[0];
        int totalMax = nums[0];
        for (int num = 1; num < nums.length; num++) {
            currMax = Math.max(nums[num], currMax + nums[num]);
            totalMax = Math.max(totalMax, currMax);
        }
        return totalMax;
    }
}
