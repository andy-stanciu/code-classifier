class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int var1 = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            if (sum > var1) {
                var1 = sum;
            }
            if (sum < 0) {
                sum = 0;
            }
        }
        return var1;
    }
}
