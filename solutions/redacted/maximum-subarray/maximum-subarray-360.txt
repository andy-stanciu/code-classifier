class Solution {

    public int method1(int[] nums) {
        int currSum = 0, maxSum = Integer.MIN_VALUE;
        for (int value : nums) {
            currSum += value;
            maxSum = Math.max(maxSum, currSum);
            if (currSum < 0) {
                currSum = 0;
            }
        }
        return maxSum;
    }
}
