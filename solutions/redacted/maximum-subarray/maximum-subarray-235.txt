class Solution {

    public int method1(int[] nums) {
        int l = nums.length;
        if (l < 1)
            return -1;
        int maxSum = nums[0];
        int prevMaxSum = nums[0];
        for (int right = 1; right < l; right++) {
            prevMaxSum = Math.max(prevMaxSum + nums[right], nums[right]);
            maxSum = Math.max(maxSum, prevMaxSum);
        }
        return maxSum;
    }
}
