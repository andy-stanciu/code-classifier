class Solution {

    public int method1(int[] nums) {
        // Get the length of the input array \'nums\'
        int n = nums.length;
        // Initialize \'max\' to the smallest possible integer value
        int max = Integer.MIN_VALUE;
        // Initialize \'sum\' to keep track of the current subarray sum
        int sum = 0;
        for (int i = 0; i < n; i++) {
            // Loop through the elements of the array
            // Add the current element to the \'sum\'
            sum += nums[i];
            if (sum > max) {
                // If \'sum\' is greater than \'max\', update \'max\'
                max = sum;
            }
            if (sum < 0) {
                // If \'sum\' becomes negative, reset \'sum\' to 0
                sum = 0;
            }
        }
        // Return the maximum subarray sum found
        return max;
    }
}
