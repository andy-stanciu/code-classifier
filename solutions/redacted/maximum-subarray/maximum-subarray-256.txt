class Solution {

    public int method1(int[] nums) {
        int MaxSum = Integer.MIN_VALUE;
        int CurrSum = 0;
        for (int i = 0; i < nums.length; i++) {
            CurrSum = CurrSum + nums[i];
            MaxSum = Math.max(CurrSum, MaxSum);
            if (CurrSum < 0) {
                CurrSum = 0;
            }
            ;
        }
        ;
        return MaxSum;
    }
}
