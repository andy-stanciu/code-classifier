class Solution {

    public int method1(int[] nums) {
        int length = nums.length;
        int[] dp = new int[length];
        dp[0] = nums[0];
        int maxOverall = dp[0];
        for (int i = 1; i < length; i++) {
            dp[i] = Math.max(nums[i], dp[i - 1] + nums[i]);
            maxOverall = Math.max(maxOverall, dp[i]);
        }
        return maxOverall;
    }
}
