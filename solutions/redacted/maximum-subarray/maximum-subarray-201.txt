class Solution {

    public int method1(int[] nums) {
        // Variable to store the current sum
        int curr_sum = nums[0];
        // Variable to store the maximum sum
        int max_sum = curr_sum;
        if (nums.length == 1) {
            // Return the only element if array length is 1
            return nums[0];
        }
        for (int i = 1; i < nums.length; i++) {
            if (curr_sum < 0) {
                // Reset current sum to 0 if it becomes negative
                curr_sum = 0;
            }
            // Add current element to current sum
            curr_sum = curr_sum + nums[i];
            // Update maximum sum if necessary
            max_sum = Math.max(curr_sum, max_sum);
        }
        // Return the maximum sum
        return max_sum;
    }
}
