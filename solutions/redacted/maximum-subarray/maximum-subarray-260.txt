class Solution {

    public int method1(int[] nums) {
        int var1 = nums[0];
        int currSum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (currSum < 0)
                currSum = 0;
            currSum += nums[i];
            var1 = Math.max(currSum, var1);
        }
        return var1;
    }
}
