class Solution {

    public int method1(int[] nums) {
        //step 1
        int[] dp = new int[nums.length + 1];
        //step 2
        dp[0] = nums[0];
        int maxSum = dp[0];
        //step 3
        for (int k = 1; k < nums.length; k++) {
            dp[k] = Math.max(dp[k - 1] + nums[k], nums[k]);
            //step 4
            maxSum = Math.max(maxSum, dp[k]);
        }
        // step 5
        return maxSum;
    }
}
