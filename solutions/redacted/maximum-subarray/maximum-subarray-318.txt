class Solution {

    public int method1(int[] nums) {
        int maxsum = Integer.MIN_VALUE;
        int sum = 0;
        int i = 0;
        while (i < nums.length) {
            sum += nums[i];
            maxsum = Math.max(sum, maxsum);
            if (sum < 0)
                sum = 0;
            i++;
        }
        return maxsum;
    }
}
