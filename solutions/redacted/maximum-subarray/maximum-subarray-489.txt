class Solution {

    public int method1(int[] nums) {
        int maxSoFar = nums[0];
        int max = nums[0];
        for (int i = 1; i < nums.length; i++) {
            maxSoFar = Math.max(nums[i], nums[i] + maxSoFar);
            max = Math.max(max, maxSoFar);
        }
        return max;
    }
}
