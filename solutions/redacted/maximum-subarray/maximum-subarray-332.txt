class Solution {

    public int method1(int[] nums) {
        int maxSum = Integer.MIN_VALUE;
        int leftToRight = 0;
        for (int j : nums) {
            if (leftToRight < 0)
                leftToRight = 0;
            // calculate product from index left to right
            leftToRight += j;
            // calculate product from index right to left
            maxSum = Math.max(leftToRight, maxSum);
        }
        return maxSum;
    }
}
