class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int j = 0, sum = 0, max = Integer.MIN_VALUE;
        while (j < n) {
            sum += nums[j];
            max = Math.max(sum, max);
            if (sum < 0) {
                sum = 0;
            }
            j++;
        }
        return max;
    }
}
