class Solution {

    public int method1(int[] nums) {
        int currentSum = nums[0];
        int globalSum = nums[0];
        for (int i = 1; i < nums.length; i++) {
            currentSum += nums[i];
            if (nums[i] >= currentSum) {
                currentSum = nums[i];
            }
            globalSum = Math.max(globalSum, currentSum);
        }
        return globalSum;
    }
}
