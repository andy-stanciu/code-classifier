class Solution {

    public int method1(int[] nums) {
        int maxsum = nums[0];
        int currentsum = 0;
        for (int i = 0; i < nums.length; i++) {
            if (currentsum < 0) {
                currentsum = 0;
            }
            currentsum = currentsum + nums[i];
            maxsum = Math.max(maxsum, currentsum);
        }
        return maxsum;
    }
}
