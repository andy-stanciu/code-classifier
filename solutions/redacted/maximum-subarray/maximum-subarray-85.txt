class Solution {

    public int method1(int[] nums) {
        //start with assuming that the first element is maxsum and it\'s also our current sum
        int currentsum = nums[0];
        int maxsum = nums[0];
        //now hover over the loop and check for either to continue the extension or break it here for maximum holding...more kind of bussiness stuff, you know if you know(capitalism)
        for (int i = 1; i < nums.length; i++) {
            //check whether to extend or not
            currentsum = Math.max(nums[i], nums[i] + currentsum);
            //now swap if the currentsum is greater than that of maxsum
            maxsum = Math.max(currentsum, maxsum);
        }
        //now at last return the max
        return maxsum;
    }
}
