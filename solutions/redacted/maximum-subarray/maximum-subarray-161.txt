class Solution {

    public int method1(int[] nums) {
        int max = nums[0];
        int maxOverall = nums[0];
        for (int i = 1; i < nums.length; i++) {
            max = Math.max(nums[i], max + nums[i]);
            maxOverall = Math.max(maxOverall, max);
        }
        return maxOverall;
    }
}
