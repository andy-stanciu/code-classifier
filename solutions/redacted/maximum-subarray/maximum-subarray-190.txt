class Solution {

    public int method1(int[] nums) {
        int start = 0, ans = 0, sum = 0, e = nums[0];
        while (start < nums.length) {
            e = Math.max(e, nums[start]);
            sum = sum + nums[start];
            if (sum < 0) {
                sum = 0;
                start++;
            } else
                start++;
            ans = Math.max(sum, ans);
        }
        if (ans == 0)
            return e;
        return ans;
    }
}
