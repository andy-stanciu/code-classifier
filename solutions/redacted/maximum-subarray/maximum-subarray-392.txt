class Solution {

    public int method1(int[] nums) {
        int maxsum = Integer.MIN_VALUE;
        int currsum = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            currsum = currsum + nums[i];
            maxsum = Math.max(maxsum, currsum);
            if (currsum < 0) {
                currsum = 0;
            }
        }
        return maxsum;
    }
}
