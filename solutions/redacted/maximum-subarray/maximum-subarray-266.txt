class Solution {

    public static int largestNegativeNumberInArray(int[] arr) {
        int largestNegativeNumber = Integer.MIN_VALUE;
        for (int n : arr) {
            if (n > largestNegativeNumber) {
                largestNegativeNumber = n;
            }
        }
        return largestNegativeNumber;
    }

    public static boolean allNegative(int[] arr) {
        for (int n : arr) {
            if (n >= 0) {
                return false;
            }
        }
        return true;
    }

    public int method1(int[] nums) {
        if (allNegative(nums)) {
            return largestNegativeNumberInArray(nums);
        }
        int maxSum = Integer.MIN_VALUE;
        int cs = 0;
        for (int i : nums) {
            cs = cs + i;
            if (cs < 0) {
                cs = 0;
            }
            maxSum = Math.max(cs, maxSum);
        }
        return maxSum;
    }
}
