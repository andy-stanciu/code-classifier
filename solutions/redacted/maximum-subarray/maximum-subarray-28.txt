class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        //assuming ans is maximum as of now
        int ans = -999999;
        // creating tempory value
        int t = 0;
        for (int i = 0; i < n; i++) {
            t += nums[i];
            if (//
            t > ans) {
                ans = t;
            }
            if (t < 0) {
                t = 0;
            }
        }
        return ans;
    }
}
