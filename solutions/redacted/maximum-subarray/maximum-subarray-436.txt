class Solution {

    public int method1(int[] nums) {
        // Mathematics make thing so easy
        int sum = 0;
        int max = Integer.MIN_VALUE;
        // assigning sum = 0 when sum -1 what is the point to carry some ngative value which willl recduce my sum value
        // completing this in
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
            max = (max < sum) ? sum : max;
            if (sum < 0)
                sum = 0;
        }
        return max;
    }
}
