class Solution {

    public int method1(int[] nums) {
        int n = nums.length;
        int cur_val = 0;
        int max_val = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            cur_val += nums[i];
            if (cur_val > max_val) {
                max_val = cur_val;
            }
            if (cur_val < 0) {
                cur_val = 0;
            }
        }
        return max_val;
    }
}
