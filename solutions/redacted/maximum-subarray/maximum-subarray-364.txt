class Solution {

    public int method1(int[] nums) {
        if (nums == null || nums.length == 0)
            return 0;
        int cm = nums[0];
        int gm = nums[0];
        for (int i = 1; i < nums.length; i++) {
            cm = Math.max(nums[i], cm + nums[i]);
            gm = Math.max(cm, gm);
        }
        return gm;
    }
}
