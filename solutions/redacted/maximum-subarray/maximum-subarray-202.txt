//Time Complexity O(N)
//Space Complexity O(1)
class Solution {

    public int method1(int[] nums) {
        //no corner cases
        int result = Integer.MIN_VALUE;
        int prev = 0;
        //loop through the array
        for (int x : nums) {
            prev = Math.max(x, prev + x);
            result = Math.max(result, prev);
        }
        return result;
    }
}
