class Solution {

    public int method1(int[] nums) {
        int max = Integer.MIN_VALUE;
        int maxEnding = 0;
        for (int n : nums) {
            maxEnding = maxEnding + n;
            if (max < maxEnding)
                max = maxEnding;
            if (maxEnding < 0)
                maxEnding = 0;
        }
        return max;
    }
}
