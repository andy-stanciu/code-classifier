class Solution {

    public int method1(int[] nums) {
        //Maximum Sum
        int ms = Integer.MIN_VALUE;
        //Current Sum
        int cs = 0;
        for (int i = 0; i < nums.length; i++) {
            cs += nums[i];
            ms = Math.max(ms, cs);
            if (cs < 0) {
                cs = 0;
            }
        }
        return ms;
    }
}
