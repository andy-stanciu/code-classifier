class Solution {

    public int method1(int[] nums) {
        int end = nums[0];
        int max_total = nums[0];
        for (int i = 1; i < nums.length; i++) {
            end = Math.max(nums[i], end + nums[i]);
            max_total = Math.max(max_total, end);
        }
        return max_total;
    }
}
