class Solution {

    public int method1(int[] nums) {
        // Initialize current sum and max sum with
        // the first element of the array
        int curr_sum = nums[0];
        int max_sum = nums[0];
        // Iterate through the array starting from the second element
        for (int i = 1; i < nums.length; i++) {
            // Update the current sum by taking the maximum of the current
            // element and the current sum plus the current element
            curr_sum = Math.max(nums[i], curr_sum + nums[i]);
            // Update the max sum by taking the maximum of the current max
            // sum and the current sum
            max_sum = Math.max(curr_sum, max_sum);
        }
        return max_sum;
    }
}
