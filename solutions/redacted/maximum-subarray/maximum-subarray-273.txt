class Solution {

    public int method1(int[] arr) {
        int n = arr.length;
        int maxsum = Integer.MIN_VALUE;
        int sumsofar = 0;
        for (int i = 0; i < n; i++) {
            if (sumsofar < 0) {
                sumsofar = arr[i];
            } else {
                sumsofar += arr[i];
            }
            if (sumsofar > maxsum) {
                maxsum = sumsofar;
            }
        }
        return maxsum;
    }
}
