class Solution {

    public int method1(int[] nums) {
        int sum = 0;
        int max = nums[0];
        for (int a : nums) {
            sum += a;
            if (sum > max) {
                max = sum;
            }
            if (sum < 0) {
                sum = 0;
            }
        }
        return max;
    }
}
