/*
Runtime 1ms, faster than 100%
Memory Usage 73.1 MB, less than 70%

Loop through the ints in nums keeping track of a maxSum and a runningTotal.
If runningTotal is less 0, reset runningTotal to 0.
*/
class Solution {

    public int method1(int[] nums) {
        //use a stack to pop off values
        //if the total is negative, reset to zero and keep trying
        //Stack<Integer>
        int maxSum = Integer.MIN_VALUE;
        int runningTotal = 0;
        for (int i : nums) {
            runningTotal += i;
            if (runningTotal > maxSum) {
                maxSum = runningTotal;
            }
            if (runningTotal < 0) {
                //effectively resetting
                runningTotal = 0;
            }
        }
        return maxSum;
    }
}
