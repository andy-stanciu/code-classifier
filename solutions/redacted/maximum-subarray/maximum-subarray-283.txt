class Solution {

    public int method1(int[] nums) {
        if (nums.length == 1)
            return nums[0];
        int max = Integer.MIN_VALUE;
        int sum = 0;
        for (int num : nums) {
            sum += num;
            if (max < sum)
                max = sum;
            if (sum < 0)
                sum = 0;
        }
        return max;
    }
}
