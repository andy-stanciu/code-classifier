class Solution {

    public String method1(String s) {
        // Array for all ASCII characters
        int[] numbers = new int[128];
        // Initialize result as an empty string
        String result = "";
        // Count the frequency of each character in the string
        for (int i = 0; i < s.length(); i++) {
            // Increment the frequency of the character
            numbers[s.charAt(i)]++;
        }
        // Sorting characters based on frequency and building the result
        for (int j = 0; j < s.length(); j++) {
            int count = 0;
            char maxChar = 0;
            // Find the character with the highest frequency
            for (int x = 0; x < numbers.length; x++) {
                if (numbers[x] > count) {
                    count = numbers[x];
                    maxChar = (char) x;
                }
            }
            // Add the most frequent character `count` times to the result
            for (int v = 0; v < count; v++) {
                result += maxChar;
            }
            // Reset the frequency of the character after processing
            numbers[maxChar] = 0;
        }
        return result;
    }
}
