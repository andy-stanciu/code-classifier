class Solution {

    public String method1(String s) {
        StringBuilder str = new StringBuilder();
        Map<Character, Integer> hm = new HashMap<>();
        // Step 1: Count the frequency of each character
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            hm.put(ch, hm.getOrDefault(ch, 0) + 1);
        }
        // Step 2: Sort the characters based on their frequency
        List<Character> list = new ArrayList<>(hm.keySet());
        list.sort((ob1, ob2) -> hm.get(ob2) - hm.get(ob1));
        // Step 3: Append characters to StringBuilder based on their frequency
        for (char ch : list) {
            for (int i = 0; i < hm.get(ch); i++) {
                str.append(ch);
            }
        }
        // Step 4: Return the result as a string
        return str.toString();
    }
}
