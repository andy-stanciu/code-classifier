class Solution {

    public String method1(String s) {
        // we need to sort the string by their frequecy so we use hashmap for storing frequencies
        HashMap<Character, Integer> hm = new HashMap<>();
        for (char ch : s.toCharArray()) {
            hm.put(ch, hm.getOrDefault(ch, 0) + 1);
        }
        // Now we create a frequency arr we called it bucket for storing no.of times character
        List<Character>[] bucket = new List[s.length() + 1];
        for (char ch : hm.keySet()) {
            int freq = hm.get(ch);
            if (bucket[freq] == null)
                bucket[freq] = new ArrayList<>();
            bucket[freq].add(ch);
        }
        //now we need to store character in string so we iterate on the bucket array from back side
        StringBuilder sb = new StringBuilder();
        for (int i = bucket.length - 1; i >= 0; i--) {
            if (bucket[i] != null) {
                for (char ch : bucket[i]) {
                    for (int j = 0; j < hm.get(ch); j++) {
                        sb.append(ch);
                    }
                }
            }
        }
        return sb.toString();
    }
}
