class Solution {

    public String method1(String s) {
        String s1 = "";
        HashMap<Character, Integer> arr = new HashMap();
        for (int i = 0; i < s.length(); i++) {
            if (arr.containsKey(s.charAt(i)))
                arr.put(s.charAt(i), arr.get(s.charAt(i)) + 1);
            else
                arr.put(s.charAt(i), 1);
        }
        ArrayList<Character> ch = new ArrayList(arr.keySet());
        ArrayList<Integer> num = new ArrayList(arr.values());
        while (num.size() > 0) {
            int max = Collections.max(num);
            int index = num.indexOf(max);
            for (int i = 1; i <= max; i++) s1 = s1 + ch.get(index);
            ch.remove(index);
            num.remove(index);
        }
        return s1;
    }
}
