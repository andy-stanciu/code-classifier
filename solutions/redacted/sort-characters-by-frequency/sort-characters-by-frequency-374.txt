class Solution {

    public String method1(String s) {
        if (s == null || s.length() <= 1) {
            return s;
        }
        // Create a Map to store character frequencies
        Map<Character, Integer> var1 = new HashMap<>();
        // Populate the character frequency map
        for (char c : s.toCharArray()) {
            var1.put(c, var1.getOrDefault(c, 0) + 1);
        }
        // Convert the map entries to a list for sorting
        List<Map.Entry<Character, Integer>> entryList = new ArrayList<>(var1.entrySet());
        // Sort the list based on frequency (value) in descending order
        entryList.sort((a, b) -> b.getValue() - a.getValue());
        // Build the result string
        StringBuilder result = new StringBuilder();
        for (Map.Entry<Character, Integer> entry : entryList) {
            char c = entry.getKey();
            int var2 = entry.getValue();
            for (int i = 0; i < var2; i++) {
                result.append(c);
            }
        }
        return result.toString();
    }
}
