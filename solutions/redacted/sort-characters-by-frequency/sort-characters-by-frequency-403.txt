class Solution {

    public String method1(String s) {
        Map<Character, Integer> var1 = new HashMap<>();
        // Count the frequency of each character
        for (char ch : s.toCharArray()) {
            var1.put(ch, var1.getOrDefault(ch, 0) + 1);
        }
        // Create a list of characters sorted by frequency
        List<Character> var2 = new ArrayList<>(var1.keySet());
        var2.sort((a, b) -> var1.get(b) - var1.get(a));
        // Build the result string based on the sorted characters
        StringBuilder result = new StringBuilder();
        for (char ch : var2) {
            int var3 = var1.get(ch);
            for (int i = 0; i < var3; i++) {
                result.append(ch);
            }
        }
        return result.toString();
    }
}
