class Solution {

    public String method1(String s) {
        Map<Character, Integer> mappy = new HashMap<>();
        for (char ch : s.toCharArray()) {
            mappy.put(ch, mappy.getOrDefault(ch, 0) + 1);
        }
        List<Map.Entry<Character, Integer>> l = new ArrayList<>(mappy.entrySet());
        l.sort((entry1, entry2) -> entry2.getValue().compareTo(entry1.getValue()));
        StringBuilder sb = new StringBuilder();
        for (Map.Entry<Character, Integer> x : l) {
            for (int i = 0; i < x.getValue(); i++) {
                sb.append(x.getKey());
            }
        }
        return sb.toString();
    }
}
