class Solution {

    public String method1(String s) {
        int[] f = new int[128];
        int n = s.length();
        for (int i = 0; i < n; ++i) {
            f[s.charAt(i)]++;
        }
        PriorityQueue<int[]> q = new PriorityQueue<>((a, b) -> b[0] == a[0] ? a[0] - b[0] : b[0] - a[0]);
        //1 - frequency , 2 - char
        for (int i = 0; i < 128; ++i) {
            if (f[i] > 0)
                q.offer(new int[] { f[i], i });
        }
        StringBuilder res = new StringBuilder();
        while (!q.isEmpty()) {
            var head = q.poll();
            for (int i = 0; i < head[0]; ++i) res.append((char) head[1]);
        }
        return new String(res);
    }
}
