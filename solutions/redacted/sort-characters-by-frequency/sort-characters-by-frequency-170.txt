class Solution {

    public String method1(String s) {
        Map<Character, Integer> charMap = new HashMap<>();
        for (char c : s.toCharArray()) {
            charMap.put(c, charMap.getOrDefault(c, 0) + 1);
        }
        LinkedHashMap<Character, Integer> collect = charMap.entrySet().stream().sorted(Map.Entry.comparingByValue(Comparator.reverseOrder())).collect(Collectors.toMap(x -> x.getKey(), x -> x.getValue(), (a1, a2) -> a1, LinkedHashMap::new));
        String ans = "";
        for (Map.Entry<Character, Integer> entry : collect.entrySet()) {
            for (int i = 0; i < entry.getValue(); i++) {
                ans += entry.getKey();
            }
        }
        return ans;
    }
}
