class Solution {

    public String method1(String s) {
        // Initialize a map to store the frequency counts of characters
        Map<Character, Integer> var1 = new HashMap<>();
        // Count the frequency of each character in the string
        for (char c : s.toCharArray()) {
            var1.put(c, var1.getOrDefault(c, 0) + 1);
        }
        // Create a list of Map.Entry objects to store characters and their frequencies
        List<Map.Entry<Character, Integer>> entryList = new ArrayList<>(var1.entrySet());
        // Sort the entry list based on the frequency counts in descending order
        Collections.sort(entryList, (a, b) -> b.getValue() - a.getValue());
        // Build the sorted string based on the sorted entry list
        StringBuilder var2 = new StringBuilder();
        for (Map.Entry<Character, Integer> entry : entryList) {
            char c = entry.getKey();
            int count = entry.getValue();
            for (int i = 0; i < count; i++) {
                var2.append(c);
            }
        }
        return var2.toString();
    }
}
