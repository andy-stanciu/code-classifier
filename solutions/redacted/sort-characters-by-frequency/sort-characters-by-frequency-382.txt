class Solution {

    public String method1(String s) {
        if (s.length() == 1)
            return s;
        int[][] freq = new int[128][2];
        for (char c : s.toCharArray()) {
            freq[c][0] = (int) c;
            freq[c][1]++;
        }
        Arrays.sort(freq, (x, y) -> y[1] - x[1]);
        StringBuilder sb = new StringBuilder();
        for (int[] fr : freq) {
            int count = fr[1];
            while (count-- > 0) sb.append((char) fr[0]);
        }
        return sb.toString();
    }
}
