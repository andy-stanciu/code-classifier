/**
 *   Compute the Frequency of each Character in the String
 * - Iterate over the frequencies in the array
 * - Find the index of the element with maximum frequency
 * - check if the Max frequency is non zero otherwise break the loop
 * - Append the Character number of time the frequency of the element
 * - Finally return the String
 */
class Solution {

    public String method1(String s) {
        int[] freq = new int[256];
        for (char ch : s.toCharArray()) {
            ++freq[ch];
        }
        StringBuilder sb = new StringBuilder();
        int max = 0;
        while (true) {
            for (int index = 0; index < freq.length; index++) {
                if (freq[index] > freq[max]) {
                    max = index;
                }
            }
            if (freq[max] == 0) {
                break;
            }
            for (int index = 0; index < freq[max]; index++) {
                sb.append((char) max);
            }
            freq[max] = 0;
        }
        return sb.toString();
    }
}
