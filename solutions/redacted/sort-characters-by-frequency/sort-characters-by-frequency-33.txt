class Solution {

    public String method1(String s) {
        int[] freq = new int[123];
        //store every character one by one
        char ch;
        //ASCII value of ch for indexes.
        int asci;
        for (int i = 0; i < s.length(); i++) {
            ch = s.charAt(i);
            asci = (int) ch;
            freq[asci] = freq[asci] + 1;
        }
        //frequency of every char is stored in freq[]
        StringBuilder ans = new StringBuilder();
        int max;
        int count;
        for (int i = 0; i < s.length(); ) {
            // max is most frequent char (max = ascii value of that character)
            max = Maximum(freq);
            // max had occurred "count" number of time
            count = freq[max];
            //now there is no use of freq[max]
            freq[max] = -1;
            while (count != 0) {
                //append that character in ans
                ans.append((char) max);
                i++;
                count--;
            }
        }
        return ans.toString();
    }

    public static int Maximum(int[] arr) {
        int maxi = -1;
        int index = -1;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] >= maxi) {
                maxi = arr[i];
                index = i;
            }
        }
        return index;
    }
}
