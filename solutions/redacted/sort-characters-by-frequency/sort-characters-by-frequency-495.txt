class Solution {

    public String method1(String s) {
        var h = s.chars().mapToObj(c -> (char) c).collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        String ans = h.entrySet().stream().sorted(Map.Entry.<Character, Long>comparingByValue().reversed()).flatMap(e -> Collections.nCopies(e.getValue().intValue(), e.getKey()).stream()).map(Object::toString).collect(Collectors.joining());
        return ans;
    }
}
