class Solution {

    public String method1(String s) {
        // Create a HashMap to store character frequency
        Map<Character, Integer> var1 = new HashMap<>();
        // Count the frequency of each character
        for (char c : s.toCharArray()) {
            var1.put(c, var1.getOrDefault(c, 0) + 1);
        }
        // Create a priority queue to sort characters based on frequency
        PriorityQueue<Character> maxHeap = new PriorityQueue<>((a, b) -> var1.get(b) - var1.get(a));
        // Add all characters to the priority queue
        maxHeap.addAll(var1.keySet());
        // Build the sorted string
        StringBuilder var2 = new StringBuilder();
        while (!maxHeap.isEmpty()) {
            char c = maxHeap.poll();
            int var3 = var1.get(c);
            for (int i = 0; i < var3; i++) {
                var2.append(c);
            }
        }
        return var2.toString();
    }
}
