class Solution {

    public String method1(String s) {
        // Create a HashMap to store the frequency of each character
        Map<Character, Integer> hp = new HashMap<>();
        for (char c : s.toCharArray()) {
            hp.put(c, hp.getOrDefault(c, 0) + 1);
        }
        // Create a list from the map\'s entries (key-value pairs)
        List<Map.Entry<Character, Integer>> list = new ArrayList<>(hp.entrySet());
        // Sort the list based on the values (frequencies) in descending order
        list.sort((e1, e2) -> e2.getValue().compareTo(e1.getValue()));
        // StringBuilder to build the final result
        StringBuilder sb = new StringBuilder();
        // Append each character \'frequency\' number of times to the StringBuilder
        for (Map.Entry<Character, Integer> entry : list) {
            Character key = entry.getKey();
            int var1 = entry.getValue();
            // Append the character \'frequency\' times
            for (int i = 0; i < var1; i++) {
                sb.append(key);
            }
        }
        // Return the sorted string based on character frequencies
        return sb.toString();
    }
}
