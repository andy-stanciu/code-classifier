class Solution {

    public String method1(String s) {
        Map<Character, Integer> charCount = new HashMap<>();
        s.chars().forEach(i -> {
            char c = (char) i;
            charCount.put(c, charCount.getOrDefault(c, 0) + 1);
        });
        List<Map.Entry<Character, Integer>> listValues = new ArrayList<>(charCount.entrySet());
        listValues.sort(Map.Entry.comparingByValue());
        String result = "";
        for (int i = listValues.size() - 1; i >= 0; i--) {
            result += concatString(listValues.get(i).getKey(), listValues.get(i).getValue());
        }
        return result;
    }

    private String concatString(Character c, int count) {
        String s = "";
        for (int i = 0; i < count; i++) {
            s += c;
        }
        return s;
    }
}
