class Solution {

    public String method1(String s) {
        int[] count = new int[123];
        Integer[] indees = new Integer[123];
        for (int i = 0; i < 123; i++) {
            indees[i] = i;
        }
        for (char c : s.toCharArray()) {
            count[c]++;
        }
        Arrays.sort(indees, (i, j) -> count[j] - count[i]);
        StringBuilder res = new StringBuilder();
        for (int i = 0; i < 123; i++) {
            int k = indees[i];
            while (count[k] > 0) {
                res.append((char) k);
                count[k]--;
            }
        }
        return res.toString();
    }
}
