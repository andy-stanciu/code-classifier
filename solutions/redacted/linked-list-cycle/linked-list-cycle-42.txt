/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        if (head == null) {
            return false;
        }
        ListNode oneStepNext = head, twoStepNext = head;
        while (oneStepNext != null && (twoStepNext != null && twoStepNext.next != null)) {
            oneStepNext = oneStepNext.next;
            twoStepNext = twoStepNext.next.next;
            if (oneStepNext == twoStepNext) {
                return true;
            }
        }
        return false;
    }
}
