/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        if (head == null)
            return false;
        ListNode torNode = head, harNode = head;
        while (harNode.next != null) {
            harNode = harNode.next;
            if (harNode.next == null) {
                return false;
            }
            harNode = harNode.next;
            torNode = torNode.next;
            if (torNode.equals(harNode))
                return true;
        }
        return false;
    }
}
