public class Solution {

    public boolean method1(ListNode head) {
        ListNode slow = head, fast = head;
        while (true) {
            if (fast == null)
                return false;
            else if (fast.next == null)
                return false;
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow)
                return true;
        }
    }
}
