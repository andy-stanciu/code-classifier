/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        ListNode current = head;
        Map<ListNode, Integer> m = new HashMap();
        while (current != null) {
            if (m.containsKey(current)) {
                return true;
            }
            m.put(current, 1);
            current = current.next;
        }
        return false;
    }
}
