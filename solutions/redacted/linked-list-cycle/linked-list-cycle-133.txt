/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        ListNode current = head;
        while (current != null) {
            if (current.val == -11111) {
                return true;
            }
            current.val = -11111;
            current = current.next;
        }
        return false;
    }
}
