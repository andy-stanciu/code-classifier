public class Solution {

    public boolean method1(ListNode head) {
        // Initialize two pointers: tortoise and rabbit
        // Tortoise starts at the head
        ListNode tortoise = head;
        // Rabbit also starts at the head
        ListNode rabbit = head;
        // Begin the exciting race \uD83C\uDFC1
        while (rabbit != null && rabbit.next != null) {
            // Tortoise moves one step
            tortoise = tortoise.next;
            // Rabbit moves two steps
            rabbit = rabbit.next.next;
            // Check if they meet at the same spot
            if (tortoise == rabbit) {
                // Hooray! They met, which means there\'s a cycle \uD83C\uDF89
                return true;
            }
        }
        // No cycle found, the race is over \uD83D\uDE22
        return false;
    }
}
