/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        if (head == null)
            return false;
        if (head.next == null)
            return false;
        ListNode t = head;
        ListNode s = head;
        ListNode f = head.next.next;
        while (s != null && f != null && f.next != null) {
            if (s == f)
                return true;
            s = s.next;
            f = f.next.next;
            // if(f==null) return false;
        }
        return false;
    }
}
