/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null)
            return false;
        List<ListNode> lst = new ArrayList<>();
        lst.add(head);
        head = head.next;
        return method2(head, lst);
    }

    boolean method2(ListNode head, List<ListNode> lst) {
        if (lst.contains(head))
            return true;
        if (head.next == null)
            return false;
        lst.add(head);
        head = head.next;
        return method2(head, lst);
    }
}
