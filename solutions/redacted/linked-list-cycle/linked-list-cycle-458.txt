public class Solution {

    public boolean method1(ListNode head) {
        ListNode slowVehicle = head;
        ListNode fastVehicle = head;
        while (fastVehicle != null && fastVehicle.next != null) {
            slowVehicle = slowVehicle.next;
            fastVehicle = fastVehicle.next.next;
            if (slowVehicle == fastVehicle)
                return true;
        }
        return false;
    }
}
