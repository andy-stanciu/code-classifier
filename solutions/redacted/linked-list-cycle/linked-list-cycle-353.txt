/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        return f(head);
    }

    public boolean f(ListNode head) {
        if (head == null || head.next == null)
            return false;
        ListNode fp = head, sp = head;
        while (sp != null && sp.next != null) {
            fp = fp.next;
            sp = sp.next.next;
            if (fp == sp) {
                return true;
            }
        }
        return false;
    }
}
