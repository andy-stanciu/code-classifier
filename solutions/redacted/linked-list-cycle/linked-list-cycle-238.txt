/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        HashMap<ListNode, Integer> hash1 = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (hash1.containsKey(temp)) {
                //(hash1.containsKey(temp))
                return true;
            }
            hash1.put(temp, 1);
            temp = temp.next;
        }
        return false;
    }
}
