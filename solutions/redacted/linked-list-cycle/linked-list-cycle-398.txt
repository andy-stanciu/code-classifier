/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        if (head == null || head.next == null) {
            return false;
        }
        ListNode heir = head;
        ListNode turt = head;
        while (!(turt == null) && !(turt.next == null)) {
            heir = heir.next;
            turt = turt.next.next;
            if (heir == turt) {
                return true;
            }
        }
        return false;
    }
}
