/**
 * Definition for singly-linked list.
 * class ListNode {
 * int val;
 * ListNode next;
 * ListNode(int x) {
 * val = x;
 * next = null;
 * }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        //this is floyd-warshall cycle detection method
        ListNode f = head, s = head;
        if (head == null)
            return false;
        // Traverse the list
        while (f != null && f.next != null) {
            s = s.next;
            f = f.next.next;
            // Cycle detected
            if (s == f)
                return true;
        }
        return false;
    }
}
