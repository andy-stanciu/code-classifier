/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public boolean method1(ListNode head) {
        if (head == null) {
            return false;
        }
        ListNode backNode = head;
        ListNode frontNode = head.next;
        while (backNode != null && frontNode != null) {
            if (frontNode != null && backNode.equals(frontNode.next)) {
                return true;
            }
            backNode = backNode.next;
            if (frontNode.next != null) {
                frontNode = frontNode.next.next;
            } else {
                return false;
            }
        }
        return false;
    }
}
