/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        ListNode dumm = new ListNode(0);
        dumm.next = head;
        ListNode slow = dumm;
        ListNode fast = dumm;
        int c = 0;
        ListNode slow2 = dumm;
        do {
            if (fast == null || fast.next == null)
                return null;
            slow = slow.next;
            fast = fast.next.next;
        } while (slow != fast);
        do {
            slow = slow.next;
            slow2 = slow2.next;
            c++;
        } while (slow != slow2);
        return slow;
    }
}
