public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast)
                return slow;
        }
        return null;
    }

    public ListNode method2(ListNode head) {
        ListNode meet = method1(head);
        ListNode start = head;
        if (meet == null)
            return null;
        while (meet != start) {
            start = start.next;
            meet = meet.next;
        }
        return start;
    }
}
