public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast)
                break;
        }
        if (slow != fast)
            return null;
        while (head != fast) {
            // use head, it is waste now
            head = head.next;
            fast = fast.next;
        }
        return head;
    }
}
