/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        ListNode start = head, fast = head;
        while (fast != null && fast.next != null) {
            start = start.next;
            fast = fast.next.next;
            if (start == fast) {
                start = head;
                while (start != fast) {
                    start = start.next;
                    fast = fast.next;
                }
                return fast;
            }
        }
        return null;
    }
}
