/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        ListNode hare = head;
        ListNode turtle = head;
        while (hare != null && hare.next != null) {
            hare = hare.next.next;
            turtle = turtle.next;
            if (hare == turtle) {
                ListNode turtle2 = head;
                while (turtle != turtle2) {
                    turtle = turtle.next;
                    turtle2 = turtle2.next;
                }
                return turtle;
            }
        }
        return null;
    }
}
