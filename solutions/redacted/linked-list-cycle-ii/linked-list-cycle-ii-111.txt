public class Solution {

    public ListNode method1(ListNode head) {
        ListNode constant = head;
        ListNode fast = head;
        ListNode entry = head;
        if (head == null || head.next == null) {
            return null;
        }
        while (fast.next != null && fast.next.next != null) {
            constant = constant.next;
            fast = fast.next.next;
            if (constant == fast) {
                while (constant != entry) {
                    entry = entry.next;
                    constant = constant.next;
                }
                return entry;
            }
        }
        return null;
    }
}
