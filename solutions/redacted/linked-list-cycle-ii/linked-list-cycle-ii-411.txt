/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        HashSet<ListNode> hashSet = new HashSet<>();
        hashSet.add(head);
        while (head.next != null) {
            if (hashSet.contains(head.next))
                return head.next;
            else
                hashSet.add(head.next);
            head = head.next;
        }
        return null;
    }
}
