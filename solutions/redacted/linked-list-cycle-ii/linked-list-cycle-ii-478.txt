/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        HashSet<ListNode> var1 = new HashSet<>();
        while (head != null) {
            if (var1.contains(head)) {
                break;
            }
            var1.add(head);
            head = head.next;
        }
        return head;
    }
}
