public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast)
                // Cycle detected, find start of cycle
                return method2(head, slow);
        }
        return null;
    }

    private ListNode method2(ListNode current1, ListNode current2) {
        while (current1 != current2) {
            current1 = current1.next;
            current2 = current2.next;
        }
        return current1;
    }
}
