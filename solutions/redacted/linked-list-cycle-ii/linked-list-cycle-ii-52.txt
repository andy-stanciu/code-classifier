public class Solution {

    public ListNode method1(ListNode cabeza) {
        //paso#1
        ListNode lento = cabeza;
        ListNode rapido = cabeza;
        //paso#2
        while (rapido != null && rapido.next != null) {
            //paso#3
            lento = lento.next;
            rapido = rapido.next.next;
            //paso#4
            if (lento == rapido) {
                //paso#4.1
                lento = cabeza;
                //paso#4.2
                while (lento != rapido) {
                    lento = lento.next;
                    rapido = rapido.next;
                }
                //paso#4.3
                return lento;
            }
        }
        //paso#5
        return null;
    }
}
