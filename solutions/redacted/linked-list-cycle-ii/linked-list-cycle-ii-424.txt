public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return head;
        ListNode fast = head, slow = head;
        int f = 0;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                f = 1;
                break;
            }
        }
        if (f == 0)
            return null;
        fast = head;
        while (fast != slow) {
            fast = fast.next;
            slow = slow.next;
        }
        return slow;
    }
}
