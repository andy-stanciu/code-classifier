/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public static ListNode method1(ListNode head) {
        HashSet<ListNode> set = new HashSet();
        ListNode copy = head;
        while (copy != null) {
            if (!set.add(copy)) {
                return copy;
            }
            copy = copy.next;
        }
        return null;
    }
}
