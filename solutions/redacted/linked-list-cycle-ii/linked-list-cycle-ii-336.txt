/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        ListNode turtle = head;
        ListNode hare = head;
        if (head == null || head.next == null) {
            return null;
        }
        while (hare != null && hare.next != null) {
            turtle = turtle.next;
            hare = hare.next.next;
            if (turtle == hare) {
                break;
            }
        }
        if (hare == null || hare.next == null) {
            return null;
        }
        ListNode temp = head;
        while (temp != turtle) {
            turtle = turtle.next;
            temp = temp.next;
        }
        return temp;
    }
}
