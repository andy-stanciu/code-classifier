/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null)
            return null;
        if (head == head.next)
            return head;
        if (head.next == null)
            return null;
        ListNode slow = head;
        ListNode fast = head;
        int flag = 0;
        while (!(fast.next == null)) {
            slow = slow.next;
            fast = fast.next;
            if (fast.next == null)
                return null;
            else
                fast = fast.next;
            if (slow == fast) {
                flag = 1;
                break;
            }
        }
        slow = head;
        while (slow != fast) {
            if (fast.next == null)
                return null;
            slow = slow.next;
            fast = fast.next;
        }
        return slow;
    }
}
