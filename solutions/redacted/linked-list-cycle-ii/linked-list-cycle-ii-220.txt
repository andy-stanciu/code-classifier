/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode slow, ListNode fast, ListNode head) {
        if (fast == null || fast.next == null) {
            return null;
        }
        slow = slow.next;
        fast = fast.next.next;
        if (slow == fast) {
            return method2(slow, head);
        }
        return method1(slow, fast, head);
    }

    public ListNode method2(ListNode meetingPoint, ListNode head) {
        ListNode entry = head;
        while (entry != meetingPoint) {
            entry = entry.next;
            meetingPoint = meetingPoint.next;
        }
        return entry;
    }

    public ListNode method3(ListNode head) {
        if (head == null || head.next == null) {
            return null;
        }
        return method1(head, head, head);
    }
}
