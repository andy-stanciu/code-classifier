/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        int length = 0;
        ListNode fast = head;
        ListNode slow = head;
        // Find the intersection point if there is a cycle
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                length = method2(slow);
                break;
            }
        }
        // If there is no cycle, return null
        if (length == 0) {
            return null;
        }
        ListNode f = head;
        ListNode s = head;
        // Move s ahead by \'length\' nodes
        while (length > 0) {
            s = s.next;
            length--;
        }
        // Move both f and s at the same pace, they will meet at the cycle start
        while (f != s) {
            f = f.next;
            s = s.next;
        }
        return s;
    }

    public int method2(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                // Calculate the length
                ListNode temp = slow;
                int length = 0;
                do {
                    temp = temp.next;
                    length++;
                } while (temp != slow);
                return length;
            }
        }
        return 0;
    }
}
