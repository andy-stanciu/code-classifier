/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        ListNode temp = head;
        // if(head==null || head.next==null ||head.next.next==null)
        // return false;
        while (true) {
            if (fast == null || fast.next == null)
                return null;
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) {
                while (temp != slow) {
                    slow = slow.next;
                    temp = temp.next;
                }
                return slow;
            }
        }
        //   return null;
    }
}
