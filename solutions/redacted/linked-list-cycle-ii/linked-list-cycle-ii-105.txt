/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        HashSet<ListNode> s = new HashSet<>();
        int index = 0;
        while (head != null) {
            if (s.contains(head)) {
                //System.out.println(head);
                return head;
            }
            s.add(head);
            head = head.next;
        }
        return null;
    }
}
