/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        Set<ListNode> hashSet = new HashSet<>();
        ListNode dummy = head;
        while (dummy != null) {
            if (hashSet.contains(dummy)) {
                return dummy;
            } else {
                hashSet.add(dummy);
            }
            dummy = dummy.next;
        }
        return null;
    }
}
