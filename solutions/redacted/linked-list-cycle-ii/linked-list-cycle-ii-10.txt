/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        //initialize with head
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            //increace slow by 1
            slow = slow.next;
            //increase fast by 2
            fast = fast.next.next;
            if (slow == fast) {
                slow = head;
                while (slow != fast) {
                    //until slow not equal to fast move slow and fast one by one
                    slow = slow.next;
                    fast = fast.next;
                }
                //return where cycle form
                return slow;
            }
        }
        //if there is no cycle form
        return null;
    }
}
