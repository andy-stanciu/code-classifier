/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        ListNode pointer = head;
        if (head == null)
            return null;
        HashSet<ListNode> set = new HashSet<>();
        while (pointer != null) {
            if (set.contains(pointer)) {
                return pointer;
            } else {
                set.add(pointer);
            }
            pointer = pointer.next;
        }
        return pointer;
    }
}
