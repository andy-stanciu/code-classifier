/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        int count = 0;
        HashMap<ListNode, Integer> node = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (node.containsKey(temp)) {
                return temp;
            }
            node.put(temp, count++);
            temp = temp.next;
        }
        return null;
    }
}
