/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {

    public ListNode method1(ListNode head) {
        if (head == null || head.next == null)
            return null;
        Map<ListNode, Integer> mpp = new HashMap<>();
        ListNode temp = head;
        while (temp != null) {
            if (mpp.containsKey(temp)) {
                return temp;
            } else {
                mpp.put(temp, 1);
            }
            temp = temp.next;
        }
        return temp;
    }
}
