public class Solution {

    public ListNode method1(ListNode head) {
        ListNode LilToeTortoise = head, hare = head;
        do {
            if (hare == null || hare.next == null)
                return null;
            LilToeTortoise = LilToeTortoise.next;
            hare = hare.next.next;
        } while (LilToeTortoise != hare);
        while (LilToeTortoise != head) {
            LilToeTortoise = LilToeTortoise.next;
            head = head.next;
        }
        return head;
    }
}
