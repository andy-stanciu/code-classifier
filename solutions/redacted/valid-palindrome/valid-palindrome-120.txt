class Solution {

    private static final String var1 = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";

    public boolean method1(String s) {
        String cleanString = Arrays.stream(s.split("")).filter(i -> var1.contains(i)).map(String::toLowerCase).collect(Collectors.joining());
        return cleanString.equals(new StringBuilder(cleanString).reverse().toString());
    }
}
