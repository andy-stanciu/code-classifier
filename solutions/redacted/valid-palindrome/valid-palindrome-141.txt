class Solution {

    public boolean method1(String s) {
        int len = s.length();
        int i = 0;
        int j = len - 1;
        while (i < j) {
            int beg = s.charAt(i);
            int end = s.charAt(j);
            if (beg >= 65 && beg <= 90) {
                beg += 32;
            } else if (!(beg >= 97 && beg <= 122) && !(beg >= 48 && beg <= 57)) {
                i++;
                continue;
            }
            if (end >= 65 && end <= 90) {
                end += 32;
            } else if (!(end >= 97 && end <= 122) && !(end >= 48 && end <= 57)) {
                j--;
                continue;
            }
            if (end != beg) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
