class Solution {

    public boolean method1(String s) {
        // Use StringBuilder for efficient string manipulation
        StringBuilder filteredString = new StringBuilder();
        // Filter only alphabetic and numeric characters
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (Character.isLetterOrDigit(ch)) {
                filteredString.append(Character.toLowerCase(ch));
            }
        }
        // Reverse the filtered string
        String reversedString = filteredString.reverse().toString();
        // Restore the original filtered string
        filteredString.reverse();
        // Check if the filtered string is equal to its reverse
        return filteredString.toString().equals(reversedString);
    }
}
