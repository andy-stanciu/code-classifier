class Solution {

    public boolean method1(String s) {
        String b = "";
        for (int i = 0; i < s.length(); i++) {
            if (Character.isLetterOrDigit(s.charAt(i)))
                b += s.charAt(i);
        }
        b = b.toLowerCase();
        int i = 0;
        int j = b.length() - 1;
        while (i <= j) {
            if (b.charAt(i) != b.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
