class Solution {

    public boolean method1(String s) {
        // StringBuilder to hold the cleaned and lowercase string
        StringBuilder sb = new StringBuilder();
        // Pattern to match alphabetic characters (both uppercase and lowercase)
        Pattern p1 = Pattern.compile("[A-Za-z0-9]");
        // Pattern to match uppercase alphabetic characters
        Pattern p2 = Pattern.compile("[A-Z]");
        // Iterate through each character of the input string
        for (int i = 0; i < s.length(); i++) {
            Matcher m = p1.matcher(Character.toString(s.charAt(i)));
            // If the character is a letter (A-Za-z)
            if (m.matches()) {
                Matcher c = p2.matcher(Character.toString(s.charAt(i)));
                if (c.matches()) {
                    // If the character is uppercase, convert it to lowercase
                    char ch = Character.toLowerCase(s.charAt(i));
                    sb.append(ch);
                } else {
                    // If the character is already lowercase, append it as-is
                    sb.append(s.charAt(i));
                }
            }
        }
        // Convert the cleaned StringBuilder to a string
        String str = sb.toString();
        // Create the reverse of the cleaned string
        String reversedStr = new StringBuilder(str).reverse().toString();
        // Compare the cleaned string with its reverse
        return str.equals(reversedStr);
    }
}
