class Solution {

    public boolean method1(String s) {
        // this will convert the upper char to lower and remove all the character that are not a to z and 0 to 9
        s = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        //if string is null or empty it returns true
        if (s == null || s.length() == 0) {
            return true;
        }
        for (int i = 0; i <= s.length() / 2; i++) {
            int start = s.charAt(i);
            int end = s.charAt(s.length() - 1 - i);
            if (start != end) {
                return false;
            }
        }
        return true;
    }
}
