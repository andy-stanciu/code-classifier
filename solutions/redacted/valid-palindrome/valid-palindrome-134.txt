class Solution {

    public boolean method1(String s) {
        String n = s.replaceAll("[^a-zA-Z0-9]", "");
        String k = n.toLowerCase();
        for (int i = 0, j = k.length() - 1; i < j; i++, j--) {
            if (k.charAt(i) != k.charAt(j)) {
                return false;
            }
        }
        return true;
    }
}
