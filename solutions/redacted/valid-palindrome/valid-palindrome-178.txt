class Finding {

    public boolean check(String s, int len, int i) {
        if (i >= (len / 2))
            return true;
        if (Character.toLowerCase(s.charAt(i)) != Character.toLowerCase(s.charAt(len - i - 1)))
            return false;
        return check(s, len, i + 1);
    }
}

class Solution {

    public boolean method1(String s) {
        if (s.isEmpty())
            return true;
        s = s.replaceAll("[^a-zA-Z0-9]", "");
        int len = s.length(), i = 0;
        Finding obj = new Finding();
        return obj.check(s, len, i);
    }
}
