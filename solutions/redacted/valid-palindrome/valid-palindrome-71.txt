class Solution {

    public boolean method1(String s) {
        // creating a mutable string object
        // Jisme hogi apni cleaned string
        StringBuilder strr = new StringBuilder();
        //cleaning the string
        for (char c : s.toCharArray()) {
            if (Character.isLetterOrDigit(c)) {
                strr.append(Character.toLowerCase(c));
            }
        }
        String str = strr.toString();
        //Creating Another mutable to use
        // .reverse() fxn of StringBuilder class
        StringBuilder st = new StringBuilder(str);
        // reverseing the string
        st.reverse();
        // using .equals() method
        return str.equals(st.toString());
    }
}
