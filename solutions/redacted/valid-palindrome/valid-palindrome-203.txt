class Solution {

    public boolean method1(String s) {
        String s2 = s.toLowerCase().replaceAll("[^A-Za-z0-9]", "");
        int i = 0;
        int j = s2.length() - 1;
        while (i <= j) {
            if (s2.charAt(i) != s2.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
