class Solution {

    public boolean method1(String s) {
        if (s.length() == 0) {
            return true;
        }
        StringBuilder sb = new StringBuilder();
        String res = "";
        String[] r = s.split(" ");
        for (int i = 0; i < r.length; i++) {
            //Gives only Alphanumeric characters(numbers,characters)
            res += r[i].replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        }
        sb.append(res);
        return res.equals(sb.reverse().toString());
    }
}
