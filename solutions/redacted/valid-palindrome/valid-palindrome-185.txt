class Solution {

    public boolean method1(String s) {
        String a = s.toLowerCase();
        int start = 0;
        int end = s.length() - 1;
        while (start <= end) {
            char cf = a.charAt(start);
            char cl = a.charAt(end);
            if (!Character.isLetterOrDigit(cf)) {
                start++;
            } else if (!Character.isLetterOrDigit(cl)) {
                end--;
            } else {
                if (cf != cl) {
                    return false;
                }
                start++;
                end--;
            }
        }
        return true;
    }
}
