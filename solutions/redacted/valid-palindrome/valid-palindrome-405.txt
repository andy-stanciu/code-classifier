class Solution {

    public boolean method1(String s) {
        if (s.isBlank()) {
            return true;
        }
        s = s.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        for (int i = 0; i < s.length(); i++) {
            // 2
            if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
