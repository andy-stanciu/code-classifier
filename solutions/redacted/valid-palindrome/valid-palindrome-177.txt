class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int i = 0;
        int j = s.length() - 1;
        if (s.length() == 0 || s.length() == 1)
            return true;
        while (i != j && j > 0 && i < s.length() - 1) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            } else {
                i = i + 1;
                j = j - 1;
            }
        }
        return true;
    }
}
