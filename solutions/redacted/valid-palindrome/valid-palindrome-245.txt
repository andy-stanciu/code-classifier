class Solution {

    public boolean method1(String s) {
        String lowerCasePhrase = s.toLowerCase();
        lowerCasePhrase = lowerCasePhrase.replaceAll("[^a-zA-Z0-9]", "");
        String reversedPhrase = new StringBuilder(lowerCasePhrase).reverse().toString();
        return lowerCasePhrase.equals(reversedPhrase);
    }
}
