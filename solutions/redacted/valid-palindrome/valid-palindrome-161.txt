class Solution {

    public boolean method1(String s) {
        StringBuilder sb = new StringBuilder("");
        if (s.equals(" ")) {
            return true;
        }
        sb.append(s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase());
        for (int i = 0; i < sb.length() / 2; i++) {
            if (sb.charAt(i) != sb.charAt(sb.length() - 1 - i)) {
                return false;
            }
        }
        return true;
    }
}
