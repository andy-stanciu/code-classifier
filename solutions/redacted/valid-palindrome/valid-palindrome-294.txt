class Solution {

    public boolean method1(String str) {
        String lowerCase = str.toLowerCase();
        int l = 0, r = lowerCase.length() - 1;
        while (l < r) {
            char leftChar = lowerCase.charAt(l);
            char rightChar = lowerCase.charAt(r);
            while (l < lowerCase.length() - 1 && !isAlpha(leftChar)) {
                l++;
                leftChar = lowerCase.charAt(l);
            }
            while (r > 0 && !isAlpha(rightChar)) {
                r--;
                rightChar = lowerCase.charAt(r);
            }
            if (l < r && rightChar != leftChar) {
                return false;
            }
            l++;
            r--;
        }
        return true;
    }

    private boolean isAlpha(char currChar) {
        if ((97 <= currChar && currChar <= 122) || (48 <= currChar && currChar <= 57))
            return true;
        return false;
    }
}
