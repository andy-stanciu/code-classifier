class Solution {

    public boolean method1(String s) {
        int j = s.length() - 1, i = 0, x, y;
        while (i < j) {
            x = s.charAt(i);
            y = s.charAt(j);
            if (!((x >= 97 && x <= 122) || (x >= 65 && x <= 90) || (x >= 48 && x <= 57))) {
                i += 1;
                continue;
            } else if (!((y >= 97 && y <= 122) || (y >= 65 && y <= 90) || (y >= 48 && y <= 57))) {
                j -= 1;
                continue;
            } else {
                if (x >= 65 && x <= 90)
                    x += 32;
                if (y >= 65 && y <= 90)
                    y += 32;
                if (x != y)
                    return false;
            }
            i++;
            j--;
        }
        return Boolean.TRUE;
    }
}
