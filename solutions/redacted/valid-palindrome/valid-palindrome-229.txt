class Solution {

    public boolean method1(String s) {
        String copyPal = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int left = 0;
        int right = copyPal.length() - 1;
        while (left < right) {
            if (copyPal.charAt(left) != copyPal.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
