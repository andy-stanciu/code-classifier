class Solution {

    public boolean method1(String s) {
        String str = s.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        return check(str, 0);
    }

    public boolean check(String str, int i) {
        int n = str.length();
        if (i >= n / 2)
            return true;
        if (str.charAt(i) != str.charAt(n - i - 1))
            return false;
        return check(str, i + 1);
    }
}
