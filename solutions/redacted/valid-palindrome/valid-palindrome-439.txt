class Solution {

    public boolean method1(String s) {
        List<String> str = s.chars().filter(Character::isLetterOrDigit).mapToObj(c -> String.valueOf((char) c).toLowerCase()).toList();
        return IntStream.range(0, str.size() / 2).allMatch(i -> str.get(i).equals(str.get(str.size() - 1 - i)));
    }
}
