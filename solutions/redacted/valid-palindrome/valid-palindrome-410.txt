class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase().replace(" ", "").replaceAll("[^a-zA-Z0-9]", "");
        System.out.println(s);
        char[] arr = s.toCharArray();
        //pointer that starts at the same char at the ends of the string
        int orig_string = 0;
        int reverse_string = arr.length - 1;
        // Loop until the two pointers meet in the middle
        while (orig_string < reverse_string) {
            // Compare characters from the start and end
            if (arr[orig_string] != arr[reverse_string]) {
                // Return false if they don\'t match
                return false;
            }
            // Move the pointers closer to the center
            orig_string++;
            reverse_string--;
        }
        // If all characters match, return true
        return true;
    }
}
