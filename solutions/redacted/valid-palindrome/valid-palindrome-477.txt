class Solution {

    public boolean method1(String s) {
        String str = "";
        for (char c : s.toCharArray()) {
            if (Character.isLetter(c) || Character.isDigit(c))
                str += c;
        }
        str = str.toUpperCase();
        System.out.print(str);
        int k = str.length() - 1;
        for (int i = 0; i < str.length() / 2; i++) {
            if (str.charAt(i) != str.charAt(k))
                return false;
            k--;
        }
        return true;
    }
}
