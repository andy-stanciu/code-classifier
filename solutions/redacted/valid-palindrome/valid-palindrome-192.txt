class Solution {

    public boolean method1(String s) {
        String ans = s.replaceAll("[^A-Za-z0-9]", "").toLowerCase();
        int i = 0;
        int j = ans.length() - 1;
        while (i < j) {
            if (ans.charAt(i) != ans.charAt(j))
                return false;
            i++;
            j--;
        }
        return true;
    }
}
