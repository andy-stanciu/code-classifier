class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase().trim().replaceAll(" ", "").replaceAll("[^a-zA-Z0-9]", "");
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
                return false;
            }
        }
        return true;
    }
}
