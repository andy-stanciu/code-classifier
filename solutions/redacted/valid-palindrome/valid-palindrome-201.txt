class Solution {

    int i = 0;

    public boolean method1(String s) {
        // Clean the string by removing non-alphanumeric characters and convert to lowercase
        s = s.replaceAll("[^A-Za-z0-9]", "").toLowerCase();
        // Now check for palindrome in the cleaned string
        return method2(s);
    }

    private boolean method2(String s) {
        // Base Condition: If i exceeds half of the string, we return true.
        if (i >= s.length() / 2) {
            return true;
        }
        // If start and end characters don\'t match, return false.
        if (s.charAt(i) != s.charAt(s.length() - i - 1)) {
            return false;
        }
        // Increment the index i and check next pair of characters.
        i++;
        // Recursive call
        return method2(s);
    }
}
