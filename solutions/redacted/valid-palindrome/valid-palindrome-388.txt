class Solution {

    public boolean method1(String s) {
        String res = "";
        for (int i = 0; i < s.length(); i++) {
            if (Character.isLetterOrDigit(s.charAt(i))) {
                res = res + Character.toLowerCase(s.charAt(i));
            }
        }
        int left = 0;
        int right = res.length() - 1;
        while (left < right) {
            if (res.charAt(left) == res.charAt(right)) {
                left++;
                right--;
            } else
                return false;
        }
        return true;
    }
}
