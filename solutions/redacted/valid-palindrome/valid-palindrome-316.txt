class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        if (s.length() == 0) {
            return true;
        }
        int i = 0, j = s.length() - 1;
        while (i <= j) {
            // skipping non-alphanumeric characters from start
            while ((s.charAt(i) < 48 || (s.charAt(i) > 57 && s.charAt(i) < 97) || s.charAt(i) > 122) && i < j) {
                i++;
            }
            // skipping non-alphanumeric characters from end
            while ((s.charAt(j) < 48 || (s.charAt(j) > 57 && s.charAt(j) < 97) || s.charAt(j) > 122) && i < j) {
                j--;
            }
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
