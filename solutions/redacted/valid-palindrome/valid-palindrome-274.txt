class Solution {

    public boolean method1(String s) {
        String str = (s.replaceAll("[\\\\W\\\\p{Punct}]", "")).toLowerCase();
        for (int i = 0, j = str.length() - 1; i <= j; i++, j--) {
            if (str.charAt(i) != str.charAt(j))
                return false;
        }
        return true;
    }
}
