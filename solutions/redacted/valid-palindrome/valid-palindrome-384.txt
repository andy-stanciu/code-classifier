class Solution {

    public boolean method1(String s) {
        String p = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int n = p.length();
        for (int i = 0; i < n / 2; i++) {
            if (p.charAt(i) != p.charAt(n - 1 - i)) {
                return false;
            }
        }
        return true;
    }
}
