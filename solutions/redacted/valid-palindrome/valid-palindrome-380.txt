class Solution {

    public boolean method1(String s) {
        s = s.trim().toLowerCase().replaceAll("[^a-z0-9]", "");
        return ispalrecursive(s, 0, s.length() - 1);
    }

    private boolean ispalrecursive(String s, int left, int right) {
        if (left >= right) {
            // cond for breaking the rec func i.e. our base condition
            return true;
        }
        if (s.charAt(left) != s.charAt(right)) {
            return false;
        }
        return ispalrecursive(s, left + 1, right - 1);
    }
}
