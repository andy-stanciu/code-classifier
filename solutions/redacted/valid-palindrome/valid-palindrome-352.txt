class Solution {

    public boolean method1(String s) {
        String new_s = s.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        int l = 0;
        int r = new_s.length() - 1;
        if (new_s.length() == 0)
            return true;
        while (l < r) {
            if (new_s.charAt(l) != new_s.charAt(r)) {
                return false;
            }
            l++;
            r--;
        }
        return true;
    }
}
