class Solution {

    public boolean method1(String s) {
        String n = s.replaceAll("[^A-Za-z0-9]", "");
        String ch = n.toLowerCase();
        StringBuilder r = new StringBuilder();
        r.append(ch);
        r.reverse();
        String str = r.toString();
        if (ch.equals(str))
            return true;
        else
            return false;
    }
}
