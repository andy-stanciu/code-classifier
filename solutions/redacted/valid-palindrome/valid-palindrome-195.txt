class Solution {

    public boolean method1(String s) {
        int i = 0;
        int j = s.length() - 1;
        s = s.toLowerCase();
        while (i < j) {
            //making sure variables are pointing at right characters.
            while (i < j && !Character.isLetterOrDigit(s.charAt(i))) {
                i++;
            }
            // here again making sure
            while (i < j && !Character.isLetterOrDigit(s.charAt(j))) {
                j--;
            }
            //compare
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            //update your pointers
            i++;
            j--;
        }
        return true;
    }
}
