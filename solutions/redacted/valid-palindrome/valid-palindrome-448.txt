class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        // Remove non-alphanumeric characters and spaces
        s = s.replaceAll("[^a-z0-9]", "");
        // Convert to char array
        char[] ch = s.toCharArray();
        int start = 0;
        int end = ch.length - 1;
        while (start <= end) {
            if (ch[start] != ch[end]) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
}
