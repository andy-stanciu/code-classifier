class Solution {

    public boolean method1(String s) {
        char[] s1 = s.toCharArray();
        String s2 = "";
        for (int i = 0; i <= s1.length - 1; i++) {
            char ch = s.charAt(i);
            if (Character.isAlphabetic(ch) || Character.isDigit(ch)) {
                s2 = s2 + ch;
            }
        }
        s2 = s2.toLowerCase();
        String s3 = "";
        for (int i = s2.length() - 1; i >= 0; i--) {
            s3 = s3 + s2.charAt(i);
        }
        if (s2.equals(s3)) {
            return true;
        } else {
            return false;
        }
    }
}
