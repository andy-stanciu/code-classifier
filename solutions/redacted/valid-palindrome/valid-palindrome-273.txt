class Solution {

    public boolean method1(String s) {
        // Initialize an empty string to store the filtered characters
        String filtered = "";
        // Loop through the input string and add only alphanumeric characters (ignore spaces, symbols, etc.)
        for (int i = 0; i < s.length(); i++) {
            char currentChar = s.charAt(i);
            if (Character.isLetterOrDigit(currentChar)) {
                // Convert to lowercase to ensure case-insensitive comparison
                filtered += Character.toLowerCase(currentChar);
            }
        }
        // Initialize an empty string to store the reversed version of the filtered string
        String reversed = "";
        // Reverse the filtered string by looping from the end to the beginning
        for (int i = filtered.length() - 1; i >= 0; i--) {
            reversed += filtered.charAt(i);
        }
        // Compare the filtered string and the reversed string
        // If they are the same, it\'s a palindrome
        return filtered.equals(reversed);
    }
}
