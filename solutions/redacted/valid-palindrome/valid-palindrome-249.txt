class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        char[] arr = s.toCharArray();
        int i = 0;
        int j = arr.length - 1;
        while (i < j) {
            if (!Character.isLetterOrDigit(arr[i])) {
                i++;
            } else if (!Character.isLetterOrDigit(arr[j])) {
                j--;
            } else if (arr[i] != arr[j]) {
                return false;
            } else {
                i++;
                j--;
            }
        }
        return true;
    }
}
