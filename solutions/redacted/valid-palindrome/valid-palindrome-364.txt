class Solution {

    public boolean method1(String s) {
        String removeWhiteSpace = "";
        for (char ch : s.toCharArray()) {
            if (Character.isLetterOrDigit(ch)) {
                removeWhiteSpace = removeWhiteSpace + ch;
            }
        }
        String reverseRemoveWhiteSpace = "";
        for (int i = removeWhiteSpace.length() - 1; i >= 0; i--) {
            reverseRemoveWhiteSpace = reverseRemoveWhiteSpace + removeWhiteSpace.charAt(i);
        }
        if (removeWhiteSpace.toLowerCase().equals(reverseRemoveWhiteSpace.toLowerCase())) {
            return true;
        } else {
            return false;
        }
    }
}
