class Solution {

    public boolean method1(String s) {
        /*String s1=s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();//replaces all the special character  
        StringBuilder sb = new StringBuilder(s1);
        return s1.equals(sb.reverse().toString());
        */
        //using two pointers
        String s_new = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int left = 0;
        int right = s_new.length() - 1;
        while (left < right) {
            if (s_new.charAt(left) != s_new.charAt(right))
                return false;
            left++;
            right--;
        }
        return true;
    }
}
