class Solution {

    public boolean method1(String s) {
        if (s.isEmpty()) {
            return true;
        }
        s = s.toLowerCase();
        String str = "abcdefghijklmnopqrstuvwxyz0123456789";
        String str1 = "";
        for (int i = 0; i < s.length(); i++) {
            char a = s.charAt(i);
            String s1 = String.valueOf(a);
            if (str.contains(s1)) {
                str1 = str1 + s1;
            }
        }
        return reverse(str1).equals(str1);
    }

    public String reverse(String str1) {
        String s2 = "";
        for (int i = str1.length() - 1; i >= 0; i--) {
            s2 = s2 + str1.charAt(i);
        }
        return s2;
    }
}
