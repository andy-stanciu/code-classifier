class Solution {

    public boolean method1(String s) {
        int n = s.length();
        if (s.equals(" ")) {
            return true;
        }
        s = s.toLowerCase();
        StringBuilder res = new StringBuilder();
        for (int i = 0; i < n; i++) {
            int a = s.charAt(i);
            if (a >= 48 && a <= 57) {
                res.append(s.charAt(i));
            } else if ((a >= 97 && a <= 122) || (a >= 65 && a <= 90)) {
                res.append(s.charAt(i));
            }
        }
        StringBuilder ans = new StringBuilder();
        ans.append(res);
        res.reverse();
        System.out.println(ans);
        System.out.println(res);
        boolean areEqual = res.toString().equals(ans.toString());
        if (areEqual) {
            return true;
        }
        return false;
    }
}
