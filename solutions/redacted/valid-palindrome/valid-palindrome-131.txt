class Solution {

    public boolean method1(String s) {
        int fIdx = 0;
        int lIdx = s.length() - 1;
        while (fIdx < lIdx) {
            // Move fIdx forward if it\'s not alphanumeric
            if (!Character.isLetterOrDigit(s.charAt(fIdx))) {
                fIdx++;
            } else // Move lIdx backward if it\'s not alphanumeric
            if (!Character.isLetterOrDigit(s.charAt(lIdx))) {
                lIdx--;
            } else // Compare characters
            {
                if (Character.toLowerCase(s.charAt(fIdx)) != Character.toLowerCase(s.charAt(lIdx))) {
                    // Not a palindrome
                    return false;
                }
                fIdx++;
                lIdx--;
            }
        }
        return true;
    }
}
