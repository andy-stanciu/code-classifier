class Solution {

    public boolean method1(String s) {
        int l = 0;
        int r = s.length() - 1;
        while (l < r) {
            char lc = s.charAt(l);
            char rc = s.charAt(r);
            if (!Character.isLetterOrDigit(lc)) {
                l++;
                continue;
            }
            if (!Character.isLetterOrDigit(rc)) {
                r--;
                continue;
            }
            if (!(String.valueOf(lc).toLowerCase().equals(String.valueOf(rc).toLowerCase())))
                return false;
            l++;
            r--;
        }
        return true;
    }
}
