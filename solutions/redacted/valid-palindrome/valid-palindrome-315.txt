class Solution {

    public boolean method1(String s) {
        String cleaned = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int n = cleaned.length() - 1;
        for (int i = 0; i < cleaned.length() / 2; i++) {
            if (cleaned.charAt(i) != cleaned.charAt(n)) {
                return false;
            }
            n--;
        }
        return true;
    }
}
