class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        StringBuilder ans2 = new StringBuilder(s);
        ans2.reverse();
        String ans3 = ans2.toString();
        if (s.equals(ans3)) {
            return true;
        }
        return false;
    }
}
