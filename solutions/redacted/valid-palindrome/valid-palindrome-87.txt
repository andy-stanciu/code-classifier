public class Solution {

    public static boolean method1(String s) {
        // paso#1
        s = s.toLowerCase();
        // paso#2
        int i = 0;
        int j = s.length() - 1;
        // paso#3
        while (i < j) {
            // paso#4
            if (get(s.charAt(i))) {
                // paso#5
                while (j > i) {
                    // paso#6
                    boolean es_alfanumerico = get(s.charAt(j));
                    // paso#6.1
                    if (!es_alfanumerico) {
                        j--;
                    } else // paso#6.2
                    {
                        // paso#6.2.1
                        if (s.charAt(i) != s.charAt(j)) {
                            return false;
                        }
                        // paso#6.2.2
                        if (s.charAt(i) == s.charAt(j)) {
                            break;
                        }
                    }
                }
                // paso 7
                j--;
            }
            // paso#8
            i++;
        }
        // paso#9
        return true;
    }

    private static boolean get(char c) {
        String string = "abcdefghijklmnopqrstuvwxyz0123456789";
        return string.indexOf(c) != -1;
    }
}
