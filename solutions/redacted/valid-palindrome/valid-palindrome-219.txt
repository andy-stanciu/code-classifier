class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
        int x = 0;
        int y = s.length() - 1;
        while (x < y) {
            if (s.charAt(x) != s.charAt(y)) {
                return false;
            }
            x++;
            y--;
        }
        return true;
    }
}
