public class Solution {

    public boolean method1(String s) {
        String newString = s.toLowerCase().replaceAll("[^a-zA-Z0-9]", "");
        Stack<Character> stack = new Stack<>();
        StringBuilder originalBuilder = new StringBuilder();
        StringBuilder reversedBuilder = new StringBuilder();
        for (var element : newString.toCharArray()) {
            stack.push(element);
            originalBuilder.append(element);
        }
        while (!stack.isEmpty()) reversedBuilder.append(stack.pop());
        return originalBuilder.toString().contentEquals(reversedBuilder.toString());
    }
}
