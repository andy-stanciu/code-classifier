class Solution {

    public boolean method1(String s) {
        String st = s.toLowerCase();
        StringBuilder filter = new StringBuilder();
        for (int i = 0; i < st.length(); i++) {
            char ch = st.charAt(i);
            if (Character.isLetterOrDigit(ch)) {
                filter.append(ch);
            }
        }
        String ori = filter.toString();
        StringBuilder temp = new StringBuilder(ori).reverse();
        String rev = temp.toString();
        if (ori.equals(rev)) {
            return true;
        } else {
            return false;
        }
    }
}
