class Solution {

    public boolean method1(String str) {
        StringBuilder clean = new StringBuilder();
        for (char i : str.toCharArray()) {
            if (Character.isLetterOrDigit(i)) {
                clean.append(Character.toLowerCase(i));
            }
        }
        String cleaned = clean.toString();
        String rev = clean.reverse().toString();
        return cleaned.equals(rev);
    }
}
