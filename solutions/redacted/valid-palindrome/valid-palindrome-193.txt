class Solution {

    public boolean method1(String s) {
        s = s.replaceAll("[^\\\\p{IsAlphabetic}\\\\p{IsDigit}]", "");
        s = s.toLowerCase();
        for (int i = 0, j = s.length() - 1; i < (s.length() / 2) && j >= 0; i++, j--) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
        }
        return true;
    }
}
