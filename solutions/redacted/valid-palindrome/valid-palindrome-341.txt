class Solution {

    public boolean method1(String s) {
        s = s.toLowerCase();
        if (s.isEmpty()) {
            return true;
        }
        int i = 0;
        int j = s.length() - 1;
        char right, left;
        while (i <= j) {
            left = s.charAt(i);
            right = s.charAt(j);
            if (!Character.isLetterOrDigit(left)) {
                i++;
            } else if (!Character.isLetterOrDigit(right)) {
                j--;
            } else {
                if (left != right) {
                    return false;
                }
                i++;
                j--;
            }
        }
        return true;
    }
}
