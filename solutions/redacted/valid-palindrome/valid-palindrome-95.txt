class Solution {

    public boolean method1(String s) {
        if (s.length() == 0)
            return true;
        //converts all uppercase to lowercase;
        s = s.toLowerCase();
        String str = "";
        for (int i = 0; i < s.length(); i++) {
            //
            int x = (int) s.charAt(i);
            if (x >= 65 && x <= 90 || x >= 97 && x <= 122 || x >= 48 && x <= 57) {
                str = str + s.charAt(i);
            }
        }
        //check for palindrome :-
        int i = 0;
        int j = str.length() - 1;
        while (i <= j) {
            if (str.charAt(i) == str.charAt(j)) {
                i++;
                j--;
            } else {
                return false;
            }
        }
        return true;
    }
}
