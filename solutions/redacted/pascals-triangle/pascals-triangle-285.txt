class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        int s = 1;
        int n = numRows;
        for (int row = 1; row <= n; row++) {
            s = 1;
            List<Integer> res = new ArrayList<>();
            for (int col = 0; col < row; col++) {
                if (col == 0 || col == row - 1) {
                    res.add(1);
                } else {
                    s = (s * (row - col));
                    s = s / col;
                    res.add(s);
                }
            }
            ans.add(res);
        }
        return ans;
    }
}
