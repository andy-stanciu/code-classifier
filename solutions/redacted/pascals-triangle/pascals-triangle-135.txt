class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<Integer> list = new ArrayList<>(numRows);
        for (int i = 0; i < numRows; i++) list.add(1);
        if (numRows == 1) {
            List<List<Integer>> ans = new ArrayList<>();
            ans.add(list);
            return ans;
        }
        List<List<Integer>> ans = generate(numRows - 1);
        List<Integer> prevList = ans.get(numRows - 2);
        for (int i = 1; i < numRows - 1; i++) {
            list.set(i, (prevList.get(i - 1) + prevList.get(i)));
        }
        ans.add(list);
        return ans;
    }
}
