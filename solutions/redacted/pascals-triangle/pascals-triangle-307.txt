class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> x = new ArrayList<>();
        x.add(new ArrayList<>());
        x.get(0).add(1);
        if (numRows > 1) {
            for (int i = 0; i < numRows - 1; i++) {
                List<Integer> y = new ArrayList<>();
                y.add(x.get(i).get(0));
                int j = 1;
                while (j <= (x.get(i).size() - 1)) {
                    int f = x.get(i).get(j - 1) + x.get(i).get(j);
                    y.add(f);
                    j++;
                }
                y.add(x.get(i).get(0));
                System.out.println(y);
                x.add(y);
            }
        }
        return x;
    }
}
