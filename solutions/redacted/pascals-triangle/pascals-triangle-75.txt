class Solution {

    public List<List<Integer>> generate(int numRows) {
        //Initializing the nested list
        List<List<Integer>> mainList = new ArrayList<List<Integer>>();
        //Loop will run numRows times
        for (int i = 0; i < numRows; i++) {
            //Initializing the current list
            List<Integer> currList = new ArrayList<>();
            for (int j = 0; j <= i; j++) {
                //If the element is at first or last index, add 1
                if (j == 0 || j == i) {
                    currList.add(1);
                    continue;
                }
                //Else add sum of 2 consecutive elements of the previous list into the list
                currList.add(mainList.get(i - 1).get(j - 1) + mainList.get(i - 1).get(j));
            }
            //Adding the list into the main List
            mainList.add(currList);
        }
        return mainList;
    }
}
