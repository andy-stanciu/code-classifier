class Solution {

    public List<List<Integer>> generate(int numRows) {
        // To store the final result
        List<List<Integer>> result = new ArrayList<>();
        // Traversing until the numRows
        for (int i = 0; i < numRows; i++) {
            // Pointers to access the elements which will be summed up and stored
            int left = 0, right = 1;
            // To store the nums of a row
            List<Integer> nums = new ArrayList<>();
            // Traversing through a row
            for (int j = 0; j <= i; j++) {
                // If at ends of the row then add 1 and if not then add the result from previous row using pointers
                if (j == 0 || j == i)
                    nums.add(1);
                else
                    nums.add(result.get(i - 1).get(left++) + result.get(i - 1).get(right++));
            }
            // Add the row\'s num values to the final list
            result.add(nums);
        }
        return result;
    }
}
