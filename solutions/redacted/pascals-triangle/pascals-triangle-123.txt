class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> var1 = new ArrayList<List<Integer>>();
        if (numRows < 1) {
            return var1;
        }
        for (int i = 0; i < numRows; i++) {
            List<Integer> row = new ArrayList<Integer>();
            for (int j = 0; j < i + 1; j++) {
                if (j == 0 || j == i) {
                    row.add(1);
                } else {
                    row.add(var1.get(i - 1).get(j - 1) + var1.get(i - 1).get(j));
                }
            }
            var1.add(row);
        }
        return var1;
    }
}
