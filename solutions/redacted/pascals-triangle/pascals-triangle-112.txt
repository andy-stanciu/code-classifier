class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> ans = new ArrayList<>();
        int count = 0;
        while (count != numRows) {
            List<Integer> answ = new ArrayList<>();
            for (int i = 0; i <= count; i++) {
                if (i == 0 || i == count)
                    answ.add(1);
                else {
                    int y = ans.get(count - 1).get(i - 1) + ans.get(count - 1).get(i);
                    answ.add(y);
                }
            }
            ans.add(answ);
            count++;
        }
        return ans;
    }
}
