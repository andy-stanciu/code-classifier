class Solution {

    private List<Integer> pascalRowGenerator(int rowNo) {
        List<Integer> l = new ArrayList<>();
        l.add(1);
        int ans = 1;
        for (int i = 1; i < rowNo; i++) {
            ans *= (rowNo - i);
            ans /= i;
            l.add(ans);
        }
        return l;
    }

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> al = new ArrayList();
        for (int i = 1; i <= numRows; i++) al.add(pascalRowGenerator(i));
        return al;
    }
}
