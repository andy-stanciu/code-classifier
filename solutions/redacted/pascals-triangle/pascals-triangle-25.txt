class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> traingle = new ArrayList<>();
        for (int r = 0; r < numRows; r++) {
            List<Integer> row = new ArrayList<>();
            // Start with the first element for each row
            long ans = 1;
            for (int c = 0; c <= r; c++) {
                // Add the current element to the row
                row.add((int) ans);
                ans = ans * (r - c);
                ans = ans / (c + 1);
            }
            // Add the completed row to the triangle
            traingle.add(row);
        }
        // Return the complete triangle
        return traingle;
    }
}
