    // Helper method to generate a specific row of Pascal\'s Triangle
    public static List<Integer> generateRow(int row) {
        // Initialize the first element as 1
        long ans = 1;
        List<Integer> ansRow = new ArrayList<>();
        // Add the first element of the row
        ansRow.add(1);
        // Calculate the rest of the elements
        for (int col = 1; col < row; col++) {
            // Update the numerator
            ans = ans * (row - col);
            // Update the denominator
            ans = ans / col;
            // Add the computed element to the row
            ansRow.add((int) ans);
        }
        // Return the generated row
        return ansRow;
    }

    // 2nd Approach: Optimal Approach - Time Complexity = O(numRows^2), Space Complexity = O(numRows^2)
    public static List<List<Integer>> generate(int numRows) {
        // List to store final result
        List<List<Integer>> ans = new ArrayList<>();
        // Store the entire Pascal\'s Triangle row by row
        for (int row = 1; row <= numRows; row++) {
            ans.add(generateRow(row));
        }
        // Return the complete Pascal\'s Triangle
        return ans;
    }
}
