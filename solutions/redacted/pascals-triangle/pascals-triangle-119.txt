class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        List<Integer> row, pre = null;
        for (int i = 0; i < numRows; i++) {
            row = new ArrayList<Integer>();
            for (int j = 0; j <= i; ++j) if (//check for first and last element
            j == 0 || j == i) {
                //adding 1 in list
                row.add(1);
            } else // for rest of the indexes adding the previous above values in the triangle
            {
                row.add(pre.get(j - 1) + pre.get(j));
            }
            //saving previous row
            pre = row;
            res.add(row);
        }
        return res;
    }
}
