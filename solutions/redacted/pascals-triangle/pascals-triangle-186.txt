class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> result = new ArrayList<>();
        for (int i = 0; i < numRows; i++) {
            // Create a new row with \'i+1\' elements initialized to 1
            List<Integer> currentRow = new ArrayList<>();
            // Add 1 to the first position
            for (int j = 0; j <= i; j++) {
                if (j == 0 || j == i) {
                    // First and last element of each row is 1
                    currentRow.add(1);
                } else {
                    // Add the sum of the two numbers directly above it
                    currentRow.add(result.get(i - 1).get(j) + result.get(i - 1).get(j - 1));
                }
            }
            // Add the current row to the result
            result.add(currentRow);
        }
        return result;
    }
}
