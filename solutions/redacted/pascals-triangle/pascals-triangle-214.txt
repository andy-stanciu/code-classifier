class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> mat = new ArrayList<>();
        List<Integer> li = new ArrayList<>();
        li.add(1);
        mat.add(new ArrayList<>(li));
        if (numRows == 1) {
            return mat;
        }
        li.add(1);
        mat.add(new ArrayList<>(li));
        if (numRows == 2) {
            return mat;
        }
        int j = 3;
        while (j <= numRows) {
            List<Integer> l = new ArrayList<>();
            l.add(1);
            List<Integer> m = mat.get(j - 2);
            int p = 0;
            int q = 1;
            while (q < m.size()) {
                int sum = m.get(p) + m.get(q);
                l.add(sum);
                p = q;
                q++;
            }
            l.add(1);
            mat.add(new ArrayList<>(l));
            j++;
        }
        return mat;
    }
}
