class Solution {

    public List<List<Integer>> generate(int n) {
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            List<Integer> temp = new ArrayList<>();
            for (int k = 0; k <= i; k++) {
                if (k == 0 || k == i)
                    temp.add(1);
                else {
                    int x = ans.get(i - 1).get(k) + ans.get(i - 1).get(k - 1);
                    temp.add(x);
                }
            }
            ans.add(temp);
        }
        return ans;
    }
}
