class Solution {

    public List<List<Integer>> generate(int numRows) {
        //Create list of list of integers
        List<List<Integer>> list = new ArrayList<>();
        if (numRows < 1) {
            return list;
        }
        //Create helper list.
        List<Integer> prevList = new ArrayList<>();
        prevList.add(1);
        //So we add [1] to first row in list [[1]].
        list.add(prevList);
        if (numRows == 1) {
            return list;
        }
        for (int i = 1; i < numRows; i++) {
            List<Integer> newList = new ArrayList<>();
            newList.add(1);
            for (int j = 1; j < i; j++) {
                newList.add(prevList.get(j) + prevList.get(j - 1));
            }
            newList.add(1);
            list.add(newList);
            prevList = newList;
        }
        return list;
    }
}
