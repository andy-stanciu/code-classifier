class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> result = new ArrayList<>();
        for (int line = 1; line <= numRows; line++) {
            List<Integer> temp = new ArrayList<>();
            int a = 1;
            for (int i = 1; i <= line; i++) {
                temp.add(a);
                a = a * (line - i) / i;
            }
            result.add(temp);
        }
        return result;
    }
}
