class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        List<Integer> current, previous = null;
        for (int i = 0; i < numRows; i++) {
            //assign memory to current
            current = new ArrayList<Integer>();
            for (int j = 0; j <= i; j++) {
                // for first and last elements put 1
                if (j == 0 || j == i) {
                    current.add(1);
                } else {
                    // for ones in between, add j-1 and j from previous array
                    current.add(previous.get(j - 1) + previous.get(j));
                }
            }
            // make current as previous for the next iteration
            previous = current;
            result.add(current);
        }
        return result;
    }
}
