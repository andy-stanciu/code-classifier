class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> var1 = new ArrayList<>();
        ArrayList<Integer> firstrow = new ArrayList<>();
        firstrow.add(1);
        var1.add(firstrow);
        for (int i = 1; i < numRows; i++) {
            ArrayList<Integer> curr_row = new ArrayList<>();
            curr_row.add(1);
            for (int j = 1; j < i; j++) {
                curr_row.add(var1.get(i - 1).get(j - 1) + var1.get(i - 1).get(j));
            }
            curr_row.add(1);
            var1.add(curr_row);
        }
        return var1;
    }
}
