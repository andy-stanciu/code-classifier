class Solution {

    public List<List<Integer>> generate(int numRows) {
        //for storing ans
        List<List<Integer>> ans = new ArrayList<>();
        //for store values at single level
        List<Integer> li = new ArrayList<Integer>();
        for (int i = 0; i < numRows; i++) {
            //adding value : 1 at 0th index each time in li.
            li.add(0, 1);
            for (int j = 1; j < li.size() - 1; j++) {
                // replacing old value with new value at index \'j\' in li
                li.set(j, li.get(j) + li.get(j + 1));
            }
            // adding \'li\' to the \'ans\' arraylist
            ans.add(new ArrayList<>(li));
        }
        return ans;
    }
}
