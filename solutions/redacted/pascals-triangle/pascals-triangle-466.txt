class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> re = new ArrayList<>();
        for (int i = 1; i <= numRows; i++) {
            re.add(nthPascal(i));
        }
        return re;
    }

    public List<Integer> nthPascal(int n) {
        List<Integer> list = new ArrayList<>();
        list.add(1);
        int prev = 1;
        for (int i = 1; i < n; i++) {
            prev = prev * (n - i);
            prev = prev / i;
            list.add(prev);
        }
        return list;
    }
}
