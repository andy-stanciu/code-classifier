class Solution {

    public List<List<Integer>> generate(int numRows) {
        List<Integer> firstRow = new ArrayList<>();
        firstRow.add(1);
        List<List<Integer>> out = new ArrayList<>();
        // out is initialize as [[1]] as per constraint requirement
        out.add(firstRow);
        for (int i = 1; i < numRows; i++) {
            // for each row we want to create (i.e., out element)
            // create the row with default first number
            List<Integer> row = new ArrayList<>();
            // each row starts with a 1
            row.add(1);
            // previous line in triangle
            List<Integer> previous = out.get(i - 1);
            for (int j = 0; j < previous.size() - 1; j++) {
                // the number of elements to compute is equals to previous.size()-1
                int nextEl = previous.get(j) + previous.get(j + 1);
                row.add(nextEl);
            }
            // each row ends up with a 1
            row.add(1);
            out.add(row);
        }
        return out;
    }
}
